སྣང་བ་ཞེས་བྱ་བ། མཁས་པ་ཆེན་པོ་སློབ་དཔོན་སེང་གེ་བཟང་པོའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།། །།སྔོན་གྱི་སྨོན་ལམ་བསོད་ནམས་ཚོགས་བསགས་རྒྱུས། །ལྷ་རིགས་བྱང་ཆུབ་སེམས་དཔའི་རྒྱུད་དུ་གྲགས། །ས་སྟེངས་རྒྱལ་པོའི་རྒྱལ་
པོ་ཉིད་གྱུར་པ། །དཔལ་གྱིས་བརྒྱན་ཅིང་སྐྱོན་མེད་གནས་སུ་འཁྲུངས། །གང་དག་གྲགས་པ་ཟླ་ཟེར་ལྟ་བུ་ཡིས། །ས་སྟེངས་ཀུན་ཏུ་རྣམ་པར་མཛེས་གྱུར་ཅིང་། །མཛད་སྤྱོད་ཕུན་ཚོགས་ཡོན་ཏན་ཀུན་གྱིས་བརྒྱན། །མི་དབང་ས་ཡི་བདག་པོ་ལྷ་ལྡེར་གྲགས། །སྐྱེས་ཆེན་
དེས་ནི་བོད་ཁམས་འགྲོ་བ་ཀུན། །སངས་རྒྱས་འབྲས་ཐོབ་བར་གྱི་ཕན་བཞེད་དེ། །རྒྱལ་ཡུམ་དོན་གསལ་བྱ་ཕྱིར་རབ་འབད་པས། །ཡོངས་སྙོམས་ཆོས་ཀྱི་སྦྱིན་བདག་ཆེན་པོ་མཛད། །དོན་འདི་སྒྲུབ་ཕྱིར་དགའ{𰹮་བས་ཐ,རིང་}ག་རིངས་ནས། །གསོལ་བཏབ་རྒྱུ་ཡིས་རྒྱལ་པོའི་
རིགས་སུ་གྲགས། །དགའ་བྱེད་རྒྱུད་བྱུང་ལེགས་བཤད་དཔལ་ཞེས་བྱའི{𰹯། །ཐ,མང་}ོས་མངས་མཁས་པའི་གྲགས་ཐོབ་བདག་གིས་བཤད། །སྒྲ་སྐད་ཐ་སྙད་ལ་བྱང་བརྩོན་པར་ལྡན། །རིན་ཆེན་སྒྲོན་འདྲའི་རིན་ཆེན་བཟང་པོ་ཡིས། །གཙང་དགེ་ཡོན་ཏན་དུ་མའི་འབྱུང་གནས་གྱུར། །
རྒྱལ་སའི་དམ་པ་པུ་རངས་གནས་སུ་བསྒྱུར། །སྙིང་རྗེས་གཞན་ལ་ཕན་པ་གཙོར་བྱས་ཏེ། །རྟོགས་སླ་ཐུན་མོང་རང་སྐད་བྱས་པ་ལས། །བསོད་ནམས་རྒྱ་ཆེན་འདིར་ནི་གང་བསགས་དེས། །འཇིག་རྟེན་མཐའ་དག་བྱང་ཆུབ་མཆོག་ཐོབ་ཤོག །དབང་ཕྱུག་དམ་པའི་མངའ་བདག་བོད་
ཀྱི་ལྷ་བཙན་པོ་ཁྲི་བཀྲ་ཤིས་ལྡེ་བཙན་གྱི་བཀས། རྒྱ་གར་གྱི་མཁན་པོ་སུ་བྷཱ་ཥི་ཏ་དང་། སྒྲ་བསྒྱུར་གྱི་ལོ་ཙཱ་བ་ཆེན་པོ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ནས། སླད་ཀྱི་རྒྱ་གར་གྱི་མཁན་པོ་པཎྜི་ཏ་ཆེན་པོ་དཱི་པཾ་ཀ་ར་ཤྲཱི་ཛྙཱ་ན་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་ཆེན་པོ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་ཡུལ་
དབུས་ཀྱི་དཔེ་དང་ཡང་གཏུགས་ནས་བཅོས་ཤིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ་ལས། དེ་ནས་དུས་ཕྱིས་པཎྜི་ཏ་ཆེན་པོ་གཞུང་འབུམ་ཕྲག་གཉིས་ཀྱིས་མགྲིན་པ་བརྒྱན་པ་དྷི་ར་པཱ་ལ་ཞེས་བྱ་བ་དང་། ལོ་ཙཱ་བ་དགེ་སློང་བློ་ལྡན་ཤེས་རབ་ཀྱིས་ལེགས་པར་བསྒྱུར་ཞིང་བཅོས་པའོ།། །།
