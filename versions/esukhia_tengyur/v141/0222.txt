དང་ཐམས་ཅད་ལ། །དམིགས་པར་དབང་བྱེད་ཕྱིར་དབང་དྲུག །མིག་ལ་སོགས་པ་ལྔ་ནི་རང་རང་གི་དོན་དམིགས་པ་ལ་དབང་བྱེད་དོ། །ཡིད་ནི་དོན་ཐམས་ཅད་དམིགས་པ་ལ་དབང་བྱེད་དེ། དེའི་ཕྱིར་དྲུག་པོ་དེ་དག་ནི་དབང་པོ་རེ་རེའོ། །ཕོ་མོ་ཉིད་དུ་དབང་བྱེད་ཕྱིར། །ལུས་ལ་ཕོ་དང་མོའི་དབང་
པོ། །ལུས་ཀྱི་དབང་པོ་ཉིད་ལ་ཕོ་དང་། མོའི་དབང་པོ་དག་ཏུ་རྣམ་པར་གཞག་སྟེ། དོན་ཐ་དད་དུ་གྱུར་པ་ནི་མེད་ཀྱི་འགའ་ལ་སྟེ། ལུས་ཀྱི་དབང་པོའི་ཆ་མདོམས་ཀྱི་ཕྱོགས་གང་ཡིན་པ་དེ་ནི་ཕོ་དང་མོའི་དབང་པོའི་མིང་འཐོབ་པོ། །དེ་གཉིས་ནི་རིམ་བཞིན་དུ་ཕོ་ཉིད་དང་མོ་ཉིད་དུ་དབང་བྱེད་
དེ། དེ་ལ་མོའི་ངོ་བོ་ནི། མོའི་ཤ་ཚུགས་དང་། ང་རོ་དང་། སྤྱོད་པ་དང་། བསམ་པ་དེ་ལྟ་བུའི་མོ་ལ་མོ་ཉིད་དོ། །ཕོའི་ངོ་བོ་ནི་ཕོའི་ཤ་ཚུགས་དང་། ང་རོ་དང་། སྤྱོད་པ་དང་། བསམ་པ་དེ་ལྟ་བུའི་ཕོ་ལ་ཕོ་ཉིད་དོ། །རིས་གནས་ཀུན་ནས་ཉོན་མོངས་ཉིད། །རྣམ་པར་བྱང་ལ་དབང་བྱེད་ཕྱིར། །སྲོག་
དང་ཚོར་དང་དད་ལ་སོགས། །དབང་པོ་དག་ནི་ལྔར་འདོད་དེ། །སྲོག་གི་དབང་པོ་ནི་རིས་མཐུན་པར་གནས་པ་དག་ལ་དབང་བྱེད་དོ། །ཚོར་བ་རྣམས་ནི་ཀུན་ནས་ཉོན་མོངས་པ་ལ་སྟེ། འདི་ལྟར་བདེ་བའི་ཚོར་བ་ལ་ནི་འདོད་ཆགས་རྒྱས་པར་འགྱུར་ལ། སྡུག་བསྔལ་ལ་ནི་ཁོང་ཁྲོ་བ། །སྡུག་
བསྔལ་ཡང་མ་ཡིན་བདེ་བ་ཡང་མ་ཡིན་པ་ལ་ནི་མ་རིག་པའོ་ཞེས་མདོ་ལས་གསུངས་སོ། །དད་པ་ལ་སོགས་པ་ལྔ་རྣམས་ནི་རྣམ་པར་བྱང་བ་ལ་སྟེ། འདི་ལྟར་དེ་དག་གིས་ཉོན་མོངས་པ་རྣམ་པར་སེལ་ཅིང་ལམ་དག་ཀྱང་འདྲེན་པར་བྱེད་དེ། དེའི་ཕྱིར་དེ་དག་ཀྱང་དབང་པོ་རེ་རེ་འདོད་དོ། །མྱ་
ངན་འདས་སོགས་འཐོབ་པ་ལ། །གོང་ནས་གོང་དུ་དབང་བྱེད་ཕྱིར། །ཀུན་ཤེས་བྱེད་དང་ཀུན་ཤེས་དང་། །དེ་བཞིན་ཀུན་ཤེས་ལྡན་དབང་པོ། །དེ་བཞིན་ཞེས་བྱ་བའི་སྒྲ་ནི་དབང་པོ་རེ་རེ་བསྟན་པར་འདོད་པའི་དོན་ཏོ། །དེ་ལ་ཀུན་ཤེས་པར་བྱེད་པའི་དབང་པོ་ནི་ཀུན་ཤེས་པའི་དབང་པོ་འཐོབ་པ་ལ་
དབང་བྱེད་དོ། །ཀུན་ཤེས་པའི་དབང་པོ་ནི་ཀུན་ཤེས་པ་དང་ལྡན་པའི་དབང་པོ་འཐོབ་པ་ལའོ། །ཀུན་ཤེས་པ་དང་ལྡན་པའི་དབང་པོ་ནི་ཡོངས་སུ་མྱ་ངན་ལས་འདས་པ་ལ་སྟེ། སེམས་རྣམ་པར་མ་གྲོལ་བ་ལ་ནི་ཡོངས་སུ་མྱ་ངན་ལས་འདས་པ་མེད་པའི་ཕྱིར་རོ། །ཡང་ཅིའི་ཕྱིར་བཅོམ་ལྡན་འདས་
