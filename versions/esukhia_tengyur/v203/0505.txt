རིམ་གྱིས་བརྒྱུད་དེ་ཤེས་གྱུར་པ། །བཤད་ཀྱིས་མཉན་ཏེ་བཟུང་བར་གྱིས། །མཆུ་ལྟག་སོར་ནི་བཞི་པ་ལ། །དེ་ནི་རི་མོ་བརྟག་བྱས་ཏེ། །ཁྱད་པར་རྣམ་པ་སྣ་ཚོགས་པ། །དེ་ལ་འཆི་བར་ཤེས་པར་བྱ། །མི་ཡི་ལག་པ་དང་འདྲ་བར། །རྟ་
རྣམས་ཀྱི་ནི་མཆུ་ལྟག་བརྟག །གང་ལ་ཚེ་ནི་ཡོད་གྱུར་པ། །དེ་ཉིད་བཤད་ཀྱིས་མཉན་པར་གྱིས། །མཆུ་ལྟག་དབུས་མར་རི་མོ་ནི། །གྱེན་དུ་སོར་གཅིག་གང་ལ་ཡོད། །དེ་ནི་ལོ་གཅིག་ཉིད་ལ་ནི། །ཁ་ནད་ཀྱིས་ནི་འཆི་བར་འགྱུར། །དེ་འདྲ་བ་ཡི་རི་མོ་གཉིས།
སོར་གཉིས་པར་ནི་སྣང་གྱུར་ན། །རྟ་དེ་ཉིད་ནི་ལོ་གཉིས་ན། །ལྕེ་རེངས་ཀྱིས་ནི་འཆི་བར་འགྱུར། །འདི་འདྲ་བ་ཡི་རི་མོ་གཉིས། །མཆུ་ལྟག་ཏུ་ནི་འཕྲེད་དུ་སྣང་། །ལོ་གསུམ་ནས་ནི་འགྲམ་ནད་ཀྱིས། །དེ་ནི་འཆི་བར་འགྱུར་བ་ཡིན། །འདི་འདྲའི་རི་མོ་དབུས་ན་
དམའ། །རྟ་གང་ལ་ནི་ཡོད་གྱུར་པ། །རྟ་དེ་ལོ་བཞི་ལོན་པ་ན། །འགྲམ་ལྐོག་ནད་ཀྱིས་འཆི་བར་འགྱུར། །གང་གི་གཡོན་ན་རི་མོ་ནི། །རྩེ་མོ་ཡོན་པོ་སོར་གཉིས་ཡོད། །རྟ་དེ་ཉིད་ནི་ལོ་ལྔ་ནས། །{𰷘རི་དགས་,རི་དྭགས་}གླལ་གྱི་ནད་ཀྱིས་འཆི། །མཆུ་ལྟག་དབུས་ན་རི་མོ་ནི། །
སོར་གཉིས་ཚད་ནི་ཡོན་ཏན་ཡོད། །གང་ཡིན་དེ་ནི་ལོ་དྲུག་ནས། །ཨ་བ་ར་གའི་ནད་ཀྱིས་འཆི། །གང་གིས་རི་མོ་འདྲ་བ། །སོར་བྱེད་ཁར་ནི་ཆུད་གྱུར་པ། །རྟ་དེ་ལོ་ནི་བདུད་ཉིད་ན། །གྲི་ཤིའི་ནད་ཀྱི་གདུངས་ནས་འཆི། །རི་མོ་དེ་འདྲ་སོར་གསུམ་པ། །རྟ་ནི་གང་ལ་
སྣང་གྱུར་པ། །རྟ་དེ་ལོ་བརྒྱད་ལོན་ནས་ནི། །རིམ་གྱི་ཉེ་བར་འཆི་བར་འགྱུར། །གང་གི་རི་མོ་འདི་འདྲ་ལས། །དབུས་ན་ནས་འདྲ་སྦོམ་པོ་ཡོད། །རྟ་དེ་ཉིད་ནི་ལོ་དགུ་ན། །རླུང་གི་གཟེར་གྱིས་སྲོག་ཟད་འགྱུར། །རི་མོ་མཆུ་ལྟག་དང་མཉམ་པ། །གྱེན་དུ་བྱུང་བ་དབུས་
ན་གནས། །རྟ་དེ་ལོ་ནི་བཅུ་པ་ན། །སྐེ་རེངས་ཀྱིས་ནི་བརླག་པར་འགྱུར། །གང་ལ་རི་མོ་འདི་འདྲ་གཉིས། །མཆུ་ལྟག་དབུས་ན་གནས་གྱུར་བ། །ལོ་ནི་བཅུ་གཅིག་ལོན་པ་ན། །རླུང་འཁྱིལ་ནད་ཀྱིས་སྲོག་ཟད་འགྱུར། །གང་ལ་དེ་ནི་སོར་གཉིས་ལས། །བཅུད་
