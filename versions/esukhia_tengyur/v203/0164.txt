དང་། །བདག་ཉིད་ཅི་ནས་རྒྱལ་བར་འགྱུར། །དེ་ནས་རྫོང་གི་འཁོར་ལོ་མཆོག །རིམ་པ་གསུམ་ལྡན་འདིར་བརྗོད་བྱ། །འདི་འདྲའི་འཁོར་ལོ་སུ་ལ་ཡོད། །བཅོམ་ལྡན་འདས་ལ་འཁོར་ལོ་མངའ། །སྨིན་དྲུག་ནས་ནི་བརྩམས་ནས་སུ། །ར་བ་གསུམ་ལ་དགོད་པར་བྱ། །སྨིན་དྲུག་རྒྱལ་སྐག་
མཆུ་དང་ནི། །ས་རི་ས་ག་ལྷ་མཚམས་དང་། །གྲོ་བཞིན་བྱི་བཞིན་མོན་གྲུ་དང་། །ཐ་སྐར་བྲ་ཉེ་བཅུ་གཉིས་པོ། །ཕྱི་ཡི་ར་བ་རྣམས་ལ་ཡང་། །རིམ་པ་བཞིན་དུ་དགོད་པར་བྱ། །སྣར་མ་ནབ་སོ་གྲེ་དང་ནི། །ནག་བ་སྣྲོན་དང་ཆུ་སྨད་དང་། །མོན་གྲེ་དང་ནི་ནམ་གྲུ་སྟེ། །བརྒྱད་པོ་བར་གྱི་ར་བར་དགོད། །
མགོ་དཔལ་གདང་དེ་བཞིན་དབོ། །མེ་བཞི་སྣྲུབས་དང་ཆུ་སྟོད་དང་། །ཁྲུམས་སྟོད་ཁྲུམས་སྨད་རྣམ་པར་བརྒྱད། །ནང་གི་ར་བར་དགོད་བར་བྱ། །ར་བ་བཞིན་ལ་དཔུང་གི་སྲོག །ནང་མ་བདག་པོའི་སྲོག་ཡིན་ཏེ། །བར་མ་མཉམ་བོར་བསྟན་པ་ཡིན། །དེ་ལ་རཱ་ཧུ་རྒྱལ་པོ་སྟེ། །ལྷག་མ་བློན་པོར་
གནས་པ་ལས། །རྒྱལ་ཕམ་མཚན་ཉིད་སྟོན་པར་བྱེད། །ནང་གི་ཕྱོགས་བཞིར་རཱ་ཧུ་སོང་། །རྫོང་ནི་བླངས་པས་ཆོག་པ་ཡིན། །ཕྱི་རོལ་བཅུ་གཉིས་རཱ་ཧུ་སོང་། །བདག་པོ་ཉལ་བསོད་ནམས་འགྲུབ། །བར་གྱི་ར་བར་རཱ་ཧུ་སོང་། །ཕྱི་ནང་མཉམ་པར་གྲགས་པ་ཡིན། །སྔར་བཤད་ཐུར་མ་བདུན་
པ་ཡི། །འཁོར་ལོ་ཇི་བཞིན་ཤེས་པར་བྱ། །སྤེན་པ་དང་ནི་ཕུར་བུ་དག །གང་གནས་པ་སྟེ་ལེགས་པར་བལྟ། །ར་བ་གསུམ་པོ་གང་ལ་གནས། །འཁོར་ལོ་འདི་ལ་མི་དགེ་ཡིན། །རྫོང་དང་གྲོང་ཁྱེར་རྣམས་སུ་ཡང་། །ཉི་མ་ཡོད་པའི་མི་འགྲོ་སྟེ། །ཉི་མ་ཕུར་བུ་སྤེན་པ་གསུམ། །ལྷན་ཅིག་ཡོད་
པར་འགྲོ་མི་རུང་། །རྫོང་གི་འཁོར་ལོའི་ཆོ་ག་ལས། །སྒྲ་གཅན་ཁ་མཇུག་ལེགས་པར་བརྟག །རྐང་པ་བརྒྱ་བ་ལ་སོགས་པའི། །སྔར་བཤད་ཆོ་ག་ཐམས་ཅད་བརྟག། །རྫོང་དང་རྫོང་གི་བདག་པོ་ཡི། །སྐར་མ་རཱ་ཧུ་ཡོད་དམ་བརྟག །གལ་ཏེ་དེ་ལ་རཱ་ཧུ་སོང་། །རྫོང་དང་བདག་པོ་དག་ལ་ངན། །ནང་
གི་ཕྱོགས་བཞིར་རྰ་ཧུ་སོང་། །རྫོང་ལ་མི་དགེ་ལྷས་མཆོད་མ། །རབ་རྣམས་ཀྱི་རིམ་པ་ལ། །ཇི་ལྟར་གནས་པ་བརྟག་པར་བྱ། །དྲག་པོ་བར་མ་བདག་ལ་སོགས། །ཡུན་རིང་ལོན་ན་འཐོབ་པར་འགྱུར། །སྡིག་ཅན་གཟའ་རྣམས་སྙིང་པོར་ཞུགས། །དེ་ཚེ་ཟླ་བ་བརྟག་པར་བྱ། །ཟླ་བ་གནས་པའི་
