གསོ་དང་། །ཐལ་མོ་བྱེད་པ་ཅན་གྱི་ནད། །ཁྲག་འཁྲུགས་པ་ཡི་ནད་གསོ་དང་། །ཚིགས་སུ་རྨ་བྱུང་ཐོག་འབབ་དང་། །གཉེའུ་རེངས་དང་སྦོས་ནད་གསོ། །འཁྲུ་བ་གསོ་བ་ཞེས་བྱ་བ། །གནས་ཆེན་བཞི་པ་གསོ་བ་ཡིན། །རྟེའུའི་གདོན་ནི་བསྟན་དང་བཤད། །འཁྲུ་དང་དེ་བཞིན་རིམས་
དང་ནི། །གཡན་པ་ལྐོག་ནད་ཤ་ལུ་ཀ། །ཁོང་པའི་སྲིན་ནད་གཅིན་སྲིར་བཅས། །བོང་བུ་ཅན་དང་རླིག་རླུགས་དང་། །གཅིན་ནད་གཟེའུ་ཞེས་བྱ་བ། །སྦོ་དང་བྲང་ནད་ཅེས་བྱ་བ། །བཀྲེས་པས་ཉེན་དང་ལུད་པ་གསོ། །ཟད་བྱེད་དང་ནི་རོ་ཧི་ནི། །རི་མོ་ཅན་དང་སྦལ་པ་འདྲ། །མེས་ཚིག་ཚིགས་ནད་
གཅིན་དམར་གསོ། །ཡིད་མི་བདེ་དང་རྣ་བའི་ནད། །སོ་ནད་དང་ནི་མཇིང་པའི་ནད། །འགྲམ་ནད་ལྕེ་ཡི་ནད་བཅས་དང་། །མིག་ནད་སྣ་ནད་བཅས་པ་དང་། །རྨིག་པའི་ནད་དང་དབྱིག་པའི་ནད། །སེམས་འཁྲུལ་པ་དང་རྔུལ་་ནད་བཅས། །སྦྲུལ་གྱིས་ཟིན་དང་མེས་ཚིག་གསོ། །དེ་བཞིན་
ཆག་གྲུགས་གསོ་བ་ནི། །ལྔ་པའི་གནས་སུ་ངས་བཤད་ཅིང་། །རྟེའུའི་ནད་རྣམས་གསོ་བ་ཡིན།། །།རྟ་ཁང་པ་ནི་བརྟག་པ་དང་། །ཁང་པའི་ལྷ་རྣམས་མཆོད་པ་དང་། །དེ་གཞིན་གདོན་ནི་བསལ་བ་དང་། །དཔལ་འཕེལ་བ་ཞེས་བྱ་བ་དང་། །རྟ་རྣམས་རྗེས་སུ་བསྲུང་བ་དང་། །རྣམ་རྒྱལ་
རྟ་ནི་དབང་བསྐུར་དང་། །མཆེ་བ་ཆག་པའི་ལྟས་བསྟན་དང་། །ལྟས་ཀྱི་མཚན་ཉིད་གཞན་བསྟན་དང་། །ཞི་བ་བསྐྱེད་པར་བྱ་བ་རྣམས། །དྲུག་པའི་གནས་སུ་བསྡུས་པ་ཡིན། །དང་པོར་རྗེས་བཏུང་ཆོ་ག་སྟེ། །དེ་འོག་ལན་ཚྭའི་ཆོ་ག་ཡིན། །སྣུམ་དང་ལན་ཚྭ་ཆང་དང་འབྲུ། །འོ་མ་ཆུ་
དང་སྦུབས་ལས་བྱུང་། །གནོད་པའི་ནད་ནི་བདུན་པོ་ཡི། །ནད་གཞི་གསོ་བར་བཅས་བསྟན་དང་། །མར་བླུད་དུགས་ཀྱི་ཆོ་ག་དང་། །དུར་བྱིད་ཀྱི་ནི་ཏིལ་མར་དང་། །དོན་གྲུབ་ཅེས་བྱའི་ཏིལ་མར་དང་། །མེ་བཙའི་ལས་ཀྱི་ཆོ་ག་དང་། །མས་གཏོང་ཆོ་ག་བསྟན་པ་དང་། །གཙགས་བུ་
ལས་གྱུར་ནད་བསྟན་དང་། །དེ་བཞིན་ལྟ་བའི་ལས་བསྟན་དང་། །སྣུམ་བླུད་པ་ཡི་ཡོན་ཏན་དང་། །རོ་ཙར་བྱེད་པ་བསྟན་པ་དང་། །སྣུམ་ལས་གྱུར་པའི་ནད་གསོ་བའི། །བྱ་བ་བསྟན་པའི་གནས་འདི་ནི། །བདུན་པ་གྲུབ་པའི་མཚན་ཉིད་ཡིན། །དང་པོ་རི་མོ་ལས་ཆེ་བརྟག །
