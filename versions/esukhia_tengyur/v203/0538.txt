རླུང་ལྔ་པོ། །སྲོག་དང་གྱེན་རྒྱུ་ཐུར་སེལ་དང་། །མེ་དང་མཉམ་གནས་ཁྱབ་བྱེད་རྣམས། །རྗེས་སུ་མཐུན་པར་རྒྱུ་བར་འགྱུར། །དེ་ཚེ་ཐུར་སེར་རླུང་གིས་ནི། །མེ་དང་མཉམ་གནས་སྐྱེད་པར་བྱེད། །མཉམ་གནས་ཀྱིས་ཀྱང་ཁྱབ་བྱེད་བརྟན། །དེས་ཀྱང་གཞན་དག་བརྟན་པར་
བྱེད། །དེ་ལྟར་གོང་ནས་གོང་དུ་ནི། །འདི་དག་རིམ་གྱིས་རྗེས་མཐུན་གནས། །དཔེར་ན་སོས་ཀའི་ཉི་མ་ཡིས། །ས་ལ་གནས་པའི་ཆུ་ཚོགས་ཀྱི། །གཟི་མདངས་བླངས་པར་གྱུར་ནས་ནི། །ནམ་མཁའི་ངོས་ལ་གནས་པ་ཡིན། །རྟ་ལ་འཇམ་རྩི་ལེགས་བཏང་པ། །
དེ་དང་འདྲ་བར་ཤེས་པར་བྱ། །ཇི་ལྟར་ལེབ་རྒན་རྩིས་བསྒྱུར་བའི། །མཚོན་པའི་འདག་ཆལ་གྱིས་སེལ་ལྟར། །དེ་བཞིན་མས་གཏོང་ནི་རུ་ཧ། །ཕྲ་བའི་ལམ་ནས་ཞུགས་གྱུར་པས། །མགོ་ནས་རྐང་པའི་བར་གྱི་ནི། །ནད་རྣམས་སེལ་བར་བྱེད་པ་ཡིན། །ཤ་ལི་ཧོ་
ཏྲའི་གསང་བའི་གནས་ཏེ་བརྒྱད་པ་ལས་མས་གཏོང་བའི་ལས་ཀྱི་ཡོན་ཏན་བསྟན་པའོ།། །།སྣ་སྨན་གཏོང་བའི་སྦུ་གུ་ནི། །སོར་བཅུ་རུ་ནི་བྱ་བར་བཤད། །ཡང་ན་སྟེང་ནས་བླུགས་བྱ་སྟེ། །དེ་ཡང་རས་བལ་ཉིད་ལ་གཟག །སྔ་དྲོ་ལྟོ་སྟོང་གྱུར་པ་ཡི། །རྟ་ཡི་ལུས་
ནི་དྲང་པོར་གཞག །འཕྲལ་དུ་བླུགས་ཀྱིས་རྒྱུན་དུ་མིན། །དབུགས་དབྱུང་ཚེ་ན་སྤྱད་པར་བྱ། །སྣ་སྨན་གཏོང་བ་རྩོལ་བ་དང་། །ཆུ་འཐུངས་པ་ལས་བསྲུང་བར་བྱ། །གསང་བའི་གནས་ལས་སྣ་སྨན་བཏང་བའི་ཆོ་གའོ།། །།རུ་རྟ་ཡུང་བ་པི་པི་ལིང་། །བཅའ་
ལྒ་ཤིང་ཀུན་སྐྱེར་ཁཎ་དང་། །པྲི་ཡང་ཀུ་དང་ར་སྣ་ནི། །ལྷན་ཅིག་སྦྱར་ཏེ་བཏགས་པར་བྱ། །སྨན་ཁུ་དེ་ལ་ར་སྣ་ནི། །བཞི་ཆ་སླར་ཡང་གཞག་པར་བྱ། །སྣ་སྨན་འདི་ནི་གཉེའུ་རེངས་དང་། །རྣ་རྩེའི་ནད་ལ་གཏང་བར་བྱ། །དེ་ནི་འཐེང་དང་བྲང་ཁ་ན། །འགྲམ་པ་དང་ནི་བརླ་
རེངས་དང་། །རླུང་དང་བད་ཀན་ལས་གྱུར་པའི། །ནད་ཀུན་སེལ་བར་བྱེད་པ་ཡིན། །གསང་བའི་གནས་ལས་རུ་རྟ་ལ་སོགས་པའི་སྣ་སྨན་ནོ།། །།བུ་རམ་བཅའ་ལྒའི་ར་བའམ། །པི་པི་ལིང་དང་རྒྱམ་ཚར་སྦྱར། །སྣ་སྨན་བཏང་ན་འགྲམ་པ་སྣད། །དཔུང་པ་མགོ་བོའི་ནད་རྣམས་དང་། །
