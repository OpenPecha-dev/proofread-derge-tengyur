ལྷ་ཁང་བརྩིག་པ་དང་། ལྷ་བཞེངས་པ་དང་། རྔམས་ཆེ་བའི་ལས་ལ་མིག་གི་འབྲས་བཞིན་གལ་ཆེའི། །བེལ་སེ་གའང་འདིའི་ནང་ན་ཞིབ་བས་ལྟད་མོ་དང་སྟོན་མོ་གལ་ཆེའོ། །རུས་སྤལ་འཁོར་ལོས་ཡུལ་གྱི་བདེ་སྡུག་དང་། ལེགས་ཉེས་ཤེས་པས་ཞི་དྲག་གི་གཟའ་ལ་
བལྟའོ། དེས་པ་ཙཀྲ་ནི་རྒྱལ་པོའི་བླ་མཆོད་དང་། མི་དཔོན་གྱི་གཡོག་བྱེད་པ་དང་། །རང་ལས་ཆེ་བ་རྣམས་བརྟག་གོ །ལེའུ་ལྔ་བའོ།། །།འཁོར་ལོ་བདུན་ཀྱི་གདན་ལ་བརྟེན་ནས། རླུང་རྒྱུ་བའི་ཡུན་ནམ་རིང་ཐུང་བལྟ་སྟེ། འཆི་ལྟས་དང་། ན་ཚ་དང་། ཁ་སྨྲས་དང་། དེ་རྣམས་ཤེས་ཏེ།
རླུང་རྒྱུ་བ་ཡུན་རིང་འཆི། འབྲིང་ནན། ཡུན་ཐུང་ན་ཁ་སྨྲས་འོང་ངོ་། །ཚེ་ཚད་ནི་ལོ་བདག་དང་སྐྱེ་སྐར་དྲི་སྟེ། །བློ་ཞིབ་ན་བརྟན་ནོ། །གཟའི་ལོ་བདག་རེ་རེ་ཞིང་ཡང་དགེ་མི་དགེ་གཞུང་ཉིད་ཀྱིས་བགོའོ། །སྐྱེས་བུ་དུས་ཇི་ཙམ་ན་སྐྱེས་པའི་དུས་ཀྱི་ཉིན་སྐར་ཁ་ལ་དགོད། དེ་ནས་རྩིའོ། །འདི་ནས་སྐྱེ་
བ་བརྟག་པའི་ཆོག་སྟེ། གཞོན་པ་ཁ་བྱང་གི་རྩིའོ། །ལེའུ་དྲུག་པའོ།། །།ལམ་གྱི་འཁོར་ལོ་ལ་སྦྲུལ་གྱི་ལུས་ལ་རྩ་གསུམ་ལ་སོགས་པ་གཞུང་ཉིད་ཀྱིས་གོ་ཞིང་། དེ་དག་ལ་ཞི་དྲག་གི་གཟའ་གང་གནས་བལྟའོ། །ལམ་དུ་ཞུགས་པའི་ཉིན་ཞག་ལ་དབང་བའི་སྐར་མ་དང་། རྒྱུད་གཅིག་
པའི་རྩ་ལ་གཟའ་དང་གནས་པའི་འབྲས་བུ་འབྱུང་ངོ་། །ས་ཚིགས་ཇི་ཙམ་དུ་འབྱུང་ན། འགྲོ་སར་དཔག་ཚད་དུ་ཡོད་པ་དེ་རྩ་རྒྱུད་ལ་གནས་པའི་གཟའ་དེ་ཡིས་བསྒྱུར། འཁོར་ལོས་བགོ་ནོར་ནི་འགྲོ་ལ་མ་ཡིན། ལྷག་མ་བཞིས་བསྒྱུར། འཁོར་ལོས་ཐོབ་པ་ནི། མདའ་འཕེན་དྲུག་ཅུའམ་རྐང་པ་
གཅིག་ཡིན་ནོ། །ལྷག་མ་དྲུག་ཅུས་བསྒྱུར་ཅིང་ཐོབ་པ་ནི་མདའ་རྒྱད་གང་དུ་འགྱུར་བ་ཡིན་ནོ། །ས་ཡུལ་འདི་ནས་བརྩེས་པའི་དྲོ་ལམ་ཀྱང་ལ། མདའ་འཕེན་རྣམས་འདིའི་འབྲས་བུ་འབྱུང་ཞེས་བྱའོ། །རྒྱན་གྱི་ཙཀྲ་ནི་མིང་ཡིག་ཉེར་གཉིས་རང་གིས་བསྒྱུར་ཏེ། བརྒྱད་ཀྱིས་བགོས་ངས་ཐོབ་པའི་
ལྷག་མ་ལ་བ་ལྟའོ། རྒྱལ་མཚན་ལ་སོགས་འཆར་རོ། །འདས་པའི་དབྱུ་གུ་ལ་བརྒྱད་ཀྱིས་བགོ། དེ་ནི་རྒྱལ་མཚན་པ་འཆར་རོ། །རེ་རེ་ཞིང་གཟའ་བདུན་ཀྱི་བདག་པོ་ཡང་ཡོད་དོ། །མིང་ཡིག་མཉམ་ན་འདུམ་ཞིང་། བཟའ་མི་འང་འགྲོགས་ལུགས་ཡུན་རིང་ངོ་། །མི་མཉམ་ན་གང་མང་རྒྱལ་ལོ། །
