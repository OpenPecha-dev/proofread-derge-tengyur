རྟགས་འདིས་སྦྱོར་ཉེས་ཡིན་ཤེས་བྱ། །སྙིང་ག་ཚ་ཞིང་ཁ་ནས་ནི། །ཁ་ཆུ་ལྦུ་བ་དང་བཅས་འབྱུང་། །ཆུ་ལ་མངོན་པར་དགའ་བ་དང་། །རྟག་ཏུ་འཐུང་བར་འདོད་པར་འགྱུར། །ཉམ་ཆུང་བཤང་བ་སྲི་བ་དང་། །ལྕེ་ཡང་འབྱིན་པར་བྱེད་པར་འགྱུར། །རྩྭ་ལ་སྲེད་པ་ཆུང་
བ་དང་། །ས་ལ་ལྕེ་ཡིས་འདེགས་པར་བྱེད། །ཀུན་དུ་དབུགས་རྒོད་ལུས་རུལ་ཞིང་། །མཁྲིས་པའི་ཚ་བས་གདུངས་པ་དང་། །སྙིང་ག་སྲེག་པ་སྙམས་སེམས་ཤིང་། །དེ་ཡི་ཁ་ཡང་སྐམས་པར་འགྱུར། །མགོ་བོ་འགུགས་ཤིང་སྐེ་བར་འཇོག །མདོག་ཉམས་
ལུས་ནི་བཅག་པར་འགྱུར། །ལན་ཚྭ་ལས་འགྱུར་ནད་དེ་འདྲར། །ཤེས་པར་བྱས་ནས་གསོ་བ་བརྩམ། །ཆུ་ལ་སྦང་དང་བླུགས་པ་དང་། །འདམ་གྱིས་ཡོངས་སུ་བསྐུ་བར་བྱ། །ཆུ་བསིལ་ཁ་ར་དང་སྦྱར་བའི། །བཏུང་བ་དེ་ལ་བླུད་པར་བྱ། །ཆུ་གྲང་ལ་ནི་སྦང་བར་
བྱ། །འཐུང་དུ་གཞུག་པར་མི་བྱའོ། །འབྲས་ཆང་གྲང་མོ་འོ་མ་དང་། །མར་དང་སྦྲང་རྩི་དང་སྦྱར་བྱིན། །པ་རུ་ཤ་ཀ་རྒུན་འབྲུ་དང་། །ཙི་ལུའི་འབྲས་བུ་ཁ་ར་དཀར། །སྦྲང་རྩི་དང་ནི་བསྲེས་པ་ཡིས། །སྨན་གྱི་ཆང་བུ་དེ་ལ་བྱིན། །སྔར་བཤད་པ་ནི་གང་ཡིན་

སེལ་བ་ཡི། །མངར་བའི་སྨན་གྱི་འཇམ་རྩི་གཏང་། །གལ་ཏེ་ཁྲག་ནི་ཤས་ཆེ་བས། །ལུས་ནི་ཤིན་ཏུ་ལྕི་གྱུར་ན། །རྟ་ལ་གཏར་ག་བྱས་གྱུར་ན། །བདེ་བ་ངེས་པར་འཐོབ་པར་འགྱུར། །ངང་པ་ངུར་པ་ཁྲུང་ཁྲུང་དང་། །ཆུ་བྱ་གཞན་དང་འོལ་བྱ་དང་། །རྨ་བྱ་ཁྱིམ་བྱ་སྲེག་
པ་ཡི། །ཤ་ལས་ཤ་ཁུ་བྱས་ཏེ་སྦྱིན། །གནས་ནི་བདེ་ཞིང་སེར་བུ་རྒྱུ། །བསིལ་ཞིང་ཆག་ཆག་བཏབ་པར་གཞག །ཨུཏྤལ་བརྒྱད་པ་དཱུར་བ་དང་། །རྟ་མ་བྲི་ཎྟ་ཨ་རྫུ་ན། །རྩྭ་ཡི་ཕྱེད་དུ་སྦྱིན་བྱས་ན། །དེས་ནི་བདེ་བ་འཐོབ་བར་འགྱུར། །ཤ་ལི་ཧོ་ཏྲའི་གནས་བདུན་པ་
