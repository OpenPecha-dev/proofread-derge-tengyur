དང་། སྐག་དང་། མཆུ་དང་། གྲེ་དང་། དབོ་དང་། མེ་བཞི་དང་། ནག་པ་དང་། ས་རི་དང་། ས་ག་དང་། ལྷ་མཚམས་དང་། སྣྲོན་དང་། སྣྲུབས་དང་། ཆུ་སྟོད་དང་། ཆུ་སྨད་དང་། བྱི་བཞིན་དང་། གྲོ་བཞིན་དང་། མོན་གྲེ་དང་། མོན་གྲུ་དང་། ཁྲུམས་སྟོད་དང་། ཁྲུམས་
སྨད་དང་། ནམ་གྲུ་དང་། ཐ་སྐར་དང་། བྲ་ཉེ་དང་། འདི་རྣམས་ནི་རྒྱུ་སྐར་ཉི་ཤུ་རྩ་བརྒྱད་དོ། །དེ་ལ་རྒྱུ་སྐར་ནི་དུ། དབྱིབས་ནི་ཇི་སྙེད་ཡོད་། ཡུད་ཙམ་དང་ལྡན་པ་ཇི་སྙེད། ཟས་ནི་ཅི། རུས་ནི་ཅི། ལྷ་རྣམས་ནི་གང་། ཀྱེ་བྲམ་ཟེ་པདྨའི་སྙིང་པོ་ཅན་སྨིན་དྲུག་ནི་སྐར་མ་དྲུག་སྟེ། དབྱིབས་ནི་སྤུ་གྲི་ལྟ་བུའོ།
ཡུད་ཙམ་སུམ་ཅུ་དང་ལྡན་པའི་ཆ་དྲུག་དང་བཅས་པའོ། །ཟས་ནི་ཞོའོ། །ལྷ་ནི་མེ་ལྷའོ། །རུས་ནི་རྗེའུ་རིགས་ཀྱི་རྒྱུད་དོ། །རྒྱུ་སྐར་སྣར་མ་ནི་སྐར་མ་ལྔ་སྟེ། དབྱིབས་ནི་ཤིང་རྟ་ལྟ་བུའོ། །ཡུད་ཙམ་ནི་བཞི་བཅུ་རྩ་ལྔ་དང་ལྡན་པའི་ཆ་བདུན་དང་བཅས་པའོ། །ཁ་ཟས་ནི་ཞོ་དང་སྦྲང་རྩིར་སྦྱར་བའོ། །ལྷ་ནི་སྐྱེ་དགུའི་བདག་
པོའོ། །རུས་ནི་གཽ་ཏ་མའོ། །རྒྱུ་སྐར་མགོ་ནི་སྐར་མ་གསུམ་སྟེ། དབྱིབས་ནི་{𰵟རི་དགས་,རི་དྭགས་}ཀྱི་མགོ་ལྟ་བུའོ། །ཡུད་ཙམ་ནི་སུམ་ཅུ་དང་ལྡན་པའི་ཆ་དང་བཅས་པའོ། །ཟས་ནི་{𰵠རི་དགས་,རི་དྭགས་}ཀྱི་ཤའོ། །ལྷ་ནི་ཟླ་བའོ། །རུས་ནི་བྷ་ར་དྷྭ་ཛའོ། །རྒྱུ་སྐར་ལག་ནི་སྐར་མ་གཅིག་སྟེ། དབྱིབས་ནི་སོག་ལེ་ལྟ་བུའོ། །ཡུད་ཙམ་བཅོ་ལྔ་
དང་ལྡན་པའི་ཆ་བདུན་དང་བཅས་པའོ། །ཁ་ཟས་ནི་ཁྲག་གོ། །ལྷ་ནི་དྲག་ཤུལ་ཅན་ནོ། །རུས་ནི་དམར་པོའི་རྒྱུད་དོ། །རྒྱུ་སྐར་ནབ་སོ་ནི་སྐར་མ་གཉིས་ཏེ། དབྱིབས་ནི་ཁང་པའི་གདུང་ལྟ་བུའོ། །ཡུད་ཙམ་བཞི་བཅུ་རྩ་ལྔ་དང་ལྡན་པའི་ཆ་བདུན་དང་བཅས་པའོ། །ཁ་ཟས་ནི་མར་དང་ཁུར་བའོ། །ལྷ་ནི་ཉི་
མའོ། །རུས་ནི་དབང་བྱེད་དོ། །རྒྱུ་སྐར་རྒྱལ་ནི་སྐར་མ་གསུམ་སྟེ། དབྱིབས་ནི་རིལ་བ་ལྟ་བུའོ། །ཡུད་ཙམ་སུམ་ཅུ་དང་ལྡན་ནོ། །ཁ་ཟས་ནི་སྦྲང་དང་སྤོས་སོ། །ལྷ་ནི་ཕུར་བུའོ། །རུས་ནི་རྩྭ་ཤུན་གྱོན་པའི་རྒྱུད་དོ། །རྒྱུ་སྐར་སྐག་ནི་སྐར་མ་དྲུག་སྟེ། དབྱིབས་ནི་སྦྲུལ་མགོའི་གདེངས་ཀ་ལྟ་བུའོ། །ཡུད་ཙམ་
བཅོ་ལྔ་དང་ལྡན་པའི་ཆ་བདུན་དང་བཅས་པའོ། །ཁ་ཟས་ནི་སྦྲུལ་གྱི་ཤའོ། །ལྷ་ཡང་སྦྲུལ་ལོ། །རུས་ནི་དམའ་བའི་རྒྱུད་དོ། །པདྨའི་སྙིང་པོ་ཅན་རྒྱུ་སྐར་འདི་བདུན་ནི་ཤར་སྒོར་གཏོགས་སོ། །རྒྱུ་སྐར་མཆུ་ནི་སྐར་མ་དྲུག་སྟེ། དབྱིབས་ནི་ཆུ་བོ་ལྟ་བུའོ། །ཡུད་ཙམ་སུམ་ཅུ་དང་འབྲེལ་བའི་མཚན་ཡིན་
