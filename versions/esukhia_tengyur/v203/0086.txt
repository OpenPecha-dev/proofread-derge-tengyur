འགྱུར་རོ། །རྒྱུ་སྐར་སྣོན་ལ་ས་གཡོས་ན་རི་ཆེན་པོ་བསྙིལ་བར་འགྱུར། འཇིག་པའི་རླུང་རབ་ཏུ་ལྡང་། ཡུལ་ཨ་མྲ་ཀ་ཞེས་བྱ་བའི་ཡུལ་དུ་ཡུལ་ཀ་ཤའི་མི་དང་། རྐུན་པོ་དང་། མི་ངས་དང་། ཐན་ཆེན་དང་། སེར་བ་དང་། ཡང་དང་ཡང་མཚོན་ཀར་འགྱེད་པ་ཡུལ་དེ་ཕྱོགས་ན་མྱ་ངན་བྱེད་ཅིང་འགྲོ་སྟེ།
ས་གཡོས་པས་གཟི་བརྗིད་ཉམས་ནས་གནོད་པ་དེ་ལྟ་བུ་འབྱུང་བར་འགྱུར་རོ། །ཚུལ་མ་ཡིན་པ་དག་གིས་ཕན་ཚུན་དུ་གནོད་པ་བྱེད་པར་འགྱུར་རོ། །མི་གཡོན་ཅན་དང་། རྒྱལ་པོ་དང་། རྒྱའི་མི་དང་། དེ་བཞིན་དུ་སྒོ་ལྕགས་འབྱེད་ཅིང་རྐུ་བའི་མི་རྣམས་དང་། ཡུལ་ཨ་ན་དྲའི་མི་རྣམས་དང་། ཡུལ་པུ་ལིནྡ་ན་
གནས་པའི་མི་རྣམས་འཇིག་པར་གནས་པར་འགྱུར། །མཛའ་བའི་དཔུང་དང་ལྡན་པ་རྣམས་དང་ཡང་ཕྱེད་པར་འགྱུར། རྒྱལ་པོ་ཆེན་པོ་དང་། སྙོམ་ལས་དང་ལྡན་པ་རྣམས་ཀྱང་ཚར་གཅོད་པ་དང་། བྱ་རོག་དང་། {𰵪རི་དགས་,རི་དྭགས་}པྲི་ཤ་དང་། གཅན་གཟན་དང་། གཅན་གཟན་དྲག་ཤུལ་ཅན་ཆེན་པོ་རྣམས་དང་། །
ཡུལ་ཀ་ར་དང་། ཡུལ་པ་ར་དང་། ཡུལ་ཤུད་དྷ་སེ་ན་ན་གནས་པའི་མི་དང་། ཡུལ་མ་ལ་ཡའི་མི་རྣམས་དང་། ཐོ་གར་གྱི་ཡུལ་བག་ལེ་དང་། པར་སེག་གི་ཡུལ་དང་། ཟ་ཧོར་གྱི་ཡུལ་དང་། ཀ་པུ་ཤའི་ཡུལ་གྱི་རི་བ་རི་ཤེ་ནའི་འཁོར་ན་གནས་པའི་མི་དང་། འདི་རྣམས་ལ་ས་གཡོས་ན་དགྲ་དྲག་པོ་མྱུར་
དུ་འབྱུང་ཞིང་གནོད་པར་འགྱུར་རོ། །རྒྱུ་སྐར་སྣུབས་ལ་ས་གཡོས་ན་རྐང་བཞི་པ་དང་། རྐང་གཉིས་པ་དང་། དེ་བཞིན་དུ་ཕུག་ན་གནས་པ་རྣམས་དང་། {𰵫ཡི་དགས་,ཡི་དྭགས་}ཤ་ཟ་རྣམས་དང་། སེམས་ཅན་གང་དག་རི་ལ་གནས་པ་རྣམས་དང་། མུ་གེ་མྱུར་དུ་འབྱུང་། ས་ལ་ཆུ་ཡང་མྱུར་དུ་སྐམས་པར་འགྱུར་རོ། །དགྲ་
རྣམས་རི་དྲུང་དུ་སོང་ནས་དེ་ནས་དགའ་བར་གནས་པ་དང་། ཆོས་ཀྱི་ཡོ་བྱད་རྣམས་དང་། གང་ན་ནགས་ཚལ་ཕུན་སུམ་ཚོགས་པ་དང་། རྒུན་ར་དང་། བུ་རམ་ཤིང་གི་ཁུ་བ་དང་། འབྲས་བུ་མང་པོ་ཅུང་ཟད་ཡོད་པ་རྣམས་དང་། ཤིང་དུ་ཀའི་ཚལ་དང་། མེ་ཏོག་སྣ་ཚོགས་རྣམས་དང་། འབྲུ་དང་ས་བོན་རྣམས་དང་།
འབྲས་ལ་སོགས་པ་རྣམ་པ་ཐ་དད་པ་སྣ་ཚོགས་དང་། ཁྱིམ་གྱི་ལས་ཀུན་བརྩམ་པ་དང་། བརྟན་པར་བྱ་བའི་རྣམ་པ་དང་། རྩ་བ་བསྐྱེད་ཅིང་བསྐྲུན་པའི་ལས་སྣ་ཚོགས་དང་། རོ་ལངས་བྱེད་པའི་ལས་རྣམས་དང་། ཡུལ་སུ་ན་དང་། གོག་ཡུལ་དང་། ཡུལ་ཀ་ཏ་ཡའི་མི་དང་། བ་ལང་རྫི་ལ་སོགས་པའི་རིའི་རྩེ་མོ་
