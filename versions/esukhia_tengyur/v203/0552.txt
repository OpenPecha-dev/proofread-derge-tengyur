ཚེ་ན་མི་སྐྲག་ཅིང་། །འདྲེས་པ་ན་ཡང་མི་འཐེན་ལ། །མདའ་དང་དབྱིག་པ་མདུང་སོགས་ཀྱིས། །བརྒྱབ་པ་ན་ནི་སྲན་ཆེ་བ། །དེ་འདྲ་བ་ཡི་རྟ་དག་ནི། །གཡུལ་གྱི་དུས་སུ་བསྔགས་པ་ཡིན། །རྣ་བར་བཟླས་པའི་ཚེ་ན་ནི། །མིག་ལས་མཆི་མ་འབྱིན་བྱེད་ཅིང་། །སྣ་སྦུབས་ཤིན་ཏུ་རྒྱས་བྱས་ནས། །
ཉི་མ་ལ་ཡང་བལྟ་བར་བྱེད། །རྐང་པ་གཡས་པ་ས་ལ་ནི། །ཡང་དང་ཡང་དུ་རྡེབས་པར་བྱེད། །འདི་ནི་སྐྱེ་བ་དྲན་པ་ཡི། །མཚན་ཉིད་དང་ལྡན་མཆོག་ཏུ་བཤད། །དེ་ལས་གཞན་པའི་རྟ་ངན་དག །ཤིང་རྟ་ལ་ཡང་སྦྱར་མི་བྱ། །ལེགས་ཐོས་རྣམ་པ་ཐམས་ཅད་དུ། །རྣ་བར་བཟླས་ལ་བརྟག་པར་
བྱ། །སྔོན་ཚེ་ང་ཡིས་ཇི་ལྟར་ཐོས། །རྣ་བར་བཟླས་པ་གང་གི་ནི། །རྟ་ནི་ཤིན་ཏུ་རིགས་ལྡན་འབྱུང་། །དོན་མེད་མི་འགྱུར་དེ་མཉན་གྱིས། །ཤིང་རྟ་ཡན་ལག་ཀུན་ལྡན་ཞིང་། །འཁོར་ལོ་ལེགས་པར་བཏགས་པ་དང་། །གདུགས་དང་རྒྱལ་མཚན་ལེགས་པར་བསྒྲེང་། །ཤིན་ཏུ་རྫོགས་པ་ཉིད་
དུ་སྦྱར། །ལེགས་པར་སྦྲེལ་ཞིང་ཤིན་ཏུ་ཕྱིས། །ཉི་མ་ལྟ་བུར་གཟི་མདངས་ལྡན། །རིགས་དང་མཚན་ཉིད་བཟང་བ་ཡིན། །རྟ་བཞི་དང་ནི་ཡང་དག་ལྡན། །ཁ་ལོ་བསྒྱུར་བ་བཟང་པོར་བཅས། །འབྲུག་སྒྲ་ལྟ་བུའི་སྒྲ་སྒྲོགས་པ། །ཤིང་རྟ་དེ་ནི་ལེགས་གནས་པས། །ཤར་ཕྱོགས་སུ་ནི་ཁ་
བསྟན་གཞག །གཡས་སུ་བསྐོར་ཞིང་བཀྲ་ཤིས་དང་། །ཤིས་པར་བརྗོད་པ་ལེགས་པར་བྱ། །དེ་ཡི་རྣ་བ་གཡས་པར་ནི། །སྔགས་ཀྱི་ཚིག་འདི་བརྗོད་པར་བྱ། །ཁྱོད་ནི་དྲི་ཟའི་རིགས་སུ་སྐྱེས། །རིགས་སུན་པར་ནི་མ་བྱེད་ཅིང་། །ཁྱོད་ནི་དངུལ་ལས་བྱུང་གྱུར་ཅིང་། །དཔལ་མོའི་ལག་ཏུ་
གནས་ན་འཕེལ། །མེ་ཡི་བུ་ཉིད་ཁྱོད་དྲན་བྱོས། །ཀོ་ཏུ་བ་ཡི་ནོར་བུའང་དྲན། །འོ་མའི་མཚེ་ལས་བདུད་རྩིའི་ཕྱིར། །ལྷ་དང་ལྷ་མིན་གྱིས་བསྲུབས་ཚེ། །དང་པོ་ཨུ་ཙ་སྲ་བ་བྱུང་། །དེ་ལས་ཁྱོད་སྐྱེས་དྲན་པར་གྱིས། །ཤུགས་མཆོག་སྟོན་དེ་ང་ཡིས་ནི། །དགྲ་དཔུང་མ་ལུས་ལས་རྒྱལ་གྱིས། །
ཚངས་པ་དང་ནི་ཟླ་བ་དང་། །ཆུ་ལྷ་ཡི་ནི་བདེན་ཚིག་དང་། །ཉི་མ་ཡི་ནི་གཟི་བརྗིད་དང་། །དེ་བཞིན་དྲང་སྲོང་དཀའ་ཐུབ་དང་། །སྟོབས་བཟང་གི་ནི་ཚངས་སྤྱོད་དང་། །དེ་བཞིན་རླུང་གི་ཤུགས་ཉིད་དང་། །མེ་ཉིད་ཀྱི་ནི་ནུས་མཐུ་ཡིས། །རྟ་ཁྱོད་རྒྱལ་བ་སྐྱེད་གྱུར་ཅིང་། །འཐབ་པའི་ཚེ་ན་རྟ་ཁྱོད་ཀྱིས། །
