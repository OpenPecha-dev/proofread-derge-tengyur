རིམ་པ་བཞིན་རྒྱུའོ། །ས་ལ་སོགས་པ་འབྱུང་བ་རེ་རེ་ལ་སུམ་བརྒྱ་བཞི་བཅུ་ཞེ་དགུའི་དབྱུག་གི་དབུགས་རྒྱུའོ། །སྣ་བུག་གཡས་རྒྱུ་ཉི་མ་གཡོན་རྒྱུ་ཟླ་བ། གཉིས་ཀ་རྒྱུ་བ་སྒྲ་གཅན། གཡས་ལྷོ་གཡོན་བྱང་། ཁྱིམ་འཕྲིན་ལས་རིང་ཐུང་རྣམས་གཞུང་ཉིད་ལ་གསལ་ལོ། །མིག་དམར་
མེ། སྤེན་པ་ཆུ། ཉི་མས། སྒྲ་གཅན་རླུང་། ལག་པ་ནམ་མཁའ། གཞན་རྣམས་ནི་སྒྲ་གཅན་ནམ་མཁར་བཞེད་དོ། །གཡས་པའི་འབྱུང་བ་དག་གཞུང་ན་གསལ་ལོ། །དེ་དག་གི་ལམ་ལྔའི་རིམ་པའོ། །སུམ་ཅུ་གཉིས་ཀྱི་མིང་ཡིག་ཤར་བ་དེ་བྱིས་པ། དེ་ནས་བགྲངས་པའི་གཉིས་པའི་གཉིས་པ་གཞོན། གསུམ་
པ་ལང་ཚོ། བཞི་པ་རྒན། དགའ་བ་རིམ་བཞིན་དེས་གོ་ཞིང་། ཞིབ་པར་གཞུང་ལས་ཤེས་སོ། །ལེའུ་དང་པོའོ།། །།གཙོ་བོར་དུས་སྦྱོར་སྟོན་པས། དེ་ཡང་ལོའང་དུས་སྦྱོར་སྟོན་པས་ཡིན་ཞིང་། འབྲས་བུས་ཞག་གཅིག་གི་དུམ་བུ་ལ་འབྱུང་བས། དེའི་རྩིས་རླུང་གྲངས་ལ་ལྟ་སྟེ།
ཞག་གཅིག་རྩིས་པ་ནས། ཁྱིམ་བདག་ཡན་ཆད་ནི། དུས་བཅུ་གཉིས་ཀྱི་འཁོར་ལོ་རྣམས་ཀྱི་དང་པོ་སྟེ་རྩ་བ་ཡིན་ནོ། །ཁྱིམ་དབང་བ་ཡང་འདིས་ཤེས་སོ། །དུས་སྦྱོར་ཀྱི་འཁོར་ལོ། འོག་མ་རྣམས་འདིའི་སྟེང་ནས་ལྟ་བ་ཡིན་ནོ། ཆུ་བདག་ངན་ཞེས་པ་ཡན་ཆད་ནི། དུས་སྦྱོར་གྱི་ནུས་པ་ཆེ་ཆུང་
ཕྱོགས་ཀྱི་སྟོབས་ཀྱི་ཙཀྲའོ། །ཟླ་བ་བཅུ་གཉིས་ཀྱི་ཉིན་མཚན་གྱི་དབྱུ་གུ་ནི་གོ་སླའོ། །ཉིན་མོ་དུ་དབང་བའི་ཆུ་ཚོད་ལ་བརྒྱད་ཀྱིས་བགོས་པས་ནོར་གྱི་ཐུན་ཕྱེད་དོ། །མཚན་མོ་ཡང་ནོར་ལྷས་བགོས་པས་ཐུན་ཕྱེད་ཅེས་ཏེ། འདི་ནི་མཚོན་ཆ་དར་ཅན་ལྟ་བུ་སྒྲ་གཅན་ལ་གལ་ཆེ། ཆུ་སྲང་ནི་བློས་སྤྱོད་
སྐར་མ་གཉིས་དང་། དབྱུ་གུ་བཅོ་ལྔས་ཁྱིམ་གཅིག་རྫོགས། དེས་གཞན་རྣམས་ཤེས་སོ། །གཞན་ཡང་ཞེས་པ་ནི་ཡན་ལག་ལྔ་རྩིས་ཀྱི་ཉིན་སྐར་དག་པ་ལ་ཟེར་རོ། །འདི་ཕྱོགས་ནི་ཟླ་བ་ལ་དབང་བའི་ཁྱིམ་བཅུ་གཉིས་ཀྱི་ཉི་མའི་འཕོ་བའོ། །ལུག་ལ་སོགས་པར་འཕོས་པའི་འདས་པའི་
ཞག་དུ་ལོན་གནས་གཉིས་སུ་བྱ་སྟེ། ཉི་མ་ཞག་རེ་ཞིང་ཆུ་ཚོད་བགོ་ཞིང་ཆུ་སྲང་ཉི་ཤུ་རྩ་ལྔ་ཟ་བས་སྟེང་བཞི་འོག་ཉི་ཤུ་རྩ་ལྔས་བསྒྱུར་རོ། །དྲུག་ཅུས་བགོས་ལྷག་དབྱེ། ནོར་སྟེང་དུ་གཞུག །ཕྱེད་ལྷག་ན་གཅིག་སྦྱིན་ནོ། །སྟེང་དུ་དྲུག་ཅུས་བགོ་བར་བྱ་ཞིང་། ནོར་གྱིས་ཆུ་སྲིན་ལ་སོགས་བྱང་བགྲོད་
