བཏུང་བ་བླུད། །སྣུམ་གྱི་ཆོ་གས་རོ་སྟོད་ཀྱི། །ནད་ནི་སེལ་བར་བྱེད་པ་ཡིན། །དེ་བཞིན་ལུས་ཀྱི་རོ་སྨད་ཀྱི། །ནད་ནི་འཇམ་རྩིས་སེལ་བར་བྱེད། །སྦེག་དང་གཞོན་ནུ་རྒན་པོའི་དང་། །རྟ་གང་སྟོབས་ནི་ཟད་པ་དང་། །ལས་ཆེན་དག་ལ་སྦྱར་བ་དང་། །དེ་བཞིན་ལུས་
ཟུངས་ཟད་པ་དང་། །རོ་སྨད་ཀྱི་ནི་ནད་རྣམས་དང་། །ནད་ནི་ཚིགས་སུ་ཞུགས་པ་དེ། །འདི་དག་གིས་ནི་ཉེན་གྱུར་པ། །འཇམ་རྩི་བཏང་ན་བདེ་འཐོབ་འགྱུར། །ཁྱད་པར་དུ་ནི་སོས་ཀ་སྦྱར། །འཇམ་རྩི་ཚད་ཙམ་བཏང་བྱས་ཏེ། །འདི་ནི་སྦྱོར་བ་རྗེས་མཐུན་པའི། །འཇམ་རྩི་
སྟོབས་ནི་སྐྱེད་བྱེད་ཡིན། །རྟ་ཡི་བཅུད་ཀྱིས་ལེན་དང་ནི། །རོ་ཙ་རྟས་བྱེད་ཉིད་དུ་འགྱུར། །སྔ་དྲོ་ཁྲུས་བྱས་སྐོམ་བླུད་ལ། །བཟའ་དང་བཅའ་བ་བྱིན་གྱུར་ནས། །སྔར་བཤད་པ་ཡི་ཆོ་ག་ཡིས། །ལེགས་པར་བཏང་ན་བདེ་འཐོབ་བྱེད། །རྩ་བ་ལྔ་པ་གཉིས་ཁུ་བ། །དེ་ལ་རྗེས་སུ་
བཏུང་ཕྱིར་བླུད། །རམ་ཉེ་ཀ་རྟི་ཀ་ཀོ་ལ། །བི་ཛ་ཀ་དང་རི་ཤ་བ། །མ་ཤ་མུདྒ་པ་རྞི་གཉིས། །མེ་དང་དེ་བཞིན་སླེ་ཏྲེས་དང་། །སྟག་པ་ཚེར་དང་མ་དུ་ཀཾ །ནཱ་ག་གེ་སར་པ་ཏ་ཀ །པྲི་ཡང་ཀུ་དང་ག་དུར་དང་། །འོ་མ་ཅན་ཤིང་ཤུན་སྤང་སྤོས། །པདྨའི་རྩ་བ་སྡོང་བུར་བཅས། །ཀ་ཤེ་རུ་
ཀ་ཤྲི་ག་ཏ། །ཀ་ར་བཱི་ར་ཨུཏྤ་ལ། །པུ་ཤེལ་ཙེ་དང་གསེར་པྲམ་དང་། །ཙ་ནྡན་དང་ནི་པ་ཡས་སྱ། །ཤ་རི་བ་དང་པྲ་ཏི་དང་། །ཏིལ་ནག་དང་ནི་བཙོད་དང་ནི། །ཤུག་པ་སྲིང་སྣ་ཤ་ར་བ། །འདི་རྣམས་ལྡེ་གུར་བསྐོལ་བའི་ཆུ། །འོ་མ་དང་ནི་ལེགས་སྦྱར་བ། །ཚགས་ཀྱིས་ལེགས་
པར་བཙགས་པ་ལ། །ཁ་ར་དང་ནི་སྦྲང་རྩི་བཏབ། །བསིལ་ཞིང་སྙིགས་མ་མེད་གྱུར་པ། །མར་དང་ལྷན་ཅིག་བསྲེས་ནས་ནི། །བཤང་གཅི་རྣམ་པར་བསལ་གྱུར་པ། །ཤར་དུ་བསྟན་དེ་མགོ་ལྡོག་ནས། །དྲང་བར་གཞག་པར་གྱུར་པ་ལ། །རྔ་མ་གཡས་ནས་བཟུང་ནས་ནི། །བློ་དང་
ལྡན་པས་མ་བལྟབས་པར། །ལན་གསུམ་མམ་ནི་ལན་ལྔ་གཏང་། །འོ་མའི་འཇམ་རྩིས་བསལ་བའི་ནད། །སོས་ཀ་སྦྱར་ཚེ་བསལ་བ་མཆོག །མཁྲིས་པ་ལས་བྱུང་ནད་རྣམས་དང་། །དེ་བཞིན་རྟ་ཡི་ཁྲག་ནད་དང་། །རིང་ནས་འོངས་དང་ལས་ཀྱིས་དུབ། །དབུགས་རྒོད་མགོ་
