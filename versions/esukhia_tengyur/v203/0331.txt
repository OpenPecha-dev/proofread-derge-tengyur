དགྲ་ཡིན། །དེ་ལྟར་འདི་ནི་དྲེའུ་སྟེ། །མ་འདྲེས་འཇུག་ལས་རྟ་ཉིད་ཡིན། །རང་དང་རྟ་ཡི་སྐྱེ་གནས་ཏེ། །བོང་བུའི་སྐྱེ་གནས་གཉིས་སུ་འདོད། །རྟ་རྣམས་གཞན་གྱི་མངལ་དུ་ནི། །འཁྲིག་པ་སྤྱོད་པར་མི་བྱེད་ལ། །གལ་ཏེ་བྱེད་པར་གྱུར་ན་ནི། །ཡུལ་དེ་འཇིག་པར་
ཤེས་པར་བྱ། །དེ་ལྟར་ང་ནི་རྟ་འབྱིན་ལ། །མཆོད་སྦྱིན་དོན་ནི་ཕྱུགས་ཀྱི་མཆོག །དགྲ་འཇོམས་རྒྱས་བྱེད་ལ་སོགས་པ། །རྒྱལ་པོ་རྣམས་ཀྱི་དཔུང་གི་མཆོག །དེ་ནས་ཁ་དོན་ལ་དགའ་ཞིང་། །སྦྱིན་སྲེག་བྱེད་ལ་རབ་གཞོལ་བ། །གཞོན་ནུ་ཚངས་པར་སྤྱོད་ལ་གནས། །ཤ་ལི་ཧོ་ཏྲ་
དཀའ་ཐུབ་ཆེ། །ཁ་བ་ཅན་གྱི་ངོས་ཀྱི་ནགས། །ལྷ་ཡི་ཚལ་འདྲར་འོངས་པར་གྱུར། །དེ་ནི་དེ་རུ་གནས་བཅས་ནས། །རི་ཡི་ཚོགས་ནི་ཉམས་དགའ་ཞིང་། །{𰶮རི་དགས་,རི་དྭགས་}ཚོགས་ནི་དུ་མས་ཁྱབ། །མིའམ་ཅི་ནི་སྣ་ཚོགས་གནས། །བྱ་ཚོགས་དུ་མ་སྒྲ་སྒྲོགས་ཤིང་། །དྲི་ཟའི་བུ་མོ་རྣམས་ཀྱིས་བསྐོར། །
གྲུབ་དང་དྲང་སྲོང་རོལ་པ་ཡི། །ཚོགས་དང་ལྷ་རྣམས་དག་གིས་མཛེས། །མེ་ཏོག་འབྲས་བུ་མང་པོ་ཡིས། །ཉམས་དགའ་ནེའུ་གསེང་བཟང་པོས་ཁྱབ། །ཤིན་ཏུ་ཡིད་དུ་འོང་བའི་གནས། །ཉམས་དགའ་དེར་ནི་གནས་པར་གྱུར། །དེར་ནི་ཕྱེ་ལེབ་ཡན་ཆད་ཀྱི། །{𰶯ཡི་དགས་,ཡི་དྭགས་}
ལ་སོགས་སྣ་ཚོགས་གནས། །རྟ་དང་དྲི་ཟ་མིའམ་ཅི། །ནེའུ་ལེ་སྦྲུལ་དང་བྱ་རྣམས་ནི། །སེང་གེ་སྟག་དང་ཕག་དང་ནི། །གླང་པོའི་དབང་པོ་སྤྲེའུ་དང་དོམ། །དེ་ལ་སོགས་པ་མང་པོ་ནི། །ཤ་ལི་ཧོ་ཏྲ་དྲང་སྲོང་ཆེ། །བདག་ཉིད་ཐུལ་ཅིང་གཙང་བ་ཡིས། །གནས་དང་ཉེ་བར་
འཁོད་པ་ན། །ཤ་ལི་ཧོ་ཏྲས་རྟ་དེ་རྣམས། །བུ་དང་འདྲ་བར་ཡོངས་སུ་བསྐྱང་། །ཁ་བ་ཅན་གྱི་ངོས་ན་རྩེ། །ཤ་ལི་ཧོ་ཏྲའི་དབང་དུ་གྱུར། །དེ་བས་ལྷ་རྣམས་མྱུར་དུ་ནི། །ཤ་ལི་ཧོ་ཏྲའི་གནས་སུ་སོང་། །དེར་ནི་ཁྱེད་ལ་བཞོན་པ་མགྱོགས། །ཤ་ལི་ཧོ་ཏྲ་སྟེར་བར་འགྱུར། །ལྷ་
མཆོག་ཚངས་པས་དེ་སྐད་དུ། །ལྷ་རྣམས་ལ་ནི་སྨྲས་པ་དང་། །དགའ་བས་དེ་ལྟར་བགྱི་ཞེས་ཏེ། །དེ་རྣམས་རི་མཆོག་དེ་རུ་ཆས། །དེ་ནས་སྐད་ཅིག་ཙམ་གྱིས་ནི། །བརྒྱ་བྱིན་གཤིན་རྗེ་ཆུ་ལྷ་དང་། །དཔལ་ལྡན་ནོར་གྱི་བདག་པོ་དང་། །བཅས་པ་དྲང་སྲོང་གནས་སུ་སོང་། །ཕྱིན་ནས་
