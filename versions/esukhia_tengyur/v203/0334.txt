ལས་ནི་མེ་ཕྱུང་སྟེ། །སྦྱིན་སྲེག་ང་ཡིས་བྱས་ནས་ནི། །ལྷ་དང་སྲིན་པོ་ཚིམ་པར་བྱས། །དེ་ནས་ཡུལ་དེའི་ས་མཚམས་ཀྱི། །ཉེ་འཁོར་ཀུན་དུ་ཁྱབ་ནས་ནི། །རྟ་དེ་དག་གིས་འབྲུ་དག་ནི། །ཐམས་ཅད་ཟ་བ་ཉིད་དུ་འགྱུར། །དེ་ནས་འབྲུ་དེ་བརླག་གྱུར་པ། །བྲམ་ཟེས་མཐོང་
ནས་ཁྲོས་པ་ཡི། །ཡིད་ཀྱིས་མིག་ནི་དམར་པོ་དང་། །དེ་ལ་ཚིག་རྩུབ་འདི་སྐད་སྨྲ། །སྔོན་ནི་དཀའ་ཐུབ་ལ་གནས་པའི། །བྲལ་ཟེ་གཞན་དག་འདིར་འོངས་ཚེ། །དཀའ་ཐུབ་ཚངས་པར་སྤྱོད་པ་དང་། །འདོན་དང་སྦྱིན་སྲེག་བྱེད་པ་ཡིས། །ལོ་གྲངས་མང་པོར་འདིར་གནས་ཤིང་། །
དེ་བཞིན་ཡམས་ཀྱང་ཞི་བྱེད་ན། །ཁྱེད་ནི་{𰶵རི་དགས་,རི་དྭགས་}འདི་འདྲ་བ། །ཁྱེར་ཏེ་ཡུལ་དུ་ཇི་ལྟར་འོང་། །ལམ་གང་ནས་ནི་ཁྱེད་འོངས་པ། །དེ་ལས་སླར་ནི་འགྲོ་བར་གྱིས། །ཁྱེད་ཀྱི་ཡམས་འདི་བདག་ཅག་གིས། །བཟློག་པར་ནུས་པ་མ་ཡིན་ནོ། །བྲམ་ཟེ་ཁྲོས་པ་དེ་ལ་ངས། །བཟོད་གསོལ་
སྔོན་འགྲོའི་ཚིག་འདི་སྨྲས། །རྟ་རྣམས་འབྲུ་ནི་ཟ་བ་དང་། །ཁྱེད་ལ་འཚེ་བར་ང་མི་དགའ། །སྦྱིན་སྲེག་བྱེད་པས་ང་བྲེལ་བས། །ཁྱེད་ལ་གནོད་པ་ངས་མ་བཀག །དེ་ནས་ང་ཡིས་རྟ་རྣམས་ནི། །ཐམས་ཅད་བསྡུས་ནས་འདི་སྐད་སྨྲས། །ང་ལ་ཡུལ་གྱི་མི་རྣམས་ནི། །ཅི་སྐད་སྨྲས་གྱུར་
པ་ལ་ལྟོས། །འདི་དག་མགུ་བར་བྱ་བའི་ཕྱིར། །རེ་ཞིག་ཁྱེད་ནི་ལེགས་པར་བཅིང་། །དེ་ནས་མཚན་མོར་གྱུར་པ་དང་། །སླར་ཡང་ཅི་དགར་སྤྱད་པར་གྱིས། །དེ་ནས་བཅིངས་པའི་ཚོར་པ་ནི། །མ་མྱོང་པས་ན་བཅིང་པར་ལྷགས། །ལྷ་རྣམས་ལྟད་མོ་སྐྱེས་པ་ཡི། །ཡིད་ཀྱིས་ང་ལ་ཡིད་
ཆེས་འགྱུར། །དེ་ནས་ཆུ་ལྷ་རང་ཉིད་ནི། །བོས་ནས་ཞགས་པས་རྟ་རྣམས་ནི། །ཡན་ལག་ལྔར་བཅས་ཤིང་ལ་ནི། །{𰶶གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བཅིངས་པར་གྱུར། །བཅིངས་པའི་རྗེས་ལ་སྟོབས་ཡོད་ཀྱང་། །འགུལ་བར་གྱུར་པ་མ་གྱུར་ཏོ། །ཉེ་བར་བཅིང་པ་བྱས་ནས་ནི། །མཚོན་ནི་
རྣོན་པོ་ཐོགས་ནས་སུ། །ལྷ་ཡིས་མདུན་དང་རྒྱབ་ཀྱི་ནི། །རྟ་རྣམས་ཀྱི་ནི་གཤོག་པ་བྲེགས། །དེ་ནས་གཤོག་པ་བཅད་ཀྱང་ནི། །ཕྱི་ཕྱིར་ཞིང་ནི་སྐྱེ་བར་གྱུར། །དེ་རྗེས་མེ་ཡིས་བསྲེགས་པས་ན། །ཚིག་པས་མེ་ལྕེ་བྱུང་བར་གྱུར། །བཅིངས་ཤིང་གཤོག་པ་མེད་བྱས་ནས། །གང་ཚེ་
