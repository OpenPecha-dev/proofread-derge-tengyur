དང་། ལམ་གྱི་བཞི་མདོར་ཁྱི་ཁ་གྱེན་དུ་བལྟས་ཏེ་ངུ་བར་བྱེད་ན་དེ་ནི་ཆོམ་རྐུན་དང་། མེའི་འཇིགས་པ་འོང་བར་སྟོན་པ་ཡིན་ནོ། །ཡང་ཁྱི་དང་། བྱ་རོག་རྒྱལ་པོའི་ལམ་དང་། ལམ་གྱི་བཞི་མདོར་ཐང་ལ་གཅིག་ཏུ་འདུས་ནས་གལ་ཏེ་སྒྲ་ཆེན་པོ་འབྱིན་པར་བྱེད་པ་དང་། ཅི་དགར་
འགྲོ་བར་བྱེད་པ་དེ་ནི་ཕ་རོལ་གྱི་དམག་ཚོགས་འོང་བ་ཡིན་ནོ། །གང་གི་ཚེ་བྱ་རོག་དང་། བུང་བ་དང་། སྦྲང་བུ་རྣམས་ར་བ་དང་། གྲོང་ཁྱེར་གྱི་སྒོ་དང་། ཐང་དང་། ལམ་གྱི་བཞི་མདོར་སྦྲང་རྩི་ཚང་དུ་བྱེད་པར་གྱུར་ན་གློ་བུར་གྱི་འཇིགས་པ་དང་། རྒྱལ་པོའི་འཇིགས་པ་འབྱུང་བར་རིགས་པར་བྱའོ། །དེ་བཞིན་
དུར་བ་དང་། ཁྱིམ་གྱི་མདུན་དག་ཏུ་ཡང་དེ་ལྟ་བུར་སྣང་ན། དེ་བཞིན་དུ་འཇིགས་པ་འབྱུང་བར་རིག་པར་བྱའོ། །དེ་ལ་གློ་བུར་གྱི་འཇིགས་པ་འོང་ཞེས་བྱ་བ་ནི་ཕ་རོལ་གྱི་དམག་ཚོགས་འོང་བ་དང་། མེ་དང་ཆོམ་རྐུན་གྱི་འཇིགས་པ་དག་འབྱུང་བར་སྟོན་པ་ཡིན་ནོ། །ཡང་གང་དུ་སྲོག་ཆགས་ཕྲ་མོ་དག་ལ་ལ་
རབ་དང་། གྲོང་ཁྱེར་གྱི་སྒོ་དང་། ཡང་ཐོག་དང་། སྒོ་ཁང་ལ་འཕྱང་མོ་ཉུག་ཅིང་འཛེགས་པར་སྣང་བ་དེར་ནི་ཕ་རོལ་གྱི་དམག་ཚོགས་ཀྱིས་རིང་པོ་མི་ཐོགས་པར་གྲོང་ཁྱེར་མྱུར་དུ་འཇིགས་པར་འགྱུར་རོ། །གང་གི་ཚེ་ཁྱི་དང་། ཝ་དང་། བྱི་ལ་ལ་སོགས་པ་དང་། བྱ་སྣ་ཚོགས་དག་དུས་མཚམས་མེད་པར་
འཁྲིག་པ་དང་ལྡན་པའི་ཚེ་ནི་རྒྱལ་པོ་དང་། ཡུལ་རྟག་ཏུ་འཇིགས་པར་བསྟན་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་མི་དང་། རྟ་དང་། རྔ་མོ་དང་། བོང་བུའི་རིགས་དང་། བ་ལང་མོ་དང་། བྱ་དང་། {𰵕རི་དགས་,རི་དྭགས་}རྣམས་དང་། ཁྱི་དང་། བྱི་བ་དང་། སྦྲུལ་དག་དང་། སྲོག་ཆགས་ཕྲ་མོ་གཞན་དག་གི་ཡང་མངལ་ཕྱིན་ཅི་
ལོག་ཏུ་གྱུར་པ་དེར་ནི་ཡུལ་རྣམས་འཁྲུག་པ་དང་། རྒྱལ་པོ་ཡང་འཆི་བར་འགྱུར་རོ། །ཡང་གང་དུ་བྱ་དང་། {𰵖རི་དགས་,རི་དྭགས་}<𰵀གྱི་,ཀྱི་>རིགས་རྣམས་དང་། ཁུང་ན་ཉལ་བ་རྣམས་ལོག་པར་འཁྲིག་པ་དང་ལྡན་པར་སྨྲ་བ་དེ་ནི་རྒྱལ་པོ་འཆི་བར་འགྱུར་རོ། །བྱ་རོག་གི་གཤོག་པ་དཀར་པོ་མཐོང་ན་མི་དགེ་བ་ཡིན་ཏེ། ཡུལ་
དེ་ནས་འགྲོ་བར་བྱའོ། །དེ་བཞིན་དུ་བྱ་རྒོད་ལ་སོགས་པའི་གཤོག་པ་ལྷུང་བར་མཐོང་བ་དང་། ཝ་ངལ་མི་སོས་པར་བརྒྱལ་བར་འགྱུར་བ་དེར་ནི་ཡུལ་གྱི་བདག་པོ་དང་། རྒྱལ་པོ་དག་འཆི་བའི་ལྟས་སུ་བསྟན་ཏོ། །ཡང་གང་གི་ཚེ་གྲོང་ཁྱེར་དུ་ཁྭའི་ཚོགས་འདུས་ནས་གྲོང་གི་ཕྱི་རོལ་དུ་འདོང་བ་དེ་ནི་ངེས་པར་
