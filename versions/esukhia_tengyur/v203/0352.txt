དེ་ལྟར་སྐྱེ་གནས་འདྲ་ཡི། །རང་རང་བཞིན་ཅི་བཞིན་རྣམ་པར་དབྱེ། །དེ་བཞིན་སྤྱོད་ཚུལ་མི་མཐུན་པ། །ཕྱན་སྡིག་པས་གོས་པར་འགྱུར། །དེ་ལྟར་ལེགས་ལྡན་རྗེས་སྟོན་པ། །ཤ་ལི་ཧོ་ཏྲས་བཤད་པ་ཡིན། །ཤ་ལི་ཧོ་ཏྲའི་མདོའི་གནས་ལས་རྟའི་རིགས་བཞིའི་
ལེའུ་སྟེ་བརྒྱད་པའོ།། །།དེ་ལ་རྟ་ཡི་མཚན་ཉིད་ནི། །ཡན་ལག་མ་གཏོགས་ཐ་དད་དེ། །གདོང་ནི་སྲབ་ལ་རྒྱ་ཆེ་ཞིང་། །ཟུར་བཞི་པ་ལ་གཡུལ་སྙོམས་པ། །དྲང་ཞིང་འཁྱོག་པ་མ་ཡིན་ལ། །ཡོངས་སུ་ཚད་བ་བསྔགས་པ་ཡིན། །ཁ་ནི་བཙུམས་

དབྱེ་སླ་ཤ་རྒྱས་དབྱིབས་ལེགས་དང་། །ཤིང་ཏུ་གཡོ་དང་འཕྱང་བའམ་མིན། །སོ་ནི་ཁེབས་པར་གྱུར་བ་དང་། །མཉམ་ཞིང་སྙོམས་ལ་ཟླུམ་པ་དང་། །ཚད་ལྡན་དྲང་ན་བསྔགས་པ་ཡིན། །སྣ་གུང་མཉམ་པར་དབུགས་རྒྱུ་ཞིང་། །ཟླུམ་ལ་ནང་ན་ཡངས་པར་
