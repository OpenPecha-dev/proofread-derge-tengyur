དང་ལྕགས་སྒྲོག་དང་དྲིལ་བུ་རྣམས་ཀྱི་ཕྱག་མཚན་བྲི་བར་བྱའོ། །དེའི་ཤར་གྱི་སྣམ་བུ་ལ་པདྨ་དང་ནགས་ཀྱི་ལྕུག་མ་དང་བགྲང་ཕྲེང་དང་གུནྡྷེ་དང་། ཀྲྀཥྞ་ས་རའི་པགས་པ་བྲིའོ། །ལྷོའི་སྣམ་བུ་ལ་རྩེ་གསུམ་དང་། ཟླ་བ་ཕྱེད་པ་དང་། རྒྱལ་མཚན་དང་། རལ་པའི་ཐོར་ཚུགས་དང་། མིག་
ཅེས་པ་ལྔའོ། །ནུབ་ཀྱི་སྣམ་བུ་ལ་འཁོར་ལོ་དང་། རལ་གྲི་དང་། ཕུབ་དང་། མདུང་དང་། བེ་ཅོན་ཏེ་ལྔའོ། །བྱང་གི་སྣམ་བུ་ལ་མདུང་རྩེ་དང་། དྲིལ་བུ་དང་། བ་དན་དང་། རྨ་བྱ་དང་། ར་སྐྱེས་ཏེ་ལྔ་བྲི་བར་བྱའོ། །དེ་ལྟ་བུའི་མེའི་ཐབ་ཀྱི་དབུས་སུ། འཁོར་ལོ་དང་། རྡོ་རྗེ་དང་། རིན་
པོ་ཆེ་དང་། པདྨ་དང་། སྣ་ཚོགས་རྡོ་རྗེ་སྟེ་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་དེ་བཞིན་གཤེགས་པའི་མཚན་མས་བརྒྱན་པར་བྱས་ལ་ལས་ཐམས་ཅད་པ་ནི་རྡོ་རྗེ་ཕྲེང་བས་རྣམ་པར་བརྒྱན་པར་བྱའོ། །ཡང་ན་ལས་ཐམས་ཅད་པ་ནི་ཟླུམ་པོ་ཁོ་ན་སྟེ། འོན་ཀྱང་ནང་གི་པདྨའི་ལྟེ་བར་སྣ་ཚོགས་རྡོ་རྗེ་
དང་། འདབ་མ་རྣམས་ལ་རྡོ་རྗེ་བརྒྱད་བྲིས་ལ། ལྷག་མ་ནི་སྔ་མ་བཞིན་ནོ། །དཀྱིལ་འཁོར་རམ་ལྷའི་ཁང་པར་ཤར་ཕྱོགས་སུ་ཞི་བའི་མེ་ཐབ་དང་། དེ་བཞིན་དུ་ལྷོ་དང་། ནུབ་དང་བྱང་ཕྱོགས་སུ་རྒྱས་པ་དང་དབང་དང་དྲག་པོའི་མེ་ཐབ་བྱའོ། །ཡང་ན་ལས་དང་རྗེས་སུ་མཐུན། །
རྡུལ་ཚོན་གྱིས་ནི་མེ་ཐབ་བྲི། །དེ་ཕྱིར་བརྐོས་དང་རྡུལ་ཚོན་གྱིས། །མེ་ཐབ་རྣམ་པ་གཉིས་སུ་འདོད། །ཅེས་འབྱུང་ངོ་། །དེ་ནི་མེ་ཐབ་ཀྱི་མཚན་ཉིད་དོ། །ཡམ་ཤིང་རྣམས་ཀྱི་མཚན་ཉིད་ནི། །བླང་དོར་དབྱེ་བས་བཤད་པར་བྱ། །བ་ལ་ཤ་<𰵊དུ་མ་,དུམ་>ཨ་མྲ་དང་། ཀ་དམ་པ་དང་ཀས་མི་རྒྱ། །ཨ་
<𰵋ཤྭང་,ཤྭད་>པླག་ཤ་ལས་འབྱུང་བ། །ཤ་མི་ཁ་དང་འོ་མ་ཅན། །འོ་མ་དང་བཅས་རློན་པ་སྟེ། །མཉམ་པོར་བཅད་པར་འདོད་པ་ཡིན། །མཐོ་གང་ཙམ་གྱི་ཚད་དང་ལྡན། །སྦྲང་རྩི་འོ་མ་མར་གྱིས་བྱུགས། །དཀར་པོའི་དྲི་ཆུས་བསྐུས་པ་ཡི། །འོ་མ་དང་བཅས་ཡམ་ཤིང་འདོད། །ཇི་སྐད་དུ།
རྒྱས་པར་སོར་ནི་བཅུ་བཞི་པ། །ཞི་བར་སོར་ནི་བཅུ་གཉིས་ཏེ། །དབང་ལ་སོར་ནི་བཅུ་པ་དང་། །སྡང་དང་དགར་བ་སོར་བརྒྱད་པའོ། །རྙིང་ཞིང་ཆག་པ་འབུས་ཟོས་པ། །འབད་དེ་རྣམ་པར་སྤང་བར་བྱ། །མཆོད་སྦྱིན་པ་ཡི་ཤིང་རྣམས་ཀྱི། །སྲོག་ཆགས་མེད་པའི་བུད་ཤིང་གང་། །མེ་
