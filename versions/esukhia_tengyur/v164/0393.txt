སྤུ་ཅན་བྱིན་གྱིས་མ་བརླབས་ཞེས་འབྱུང་བས་ན། ལ་བ་དང་བེའུ་རས་ནི་སྤུ་བྲེག་པ་ལ་བྱིན་གྱིས་རློབས་ཤིག་པའོ། །མལ་སྟན་མི་གཙང་བས་རེག་ན་དེའི་མོད་ལ་བཀྲུ་ཞེས་པ་ནི་འདི་ལྟར་གྱུར་ན་མ་བཤོལ་བར་མོད་ལ་ཁྲུས་ཤིག་པ་སྟེ། དགེ་འདུན་གྱི་ཡང་རུང་གང་ཟག་
གི་ཡང་རུང་བའོ། །དགེ་འདུན་གྱི་ཁྲིའུ་དང་། ཁྲི་ངན་པ་མ་ཡིན་ན་འདུག་པའི་བར་དུ་དེ་མ་ཡིན་པའི་གདིང་བ་མེད་པར་མི་བྱ་བ་དང་། རྔུལ་གཟན་མེད་པར་ཡང་བགོ་བ་ཡང་མི་བྱ་ཞེས་པ་ནི་ཁྲི་སྟན་ངན་པ་རབ་མ་ཡིན་ན་དེ་ལ་གང་ཟག་གི་གདིང་བ་མེད་པར་འདུག་པ་ཡང་མི་བྱ་ན། །ཉལ་
བ་ལྟ་ཅི་སྨོས་པ་དང་། ཉིད་ཀྱི་རྔུལ་གཟན་མེད་པར་མལ་གོས་སུ་བགོ་བ་ཡང་མ་སྤྱོད་ཅིག་པའོ། །རི་མོ་ཕལ་ཆེར་མ་ཉམས་པ་ཐམས་ཅད་མ་ཡིན་ཞེས་པ་ནི་མལ་སྟན་རི་མོ་ཅན་ཞིག་ལ་རི་མོ་ཕལ་ཆེར་བྱེད་དམ། ཉམས་པ་མ་ཡིན། གང་ཟག་གམ། དགེ་འདུན་སུའི་ཡང་རུང་སྟེ།
དེ་ལ་གདིང་བ་མེད་པར་མ་སྤྱོད་ཅིག་པའོ། །ཁྲལ་ཀ་དང་དྲི་མ་ཅན་ཡང་མ་ཡིན་ཞེས་པ་ནི་གདིང་བ་ཁྲལ་ཀོག་གམ་དྲི་མ་ཅན་གྱིས་ཀྱང་མ་བྱེད་ཅིག་པའོ། །གདིང་བ་བཅང་ཞེས་པ་དང་། སྟུག་ན་ཅིག་རྐྱང་དགེ་ཞེས་པ་དང་། སྲབ་ན་ཉིས་རིམ་དུའོ་ཞེས་པ་དང་། ཁྲ་བོ་ནི་མི་རུང་ཞེས་པ་དང་། འདིར་འདབ་ངོས་
བྱ་ཞེས་པ་དང་། འཕོངས་ཀྱི་སུམ་ཆར་རོ་ཞེས་པ་དང་། རྐྱང་པ་ལ་བྱིན་གྱིས་མི་བརླབ་ཅེས་པ་ནི་གདིང་བའི་ཚུལ་འདི་དག་ལྟར་གྱིས་ཤིག་པ་སྟེ། འཕོངས་ལོགས་ཀྱི་སུམ་ཆར་ཚལ་བུ་ཕྲེད་དུ་འགྲོ་བས་མ་གླན་པར་བྱིན་གྱིས་བརླབ་ཏུ་མི་རུང་བའོ། །དེ་ཡོད་ན་གཡན་པ་བགབ་པ་
བཅང་ཞེས་པ་དང་། དེ་ནི་ཞག་ལྔའམ་དྲུག་ཅིང་བཀྲུ་ཞེས་པ་ནི་གཡན་བ་འམ་ཤུ་བ་ཡོད་ན་དེའི་དགབ་པ་ཆོངས་ཤིག་པ་དང་། ཞག་ཚིགས་འདི་ཙམ་ཞིང་ཁྲུས་ཤིག་པའོ། །སྲིན་བལ་དང་བལ་དང་། ཟར་མ་དང་། ཤ་ན་ཀའི་རུང་ཞེས་པ་ནི་རྒྱུ་འདི་དག་ལས་བྱས་པ་ནི་བྱིན་
གྱིས་བརླབ་པ་དང་། ལྷག་པའི་གོས་སུ་རུང་བར་སྟོན་པའོ། །མ་བཏགས་པ་དང་གསོ་མའི་མ་ཡིན་ཞེས་པ་ནི་བྱང་གི་སྒྲ་མི་སྙན་ནམ། ལྷའི་གོས་ལྟ་བུ་རྒྱུ་སྤུན་མེད་པ་དང་། གསོ་མ་ལས་བྱས་པ་ནི་དགེ་སློང་གི་གོས་སུ་མི་རུང་བའོ། །སྐྲ་ལས་བྱས་པ་དང་། གཅེར་བུ་
