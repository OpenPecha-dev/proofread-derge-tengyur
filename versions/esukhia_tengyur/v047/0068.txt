རྒྱས་ཆོས་ཆེན་པོས། །རྡོ་རྗེ་སེམས་དཔའ་དབང་བསྐུར་ལྟར། །དེ་བཞིན་བདག་ཀྱང་དབང་བསྐུར་ཏེ། །བདག་ལ་དོན་མཆོག་སྩལ་དུ་གསོལ། །གཙོ་ལ་ཡོན་སོགས་རབ་ཏུ་དབུལ། །གཏང་སེམས་ཡང་དག་བསྐྱེད་པར་བྱ། །དེ་ཡི་སྙིང་གར་རྡོ་རྗེ་ནི། །མེ་ཏོག་བཅས་པའི་སྙིང་པོར་བཞག །
ཨོཾ་སརྦ་ཡོ་ག་ཙིཏྟ། ཨུཏྤཱ་ད་ཡཱ་མི་སུ་ར་ཏ་ས་མ་ཡ་སྟྭཾ་ཧོ་སིདྡྷི་བཛྲ་ཡ་ཐཱ་སུ་ཁཾ། དེ་རིང་ཁྱོད་ལ་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བྱིན་གྱིས་བརླབ་པར་འགྱུར་རོ། །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་དམ་པའི་གསང་ཆེན་འདི། དཀྱིལ་འཁོར་དུ་མ་ཞུགས་པ་རྣམས་ལ་སྨྲ་བར་མི་
བྱའོ། །དད་པ་མེད་པ་ལ་ཡང་མི་རུང་ངོ་། །ཞེས་བརྗོད་པར་བྱའོ། །ཐོ་སོགས་སྔགས་ཀྱིས་དྲང་བྱ་ཞིང་། །ཡི་གེ་གསུམ་གྱིས་གཞུག་པར་བྱ། །ནང་དུ་སྡོམ་པ་འཛིན་དུ་གཞུག །དེ་ནས་ཚིག་འདི་བརྗོད་པར་བྱ། །དེ་རིང་ཁྱོད་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་རིགས་ལ་རབ་ཏུ་ཆུད་དོ། །དེ་བས་
བདག་གིས་ཁྱོད་ཀྱི་རྡོ་རྗེ་ཡེ་ཤེས་བསྐྱེད་པར་བྱའོ། །ཡེ་ཤེས་གང་གིས་ཁྱོད་ལ་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་དངོས་གྲུབ་ཀྱང་ཐོབ་པར་བྱ་སྟེ། གཞན་ལྟ་སྨོས་ཀྱང་ཅི་དགོས། དངོས་གྲུབ་འདི་ནི་ཁྱོད་ཀྱིས་དཀྱིལ་འཁོར་མ་མཐོང་བ་རྣམས་ཀྱི་མདུན་དུ་སྨྲ་བར་མི་བྱའོ། །ཁྱོད་ནི་དམ་ཚིག་ལ་
འདའ་བར་མི་རུང་ངོ་། །དེ་རིང་རྡོ་རྗེ་སེམས་དཔའ་ཉིད། །ཁྱོད་ཀྱི་སྙིང་ལ་ཡང་དག་གནས། །གལ་ཏེ་ཚུལ་འདི་སྨྲས་ན་ནི། །དེ་མ་ཐག་ཏུ་དྲལ་ཏེ་གཤེགས། །ཨོཾ་བཛྲ་སཏྭ་ཧཱུྃ། པད་གནས་ཡིག་གསུམ་འབར་བ་ཡི། །བདུད་རྩི་ལྔ་ནི་འཐུང་བཅུག་སྟེ། །འདི་ནི་ཁྱོད་ཀྱི་དམྱལ་བའི་ཆུ། །དམ་ཚིག་
འདས་ན་སྲེག་པར་བྱེད། །དམ་ཚིག་བསྲུངས་ན་དངོས་གྲུབ་ནི། །རྡོ་རྗེ་བདུད་རྩིའི་ཆུ་འདིས་འགྲུབ། །ཨོཾ་བཛྲོད་ཀ་ཋཿདེ་ནས་འདི་སྐད་སྨྲ་བར་བྱའོ། །དེ་རིང་ཕན་ཆད་ཁྱོད་ཀྱི་ཕྱག་ན་རྡོ་རྗེ་ང་ཡིན་གྱིས། ངས་ཅི་སྨྲས་པའི་བྱ་བ་འདི་ཁྱོད་ཀྱིས་བྱ་དགོས་སོ། །ང་ལ་ཡང་བརྙས་པར་མ་བྱེད་ཅིག །མི་བདེ་བ་མ་
སྤངས་པར་ཤི་བའི་དུས་བྱས་ཏེ། དམྱལ་བར་ལྟུང་བར་མ་འགྲོ་ཤིག །དེ་ནས་དེ་ལྟར་སྨྲ་བ་ལ། སློབ་མས་འདི་ལྟར་སྨྲ་བར་བྱའོ། །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བྱིན་གྱིས་བརླབས་ནས། རྡོ་རྗེ་སེམས་དཔའ་བདག་ལ་འབབ་པར་གྱུར་ཅིག །ཅེས་སྨྲ་བར་བྱའོ། །གསོལ་འདེབས་དེ་ཡི་
