ལ། །ཆོ་ག་བཞིན་དུ་རྣམ་པར་བསམ། །པྲཛྙཱ་དྷྲྀཀ་ཅེས་བྱ་བ་འདིས། །ཕྲག་དོག་གཤིན་རྗེ་ཟད་བྱེད་པ། །བྱང་རྩིབས་ཆུ་སྐྱེས་ཉི་མ་ལ། །ཆོ་ག་བཞིན་དུ་དགོད་པར་བྱ། །མུངྒར་དྷྲྀཀ་ཅེས་བྱ་བ་ཡིས། །ཤར་སྒོར་ཆུ་སྐྱེས་ཉི་མ་ལ། །ཐོ་བའི་མཐའ་འཇོམས་ཟ་བྱེད་པ། །བློ་དང་ལྡན་
པས་བསམ་པར་བྱ། །དཎྜ་དྷྲྀཀ་པའི་སྔགས་འདི་ཡིས། །ལྷོ་ཡི་སྒོ་རུ་ཡང་དག་གནས། །དབྱུག་པ་གཤིན་རྗེ་འདུལ་དཀའ་བ། །སྣ་ཚོགས་ཆུ་སྐྱེས་ཉི་མར་དགོད། །པདྨ་དྷྲྀཀ་ཅེས་བྱ་བ་འདིས། །པདྨ་གཤིན་རྗེ་གཤེད་འཇོམས་པ། །ནུབ་སྒོར་ཆུ་སྐྱེས་ཉི་མ་ལ། །གཏུམ་པོ་
སྟོབས་ཆེན་བསྒོམ་པར་བྱ། །ཁཌྒ་དྷྲྀཀ་ཅེས་བྱ་བ་འདིས། །རལ་གྲི་གཤིན་རྗེ་ཟད་པར་བྱེད། །སྣ་ཚོགས་པདྨ་ཉི་མར་བསམ། །བྱང་དུ་བདུད་ནི་འཇོམས་པའོ། །སྣ་ཚོགས་རྡོ་རྗེའི་མཚམས་བཞི་ལ། །ལྷ་མོ་བསྒོམ་བྱ་དེ་བཞིན་མཆོག །ཤར་ལྷོ་ལྷ་མོ་སེམས་མ་སྟེ། །གཏུམ་མོ་མཛེས་
མའི་འོད་ཟེར་ཅན། །མོ་ཧ་ར་ཏི་ཞེས་སྔགས་ཀྱིས། །སྣ་ཚོགས་ཆུ་སྐྱེས་ཟླ་བར་གནས། །དེ་ནས་ལྷོ་ནུབ་རྡོ་རྗེ་མོ། །ཕག་མོ་རིགས་ཀྱི་གཙོ་མོ་སྟེ། །དྭེ་ཥ་ར་ཏི་དེ་ཉིད་འདིས། །པདྨ་ཟླ་བར་མཁས་པས་དགོད། །དེ་ནས་ནུབ་བྱང་མཚམས་སུ་ནི། །ཟླ་བར་དབྱངས་ཅན་དགོད་པར་བྱ། །རཱ་ག་
ར་ཏིའི་སྔགས་འདི་ཡིས། །དེ་ནས་བྱང་ཤར་གཽ་རཱི་སྟེ། །བཛྲ་ར་ཏིའི་སྔགས་འདི་ཡིས། །ཟླ་བར་ཨོཾ་གྱིས་མཚན་པའོ། །རྩ་བའི་དཀྱིལ་འཁོར་གྲྭ་བཞི་རུ། །དེ་ཉིད་གསུམ་གྱི་ཐོད་པའོ། །འདི་ལྟར་སྤྲོ་བྱ་དཀྱིལ་འཁོར་བཅས། །བརྟན་པར་བྱས་ནས་བདེ་ལྡན་པ། །སྔོན་གྱི་འཁོར་ལོ་དེར་ཐིམ་པར། །
མཉམ་པར་གཞག་པས་བལྟ་བར་བྱ། །འདི་ནི་ཤིན་ཏུ་རྣལ་འབྱོར་རོ། །གཤིན་རྗེ་ནག་པོ་འཇོམས་པ་དང་། །དཀྱིལ་འཁོར་པ་ཡི་རང་བཞིན་གྱིས། །མིག་དང་སྐུ་སོགས་ངོ་བོ་ལ། །ལྷ་རྣམས་བྱིན་གྱིས་བརླབ་པར་བྱ། །ཀྵིཾ་ལས་འཁོར་ལོ་གཏི་མུག་མདངས། །ཅུང་ཟད་ཁྲོ་བོ་མིག་ལ་ནི། །རྡོ་རྗེ་
མིག་དགོད་བློ་ལྡན་གྱིས། །རྣ་བར་ཛཾ་ཡིག་རྡོ་རྗེ་ལས། །རྡོ་རྗེ་རྣ་བ་ནག་པོའི་མདངས། །སྣ་རུ་ཁཾ་ཡིག་རིན་ཆེན་འབྱུང་། །ཕྲ་མ་ཞེས་བསྒྲགས་དང་མཚུངས་པ། །ཟླ་བར་རྡོ་རྗེ་སྣ་ཚོགས་སོ། །ལྕེ་ལ་རཾ་ཞེས་པདྨ་ལས། །རྡོ་རྗེ་ལྕེ་བྱུང་པདྨར་སོང་། །འདོད་ཆགས་གཤིན་རྗེ་མདངས་ལྡན་པའོ། །ཀཾ་སྐྱེས་
