བྱ་བས་ཁྱད་པར་དུ་མཛད་དེ་འཇིག་རྟེན་གསུམ་པ་ལའོ། །འབྲས་བུ་མཆོག་སྒྲུབ་ཅིང་འགྲུབ་པར་མཛད་པས་ན་མཆོག་སྒྲུབ་པའོ། །དམ་པ་སྩོལ་བ་ས་བཅུ་པའི་དབང་ཕྱུག་རྣམས་ཀྱི་ནང་ན་མཆོག་སྒྲུབ་པ་གང་ཡིན་པ་དག་ནི་དམ་པ་སྩོལ་མཆོག་སྒྲུབ་པ་ཡིན་ནོ། །སངས་རྒྱས་ཐམས་ཅད་
འདུ་བ་ལས་ནི་མཆོག་གི་དངོས་གྲུབ་སྩོལ་ཞེས་འབྱུང་སྟེ། དེར་ནི་མཆོག་སངས་རྒྱས་ཀྱི་ཆོས་རྣམས་ཀྱི་དངོས་གྲུབ་ཡོངས་སུ་གྲུབ་པ་དེ་སྩོལ་བ་ཞེས་རྣམ་པར་བཤད་པར་བྱའོ། །བདེ་གཤེགས་ནི་རབ་ཏུ་ཞི་བར་ཕྱིར་མི་ལྡོག་པར་མ་ལུས་པར་གཤེགས་པའི་ཕྱིར་རོ། །གང་དག་ལ་དུས་
གསུམ་ལ་ཐོགས་པ་མི་མངའ་བའི་མཁྱེན་པའི་བློ་མངའ་བ་དེ་དག་ལ་དེ་སྐད་ཅེས་བྱ་སྟེ། འདས་པ་དང་མ་འོངས་པ་དང་ད་ལྟར་དག་ལ་ཐོགས་པ་མེད་པའི་ཡེ་ཤེས་གཟིགས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེའང་ཅིའི་ཕྱིར་ཞེ་ན། འདི་ལྟར་དེ་དག་ནི་སྒྲིབ་པ་མི་མངའ་བའོ། །སྒྲིབ་པ་ནི་ཆོད་ཅིང་སྒྲིབ་
པར་བྱེད་པ་སྟེ། ཉོན་མོངས་པའི་སྒྲིབ་པ་དང་། ཤེས་བྱའི་སྒྲིབ་པ་ཞེས་བྱ་བ་གང་ཡིན་པའོ། །གང་དག་ལ་དེ་དག་སྤངས་པའི་ཕྱིར་མེད་པ་དེ་དག་ནི་སྒྲིབ་པ་མི་མངའ་བ་སྟེ། འདི་ནི་བདེ་བར་གཤེགས་པའི་འབྲས་བུ་ཡིན་ནོ། །ལ་ལ་ལས་ནི་མགོན་པོ་ཞེས་འབྱུང་གི་བདེ་བར་
གཤེགས་ཞེས་མི་འབྱུང་ངོ་། །དེར་ནི་སྡེབ་སྦྱོར་ཉམས་ཏེ་ལན་ཡང་སྔ་མ་ཡིན་ནོ། །མགོན་པོ་ནི་ས་གསུམ་གྱི་རྗེའོ། །ཀུན་གྱི་ཞེས་བྱ་བ་ནི་ཕྱོགས་བཅུ་མཐའ་ཡས་པའི་འཇིག་རྟེན་གྱི་ཁམས་ན་བཞུགས་པ་རྣམས་སོ། །དམ་ཚིག་གསུམ་མཆོག་ཅེས་བྱ་བ་ལ་སོགས་པ་ལ། དམ་
ཚིག་གསུམ་ཞེས་བྱ་བ་ནི། དམ་ཚིག་གསུམ་པའི་རྒྱུད་ལ་བརྩོན་པ་བདག་ལའམ། དེ་ལས་གསུངས་པའི་མཆོག་གི་དངོས་གྲུབ་བམ། དེ་རྟོགས་པའི་རྒྱུ་མཚན་བདག་ལ་སྩོལ་ཅིག་པའོ། །རྒྱུད་གང་ལས་དམ་ཚིག་སྟེ་ངེས་པ་གསུམ་པོ་སངས་རྒྱས་དང་། ཆོས་དང་། ཕྱིར་མི་
ལྡོག་པའི་དགེ་འདུན་ལ་དང་བའི་བདག་ཉིད་འབྱུང་བ་དེ་ནི་དམ་ཚིག་གསུམ་པའོ། །ཡང་ན་དམ་ཚིག་གསུམ་ནི། དོན་ཀུན་གྲུབ་པར་བྱ་བའི་ཕྱིར། །བྱང་ཆུབ་སེམས་ཡོངས་མི་གཏོང་དང་། །ཐུབ་པའི་བརྟུལ་ཞུགས་བརྟན་པ་དང་། །དེ་ཉིད་ཤེས་པ་བཙལ་བ་སྤྱད། །ཅེས་བྱ་བ་འདི་དག་
