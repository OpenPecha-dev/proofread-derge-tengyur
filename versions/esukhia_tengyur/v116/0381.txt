སྡིག་པ་ནི་མི་དགེ་བ་ཡིན་པས་སྡིག་པའི་ལས་ཞེས་པ་སྟེ། ལུས་ངག་ཡིད་གསུམ་གྱིས་བྱས་པའོ། །རང་གིས་བྱས་པ་དང་། གཞན་དག་ལ་བཅོལ་བ་དང་། གཞན་བྱེད་པ་ལ་དགའ་བར་བྱས་པ་ཕྱིར་བཤགས་པའོ། །མཆོད་རྟེན་ནི་སྐུ་གཟུགས་དང་། གསུང་རབ་དང་། རིང་
བསྲེལ་དང་། ལྷ་ཁང་ལ་སོགས་པའོ། །དགེ་འདུན་ནི་སྡེ་པའོ། །ཕྱོགས་བཅུའི་དགེ་འདུན་ནི་གཞན་འདྲོན་པོའོ། །དེ་རྣམས་ཀྱི་དཀོར་ཕྲོགས་ན་དམྱལ་བར་འགྲོ་སྟེ། དེ་ཡང་མཆོད་རྟེན་དང་གཙུག་ལག་ཁང་བཤིག་པ་དང་། དཀོན་མཆོག་གསུམ་གྱི་དཀོར་སྦས་ནས་བདག་ཉིད་ཀྱི་ཕྱིར་ཡོངས་སུ་སྤྱོད་པ་གང་ཡིན་པ་
དེ་ནི་ཚེ་འཕོས་ནས་དམྱལ་བ་ཆེན་པོ་མནར་མེད་པར་ལྟུང་ངོ་ཞེས་གསུངས་པ་དང་། དགེ་འདུན་གྱི་དཀོར་བརྐུས་པས་རང་འཐག་འདྲེན་པའི་གླང་དང་བོང་བུར་སྐྱེ་བར་འགྱུར་རོ། །དགེ་འདུན་གྱི་དཀོར་ལ་ནན་གྱིས་སློང་ཞིང་གསང་སྐྱོན་བྱེད་ན་ཕུག་རོན་དུ་སྐྱེ་བར་འགྱུར་རོ། །དགེ་འདུན་གྱི་ཚོད་ལུམ་ཟོས་ན་ཚོད་མ་
ནང་གི་སྲིན་བུར་སྐྱེ་བར་འགྱུར་རོ་ཞེས་གསུངས་སོ། །དེ་ཡང་རང་གིས་བྱས་པ་དང་། གཞན་ལ་བཅོལ་བ་དང་། གཞན་བྱེད་པ་ལ་ཡི་རང་བ་བཤགས་སོ། །མཚམས་མེད་པ་ལྔ་ནི་ཕ་མ་དང་དགྲ་བཅོམ་པ་གསོད་པ་དང་། དེ་བཞིན་གཤེགས་པ་ལ་ངན་སེམས་ཀྱིས་ཁྲག་བཏོན་པ་དང་། དགེ་འདུན་མཐུན་
པའི་ནང་དུ་དབྱེན་བྱེད་པའོ། །དེ་རྣམས་བྱས་ན་བར་མཚམས་མེད་པར་དམྱལ་བར་འགྲོའོ། །མི་དགེ་བ་བཅུ་ནི་སྲོག་གཅོད་པ་དང་། མ་བྱིན་པར་ལེན་པ་དང་། མི་ཚངས་པར་སྤྱོད་པ་དང་། <བརྫུན་,རྫུན་>དུ་སྨྲ་བ་དང་། ཚིག་ཕྲ་མ་དང་། ཚིག་རྩུབ་པོ་དང་། ངག་འཁྱལ་བ་དང་། བརྣབ་སེམས་དང་། གནོད་སེམས་དང་།
ལོག་པར་ལྟ་བའོ། །དེ་རྣམས་སེམས་མཚམས་མེད་པར་སྤྱད་ན་དམྱལ་བར་འགྲོའོ། །དུད་འགྲོར་འགྲོའོ། །ཐ་མ་<ཡི་དགས་,ཡི་དྭགས་>སུ་སྐྱེའོ། །ལས་ཚབས་ཆུང་ལ་དགེ་བ་གཞན་གྱི་དབང་གིས་མིར་སྐྱེས་ན་ཡང་། སྲོག་བཅད་པས་ཚེ་ཐུང་བ་དང་། ནད་མང་བའོ། །མ་བྱིན་པར་བླངས་པས་ནི་དབུལ་ཞིང་ཕོངས་པ་དང་
ནོར་གཞན་གྱིས་རང་དབང་མེད་པར་སྤྱོད་པའོ། །མི་ཚངས་པར་སྤྱད་པས་ནི་མངལ་གྱི་གནས་མིན་པ་མི་ལེན་པ་དང་། ལུས་མི་ཚངས་པའོ། །འདོད་པ་ལོག་པར་གཡེམ་པ་ནི་ཆུང་མ་ལ་གཞན་སྤྱོད་པ་དང་། ཆུང་མ་དང་འཐབ་ཅིང་བསམས་བཞིན་དུ་མི་བཏུབ་པའོ། །<བརྫུན་,རྫུན་>དུ་སྨྲ་བས་ནི་སྐུར་པ་མང་བ་དང་།
