རྟོགས་ན་རྟོགས་པར་སླ་བའི་ཕྱིར་དང་། དུ་མ་མ་བཟློག་པ་ཡིན་པའི་ཕྱིར་བྱེད་པ་པོ་གཅིག་ཏུ་རྟོགས་པ་ཁོ་ནར་རིགས་པ་མ་ཡིན་ནམ་ཞེ་ན། འདི་ལ་ཚད་མའི་བྱེད་པ་ནི་མེད་དོ། །སླའོ་ཞེས་བྱ་བ་ཁོ་ནས་ནི་རྟོགས་པ་ཚད་མ་དང་ལྡན་པ་མ་ཡིན་ཞིང་དཀའོ་ཞེས་བྱ་བས་ཀྱང་བྲལ་བ་མ་ཡིན་ཏེ། བྱེད་
པ་མ་མཐོང་བའི་བུམ་པ་ལ་སོགས་པ་ལ་གཞན་དུ་མཐོང་བས་ན། རྟོགས་པར་སླ་བ་ལ་ཆགས་པ་ཁྱོད་ཀྱིས་བྱེད་པ་པོ་གཅིག་ཏུ་བརྟག་པར་བྱ་བར་འགྱུར་རོ། །བརྡ་སྤྲོད་པའི་མདོར་བྱས་པ་ཁོ་ན་དཀའ་བ་དང་། ཟོར་ཡང་པའི་བསམ་པ་འཐད་པ་ཡིན་ན་ཁྱེད་ནི་སྔར་མ་བྱུང་བའི་ཚད་
མ་བ་ཡིན་ནོ། །འདིར་ཡང་དེ་ལ་གཅེས་སུ་བྱེད་པས་ན་ཅི་ཞིག་སྨྲར་ཡོད། ཐ་དད་པ་སྒྲུབ་པར་བྱེད་པའི་ཚད་མ་མེད་པ་ཡང་བརྟག་པར་སླ་བས་བཤད་ཟིན་ཏོ། །དེའི་ཕྱིར་བྱེད་པ་པོ་གཅིག་ཏུ་ངེས་པ་ལ་ཐབས་ཅི་ཡང་མེད་ཅིང་། དེ་མ་ངེས་པར་ཡང་འདོད་པས་ཡུལ་དུ་བྱས་
པའི་ཐམས་ཅད་ཤེས་པ་ཉིད་མི་ངེས་པས་ན། བྱེད་པ་པོའི་སྒྲུབ་བྱེད་ཐམས་ཅད་ནི་འདོད་པའི་བསྒྲུབ་བྱ་ལ་ལྟོས་ནས་ལྡོག་པ་ལ་ཐེ་ཚོམ་ཟ་བ་ཡིན་ནོ། །དེའི་ཕྱིར་དེ་ལྟར་གཞན་གྱིས་རབ་ཏུ་སྦྱར་བའི་དབང་ཕྱུག་གི་སྒྲུབ་བྱེད་རྣམས་ལ་ཁོ་བོ་ཅག་འགལ་བ་མི་འཁྲུལ་བ་འགོད་པ་མ་
ཡིན་གྱི། འོན་ཀྱང་སྒྲུབ་བྱེད་སུན་འབྱིན་པ་ཁོ་ན་སྟེ། མང་དུ་སྨྲ་བས་ཆོག་གོ། །དེའི་ཕྱིར་འདི་སྒྲུབ་པར་བྱེད་པའི་ཚད་མ་མེད་ཅིང་གནོད་པར་བྱེད་པ་ནི་ཁ་ན་མ་ཐོ་བ་མེད་པར་ཡོད་པས་ན་འདུས་བྱས་བྱེད་པ་པོ་མེད་པར་གྲུབ་པ་ཡིན་ཏེ། དེ་སྐད་དུ་ཡང་། དབང་ཕྱུག་{𰵋སྩོགས་,སོགས་}མིན་རིམ་
{𰵌སྩོགས་,སོགས་}ཕྱིར། །ཞེས་གསུངས་སོ། །འོན་ཏེ་ཇི་ལྟར་མདོ་སྡེ་པ་རྣམས་ཀྱི་རྣམ་པ་དང་བཅས་པའི་རྣམ་པར་ཤེས་པ་དོན་ཡུལ་ཅན་ཡིན་པ་དེ་བཞིན་དུ་འདི་དག་གི་ལྟར་ན་ཡང་ཡིན་ནམ། དེ་ལྟ་ནི་མ་ཡིན་གྱི་འོན་ཀྱང་། རྣམ་པ་མེད་པའི་དབང་སྐུ་བློས། །རྡུལ་ཕྲན་བསགས་
པ་དངོས་སུ་རིག །དབང་པོ་སྟེ་དབང་བྱེད་པ་རྣམས་ལས་སྐྱེས་པའི་བློ་མངོན་སུམ་ཞེས་བྱ་བའི་བློ་གྲོས་ཡུལ་གྱི་རྣམ་པ་མེད་པས། དེ་དག་གི་བཞིན་རྣམ་པས་བར་དུ་ཆོད་པ་བཏགས་པ་ལྟར་མ་ཡིན་པར་དངོས་སུ་རྡུལ་ཕྲན་ཏེ་རྡུལ་ཕྲ་རབ་བསགས་པ་ཕུང་པོའི་རིག་སྟེ་ཤེས་པ་ཡིན་གྱི། གཞན་གྱི་བཞིན་
