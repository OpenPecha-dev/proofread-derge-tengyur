རྟོགས་པ་ཡིན་པའི་ཕྱིར་རོ། །དེ་བས་ན་བསོད་ནམས་མི་ཉུང་བ་བསགས་པའི་གནས་ཡིན་པའི་ཕྱིར་དེ་ཉིད་མཆོད་གནས་སུ་འོས་པ་ཡིན་ནོ། །ཡི་གེ་ཁ་ཅིག་ལས་ཉེས་པར་སྒྲོ་འདོགས་པ་ལ་སྤྱོད་པ་ཞེས་འབྱུང་སྟེ། དེ་ལ་ཉེས་པ་ཞེས་བྱ་བ་ནི་ལྟུང་
བ་ལ་སོགས་པ་ལ་བྱ་ལ། དེ་ཆོས་ཀྱི་དབྱིངས་གང་ལ་སྒྲོ་འདོགས་པ་ཡོད་པ་དེ་ལ་དེ་སྐད་ཅེས་བྱ་བ་འབྲུ་མང་པོར་བྱའོ། །བཙུན་པ་ཤ་ར་དྭ་ཏིའི་བུ་ལྷག་པར་སྒྲོ་འདོགས་པ་ལ་སྤྱོད་པ་ཞེས་བྱ་བ་འདིའི་སྒྲའི་དོན་བཟུང་ན་ལྷག་པར་སྒྲོ་འདོགས་པ་ལ་སྤྱོད་པའམ། རྩོད་པར་སྒྲོ་འདོགས་
པ་ལ་སྤྱོད་པ་ནི་ཡོན་གནས་ཡིན་ནོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན་ཞེས་བྱ་བ་ནི་རྒྱུད་རྣམ་པར་མ་དག་པ་ཇི་ལྟར་ཡོན་གནས་སུའོ་ཞེས་སེམས་པའོ། །རྣམ་པར་ཕྱེ་སྟེ་མདོར་སྟོན་པ་ནི་འདི་ལྟར་མཉམ་པ་ལ་དེ་ལྷག་པར་སྒྲོ་འདོགས་པའི་ཕྱིར་རོ་ཞེས་གསུངས་ཏེ། མཉམ་པ་འདྲ་བ་ཀུན་ལ་ཁྱབ་
པ་དེ་ལ་ལྷག་པར་སྒྲོ་འདོགས་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ལ་བྱ་སྟེ། འདི་ལ་ལྷག་པར་སྒྲོ་འདོགས་པ་ཞེས་སྔ་མ་བཞིན་དུ་སྦྱར་རོ། །མ་མོའི་དོན་ལ་བརྩོན་པ་དག་སེམས་པ་སྣ་ཚོགས་པ་མེད་ན་དད་པ་དང་དད་པ་མེད་པ་དང་། ལྷག་པའི་སེམས་ལ་སྤྱོད་པ་ཞེས་བྱ་བའི་བར་དུ་འདི་དག་ཇི་
ལྟར་རྣམ་པར་དབྱེ། འདི་ལྟར་དད་པ་ཞེས་བྱ་བ་ནི་ལས་དང་འབྲས་བུ་དང་བདེན་པ་དང་། དཀོན་མཆོག་ལ་མངོན་པར་ཡིད་ཆེས་པ་གང་ཡིན་པའོ། །རུང་བ་འདི་ལ་ཡོད་པས་རུང་བ་ཅན་ཏེ། འདུལ་བ་ལས་གསུངས་པ་བཞིན་དུ་རུང་བ་ལ་རྗེས་སུ་གནས་པའོ། །ཚུལ་ཡང་དག་
པར་མ་བཅོམ་པ་ནི་སྲིད་པའི་མཚན་ཉིད་འཁོར་བ་ལ་མངོན་པར་དགའ་བ་དང་། གཟུགས་ལ་དགའ་བའི་ཚུལ་རྩ་བ་ནས་བཏོན་པའོ། །སྐྱེ་བ་ལ་སོགས་པའི་འཇིགས་པ་ལས་རྒལ་བ་ནི་གང་གིས་འཇིགས་པའི་རྒྱུ་ཉོན་མོངས་པའི་ཚོགས་སྤངས་པའི་ཕྱིར་ཏེ། དེ་ནི་འཇིགས་པ་ལས་
རྒལ་བའོ། །བཟོད་པ་སྐྱེས་པ་ཅན་ནི་གང་ཞིག་སྡུག་བསྔལ་ལ་ཆོས་ཤེས་པའི་བཟོད་པ་སྐྱེས་པའོ། །དུལ་བ་ནི་སྤྱོད་པ་བཟང་པོ་ཅན་ནོ། །ལྷག་པའི་སེམས་ལ་སྤྱོད་པ་ནི་ཟག་པ་མེད་པའི་ཏིང་ངེ་འཛིན་ཐོབ་པའོ། །བཟློག་པ་ནི་མ་དད་པ་ལ་སོགས་པ་ཡིན་ནོ་ཞེ་ན། རྣམ་པར་དབྱེ་
