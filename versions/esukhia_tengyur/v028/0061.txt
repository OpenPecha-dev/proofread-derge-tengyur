བཏབ་ལ། །སངས་རྒྱས་ཀུན་བཞུ་ལྷ་མོ་ཡི། །རྣལ་འབྱོར་ཀུན་ལྡན་ཡེ་ཤེས་ལྔའི། །དབང་བསྐུར་སློབ་མ་མཆོག་ལ་སྦྱིན། །ཨོཾ་ཨཱ་བཛྲཱ་དཪྴ་ཛྙཱ་ན་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་བཛྲ་ས་མ་ཏ་ཛྙཱ་ན་རཏྣ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་བཛྲ་པྲ་ཏྱ་བེཀྵ་ཎ་ཛྙཱ་ན་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་ཀྲྀ་ཏྱ་ཨ་ནུཥྛ་ན་ཛྙཱ་ན་བཛྲ་
ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་ཙཀྲ་ཨ་མོ་གྷ་ཛྙཱ་ན་ཨ་བྷི་ཥིཉྩ་མཱཾ། དེ་ནས་རྡོ་རྗེ་སྦྱིན་ཏེ། ཨོཾ་བཛྲ་ས་མ་ཡ་སྟྭཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། དེ་ནས་རྒྱན་དང་སྲུང་བ་སྦྱིན་ལ། ཨོཾ་བཛྲ་རཀྵ་རཀྵ་ཧཱུཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ལུང་བསྟན་ཅིང་དབུགས་ཕྱུང་ལ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སིདྡྷ་ས་མ་ཡ་སྟྭཾ་སྭ་བྷཱ་བ་ཨ་བྷི་ཥིཉྩ་མཱཾ།
རྡོ་རྗེ་སློབ་དཔོན་ཉིད་དུ་དབང་བསྐུར་ལ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་ཧེ་ཧེ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཕྱིར་མི་ལྡོག་པར་དབང་བསྐུར་ལ། ཨོཾ་བཛྲ་བེ་ཝརྟི་ཨ་བྷི་ཥེ་ཀ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཆོས་ཀྱི་རྒྱལ་པོར་དབང་བསྐུར་ལ། ཨོཾ་དྷརྨཱ་རཱ་ཛཱ་པྲ་ཏི་སྭཱ་ཧཱ། རྗེས་སུ་གནང་བའི་དབང་བསྐུར་ལ། ཨོཾ་བཛྲ་ཨ་

བདུད་རྩི་བཞི་པས་རྡོ་རྗེ་འཛིན། །རྗེ་བཙུན་མ་སྒྲོལ་མ་མྱ་ངན་སེལ་བྱེད་ཀྱི་དཀྱིལ་འཁོར་དུ་འཇུག་པའི་ཡན་ལག་ཅེས་བྱ་བ་རྫོགས་སོ།། །།ཕྱག་འཚལ་ས་གཞི་སྐྱོང་བའི་ཚོགས་རྣམས། །ཐམས་ཅད་འགུགས་པར་ནུས་པ་ཉིད་མ། །ཁྲོ་གཉེར་གཡོ་བ་ཡི་གེ་ཧཱུཾ་གིས། །
ཕོངས་པ་ཐམས་ཅད་རབ་ཏུ་སྒྲོལ་མ། །དཔལ་ལྡན་ས་སྐྱོང་ཕྱོགས་རྣམས་དང་། །ཕྱོགས་བྲལ་མ་ལུས་འགུགས་བྱེད་མ། །དེ་ཉིད་སྙིང་པོ་རབ་གཡོ་བས། །ཕོངས་པ་མཐའ་དག་སེལ་མཛད་བསྒོམ། །རྗེས་མཐུན་ཆོ་གས་ལས་བསྒྲུབས་ནས། །པད་ཉི་ལ་བཞུགས་གཡས་
