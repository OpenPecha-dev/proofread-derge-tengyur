ལུས་ནི་འདས་གྱུར་ཀྱང་། །སེམས་ཀྱིས་ཀྱང་ནི་མ་ཡིན་ནོ། །བསམ་མེད་བསམ་པར་བྱ་བ་ཉིད། །བསམ་པ་དེ་ཡང་མི་ཤེས་སོ། །སེམས་དཔའ་འམ་ནི་བསམ་མི་ཁྱབ། །དེ་ནས་གཟུངས་ནི་ཐོབ་པར་འགྱུར། །ཞེས་གསུངས་སོ། །དེའི་རྗེས་ལ་ལྟེ་བའི་ས་བོན་གྱི་ཐིག་ལེ་ལ་ཉི་
མའི་གཟུགས་ཀྱི་སྟེང་དུ། སྔགས་ཀྱི་ཕྲེང་བའི་སྟེང་དུ་བལྟས་པས་ཤིན་ཏུ་ཕྲ་བ་མངོན་པར་བལྟས་ཏེ། བརྟན་པའི་སེམས་ཀྱིས་འཕྲོ་བ་དང་འདུ་བའི་རིམ་པས་མན་ངག་ཇི་ལྟ་བ་བཞིན་དུ་བཟླས་པར་བྱའོ། །དེ་ལ་ཨོཾ་ཤྲཱི་བཛྲ་ཧེ་ཧེ་རུ་རུ་ཀཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཌཱ་ཀི་ནཱི་ཛཱ་ལ་སཾ་བ་ར་སྭཱ་ཧཱ། ཨོཾ་ཧྲཱིཿཧ་ཧ་ཧཱུཾ་ཧཱུཾ་ཕཊ།
རྗེ་བཙུན་གྱི་སྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ནཱི་ཡེ་བཛྲ་བརྞ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ལྷ་མོའི་སྙིང་པོ་དང་ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ལཱ་མཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁཎྜ་རོ་ཧཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་རཱུ་པི་ཎཱི་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀཱ་ཀཱ་སྱེ་ཧཱུཾ་
ཧཱུཾ་ཕཊ། ཨོཾ་ཨུ་ལཱུ་ཀཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤྭཱ་ནཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སཱུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཱ་ཌྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཱུ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཾཥྚྲཱི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། དཔའ་མོ་རྣམས་ཀྱི་སྔགས་སོ། །ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་ཝཱི་རེ་ཤཱ་ཡ། མ་ཧཱ་
ཀལྤཱགྣི་སནྣི་བྷཱ་ཡ། ཛ་ཊ་མ་ཀུ་ཊོཏྐ་ཊཱ་ཡ། དཾཥྚྲཱ་ཀ་རཱ་ལོ་གྲ་བྷཱི་ཥ་ཎ་མུ་ཁཱ་ཡ། སཱ་ཧ་སྲ་བྷུ་ཛ་བྷཱ་སུ་རཱ་ཡ། པར་ཤུ་པཱ་ཤོ་དྱ་ཏ་ཤཱུ་ལ་ཁ་ཊྭཱཾ་ག་དྷཱ་རི་ཎེ། བྱཱ་གྷྲ་ཛི་ནཾ་བ་ར་དྷ་རཱ་ཡ། མ་ཧཱ་དྷུམྦྲཱནྡྷ་ཀ་ར་ཝ་པུ་ཥཱ་ཡ། ཀ་ར་ཀ་ར། ཀུ་རུ་ཀུ་རུ། བནྡྷ་བནྡྷ། ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ། ཀྵོ་བྷ་ཡ་ཀྵོ་བྷ་ཡ། ཧྲཽཾ་ཧྲཽཾ།
