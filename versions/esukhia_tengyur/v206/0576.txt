དང་བོག་རི་ལྟ་བུ་ནི་སྒང་ངོ་། རྫིང་གི་ནང་ལྟ་བུ་ནི་གཤོང་ངོ་། །མཚུངས་པ་བཞི་གང་ཞེ་ན། དུས་གསུམ་དུ་གཤེགས་པའི་སངས་རྒྱས་རྣམས་ཀྱི་ཡོན་ཏན་མཉམ་པ་ལྟ་བུ་སྟེ། སངས་རྒྱས་ཞེས་བྱ་བའི་ཡིག་འབྲུ་མང་ཉུང་མེད་པ་ནི་ཡི་གེ་མཚུངས་པའོ། །གསུང་གི་ཡན་ལག་དྲུག་ཅུ་དང་ལྡན་ཞིང་
ཚངས་པའི་སྒྲ་སྐད་ལྟ་བུ་དང་ལྡན་པར་འདྲ་བ་ནི་གསུང་མཚུངས་པའོ། །ཆོས་ཐམས་ཅད་སྐྱེ་བ་དང་འགག་པ་མེད་ཅིང་བདག་མེད་པ་རྣམ་གཉིས་སུ་ཐུགས་སུ་ཆུད་པར་འདྲ་བ་ནི་ཆོས་མཚུངས་པའོ། །སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཆོས་ཀྱི་སྐུར་མཉམ་སྟེ་བྱེ་བྲག་ཆེ་ཆུང་མེད་པ་ནི་སྐུ་མཚུངས་པའོ། །བདེན་
པ་བཞི་གཅིག་ཅིང་ཡང་བཞི་བཞིར་བརྟགས་པ་ལ། འགོག་པ་ལ་ནི་འགོག་པ་དང་ཞི་བ་དང་གྱ་ནོམ་པ་དང་ངེས་པར་འབྱུང་བའོ། །ལམ་ལ་ནི་ལམ་དང་། རིགས་པ་དང་། བསྒྲུབ་པ་དང་། ངེས་པར་འབྱིན་པའོ། །སྡུག་བསྔལ་ལ་ནི་མི་རྟག་པ་དང་། སྡུག་བསྔལ་བ་དང་། སྟོང་པ་དང་། བདག་མེད་པའོ། །
ཀུན་འབྱུང་ལ་ནི་རྒྱུ་དང་། ཀུན་འབྱུང་བ་དང་། རྐྱེན་དང་། འབྱུང་བའོ། །དེ་ལ་འདོད་ཆགས་དང་ཞེ་སྡང་ལ་སོགས་པའི་ཉོན་མོངས་པ་དང་། སྲོག་གཅོད་པ་ལ་སོགས་པ་སྡིག་པ་མི་དགེ་བའི་ལས་རྣམས་སྤངས་ཏེ་མི་འབྱུང་བ་ནི་འགོག་པའོ། །ལས་དང་ཉོན་མོངས་པས་བསྐྱེད་པའི་འབྲས་བུ་སྡུག་བསྔལ་
ཅི་ཡང་མི་འབྱུང་བ་ནི་ཞི་བའོ། །ཕྱིས་ཀྱང་ཁམས་གསུམ་དུ་སྐྱེ་ཞིང་སྡུག་བསྔལ་མྱོང་བར་མི་འགྱུར་བས་ན་གྱ་ནོམ་པའོ། །ཁམས་གསུམ་བཙོན་ར་དང་འདྲ་བ་ནས་ཐར་ཅིང་མྱ་ངན་ལས་འདས་པའི་གནས་བདེ་བར་ཕྱིན་པས་ན་འབྱུང་བའོ། །ལམ་ནི་སོ་སོའི་སྐྱེ་བོའི་ས་ནས་འཕགས་པའི་གནས་སུ་འགྲོ་
བས་ན་ལམ་མོ། །མི་རིགས་པ་ཉོན་མོངས་པ་སྤངས་པའི་གཉེན་པོར་གྱུར་པས་ན་རིགས་པའོ། །སེམས་ཕྱིན་ཅི་ལོག་སྤངས་ཏེ་ཕྱིན་ཅི་མ་ལོག་པར་སྦྱོར་བས་ན་སྒྲུབ་པའོ། །འཁོར་བའི་ཉེས་པ་དང་གཏན་བྲལ་ཞིང་མྱ་ངན་ལས་འདས་པའི་གནས་སུ་ཕྱིན་པར་བྱེད་པས་ན་ངེས་པར་འབྱུང་བའོ། །
སྡུག་བསྔལ་ནི་སྐད་ཅིག་མ་རེ་རེ་ལས་སྐྱེས་ཤིང་འཆི་བའི་ཆོས་ཅན་ཡིན་པས་ན་མི་རྟག་པའོ། །སྡུག་བསྔལ་གསུམ་དང་བརྒྱད་ཀྱིས་རྒྱུན་མི་འཆད་པར་འཚེ་བས་ན་སྡུག་བསྔལ་བའོ། །ཕུང་པོ་ལྔའི་ནང་ན་མུ་སྟེགས་ཀྱིས་བརྟགས་པའི་བདག་རྟག་པ་མེད་པས་ན་སྟོང་པའོ། །ཕུང་པོ་ལྔ་པོ་ཉིད་ཀྱང་
