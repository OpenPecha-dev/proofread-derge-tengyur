བའི་ནང་དུ་ཟན་དང་། ཁུར་བ་དང་། ཆན་དང་། འབྲུའི་ཚོགས་དང་། ཤིང་ཐོག་དང་། རིན་པོ་ཆེའི་ཚོགས་དང་། སྨན་དང་ལྡན་པར་བྱ་སྟེ། དེ་ནས་བདག་ཉིད་ལྷའི་རྣལ་འབྱོར་པས་ཨ་ཀཱ་རོས་སྦྱངས་ཏེ། ནམ་མཁའ་མཛོད་ཀྱིས་སྤེལ་ལ། བདུད་རྩིའི་
རྒྱ་མཚོར་བསམས་ཏེ། དེའི་ཕྱོགས་རྣམས་སུ་ཧྲཱིཿལས་པདྨའི་ཚོགས་བསྐྱེད་པ་ལ་ཕྱོགས་སྐྱོང་བ་བཅུ་ལ་སོགས་པ་ཆོས་ཀྱི་སྲུང་མ་དང་། ཁྱད་པར་དུ་ཚངས་པ་དང་། དབང་པོ་དང་ལེགས་ལྡན་ལ་སོགས་འཁོར་དང་བཅས་པ་སྤྱན་དྲང་བར་བྱའོ། །ཡང་འབྱུང་པོའི་གཏོར་མ་གཉིས་པ་
ལ་ལྷའི་གདོན་ལ་སོགས་པ་དང་། ཟས་སུ་མདངས་ཟ་བ་ཀུན་ལ་སྦྱིན་པར་བྱའོ། །དེ་ཡང་ལྕེ་རྡོ་རྗེར་བསམ་ཞིང་། སྦུ་གུས་བདུད་རྩི་ཐམས་ཅད་ངོམས་ཤིང་ཚིམ་པར་བྱ་བ་དང་། འདོད་པའི་ཡོན་ཏན་ལྔ་ཆར་བཞིན་དུ་འབྱུང་བར་བསམ་མོ། །དེ་ནས་ཐ་མར་བཟོད་པར་གསོལ་ཏེ། གསོལ་
བ་བཏབ་ནས་ཡི་གེ་བརྒྱ་པ་བརྗོད་དོ། །ཐམས་ཅད་གཤེགས་སོ། །དེ་ཡང་འབྱུང་པོའི་གཏོར་མ་ནི་མངལ་དང་། ཁྲག་དང་། ཞག་དང་། ཤ་དང་། ཚིལ་དང་། རྐང་དང་། བཙས་པ་ཟ་བ་ལ་སོགས་པ་ལ་སྦྱིན་ནོ། །དེ་ཡང་། འབྱུང་པོ་གང་དག་རི་རབ་སྟེང་གནས་དང་། གཟུངས་ཀྱི་
ལྷའི་གདོན་དང་། བྱ་བ་ནས་ཟས་སུ་མངལ་ཟ་བ་ཀུན་གྱིས་སྦྱིན་ནོ། །གཏོར་མའི་ཆོ་ག་རྫོགས་སོ།། །།པཎྜི་ཏ་མ་ཧཱ་ཛ་ན་ཉིད་ཀྱི་རང་འགྱུར་ཡིན་གསུང་།། {D3096}༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་ཏ་ཐཱ་ག་ཏོཥྞཱི་ཥ་སི་ཏཱ་ཏ་པ་ཏྲེ་ནཱ་མ་དྷཱ་ར་ནི་བི་དྷི་ནཱ་མ། བོད་སྐད་དུ། འཕགས་

བསྡུས་ཏེ། །གཟུངས་ཀྱི་ཆོ་ག་དགོད་པར་བྱ། །དང་པོར་སྔགས་པ་དམ་ཚིག་ལྡན། །བྱང་ཆུབ་སེམས་ལྡན་རབ་སྦྱངས་ཏེ། །ཆོ་ག་བསྡུས་རྒྱས་མཁས་པ་ཡིས། །མཎྜ་ལ་ཆག་ཆག་བཏབ་སྟེང་དུ། །དྲི་ཡི་བྱུག་པ་ལེགས་བསྒོས་ཏེ། །དང་པོར་དབུས་སུ་པདྨ་ནི། །འདབ་བརྒྱད་གེ་སར་
