བུ་འབྱིན་པའི་སངས་རྒྱས་ཀྱིས། །ཤི་ལ་ཕན་གདགས་དོན་ཆེད་དུ། །མཆོག་སྦྱིན་རྡོ་རྗེ་མཆོག་ཁྱོད་འདུད། །དེས་ནི་གསོལ་བ་གདབ་བྱས་ནས། །ཕྱོགས་བཅུའི་སངས་རྒྱས་ནམ་མཁའ་ལ། །མངོན་སུམ་འདྲ་བར་གང་བར་བསྒོམ། །དེ་ནས་ཕྱོགས་བཅུའི་
རྒྱལ་བ་དང་། །དང་པོའི་སངས་རྒྱས་ལ་སོགས་པ། །ནམ་མཁའ་གང་བར་འདི་བཞེས་ལ། །དེང་བདག་འདུས་པའི་མཆོག་ཡིན་ཏེ། །དེ་ནས་ཏིལ་གྱིས་དེ་ལ་ནི། །བསྐོར་ཏེ་སྦྱིན་སྲེག་དབུལ་བར་བྱ། །ཤི་བ་སྟ་གོན་ལ་གནས་པའི་ཏིང་ངེ་འཛིན་ཏོ་ཞེས་བྱ་བས། རོ་སྟ་གོན་ལ་གནས་
པ་བསྟན་ཏེ། དེ་ཡང་དེ་ནས་རོ་སྟ་གོན་ལ་གནས་པ་ནི། ཤིན་ཏུ་དབེན་པའི་གནས་སུ། དཀྱིལ་འཁོར་འདོམ་གང་གྲུ་བཞི་པ་བྱས་ལ། ལྷ་རྣམས་ཀྱི་གནས་སུ་མེ་ཏོག་གི་ཚོམ་བུ་བཀོད་དེ། དབུས་སུ་ཁྲི་བྱས་ལ། རས་བཟང་པོ་ཡུག་གཅིག་གིས་དཀྲིས་ཏེ། དེའི་སྟེང་དུ་རོ་
བཞག་ལ། རྡོ་རྗེ་སློབ་དཔོན་གྱིས་དཀར་གསུམ་བཟའ་ཞིང་ཟ་གཅིག་པའི་སྨྱུང་བ་བྱས་ལ། རྡོ་རྗེ་འཆང་ཆེན་པོའི་ཕྱག་རྒྱ་ཆེན་པོའི་ཏིང་ངེ་འཛིན་གསུམ་དང་ལྡན་པར་བསྒོམས་ནས། དེ་ནས་ཡང་སྔོན་བཞིན་རོའི་བགེགས་བསྐྲད་ལ། དེའི་འོག་ཏུ་རྡོ་རྗེ་འཁོར་ལོའི་སྔགས་དང་ཕྱག་རྒྱས་
དཀྱིལ་འཁོར་ནམ་མཁའ་ལ་སྤྲུལ་ལ། ཕྱག་དང་མཆོད་པ་དང་བསྟོད་པ་ཡན་ཆད་སྔོན་བཞིན་བྱས་ལ། དེའི་འོག་ཏུ་ཚིག་འདི་བརྗོད་པར་བྱ་སྟེ། དེང་དུས་བདག་ཚེ་འབྲས་བུ་ཡོད། །དེང་བདག་འདུས་པ་མཆོག་ཡིན་ནོ། །ཞེས་པའི་བར་ཏེ། དེ་ལྟར་ཞག་ལྔའི་བར་དུ་ཉིན་ཞག་རེ་ལ་
ཐུན་བཞིའི་སྦྱོར་བས་ལྷ་ལ་མཆོད་པ་དང་རོ་ལ་ཡུངས་ཀར་རམ་བྱེ་མས་བྲབ་ཅིང་སྔགས་{𰵍རིངས་ཀྱི་,རིང་གི་}བཟླས་པ་བྱའོ། །དེའི་འོག་ཏུ་ནམ་མཁའ་ན་བཞུགས་པ་གཤེགས་སུ་གསོལ་ལ། དེ་ནས་ཞག་ལྔའི་ནང་པར་སློབ་དཔོན་གྱིས་ཏིང་ངེ་འཛིན་རྣམ་པ་གསུམ་སྔོན་བཞིན་བསྒོམས་ལ། དེ་ནས་
སྤོས་ཐོགས་ལ་སྤྱན་དྲངས་ཚིག་འདི་བརྗོད་པར་བྱ། ཀྱེ་དབང་བསྐུར་ངེས་ཏེ་བླ་ན་མེད། །ཕྱོགས་བཅུར་བཞུགས་པའི་སངས་རྒྱས་ཀུན། །མ་ལུས་བདག་ལ་དགོངས་སུ་གསོལ། །བདག་མིང་འདི་ཞེས་བྱ་བ་ཡིས། །འདི་ལ་ཕན་གདགས་དོན་ཆེད་དུ། །གནས་འདིར་སངས་རྒྱས་ཀུན་
