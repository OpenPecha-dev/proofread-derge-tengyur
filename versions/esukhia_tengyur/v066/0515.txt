རྗེས་མཚམས་སྦྱར་ནས་ཟླ་བའི་ངོ་བོར་བྱས་ཏེ། གསལ་བར་བྱས་ལ་དེའི་དབུས་སུ་བྲིས་པ་ལྟར་གནས་པའི་རང་གི་སྙིང་པོ་བསམས་ལ། ཨོཾ་ས་མ་ཡ་སྟྭཾ་ཞེས་བརྗོད་དོ། །ཨོཾ་བཛྲ་སཏྭ་ཨཱཿཞེས་པས་ཕབ་ལ། རང་གི་ལྷའི་མཚན་མར་གྱུར་པ་དང་། ཨོཾ་བཛྲ་
ཨཱཏྨ་ཀོ྅ཧཾ་ཞེས་བྱ་བ་ལ་སོགས་པའི་ང་རྒྱལ་བྱས་ཏེ། དེ་ནས་ཛཿཧཱུཾ་བཾ་ཧོཿཞེས་བརྗོད་དོ། །རང་གི་སེམས་དཔའི་གཟུགས་སུ་གྱུར་པར་བྱ་བ་ནི། ཨོཾ་མ་ཧཱ་ས་མ་ཡ་སྟྭཾ་ཧཾ་ཞེས་བྱ་བས་མེ་ལྕེ་མང་པོ་འཁྲུགས་པའི་སྦྱོར་བ་ལས། རང་གི་ལྷའི་སྐུར་གྱུར་པར་བརྟག་ཅིང་། ཨོཾ་མ་ཧཱ་ས་
མ་ཡ་སྟྭཾ་ཞེས་བྱ་བ་དང་། ཨོཾ་ས་མ་ཡ་སྟྭཾ་ཞེས་བྱ་བ་དང་། རྒྱབ་ཏུ་ཟླ་བ་གྲུབ་པར་བྱ་སྟེ། དེ་ནས་ཨོཾ་སུ་ར་ཏ་སྟྭཾ་ཞེས་བྱ་བ་དང་། ཨོཾ་ས་མ་ཡཱ་ཧཾ། ཨོཾ་ས་མ་ཡ་སཏྭ་ཨ་དྷི་ཏིཥྛ་སྭ་མཱཾ་ཞེས་བྱ་བ་དང་། དེ་ན་ཨོཾ་བཛྲ་མུཥྚི་ཞེས་བརྗོད་ལ། ཨོཾ་སཏྭ་བཛྲི་ཎི་སྟ་ནཾ་ཞེས་པ་དང་། གནས་
པ་གྲུབ་པར་བྱས་ལ། ཨོཾ་བཛྲ་དྷཱ་ཏྭཱི་ཤྭ་རི་ཧཱུཾ། བཛྲ་རཏྣ་ཨ་བྷི་ཥིཉྩ་མཱཾ་ཞེས་བརྗོད་དེ། ཨོཾ་བཛྲ་དྷཱ་ཏུ་ཞེས་པ་དང་། ཨོཾ་བཛྲ་བཛྲི་ཎི་ཧཱུཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་རཏྣ་བཛྲི་ཎི་ཧཱུཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་དྷརྨ་བཛྲི་ཎི་ཧཱུཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་ཀརྨ་བཛྲི་ཎི་ཧཱུཾ་ཨ་བྷི་ཥིཉྩ་མཱཾ། ཨོཾ་ཨཀྵོ་བྷྱ། ཨོཾ་རཏྣ་
སཾ་བྷ་བ། ཨོཾ་ཨ་མི་ཏ་བྷཱ། ཨོཾ་ཨ་མོ་གྷ་སིདྡྷི། ཨོཾ་བཛྲ་དྷཱ་ཏུ་ཞེས། དཔའ་བོའི་ང་རྒྱལ་བྱ་བ་དང་། དེ་ནས་གསང་བའི་མཆོད་པ་བརྗོད་དེ། དེ་ལྟར་ན་བསྟན་པ་ནི་རྒྱུད་འདི་ཉིད་ལས། །ཆོས་རྣམས་ཐམས་ཅད་དངོས་མེད་པར། སེམས་ཀྱིས་བསམ་པ་བྱས་ནས་ནི། །ཨ་ལས་ཟླ་
བའི་དཀྱིལ་འཁོར་བསམ། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྟོན་ཏོ། །དེ་ལྟར་ན་དམ་ཚིག་སེམས་དཔའ་གཉིས་ལས་གང་ཡང་རུང་བ་ཅིག་ཏུ་བདག་ཉིད་བསམས་ཏེ་དང་པོར་ཨོཾ་ཐམས་ཅད་རང་བཞིན་གྱིས་དག་གོ་ཞེས་བརྗོད་དེ། དེའི་དོན་གྱིས་ཆོས་ཐམས་ཅད་སྟོང་པ་ཉིད་དུ་
མོས་པས་བདག་ནམ་མཁའ་ལ་གནས་ཏེ། བདག་གི་སེམས་ཀྱི་དབང་ཙམ་ལས། དང་པོ་རླུང་དང་ཆུ་དང་དབང་ཆེན་ནག་པོ་དང་། སྔོན་པོ་དང་། དཀར་པོ་དང་། སེར་པོ་ཡི་གེ་ཡཾ་དང་བཾ་དང་ལཾ་དག་ལས་བྱུང་བར་བསམས་ཏེ། དེའི་དབུས་སུ་ཀཾ་གི་ཡི་གེ་སེར་པོ་
