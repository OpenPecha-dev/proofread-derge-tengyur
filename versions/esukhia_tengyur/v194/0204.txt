རབ་ཏུ་སྦྱིན་པ་སྟེ། ཡང་དག་པར་རབ་ཏུ་སྦྱིན་པ་ལ་བཞི་པའོ། །འདི་ལ་འབྱུང་ཁུངས་བྱེད་དོ་ཞེས་པས། ཨ་དྷི་ཀ་ར་ཎཾ། གནས་བཞི་ལ། ཥེ་བ་ནཾ། ཤཱི་བ་ནཾ། ནིཥྛཱ་བ་ནཾ། ནིཥྛཱེ་བ་ནཾ། ནི་ཤཱི་བ་ནཾ། ཥཱི་བུ་ཤི་བ་ལས་རེང་པོ་ཡང་ངོ་ཞེས་པས་ལྷུཊའོ། །མང་པོ་ལ་རིང་པོའོ། །ཀ་ར་ཎཾ་ཀྲིཉ་ལས་བྱེད་པ་པོ་
ལ་ཞེས་པས་ལྷུཊའོ། །མང་པོ་ཡང་ངོ་། །ཀྟིན་ནི་དཔེར་ན། པུ་ཎྱ་ཀྲྀ་ཏ། འབྲེལ་པ་འཆད་པར་འདོད་པའི་ལས་ལ་དྲུག་པའོ། །བྷོ་ཛ་ནཾ། བྷུཀྟིཿ། ཤྲིཏིཿ། བྷཀྟིཿ། སྟུ་ཏིཿ། བུདྡྷཿ། མ་ཏིཿ། བྷཱི་ཏིཿ པྲཱ་ཏིཿ པྲ་ནིཿ པ་བ་ནི། པཱ་ཏི་པཀྟིཿ། བུད་མེད་ལ་ཀི་ནའོ་ཞེས་པས་བུད་མེད་ཀྱི་རྟགས་ལ་
དངོས་པོ་ལ་ཀིན་ནའོ། །མཱ། སྠཱ། སཱ། གཱ། པི་བ། ཧ་ཀ །དཱ། དྷཱ་རྣམས་ཀྱི་ཧ་ལའོ། །ཞེས་པས་ཨི་ཉིད་དོ། །གྷཉ་ནི་དཔེར་ན། ཨོ་ད་ན་སྱ་པཱ་ཀཿ། པཱ་པ་སྱི་ཡོ་གཿ། ཏྱཱ་གཿ རཱ་གཿ བཱ་དཿ བྷེ་དཿ ཙྪེ་དཿ པྲ་པཱ་དཿ སྐནྡཿ གྲནྠཿ བྷངྒཿ སངྒཿ། ཞེས་པ་དེ་ལྟ་བུ་ལ་སོགས་པ་དངོས་པོ་ལ་གྷཉ་
སྟེ་བྱེད་པ་པོ་ལའང་ངོ་། །ཞེས་པས་གྷ་ཉའོ། །དེ་བཞིན་དུ། པཱ་དཿཔེ་ཤཿསྤརྴཿསཱ་རཿཨ་ནུ་བྷཱ་བཿཔ་བྷཱ་བཿ། ཞེས་པ་ཡང་པྲ་ལ་སོགས་པའི་བསྡུ་བའོ། །རྗེས་སུ་རྟོགས་པའི་དངོས་པོ་ནི། ཨ་ནུ་བྷཱ་བཿ། རབ་ཏུ་འགྱུར་བའི་དངོས་པོ་ནི། པྲ་བྷཱ་བཿ། ཀུ་པྲ་ལ་སོགས་པ་སུ་པའི་སྒྲུབ་པ་ལ་རྟག་ཏུའོ་ཞེས་པས་
