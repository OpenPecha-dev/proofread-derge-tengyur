མེད་པ་དང་ཏ་དབྱིའོ། །སྭ་སཱ། སྲིང་མོ། ནཉ་སྔོན་མའི་བི་པ་ཏིའི་ཨ་ཡིག་དབྱིའོ། །ནབྷཱཿ། ཚ་བོའམ་བུའི་བུ། ན་ཡ་ཏི་ལས་ཥུ་ཀའོ། ཨི་ཡིག་བརྗོད་དོན། ཀ་ཡིག་ཀི་ཏ་བྱ་བའི་དོན། ཊུ་ཉིད་ཨེ་ཉིད་དོ། ཎེཥྚཱཿ། ཏྭཀྵིའི་ཀ་ཡིག་དབྱེ། ཏྭ་ཥྚཱཿ། ཀྵ་དེའི་ད་ཡིག་གི་ཏ་ཡིག་གོ། ཀྵ་རྟཱ། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་
ཨོ་ཉིད་དོ། ཧོ་རཱཿ། པོ་ཏཱ། པྲ་ཤཱ་སུ་ལས་ཨིཊ་མེད་དོ། ཤཱ་སྟྭ། པྲ་ཤཱསྟྲཱ། དེ་བཞིན་དུ་གཞན་ཡང་བལྟ་བར་བྱའོ། ཨིའོ་བྱ་བའི་དོན་ལས་ཨི་ཕ་རོལ་ཏུ་འགྱུར་རོ། ཊུ་ཀྵུ་རུ་ཀུ་སྒྲ་ལའོ། ཁྱབ་བདག། ཌུ་ཀྲྀཉ་བྱེད། སྐྱེ་དགུའི་བདག་པོ། པ་ལའོ། དྲི་རྣམ་པར་སྒོ་སྲུང་བ། འཇོམས་པ་ལའོ། ཤྲཱྀ་
འཚོ་བ་ལའོ། སྦྱིན་སྲོག་མཁན་བྱིས་པ། ཤྲནྠ་གྲནྠ་གཞུང་སྒྲིག་སྟོན་པ། མཁན་པོ། བ་ལའོ། བ་ལའོ་སྡོམ་པ་ལའོ། བལླ་ཞེས་པ་མདོའི་བྱིངས་སོ། དྷྭ་ན་སྒྲ་ལའོ། ཨ་བ་རཀྵ་སྐྱོང་བ་ལའོ། ཧྲྀ་ཉ་འཕྲོག་པ་ལའོ། མ་ལ། མ་ལླ་འཛིན་པ་ལའོ། བྷ་ལ། བྷ་ལླ་ཡོངས་སུ་བཤད་པ་ལའོ། ཙུལླ་རྩེད་
འཇོ་བྱེད་པ་ལ། དེ་ལ་ཨི་ཀའི་ཨ་ཏ་ཨེངའོ་ཞེས་པས་ཨོ་ཉིད། ཨེ་ཙའི་ཨ་ཡ་ཨ་པ་ཨཱ་ཡ་ཨ་པ་རྣམས་སོ་ཞེས་པས་ཨ་བར་བརྒྱུར་རོ། ར་བིཿ། ཀ་བིཿ། ཨི་ཀའི་ཨཏ་ཨེངའོ་ཞེས་པས་ཨ་ཉིད་དོ། ད་རི། ཤ་རི། ཤྲནྠི། གྲནྠི བ་ལི་བ་ལླིཿ། དྷྭ་ནིཿ། ཨ་བིཿ། ཨི་ཀའི་ཨཏ་ཨེང་། ར་ཕ་རོལ་
ཉིད་དོ། ཧ་རིཿ། མ་ལླིཿ། དེ་བཞིན་དུ་གཞན་ཡང་བལྟ་བར་བྱའོ། ཨི་ཀ་མཐའ་ཉེ་ལས་ཀིའོ། ཨི་ཀ་ཞེས་པ་སོ་སོར། ཉི་མ། སྙན་ངག་བྱེད་པོ། བསྡུ་བ་བཟུང་བ་སྟེ། ཨི་ཨུན། རྀ། གཡངས་མདའ། གཞུང་སྒྲིག་ཚིག་ངམ་གརྡྷོ་བས་ལྡན་ནམ་ལྷ་མིན། ཡལ་ག། སྒྲ། ལུག །ལྀཀ་ཅེས་པ་ཨིཀའོ། མཐའི་
ཉེ་བ་ནི། ཁྱབ་འཇུག་མེ་ཏོག་རིགས་མཆོད་སྦྱིན་སྐྱེད་བུ། མཐའ་ཉེ་སྟེ། ཨི་ཀ་ཉེ་བའི་མཐའ་གང་ཡིན་པ་དེ་འདིར་ཨི་ཀ་མཐའ་ཉེའོ། དེ་ལས་ཏེ་ཨི་ཀ་ལས་མཐའ་ཉེ་ལས་བྱ་བའི་དོན་ལ་ཀི་ཕ་རོལ་དུ་འགྱུར་རོ། དེ་ལ་ཀྲྀ་ཥ་རྣམ་པར་བརྐོ་བ་ལའོ། ཤུ་ཙ་མྱ་ངན་ལའོ། རུ་ཙིར་གསལ་བ་ལའོ། ཀྵི་པ་འཕེན་
པ་ལའོ། ཀྲྀཥིཿཞིང་ལས་ཤུ་ཙིཿ། གཙང་བའམ་དབེན་པ། རུ་ཙིཿ། མཛེས་པ།། ཀྵིཔིཿ། གཡུལ་འཁྲུག་པ། ཀྲ་མུའི་ཨཏ་ཨིཏ་འང་ངོ་། །ཀྲ་མུ་རྐང་པ་སྐྱེད་པ་ལའོ། །ཨུ་ཡིག་ནི་ཨུ་དེཏ་ལས་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན། དེ་ལ་ཀི་ཕ་རོལ་དུ་འགྱུར་རོ། །ཨཏ་ཨིཏ་འང་ངོ་ཞེས་པས་ཀྲ་མའི་ཨི་ཉིད་དོ། མིཿ།
