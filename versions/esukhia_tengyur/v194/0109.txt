ཤྣུ་བྱིངས་བྷཱུ་རྣམས་ཀྱི་ཨཱི་ཨཱུ་དག་གི་ཨི་ཡ་ཨུ་ཝ་དག་གོ་ཞེས་པ་ཨི་ཡ་རུ་བསྒྱུར་རོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །ཨེཙ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་ནུཊ་ཡང་ནའོ། །ཚིག་གཅིག་ལ་རཱ་ཥ་ལས་
ནའི་ཎའོ། །ངིའི་ཨཱམ་དེ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །རྣམ་པར་གཅད་པའོ། །ཤྲཱིཿ། ཤྲི་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཾ། ཤྲི་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཱ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷིཿ། ཤྲི་ཡཻ། ཤྲི་ཡེ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ། ཤྲི་ཡཱཿ། ཤྲི་ཡཿ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ། ཤྲི་ཡཿ།
ཤྲི་ཡཱཿ། ཤྲི་ཡོཿ། ཤྲཱི་ནཱཾ། ཤྲི་ཡཱཾ། ཤྲི་ཡཱཾ། ཤྲི་ཡི། ཤྲི་ཡོཿ། ཤྲཱི་ཥུ། ཧེ་ཤྲཱིཿ། ཧེ་ཤྲི་ཡཽ། ཧེ་ཤྲཱི་ཡཿ། དེ་ལྟར། སྟྲཱི། སྟྲི་ཡཽ། སྟྲི་ཡཿ། སུ་དྷཱི་ཞེས་པ་འདིའི་ཤྲཱིའི་སྒྲ་བཞིན་ནོ། །སུ་དྷཱིཿ། སུདྷི་ཡཽ། སུ་དྷི་ཡཿ། སུ་དྷི་ཡཾ། སུ་དྷི་ཡཽ། སུ་དྷི་ཡཿ། བྷྲཱུ་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། ཤྣུ་
བྱིངས་བྷྲཱུ་རྣམས་ཀྱི་ཨིཡ་ཨུཝ་དག་གོ་ཞེས་པ་ཨུ་ཝ་རུ་བསྒྱུར་རོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །ཨེཙ་ལའོ། །ཨཀའི་ཨཀ་ལ་རིང་པོའོ། །དེ་ཨིཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་ནུཊ་ཡང་ནའོ། །ཚིག་གཅིག་ལར་ཥ་ལས་ནའི་ཎའོ། །དེ་ལ་ངིའི་ཨཱ་མའོ། །ཨཀའི་ཨཀ་ལ་
རིང་པོའོ། །རྣམ་པར་གཅད་པའོ། །བྷྲཱུཿ། བྷྲུ་ཝཽ། བྷྱ་ཝཿ། བྷྲུ་ཝཾ། བྷྲུ་ཝཽ། བྷྲུ་ཝཿ། བྷྲུ་ཝཱ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷིཿ། བྷྲུ་ཝཻ། བྷྲུ་ཝེ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷྱཿ། བྷྲུ་ཝཱཿ། བྷྲུ་ཝཿ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷྱཿ། བྷྲུ་ཝཱཿ། བྷྲུ་ཝ། བྷྲུ་ཝོཿ། བྷྲཱུ་ཎཱཾ། བྷྲུ་ཝཱཾ། བྷྲུ་ཝཱཾ། བྷྲུ་ཝི། བྷྲུ་ཝོཿ།
བྷྲཱུ་ཥུ། ཧེ་བྷྲཱུཿ། ཧེ་བྷྲུ་ཝཽ། ཧེ་བྷྲུ་ཝཿ། འདི་ལྟར། བྷཱུཿ། བྷུ་ཝཽ། བྷུ་ཝཿ། སྭ་ཡཾ་བྷུའི་སྤྲ་ནི་བྷྲའི་སྒྲ་བཞིན་ནོ། །ཀཱི་ལཱ་ཡ་བཱ་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། །ཨེ་ཙ་ལའོ། །དང་པོ་དག་གི །ཨ་ཙ་ལའོ། །ཨ་མ་ལ་སྔ་མའོ། །ཤམ་ལ་སོགས་པ་ཨ་ཙ་དང་པོ་ཅན་རྣམས་ལ་ཨཱ་ཏའིའོ་ཞེས་པ་ཨཱ་ཡིག་གི་དབྱིའོ། །རྣམ་པར་
