ལ་གོ་མིན་ནོ། །བཱ་ཙཱ་ལས་གྨི་ནིའོ། །སྨད་པ་ལ་ཨཱ་ལཙ། ཨཱ་ཊཙ་དག་གོ། །ཨཪྵས་ལ་སོགས་པ་རྣམས་ལས་ཨཙའོ། །ཏུཎྜི། བ་ལི། བ་ཊི་ལས་བྷའོ། །ཀམ། ཤམ་དག་ལས་སོ། །ཏི། ཏུ། བ། ཡས། ཏ་རྣམས་སོ། །ཡུས་སོ། །ཨཱུརྞྞཱ་ཨཧཾ། ཤུ་བཾ་རྣམས་ལས་སོ། །ལེགས་པར་བརྗོད་པ་དང་
མཉམ་པ་དག་ལ་ཚའོ། །འདོད་པ་དང་རྗེས་ཀྱི་ཚིག་དག་ལས་འཇིག་པ་ཡང་ནའོ། །བི་མུགྟ་ལ་སོགས་པ་རྣམས་ལས་ཨཎ་འོ། །གོ་སད་ལ་སོགས་པ་རྣམས་ལས་བུན་ནོ། །ནི་ཏྲཱ། ཏནྟྲཱ། ཤྲདྡྷཱ། ད་ཡཱ། ཧྲྀ་ད་ཡ་ལས་ཨཱ་ལུ་ཙ་ཡང་ནའོ། །ཤཱི་ཏ། ཨུཥྞ། ཏྲྀ་པྲ་ལ་མི་བཟོད་པ་ལའོ། །ཧི་མ་ལ།
བཟོད་པ་ལ་ཙེ་ལུའོ། །བ་ལ། བཱ་ཏ་ལ་ཙཱུ་ལའོ། །སྐབས་བཞི་པའི་རྐང་པ་གཉིས་པ་རྫོགས་སོ།། །།རྟེན་ཐ་དད་པ་ལ་དྲུག་པ་ལ་ཏས་སོ། །ནད་ལས་སེལ་བར་བྱེད་པ་ལའོ། །འཕེན་པ་འཛིན་པ་མི་འཁྲུགས་པ་རྣམས་ལ་བྱེད་པ་པོ་མ་ཡིན་པའི་གསུམ་པ་ལས་སོ། །ཧཱི་ཡ་ལྡན་པ་དང་
པཱ་པ་སྦྱོར་བ་ལས་སོ། །ལྔ་པ་ལས་པྲ་ཏིའིས་སོ། །འབྱུང་ཁུངས་ལ་ཧཱཀ །རུ་ཧ་དག་མ་ཡིན་པའིའོ། །དྭི་སོགས་རྣམས་མ་ཡིན་པ་ལས་སརྦ་ལ་སོགས་པ་དང་བ་ཧུ་རྣམས་ལས་སོ། །ཀུ་ཏས། ཨ་ཏས། ཨི་ཏས། ཨི་ཏས་སོ། །ཨཱ་དི་ལ་སོགས་པ་རྣམས་ལས་སོ། །བདུན་པ་ལས་ཏྲལ་ལོ། །ཀྭ་ཀུ་ཏྲ། ཨི་ཧ། ཨ་ཏྲའོ། །
བྷ་བཏ། དཱིརྒྷཱ་ཡུས། ཨཱ་ཡུཥྨཏ། དེ་བཱ་ནཾ། པྲི་ཡ་རྣམས་ཀྱིས་དེ་རྣམས་གཞན་ལས་ཀྱང་ངོ་། །སརྦྦ་ཨེ་ཀ །ཨ་ནྱ། ཀིམ། ཡཏ། ཏ་ཏ་རྣམས་ལས་དུས་ལ་དཱའོ། །སདྷཱ། ཨ་དྷུ་ནཱ། ཨི་དཱ་ནཱིམ། ཏ་དཱ་ནཱིམ་མོ། །ཀིམ། ཡཏ། ཨནྱ་ལས་{𰵄དི་རིང་,དེ་རིང་}མ་ཡིན་པ་ལརྷིལ་ཡང་ནའོ། །ཏརྷི། ཨེ་ཏརྷི། ས་དྱཿ། པ་རེ་
དྱ་མིའོ། །བཱུརྦྦ། ཨ་ནྱ། ཨ་ནྱ་ཏ་ར། ཨི་ཏ་ར། ཨ་པ་ར། ཨ་དྷ་ར། ཨུཏྟ་ར་ལས་ཨེ་དྱུས་སོ། །ཨུ་བྷ་ཡ་ལས་དྱུས་ཡང་ངོ་། །རྣམ་པ་ལ་ཐལ་ལོ། །གྲངས་ལས་དྷཱའོ། །ཥོ་ཌྷཱ་ཡང་ནའོ། །ཨཻ་ཀ་དྷྱཾ་མོ། །དྭི་ཏྲི་ལས་དྷ་མུཉའོ། །ཨེ་དྷཱའོ། །དེ་དང་ལྡན་པ་ལ་དྷ་ཎའོ། །ཛཱ་དཱི་ཡར་རོ། །སྠུ་ལ་ལ་སོགས་པ་
རྣམས་ལས་ཀན་ནོ། །ཕྱོགས་ཀྱི་སྒྲ་ལས་ཕྱོགས་ཡུལ་དུས་ཀྱི་དོན་ལས། བདུན་པ། ལྔ་པ། དང་པོ་རྣམས་ལས་ཨསྟཱ་ཏིའོ། །ཨཉྩ་ལས་འཇིག་གོ། །ཨུ་པ་རི། ཨུ་པ་རིཥྚ་ཏ་དོ། །པཱུརྠ་ཨ་དྷ་ར་དག་གི་པུ་ར་ཨ་དྷ་དག་ཀྱང་ངོ་། །ཨས་སོ། །ཨ་བ་རའི་ཨ་བའོ། །ཨསྟཱ་ཏི་ཡང་ནའོ། །པཉྩཱད་དོ། །པཉྩ་དྡྷ་མའོ། །
