པོའི་ལས་ནི་དཔེར་ན། ཤི་ཥྱངྒ་མ་ཡ་ཏི་གྲཱ་མངྒུ་རུཿ། པོ་ཡ་ཏི་ཤཱསྟ་མུ་པཱ་དྷྱ་ཡཿ། ཤི་ཥྱ་བོ་ཛྱ་ཏི་ཨོ་ད་ནཾ་དེ་བ་དཏྟཾ་ཡཛྙ་དཏྟཿ། པཱ་ཋ་ཏི་མཱ་ན་བ་ཀིམྦ་ད་མུ་པ་དྷྱཱ་ཡཿ། ཨཱ་ཤ་ཡ་ཏི་དེ་བ་དཏྟཾ་ཡཛྙ་དཏྟཿ། པཱ་ཤ་ཡ་ཏི་པ་ཏིམྦྷཱརྱ། འགྲོ་བ་དང་རྟོགས་པ་དང་། ཟས་དང་སྒྲའི་དོན་དང་ལས་མེད་པ་རྣམས་ཀྱི་སྦྱར་བྱ་ལའོ་ཞེས་པས་
སྦྱར་བར་བྱ་བའི་བྱེད་པ་པོ་ལ་གཉིས་པ་འགྱུར་བའོ། །བཤད་པའི་ལས་ནི་དཔེར་ན། ཀ་ཊཿ། ཀྲི་ཡ་ཏེ་དེ་བ་དཏྟེ་ན། བྷཱ་ར་དྷི་ཡ་ཏེ་དེ་བ་དཏྟེ་ན། ཨོ་ད་ནཿ། པ་ཙྱ་ཏི་ཤཱུ་དྲེ་ཎ། ཤཱསྟྲམྤ་ཋྱ་ཏེཙྪཱ་ཏཻཿ། བེ་དོ྅དྷི་ཡ་ཏེ་མཱ་ན་བ་ཀེ་ན། བཤད་པའི་ལས་ལ་དོན་ཙམ་ལ་དང་པོའོ་ཞེས་པ་དང་པོའོ། །བྱེད་པ་པོ་ལ་ཇི་སྐད་བཤད་
པ་དེ་བཞིན་དུ་འདིར་ཡང་སྦྱར་བར་བྱའོ། །མ་བཤད་པའི་བྱེད་པ་པོ་ནི་དཔེར་ན། ཀ་ཊཾ་ཀ་རོ་ཏི་ན་རཿ། ཀུམྦྷ་ནིཥྤཱ་ད་ཡ་ཏི་ཀུམྦྷ་ཀཱ་རཿ། ཀཱཎྜ་ལུ་ནཱ་དེ་བ་དཏྟཿ། སུ་བརྞྞ་ཀུཎྜ་ལ་ཀ་རོ་ཏི་སུ་བརྞྞ་ཀཱ་རཿ། གྲི་ཧ་བྲ་བི་ཤྱ་ཏི་ནཱི་རཱིཿ། བེེ་ད་མ་དྷཱི་ཏེ་མཱ་ན་བ་ཀ་བ་ཀཿ། དྷརྨྨ་ཤྲྀ་ཎོ་ཏི་པྲཛྙཿ། །བུདྡྷཾ་བྷ་ག་བཱ་ནཾ་བནྟ་ཏེ་ཤྲཱདྡྷཿ།
དེ་ལྟར་ཡང་བཤད་པ། །ངེས་འཇུག་རྣམ་འགྱུར་ཐོབ་བྱ་དང་། །འདི་དག་རྣམ་པ་གསུམ་དུ་འདོད། །བྱེད་པ་པོ་དང་བྱ་བ་ལས། །རེ་ལྡེ་ཚེར་མ་ཁྱིམ་དུ་བརྗོད། །དེ་ལ་མ་བཤད་པའི་ལས་ནི་བསྡུ་བ་ལ་སོགས་པ་རྣམས་ཀྱིས་བཤད་པའི་ཕྱིར་རོ། །བྱ་བའི་ལས་ལ་གཉིས་པའོ། །ཞེས་པས་རྣམ་དབྱེ་གཉིས་པ་
འགྱུར་རོ། །བཤད་པའི་ལས་ནི་ཕྱི་དང་ནང་གི་བྱེད་པས་རྣམ་པ་གཉིས་ཏེ། ནང་གི་ནི་ལུས་ཀྱི་ཡན་ལག་ལ་བརྟེན་པའོ། །ཕྱིའི་ནི་དེ་ལས་བཟློག་པའོ། །དེ་སྐད་དུ་ཡང་། བརྗོད་པར་འདོད་པའི་བྱ་བའི་ལས། །བྱེད་ཅན་འདི་ནི་མཆོག་ཏུ་གྲུབ། །བྱེད་པ་ཕྱི་དང་ཡིད་ཀྱང་རུང་། །རྣམ་གཉིས་མིག་དང་ཟོར་བ་སོགས། །
ཞེས་སོ། །ནང་གི་ནི་དཔེར་ན། ནེ་ཏྲ་ཎ་ནི་རཱཀྵི་ཏ་སུ་ཏྲཱཉྫ་ན་ཀཿ། བྱ་བའི་ཁྱད་པར་ཡང་བྱ་བའི་ལས་ཉིད་ཀྱི་ཕྱིར་གཉིས་པའོ། །སྟོ་ཀཾ་ཙ་ལ་ཏི། གོ་དོ་ཧ་མ་སྟེ། ཨོ་ད་ན་སྲ་ཀ་སེ་ཏ་ཞེས་སོ། །དཔེར་ན། མཱ་ན་སཱ་པཱ་ཊ་ལི་པུ་ཏྲཱཾ་གཙྪ་ཏི། གྷྲ་ཎེ་ན་པུཥྤཱ་ནི་ཛི་གྷྲ་ཏི་ལོ་ཀཿ། ཛི་ཧྲ་ཡཱ་ར་ས་མཱ་སྭཱ་ད་ཡ་ཏི།
ཤྲ་བཎ་ན་ཤབྷཱ་ན་གྲྀཧྞཱ་ཏི། ཀཱ་ཡེ་ན་སུ་ཁ་མ་ནུ་བྷ་བ་ཏི་མོ་ཀཿ། ཕྱི་རོལ་གྱི་ནི་དཔེར་ན། དཱ་ཏྲེ་ཎ་དྷཱ་ནྱ་ལུ་ནཱ་ཏི། པ་ར་ཤུ་ནཱ་ཙྪི་ནནྟི། ཨ་ཤྭེ་ན་པྲཛ་ཏི། ཧསྟི་ཏ་གཙྪ་ཏི་བུཥྞུ་མི་ཏྲཿ། པྲཛྙ་ཁཌྒེ་ནཱ་ཛྙཱ་ནཾ། ཧིནྟ། བྷ་ག་བཱན་མཉྫུ་གྷོ་ཥ་ལོ་ཀ་སྱ་སརྦ་དྷ། ཀྲི་ཡ་ཏེ་ཡ་མ་ཏི་ཀ་ར་ཎི། ཨསྱ་བཱ་ཙ་སྱ་ཏྲིཔྟཱི་ཀ་རོ་ཏི། བྱེད་
