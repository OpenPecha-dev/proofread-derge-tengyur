རྟགས་དོན། མ་ན་ཤེས་པ་ལའོ། །ནའི་ཨ་བརྗོད་དོན། ཏ་སུ། ད་སུ། ཉེ་བར་འཕེན་པ་ལའོ། །ཨུ་ཡིག་ནི་ཨུ་དི་ཏ་ལས་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན། བྷུ་ཛ་སྐྱོང་བ་ལའོ། །ཛའི་ཨ་ཡིག་ཀློག་དོན་ཏོ་ཞེས་པ་འདི་རྣམས་ལས་བྱ་བའི་དོན་ལ་དཀྱུས་ཕ་རོལ། ཀ་ཡིག་ནི་ཨཏ་
ཨེང་དགག་པའི་དོན། ས་ཡིག་ནི་ས་མེད་པའི་ཡུ་བུ་དག་གི་ཨ་ན་ཨཀ་དག་གོ་ཞེས་པའི་རྟགས་དོན་ཏེ་ཚིག་ཕྲད་ལས་ཡུ་ཨ་ནར་མི་འགྱུར་རོ། ཛནྻུཿ། མ་ནྻུཿ། ད་སྻུ། བྷུ་ཛྻུཿ། ཧྲཱི་ཨི་ཥི་ཀྲྀ་ཥྀ་རྣམས་ལས་ཀུཀ་སུཀ་ཨཱ་ནུཀའོ། །ཧྲཱི་ངོ་ཚ་བ་ལའོ། །ཨིཥུ་འདོད་པ་ལའོ། །ཨུ་ཡིག་ནི་
ཨུ་དིཏ་ལས་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན་ཏོ། །ཀྲྀ་ཤ་ཕྲ་མོར་བྱེད་པ་ལའོ། །ཤའི་ཨ་བརྗོད་དོན་ཞེས་པ་འདི་རྣམས་ལས་རིམ་བཞིན་ཀུཀ་དང་། སྲོག་ཅན། ཁྲོ་བ། རྐུན་པོ། ཟིན། སུཀ་དང་ཨ་ནུ་ཀ་རྣམས་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ནི་ཀིཏ་ལའོ་ཞེས་པས་ཨཏ་ཨེང་དགག་པའི་དོན། ཧྲཱི་ཀུཿ། ཀ་པི་
རི་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པས་ལ་ཉིད་དོ། །ཧླཱི་ཀུཿདེ་ཉིད། ས་ལ་ཥ་ཌྷ་དག་གི་ཨའོ་ཞེས་པས་ཥའི་ཀའོ། །ཀུ་ལས་ཀྱང་ཞེས་པས་ཥ་ཉིད་དོ། །ཨིཀྵུཿ། ཀྲྀ་ཤཱ་ནུཿ། མྲྀང་ལས་ཏྱུཀའོ། །མྲྀང་སྲོག་གཏོང་བ་ལའོ། །ང་ཡིག་ནི་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། །འདི་ལས་ཏྱུཀ་ཕ་ཞུམ་པ། རོལ་
ཀ་ཡིག་སྔར་བཞིན་དབྱིའོ། །མྲྀ་ཏྱུཿ། ཤཱིང་ལས་དྷུཀའོ། །ཤཱིང་ཉལ་བ་ལའོ། །ང་ཡིག་ནི་ལིཊ་མ་ཡིན་པ་ལ་ཤིང་གི་ཨེཏའོ་ཞེས་པའི་རྟགས་དོན། བུད་ཤིང་། མེ། འདི་ལས་ཌྷྱུག་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ནི་ཀིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། །དགག་བའི་དོན་ཏོ། །ཤཱི་དྷུཿ། པྲཱྀ་ལས་ཤུན་
དང་ཨ་ཤུཙ་དག་གོ། །པྲཱྀ་སྐྱོང་འཆིང་བ། བ་ལ་འོ། །འདི་ལས་ཤུན་དང་ཨ་ཤུ་ཙ་ཕ་རོལ་ཏེ། ན་དང་ཙ་ཡིག་དབྱངས་དོན་ནོ། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད། ར་ཕ་རོལ་ལོ་རེ་ཕའི་ཨ་ཡིག་བརྗོད་དོན། པཪྴུཿ། པ་ར་ཤུཿ པཱི། མཱི་དག་ལས་རུའོ། །པཱིང་བཏུང་བ་ལའོ། ཆང་གི་ཁྱད་པར། མཱིང་འཚེ་
བ་ལའོ། །ང་ཡིག་ནི་སྟོང་ལྡན་པ་ལས་ཧལ་དང་པོ་ང་མེད་པ་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ནོ། །འདི་དག་ལས་རུ་ཕ་རོལ་ཏུ་འགྱུར། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཉིད། པེ་རུཿ། ཉི་མ། མེ་རུཿ། རིའི་རྒྱལ་པོ། །སུ་སྔ་དགྲ་སྟ། རྩིབ་མའི་རུས་པ། མ་ལས། སུ་མེ་རུཿ། ཛ་ཏྲུ་ལ་སོགས་པ་རྣམས་སོ། །ཛ་ན་
