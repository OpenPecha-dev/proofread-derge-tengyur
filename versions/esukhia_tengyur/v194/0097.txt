ཨ་ཡ་ཨཝ་ཨཱ་ཡ་ཨཱཝའོ། །དེ་སི་ངས་དག་ལའོ་ཞེས་པ་ཕ་རོལ་སྔ་མ་ཉིད་དོ། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ་ཞེས་པ་ཡཎ་རུ་བསྒྱུར་བའོ། །ཐུང་དུ་ཨཱ་པ་ལ་ནུ་ཊའོ། ཨུ་ཡིག་དག་དང་ཊ་ཡིག་དག་སྔ་མ་བཞིན་དབྱིའོ། །ནཱ་མཱ་ལ་ཏིསྲྀ་ཙ་ཏ་སྀ་མ་ཡིན་པ་དག་ལའོ་ཞེས་པ་རིང་བོ་ཉེ་དེ་དོ། ཨེ་བའི་
ཨ་ཏའང་ངོ་ཞེས་པ་ཨཽ་ཉིད་དོ། ཨེ་ཡིག་གི་ཡང་ཨ་ཡིག་གོ། །ཨེ་ཙ་ལའོ། །ཀུན་ལས་ཀྱང་ཞེས་པ་ཥ་ཉིད་དོ། །ཀུན་དུ་བོད་པ་ལ་ཨིཏ་ཨུཏ་དག་གི་ཨེང་ཉིད་དོ། །ཨེང་ཐུང་ངུ་ཨཏ་ལས་ཀུན་དུ་བོད་པ་ལའོ་ཞེས་པ་དབྱིའོ། མཐའི་ས་ཡིག་རྣམས་ཀྱི་རུ་ཉིད་དེ། །རྣམ་པར་གཅད་པའོ། །ཨགྣིཿ། ཨགྣཱི།
ཨགྣ་ཨཿ། ཨགྣི་མ། ཨགྣཱིནཱ། ཨགྣིནཱ། ཨགྣི་བྷྱཱཾ། ཨགྣི་བྷིཿ། ཨགྣི་བྷིཿ། ཨགྣ་ཡེ། ཨགྣི་བྷྱམྲ། ཨགྣི་བྷྱཿ། ཨགྣེཿ། ཨགྣི་བྷྱཱམྲ། ཨགྣེ་བྷྱཱཿ། ཨགྣེཿ། ཨགྣོ། ཨགྣཱི་ནཱཾ། ཨགྣཽ། ཨགྣྱཽཿ། ཨགྣིཥུ། ཧེ་ཨགྣྭི། ཧེ་ཨགྣ་ཡཿ། ས་ཁི་ཞེས་པ་འདིའི། རི་དུ་གནས། པུ་རུ་དཾ་ཤ་ནེ་ཧ་རྣམས་
ཀྱིའང་ཨ་ནང་སུ་ལའོ་ཞེས་པ་ཨ་ནང་རུ་བསྒྱུར་རོ། ང་ཡིག་ནི་ངིདའི་དོན་ཏོ། ན་ཡིག་ལ་ཨ་ཡིག་ནི་བརྗོད་པའི་དོན་ཏོ། ཀུན་དུ་བོད་པ་མ་ཡིན་པའི་སུ་ལའོ་ཞེས་པའི་རིང་པོ་ཉིད་དོ། སུའི་སུ་དབྱིའོ། །སུ་པའི་རང་བཞིན་གྱི་ཞེས་པ་ནི་དབྱིའོ། །ཤི་མ་ཡིན་པ་ལས་ཁིའི་ཨེ་ཏའོ་ཞེས་པ་ཨེ་ཉིད་དོ། །ཨེ་ཙའི་
ཨ་ཡ་ཨ་ཝ་ཨཱ་ཡ་ཨཱ་ཝའོ། །དང་པོ་དག་གི་ཨ་ཙའོ། །དེ་ལས་ཤས་ཀྱི་ན་སྐྱེས་པ་ལའོ། །ཡི་ཀའི་ཡཎ་ཨ་ཙ་ལའོ། །ས་བྷྱུཿ། པ་ཏྱུཿཞེས་པའི་ཚིག་ཕྲད་ལས་ཁྱུའོ། །ཐུང་ངུ་ཨཱ་པ་ལས་ནུ་ཊའོ། །ནཱ་མ་ལ་ཏ་སྲྀ་ཙ་ཏ་སྲྀ་མ་ཡིན་པ་དག་ལའོ། །ཨི་ཏ་ཨུ་ཏ་ན་ག་ལས་ཨཽ་ཏའོ་ཞེས་པ་ཨཽ་ཉིད་དོ། །ཨཎ་རུ་བསྒྱུར་
བ་སྟེ། ཨི་ཀའི་ཡཎ་ཨ་ཙ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་ཥ་ཉིད་དོ། །ཀུན་དུ་བོད་པ་ལ་ཨི་ཏ་ཨུ་ཏ་དག་གི་ཨེ་ངའོ། །ཨེང་ཐུང་དུ་ཨ་ཏ་ལས་ཀུན་དུ་བོད་པ་ལའོ། །སའི་རུ་ཉིད་ད་རྣམ་པར་གཅད་པའོ། །ས་ཁཱ། ས་ཁ་ཡཽ། ས་ཁཱ་ཡཿ། ས་ཁཱ་ཡཾ། ས་ཁཱ་ཡཽ། ས་ཁཱིན། ས་ཁྱཱ། ས་ཁིབྷྱཱཾ། ས་ཁི་བྷིཿ། ས་ཁྱེ།
ས་ཁི་བྷྱཱཾ། ས་ཁི་བྷྱཿ། ས་ཁྱུ། ས་ཁི་བྷྱཾཿ། ས་ཁི་བྷྱཿ། སཁྱུ། ས་ཁྱོཿ། ས་ཁཱི་ནཱཾ། ས་ཁྱཽ། སཁྱོཿ། ས་ཁྱོཿ། ས་ཁྱིུཥ། ཧེ་ས་ཁེ། ཧེས་ཁཱ་ཡཽ། ཧེས་ཁཱ་ཡཿ། པ་ཏི་ཞེས་པ་འདིའི་སུ་ལ་སོགས་པའི་སྔ་མ་བཞིན་ཕྱིས་པ་ལས་ཀྱང་། སྔ་མ་བཞིན་དུ་སུའི་རུ་ཉིད་དོ། རྣམ་པར་གཅད་པའོ། དང་པོ་དག་གི་ཨ་ཙ་
