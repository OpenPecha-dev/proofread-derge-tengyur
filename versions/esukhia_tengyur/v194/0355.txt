ཙཱརྻཱཿ། ཨ་བ་དྱ། པ་ཎྱ། བརྻྰ་རྣམས་སྨད་བྱ། བཙོང་བྱ། འགོག་པ་མེད་པ་རྣམས་ལའོ། །ཚིག་ཕྲད་དེ། སྨྲ་བར་བྱ་བ་མ་ཡིན་ཅིང་སྨད་པའི་ཕྱིར་ཨ་བ་དྱཾ་སྟེ་སྨད་བྱའོ། །ཨ་ནཱུ་དྱ་མ་ནྱར། ཐ་སྙད་དུ་བྱ་བས་ན་པ་ཎྱཾསྟེ་བཙོང་བྱའོ། །པ་ཎྱ་མ་ནྱཏ། དོན་དུ་གཉེར་བས་བརྒྱ་ལས་འདམ་
པས་ན་བཱརྻ་ཏེ་འགོག་པ་མེད་པའོ། །སུ་གྲཱི་བོ་ནཱ་མ་བརྻོ་སོ་པ་ཎྱ་བྷྲཱ་ཏི་བ་དྷཿ། ཀ་པིཿ། བདག་པོ་འདམ་པས་བརྻྰ་ཞེས་ཁ་ཅིག་གོ། །གཞན་དུ་ན་བྲྀ་ཏྱཱ་ནི་འགོག་པ་མེད་པ་ཞེས་པའི་དོན་ཏོ། །བྲྀང་ལས་བུད་མེད་ལ་ཞེས་པ་ཁ་ཅིག་སྟེ། དེས་ན་བཱརྻྰ། དུས་མཆོད་དོ། །བ་ཧ་ལས་བྱེད་པ་ལའོ། །ཡཏ་
འགྱུར་ཏེ། འདིས་བྱེད་དོ་ཞེས་པས་བ་ཧྱཾ་སྟེ། ཤིང་རྟ་བྱེད་པར་རློམ་པའོ། །ཨརྟི་ལས་ཇོ་བོ་རྗེ་རིགས་དག་ལའོ། །ཡཏ་འགྱུར་ཏེ། ཨརྻཿཇོ་བོའོ། ཨརྻཿརྗེ་རིགས་སོ། །གཞན་དུ་ཨཱརྱའོ། །ཨུ་པ་སརྻྰ་དུས་ལྡན་མའོ། །དུས་ནི་བུད་མེད་རྣམས་ཀྱི་མངལ་འཛིན་པའི་དུས་ལ་བརྗོད་དོ། །མངལ་
གྱིས་ཉེ་བར་སྐྱེད་པ་ལ་ཨུ་པ་སརྻྰ་ཚིག་ཕྲད་དེ་གལ་ཏེ་དུས་ལྡན་མ་ལའོ། །གཞན་ལ་ཨུ་པ་སཪྻཱ། ཨ་ཛརྻཾ་ཡང་དག་པར་སོང་བ་ལའོ། །རྒས་པ་མེད་པ་ལ་ཨ་ཛརྻཾ་ཚིག་ཕྲད་དེ། གལ་ཏེ་བྱམས་པ་ནའོ། །ཨ་ཛརྻཾ། འཕགས་པ་ཡང་དག་པར་སོང་བའོ། །དེས་ན་ཡང་དག་པར་སོང་བའོ། །དེས་ན་ཡང་དག་པར་སོང་བ་ལ། ཨཱརྻེ་ཎ་རཱ་
མཱ་ཛརྻཾ་ཀུ་རུ་དྲུ་ཏཾ། ཡང་དག་པར་སོང་བ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨ་ཛ་རི་ཏཱ་ཀམྦ་ལ། གྷ་དང་བྷཱུ་ལས་བྱེད་པ་པོ་ལའང་ངོ་། །གཱ་ཡ་ཏི་དང་བྷཱུ་ལས་ཀྱང་བྱེད་པ་པོ་ལ་ཡཏ་འགྱུར་ཏེ། འང་ཡིག་ལས་དངོས་པོ་དང་ལས་དག་གོ། །གེ་ཡཿ། གེ་ཡཾ། བྷ་བྱཿབྷ་བྱཾ། ཧན་ལས་བ་དྷའང་ཡང་ནའོ། །ཧནྟི་ལས་ཡཏ་
ཡང་ན་འགྱུར་ཞིང་དེ་ཚིག་ཕྲད་ལས་བ་དྷ་རུ་བསྒྱུར་བའང་ངོ་། །བདྷྱཿ། གྷཱ་དྷཿ། སུཔ་ལས་བད་ལས་ཀྱཔ་འགྱུར་ཏེ། ཡཏའང་ངོ་། །བྲ་ཧྨ་བ་དྱཾ། ས་ཏྱོ་དྱཾ། ས་ཏྱ་པ་དྱཾ། བྱེད་པ་པོ་ལང་བལྟ་བར་བྱ་སྟེ། ས་ཏྱ་བ་དྱ་ར་གྷཱུཏྟ་མཿ། ཨ་ནྲྀ་ཏོ་དྱཾ་ན་ཏ་ཏྭ་སྟི་བ་དྱ་བྲ་པཱི་མི་ཏེ། སུཔ་ལས་ཞེས་པ་ཅི་ཞེ་ན། པ་དྱཾ། ཉེར་
བསྒྱུར་མེད་པ་ལས་ཞེས་པ་ཉིད་དེ། ཨ་བྷི་བཱ་དྱཾ། ཀ་ཐ་མ་ནཱུ་དྱཾ། ཨ་ནུ་བ་དྷྱཾ། ཉེར་བསྒྱུར་མེད་པ་ལ་ཨ་ནུ་ཡིས་འགྱུར་རོ། །བྷཱུ་ལས་དངོས་པོ་ལའོ། །སུཔ་ལས་ཕ་རོལ་བྷ་བ་ཏི་ལས་དངོས་པོ་བརྗོད་པ་ལས་ཀྱཔ་འགྱུར་ཏེ། བྲཧྨ་བྷཱུ་ཡ་ག་ཏཿ། ཚངས་པ་ཉིད་ཐོབ་ཅེས་པའི་དོན་ཏོ། །སུཔ་ལས་ཞེས་
