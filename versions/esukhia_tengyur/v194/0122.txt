ཡ། པཱུརྦཱ་བྷྱཱཾ། པཱུརྦཱ་བྷིཿ། པཱུརྦ་སྱཻ། པཱུརྦཱ་བྷྱཱཾ། པཱུརྦ་བྷྱཿ། པཱུརྦ་སྱཻ། པཱུརྦ་བྷྱཱཾ། པཱུརྦ་བྷྱཿ། པཱུརྦ་སྱཱ། པཱུརྦ་ཡོཿ། པཱུརྦ་སཱཾ། པཱུརྦ་སྱཾཱ། པཱུརྦ་ཡོཿ། པཱུརྦ་སུ། མ་ནིང་ལ། པཱུརྦཾ། པཱུརྦེ། པཱུརྦཱ་ཎི། ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཨཀ་ཙ་སྟེ། པཱུརྦ་ཀཿ། པཱུརྦ་ཀཽ །པཱུརྦ་ཀེ །བུད་མེད་ལ། པཱུརྦ་ཀཱཿ།
པཱུརྦ་ཀེ །པཱུརྦ་ཀཱཿ། མ་ནིང་ལ། པཱུརྦ་ཀཾ །པཱུརྦ་ཀེ །པཱུརྦ་ཀཱ་ཎི། འདི་ལྟར། པ་རཿ། པ་རཽ། པ་རེ། བུད་མེད་ལ། པ་རཱ། པ་རེ། པ་རཱཿ། མ་ནིང་ལ། པ་རཾ། པ་རེ། པ་རཱ་ཎི། འདི་ལྟར། ཨ་ཝ་རཿ། ཨ་ཝ་རཽ། ཨ་ཝ་རེ། བུད་མེད་ལ། ཨ་ཝ་རཱ། ཨ་ཝ་རེ། ཨ་ཝ་རཱཿ། མ་ནིང་ལ། ཨ་ཝ་རཱཾ། ཨ་
ཝ་རེ། ཨ་ཝ་རཱ་ཎི། འདི་ལྟར། དཀྵི་ཎཿ། དཀྵི་ཎཽ། དཀྵི་ཎེ། བུད་མེད་ལ། དཀྵི་ཎཱཿ། དཀྵི་ཎེ། དཀྵི་ཎཱཿ། མ་ནིང་ལ། དཀྵི་ཎཾ། དཀྵི་ནེ། དཀྵི་ཎཱ་ནི། འདི་ལྟར། ཨུ་ཏྟར། ཨུ་ཏྟ་རཽ། ཨུ་ཏྟ་རེ། བུད་མེད་ལ། ཨུཏྟ་རཱཿ། ཨུཏྟ་རེ། ཨུཏྟ་རཱཿ། མ་ནིང་ལ། ཨུཏྟ་རཾ། ཨུཏྟ་རེ། ཨུཏྟ་རཱ་ཎི། འདི་ལྟར།
ཨ་པ་རཿ། ཨ་པ་རཽ། ཨ་པ་རེ། བུད་མེད་ལ། ཨ་པ་རཱ། ཨ་པ་རེ། ཨ་པ་རཱཿ། མ་ནིང་ལ། ཨ་པ་རཾ། ཨ་པ་རེ། ཨ་པ་རཱ་ཎི། འདི་ལྟར། ཨ་དྷ་རཿ། ཨ་དྷ་རཽ། ཨ་དྷ་རེ། བུད་མེད་ལ། ཨ་དྷ་རཱ། ཨ་དྷ་རེ། ཨ་དྷ་རཱཿ། མ་ནིང་ལ། ཨ་དྷ་རཾ། ཨ་དྷ་རེ། ཨ་དྷ་རཱ་ཎི། འདི་ལྟར། སྭཿ། སྭཽ།
སྭེ། བུད་མེད་ལ། སྭཱ། སྭེ། སྭཱཿ། མ་ནིང་ལ། སྭཾ། སྭེ། སྭཱ་ནི། འདི་ལྟར། ཨནྟ་རཿ། ཨནྟ་རཽ། ཨནྟ་རེ། བུད་མེད་ལ། ཨནྟ་རཱ། ཨནྟ་རེ། ཨནྟ་རཱཿ། མ་ནིང་ལ། ཨནྟ་རཾ། ཨནྟ་རེ། ཨནྟ་རཱ་ཎི། འདི་རྣམས་ནི། བཱུརྦ་ལ་སོགས་པ་རྣམས་སོ། །ཏྱད་ཅེས་པ་འདིའི། ཏྱད་རྣམས་ཀྱི་ཏྱས་ལ་
