ཏིཥྠ་ཏིའི་ས་ཡིག་འོ། ཥ་དབྱི་བ་དང་ཎ་ཉིད་ཀྱང་ཚིག་ཕྲད་དོ། སྠཱ་ནཱ། པཱི་ནཱ། ཏྲྀ་ཥའི་ཥ་དབྱི་བ་ཚིག་ཕྲད་དོ། ཏྲཱྀ་ཎི། རྫ། ཨརྟིའི་ཨཏ་ཨེང་དང་ར་ཕ་རོལ་ཉིད་དང་ཎ་ཉིད་ཀྱང་ངོ་། ཨརྞཿ། བིཊ་པཿ། ཛཱི་རྞྞཿ། རྒས་པ། དེ་བཞིན་དུ་གཞན་ཡང་
བལྟ་བར་བྱའོ། ཀྲྀ། པྲྀ། ཏྲྀ། ཡམི། དཱ་རི། ཨ་རྫི་རྣམས་ལས་ཨུ་ན་ནའོ། ཌཱ་ཀྲྀཉ། ཀ་བ། གི་ཝི། བྱེད་པ་ལའོ། བྲྀཉ་སྒྲིབ་པ་ལའོ། ཏྲྀ་རྐྱལ་པ་ལའོ། ཡ་མུ་ཉེ་བར་འགོག་པ་ལའོ། ཏྲཱྀ་རྣམ་པར་འཇོམས་པ་ལའོ། སྦྱོར་བ་པོའི་བྱ་བ་ལ་ཎི་ཙའི་ཨ་ཙའི་འོ་ཞེས་པས་ཨཱ་ཉིད། ར་ཕ་རོལ།
ཨརྫ། སརྫ་སྒྲུབ་པ་ལའོ། འདི་རྣམས་ལས་ཨུ་ནན་ཕ་རོལ་ཏུ་འགྱུར་རོ། ཨ་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། ར་ཥ་ལས་ཎ་ཉིད་དོ། ཌཱ་པ། ཀ་རུ་ཎཱཿ། ཀ་པི་རི་ཀ་ལ་སོས་པ་རྣམས་སོ་ཞེས་པས་ལ་ཉིད། རྒྱུ་མཚན་མེད་ན་རྒྱུ་མཚན་ཅན་གྱི་ཡང་
མེད་དོ་ཞེས་པས་ཎའི་ནའོ། ཏ་ལུ་ནཿ། ཌཱ་པ། ཡ་བུནཱཿ། ཎིའི་ཨིཊ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཎི་དཔྱིའོ། དཱ་རུ་ཎཱཿ། ལ་རྫུ་ཎ། ཤིང་གི་ཁྱད་པར་རམ་དཀར་པོའང་ངོ་། །སྙིང་རྗེའམ་བརྩེ་བ། ཆུ་ལྷའམ་ཤིང་གི་རྒྱལ་པོ། ལང་ཚོ། ཤ་ཀི་ལས་ཨུཎའོ། །ཤླྀ་ནུས་པ་ལའོ། །ལྀ་ཡིག་ན་ལྀ་དིཏ་བྻུ་ཏ་སོགས་སུ་ཥ་སོགས་
རྣམས་ལ་ཨ་ཏང་ལའོ་ཞེས་པའི་ལང་ཚོ། རྟགས་ཆུ་ཀླུང་དོན་ཏོ། །འདི་ལས་ཨུ་ནན་ཕ་རོལ་ཏུ་འགྱུར་རོ། དྲག་ཤུལ། སྲིང་སྒྲུབ་བམ་སྒྲུབ་བྱེད། ཤ་ཀུ་ནཿ། པཱ་ཏ་ལིས་རྣམས་ལས་སའོ། །པྲཱྀ་རྫོགས་པ་ལའོ། །པཱ་བཏུང་བ་ལའོ། །ཏ་ལ་རབ་ཏུ་གནས་པ་ལའོ། །འདི་ལས་པ་ཕ་རོལ་ཏུ་འགྱུར་
རོ། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད། རྀ་ཀའི་ཨཎ་ལ་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད། པརྦཿ། པཱ་པ། །ཏལྤཾ། སྟུ་ལས་ འདབ་ཆགས། ཨཱུཏ་ཀྱང་ངོ་། །ཥྚུཉ་བསྟོད་པ་ལའོ། །ཥྛི་བུ་ཥྭཀྐ་མ་ཡིན་པའི་དང་པོའི་ཥའི་སའོ། །རྒྱུ་མཚན་མེད་ན་ཞེས་པས་ཊའི་ཏ་ཡིག་གོ། །འདི་
ལས་པ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་ཨུའི་ཨཱུཏའོ། །སྟཱུ་པཿ། མཆོད་རྟེན། ཡུ། ཀུ། སུ། རྣམས་ལ་སྟེ་ཀིཏའང་ངོ་། །ཁྱིམ། སྡིག་པ། མ་ལས། ཡུ་སྤེལ་བ་ལའོ། །ཀུ་སྒྲ་ལའོ། །ཥུན་མངོན་པར་སྐྱེ་བ་ལའོ། །སྔ་མ་བཞིན་པ་དབྱི། འདི་རྣམས་ལས་པ་ཕ་རོལ་དང་ཀིཏ་བཞིན་ནོ། །ཨུ་ཡིག་གི་ཨཱུ་ཡིག་
