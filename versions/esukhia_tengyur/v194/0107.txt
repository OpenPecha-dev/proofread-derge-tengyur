ས་དྲྀག་བྷྱཿ། ས་དྲྀ་ཤཿ། ས་དྲྀ་ཤོཿ། ས་དྲྀ་ཤཱཾ། ས་དྲྀ་ཤི། ས་དྲྀ་ཤོཿ། ས་དྲྀཀྵུ། ཧེ་ས་དྲྀཀ། ཧེ་ས་དྲྀ་ཤཽ། ཧེ་ས་དྲྀ་ཤཿ། དིཝ་ཞེས་པ་འདིའི། དིཝའི་ཨཽཏའོ་ཞེས་པ་ཝ་ཡིག་གི་ཨཽ་ཉིད་དོ། །ཨིཀའི་ཡཎ་ཨ་ཙ་ལའོ། །དིཝའི་མཐའ་ལའང་ཨུཏའོ་ཞེས་པ་ཝ་ཡིག་གི་ཨུ་ཉིད་དོ། །
ཡཎ་རུ་བསྒྱུར་བའོ། །ཀུ་ལས་ཀྱང་ཞེས་པ་ཥ་ཉིད་དོ། །རྣམ་པར་གཅད་པའོ། དྱཽཿ། དི་ཝཽ། དི་ཝཿ། དི་ཝཾ། དི་ཝཽ། དི་ཝཿ། དི་ཝཱ། དྱུ་བྷྱཱཾ། དྱུ་བྷིཿ། དི་ཝེ། དྱུ་བྷྱཱཾ། དྱུ་བྷྱཿ། དི་ཝཿ། དྱུ་བྷྱཱཾ། དྱུ་བྷྱཿ། དི་ཝཿ། དི་ཝོཿ། དི་ཝཾ། དི་ཝི། དི་ཝོཿ། དྱུ་ཥུ། ཧེ་དྱཽ། ཧེ་དི་ཝཽ། ཧེ་དི་ཝཿ། མྲྀད་ཅེས་པ་
