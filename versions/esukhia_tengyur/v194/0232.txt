སྔ་མ་བཞིན་དུ་ཨཱ་ཉིད། ཨཱ་ཏིཿ། གྷཱ་སིཿ། རཱ་སིཿ། པཱཎིཿ། བེ་ཉ་ལས་ཌིའོ། བེ་ཉ་སྣ་ལ་མའི་རྒྱུད་ལའོ། ཉ་ཡིག་ནི་བེ་ཉའི་ལྀཊ་ལ་བ་ས་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན། ཤིཏ་མ་ཡིན་པ་ལ་ཨེ་ཙའི་ཨཱ་ཉིད་དོ། འདི་ལས་ཌི་ཕ་རོལ་དུ་འགྱུར་རོ། ཌ་ཡིག་ལས་ཨི་ཡིག་བཟུང་ནས་ཌ་ཡིག་
ནི་ཌེཏ་བྱ་བའི་དོན་ཏོ། དེས་འགྲོག མེ། ཁྱིམ། ལག་པ། ན་ཌིཏ་ལ་ཞེས་པས་མཐའི་ཨ་ཙ་ལ་སོགས་པ་དབྱིའོ། བཱིཿནི་ལས་ཨཱིཏ་འང་ངོ་། བེ་ཉ་ཞེས་པ་འཇུག་པས་ནི་སྔ་མ་ཅན་གྱི་བེཉལ་ཌི་འགྱུར་ཞིང་ཌ་ཡིག་དབྱི་བ་ལ་སོགས་པ་སྔར་བཞིན་ནོ། ཨིཏ་འང་ངོ་ཞེས་པས་ནིའི་ཨཱི་ཉིད་དོ། ནཱི་
བིས་ཁི་ལ་སོགས་པ་རྣམས་སོ། ཁྱ་རབ་ཏུ་གཏམ་པ་ལའོ། ས་མཱ་ན་སྔོན་དུ་ས་མཱ་ནའི་པ་ཀྵ་ལ་སོགས་རྣམས་བྱ། ལའོ་ཞེས་པས་སའི་དངོས་པོའོ། བྷ་ཛཿ། ཤྲི་བ་སྟེན་པ་ལའོ། ཨཱང་སྔོན་དུ། ང་ཡིག་ནི་ཨཱད་མཱངའིའོ་ཞེས་པའི་རྟགས་དོན། ཨོ་ཧཱ་ཀ་གཏོང་བ་ལའོ། ཨོ་ཡིག་
སྐ་རགས། ནི་ཨོ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན་ཏོ། ཀ་ཡིག་ནི་ཧཱ་ཀའི་ཏ་ལའོ་ཞེས་པའི་རྟགས་དོན་ཏོ། བྲ་སྔོན་དུ། བྷྲ་མུ་གཡོ་བ་ལའོ། ཌུ་ཀྲིཉ་བྱེད་པ་ལའོ། འདི་རྣམས་ལས་ཌི་ཕ་རོལ་ཏུ་འགྱུར་རོ། མཐའི་ཨ་ཙ་ལ་སོགས་དབྱིའོ། ཚིག་ཕྲད་ལས་ཨ་ཡིག་དབྱིའོ། །
ས་ཁི་ཞེས་གནས་པ་ལ་དང་པོའི་གཅིག་ཚིག་སུའོ། སུའི་ཨུ་ཡིག་སུའི་འོ་ཞེས་པའི་རྟགས་དོན། རེཏ་ཨུ་ཤ་ནས་པུ་རུ་དམ་ཤས་ཨ་ནེ་ཧས་རྣམས་ཀྱི་སུ་ལ་ཨ་ནང་ངོ་ཞེས་བས་ཨིའི་ཨ་ནང་ངོ་། ད་ཡིག་ངིཏ་བྱ་བའི་དོན། ནའི་ཨ་བརྗོད་དོན་ནོ། ས་མཱུ་དྡྷི་མ་ཡིན་པའི་སུ་ལའོ་ཞེས་པས་
རིང་པོའོ། སུའི་འོ་ཞེས་པས་སུ་དབྱིའོ། སུས་ཀྱི་རང་བཞིན་ཞེས་པས་ན་དབྱིའོ། ས་ཁཱ་མཐའི་ཨ་ཙ་སོགས་དབྱི་བས་ཨི་ཡིག་དང་ཨ་ཡིག་དབྱིའོ། ཨཱ་ངའི་ཐུང་ངུ་ཉིད་དོ། ཨ་ཤྲིཿ། པྲ་ཧིཿ། བྷྲ་མའི་མཐའི་ཨཾ་ཙ་ལ་སོགས་པ་ཚིག་ཕྲད་ལས་མི་དབྱི་བ་དད། རེ་པའིའང་རྀ་ཡིག་གོ།
བྷྲྀ་མིཿརླུང་། ཀ་རོ་ཏིའི་རྀ་ཡིག་གི་ཨ་ཉིད་དོ། ཀཱ་རིཿ། བཟོ་མོ། དེ་བཞིན་སྒྲ་གཞན་ཡང་ཤེས་པར་གྲོགས་པོའམ་མཛའ་བཤེས། བྱའོ། སཉྫི་ཨ་སི་དག་ལས་ཀྠིནའོ། །ཥཉྫ་འགྲོགས་པ་ལའོ། མཐའ། ཁྲོ་ནག། །ཥྠི་བུ་ཥཀྐ་མ་ཡིན་པའི་དང་པོའི་ཥའི་སའོ། ཨ་སུ་འཕེན་པ་ལའོ། །འདི་དག་ལས་ཥྠི་ན་ཕ་
