ལའོ།། ཥྛི་བུ་དང་། ཥྭཀྐ་མ་ཡིན་པའི་དང་པོའི་ཥའི་སའོ་ཞེས་པས་ཥ་ཡིག་གི་ས་ཡིག་གོ། །རྒྱུ་མཚན་མེད་ན་རྒྱུ་མཚན་ཅན་ཡང་མེད་དོ་ཞེས་པས་ཋ་ཡིག་གི་ཐ་ཡིག་གོ། །ཨ་པ་སུ་དུར་སྔོན་དུ། དྲཱ་སྨད་པ་ལའོ། །ཡཱ་ཐོབ་པ་ལའོ། །ཧ་རི། མི་ཏྲ། སྔོན་དུ་ཞེས་པ་
འདི་རྣམས་ལས་ཀུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཤཾ་ཀུཿ། ཀྲྀ་ཏིའི་དང་པོ་དང་མཐའ་བཟློག་པར་བྱ་སྟེ། དང་པོར་གྱུར་པ་ཡ་མཐའ་དང་། མཐར་གྱུར་པ་ཡང་དང་པོའོ། །དེ་ལ་ཨཀ་ནི་ཡང་བ་མཐའ་ཉེའི་ཞེས་པས་རྀ་ཡིག་གི་ཨ་ཉིད། རྀ་ཀའི་ཨཎ་ལས་ར་ལ་དག་གོ་ཞེས་པས་ར་ཕ་རོལ་ཉིད་དོ། །རེ་
པའི་ཨ་ཡིག་བརྗོད་དོན། ཏ་ཡིག་ཨ་ཡིག་ཕུལ་བུའམ་རྟགས། དང་སྦྱར་རོ། །ཏརྐུཿ། ཙྪ་རུཿ། དྷ་ནུཿ། ཨི་ཀའི་ཨ་ཏ་ཨེང་ཞེས་པས་ཨི་ཉིད་དོ། །ཨེ་ཙའི་ཨམ་ཨབ་ཨཱ་ཡ་ཨཱབ་རྣམས་སོ་ཞེས་པས་ཨ་ཡར་བསྒྱུར་རོ། །མ་ཡུཿ། ཏིཥྛ་ཏིའི་ནི་ཨ་ཙ་ལ་ཀིཏ་ལའང་ཨཱཏ་ཀྱི་དབྱིའོ་ཞེས་པས་
ཨ་ཡིག་དབྱིའོ། །སུ་ཥཱ་མ་ལས་སོགས་པ་རྣམས་སོ་ཞེས་པས་ཥ་གཅོད་པར་བྱེད་པའི་རྫས། མེ་ལོང་གི་ཡུང་དམ་དག་གཞུ། ཉིད་དོ། །ཨ་པ་སྔོན་དུ། པཱུའི་ཤྩུ་དང་ཥྚུ་དག་གི་དེ་དག་གོ་ཞེས་པས་ཐ་ཡིག་གི་ཋ་ཡིག་གོ། །ཨ་བཥྛུཿ། སུ་སྔོན་མིའམ་ཅི། དུ། སུཥྛུཿ། དུར་སྔོན་དུ། དུར་གྱི་རེ་ཕའི་ཁར་ལའོ་ཞེས་
པས་རྣམ་པར་བཅད་པའོ། །དུཿཥྛུཿདུརྦི་ནཱི་ཏཿསྔ་མ་བཞིན་དུ་ཨཱ་ཡིག་དབྱིའོ། །ཧ་རི་དྲུཿ། མི་ཏྲ་དྲུཿ། མི་ཏྲ་ཡུཿ། ཤ་ཏ་དྲུཿ། དེ་བ་ཡུཿ། ཀུ་མཱ་ར་ཡུཿ། མྲྀ་ག་སྔོན་དུ། མྲྀ་ག་ཡུཿ། ཛ་ཊཱ་ཡུཿ། པ་ཊཱ་ཡུཿ། དེ་བཞིན བྱིས་པ། དུ་གཞན་ཡང་ཤེས་པར་བྱའོ། ལེགས་པ། བྱའོ །པི་ཏ་ནི་ག་མི་མ་སི་ས་ཙ་ཨ་བི་
དྷཱ་ཉ། ཀྲུ་ཤི་རྣམས་ལས་ཏུནའོ། གདུལ་དཀའ་བ། །ཥི་འཆིང་བ་ལའོ། །ཥྛི་བུ་དང་ཤིང་གི་རིགས། རྒྱ་མཚོ། མཛའ་བཤེས་ཆུ་ཀླུང་། ཆོས་ལྡན། གཞོན་ནུ་ཐ་མ། ཥྚཀྐ་མ་ཡིན་པའི་རྔོན་པ། བྱའོ། གོས་སུ། དང་པོའི་ཥའི་སའོ་ཞེས་པས་ཥ་ཡིག་གི་ས་ཡིག་གོ། །ཏ་ནུ་རྒྱས་པ་ལའོ། །ཨུ་ཡིག་ནི་ཨུ་དིཏ་
ལས་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན། གམླྀ་སྲྀ་སླྀ་འགྲོ་བ་ལའོ། །ལྀ་ཡིག་ནི་ལྀ་དིཏ་དང་། དྱུ་ཏ་ལ་སོགས་པ་དང་པུཥྱ་ཏི་ལ་སོགས་པ་རྣམས་ལས་ཨ་ཏང་ལའོ་ཞེས་པའི་རྟགས་ཀྱི་དོན་ཏོ། །པཱ་སཱི་མ་སཱ་ཡོངས་སུ་འཇལ་བ་ལའོ། །ཨཱི་ཡིག་ནི་སྭི་ཨཱི་དིཏ་ལས་ཏ་ཏ་བཏ་དག་གིའོ་ཞེས་པའི་རྟགས་དོན་
