ཡོ྅བ་རོ྅རྡྷ་མཱ་སཿ། ཡོ་ཡཾ་ཏྲིཾ་ཤད་རཱ་ཏྲ་ཨཱ་གཱ་མི་ཏ་སྱ་ཡོ྅བ་ར་པཉྩ་ད་ཤ་རཱ་ཏསྟྲ་ཏྲ་ཡུཀྟཱ་ཨ་དྷྱེ་ཏཱསྨ་ཧེཿ། རྣམ་པ་གསུམ་ཀ་ཡང་ཉིན་མཚན་དང་མ་འབྲེལ་པའི་ཕྱིར་བཀག་གོ། །པ་ར་ཡང་ནའོ། །གཞན་ལ་ཞེས་པ་སྤངས་ནས་མདོ་གཉིས་རྗེས་སུ་འཇུག་གོ། །མཚམས་བརྗོད་
པ་ལ་འཇུག་པ་དུས་གཞན་རྣམ་པར་འབྱེད་པ་འབྱུང་བ་དེ་རིང་མ་ཡིན་པ་དག་ལ་བྱས་པ་དེ་མི་འགྱུར་ཏེ། གལ་ཏེ་ཉིན་མཚན་དང་འབྲེལ་པས་རབ་ཏུ་ཕྱེ་བ་མ་ཡིན་ནོ། །ཡོ་ཡཾ་སཾ་བཏྶ་ར་ཨཱ་གཱ་མི་ཏ་སྱ་ཡཏྤ་ར་མཱ་གྲ་ཧཱ་ཡ་ཎྱཱསྟ་ཡུཀྟཱ་ཨ་དྷྱེ་ཥྱཱ་མ་ཧེ། ཨ་དྷྱེ་ཏཱསྨ་ཧེ། ཏ་ཏྲ་སཀྟཱུན་པཱ་སྱཱ་མཿ། སཀྟཱུན་པཱ་
ཏཱསྨཿ། ཉིན་མཚན་མ་ཡིན་པ་ལ་ཞེས་པ་ཉིད་དེ། ཨིསྟྲཾ་ཤད་རཱ་ཏྲ་ཨཱ་གཱ་མི་ཏ་སྱ་ཡཿ། པཉྩ་ད་ཤ་རཱ་སྟ་ཏྲ་ཡུཀྟཱ་ཨ་དྷྱེ་ཏཱསྨ་ཧེ། ཏ་ཏྲ་སཀྟཱུན་པཱ་ཏཱསྨ། བྷཱ་བི་ནི་ཞེས་པ་ཉིད་དེ། ཡོ་ཡཾ་སམྦཏྶ་རོ྅ཏཱི་ཏས་སྱ་ཡ་ཏ་པ་ར་མཱ་གྲ་ཧཱ་ཡ་ཎྱས་ཏྲ་ཡུགྟཱ་ཨ་དྷྱེ་མ་ཧི། མཚམས་བརྗོད་པ་ལ་ཞེས་པ་ཉིད་དེ། ཡོ་
ཡཾ་ནི་ར་བ་དྷི་ཀཿ། ཀཱ་ལ་ཨཱ་གཱ་མི་ཏ་སྱ་ཡ་ཏ་པ་ར་མཱ་གྲ་ཧཱ་ཡ་ཎྱ་སྟ་ཏ་ཡུཀྟཱ་ཨ་དྷྱེ་ཏཱསྨ་ཧེ། དུས་ལ་ཞེས་པ་ཉིད་དེ། ཡོ་ཡ་མ་དྷྭཱ་གནྟ་བྱ་ཨཱ་པཱ་ཊ་ལི་པུ་ཏྲཱཏ་སྱ་ཡ་ཏ་པ་རཾ་ཀཽ་ཤཱམྦྷྱཱ་སྟཱསྨ་ཧེ། ལིང་ལ་བྱ་བཤོར་བ་ལ་ལྲྀངའོ། །ལིང་གི་ཡུལ་ལ་བྱ་བ་ཤོར་བ་ལ་ཐམས་ཅད་དུ་ལྲྀང་འགྱུར་ཏེ། སྨད་པ་ལ་ཀ་ཐཾ་ལ་ལིང་འོ་
ཞེས་འཆད་པར་འགྱུར་རོ། །དེ་ལ་བྱུང་བའི་དུས་ལ་བྱ་བ་ཤོར་བ་ལ་རྟག་ཏུ་ལྲྀང་ཉིད་འགྱུར་ཏེ། ཀ་ཐཾ་ནཱ་མ་ཏ་ཏྲ་བྷ་བཱན་བྲྀ་ཥ་ལ་མ་ཡཱ་ཛ་ཡི་ཥྱཏ། བྱུང་བ་ལ་རྣམ་པར་བརྟག་པ་འཆད་པར་བྱེད་དོ། །ཀིཾ། ཀ་ཏ་ར། ཀ་ར་མ་རྣམས་ལ་ལིང་ལིཊ་དག་ལ་ཞེ་ན། ཀོ་ནཱ་མ་བྲི་ཥ་ལོ་ཡཾ་བྷ་བཱ་ན་ཡཱ་ཛ་ཡི་
ཥྱཏ། འདོད་པ་བརྗོད་པ་ལ་ལིངའོ་ཞེས་བཤད་དེ། དེ་ལ། ཨུ་པཱ་དྷྱ་ཡཤྩེ་དག་མི་ཥྱཏ། ཨཱ་ཤཾ་སེ་ཡུགྟཱ་བྱཱ་ཀ་རཎ་མ་དྷྱཻཥྱཱ་མ་ཧི། དེ་བཞིན་དུ་གཞན་ལའང་ངོ་། །ཤེ་ཥའི་བར་ལས་བྱུང་བ་ལ་ཡང་ནའོ། །ཤེ་ཥའི་སྒྲའི་བར་ལས་བྱུང་བའི་དུས་ལ་ལུང་གི་ཡུལ་ལ་བྱ་བ་ཤོར་བ་ལ་རྟག་ཏུ་ལྲྀང་
ཡང་ན་འགྱུར་རོ་ཞེས་སྤྱིར་བཏང་བར་རིག་པར་བྱའོ། །ཀ་ཐཾ་ནཱ་མ་ཏ་ཏྲ་བྷ་བཱ་ན་བྲྀ་ཥ་ལ་མ་ཡཱ་ཛ་ཡི་ཥྱཏ། ཀ་ཐཾ་ནཱ་མ་ཏ་ཏྲ་བྷ་བཱན་བྲྀ་ཥ་ལཾ་ཡཱ་ཛ་ཡ་ཏེ། ཀོ་ནཱ་མ་བྲྀ་ཥ་ལོ྅ཡཾ་ཏཾ་ཏྲ་བྷ་བཱན་ཡཱ་ཛ་ཡི་ཥྱཏ། ཡང་ན་ཡཱ་ཛ་ཡེཏ། བྱེད་པ་གང་གིས་བྱ་བ་ལས་བྱུང་ཉིད་དོ། །ཡང་ན་བྱུང་བ་ཉིད་བྱ་བ་ཤོར་
