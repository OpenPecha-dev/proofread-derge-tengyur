རྒྱན་གྱི་དོན་དང་ཤྲི་ཤྲནྠི་ཎི། སན་རྣམས་ལ་ཙིཎ་འང་ངོ་། །འདི་རྣམས་རབ་ཏུ་སྦྱོར་བའི་བྱེད་པ་པོ་ལ་ཙིཎ་མི་འགྱུར་ཞིང་། འང་ཡིག་གིས་ཡཀ་ཀྱང་ངོ་། །བྲཱུ་ཏེ་དྷརྨཿ། སྭ་ཡ་མེ་བ། ཨ་བོ་ཙཏ། ཨ་བ་ཀི་ར་ཏེ་ཧསྟཱི་སྭ་ཡ་མེ་བ། ཨ་བཱ་ཀིརྵྰ་པྲ་སྒུ་ཏེ་གཽཿ། སྭ་ཡ་མེ་བ། པྲཱ་སྣོཥྚ། ན་མ་ཏེ་དཎྜ་སྭོ་ཡ་མེ་བ།
ཨ་ནཾ་སྟ་ཨ་ལཾ་ཀུ་རུ་ཏེ་ཀ་ནྱཱ་སྭ་ཡ་མེ་བ། ཨ་ལམ་ཀྲྀཏ། ཨུཙྪ་ཡ་ཏེ་དཎྜཿ། སྭ་ཡ་མེ་བ། ཨུ་ད་ཤི་ཤྲི་ཡཏ། ཤྲ་ཐྣཱི་ཏེ་མཱ་ལཱ་སྲ་ཡ་མེ་བ། ཨ་ཤྲནྠིཥྚ། ཀཱ་ར་ཡ་ཏེ་ཀ་ཊཿ། སྭ་ཡ་མེ་བ། ཨ་ཙ་ཀཱིརྵྚ། ལས་མ་ཡིན་པའི་ཏང་གི་མཐའ་ལས་ཅན་མ་ཡིན་པའི་ཕྱིར་དང་ལས་དང་བྱེད་པ་པོ་མེད་པའི་ཕྱིར་དཔེར་བརྗོད་
མི་རིགས་སོ། །བི་ཀུ་རུ་ཏེ། ཀྲོཥྚཱ་ཡ་མེ་བ། བྱ་ཀྲྀ་ཏ། པ་རི་བཱ་ར་ཡནྟེ་ཀཎྜེ་ཀཱཿ། སྭ་ཡ་མེ་བ་ཞེས་བྱེད་པ་དང་བྱེད་པ་པོ་བརྗོད་པར་འདོད་པ་ལ་བྱེད་པ་པོར་བསམས་པའི་འབྲས་བུ་ཏང་གིས་གྲུབ་བོ། །ཅི་ལྟར་ཨུ་དུམྦ་རསྶ་ལོ་ཧི་ཏཾ་ཕ་ལཾ་པ་ཙྱ་ཏེ་ཞེ་ན། ཨུ་དུམྦ་རའི་སྒྲ་ནི་འདིར་འབྲས་བུ་ལ་འཇུག་པ་ཉིད་ཀྱིས་རཽ་
དྲ་སོགས་ཀྱི་ལོ་ཧི་ཏ་ལ་ཇི་ལྟ་བར་འགྱུར་ཏེ། དེ་ལྟར་སྨིན་པར་བྱེད་པ་དང་སྨིན་པ་དག་མངོན་པར་འབྲེལ་པར་བསམས་སོ་ཞེས་པའི་དོན་ཏོ། །སྣུ་ནམ་ལ་སོགས་པའི་ནང་དུ་ཆུད་པ་ཎིའི་དོན་རྣམས་སོ། །འདི་ནི་བྱ་བ་ཤིན་ཏུ་བསྟན་པ་སྟེ། དེས་ན་ཡཀྟིཎླི་འཇུག་པའི་སྤྱིར་བཏང་གིས་གྲུབ་པས་<ཤཔ་,ཤབ་>ལ་སོགས་པ་ཉིད་
དོ། །བྱེད་པ་པོ་ལ་ཤབ་འོང་ངོ་། །བྱེད་པ་པོ་བརྗོད་པ་ལ་བྱིངས་ལས་ཤ་པ་འགྱུར་ཏེ། ཏ་ར་ཏི། ན་ཡ་ཏི། ཙི་ཀཱིརྵ་ཏི། དྷཱ་ར་ཡཿ། པྲཱ་ར་ཡ། ཀུན་འགྲོ་ལ་ཞེས་པ་ཉིད་དེ། པ་བྷཱུ་བ། བྷཱུ་ཡཱ་ཏ། ཤ་ཡིག་ནི་ཤིད་ཀྱི་བྱ་བའི་དོན་ཏོ། །ཤ་ཡིག་ནི་ཤཔ་ཤྱན་ལས་སོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཎི་ལས་སོ། །བྱིངས་
ལས་ཎིའི་མཐའ་ལས་བྱེད་པ་པོ་ལ་ཤབ་པ་འགྱུར་ཏེ། བི་ཤྲཱ། ཤྲཱ་ཡ་ཏི། བི་པྲྀ། པཱ་ར་ཡ་ཏི། བི་དཾ་བུ། དེ་བ་ཡ་ཏི། ཨེ་ལ་བི། ཨེ་ལ་ཡ་ཏི། བཱི་འགྲོ་བ་ལས། བཱ་ཡ་ཏི། རང་དོན་ལ་ཎི་ཙའོ། །བརྗོད་པ་འདི་ནི་འཇིག་པ་ལ་སོགས་པའི་ཚོགས་ཀྱི་བྱ་བ་དགག་པའི་དོན་ཏོ། །ཨད་སོགས་ལས་འཇིག་གོ། །ཨད་
སོགས་རྣམས་ལས་ཤཔ་འཇིག་པར་འགྱུར་ཏེ། ཨརྟི། ཧནྟི་ཡང་ལས་མང་པོ་ལའོ། །འཇིག་པར་འགྱུར་ཏེ། བོ་བྷ་བཱི་ཏི། བོ་བྷཱུ་ཡ་ཏེ། ཁ་ཅིག་ཏུ་རྟག་ཏུ་སྟེ། ལོ་ལུ་བ། པོ་པུ་བཿ། ཁ་ཅིག་ཏུ་མི་འགྱུར་ཏེ། ལོ་ལཱུ་ཡཱ། ཛུ་ཧོ་ཏི་སོགས་ལས་ཤྣུའོ། །ཛུ་ཧོ་ཏི་སོགས་རྣམས་ལས་ཕ་རོལ་ཤབ་ལས་ཤྣུ་འགྱུར་ཏེ།
