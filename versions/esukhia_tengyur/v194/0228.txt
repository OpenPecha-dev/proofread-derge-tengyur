རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཤ་དླྀ་རྣམ་པར་རྒྱུ་བ་ལའོ། །ཏུཀྵུ་རུ་ཀུ་སྒྲ་ལའོ། །ཁ་ནུ་ངེས་པར་འཇོམས་པ་ལའོ། །ཤཱིང་ཉལ་བ་ལའོ། །ཤྲུ་ཐོབ་པ་ལའོ། །འདི་དག་ལས་རུ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །དེ་ལ་ཛ་ནི་དང་ཤ་དིའི་མཐའ་ཏ་ཡིག་ཏུ་བསྒྱུར་བ་ཚིག་ཕྲད་དོ། །ཛ་ཏྲུཿ། །རུ་རུཿ། ཤ་ཏྲུཿ། ཁ་ནའི་ན་དབྱི། ཁ་རུཿ།
ཤེ་ཏིའི་ཐུང་ངུ་དང་གུཀ་ཨཱ་ག་མའོ། །ཤི་གྲུཿ། ཀརྐནྡྷུཿ། ཨ་ཌིའི་རིང་པོ་ཉིད། ཨཱ་ཌཱུཿ། ལམ་བིའི་ན་དབྱི་བ་དང་རིང་པོ་ཉིད་དོ། །ཨ་ལཱ་བཱུཿ། ཤ་ཀི་ལས་རུ་ཀའོ། །ཀ་ཡིག་གི་ཏ་བྱ་བའི་དོན། ཀ་ཥ་རཱུཿ། ཀཱ་སཱུཿ། པཱ་དིའི་རིང་པོ་ཉིད། པཱ་དཱུཿ། མགྲིན་པའི་ཕྱོགས་གཅིག། {𰵋རི་དགས,རི་དྭགས}། མི་མཛའ་བ། མེ་ལོང་གི་ཡུ་བ།
པྲྀ་ཛིའི་ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་མིག་སྨན་བཟང་པོ། རྒྱ་ཤུག་གི་འབྲུ། ཨ་ཉིད་དང་། ཆུ་སྒྲོལ་བ། ར་ཕ་རོལ་ཉིད། སརྫཱུཿ། ཁརྫུཿ། མྲྀ་ཛིའི་སྔ་མ་བཞིན་ཨ་ཉིད། མརྫུཿ། ནྲྀ་ཏི་དང་སྲྀ་གྲོམ། དགེ་མ་ཚང་ངག །དྷི་དང་གྲྀ་དྷི་རྣམས་རྐང་རྟེན། ཀྱི་ཨཏ་ཨེང་དུ་འགྱུར་བ་ཚིག་ཕྲད་དོ། །ནྲྀ་དྷཱུཿ། སྲྀ་དྷཱུཿ།
གྲྀ་དྷཱུཿ། ཀྲྀ་ཥའི་ཡང་བ་མཐའ་ཉེའི་ཞེས་པས་ཨཏ་ཨེངའོ། །ར་ཕ་རོལ་ཉིད། ཀཪྵཱུཿ། ཨནྡཱུཿ། གློག་གམ་ལེགས་པར་ཕན་བྱེད། ཤིང་འབྲ་གོ། ཀ་པི་ལས་ཨ་ལུཀ་ཨཱ་ག་མའོ། །ཀ་པོ་ལུཿ། དྲི་མ་དག་པའི་སྣོད། ཀ་མའི་བུཀ་ཨཱ་ག་མའོ། །ཨུ་ཡིག་བརྗོད་པའི་དོན། ཀ་ཡིག་ཀིཏ་བྱ་བའི་དོན་ཏོ། འབུ་རིང་པོ་གོ། གནོད་སྦྱིན་
འདོད་པ། ཀརྦཱུཿ། རི་ཏིའི་རྀ་ཡིག་རེ་ཕའོ། །ར་ཏུཿ། ཀ་ཕེའི་ཨེ་ཉིད། ལྕི་བསྐམ་པོ། རྒྱན་གྱི་རིགས། ཀེ་ཕ་ལཱུཿ། དེ་བཞིན་དུ་ཨུ་ཡིག་མཐའ་ཅན་གཞན་ཡང་མཆོག་གི་འཁུར་ཆོས། སྦྱོར་བ་ལས་འདི་བལྟ་བར་བྱའོ། །དི་པི་ལས་རྀ་ནའོ། །དི་བུ་བརྩེ་བ་ལའོ། །ཨུ་ཡིག་ནི་ཨུ་དིཏ་ལས་ཡང་ནའོ་ཞེས་
གཞན་གྱི་རྫས་འཕྲོག་པ། པའི་རྟགས་དོན། འདི་ཆུ་ཀླུང་། ལས་རན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་བད་ཀན་སེལ་བ། དབྱངས་དོན་ཡང་བ་མཐའ་ཉེའོ་ཞེས་པས་ཨེ་ཉིད། དང་པོ་གཅིག་ཚིག་སུའོ། །སུའི་ཨུ་ཡིག་སུའི་འོ་ཞེས་པའི་རྟགས་དོན། རིཏ་ཨུ་ཤ་ནས། པུ་རུ་དམ་ཤས་ཨ་ནེ་ཧས་རྣམས་
ཀྱིའང་སུ་ལ་ཨ་ནང་ངོ་ཞེས་པས་ཨ་ནའོ། །ང་ཡིག་ནི་ངི་ཏ་ལའོ་ཞེས་པས་མཐའ་བསྒྱུར་བའི་དོན་ཏོ། །ནའི་ཨ་བརྗོད་དོན། སཾ་བུདྡྷི་མ་ཡིན་པའི་སུ་ལའོ་ཞེས་པས་རིང་པོའོ། །སུའི་འོ་ཞེས་པས་སུ་དབྱིའོ། །སུས་ཀྱི་རང་བཞིན་གྱི་ཞེས་པས་ན་དབྱིའོ། །དེ་བཱ། ལྷ། དེ་བ་རཽ། དག་དེ་བ་རཿ། རྣམས།
