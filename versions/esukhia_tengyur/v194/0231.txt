མིཿ། མ་ནིའི་ཨུཏ་ཀྱང་ངོ་། །མ་ན་ཤེས་པ་ལའོ། །འདི་ལས་ཀི་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་། ཨུཏ་ཀྱང་ངོ་ཞེས་པས་མའི་ཨུ་ཉིད་དོ། །མུ་ནིཿ། ཨཾ་ཧིཿ། ཀལྤི་དག་གི་ན་དབྱི་བའང་ངོ་། །ཨ་ཧི་འགྲོ་བ་ལའོ། །ཀ་པི་གཡོ་བ་ལའོ། །ཨི་དིཏ་ཀྱི་ནུམའོ། །ཨུ་ཡིག་བརྗོད་དོན། མ་ཡིག་
ལིཏ་ཨ་ཙ་མཐའ་སྐྱེ་བ་ཕྲ་མོའམ་འབུ། ལས་ཕ་རོལ་ལོ་ཞེས་པའི་རྟགས་དོན་ནོ་ཞེས་པ་འདི་དག་ལ་ཀཾ་ཕ་རོལ་ཏུ་འགྱུར་ཞིང་། ན་དབྱེ་བའང་ངོ་། །ཨ་ཧིཿ། ཐུབ་པའི་དྲང་སྲོང་། ཀ་བིཿ། ཤྀ་པ་སི་པ་པི་རཱ་ཛི་བྲྀ་ཧ་ནི་བྷ་ནི་རྣམས་ལས་ཨིཉའོ། །ཤྲཱི་འཚེ་བ་ལའོ། །བ་ས་ངེས་པར་གནས་པ་ལའོ། །ཊུ་བ་
ས་བོན་བསགས་པ་ལའོ། །ཊུ་ཡིག་ནི་ཊུ་འགྲོ་བ་ལས་ཨ་ཐུ་ཙའོ་ཞེས་པའི་རྟགས་དོན། རྀ་ཛྲ་འབར་བ་ལའོ། །རྀ་ཡིག་ནི་ཨཀ་ཕྱིས་པ་དང་། སྲུལ། ཤཱ་སུ་རྀ་དིཏ་རྣམས་སྤྲེའུ། ཀྱི་མ་ཡིན་ནོ་ཞེས་པའི་རྟགས་དོན། བྲྀཉ་སྒྲིབ་པ་ལའོ། །ཉ་ཡིག་ནི་ཉ་ཏི་ལའོ་ཞེས་པའི་རྟགས་དོན། ཧ་ན་འཚེ་བ་ལའོ། །ཎ་བྷ་
ཏུ་བྷ་འཚེ་བ་ལའོ། །ཎའི་ནའོ་ཞེས་པ་འདི་རྣམས་ལས་ཨིཉ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཉ་ཡིག་ཉྞིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། ཨའང་ངོ་ཞེས་པས་ཨཱ་ཉིད་དོ། །ར་ཕ་རོལ་ཉིད། ཤཱ་རིཿ། །ཉྞིཏ་ལ་མཐའ་ཉེའི་འོ་ཞེས་པས་རིང་བོ་ཉིད་དོ། །བཱ་སི། དཱིཀྱཱབྷཱཎྜཿ། བཱ་སི། རཱ་ཛིཿ། ཨའང་ངོ་ཞེས་པས་ཨཱ་ཉིད་
དང་ར་ཕ་རོལ་ཉིད་དོ། སུ་ཨམ་དག་གི་མ་ནིང་ལས་སོ་ཞེས་སུ་དབྱིའོ། བཱ་རི། ཉྞཏ་ན་ལ་ཧ་ནའི་ཧོ་ཞེས་པས་གྷ་ཉིད་དོ། ཙི་ཎ་ཎལ་དག་མ་ཡིན་པ་ལ་ཧ་ནའི་ཏའོ་ཞེས་པས། བྱ་རི་སྐེག །ན་ཡིག་ཏ་ཡིག་གོ། །བྷྲ་ཏིཿ། ཉྞིཏ་ལའོ་ཞེས་པས་ཨཱ་ཉིད། རྫིང་བུ། གྲལ་ལ་མཚར་གཏིང་དུ་གནས་པ། ནཱ་བྷྲིཿ།
ཨ་ཛི། ཛ་ནི། ཨ་ཏི་གྷ་སི ར་སི། བ་ཎི་རྣམས་ལས་ཨིཎའོ། ཨཛ་བྲཛ་ཆུ། འགྲོ་བ་ལའོ། ཛ་ནཱི་རབ་ཏུ་སྐྱེ་བ་ལའོ། ཨ་ཏ་མགྱོགས་པར་འགྲོ་བ་ལའོ། གྷ་སླི་ཟ་བ་ལའོ། ལྀ་ཡིག་ནི་ལྀདིཏ་དྻུཏ་སོགས་ཞེས་པའི། མཚན་ཆ། རྟགས་དོན། ར་ས་ཞེས་པ་མདོའི་བྱིངས་སོ། པཎ་ཐ་སྙད་ལའོ། ལྟ་བ།
འདི་རྣམས་ཨིཎ་ཕ་རོལ་དུ་འགྱུར་རོ། ཎ་ཡིག་ནི་ཉྞིཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། ཡུ་གྷཉ་ཨས་རྣམས་མིན་པ་ལ་ཨ་ཛིའི་བཱིའོ་ཞེས་པས་བཱིའི་དངོས་པོར་མི་འགྱུར་རོ། མང་པོའི་ཚིག་ལས་ཉྞིཏ་ལའོ་ཞེས་པས་ཨཱ་ཉིད་དོ། ཨཱ་ཛིཿ། འཐབ་པ། ཛ་ནི་པདྷ་དག་གིའོ་ཞེས་པས་ཨ་ཉིད་བཀག་གོ། ཛ་ནིཿམའོ།
