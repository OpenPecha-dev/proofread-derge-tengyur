རྣམས་དོན་བསྒྲུབ་པའི་ཕྱིར། །ཡིད་བཞིན་གྱི་ནི་ལུས་སུ་བསྒྱུར། །དེ་ལྟར་རང་དབང་ཡོད་ཀྱིས་དེ། །རྟག་ཏུ་འཛུམ་པའི་བཞིན་དུ་གྱིས། །ཁྲོ་གཉེར་ངོ་ཟུམ་ཡོངས་ཐོང་སྟེ། །འགྲོ་བའི་བཤེས་དང་གསོང་པོར་གྱིས། །ཁྲི་ལ་སོགས་པ་བབ་ཅོལ་དུ། །སྒྲ་དང་བཅས་
པར་མི་དོར་རོ། །སྒོ་ཡང་དྲག་ཏུ་མི་དབྱེ་སྟེ། །རྟག་ཏུ་གཅོམ་བསྐྱུངས་དགའ་བར་བྱ། །ཆུ་སྐྱར་བྱི་ལ་ཆོམ་རྐུན་དག །སྒྲ་མེད་འཇབ་ཅིང་འགྲོ་བ་ཡིས། །མངོན་པར་འདོད་པའི་དོན་སྒྲུབ་པ། །ཐུབ་པས་རྟག་ཏུ་དེ་བཞིན་སྤྱད། །གཞན་ལ་བཞེན་བསྐུལ་འདེབས་
མཁས་ཤིང་། །མ་བཅོལ་ཕན་པ་བྱེད་པའི་ངག །གུས་པས་སྤྱི་བོས་བླང་གྱིས་ཏེ། །རྟག་ཏུ་ཀུན་གྱི་སློབ་མར་གྱུར། །ལེགས་པར་སྨས་པ་ཐམས་ཅད་ལ། །དགེ་བར་གསུངས་ཞེས་བརྗོད་པར་བྱ། །བསོད་ནམས་བྱེད་པ་མཐོང་གྱུར་ན། །བསྟོད་པས་ལེགས་པར་དགའ་
བ་བསྐྱེད། །ལྐོག་ན་ཡོན་ཏན་བརྗོད་བྱ་ཞིང་། །ཡོན་ཏན་བརྗོད་ན་རྗེས་སུ་བརྗོད། །རང་གི་ཡོན་ཏན་བརྗོད་ན་དེ། །ཡོན་ཏན་ཤེས་པར་རིག་པར་བྱ། །རྩོམ་པ་ཐམས་ཅད་དགའ་ཕྱིར་ཏེ། །དེ་ནི་རིན་གྱིས་ཉོ་ནའང་དཀོན། །དེ་ལྟས་གཞན་བྱས་ཡོན་ཏན་གྱིས། །དགའ་
བའི་བདེ་བ་སྤྱད་པར་བྱ། །བདག་ལ་འདིར་གོང་ཡོང་མེད་ལ། །ཕ་རོལ་དུ་ཡང་བདེ་བ་ཆེ། །ཉེས་པས་མི་དགའ་སྡུག་བསྔལ་ཞིང་། །ཕ་རོལ་དུ་ཡང་སྡུག་བསྔལ་ཆེ། །སྨྲ་ན་ཡིད་ཕེབས་འབྲེལ་པ་དང་། །དོན་གསལ་ཡིད་དུ་འོང་བ་དང་། །ཆགས་དང་ཞེ་སྡང་སྤངས་
བ་དང་། །འཇམ་ཞིང་རན་པར་སྨྲ་བར་བྱ། །མིག་གིས་སེམས་ཅན་ལྟ་ན་ཡང་། །འདི་དག་ཉིད་ལ་བརྟེན་ནས་བདག །སངས་རྒྱས་ཉིད་དུ་འགྱུར་རོ་ཞེས། །དྲང་ཞིང་བྱམས་པའི་ཚུལ་གྱིས་བལྟ། །རྟག་པར་མངོན་ཞེན་གྱིས་བསླང་བའམ། །གཉེན་པོ་ཡིས་ནི་བསླང་
