རཱ་ཛ། ཨོཾ་བཛྲ་རཱ་ཛ་ཛ། བཛྲ་རཱ་ག །ཨོཾ་བཛྲ་རཱ་ག་ཧོ། བཛྲ་སཱ་དྷུ། ཨོཾ་བཛྲ་སཱ་དྷུ་ས། བཛྲ་རཏྣ། ཨོཾ་བཛྲ་རཏྣ་ཨོཾ། བཛྲ་ཏེ་ཛ། ཨོཾ་བཛྲ་ཏེ་ཛ་ཨ། བཛྲ་ཀེ་ཏུ་ཨོཾ་བཛྲ་ཀེ་ཏུ་ཏྲཾ། བཛྲ་ཧཱ་སྱ། །ཨོཾ་བཛྲ་ཧཱ་སྱ་ཧ། བཛྲ་དྷརྨ། ཨོཾ་བཛྲ་དྷརྨ་ཧྲིཿ། བཛྲ་ཏཱིཀྵྞ། ཨོཾ་བཛྲ་ཏཱིཀྵྞ་དྷཾ། བཛྲ་
ཧེ་ཏུ། ཨོཾ་བཛྲ་ཧེ་ཏུ་མཾ། བཛྲ་བྷཱ་ཥ། ཨོཾ་བཛྲ་བྷཱ་ཥ་རཾ། བཛྲ་ཀརྨ། ཨོཾ་བཛྲ་ཀརྨ་ཀཾ། བཛྲ་རཀྵ། ཨོཾ་བཛྲ་རཀྵ་ཧཾ། བཛྲ་ཡཀྵ། ཨོཾ་བཛྲ་ཡཀྵ་ཧཱུཾ། བཛྲ་སནྡྷི། ཨོཾ་བཛྲ་སནྡྷི་བཾ། བཛྲ་ལཱ་སྱེ། ཨོཾ་བཛྲ་ལཱ་སྱེ་ཧཱུཾ། བཛྲ་མཱ་ལེ། ཨོཾ་བཛྲ་མཱ་ལེ་ཏྲཱཾ། བཛྲ་གཱི་ཏི། ཨོཾ་བཛྲ་གཱི་ཏི་
ཧྲཱིཿ། བཛྲ་ནྲྀ་ཏི། ཨོཾ་བཛྲ་ནཱིརྟི་ཨ། བཛྲ་དྷཱུ་པེ། ཨོཾ་བཛྲ་དྷཱུ་པེ་ཧཱུཾ། བཛྲ་པུཥྤེ། ཨོཾ་བཛྲ་པུཥྤེ་ཏྲཱཾ། བཛྲ་ཨཱ་ལོ་ཀེ། ཨོཾ་བཛྲ་ཨཱ་ལོ་ཀེ་ཧྲཱིཿ། བཛྲ་གནྡྷེ། ཨོཾ་བཛྲ་གནྡྷེ་ཨཱ། བཛྲ་ཨཾ་ཀུ་ཤ ཨོཾ་བཛྲ་ཨཾ་ཀུ་ཤ་ཛཿ། བཛྲ་པཱ་ཤ ཨོཾ་བཛྲ་པཱ་ཤ་ཧཱུཾ། བཛྲ་སྥོ་ཊ། ཨོཾ་བཛྲ་སྥོ་ཊ་བཾ།
བཛྲ་ཨཱ་བེ་ཤ ཨོཾ་བཛྲ་ཨཱ་བེ་ཤ་ཨ། དེ་ནས་རང་གི་ལུས་དང་རྡོ་རྗེ་སེམས་དཔའ་ལ་སོགས་པའི་སྐུ་ལ། རྡོ་རྗེ་བསྡུ་བའི་ཕྱག་རྒྱ་བཅིངས་ཏེ། ཛཿཧཱུྃ་བཾ་ཧོཿ། ཞེས་བརྗོད་དེ། ཡིད་ཀྱིས་ལེགས་པར་སྦྱོར་བ་ཡིས། །སངས་རྒྱས་ཐམས་ཅད་གཞུག་པར་བྱ། །དེ་ནས་སྦྱོར་བ་འདིས་
དེ་རྣམས་རང་གི་དམ་ཚིག་དང་། ཆོས་དང་། ལས་དང་། ཕྱག་རྒྱ་ཆེན་པོས་རྒྱས་གདབ་པར་བྱའོ། །རྡོ་རྗེ་དབྱིངས་ཀྱི་དམ་ཚིག་གི་ཕྱག་རྒྱ་བཅིངས་ལ། བཛྲ་དྷཱ་ཏུ་དྲྀ་ཤྱ། ཛཿཧཱུཾ་བཾ་ཧོཿ། ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ་ཞེས་བརྗོད་དོ། །དེ་ནས་བཛྲ་དྷཱ་ཏུ་ཞེས་ལན་གསུམ་བརྗོད་ལ། ཨོཾ་སརྦ་ཏ་ཐཱ་
ག་ཏ་མ་ཧཱ་ཡོ་གཱི་ཤྭ་ར་ཧཱུཾ་ཞེས་བྱ་བ་བཅོམ་ལྡན་འདས་རྣམ་པར་སྣང་མཛད་ཀྱི་སྔགས་ལན་གསུམ་དུ་བརྗོད་པར་བྱའོ། །དེ་བཞིན་དུ་མི་བསྐྱོད་པ་དང་། རིན་ཆེན་འབྱུང་ལྡན་དང་། འོད་དཔག་ཏུ་མེད་པ་དང་། དོན་ཡོད་གྲུབ་པའི་ཕྱག་རྒྱ་བཅིངས་ལ། འདི་སྐད་བརྗོད་དེ། བཛྲ་དྷཱ་ཏུ་དྲྀ་
ཤྱ་ཛཿཧཱུཾ་བཾ་ཧོཿ། །ས་མ་ཡ་སྟྭཾ། ས་མ་ཡ་སྟྭཾ། ཨ་ཧཾ་ཞེས་བྱ་བ་དང་། བཛྲ་དྷཱ་ཏུ་ཞེས་བྱ་བའི་སྤྱིའི་སྙིང་པོ་ལན་གསུམ་གསུམ་བརྗོད་པར་བྱ་ཞིང་། ཨོཾ་བཛྲ་སཏྭ་ཧཱུཾ་ཞེས་བྱ་བ་མི་བསྐྱོད་པའི་སྙིང་པོ་དང་། ཨོཾ་བཛྲ་རཏྣ་ཏྲཱཾ་ཞེས་བྱ་བ་རིན་ཆེན་འབྱུང་ལྡན་གྱི་སྙིང་པོ་དང་། ཨོཾ་བཛྲ་
