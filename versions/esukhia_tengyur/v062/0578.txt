རྡོ་རྗེའི་གར་སྔོན་དུ་འགྲོ་བས་དྲིལ་བུ་དཀྲོལ་ཞིང་རྡོ་རྗེ་ཡང་གསོར་བར་བྱའོ། །དབང་ལྡན་གྱི་ཕྱོགས་ནས་བརྩམས་ཏེ་དང་པོ་དང་རྩེ་མོ་ཅན་དང་ལྷོ་དང་མཚན་མོ་སྤྱོད་པ་དང་ཀླུ་དང་རླུང་དང་གནོད་སྦྱིན་དང་འོག་དང་སྟེང་གི་ཕྱོགས་རྣམས་སུ་གོ་རིམས་ཇི་ལྟ་བ་བཞིན་དུ་སོ་སོ་རང་རང་གི་ཚུལ་བཞི་བཅུ་རྩ་གཉིས་
བྱའོ། །དེ་ལ་གཡས་བརྐྱང་ལ་ཚུལ་བཅུ་གཅིག་སྟེ། སེང་གེའི་རྐང་སྟབས་དང་། སེང་གེའི་གཡས་དང་། རྡོ་རྗེ་ཐོ་བ་དང་། བུད་མེད་སྒྱུ་མ་བྱེད་པ་དང་། རྡོ་རྗེ་རྒྱས་པ་དང་། རྡོ་རྗེ་མཆེ་བ་དང་། རལ་གྲི་དང་། ལྕགས་ཀྱུ་དང་། མདུང་དང་། འུར་རྡོ་དང་། གཞུའི་ཚུལ་རྣམས་སོ། །གཡོན་
བརྐྱང་ལ་ཚུལ་བཅུ་སྟེ། རྡོ་རྗེ་འཕེན་པ་དང་། རྡོ་རྗེ་འཇིགས་བྱེད་དང་། སྟོབས་པོ་ཆེ་འགུགས་པ་དང་། རྡོ་རྗེ་བཞད་པ་དང་། རྡོ་རྗེ་ཚུལ་དང་། རྡོ་རྗེ་ཞགས་པ་དང་། དབྱུག་པ་དང་། མདུང་དང་མཚོན་རྩེ་གཅིག་པ་དང་། མདའ་འཕེན་པའི་ཚུལ་རྣམས་སོ། །ས་གའི་རྐང་སྟབས་ལ་ཚུལ་དྲུག་སྟེ། རྡོ་རྗེ་
འགུགས་པ་དང་། རྡོ་རྗེ་སྒེག་པ་དང་། མི་འཇིགས་པ་དང་། འབྲས་བུ་དང་། མཆོད་སྦྱིན་དང་། ཕྱག་འཚལ་བའི་ཚུལ་རྣམས་སོ། །ཟླུམ་པོའི་རྐང་སྟབས་ལ་ཚུལ་བདུན་ཏེ། རྡོ་རྗེ་གར་དང་། ཞལ་བཞི་བྱིན་གྱིས་བརླབ་པ་དང་། རྡོ་རྗེ་འཁྱིལ་བ་དང་། རྡོ་རྗེ་དྲིལ་བུའི་སྒྲ་དང་། འཁོར་ལོ་དང་། པདྨ་
དང་། མདའ་འདྲེན་པའི་ཚུལ་རྣམས་སོ། །མཉམ་པའི་རྐང་སྟབས་ལ་ཚུལ་ལྔ་སྟེ། རྡོ་རྗེ་རྒྱས་པ་དང་། རྡོ་རྗེ་སྟོབས་བསྐྱེད་པ་དང་། རྡོ་རྗེ་རོ་ལངས་དང་། གཏུན་ཤིང་དང་། བ་དན་རྣམས་སོ། །རུས་སྦལ་གྱི་རྐང་སྟབས་ཀྱིས་རྡོ་རྗེ་སེམས་དཔའི་མཆུའི་ཚུལ་ལོ། །རྐང་གཅིག་གི་སྟབས་ལ་
ཚུལ་གཉིས་ཏེ། སྙེམས་པ་གཉིས་དང་བུང་བའི་དྲ་བའོ། །དེ་ལ་རེ་རེ་ཞིང་སོ་སོར་རྣམ་པར་དབྱེ་བར་བྱའོ། །རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་བཅིངས་ནས། མཐེའུ་གཉིས་ལྕགས་སྒྲོག་གི་རྣམ་པར་ཡང་དག་པར་སྦྱར་ཏེ། མཛུབ་མོ་གཉིས་རྩེ་མོར་བྱས་ནས་ཕྱིར་བཟློག་སྟེ་གདོང་གི་མདུན་དུ་གཟུང་བར་
བྱའོ། །ལག་པ་དང་རྐང་པ་དག་རུས་སྦལ་གྱི་ཚུལ་དུ་བྱའོ་ཞེས་པ་ནི་རྡོ་རྗེ་སེམས་དཔའི་མཆུའི་ཚུལ་ལོ། །རྡོ་རྗེ་སེམས་དཔའི་ཁྲོ་བོའི་ང་རྒྱལ་བྱའོ། །སྔགས་ནི། ཨོཾ་བཛྲ་ཏུཎྜ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཀྲ་མ་ཡ་ཀྲ་མ་ཡ་སརྦ་བི་གྷུརྞ་ཧཱུཾ་ཕཊ། བརྟན་པ་དང་གཡོ་བའི་རང་བཞིན་གྱི་དངོས་
