མན་ངག་སྟོབས་ཀྱི་སྦྱོར་བ་ཡིས། །རྣ་བའི་དྲུང་དུ་སྔགས་འདི་བཟླས། །རྡོ་རྗེ་འཛིན་པའི་རྡོ་རྗེ་བསམ། །ཨོཾ་ན་མོ་བྷ་ག་བ་ཏེ། སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནེ་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ་ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་བྷྱ་མ་བྷྱ་པི་ནི། བྱོ་མ་རཱུ་པ་ཡ། སརྦ་བྱ་པི་ནི་ཤི་པ་ཡ་ཧྲི། བྲརྨ་མཱ་ཏ་
ཧྲིམ། བཥྤྷྱི་དང་ཡ་ཏེ། ཧཱུཾ་རུ་དྲ་དང་ཡི་ཏེ། ཀྵུ་རུ་དྲཀྵ་ར་པི་སརྦ་ནི་ཧཱུཾ་རུ་དྲཀྵ་ར་པི་སརྦ་ནི་ཧཱུཾ་རུ་དྲེ་ས་རུ་པེ། ཧྲི་ན་ད། ཨཱཏྨ་ཀེ །ཧཱུཾ་བིནྡུ་ཤ་རེ་རེ། ཨོ་ཤག་ཏྲི་སཾ་ཤག་ཏྲི་སམ་ཤ་ཏྲི་ས་ཡོ་ཛ་ནི། མ་བྱོ་མ་ཡོ་ཛ་ནི། བྷོ་ཥ་ཏ་ཀཻ་བ་ལྱཻ་བ་དེ་ཧྨ་བིདྷྱེ་ན་མ། ཨཱུཾ་ཧཱུཾ་ཧཱུཾ་ལྟུཾ་ཀྵུཾ་ཀྵུཾ་ཀྵུཾ་ན། ཨོཾ་ཨོཾ་ཨཱི་ཨུ། བྱོ་མ་

ཆོ་ག་དང་། །སློབ་དཔོན་ལས་དང་དཀྱིལ་འཁོར་བསྒྲུབ། །དེ་ནས་རོ་ནི་བསྲེག་པར་བྱ། །ཡང་ན་སྟ་གོན་བྱས་རྟིང་ལ། །བསྲེག་པར་བྱ་བ་ཉིད་དུ་བྱ། །ས་ནི་བརྟག་དང་བསླང་བ་དང་། །སྦྱང་དང་བྱིན་གྱིས་བརླབས་པ་དང་། །གཟུང་དང་བསྲུང་བ་ཉིད་དུ་བྱ། །དེ་རྣམས་ཞིབ་ཏུ་བྱ་བ་
