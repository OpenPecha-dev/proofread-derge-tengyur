དེ་སྐད་དུ་མ་བསྟན་ཏེ། སྡེ་པ་ལ་ནི་ཀུན་གཞིའི་རྣམ་པར་ཤེས་པ་དང་མཚུངས་པར་ལྡན་པའི་རེག་པ་ལ་སོགས་པ་མ་གྲགས་སོ། །བསྒོ་བར་བྱ་བའི་མཚན་ཉིད་འདི་ནི་གཞན་ལ་གྲགས་པའི་བསྒོ་བར་བྱ་བ་དགག་པས་རྣམ་པར་གཞག་པ་ཡིན་ཏེ། དགག་པ་རྒྱས་
པར་ནི་འཆད་པར་འགྱུར་རོ། །དེ་ནི་བག་ཆགས་ཀྱི་མཚན་ཉིད་དོ་ཞེས་བྱ་བ་ནི་བག་ཆགས་ནི་བསྒོ་བར་བྱ་བའི་ཆོས་ཡིན་པས་བསྒོ་བར་བྱ་བའི་མཚན་ཉིད་གང་ཡིན་པ་དེ་ཡང་བག་ཆགས་ཀྱི་ཡིན་པའི་ཕྱིར་རོ། །མདོ་སྡེ་པ་གང་དག་རྣམ་པར་ཤེས་པའི་ཚོགས་ལྔ་དང་དབང་པོ་རྟེན་དང་
བཅས་པའི་གཟུགས་བསྒོ་བར་བྱ་བ་ཡིན་པར་སེམས་པ་དེ་དག་གི་དགག་པ་འཆད་པར་འགྱུར་རོ། །གང་དག་རྣམ་པར་ཤེས་པའི་ཚོགས་དྲུག་ཁོ་ན་བསྒོ་བར་བྱ་བ་ཡིན་གྱི་གཟུགས་ནི་མ་ཡིན་ནོ་ཞེས་བྱ་བར་སེམས་པ་དེ་དག་གི་དགག་པ་ནི་ལུང་དུ་མ་བསྟན་པ་མ་ཡིན་པའི་ཕྱིར་ཞེས་
བྱ་བར་བཤད་ཟིན་མོད་ཀྱི། དེའི་ཞར་ལ་དྲུག་པོ་དག་ལ་ཞེས་བྱ་བ་ལ་སོགས་པས་དགག་པ་གཞན་བརྩམ་མོ། །དེ་ལ་ཁ་ཅིག་ན་རེ་འཇུག་པའི་རྣམ་པར་ཤེས་པ་དྲུག་པོ་དག་ནི་ཕན་ཚུན་བག་ཆགས་འཇོག་པ་མ་ཡིན་ཏེ། གཞན་ཁས་ལེན་པ་ཉིད་ཀྱིས་ལྷན་ཅིག་སྐྱེ་བ་དང་འགག་པའི་
མཚན་ཉིད་ཀྱིས་འབྲེལ་པ་མེད་པའི་ཕྱིར་རོ། །ཡང་གཞན་གྱི་ཇི་ལྟར་དེ་ལྟར་ཁས་བླངས་ཤེ་ན། ཐ་དད་གསུམ་དང་འགལ་བའི་ཕྱིར། །འདི་ལྟར་གསུམ་པོ་རྟེན་དང་དམིགས་པ་དང་ཡིད་ལ་བྱེད་པ་དག་ནི་རྣམ་པར་ཤེས་པའི་ཚོགས་དྲུག་ལ་སོ་སོ་ཐ་དད་པར་མཐོང་སྟེ། གལ་ཏེ་ཅིག་ཅར་སྐྱེ་བར་
གྱུར་ན་དེ་དང་འགལ་བར་འགྱུར་རོ་ཞེས་ཟེར་རོ། །དེ་ནི་རིགས་པ་མ་ཡིན་ཏེ། དབང་པོ་ལ་སོགས་པ་ཐ་དད་པ་ཉིད་མ་ཡིན་ཡང་ཅིག་ཅར་སྐྱེ་བར་མ་མཐོང་བའི་ཕྱིར་དང་། གང་དང་ཟླ་བ་ལ་ལྟ་བ་ལ་སོགས་པ་ལ་སྲོག་ཆགས་རྣམས་ཀྱི་དབང་པོ་ལ་སོགས་པ་ཐ་དད་ཀྱང་ཅིག་ཅར་སྐྱེ་བ་
མཐོང་བའི་ཕྱིར་རོ། །དེ་ལྟ་བས་ན་གཞན་དུ་རྣམ་པར་བྱ་སྟེ། འཇུག་པའི་རྣམ་པར་ཤེས་པ་དྲུག་པོ་དག་ནི་འབྲེལ་པ་མེད་པའི་ཕྱིར་ཕན་ཚུན་བསྒོ་བར་བྱ་བ་མ་ཡིན་ནོ། །རེ་ཞིག་གཟུགས་ཅན་མ་ཡིན་པའི་ཕྱིར་ཕྲད་པའི་འབྲེལ་པ་ནི་མེད་དོ། །དུས་མཉམ་དུ་སྐྱེ་བ་ཙམ་གྱིས་འབྲེལ་
