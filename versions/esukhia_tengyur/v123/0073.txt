རྟོགས་པའི་རྒྱུ། །དེས་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བྱའོ། །མ་རྟོགས་རྟོགས་དང་རྗེས་སུ་རྟོགས་པ་རྟོགས། །དངོས་མེད་རྟོགས་དང་རབ་ཏུ་སྐྱེ་བ་རྟོགས། །མ་རྟོགས་རྟོགས་པ་སོ་སོར་རྟོགས་པའི་རྒྱུ། །དེས་ན་བྱང་ཆུབ་སེམས་དཔའ་
ཞེས་བྱའོ། །དོན་མེད་རྟོགས་དང་དོན་དམ་རྟོགས་པ་དང་། །དོན་ཀུན་རྟོགས་དང་མཐའ་དག་དོན་རྟོགས་དང་། །རྟོགས་བྱ་རྟོགས་གཞི་རྟོགས་པ་རྟོགས་པའི་རྒྱུ། །དེས་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བྱའོ། །གྲུབ་པར་གྱུར་པ་རྟོགས་དང་གནས་རྟོགས་དང་། །
ལྷུམས་རྗེས་རྟོགས་དང་རིམ་པ་རྟོགས་པ་དང་། །རབ་ཏུ་རྟོགས་དང་ཐེ་ཚོམ་བསལ་རྟོགས་རྒྱུར། །དེས་ན་བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བྱའོ། །བློ་དང་ལྡན་པ་བློ་ཐོབ་མ་ཐོབ་དང་། །རབ་གནས་རྟོགས་དང་རྗེས་སུ་རྟོགས་པ་དང་། །སོ་སོར་འཆད་བློ་རྗོད་མེད་ང་བཅོམ་

དག་བློ་ལྡན་རྣམས་ཀྱི་རྟགས། །ཡོངས་གཟུང་མོས་པ་ཐོབ་པ་དང་། །མི་སྐྱོ་གཉིས་ཀྱིས་བསྡུ་བའི་ཕྱིར། །བསམ་པ་དང་ནི་སྦྱོར་བ་ལས། །རྟགས་ལྔར་རྣམ་པར་ཤེས་པར་བྱ། །བྱང་ཆུབ་སེམས་དཔའ་རྟག་ཏུ་ནི། །ཁྱིམ་པ་འཁོར་ལོས་སྒྱུར་རྣམས་སུ། །

ཚད་མེད་རྣམས་དང་ལྡན། །དེ་ལྟས་སྡོམ་བརྩོན་ཁྱིམ་པ་ཡི། །བྱང་ཆུབ་སེམས་དཔའ་ལས་ཁྱད་འཕགས། །ས་རྣམས་ཀུན་ལ་བརྟན་པ་ཡི། །ཕ་རོལ་སྡུག་པའི་འབྲས་བུ་དང་། །འདི་ལ་དགེ་བ་འཇུག་འདོད་དང་། །མྱ་ངན་འདས་པར་འདོད་པ་དང་། །
