པ་དག་སྟོན་པའི་བརྗོད་པ་གང་ཡིན་པའོ། །རྣམ་པར་བྱང་བ་དང་མཐུན་པ་ནི་བརྗོད་པ་ཙམ་དུ་ཟད་དེ་རང་གི་མཚན་ཉིད་མེད་ན་ཇི་ལྟར་ན་ལེགས་པར་བཤད་པའམ། ཉེས་པར་བཤད་པར་འགྱུར། གཞན་ཡང་ཀུན་རྫོབ་ཀྱང་ཁ་ཅིག་གདོན་མི་ཟ་བར་དངོས་སུ་ཁས་བླང་བར་བྱ་བར་འགྱུར་རོ། །དེ་ཁས་བླངས་
ན་ནི་ཐམས་ཅད་མེད་པ་ཡང་མ་ཡིན་ནོ། །ཡང་ན་དེ་ཡང་མེད་ན་ནི་ཇི་ལྟར་ཀུན་རྫོབ་ཏུ་ཡོད་པ་ཉིད་དུ་གསུངས་པ་ཡིན། གཞན་ཡང་དེ་སྐད་དུ་སྨྲ་བ་དེ་དག་ཀུན་ནས་ཉོན་མོངས་པ་དང། རྣམ་པར་བྱང་བ་ཐམས་ཅད་ལ་སྐུར་པ་བཏབ་པས་གནས་དང་གནས་མ་ཡིན་པ་གཉིས་ཏེ་སྲིད་པ་དང་མི་སྲིད་པ་གཉིས་ལ་མི་
གནས་པའི་ཕྱིར་ཏེ། འདི་ལྟར་དེ་ཉིད་ལ་མི་གནས་པས་དེའི་ཕྱིར་བརྗོད་དུ་མེད་པར་འགྱུར་རོ། །སྨྲར་མེད་པར་འགྱུར་རོ། །རང་གིས་རང་སྨྲ་བ་བཀག་པའི་ཕྱིར་འདི་ལྟར་རང་གིས་རང་སྨྲ་བ་སྟེ་མེད་པར་སྨྲ་བ་འགོག་པར་བྱེད་དོ། །ཇི་ལྟ་ཞེ་ན། གང་ངེད་ཀྱིས་མེད་དོ་ཞེས་སྨྲ་བ་འདི་ཡང་མེད་དོ་ཞེས་ཟེར་རོ། །
དེའི་ཕྱིར་ཡང་བརྗོད་དུ་མེད་པར་འགྱུར་རོ། །མིང་གིས་ཀྱང་ནི་མིང་ཉིད་སྟོང་། །མིང་ལ་མིང་ནི་ཡོད་མ་ཡིན། །ཆོས་རྣ་མས་ཐམས་ཅད་མིང་མེད་ཅིང། །མིང་གིས་ཡོངས་སུ་བརྗོད་པ་ཡིན། །བྱིས་པ་རྣམས་ཀྱིས་ཀུན་དུ་བརྟགས་པ་བཞིན་དུ་བརྗོད་པས་སེམས་དང། རྣམ་པར་ཤེས་པ་དང། དམིགས་པ་དང་
ཚོར་བ་མེད་པ་ཞེས་བྱ་བ་དེ་ལྟ་བུ་ལ་སོགས་པ་རྗོད་པར་བྱེད་པ་ཡིན་ནོ། །དེ་ནི་སྟོན་པས་བཏགས་པའི་ཚིག་ཅེས་བྱ་བ་ནི་སྟོན་པས་གསུངས་པའི་ཚིག་ཡིན་ནོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེ་ནི་ཀུན་དུ་རྟོག་པ་ལས་བྱུང་བ་ཐ་སྙད་དུ་བརྗོད་པ་ཡིན་ལ་ཞེས་བྱ་བ་ནི་ཀུན་དུ་རྟོག་པ་ལས་སྐྱེས་པ་ཐ་སྙད་དུ་བརྗོད་པ་
ཡིན་ནོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ཀུན་དུ་རྟོག་པ་སྣ་ཚོགས་ཞེས་བྱ་བ་ལས་ཀུན་དུ་རྟོག་པ་སྣ་ཚོགས་ཀྱི་ཐ་སྙད་དུ་བརྗོད་པ་ནི་ཀུན་དུ་རྟོག་པ་སྣ་ཚོགས་ཀྱི་ཐ་སྙད་དུ་བརྗོད་པའོ། །དེ་གཏན་ནི་གཏན་དུ་ཞེས་བྱ་བའི་ཐ་ཚིག་སྟེ། ཡོངས་སུ་མ་གྲུབ་པ་ནི་ཀུན་དུ་རྟོག་པ་སྣ་ཚོགས་ཀྱི་ཐ་སྙད་དུ་བརྗོད་པ་གཏན་
ཡོངས་སུ་མ་གྲུབ་པའོ། །ཀུན་དུ་རྟོག་པ་སྣ་ཚོགས་ཀྱི་ཐ་སྙད་དུ་བརྗོད་པ་གཏན་<ཡངས་,ཡོངས་>སུ་མ་གྲུབ་པ་དེས་ཡོངས་སུ་མ་གྲུབ་པའི་ཕྱིར་རོ། །འདུས་བྱས་ཇི་ལྟ་བ་བཞིན་དུ་འདུས་མ་བྱས་ཀྱང་དེ་བཞིན་ནོ་ཞེས་དེ་བཞིན་དུ་སྦྱར་རོ། །ཇི་ལྟ་ཞེ་ན། རིགས་ཀྱི་བུ་འདུས་མ་བྱས་<ཞས་,ཞེས་>བྱ་བ་དེ་ནི་སྟོན་པས་བཏགས་
