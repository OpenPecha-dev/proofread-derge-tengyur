ཡང་ཉམས་པར་བྱེད་པ་ཡིན་ནོ། །ཤེས་རབ་མྱུར་བ། ཤེས་རབ་མགྱོགས་པ། ཤེས་རབ་རྣོ་བ། ངེས་པར་འབྱུང་བའི་ཤེས་རབ་ཅན། ངེས་པར་འབྱེད་པའི་ཤེས་རབ་ཅན། ཤེས་རབ་ཆེ་བ། ཤེས་རབ་ཡངས་པ། ཤེས་རབ་ཟབ་པ། ཤེས་རབ་མཉམ་པ་མེད་པ།
ཤེས་རབ་རིན་པོ་ཆེ་དང་ལྡན་པ་ཞེས་བྱ་བ་ནི་མདོ་སྡེའི་དུམ་བུ་ཡིན་ནོ། །ཇི་ལྟར་ན་<གོ་རིམས་,གོ་རིམ་>བཞིན་ཞེ་ན། དོན་ལ་མྱུར་བར་འཇུག་པའི་ཕྱིར་ཤེས་རབ་མྱུར་བ་ཡིན་ནོ། །དོན་མང་པོ་ལ་འཇུག་པའི་ཕྱིར་ཤེས་རབ་མགྱོགས་པ་ཡིན་ནོ། །དོན་ཕྲ་བ་ལ་འཇུག་པའི་ཕྱིར་ཤེས་རབ་རྣོ་བ་ཡིན་ནོ། །འཁོར་བ་
ལས་ངེས་པར་འབྱུང་བ་དང་ཞེས་བྱ་བ་རྒྱས་པར་འབྱུང་བ་ལ། འཁོར་བ་ལས་ངེས་པར་འབྱུང་བ་ནི་ངེས་པར་འབྱུང་བ་སྟེ་མྱ་ངན་ལས་འདས་པའོ། །འདི་ལ་ངེས་པར་འབྱུང་བ་དེའི་ཤེས་རབ་ཡོད་པས་དེའི་ཕྱིར་ངེས་པར་འབྱུང་བའི་ཤེས་རབ་ཅན་ནོ། །དེ་ཁོ་ན་ངེས་པར་འབྱེད་པ་ཞེས་བྱ་བ་ལ། དེ་ཁོ་ན་
འཕགས་པའི་བདེན་པ་བཞིའི་མཚན་ཉིད་ངེས་པར་འབྱེད་པ་ཡང་དག་པ་ཇི་ལྟ་བ་བཞིན་རབ་ཏུ་ཤེས་པར་བྱེད་པས་ན་ངེས་པར་འབྱེད་པའི་ཤེས་རབ་བོ། །ངེས་པར་འབྱེད་པའི་ཤེས་རབ་ཡོད་པས་ན་དེའི་ཕྱིར་ངེས་པར་འབྱེད་པའི་ཤེས་རབ་ཅན་ནོ། །ཇི་ལྟར་བྱས་པས་<གོ་རིམས་,གོ་རིམ་>བཞིན་ཞེ་ན། འདི་ལ་ཤེས་རབ་
ཆེན་པོ་ཡོད་དེ། འདྲི་བ་དང་ལན་གདབ་པ་དག་ལ་ཕུགས་ཐུག་པར་མི་འགྱུར་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་ནི་དྲི་བ་རྣམས་དང་ལན་འདེབས་པ་རྣམས་ལ་ཡོངས་སུ་གཏུགས་པར་མི་འགྱུར་བའི་ཕྱིར་རོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ཤེས་རབ་ཆེ་བ་ཡིན་ནོ། །ཤེས་རབ་ཆེ་བ་ནི་ཡངས་པ་ཡང་ཡིན་ཏེ། གང་གི་ཕྱིར་
ཞེ་ན། མངོན་པར་ཤེས་པ་ལྔ་རྒྱ་ཆེ་བ་དང་ཡོངས་སུ་རྫོགས་པའི་ཕྱིར་རོ། །མངོན་པར་ཤེས་པ་རྒྱ་ཆེ་བའི་ཕྱིར་དང་མངོན་པར་ཤེས་པ་ལྔ་ཆར་ཡོངས་སུ་རྫོགས་པའི་ཕྱིར་ཤེས་རབ་ཡངས་པ་ཡིན་ནོ། །རྒྱ་ཆེ་བ་ནི་དེ་དག་གི་ཡུལ་ཐག་རིང་བ་དང་། ཕྲ་བ་ཉིད་ཡིན་ནོ། །ཡོངས་སུ་རྫོགས་པ་ཉིད་ནི་གཅིག་
མ་ཡིན་ལ་བཞིའི་བར་མ་ཡིན་<གྱིའོ་,གྱི་འོ་>ན་ཅི་ཞེ་ན་ལྔ་ཆར་ཡིན་ནོ། །ཤེས་རབ་དེ་ནི་ཟབ་པ་དང་ཡངས་པ་ཡིན་ཏེ། གཞན་དག་གིས་རྟོགས་པར་དཀའ་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་ནི་ཤེས་རབ་ཟབ་པ་ཡིན་ནོ། །ཤེས་རབ་དེ་ནི་མཉམ་པ་མེད་པ་ཡང་ཡིན་ཏེ། འཇིག་རྟེན་ཐམས་ཅད་ལས་ཁྱད་པར་དུ་འཕགས་
