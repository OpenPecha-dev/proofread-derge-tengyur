ཨི་ཉིད་དོ། །བི་ཡིག་ལས་ཨི་ཡིག་བཟུང་ནས་བ་ཡིག་དབྱི་བ་ཚིག་ཕྲད་དུ་བྱའོ། །ཀི་ཀི་ བྱ། པཀྵེཿ དི་བིའི་བ་ཡིག་དབྱིའོ། །དི་བི་ ཡཱ་དིདྱ་ ཉི་མ། བྱིངས་དེ་ཉིད་ཀྱི་གཉིས་ཉིད་དོ། དི་བ་དི་བའི་ཧལ་དང་པོ་མ་ཡིན་པའི་ཞེས་པས་བ་ཡིག་དབྱིའོ། །རིང་པོ་
ཉིད་དང་བ་ཡིག་དབྱི་བའང་ཕ་རོལ་གྱི་སྒྲུབ་པའོ། །དཱི་དིབིཿ སརྦཿ ཀྲྀ་བིཿ དྷུ་པཿ པྲ་ཥིཿ སརྒཿ ཛ་ལཿ ཛཱི་བིའི་བ་ཡིག་དབྱིའོ། །ཛཱི་བིཿ སྨཥ། ཨོཥ་དྷིཿ ནཱི་བིཿ བྲཱཔྟིཿ བྲི་བསི་དག་ལས་ཀྟིནའོ། །དྲྀང་མ་གུས་པ་ལའོ། །པས་ཀུན་ཏུ་སྒྲིབ་པ་ལའོ། །འདི་
དག་ལས་ཀྟིན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ཨཏ་ཨེང་འགེགས་པའི་དོན་ནོ། །ན་ཡིག་དབྱངས་དོན། དྲྀ་ཏིཿ པགས་པ། ཙ་རྨཿ བསྟི་ མཱུ་ དྲཱ་ ཤཱ་ཡཾ་ བཱ་ཏི་ལས་ཌ་ཏིའོ། །པཱ་བསྲུང་པ་ལའོ། །པཱ་ཏི་ལས་ཌ་ཏི་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཌ་ཡིག་ལས་ཨ་ཡིག་ཕྱུང་
