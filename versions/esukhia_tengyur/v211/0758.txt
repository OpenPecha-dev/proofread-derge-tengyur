བློ་དན་ཡུལ་དུ་མ་གྱུར་ལས། །ནམ་མཁའ་འདི་ཞེས་དཀའ་དང་འདྲ། །ལོག་རྟོག་ཕྱོགས་རིགས་རྣལ་འབྱོར་དང་། །ཕྱོགས་གསལ་མི་གནས་རྟེན་པ་ཅན། །ཕྱི་པའང་འཛིན་པའི་གཞི་ཅན། །གནས་པའི་གཙོ་བོར་བདག་གིས་བརྗོད། །བླང་མེད་དོར་མེད་མི་གནས་
པ། །བསྒོམ་བྱ་སེམས་ཕྱིར་རྟེན་བྲལ་བ། །གནས་དང་མི་གནས་སྤོང་ལེན་མེད། །ཅི་ལའང་ཡིད་རྟེན་མི་འཆའ་བས། །མི་གནས་ཉིད་ལ་གནས་མི་བྱེད། །རྟེན་དང་རྟེན་ཅན་མེད་པ་ནི། །མི་གནས་ཆེན་པོ་མཆོག་ཡིན་ཏེ། །ནམ་མཁའ་ལྟ་བུ་རྟེན་དང་བྲལ། །གང་
གི་མི་དམིགས་ཤེས་བརྗོད་པའང་། །ལོག་དང་འཇིག་རྟེན་ཏིང་འཛིན་དང་། །འགོག་གཉིས་ཕྱོགས་གཅིག་ལ་གནས་དང་། །དབུ་མ་བ་དག་མི་དམིགས་གསུམ། །འདོད་ལས་གསེལ་བའི་དམིགས་པ་ཅན། །ཕྱོགས་གཅིག་རྟེན་མེད་དམིགས་པའོ། །དམིགས་
པ་ཀུན་བདག་མི་དམིགས་ལ། །མཚོན་ཆེན་ཅན་དུ་དེ་བརྗོད་ཀྱང་། །མི་དམིགས་འདོད་པའི་ཆགས་པ་མེད། །དམིགས་སྤོ་ཞེ་སྡང་<ཕྲང་,ཕྲག་>དོག་མེད། །གཉིས་སུ་མེད་པར་མི་དམིགས་ལ། །མཁའ་ལ་འདི་ཞེས་དམིགས་བྲལ་འདྲའོ། །རྣམ་པར་མི་རྟོག་བརྗོད་པའང་། །
འཇིག་རྟེན་ཞེན་པའི་རྣམ་མི་རྟོག །ལོག་པར་སྒོམ་པ་མི་རྟོག་དང་། །བྱིས་པ་དགག་སྒྲུབ་མི་རྟོག་གཉིས། །ཕྱོགས་འཛིན་གསལ་བ་མི་རྟོག་པ། །རྣམ་དག་གསལ་བ་མི་རྟོག་རྣམས། །བཞི་ལ་རྟོག་པ་མེད་སྨྲ་ཞིང་། །མི་རྟོག་མི་མཐུན་བགེགས་ལྟ་བས། །དགག་
སྒྲུབ་བླང་དོར་བྱེད་པ་དེ། །ངེས་མཆོག་བཞི་ཡི་རྟོག་པ་ཅན། །ཐབས་ཆེན་རྣམ་པར་མི་རྟོག་པ། །རྟོགས་ཚོགས་མིང་ཙམ་ཡེ་ཤེས་ཕྱིར། །རྟོག་པ་ཉིད་ནས་མ་སྐྱེས་པས། །སྐྱེས་བརྡའི་འདུ་ཤེས་དབྱེ་བ་མེད། །དགག་དང་སྒྲུབ་མེད་མི་རྟོག་པ། །མི་རྟོག་ཉིད་
དུའང་རྟོག་མེད་པས། །མེ་ལོང་ནང་ལ་རྟོག་མེད་བཞིན། །གྲོལ་རྟོགས་བྱ་ཕྱིར་མཐའ་བྲལ་བ། །མཐའ་ཞེས་ཐུབ་པར་གྲགས་པ་རྣམས། །ལོག་པའི་མཐའ་གཉིས་བྲལ་གཟུགས་མེད། །སྐྱེ་འགག་ཆ་ཤས་བྲལ་བ་དང་། །རང་རིག་མཐའ་བརྒྱད་བཞི་གཉིས་བྲལ།
