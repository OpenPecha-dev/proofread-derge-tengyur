བྱེ་ཀར་དང་མཉམ་དུ་བཏང་ན་མཁྲིས་པ་ལ་ཕན། སྦྲང་རྩི་དང་མཉམ་དུ་ཟོས་ན་ལུས་སྟོབས་བསྐྱེད་པར་འགྱུར་རོ། །པ་ཐུ་བའི་ལོ་མ་བརྡུངས་པ་དང་མཉམ་དུ་བཏང་ན་མཁྲིས་པ་ཤ་སེར་དང་རྡེལ་ནད་ལ་ཕན། མ་ནུ་ཁྲག་ཅན་ཆུང་ངུ་དང་མཉམ་དུ་བཏང་ན་གསེབ་
དྲི་སོགས་ལུས་ཀྱི་རྔུལ་ཆེན་པོའི་དྲི་དང་ཆུ་རྟགས་ཀྱི་དྲི་ངན་པ་ལ་ཕན། སླེ་ཏྲེས་དང་མཉམ་དུ་བཏང་ན་གྲང་བ་འཛག་པ་ཚ་གྲང་མཐའ་དག་ལ་ཕན། ཉེ་ཤིང་པ་དང་མཉམ་དུ་བཏང་ན་རླུང་གིས་ན་ཟུག་བྱེད་པ་དང་བ་བྱི་བ་སོགས་ལ་ཕན། ལ་ལ་ཕུད་ཤུ་དག་གཉིས་ཕྱེ་
མར་བཏགས་པ་དང་མཉམ་དུ་བཏང་ན་རྐེད་པ་ན་བ་ལ་ཕན། བག་ནི་དྲིལ་ནས་ཟ ཆུས་རྟ་བྱས་ནས་ལོ་གསུམ་ཟླ་བ་གསུམ་ཞག་གསུམ་ཉི་མའི་སུམ་ཆ་གསུམ་གྱི་བར་<དྲིལ་,རིལ་>བུ་རེ་བར་མ་ཆད་པར་བསྟེན་ན་ལུས་ཀྱི་ནད་གསར་རྙིང་ཐམས་ཅད་འབྱང་ཞིང་ལུས་སྟོབས་འཕེལ་བ་དང་རྒས་ཁ་སྲབ་
སོགས་ཕན་ཡོན་བསམ་ལས་འདས་སོ། །ནད་པར་གཏོང་སྐབས་ཟས་སྤྱོད་སོགས་ནད་བབས་དང་བསྟུན་ཏེ་སྐབས་དང་སྦྱར་རོ་། །སྐྲན་གྱི་སྨན་ལ་དནྡའི་རྩ་བ་ཞོ་བཞི། ལྒ་དང་ཁ་རུ་ཚ་ཞོ་ཕྱེ་དོ་རེ། ཤིང་ཀུན་སེ་བ་ལྔ་རྣམས་ཆུ་ཕོར་པ་གསུམ་ལ་བཏབ་ལ་ཚ་གཉིས་
གདུས་པའི་ཕོར་གང་ཁོང་དུ་བཏང་ན་རླུང་སྐྲན་འཕྲལ་དུ་ཞིག་པར་འགྱུར་རོ། །ཡང་འགྲོན་བུ་མེར་བསྲེགས་པའི་ཐལ་བ ཕོ་རིས པི་པི་ལིང་། ལྒ། ཙི་ཏྲ། སོ་ཙལ་ལོ་ཎ ཁ་རུ་ཚྭ་རྣམས་ཞིབ་པར་བཏགས་པ་དང་ལ་བཙགས་ལ་ཐུར་མགོ་དོ་
ཙམ་ཆུ་སྐོལ་གྱིས་རྟ་བྱས་བཏང་ན་སྐྲན་གྱི་རིགས་ཐམས་ཅད་ལ་ཕན་ནོ། །ཤིང་ཀུན་ཆ་ཕྱེད། ཟི་ར་དཀར་པོ། ལྒ། ཁ་རུ་ཚ། འུ་སུ་རྣམས་ཆ་རེ་བཅས་ཐང་ཐུན་ཙམ་ཆུ་ཕོར་པ་གསུམ་གྱི་ནང་དུ་བཏབ། ཆ་གཉིས་གདུས་ལ་བཏང་ན་མཁྲིས་སྐྲན་ཞིག་པར་
འགྱུར་རོ། །ཀཱ་ཀ་ར་སིངྒི། དྷ་བ་རའི་མེ་ཏོག །ལྒ། བུ་རམ་རྣམས་ཆ་མཉམ་ཞིབ་པར་བཏགས་པའི་རིལ་བུ་བ་རུ་ར་ཙམ་རེ་བཏང་ན་ ངྷནི་ཉ ཕོ་སྐྲན་ཞིག་པར་འགྱུར་རོ། །རྒྱ་ཚ། སོཙྔྷོ། །བ་རུ་ཚེ། ཙི་ཏྲ། ལྒ། པི་པི་ལིང་གི་རྩ་བ་རྣམས་ཆ་མཉམ་
