ཕྱེ་མར་བྱས་ལ། རྩབས་སྐྱུར་པོ་དང་སྦྱར་བས་ལུས་ལ་བསྐུ་བར་བྱའོ། །ཡང་ན་སྨན་གྱི་སྡེ་ཚན་དེ་གཉིས་ལས་གང་ཡང་རུང་བ་སྐོལ་བའི་ཁུ་བ་བསྒྲངས་བའི་ནང་དུ། ཚ་བ་མི་བཟོད་པའི་མི་ཉིད་འཇུ་བར་བྱའོ། །ཡང་ན་བུ་རམ་ཆང་ངམ། རྒུན་ཆང་གི་
ཚའམ། ཆང་ཐམ་ལེ་པའམ། རྩབས་སྐྱུར་བའམ། འོ་མའི་རྒུན་ལུས་ལ་བྱུག་ན་ཡང་ཚ་བ་སེལ་ཏེ་བཟང་ངོ་། །ཤུག་པ་དང་། མེ་ཏོག་ཨུཏྤལ་དང་། ཀ་ལྟ་ར་དང་། པདྨའི་རྩ་ལག་དང་། ལེ་བརྒན་རྩི་དང་། པུ་ཤེལ་ཙེ་དང་། བཙོད་དང་། པདྨ་དང་།
བཙག་དང་། ཀ་ཏལ་དང་། ཐལ་ཏྲེས་རྣམ་གཉིས་དང་། སེང་ཕྲོམ་དང་། གླ་སྒང་དང་། རྒུན་དང་། འབྲ་གོ་དང་། ཙན་དན་དཀར་པོ་དང་། སྐྱུ་རུ་ར་དང་ཉེའུ་ཤིང་རྣམས་ལྷན་ཅིག་ཏུ་སྐོལ་བའི་ཁུ་བའི་ནང་དུ། སྨན་དེ་དག་ཉིད་ཀྱི་ཕྱེ་མ་དང་། རྒྱ་སྐྱེགས་ཀྱི་ཁུ་
བ་དང་། འོ་མ་དང་། ཚ་དང་། རྩབས་ཀྱི་དངས་མ་དང་། ཞོ་ཁ་ཆུ་དང་ཏིལ་མར་རྣམས་བླུགས་ཏེ་བསྐོལ་ལ་ཏིལ་མར་ཉི་ཚེ་ལུས་པར་བྱས་ལ་བཙགས་པ་འདི་ནི་པགས་པའི་མདོག་བཟང་ཞིང་། ལུས་ཚ་བ་དང་། སྐོམ་པའི་ནད་སེལ་pའི་མཆོག་ཡིན་ཏེ་ལུས་ལ་བསྐུ་བར་
བྱའོ། །ཀ་ལིང་ཡ་ཀ་དང་། རྒྱ་ཤུ་ཀ་དང་། ཨ་དནྟ་དང་། ཤིང་མངར་དང་། བཙག་རྣམས་བཏགས་ལ་ཞུན་མར་དང་སྦྱར་ཏེ་མགོར་བསྐུས་ན་ཡང་སྐོམ་པའི་ནད་དང་། ལུས་ཚ་བའི་གནོད་པ་སེལ་ཏོ། །བལ་པོ་སེའུ་དང་། རྒྱ་ཤུག་དང་། སེང་ཕྲོམ་དང་། ད་དྷི་ཐ་
དང། ཁ་ལུང་རྣམས་བཏགས་ལ་ཞུན་མར་དང་སྦྱར། འདིས་ཀྱང་སྐོམ་པའི་ནད་དང་ལུས་ཚ་བ་སེལ་ཏོ། །མགོ་བསྐུ་བར་བྱའོ། །པདྨའི་ལོ་མ་དང་། ཆུ་ཤིང་གི་ལོ་མ་དང་ཉི་སྔོ་བའི་སྟེང་དུ་ཉལ་ལ། ཤིང་ཏཱ་ལའི་ལོ་མ་ཙན་དན་བརྡར་བའི་ཁུ་བའི་ནང་དུ་དམྱུགས་པའི་
བསིལ་གཡོབ་ཀྱིས་གཡབས་ན་ཡང་ལུས་ཚ་བའི་གནོད་པ་སེལ་ཏོ། །རྫིང་བུའི་མེ་ཏོག་པདྨ་ཁ་བྱེ་བ་དང་ལྡན་པའི་ཕྱོགས་སམ། ཆུ་འདྲེན་བའི་འཁྲུལ་འཁོར་ཡོང་བའི་ཕྱོགས་སུ་བསིལ་ཁང་བཟང་པོ་བྱས་པའི་ནང་དུ་འདུག་ཅིང་། བུད་མེད་ཀྱི་ལུས་ཙན་དན་གྱི་འདག་
