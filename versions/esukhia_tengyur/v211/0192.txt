ཛ་ལ་སོགས་པ་དང་ཨཏ་ལས་སོ་ཞེས་པས་ཌཱཔའོ། །ར་སནཱ་ སྐ་རགས། མེ་ཁ་ལས་ ཡང་བ་མཐའ་ཉེའོ་ཞེས་པས་ཨིང་ངོ་། །རོ་ཙ་ནཱ་གོ་བིཏྟཾ་བྱ་བའི་དོན་གྱི་ཨེཀ་གི་ཨཏ་ཨེངའོ་ཞེས་པས་ཨ་ཉིད་དོ། །རྀཀ་གིཨཎ་ལས་རལ་དག་གོ་ཞེས་པས་
ར་ཕ་རོལ་ཉིད་དོ། །རཥ་ལས་ཞེས་པས་ན་ཉིད་དོ། །བ་རཎཿ ན་དཱིཿ ཆུ་ཀླུང་། ཨཏ་ཨེང་ཞེས་པས་ཨོ་ཨེཙའི་ཞེས་བས་ཨ་ཝ་རུ་བསྒྱུར་རོ། །ར་ཥ་ལས་ཞེས་པས་ཎ་ཉིད་དོ། །ར་ཝ་ཎཿ ཉི་མ། ཨཱ་དི་ཏྱ། ཨུནྡིའི་ན་དབྱི་བ་ཡང་ངོ་། །ཨུནྡཱི་ཀླས་པ་
ལའོ། །འདི་ལས་ཨུཙ་ཕ་རོལ་ཏུ་འགྱུར་རོ། ན་དབྱི་བ་ཡང་ངོ་། །ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨོ་ཉིད་དོ། །ཨོ་ད་ནཿ ཨནྣོོ: རཉྫི་ལས་ཀྱུ་ནའོ། །རཉྫ་ཚོས་ལའོ། །འདི་ལས་ཀྱུན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ཀྲིན་བྱ་བའི་དོན་ནོ། །ན་ཡིག་
དབྱངས་དོན་ནོ། །ས་མེད་པའི་ཡུ་ཝུའི་ཨ་ན་ཨཀ་དག་གོ་ཞེས་པས་ཨུ་སྒྲའི་ཨ་ན་རུ་བསྒྱུར་རོ། །རཛ་ན: རདྒིཿ གཽ་ར་ལ་སོགས་བ་རྣམས་ལས་སོ་ཞེས་བས་དི་ཤའོ། །ར་ཛ་ནཱི: རདྷཱིཿ ཀྲ་པྲི: བྲི་ཛི་མཎྜིནི་དྷཉ་རྣམས་ལས་སོ། །ཀྲཱི་རྣམ་པར་གཡོ་བ་ལའོ། །
བྲི་རྫོགས་བ་ལའོ། །པྲི་ཛྭ་འདོར་བ་ལའོ། །མ་ཌི་རྒྱན་ལའོ། །ཨི་ཡིག ཨི་དིཏ་ཀྱི་ནུ་མའོ་ཞེས་བས་རྟགས་དོན་ནོ། །ནུ་མའོ། །ཌུ་དྷཱཉ་འཛིན་པ་ལའོ། །ནི་སྔོན་དུའོ། །འདི་རྣམས་ཀྱུན་འགྱུར་རོ། །ཀ་ཡིག་ཨཏ་ཨེང་འགེགས་པའི་དོན་ནོ། །ས་མེད་
པའི་ཡུ་བུའི་ཨ་ན་ཨ་ཀ་དག་གོ་ཞེས་བས་ཨ་ན་རུ་བསྒྱུར་རོ། །བྱིངས་ཀྱི་ཨིཏའོ་ཞེས་པས་ཨི་ཉིད་དོ། །རིག་གི་ཨཎ་ལས་རལ་དག་གོ་ཞེས་བས་ར་ཕ་རོལ་ཉིད་དོ། །ཎ་ཉིད་ཀྱང་ངོ་། །ཀི་ར་ནཿ པྲཱ་བྷྲཿ མཆུ་བྱུང་ལས་ཨུཏའོ་ཞེས་པས་ཨུ་ཉིད་དོ། །ར་ཕ་
རོལ་ཉིད་དོ། །བྱིངས་ཀྱི་ར་བ་དག་ལ་ཨཙ་མེད་པ་ལ་ཨིཀ་གི་རིང་པོའོ་ཞེས་བས་རིཏཱ་པ་ཉིད་དོ། །བཱུ མ་ཧཿ བྲི་ཛན། ཨནྟ་རྭི་ཀྵཾ མཎྜ་ནཾ: རྒྱན། བྷཱུ་ཥ་ན: ནདྷཱ་ན: གཏེར། ནིདྷྲིཿ དྷྲི་ཥིའི་དྷི་ཥ་ཡང་ངོ་། །དྷྲྀ་ཥཱ་སྦྱི་བརྟོལ་ལའོ། །ཨཱ་ཡིག་ཨཱ་དི་ཏལས་
