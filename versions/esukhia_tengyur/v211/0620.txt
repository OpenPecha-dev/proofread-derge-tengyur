སིནྔྷཱར་བསྲེས་པས་ནད་པ་བྲབ། རྐང་ལག་གང་ཡིན་ཀྱང་མི་གཉིས་ཙམ་གྱིས་ཟིན་ཀྱང་མི་ཐུབ་པར་སྔགས་མཁན་གྱི་སྔར་སླེབ་ཡོང་བས་ཁྲུ་གང་ཙམ་དུ་སླེབ་པ་དང་གྲིས་འཕྲེད་ལ་རེ་ཁཱ་གསུམ་འཐེན། གང་ན་བ་དེ་འབྲས་བཙོས་པའི་ཟན་དང་སིནྔྷཱུ་ར་བསྲེས་པས་
གཏུམས་ལ་བཞག་པས་ཕན་ནོ། །སྔགས་ནི། ཨོཾ་གཱ་ཛཾ་ཏ་གྷོ་རཏ་ཨཱ་བ་ཏཱ་ཚ་ར་ཚེ་དུ་ཨུ་ཏཱ་ར་ཏ་ཨ་བ་ཏཱ་ཧ་ཡ། ན་བ་ནཱ་རི་བ་ཧ་ཏྲ་ཀོ་ཋཱ་སུ་ཥ་བ་ར་ཏཱ་ཨཱ་བ་ཏ་ཧ་ཡ། ཀཱ་ཧཱ་ཏེ་ཨཱ་བ་ར་ཧ་ཡ་ག་ཌྔྷ་ག་ཡ་ནི་ཏེ་ཨཱ་བ་ཏཱ་ཧ་ཡ། ཀཱ་བ་རུ་ཀོ་ཀོ་ཏ་བཱ་ལ་
དི་ལཱི་ཀོ་པ་ཏ་སཱ་ཧ། སེ་ཥ་སིལཱ་ར་རུ་རུ་ཀ་ནི་ཀེ་པུ་ཏ་མ་ཧ་མ་དཱ་པི་ར་མནྟྲ་པ་ཌྔྷ་ཏ། བ་དྷིཥུ་རཱ་ས་ན་བྷེ་ཛྱེ་བྷུ་ཏ་པཱཾ་ངྷཾཔྲེ་ཏ་པཱཾ་དྷིཔྲེ་ཏ་པཱཾ་ངྙི་དཱ་ནོབཱཾངྟི། མ་རི་ཡཱ་མ་སཱ་ན་བཱཾ་དྷི་ཀཱལ་བྷི་རོ་བཱཾདྷི། ཨཥྚ་བྷཻ་ར་བ་པཱཾ་དྷི་ལོ་ཧ་རོ་ཧ་བོ་ནཱ་སི་དྷ་ཤ་དྷ་པཾ་དྷི་པཱ་ན་ཕུ་ལ་ཀོ་བཱཾ་ངྷི། ཏེ་ལ་ཏ་ལེ་ཀོ་
པཱཾ་དྷི་བྷཱ་ཏ་དུ་དྷ་ཀོ་པཱཾདྷི། ཀ་ར་ན་ཀ་ར་ཏུ་ཏི་ཀོ་བཱདྷ་ཥི་ར་ཥེ་རཾ་ཥེ་ཧྲ་པཱ་ལ་བཱཾ་ངྷཾ། ཨ་ཋོ་ཏ་ར་སོ་ནཱིནཱ་ཨི་པཱཾ་དྷི་བཱ་རཱ་སྱ་མ་སཱན་པཱཾ་ངྷི་ཏེ་རཱ་སྱ་ཀ་རུ་བཱ་པཱཾ་དྷི། ཙོ་དཱ་སྱ་བྷཾསཱ་སུ་ར་པཱཾ་དྷི་སཾ་ཥ་ནཱི་ཌི་ཀ་ནཱི་པཱཾ་དྷི་པ་ཏ། ཚ་ནཱི་པཱཾ་དྷི་བྷེ་ག་ན་པཱཾ་དྷ་ཏོ་བཱ་བཱ་ཨཱ་ད་མ་ཀཱི་དོ་ཧཱ་ཨཱི། བི་བི་ཕཱཏ་
མཱ་ཀི་དུ་ཧཱ་ཨི་་སུ་བ་ར་མུ་ར་ཌཱ། པི་པི་ཕཱ་ཏ་མཱ་ཀེ་དཱ་བ་ན་གི་ར། ན་བ་ལཱ་ཥ་ཨ་སི་ཧ་ཛཱ་ར་པི་ར་པི་ཀཾ་བ་ར་བཱ་ངྷི་ཌ་རི། ཨ་བ་དེ་ཥོརེ་མ་ཧ་མ་དྭཱ་པི་ཏེ་རེ་མནྟ་ཀི་སཀྟཱི། ཨུ་ཏ་ར་ཀོ་ཨུ་ཏ་ར་པཱཾ་དྷཱི། ད་ཚ་ན་ཀོ་ད་ཚ་ན་ནཱཾ་ངྷི་པ་ཚཱི་མ་ཀོ་པ་ཚི་མ་བཱྀ་དྷི། པུརྦ་ཀོ་པུརྦ་
པཱཾདྷི། ཀཱ་ཋ་ཀཱི་རོ་ཊི་བཱཾ་དྷི་ཕི་རེ་མ་ཧ་མ་དཱ་པི་རཱ་སེ་ཥ་སི་ལ་ར་པཊྡྟ་ཏ་མནྟྲ། བཱ་དྷི་ཌཱ་རཱི་མེ་རཱི་བྷཀྟཱིགུ་རུ་ཀཱི་སིཀྟ་ཕུ་རོ་མནྟྲ་སཱི་སྭ་རི་བཱ་ཙཱ། ཞེས་པའོ། །རཱ་ས་རཱ་ཛེ་ཤྭ་ར་ཏེ་བཅུད་ལེན་དབང་ཕྱུག་རྒྱལ་པོའི་སྦྱོར་བ་ནི། དངུལ་ཆུ་དུག་བྲལ། མུ་ཟི་སེར་པོ། །མྲྀ་ཏ་
སཱ་ར། ཆུ་ཤིང་ནག་པོས་གོ་ཆོད་ཙམ། ཟངས་ཕྱེ། ཏཱ་ལ་ཀ །བཙན་དུག །ཚལ་དཀར། སི་མཾ་ལ། ལྡོང་རོས །རྒྱ་ཚྭ། ལྕགས་ཕྱེ། ཞ་ཉེ་དཀར་པོའི་ཕྱེ་མ། ཛཱཏི། བོང་ང་དམར་པོ་ནང་། ཧི་ཏི་མྦྷི་སི་ ལྗང་གུ། མུ་ས་ལི། ཛ་བ་ཥ་ར། ནཱི་བ།
