རོ༑ །ཤིཾ་གྷ་ཎ་ཀཿ སྣབས། ནཱ་སཱ་ཤྲ་བཿ སྣའི་དྲི་མ། ཀྲྀ་ཏི་བྷི་དི་ལ་ཏི་ལས་ཀྟི་ཀ་ནའོ། །ཀྲྀ་ཏཱི་གཅོད་པ་ལའོ། །བྷི་དིར་འཇོམས་པ་ལའོ། །བྷི་དིར་འཇོམས་པ་ལའོ། །ལ་ཏ་ཞེས་པ་མདོའི་བྱིངས། འདི་རྣམས་ལས་ཀྟི་ཀན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་
ཀིད་བྱ་བའི་དོན་ནོ། །ཌཱ་པ་ཨིཏྟ་ཀཱ་ སྨིན་དྲུག །ནཀྵ་ཏྲཱཿ རྒྱུ་སྐར། བིཏྟི་ཀཱ་ ཀུཌྱཾ། ལཏྟི་ཀཱ་ གོ་དྷཱ་ ཨི་ཥི་ལས་ཀྟ་ཀནའོ། །ཨི་ཥུ་འདོད་པ་ལའོ། །འདི་ལས་ཀྟ་ཀན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །སྟུའི་ཤྩུ་དང་ཥྚུ་དག་ལས་དེ་དག་གོ་ཞེས་པས་ཏ་ཡིག་གོ། །ཌཱབ྄་
