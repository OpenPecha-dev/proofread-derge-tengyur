ཨཱ་ཀཪྵ་ཡ་མཱ་ཏི་ཀྲ་མ་ཧཱུཾ་ཛཿཛཿ། སློབ་དཔོན་ཤར་དུ་ཁ་བལྟས་ལ། །གྲོགས་པོ་ནུབ་ཏུ་བལྟ་བར་བྱ། །སློབ་དཔོན་བྱང་དུ་ཁ་བལྟས་ལ། །གྲོགས་པོ་ལྷོ་རུ་བལྟ་བར་བྱ། །མེ་དང་འབྱུང་པོར་ཁ་བལྟས་ཏེ། །ཤར་གྱི་ཆར་ནི་ཐིག་གདབ་བྱ། །རླུང་དང་སྲིན་པོའི་ཕྱོགས་གནས་ཏེ། །ནུབ་
ཀྱི་ཐིག་ནི་གདབ་པར་བྱ། །དེར་གནས་ནས་ནི་བྱང་གི་ཐིག །སླར་ཡང་མེ་སྲིན་ལྷོ་ཕྱོགས་སུའོ། །ཁ་ཁྱེར་ཐིག་ཀྱང་དེ་བཞིན་ཏེ། །ཕྱོགས་ཀུན་མཉམ་པ་ཉིད་དུ་བསྒོམས། །མེ་དང་རླུང་གི་ཕྱོགས་གནས་ཏེ། །སྲིན་པོ་འབྱུང་པོའི་ཕྱོགས་སུ་བལྟ། །གཡས་སུ་བསྐོར་བའི་སྦྱོར་བ་
ཡིས། །ཕྱོགས་བཞིར་ཐིག་ནི་གདབ་པར་བྱ། །གཞན་དུ་བྱས་ན་སློབ་དཔོན་དང་། །གྲོགས་ཀྱི་ཉེས་པ་ཐོབ་པར་འགྱུར། །སྒོ་ནི་འཁོར་ལོའི་བརྒྱད་ཆ་སྟེ། །སྒོ་ཁྱུད་ཀྱང་ནི་སྒོ་ཡི་ཚད། །ཁྱམས་དང་འདྲ་བ་ཆ་ཡི་ཚད། །དེ་ཕྱེད་ཚོན་སྣ་ལྔ་ཡི་ས། །རྟ་བབས་ཀྱིས་ནི་ཚོན་གྱི་ས། །
སྒོ་ཡི་སུམ་འགྱུར་ཚད་དུ་བྱ། །གྲུ་བཞི་པ་ལ་སྒོ་བཞི་པ། །སྒོ་ཁྱུད་བཞིས་ནི་རྣམ་པར་བརྒྱན། །དྲ་བ་དྲ་ཕྱེད་ཉི་མ་དང་། །ཟླ་བ་སོགས་ཀྱིས་བརྒྱན་པར་བྱ། །རྟ་བབས་ཀྱི་ནི་ས་ཡི་ཆ། །ཇི་ལྟར་མཛེས་པ་རྣམ་པར་བརྟག །འཁོར་ལོའི་ཚུལ་ལ་དཀྱིལ་འཁོར་དགུ། །{𰵻གོ་རིམས་,གོ་རིམ་}
བཞིན་དུ་བྲིས་པ་ཡི། །འཁོར་ལོ་རྩིབས་བརྒྱད་མཛེས་པར་བྱ། །རྩིབས་བརྒྱད་འཁོར་ལོའི་རྒྱབ་ཏུ་ནི། །རྡོ་རྗེ་ཕྲེང་བ་ཡང་དག་བྲི། །འཁོར་ལོ་རྩིབས་བརྒྱད་དབུས་སུ་ནི། །ཤཱཀྱ་སེང་གེ་ངེས་པར་བྲི། །པདྨ་དབུ་མའི་འདབ་མ་ལ། །རྡོ་རྗེ་སེམས་མ་བྲི་བར་བྱ། །མཚམས་
ལ་རིན་ཆེན་སེམས་མ་སོགས། །{𰵼གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བྲི་བར་བྱ། །ཤར་གྱི་རྩིབས་ལ་བྲི་བ་ནི། །རྡོ་རྗེ་གཙུག་ཏོར་དེ་བཞིན་གཤེགས། །དེ་བཞིན་རྩིབས་ནི་ཐམས་ཅད་ལ། །{𰵽གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བྲི་བར་བྱ། །མཚམས་ཀྱི་ཆར་ནི་སྒེག་མོ་སོགས། །{𰵾གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བྲི་བར་བྱ། །
དེ་བཞིན་མེ་ཏོག་སྤོས་མེ་སོགས། །ཤིན་ཏུ་མཉམ་པར་བཞག་སྟེ་བྲི། །སྒོ་ནི་སྲུང་བ་བཞི་པོ་དག །སྒོ་ཡི་དབུས་སུ་བྲི་བར་བྱ། །བྱམས་པ་ལ་སོགས་བཅུ་དྲུག་པོ། །ཤར་སོགས་ཁྱམས་སུ་གནས་པར་བྱ། །སྒོ་བཞི་ཡི་ནི་ལོགས་དག་ཏུ། །གླང་ཆེན་དང་ནི་སེང་གེ་བྲི། །
