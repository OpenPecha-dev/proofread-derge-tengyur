ཚུལ་ཁྲིམས་ཀྱི་འབྲས་བུ་དང་། བཟོད་པ་དང་། བཟོད་པའི་འབྲས་བུ་དང་། བརྩོན་འགྲུས་དང་། བརྩོན་འགྲུས་ཀྱི་འབྲས་བུ་དང་། བསམ་གཏན་དང་། བསམ་གཏན་གྱི་འབྲས་བུ་དང་། ཤེས་རབ་དང་། ཤེས་རབ་ཀྱི་འབྲས་བུ་ལ་དད་པའོ་ཞེས་གསུངས་སོ། །ཡང་དེ་ཉིད་ལས་
འབྱུང་བ། དད་པའི་མཚན་ཉིད་ནི་དད་པ་སྟེ། གང་ཐེག་པ་ཆེན་པོ་ལ་མོས་པ་འདི་ནི་དད་པའི་སྟོབས་ཞེས་བྱའོ། །གང་ཆགས་པ་མེད་པ་དང་། ཁོང་དུ་ཆུད་པ་དང་། དད་པའི་དབང་པོ་དང་། དད་པའི་སྟོབས་འདི་ནི་དད་པ་ཞེས་བྱའོ། །གཞན་ཡང་དད་པའི་སྟོབས་ནི་དད་པ་གང་གིས་
