གང་དག་ཡི་གེ་དང་། དོན་དང་། ཡིད་ལ་བྱེད་པ་དང་། མི་འཕྲོ་བ་དང་། རང་གི་མཚན་ཉིད་དང་། སྤྱིའི་མཚན་ཉིད་དང་། མ་དག་པ་དང་། དག་པ་དང་། གློ་བུར་དང་། མི་སྐྲག་པ་དང་། ཁེངས་པ་མེད་པ་ལ་ཕྱིན་ཅི་མ་ལོག་པ་དེ་དག་ངོ་བོ་ཉིད་གསུམ་གྱི་དབང་དུ་
བྱས་པའོ་ཞེའོ། །དེ་ལ་ཡི་གེ་དང་དོན་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཀུན་བརྟགས་པའི་ངོ་བོ་ཉིད་ཀྱི་དབང་དུ་བྱས་པ་སྟེ། འབྲེལ་པ་དང་གཉིས་སུ་ལྟུང་བའི་དོན་ནི་བརྟགས་པའི་ཕྱིར་རོ། །དེ་ཁོ་ནའི་ཡུལ་མ་ཡིན་དུ་ཟིན་ཀྱང་། དེ་བཞིན་དུ་ཕྱིན་ཅི་མ་ལོག་པ་དེ་བཞིན་ཉིད་ཀྱི་ཕྱིར་ཏེ། དེ་གཉིས་ཕྱིན་ཅི་མ་ལོག་
པ་ཉིད་དོ། །ཡིད་ལ་བྱེད་པ་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་གཞན་གྱི་དབང་གི་ངོ་བོ་ཉིད་ཀྱི་དབང་དུ་བྱས་པ་སྟེ། རྟོག་པ་དང་ཡིད་ལ་བྱེད་པ་གཉིས་རང་བཞིན་གྱིས་ཀུན་གཞི་རྣམ་པར་ཤེས་པ་ཡིན་པའི་ཕྱིར་རོ། །མི་འཕྲོ་བ་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཡོངས་སུ་གྲུབ་པའི་ངོ་བོ་ཉིད་ཀྱི་དབང་དུ་བྱས་པ་སྟེ།
ཅིའི་ཕྱིར་ཞེ་ན། རྣམ་པར་མི་རྟོག་པའི་ཡེ་ཤེས་ནི། ཡོད་པ་དང་མེད་པ་ལ་གཡེང་བ་དང་བྲལ་བའི་ཕྱིར་དང་། ཕྱིན་ཅི་མ་ལོག་པར་ཡོངས་སུ་གྲུབ་པས་ཡོངས་སུ་གྲུབ་པའི་ངོ་བོ་ཉིད་དུ་བསྡུས་པའི་ཕྱིར་རོ། །རང་དང་སྤྱིའི་མཚན་ཉིད་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ཉིད་ལ་སྟེ། རྣམ་པར་
རྟོག་པའི་ཡུལ་ལས་འདས་པའི་ངོ་བོ་ཉིད་ཀྱི་ཕྱིར་རོ། །མ་དག་པ་དང་དག་པ་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ཉིད་ཀྱི་གནས་སྐབས་ཀྱི་བྱེ་བྲག་ལའོ། །གློ་བུར་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ཉིད་ལ་སྟེ། ནམ་མཁའ་བཞིན་དུ་རང་བཞིན་གྱིས་རྣམ་པར་དག་པ་དེ་ལ། སེམས་ཀུན་ནས་ཉོན་
མོངས་པ་དང་རྣམ་པར་བྱང་བ་འདི་གཉིས་ནི་གློ་བུར་ཡིན་པར་རྣམ་པར་གཞག་གོ། །མི་སྐྲག་པ་དང་ཁེངས་པ་མེད་པ་ལ་ཕྱིན་ཅི་མ་ལོག་པ་ནི་ཆོས་ཀྱི་དབྱིངས་ཉིད་ལ་སྟེ། དེ་ནམ་མཁའ་བཞིན་དུ་བཅོས་མ་མ་ཡིན་པའི་ཕྱིར་འབྲི་བ་དང་ཁྱད་པར་དུ་གྱུར་པ་མེད་པའི་ཕྱིར་རོ། །དེ་ལྟར་ཕྱིན་ཅི་མ་ལོག་པ་བཅུ་པོ་འདི་དག་ངོ་བོ་
ཉིད་གསུམ་དུ་རིག་པར་བྱའོ། །བཤད་མ་ཐག་པའི་ཕྱིན་ཅི་མ་ལོག་པ་བཅུ་པོ་འདི་དག་ནི་མདོ་སྡེ་ལས་གསུངས་པའི་རྡོ་རྗེའི་ཚིག་བཅུ་དག་དང་{𰷄གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་སྦྱར་བར་བྱ་སྟེ། རྡོ་རྗེའི་ཚིག་བཅུ་པོ་གང་ཞེ་ན། ཡོད་པ་དང་མེད་པ་དང་། ཕྱིན་ཅི་མ་ལོག་པ་དང་། གནས་དང་། སྒྱུ་མ་ལྟ་བུ་ཉིད་དང་།
