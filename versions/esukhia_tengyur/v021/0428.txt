འཁྱུད་ནས། །རྡོ་རྗེ་ཆེ་དང་དྲིལ་བུ་ཆེ། །ཟུང་ཤིག་རྡོ་རྗེས་བྱིན་བརླབས་ནས། །དེ་རིང་རྡོ་རྗེ་སློབ་དཔོན་གྱུར། །སློབ་མའི་ཚོགས་རྣམས་བསྡུ་བར་གྱིས། །དེ་ལྟར་དམ་ཚིག་བརྟན་བྱ་སྟེ། །རྗེས་སུ་གནང་བ་སྦྱིན་པར་བྱ། །ཁ་ཊྭཱཾ་ག་དང་ཌ་མ་རུ། །ཁ་ཊྭཱཾ་ག་ནི་ལྷ་ཡི་སྐུ། །ཤེས་རབ་ཌ་མ་རུ་ཡི་སྒྲ། །
ཁ་ཊྭཱཾ་རྡོ་རྗེ་དང་བཅས་ཤིང་། །དྲིལ་བུ་སིལ་སིལ་སྒྲ་སྒྲོགས་པ། །ཚངས་པའི་སྐུད་ཐོགས་རྣལ་འབྱོར་བདག །ཁ་ཊྭཱཾ་ག་དང་རྣལ་འབྱོར་སྣོད། །དཔའ་བོ་ཆེན་པོ་བླང་བར་གྱིས། །སྤྱོད་པའི་རྗེས་སུ་གནང་བའོ། །དེང་ནས་བརྩམས་ཏེ་རྫོགས་སངས་རྒྱས། །སྲས་དང་བཅས་ལ་རྣམ་རོལ་ཅིང་། །རྡོ་རྗེ་
སྙེམས་མའི་རྣལ་འབྱོར་གྱིས། །དགའ་སྟོན་བདེ་བའི་མཆོད་པ་གྱིས། །ཨཱ་སྒེག་པའོ། །སྒེག་དང་བཅས་པའི་ཕྲེང་བ་འཕང་། །སངས་རྒྱས་ཐམས་ཅད་མཆོད་པར་བྱ། །དེ་ནི་བཅོམ་ལྡན་འབྲལ་མི་འགྱུར། །ཕྲེང་བ་དེ་ཡི་རྒྱན་ཡིན་ནོ། །ཨོཾ། ཕྲེང་བའོ། །ལུས་ཅན་ཀུན་ལ་དམ་ཆོས་ཀྱི། །ཁྱོད་
ཀྱིས་གླུ་ནི་བླང་བར་གྱིས། །དེ་ཡིས་བདེ་གཤེགས་དངོས་གྲུབ་ཀྱིས། །བདེ་བ་མྱུར་དུ་མྱོང་བར་འགྱུར། གཱི། གླུའིའོ། །དེ་ཉིད་ཐལ་མོ་གར་གྱིས་ནི། །སངས་རྒྱས་ཐམས་ཅད་ཆགས་པར་བྱ། །སངས་རྒྱས་གདུལ་བའི་ཕྲིན་ལས་ཀྱིས། །གར་མཁན་ཚུལ་དུ་སྒྱུར་བར་གྱིས། ཧོ། གར་རོ། །རྫོགས་

ནི། །བདེ་གཤེགས་སྐུ་ནི་ཐོབ་པར་གྱིས། བྷྲཱུཾ། མེ་ཏོག་གོ། །སེམས་ཅན་མ་རིག་ལྡོངས་པ་ལ། །ཆོས་ཀྱི་མེ་ནི་ལེགས་སྦར་ནས། །རྡོ་རྗེ་ཆོས་དང་འདྲ་བ་རུ། །ཆོས་ཀྱི་རྣམ་པར་འབར་བར་གྱིས། །མར་མེའོ། །ཁྱོད་ཀྱིས་རྒྱལ་ལ་དྲི་ཕུལ་བས། །ཚུལ་ཁྲིམས་ཏིང་འཛིན་ཤེས་རབ་དང་། །
རྣམ་གྲོལ་ཡེ་ཤེས་མཐོང་བ་ཡི། །བྱང་ཆུབ་ལུས་ཅན་རྣམས་ལ་བསྒོས། །དྲིའོ། ཇི་ལྟར་མིག་མཁན་རྒྱལ་པོ་ཡིས། །འཇིག་རྟེན་ལིང་ཐོག་བསལ་བ་ལྟར། །བུ་ཁྱོད་ཀྱིས་ནི་མི་ཤེས་པའི། །རབ་རིབ་རྒྱལ་བ་རྣམས་ཀྱིས་བསལ། །རྒྱལ་རྣམས་ཀྱིས་ནི་ཐུར་མས་སོ། །ཆོས་རྣམས་
