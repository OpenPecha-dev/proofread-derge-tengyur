དག་ལྡན། །ཇི་སྟེ་ཆོ་ག་བཞིན་བསྟེན་ན། །དེ་ཡི་མདུན་ན་ང་གནས་ཏེ། །སངས་རྒྱས་བསྙེན་པ་ལ་སོགས་པའི། །བྱ་བ་ཐམས་ཅད་པདྨས་བྱེད། །བདེ་ཆེན་རྒྱལ་པོ་རང་ཉིད་ནི། །འདི་ཉིད་དུ་ནི་ཡང་དག་གནས། །བྷཉྫ་མོཀྵ་ཧོ། རང་གི་ལྷ་ཡི་སྦྱོར་བ་ཡིས། །རྡོ་རྗེ་པདྨ་
ལེགས་སྦྱངས་ནས། །བླ་མས་ཆོ་ག་བརྗོད་བྱས་སྟེ། །སྦྱོར་སྟངས་དྲུག་པོ་སྔོན་འགྲོ་བས། །ཕྲ་བའི་ཡེ་ཤེས་བསྐྱེད་པར་བྱ། །ཡེ་ཤེས་ཐིག་ལེ་བརྟན་པ་དང་། །དམ་ཚིག་འདི་ཡང་བསྒྲག་པར་བྱ། །མཚན་ལྡན་བུད་མེད་མཐོང་བ་དང་། །གཡོན་ནས་ལན་གསུམ་བསྐོར་བྱས་ཏེ། །ཐལ་སྦྱར་
ནས་ནི་གསོལ་བ་གདབ། །འདོད་ཆགས་བྲལ་བ་ལྟ་བུ་ཡི། །ཁམས་གསུམ་ནས་ནི་སྡིག་མེད་པས། །དེ་བས་འདོད་ལ་ཆགས་བྲལ་བར། །ནམ་དུ་ཡང་ནི་མི་བྱའོ། །མ་ཧཱ་ས་མ་ཡ་ཧཱུཾ་ཕཊ། །དེ་ནས་ཚུལ་བཞིན་གནས་པ་ལ། །དག་པ་ཡང་ནི་རབ་ཏུ་བརྗོད། །འབྲས་བུ་རྣམས་ནི་ཕུལ་ནས་
ཀྱང་། །ཞབས་ལ་སྤྱི་བོས་ཕྱག་བྱས་ཏེ། །ཁྱོད་ཞབས་པདྨའི་བཀའ་དྲིན་གྱིས། །གཙོ་བོ་དབང་བསྐུར་རྣམ་གསུམ་ཐོབ། །དེ་རིང་རིན་ཆེན་བཞི་པ་ཡང་། །བདག་ལ་བཀའ་དྲིན་རབ་ཏུ་སྩོལ། །སློབ་མས་ལན་གསུམ་བརྗོད་ཐོས་ནས། །{གོ་རིམས་,གོ་རིམ་}རབ་ཏུ་བརྗོད་བྱས་ཏེ། །རིན་ཆེན་ཚིག་གིས་
དབང་བསྐུར་ནས། །འདི་ཡིས་རབ་ཏུ་བསྟན་པར་བྱ། །འདི་ནི་རང་ཉིད་མཛེས་མ་སྟེ། །ཕྱི་རོལ་ཡུལ་ནི་རྣམ་སྤངས་པ། །རིག་འདི་བོར་ནས་དེ་ལས་ནི། །གཞན་པའི་ཐབས་ཀྱིས་སངས་རྒྱས་མེད། །རིག་མའི་བརྟུལ་ཞུགས་སྤྱོད་ཅིག་པུ། །བརྟུལ་ཞུགས་རྣམས་ཀྱིས་བླ་མའི་བླ། །མཚན་མེད་
བསམ་གྱིས་མི་ཁྱབ་པ། །འདི་ལ་རྟག་ཏུ་གནས་པར་གྱིས། །དེ་ནས་དག་པ་རབ་ཏུ་བརྗོད། །ལམ་གྱི་དམ་ཚིག་འདི་ཡང་སྦྱིན། །ཁྱོད་ཀྱིས་སྲོག་ཆགས་གསད་པར་བྱ། །གཞན་གྱི་བུད་མེད་རྣམས་སྤྱད་ནས། །མ་བྱིན་པར་ཡང་ཁྱོད་ཀྱིས་ལོང་། །ཐམས་ཅད་ལ་ཡང་{བརྫུན་,རྫུན་}ཚིག་སྨྲོས། །བདེ་བའི་
དོན་གྱིས་སྤྲོ་བཞིན་གནས། །དཀྱིལ་འཁོར་ཉི་མ་མ་ཤར་བར། །ལེགས་པར་གཤེགས་སུ་གསོལ་བར་བྱ། །བུམ་པ་ལ་སོགས་རྫས་རྣམས་ཀུན། །ཅི་བདེ་བར་ནི་གནས་པར་བྱ། །དེ་བས་རྟོག་པ་ཐམས་ཅད་ཀྱིས། །དགེ་འདུན་རྣམས་ནི་མཆོད་པར་བྱ། །བྱིས་པ་རྣམས་ཀྱང་ཚིམ་པ་དང་། །
