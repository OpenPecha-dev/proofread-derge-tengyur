སོར་རྟོག་པའི་ཕྱིར་སྒྲོ་འདོགས་པ་མེད་པ་དང་སྐུར་པ་འདེབས་པ་མེད་པ་གཉིས་ཀྱིས་མཉམ་པ་ལ་འཇོག་པར་བྱེད་པའི་ཕྱིར་འཇལ་བར་བྱེད་པ་ཡིན་ནོ། །བསྒོམ་པའི་ལམ་གྱིས་ནི་ཇི་ལྟར་འཇལ་བ་ལ་ཡང་དང་ཡང་དུ་རྟོག་པར་བྱེད་པའི་ཕྱིར་ཉེ་བར་རྟོག་པར་
བྱེད་པ་ཡིན་ནོ། །བདེན་པ་དེ་ཡང་ཞེས་བྱ་བ་ནི་གང་གི་དབང་དུ་མཛད་ནས་ཆོས་བསྟན་པའི་མྱ་ངན་ལས་འདས་པ་ཡིན་ནོ། །ལུས་ཀྱིས་མངོན་སུམ་དུ་བྱེད་ཅིང་ཞེས་བྱ་བ་ནི་གནས་གྱུར་པ་ཉིད་ཀྱིས་ཏེ། རྣམ་པར་གྲོལ་བ་ཐོབ་པའི་ཕྱིར་རོ། །ཤེས་རབ་ཀྱིས་སོ་སོར་འབིགས་པར་བྱེད་
དོ་ཞེས་བྱ་བ་ནི་རྣམ་པར་གྲོལ་བའི་ཡེ་ཤེས་ཀྱིས་སོ། །དེ་ལྟར་ན་དད་པ་ལ་བརྟེན་ནས་ཚོགས་དང་བཅས་པའི་ལམ་དང་། ལམ་གྱི་འབྲས་བུ་རྣམ་པར་གྲོལ་བ་དང་། རྣམ་པར་གྲོལ་བའི་ཡེ་ཤེས་ཀུན་ཏུ་སྟོན་པར་བྱེད་དོ། །གཞི་དེ་ལས་སྦྱིན་པ་པོ་དང་སྦྱིན་བདག་གི་བསོད་
ནམས་རྒྱུན་རྒྱས་པ་ཞེས་བྱ་བ་ནི་མདོ་སྡེའི་དུམ་བུའོ། །དེ་ལ་སྦྱིན་པ་པོ་ནི་གཞན་གྱི་ནོར་ཡང་བདག་གིས་སྦྱིན་པར་བྱེད་པའོ། །སྦྱིན་བདག་ནི་གང་གིས་ནོར་སྦྱིན་པར་བྱེད་པའོ། །བསོད་ནམས་ཀྱི་ལས་བྱེད་པ་ན་རྒྱུད་ལ་བག་ཆགས་རྣམ་པ་གཉིས་གནས་ཏེ། འབྲས་
བུ་རྣམ་པར་སྨིན་པ་ཡིད་དུ་འོང་བ་འགྲུབ་པར་བྱེད་པ་དང་། དད་པ་ལ་སོགས་པ་དགེ་བའི་ཆོས་ཀྱི་རྒྱུ་མཐུན་པའི་འབྲས་བུ་འགྲུབ་པར་བྱེད་པ་སྟེ། ཡོན་གནས་ཀྱི་ཁྱད་པར་གྱིས་དེའི་སྦྱིན་པར་བྱ་བའི་དངོས་པོ་ཡོངས་སུ་སྤྱད་པའི་ཕྱིར་རོ། །དེ་གཉིས་བརྟས་པའི་ཁྱད་པར་དུ་འགྲོ་བ་
ནི་{𰵭གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བསོད་ནམས་རྒྱུན་རྒྱས་པ་དང་། དགེ་བ་རྒྱུན་རྒྱས་པ་ཚད་མེད་པ་རིག་པར་བྱའོ། །དེ་རྒྱུན་རྒྱས་པ་ཡང་བདེ་བའི་རྒྱུན་ཡིན་པའི་ཕྱིར་བདེ་བ་འདྲེན་པ་ཞེས་བྱ་བ་ནི་ཕན་ཡོན་ཡོངས་སུ་སྟོན་པར་བྱེད་པ་ཡིན་ནོ། །གཞན་ཡང་སྦྱིན་པ་ལས་བྱུང་བའི་བསོད་ནམས་གང་ཡིན་
པ་དང་། དེ་རབ་ཏུ་འཇུག་པར་བྱེད་པའི་དགེ་བ་ཡང་དག་པའི་ལྟ་བ་གང་ཡིན་པ་དེ་གཉིས་ཀྱིས་ཁམས་བརྟས་པ་ནི་{𰵮གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་བསོད་ནམས་རྒྱུན་རྒྱས་པ་དང་། དགེ་བ་རྒྱུན་རྒྱས་པ་ཡིན་ནོ། །དེ་གཉིས་ཀྱི་འབྲས་བུ་ཡིད་དུ་འོང་བའི་ས་བོན་བརྟས་པ་ནི་བདེ་བ་འདྲེན་པ་ཡིན་ནོ། །
