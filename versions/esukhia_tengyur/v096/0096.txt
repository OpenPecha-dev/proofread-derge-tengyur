བྱ་བ་མེད་པའི་ཚེ། ཇི་ལྟར་སྐྱེ་བ་ལ་བརྟེན་ནས་སྐྱེ་བཞིན་པ་ཞེས་བརྗོད་པར་བྱ། ཡང་གཞན་ཡང་། རྟེན་ཅིང་འབྱུང་བ་གང་ཡིན་པ། །དེ་ནི་ངོ་བོ་ཉིད་ཀྱིས་ཞི། །དེ་ཕྱིར་སྐྱེ་བཞིན་ཉིད་དང་ནི། །སྐྱེ་བ་ཡང་ནི་ཞི་བ་ཉིད། །འདི་ལྟར་བརྟེན་ནས་འབྱུང་བ་གང་ཡིན་པ་དེ་ནི་ངོ་བོ་ཉིད་ཀྱིས་ཞི་བ་
དེའི་ཕྱིར་སྐྱེ་བཞིན་པ་ཉིད་དང་སྐྱེ་བ་ཡང་ཞི་བ་ཉིད་དོ། །འདིར་སྨྲས་པ། དུས་གསུམ་རྣམ་པར་བཞག་པ་ལས་མ་འོངས་པའི་དངོས་པོ་མ་འོངས་པའི་བདག་ཉིད་ཀྱིས་ཡོད་པས་དེ་སྐྱེ་བར་འགྱུར་རོ། །འདིར་བཤད་པ། གལ་ཏེ་དངོས་པོ་མ་སྐྱེས་པ། །འགའ་ཞིག་གང་ན་ཡོད་གྱུར་ན། །དེ་ནི་
ཅི་ཕྱིར་དེར་སྐྱེ་འགྱུར། །ཡོད་ན་སྐྱེ་བར་མི་འགྱུར་རོ། །གལ་ཏེ་དངོས་པོ་མ་སྐྱེས་པ་འགའ་ཞིག་མ་འོངས་པའི་དུས་གང་ན་ཡོད་པར་གྱུར་ན། དེ་ཅིའི་ཕྱིར་ད་ལྟར་གྱི་དུས་དེར་སྐྱེ་བར་འགྱུར་ཏེ། དེ་དེ་ན་ཡོད་ན་ཡོད་པའི་ཕྱིར་ད་འདིར་སྐྱེ་བར་མི་འགྱུར་རོ། །ཅི་སྟེ་སྐྱེ་བར་འདོད་ན། དེ་ལ་བཤད་པར་
བྱ་སྟེ། དངོས་པོ་དེ་འདིར་སྐྱེ་བར་གྱུར་ན། དེ་ལྟ་ན་དངོས་པོ་དེ་མ་འོངས་པའི་དུས་དེ་ན་ཡོད་པ་མ་ཡིན་པར་འགྱུར་ཏེ། དེ་ཡང་མི་འདོད་པས། དེའི་ཕྱིར་མ་འོངས་པའི་དངོས་པོ་ཡོད་པ་དེ་སྐྱེ་བར་འགྱུར་རོ་ཞེས་བྱ་བ་དེ་ཡང་མི་རིགས་སོ། །ཡང་གཞན་ཡང་། ཁྱོད་ཀྱིས་སྔར་སྐྱེ་བས་སྐྱེ་བཞིན་པ་
སྐྱེད་པར་བྱེད་དོ་ཞེས་གང་སྨྲས་པ་དེ་ལ་བཤད་པར་བྱ་སྟེ། གལ་ཏེ་སྐྱེ་བ་དེ་ཡིས་ནི། །སྐྱེ་བཞིན་པ་ནི་སྐྱེད་བྱེད་ན། །སྐྱེ་བ་དེ་ནི་སྐྱེད་བྱེད་པ། །སྐྱེ་བ་ཡང་ནི་གང་ཞིག་ཡིན། །གལ་ཏེ་སྐྱེ་བ་དེ་དེས་སྐྱེ་བཞིན་པ་སྐྱེད་པར་བྱེད་ན། འོ་ན་ད་སྐྱེ་བ་སྐྱེད་པར་བྱེད་པའི་སྐྱེ་བ་ཡང་གང་ཞིག་ཡིན། དེ་ལ་འདི་
སྙམ་དུ་སྐྱེ་བ་གཞན་ཞིག་གིས་སྐྱེ་བ་དེས་སྐྱེད་པར་སེམས་ན། དེ་ལ་བཤད་པར་བྱ་སྟེ། གལ་ཏེ་སྐྱེ་བ་གཞན་ཞིག་གིས། །དེ་སྐྱེད་ཐུག་པ་མེད་པར་འགྱུར། །གལ་ཏེ་སྐྱེ་བ་གཞན་ཞིག་གིས་སྐྱེ་བཞིན་པ་སྐྱེད་པར་བྱེད་པའི་སྐྱེ་བ་དེ་སྐྱེད་པར་བྱེད་ན། དེ་ལྟ་ན་ཐུག་པ་མེད་པར་ཐལ་བར་འགྱུར་རོ། །
ཅི་སྟེ་འདི་སྙམ་དུ་སྐྱེ་བཞིན་པ་སྐྱེད་པར་བྱེད་པའི་སྐྱེ་བ་དེ་ནི་སྐྱེད་པ་གཞན་མེད་པ་ཁོ་ནར་སྐྱེས་པར་སེམས་ན། དེ་ལ་བཤད་པར་བྱ་སྟེ། ཅི་སྟེ་སྐྱེ་བ་མེད་སྐྱེས་ན། །ཐམས་ཅད་དེ་བཞིན་སྐྱེ་བར་འགྱུར། །ཅི་སྟེ་སྐྱེ་བཞིན་པ་སྐྱེད་པའི་སྐྱེ་བ་དེ་སྐྱེད་པ་གཞན་མེད་པར་སྐྱེས་པར་རྟོག་ན་གཞན་ཐམས་ཅད་ཀྱང་དེ་
