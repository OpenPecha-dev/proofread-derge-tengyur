མ་བྱས་པ་རྒྱུ་མེད་པ་ལས་ཇི་ལྟར་འབྱུང་བར་འགྱུར་ཏེ། སྐྱོན་དུ་མར་ཐལ་བར་འགྱུར་བའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་སྒྲ་ལ་སོགས་པ་དངོས་པོ་ཐམས་ཅད་ལ་ཡང་རྣམ་པ་བཞི་པོ་དག་མི་འཐད་པས་འགྲུབ་པར་བལྟ་བར་བྱའོ། །སྡུག་བསྔལ་བརྟག་པ་ཞེས་བྱ་བ་སྟེ་རབ་ཏུ་བྱེད་པ་བཅུ་གཉིས་
པའོ།། །།སྨྲས་པ། སྡུག་བསྔལ་ཡང་ཡོད་ཕྱི་རོལ་གྱི་དངོས་པོ་རྣམས་ཀྱང་ཡོད་དེ། དེ་དག་ཡོད་པ་ལ་རྣམ་པ་བཞི་པོ་འབའ་ཞིག་མི་འཐད་དོ། །རྣམ་པ་བཞི་པོ་དག་མེད་དུ་ཟིན་ཀྱང་རེ་ཞིག །དངོས་པོ་རྣམས་ནི་རབ་ཏུ་གྲུབ་པོ། །བཤད་པ། ཅི་ཁྱོད་སྒྱུ་མའི་གླང་པོ་ཆེས་
འགྲོ་བར་འདོད་དམ། ཁྱོད་རྣམ་པ་བཞི་པོ་དག་གིས་མ་བྱས་པའི་དངོས་པོ་རྣམས་ཡང་དག་པར་ཡོད་པར་རྟོག་གོ། །འདིར་ཡང་དག་པ་གང་ཡིན་པ་དེ་ཉིད་གཟུང་བར་བྱ་བའི་རིགས་པ་སྙམ། སྨྲས་པ། འདིར་ཡང་དག་པ་གང་ཡིན། བཤད་པ། ཆོས་གང་སླུ་བ་དེ་{𰵬བརྫུན་,རྫུན་}ཞེས། །
བཅོམ་ལྡན་འདས་ཀྱིས་དེ་སྐད་གསུངས། །འདུ་བྱེད་ཐམས་ཅད་སླུ་བའི་ཆོས། །དེས་ན་དེ་དག་{𰵭བརྫུན་,རྫུན་}པ་ཡིན། །འདི་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་མདོ་སྡེ་གཞན་ལས་ཆོས་གང་སླུ་བ་དེ་ནི་{𰵮བརྫུན་,རྫུན་}པའོ། །དགེ་སློང་དག་འདི་ལྟ་སྟེ། མི་སླུ་བའི་ཆོས་མྱ་ངན་ལས་འདས་པ་དེ་ནི་བདེན་པའི་
མཆོག་གོ་ཞེས་གསུངས་སོ། །དེ་བཞིན་དུ་བདེན་པ་གཅིག་སྟེ། གཉིས་པ་མེད་ཅེས་ཚིགས་སུ་བཅད་པ་ཡང་གསུངས་སོ། །དེ་བཞིན་དུ་གཞན་ནས་ཀྱང་འདུས་བྱས་དེ་ནི་སླུ་བའི་ཆོས་ཀྱང་ཡིན། དེ་ནི་རབ་ཏུ་འཇིག་པའི་ཆོས་ཀྱང་ཡིན་ནོ་ཞེས་འདུ་བྱེད་ཐམས་ཅད་སླུ་བའི་ཆོས་ཅན་ཡིན་
པར་གསུངས་སོ། །དེའི་ཕྱིར་འདུ་བྱེད་ཐམས་ཅད་སླུ་བའི་ཆོས་ཉིད་དེས། ཐམས་ཅད་{𰵯བརྫུན་,རྫུན་}པ་ཉིད་ཡིན་ཏེ། གང་དག་{𰵰བརྫུན་,རྫུན་}པ་དེ་དག་ཇི་ལྟར་རབ་ཏུ་འགྲུབ་པར་འགྱུར། ཁྱོད་ཀྱིས་དངོས་པོ་རྣམས་ནི་རབ་ཏུ་གྲུབ་པོ་ཞེས་གང་སྨྲས་པ་དེ་ནི་སྲེད་པས་བསྐྱོད་པར་ཟད་དོ། །སྨྲས་པ། གལ་
ཏེ་འདུ་བྱེད་ཐམས་ཅད་{𰵱བརྫུན་,རྫུན་}པ་ཡིན་ན་འཛིན་བཞིན་དུ་ཡང་དངོས་པོ་ཐམས་ཅད་མེད་དོ་ཞེས་དེ་དག་མི་གསལ་བར་བྱས་པར་མི་འགྱུར་རམ། བཤད་པ་མི་འགྱུར་ཏེ། གལ་ཏེ་སླུ་ཆོས་གང་ཡིན་པ། །དེ་{𰵲བརྫུན་,རྫུན་}དེ་ལ་ཅི་ཞིག་སླུ། །བཅོམ་ལྡན་འདས་ཀྱིས་དེ་གསུངས་པ། །
