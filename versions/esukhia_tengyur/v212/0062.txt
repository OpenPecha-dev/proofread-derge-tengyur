ཨི་ཛྱེ་ཏེ། བ་ས། ཨུ་པྱ་ཏེ། བ་ཧ་ཨུ་ཧྱ་ཏེ། བེ་ཉ། མཚམས་སྦྱོར་གྱི་ཡི་གེའི་མཐའ་ཉིད་ཀྱི་ཕྱིར་ཨཱ་ཉིད་དོ། །སཾ་པྲ་སཱ་ར་ཎའོ། །ནཱ་མིན་མཐའ་རྣམས་ཀྱི་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཨཱུ་ཡ་ཏ། བྱེཉ། བཱི་ཡ་ཏེ། ཧྭེཉ། ཧཱུ་ཡ་ཏེ། ཨཱ་ཧཱུ་ཡ་ཏེ། བ་དཱ། ཡུརྱ་ཏེ། བ་ས། ཤཱི་
སི་བ་སི་གྷ་སི་རྣམས་ཀྱང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཨུ་ཥྱ་ཏེ། ཊུ་ཨོ་ཤྭི། ཤཱུ་ཡ་ཏེ། ཨས་ཡོད་པ་ལའོ། །ཨ་སྟིའི་བྷཱུའོ་ཞེས་པས་བྷཱུ་རུ་བསྒྱུར་རོ། །བྷཱུ་ཡ་ཏེ། ཙཀྵི། ཙཀྵི་ངའི་ཁྱཱ་ཉ་རུ་བསྒྱུར་རོ། །ཁྱཱ་ཡ་ཏེ། ཨཱ་ཁྱཱ་ཡ་ཏེ། ཨ་ཛ། ཨ་ཛིའི་བཱིའོ་ཞེས་པས་པཱ་རུ་བསྒྱུར་རོ། །
བཱི་ཡ་ཏེ། ཨཱུ་ཧ། རྣམ་པར་རྟོག་པ་ལའོ། །སཾ་སྔོན་དུའོ། །ཨཱུ་ཧིའི་ཉེར་བསྒྱུར་ལས་ཐུང་ངུའོ་ཞེས་ཁ་ཅིག་གོ། །ས་མུ་ཧྱ་ཏེ། འདིར་བ་ཧིའི་ནཾ་རབ་ཏུ་སྦྱོར་བ་ལའོ། །ཤྲནྶུ་དྷཱནྶུ་ཨིཏ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར་ཨ་ནུ་ཥངྒ་དབྱིའོ། །ཤྲ། བྱ་ཏེ། དྷ་སྱ་ཏེ། ཀ་པི་བཱ་

པོ་ལས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །པ་ཙྱ་ཏེ། ཨོ་ད་ནཿ རང་ཉིད་ཁོ་ན། བྷིདྷྱ་ཏེ། ཀུ་ཤཱུ་ལ། རང་ཉིད་ཁོ་ནའོ། །ལས་བྱེད་པ་པོ་ལས་བཞིན་ནོ་ཞེས་པ་ལས་བཞིན་དུ་ཡོད་པ་ཐོབ་པ་དེ་འགོག་པའོ། །དུ་ཧ་ཞེས་པ་བྱེད་པ་པོ་ལ་ཨཱཏྨ་ནེའི་ཚིག་གི་ཆོ་ག་བྱའོ། །དུགྷེ་གཽ་ཡ་ཡཱ་མེ་པ།
དེ་བཞིན་དུ་བསྟོད་པ་དང་ཕྱག་འཚལ་བ་དག་རང་ཉིད་དོ་ཞེས་པས་བྱེད་པ་པོ་ལ་ཨཱཏྨ་ནེའི་ཚིག་གོ། །པྲ་སྟུ་ཏི་གཽཿསྭ་ཡ་མེ་བ། ན་མ་ཏེ། ད་ཎྚཿ་སྭ་ཡ་ན་མེ་པ་ཀུ་ཥ། ངེས་པར་འགུགས་པ་ལའོ། །རནྫས་ཚོས་ལའོ། །ཀུ་ཥ་རི་རྫ་དག་ལ་པ་རསྨཻའི་ཚིག་ཉིད་དོ་ཞེས་བརྗོད་པར་
