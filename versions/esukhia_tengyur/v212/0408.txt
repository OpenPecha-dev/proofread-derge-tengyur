གིས་བྱས་པ་ལ། །ཨ་བྷི་མཱ་ན་སྐྱེས་བུའི་སྙེམས། །ཤེས་དང་། པྲ་ཎ་ཡ་དང་འཚེ། །འགྲོ་བྱེད་ཉི་མ་ཉི་མའི་ཟས། །ཕ་མེས་རྣམས་ཀྱི་ལྷ་ལའོ། །ཨ་བ་དཱ་ནི། ལས་དེ་དག །དེ་ལྟ་བུ་བྱུང་རྟོགས་བརྗོད་དུམ། །ལུས་དབུགས་དང་ནི་ཀུན་སྦྱོར་དང་། །ཀུན་ནས་སྦྱོར་བ་མངོན་
པར་བརྗོད། །ཨཱ་ཀ་ལ་ནི་སོམ་ཉི་དང་། །ཡོངས་སུ་བགྲང་དང་རྣམ་པར་འཆིང་། །ཨཱཙྪ་ད་ནཾ་ཁ་ཆོད་དང་། །གོས་དང་། སྒྲིབ་པར་བྱེད་པ་ལ། །ཨཱ་ཏ་ནཾ་ནི་སོ་སོར་གནས། །རྒྱུག་པར་བྱེད་དང་བགྲོད་པར་འདོད། །ཨཱཏྨ་ཡོ་ནི་རྣམ་སྤྲོ་བྱེད། །ཡིད་ལས་བྱུང་བ་མངོན་
པར་བརྗོད། །ཨཱ་བ་ཤ་ནཾ། བཟོ་ཁང་དང་། །འབྱུང་པོའི་གནས་དང་རབ་འཇུག་པ། །ཨཱ་ཡོ་དྷ་ནོ་འཁྲུག་དང་གསོད། །འཐབ་པར་བྱེད་པ་ལ་འདོད་དོ། །ཨཱ་ར་དྷ་ནཾ་མཆོད་པ་དང་། །འཚེད་དང་འཐབ་པ་དག་ལ་ཡང་། །ཨ་སྐནྡ་ནི་མི་མངོན་དང་། ཀུན་ནས་སྐེམས་དང་
སྒྲ་ལའོ། །ཨུཏྤ་ཏ་ནཾ་ཀུན་ནས་སྐྱེས། །འབྱུང་དང་གྱེན་དུ་བགྲོད་པ་ལ། །ཨུཏྤ་ད་ནི་ཀུན་ནས་བྲིས། །ལེགས་པར་འཇུག་དང་། འབྱུང་དོན་ཅན། །ཨུ་ད་ཡན་ཞེས་བྱ་བ་ལ། པཏྶའི་རྒྱལ་དང་བུམ་པ་སྐྱེས། །ཨུད་བརྟ་ནཾ་ཀུན་སྐྱེས་དང་། །རྣམ་པར་འཇུག་དང་རྣམ་པར་བྲིས། །
