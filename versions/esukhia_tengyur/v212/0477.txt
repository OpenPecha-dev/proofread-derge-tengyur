པ་ཅན་དང་མེ། །དཀྵ་ས་གཞིའོ། དྡྷཾ་ཀྵ་ནི། །བྱ་རོག་དང་ནི་ཀཾ་ཀ་དང་། །སློང་བ་པོ་དང་རྟོག་གེ་ཁྱིམ། །དྷཱུཀྵཱཀྐ་ཨི་ཀ་དང་། །ནྱ་ཀྵ་མཐའ་དག་ནག་པོ་དང་། །དགྲ་སྟ་རཱ་མ་ལ་ཡང་ངོ་། །བཀྵ་ཕྱོགས་དང་ངོས་གཅིག་དང་། །ཁྱིམ་དང་། རྩིག་པ་ཟླ་བླ་མི། །གྲོགས་པོ་

འཛིན་དང་མུན་པ་འཇིག །ཉལ་དང་། སློང་བའི་དངོས་པོ་ལ། །མོ་ཀཱ་བྱད་གྲོལ། གྲོལ་བ་ཐར། །མུཥྐ་ཀ་ཡི་རྐང་འཐུང་ལའོ། །ཡཀྵ། ལུས་ངན། གསང་བ་པ། །སྲུང་བ་སྲུང་བྱེད་མཚོན་པ་ལ། །རཱུ་ཀྵ་ཤིང་གི་ཁྱད་པར་དང་། །རྩུབ་པ་དགའ་བ་སྟོང་པའོ། །
