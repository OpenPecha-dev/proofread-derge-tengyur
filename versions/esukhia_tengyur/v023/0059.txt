ནི་ཡི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏུརྦྷ་གཱི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏུརྦྷ་གི་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཱ་ཙ་མ་ཏུ་ལ་སྱ་བྷཱརྱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཱ་ཙ་མཏུ་ལ་སྱ་བྷཱརྱཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱཱ་བྷ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱཱ་བྷ་གི་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་
ཏཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་ཏ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་སྱཻ་བ་པི་ཏུརྨཱ་ཏྲི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་སྱཻ་བ་པི་ཏུརྨཱ་ཏྲི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱ་པི་ཏ་མ་ཧཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱཱ་པི་ཏ་མ་ཧ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཏུརྨཱ་ཏཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཏུརྨཱ་ཏ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
བནྡྷ་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བནྡྷ་བ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་ཏུརྦྷ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་ཏུརྦྷ་གི་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷ་གི་ནྱ་ཡེ་ཀཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷ་གི་ནྱ་ཡི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྭ་མཱ་ཏུ་རྨཱ་ཏ་བྷ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྭ་མཱ་ཏུ་རྨ་ཏ་བྷ་གི་ན་ཧུཾ་ཧཱུཾ་
ཕཊ། ཨོཾ་བཛྲ་བྷ་གི་ནྱ་ཡཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སྱ་པུ་ཏྲི་ཀཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་སྱ་པུ་ཏྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏུ་རྨཱ་ཏཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏུ་རྨཱ་ཏ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏཱ་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏཱ་མ་ཧཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་པི་ཏུ་ལཱ་སྱ་བྷཱརྱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པི་ཏུ་ལཱ་སྱ་བྷཱརྱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱ་པུ་ཏྲ་བྷཱ་རྱཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱ་པུ་ཏྲ་བྷཱཪྻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱ་ཡ་བྷ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱརྱ་ཡ་བྷ་གི་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྭཿཔི་ཏུརྦྷ་གི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྭཿཔི་ཏུརྦྷ་གི་
ནི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་སྱཻ་བ་ཏུ་སྭ་རྒོ་ཏྲ་ཛཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་སྱཻ་བ་ཏུ་སྭ་རྒོ་ཏྲ་ཛ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱ་ཏཱ་ཡ་བྷཱརྱཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱ་ཏཱ་ཡ་བྷཱརྱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་སྱཻ་བ་ཏུ་
བྷཱརྱ་ཀཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་སྱཻ་བ་ཏུ་བྷཱརྱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱ་ཡ་བྷཱརྟི་མཱ་ཏུཿཔུ་ཏྲ་སྱཻ་བ་ཏུ་སྭ་སྲིཀྐཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱ་ཡ་བྷཱརྟྲི་མཱ་ཏུཿཔུ་ཏྲ་སྱཻ་བ་ཏུ་སྭ་སྲིཀྐ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུ་ཧི་ཏཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུ་ཏྲ་ཧུཾ་ཧཱུཾ་ཕཊ།། །།
