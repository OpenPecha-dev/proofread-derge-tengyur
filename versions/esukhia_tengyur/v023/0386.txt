བསྒྱུར་བའོ།། {D1546}༄༅༅། །རྒྱ་གར་སྐད་དུ། ཤྲཱི་བཛྲ་ཡོ་གི་ནཱི་མནྟྲ་ཏཏྟྭ། ཨརྠ་ནིར་དེ་ཤ་སླ་ཨ་དྷིཥྛཱ་ན་ནཱ་མ། བོད་སྐད་དུ། དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྔགས་ཀྱི་དེ་ཁོ་ན་ཉིད་ཀྱིས་བྱིན་གྱིས་བརླབས་པའི་རིམ་པ་ཞེས་བྱ་བ། དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་ཕྱག་འཚལ་ལོ། །སྔགས་

པའི་གོ་འཕང་འགྲོ། །བླ་ན་མེད་པའི་ལམ་སྤྱད་དེ། །ཡང་དག་རྫོགས་སངས་རྒྱས་ཀྱིས་གསུངས། །ས་ཡིག་རི་བོང་ཅན་གཟུགས་གནས། །ཐིག་ལེ་ལྷན་སྐྱེས་མི་འཇིགས་པ། །དེ་ནི་སྲོག་གིས་རྣམ་པར་ཤེས། །བླ་མ་དམ་པའི་མན་ངག་གིས། །མཛོད་སྤུ་ཆེན་པོའི་སྒོ་རུ་ནི། །སྣ་ཚོགས་ཐིག་ལེ་
