འཁོར་རོ། །དེའི་སྟེང་དུ་ཨོཾ་ཧ་བཾ་ཧཱ་བཱཾ། ཧི་བི་ཧཱི་བཱི། ཧུ་བུ་ཧཱུ་བཱུ། ཧེ་བེ་ཧཻ་བཻ་ཧོ་བོ་ཧཽ་བཽ་ཧཾ་བཾ་ཧཿབཿཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ཁ་དོག་དཀར་དམར་གཡོན་སྐོར་དུ་སྦྲུལ་འཁྱིལ་བ་ལྟར་གནས་པ་རྡོ་རྗེ་སེམས་དཔའ་དང་ཕག་མོའི་ངོ་བོ། །མགོ་བོར་དེ་བཞིན་དུ་ཉི་མ་ལ་
ཨོཾ་ན་ཧཾ་ནཱ་ཧཱཾ་ནི་ཧིཾ་ནཱི་ཧཱིཾ་ནུ་ཧུཾ་ནཱུ་ཧཱུཾ་ནེ་ཧཻཾ་ནཻ་ཧཻཾ་ནོ་ཧོཾ་ནོ་ཧཽཾ་ན་ཧཾ་ནཿཧཿཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་སྭཱ་ཧཱ། སེར་ཞིང་སྔོ་བས་རྣམ་པར་སྣང་མཛད་དོ། །གཤིན་རྗེ་མའི་རང་བཞིན་ནོ། །སྤྱི་གཙུག་ཏུ་ཉི་མ་ལ་ཨོཾ་སྭ་ཧྲཾ་སྭཱ་ཧྲཱཾ་སྭི་ཧྲིཾ་སྭཱི་ཧྲཱིཾ་སྭུ་ཧྲུཾ་སཱུཾ་ཧྲུཾ་སྭེ་ཧྲེཾ་སྭཻ་ཧྲེེཾ་སྭོ་ཧྲོཾ་སྭཽ་ཧྲཽཾ་སྭ་ཧྲཾ་སྭཿཧྲཱཿཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་
ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་དམར་ཞིང་དཀར་བ་པདྨ་གར་གྱི་དབང་ཕྱུག་དང་རྨོངས་བྱེད་མའི་ངོ་བོའོ། །དཔུང་པ་གཉིས་ཀྱི་ནང་དུ་ཉི་མ་ལ། ཨོཾ་བ་ཧྲཾ་བཱ་ཧྲཱཾ་བི་ཧྲིཾ་བཱི་ཧྲཱིཾ་བུ་ཧྲུཾ་བཱུ་ཧྲཱུཾ་བེ་ཧྲཻཾ་བཻ་ཧྲཻཾ་བོ་ཧྲོཾ་བཽ་ཧྲཽཾ་བ་ཧྲ་བཿཧྲཿཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་གནག་ཅིང་སེར་བ་ཧེ་
རུ་ཀ་དང་སྐྲོད་བྱེད་མའོ། །སྤྱན་གཉིས་པ་ལ་ཉི་མའི་སྟེང་དུ། ཨོཾ་ཧཱུཾ་ཧཱུཾ་ཧཾ་ཧཾ་ཧཱཾ་ཧཱཾ། ཧིཾ་ཧིཾ་ཧཱིཾ་ཧཱིཾ། ཧཱུཾ་ཧཱུཾ་ཧེཾ་ཧེཾ་ཧཻཾ་ཧཻཾ་ཧོཾ་ཧོཾ་ཧཽཾ་ཧཽཾ་ཧཾ་ཧཾ་ཧཿཧཿཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཕཊ་ཕཊ་སྭཱ་ཧཱ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་རྣམས་དམར་ཞིང་ལྗང་བ་རྡོ་རྗེ་ཉི་མ་དང་སྐྲག་བྱེད་མའོ། །མཛོད་སྤུའི་ཐད་ཀའི་ནང་དུ་ཉི་མ་ལ་

ཕྱི་རོལ་དུ་ཨོཾ་ཧ་དཀར་པོ་སྙིང་ག་དང་། ན་མ་ཧི་སེར་པོ་མགོ་བོ་ལ་དང་། སྭཱ་ཧཱ་ཧུ་དམར་པོ་སྤྱི་གཙུག་ཏུ་དང་། བཽ་ཥཌ་ཧེ་ནག་པོ་དཔུང་པ་གཉིས་ལ་དང་། ཧཱུཾ་ཧཱུཾ་ཧོ་དམར་སེར་མིག་གཉིས་ལ་དང་། ཡན་ལག་ཐམས་ཅད་ལ་ཕཊ་ཧཾ་ལྗང་གུའོ། །རྡོ་རྗེ་ཕག་མོའི་ལྟེ་བར་ཨོཾ་བཾ་
དམར་པོ་དང་། ཧཾ་ཡོཾ་སྔོན་མོ་སྙིང་གར་རོ། །ཧྲིཾ་མོཾ་དཀར་མོ་ཁ་ལའོ། །ཧྲེཾ་ཧྲིཾ་སེར་མོ་མགོ་བོ་ལའོ། །ཧཱུཾ་ཧཱུཾ་ལྗང་གུ་སྤྱི་གཙུག་ཏུའོ། །ཕཊ་ཕཊ་དུད་ཀ་སྨིན་མཚམས་སུ་བཀོད་དེ། དེ་ལས་འཕྲོས་པའི་ལུས་ཐམས་ཅད་དགང་བའོ། །དེ་ནས་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་དང་གཉིས་
