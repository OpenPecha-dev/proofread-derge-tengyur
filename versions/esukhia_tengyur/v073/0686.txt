མཆོག་ཏུ་སེམས་བསྐྱེད་དོ། །དེ་ནས་སྟོང་པའི་ཏིང་འཛིན་ལས། །བདག་ཉིད་ཁྲོ་བོའི་སྐུར་བསྐྱེད་ནས། །སྤྱན་དྲངས་བྱིན་རླབས་རྫོགས་པ་ཡིས། །མདུན་དུ་མཎྜལ་རིན་ཆེན་གནས། །པདྨ་འདབ་བཞིར་བཅས་པ་ཡི། །སྟེང་དུ་བུམ་པ་ལྔ་བསྐྱེད་ནས། །ནང་དུ་ཟླ་སྟེང་ས་བོན་ནི། །ཧཱུཾ་དྷཱིཿཏྲཱཾ་རཱཾ་ཏཱཾ་ངོ་། ན་
མཤྩཎྜ་བཛྲ་ཀྲོདྷཱ་ཡ་ཧཱུཾ། ཧུ་ལུ་ཧུ་ལུ། ཏིཥྛ་ཏིཥྛ། བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན། ཨ་མྲྀ་ཏེ་ཧཱུཾ་ཕཊ། བུམ་པ་བཞི་ལ་སོ་སོ་སྟེ། །དེ་ལྟར་བསྐྱེད་ནས་སངས་རྒྱས་བསྟིམ། །བསྒྲུབ་པའི་རིག་སྔགས་འདི་བཟླས་སོ། །ན་མཤྩཎྜ་བཛྲ་ཀྲོ་དྷཱ་ཡ། ཧུ་ལུ་ཧུ་ལུ། ཏིཥྛ་ཏིཥྛ། བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན། ཨ་མྲྀ་ཏེ་ཧཱུཾ་

དཀྱིལ་འཁོར་ལ་ནི་འཇུག་པའི་ཕྱིར། །ལུས་ནི་བྱིན་གྱིས་བརླབ་པར་བྱའོ། །མ་ཏི་སྠི་ར་བཛྲ། སུ་ཏི་སྠི་ར་བཛྲ། ཨོཾ་ཨཱཿཧཱུཾ། མ་ཧཱ་བཛྲ། ཨ་པྲ་ཏི་ཧ་ཏ་བཛྲ། ཨ་མོ་གྷ་བཛྲ། ཨེ་ཧི་བཛྲ། ཤཱི་གྷྲཾ་བཛྲ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། རྡོ་རྗེ་འཁོར་ལོ་པདྨའི་སྟེང་། །འདུག་སྟེ་ཕྲེང་བ་དོར་བར་བྱའོ། །དྷ་ར་དྷ་ར། དྷི་རི་
