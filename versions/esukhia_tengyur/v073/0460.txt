བྱས་པ། །ཡང་དག་སྐྱེས་ལས་རྩེ་ལྔ་པའི། །རྡོ་རྗེ་ལྟེ་བར་ཧཱུཾ་གིས་མཚན། །སོ་སོར་ཀུན་རྟོག་ཡེ་ཤེས་བསྒོམ། །ཧཱུཾ་ལས་སངས་རྒྱས་སྤྲོ་བྱ་སྟེ། །བསྡུས་ཏེ་དེ་ལ་དེ་རྣམས་བཞུགས། །ཡང་དག་འདུས་ནས་དབྱེར་མེད་པ། །དེ་ནི་བྱ་བ་གྲུབ་པར་བསྒོམ། །ཟླ་བ་ཉིས་མཚན་ཧཱུཾ་ཡིག་ལས། །སྐྱེས་པའི་རང་
ཉིད་གོས་སྔོན་ཅན། །རྣམ་དག་ཡེ་ཤེས་ངོ་བོ་ཉིད། །ཡུན་རིང་ཕྱག་ན་རྡོ་རྗེ་བསམ། །དབུ་སྐྲ་ཁམ་སེར་གྱེན་དུ་འགྲེང་། །ཁྲོ་གཉེར་སྤྱན་ནི་དམར་བ་སྟེ། །མཆེ་བ་གཙིགས་པ་ཡི་ནི་ཞལ། །གློག་གི་ལྗགས་ཀྱིས་འཇིགས་བྱེད་པ། །འཇིག་དུས་རླུང་གི་དབུགས་དང་ནི། །ཧཱུཾ་ཡིག་འདོན་པ་ཡི་ནི་ཞལ། །
ཧ་ཧར་དགོད་པ་རྩུབ་ཅིང་དྲག །རྒྱ་མཚོ་ཆེན་པོ་ལྟོ་ཆེ་བ། །པདྨ་དབུ་ལ་རྒྱས་པར་རོ། །དགའ་བོ་ཉེར་དགའ་རྣ་ཆ་སྟེ། །པདྨ་ཆེན་པོ་ཚངས་སྐུད་དོ། །དུང་སྐྱོང་ཏཀྵ་ལག་གདུབ་བོ། །རིགས་ལྡན་ཀརྐོ་དབུ་རྒྱན་ཏེ། །ལྷག་ཅན་ནོར་རྒྱས་རྐང་གདུབ་བོ། །ཕྲག་པ་ཟླ་བ་ཉི་མས་བརྒྱན། །སེང་གེ་རྐང་བརྒྱད་
པགས་པའི་གོས། །རྡོ་རྗེ་ལྕགས་སྒྲོག་རྐེད་པར་དཀྲིས། །བྲང་འགྲོ་ཆེན་པོ་གསོལ་བས་སོ། །རང་མདོག་རྡོ་རྗེ་རྩེ་གསུམ་ནི། །གཡས་ཀྱི་ཕྱག་ན་བསྣམས་པ་སྟེ། །རྡོ་རྗེ་དབྱུག་ཆེན་མི་བཟད་པ། །གཡོན་གྱི་ཕྱག་ན་བསྣམས་པའོ། །ཉི་མའི་གདན་དང་རྒྱབ་རྟེན་བྲག །གཡོན་བརྐྱང་པ་ཡི་ཞབས་ཀྱིས་
གནས། །དེ་ནས་དབྱུག་པའི་སྟེང་བསམ་པ། །མཁའ་ལ་འགྲོ་བདག་བྱ་ཁྱུང་སྟེ། །ཁྲོས་ཞིང་ལུས་ནི་ཆེ་ལ་དཔའ། །ཁ་གདངས་སྣ་ཚོགས་མདོག་ཅན་ནོ། །རྐང་པ་གཉིས་ཀྱིས་སྦྲུལ་རྒྱལ་པོ། །ཟ་བཞིན་པ་ནི་ཉིད་དུ་བྱེད། །ཟླུམ་ཞིང་དམར་ལ་འབར་བའི་སྤྱན། །འཇིགས་ཤིང་ཀུན་ཏུ་སེང་
སེང་ལྟ། །དེ་ནས་སྙིང་གར་ཉིས་མཚན་ཧཱུཾ། །དེ་ཉིད་འོད་ཟེར་འཕྲོ་བའོ། །དེ་བཞིན་སྤྱི་བོ་མགྲིན་པ་སྙིང་། །ཨོཾ་ཨཱཿཧཱུཾ་ཡོངས་བསྒོམ་པར་བྱ། །ཞེས་བྱ་བ་ནི་སྦྱོར་བ་དང་པོ་ཞེས་བྱ་བའི་ཏིང་ངེ་འཛིན་ཏོ། །དེ་ནས་སྙིང་གར་ཉིས་གནས་པའི། །ས་བོན་ལས་བྱུང་ཟླ་བ་ནི། །འཇིགས་བྱེད་ལ་སོགས་
བརྒྱད་ལ་དགོད། །དེ་རྣམས་ཉི་གནས་དེ་བཞིན་ནོ། །དེ་ལ་ཟླ་ཉི་སྦྱོར་བ་ལས། །དེ་བཞིན་ས་བོན་བརྒྱད་དུ་གྱུར། །ཧཱུཾ་ཧཱུཾ་ཕུཾ་ཀུཉྫ་ཤཾ་རཾ་མཾ། །ཐིག་ལེ་ནཱ་ད་ལྡན་རིམ་བཞིན། །དེ་རྣམས་ཡོངས་སུ་གྱུར་པ་ལས། །མཚན་མ་བརྒྱད་ནི་{𰷓གོ་རིམས་,གོ་རིམ་}བཞིན། །འོག་ནས་འབྱུང་ཞིང་སྒྲུབ་པ་པོས། །རང་
