བྱ་བ་ནི་ཆོས་ཀྱི་དབྱིངས་རབ་ཀྱི་མཐར་ཐུག་པ་འགྲོ་བས་གནས་སོ། །གསལ་ཞེས་བྱ་བ་ནི་རྣལ་འབྱོར་པའི་རྒྱུད་ལ་གསལ་བ་ཉིད་དུ་སྣང་བའི་ཕྱིར་རོ། །དེ་ལྟར་དོན་རྣམ་པ་བཞིས་གསང་བའི་རྗེས་སུ་འབྲངས་ནས། དེའི་དབྱེ་བ་བརྗོད་པའོ། །དེ་ལྟར་རྣམ་པ་གཉིས་ཐུན་མོང་བའི་
ཕྱག་རྒྱའི་དབྱེ་བ་བསྟན་ནས། ད་ནི་རིམ་པ་གཞན་གྱིས་དེ་བརྗོད་པའི་ཕྱིར། དེའི་ཡན་ལག་བཤད་པར་དམ་བཅའ་བ་ནི། དེ་ནས་གཞན་ཡང་ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། སྔོན་ལ་ལྟོས་པས་ཐ་དད་པའི་རིམ་པ་གཞན་གྱིས་སོ། །ཕྱག་རྒྱའི་དབྱེ་བ་ཞེས་བྱ་བ་ནི་འཆད་བཞིན་པའི་མཚན་ཉིད་དོ། །བཤད་
བྱ་བ་ཞེས་བྱ་བ་ནི་བརྗོད་པར་བྱ་ཞེས་བྱ་བའི་དོན་ཏོ། །བླ་ན་མེད་ཅེས་བྱ་བ་ནི་བླ་ན་མེད་པའི་དེ་ཁོ་ན་ཉིད་ཀྱི་རྒྱུར་གྱུར་པའི་ཕྱིར་རོ། །དེ་ལྟར་བཤད་པར་དམ་བཅའ་བ་བསྟན་ནས། གྲངས་ཀྱི་ཆོ་ག་བརྗོད་པའི་ཕྱིར། རིགས་དང་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། རིགས་ནི་དྲུག་སྟེ། ཕ་རོལ་ཏུ་ཕྱིན་
པ་སྦྱིན་པ་ལ་སོགས་པ་ཡང་དེ་བཞིན་ནོ། །རིགས་ཀྱང་ཡིན་ལ་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཡང་ཡིན་པས་ན། རིགས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །དེ་རྣམས་ཀྱི་དབྱེ་བ་ནི་ཁྱད་པར་རོ། །ཕྱག་རྒྱ་དྲུག་ཏུ་ཞེས་བྱ་བ་ནི་འཆད་བཞིན་པ་ལས་སོ། །ཡང་དག་བཤད་ཅེས་བྱ་བ་ནི་བརྗོད་པའོ། །དེ་ལྟར་གྲངས་ཀྱི་ཆོ་ག་
བརྗོད་ནས། ད་ནི་དེ་རྣམས་སྔགས་ཀྱི་སྒོ་ནས་མཆོད་པར་བྱ་བ་ནི། ཡཾ་རཾ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། ལྷ་མོ་རྣམས་ཞེས་བྱ་བ་ནི་འཆད་བཞིན་པའི་གཏུམ་མོ་ལ་སོགས་པ་རྣམས་ལའོ། །རིམ་པ་ལས་ཞེས་བྱ་བ་ནི་འཆད་པར་འགྱུར་བའི་རིམ་པས་སོ། །བྱ་ཞེས་བྱ་བ་ནི་བྱ་བའོ། །མཆོད་པར་
བྱ་ཞེས་བྱ་བ་ནི་ཁྱད་པར་དུ་འཕགས་པའི་ཉེ་བར་མཁོ་བ་ལ་སོགས་པའོ། །རྟག་ཏུ་ཞེས་བྱ་བ་ནི་ཚོགས་ཀྱི་འཁོར་ལོ་ལ་སོགས་པ་ལ་དུས་རྟག་ཏུའོ། །གང་དུ་ཞེ་ན། དེའི་ཕྱིར། མཆོད་པའི་རིམ་པ་རོ་གཅིག་པས། །ཞེས་བྱ་བ་གསུངས་ཏེ། མཆོད་བྱ་དང་མཆོད་བྱེད་དང་མཆོད་པ་རྣམས་ལ་ཐ་
དད་པའི་བློ་དོར་བས་སོ། །མཆོད་པ་ལ་རོ་གཅིག་པའི་རིམ་པ་ནི། མཆོད་པའི་རིམ་པར་རོ། །གཅིག་པ་སྟེ་དེ་ལས་སོ། །དེ་ལྟར་དེ་རྣམས་ཀྱི་སྔགས་ཀྱི་སྒོ་ནས་མཆོད་པ་བསྟན་ནས། ད་ནི་ཕྱག་རྒྱའི་ངོ་བོ་ཉིད་བརྗོད་པའི་ཕྱིར། གདོལ་པ་མོ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། ཁྲུས་མཁན་མོ་
