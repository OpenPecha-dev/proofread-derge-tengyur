ཨོཾ་ཤྲཱི་དུདྡྷ་ས་ཏྱ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་དྷརྨ་ས་ཏྱ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སཾ་གྷ་ས་ཏྱ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་གུརྦི་ཎཱི་སུ་ཁེ་ན་པྲ་སཱུ་ཏཱ་མ་ཧཱ་ཏེ་ཛཱཿཧྲཱིཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མ་ཧཱ་ཤཱནྟཾ་མ་ཏི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མ་ཧཱ་པཽཥྚི་མ་ཏི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ཛ་ན་བ་ཤཾ་ཀ་རི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་དུཥྚཱ་ནི་ཀྲྀནྟ་ནི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་དུཥྚ་ཤ་ཏྲུ་
བི་ནཱ་ཤ་ནི་སྭཱ་ཧཱ། ཧཱུཾ་ཕཊ་ཨོཾ་ཨཱཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨཱ་གཙྪ་ཨཱ་གཙྪ་ས་མ་ཡ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨཱ་བ་ར་ཎ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨཱ་དྷཱ་ར་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བྷཱུ་ཏི་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཛ་ཡ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བི་ཛ་ཡ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་སཏྭ་ས་མ་ཡ་མ་ནུ་སྨ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུནྡྷ་
རེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ལཀྵྨཱི་བྷཱུ་ཏ་ལ་ནི་བཱ་སི་ནཱི་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུནྡྷ་རི་སྭཱ་ཧཱ། ཨོཾ་བ་སུ་དྷཱ་རི་ཎི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུ་དྷཱ་རི་ཎི་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཤྲཱི་ཀ་ར་དྷ་ནི་ཀ་རི་དྷཱ་ནྱ་ཀ་རི་སྭཱ་ཧཱ། ཨཱ་གཙྪ་ཨཱ་གཙྪ་ཤྲཱི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨུཀྐེ་ཏུཀྐེ་བྷུཀྐེ་བྷུཀྐེ་བཪྵ་ཎི་པྲ་བཪྵ་ཎི་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུ་དྷེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུ་དྷཱ་དཱིཔྟ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་
ས་མ་ཡེ་སཽ་{𰵞མྱེ་,མེ་}ས་མ་ཡངྐ་རི་མ་ཧཱ་ས་མ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བ་སུ་དྷཱ་ར་ཨེ་ཧྱེ་ཧི་བྷ་ག་བ་ཏི་ས་མ་ཡེ་ས་མ་ཡ་མ་ནུ་སྨ་ར་སིདྡྷིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ལོ་ཀ་པཱ་ལཱ་ཡ་ན་མཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་མཱ་ཏ་རཱི་བྷྱོ་ན་མཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ཤྲིཾ་གཱ་རཱི་བྷྱོ་ན་མཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་བྷཱུ་ཏི་ནི་བྷྱོ་ན་མཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་སརྦ་ཡཀྵི་ཎཱི་བྷྱོ་ན་མཿསྭཱ་ཧཱ།
ཨོཾ་ཤྲཱི་གུ་ཧྱ་ས་ཀ་ཊི་ཀེ་སརྦ་དུཥྚཱ་ནཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱིཿཧཱུཾ་ཙཾ་སཿསྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཛམྦྷ་ལ་ཛ་ལེནྡྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨཱཪྻཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མ་ཎི་པདྨེ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བཛྲ་དྷ་རཱ་ཡ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མ་ཎི་བྷ་དྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་པཱུརྞྞ་བྷ་དྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་དྷ་ན་དཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་བཻ་
ཤྲ་བ་ཎཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཙི་པི་ཀུཎྜ་ལི་ནེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཀེ་ལི་མཱ་ལི་ནེ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་མུ་ཁེནྡྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཙ་རེནྡྲཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཕུཿཤཾ་ཁ་པཱ་ལ་ནཱ་ག་རཱ་ཛཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཨི་ལཱ་དེ་བྱཻ་སྭཱ་ཧཱ། ཨོཾ་མ་ཎི་པདྨེ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷ་ར་སཱ་ག་ར་ནིརྒྷོ་ཥཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷརྨེ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཏ་བྷ་དེ་བྱཻ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་གུཔྟ་
དེ་བྱཻ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ས་ར་སྭ་ཏཱི་དེ་བྱཻ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཙནྡྲ་ཀཱནྟཱ་དེ་བྱཻ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཤཾ་ཁ་ནི་དྷཱ་ནཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་པདྨ་ནེ་དྷཱ་ནཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཤྲཱི་ཨཥྚཽ་མ་ཧཱ་ཡཀྵི་ཎཱི་བྷྱོ་ན་མཿསྭཱ་ཧཱ། ཐམས་ཅད་མཆོད་པའི་དོན་དུ་ཧཱུཾ་སྭཱ་ཧཱའོ། །ཨོཾ་བཅོམ་ལྡན་འདས་མ་དཔལ་ནོར་རྒྱུན་མ་ཡེ་ཤེས་ཀྱི་སྐུ་མངའ་བ་ལ་ཕྱག་འཚལ་
