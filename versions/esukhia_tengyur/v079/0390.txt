རྫས་ཀྱི་ཁྱད་པར་བསམ་གྱིས་མི་ཁྱབ་པ་དང་། སེམས་ཅན་ཐམས་ཅད་ཀྱི་མོས་པ་ཇི་ལྟ་བ་བཞིན་ནམ་མཁའི་ཁམས་འདོད་པའི་ཡོན་ཏན་ལྔས་གང་བར་བྱིན་གྱིས་བརླབ་བོ། །དེ་ལྟར་སྤེལ་ནས་གསུམ་པས་ནི་བདུད་རྩིའི་རྒྱ་མཚོ་དང་། ཟས་ལ་སོགས་པ་ལོངས་སྤྱོད་ལ་ཇི་ལྟར་སོ་སོར་མོས་པ་བཞིན་དུ་
ལོངས་སྤྱོད་ཀྱི་རྐྱེན་དུག་ལྔའི་གསོས་དང་། ཡེ་ཤེས་ལྔར་བྱིན་གྱིས་བརླབ་བོ། །བཞི་པས་ནི་འཕགས་པ་རྣམས་ཀྱི་སྤྱན་ལམ་དུ་ཡང་གཞལ་མེད་ཁང་བརྩེགས་མ་ལ་སོགས་པ་མཆོད་པའི་སྤྲིན་ཕུང་མང་པོར་སྤྲུལ་ནས་མཆོད་པ་དང་། སེམས་ཅན་རྣམས་ཀྱི་དབང་པོའི་སྤྱོད་ཡུལ་དུ་ཇི་ལྟར་འདོད་པ་
བཞིན་འདོད་པའི་ཡོན་ཏན་ལྔ་ཆར་བཞིན་དུ་འབྱུང་བར་བསམ་ཞིང་། ཁྱད་པར་དུ་ཡང་{𰶜ཡི་དགས་,ཡི་དྭགས་}ཀྱི་རྣམ་པ་མང་པོ་དང་སེམས་ཅན་ཀུན་གྱི་བསམ་པ་ཡོངས་སུ་སྐོང་བར་བྱིན་གྱིས་བརླབ་བོ། །ལྔ་པས་ནི་བདག་གི་ལུས་ཞིང་གི་རྡུལ་སྙེད་དུ་སྤྲུལ་ནས་འཕགས་པ་རྣམས་ལ་མཆོད་པ་ཕུལ་ནས་མཉེས་པ་
དང་། སེམས་ཅན་ལ་བྱིན་པས་ཕམ་རྒྱལ་མེད་པ་དང་འཐབ་རྩོད་མེད་པ་སོ་སོའི་མོས་པ་བཞིན་ཐོབ་པར་བྱིན་གྱིས་བརླབ་བོ། །དྲུག་པས་ནི་སྲི་ཞུའི་མགྲོན་དང་ཡོན་ཏན་གྱི་མགྲོན་དང་། སྙིང་རྗེའི་མགྲོན་དང་། ལན་ཆགས་ཀྱི་མགྲོན་དུ་གྱུར་པའི་རིགས་ལ་ཡང་དམ་ཚིག་གི་དབང་གིས་འགོ་
ཞིང་སྟོང་གྲོགས་སུ་གྱུར་པར་མངོན་སུམ་དུ་དམིགས་སོ། །སེམས་ཅན་གྱི་དོན་དུ་འགྱུར་བའི་ལས་མི་འགལ་བར་ཅི་བཅོལ་བ་ཡང་འགྲུབ་སྟེ་ཕྱག་རྒྱ་ཡང་ཕྱོགས་བཅུར་བསྟན་ལ་དབང་དུ་བྱའོ། །དེ་ནས་ཚིགས་སུ་བཅད་པ་འདི་དག་གིས་ཀྱང་བསྔོ་བར་བྱའོ། །གཏན་པ་མེད་པའི་མཆོད་སྦྱིན་རྒྱལ་པོ་
འདི། །ཆོས་ཉིད་རྣམ་པར་དག་པའི་བྱིན་རླབས་དང་། །དཀོན་མཆོག་ཡོན་ཏན་དཔག་མེད་ཆེ་བ་དང་། །སྙིང་པོ་ཕྱག་རྒྱ་ངེས་པའི་དམ་ཚིག་དང་། །རྒྱུ་རྐྱེན་རྟེན་འབྲེལ་ཚོགས་པའི་ནུས་མཐུ་དང་། །དྲང་སྲོང་རིག་འཛིན་རྣམས་ཀྱི་གྲུབ་པ་དང་། །བདག་གི་སྨོན་ལམ་ཏིང་འཛིན་ཤེས་རབ་ཀྱིས། །ཡེ་ཤེས་
ལྔ་ཡི་འོད་ཟེར་རྣམ་པར་འཕྲོ། །དབང་པོ་ལྔ་ཡི་ལོངས་སྤྱོད་མཁོ་དགུར་ལྡན། །ནམ་མཁའི་མཐའ་ཀླས་གཏེར་ཆེན་གྱུར་པ་འདི། །མཆོད་གནས་བླ་མ་ཁྱེད་ཀྱི་ཕོ་བྲང་དུ། །མཆོད་པའི་སྤྲིན་གྱིས་རྒྱས་ཤིང་འཕེལ་བར་བསྔོ། །སེམས་ཅན་ཐམས་ཅད་བདེ་གྱུར་ཅིག །ཀུན་ཀྱང་གནོད་པ་
