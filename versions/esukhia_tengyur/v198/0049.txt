དང་། ཏུ་ལི་སི་དང་། ཀྲྀཥྞ་དང་། བག་ཀྲ་པརྞི་དང་། སུ་མཉྫ་རི་དང་། བྷུ་ཏ་བྲི་ཡ་དང་། ན་ས་ག་ན་ས་དང་། ཀ་པིད་ཚ་དང་། སུར་བི་ར་ས་དང་། དེ་དག་ནི་བྱེའུ་རུག་པ་ནག་པོའི་མིང་ཡིན་ནོ། །སུ་ར་སུ་དང་། གཽ་རི་དང་། ཤྭེ་ཏ་དང་། སུ་ར་བ་མཉྫ་རི་དང་། བྲི་ནན་ཛ་
རི་དང་། བུ་ཏ་པ་ཏི་དང་། རུ་ཏུ་ལ་སི་དང་། དེ་དག་ནི་བྱེའུ་རུག་པ་དཀར་པོའི་མིང་དུ་བཤད་པའོ། །མ་ནི་ཛ་ཀ་དང་། ཁ་ར་བཏྟ་དང་། པ་རུ་རུ་ཀ་དང་། ཛམྦི་ར་ཀ་དང་། མུ་རུ་བ་ཀ་དང་། ཏིཀྵྞ་གནྡྷ་དང་། སུ་གནྡྷ་ཀྲི་དང་། དེ་དག་ནི་ཕ་ནིད་ཛའི་མིང་ངོ་། །ཀ་ལ་ད་མ་ལ་དང་། ཀ་ར་ལ་དང་། མ་
ལུ་ཀ་དང་། ཤ་ཀྲྀཥྞ་མ་ལ་ཀ་དང་། ཀྲི་ཤནྣ་ཛ་ཀ་དང་། ཏིཀྵྞ་ག་ད་དང་། ཀུ་ཐེ་ར་དང་། ཀྲྀཥྞ་མ་ལུ་ཀ་དང་། དེ་དག་ནི་ཟིམ་ཐིག་ལེ་ནག་པོའི་མིང་དུ་བཤད་པའོ། །ཀུ་ཐི་ར་ཀ་དང་། བེ་ཀུནྠ་དང་། པརྞ་སུ་དང་། གནྡྷ་བྷ་དྲ་ཀ་དང་། ཨརྫ་ཀ་དང་། བིལ་བ་གནྡི་ཛ་དང་། ཀྵུ་ཏྲ་པ་ཏྲ་དང་། བྲ་
ད་ཚད་དང་། དེ་དག་ནི་ཞིམ་ཐིག་ལེའི་མིང་ངོ་། །མ་རུ་བ་ཀ་དང་། ཁ་ར་པུ་ཤ་དང་། བེ་ར་པ་ཏྲ་དང་། མཱུ་ལེ་མ་ཀ་དང་། སུག་སྨེ་ཧི་ན་དང་། སུ་གནྡྷི་ཙི་དང་། དི་བ་ནི་དང་། གི་ནི་ན་ཤན་དང་། དེ་དག་ནི་མ་རུ་བ་ཀའི་མིང་ངོ་། །བྲི་ཤ་ཀརྞ་དང་། གྲི་མི་ཧ་ར་དང་། མུ་ལི་ཀ་ཧ་དང་། ས་བ་
ནི་དང་། ལི་ཀ་དང་། ཨུནྡུ་ར་ཀརྞ་ཀ་དང་། དེ་དག་ནི་མུ་ཤཀ་ཀརྞིའི་མིང་ངོ་། །ཀྵ་པ་ཀ་དང་། ཏིཀྵྞ་ན་དང་། ཀྲུར་དང་། ཨུ་དྭ་ཛ་ན་དང་། ཀྵོད་ཏི་ཛན་དང་། ར་ཛ་ཀྵོ་བ་དང་། ཀྵོ་ཏྲ་བོད་ན་དང་། དེ་དག་ནི་ཀྵ་པ་ཀའི་མིང་ངོ་། །ཀ་པིད་ཏ་པ་ཏྲ་དང་། བ་རི་ཤི་དང་། གིར་ཛ་ར་དང་། ཙེ་ཏ་པ་ཏྲ་དང་།
བི་རུ་པ་དང་། ཀུ་ལ་ཛ་དང་། བ་ན་ཛ་ན་དང་། པ་ཤྭ་པ་ཏྲི་ཀ་དང་། དེ་དག་ནི་ཤིང་ཀ་པིད་པའི་ལོ་མའི་མིང་ངོ་། །པྲ་ཙི་བ་ལ་དང་། ན་ཛི་ཀཎྛ་དང་། ཛ་ལ་ཛ་དང་། རཀྟ་མཉྫ་རི་དང་། ནིམྣ་ཛ་རུ་དང་། ཛ་ཡ་ད་ནི་དི་ཀ་མུ་ཁ་དང་། དེ་དག་ནི་རཀྟ་མཉྫ་རིའི་མིང་ངོ་། །ཀ་ཀ་མ་ཛི་དང་། བ་ཧུ་བ་ལ་དང་།
ཀ་མ་ཏ་དང་། ཛ་ག་གི་ཕ་ལ་དང་། ད་ཀྵ་མ་ཙི་དང་། གུ་ད་པ་ལ་དང་། ཀུཤྣ་ག་ནི་དང་། བ་ཡུ་སེ་དེ་དག་ནི་མོན་སྣེའུ་མེ་རུའི་མིང་ཡིན་ནོ། །ཀུ་ལ་ཧ་ལ་དང་། ལ་བུ་ས་དང་། གུད་ཚལ་དང་། ཁ་ཤ་ཁ་དང་། ནི་ཙ་དམྤ་དང་། བེཏ་དང་། སྟ་བིར་དང་། བུ་ཀ་ཏམྤ་དང་། དེ་དག་ནི་ཀུ་ལའི་
