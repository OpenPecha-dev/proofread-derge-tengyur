བྱེ་བྲག་ཡིན་ནོ། །པྲ་པུ་ས་དང་། ཀུ་ཏུ་ཀ་དང་། ཏིག་ཏ་དང་། མུ་ཏྲ་ལ་དང་། ཧ་ཏི་པ་རྞི་དང་། ཚར་དང་པརྞི་དང་། མུཏྲཱ་ཕ་ལ་དང་། པན་དུ་པ་ཏྲ་དང་། མུ་ཁ་ཕྲི་ཡང་དང་། དེ་དག་ནི་ག་གོན་གྱི་མིང་ཡིན་ནོ། །ཀུ་ཏ་ཛ་དང་། ཙད་ཙ་ཀ་དང་། ཏུད་ཚ་དང་། བྲྀཀྵ་ཀ་དང་།
ག་རིམ་ལ་ལི་ཀ་དང་། བ་རི་ཏིག་ཏ་དང་། ཤང་ཀ་བྲྀཀྵ་དང་། ཀུ་ཏ་ཙི་དང་། ནི་ལ་ཡཥྚི་ཀ་དང་། དེ་དག་ནི་དུག་མོ་ཉུང་གི་ཤིང་གི་མིང་ངོ་། །བད་ཚ་ཀ་དང་། ཀོ་ཏ་ཛ་དང་། ཀ་ལི་ག་ཀ་དང་། ཤ་ཀུ་པཱ་ཧྭ་དང་། ཨིནྡྲ་ཡ་བ་དང་། ཨ་བཱ་ཧྭ་ཡ་དང་། དེ་དག་ནི་དུག་མོ་ཉུང་གི་ས་བོན་གྱི་
མིང་ངོ་། །མུརྦ་དང་། མ་དྷུ་ར་ས་དང་། དེ་བི་དང་། པཱི་ལུ་པརྞི་དང་། མ་དྷུ་པྲ་བ་དང་། སཥྚོ་ན་དང་། མ་དྷུ་པ་ཏི་དང་། ལ་ཀུ་པརྞ་དང་། པྲི་ཐག་ཏྭ་ཙ་དང་། དེ་དག་ནི་མུར་བའི་མིང་དུ་བཤད་དོ། །ཛི་མུ་ཏ་ཀ་དང་། དེ་བར་ལི་དང་། བྲིད་ཏྲ་ཀོ་ཤ་དང་། གྷ་རྭ་ག་རི་དང་། ཅེ་ནི་དང་། ལོ་
མ་ཤ་པ་ཏྲ་དང་། ཀརྐོ་ཏི་དང་། ཨ་ཀུ་བི་ཤ་པ་ཧ་དང་། དེ་དག་ནི་ཛི་མུ་ཏ་ཀའི་མིང་ངོ་། །བྱི་དང་ག་དང་། ཀྲ་མིང་ཛིད་དང་། ཅེ་ལ་ཀ་དང་། ཨ་ལམྦ་དང་། ཙིཏྟ་ཏན་དུ་ལ་དང་། ཀྲི་མི་ཧར་དང་། ཛན་དུག་ན་དང་། ཀྲི་མིན་ཤ་ན་དང་། དེ་དག་ནི་བྱི་དང་གའི་མིང་ཡིན་ནོ། །བེ་ཧ་ས་དང་། བྱི་དུ་
ལ་དང་། ཏཱ་མ་མྲ་དང་། བཉྫུ་ལ་དང་། དིརྒྷ་པ་ཏྲ་ཀ་དང་། ནཏི་ཡ་དང་། གནྦྷ་པུཥྤ་དང་། ཛ་ལོ་ཀཱ་སཱ་མ་ཏྲི་ཏཱ་དང་། དེ་དག་ནི་ལྕང་མའི་མིང་ཡིན་ནོ། །ན་དི་གུ་ལ་པྲི་ཡ་དང་། སུ་ཤི་ཏ་དང་། མེ་གྷ་པུཥྤ་དང་། ཛ་ལེ་ཛ་ཏུ་དང་། ཏོ་ཡ་ཀ་མ་དང་། ནི་ཙུ་ལ་དང་། ཛ་ལེ་པེ་ཏ་པ་དང་། དེ་དག་ནི་
ཆུ་ལྕང་གི་མིང་གི་རྣམ་གྲངས་སོ། །ཙི་ཏྲ་ཀ་དང་། ད་ཧ་ན་དང་། ཅན་ནི་དང་། པཱ་ཅ་ཀ་དང་། ཨ་ན་ལ་དང་། ཛོ་ཏི་དང་། བྱ་མ་དང་། དྭི་པ་དང་། པཱ་ཐིན་དང་། ད་རུ་ན་དང་། དེ་དག་རྣམས་ནི་ཀྲུ་ཏྲུག་ཏྲེས་ཀྱི་མིང་ཡིན་ནོ། །དནྟི་དང་། བུ་ཀ་ལ་ཀ་
དང་། ཙིཏྟྲ་དང་། ནི་ཀུམྤ་དང་། ཨུ་པ་ཙ་ཏྲ་ཀ་དང་། ཀུ་ལ་དང་། ཨུ་དུ་མྦཱར་པ་རྞི་དང་། བྱི་ཤ་ལི་དང་། གྷུ་ན་པྲི་ཡ་དང་། དེ་དག་ནི་དནྟིའི་མིང་གི་རྣམ་གྲངས་སོ། །ཏྲ་བནྟི་ན་དང་། ཤམྤ་རི་དང་། ཙིཏྟྲ་དང་། ནི་ཀུན་པ་དང་། ཉ་གྲོ་ཀྲོ་ཏྲ་བདྷ་ན་དང་། མུ་ཤཱི་ཀཱ་ཡཱ་དང་། པྲ་ཏྱན་ཤྲི་ནི་དང་། བྲེ་ཤེ་རཎྜ་དང་། སུ་ཏ་ཤྲི་ནི་དང་། ཨ་ཀུ་ཀརྞི་ཀ་དང་།
