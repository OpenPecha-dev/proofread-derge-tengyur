གྱིས་མེ་དྲོད་སྐྱེད་པར་བྱེད། །ཚ་བ་ལྔ་དང་ཨ་རུ་ར། །འུ་སུ་པ་ཏ་པ་ལ་ཤ།།སྐྱུར་རྩེ་ཆེན་པོའི་མྱུ་གུ་དང་། །བསྐོལ་བའི་ཐུག་པ་ལ་སོགས་བླུད། །མ་དུ་ཀ་ཡི་མེ་ཏོག་བྲེ། །བཅུ་དྲུག་<བྱེ་,བྱི>་དང་ག་དེའི་ཕྱེད། །ཙི་ཏྲ་ཀ་ནི་དེ་ཕྱེད་དང་། །བསེ་ཤིང་འབྲས་བུ་བྲེ་བཞི་དང་། །བཙོད་སྲང་
བརྒྱད་དེ་འདི་དག་ནི། །ཆུ་བྲེ་སུམ་ཅུ་རྩ་གཉིས་བསྐོལ། །ཆུ་བྱེ་བཅུ་དྲུག་ལུས་བསྒྲངས་པར། །སྦྲང་རྩི་བྱེ་དོ་གཞུག་པ་དང་། །སུག་སྨེལ་པདྨའི་རྩ་བ་དང་། །ཨ་ཀར་ཙནྡན་ལྡེ་གུ་བྱུག །བུམ་པར་ཟླ་བ་གཅིག་གཞག་སྟེ། །ལངས་ནས་མནན་ཆང་བླུད་གྱུར་ན། །འདི་ཡིས་ཕོ་བའི་
དྲོད་སྐྱེད་ཅིང་། །ཉམས་བརྟས་ཁྲག་མཁྲིས་སེལ་བར་བྱེད། །སྐེམས་དང་མཛེ་དང་ཤ་བཀྲ་དང་། །གཅི་སྙི་བ་ཡི་ནད་རྣམས་འཇོམས། །མ་དུ་ཀ་ཡི་མེ་ཏོག་ནི། །ཁུ་བ་བསྐོལ་བ་ཕྱེད་དུ་བ། །བསྒྲངས་པར་སྦྲང་རྩི་བཞི་ཆ་བཏབ། །སྔ་མ་བཞིན་དུ་བྱས་པའི་ཆང་། །ཕན་པའི་
ཟས་དང་བཅས་བསྟེན་ན། །ཕོ་བའི་ནད་ནི་ཐམས་ཅད་སེལ། །དེ་བཞིན་རྒུན་འབྲུ་བུ་རམ་ཤིང་། །འབྲ་གོའི་ཁུ་བ་མནན་པ་བླུད། །ཤིང་ཀུན་ཤུ་དག་པུ་ཙེ་ཤེལ། །བོང་ང་པ་ཏ་དུག་མོ་ཉུང་། །གཟེ་མ་ཚ་བ་ལྔ་པོ་ཞོ། །གཉིས་གཉིས་ལན་ཚྭ་ལྔ་སྲང་རེ། །མར་དང་ཏིལ་མར་
ཕུལ་དོ་དོ། །ཞོ་བྲེ་གཉིས་དང་བསྲེས་ནས་ནི། །མེ་ནི་ཆུང་ངུ་ལ་བཙོས་སྟེ། །རྙིང་ཞིང་ཁུ་བ་ཐིམ་པ་དང་། །དེ་ནས་དུ་བ་འཁྱིམས་པར་བསྲེག །བཏགས་ཏེ་མར་དང་བསྲེས་པ་དག །དེ་ལས་ཞོ་གཉིས་འཐུངས་ནས་ནི། །ཞུ་ནས་ཁ་ཟས་མངར་ཟོས་ནས། །དེས་ནི་བད་ཀན་རླུང་ནད་ནི། །ཀུན་
དང་དུག་དང་དབྱིག་དུག་འཇོམས། །རྩྭ་མཁྲིས་དང་ནི་པུ་ཙེ་ཤེལ། །པ་ཏོ་ནིམ་པ་པར་པ་ཏ། །བསྲེགས་ལ་མ་ཧེའི་གཅིན་དང་ནི། །འཐུངས་ན་མེ་དྲོད་སྐྱེད་བྱེད་ཡིན། །ཤུ་དག་རུ་རྟ་སེར་པོ་གཉིས། །ཙི་ཏྲ་ཀ་དང་པུ་ཙེ་ཤེལ། །གླ་སྒང་ར་ཡི་གཅིན་དང་ནི། །བསྐོལ་བ་མེ་དྲོད་སྐྱེད་
བྱེད་ཡིན། །སུ་ཏའི་སྡོང་བུ་སྲང་བཞི་དང་། །ལན་ཚྭ་གསུམ་ལས་སྲང་རེ་དང་། །པར་ཏ་ཀ་ནི་ཕུལ་གང་དང་། །ཨརྐ་སྲང་བརྒྱད་ཙི་ཏྲ་ཀ །སྲང་གཉིས་བསྲེགས་ལ་པར་ཏ་ཀའི། །ཁུ་བས་རི་ལུ་བྱས་ནས་ནི། །ཟས་རྗེས་ཟོས་ན་ཟོས་པའི་ཟས། །མྱུར་དུ་ལུད་པ་དབུགས་མི་བདེ། །
