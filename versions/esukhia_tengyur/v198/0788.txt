ཞིང་ཟད་པའི་ནད་དམིགས་ནི་རྐང་ཟད་པས། རྐང་གི་སྦུ་གུའི་ནང་སྟོང་། མགོ་འཁོར། མིག་རབ་རིབ་ཟའོ། །ཁུ་ཆུ་ཟད་པ་ནི་བུད་མེད་དང་ཉལ་བ་ན། ཁུ་བ་ཡུན་རིང་དུ་ལོན་ནས་འབྱུང་བའམ། ཡང་ན་ཁུ་བ་ཁྲག་ཉིད་དུ་འབྱུང་། རླིག་པའི་འབྲས་བུ་གཉིས་རབ་ཏུ་ཟུག་
ཅིང་ན། ཕོ་མཚན་ཚིག་པ་སྙམ་དུ་ཚ་ཞིང་མི་བདེའོ། །བཤང་བ་བྲི་ཞིང་ཟད་པ་ནི། རྒྱུ་མའི་ནང་ཀུན་དུ་རླུང་སྒྲ་ཅན་དུ་འཁྲུལ་ཞིང་རྒྱུ་བ་ནི། རྒྱུ་མའི་ནང་དུ་འཁྱིལ་ཞིང་འཁོར་བ་སྙམ་བྱེད་དོ། །ལྟོ་བའི་ནང་ན་ཕན་ཚུན་རླུང་རྒྱུ་ལ། རླུང་དེ་གྱེན་དུ་རྒྱུ་ཞིང་སོང་ན། སྙིང་ག་དང་རྩིབ་
ལོགས་སུ་རབ་ཏུ་ན་ཞིང་ཟུག་གོ། །གཅིན་བྲི་ཞིང་ཟད་པའི་ནད་ཐབས་ནི་གཅིན་ཟད་ན་གཅིན་འབྱུང་བ་དཀའ། གཅིན་གྱི་མདོག་འགྱུར་ཏེ་སྐྱ་བའམ། རྩ་དམར་བའམ། ཁྲག་ཅན་ཉིད་དུ་འབྱུང་བའམ། ཁྲག་དང་འདྲེན་མར་འབྱུང་བ་ཡིན་ནོ། །རྡུལ་ཟད་པའི་ནད་
དམིགས་ནི་སྤུ་འབྱི། སྤུ་ཀུན་ལངས་ཤིང་འགྲེང་བར་འགྱུར། པགས་པ་གས། བཤང་གཅི་དང་། རྡུལ་དང་། དྲི་མ་འདི་གསུམ་ནི་ཕྲ་བས་ན་བརྟག་དཀའ་སྟེ། ནད་གློ་གཅོང་ལྟ་བུ་ཟད་བྱེད་སྐམས་པའི་ནད་ཀྱིས་བཏབ་པའི་ཚེ། གདོད་བཤང་གཅི་དང་རྡུལ་ཟད་པར་
གྱུར་པར་ཤེས་སོ། །དྲི་མ་བཤང་གཅི་རྡུལ་ལ་སོགས་པ་རང་རང་གི་གནས་ན་ཟད་དེ་སྐམས་པ་དང་། ཟུག་ཅིང་ན་ལ་རང་གི་གནས་ན་དྲི་མེད་ཅིང་སྟོང་པས་ཡང་བར་འདུག་པ་འདི་ལྟ་བུའི་ཚེ་ནི། གློ་གཅོང་ལ་སོགས་པའི་ནད་ཀྱི་རང་བཞིན་ཡིན་པར་ཤེས་པར་བྱའོ། །རླུང་དང་བད་
ཀན་དང་མཁྲིས་པ་ལ་སོགས་པ་སོ་སོ་རང་རང་གི་ནད་དམིགས་སྐྱེས་པ་དང་། ནད་དེ་བྱང་སྟེ་ཟད་པར་གྱུར་པ་སྨན་པས་ཤེས་པར་གྱིས་ཤིག །དེ་ཡང་མིའི་ལུས་ལ་རླུང་ནད་ལྟ་བུ་གྲང་ཞིང་རྩུབ་པ་མེད་དེ་བདེ་བའི་ཡོན་ཏན་དང་ལྡན་པ་འདུག་པ་ལས། རླུང་ནད་དུ་གྱུར་ཏེ་ཁོང་གྲང་ཞིང་
རྩུབ་པས་ན་བ་མེད་པའི་ཡོན་ཏན་དེ་ཉམས་ཤིང་ཟད་པའི་ཚེ་ན་རླུང་ནད་སྐྱེས་པ་ཡིན་ནོ། །རླུང་ནད་ཁོང་གྲང་ཞིང་རྩུབ་པའི་རྣམ་པ་ཞེས་བྱ་སྟེ། ཁོང་པ་དྲོད་འབབས་ལ། ལུས་ཀྱི་ཁོང་འཇམ་ཞིང་སྣུམ་པར་གྱུར་པ་ན་བདེ་བའི་ཡོན་ཏན་སྐྱེས་པ་ཞེས་བྱ་བ་སྟེ། འདི་གཉིས་ཀྱི་ཚུལ་ཕན་
