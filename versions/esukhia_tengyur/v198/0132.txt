རྣམ་གསུམ་ནི། །བརྟགས་ནས་རབ་ཏུ་སྦྱར་བར་བྱ། །དེ་ལ་ཆུང་ལ་སྨྱུང་བ་ཕན། །འབྲིང་ལ་སྨྱུང་དང་བཞུ་བྱ་བ། །ཆེན་པོ་ལ་ནི་སྦྱང་བ་སྟེ། །དེ་ཡིས་དྲི་མ་རྩ་ནས་འབྱིན། །དེ་བཞིན་ནད་གཞན་དག་ལ་ཡང་། །རང་ནད་གཞི་ལས་གོ་བཟློག་པས། །གསོ་བ་
དེ་ལས་རྗེས་འབྲེལ་ཡོད། །རྒྱུ་ལས་གོ་བཟློག་རྣམ་སྤང་བ། །སྨན་པས་ཇི་ལྟ་ཇི་ལྟ་བར། །ནད་ལས་གོ་བཟློག་སྦྱར་བའམ། །དེའི་དོན་བྱེད་པ་སྦྱོར་བར་བྱ། །ཉེས་སྨིན་མེ་དྲོད་སྐྱེད་འགྱུར་ནས། །ཉམས་རྟས་སྣུམ་གྱིས་བྱུག་པ་དང་། །བཏུང་དང་འཇམ་རྩི་{𰶀སྩོགས་,སོགས་}རིགས་
སྦྱར། །བད་ཀན་མ་ཞུའི་བེ་སྣབས་ཏེ། །དེ་ནི་མིག་དང་མཁུར་བ་སྐྲང་། །སྒྲེགས་པ་ཟོས་མ་ཐག་འདྲ་འབྱུང་། །མཆིལ་མ་མང་ཞིང་ཉེས་གཡོ་ལྕི། །རླུང་ལས་རེངས་བག་གཟེར་བ་དང་། །བཤང་གཅི་འགགས་ཤིང་སྦོས་དྲོད་ཉམས། །མཁྲིས་ལས་ཁོང་ཚ་སྐོམ་མགོ་
འཁོར། །རྨོངས་བག་སྒྲེགས་པ་སྐྱུར་ཞིང་ཚ། །མ་ཞུ་བེེ་སྣབས་སྨྱུང་བར་བྱ། །རེངས་བག་ལས་ནི་དུགས་རབ་བྱ། །ཚ་བའི་མ་ཞུ་སྐྱུག་སྨན་དག །བཏང་བའམ་གནས་སྐབས་གང་ཕན་བྱ། །མ་ཞུའི་བེ་སྣབས་ལྕི་བ་ན། །རྩ་སྦུབས་ཐིམ་ལས་ཡུན་{𰶁རིངས་,རིང་}ཅན། །

གགས་བྱེད་ཚིག་མ་ཚོས། །ལྕི་དང་རྩུབ་དང་མི་གཙང་དྲང་། །ཁོང་ཚ་སྐོམ་དད་ཆུ་མང་དང་། །བསྲེས་པའི་ཟས་ནི་འཇུ་མི་འགྱུར། །མྱ་ངན་ཁྲོ་དང་བཀྲེས་པ་ཡིས། །ཉེ་བར་གདུངས་ཤིང་ཟས་མི་འཇུ། །ཕན་དང་མི་ཕན་བསྲེས་པ་དག། །ཟོས་པ་མཉམ་ཟ་
