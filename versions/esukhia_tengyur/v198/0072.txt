ཡོས་ཀྱི་ཕྱེ་དང་། མར་བཏབ་པ་ནི་སྙིང་པོ་ལྔའི་ཕྱེ་མ་ཞེས་བྱའོ། །ཤ་ཁུ་དང་། ལུ་སུའི་ཁུ་བ་བསྲེས་པའི་ནང་དུ། འབྲས་བུ་གསུམ་དང་། ཏིལ་གྱི་ཚིགས་མ་དང་། ལ་ཕུག་སྐམ་པོ་དང་། ཏིལ་བཏགས་པ་དང་། སྐྱུར་གང་ཡང་རུང་བ་མང་དུ་བསྲེས་པ་ནི་ཀམྦི་ལ་ཞེས་བྱའོ། །
ནད་ཀྱི་བྱེ་བྲག་གིས་སྣུམ་འཆོས་དང་། རྩུབ་འཆོས་གང་རིགས་པར་བྱེད་པར་ནི་ཀྲི་ཏ་ཀྲ་ཏ་ཞེས་བྱའོ། །ཤ་ཉུང་ཤས་དང་། འུ་སུ་དང་། སེའུ་འབྲུ་ལ་སོགས་པ་བསྐོལ་བའི་དངས་མ་ནི་དྷ་ཀལ་དང་། བ་ནི་ཞེས་བྱའོ། །ཞེས་དྲང་སྲོང་དག་འཆད་དོ། །ཧ་ཏི་དང་། ཨི་པུ་དང་། བ་ར་ནུ་
དང་། དུནྟི་དང་། ཏྲི་ར་དུ་དང་། མ་ཏང་གདང་། དྭ་བི་དང་། ག་ལ་བྷ་དང་། པུ་ཀྵ་རི་དང་། ནཱ་ག་དང་། ཀུཉྫ་ར་ཛ་དང་། ཤ་གྷི་ཧ་ཡན་དང་། དེ་དག་ནི་གླང་པོ་ཆེའི་མིང་ངོ་། །ཀཻ་རེ་ཎུ་ཀ་དང་། ཧ་ཏི་བུ་ག་དང་། དེ་དག་ནི་གླང་པོ་ཆེ་མོའི་མིང་ངོ་། །ཨ་ཤྭ་དང་། བི་ཛ་དང་། ཧ་ཡ་དང་། ཤ་
ཤ་ལི་རུ་རྫ་དང་། ཏུ་རཾ་ག་མ་དང་། གོ་ཏ་དང་། བ་ཧུ་དང་། པ་རི་དང་། སབ་ཏི་དང་། སུ་རཾ་ག་དང་། སྟུ་རཾ་གོ་དང་། ཛ་བི་དང་། དེ་དག་ནི་རྟའི་མིང་ངོ་། །བ་ཏྲ་བ་དང་། གྷོ་ཏོ་ཀ་དང་། དེ་དག་ནི་རྒོད་མའི་མིང་ངོ་། །སིཾ་ཧ་དང་། མྲི་ག་རི་དང་། ཤ་རུ་ལ་དང་། ཀེ་ས་རི་དང་། ཤེ་ཏ་པ་ཀ་ལི་
དང་། ཧ་རེ་དང་། པཉྪུ་མུ་ཀ་ཁ་དང་། ཀྲ་བྱ་དང་། མི་ར་ག་ར་ཛ་དང་། སྲ་ཀྲི་པྲ་ཛ་དང་། དེ་དག་ནི་སེང་གེའི་མིང་ངོ་། །བྷྱ་གྷ་དང་། སྟ་ར་སི་དང་། མྲི་ག་ར་དང་། དཥྚི་དང་། ཙིད་ཏྲ་དང་། གུ་པ་ཀ་ཡ་དང་། བྷྱ་ལ་དང་། པཉྪ་ནཱ་ག་དང་། དྭི་བི་དང་། མྲི་ག་ཏ་དང་། བྷི་མ་དཪྴ་ན་དང་། དེ་དག་ནི་
སྟག་གི་མིང་ངོ་། །ཨུ་ཤོ་དང་། བྷུ་རི་ག་མོ་དང་། བྷ་ཡ་དང་། ཀ་ར་བུ་དང་། དྷིརྒྷ་དང་། མརྒྷ་དང་། དྭ་ཀ་ཀུ་དང་། ཤ་ཀུ་ས་ནི་དང་། བག་ཀྲ་གྷུ་ལ་བ་དང་། སུ་ཅིག་ཀྲ་མ་དང་། དེ་དག་ནི་རྔ་མོའི་མིང་ངོ་། །གྷར་དྷ་པ་དང་། ར་ས་གུ་དང་། བ་ས་ཧྭ་དང་། དྭེ་ར་ཏ་དང་། དྷུ་ར་ཀ་དང་། ཁ་
ར་དང་། བཱ་ལི་ཡ་དང་། ཤཾ་ཀུ་ཀརྞ་དང་། བྷག་ཀྲ་བ་དང་། གྷ་ལི་དུ་ས་ར་དང་། དེ་དག་ནི་ལུང་བོང་གི་མིང་ངོ་། །དནྡུ་དང་། གོ་པ་ཏི་དང་། ཀ་ཀུ་སྨན་དང་། བྲི་ཤ་བཅུ་དང་། ཨུ་གྲུབ་དང་། དན་མྱོ་དང་། བལླི་བྲྡྷ་དང་། གའུ་རཀྵ་ཤ་དང་། དྷུ་པ་ར་ཏ་དང་། དེ་དག་ནི་གླང་གི་མིང་ངོ་།
