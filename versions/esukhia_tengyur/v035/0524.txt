པ་ལ་བཏུད་དེ། །དེ་ཡི་མན་ངག་ཇི་བཞིན་དུ། །གཏོར་མའི་ཆོ་ག་བྲི་བར་བྱ། །དེ་ལ་ཚོགས་བསགས་པ་དང་། ལས་བསྒྲུབ་པར་བྱ་བ་དང་། བར་དུ་གཅོད་པ་རྣམས་ཞི་བར་བྱ་བའི་ཕྱིར་གཏོར་མ་ནར་མར་བྱེད་པར་འདོད་པས་སྲོད་ཀྱི་དུས་སུ་ཤ་ལྔ་དང་། བདུད་རྩི་ལྔའི་རི་ལུ་དང་ལྡན་པའི་གཏོར་མས་སྣོད་མཛེས་པར་བཀང་སྟེ། སེམས་ཅན་ཐམས་ཅད་ལ་
སྙིང་རྗེ་ཆེན་པོ་བསྐྱེད་ལ། དེ་ནས་སྟོང་པ་ཉིད་དུ་མོས་པར་བྱ་སྟེ། དེའི་འོག་ཏུ་ཧཱུཾ་ཞེས་བརྗོད་པས་རྡོ་རྗེའི་ས་གཞི་དང་། ར་བ་དང་། དྲ་བ་དང་། ཕྱི་རོལ་དུ་མེ་ལྕེ་འབར་བ་དང་། རྣམ་པར་འཁྲིགས་པར་བསམས་ཤིང་ནང་དུ་རིན་པོ་ཆེའི་ཁང་པ་ཉམས་དགའ་བ་བསམ་པར་བྱའོ། །དེར་བདག་ཉིད་སྐད་ཅིག་ཙམ་གྱིས་ལྷག་པའི་ལྷར་བསྒོམས་ཏེ་རྡོ་རྗེ་
གསུམ་དང་། ཡེ་ཤེས་སེམས་དཔའ་དང་། ཅོད་པན་གྱི་ལྷ་དྲན་པས་བསྲུང་བར་བྱའོ། །དེ་ནས་རྡོ་རྗེ་ཐལ་མོ་སྙིང་གར་ཕྱེ་ལ་གཏོར་མར་ཧཱུཾ་ཨཱཿཨོཾ་བཾ་བསམས་ཏེ། ཧཱུཾ་ཆུ་ལྟར་ཞུ་བས་ཁ་དོག་ཕུན་སུམ་ཚོགས་པ་དང་། ཨཱཿཞུ་བས་དྲི་ཕུན་སུམ་ཚོགས་པ་དང་། ཨོཾ་ཞུ་བ་རོ་ཕུན་སུམ་ཚོགས་པར་བསམ་པར་བྱའོ། །དེ་ནས་ཨོཾ་གྱི་འོད་ཟེར་
གྱིས་ཡེ་ཤེས་ཀྱི་བདུད་རྩི་བཀུག་སྟེ་གཉིས་སུ་མེད་པར་བྱས་ལ་འབྲུ་གསུམ་བརྗོད་པས་བརྟན་པར་བྱའོ། །དེ་ནས་བདག་གི་སྙིང་གའི་འོད་ཟེར་གྱིས་སེམས་ཅན་ཐམས་ཅད་དམ{𰶅་ཡི་དགས,ཡི་དྭགས་}་ཀྱི་ཚོགས་བཀུག་སྟེ། དེ་དག་གི་སྙིང་ག་དང་སྤྱི་བོར་ལྷ་དང་བླ་མ་རིགས་ཀྱི་བདག་པོ་བསྒོམ་པར་བྱའོ། །དེ་ནས་ནང་དུ་བཀུག་ལ་ཐམས་ཅད་ཀྱི་ལྕེ་ལ་ཧཱུཾ་
ལས་རྡོ་རྗེ་བསྐྱེད། ཨོཾ་གྱིས་ནོར་བུ་བསྐྱེད་ལ། ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷ་རྨཱ་ཎཱཾ་ཨཱ་དྱ་ནུཏྤནྣ་ཏྭཱ་ཏ་ཨོཾ་ཨཱཿཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཞེས་བརྗོད་ནས་ནོར་བུ་འོད་ཟེར་གྱིས་ལྗགས་ཀྱི་ལྕེའུ་ཆུང་དྲངས་པ་བཞིན་དུ་སོ་སོར་ཐིམ་པས་ཐམས་ཅད་གཉིས་མེད་ཀྱི་ཡེ་ཤེས་དང་ལྡན་པར་བསམས་ཏེ། ཡིད་ལ་བྱུང་བའི་མཆོད་པའི་ཚོགས་ཀྱིས་ཀྱང་མཆོད་པར་བྱའོ། །
དེ་ནས་མཆོད་སྦྱིན་གྱི་ཕྱག་རྒྱ་དང་བཅས་ལ། ཆོས་རྣམས་གཟུགས་བརྙན་ལྟ་བུ་སྟེ། །གསལ་ཞིང་དག་ལ་རྙོག་པ་མེད། །གཟུང་དུ་མེད་ཅིང་བརྗོད་དུ་མེད། །རྒྱུ་དང་ལས་ལས་ཀུན་དུ་བྱུང་། །ངོ་བོ་གཉིས་མེད་གནས་མེད་པར། །དེ་ལྟར་ཆོས་རྣམས་ཤེས་ནས་ཀྱང་། །སེམས་ཅན་དོན་ནི་མཉམ་མེད་བྱོས། །འོན་ཀྱང་སངས་རྒྱས་རིགས་
སུ་སྐྱེ། །ཞེས་བྱ་བ་ལ་སོགས་པས་ཆོས་བཤད་པར་བྱའོ། །དེ་ནས་ཨོཾ་ཨཱཿཧཱུཾ། ཡུལ་ཕྱོགས་འདི་ལ་སོགས་པའི། སརྦ་ཀྵེ་ཏྲ་པཱ་ལ། སརྦ་རཱཀྵ་ས། ཡཀྵ། ཨུནྨ་ནྡ། ཨ་པསྨ་ར། སརྦ་དེ་བ། སརྦ་ནཱ་ག །སརྦ་ཡ་མ། སརྦ་ཌཱ་ཀི་ནྱ། སརྦ་སཏྭ། སརྦ་བྷཱུ་ཏ། ཨ་ཏ་ཡ། ཨི་དཾ་བ་ལིཾ། གྲྀ་ཧྞ་གྲྀ་ཧྞ། མ་མ་སརྦ་ས་པ་རི་ཝ་ར་ན་སརྦ་སཏྭ་ནཉྩ། སརྦ་པཱ་པཾ་
