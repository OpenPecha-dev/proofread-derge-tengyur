ཆེན་པོ་དང་། །དུང་སྐྱོང་དང་ནི་རིགས་ལྡན་རྣམས། །ལྷ་མོ་ལྷ་མོ་ཆེན་མོ་དང་། །ཟླ་བའི་སྤྱི་གཙུག་སྤྱི་གཙུག་ཆེ། །བེ་ཅོན་འཛིན་དང་བེ་ཅོན་ཆེ། །སྙིང་པོ་མེད་དང་གུ་ལི་ཏ། །དགའ་བོ་དང་ནི་ཉེ་དགའ་དང་། །རྒྱ་མཚོ་དང་ནི་རྒྱ་མཚོ་ཆེ། །གདུང་བ་དང་ནི་གདུང་ཆེན་མོ། །དཔལ་གྱི་
འོད་དང་འོད་པོ་ཆེ། །རིན་ཆེན་འོད་དང་གཟུགས་བཟང་དང་། །གཟུགས་ཆེན་སྦྲུལ་བཟང་ལྟོ་འཕྱེ་ཆེ། །སྨིག་རྒྱུ་དང་ནི་སྨིག་རྒྱུ་ཆེ། །ཨོཾ་ཨཱ་གཙྪ་ཨཱ་གཙྪ་མ་ཧཱ་ནཱ་ག་ཨ་དྷི་པ་ཏི་སརྦ་བྷུར་བྷུ་བ་ཕྲུཾ་ཕཊ་སྭཱ་ཧཱ། བྱང་ཕྱོགས་སུ་ཀླུ་ལ་གཏོར་མ་བྱིན་ནས་སེ་གོལ་གསུམ་གྱིས་བཏང་པར་བྱའོ། །
ཟླུམ་པོའི་སྟབས་གནས་སྦྱོར་བའི་བདག །ནུབ་ཀྱི་ཕྱོགས་<𰵦གྱི་,ཀྱི་>ཆ་རུ་ནི། །མཎྜལ་དམར་པོས་བྱུག་ནས་སུ། །མེ་ཏོག་དམར་པོ་གཅལ་དུ་བཀྲམ། །གནོད་སྦྱིན་གྱི་ནི་དབང་པོ་རྣམས། །སྔར་གྱི་ཕྱག་རྒྱས་དགུག་པར་བྱ། །མེ་ཏོག་དྲི་ལ་སོགས་ལྡན་པའི། །གཏོར་མ་བདུད་རྩི་ལྔ་དང་བཅས། །
སྔོན་གྱི་རྐང་སྟབས་བསྐྱོད་ནས་ནི། །ནུབ་ཏུ་གནོད་སྦྱིན་དབང་པོ་མཆོད། །ཨོཾ་ནོར་བུ་བཟང་དང་གང་བ་བཟང་། །རྒྱལ་བ་བཟང་པོ་དགའ་བྱེད་ཕུར། །ཨོཾ་ཕུར་བུ་ཆེ་དང་པདྨའི་གཙུག །དུང་གི་གཙུག་ཕུད་དཀར་དང་ནི། །གར་ཆེན་འབིགས་པར་བྱེད་པ་དང་། །གཅོད་པར་བྱེད་དང་གང་
བར་བྱེད། །ནོར་དང་ནོར་མིན་ཤིན་ཏུ་ནོར། །གཟུགས་བཟང་རོལ་པ་རོལ་ཆེན་པོ། །དགོད་དང་རྩེ་དང་རྣམ་པར་བསྐྱོད། །དཔལ་བཟང་དང་ནི་གཟུགས་མཛེས་རྣམས། །བདེ་བ་ཆེན་པོའི་བཀའ་འདི་ཡིས། །གནོད་སྦྱིན་གནོད་སྦྱིན་མོ་འདིར་ཤོག། །ཧཱུཾ་ཨཱ་བ་ད་ར་ཨཱ་བ་ད་ར། ཨོཾ་བཏྟཱ་
ལི་བཏྟཱ་ལི་བྷཀྵ་པྲི་ཡ། །ཡོ་གཱི་ནི་མ་དྷེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་སྭཱ་ཧཱ། ཁུ་ཚུར་གཉིས་ནི་སྦྱར་ནས་སུ། །སྙིང་གའི་ཕྱོགས་སུ་བཙིར་བར་བྱ། །ཨོཾ་དཱ་ན་ཨ་དྷི་པ་ཏ་ཡེ་བཏྟཱ་ལི་ཕུཾ་སྭཱ་ཧཱ། ནུབ་སྒོར་གནོད་སྦྱིན་གྱི་གཏོར་མ་བྱིན་ལ། སེ་གོལ་ལན་གསུམ་གཏོགས་པར་བྱའོ། །གསུམ་གྱི་གསུམ་པ་དབུ་མ་ལ། །རྡོ་རྗེ་
རྐང་པས་མནན་ནས་སུ། །ཁྲག་གམ་དམར་པོའི་རྫས་ཀྱིས་ནི། །མཎྜལ་གྲུ་གསུམ་བྱུག་པར་བྱ། །མེ་ཏོག་དམར་པོ་བདུག་པ་སོགས། །ཉེ་བའི་ལོངས་སྤྱོད་གཏོར་མར་ལྡན། །ཤ་སྐམ་དང་ནི་ཉ་ལ་སོགས། །བཙོད་དང་ཁྲག་གིས་གང་བའི་སྣོད། །སྲན་ཆན་དང་ནི་ཏིལ་ཁུར་བཅས། །
