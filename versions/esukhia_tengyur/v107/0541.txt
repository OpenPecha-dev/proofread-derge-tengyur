དེས་ན་སྐྱེ་བ་བཀག་གྱུར་ཀྱང་། །བསྒྲུབ་པར་བྱ་ལ་ངེས་འབྱུང་བ། །གཟུགས་ལས་གཟུགས་བརྙན་ཇི་བཞིན་དུ། །ཀུན་རྫོབ་ཏུ་ནི་ཡོད་མིན་པ། །ཇི་སྟེ་ཤེས་སོགས་དངོས་ཡོད་པ། །མི་འདོད་ན་ཡང་ཀུན་རྫོབ་ནི། །རྒྱུ་མེད་པར་ནི་མི་རིགས་ཕྱིར། །གང་དུ་བདག་དང་ཆོས་འདི་
འདོགས། །འོན་ཏེ་ཡང་ནི་རྒྱུ་བརྗོད་ན། །དེ་ཉིད་དངོས་པོ་གྲུབ་པ་ཡིན། །ཀུན་ནས་ཉོན་མོངས་རྣམ་བྱང་དུ། །འགྱུར་བའི་རྒྱུ་ཡང་ཅི་ཞིག་ཡིན། །ཀུན་རྫོབ་རྣམས་ཀྱི་རྒྱུ་ཙམ་ཞིག །རབ་ཏུ་བསྒྲུབ་ནའང་གྲུབ་པ་སྒྲུབ། །འདི་ཡི་ས་བོན་ཀུན་རྫོབ་ཏུ། །རང་དང་རྗེས་སུ་མཐུན་པར་འདོད། །ཇི་སྟེ་
དངོས་ཡོད་དེ་སྒྲུབ་ན། །དེ་དང་རྗེས་སུ་འགྲོ་བ་མེད། །ཤིང་ལ་སོགས་ལ་གྲུབ་པ་ནི། །ཀུན་རྫོབ་ཁོ་ནར་ཀུན་དུ་གྲུབ། །རང་གི་རྒྱུ་ལ་ལྟོས་པར་ནི། །མ་བརྟགས་ཤིན་ཏུ་གྲུབ་པ་ཡིན། །གང་ཕྱིར་ཀུན་དུ་ཀུན་རྫོབ་ནི། །གནོད་སྦྱིན་རྗེས་མཐུན་གཏོར་མ་བཞིན། {𰸮།བརྫུན,རྫུན་}་པའི་དངོས་པོ་དེ་ལ་ནི། །
ཇི་སྟེ་དངོས་པོ་ཡོད་དོ་ཞེས། །མིང་དུ་བརྗོད་ན་མིང་ལ་ནི། །བརྩད་པ་ནམ་ཡང་ཡོད་པ་མིན། །ཇི་ལྟར་རྨི་ལམ་སྦྲུལ་གདུག་པས། །ཟིན་དེ་བརྒྱལ་དང་དེ་བྲལ་ལྟར། །ཀུན་ནས་ཉོན་མོངས་རྣམ་བྱང་ཡང་། །དེ་བཞིན་འདོད་ཀྱི་ཡང་དག་མིན། །སྐྱེ་བ་དང་ནི་འགག་པ་དང་། །ཀུན་ནས་ཉོན་
མོངས་རྣམ་བྱང་སོགས། །འཕགས་པ་ཤེས་རབ་ཕ་རོལ་ཏུ། །ཕྱིན་སོགས་ལ་བརྗོད་ཐ་སྙད་ཡིན། །དོན་དམ་པ་ཡི་དངོས་པོ་དང་། །རབ་ཏུ་འབྲལ་བ་མ་གྲུབ་ན། །དེ་མེད་པ་ཡི་ཉོན་མོངས་སོགས། །དངོས་པོ་མེད་པར་རིགས་པ་མིན། །མ་བརྟགས་རབ་ཏུ་དགའ་རྣམས་ཀྱིས། །ཤིན་ཏུ་
འཇུག་དང་ལྡོག་པ་ལས། །རྨོངས་ལ་སོགས་ལ་ཉོན་མོངས་དང་། །རྣམ་པར་བྱང་ལ་རྣམ་པར་གནས། །ཀུན་རྫོབ་ཉིད་དུ་ཐམས་ཅད་ཡོད། །དམ་པའི་དོན་དུ་མེད་མིན་ནམ། །དངོས་དང་དངོས་མེད་ལ་སོགས་པའི། །ཆོས་རྣམས་གཅིག་ལ་ཇི་ལྟར་འཐད། །ཀུན་རྫོབ་ཅི་ཡིན་
བརྗོད་བྱ་དགོས། །ཐ་སྙད་ལས་བྱུང་བློ་ཡིན་ན། །དེ་ལྟར་གྱུར་ན་བློ་ཙམ་སྟེ། །རྣམ་རིག་ཙམ་ཉིད་གྲུབ་པ་ཡིན། །སྒྲ་ནི་ཀུན་རྫོབ་ཡིན་ནའང་། །ལེགས་བཤད་ཉེས་བཤད་དབྱེ་མེད་འགྱུར། །སྒྲ་ལས་ཐམས་ཅད་གྲུབ་པའི་ཕྱིར། །ཕོངས་པ་འགའ་ཡང་མེད་པར་འགྱུར། །
