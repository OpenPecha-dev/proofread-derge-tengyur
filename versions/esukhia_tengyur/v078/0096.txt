དང་། རྣལ་འབྱོར་པ་ནི་སྒྲོ་བཏགས་དང་། །སྐུར་མཐའ་གང་དང་གང་དག་གི །དབྱེ་བ་དེ་དང་དེར་བརྟེན་པ། །དེ་ཉིད་ཚུལ་དུ་ལྡོག་པ་ཡིན། །དེ་དང་དེ་ཚུལ་ལ་གནས་པ། །ཀུན་མཁྱེན་ཡེ་ཤེས་དངོས་རང་ཉིད། །རྫོགས་པའི་བྱང་ཆུབ་སེམས་དཔའི་བདག །ཕ་རོལ་ཕྱིན་ཀུན་རང་བཞིན་
ཅན། །དེ་ཁོ་ན་ཉིད་ཀྱི་རྣལ་འབྱོར་པ་དེ་སྒྲོ་བཏགས་པ་མ་ལུས་པ་ལས་ལྡོག་པས་མངོན་པར་བྱང་ཆུབ་པ་རྣམ་པ་ལྔའི་རང་བཞིན་ཅན་གྱི་དཀྱིལ་འཁོར་བའི་ལྷ་རྣམས་ཀྱི་བདག་ཉིད་ཅན་དང་། ཕ་རོལ་ཏུ་ཕྱིན་པ་ཐམས་ཅད་ཀྱི་རང་བཞིན་དུ་རྣམ་པར་གཞག་པ་ཡིན་ནོ། །དེ་ལ་འདོད་
ཆགས་ལ་སོགས་པའི་དྲི་མ་མ་ལུས་པ་དང་བྲལ་བ་ཉིད་དེ་ཉིད་དེ་ལས་གཞན་པའི་ཁྱད་པར་ཅན་གྱི་རྒྱུར་གྱུར་པ་ཉིད་ཀྱིས་ན་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས་ཀྱི་ངོ་བོ་ཉིད་དང་། དབྱེ་བ་མ་ལུས་པ་དང་བྲལ་བའི་རང་བཞིན་ཉིད་ཀྱིས་ན་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས་ཀྱི་ངོ་བོ་ཉིད་དང་། ཕྱིན་ཅི་ལོག་མ་
ལུས་པ་དང་བྲལ་བའི་རང་བཞིན་ཉིད་ཀྱིས་ན་སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་ཀྱི་ངོ་བོ་ཉིད་དང་། འགྲོ་བ་མ་ལུས་པའི་ཕན་པ་བསྐྱེད་པའི་རྒྱུར་གྱུར་པས་བྱ་བ་སྒྲུབ་པའི་ཡེ་ཤེས་ཀྱི་ངོ་བོ་ཉིད་དང་། གཟུང་བ་དང་། འཛིན་པ་རྣམ་པ་གཉིས་དང་བྲལ་བས་ཆོས་ཀྱི་དབྱིངས་ཤིན་ཏུ་རྣམ་པར་དག་
པའི་ངོ་བོ་ཉིད་ཡིན་ནོ། །དེ་བས་ན་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས་ལ་སོགས་པའི་བདག་ཉིད་ཅན་མི་བསྐྱོད་པ་ལ་སོགས་པ་དེ་བཞིན་གཤེགས་པ་ལྔའི་བདག་ཉིད་ཅན་ཡང་དེ་ཉིད་ཡིན་པས་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ངོ་བོ་ཉིད་ཡིན་ནོ། །དེ་ཉིད་ནི་ཉེ་བར་དམིགས་པ་རྣམས་ལས་ལྡོག་པས་སྟོང་
པ་ཉིད་ཀྱི་བདག་ཉིད་ཅན་ཡིན་པའི་ཕྱིར་རྡོ་རྗེ་སེམས་མའོ། །མཚན་མ་མ་ལུས་པ་ལས་ལྡོག་ཅིང་མཚན་མ་མེད་པའི་བདག་ཉིད་ཅན་ཡིན་པས་རིན་ཆེན་སེམས་མའོ། །ལོག་པའི་སྨོན་ལམ་ལས་ལོག་ཅིང་སྨོན་པ་མེད་པའི་བདག་ཉིད་ཅན་ཡིན་པས་ན་ཆོས་ཀྱི་སེམས་མའོ། །ཕྱིན་ཅི་ལོག་
གི་མངོན་པར་འདུ་བྱེད་པ་ལས་ལྡོག་ཅིང་མངོན་པར་འདུས་མ་བྱས་པའི་བདག་ཉིད་ཅན་ཡིན་པས་ན་ལས་ཀྱི་སེམས་མ་སྟེ། རྡོ་རྗེ་སེམས་མ་ལ་སོགས་པ་རིགས་ཀྱི་ཡུམ་བཞི་ནི་{𰶀གོ་རིམས་,གོ་རིམ་}བཞིན་དུ་སྟོང་པ་ཉིད་ལ་སོགས་པ་རྣམ་པར་ཐར་པའི་སྒོ་བཞིའི་ངོ་བོ་ཉིད་ཡིན་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་དེ་
