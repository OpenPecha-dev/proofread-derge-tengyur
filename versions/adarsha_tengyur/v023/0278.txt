པ་ལྟེ་བར་བཅས། །ཕྱི་རོལ་རྡོ་རྗེ་ཕྲེང་བས་བསྐོར། །འཕྲོ་བའི་རྡོ་རྗེའི་ཚོགས་སྤྲུལ་པའོ། །འཁོར་ལོ་རྩིབས་བརྒྱད་ལེགས་པར་བྲི། །དངོས་གྲུབ་ཀུན་དང་ཡོན་ཏན་གཞི། །ཕྱི་རོལ་དུ་ནི་ནག་པོས་བསྐོར། །མིང་ནི་མཁའ་སྤྱོད་ཅེས་བྱའོ། །རྩིབས་བརྒྱད་ལ་སོགས་འཁོར་ལོ་སྟེ། །པདྨའི་ཀ་བའི་
ཕྲེང་གིས་བསྐོར། །ས་འོག་གནས་པའི་ལྷ་མོ་ནི། །འཁོར་ལོའི་ཕྲེང་བས་དང་པོར་བསྐོར། །རྩིབས་ཀྱི་མཚམས་སུ་རྡོ་རྗེ་ནི། །མཉམ་གཞག་རྣལ་འབྱོར་པ་ཡིས་བྲི། །འཁོར་ལོ་ཕྲེང་བའི་མཐར་ཐུག་པར། །འབར་བའི་རི་མོ་དེ་བཞིན་ནོ། །ཧེ་རུ་ཀ་དཔལ་རྣལ་འབྱོར་ཆེ། །འཁོར་ལོའི་
དབུས་སུ་ལེགས་གནས་ནས། །ཧི་སྒྲོགས་ཁྲོ་བོའི་ལྟ་བ་ཡིས། །དེ་ནས་སྒོ་ནི་དབྱེ་བར་བྱ། །སྒོ་ནི་འཁོར་ལོའི་བརྒྱད་ཆ་སྟེ། །ཁ་ཁྱེར་དག་ནི་ཡངས་པའོ། །སྒོ་ཁྱུད་དང་ནི་སྒོ་ཡི་ཚད། །ལྷ་ཡི་སྣམ་བུའང་དེ་བཞིན་ནོ། །ཁ་ཁྱེར་ཀུན་ནས་སྒོ་ཡི་ཕྱེད། །འགྲམ་དང་ལོགས་དང་དེ་བཞིན་
ནོ། །དྲ་བ་དྲ་ཕྱེད་ཉི་ཟླ་དང་། །འཕན་དང་མེ་ཏོག་ཕྲེང་བས་སྤྲས། །ཚོན་གྱི་ས་ནི་དེ་ཡི་ཕྱེད། །རྩ་བའི་ཐིག་གི་ས་ཡི་རྒྱབ། །རྡོ་རྗེ་ལ་སོགས་ཀ་བ་ཡི། །ས་ནི་ཚོན་གྱི་ས་དང་མཉམ། །རྟ་བབས་སྒོ་ལས་སུམ་འགྱུར་ཏེ། །འཕན་དང་དྲིལ་བུ་དང་ལྡན་པ། །དྲིལ་བུ་རླུང་གིས་བསྐྱོད་པ་ཡི། །འཕན་
ནི་ཕྱི་རོལ་མཚམས་སུའོ། །དང་པོར་ནམ་མཁའི་ཐིག་ཡིན་ནོ། །དེ་ཡི་འོག་ཏུ་ཡེ་ཤེས་ཐིག །ཕྱོགས་ཀུན་མཉམ་པར་བགྲོད་ནས་ནི། །རྣལ་འབྱོར་པ་ཡི་ཆོ་གས་གདབ། །ལུགས་བྱུང་ལུགས་བཟློག་པ་འདིས་ནི། །ཕྱི་རོལ་དཀྱིལ་འཁོར་ཐིག་ཡིན་ནོ། །དེ་བཞིན་ནང་གི་བདག་ཉིད་ལུས། །
ཚོགས་པ་ལས་བྱུང་མགོན་པོ་ཡིས། །ལུས་ཀྱི་དཀྱིལ་འཁོར་ཐིག་གདབ་བོ། །ལྷན་ཅིག་སྐྱེས་ཤིང་རང་བཞིན་དག །དཔའ་བོ་དཔའ་མོའི་དབང་ཕྱུག་ལྡན། །ཕྱི་དང་ནང་གི་ཐིག་གདབ་པའི་ཆོ་གའོ།།      །།དུར་ཁྲོད་རུས་པ་ལེག་ས་པར་བླང་། །ཐལ་བ་ལོ་མ་ལ་སོ་ཕག་དང་། །བ་བླ་
དག་དང་ལྡན་པ་ཡིས། །ལེགས་བྲིས་ཧེ་རུ་ཀ་འཁོར་ལོ། །ཐོད་པར་རྡུལ་ཚོན་བཞག་ནས་ནི། །བློ་བཟང་ཐིག་གདབ་ཆོ་གས་སོ། །འབར་བའི་ལྟ་བས་བཀུག་ནས་ནི། །ཡེ་ཤེས་འཁོར་ལོའི་སྙིང་པོ་གཞུག །སངས་རྒྱས་ཀུན་གྱི་མཁའ་འགྲོ་མའི། །གསང་སྔགས་ཉིད་ཀྱིས་མངོན་
