བྱེད་དང་གངས་ཅན་གྱི་བློ་བཞིན་ནོ་ཞེས་བྱ་བ་དང་ལྡན་པ་མ་ཡིན་པའི་བློ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་ཅི་རིགས་པའི་དོན་ཏོ། །ཅི་སྟེ་གང་གིས་གཏན་ཚིགས་མི་མཐུན་པའི་ཕྱོགས་ལ་འཇུག་པའི་དོགས་པ་ལྡོག་པར་འགྱུར་བ་འདི་ལ་བསྒྲུབ་པར་བྱ་བ་ལས་བཟློག་པ་ལ་གཏན་ཚིགས་གནོད་པ་ཅན་གྱི་
ཚད་མ་ཅི་ཡོད་ཅེ་ན། གཅིག་ནི་དུ་མ་ལ་འཇུག་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཇི་ལྟར་གཅིག་དུ་མ་ལ་འཇུག་པ་མི་རིགས་པ་དེ་ལྟར་དེ་གཅིག་ལ་འཇིག་རྟེན་ངོ་བོས་ཞེས་བྱ་བ་ལ་སོགས་པས་ཡན་ལག་ཅན་དགག་པར་རྒྱས་པར་བསྟན་པའི་ཕྱིར་ཡང་འདིར་རྒྱ་མི་བསྐྱེད་དོ། །གཞན་ཉིད་དང་གཞན་མ་ཡིན་པ་
དག་ཉིད་སུན་འབྱུང་བར་བཤད་པ། གཞན་དང་གཞན་མིན་བརྗོད་ལ་སོགས། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། གང་ལས་འདི་ནི་གཞན་ནོ། །འདི་ནི་གཞན་མ་ཡིན་པའོ་ཞེས་བྱ་བའི་བརྗོད་པ་དང་རྟོགས་པ་དག་ཏུ་འགྱུར་བ་གཞན་དང་གཞན་མ་ཡིན་པ་དེའི་རྒྱུ་མཚན་ནི་གོ་རིམས་བཞིན་དུ་གཞན་ཉིད་དང་གཞན་མ་
ཡིན་པ་ཉིད་ཡིན་པར་གྲུབ་པོ། །སོགས་པའི་སྒྲས་རྟོགས་པ་བཟུང་ངོ་། །སྦྱོར་བ་ནི་གང་འདི་གཞན་དང་གཞན་མ་ཡིན་པའི་བློ་དེ་ནི་བུམ་པ་ལས་ཐ་དད་པའི་དོན་གཞན་དུ་གྱུར་པའི་རྒྱུ་མཚན་ཅན་མ་ཡིན་ཏེ། དེའི་ཤེས་པ་དང་མི་མཐུན་པའི་ཕྱིར་བདེ་བ་ལ་སོགས་པའི་བློ་བཞིན་ནོ་ཞེས་བྱ་བའོ། །འདི་ལྟར་ཕྱོགས་གཅིག་
ན་གནས་པའི་གང་བུ་དག་ལ་གཞན་དང་གཞན་མ་ཡིན་ཞེས་བྱ་བའི་བློ་འཇུག་པའི་ཕྱིར། རེ་ཞིག་འདི་ཕྱོགས་ཀྱི་རྒྱུ་མཚན་ཅན་མ་ཡིན་ལ། དུས་ཀྱི་རྒྱུ་མཚན་ཅན་མ་ཡིན་ཏེ། ད་ལྟར་བའི་དུས་གཅིག་ལ་ད་ལྟར་བར་གནས་པ་ཕྱོགས་དང་ཡུལ་མ་ངེས་པ་དང་ལྡན་པའི་གཞོན་ནུ་དང་གནས་བརྟན་དག་ལ་རྣམ་པར་
དབྱེ་ནས་གཞན་དང་གཞན་མ་ཡིན་པའི་བློ་འཇུག་པའི་ཕྱིར་ཏེ། དེ་ཁྱད་པར་མེད་ན་ཡང་ཁྱད་པར་ཡོད་པའི་ཕྱིར་རོ་སྙམ་དུ་བསམས་པ་ཡིན་ནོ། །བློ་འདིའི་རྒྱུ་མཚན་གཞན་བརྗོད་པར་ཡང་མི་ནུས་ཏེ། དེ་ལྟ་བས་ན་འདིའི་རྒྱུ་མཚན་གང་ཡིན་པ་དེ་ནི་གཞན་ཉིད་དང་གཞན་མ་ཡིན་པ་ཉིད་དོ་ཞེས་བྱ་བར་གྲུབ་པོ། །ཕྱོགས་དུས་དེ་
ཡི་རྒྱུ་མཚན་མིན། །ཞེས་བྱ་བ་ནི་ཕྱོགས་དང་དུས་ཀྱིས་ཕྱོགས་དང་ལྡན་པའི་ཉེ་བ་དང་བསྐལ་པའི་དངོས་པོ་ལ་རྒྱུ་མཚན་ཅན་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཕྱོགས་དང་དུས་ཀྱི་སྒྲ་དག་གིས་བཏགས་ནས་ཕྱོགས་དང་དུས་ཀྱི་ཡུལ་ཕྱོགས་དང་ལྡན་པ་དག་བསྟན་ཏོ། །འདི་ལྟར་གཞན་གྱིས་གཞན་ཉིད་དང་གཞན་
