རྣམས་འཛིན་གྱུར་པས། །སངས་རྒྱས་བྱང་ཆུབ་སེམས་དཔའ་དང་། །ལྷ་རྣམས་ཀུན་ཀྱིས་མཐུན་པར་གསུངས། །སེམས་ཅན་ཀུན་ལ་སེམས་བརྩེ་བས། །དཀྱིལ་འཁོར་ཆོ་ག་བཞིན་དུ་བྲི། །ཁྱོད་ཀྱིས་ནན་ཏན་བསྐྱེད་ནས་སུ། །སྒྲུབ་པ་པོ་རྣམས་སྔགས་ལ་སྦྱོར། །དེ་ནས་སླར་ཡང་བསྟོད་པར་བྱ་སྟེ།  མཆོག་ཏུ་གསང་བའི་
དཀྱིལ་འཁོར་འདིར། །བཅུག་པ་དང་ནི་མཐོང་བ་ཡིས། །དེ་རིང་སྡིག་ལས་རྣམ་གྲོལ་ནས། །དེ་རིང་ཁྱེད་རྣམས་དགེ་བར་གནས། །བདེ་ཆེན་ཐེག་པ་འདི་ལ་ནི། །ཕྱིན་ཆད་འཆི་བ་ཡོད་མིན་ཞིང་། །ཐུབ་པ་ཚུགས་པ་ཡོད་མིན་པས། །ཅིས་ཀྱང་མི་འཇིགས་དགའ་བར་གྱིས། །སྔིག་པ་ཤིན་ཏུ་དག་པས་ན། །སྲིད་པའི་སྡིག་པ་ལས་འདས་
པའོ། །བྱང་ཆུབ་སེམས་ནི་ཡོངས་མི་སྤོང་། །རྡོ་རྗེ་ཞེས་བྱའི་ཕྱག་རྒྱ་དེ། །གང་ཞིག་བསྐྱེད་པ་ཙམ་གྱིས་ནི། །སངས་རྒྱས་ཉིད་འགྱུར་ཐེ་ཚོམ་མེད། །དམ་པའི་ཆོས་ནི་སྨད་མི་བྱ། །ནམ་ཡང་བོར་བར་ཡང་མི་རུང་། །མི་ཤེས་པའམ་རྨོངས་པ་ལ། །དེ་ནི་བསྟན་པར་མི་བྱའོ། །རྡོ་རྗེ་དྲིལ་བུ་མི་སྤང་སྟེ། །སློབ་དཔོན་ནམ་ཡང་སྨད་མི་བྱ། །སངས་རྒྱས་ཀུན་
དང་མཚུངས་པ་ཡིན། །རང་བདག་ཡོངས་སུ་སྤངས་ནས་སུ། །དཀའ་ཐུབ་ཀྱིས་ནི་གདུང་མི་བྱ། །ཇི་ལྟར་བདེ་བ་བདེ་བཟུང་ནས། །འདི་ནི་མ་འོངས་རྫོགས་སངས་རྒྱས། །ཞེས་བྱ་བས་དབུགས་དབྱུང་ངོ་། །སློབ་མས་ཀྱང་ཕྱག་བྱས་ཏེ། གཙོ་བོས་ཅི་ལྟར་བཀའ་སྩལ་པ། །དེ་དག་ཐམས་ཅད་བདག་གིས་བགྱི། །ཞེས་ཁས་བླང་ངོ་། །སློབ་དཔོན་གྱི་དབང་
བསྐུར་བའི་ཆོ་གའོ།།      །།དེའི་རྗེས་ལ་སློབ་མ་ཡིད་དུ་འོང་བའི་གནས་དབེན་པར་ཡོལ་བ་ལ་སོགས་པས་བྲེས་ལ། གཟུགས་ལ་སོགས་པ་ཡོན་ཏན་དང་ལྡན་པ་དམ་ཚིག་ཅན་གྱི་བུ་མོ་རྡོ་རྗེ་སློབ་དཔོན་ལ་ཕུལ་ཏེ། བཟའ་དང་བཅའ་བ་དང་བཏུང་བ་ཞིམ་པོ་རྣམས་ཀྱིས་བླ་མ་མཉེས་པར་བྱས་ལ་མཎྜལ་བྱས་ཏེ། ཚིགས་སུ་བཅད་པ་
འདིས་གསང་བ་ནོད་པའི་དོན་དུ་གསོལ་བ་གདབ་བོ། །ཁྱོད་ཀྱི་བཀའ་དྲིན་གང་གིས་བདག །བདག་དང་གཞན་དོན་བྱེད་ནུས་པ། །ཡང་དག་གསང་བའི་དབང་བསྐུར་ནི། །སྙིང་རྗེའི་གཏེར་གྱིས་བདག་ལ་སྩོལ། །ཞེས་བྱའོ། །དེ་ལྟར་གསོལ་བ་བཏབ་པའི་མཐར་བླ་མས་སློབ་མ་ལ་སྙིང་རྗེ་ཆེན་པོ་བསྐྱེད་དེ་སྒྲུབ་ཐབས་ལས་འབྱུང་བའི་རིམ་པ་
བཞིན་བདག་ཉིད་དང་ཤེས་རབ་གསུང་རྡོ་རྗེའི་གཟུགས་སུ་མངོན་པར་འདུས་བྱས་ལ་སྙོམས་པར་འཇུག་པར་བྱའོ། །དེ་ལས་བྱུང་བའི་བདུད་རྩི་དང་བདུད་རྩི་ལྔ་བསྲེས་ལ་ཚིགས་སུ་བཅད་པ་འདིས་སློབ་མའི་ཁར་སྦྱིན་ནོ། །ཇི་ལྟར་འཕགས་པ་སངས་རྒྱས་ཀྱིས། །བྱང་ཆུབ་སྲས་ལ་དབང་བསྐུར་བ། །དེ་བཞིན་གཤེགས་པའི་དབང་
