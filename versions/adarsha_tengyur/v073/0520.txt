གྱིས་གདུག་པ་རྣམས། །སྐྱེ་བོ་རྣམས་ཀྱི་དོན་དུ་ཆོམ། །དེ་ལྟར་བྱས་ན་སངས་རྒྱས་ལ། །ཕན་འདོགས་དྲིན་དུ་གཟོ་ཞེས་བྱ། །རྡོ་རྗེ་འཛིན་པ་དེ་ཀུན་ཀྱང་། །ཁྱོད་ལ་རྟག་ཏུ་སྐུལ་བར་བྱེད། །སེམས་ཅན་ཀུན་ལ་ཕན་པའི་ཕྱིར། །སེམས་ཅན་མ་ལུས་ཐམས་ཅད་ལ། །ཇི་ལྟར་འདུལ་གྱུར་སྣ་ཚོགས་པ། །
ཆོས་ཀྱི་འཁོར་ལོ་རབ་ཏུ་བསྐོར། །དེ་བཞིན་རྡོ་རྗེ་རིན་ཆེན་པད། །ལས་ཀྱི་འཁོར་ལོའི་རྗེས་གནང་སྦྱིན། །དེ་ནས་ལུང་བསྟན་དབུགས་དབྱུང་སོགས། །ཇི་སྐད་གསུང་བཞིན་སྦྱིན་པར་བྱ། །འདི་ནི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །ཕྱག་ན་རྡོ་རྗེའི་ལག་ན་འདུག །རྟག་ཏུ་ཁྱོད་ཀྱིས་གཟུང་བར་བྱ། །
ཕྱག་ན་རྡོ་རྗེའི་བརྟུལ་ཞུགས་བརྟན། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་སིདྡྷི་བཛྲ་ས་མ་ཡ་ཏིཥྛ་ཨེ་ཥ་ཏྭཾ་དྷ་ར་ཡཱ་མི་ཧི་ཧི་ཧི་ཧི་ཧཱུཾ། ཞེས་པས་བརྟུལ་ཞུགས་སྦྱིན་པར་བྱ། །དེ་ནས་འབྱོར་པའི་ཡོན་བདག་གིས། །བླ་མ་རྗེས་སུ་མཉེས་པར་བྱ། །ཞི་བ་ཡི་ནི་ཐབ་ཁུང་དུ། །སྦྱིན་སྲེག་གིས་ནི་ཚིམ་པར་བྱ། །དེ་
ནས་རྡུལ་ཚོན་ལ་སོགས་ནི། །འདོད་པའི་མི་ལ་སྦྱིན་པའམ། །ཆུ་བོ་ཆེན་པོར་དོར་བར་བྱ། །ཀླུ་རིགས་ཀུན་གྱི་དགྲར་གྱུར་པའི། །གོས་སྔོན་དཀྱིལ་འཁོར་ཆོ་ག་བྱས། །འདི་ལ་འཁྲུལ་པ་མཐོང་ན་ཡང་། །མཁས་པ་རྣམས་ཀྱིས་བཟོད་པར་མཛོད། །འདི་བྱས་དགེ་བ་གང་ཐོབ་པ། །
དེས་ནི་འགྲོ་བ་མ་ལུས་པ། །རྡོ་རྗེ་ཅན་གྱི་ཞལ་མཐོང་ནས། །སྡུག་བསྔལ་མཚོ་ལས་སྒྲོལ་བར་ཤོག །ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག །སློབ་དཔོན་ཆེན་པོ་སྟོང་པ་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་སྙིང་རྗེའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་
མཁན་པོ་པཎྜི་ཏ་དོན་ཡོད་པའི་རྡོ་རྗེ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་ཁམས་པ་བ་རི་དགེ་སློང་རིན་ཆེན་གྲགས་པས་བསྒྱུར་ཅིང་ཞུས་པའོ། །༄༅༅། །རྒྱ་གར་སྐད་དུ། བཛྲ་ཀྲོ་དྷ་དྷ་ར་སཱ་དྷ་ན་ནཱ་མ། བོད་སྐད་དུ། ཁྲོ་བོ་རྡོ་རྗེ་འཛིན་པའི་སྒྲུབ་པའི་ཐབས་ཞེས་བྱ་བ། བླ་མ་རྗེ་བཙུན་
པའི་ཞབས་ལ་ཕྱག་འཚལ་ལོ། །རྣལ་འབྱོར་པ་ལ་གནོད་པ་ཡི། །འབྱུང་པོའི་ཚོགས་རྣམས་བསྲུང་བའི་ཕྱིར། །རྡོ་རྗེ་འཛིན་མཆོག་སྒྲུབ་པའི་ཐབས། །འབད་པའི་བློ་ཡིས་བསྒྲུབ་པར་བྱ། །དང་པོར་སྟན་བདེ་ལ་འདུག་སྟེ། །སྐད་ཅིག་གཅིག་གིས་རྡོ་རྗེ་འཛིན། །ཁྲོ་བོའི་རྒྱལ་པོར་
