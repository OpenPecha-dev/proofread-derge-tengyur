།དེ་ནས་ཕྱིར་མི་ལྡོག་པ་རྡོ་རྗེ་སློབ་དཔོན་གྱི་དབང་བསྐུར་ཏེ། དབང་སྟེགས་པདྨ་འདབ་བརྒྱད་བྲི། །དེ་སྟེང་སློབ་མ་བཞག་ནས་ནི། །ཚིག་གིས་གསོལ་བ་གདབ་པར་བྱ། །མི་ལྡོག་འཁོར་ལོའི་དབང་སྩོལ་ཅིག་དཀྱིལ་འཁོར་ལྷ་ཡི་དེ་ཉིད་དང། །སྔགས་དང་ཕྱག་རྒྱའི་དེ་ཉིད་རྣམས། །
བདག་ཉིད་སློབ་དཔོན་བགྱིད་པར་འཚལ། །དེ་ནས་བུམ་ཆུས་དབང་བསྐུར་ཏེ། །གསང་བ་བྱང་ཆུབ་སེམས་ཀྱི་ཆུ། །ལྷ་ཡི་ཆུ་ནི་དག་པ་སྟེ། །དྲི་མེད་རྡོ་རྗེ་ཆུ་ཡི་དབང། །རྡོ་རྗེ་རིགས་མཆོག་ཆུ་དབང་སྦྱིན། །རྡོ་རྗེ་དམ་ཚིག་སྦྱིན་ཏེ། ཧཱུཾ་ཐོག་མཐའ་མེད་པའི་སེམས་དཔའ་ནི། །རྡོ་རྗེ་སེམས་དཔའ་
དགའ་བ་ཆེ། །ཀུན་དུ་བཟང་པོ་ཀུན་བདག་ཉིད། །རྡོ་རྗེ་དང་པོའི་བདག་པོའི་གཙོ། །ཡང་དག་ཐོག་མ་མེད་པ་ཡི། །སྐྱེས་བུ་ཆེན་པོ་བཅོམ་ལྡན་འདས། ཨོཾ་བཛྲ་ཨ་བཧི་ཥིཉྩཱ་མི། དྲིལ་བུའི་དམ་ཚིག་སྦྱིན་ཏེ། འདི་དེ་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །ཤེས་རབ་དབྱིངས་ཞེས་འགྲོ་བར་འདོད། །རྟག་ཏུ་
ཁྱོད་ཀྱིས་ཀྱང་ཟུངས་ཏེ། །རྒྱལ་བའི་བྱང་ཆུབ་མཆོག་ཏུ་བཞེད། །ཨོཾ་བཛྲ་དཱཔྟི་དྲིཥྚི་ཨ་བཧི་ཥིཉྩཱ་མི། ཕྱག་རྒྱའི་དམ་ཚིག་སྦྱིན་ཏེ། ཐབས་དང་ཤེས་རབ་གཉིས་མེད་རྒྱ། །མི་ཕྱེད་རྡོ་རྗེའི་ངོ་བོ་ཉིད། །འཁྱུད་པའི་དམ་ཚིག་ཕྱག་རྒྱ་སྟེ། །ཁྱོད་ཀྱིས་ཀྱང་ནི་རྟག་ཏུ་བཅང། །ཨོཾ་ཀཱ་ཡ་མུ་ཏྲ་ཨ་
བཧི་ཥིཉྩ་མི། དཀྱིལ་འཁོར་གྱི་ནི་དེ་ཉིད་དང། །སློབ་དཔོན་དེ་ཉིད་རབ་ཏུ་སྦྱིན། །དེ་བ་མཎྜལ་གུ་རུ་ཨ་བཧི་ཥིཉྩཱ་མི། དེ་ལ་ལྷའི་གནང་བ་སྦྱིན་པའོ། །སེམས་ཅན་རྣམས་ལ་ཕན་དོན་ཕྱིར། །འཇིག་རྟེན་ཁམས་ནི་ཐམས་ཅད་དུ། །ཇི་ལྟ་ཇི་ལྟར་འདུལ་འགྱུར་བ། །དེ་ལྟར་ཆོས་ཀྱི་འཁོར་ལོ་
བསྐོར། །རྗེས་སུ་གནང་བ་སྦྱིན་པའོ། །འདི་ནི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི། །རྡོ་རྗེ་སེམས་དཔའི་ཕྱག་ན་གནས། །རྟག་ཏུ་ཁྱོད་ཀྱིས་ཀྱང་བཟུང་སྟེ། །ལག་ན་རྡོ་རྗེའི་བརྟུལ་ཞུགས་བརྟན། །ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཏིཥྛ་ས་མ་ཡ། ཨ་ཏི་དེ་ཤ་སྟྭཾ་དྷ་ར་ཡ་མི་བཛྲ་སཏ་ཧི་ཧི་ཧི་ཧི་ཧཱུཾ་ཞེས་བརྟུལ་ཞུགས་
སྦྱིན་ནོ། །དེ་ནས་སློབ་དཔོན་གཙོ་བོར་བསྒོམས་ཏེ། ཆོས་གོས་གྱོན་པའི་ཐ་མ་ལག་པ་གཡོན་པའི་ཁུ་ཚུར་གྱིས་སྙིང་གར་བཟུང་ལ། ལག་པ་གཡས་པས་མཆོད་སྦྱིན་དུ་བྱས། །ཀྱེ་འདོད་ཆགས་རྡོ་རྗེ་རྫོགས་སངས་རྒྱས། །ཞེས་བོས་ལ། སྔགས་འདི་བརྗོད་པར་བྱའོ། །སིདྡྷིས་ས་མ་ཡ་སྟྭཾ་
