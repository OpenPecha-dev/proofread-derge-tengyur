ཀྱིས་དགའ་བར་བྱ། །ཨོཾ་བཾ་བ་རུ་ཎ་ཡ་གཱ་ཏི་པ་ཏ་ཡེ་ན་ག་ནཾ་པཪྵ་ཡ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་ཨཾ་ཨ་ནནྟ་ཡ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་བཾ་བ་སུ་ཀེ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་ཏཾ་ཏཀྵ་ཀ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་ཀཾ་ཀརྐོ་ཊ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་པཾ་པདྨ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་མཾ་མ་ཧཱ་པདྨ་ཧཱུྃ་ནཾ་ཧཱུྃ་སྭཱ་ཧཱ། ཨོཾ་སཾ་ཤཾ་

ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ། ཨོཾ་ཨུ་གྲ་ལ་ཤུ་ལ་བ་ནི་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ། ཨོཾ་ཛོ་ཏི་ད་ནི་ད་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཕཊ། ཨོཾ་ཨོཾ་ཨོཾ། མ་ཧཱ་པ་ལཱ་ཡ་སྭཱ་ཧཱ། དེ་ཡི་དབང་དུ་རབ་བྱས་ནས། །ས་གཞི་ཐམས་ཅད་ཆར་གྱི་ནི། །རྒྱུན་གྱིས་ངོམས་པར་བྱས་ནས་ནི། །ཐམས་ཅད་བདེ་ལ་གནས་པར་བྱེད། །

བསྐོར་བར་བྲི། །འཁོར་ལོ་འཁོར་ལོ་ཕྲེང་བསྐོར་ལ། །རྡོ་རྒྱུས་སོགས་ཀྱིས་བྱུགས་བྱས་ཏེ། །ཇི་བཞིན་པ་ཡི་ཆོ་ག་ཡིས། །མེ་ལྷ་ལ་སོགས་མཆོད་པར་བྱ། །དེ་ཡི་དབུས་སུ་སྟོབས་པོ་ཆེ། །འོག་ནས་འབྱུང་བ་བཞིན་དུ་བྱ། །ཕྱོགས་དང་མཚམས་སུ་ཀླུ་བརྒྱད་ཀྱང་། །དེ་ཡི་ཞལ་
བལྟས་ཐལ་སྦྱར་བསམ། །གོང་གི་སྔགས་ཀྱིས་སྤྱན་དྲངས་ནས། །བསང་གཏོར་ཞབས་བསིལ་ལ་སོགས་པ། །བྱས་ནས་དེ་ལ་ཐིམ་པར་བསམ། །དེ་རྣམས་ལྕེ་ཡི་དབུས་སུ་ནི། །ཡི་གེ་ནཾ་ནི་བསམ་བྱས་ཏེ། །བླུགས་གཟར་དགང་གཟར་ཆོ་ག་བཞིན། །བྱས་ནས་བསང་གཏོར་ལ་
