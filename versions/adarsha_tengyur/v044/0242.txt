ཅིག་ནུབ་ཀྱི་ཆའི་ཐིག་གདབ་བོ། །ཡང་བསྐོར་ཕྱོགས་སུ་སོང་སྟེ། རླུང་གི་ཕྱོགས་སུ་འདུག་ལ་སློབ་མ་དབང་ལྡན་གྱི་ཕྱོགས་སུ་འདུག་པ་དང་ལྷན་ཅིག་བྱང་གི་ཆར་ཐིག་གདབ་བོ། །དེའི་རྗེས་ལ་དེ་བཞིན་དུ་མེའི་ཕྱོགས་སུ་སོང་སྟེ། དེ་ཉིད་དུ་འདུག་ལ་སློབ་མ་བདེན་བྲལ་གྱི་ཕྱོགས་
སུ་འདུག་པ་དང་ལྷན་ཅིག་ལྷོའི་ཆའི་ཐིག་གདབ་བོ། །དེ་ཉིད་དུ་འདུག་ལ་སློབ་མ་རླུང་གི་ཕྱོགས་སུ་འདུག་པ་དང་ལྷན་ཅིག་མཚམས་ཀྱི་ཐིག་གདབ་བོ། །དེ་ནས་བདེན་བྲལ་གྱི་ཕྱོགས་སུ་སོང་ལ་ལྷན་ཅིག་བྱང་གི་མཚམས་ཀྱི་ཐིག་གདབ་བོ། །དེ་ཉིད་དུ་འདུག་ལ་སློབ་མ་རླུང་གི་
ཕྱོགས་སུ་འདུག་པ་དང་ལྷན་ཅིག་མཚམས་ཀྱི་ཐིག་གདབ་བོ། །དེ་ནས་བདེན་བྲལ་གྱི་ཕྱོགས་སུ་སོང་ལ། སློབ་མ་དབང་ལྡན་གྱི་ཕྱོགས་སུ་སོང་བ་དང་ལྷན་ཅིག་བྱང་གི་མཚམས་ཀྱི་ཐིག་གདབ་བོ། །ཐིག་དེ་ཉིས་ལྡབ་ཏུ་བྱས་ལ་སློབ་དཔོན་དཀྱིལ་འཁོར་གྱི་དབུས་སུ་འདུག་པས་
བཟུང་ལ། སློབ་མས་ཐིག་གི་ཕྱོགས་གཅིག་ནས་བཟུང་སྟེ། དབང་ལྡན་གྱི་ཕྱོགས་ནས་བརྩམས་ནས་གཡས་ཕྱོགས་ནས་བསྐོར་ཏེ་དེ་དང་ལྷན་ཅིག་རྡོ་རྗེ་ལྟ་བུའི་ཕྱིའི་ཐིག་གདབ་བོ། །སྲད་བུ་དེ་ཉིད་ཡང་ཉིས་ལྡབ་ཏུ་བྱས་ལ་སློབ་དཔོན་རང་ཉིད་ཀྱིས་དབང་ལྡན་གྱི་ཕྱོགས་ནས་བརྩམས་
ཏེ། བསྐོར་ཕྱོགས་སུ་ནང་གི་འཁོར་ལོའི་ཐིག་གདབ་བོ། །དེ་ལྟར་ཐིག་བཏབ་ནས། ཁྲོ་གཉེར་གཉེར་མ་རབ་བསྡུས་པ། །ཁྲོ་བའི་མིག་ནི་བཙུམ་པ་སྟེ། །ཞེས་བྱ་བའི་ཁྲོ་བའི་ལྟ་བ་དང་། མཉམ་པར་གཞག་པའི་ས་ཧྲི་ཞེས་བརྗོད་དེ་སྒོ་བཞི་དབྱེ་བར་བྱའོ། །དེ་ནས་དེ་ཁོ་ན་ཉིད་
བསྡུས་པ་ལས། དེ་ནས་དབུས་སུ་འདུག་ནས་ནི། །རྡོ་རྗེ་སློབ་དཔོན་མཉམ་གཞག་པས། །རྡོ་རྗེའི་སྒོ་ནི་རྣམ་བཞིའོ། །ཡིད་ཀྱིས་སུ་ནི་དབྱེ་བས་བྱ། །ཞེས་གསུངས་པ་ཡིན་ནོ། །དེ་ནས་ཚོན་སྣ་ལྔ་པའི་སྲད་བུ་དེ་ཡི་གེ་མུཿདང་། ཡི་གེ་གསུམ་གྱིས་གཤེགས་སུ་གསོལ་ལ་
གནས་སུ་གཞག་པར་བྱའོ། །ཡེ་ཤེས་ཀྱི་ཐིག་གི་ཆོ་གའི་རིམ་པའོ། །དེ་ནས་རྣམ་པར་སྣང་མཛད་ལ་སོགས་པའི་དེ་བཞིན་གཤེགས་པ་ལྔའི་ཁ་དོག་ཅན་གྱི་ཚོན་རྩི་མུ་ཏིག་གི་ཕྱེ་མ་ལ་སོགས་པའམ། དེ་མ་འབྱོར་ན་འབྲས་སཱ་ལུའི་ཕྱེ་མ་ལེགས་པར་བཏགས་ཤིང་། ལེགས་པར་
