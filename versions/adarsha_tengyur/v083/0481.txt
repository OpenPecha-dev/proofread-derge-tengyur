བར་རིག་པར་བྱའོ། །རབ་འབྱོར་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་ཡང་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཀྱི་ཕྱིར་དད་པ་ཡང་ཡོད། བཟོད་པ་ཡང་ཡོད། དགའ་བ་ཡང་ཡོད། ལྷག་པའི་བསམ་པ་ཡང་ཡོད། འདོད་པ་ཡང་ཡོད། མོས་པ་ཡང་ཡོད།
 འདུན་པ་ཡང་ཡོད། མི་གཏོང་བ་ཡང་ཡོད། བརྩོན་འགྲུས་མི་འདོར་བ་ཡོད་མོད་ཀྱི། དེ་དག་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པས་ཡོངས་སུ་མ་བཟུང་། ཐབས་མཁས་པས་ཡོངས་སུ་མ་བཟུང་། བསམ་གཏན་གྱི་ཕ་རོལ་ཏུ་ཕྱིན་པས་ཡོངས་སུ་མ་བཟུང་། བརྩོན་འགྲུས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པས་
ཡོངས་སུ་མ་བཟུང་། བཟོད་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པས་ཡོངས་སུ་མ་བཟུང་། ཚུལ་ཁྲིམས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པས་ཡོངས་སུ་མ་བཟུང་། སྦྱིན་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པས་ཡོངས་སུ་མ་བཟུང་། སྟོང་པ་ཉིད་ཐམས་ཅད་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་སུམ་ཅུ་རྩ་བདུན་གྱིས་
ཡོངས་སུ་མ་བཟུང་། ཚད་མེད་པ་དང་། བསམ་གཏན་དང་། གཟུགས་མེད་པའི་སྙོམས་པར་འཇུག་པ་རྣམས་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། འཕགས་པའི་བདེན་པ་རྣམས་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། རྣམ་པར་ཐར་པ་བརྒྱད་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། མཐར་གྱིས་གནས་པའི་སྙོམས་
པར་འཇུག་པ་དགུས་ཡོངས་སུ་མ་བཟུང་། སྟོང་པ་ཉིད་དང་། མཚན་མ་མེད་པ་དང་། སྨོན་པ་མེད་པས་ཡོངས་སུ་མ་བཟུང་། ཏིང་ངེ་འཛིན་ཐམས་ཅད་དང་། གཟུངས་ཀྱི་སྒོ་རྣམས་དང་། མངོན་པར་ཤེས་པ་རྣམས་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། དེ་བཞིན་གཤེགས་པའི་སྟོབས་བཅུས་
ཡོངས་སུ་མ་བཟུང་། མི་འཇིགས་པ་བཞི་རྣམས་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། སོ་སོ་ཡང་དག་པར་རིག་པ་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་བཅོ་བརྒྱད་རྣམས་ཀྱིས་ཡོངས་སུ་མ་བཟུང་། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་བར་གྱིས་ཡོངས་སུ་མ་བཟུང་ན། རབ་འབྱོར་བྱང་ཆུབ་
སེམས་དཔའ་དེ་ནི་བར་མ་དོར་ཉེ་ཞོར་འགྱུར་ཏེ། འདི་ལྟར་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་རིན་པོ་ཆེའི་ཕུང་པོའི་དོན་ཆེན་པོ་ལས་ཡོངས་སུ་ཉམས་པར་འགྱུར་བར་རིག་པར་བྱའོ། །རབ་འབྱོར་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་རྣམས་ཀྱི་བར་མ་དོའི་ཉེ་ཞོ་གང་ཞེ་ན། འདི་ལྟ་སྟེ་ཉན་ཐོས་ཀྱི་
