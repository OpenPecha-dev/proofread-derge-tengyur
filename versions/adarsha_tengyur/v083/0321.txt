རྨི་ལམ་དང་། སྒྲ་བརྙན་དང་། མིག་ཡོར་དང་། གཟུགས་བརྙན་དང་། དྲི་ཟའི་གྲོང་ཁྱེར་རྣམས་ལ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བར་ནུས་སམ། བརྒྱ་བྱིན་གྱིས་སྨྲས་པ། བཙུན་པ་རབ་འབྱོར་དེ་ནི་མི་ནུས་སོ། །རབ་འབྱོར་གྱིས་སྨྲས་པ། ཀོོ་ཤི་ཀ་དེ་བཞིན་དུ་བྱང་
ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སྤྱོད་པ་རྣམས་ལ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བར་གང་སེམས་པ་དེ་ནི་བསམས་པ་ལྟར་མི་འགྲུབ་ཅིང་དུབ་པའི་སྐལ་བ་ཅན་དུ་འགྱུར་རོ། །ཀཽ་ཤི་ཀ་དེ་ཇི་སྙམ་དུ་སེམས། ཁྱོད་ཀྱིས་དེ་བཞིན་གཤེགས་པའམ།
 དེ་བཞིན་གཤེགས་པའི་སྤྲུལ་པ་ལ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བར་ནུས་སམ། བརྒྱ་བྱིན་གྱིས་སྨྲས་པ། བཙུན་པ་རབ་འབྱོར་དེ་ནི་མི་ནུས་སོ། །རབ་འབྱོར་གྱིས་སྨྲས་པ། ཀཽ་ཤི་ཀ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སྤྱོད་པ་
རྣམས་ལ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བར་གང་སེམས་པ་དེ་ནི་བསམས་པ་ལྟར་མི་འགྲུབ་ཅིང་དུབ་པའི་སྐལ་བ་ཅན་དུ་འགྱུར་རོ། །ཀཽ་ཤི་ཀ་དེ་ཇི་སྙམ་དུ་སེམས། ཁྱོད་ཀྱིས་ཆོས་ཀྱི་དབྱིངས་དང་། ཡང་དག་པའི་མཐའ་དང་། དེ་བཞིན་ཉིད་དང་། བསམ་གྱིས་མི་ཁྱབ་པའི་དབྱིངས་ལ་
བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་བར་ནུས་སམ། བརྒྱ་བྱིན་གྱིས་སྨྲས་པ། བཙུན་པ་རབ་འབྱོར་དེ་ནི་མི་ནུས་སོ། །རབ་འབྱོར་གྱིས་སྨྲས་པ། ཀཽ་ཤི་ཀ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སྤྱོད་པ་རྣམས་ལ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་། སྦ་
བར་གང་སེམས་པ་དེ་ནི་བསམས་པ་ལྟར་མི་འགྲུབ་ཅིང་དུབ་པའི་སྐལ་བ་ཅན་དུ་འགྱུར་རོ་ཞེས་བྱ་བ་ནི་གཞན་གྱི་རྐྱེན་གྱིས་མི་འགྲོ་བའི་སྦྱོར་བ་ཡིན་ནོ།    །བརྒྱ་བྱིན་གྱིས་སྨྲས་པ། བཙུན་པ་རབ་འབྱོར་ཇི་ཙམ་གྱིས་ན་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སྤྱོད་ན་
ཆོས་ཐམས་ཅད་རྨི་ལམ་ལྟ་བུ་དང་། སྒྱུ་མ་ལྟ་བུ་དང་། སྨིག་རྒྱུ་ལྟ་བུ་དང་། སྒྲ་བརྙན་ལྟ་བུ་དང་། གཟུགས་བརྙན་ལྟ་བུ་དང་། དྲི་ཟའི་གྲོང་ཁྱེར་ལྟ་བུ་དང་། སྤྲུལ་པ་ལྟ་བུར་ཆོས་ཐམས་ཅད་ཡོངས་སུ་ཤེས་པར་འགྱུར་བ་ཡིན། རབ་འབྱོར་གྱིས་སྨྲས་པ། གང་གི་ཚེ་བྱང་ཆུབ་སེམས་དཔའ་
