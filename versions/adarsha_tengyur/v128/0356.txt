བྱེད་པ་དང་། ཚིག་ཉུང་ངུ་སྨྲ་བ་དང་། རབ་ཏུ་ཞི་བར་སྨྲ་བ་དང་། བྲེ་མོ་གཏམ་ལ་མི་དགའ་བ་དང་། འདུ་འཛི་ལ་མི་དགའ་བ་དང་། དེའི་ཚིག་བརྟན་པར་འབྱུང་བ་དང་། མིག་གིས་གཟུགས་རྣམས་མཐོང་ནས་གཟུགས་སོ་སོར་ཡང་དག་པར་རིག་པར་བྱེད་ཅིང་གཟུགས་ལ་འདོད་ཆགས་ཀུན་དུ་
མི་འབྱུང་བ་དང་། དེ་བཞིན་དུ་སྒྲ་དང་། དྲི་དང་། རོ་དང་། རེག་བྱ་སོ་སོར་ཡང་དག་པར་རིག་པར་བྱེད་ཅིང་། རེག་བྱའི་བར་ལ་འདོད་ཆགས་ཀུན་དུ་མི་འབྱུང་བ་དང་། འཇིགས་པ་མེད་པ་དང་། བློ་ཟབ་པ་དང་། ཤིན་ཏུ་སྦྱངས་པ་ཆེན་པོས་ལུས་དང་སེམས་ཉེ་བར་བསྐོར་བ་དང་། བརྣབ་
སེམས་མེད་པ་དང་། འཁྲུག་པ་མེད་པ་དང་། བཟོད་པ་དང་ལྡན་པ་དང་། དེའི་སེམས་ལ་འདོད་པའི་རྣམ་པར་རྟོག་པ་ལ་སོགས་པས་ཀུན་ནས་འཁྲུག་པ་མེད་པ་དང་། དེ་ལྟ་བུ་དང་མཐུན་པ་དག་ནི་འདོད་ཆགས་དང་བྲལ་བའི་རྟགས་རྣམས་ཡིན་པར་རིག་པར་བྱ་སྟེ། དེ་ནི་རེ་ཞིག་འཇིག་རྟེན་པའི་
ལམ་གྱིས་འགྲོ་བའི་རྣམ་པ་དབྱེ་བ་ཡིན་ནོ།    །གལ་ཏེ་དེ་འཇིག་རྟེན་ལས་འདས་པའི་ལམ་གྱིས་འགྲོ་བར་འདོད་ན་ནི། འཕགས་པའི་བདེན་པ་བཞི་པོ་དག་ལས་བརྩམས་ནས་དགྲ་བཅོམ་པ་ཉིད་ཐོབ་ཀྱི་བར་དུ་དེ་ལ་མཚན་ཉིད་སོ་སོར་རིག་པའི་ཡིད་ལ་བྱེད་པ་ལ་སོགས་པ་ནས་སྦྱོར་བ་མཐར་ཐུག་པའི་
འབྲས་བུ་ཡིད་ལ་བྱེད་པའི་བར་གྱི་ཡིད་ལ་བྱེད་པ་བདུན་པོ་དག་རིམ་གྱིས་འབྱུང་བར་འགྱུར་ཏེ། དེ་ལ་འཕགས་པའི་བདེན་པ་བཞི་པོ་མདོར་བསྟན་པ་དང་། རྒྱས་པར་བཤད་པ་དང་བཅས་པ་ཐོས་ཤིང་བཟུང་ལ་ལེགས་པར་བསྒོམས་ཤིང་ཡིད་ལ་བྱེད་པའམ། བསམ་གཏན་དང་གཟུགས་མེད་
པའི་དངོས་གཞི་ཐོབ་པའི་རྣལ་འབྱོར་སྤྱོད་པ་ལ་ནི་རྣམ་པ་བཞི་པོ་འདི་ལྟ་སྟེ། མི་རྟག་པའི་རྣམ་པ་དང་། སྡུག་བསྔལ་བའི་རྣམ་པ་དང་། སྟོང་པའི་རྣམ་པ་དང་། བདག་མེད་པའི་རྣམ་པས་བདེན་པ་སྡུག་བསྔལ་གྱི་མཚན་ཉིད་སོ་སོར་རིག་པར་བྱེད་དོ། །རྣམ་པ་བཞི་པོ་འདི་ལྟ་སྟེ། རྒྱུའི་
རྣམ་པ་དང་། ཀུན་འབྱུང་བའི་རྣམ་པ་དང་། རབ་ཏུ་སྐྱེ་བའི་རྣམ་པ་དང་། རྐྱེན་གྱི་རྣམ་པས་བདེན་པ་ཀུན་འབྱུང་བའི་མཚན་ཉིད་སོ་སོར་རིག་པར་བྱེད་དོ། །རྣམ་པ་བཞི་པོ་འདི་ལྟ་སྟེ། འགོག་པའི་རྣམ་པ་དང་། ཞི་བའི་རྣམ་པ་དང་། གྱ་ནོམ་པའི་རྣམ་པ་དང་། ངེས་པར་འབྱུང་བའི་
