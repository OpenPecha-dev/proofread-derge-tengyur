རྒྱ་གྲམ་ལ། །འཁོར་ལོ་རྩིབས་བཅུ་འབར་བས་མཚན། །ཟླ་བའི་སྟེང་དུ་གནོད་མཛེས་སྤྲུལ། །ཡི་གེ་ཧཱུཾ་གི་འཕྲོ་འདུ་ལས། །ཁྲོ་བོ་བཅུ་སྤྲུལ་ཧཱུཾ་བཞི་ཡི། །གསང་སྔགས་བཟླས་ནས་བཀའ་བསྒོ་བྱ། །ཁྲོ་བོ་སྤྲུལ་པས་བགེགས་རྣམས་བཀུག །ཡང་སྒོ་སྐུ་གསུང་ཐུགས་ཀྱི་ནི། །ཕུར་བུའི་
སྔགས་བརྗོད་ཐོ་བས་བརྡུང་། །བསྲེགས་ནས་ནམ་མཁའ་ཉིད་དུ་འགྱུར། །དེ་ནས་ཧཱུཾ་གི་འཕྲོ་འདུ་ཡིས། །རྡོ་རྗེ་གུར་དང་ར་བར་འགྱུར། ཡི་གེ་བྷྲཱུཾ་གྱི་འཕྲོ་འདུ་ལས། །གནོད་མཛེས་འོད་དུ་ཞུ་ནས་ཀྱང་། །གཞལ་ཡས་ཁང་ནི་མཛེས་པར་གྱུར། །དེ་སྟེང་པདྨ་འདབ་བརྒྱད་གདན། །རང་
གི་སེམས་ཀྱི་ཆོ་འཕྲུལ་ལས། །ཟླ་བའི་དཀྱིལ་འཁོར་ཆེན་པོ་ལ། །ཡི་གེ་ཨཱ་ལིས་མཚན་པར་བྱ། །ཉི་མའི་དཀྱིལ་འཁོར་ཆེན་པོ་ལ། །ཡི་གེ་ཀཱ་ལིས་མཚན་པར་བྱ། །ཧཱུཾ་ལས་རྡོ་རྗེ་དེ་ཉིད་མཚན། །བྱིན་རླབས་བསྡུས་ནས་འོད་དུ་ཞུ། །དེ་ལས་སྐུ་ནི་བསྐྱེད་པར་བྱ། །པདྨ་དབུས་ཀྱི་ལྟེ་བ་
