ཡན་ཆད་ཀྱིས་རྒྱུ་བསྟན་ཏོ། །གོང་དུ་ལྷའི་མིག་གི་སྐབས་སུ་བཤད་པ་དང་དོན་མཐུན་པའི་ཚིག་རྣམས་འདིར་མ་བཤད་དོ། །སེམས་ཀྱི་དམིགས་པ་དགེ་བ་དང་མི་དགེ་བ་དང་། ལུང་དུ་མི་སྟོན་པ་གང་དག་གིས་སེམས་ཅན་དེ་དག་གི་ངག་གི་ལས་འབྱུང་བ་ཐམས་ཅད་ཀྱང་ཡང་དག་པ་
ཇི་ལྟ་བ་བཞིན་དུ་རབ་ཏུ་ཤེས་སོ་ཞེས་བྱ་བ་ནི་སེམས་ཅན་དེ་དག་གི་ཚིག་རབ་ཏུ་སྨྲ་བ་དེའི་རྒྱུ་ནི་སེམས་ཀྱི་དགེ་བ་དང་མི་དགེ་བ་དང་ལུང་དུ་མི་སྟོན་པའི་ཆོས་ལ་དམིགས་པ་ཡིན་ཏེ། དེ་དག་གི་སེམས་ཀྱི་ཆོས་གང་ལ་དམིགས་ནས་ངག་གི་ལས་ཚིག་ཏུ་སྨྲ་བ་ནི་སྒྲ་གང་བྱུང་བའི་རྒྱུ་རྣམས་
ཀྱང་རབ་ཏུ་ཤེས་སོ་ཞེས་བྱ་བའི་དོན་ཏོ། །རྒྱུ་གང་ལས་བྱུང་བ་ཤེས་པ་བཤད་ཟིན་ཏོ། །ངག་གི་ལས་གང་དགེ་བའི་རྒྱུ་ཡོངས་སུ་འཛིན་པ་ལ་སོགས་པ་ནི་སེམས་ཅན་དེ་དག་གིས་ཚིག་ཏུ་སྨྲ་བ་དེ་དག་ཐོས་ནས་ཚིག་དེ་དག་སྨྲས་པའི་རྒྱུས་འབྲས་བུ་ཅི་དང་ཅིར་འགྱུར་བ་ཕྱིན་ཅི་མ་
ལོག་པར་ཤེས་པའི་དོན་ཏོ། །འབྲས་བུ་གང་བྱུང་བ་ཡང་བཤད་ཟིན་ཏོ། །འདོད་ཆགས་བག་ལ་ཉལ་བའི་ངག་གི་ལས་གང་སྡང་བ་དང་འབྱིན་པ་དང་ཞེས་བྱ་བ་ལ་སོགས་པས་ནི་འདོད་ཆགས་དང་ཞེ་སྡང་དང་གཏི་མུག་གསུམ་ལས་གཅིག་ནི་ཚིག་མངོན་པར་སྨྲ་བའི་རྒྱུའི་གཙོ་བོ་
ཡིན་པར་སྟོན། གཅིག་ནི་གྲོགས་ཡིན་པར་སྟོན་ཏེ། དེ་ལ་འདོད་ཆགས་བག་ལ་ཉལ་ལ་སྡང་བས་འབྱིན་པ་ནི་ཚིག་འབྱུང་བའི་གཞི་འདོད་ཆགས་ལས་བླངས་ལ། འདོད་ཆགས་དངོས་སུ་མ་བྱུང་བར་བག་ལ་ཉལ་ཏེ་ཞེ་སྡང་དངོས་སུ་བྱུང་བས་ཚིག་ཏུ་སྨྲས་པས་ཤེས་སོ་ཞེས་བྱ་བའི་
དོན་ཏོ། །དེ་བཞིན་དུ་ལྷག་མ་རྣམས་ལ་ཡང་སྦྱར་རོ། །ངག་གི་ལས་གང་བསམ་པ་དག་ལ་སྦྱོར་བ་ཉམས་པ་དང་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། བསམ་པ་མ་ནོར་ལ་ཚིག་ཏུ་སྨྲ་བའི་སྦྱོར་བ་འཁྲུལ་ཞིང་ནོར་བ་དང་། ཚིག་ཏུ་སྨྲ་བའི་སྦྱོར་བ་མ་ནོར་མ་འཁྲུལ་ལ། བསམ་པ་འཁྲུལ་
པ་ལ་སོགས་པ་ཕྱིན་ཅི་མ་ལོག་པར་ཤེས་པ་ལ་བྱའོ། །རྒྱུ་དང་འབྲས་བུ་མི་མཐུན་པར་ཤེས་པ་བཤད་ཟིན་ཏོ། །འཕགས་པའི་སྒྲ་ནི་ཉན་ཐོས་དང་རང་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་དང་། དེ་བཞིན་གཤེགས་པ་རྣམས་གསུངས་པ་དང་། ཆོས་འཆད་པའི་སྒྲའོ། །
