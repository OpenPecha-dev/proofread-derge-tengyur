རྣམས་སོ། །ཙཀྵ་བརྗོད་པ་ལའོ། །ཨཱི་ར་འགྲོ་བ་ལའོ། །ཨཱི་ཌ་བསྟོད་པ་ལའོ། །ཨཱི་ཤ་དབང་ཕྱུག་ཉིད་ལའོ། །ཨཱས་ཉེ་བར་སྡོད་པ་ལའོ། །ཨཱང་ལས་ཤཱ་སུ་འདོད་པ་ལའོ། །བ་ས་ཀུན་ནས་འགེབས་པ་ལའོ། །ཀ་སི་འགྲོ་བ་ལའོ། །ཎི་སིའོ་བྱེད་པ་ལའོ། །ཎི་ཛི་དག་པ་ལའོ། །ཤི་
ཛི་སྒྲ་ལའོ། །བྲྀ་ཛཱི་འདོར་བ་ལའོ། །བྲྀ་ཙཱི་སྤེལ་བ་ལའོ། །ཥཱུང་རབ་ཏུ་སྐྱེ་བ་ལའོ། །ཤྲཱིང་ཉལ་བ་དག་ལའོ། །ཨིང་ཚིག་འདོན་པ་ལའོ། །དཱི་དྷཱིང་འབར་བ་ལའོ། །བེ་བཱིང་འགྲོ་བ་ལའོ། །ཧྣུང་སེལ་བ་ལའོ། །ཏང་ཨཱ་ན་རྣམས་སོ། །དྭི་ཥ་མི་དགའ་བ་ལའོ། །དུ་ཧ་འཇོ་བ་ལའོ། །དི་ཧ་ཉེ་བར་སོགས་པ་
ལའོ། །ལི་ཧ་རོ་མྱང་བ་ལའོ། །ཨཱུརྞྞཉ་འགེབས་པ་ལའོ། །ཥྚུ་བསྟོད་པ་ལའོ། །བྲཱ་བརྗོད་པ་ལའོ། །རྣམ་པར་བཤད་པ་རྣམས་སོ། །ཤས་འཇིག་པ་རྣམས་ཨད་ལ་སོགས་པ་རྫོགས་སོ།།      །།ཧུ་སྦྱིན་པ་ལའོ། །བྷཱི་འཇིགས་པ་ལའོ། །ཧྲི་ངོ་ཚ་བ་ལའོ། །པྲཱྀ་སྐྱོང་བ་ལའོ། །ཨེ་ཧཱ་ཀ་གཏོང་བ་
ལའོ། །གྷྲྀ་འགྱུར་བར་བྱེད་པ་ལའོ། །རྀ། སྲྀ་འགྲོ་བ་ལའོ། །བྷ་ས་བརྒྱད་བཀག་པ་ལའོ། །ཀི་ཀིཏ་ཤིས་པ་ལའོ། །ཏུ་ར་མགྱོགས་པ་ལའོ། །དྷི་ཥ་སྒྲ་ལའོ། །དྷ་ན་སོ་བ་ལའོ། །ཛ་ན་སྐྱེད་པར་བྱེད་པ་ལའོ། །ཏང་ཨཱ་ན་མ་ཡིན་པ་རྣམས་སོ། །ཎི་ཛིར་གཙང་བ་ལའོ། །བི་ཛིར་ཐ་དད་པར་གྱུར་
པ་ལའོ། །བི་ཥླྀ་ཁྱབ་པ་ལའོ། །ཌུ་དཱཉ་སྦྱིན་པ་ལའོ། །ཌུ་དྷཱཉ། ཌུ་བྷྲྀཉ་འཛིན་པ་ལའོ། །རྣམ་པར་བཤད་པའོ། །མཱང་ཚད་ལའོ། །ཨོ་ཧཱང་འགྲོ་བ་ལའོ། །ཏང་ཨཱ་ན་དག་གོ། །ཤས་འཇིག་པ་རྣམས་སུ་ཉོ་ཏི་ལ་སོགས་པ་རྫོགས་སོ།།      །།དི་བུ་རྩེ་བ་ལའོ། །ཥི་བུ་སྣལ་མའི་རྒྱུད་ལའོ། །ཤྲྀ
བུ་འགྲོ་བ་ལའོ། །ཥྛི་བུ་སེལ་བ་ལའོ། །ཀྣ་སུ་དྲག་པོར་འབོད་པ་ལའོ། །ཎྲྀ་ཏཱི་ཟློས་གར་ལའོ། །ཏྲ་སཱི་འཇིགས་པ་ལའོ། །ཀུ་ཐ་རུལ་བར་གྱུར་པ་ལའོ། །ཥུ་ཐ་འཚེ་བ་ལའོ། །གྷུརྡ་དཀྲིས་པ་ལའོ། །ཀྵི་པ་དོད་པ་ལའོ། །པུཥྤ་རྣམ་པར་རྒྱས་པ་ལའོ། །ཏི་མ། ཥྚི་མ། ཥྚཱི་མ་གཤེར་བར་གྱུར་པ་ལའོ། །བྲཱི་ཊ་
