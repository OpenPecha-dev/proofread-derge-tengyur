དམིགས་བསལ་ལོ། །ནུས་པ་གསལ་བར་བྱེད་པ་ནི་རྒྱུ་སྟེ། བིདྱཱ་ཡ་ཤསྐ་རཱི། ངང་ཚུལ་ནི་རང་བཞིན་ཏེ། མྲྀནྨརྡ་ན་ཀ་རཿ། ཏྲྀ་ཎ་ཙྪེ་ད་ཀ་རཿ། རྗེས་སུ་མཐུན་པ་ནི་རྗེས་སུ་འགྲོ་བ་སྟེ། བ་ཙ་ན་ཀ་རཿ། དེ་བཞིན་དུ། དི་བཱ། བི་བྷཱ། པྲ་བྷ། བྷཱས་ཀཱ་ར། ཨ་ནནྟ། ཨཱ་ཧ་ར། ཨ་དི།
 བ་ཧུ། ལི་པི། ལི་བི། བ་ལི། བྷཀྟི། ཀརྟྲྀ། ཙིཏྲ། ཀྵེ་ཏྲ། དྷ་ནུས། ནཱ་ཌི། ཨ་རུས། སཾ་ཁྱཱ། དི་ན། དི་བས། ཀྵ་པཱ། ཀྵ་ཎ་དཱ་རཱ་ཏྲི། ར་ཛ་ནཱི། གྷ་ཊཱི་རྣམས་ལས་སོ། །ཀྱང་སྟེ་ཉི་ཤུ་རྩ་བདུན་པོ་འདི་རྣམས་ལས་ཕ་རོལ་ཀ་རོ་ཏི་ལས་ཊ་བསྒྲུབ་པར་བྱའོ། །དི་པའི་སྒྲ་ནི་གཞི་ཉིད་ལས་གྲངས་མེད་པ་
