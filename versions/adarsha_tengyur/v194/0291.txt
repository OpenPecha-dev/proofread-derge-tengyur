ཐལ་བ་མི་ཤེས་པས་གྲུབ་བོ། །དཔེར་ན། བྷིནྡྷ་ཀི་ཞེས་པ་འདིར་དྷིའི་ངོ་བོའི་བྱ་བ་ལ་ཕྱི་ནས་ཨཀའོ། །ཨི་ཡ་རུ་ཞེས་པ་འདིར་ཀྲྀཏ་ལ་ཡ་ཎ་གཉིས་བརྗོད་པའོ། །སྤྱིར་བཏང་བ་དང་དམིགས་ཀྱིས་བསལ་བ་དང་རྟག་པ་དང་མི་རྟག་པ་དང་ནང་གི་ཡན་ལག་དང་ཕྱིའི་ཡན་ལག་རྣམས་ལ་བལ་
མ་བཀག་པར་མཚུངས་པའི་ཕྱིར་འདིའི་ཡུལ་ལ་ཨིཊ་བལ་དང་འདྲ་བ་ཁོ་ནར་འགྱུར་རོ། །ཨིཏ་འགྲོ་བ་བྱ་བའི་དོན་ཏོ། །འདིར་ཨིཏ་ནི་བྱ་བའི་དོན་གྱི་ཡི་གེ་གང་ཡིན་པ་སྟེ། རྟགས་ཀྱི་དོན་དང་བརྗོད་པའི་དོན་ཡང་མ་ཡིན་ནོ། །དེ་འགྲོ་བ་ཞེས་རིག་པར་བྱ་སྟེ། ཌུ་འགྲོ་བ་ལས་ཀྟྲིའོ། །པཀྟྲི་མ། ཌུ་འགྲོ་
བ་ལས་ཨ་ཐུ་ཙའོ། །ཤྭ་ཡ་ཐུཿ། ཏཱ་ཏང་། ཛུ་ཧུ་ཏཱཏ། ཨིཏ་སྨོས་པ་ཅི་ཞེ་ན། ཨགྣི་ཙིདྦྲཱ་ཧྨ་ཎོ་བྲཱཧྨ་ཎཱི་ཞེས་པ་འདིར་ནུཾ་དང་ཏཱིཔ་དག་ཏུ་ཐལ་བ་ལ་ཡཱ་ཏཱཾ། ངེས་པ་མེད་པ་ཉིད་ཀྱིས་ལྡོག་པ་ཉེ་བར་བསྟན་པའི་གནས་སྐབས་སུ་ཨུཀ་འགྲོ་བ་ཉིད་ལྡོག་པའི་ཡུལ་ཉིད་ཀྱིས་གཙོ་བོའི་ཕྱིར་རོ། །མི་བདེན་པར་
བཤད་པ་ནི་དོན་གྱི་ཤུགས་ཀྱིས་བསྒྲུབས་པ་ཉིད་ཀྱིས་འདི་སུ་ན་ཕྱུང་ངོ་། །གནས་སུ་ཆེས་གཞན་ནོ། །མང་པོ་རྣམས་སུ་མ་ངེས་པ་བསྒྱུར་བ་ལ་གནས་སུ་ཆེས་འདྲ་བ་ཉིད་དུ་འགྱུར་རོ། །ཨཀ་ནི་འདྲ་བའི་ཡི་གེ་ལ་རིང་པོའོ་ཞེས་འཆད་དོ། །ལ་ཏཱ་གྲཾ། དེ་ཡང་མགྲིན་པ་དང་མཆུ་དང་རྐན་ལ་སོགས་པ་ལས་
སྐྱེས་པའི་དབྱངས་རིང་པོ་ཉིད་ལ་མགྲིན་པའི་གནས་ལ་མགྲིན་པ་ལས་སྐྱེས་པ་དང་ཆེས་འདྲ་བ་ཉིད་དུ་འགྱུར་རོ། །ཨ་མཱུ་ཨ་མུཥྨེ་ཞེས་པ་འདིར་ཕྱི་མོ་གཉིས་པ་དང་ཕྱི་མོ་གཅིག་པ་དག་གི་གནས་སུ་ཕྱི་མོ་གཉིས་པ་དང་ཕྱི་མོ་གཅིག་པ་དག་ཉིད་དེ། ཕྱི་མོ་གཅིག་པ་མ་ཡིན་ཞིང་ཕྱི་མོ་གསུམ་པ་ཡང་མ་
ཡིན་ནོ། །གནས་སུ་ཞེས་པ་ཅི་ཞེ་ན། ཙེ་ཏཱ། སྟོ་ཏཱ། གནས་ཀྱི་དོན་ནི་ཡོན་ཏན་གྱི་ཚད་རྣམས་ལ་གནས་གཞན་ཇི་ལྟ་བར་འགྱུར་རོ། །ཚད་གཞན་ལས་ཨ་ཡིག་ཏུ་མི་འགྱུར་རོ། །ཏ་མས་སྨོས་པ་ཅི་ཞེ་ན། མང་པོ་རྣམས་ལས་ཆེས་གཞན་པ་ལ་ཏ་མས་ཇི་ལྟ་བ་བཞིན་འགྱུར་རོ། །བཱགྦྷ་ས་ཏི། ཏྲིཥྚུ་བྦྷ་
ས་ཏི། ཡི་གེ་རྣམས་ཀྱི་ཡི་གེ་ལས་ཆེས་གཞན་པ་ནི་སློབ་མའི་མདོར་རྟོགས་པར་བྱའོ། །ཨེ་ཙའི་ཐུང་ངུ་ནི་ཨིཀའོ། །ཨེ་ཙའི་ཐུང་ངུར་བསྒྱུར་བ་ནི་ཨིཀ་ཉིད་དུ་འགྱུར་ཏེ། ཐུང་ངུའི་ཡན་ལག་གིའོ་ཞེས་འཆད་དོ། །ཨ་རི་རི། ཨ་ཏི་ནུ། འདིར་ཡང་མགྲིན་པ་དང་རྐན་ལས་བྱུང་བ་དག་གི་མགྲིན་པ་ལས་
