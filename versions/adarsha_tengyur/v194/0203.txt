ཤཱཥྱཿ། སི་ཥ་ཏཿ། པ་ཏི་མཱ་སམྦྷཱཪྻཱ་སཱཥ་ཡི་ཏ་བྱཿ། སཱ་ས་ནཱི་ཡཿ། སཱ་སྱཿ། སཱ་ཥི་ཏཿ། ས་ཏི་མཱ་སོ་བྷཱཪྻཱཿ། ཡོ། བྷཪྻ་ཡཱ་བཱ། ཨཱ་ས་ཡི་ཏ་བྱ། ཨཱ་ས་ཡི་ནཱ་ཡ། ཨ་སྱ། ཨ་སི་ཏི། པ་ཏི་མཱ་སམྦྷཱཪྻཱ། ཨཱ་ས་ཡི་ཏ་བྱ། ཨཱ་སི་ནཱ་ཡཿ། ཨཱ་སྱ། ཨཱ་སི་ཏཿ། པ་ཏིམྨཱ་སོ་བྷཱཪྻཱ་ཡཱ་ནཱ་ཡི་ཏ་བྱོ། ནཱ་ཡ་ནཱི་ཡོ་
ནཱ་ཡྱོ། ནཱ་ཡི་ཏོ་གྲ་མནྡེ་བ་དཏྟེ་ན། ཡ་ཛྙ་དཏྟེ་ན། ཨ་ད་ཡི་ཏ་བྱཿཨད་ནཱི་ཡ། ཨཱ་དྱ། ཨ་དི་ཏ། ཨོ་ནམྦཱ་ལ་བིཥྞུ་མི་ཏྲི་ཎ། བཱ་ཧ་ཡི་ཏ་བྱོ། བཱ་ཧ་ནཱི་ཡོ། བཱ་ཧྱོ། བཱ་ཧི་ཏེ། བྷཱ་རངྒྲཱ་མཾ་དེ་བ་དཏྟེ་ན། ཁཱ་ད་ཡི་ཏ་བྱཿ། ཁཱ་ད་ནཱི་ཡཿ། ཁཱ་དྱཿ། ཁཱ་དི་ཏོ། གྷཱ་ས་མ་སྭོ་དེ་བ་དཏྟེ་ན། ཨཱ་ད་ཡི་ཏ་བྱ། ཨཱ་ད་ནི་ཡ། ཨཱ་དྱཿ། ཨཱ་དི་
ཏཿ། བྷ་སོ྅སྲཾ་དེ་བ་དཏྟེ་ན། བཱ་ཧ་ཡི་ཏ་བྱོ། །བཱ་ཧ་ནཱི་ཡོ། བཱ་ཧྱོ། བཱ་ཧི་ཏཱ། བྷཱ་རངྒྲ་མམྦ་ལཱི་བརྡྷཱ་དེ་བ་དཏྟེ་ན། བཀྵ་ཡི་ཏ་བྱོ། བཀྵ་ནཱི་ཡོ། བཀྵྱོ། བཀྵི་ཏོ་བཱ་སཎྜཱི། དེ་བ་དཏྟོ་ཡཛྙ་དཏྟེ་ན། བཀྵི་ཏ་བྱོ། བཀྵི་ཎཱི་ཡོ། བཀྵྱོ། བཀྵི་ཏཱ། བཱ་བ་ལཱི་བརྡྷཱ་ཡ་བནྡེ་བ་དཏྟེ་ན། ཞེས་པ་ལ་སོགས་པ་རྣམས་ལ། 
ཏ་བྱ། ཨ་ནཱི་ཡར། ཀེ་ལ་མར་རྣམས་སོ། །ཡ་ཏའོ། །རི་དང་ཧ་ལས་ཎྱ་ཏའོ། །དངོས་པོ་དང་ལས་དག་ལ་ཞེས་པས་ཀྟའོ། །བྱ་བའི་དོན་ཨིག་གི་ཨ་ཏ་ཨེང་ངོ་ཞེས་པས་ཨཏ་ཨེང་ངོ་། །ཨེ་ཙའི་ཨ་ཡ་ཨ་ཝ་ཨཱ་ཡ་ཨཱ་ཝ་རྣམས་སོ། །སྦྱོར་བ་པོའི་བྱ་བ་ལ་ཎི་ཙའོ། །ཨ་ཙའི་ཞེས་པས་ཨཱ་
ཨི་ཙའོ། །རྀ་ཀའི་ཨ་ཎ་ལ་ར་ལ་དག་གོ། །ཎའི་ཨིཊ་མ་ཡིན་པ་ལའོ། །ཏ་ཏ་བ་ཀཱི་ཀི་ཎི་བྱའོ། །ཝལ་དང་པོའི་ཨིཊའོ། །ཚིག་གཅིག་ལ་ར་ཥ་ལས་ནའི་ཎའོ་ཞེས་པ་ལ་སོགས་པའི་ཚིག་གིས་སྒྲུབ་པ་རྣམས་ཇི་ལྟར་འབྱུང་བ་བཞིན་བསྒྲུབ་པར་བྱའོ་ཞེས་པ་འབྲེལ་པ་གྲུབ་པ་ལས་གཉིས་དང་ལྡན་
པའི་བྱིངས་རྣམས་ཀྱི་གཙོ་བོ་དང་གཙོ་བོ་མ་ཡིན་པ་དང་བཤད་པ་དང་མ་བཤད་པའི་ལས་རྣམ་པར་དབྱེའོ།།      །།ཏིང་གི་རབ་ཏུ་སྦྱོར་བ་རྣམས་མངོན་པར་བརྗོད་ནས། དེའི་འོག་ཏུ་སངའི་རབ་ཏུ་སྦྱོར་བ་སྐབས་སུ་བབ་པས་བཤད་པར་བྱ་སྟེ། སང་དང་གསུམ་རྣམས་ཀྱང་འཇུག་སྟེ། །དངོས་དང་
དེའི་ལས་བྱེད་པོ་ལ། །ཞེས་པའི་དོན་ཅི་ཞེ་ན། ཇི་ལྟར་ཏིང་གི་རབ་ཏུ་སྦྱོར་བ་རྣམས་དངོས་པོ་དང་བྱ་བ་དང་ལས་བྱེད་པ་པོ་རྣམས་ལ་བསྟན་པ་དེ་བཞིན་དུ། རྣམ་པ་དྲུག་པོ་དང་གསུམ་པོ་ཡང་བལྟ་བར་བྱའོ། །སང་ཞེས་པས་ནི། བྱ་བའི་དོན་ལ་བྱས་པའི་ཕ་རོལ་གྱི་སྒྲུབ་པ་རྣམས་གཟུངས་ཏེ། ཏི་ཛ་ལས་
