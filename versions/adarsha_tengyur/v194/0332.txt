ཀ་ཊ་ཀ་ར་ཎེ། སྨ་ལ་ལོཊའོ། །སྨའི་སྒྲ་སྦྱོར་བ་ལ་སྤྲིང་བསྐུལ་ལ་སོགས་པ་རྣམས་ལ་ཨཱུརྡྷྭ་མཽ་ཧཱུརྟྟཱི་ཀ་ལ་ད་ལྟར་བ་ལས་ལོཊ་ཉིད་འགྱུར་ཏེ། ཨཱུརྡྷྭ་མུ་ཧཱུརྟྟཱངྒྲཱ་མཾ་གཙྪ་ཏུསྨ་བྷ་བཱན། ཀ་ཊ་ཀ་རོ་སྟུ་བྷ་བཱན། ཁྱོད་ནི་རེ་ལྡེ་བྱེད་པའི་དུས་ལ་བབ་ཅེས་པའོ། །ལྷག་པར་ཚོལ་བ་ལའོ། །ལྷག་པར་ཚོལ་བ་ལ་སྨའི་

ལ་ཏུཾ་རྐྱེན་དུ་འགྱུར་བ་ཉིད་དོ། །ཀཱ་ལ། ས་མ་ཡ། བེ་ལཱ་རྣམས་ལ་ཡད་ལ་ལིངའོ། །ཀཱ་ལ་ལ་སོགས་པའི་སྒྲ་ལྷན་ཅིག་སྦྱོར་བའི་བྱིངས་ལས་ལིང་འགྱུར་ཏེ། ཀཱ་ལ་ཡཾ་ཡདྦྷུཉྫཱི་ཏ་བྷ་བཱན། ས་མ་ཡོ་ཡཾ་ཡདྦྷུཉྫཱི་ཏ་བྷ་བཱན། བེ་ལོ་ཡཾ་ཡདྦྷུཉྫཱི་ཏ་བྷ་བཱན། འོས་པ་དང་ནུས་པ་དག་ལའོ། །འོས་པ་དང་ནུས་པ་
དག་ལའང་ལིང་འགྱུར་ཏེ། བྷ་བཱན་ཁ་ལུ་ཀ་ནྱཱཾ་བ་ཧེ་ཏ། ཨརྡ་ཏི་ཀ་ནྱཱ་བོ་ཌྷུཾ། བྷ་བཱན་ཁ་ལུ་བྷཱ་རཾ་བ་ཧེ་ཏ། ཤཀྟོ་བྷ་བཱན་བྷཱ་རཾ་བོ་ཌྷུཾ། །བྷ་བ་ཏཱ་ཁ་ལུ་ཀ་ནྱཱ་བོ་ཌྷ་བྱཱ། བ་ཧ་ནཱི་ཡཱ། བཱ་ཧྱ་བྷ་བཱན། ཁ་ལུ་ཀ་ནྱཱ་ཡ་བོ་ཌྷ་ཞེས་པ་གནས་པའི་ཚུལ་གྱི་ཆོ་གས་འགྲུབ་བོ། །བྱ་བའི་ཁྱད་པར་ལས་དུས་གཞན་དུ་གྱུར་པ་
ལ་ཡང་རྣམ་པ་གཞན་དུ་མ་བྱས་སོ། །ཚིག་གི་ཁྱད་པར་བརྗོད་པའི་བྱ་བ་དང་འབྲེལ་པའི་དུས་གཞན་དུ་གྱུར་པ་ལ་ཡང་དུས་ཀྱི་ཁྱད་པར་ལ་བྱས་པའི་ཚིག་གང་ཡིན་པ་དེ་རྣམ་པ་གཞན་དུ་མི་འགྱུར་ཏེ། ཨགྣི་ཥྚོ་མ་ཡཱ་ཛྱ་སྱ་པུ་ཏྲོ་ཛ་ནཱི་ཏཱ། གོ་མནཱ་སཱི་ཏ། གོ་མཱན་བྷ་བི་ཏཱ། དེ་བཞིན་དུ། ལུ་ནཱི་ཧི་ལཱུ་ནཱི་ཞེས་པ་
འདི་ཉིད་ལུ་ནཱ་ཏིའོ། །ཨ་ལཱ་བིཏ། ལ་བི་ཥྱ་ཏི། ལུ་ནཱ་སི། ཨ་ལ་བཱི། ལ་བི་ཥྱ་སི། ལུ་ནཱ་སི། ཨ་ལཱ་བི་ཥཾ། ལ་པི་ཥྱཱ་མི། དེ་བཞིན་དུ་གཉིས་ཚིག་དང་མང་ཚིག་དག་ལ་ཡང་དཔེར་བརྗོད་དོ། །ཨ་དྷཱི་ཥྭ་ཨ་དྷཱི་ཥ་ཞེས་པ་འདི་ཉིད་ཨ་དྷཱི་ཏེ་ཞེས་ཏེ། ལོཊའི་མཐའ་ཉིད་ཡང་ཡང་ལ་གཉིས་བརྗོད་པའོ། །འདི་
