དེ་བཞིན་དུ་སྦྱིན་བྱ་དང་བྱ་བ་ལའང་ངོ་། །བྻུཥྚ་སོགས་ལས་ཨ་ཎའོ། །བདུན་པའི་མཐའ་ལས་བྱ་བ་དང་སྦྱིན་བྱ་དག་ལ་འགྱུར་ཏེ། བྻུཥྚའི་བྱ་བའམ་སྦྱིན་བྱ་ལ། བཻ་ཡུཥྚཾ། ནཻཏྱཾ། བྻུཥྚ་ནི་དྱ། ནིསྐྲ་མ་ཎ། ནི་བི་ཤ་ན། པྲ་བེ་ཤ་ན། ཏཱིརྠ། སཾ་གྲཱ་མ། ས་གྷཱ་ཏ། པྲ་བཱ་ས། ཨུ་པ་བཱ་ས། ཨགྣི་པ་ད།

སྦྱིན་བྱ་དང་བྱ་བ་དག་ལ་འགྱུར་ཏེ། ལག་པས་སྦྱིན་པར་བྱ་བའམ་བྱ་བ་ནི། ཧསྟྱཾ། མཛེས་པར་བྱ་བའི་དོན་ལ་གསུམ་པ་ལས་ཋཉ་འགྱུར་ཏེ། རྣ་བ་དཀྲིས་པས་མཛེས་པར་བྱེད་པ་ནི། ཀཱརྞ་བཥྚ་ཀི་ཀཾ །གདོང་ངོ་། །བཱསྟྲ་ཡུ་ག་ལ་ལི་ཀཾ །ལུས་སོ། །ཀརྨ་དང་བེ་ཤ་ལས་ཡཏའོ། །མཛེས་པར་བྱེད་པ་ལ་གསུམ་པའི་མཐའ་
ལས་འགྱུར་ཏེ། ལས་ཀྱི་མཛེས་པ་ནི། ཀརྨྨ་ཎྱཾ་དཔའ་བོའོ། །ཟློས་གར་ཆ་བྱད་ནི། བཻ་ཤྱཱ། ཚོགས་ཅན་མའོ། །བཻཤའི་སྒྲ་ནི་ལུས་དང་བཅས་པ་ཞེས་ཁ་ཅིག་ཟེར་རོ། །སནྟྲཱ་པ་སོགས་ལས་པྲ་བྷ་བ་ཏི་ལས་བཞི་པ་ལས་སོ། །སནྟཱ་པ་སོགས་ལས་བཞི་པའི་མཐའ་ལས་པྲ་བྷ་བ་ཏིའི་དོན་ལ་ཋཉ་འགྱུར་ཏེ། སརྦ་
པཱ་ཡ་པྲ་བྷ་བ་ཏི་ས་མརྠཿ། སནྟ་པི་ཀཿ། སཾ་གྲ་མ། སནྣཱ་ཧ། སཾ་ཡོ་ག །སཾ་པ་རཱ་ཡ། སཾ་བེ་ཤཿ། ནིརྒྷོ་ཥ། ནི་སརྒྒ་ཨུ་པ་སརྒྒ་པ་བཱ་ས། སམྦཱ་ད་ན་ཕྱེ་དང་ཤ་དང་ཟན་ལས་རྣམ་པར་གཟུང་བ་ལས་ཀྱང་སྟེ། མཱཾ་སི་ཀཾ །ཨཽ་ད་ནི་ཀཾ །སཀྟུ་མཾ། སཽ་ད་ནི་ཀཾ །ཡོ་ག་ལས་ཡཏ་ཀྱང་ངོ་། །ཡོ་གའི་སྒྲ་ལས་བཞི་པའི་མཐའ་ལས་པྲ་བྷ་
བ་ཏིའི་དོན་ལ་ཡཏ་འགྱུར་ཏེ། རྣལ་འབྱོར་གྱི་ཕྱིར་རབ་ཏུ་སྐྱེ་བ་ནི། ཡོ་གྱཿ། ཡོ་གི་ཀཿ། ཀརྨཱན་ལས་ཨུ་ཀནའོ། །བཞི་པའི་མཐའ་ལས་འགྱུར་ཏེ། ལས་ལ་རབ་ཏུ་སྐྱེ་བ་ནི། ཀཱརྨུ་ཀཾ །གཞུའོ། །སུ་ལས་ས་མ་ཡ་ལས་འདིའི་ཐོབ་པ་ལའོ། །ས་མ་ཡའི་སྒྲ་ལས་འདིའི་ཞེས་པ་དྲུག་པའི་དོན་ལ་དང་པོའི་མཐའ་ལས་ཋཉ་
འགྱུར་ཏེ། དེ་གལ་ཏེ་དུས་ལ་བབ་པ་ནི། སཱ་མ་ཡི་ཀཾ །རྀ་ཏུ་སོགས་ལས་ཨ་ཎའོ། །འདིའི་ཐོབ་པ་ཞེས་པའི་དོན་ལ་འགྱུར་ཏེ། འདིའི་དུས་ལ་བབ་པ་ནི། ཨཱརྟྟ་བཾ་ལོ་ཏོག་གོ། །འདིའི་གོས་ཉེ་བར་ཐོབ་པ་ནི། ཨཽ་པ་བསྟྼཾ། འདིའི་རབ་ཏུ་ཟ་བ་བབ་པ་ནི་པྲ་ཤི་ཏྲཾ་བྷཀྟ། ཀ་ལ་ལས་ཡཏ་འང་ངོ་། །འདིའི་ཐོས་པ་ཞེས་པའི་དོན་ལ་འགྱུར་ཏེ།
