ཌཱཔ་ཀྱི་དོན། རྒྱུ་མེད་ན། རྒྱུ་མཚན་ཅན་ཡང་མེད་དོ་ཞེས་པས་ཊའི་ཏ། ཥིཏ་ལས་དཱིཥ་ཞེས་པས་དཱི་ཥ་འོ། །ང་ཡིག་དཱི་ཨུང་ཞེས་པའི་རྟགས་དོན། ཥ་ཡིག་དཱིས་དང་དཱི་ཥ་ཐུན་མོང་བ་གཟུང་བའི་དོན་ཏེ། དཱི་ཨཱམ་ཞེས་པས་ཨ་ཡིག་དབྱིའོ། །སུའིའོ། །དཱི་ཨས་ཀྱི་རིང་པོ་ལས་སོ་ཞེསཔས་སུ་དབྱིའོ། །དྷཱ་ཏྲཱི། དཱ་ཏྲཱཾ། ཨི་ཀའི་ཨཏ་ཨེང་། ནེ་ཏྲཾ། བལ་དང་པོའི་ཨི་ཊ་ཞེས་པས་ཨིཏ་མི་འགྱུར་ཏེ། མང་པོ་ལས་སྤྱིར་བཏང་བའི་ཕྱིར་རོ། །པཏྟྲཾ། པཱ་ཏྲཾ། ཤཱ་སྟྲཾ། དེ་ལ་སོགས་པ་གཞན་ཡང་ཤེས་པར་བྱའོ། །ཨུ་ཥི་སཱུ་མཱུ་རྣམས་ལས་ཀི་ཏའོ། །ཨུ་ཥ་སྲེག་པ་
ལའོ། །ཥཱུ་དེད་པ་ལའོ། །[ཡ་མ། ཟོར་བ།] མཱུང་འཆིང་བ་ལའོ། །[མིག] །འདི་རྣམས་ལས་ཥྚྲན་ཕ་རོལ་ཏུ་འགྱུར་ཏེ། ཀིཏ་བཞིན་ནོ། །དེས་ན་ཨཏ་ཨེང་དུ་མི་འགྱུར་རོ། [ལོ་མ། ལྷུང་བཟེད། བསྟན་བཅོས།] །སྟུའི་ཤྩུ་དང་ཥྚུ་དག་གི་དེ་དག་གོ་ཞེས་པས་ཏ་ཡིག་གི་ཊ་ཡིག་གོ། །ཨུཥྚཿ། སཱུ་ཏྲཾ། མཱུ་ཏྲཾ། 
ཨ་མི། ནཀྵི། ཀ་ཌི་རྣམས་ལས་ཨ་ཏྲནའོ། །ཨ་མ་ནད་ལའོ། །ནཀྵ་འགྲོ་བ་ལའོ། །ཀ་ཌ་རྒྱགས་པ་ལའོ། །འདི་རྣམས་ལས་ཨ་ཏྲན་ཕ་རོལ་དང་། ན་ཡིག་དབྱངས་དོན། ཨ་མ་ཏྲཾ། ནཀྵ་ཏྲཾ། ཀ་ཌ་ཏྲཾ། ཀ་པི་རི་ཀ་ལ་སོགས་པས་ལ་ཉིད་དོ། །ཀ་ལ་ཏྲཾ། [རྔ་མོ། སྲད་བུའམ་སྣལ་མ། གཅིན།]
བྲྀཉ་ལས་ཨ་ཏྲ་ཙའོ། །བྲྀཉ་སྒྲིབ་པ་ལའོ། །འདི་ལས་ཨ་ཏྲ་ཙ་ཕ་རོལ། ཙ་དབྱངས་དོན། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དོ། །ཌཱཔའོ། །བ་ར་ཏྲཱ། [འབྲང་བ།] ཨ་མི། ཙི། [སྣོད། རྒྱུ་སྐར། ཆུང་མ།] མི་དི་ལས་ཏྲ་ཀའོ། །ཨ་མ་ནད་ལའོ། །[ཆུང་མ།] ཙི་སོགས་པ་ལའོ། །མི་དཱ་
སྣུམ་པ་ལའོ། །ཨཱ་ཡིག་ཨཱ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། འདི་དག་ལས་ཏྲཀ་ཕ་རོལ། ཀ་ཡིག་ནི་ཀིཏ་ལས་སོ་ཞེས་པས་ཨཏ་ཨེང་དགག་པའི་དོན། མ་བ་དག་གིའང་མའི་ནའོ་ཞེས་པས་མའི་ནའོ། །ཨནྟཾ། ཙི་ཏྲཾ། མི་དིའི་ང་ཡིག་གི་ཏ་ཡིག་གོ། །མཾཏྟྲཾ། པཱུཉ་ལས་ཐུང་ངུ་ཡང་ངོ་། །པཱུཉ་དག་བྱེད་
ལའོ། །འདི་ལས་ཏྲཀ་ཕ་རོལ་ཏེ་ཐུང་ངུ་ཡང་ངོ་། །པུ་ཏྲཿ། བུ། བ་ཧ་དང་ལ་ལ་སོགས་པ་རྣམས་ལས་ཨི་ཏྲ་ཨུ་ཏྲ་དག་གོ། །བ་ཧ་ལ་སོགས་དང་ལ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་གྲངས་ཇི་ལྟ་བ་བཞིན་ [སྒྱུ་མ། བཀྲ་བའམ་རྨད་དུ་བྱུང་བ།] དུ་ཨི་ཏྲ་དང་ཨུ་ཏྲ་ [བཤེས་གཉེན་ནམ་མཛའ་བཤེས།] ཞེས་པ་འདི་
