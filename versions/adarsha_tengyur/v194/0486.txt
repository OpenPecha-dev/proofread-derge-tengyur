རཱ་ས་ནཿ། རེག་བྱ་ནི་སྤཪྵ་ནཿ། བཅུ་བཞི་བལྟ་བར་བྱ་བ་ནི། རཀྵ་ཤྩཏུ་རྡ་ཤཾ །རྡོ་བའི་ཕྱེ་མ་རྣམས་ནི། ཤཀྟ་བོ་དཱཪྵ་དཱཿ། ཨུ་དཱུ་ཁ་ལེ་ཀྵུ་ཎྜཱ་ཡ། ཡང་ན་ཨཽ་དཱ་ཁ་ལཱཿ། རྟ་རྣམས་ཀྱིས་འགྲོ་བའི་ཤིང་རྟ་ནི། ཨཱ་ཤྭཿ། བཞི་རྣམས་ཀྱིས་འགྲོ་བ་ནི་ཙཱ་ཏུཪྵ་ཀ་ཊཾཿ། ལྷག་མ་ལའོ་
ཞེས་པའི་སྤྱིར་གཏང་བ་ནི་རྐང་པ་ཡོངས་སུ་རྫོགས་པའི་བར་རོ། །དེ་ཉིད་ཀྱི་ཕྱིར་དོན་གཞན་གྱི་ཁྱད་པར་རྗེས་སུ་རྟོགས་པ་འདིས་རང་བཞིན་གྱི་ཁྱད་པར་ཡང་སྟེ། རྐྱེན་གྱི་ཁྱད་པར་སྒྲུབ་པ་གང་ཡིན་པ་དེ་དག་ཐམས་ཅད་ལྷག་མ་རྣམས་ཡང་ངོ་། །དོན་གྱི་ཁྱད་པར་རྟོགས་པ་ལ་ནི་ལྷག་མ་ཉིད་
མ་ཡིན་ནོ། །འོན་ཀྱང་ལྷག་མ་ཁྱད་པར་ཉིད་སོ་སོར་ངེས་པའི་དོན་གྱི་དངོས་པོའི་ཕྱིར། རཱཥྚ་ལས་གྷའོ། །ལྷག་མ་ལ་འགྱུར་ཏེ། རཱཥྚཱི་ཡཿ། ཨཎ་གྱི་དམིགས་བསལ་ཏེ་འདོད་པ་ཐོབ་པ་ལ་བརྗོད་པའོ། །པཱ་ར། ཨ་བཱ་ར། ཨ་བཱ་ར་བཱ་ར་ལས་ཁའོ། །ལྷག་མ་ལ་འགྱུར་ཏེ། པཱ་རི་ཎཿ། ཨ་པཱ་རཱི་
ཎཿ། པཱ་རཱ་བཱ་རི་ཎཿ། ཨ་བཱ་ར་པཱ་རཱི་ཎཿ། ཚོགས་པ་རྣམ་པར་བཟུང་བ་ཉིད་ཀྱིས་བསྐྱེད་པས་གཟུགས་བཞིའོ། །དང་པོ་དང་གསུམ་པ་དག་ལ་ཚེ་ཐོབ་པ་དང་ལྷག་མ་ལ་ཨཎ་ཐོབ་པ་ལ་བརྗོད་པའོ། །གྲཱ་མ་ལས་ཡཏ་ཁཉ་དག་གོ། །གྲཱ་མའི་སྒྲ་ལས་ལྷག་མ་ལ་ཡཏ་དང་ཁཉ་དག་སྟེ། ཚའི་དམིགས་
བསལ་དུ་འགྱུར་རོ། །གྲཱ་མྱཿ། གྲཱ་མི་ཎཿ། ཉིཏ་བྱེད་པ་ནི་གྲ་མཱི་ཎཱ་བྷྱཱརྱ་ཞེས་པ་འདིར་སྐྱེས་པ་བཞིན་གྱིས་དངོས་པོ་དག་པའི་དོན་ཏོ། །ཀརྟྲི་སོགས་ལས་ཀྱང་ཌྷཉའོ། །ཀརྟྲི་སོགས་རྣམས་དང་གྲ་མ་ལས་ཀྱང་ཌྷཀ་འགྱུར་ཏེ། ཀུརྦི་རཱ་སྟ་ཡཿ། ཀརྟྲ་ཡཿ། དེ་རྣམས་ལས་བྱུང་བའམ་དེ་རྣམས་ལས་
འོངས་པའོ། །ཀཱརྟྲེ་ཡ་ཀ །གྲ་མེ་ཡ་ཀཾ། ཨཽམྦྷེ་ཡ་ཀཾ །ཀརྟྲི་ཨུ་མྦྷི། པུཥྐ་ར། པུཥྐ་ལ། ཀུཎྜི་ནཱ། ན་ག་ར་མཱ་ཧིཥྨ་ཏྲཱི། བརྨ་ཏཱི། ཀུམྦྷི། ཀུཌྱཱའི་ཡ་དབྱི་བའང་ངོ་། །ཀཽ་ཌེཡ་ཀཿ། ཀུ་ལ་ཀུཀྴི། གྲཱི་བ་རྣམས་ལས་ཁྱི་དང་རལ་གྲི་དང་རྒྱན་ལའོ། །རིམ་པ་བཞིན་དུ་ཌྷ་ཀཉ་འགྱུར་ཏེ། རིགས་ལས་
སྐྱེས་པ་ནི། ཀཽ་ལེ་ཡ་ཀཿ། ཁྱིའོ། །གཞན་ལ་ཀཽ་ལཿ། ཀཽ་ཀྵེ་ཡ་ཀཿ། རལ་གྲིའོ། །གཞན་ལ་ནི་ཀཽཀྵཿ། གྲཻ་བ་ཡ་ཀཾ། རྒྱན་ནོ། །གཞན་ལ་གྲེ་བཿ། ན་དཱི་སོགས་ལས་ཌྷ་ཀའོ། །འགྱུར་ཏེ། ནཱ་དེ་ཡཿ། མཱ་ཧེ་ཡཿ། ནཱ་དཱིཿ། མ་ཧཱིཿ། བཱ་རཱ་ཎ་སཱིཿ། ཤྲཱ་བསྟྲི། ཀཽ་ཤམྦཱི། བ་ན་ཀཽ་ཤམྦཱི། ཀཱ་ཤ་པཱ་རི།
