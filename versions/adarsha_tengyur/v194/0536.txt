ནཱ་བྷྱཿ། ཙཱཀྵུ་ཥྱཾ། ཁ་ལ། ཡ་བ། མཱ་ཥ། ཏི་ལ། བྲྀ་ཥ། བྲཧྨ་ར་ཐ་ལས་སོ། །བཞི་པའི་མཐའ་ལས་ཕན་པ་ལ་ཡཏ་འགྱུར་ཏེ། ཐ་ཆད་ལ་ཕན་པ་ནི། ཁ་ལྱཾ། ཡ་བྱཾ། མཱ་ཥྱཾ། ཏི་ལྱཾ། ཀྲིཥྞ་ཏི་ལྱཾ། བྲྀཥྱཾ། བྲཧྨ་ཎྱ། ཤིང་རྟའི་ཆེད་དུ་ཕན་པ་ནི། ར་ཐྱཱཿ། པྲ་ཧོ་ལཱི། ཨཛ། ཨ་བྷི་དག་ལ་ཐྱནའོ། །བཞི་
པའི་མཐའ་ལས་ཕན་པ་ལ་འགྱུར་ཏེ། ར་འམ་ལུག་གི་ཆེད་དུ་ཕན་པ་ནི། ཨ་ཛ་ཐྱཱ། ཨཱརྨ་དང་བྷོ་གའི་མཐའ་དག་ལ་ཁའོ། །བཞི་པའི་མཐའ་ལས་ཕན་པའི་དོན་ལ་ཁ་འགྱུར་ཏེ་བདག་ལ་ཕན་པ་ནི། ཨཱཏྨ་ནཱི་ནཾ། མིའི་ལོངས་སྤྱོད་ལ་ཕན་པ་ནི། མཱ་ཏྲི་བྷོ་གཱི་ཎཿ། བྷོ་གཿ། ཤ་རཱི་རཾ། རཱ་ཛ་བྷོ་གཱི་ནི། ཨཱ་ཙཱརྱ་བྷོ་གཱི་ནི། ཀྵུ་བྷྣ་སོགས་ཉིད་ཀྱིས་
ཕྱིར་ན་ཎ་ཉིད་དོ། །པཉྩ་བི་ཤྭ་སོགས་ལས་ཛི་ན་ལས་ལས་འཛིན་ལའོ། །པཉྩ་དང་བི་ཤྭ་དང་པོའི་ཛི་ནའི་སྒྲ་བཞི་པའི་མཐའ་ཅན་ལས་ཕན་པའི་དོན་ལ་ཁ་འགྱུར་ཏེ། ལྔ་རྣམས་ལས་སྐྱེས་པ་ལ་ཕན་པ་ནི། པཉྩ་ནཱི་ནིཿ། བི་ཤྭ་ཛ་ནཱི་ནཿ། ལས་འཛིན་ལ་ཞེས་པ་ཅི་ཞེ་ན། འདི་སྣ་ཚོགས་ལས་སྐྱེས་པ་ནི། བི་ཤྭ་ཛ་ནཿ། དེའི་ཆེད་དུ་
ཕན་པའམ། སྣ་ཚོགས་རྣམས་སྐྱེ་བའི་ཆེད་དུ་ཕན་པ་ནི། བི་ཤྭ་ཛ་ནཱི་ཡཿ། སརྦ་ལས་སོ། །ཕ་རོལ་ཛ་ན་ལས་བཞི་པའི་མཐའ་ལས་ཕན་པའི་དོན་ལ་ཁ་འགྱུར་ཏེ། ཐམས་ཅད་རྣམས་ལས་སྐྱེས་པ་རྣམས་ལ་ཕན་པ་ནི། སརྦ་ཛ་ནཱི་ནཿ། ལས་འཛིན་ལ་ཞེས་པ་ཉིད་དེ། སརྦ་ཛ་ནཱི་ཡཿ། མ་ཧ་ཏ་
ལས་ཐཉའང་ངོ་། །མ་ཧ་ཏ་ཀྱི་སྒྲ་དང་སརྦ་ལས་ཕ་རོལ་གྱི་ཛ་ན་ལས་བཞི་པའི་མཐའ་ལས་ཕན་པ་ལ་ཋཉ་འགྱུར་ཏེ། མཱ་ཧཱ་ཛ་ནི་ཀཿ། སརྦ་ཛ་ནི་ཀཿ། ལས་འཛིན་ལ་ཞེས་པ་ཉིད་དེ། མཱ་ཧཱ་ཛ་ནཱི་ཡཿ། སརྦ་ཛི་ནཱི་ཡཿ། སརྦ་ལས་ཎ་ཡང་ནའོ། །སརྦའི་སྒྲ་བཞི་པའི་མཐའ་ལས་ཕན་པ་ལ་ཎ་ཡང་
ན་འགྱུར་ཏེ། ཐམས་ཅད་ཀྱི་ཆེད་དུ་ཕན་པ་ནི། སཱརྦཿ། སརྦཱ་ཡཿ། སླར་ཡང་སརྦ་སྨོས་པ་ནི། ཛ་ན་བཟློག་པའི་དོན་ཏོ། །མཱ་ཎ་བ་དང་ཙ་ར་ཀ་ལས་ཁཉའོ། །བཞི་པའི་མཐའ་ལས་ཕན་པ་ལ་འགྱུར་ཏེ། མཱ་ཎ་བཱི་ནཾ། ཙཱ་ར་ཀཱི་ཎཾ། བྱ་བ་ལས་ཉེ་བར་ལེན་པ་ལའོ། །བྱ་བ་བརྗོད་པ་བཞི་པའི་མཐའ་ལས་ཉེ་བར་ལེན་
པའི་རྒྱུ་མངོན་པར་བརྗོད་པ་ལ། ཆོ་ག་ཇི་ལྟ་བ་བཞིན་དུ་ཚ་དང་ཡཏ་དག་འགྱུར་ཏེ། སོལ་བ་རྣམས་ལ་ཕན་པའི་ཤིང་རྣམས་ནི། ཨངྒ་རཱི་ཡཱ་ཎིཿ། པྲཱ་བཱ་དཱི་ཡཱཿ། སོ་ཕག་རྣམས་སོ། །པི་ཙ་བྱཿ། རས་བལ་ལོ། །ཉེ་བར་ལེན་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨུ་ད་ཀཱ་ཡ་ཀཱུ་པཿ། བྱ་བ་ལས་ཞེས་པ་ཅི་ཞེ་ན།
