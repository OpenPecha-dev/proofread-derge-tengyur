སྐྲའི་སྒྲ་ནི་སྐྲའི་རྣམ་གྲངས་ཏེ། དེ་ལ་སྐྲ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་སོ། །སྐེ་རགས་ནི་ཨ་བ་དྷཱུ་ཏཱི་བདག་མེད་མ་ཕན་ཚུན་དུ་འབྲེལ་བའི་ཕྱིར་སྐོར་གཉིས་སོ། །ཡང་ན་ཨ་བ་དྷཱུ་ཏཱི་དང་འབྲེལ་བ་ཉིད་བྱང་ཆུབ་ཀྱི་སེམས་སོ། །སྐེ་རགས་ནི་ཨ་བ་དྷཱུ་ཏཱི་བདག་མེད་མ་ཕན་ཚན་དུ་འབྲེལ་བའི་
ཕྱིར་སྐོར་གཉིས་སོ། །ཡང་ན་ཨ་བ་དྷཱུ་ཏཱི་དང་འབྲེལ་བ་ཉིད་བྱང་ཆུབ་ཀྱི་སེམས་སོ། །དེ་ཡང་སྐྲའི་སྐེ་རགས་ཡིན་ལ། ཐབས་དང་ཤེས་རབ་རང་བཞིན་གྱིས་ཉིས་སྐོར་རོ། །གཉི་ག་བསྐོར་བ་ཉིད་ནི་ཨཱ་ལི་ཀཱ་ལིའི་ནཱ་ད་གཉིས་དབྱེར་མེད་པ་ཉིད་དོ། །འཛིན་པ་ནི་གཡོ་བ་མེད་པར་
བྱས་པའོ། །གཞན་ཡང་ཕྱག་རྒྱའི་རྒྱན་རྣམ་པར་དག་པ་གསུངས་པ། ཐལ་བ་སྐྲ་ཡི་མཆོད་ཕྱིར་ཐོགས། །རྣལ་འབྱོར་པ་ཡི་སྤྱོད་པས་འཛིན། །ཞེས་པ་སྨོས་སོ། །ཐལ་བ་ནི་རོ་འདུས་བྱས་པའི་ཐལ་བའམ། ཡང་ན་རོ་འདུས་པའི་རུས་པའི་ཕྱེ་མའོ། །སྐྲའི་མཆོད་
ཕྱིར་ཐོགས་ཞེས་པ་ནི། རྐུན་མ་དཔྱངས་སྐྲ་ལས་བྱས་པའི་རྣལ་འབྱོར་པའི་མཆོད་ཕྱིར་ཐོགས་སོ། །སྤྱོད་པས་འཛིན་ནི་འཁྱམས་པས་སོ། །ཡང་ན་ཐལ་བའི་སྒྲས་རུས་པ་ཡིན་པའི་ཕྱིར་བདག་མེད་མའོ། །སྐྲའི་སྒྲས་རྡོ་རྗེ་འཛིན་པ་སྟེ་ཐལ་བ་དང་སྐྲ་དག་གོ། །མཆོད་ཕྱིར་
ཐོགས་ནི་དྲི་མ་མེད་པར་བྱས་པ་ཤིན་ཏུ་རྣམ་པར་དག་པ་ཆོས་ཀྱི་དབྱིངས་ཀྱི་རང་བཞིན་ནོ། །དེ་ཉིད་རྣལ་འབྱོར་པ་སྟེ་རྣལ་འབྱོར་དང་ལྡན་པས་སོ། །འཛིན་པ་ནི་གཡོ་བ་མེད་པར་འགྱུར་བའོ། །སྤྱོད་པ་ནི་འཁྱམ་པའི་བྱ་བའོ། །ཁ་ཊྭཱཾ་ག་གཟུགས་ཤེས་རབ་ཏེ། །ཞེས་གསུངས་
པ་དང་། ཐབས་ཀྱི་གཟུགས་ཀྱིས་ཅང་ཏེའུ་ཉིད། །ཅེས་པའི་རྣམ་པར་དག་པ་ལུགས་ལས་བཟློག་པས་གསུངས་པ། ཅང་ཏེའུའི་སྒྲས་ནི་བཟླས་པ་སྟེ། ཁ་ཊྭཱཾ་ག་བསྒོམ་ཤེས་རབ་ཉིད། །ཅེས་པའི་དོན་ནི་གསལ་བ་ཉིད་དོ། །ཡང་ན་ཅང་ཏེའུའི་སྒྲས་ཨཱ་ལི་ཀཱ་ལིའི་ནཱ་དའི་བདག་ཉིད་
གཞོམ་དུ་མེད་པའི་རྡོ་རྗེ་འཛིན་པའོ། །དེ་ཡང་བཟླས་པ་ནི་རྒྱུན་མི་འཆད་པར་བཟླས་པའི་རང་བཞིན་ཡིན་པས་བར་མ་ཆད་པའི་རང་བཞིན་དམ་ཚིག་གི་དཀྱིལ་འཁོར་རོ། །ཁ་ཊྭཱཾ་ག་བསྒོམ་ཤེས་རབ་ཉིད། །ཅེས་པ་ནི་མཆོག་ཏུ་བྱུང་བའི་ཡེ་ཤེས་ནི་ཤེས་རབ་བོ། །ཅི་ཞེ་ན། ཁ་ཊྭཱཾ་ག་
