གལ་ཏེ་དེ་གཉིས་ཀ་ཡང་དུས་གཅིག་ཏུ་ཁྲུས་རྩོམ་པ་ན་གལ་ཏེ་མཉམ་དུ་འཇུག་པར་གྱུར་ན་ཁ་ན་མ་ཐོ་བ་ཡོངས་སུ་རྫོགས་པ་གཅིག་གི་ཕྱེད་དུ་འགྱུར་རོ། །གལ་ཏེ་མ་ཡིན་ན་མ་ཆོག་པ་ཉིད་དོ། །དེ་བས་ན་གོང་མའམ་འོག་གི་ཆ་དག་ལ་ཐམས་ཅད་དུ་གྱུར་པ་ཉིད་ཅེས་བྱ་བ་འདི་ལ་ལྟེ་བར་ཞེས་སྨོས་པ་
ནི་གཙོ་བོ་ཡིན་ཏེ། དེ་ལྟ་བས་ན་མི་སྲིད་པ་ཉིད་ལ་ཡང་ལྟེ་བའི་ཕྱོགས་གཏམས་ཏེ་རྫོགས་པ་ལ་རྫོགས་པ་ཉིད་དོ། །འདིར་ལྟེ་བར་མ་ཕྱིན་པའི་མདུན་རོལ་དུ་ཅི་ཞིག་རྟོགས་པར་བྱ་ཞེ་ན། དེའི་ཕྱིར་མདུན་རོལ་དུ་ནི་ཉེས་བྱས་སོ་ཞེས་བྱ་བ་སྨོས་ཏེ། ཆུ་ལ་རེག་པའི་མདུན་རོལ་དུ་སྦྱོར་བར་བྱེད་པ་ནི། དེ་
སྦྱོར་བའི་ཉེས་བྱས་སུ་རྣམ་པར་གནས་པ་ཡིན་ནོ། །ཐམས་ཅད་ལ་དུས་དག་ཡོད་ཀྱང་བྱིན་གྱིས་མ་བརླབས་ན་ཞེས་བྱ་བ་ནི་དུས་དག་ལ་ཐམས་ཅད་དུ་བྱས་པ་གང་ཡིན་པ་བསླབ་པའི་གཞི་གཞན་དང་འབྲེལ་པ་དེ་ལ་དུས་དག་ཡོད་དུ་ཟིན་ཀྱང་བྱིན་གྱིས་མ་བརླབས་ཤིང་མངོན་སུམ་དུ་མ་བྱས་ན་ཉེས་
བྱས་སོ། །འདིར་གཞུང་ནི་ཞུ་བ་ལས། ཉེས་བྱས་ལས་ལྡང་བ་ནི་འདི་ཡིན་ཏེ། གང་ཞིག་རྒྱུར་གྱུར་བ་དུས་ཀྱི་གནས་སྐབས་སམ་གཞན་དེ་ནི་དུས་སོ། །ཞག་བཅོ་ལྔ་ན་ཁྲུས་ཀྱི་ཉིན་མོ་ལ་ཁྲུས་བྱ་བ་གང་ཡིན་པ་དེ་ནི་གྲགས་པ་ཡིན་པས་ན་འདིར་དུས་མེད་པའོ། །འདི་ནི་དུས་བྱིན་གྱིས་བརླབ་
པར་བྱ་བ་ལ་ཡུལ་མ་ཡིན་ཞེས་བྱ་བར་རིག་པར་བྱའོ། །དོན་གཞན་ལ་གཞོལ་བར་ཆུ་ཉམས་སུ་མྱོང་བ་ནི་ཁྲུས་མ་ཡིན་ནོ། །དེའི་ཕྱིར་མ་བསམས་པར་བརྒལ་བ་དང་བརྐྱལ་བ་ལ་གདབ་པ་དག་ནི་ལྟུང་བ་མེད་དོ་ཞེས་བྱ་བ་ནི་ཁྲུས་ལ་རྩེ་གཅིག་པ་ཉིད་ཀྱི་ཆུ་ཉམས་སུ་མྱོང་བ་གང་ཡིན་པ་དེ་ནི་ཁྲུས་གཞན་
ལ་ནི་མ་ཡིན་ཞེས་བྱ་བའི་དོན་ཏོ། །འདིར་གཞུང་ནི་དགེ་སློང་དག་ཆུ་ཀླུང་རྒལ་བར་བྱེད་དོ་ཞེས་གསུངས་པ་དང་། འདིར་གཞུང་གཞན་ལས་ཀྱང་རྐྱལ་བ་ལ་ཆུས་དབང་བསྐུར་བ་དང་། ཟམ་པ་ལས་ལྟུང་བ་ཆུ་ཀླུང་ལས་རྒལ་བར་བྱེད་པ་དང་། ཆུས་གང་བ་ལ་མཆོང་བར་བྱེད་པ་ལ་ཡང་ལྟུང་བ་མེད་དོ། །
དེ་བཞིན་དུ་མདུན་རོལ་དུ་འགྲོ་ཞིང་རྒྱ་མཚོ་ལ་གཟིངས་ནས་འགྲོ་བ་རྣམས་ལ་བཤང་བ་དང་གཅི་བ་བྱེད་པ་ཡང་ལྟུང་བ་མེད་དོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །དེ་དང་དེའི་ངོ་བོ་གཞན་གྱི་རྐྱེན་རྣམས་ལ་ཆུ་ཉམས་སུ་མྱོང་བ་ལ་ཉེས་པ་མེད་པ་ཉིད་དུ་བསྟན་པའོ། །ཇི་སྲིད་དུ་ཁྲུས་ཞེས་བྱ་བ་ལ་སོགས་པ་རྣམས་
