བྱ་བ་ནི། བྱང་ཆུབ་ཀྱི་མཆོག་ཏུ་སེམས་བསྐྱེད་པ་ནས་བཟུང་ནས་བྱང་ཆུབ་ཀྱི་སྙིང་པོ་ལ་གནས་ཀྱི་བར་རོ། །འབྲས་བུའི་གནས་སྐབས་ཞེས་བྱ་བ་ནི། ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཀྱི་ཡེ་ཤེས་བསྐྱེད་པ་དང་། ཉོན་མོངས་པ་ཐམས་ཅད་སྤང་བ་ཡོན་ཏན་ཐོབ་པའི་གནས་
སྐབས་སོ། །སེམས་ཅན་གྱི་དོན་བྱེད་པའི་གནས་སྐབས་ཞེས་བྱ་བ་ནི། དང་པོ་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བ་ནས་བཟུང་སྟེ། བསྟན་པ་མ་ནུབ་ཀྱི་བར་རོ། །དེ་ལ་རྒྱུའི་གནས་སྐབས་ནི་གསུམ་སྟེ། བསམ་པའི་གནས་སྐབས་དང་། སྦྱོར་བའི་གནས་སྐབས་དང་། དབང་གི་
གནས་སྐབས་སོ། །དེ་ལ་བསམ་པའི་གནས་སྐབས་ནི། སེམས་ཅན་ཐམས་ཅད་ངེས་པར་ཐར་པའི་སྨོན་ལམ་མོ། །དེའི་ཉེ་བའི་དཀའ་བ་བཞི་ནི་འདི་ལྟ་སྟེ། གཞན་གྱི་སྡུག་བསྔལ་ཉེ་བར་བསལ་བ་ལ་སོ་སོར་རྟོག་པ་དང་། དགོས་པ་ལ་སོ་སོར་རྟོག་པ་དང་། གྲོགས་ལ་སོ་སོར་
རྟོག་པ་དང་། ལོངས་སྤྱོད་ལ་སོ་སོར་རྟོག་པའོ། །དེ་ལ་ཐབས་སུ་འགྲོ་བ་བྱང་ཆུབ་ཏུ་སྨོན་པ་སྟེ། དེའི་རྗེས་སུ་ཚོལ་བ་བཞི་ནི་འདི་ལྟ་སྟེ། རིགས་དང་བཤེས་གཉེན་དམ་པ་དང་སྙིང་རྗེ་དང་སྡུག་བསྔལ་ལ་ཇི་མི་སྙམ་པའི་བཟོད་པ་དང་། རྐྱེན་བཞི་བཞི་རྣམས་ལས་བྱང་ཆུབ་ཀྱི་
སེམས་སྐྱེ་བར་འགྱུར་ཏེ་ཞེས་བྱ་བའི་ཚིག་བཅུ་པོ་འདི་དག་གིས་སྟོན་ཏེ། ལྷག་པའི་བསམ་པའི་གནས་སྐབས་སོ། །ལྷག་པར་མོས་པ་སྦྱོར་བས་སྦྱོར་ལམ་ཕ་རོལ་ཏུ་ཕྱིན་པ་བཅུ་དང་། ས་ཐོབ་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བདུན་ནོ། །དེ་ལ་ལྷག་པར་མོས་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བཅུ་ནི། སྦྱིན་
པ་དང་། ཚུལ་ཁྲིམས་དང་། བཟོད་པ་དང་། བརྩོན་འགྲུས་དང་། བསམ་གཏན་དང་། ཤེས་རབ་དང་། ཐབས་དང་། སྨོན་ལམ་དང་། སྟོབས་དང་། ཡེ་ཤེས་ཏེ། དེ་དག་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བཅུའོ། །ས་ཐོབ་པའི་ཕུན་སུམ་ཚོགས་པ་བཞི་ནི། བསམ་པ་དང་། སྦྱོར་བ་དང་། 
ཡོངས་སུ་འཛིན་པ་དང་། སྦྱིན་པ་ཕུན་སུམ་ཚོགས་པ་སྟེ། སྦྱིན་པ་དང་། ཚུལ་ཁྲིམས་དང་། བཟོད་པ་དང་། བརྩོན་འགྲུས་དང་། བསམ་གཏན་དང་། ཤེས་རབ་དང་། ཐབས་ཏེ། དེ་དག་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བདུན་ནོ། །ལྷག་པར་མོས་པའི་སྦྱོར་བས་ཕ་རོལ་ཏུ་ཕྱིན་པ་དག་བྱེ་བྲག་
