བ་ཀུན་ལས་བཏུས་པ། ཤློ་ཀ་ཁྲི་ལྔ་སྟོང་སྟེ། བམ་པོ་ལྔ་བཅུ། སོ་སོར་ཐར་པའི་གཞུང་འགྲེལ། ཤློ་ཀ་བརྒྱད་སྟོང་ཆིག་བརྒྱ་སྟེ་བམ་པོ་ཉི་ཤུ་རྩ་བདུན། ཡང་སོ་སོར་ཐར་པའི་འགྲེལ་པ། ཤློ་ཀ་ཉིས་སྟོང་ཆིག་བརྒྱ་སྟེ་བམ་པོ་བདུན། སོ་སོར་ཐར་པའི་ཚིག་
གི་བརྗེད་བྱང་། ཤློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་གཉིས། དགེ་སློང་མའི་སོ་སོར་ཐར་པའི་འགྲེལ་པ། ཤློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་གཉིས། འདུལ་བའི་མདོའི་རྒྱ་ཆེར་འགྲེལ་པ། ཤློ་ཀ་ཉི་ཁྲི་ཆིག་སྟོང་སྟེ་བམ་པོ་བདུན་ཅུ།།      །།ཐམས་ཅད་ཡོད་པར་སྨྲ་བ་དགེ་ཚུལ་གྱི་ཚིག་ལེའུར་བྱས་
པ། ཤློ་ཀ་ལྔ་བཅུ། ཚིག་ལེའུར་བྱས་པ་སུམ་བརྒྱ་པ། ཤློ་ཀ་སུམ་བརྒྱ་སྟེ་བམ་པོ་གཅིག །སུམ་བརྒྱ་པའི་རྒྱ་ཆེར་འགྲེལ་པ་འོད་ལྡན། ཤློ་ཀ་ཉིས་སྟོང་བཞི་བརྒྱ་སྟེ་བམ་པོ་བརྒྱད། འདུལ་བ་དྲི་བའི་ཚིག་ལེའུར་བྱས་པ། ཤློ་ཀ་བརྒྱ། འདུལ་བ་དྲི་བའི་ཚིག་གི་རྒྱ་ཆེར་བཤད་པ། ཤློ་ཀ་
སྟོང་དྲུག་བརྒྱ་དྲུག་ཅུ་རྩ་བཞི་སྟེ་བམ་པོ་ལྔ་དང་ཤློ་ཀ་བརྒྱ་དྲུག་ཅུ་རྩ་བཞི། སྡེ་པ་བཅོ་བརྒྱད་ཀྱི་མིང་དང་རིམ་པ། སློབ་དཔོན་དུལ་བའི་ལྷས་མཛད་པ། ཤློ་ཀ་དྲུག་ཅུ་རྩ་ལྔ། གཞུང་ཐ་དད་པ་བཀླག་པའི་འཁོར་ལོ་ལས་སྡེ་པ་ཐ་དད་པ་བསྟན་པ། སློབ་དཔོན་དབྱིག་བཤེས་ཀྱིས་མཛད་པ། ཤློ་ཀ་ལྔ་བཅུ།
 དགེ་སློང་གི་ཚུལ། མཁན་པོ་བོ་དྷི་ས་ཏྭས་མཛད་པ། ཤློ་ཀ་ལྔ་བཅུ། མ་བྱིན་པར་ལེན་པའི་སྐབས་ན། མཁན་པོ་བཞིས་དཔྱད་པ་ཤློ་ཀ་ཉི་ཤུ།།      །།ཐེག་པ་ཆེན་པོའི་མདོ་སྡེའི་ཏཱི་ཀཱ་ལ། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འབུམ་གྱི་འགྲེལ་པ་ཆེན་པོ། ཤློ་ཀ་ཉི་ཁྲི་སུམ་སྟོང་བཞི་བརྒྱ་སྟེ་བམ་
པོ་བདུན་ཅུ་རྩ་བརྒྱད། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འབུམ་དང་ཉི་ཁྲི་ལྔ་སྟོང་དང་། ཁྲི་བརྒྱད་སྟོང་པའི་རྒྱ་ཆེར་བཤད་པ་ཤློ་ཀ་བརྒྱད་སྟོང་ཆིག་བརྒྱ་སྟེ་བམ་པོ་ཉི་ཤུ་རྩ་བདུན། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བརྒྱད་སྟོང་པའི་རྒྱ་ཆེར་བཤད་པ། མངོན་པར་རྟོགས་པའི་རྒྱན། ཤློ་ཀ་དྲུག་བརྒྱ་སྟེ་བམ་པོ་
གཉིས། མངོན་པར་རྟོགས་པའི་རྒྱན་གྱི་རྒྱ་ཆེར་འགྲེལ་པ། ཤློ་ཀ་སྟོང་ལྔ་བརྒྱ་སྟེ་བམ་པོ་ལྔ། འཕགས་པ་སྡུད་པའི་ཚིགས་སུ་བཅད་པའི་འགྲེལ་པ། ཤློ་ཀ་སྟོང་བརྒྱད་བརྒྱ་སྟེ། བམ་པོ་དྲུག །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བདུན་བརྒྱ་པའི་འགྲེལ་པ། སློབ་དཔོན་བི་མ་ལ་མི་ཏྲས་མཛད་པ།
