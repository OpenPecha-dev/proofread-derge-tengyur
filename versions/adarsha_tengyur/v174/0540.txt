ངོ་བོ་ཉིད་སྟོན་ཏོ། །བྱས་པའི་ངོ་བོ་ཉིད་སྐྱེད་པར་བྱེད་པ་གང་ཡིན་པ་དེ་ནི་མི་རྟག་པའི་ངོ་བོ་ཉིད་མེད་ན་སྐྱེད་པར་བྱེད་དོ་ཞེས་ཚད་མ་དཔེས་རབ་ཏུ་སྟོན་ཏོ། །དེ་ལྟ་མ་ཡིན་ན་ཆོས་གཅིག་ཡོད་པའི་ཕྱིར་དེ་ལས་གཞན་པ་ཡང་ཡོད་པར་འགྱུར་རོ་ཞེས་བྱ་བར་ངེས་པ་མེད་པའི་ཕྱིར་འཁྲུལ་
པར་དགོས་པར་འགྱུར་རོ། །ཚད་མ་ངེས་ཀྱང་བསྒྲུབ་པར་བྱ་བའི་ཆོས་ཀྱི་དེ་ཙམ་དང་འབྲེལ་པ་སྟོན་ཏེ། རང་གི་རྒྱུ་ཁོ་ན་ལས་བྱས་པ་འཇིག་པའི་ངང་ཚུལ་ཅན་སྐད་ཅིག་མར་གནས་པའི་ཆོས་གང་ཡིན་པ་དེ་ལྟ་བུ་སྐྱེས་ཏེ། གཞན་ལས་དེ་ཡོད་པ་བཀག་པའི་ཕྱིར་རོ། །ཡང་ན་རྒྱུའི་
དངོས་པོར་ཏེ། ཡོད་པ་ཉིད་ཡིན་ན་ཡོད་པའི་ཕྱིར་རོ་ཞེས་དོན་གཞན་རབ་ཏུ་སྟོན་ཏོ། །དེ་ལྟར་དེའི་ངོ་བོ་དང་རྒྱུའི་དངོས་པོ་རབ་ཏུ་གྲུབ་ན་མི་རྟག་པ་ཉིད་མེད་ན་བྱས་པ་ཉིད་མེད་ལ་མེ་མེད་ན་ཡང་དུ་བ་སྟེ། འདི་ལྟར་དེ་ནི་རང་གི་ངོ་བོ་དང་རྒྱུ་ཡིན་ན། ཇི་ལྟར་རང་གི་ངོ་བོ་དང་རྒྱུ་མེད་པར་ཡོད་
པར་འགྱུར། དེ་བས་ན་གཞི་མེད་པ་ཡང་ཆོས་མི་མཐུན་པའི་དཔེ་ལ་ལྡོག་པ་རབ་ཏུ་གྲུབ་པོ། །གང་དག་ལ་དེའི་ངོ་བོ་དང་རྒྱུའི་དངོས་པོ་དག་རབ་ཏུ་གྲུབ་པའི་མཁས་པ་རྣམས་ལ་ནི། གཏན་ཚིགས་འབའ་ཞིག་བརྗོད་པར་ཟད། གང་གི་དོན་དུ་དཔེ་བརྗོད་པའི་དོན་དེ་གྲུབ་པ་
ཁོ་ན་ཡིན་པས་དེའི་ཚེ་དེ་བརྗོད་པས་ཅི་ཞིག་བྱ། དེ་རབ་ཏུ་བསྟན་ན་ཡང་ཆོས་མི་མཐུན་པའི་དཔེའི་གཞིས་ཅི་ཞིག་བྱ་སྙམ་དུ་དགོངས་ཤིང་གཞི་སྤང་བར་མཛད་དོ། །དེ་ཉིད་ཀྱི་ཕྱིར་འབྲེལ་ཤེས་ན། །གཉིས་ལས་གང་ཡང་རུང་སྨྲས་པས། །དོན་གྱིས་གོ་བས་ཅིག་ཤོས་ལའང༌། 
དྲན་པ་ཡང་དག་སྐྱེ་བར་འགྱུར། །ཡང་ན་དོན་གྱིས་གོ་བས་གང་ཡང་རུང་བས་གཉི་ག་རབ་ཏུ་བསྟན་པའི་ཕྱིར་རོ་ཞེས་བཤད་པ་གང་ཡིན་པ་སྟེ། དེར་ཡང་དཔེས་དེའི་ངོ་བོ་དང་རྒྱུའི་དངོས་པོ་རབ་ཏུ་སྟོན་པར་དགོངས་ཏེ། དོན་གྱིས་གོ་བས་གཅིག་བརྗོད་པས་ཅིག་ཤོས་འགྲུབ་པར་
བཤད་དོ། །འདི་ལྟར་གང་བྱས་པ་དེ་ནི་མི་རྟོག་པའོ་ཞེས་སྨྲས་ན། དོན་ཐ་དད་པ་མེད་ན། གསལ་བར་འདི་ནི་འདིའི་ངོ་བོ་ཉིད་དེ་ཙམ་དང་འབྲེལ་པ་ཚད་མས་མཐོང་བ་ཡིན་ནོ་ཞེས་བྱ་བར་ངེས་པར་འགྱུར་ཏེ། དེ་ཡོད་པར་ངེས་པའི་ཕྱིར་རོ། །དེ་བས་ན་དེའི་ངོ་བོར་ཤེས་པས་ནི་དོན་གྱིས་གོ་
