རོ་ཧི་ཏཾ་ནི་ཐང་ཤིང་དང་། །ཡུལ་དུ་བརྟན་དང་ཁྲག་ལའོ། །རོ་ཧི་ཏཱ་ནཱི་དང་པོ་དང་། །ཉ་དང་རི་དགས་དབྱེ་བ་ལ། །རོ་ཧི་ཎི་དང་རོ་ཧི་ཏ་དང་། །ཉི་མ་ཁྲི་ཤིང་དབྱེ་བ་ལ། །ལ་ལི་ཏི་ནི་མཛེས་དད་འདོད། །དོ་ཤལ་གྱི་ནི་ཁྱད་པར་ལ། །ལོ་ཧི་ཏཾ་ནི་ཙནྡན་དམར། །
ས་མཆོག་དང་ནི། གུར་ཀུམ་དང་། །ཁྲག་དང་ཁ་དོག་དམར་པོ་ལ། །ལོ་ཧི་ཏ་ནི། སའི་འདབ་མ། །ཆུ་ཀླུང་བྱེ་བྲག་ལ་ཡང་ངོ་། །པ་ནི་ཏཱ་ནི་སྐྱེད་མ་དང་། །ཤིན་ཏུ་ཆགས་དང་བཙུན་མོ་ལ། །བ་ནི་ཏཾ་ནི། སློང་དང་བསྟེན། །བ་ས་ཏི་ནི་མཚན་མོ་ཁྱིམ། །ཀུན་ཏུ་གནས་
དང་མཆོད་ངལ་གསོ། །བ་ཧ་ཏུ་ནི་ཁྱུ་མཆོག་ལམ། །བས་ཏི། གླང་པི་ཙི་བ། །བི་བ་ཏཾ་ནི་གཉིས་སྐྱེས་དང་། །བྱས་དང་མགོ་བོའི་ཆ་དག་ལ། །བ་སནྟ་ནི་ཁུ་བྱུག་དང་། །སྲན་མ། མངྒ་གླང་ཕྲུག་དང་། །ཐོབ་པ་དང་ནི་ཡལ་ཕྲན་ལའོ། །བཱ་སནྟཱི་ནི། སྤང་ལྡན། ཁྱུ། །ལྗོན་
ཤིང་སྐྱ་ནར་དག་ལ་ཡང་། །བ་སི་ཏཱ་ནི། གླང་པོ་དང་། །བུད་མེད་ལའོ་བཱ་སི་ཏཾ། །ནམ་མཁར་འགྲོ་བའི་སྐྲ་དང་ནི། །ཤེས་དང་གོས་ཀྱིས་དཀྲིས་པ་དང་། །དྲི་ཞིམ་བྱས་པ་དག་ལའོ། །བི་ཀྲྀ་ཏི་ནི་འཇིགས་རུང་དང་། །ནད་ཅན་དང་ནི་འདུས་མ་བྱས། །སྤྱོ་བ་དང་ནི་ནད་
ལའོ། །པི་ཀྲི་ཏི། བྲལ། འོང་མེད་འགྲོ། །བི་ཙིཏྟི་ནི་ལུས་ཆགས་དང་། །རྣམ་པར་ཆང་དང་སྒེག་པ་ལ། །བི་ཛཱ་ཏཱ་ནི་རབ་བཅོས་ལ། །རྣམ་པར་འགྱུར་དང་སྐྱེད་པ་ལ། །བི་ཏ་ཏཾ་ནི་ཁྱབ་པ་དང་། །རྒྱ་ཆེ་བ་ལ་མངོན་པར་བརྗོད། །བིརླུ་ཏ་ནི་གློག་གི་འོད། །ཐུན་མཚམས་
དད་ནི་འོད་མེད་ལ། །བི་དི་ཏཾ་ནི། ཤེས་དང་དོན། །བི་དྷཱ་ཏཱ་ནི། ཚངས་པ་དྲན། །པི་ན་ཏ་ནི་རབ་འདུད་དང་། །ཉམས་པ་དང་ནི་ལྷོད་པ་ལ། །བི་ན་ཏཱ་ནི་ཁྱུང་གི་མ། །ཕ་ཡི་ཁྱད་པར་དག་ལ་ཡང་། །བེ་ནཱི་ཏཾ་ནི་འདུལ་པ་ཐོབ། །ལེགས་འབབ་རྟ་དང་དབང་པོ་ཐུལ། །ཉེ་བར་
དྲངས་དང་ཁྲིད་པ་དང་། །ངེས་པར་ཟུང་དང་དག་སྐྱེས་སོ། །རྒྱལ་པོ་ཡུལ་ཅན་བི་ནེ་ཏཱ། །འོངས་དང་། སློང་བ་དི་པཏྟི། །ནད་དང་ཕྲན་ཚེགས། བི་པྲྀ་ཏཱ། །བི་པྲྀ་ཏཱཾ་ནི་གསལ་བའོ། །བི་བཏྟི་ནི་ཀུན་སྤྱོད་དང་། །རབ་ཏུ་འཇུག་དང་གར་དག་ལ། །བི་བིཀྟཾ་ནི་སྐྱེ་བོ་བྲལ། །
