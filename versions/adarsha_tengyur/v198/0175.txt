སྨན་ནི་ལྷག་མ་ཀུན། །སྲང་བཅུ་ཡོངས་སུ་སྦྱར་བར་བྱ། །སྦྲང་རྩི་རྒྱམ་ཚ་སྣུམ་རྣམས་དང་། །སྨན་ཕྱེ་ཁུ་བའི་རིམ་པ་ནི། །མས་གཏོང་རྐྱལ་བར་བཅུག་བྱས་ཏེ། །འདི་ནི་སྦྱོར་བའི་ཆོ་གར་བཤད། །གན་རྐྱལ་བཞག་ནས་བཏང་མ་ཐག །སྔས་ཀྱིས་བརྟེན་ཏེ་མས་གཏོང་ལ། །ཡིད་ནི་གཞན་
དུ་མ་ཡེངས་པས། །འཁྲུ་བ་ཙོག་པུས་གཏང་བར་བྱ། །འོང་དུས་དམ་པ་ཡུད་ཙམ་སྟེ། །དེ་ལས་གཞན་དུ་འཆི་བྱེད་ཡིན། །དེ་ལ་བཅོས་པའི་ཐབས་གྱུར་སྣུམ། །ཐལ་སྨན་བ་གཅིན་ཁུ་སྐྱུར་སྦྱར། །མྱུར་བས་སྣུམ་ལ་རྣོ་དྲོ་བའི། །མང་གཏོང་གཞན་དག་གཏང་བར་བྱ། །ཡང་ན་སྨན་གྱི་རེང་བུ་
དག །གཏོང་ཞིང་དུགས་ལ་སོགས་པ་བྱ། །རང་ཉིད་འོངས་པར་གྱུར་ན་ནི། །མས་གཏོང་གཉིས་པ་གཏང་བའམ། །ལན་གསུམ་ལན་བཞིའི་བར་དུ་ནི། །ཇི་སྲིད་ལེགས་པར་གྱུར་པར་གཏང་། །དེ་ཡིས་སྦྱོར་སོགས་བཀྲུ་སྨན་དང་། །འདྲ་བར་ཤེས་བྱ་སྦྱོར་ལེགས་ན། །ཆུ་དྲོན་འཇམ་པོས་
བཀྲུས་ནས་ནི། །ཤ་ཁུ་འབྲས་ཆན་ཉུང་ཟད་སྦྱིན། །མས་གཏོང་དྲག་པོས་ནད་རྣམས་ཀྱི། །དྲི་མ་བསྐྱོད་པས་བྱུང་གྱུར་པ། །དེ་ནི་ཆུ་དྲོན་གྱིས་བཀྲུ་དང་། །ཁ་ཟས་ཚུལ་བཞིན་ཐོས་བས་འབྱུང་། །གལ་ཏེ་རླུང་གིས་ཉམ་ཐག་ན། །སླར་ཡང་འཕྲལ་དུ་འཇམ་རྩི་གཏང་། །དེ་ནི་ལེགས་དང་མ་
ལེགས་དང་། །གྲགས་པ་སྣུམ་འཐུངས་འདྲར་ཤེས་བྱ། །གང་ཞིག་དུས་ནི་ཅུང་ཟད་གནས། །བཤང་དང་བཅས་པར་འབྱུང་འགྱུར་ཞིང་། །ཐུར་དུ་དབུགས་དང་སྣུམ་ལྷན་ཅིག །འབྱུང་འགྱུར་འཇམ་རྩི་ལེགས་པ་ཡིན། །བད་ཀན་ལ་ནི་མས་གཏོང་བའི། །འཇམ་རྩི་གཅིག་གམ་གསུམ་
གཏང་བྱ། །མཁྲིས་པ་ལ་ནི་ལྔའམ་བདུན། །རླུང་ལ་དགུའམ་བཅུ་གཅིག་གཏང་། །སླར་ཡང་དེ་ལ་ཟུང་མིན་པ། །མས་གཏོང་རྗེས་ལ་བྱ་བ་ཡིན། །བད་མཁྲིས་རླུང་ལ་རིམ་བཞིན་ཟས། །ཡུ་ཤ་འོ་མ་ཤ་ཁུ་གཏང་། །རླུང་འཇོམས་སྨན་ནི་ལེགས་བསྐོལ་བ། །དུར་བྱིད་རྒྱ་མཚར་བཅས་པར་
བྱ། །སྣུམ་མངར་སྐྱུར་ཤ་ཁུ་དྲོན་སྦྱར། །མས་གཏོང་གཅིག་ནི་རླུང་ལ་ཕན། །ནྱ་གྲོ་དྷ་སོགས་སྡེ་ཚན་བསྐོལ། །ཤུག་སོགས་ཀར་དཀར་དང་བཅས། །མངར་བསིལ་བུ་རམ་ཤིང་ཁུ་མར། །འོ་མ་སྦྲང་རྩིར་བཅས་མཁྲིས་སེལ། །དོང་ག་ལ་སོགས་བསྐོལ་བ་དང་། །དུག་མོ་ཉུང་སོགས་བཅས་པ་
