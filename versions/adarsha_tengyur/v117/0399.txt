རྒྱས་ཀྱི་ཞིང་རྣམས་རིན་པོ་ཆེ་སྣ་ཚོགས་དང་། ན་བཟའ་དང་། བླ་རེ་དང་། གདུགས་དང་། རྒྱལ་མཚན་དང་། བ་དན་དང་། རིན་པོ་ཆེའི་དྲིལ་བུའི་ཕྲེང་བ་དང་སྤོས་དང་། མེ་ཏོག་དང་། བདུག་པ་དང་། མར་མེ་དང་། ཕྲེང་བ་ལ་སོགས་པ་དང་། སིལ་
སྙན་མིའི་ལས་འདས་པ་འཕྲུལ་གྱི་རྒྱན་རྣམས་ཀྱིས་རབ་ཏུ་བརྒྱན་པ་དང་། མིའི་མ་ཡིན་པའི་འཕྲུལ་གྱི་ཞལ་ཟས་རོ་ལྔ་བརྒྱ་དང་ལྡན་པ་མཆོག་ཏུ་བདེ་བ་བསྐྱེད་པ་རྣམས་སངས་རྒྱས་ཀྱི་ཞིང་གི་རྡུལ་དང་མཉམ་པའི་བསྐལ་པར་སངས་རྒྱས་བཅོམ་ལྡན་འདས་ལ་གསོལ་ཞིང་ཕུལ་བ་བས། 
གང་གིས་བསྔོ་བའི་རྒྱལ་པོ་འདི་ཐོས་ནས། །ལན་ཅིག་ཙམ་ཡང་དད་པ་བསྐྱེད་པ་ནི། །ཞེས་བྱ་བ་སྟེ། འདི་ནི་འདི་ཉིད་དོ་ཞེས་ངེས་པ་བསྐྱེད་པར་བྱས་པ་སྟེ། དེ་ཡང་གང་ཞེ་ན། བྱང་ཆུབ་མཆོག་གི་རྗེས་སུ་རབ་མོས་ཤིང་། །ཞེས་བྱ་བ་ཐེག་པ་ཆེན་པོ་ལ་ཞུགས་པ་དེས་སྔ་
མའི་སྦྱིན་པའི་བསོད་ནམས་བས་འདིའི་བསོད་ནམས་ཁྱད་པར་ཅན་དུ་འགྱུར་རོ། །སྡིག་པའི་གྲོགས་པོ་ཡོངས་སུ་སྤངས་པས་དེ་བཞིན་གཤེགས་པ་མཐོང་བ་སྟེ། དེ་ཡང་གང་ཞེ་ན། དེས་ནི་ངན་སོང་ཐམས་ཅད་སྤང་བར་འགྱུར། །ཞེས་བྱ་བ་ལ་སོགས་པའོ། །དེས་
སྤང་བར་བྱ་བ་ནི་སྡིག་པ་དང་ངན་སོང་གསུམ་མོ། །དེས་ནི་གྲོགས་པོ་ངན་པ་སྤང་བར་འགྱུར། །ཞེས་བྱ་བ་ནི་ཉན་ཐོས་ལ་སོགས་པ་མི་དགེ་བའི་བཤེས་གཉེན་རྣམས་སོ། །འཆི་བའི་ཚེ་བཅོམ་ལྡན་འདས་སྣང་བ་མཐའ་ཡས་དེ་བཞིན་གཤེགས་པ་དེ་ཡང་དེས་མྱུར་དུ་གདོན་མི་ཟ་བར་མཐོང་
བར་འགྱུར་ཏེ། གང་གིས་མཐོང་ཞེ་ན། གང་གིས་འདི་ལྟར་བཟང་པོ་སྤྱོད་པའི་སྨོན་ལམ་འདིའི་དོན་དང་ཚིག་བཅས་པར་སེམས་ཀྱིས་ཆུབ་པའོ། །ཚེའི་འཚོ་བ་དགེ་བ་ཐོབ་པ་ནི་བྱང་ཆུབ་སེམས་དཔའ་དང་སྐལ་བ་མཐུན་པ་ཐོབ་པ་སྟེ་དེ་ཡང་གང་ཞེ་ན། དེ་གང་རྙེད་པ་རབ་རྙེད་བདེ་བར་
འཚོ། །ཞེས་བྱ་བ་ལ་སོགས་པའོ། །བླ་ན་མེད་པའི་སྨོན་ལམ་རིན་པོ་ཆེ་ཐོབ་པ་ནི་རབ་ཏུ་རྙེད་པ་སྟེ། བདེ་བར་འཚོ་བ་རྙེད་པ་ཡང་སྨོན་ལམ་རིན་པོ་ཆེ་འདི་ལྟ་བུ་ཐོབ་པས་སོ། །དེ་ལྟ་བས་ན་ཚེའི་མཆོག་ཐོབ་པར་བྱ་བའི་དངོས་པོ་ཐོབ་པར་འགྱུར་བའི་ཕྱིར་རོ། །མི་ཚེ་འདིར་ཡང་དེ་དག་
