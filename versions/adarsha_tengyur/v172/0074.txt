དང་གཉིས་སྐྱེས་གཉིས། །དེ་ཡི་བསྟན་པ་ཉིད་དང་ངེས་པ་ཐོས་གྱུར་ནས། །གཏི་མུག་བཞིན་ལྡན་རྗེས་སུ་སོང་ནས་སྡུག་བསྔལ་བ། །དལ་བུས་སོང་ནས་གྲོང་ཁྱེར་ཉིད་དུ་སོང་བར་གྱུར། །དེ་ནས་དེ་ལ་བརྩེ་དང་མི་སྐྱོང་ལ་གུས་ལས། །དེ་དག་ལྟོས་བྱས་རབ་ཏུ་སོང་ཞིང་གནས་གྱུར་
ལ། །གསལ་ཞིང་སྣང་བས་ཉི་མ་བལྟ་བར་དཀའ་བ་ལྟར། །དེ་ནི་ལམ་དུ་ལྟ་བར་མ་ནུས་འདོར་བར་མིན། །མཆོག་ཏུ་གཤེགས་ཏེ་གཤེགས་པ་ཤེས་ཕྱིར་དེ་དག་གིས། །སོ་པའི་སྐྱེས་བུ་གཙང་མ་སྦས་པ་རྣམས་བྱས་ནས། །དགའ་ཞིང་སྲས་པོ་འདོད་པའི་རྒྱལ་པོ་ལ་སོང་ནས། །ཇི་ལྟར་
བལྟའམ་ཞེས་ནི་ཇི་ལྟར་ཡང་སོང་ངོ་། །སངས་རྒྱས་ཀྱི་སྤྱོད་པ་ཞེས་བྱ་བའི་སྙན་དངགས་ཆེན་པོ་ལས། གཞོན་ནུ་རྗེས་སུ་ཚོལ་བའི་ལེའུ་སྟེ་དགུ་པའོ།།      །།སྐུ་སྟོད་རྒྱས་ཤིང་རྒྱ་ཆེན་རྒྱལ་པོའི་སྲས་པོ་སྟེ། །བསྲེག་བྱ་དང་བསྔགས་བདག་པོ་དེ་དག་རྣམས་སྤངས་ཏེ། །དབའ་
རླབས་གཡོ་བའི་ཆུ་བོ་གང་གཱ་རྒལ་ནས་ནི། །དཔལ་དང་ལྡན་པའི་རྒྱལ་པོའི་ཁབ་ཏུ་གཤེགས་པར་གྱུར། །རི་རྣམས་དག་གིས་ལེགས་པར་སྦས་དང་བརྒྱན་པ་དང་། །དགེ་དང་དཀའ་ཐུབ་གཏེར་བས་གཟུང་དང་གཙང་བ་སྟེ། །ཞི་བ་རང་བྱུང་ཚངས་པ་མཐོ་རིས་ལོགས་སུ་བཞིན། །རི་
ལྔས་མཚན་པའི་གྲོང་ཁྱེར་རབ་ཏུ་ཐོབ་པར་གྱུར། །ཟབ་མོར་གྱུར་པའི་མདངས་དར་དེ་ཡི་ལུགས་གསལ་བ། །སྐྱེས་བུ་རྣམས་ནི་ཤིན་ཏུ་གནོན་པ་མཐོང་གྱུར་ནས། །ཁྱུ་མཆོག་རྒྱལ་མཚན་དེ་ཡི་བརྟུལ་ཞུགས་བརྟན་པ་བཞིན། །དེར་ནི་དེ་ཚེ་སྐྱེ་བོ་རྣམས་ནི་ཡ་མཚན་གྱུར། །དེ་རབ་
མཐོང་ནས་གཞན་གང་མ་སོང་དེ་གནས་ཤིང་། །དེར་ནི་གང་ཞིག་ལམ་ན་གནས་དེ་རྗེས་འགྲོ་ཞིང་། །གང་ཞིག་སྐྱེན་པོར་སོང་སྟེ་བརྟན་པོར་སོང་གྱུར་ལ། །ཁ་ཅིག་བསྡད་པ་གང་ཡིན་དེ་ནི་ལངས་པའོ། །སྐྱེ་བོ་ཁ་ཅིག་དེ་ལ་མཆོད་པའི་ལག་པ་ཡིས། །བསྟི་སྟང་བྱས་ནས་ཁ་ཅིག་
མགོ་བོས་ཕྱག་འཚལ་ཞིང་། །ཁ་ཅིག་གཉེན་པོའི་ཚིག་གིས་མངོན་པར་སྟོད་བྱེད་ཅིང་། །ཁ་ཅིག་དེ་ལ་རབ་ཏུ་མཆོད་ནས་སོང་བར་གྱུར། །དེ་རབ་མཐོང་ནས་སྣ་ཚོགས་ཆ་ལུགས་རྣམས་ངོ་ཚ། །རབ་བཀྲམ་ཚིག་ནི་ལམ་དུ་སྨྲ་བ་མེད་པར་སོང་། །དངོས་སུ་གཞན་དུ་ཆོས་རྣམས་
