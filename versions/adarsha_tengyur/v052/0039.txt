མི་བྱའོ་ཞེས་པའི་དོན་ཏོ། །ཡང་དང་ཡང་དུ་བརྟགས་པ་ནི། །ཞེས་པ་ནི་གདན་དང་ས་བོན་དང་མཚན་མ་ལ་སོགས་པའམ། རྩ་འམ། ཐིག་ལེ་འམ། རླུང་གི་གོང་བུ་འམ། སྒྲ་འམ། ཟླ་བའི་དུམ་བུ་འམ། ལྟེ་བར་མེའི་གཟུགས་ལ་སོགས་
པ་ཡང་དང་ཡང་དུ་བརྟག་པར་མི་བྱའོ་ཞེས་པའི་དོན་ཏོ། །ཅིའི་ཕྱིར་མི་བརྟག་སྙམ་པ་ལས། །རྟོག་པ་བསྒོམ་པའི་བདག་ཉིད་ཀྱིས། །དངོས་གྲུབ་དོན་དུ་སྒྲུབ་པའི་ཐབས། །ཞེས་བྱ་བ་སྨོས་ཏེ། རྡོ་རྗེ་འཆང་ཆེན་པོའི་གནས་ཀྱི་དངོས་གྲུབ་འདོད་པ་གང་དག་ཡིན་པ་དེ་
དག་གིས་ཀུན་བརྟགས་པའི་རྣལ་འབྱོར་གྱིས་རྡོ་རྗེ་འཆང་གི་གནས་ལ་རེག་པ་མེད་དེ་བརྟགས་པ་ཡིན་པའི་ཕྱིར་རོ། །གང་ཡང་གནས་དེ་ནི་མ་བརྟགས་པ། འདུས་མ་བྱས་པ། རྒྱུ་རྐྱེན་གྱིས་མ་བསྐྱེད་པ། རྡོ་རྗེ་ལྟར་གཞིག་ཏུ་མེད་པའི་ཕྱིར་རོ། །དེ་བས་ན་དེའི་ཕྱིར། 
དེ་ཀུན་རྟག་ཏུ་རྣལ་འབྱོར་པས། །དོར་བ་ཉིད་དུ་ཡོངས་སུ་བརྟག །ཅེས་བྱ་བ་སྨོས་སོ། །བརྟག་པའི་བདག་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་གང་ཡིན་པ་དེ་དག་ཡོངས་སུ་སྤངས་ཞེས་བྱའོ།    །དེ་བཞིན་དུ་ཡང་ཞེས་པ་ནི་མདོར་བསྡུས་པའི་ཚུལ་ལོ། །ཇི་སྲིད་ཇི་སྲིད་རྣམ་རྟོག་པ། །དེ་སྲིད་
ཐམས་ཅད་ཀུན་ནས་བརྫུན། །ཞེས་བྱ་བ་ནི་མདོར་ན་ཇི་སྲིད་དུ་རྣམ་པར་རྟོག་པ་ཡིན་པ་དེ་སྲིད་དུ་བརྫུན་པ་དང་། སླུ་བའི་ཆོས་ཅན་ཡིན་པས་སྤང་ཞེས་པའོ། །མི་འགྱུར་བའི་ཆོས་ཅན་གང་ཡིན་ཞེ་ན། གང་ཞིག་དེ་ཉིད་མ་བརྟགས་པས། །དེ་བཞིན་ཡང་དག་དེ་བདེན་
པ། །ཞེས་བྱ་བ་སྨོས་ཏེ་གང་ཞིག་སྟེ་ཏིང་ངེ་འཛིན་གང་ཞིག་མ་བརྟགས་པ་དང་། མ་བཅོས་པ་དང་། བཙལ་བ་ཐམས་ཅད་དང་བྲལ་བ་དེ་ནི་བདེན་པ་དང་། བརྟག་པ་དང་། བརྟན་པ་དང་། གཞིག་ཏུ་མེད་པའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ཡང་འདི་སྐད་དུ། གང་དག་
སྟོང་ཉིད་མི་ཤེས་པ། །དེ་དག་མྱ་ངན་འདས་མི་ཤེས། །དེ་ཕྱིར་དངོས་དང་དངོས་མེད་པ། །འཇིག་པས་སྟོང་ཉིད་ཤེས་པར་བྱ། །ཞེས་གསུངས་པ་དང་། གཞན་ཡང་། མ་བཅོས་ལམ་ལ་གནས་ན་ནི། །ཐམས་ཅད་བྱས་ཀྱང་གྲོལ་འགྱུར་ཞིང་། །ཞེས་བྱ་བ་ལ་
