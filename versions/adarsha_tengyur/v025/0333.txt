ལུས་དང་ནི། །བླ་མའི་མན་ངག་གིས་མཚོན་ཆ། །དེ་ཉིད་ཁ་དོག་མེད་ཅིང་མཚན་ཉིད་མེད། །ཐུབ་དབང་འཆི་བའི་གནས་སྐབས་ལྟ་བུར་རྟོགས། །དག་པའི་ལུས་སུ་བསྒོམས་ནས་སྔགས་ཀྱིས་ནི། །མན་ངག་ཇི་བཞིན་སྒོ་རྣམས་བསྡམས་ནས་སུ། །རྣམ་
ཤེས་ས་བོན་རླུང་ལ་ཡང་དག་གནས། །ཚིགས་དགུ་ནས་ནི་སྟེང་དུ་འཕོ་བར་བྱ། །འཁོར་བའི་རྒྱ་མཚོ་འཇིགས་སུ་རུང་བའི་འདམ་དུ་བྱིང་བ་ཡི། །འཚེ་ཞིང་རྨོངས་པར་འགྲོ་བར་ཞུགས་གང་ཡིན། །ཉི་མ་སྟོང་གིས་མངོན་ཤེས་དྲུག་ལྡན་ལུས། །བླ་ན་མེད་པའི་གོ་འཕང་
ཐོབ་པར་འགྱུར། །ནང་གི་སེམས་ལས་སྤྲུལ་པའི་སྦྱོར་བའི་གདན་གྱི་བསམ་གཏན་གྱི་དེ་ཁོ་ན་ཉིད་དང་། བདག་མེད་མའི་མངོན་པར་བྱང་ཆུབ་པ་དང་། གོང་དུ་འཕོ་བ་སྟེ་དུམ་བུ་གསུམ་པའོ།།      །།རི་ཡི་རྩེར་གནས་གེ་སར་དང་། །སྣ་ཚོགས་འདབ་མ་འཇམ་པ་
ལ། །དེ་ཡི་དབུས་གནས་དཔའ་བོ་ནི། །འདོད་དོན་ཐམས་ཅད་སྒྲུབ་བྱེད་པ། །འབྱུང་པོ་ཀུན་གྱི་གནས་ཡིན་ཏེ། །ཐམས་ཅད་བསྐྱེད་པའི་རྒྱུ་ཡང་ཡིན། །ལུས་ཅན་ཀུན་ལ་ཁྱབ་པ་སྟེ། །ཐམས་ཅད་ཐམས་ཅད་བདག་ཉིད་གནས། །ཧཾ་གི་ས་བོན་ཡང་དག་འབྱུང་། །འཇིག་    
རྟེན་གསུམ་བསྐྱེད་གཅོ་བོ་ཡིན། །ཡིད་བཞིན་ནོར་བུའི་བདག་ཉིད་དེ། །རིན་ཆེན་སྣ་ཚོགས་འབྲས་བུ་འབྱུང་། །མཚན་ནི་སུམ་ཅུ་རྩ་གཉིས་ལྡན། །དཔེ་བྱད་བཟང་པོ་བརྒྱད་ཅུར་བཅས། །རླུང་བསྐྱོད་པ་ལ་ཞོན་ནས་སུ། །རྟག་ཏུ་ལྟེ་བར་འགྲོ་འོང་བྱེད། །མགོ་བོ་མཐའ་ན་རླུང་
དཀྱིལ་འཁོར། །གཞུ་ཡི་དབྱིབས་ལྟར་གནས་པ་སྟེ། །སྟེང་དང་འོག་གི་རླུང་ལ་ནི། །འདབ་མ་བཞི་པོ་ཡང་དག་གནས། །ལྕེ་དང་རྐན་ནི་སྦྱར་ནས་སུ། །རྟག་ཏུ་སྒྲ་དང་བཅས་བྱེད་པ། །གཅིག་ཏུ་གྱུར་ཅིང་ཐིམ་གྱུར་པ། །དེར་ནི་ཕྱག་རྒྱ་ཆེན་པོར་བརྗོད། །འཇུག་པའི་རླུང་ནི་བསྡམས་
ནས་སུ། །སྟེང་དུ་དྲངས་ཏེ་སྦྱར་བར་བྱ། །ཇི་སྙེད་ལྡོག་པར་འགྱུར་བར་དུ། །རྣལ་འབྱོར་པ་ཡིས་བཙིར་བར་བྱ། །ཇི་ལྟར་མར་ལ་མར་བླུགས་ཤིང་། །ཇི་ལྟར་ཆུ་ལ་ཆུ་བཞག་དང་། །ནམ་མཁའ་དང་ནི་ནམ་མཁའ་བཞིན། །རང་གིས་རང་ནི་རིག་པར་འགྱུར། །འཇུག་དང་འབྱུང་དང་
