དང་། རྡོ་རྗེ་གར་དང་། རྡོ་རྗེ་འཇིགས་སུ་རུང་བ་དང་། རྡོ་རྗེ་རོ་ལངས་དང་། རྡོ་རྗེ་ཐོ་བ་དང་། སྒོ་བཞི་བྱིན་གྱིས་བརླབ་པ་དང་། རྡོ་རྗེ་གདུ་བུ་དང་། བུད་མེད་ཀྱི་འཕྲུལ་གྱི་སྟང་སྟབས་དང་། རྡོ་རྗེ་བྲལ་བ་དང་། སྟོབས་ཆེན་གདེངས་པ་དང་། རྡོ་རྗེ་མཆེ་བ་དང་། རྡོ་རྗེ་གད་མོ་དང་། རྡོ་རྗེ་སྒྲ་དང་། 
རལ་གྲིའི་ཐབས་དང་། རྡོ་རྗེའི་ཐབས་དང་། འཁོར་ལོའི་ཐབས་དང་། གཏུན་ཤིང་གི་ཐབས་དང་། བ་དན་གྱི་ཐབས་དང་། གཞུའི་ཐབས་དང་། ཞགས་པའི་ཐབས་དང་། པདྨའི་ཚུལ་དང་། བེ་ཅོན་གྱི་ཚུལ་དང་། མི་འཇིགས་པའི་ཚུལ་དང་། ལྕགས་ཀྱུའི་ཐབས་དང་། 
གཤོལ་བའི་ཚུལ་དང་། མདུང་གི་ཐབས་དང་། མདའ་གཟུང་བའི་ཐབས་དང་། མདའ་འཕང་བའི་ཐབས་དང་། ཐག་མདུང་བསྐོར་བའི་ཐབས་དང་། མཆོག་སྦྱིན་པའི་ཚུལ་དང་། མདུང་ཐུང་གི་ཐབས་དང་། མཚོན་རྩེ་གཅིག་པའི་ཐབས་དང་། ཕྱག་འཚལ་བའི་

མཁར་བྱས་ཏེ། ཧཱུཾ་ལཾ་ཧཱུཾ་ཞེས་བརྗོད་པས་གསེར་གྱི་རང་བཞིན་དུ་བྱས་ལ། དེའི་དབུས་སུ་པདྨ་ནས་བཛྲ་བནྡྷ་ཧཱུཾ་ཞེས་ཅི་ནུས་བཟླས་པའི་བར་དུ་བྱ་སྟེ། དེ་ནས་ཡང་ས་གཞི་དྲི་དང་བའི་ལྕི་བས་བྱུགས་ལ། ཕྱོགས་མཚམས་བརྒྱད་དུ་མར་མེ་བརྒྱད་དང་བདུག་སྤོས་དུ་བ་འཐུལ་བ་
