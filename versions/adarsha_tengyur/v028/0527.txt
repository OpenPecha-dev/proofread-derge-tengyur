བཏགས་པའི་ཕྱི་མ་སྦྱར་བ་བྱས་ལ། སྔགས་བཟླས་ནས་གཟུགས་བརྙན་ལ་གདབ། ཕུར་པ་དམར་ལྷང་ངེ་བ་བསྲེགས་ལ་སྙིང་གར་གདབ། དེ་ལྟར་མཚོན་རྣོན་པོས་གཟུགས་དུམ་བུར་གཏུབས་ལ། ཕྱེ་མར་བྱས་ལ། ད་དུ་རའི་འབྲས་བུ་ཡོད་ན་རབ། མེད་ན་ཡང་ལུང་
ཏང་ཙེའི་འབྲས་བུ་བརྒྱ་རྩ་བརྒྱད་བླངས་ལ།  རྩ་བ་མོའི་ནང་དུ་བཙགས་ནས་གཟུགས་བརྙན་གྱི་ཕྱེ་མ་དང་བསྲེས་ནས། དཔལ་ལྷ་མོའི་སྦྱོར་བས་སྐྲ་བཤིག་གཅེར་བུར་ཕྱུང་ནས། ཁ་ལྷོ་ཕྱོགས་སུ་བལྟས་ཏེ་ཚེར་མ་ལ་སྦྱིན་སྲེག་བྱའོ། །དེ་ནས་སྔགས་ལ། ཨོཾ་ཧཱུཾ་ཤྲཱི་དེ་ཝཱི་ཀཱ་ལི་
མ་ཧཱ་ཀཱ་ལི་ཧཱུཾ་བྷྱོཿ། མ་མ་ཤ་ཏྲཱུཾ་ཨཱ་མུ་ཀཾ་མཱ་ར་ཡ་ཨུན་མནྟ་ཀུ་རུ་ཧཱུཾ་ཕཊ་ཅེས་བརྒྱ་རྩ་བརྒྱད་བཟླས་ཤིང་། རྫས་དང་བསྟུན་ཞིང་བསྲེག་པར་བྱའོ། །དེ་ལྟར་བྱས་ན་སྨྱོ་ནས་འཆི་བར་འགྱུར་རོ། །ཧཾ་གིས་ཐལ་བ་དེ་ཉིད་བླངས་ཏེ། དེ་ཉིད་སྤྱི་བོར་བཏབ་ན་ལོ་ཉི་ཤུ་རྩ་གཅིག་ཏུ་སྨྱོ་བར་འགྱུར་རོ་ཞེས་
ཀྱང་བཤད་དོ། །གཞན་ཡང་བདག་གི་རྨི་ལྟས་སུ། །བུད་མེད་ནག་མོ་ཁྲོས་པ་ཡིས། །གང་ཞིག་འཆིང་བས་བཅིངས་གྱུར་ན། །མཚན་མོ་ནག་མོར་དེ་ཤེས་བྱ། །བསྒྲུབ་བྱ་ཉི་མ་བདུན་ན་ནི། །འཆི་སྟེ་གཤིན་རྗེའི་ཡུལ་དུ་འགྲོ། །གལ་ཏེ་རེ་ཞིག་ཅུང་ཟད་འགོར་ན་ཡང་། །ཉིན་ཞག་
བདུན་དུ་བསམ་གཏན་འདི་བྱའོ། །བསྒྲུབ་བྱ་ལུས་པོ་སྐམས་ལ་རིད་པ། སྐྲ་འཛིངས་པ། སྦུབ་པ། ཁུ་བ་དང་། རྡུལ་དང་ཁྲག་འཛག་པ། འདར་བའི་ཆ་བྱད་ཅན་ཞིག་མདུན་དུ་མངོན་སུམ་དུ་བསམ་མོ། །བདག་ལྷ་མོར་གནས་པའི་ཧཱུཾ་ལས་འོད་ཟེར་ཉི་མ་དང་མེ་སྟག་ལྟ་བུ་འབྱུང་
བ་ལས། བསྐལ་པའི་མེ་བཞིན་འབར་ཞིང་། འུར་འུར་དང་། ཏོག་ཏོག་དང་། ཆེམ་ཆེམ་ཅན་དུ་དྲག་པོའི་ཡི་གེ་ཧཱུཾ་དུ་གྱུར་ནས། ཧཱུཾ་དེ་ལས་རྡོ་རྗེའི་ཁབ་ཏུ་གྱུར་པར་བསམ་མོ། །རྡོ་རྗེ་འབར་བའི་ཁབ་དེ་དགྲ་བོའི་བཤང་ཁུང་གི་སྒོར་ཞུགས་ཏེ། སྙིང་ཕུག་ནས་ཁའི་ནང་ནས་སམ། སྤྱི་གཙུག་
གི་ཚངས་པའི་བུ་ག་ནས་ཕྱུང་ནས་ཆེར་འབར་བར་བསམས་ཏེ། སྙིང་པོའི་སྔགས་ཀྱི་མཇུག་ལ། ཆེ་གེ་མོ་ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཅེས་པའི་སྔགས་ཀྱིས་ཁས་བུབ་ཏུ་ལྷུང་བར་བྱའོ། །དེ་ལྟ་བུས་བསམ་གཏན་གྱི་ལྷ་ཡང་འཆི་ན་འཁོར་བ་ན་གནས་པའི་མི་རྣམས་ལ་ལྟ་
