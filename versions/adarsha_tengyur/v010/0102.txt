དཀྱིལ་འཁོར་མི་བརྩམ་བློ་ལྡན་གྱིས། །ཨ་ཤོ་ཀ་སོགས་ལྡན་པ་ཡི། །ས་གཞི་དཀར་པོ་འཇམ་པ་ཡི། །བྱང་ཤར་གཡས་འབབ་ཆུ་འབབ་ན། །དེར་ནི་བླ་མེད་དངོས་གྲུབ་ཐོབ། །བརྐོས་པས་ས་ཆུས་མ་ཁེངས་ན། །དེས་ནི་དགོངས་པ་ཉམས་པར་འགྱུར། །བརྐོས་པས་ས་ཆུས་གང་
ལྷག་ན། །བསམས་པའི་དགོངས་པ་འགྲུབ་པར་འགྱུར། །ས་གཞིའི་མཚན་ཉིད་བརྟག་པའོ།།      །།དེ་ནས་སྔགས་པས་ས་གཞི་ལེགས་པར་བརྟགས་པའི་དབུས་སུ་མཐོ་ཞིང་བདེ་བའི་སྟན་ལ་འདུག་སྟེ། ཁྲུས་བྱས་ཏེ་དྲི་ཞིམ་པོས་བྱུགས་ལ། ན་བཟའ་དང་ཕྱག་རྒྱ་དྲུག་གིས་ལེགས་
པར་བརྒྱན་ལ། དགོས་པ་ཐམས་ཅད་ཡོངས་སུ་ཚོགས་པར་བྱས་ནས་རྒྱུན་དུ་བསྲུང་བའི་སྔགས་བཟླས་ཏེ། སྔགས་པ་མཉམ་པར་གཞག་པ་དང་ལྡན་པས་རྡོ་རྗེ་དང་དྲིལ་བུ་གཟུང་སྟེ། མཆོད་པ་དང་གཏོར་མ་སྔོན་དུ་འགྲོ་བ་བྱས་ནས། དགོངས་ཀའི་དུས་སུ་ཏིང་ངེ་འཛིན་གསུམ་རྫོགས་པར་
བྱས་ལ། ཇི་ལྟར་བཤད་པའི་དྲག་པོའི་སྦྱིན་སྲེག་གིས་བགེགས་བསྐྲད་ནས་ཕུར་བུ་བཏབ་སྟེ། རྡོ་རྗེ་ར་བ་དང་གུར་ལ་སོགས་པ་བསྒོམ་པར་བྱའོ། །དེ་ནས་སློབ་མ་བསྲུང་བའི་དོན་དུ་ཞི་བའི་སྦྱིན་སྲེག་བྱའོ། །དེ་ནས་གཏོར་མ་རྒྱ་ཆེན་པོ་ཇི་ལྟ་བའི་གདམས་ངག་གིས་བྱིན་གྱིས་བརླབས་ལ།
