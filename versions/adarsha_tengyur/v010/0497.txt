རྒྱལ་བའི་ཕྱག་རྒྱར་གྱུར་པ་ནི། ལག་མཐིལ་གཉིས་རྒྱབ་སྦྱར་ཏེ། སྲིན་ལག་གཉིས་མདུད་པ་བཅིང་། མཐེའུ་ཆུང་དང་གུང་མོ་དང་། མཐོ་བོང་རྩེ་སྤྲད། མཛུབ་མོ་ལྕགས་ཀྱུ་ལྟར་བྱས་པ་འདི་དཔྲལ་བའི་གནས་སུ་གཞག་པ་ནི་འཇིག་རྟེན་གསུམ་ལས་རྒྱལ་བའི་ཕྱག་
རྒྱའོ། །འདི་ལྟར་བསྒོམ་པའི་དུས་སུ་གསང་སྔགས་བཟླས་པ་ནི། ཨོཾ་ཀུ་རུ་ཀུལླེ་ཧཱུཾ་ཧྲཱིཿསྭཱ་ཧཱ། ཁྲིར་བཟླས་བརྗོད་བྱས་ན་འགྲོ་བ་ཐམས་ཅད་དབང་དུ་འགྱུར་རོ། །ཁྲག་ཁྲིག་གིས་སྔགས་པའོ། །འབུམ་གྱིས་རྒྱལ་པོ་དབང་དུ་འགྱུར་རོ། །མེ་ཏོག་ཨུཏྤ་ལ་དང་ཤིང་ཨརྐས་སྦྱིན་སྲེག་བྱས་ན་
འདོད་པའི་དོན་འཐོབ་པོ། །བྷེ་ཏའི་མེ་སྦར་ལ། མ་དུ་བ་དང་འབྲས་ཀྱི་ལག་པན་གྱིས་ལན་བརྒྱ་སྟོང་དུ་སྦྱིན་སྲེག་བྱས་ན། རྒྱལ་པོའི་བློན་པོ་ཡང་དབང་དུ་འགྱུར་རོ། །ཨུཏྤལ་སྟོང་གིས་བློན་པོ་ཆེན་པོ་ཡང་དབང་དུ་འགྱུར་རོ། །རྩི་མར་གྱི་མར་མེ་སྟོང་ལྷ་མོ་ལ་ཕུལ་བས་རྒྱལ་པོ་ཡང་དབང་དུ་འགྱུར་
རོ། །ལྷ་མོ་ཀུ་རུ་ཀུལླེ་བདག་ཉིད་བཅུ་གསུམ་པ་རྫོགས་སོ།།      །།༄༅༅། །བཅོམ་ལྡན་འདས་མ་ཀུ་རུ་ཀུལླེ་ལ་ཕྱག་འཚལ་ལོ། །ལྷ་མོ་ཆེན་མོ་ཀུ་རུ་ཀུལླེ། །མཆོག་ཏུ་དགའ་བའི་གཟུགས་ཅན་མ། །འདོད་དོན་ཐམས་ཅད་བསྒྲུབ་པ་ལ། །ཡང་དག་གུས་པས་ཕྱག་འཚལ་ལོ། །ཡེ་ཤེས་འདོད་པའི་བླ་
མ་ཡིས། །སློབ་མའི་བསོད་ནམས་དོན་དུ་ནི། །དེ་ཡི་སྒྲུབ་ཐབས་མདོར་བསྡུས་པ། །རྒྱུད་ཀྱི་རིམ་བཞིན་བདག་གིས་བཤད། །དང་པོ་ཁོ་ནར་རྣལ་འབྱོར་པས་ལྷའི་ཁང་པར་ཉེ་བར་འདུག་སྟེ། ས་གཞི་ཀུན་དུ་དྲི་བཟང་པོའི་མཎྜལ་བྱས་ལ། མདུན་དུ་བླ་མ་དང་། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་
དཔའ་དང་། ལྷ་མོའི་སྐུ་བསམས་ལ། མེ་ཏོག་དམར་པོ་དང་། མར་མེ་ལ་སོགས་པས་ཡང་དག་པར་མཆོད་དེ། སྡིག་པ་བཤགས་པ་ལ་སོགས་པ་བྱའོ། །སྡིག་བྱས་ཐམས་ཅད་བདག་གིས་བཤགས། །དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་འགྲོ་བྱ། །འགྲོ་བའི་བསོད་ནམས་རྗེས་ཡི་རང་། །སངས་
རྒྱས་བྱང་ཆུབ་ཡོངས་སུ་བསྔོ། །དང་པོ་བྱམས་པའི་བསམ་གཏན་བྱ། །དེ་བཞིན་གཉིས་པ་སྙིང་རྗེ་བསྒོམ། །གསུམ་པ་དགའ་བ་བསྒོམ་པར་བྱ། །ཐམས་ཅད་རྗེས་ལ་བཏང་སྙོམས་སོ། །དེ་རྗེས་སྔགས་འདི་བརྗོད་པར་བྱ། །ཨོཾ་ཤུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་ཝ་ཨཱཏྨ་ཀོ྅ཧཾ། རང་བཞིན་སྟོང་པར་རྣམ་པར་བསྒོམ། །
