སུ་བྱིན་གྱིས་བརླབ་པར་བྱ། །ཨོཾ་ཤོ་དྷ་ནི་ཤོ་དྷ་ནི་སརྦ་པཱ་པཾ་ཨ་པ་ན་ཡ་ཧཱུཾ། སྡིག་པ་ཐམས་ཅད་དགུག་པའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་པཱ་པཾ་ཤོ་དྷ་ནི་ཧཱུཾ་སྭཱ་ཧཱ། སྡིག་པ་ཐམས་ཅད་སྦྱོང་བའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་ཏྲ་ཐ། ངན་སོང་ཐམས་ཅད་དྲལ་བའི་སྔགས་སོ། །ཨོཾ་སརྦ་
བིཏ་སརྦ་ཨཱ་བ་ར་ཎ་བི་ཤོ་དྷ་ནི་མུ་ནེ་ཧཱུཾ་ཕཊ། ཤཱཀྱ་ཐུབ་པའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་ཧཱུཾ་ཧཱུཾ། རྡོ་རྗེ་གཙུག་ཏོར་གྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་ཏྲཱཾ་ཏྲཱཾ། རིན་ཆེན་གཙུག་ཏོར་གྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་ཧྲཱིཿཧྲཱིཿ། པདྨ་གཙུག་ཏོར་གྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་ཨཿཨཿ། སྣ་ཚོགས་གཙུག་ཏོར་
གྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་ཨོཾ་ཨོཾ། གཙུག་ཏོར་གཟི་བརྗིད་ཀྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་ཧཱུཾ་ཧཱུཾ། གཙུག་ཏོར་རྒྱལ་མཚན་གྱི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་ཕཊ། གཙུག་ཏོར་རྣོན་པོའིའོ། །ཨོཾ་སརྦ་བིཏ་ཀྲིཾ་ཏྲཊ། གཙུག་ཏོར་གདུགས་དཀར་གི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་མ་ཧཱ་བཛྲོད་བྷ་
བ་དཱ་ན་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཧཱུཾ། སྒེག་མོའིའོ། །ཨོཾ་སརྦ་བིད་མ་ཧཱ་བཛྲོད་བྷ་བ་ཤཱི་ལ་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཏྲཱཾ། །ཕྲེང་བ་མའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་མ་ཧཱ་བཛྲོད་བྷ་བ་ཀྵཱ་ནྟི་པ་ར་མི་ཏ་པཱུ་ཛེ་ཧྲཱིཿ། གླུ་མཁན་མའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིད་མ་ཧཱ་བཛྲོད་བྷ་བ་བཱིརྱ་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཨཿ། གར་མཁན་མའི་
སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ཨ་པ་ཡ་གནྡྷ་ན་དྷ་མ་དྷ་མ་དྷ་མ་དྷུ་པ་ཡ་དྷྱན་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། བདུག་པ་མའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་དུརྒ་ཏི་པ་རི་ཤོ་དྷ་ནི་ཀླེ་ཤཙྪེ་ད་ན་པུཥྤ་བི་ལོ་ཀི་ནི་པྲ་ཛྙཱ་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཏྲཱི་ཧཱུཾ་ཕཊ། མེ་ཏོག་མའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ཨ་པ་ཡ་
བི་ཤོ་དྷ་ནི་ཛྙཱ་ན་ཨཱ་ལོ་ཀ་ཀ་རི་པྲ་ཎི་དྷ་ན་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཧྲཱི་ཧཱུཾ་ཕཊ། མར་མེའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ཨ་པ་ཡ་གནྡྷ་བི་ནཱ་ཤ་ནི་བཛྲ་གནྡྷ་ཨུ་པ་ཡ་པཱ་ར་མི་ཏཱ་པཱུ་ཛེ་ཨཿཧཱུཾ་ཕཊ། བྱུག་པ་མའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ན་ར་ཀ་ག་ཏེ་ཨཱ་ཀཪྵ་ཎི་ཧཱུཾ་ཛཿཕཊ། རྡོ་རྗེ་ལྕགས་ཀྱུའི་སྔགས་
སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ན་ར་ཀ་ཨུདྡ་ར་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། རྡོ་རྗེ་ཞགས་པའི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ཨ་པ་ཡ་བྷན་དྷ་བི་མོ་ཙ་ནི་ཧཱུཾ་བཾ་ཕཊ། རྡོ་རྗེ་ལྕགས་སྒྲོག་གི་སྔགས་སོ། །ཨོཾ་སརྦ་བིཏ་སརྦ་ཨ་པ་ཡ་ག་ཏི་ག་ཧ་ན་བི་ཤོ་དྷ་ནི་ཧཱུཾ་ཧོཿཕཊ། རྡོ་རྗེ་འབེབས་པའི་སྔགས་སོ། །ཨོཾ་མཻ་ཏྲི་ཧ་
