སུ་ཁ་སཱ་དྷུ་སཱ་དྷུ། སུ་མ་ཧཱ་ཏྟྲཾ། རཱུ་པོ་དྱོ་ཏྱ། ཨར་ཏ་པ་ཏི། ཧ་ཧ་ཧཱུཾ་ཏེ། སརྦ་ཀ་རི། དུཿཁཙྪེད། བུདྡྷ་བོ་དྷི། པྲ་ཏི་སཔྟ། སུ་པ་ཤི་ཏྲཾ། ནི་ར་པཱ་ཡ་ཏྟཾ། ཤ་ཏྲཱུཾ་བྷཀྵ། སརྦ་སིདྡྷི། མ་ཧཱ་ར་ཏེ། རཱུ་པ་ཤོ་བྷེ། སྲོ་ཏ་སུ་ཁྱེ། སརྦ་པཱུ་ཛེ། པྲཧླ་ད་ནི། ཕ་ལ་ག་མེ། སུ་ཏི་ཙི་ག །སུ་གནྡྷ་གྲེ། 
ཨ་ཡ་ཧེ་ཛ། ཨ་ཧེ་ཧཱུཾ་ཧཱུཾ། ཧེ་ཕོ་ཊ་བཾ། གྷཎྚ་ཨ་ཨ། དེ་ནས་ལས་ཀྱི་ཕྱག་རྒྱ་བཅིང་བར་བྱ་སྟེ། གུང་མོ་སྦྱར་བ་བྱས་པ་ལས། །གཡས་པ་ཡི་ནི་གུང་མོ་བསྒྲེང་། །གཡོན་པའི་སྦུ་གུར་བཅུག་ནས་ནི། །གཞན་རྣམས་བརྐྱང་པ་བྱང་ཆུབ་མཆོག །ངན་སོང་ཀུན་སྦྱོང་ས་ལ་རེག །རིན་
ཆེན་གྱི་ནི་མཆོག་སྦྱིན་ཏེ། །ཤཱཀྱ་ཐུབ་པའི་ཆོས་བཤད་མཆོག །མི་འཇིགས་སྦྱིན་པ་མེ་ཏོག་གིའོ། །སྤྱན་གྱི་མིག་དང་འདྲ་བ་སྟེ། །མཱ་མ་ཀཱི་ནི་རྡོ་རྗེ་འདྲ། །དཀར་མོ་པདྨ་ཉིད་དུ་བཤད། །ཨུཏྤལ་ནི་སྒྲོལ་མར་དགོངས། །རྡོ་རྗེ་བླངས་ནས་གསོར་བ་དང་། །ལག་གཉིས་ལྕགས་ཀྱུའི་
