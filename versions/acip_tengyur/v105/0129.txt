གལ་ཏེ་བསྟོད་དམ་སྨད་ཀྱང་རུང། །ཐུབ་པ་དེ་ལ་ཇི་སྐད་བརྗོད། །ཅེས་པ་སྟེ། སྐྱོན་དང་ཡོན་ཏན་བརྗོད་པའི་སྒོ་ནས་བསྟོད་པ་དང་སྨད་པ་ནི་བྱིས་པ་རྣམས་ཀྱིས་རྣམ་པར་རྟོག་པས་བརྟགས་པ་ཡིན་གྱི་སངས་རྒྱས་ཀྱི་རང་བཞིན་ལ་དེ་འདྲ་བ་མེད་ཅེས་པའོ། །མྱ་ངན་མི་
འདའ་འདའ་ལ་མིན། །ཁྱོད་ལ་མྱ་ངན་འདའ་མི་གནས། །ཞེས་པས་ནི་སྙིང་རྗེས་ཞི་བ་ལ་མི་གནས་པ་ཉིད་དུ་སྟོན་ཏེ། སྟོང་པ་ཉིད་ཀྱི་མཐའ་མངོན་དུ་བྱས་ནས་སེམས་ཅན་གྱི་དོན་རྒྱུན་མི་གཅོད་པའོ། །སྤྲུལ་པའི་སྐུ་མྱ་ངན་ལས་འདའ་བའི་ཚུལ་སྟོན་པ་ནི། གདུལ་བྱ་རྣམས་ཀྱིས་རྟག་པར་འཛིན་
པ་སྤང་པའི་དོན་དུ་ཡིན་གྱི་སངས་རྒྱས་ཀྱི་རང་བཞིན་ལ་ནི་དེ་འདྲ་བ་མེད་དོ། །དེ་ལྟར་ཡང་མདོ་འདི་ཉིད་ལས། །མཚན་ཉིད་དང་ནི་བྱ་བ་དང། །དངོས་པོར་མྱ་ངན་ང་མི་འདའ། །ཞེས་པ་ལ་སོགས་པས་སྟོན་ཏོ། །མདོ་སྡེ་གཞན་ལས་ཀྱང། །སངས་རྒྱས་མྱ་ངན་ཡོད་མི་འདའ། །ཞེས་པ་
ལ་སོགས་པ་གསུངས་སོ། །འདའ་ལ་མིན་ཞེས་པ་མྱ་ངན་ལས་འདས་པའི་མཐའ་མངོན་དུ་མི་བྱེད་པའོ། །ཁྱོད་ལ་མྱ་ངན་འདའ་མི་གནས། །ཞེས་པས་དེ་ཉིད་གསལ་བར་བསྟན་པའོ། །བློ་དང་རིག་པ་བགྱི་དང་བྲལ། །ཡོད་དང་མེད་པའི་ཕྱོགས་ཀྱང་སྤངས། །ཞེས་པས་ཤེས་བས་སྲིད་
པ་ལ་མི་གནས་པའམ། ཡང་དེའི་རང་བཞིན་ཤེས་པའི་ཡུལ་ལས་འདས་ཤིང། མཐའ་ཐམས་ཅད་དང་བྲལ་བ་ཉིད་དུ་བསྟན་ཏེ། རྒྱས་བར་ནི་འོག་ནས་སྟོན་པར་འགྱུར་རོ། །དེ་ལྟར་རྟོགས་པའི་ཕན་ཡོན་བསྟན་པ་ནི་ཕལ་ཆེར་ཤེས་པར་ཟད་དོ། །ཆགས་མེད་ཉེར་ལེན་མ་མཆིས་པ། །
ཞེས་པ་ནི་སྲེད་པ་དང་མ་རིག་པ་ལ་སོགས་པ་མེད་པས་ཕུང་པོ་ཉིད་མཚམས་འབྲེལ་པར་བྱེད་པ་མེད་ཅེས་པའོ། །དྲི་བའི་ཚིག་མདོར་བསྟན་པ་ནི་ཤེས་པར་ཟད་དོ། །བཤད་པར་སྐབས་དབྱེ་བ་ལ། ཐམས་ཅད་འཁོར་ལ་རྣམ་གཟིགས་ནས། །ཞེས་པ་ནི་འཁོར་རྣམས་འདུལ་བའི་དུས་ལ་
བབ་པ་ཉིད་དུ་རྟོགས་པའོ། །གཞན་ནི་ཤེས་པར་ཟད་དོ། །དྲི་བའི་ཚིག་རྒྱས་པར་བསྟན་པ་ལ་དང་པོར་རྟོག་གེའི་ཚུལ་འདྲི་བ་ནི། ཇི་ལྟར་ཞེས་པ་ལ་སོགས་པ་སྟེ། ཇི་སྐད་དུ་ཕྱིའི་དོན་ནི་ཀུན་རྫོབ་ཙམ་དུ་ཡང་མེད་ཅིང་སྣང་བ་སེམས་ཀྱི་རང་བཞིན་ཡིན་ལ། དོན་དམ་པར་སེམས་ཀྱང་མེད་ན་དེ་བཞིན་
