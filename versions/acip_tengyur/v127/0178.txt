རྒྱལ་པོའི་འོ། །བསྐོས་པ་ཞེས་བྱ་བ་ནི་དེའི་སྣ་ལ་གཏོགས་པའོ། །འཁོར་ཞེས་བྱ་བ་ནི་ཚོང་བའི་སྡེ་ཚན་ཐ་དད་པའོ། །འདུས་པ་ཞེས་བྱ་བ་ནི་དེ་དག་དང་ཕྱོགས་བཞིའི་སྐྱེ་བོའི་ཚོགས་སོ། །ཤེས་པ་ཞེས་བྱ་བ་དེ་ནི་ཐ་སྙད་རྣམ་པ་གསུམ་གྱིས་མྱོང་བ་གང་ཡིན་པའོ། །མཐོང་
བ་ཞེས་བྱ་བ་ནི་མཐོང་བ་ཁོ་ནས་མྱོང་བའོ། །བདག་གི་ཕྱིར་ཞེས་བྱ་བ་ནི་འཇིགས་པའི་ཕྱིར་རམ། ཟང་ཟིང་ཅུང་ཟད་ཙམ་གྱི་ཕྱིར་རོ། །བདག་གི་ཕྱིར་ཇི་ལྟ་བ་བཞིན་དུ་གཞན་གྱི་ཕྱིར་རམ། གཉི་གའི་ཕྱིར་ཡང་དེ་དང་འདྲའོ། །འཇིགས་པའི་ཕྱིར་ཞེས་བྱ་བ་ནི་གསོད་པ་དང། འཆིང་བ་
དང། ཆད་པ་དང། ཉམས་པ་དང། སྨད་པ་ལ་སོགས་པས་འཇིགས་པའི་ཕྱིར་རོ། །ཟང་ཟིང་ཅུང་ཟད་ཙམ་གྱི་ཕྱིར་ཞེས་བྱ་བ་ནི་ནོར་དང། འབྲུ་དང་དབྱིག་ལ་སོགས་པའི་ཕྱིར་རོ། །ཤེས་བཞིན་དུ་བརྫུན་གྱི་ཚིག་སྨྲ་བ་ཞེས་བྱ་བ་ནི་འདུ་ཤེས་དང། འདོད་པ་དང། ལྟ་བ་བསྒྱུར་ནས་སྨྲ་
བའི་ཕྱིར་ཏེ། དོན་མདོར་བསྡུ་ན་གནས་པ་དང། གཞན་དུ་སྨྲ་བ་དང། གཞི་དང། འདུ་ཤེས་བསྒྱུར་བ་ལ་བརྫུན་དུ་སྨྲ་བ་ཡིན་པར་རིག་པར་བྱའོ། །ཕྲ་མ་ཅན་ཡིན་ཞེས་བྱ་བ་ནི་བསྟན་པའི་ཚིག་གོ། །རྣམ་པར་འབྱེད་པར་བྱེད་པ་ཞེས་བྱ་བ་ནི་དབྱེ་བའི་བསམ་པ་དང་ལྡན་པའི་ཕྱིར་རོ། །འདི་
དག་ལས་ཐོས་ནས་དེ་དག་ལ་ཟློས་པ་དང། དེ་དག་ལས་ཐོས་ནས་འདི་དག་ལ་ཟློས་པ་ཞེས་བྱ་བ་ནི། དབྱེ་བ་དང་མཐུན་པའི་ཚིག་ཇི་སྐད་ཐོས་པའོ། །མཐུན་པ་རྣམས་འབྱེད་པར་བྱེད་ཅེས་བྱ་བ་ན་མི་དགའ་བ་བསྐྱེད་པས་སོ། །བྱེ་བ་རྣམས་མི་འདུམ་པར་བྱེད་ཅེས་བྱ་བ་ནི། དགའ་བར་
འགྱུར་བ་གཅོད་པས་སོ། །མི་འདུམ་པར་དགའ་བ་ཡིན་ཞེས་བྱ་བ་ནི་ཉོན་མོངས་པ་ཅན་གྱི་སེམས་ཀྱིས་མི་དགའ་བ་བསྐྱེད་པའོ། །མི་འདུམ་པར་མོས་པ་ཞེས་བྱ་བ་ནི་ཉོན་མོངས་པ་ཅན་གྱི་སེམས་ཀྱིས་དགའ་བར་འགྱུར་བ་དང་མི་མཐུན་པར་བྱེད་པའོ། །མི་མཐུན་པར་བྱ་བའི་ཚིག་སྨྲ་བ་
ཞེས་བྱ་བ་ནི། མ་ཐོས་པའམ། གཞན་ལ་འཚོལ་བ་སྟེ། དོན་མདོར་བསྡུ་ན། དབྱེ་བའི་བསམ་པ་དང། མ་བྱེ་བ་དབྱེ་བའི་སྦྱོར་བ་དང། བྱེ་བ་མི་མཐུན་པར་བྱ་བའི་སྦྱོར་བ་དང། བྱེ་བ་ལ་ཉོན་མོངས་པ་ཅན་གྱི་སེམས་ཡོད་པ་དང། གཞན་ལ་འཆོལ་བ་ཡང་ཡོངས་སུ་བ་སྟན་པ་
