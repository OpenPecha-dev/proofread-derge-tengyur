། །རྒྱ་གར་སྐད་དུ། མ་ཧཱ་ཡཀྵ་སེ་ནཱ་པ་ཏ་ཡེ་ནཱི་ལཱམྦ་ར་དྷ་ར་བཛྲ་པཱ་ཎི་མ་ཧཱ་ག་ན་བི་དཧིཿ། བོད་སྐད་དུ། གནོད་སྦྱིན་གྱི་སྡེ་དཔོན་ཆེན་པོ་ལག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་གྱི་སྦྱོར་ཀ་ཆེན་པོའི་ཆོ་ག་ 
བཅོམ་ལྡན་འདས་ཕྱག་ན་རྡོ་རྗེ་གོས་སྔོན་པོ་ཅན་ལ་ཕྱག་འཚལ་ལོ། །དུས་གསུམ་བདེ་གཤེགས་ཐམས་ཅད་ཀྱིས། །རིག་སྔགས་གསང་སྔགས་བདག་པོ་རུ། །དབང་བསྐུར་འབྱུང་པོ་ཚར་གཅོད་པ། །གསང་སྔགས་
སྡུད་མཛད་ཁྱོད་ཕྱག་འཚལ། །བཅོམ་ལྡན་ཕྱག་ན་རྡོ་རྗེ་ཡི། །རྒྱུད་རྣམས་མ་ལུས་ཀུན་དང་ནི། །རྣལ་འབྱོར་ཕ་རྒྱུད་མ་རྒྱུད་ཀྱི། །རྣལ་འབྱོར་ཆེན་པོའི་རྒྱུད་དང་ནི། །རྣལ་འབྱོར་རྒྱུད་དང་སྤྱོད་པའི་རྒྱུད། །རྒྱུད་རྣམས་
ཀུན་དང་མི་འགལ་ཕྱིར། །མན་ངག་ཅུང་ཟད་དྲན་པ་འདི། །བློ་དམན་བསྒྲུབ་པའི་དོན་དུ་ནི། །རྣལ་འབྱོར་ཆ་རྐྱེན་བསྟན་པར་བྱ། །སྦྱོར་བའི་རིམ་པ་དཔག་མེད་ཀྱང། །སྒྲུབ་པ་པོ་ཡི་དོན་སླད་དུ། །དགོས་
པ་བསྡུས་པ་བསྟན་པ་འདི། །སྒྲུབ་པོ་སློབ་དཔོན་དང་བྲལ་ན། །སྦྱོར་བ་འདི་རྣམས་བསྟན་པར་གསུངས། །དེ་ལ་སྦྱོར་བ་རིམ་བཞིན་བསྟན། །དང་པོ་ཚོགས་གཉིས་རྫོགས་བྱའི་ཕྱིར། །བདག་ཉིད་སྐད་
