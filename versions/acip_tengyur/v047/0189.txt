བག་ཆགས་རྣམས། །བསྲེགས་ནས་སྟོང་པ་ཉིད་དུ་བསྒོམ། །དེ་ནི་ཡེ་ཤེས་ཚོགས་ཡིན་ནོ། །སེམས་ནི་ཧཱུཾ་གི་རྣམ་པར་བསམ། །ཧཱུཾ་ལས་རྡོ་རྗེ་རྩེ་ལྔར་གྱུར། །རྡོ་རྗེ་ལས་ནི་སྐུ་བསྐྱེད་དོ། །ཞལ་གསུམ་ཕྱག་དྲུག་སྐུ་མདོག་གནག་གཡས་དཀར་གཡོན་དམར་དབུས་མཐིང་
ཞལ༑ །སྤྱན་གསུམ་མཆེ་བ་གཙིགས་པའི་ཞལ། །དབུ་སྐྲ་ཁམ་ནག་གྱེན་དུ་བརྫེས། །གྲི་གུག་རྡོ་རྗེ་རལ་གྲི་གསུམ། །གཡས་པའི་ཕྱག་ན་བསྣམས་པ་ཡིན། །བྷནྡྷ་པདྨ་འཁོར་ལོ་གསུམ། །གཡོན་པའི་ཕྱག་ན་བསྣམས་པ་ཡིན། །གཡས་བསྐུམ་གཡོན་བརྐྱང་གསུས་པོ་
