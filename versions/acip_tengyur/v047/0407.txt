ཨོཾ་ཀྲོ་དྷ་ཨགྣ་ཡེ་མ་མ་སརྦ་ཤ་ཏྲཱུཾ་མ་ར་ཡ་ཧཱུཾ་ཕཊ། ཅེས་བྱ་བའི་སྔགས་ཀྱིས་བརྒྱ་རྩ་བརྒྱད་དུ་སྦྱིན་སྲེག་བྱའོ། །དེ་ནས་ཡུངས་ཀར་ཐབ་ཏུ་གཏོར་ལ་གསང་སྔགས་དེ་ཉིད་ཀྱིས་ཕྲིན་ལས་བཅོལ་ལོ། །དེ་ནས་རྡོ་རྗེ་དྲིལ་བུ་བླངས་ཏེ། ཚིགས་སུ་བཅད་པ་འདི་བརྗོད་དོ། །
སྲིད་པ་རང་བཞིན་གྱིས་དག་ཕྱིར། །ངོ་བོ་ཉིད་ཀྱིས་དག་པར་བྱེད། །རང་བཞིན་དག་པའི་སེམས་དཔའ་རྣམས། །དག་པའི་སྲིད་པ་བྱེད་པའོ། །ཞེས་བརྗོད་ནས་མེ་ཐབ་ལན་གསུམ་བསྐོར་བར་བྱའོ། །མེ་ལྷ་མཆོད་པའི་རིམ་པའོ།། །།དེ་ནས་མེ་ལྷ་དེ་མེ་འབར་བའི་རང་བཞིན་དུ་
གྱུར་པར་བསམས་ཏེ། དེའི་དཀྱིལ་དུ་ཡི་གེ་ཧཱུཾ་ནག་པོ་ལས་དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་རབ་ཏུ་འཁྲུགས་པ་ཞལ་རབ་ཏུ་གདངས། གྲུ་གསུམ་དུ་འདུག་པ། སྐུ་དང་ཞལ་གསུམ་དུ་མེ་སྨུག་ནག་འབར་བ་ཅིག་ཆུ་བསམ་མོ། །དེ་ནས་བདག་གི་སྙིང་ནས་འོད་ཟེར་ལྕགས་ཀྱུ་འདྲ་བ་འཕྲོས་
པས༑ ཡེ་ཤེས་ཀྱི་དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་སྤྱན་དྲངས་ཏེ། ལྕགས་ཀྱུའི་ཕྱག་རྒྱ་བཅངས་ལ། ཨོཾ་ཧྲཱི་བཧོ་བཧོ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་གྩྪ་ཨཱ་གྩྪ་པཱུ་ཛ་པྲ་ཏི་གྲྀཧྞཱ་པྲ་ས་དྷ་མེ་དྷཱི་མཾ་ཀུ་རུ་སཱ་ཧཱ། ཞེས་བརྗོད་པས་ཐབ་ཀྱི་དཀྱིལ་དུ་གཉིས་སུ་མེད་པར་བསམ་པར་བྱའོ། །དེ་ནས་བདུད་རྩི་ལྔ་དང་ལྡན་
པའི་ཞལ་བསིལ་དབུལ་ཏེ། ཨོཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་ཅེས་སྐུ་ཁྲུས་གསོལ་ཏེ། ཨོཾ་ཧྲཱི་ཥྚྲྀ་ཨ་ཞེས་དམར་གྱི་ཨརྒྷ་ཕུལ་ནས། དེ་ནས་མཆོད་པ་རྣམ་ལྔ་དབུལ་ཏེ། ཨོཾ་ཀུ་མཱ་རཱ་རུ་པི་ཎི་ཛཿ་ཛཿ་ཧཱུཾ་ཧ། མེ་ཏོག་ནག་པོའིོ། །གཏོར་རོ། ཨོཾ་ཧྲཱི་ཧ་ཧ་ཧེ་ཧེ་ཕཊ་ཕཊ། ཤེ་ཆེན་པོའི་བདུག་པའིོ། །
ཨོཾ་དཱིཔྟ་ལོ་ཙྪ་ནི་ཝི་ཀྲྀ་ཏཱ་ན་ན་མ་ཧཱ་ཨཊྚ་ཊྚ་ཧཱ་ས་ན་དི་ནི་དཱིཔྟ་དཱིཔྟཱ་ལ་སཱ་ཧཱ། ཏིལ་ཆེན་པོའི་མར་མེའོ། །ཝི་ཀྲྀ་ཏཱ་ན་ན་དུཥྚ་སཏ་མ་ཀ་གཿ་གཿ་བདུད་རྩི་ལྔ་དང་སྦྱར་བའི་དྲིའོ། །དེ་ནས་ཤ་ལ་སོགས་པའི་བཤོས་གཙང་དབུལ། ཨོཾ་ཧྲཱི་ཥྚྲྀ་ཡེ་དེ་ནས་མར་ལ་སོགས་པའི་སྲེག་རྫས་དབུལ་བར་
བྱ་སྟེ། ཨོཾ་ཧྲཱི་ཥྚྀ་ཝི་ཀྲྀ་ཏཱ་ན་ན་ཧཱུཾ། ཆེ་གེ་མོ་མཱ་ར་ཡ་ཕཊ། ཅེས་བརྒྱ་རྩ་བརྒྱད་དབུལ་བར་བྱའོ། །ཡོངས་ཀར་ལ་སྔགས་དེ་ཉིད་ཀྱིས་བཟླས་ཏེ་ཕྲིན་ལས་བཅོལ་ལོ། །དེ་ནས་རྡོ་རྗེ་དྲིལ་བུ་བླངས་ཏེ། ཚིགས་སུ་བཅད་པ་འདིས་བསྟོད་པ། །སྲིད་པའི་རང་བཞིན་གྱིས་དག་ཕྱིར། །ངོ་བོ་ཉིད་ཀྱིས་
