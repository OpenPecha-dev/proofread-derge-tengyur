༄༅། །དེ་བཞིན་དུ་རྡོ་རྗེ་སྒེག་མོ། །བདེ་གཤེགས་སྤྱན་རྣམས་ཚིམ་མཛད་མ། །ཞེས་པ་ནས། སྒྲ་དང། དེའི་དང། རོ་དང། རིག་བྱའི་བར་དུ་ཇི་སྐད་གསུངས་པ་ལྟ་བུ་དང། དེ་བཞིན་དུ། རྡོ་རྗེ་གུར་ཁྲོད་གསང་བ་མོ། །བདེ་གཤེགས་བྱང་ཆུབ་སེམས་ཀྱིས་མཉེས། །ཞེས་
བྱ་བ་ལ་སོགས་བའི་ཚིགས་སུ་བཅད་བའི་རྒྱུད་ཆེན་པོ་ལས་བྱུང་བ་རྣམས་ཀྱིས་རྒྱུད་ཆེན་པོའི་མཆོད་བའི་ཆོ་ག་ཡོངས་སུ་བསྟན་ན། ད་ནི་མཆོད་བའི་དེ་ཁོ་ན་ཉིད་བསྟན་པར་བྱ་སྟེ། དེ་ལ། ཇི་སྲིད་ཐམས་ཅད་མཆོད་པ་ཆེ། །ཞེས་བ་ནི་ཐམས་ཅད་སྦོས་ལ་སོགས་བའི་ལྷ་མོར་བྱས་
ཏེ་དབུལ་བའི་ཕྱིར་རོ། །ཇི་སྲིད་མཆོད་པ་བདེ་བ་ཆེ། །ཞེས་པ་ནི། འདོད་བའི་ཡོན་ཏན་གྱི་མཆོད་པ་སྟེ། དབང་པོའི་རྒྱུད་ཚིམ་བར་བྱེད་བའི་ཕྱིར་རོ། །ཇི་སྲིད་མཆོད་པ་བླ་ན་མེད། །ཅེས་བ་ནི། གསང་བའི་མཆོད་པ་སྟེ། གང་གིས་གང་ལ་དེ་ཕུལ་བ་དེ་ལ་དམ་པ་བསྐྱེད་
པའི་ཕྱིར་རོ། །མཚན་མའི་མཆོག་ཏུ་གྱུར་བའི་ཕྱིར། །ཇི་སྲིད་ཡང་དག་མཆོད་པ་ཆེ། །ཞེས་པ་ནི་མཆོད་པ་དེའི་དེ་བཞིན་ཏེ་གང་གིས་མཆོད་པ་ཐམས་ཅད་སྤྲོས་པ་དང་པ་བར་ཤེས་ན་སྒྲིབ་པ་ཐམས་ཅད་ལས་གྲོལ་བར་བྱེད་པའི་ཕྱིང་གོ། །དེ་དག་གི་དོན་ཡང་ཐོག་མ་ཉིད་དུ་
བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་དོ། །བལ་མ་དང་སངས་རྒྱས་ཐམས་ཅད་མདུན་དུ་བསམས་ལ་རྫས་ལས་བྱུང་བའམ་ཡོ་བྱད་ལས་བྱུང་བའི་ལྷ་མོ་ལ་སོགས་བས་མཆོད་དེ་བསྟོད་པ་ལ་སོགས་པ་བྱས་ན། དེ་བཞིན་གཤེགས་བའི་དངོས་གྲུབ་དམ་པ་ཐོབ་པར་འགྱུར་རོ། །གཞན་
ཡང་ཀུན་དགའ་འགྱུར་བ་ཡི། །རྫས་ནི་རྣལ་འབྱོར་པ་ཡི་ས་བཟུང། །རྫོགས་སངས་རྒྱས་ལ་འཕུང་གྱུར་ན། །འབྲས་བུ་དམ་བ་ཐོབ་བར་འགྱུར། །སྒྱུར་མ་ཙམ་གྱི་བློ་གཏད་དེ། །རྟོག་མེད་བདུད་རྩི་ལྔ་ལ་སྤྱད། །གང་གིས་གཞན་ལ་ཕུལ་གྱུར་ན། །རྟོག་བྲལ་འདོད་པ་ཕུན་སུམ་
ཚོགས༑ །མཆོད་བའི་ཚོགས་རབ་བཟུང་ནས་ནི། །གང་གིས་སྤྲོས་དང་བྲལ་བར་བྱ། །སྤྲོས་མེད་དོགས་བ་མེད་བ་ཡིས། །ཕྱག་རྒྱ་ཆེན་པོ་ཐོབ་བར་འགྱུར། །ཆོད་བ་འདི་ནི་ཕྱག་རྒྱ་ཆེན། །རྫོགས་བའི་སངས་རྒྱས་ཀུན་གྱིས་བསྟན། །སངས་རྒྱས་ཀུན་གྱི་དམ་ཚིག་ནི། །
