༄༅། །གཉིས་ཀྱིས་བཤད་དེ། བློ་གྲོས་མི་མཉམ་པ་ཞེས་བྱ་བས་ནི་གཉིས་སོ། །ཇི་ལྟ་ཞེ་ན་དེ་ལ་འཇིག་ཚོགས་ལ་ལྟ་བ་དང། མཐར་འཛིན་པའི་ལྟ་བ་ནི་བློ་གྲོས་མི་མཉམ་པར་བཤད་དེ། དེ་གཉིས་ཀྱིས་བདག་དང་རྟག་པར་ཕྱིན་ཅི་ལོག་པ་དག་རབ་ཏུ་ཕྱེ་ཞིང་
བསྡུས་སོ། །བསམ་པ་མི་མཉམ་པ་ནི་ཚུལ་ཁྲིམས་དང་བརྟུལ་ཞུགས་མཆོག་ཏུ་འཛིན་པས་བསྡུས་པ་ཡིན་ནོ། །འདིས་ནི་བདེ་བ་དང་གཙང་བར་ཕྱིན་ཅི་ལོག་པ་གཞན་གཉིས་རབ་ཏུ་ཕྱེ་བ་སྟེ། བསམ་པ་མི་མཉམ་པ་ཞེས་བྱ་བ་འདིས་སྟོན་ཏོ། །དེ་དག་དང་མཐུན་པའི་གཉེན་པོར་
དམ་པའི་ཆོས་ནི་མ་ནོར་བའི་ལམ་ཞེས་བྱ་བ་སྟོན་ཏོ། །གཞན་དག་ན་རེ་རྟག་པར་འདུ་ཤེས་པ་དང། བདེ་བར་འདུ་ཤེས་པ་གཉིས་ནི་ངེས་པར་སེམས་པ་ལས་བྱུང་བའི་ཕྱིར་བློ་གྲོས་མི་མཉམ་པས་རབ་ཏུ་ཕྱེ་བའོ། །བདག་ཏུ་འདུ་ཤེས་པ་དང། གཙང་བར་འདུ་ཤེས་པ་ཉིད་ནི་ངེས་པར་
སེམས་པ་མེད་པར་ཡང་འབྱུང་བའི་ཕྱིར་བསམ་པ་མི་མཉམ་པས་རབ་ཏུ་ཕྱེ་བའོ་ཞེས་ཟེར་རོ། །ཚུལ་བ་ཞིན་ཡིད་ལ་བྱེད་པ་ནི་ཡང་དག་པའི་ལྟ་བའི་ལམ་ཞེས་བྱ་བ་འདིས་སྟོན་ཏོ། །ཡང་དག་པའི་ལྟ་བ་འཇིག་རྟེན་ལས་འདས་པ་ཇི་ལྟ་བ་དེ་བཞིན་ཉིད་ལ་ཞེས་བྱ་བ་འདིས་སྟོན་ཏོ། །ད་ནི་
སེམས་ཅན་རྣམ་པ་དགུ་པོ་དེ་དག་ཚིག་དགུ་དང་ཐུན་མོང་མ་ཡིན་པའི་ཡོངས་སུ་སྤྱོད་པ་ལས་བརྩམས་ཏེ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྦྱོར་བར་བྱེད་དེ། དེ་ལ་མི་འཕྲོད་པ་སྨོས་པ་ནི་མི་མཐུན་པ་ཁྱད་པར་མེད་པར་བསྟན་པ་དེ་སེམས་དང་དག་གི་བྱེ་བྲག་གིས་མི་མཐུན་པར་འཛིན་པ་དང་རྩོད་པ་
དག་གིས་བསྟན་ཏོ། །རྩོད་པ་ནི་མི་མཐུན་པར་སྨྲ་བ་སྟེ་འཐོབ་པའོ། །གནོད་པར་བྱ་བ་མངོན་པར་འདུ་བྱེད་པས་ཞེས་བྱ་བ་ནི་ཁྲོ་བ་དང་ཁོན་དུ་འཛིན་པ་དག་གིས་སོ། །དེ་དག་དང་མཐུན་པའི་གཉེན་པོ་ནི་བྱམས་པ་ཉེ་བར་སྒྲུབ་པའོ། །ཆོག་མི་ཤེས་པས་ཡོངས་སུ་སྤྱོད་པ་ལས་བརྩམས་ཏེ་ཞེས་
བྱ་བ་ལ་སོགས་པ་ལ་གཞན་གྱི་ནོར་ལ་མངོན་པར་འདོད་པ་ཅན་ཞེས་བྱ་བ་ནི་བརྣབ་སེམས་ཀྱིས་སོ། །ཡ་ང་བའི་འཚོ་བས་རྗེས་སུ་སྤྱོད་པ་ཡིན་ཞེས་བྱ་བ་ནི་ལུས་དང་ངག་གཉིས་ཀྱིས་ཏེ། དེ་ལུས་ཀྱིས་ནི་རྐུ་བ་དང་གཞལ་ལུགས་ལ་གཡོ་བྱེད་པ་དག་གིས་སོ། །ངག་གིས་ནི་བརྫུན་
