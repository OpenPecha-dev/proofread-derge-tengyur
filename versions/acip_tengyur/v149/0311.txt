ཅང་ཤེས་ནི་སྐྱེས་བུ་དམ་པའོ། །ཅིའི་ཕྱིར་དཀོན་པ་ཡིན་ཞེ་ན། དེ་དག་ཀུན་དུ་འབྱུང་མ་ཡིན། །ཞེས་བྱ་བ་སྨོས་ཏེ། གང་གི་ཕྱིར་དེ་ནི་རྟ་དང་གླང་པོ་ཅང་ཤེས་ལ་སོགས་པ་དང་ཡིད་བཞིན་གྱི་ནོར་བུ་ལ་སོགས་པ་བཞིན་དུ་ཀུན་དུ་སྐྱེ་བ་མ་
ཡིན་ནོ། །གང་དུ་བརྟན་པ་དེ་སྐྱེས་པ། །ཞེས་བྱ་བ་ནི་རིགས་གང་དུ་ཡང་བརྟན་པ་དེ་སྐྱེས་པ་སྟེ། བློ་ལ་སོགས་པའི་ཡོན་ཏན་རིན་པོ་ཆེ་དང་ལྡན་པར་བསྟན་ཏོ། །དེ་ཡི་རིགས་ཀྱིས་བདེ་བ་འཐོབ། །ཅེས་བྱ་བ་ནི་དེ་ལྟར་ཐམས་ཅད་དུ་སྔ་མ་དང་ཕྱི་མ་དག་ཏུ་ཡང་མཁས་པས་ནི་
རིགས་འཕེལ་བ་ཉིད་དེ། ཉིན་རེ་ཞིང་རྒྱས་པར་འགྱུར་བ་དང་བདེ་བ་སྐྱོན་མེད་པ་འཐོབ་པའོ། །བརྟན་པས་ནི་བདག་རིགས་བརྒྱན་པར་བྱེད་པ་གཅིག་པུར་མ་ཟད་དེ། བདག་ཉིད་ཀྱང་བདེ་བ་ལ་གནས་པ་ཡིན་ནོ་ཞེས་བསྟན་ཏེ། བྲམ་ཟེ་ཞེས་བྱ་བ་ལ་སོགས་པ་
སྨོས་སོ། །ཁྱིམ་བདག་མགོན་མེད་ཟས་སྦྱིན་གྱིས་བཅོམ་ལྡན་འདས་ལ་གང་གིས་བདེ་བར་གནས་པར་འགྱུར་ཞེས་ཞུས་པ་ལས་ཚིགས་སུ་བཅད་པ་གཉིས་གསུངས་སོ། །རྣམ་པ་ཀུན་དུ་ཞེས་བྱ་བ་ནི་རྣམ་པ་ཐམས་ཅད་དུའོ། །བདེ་བར་འདུག་ཅེས་བྱ་བ་ནི་འཚེ་བ་མེད་པར་
གཉིད་ལོག་པའམ་འདུག་པའོ། །བདེ་བ་ཉིད་འཐོབ་ཀྱི་སྡུག་བསྔལ་དང་མི་བདེ་བ་ནི་མ་ཡིན་ནོ་ཞེས་བསྟན་ཏེ། བྲམ་ཟེ་ཞེས་བྱ་བ་ནི་སྡིག་པ་ཕྱི་རོལ་དུ་སྤངས་པའི་ཕྱིར་ཏེ། སྡིག་པ་ལས་ཕྱི་རོལ་དུ་གྱུར་པ་ནི་བྲམ་ཟེའོ་ཞེས་བྱ་བའི་ཚིག་གིས་སོ། །ཕྱི་རོལ་དུ་ཡིད་འབྱུང་བ་
སྤངས་པས་ན་ཡོངས་སུ་བཟུང་བ་ཐམས་ཅད་སྤངས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཡོངས་སུ་བཟུང་བ་རྩ་བ་ནས་སྤངས་པས་ཡོངས་སུ་བཟུང་བ་སྤངས་པ་ཡིན་གྱི་གཞན་དུ་ནི་མ་ཡིན་ནོ་ཞེས་བསྟན་པའི་ཕྱིར། མྱ་ངན་འདས་པ་ཞེས་བྱ་བ་སྨོས་ཏེ། འདོད་པའི་འདོད་ཆགས་ལ་
སོགས་པའི་མེ་ཉེ་བར་ཞི་བ་ཞེས་བྱ་བའི་དོན་ཏོ། །རྟགས་གང་གིས་འདི་ལྟར་ངེས་པར་བྱ་ཞེ་ན། གང་ཞིག་འདོད་པས་མགོས་ཤིང་། །ཞེས་བྱ་བ་སྨོས་ཏེ། གང་གི་ཕྱིར་ཉོན་མོངས་པ་དང་ཡུལ་གྱི་འདོད་པས་ཟིལ་གྱིས་མ་མནན་པའོ། །གང་འདོད་པ་ལས་འདོད་ཆགས་
