ནི་རང་གི་ལུས་ལས་བྱུང་བ་ཉིད་ཀྱི་འོད་ཀྱིས་བྱ་བར་གྲུབ་པའི་ཕྱིར་རོ། །ནམ་མཁའ་ལ་འགྲོ་བ་ཞེས་བྱ་བ་ནི་བར་སྣང་ལ་འགྲོ་བ་དང་ལྡན་པའོ། །དགའ་བ་ཟ་བ་ཞེས་བྱ་བ་ནི་དགའ་བ་ཙམ་ཟ་བའི་ཕྱིར་རོ། །དགའ་བའི་ཟས་དང་ལྡན་པ་ཞེས་བྱ་བ་ནི་དེ་དག་གི་གནས་པའི་ཟས་དགའ་བ་ཉིད་
ཡིན་པའི་ཕྱིར་རོ། །ཚེ་རིང་ཞིང་ཞེས་བྱ་བ་ནི་ཚེ་བསྐལ་པ་དུ་མར་ཐུབ་པའི་ཕྱིར་རོ། །ཡུན་རིང་དུ་གནས་སོ་ཞེས་བྱ་བ་ནི་འཚོ་བའོ། །ཆུ་འབའ་ཞིག་ཅེས་བྱ་བ་ནི་ཆུས་ཁྱབ་པའོ། །རྒྱ་མཚོ་འབའ་ཞིག་ཅེས་བྱ་བ་ནི་རྒྱ་མཚོ་དག་དང་འདྲེས་པའི་ཕྱིར་རོ། །ཕྲི་འམ་ཞེས་བྱ་བ་ནི་སྟེང་རིམ་མོ། །
ཀུན་དུ་ཆགས་པར་བྱེད་ཅེས་བྱ་བ་ནི་ཀུན་ཏུ་སྐྱེ་བ་བྱེད་པའོ། ཀུན་དུ་འཐས་པར་བྱེད་ཅེས་བྱ་བ་ནི་ཀུན་དུ་བརྟས་པར་བྱེད་པའོ། །ཀུན་དུ་ཁྱབ་པར་བྱེད་ཅེས་བྱ་བ་ནི་ཀུན་དུ་འབྱེད་པར་བྱེད་པའོ། །འོ་མ་ཞེས་བྱ་བ་ནི་འོ་རློན་ནོ། །སྦྲང་བུ་ཞེས་བྱ་བ་ནི་སྦྲང་རྩི་བྱེད་པའོ། །སྦྲང་རྩི་ཞེས་བྱ་བ་ནི་སྦྲང་
བུ་ལས་བྱུང་བའོ། །མ་བསྐོལ་བ་ཞེས་བྱ་བ་ནི་ཁོལ་བར་མ་བྱས་པའོ། །དེ་ཡང་བུང་བ་ལ་སོགས་པའི་བྱེ་བྲག་གིས་སྤྲང་རྩི་རྣམ་པ་བཞི་ཡོང་པས་དེའི་ཕྱིར། སྤྲང་བུའི་ཞེས་བྱ་བ་གསུངས་ཏེ། སྦྲང་རྩིའི་རྣམ་པ་དག་ནི་སྦྲང་མ་ལས་བྱུང་བ་དང། རུང་བ་ལས་བྱུང་བ་དང། སྦྲང་བུ་
ལས་བྱུང་བ་དང། སྦྲང་རྒོད་ལས་བྱུང་བའོ་ཞེས་ཟེར་བའི་ཕྱིར་རོ། །འཆགས་པ་ཞེས་བྱ་བ་ནི་འབྱུང་ཞིང་འཕེལ་བའོ། །ཚེ་ཟད་ཅེས་བྱ་བ་ནི་འཚོ་བ་ཟད་པའོ། །ལས་ཟད་ཅེས་བྱ་བ་ནི་འཕེན་པར་བྱེད་པའི་ལས་ཟད་པའོ། །ལས་ཟད་ཅེས་བྱ་བ་ནི་འཕེན་པར་བྱེད་པའི་ལས་ཅད་པའོ། །བསོད་ནམས་ཟད་པས་ཞེས་བྱ་བ་ནི་ཡོངས་སུ་རྫོགས་
པར་བྱེད་པའི་ལས་ཟད་པས་སོ། །འདི་ལྟ་བུ་ཉིད་དུ་ཞེས་བྱ་བ་ནི་རྣམ་པ་འདི་ལྟ་བུ་ཉིད་དུའོ། །དེ་གང་ཞེ་ན་མི་རྣམས་དང་ཞེས་བྱ་བ་ལ་སོགས་པས་སྟོན་པར་བྱེད་པའོ། །སྐལ་བ་མཐུན་པ་ཉིད་དུ་ཞེས་བྱ་བ་ནི་རང་བཞིན་མཐུན་པ་ཉིད་དུ་སྟེ། ངོ་བོ་ཉིད་འདྲ་བ་ཉིད་དུའོ། །དེའི་ཚེ་ན་
