༄༅། །ཆེན་པོས་རྒྱུད་ཉེ་བར་བསྟན་པར་བྱ་བའི་ཕྱིར་ཇི་ལྟར་ས་དང་པོ་མངོན་སུམ་དུ་བྱེད་པ་དང་ཉེ་བར་གནས་པའི་གནས་སྐབས་དང། རྟོགས་པའི་ཁྱད་པར་ཐོབ་པས་དང་པོ་ལ་ཞུགས་པའོ། །རྒྱུ་གཉིས་པ་མེད་པར་བསྟན་པའི་ཕྱིར། ཉེ་བར་གནས་པར་གྱུར་ན་ཞེས་བྱ་བ་ལ་སོགས་
པ་ལ༑ འདི་དག་གིས་ནི་སྦྱིན་པའི་རྒྱུ་མེད་པས་སྦྱིན་པར་མི་བྱའོ་ཞེས་བསྟན་པར་འགྱུར་རོ། །བདུད་ཀྱི་རིས་ཀྱིས་ལྷ་ལ་སོགས་པ་ཐོ་འཚམས་བའི་བསམ་པས་ཡིན་པའི་ཕྱིར་དང། མིའི་ལུས་ཀྱིས་ལྷ་རྣམས་ལ་དགོས་པ་མེད་པའི་ཕྱིར་སློང་བར་གྱུར་ཀྱང་སྦྱིན་པར་མི་བྱེད་དོ། །དེ་དག་ཉིད་ལ་གནོད་པ་དང་
ཞེས་བྱ་བ་ནི་བསོད་ནམས་མ་ཡིན་པ་མང་པོ་སོགས་པར་འགྱུར་བའི་ཕྱིར་རོ། །ཕུང་བ་ཆེན་པོར་ཞེས་བྱ་བ་ནི་ཕ་རོལ་དུ་མི་འདོད་པའི་འབྲས་བུ་འཐོབ་པར་འགྱུར་རོ། །དེས་བྱིན་གྱིས་བརླབས་པ་རྣམས་ལ་ཡང་དེ་བཞིན་དུ་སྦྱིན་པར་མི་བྱ་བར་ཤེས་པར་བྱའོ། །སྨྱོན་པ་ནི་ལས་ཀྱི་རྣམ་པར་སྨིན་པས་སེམས་
འཇིགས་པ་ལ་སོགས་པའི་དབང་དུ་གྱུར་པས་རྣལ་དུ་མི་གནས་པར་གཡེངས་པ་སེམས་རང་གི་གནས་སྐབས་ལ་མི་གནས་པ་རང་གི་རང་བཞིན་ལ་མི་གནས་པའི་སེམས་སོ། །སེམས་འཁྲུགས་པ་ནི་དབང་དུ་མ་གྱུར་པའོ། །དེ་དག་གིས་ནི་བློ་སྔོན་དུ་སོང་བས་སློང་བ་ཡང་མ་ཡིན་ལ། འདོད་ནས་སློང་
བ་ཡང་མ་ཡིན་པར་བསྟན་པ་ཡིན་ནོ། །བླ་རྡོལ་ཙམ་དུ་ཟད་པས་ཞེས་བྱ་བ་ནི་བླ་རྡོལ་ཙམ་གྱིས་སློང་བ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཕྱི་རོལ་གྱི་དངོས་པོ་ཡང་གཞན་དག་གི་སྲོག་ལ་གནོད་པའམ། ལོངས་སྤྱོད་ལ་གནོད་པའམ་བུད་མེད་ལ་ཉེ་བར་གནོད་པ་སྦྱིན་པར་མི་བྱེད་དོ། །གང་ཡང་གཞན་དག་ལ་བག་མེད་
པ་དང། ཉེས་པར་སྤྱོད་པའི་གནས་སུ་གྱུར་པ་ཡང་སྦྱིན་པར་མི་བྱའོ་ཞེས་བསྟན་པའི་ཕྱིར་སྨྲས་པ། ཡང་ཕྱི་རོལ་གྱི་དངོས་པོ་ལས་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། བདག་ལ་གནོད་པར་བྱ་བའམ་ཞེས་བྱ་བ་ནི་བདག་ཉིད་འཆི་བར་བྱ་བའི་ཕྱིར་སློང་བའོ། །ཕ་རོལ་ལ་གནོད་པར་བྱ་བའི་ཕྱིར་ཞེས་བྱ་བ་ནི་
གཞན་གསད་པ་ལ་སོགས་པའི་དོན་དུ་སློང་བའོ། །ཕན་པར་བྱ་བའི་ཕྱིར་ཞེས་བྱ་བ་ནི་དུག་ཀྱང་སྨན་གྱི་དོན་དུ་ཡིན་ན་སྦྱིན་པར་བྱའོ། །མེ་ཡང་བཙོ་བ་ལ་སོགས་པའི་དགོས་པ་བསྒྲུབ་པའི་ཕྱིར་སྦྱིན་པའོ། །མཚོན་ཆ་ཡང་བསྲུང་བ་དང། མཛེས་པར་བྱ་བའི་ཕྱིར་སྦྱིན་པའོ། །ཆང་ཡང་སྨན་གྱི་དོན་དུ་
