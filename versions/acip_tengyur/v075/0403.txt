༄༅། །འཇམ་དཔལ་གྱི་སྒྲུབ་ཐབས། སྔར་བཤད་པའི་ཆོ་གས་སྟོང་པ་ཉིད་བསྒོམས་པའི་རྗེས་ལ། པདྨའི་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཐིག་ལེ་དཀར་པོ་རྣམ་པར་བསྒོམ་པར་བྱ་སྟེ། སྤྲོ་བར་བྱ་བ་ལ་སོགས་པ་སྔོན་དུ་བཏང་ཞིང་དེ་ཡོངས་སུ་གྱུར་པ་ལས་ཡི་གེ་དྷཱི་དཀར་པོས་བསྐྱེད་
པའི་འཇམ་དཔལ་ཨ་ར་པ་ཙ་ན་དཀར་པོ་གཅིག་པུ་ཉིད་འབའ་ཞིག་གོ། །རང་ཉིད་ཀྱི་ས་བོན་སྙིང་གར་རྣམ་པར་བསམས་ལ། ཨོཾ་དྷཱི་སཱ་ཧཱ། ཞེས་བཟླས་པར་བྱ་བའོ། །རིག་པ་འཛིན་པའི་སྡེ་སྣོད་ནས་འབྱུང་བའི་འཇམ་དཔལ་གྱི་སྒྲུབ་ཐབས་མདོར་བསྟན་པའོ། །ཨ་བྷ་ཡ་ལས་དགེ་སློང་ཚུལ་
ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ།།  །རྒྱ་གར་སྐད་དུ། པྲ་ཛྙཱ་བརྡྷ། བོད་སྐད་དུ། ཤེས་རབ་བསྐྱེད་པ། སྐད་ཅིག་གིས་འཇམ་དཔལ་གྱི་རྣལ་འབྱོར་དུ་དམིགས་ལ། ལྟེ་བར་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཐིག་ལེ་དཀར་པོ་མར་མེ་གསལ་བར་འབར་པ་ལྟ་བུ་འོད་
ཟེར་དང་བཅས་པ་རྣམ་པར་བསམས་ཏེ། མགོ་པོ་ལ་ཉམས་པ་མེད་པའི་ཡི་གེ་ཧཾ་དཀར་པོ་མགོ་བོ་ལས་འཛག་པར་གྱུར་པ་འོད་ཟེར་དང་བཅས་པ་བསྒོམ་པར་བྱ་བའོ། །དེའི་རྗེས་སུ་ཐིག་ལེའི་འོད་ཟེར་གྱིས་ལེགས་པར་བསྐུལ་བས་ཁུ་བར་གྱུར་ནས་ཡི་གེ་ཧཾ་གིས་ཐིག་ལེ་ཀུན་ནས་ཁྱབ་པར་གྱུར་པ་
ལ་སེམས་བརྟན་པའི་སྔགས་པ་ནི་རིང་པོ་མི་ཐོགས་པར་ཤེས་རབ་མངོན་པར་འཕེལ་བར་འགྱུར་རོ། །ཤེས་རབ་མངོན་པར་བསྐྱེད་པ། ཨ་བྷ་ཡ་ལས་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་པའོ།།  །རྒྱ་གར་སྐད་དུ། པྲཛྙཱ་བྲྀདྡྷི། བོད་སྐད་དུ། ཤེས་རབ་འཕེལ་བར་
བྱེད་པ། ལྟེ་ཡུལ་སྟེང་དུ་ལེགས་གསལ་བའི། །ཆོས་འབྱུང་དམར་པོའི་ནང་དུ་ནི། །རྡོ་རྗེ་སྔོན་པོ་རྩེ་ལྔ་པ། །ལྟེ་བར་གཟི་བརྗིད་འོད་མཆོག་ཅན། །ཡི་གེ་ཧཱུཾ་དཀར་སྒྲས་བསྐྱེད་པ། །མཚོག་མའི་མཚམས་སུ་ངེས་འབྱུང་བའི། །ཧཱུཾ་ཡིག་དུད་ཁས་ལེགས་བསྐྱེད་པ། །ནམ་མཁའ་ཁྱབ་
པར་བལྟ་བྱ་བ། །སྒྲ་ཡི་འོད་ཟེར་རྩེ་མོའོ། །ཤེས་རབ་འཕེལ་བའི་ཆོ་ག། ཨ་བྷ་ཡ་ཀ་རའི་ཞལ་སྔ་ནས་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ།།  །རྒྱ་གར་སྐད་དུ། མཉྫུ་ཤྲཱི་ཡ་པྲཛྙཱ་ཅ་ཀྲ། བོད་སྐད་དུ། འཇམ་དཔལ་གྱི་ཤེས་རབ་ཀྱི་འཁོར་ལོ་
