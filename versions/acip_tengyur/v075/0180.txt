ཨོཾ་ཨཱ་ཆེ་གེ་མོ་བཛྲ་ཞེས་རང་གི་འཁོར་ལོའི་དབང་པོ་ལ་སོགས་པ་གནོད་མཛེས་ཀྱི་བར་གྱི་ལྷ་རྣམས་རེ་རེ་ཞིང་བོས་ནས་དེའི་ས་བོན་ཡང་བརྗོད་དེ། སརྦ་དུཥྚཱ་ན་ས་མ་ཡ་མའུ་ཏྲཱ་པྲ་བྷཾ་ཛ་ཀ་མ་མ་ཤཱནྟི་ཀུཾ་རུ་ཙ་ཀུ་རུ་སཱ་ཧཱ་ཧཱུཾ་དང། ཨོཾ་ཨཱ་སརྦ་དྲྱ་དཧྭ་ཛ་ད་ཤ་དི་ཀ། ལོ་ཀ་དྷཱ་ཏྭ་ནནྟ་ག་ན། ས་མུ་
དྲ་མེ་གྷ་བྱཱུ་ཧ་པྲ་ས་ར། པ་ར་མཱ་ཎུ་རྫོ་མཎྜལ། པ་རཾ་པ་ར་ཨནྟ་རྒ་ག་ཏ། ས་མཱ་བ་ཏྟི་ཨ་བ་སྠི་ཏཱ། དྷརྨ་དྷཱ་ཏུ་ས་མ་བ་ས་རནྟ། ཨཱ་ཀཱ་ཤཱ་དྷཱ་ཏུ་པརྱ་བ་སཱ་ནཱ། སརྦ་ཏྲྱ་དཧྭ་ཛ་ད་ཤ་དི་ཀ། ལོ་ཀཱ་དྷཱ་ཏྭ་ནནྟ། མེ་གྷ་བྱཱུ་ཧ་པྲ་ས་ར་ག་གན། ས་མཱ་སརྦ་ལོ་ཀ་པཱ་ལཱ་སརྦ་ས་ཏཱ་ཤྩ། ཏདྱ་ཐཱ། བཛྲཱ་ཡུ་དྷ་མཱ་ཡཱ་བཛྲ་བཛྲཱ་
ན་ལ༑ བཛྲ་ཀཱ་ལ་བཛྲ་མུ་ཥ་ལ་ནཱ་ག་བཛྲ། བཛྲ་ཨ་ནཱ་ལི། བཛྲ་བཧཻ་ར་བ། བཛྲ་ཤཽཎྜ། བཛྲ་ཀྲོདྷ། བཛྲ་ཀུཎྜ་ལི། བཛྲ་པྲ་བྷ། མཽ་ན་བཛྲ། བེ་མ་ཙི་ཏྲི། པྲྀ་ཐི་བའི་དེ་བ་ཏྲཿ་ས་པ་རི་ཝཱ་རཿ་ཨི་མཾ་པུཥྤ། དྷཱུ་པ། དའི་པ། གནྡྷེ་ནཻ་བིཏྱ་ཨ་དའི་སཾ་ཡུཀྟཾ། བ་ལི་ཨུ་བ་ཧ་རཾ། པྲ་ཏཱིཙྪ་ཨུ་པ་བཧུ་ཛྙ། མ་མ་ཧི་ར་ཎྱ་སུ་
ཝརྞྞ་དྷ་ན་དྷཱ་ཎྱ། ཨཱ་ཡུཿ་ཡཽ་བ་ན་ཨ་རོ་གྱ། ས་ཏུ་ཁ་ཨ་པ་ཧ་ར་ཀཱ་སརྦ་བིགྷྣཱཾ་བི་ནཱ་ཡ་ཀཱ་ན། སརྦ་དུཥྚ་པྲ་དུཥྚ་ན། མ་ནུ་ཥྱ་ཎ་ཨ་མ་ནུ་ཥྱ་ཎ་ཛཾ་བྷ་ཡ་ཏ། སྟཾ་བྷ་ཡ་ཏ། བནྡྷ་ཡ་ཏ། བི་དཧྭཾ་ས་ཡ་ཏ། མ་མ་ཧི་ར་ཎྱ་སུ་པ་རྞ་དྷ་ན་དྷཱ་ཎྱ་ཨཱ་ཡུ་ར་ཡཽ་བ་ན། ཨ་རོ་གྱ་ས་ཧཱུཾ་ཁཱ་ནི་མ་ཧཱ་སུ་ཁ་པྲ་པྲྀ་དྷ་ཡེ། ཡཱ་བ་ད་ཨཱ་
བོ་དྷ་མཎྜལ་པྲྱནྟཾ། དཧོ་ཀ་ཡ་ད་ས་ཏ་ས་ཧཱ་ཡ་ཏཾ། ཤཱནྟིཾ་རཀྵ་ཉྩ་ཀུ་རུ་ཏ་ཧཱུཾ་ཞེས་པའི་སྔགས་དག་གིས་དང། ཀླུ་གཏོར་གྱི་སྔགས་ཀྱིས་ཀྱང་དབུལ་ལོ། །དེ་རྣམས་ཀྱང་དཀྱིལ་འཁོར་གྱི་བདག་པོ་ནས་བརྩམས་ཏེ་གནོད་མཛེས་ཀྱི་བར་དང། དེ་ནས་དབང་པོ་ལ་སོགས་པ་འཁོར་དང་བཅས་པ་དང། དེ་
ནས་ཀླུ་རྣམས་དང། དེ་ནས་སེམས་ཅན་ཐམས་ཅད་ལྕེ་ཧཱུཾ་ལས་བྱུང་བའི་རྡོ་རྗེ་རྩེ་གཅིག་པ་དམར་པོ་རྣམས་དེའི་འོད་ཟེར་གྱི་སྦུ་གུ་རྣམས་ཀྱིས་སམ། ཡིད་ཀྱིས་བདུད་རྩི་དྲངས་པ་གསོལ་བས་བདེ་བ་ཆེན་པོའི་རང་བཞིན་སྟོང་པ་ཉིད་དང་རོ་གཅིག་པར་ཇི་སྲིད་འདོད་ཀྱི་བར་བསམ་པར་བྱའོ། །ཞལ་བསིལ་
དང༑ ཕྱག་བཀྲུ་བ་དང། སོ་རྩི་ལ་སོགས་པ་བྱིན་ནས་མཆོད་དེ། བྱ་བ་སྒྲུབ་པ་ལ་ལྷུར་བྱེད་པ་རྣམས་ལ་བཟོད་བར་གསོལ་ཏེ། སྔར་བརྗོད་པའི་ཡི་གེ་བརྒྱབ་བཟླས་ནས། ཨོཾ་ཡོ་ག་ཤུདྡྷ་སརྦ་དྷརྨཱ་ཡོ་ག་ཤུདྡྷོཉྷཾ། ཞེས་བརྗོད་ཅིང་བད་སྐོར་དང་འཁྱུད་པའི་ཕྱག་རྒྱ་བྱེད་པས་རང་གི་ལྷའི་འཁོར་ལོ་
