མཉམ་པ་ཉིད་དོ། །གཏོང་བ་ནི་སྦྱིན་པའི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །ཚོལ་བ་ནི་ཡོངས་སུ་ཚོལ་བའོ། །དེ་ཡང་དམ་པའི་ཆོས་རྣམས་ལ་དམིགས་པ་སྟེ། ཆོས་ཡོངས་སུ་ཚོལ་བ་ཞེས་བྱ་བའི་དོན་ཏོ། །ངེས་པར་འབྱུང་བ་ནི་ཁྱིམ་གྱི་གནས་ནས་ངེས་པར་འབྱུང་བའོ། །ངེས་པར་འབྱུང་བ་དེ་
ལ་སེམས་པ་ཉིད་དེ་འདོད་པ་ཉིད་དོ། །དགའ་བ་ཞེས་བྱ་བ་ལ་ཡང་འདིར་རྟག་ཏུ་ཞེས་བྱ་བ་རྗེས་སུ་སྦྱོར་རོ། །བདེན་པའི་ཚིག་ཅེས་བྱ་བ་ནི་བདེན་པའི་ངག་སྟེ་དེ་དག་ནི་བཅུའོ། །འདི་དག་ཅེས་བྱ་བ་ནི་བསམ་པ་ལ་སོགས་པའོ། །ཚུལ་ཁྲིམས་བྱས་པ་གཟོ་དང་བཟོད། །རབ་ཏུ་དགའ་དང་སྙིང་
བརྩེ་ཆེ། །བཀུར་སྟི་གུས་པར་ཉན་པ་དང། །བརྒྱད་པ་སྦྱིན་ལ་སོགས་ལ་བརྩོན། །ཞེས་བྱ་བ་ལ་ཚུལ་ཁྲིམས་ནི་ཚུལ་ཁྲིམས་ཡོངས་སུ་དག་པའོ། །བཟོད་པ་ནི་བཟོད་པའི་སྟོབས་སོ། །རབ་ཏུ་དགའ་བ་ཉིད་ནི་རབ་ཏུ་དགའ་བའོ། །སེམས་ཅན་ཐམས་ཅད་ཡོངས་སོ་མ་བཏང་བ་ནི་སྙིང་
བརྩེ་བ་ཆེན་པོའོ། །བཀུར་སྟི་ནི་བཀུར་བར་བྱ་བར་འོས་པའོ། །གུས་པར་ཉན་པ་ནི་དགེ་བའི་བཤེས་གཉེན་རྣམས་ལ་བརྟེན་པའི་འདུ་ཤེས་སོ། །སྦྱིན་པ་ལ་སོགས་པ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དག་ལའོ། །བརྒྱད་པ་སྨོས་པ་དེ་ནི་ཡོངས་སུ་སྦྱངས་པ་གཉིས་པ་ཉིད་བརྒྱད་ཡིན་ནོ། །ཐོས་པས་མི་ངོམས་
ཉིད་དང་ནི། །ཟང་ཟིང་མེད་པར་ཆོས་སྦྱིན་དང། །སངས་རྒྱས་ཞིང་ཀུན་སྦྱོང་བ་དང། །འཁོར་བས་ཡོངས་སུ་མི་སྐྱོ་དང། །ངོ་ཚ་ཁྲེལ་ཡོད་ཅེས་བྱ་བ། །རློམ་མེད་བདག་ཉིད་དེ་རྣམ་ལྔ། །ཞེས་བྱ་བ་ལ་ཐོས་པས་ཞེས་བྱ་བ་ནི་མང་དུ་ཐོས་པས་སོ། །ཞིང་ཀུན་སྦྱོང་ཞེས་བྱ་བ་ནི་ཡོངས་སུ་དག་པར་རོ། །
དེ་ལ་དེ་དག་ཉིད་བསོད་ནམས་ཡོངས་སུ་བསྔོ་བས་རློམ་སེམས་མེད་པའི་བདག་ཉིད་དེ་དེས་རློམ་པ་མེད་པས་དེ་རྣམ་ལྔ་ཞེས་བྱ་བའི་ཚིག་གིས་སོ། །ཡོངས་སུ་སྦྱོང་བ་ལྔ་པ་ནི་གསུམ་པ་ཡིན་ནོ། །ནགས་གནས་འདོད་ཆུང་ཆོག་ཤེས་དང། །སྦྱང་པ་ཡང་དག་སྡོམ་བསྟེན་དང། །བསླབ་པ་
