བྱ་བ་ལ་སོགས་པའི་སྒྲ་ནི་དོན་མཐུན་པ་དང་གཞན་ཡང་ཡོན་ཏན་དག་གོ། །དོན་མཐུན་པར་སྤྱོད་པ་ནི་དོན་མཐུན་པ་སྟེ་དགེ་བའི་ཆོས་སོ། །གཞན་ཡང་ཡོན་ཏན་གྱི་ཆོས་རྣམས་ཡོངས་སུ་གཟུང་ངོ། །ཤེས་རབ་ཀྱིས་ཡོངས་སུ་ཟིན་པས་ཞེས་བྱ་བ་ནི་འདིས་ཡོངས་
སུ་བཟུང་བས་ན་ཡོངས་སུ་ཟིན་པ་སྟེ། ཤེས་རབ་ཉིད་ཀྱིས་ཡོངས་སུ་གཟུང་བའི་ཕྱིར་ཤེས་རབ་ཀྱིས་ཡོངས་སུ་ཟིན་པའོ། །ཤེས་རབ་ཀྱིས་བཟུང་བས་ན་སྐྱོན་མེད་པ་སྟེ་མ་སྨད་པའོ་སྙིང་རྗེའི་རྒྱུ་མཐུན་པ་ཞེས་བྱ་བ་ནི་སྙིང་རྗེའི་རྒྱུས་སྙིང་རྗེ་དང་རྗེས་སུ་མཐུན་པ་སྟེ། རྒྱུ་
མཐུན་པ་ནི་རྒྱུ་དང་འདྲ་བའི་ཚིག་གོ། །ཡོངས་སུ་འཛིན་པ་ནི་མགུ་བར་བྱས་པའོ། །རང་གི་ཆོས་ཡིན་པས་ན་རང་གི་ཆོས་དེ་ལས་རྣམ་པ་དྲུག་གོ། །དེ་ཡང་མཆོད་སྦྱིན་དང། མཆོད་སྦྱིན་བྱེད་དུ་འཇུག་པ་དང། འདོན་པ་དང། འདོན་དུ་འཇུག་པ་དང། སྦྱིན་པ་དང། ལེན་པའོ། །ཆོས་དེ་རྣམས་
ལ་མངོན་པར་དགའ་བ་ལས་གྱུར་པའི་རིགས་ཀྱི་གཙང་སྤྲ་སྤྱོད་པ་གང་ཡིན་པ་དེ་ནི་རང་གི་ཆོས་ལ་མངོན་པར་དགའ་བ་ལས་གྱུར་པའི་གཙང་སྤྲའི་སྤྱོད་པ་སྟེ། དེ་ལ་གཙང་སྤྲ་ནི་དག་པའི་སྤྱོད་པ་སྟེ། དག་པར་སྤྱོད་པའི་ཚུལ་ཀུན་ཏུ་སྤྱོད་པའོ། །ཆེན་པོ་ཞེས་བྱ་བ་ནི་རིགས་
གཞན་ལ་ལྟོས་པས་ཆེན་པོའོ། །ཇེ་མཐོ་ཇེ་མཐོར་ཞེས་བྱ་བ་ནི་སྐྱེས་ཤིང་སྐྱེས་པ་ནའོ། །ཡང་ན་མཐོ་བའི་རིགས་དེ་ལས་བྱང་ཆུབ་སེམས་དཔའ་མཐོ་བར་གྱུར་པའི་ཕྱིར་རོ། །སྐྱེ་བ་ཞེས་བྱ་བ་ནི་འབྱུང་བར་འགྱུར་བའོ། །ཡོངས་སུ་བཟུང་བ་ནི་ཐོབ་པར་བྱས་པའོ། །
གྲགས་སོ་ཞེས་བྱ་བ་ནི་མངོན་སུམ་དུ་མ་གྱུར་བའི་ཚིག་གོ། །དེ་ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའོ། །སྐྱེས་པའི་ལས་ནི་རུམ་སྐྱེད་པ་དང། མིང་གདགས་པ་དང། །བཟས་སྦྱིན་པ་དང། གཙུག་ཕུད་བྱ་བ་དང། ཚངས་སྐུད་གདགས་པ་དང། གཙུག་ཕུད་
