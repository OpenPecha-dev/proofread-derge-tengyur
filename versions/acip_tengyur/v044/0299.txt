། །།རྒྱ་གར་སྐད་དུ། བཛྲ་བཧཻ་ར་བ་ཏནྟྲ་པཉྩི་ཀཱ། བོད་སྐད་དུ། རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་རྒྱུད་ཀྱི་དཀའ་འགྲེལ། དཔལ་རྡོ་རྗེ་འཇིགས་བྱེད་ལ་ཕྱག་འཚལ་ལོ། །དཔའ་བོ་རྡོ་རྗེ་འཇིགས་བྱེད་དཔལ་བཏུད་ནས། དཔལ་ལྡན་རྡོ་རྗེ་
འཇིགས་བྱེད་ཀྱི། །རྒྱུད་ཀྱི་དགའ་འགྲེལ་ཅུང་ཟད་ནི། །གཞོན་ནུ་ཟླ་བ་ཡིས་ནི་བྱས། །དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་ནས་ཞེས་པ་ནི་འཇམ་དཔལ་གྱི་རྒྱུད་ཆེན་པོ་ཐམས་ཅད་དང་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་སྒྲུབ་ཐབས་ལས་སོ། །དེ་ནས་ནི་རྗེས་ཐོགས་དང་བཀྲ་ཤིས་ཏེ་བསྟན་བཅོས་ཀྱི་དང་
