༄༅། །པོ་གཞན་ལས་སྐྱེ་བའི་སྡུག་བསྔལ་གྱི་དབང་དུ་བྱས་པའོ། །ད་ནི་རང་གི་ཡུལ་འཁོར་གྱི་དབང་དུ་བྱས་ནས། གལ་ཏེ་ནོར་འདོད་ཅེས་བྱ་བའི་ཚིགས་བཅད་ཡིན་ཏེ། ནོར་མ་ཚང་བའམ། གནོད་པ་ཐོབ་པས་རང་གི་ཡུལ་ལས་པས་མཐར་འགྲོ་ཞེས་བྱ་བ་དེའི་དབང་
དུ་བྱས་ནས། ཅི་ཞིག་འདོད་ཕྱིར་ཞེས་བྱ་བ་ཡིན་ཏེ་རྒྱུ་གཞན་གང་གིས་སོ། །གལ་ཏེ་ནོར་འདོད་ཅེས་བྱ་བ་དེས་ནི་སྡུག་བསྔལ་ལམ་གནོད་པ་སྟེ། སྔར་བསྟན་པའི་སྡུག་བསྔལ་གང་གིས་ཁྱོད་ནི་བས་མཐར་འགྲོ་ཞེས་བྱ་བ་དེ་གསལ་བ་ཡིན་ནོ། །སྐྱེ་བོའི་ཚིག་བསྲུངས་པ་ཙམ་གྱིས་
ཞེས་བྱ་བ་ནི་གང་དེ་འཇིག་རྟེན་གྱི་ཚིག་ནི་འཇིག་རྟེན་གྱི་གཏམ་ལ་ཞེན་པའོ། །དེ་ལས་ཞེས་བྱ་བ་ནི་དགེ་བའི་ཆོས་ལས་ཉམས་པར་མི་བགྱི་བ་སྟེ། ཆོས་ཞེས་བྱ་བ་ནི་བདག་ལ་ཕན་པར་འདོད་པ་སྟེ། དེ་ལྟར་བདག་ཉིད་ཀྱི་ཆོས་ལ་ལྟོས་པར་བསྟན་པའོ། །ངོ་ཚའི་ཕྱིར་ཡང་ཞེས་བྱ་བ་ནི་བལྟས་
པས་ཀྱང་སྟེ། ངོ་ཚའི་ཕྱིར་ཡང་རབ་ཏུ་འབད་པའི་རིགས། །ཞེས་བྱ་བ་ཡིན་ཏེ། ངོ་ཚའི་ཕྱིར་ཡང་ཆོས་དང་ལྡན་པ་ལ་འབད་པར་བྱ་བའོ། །དེའི་ཕན་ཡོན་བསྟན་པའི་ཕྱིར། ཡོན་ཏན་ཡོད་ཅེས་གྲགས་ཤིང་རེ་བ་དག་ཅེས་བྱ་བའི་ཚིགས་བཅད་ཡིན་ཏེ། ཡོན་ཏན་འབྱུང་བ་ཡིན་པས་ཡོན་
ཏན་ཡོད་ཅེས་པའོ། །ཇི་ལྟར་གྱུར་ཀྱང་རུང་སྟེ་ཞེས་བྱ་བ་ནི་ངོ་ཚ་བའི་ཕྱིར་ཡང་གཏམ་སྙན་འཕེལ་བར་འགྱུར་ཏེ། ཅིའི་ཕྱིར་ཞེ་ན། གྲགས་པ་རབ་ཏུ་འགྲུབ་པས་ངོ་ཚའི་ཕྱིར་ཡང་དེ་རབ་ཏུ་འབད་པའི་རིགས་ཏེ། འདི་ལྟར་གྲགས་པས་རབ་ཏུ་འགྱུར་བས་སོ། །གཞན་དུ་བགྱིས་ན་ཞེས་བྱ་བ་ནི་
བདག་ཉིད་ཀྱི་ཕུན་སུམ་ཚོགས་པའི་ཡོན་ཏན་ཐམས་ཅད་མེད་བར་བྱས་པ་སྟེ། རེ་བའི་ཚུལ་གྱིས་ལྟ་བ་ན་ཆུ་མིག་སྐམ་པོའི་གནས་དང་འདྲ་བར་འགྱུར་ཏེ། དེ་ལྟར་ཆུ་མིག་སྐམ་པོ་ལ་སྐོམ་ཞིང་ཆུ་འདོད་པ་རྣམས་ཀྱིས་ཆུ་དོན་དུ་གཉེར་ཞིང་རེ་བ་དང་བཅས་པ་ལས་སྐམས་པར་གྱུར་ན་དེ་དག་གི་རེ་
བ་ཉམས་པར་འགྱུར་ཏེ། དེ་བཞིན་དུ་འདིར་ཡང་ཡོན་ཏན་ཡོད་པར་རེ་བ་དང་བཅས་པ་རྣམས་ཏེ། ཡོན་ཏན་ལ་སྐོམ་པའི་སྐྱེ་བོ་དགའ་བ་རྣམས་ཀྱིས་རེ་བ་དང་བཅས་པ་ལས་མ་མཐོང་ན་ནི་འཇིག་པ་ཡིན་ཏེ། ཡོན་ཏན་གཉེར་བ་རྣམས་ཀྱིས་ཁྲོན་པ་སྐམ་པོ་བཞིན་དུ་མཐོང་བར་
