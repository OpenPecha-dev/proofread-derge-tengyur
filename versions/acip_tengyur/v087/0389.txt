༄༅། །མེད་པའི་སེམས་ཅན་འབྱུང་བར་ཡང་འགྱུར་ན། དེ་ནི་དེ་ལྟར་ཡང་མ་ཡིན་ཏེ། དེའི་ཕྱིར་བདག་བྱས་པ་ཡང་མ་ཡིན་ནོ། །གཞན་ཡང། འབྱུང་བའང་རྒྱུ་མེད་ཅན་དུ་འགྱུར། །བདག་མ་བྱུང་བ་ལས་བྱུང་ན་རྒྱུ་མེད་ཅན་ཁོ་ནར་འགྱུར་ཏེ། སྔར་བདག་མེད་
པས་བྱེད་པ་པོ་མེད་པ་ནི་རྒྱུ་མེད་པ་ཅན་དུ་འགྱུར་རོ། །འང་གི་སྒྲ་ནི་རྣམ་པར་རྟོག་པ་སྟེ། བདག་ནི་བྱས་པར་འགྱུར་བའམ། ཡང་ན་འདས་པའི་དུས་ན་མ་བྱུང་བ་ཞིག་གམ་ཞེས་བྱ་བ་འདི་ནི་ཁས་མི་བླང་ངོ་ཞེས་བྱ་བ་དང། བདག་འབྱུང་བ་རྒྱུ་མེད་པ་ཅན་དུ་འགྱུར་བའམ། ཡང་ན་
འདས་པའི་དུས་ན་མ་བྱུང་བ་ཞིག་གམ་ཞེས་བྱ་བ་དེ་མི་འཐད་དོ་ཞེས་བྱ་བ་འདི་ཁས་བླང་བར་བྱའོ་ཞེས་བྱ་བའོ། །ད་ནི་ཇི་སྐད་བསྟན་པའི་དོན་དེ་ཉིད་མཇུག་བསྡུ་བའི་ཕྱིར། དེ་ལྟར་བདག་བྱུང་བདག་མ་བྱུང། །གཉི་ག་གཉི་ག་མ་ཡིན་པར། །འདས་ལ་ལྟ་བ་གང་ཡིན་པ། །དེ་ནི་འཐད་པ་མ་
ཡིན་ནོ། །ཞེས་གསུངས་ཏེ། དེ་ལྟར་ཇི་སྐད་བསྟན་པའི་ཚུལ་དུ་འདས་པའི་དུས་ན་བྱུང་བ་ཞིག་གམ་ཞེས་བྱ་བར་ལྟ་བ་གང་ཡིན་པ་དང། འདས་པའི་དུས་ན་མ་བྱུང་བ་ཞིག་གམ་ཞེས་བྱ་བ་ལྟ་བ་གང་ཡིན་པ་དེ་འཐད་པ་མ་ཡིན་པ་ཉིད་དོ། །དེ་གཉིས་མེད་པའི་ཕྱིར་ན་གཉི་ག་ཡང་མི་
འཐད་དོ། །ཅིའི་ཕྱིར་ཞེ་ན། འདི་ལྟར། དེ་གཉིས་འདུས་པ་ལ་གཉི་ག་ཞེས་བྱ་བར་བརྟག་པར་བྱ་ན། རེ་རེ་སོ་སོ་སོ་སོར་མེད་པས་དེ་དག་འདུས་པ་ག་ལ་ཡོད་དེ། དེའི་ཕྱིར་གཉི་ག་ཡང་ཡོད་པ་མ་ཡིན་ནོ། །གཉི་ག་མེད་པའི་ཕྱིར་ན་དེ་དག་བཀག་པའི་སྒོ་ནས་གཉིས་ཀ་མ་ཡིན་པ་
ཡང་ཡོད་པར་ག་ལ་འགྱུར། དེའི་ཕྱིར་བྱུང་བ་ཡང་མ་ཡིན་མ་བྱུང་བ་ཡང་མ་ཡིན་པ་ཞིག་གམ་ཞེས་བྱ་བ་དེ་ཡང་འཐད་པ་མ་ཡིན་ནོ། །དེའི་ཕྱིར་དེ་ལྟར་སྔོན་གྱི་མཐའ་ལ་བརྟེན་པའི་ལྟ་བ་བཞི་མི་སྲིད་པར་བསྟན་ནས། ད་ནི་ཕྱི་མའི་མཐའ་ལ་བརྟེན་པ་དགག་པ་གསུངས་པ། མ་
འོངས་དུས་གཞན་འབྱུང་འགྱུར་དང། །འབྱུང་བར་མི་འགྱུར་ཞེས་བྱ་བར། །ལྟ་བ་གང་ཡིན་དེ་དག་ནི། །འདས་པའི་དུས་དང་མཚུངས་པ་ཡིན། །ཇི་ལྟར་འདས་པའི་དུས་ལ་ལྟ་བ་བཞི་བཀག་པ་དེ་ལྟར་མ་འོངས་པའི་དུས་ལ་བཀག་པ་བཞི་ཡང་བཤད་ཟིན་པའི་འདོན་
