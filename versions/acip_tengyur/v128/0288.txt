།རྒྱན་ཞེས་བྱ་བ་ནི་མགོ་རྒྱན་དང་དཔུང་རྒྱན་དང་རྒྱན་སྣ་ཚོགས་ལ་བྱའོ། །ལྟུང་བ་བརྒྱ་བདུན་ཅུ་གཅིག་པ་རྫོགས་སོ།། །།སྟོན་པ་མཉན་དུ་ཡོད་པ་ན་སྟེ། བཅུ་གཉིས་སྡེ་དག་ཁྱིམ་མི་མོ་ལྷན་ཅིག་ཆུ་བོ་ཨ་ཙི་ར་རར་དིར་ཁྲུས་རས་མེད་པར་ཆུའི་ནང་དུ་
ཞུགས་ཏེ། ལག་པ་དག་སྦྲེལ་ཞིང་ཁྲུས་བྱེད་དེ། ཁྱིམ་མི་མོ་དེ་དག་གིས་མོ་མཚན་ལ་མཛུབ་ཚུགས་དག་བྱེད་ཅིང་གཅིག་ལ་གཅིག་འཕྱས་བཏགས་པ་ལས། ཁྱིམ་མི་མོ་རྣམས་མ་དད་ཅིང་འཕྱ་བ་ལས་བཅས། ཡང་དགེ་སློང་མ་གང་ལག་སྦྲེལ་ཏེ་ཁྲུས་བྱེད་ན་ལྟུང་བའོ། །དགེ་སློང་
མས་ཁྱིམ་མི་མོ་དང་ལྷན་ཅིག་ཁྲུས་ཕྱེད་ན་ལྟུང་བར་འགྱུར་རོ། །ལྟུང་བ་བརྒྱ་བདུན་ཅུ་གཉིས་པ་རྫོགས་སོ།། །།སྟོན་པ་རྒྱལ་པོའི་ཁབ་ན་སྟེ། དེའི་དུས་ན་རྒྱལ་རིགས་དང་བྲམ་ཟེའི་རིགས་མཛའ་སྟེ་དེ་དག་འདུས་ཤིང་རྩེ་སྟེ། འཕོང་རྩལ་དང། ལས་སྣ་ཚོགས་དག་གིས་འགྲན་ནོ།།
དེ་ལྟར་འགྲན་པ་ལས་རྒྱལ་རིགས་རྣམས་ནི་དམག་འགྱེད་པ་ལ་མཁས་སོ། བྲམ་ཟེའི་རིགས་རྣམས་ནི་གླུ་དང་གར་ལ་མཁས་སོ། །དེ་ནས་རྒྱལ་རིགས་རྣམས་བྲམ་ཟེ་ལ་གྲ་མ་ཐུབ་ནས་དེ་དག་སྦོམ་དགའ་མོའི་གན་དུ་དོང་སྟེ། འཕགས་མ་གར་མཁྱེན་ནམ་ཞེས་དྲིས་སོ། །སྦོམ་དགའ་
མོས་ངས་ཤེས་སོ་ཞེས་སྨྲས་པ་དང། དེ་དག་གིས་བདག་ཅག་ལ་བསླབ་ཏུ་གསོལ་ཞེས་བྱས་པ་དང། ང་ལ་རྔན་པ་ཡོད་ན་རུང་ངོ་ཞེས་སྨྲས་པ་དང། དེ་དག་གིས་ཀྱང་རྔན་པ་མང་དུ་བྱིན་ནོ། །དེ་ནས་སྦོམ་དགའ་མོས་གར་ཤ་རྫ་ག་དང། ཀེའུ་ལ་སོགས་པ་སྣ་ཚོགས་བསླབས་པས་
རྒྱལ་རིགས་ཀྱིས་བྲམ་ཟེ་ལ་འགྲན་ཐུབ་པས། བྲམ་ཟེ་རྣམས་སྦོམ་དགའ་མོ་ལ་འཕྱ་ཞིང་མ་དད་པ་ལས་བཅས། ཡང་དགེ་སློང་མ་གང་གར་བྱེད་ན་ལྟུང་བའོ། །དགེ་སློང་མས་བདག་གར་བྱེད་ན་ལྟུང་བའོ། །གཞན་གར་བྱེད་དུ་བཅུག་ན་འང་ལྟུང་བའོ། །གར་བྱེད་པ་ལ་ལྟ་ན་འང་ལྟུང་
བའོ༑ །དཀོན་མཆོག་གསུམ་ལ་མཆོད་པའི་ཕྱིར་བྱེད་དུ་བཅུག་པ་ལ་ཉེས་མེད། ལྟུང་བ་བརྒྱ་བདུན་ཅུ་གསུམ་པ་རྫོགས་སོ།། །།སྟོན་པ་རྒྱལ་པོའི་ཁབ་ན་སྟེ། སྦོམ་དགའ་མོས་རྒྱལ་རིགས་དག་ལ་གླུ་བསླབས་པ་ལས། བྲམ་ཟེ་དག་མ་དད་དེ་འཕྱ་བ་ལས་བཅས། ཡང་
