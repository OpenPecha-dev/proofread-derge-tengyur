རབ་ཏུ་རྒྱས་པར་བཤད། །ཅེས་བྱ་བ་ལ་གྲགས་ལྡན་ཞེས་བྱ་བ་ནི་དེ་བཞིན་གཤེགས་པ་བདུན་པོ་དེ་དག་གོ། །གཞན་ནི་གོ་སླ་བས་བཤད་པ་མ་བྱས་སོ། །སོ་སོར་ཐར་པའི་བསྡུས་པའི་དོན་རྣམ་པ་བརྒྱད་དུ་ཕྱེ་བའི་བཞི་པ་སྔོན་གྱི་ཚུལ་བསྟན་པ་ཞེས་བྱ་བ་བཤད་ཟིན་ཏོ།། །།
ད་ནི་སོ་སོར་ཐར་པ་ཉིད་ཀྱི་འབྲས་བུ་བསྟན་པ་ཞེས་བྱ་བ་འདིར་སྟོན་ཏེ། དེའི་ཕྱིར། འདི་ལ་སངས་རྒྱས་རྣམས་དང་གང། །ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །འདི་ལ་ཡང་རྣམ་པ་གསུམ་དུ་དབྱེ་སྟེ། སོ་སོར་ཐར་པ་ཉིད་ཀྱི་འབྲས་བུ་མདོར་བསྟན་པ་དང། སྤྲོ་བ་བསྐྱེད་པ་དང་
