གྲོལ་བར་བྱ་བ་དང། མྱ་ངན་ལས་འདས་པར་བྱ་བ་ཙམ་ལས་བྱ་བ་གཞན་མི་བྱེད་པའི་ཕྱིར་རོ། །བྱེད་པ་ཉུང་ཞེས་བྱ་བས་གཞན་གྱི་དོན་མི་བྱེད་པ་བསྟན་ཏེ། གཞན་ཞི་བར་བྱ་བ་དང། མྱ་ངན་ལས་འདས་པར་བྱ་བ་ལ་མི་འབད་པར་བྱ་བས་སོ། །ཉོན་མོངས་པ་ཆུང་བའི་
རིགས་ཀྱང་རྣམ་པ་གསུམ་སྟེ། ལུས་དབེན་པ་ལ་དགའ་བ་དང། སེམས་དབེན་པ་ལ་དགའ་བ་དང། གནས་དབེན་པ་ལ་དགའ་བའོ། །འདུ་འཛི་ལ་མི་དགའ་ཞིང་ཞེས་བྱ་བས་ལུས་དབེན་པ་ལ་དགའ་བ་བསྟན་ཏེ། རང་བཞིན་གྱིས་འགྲོ་བ་མང་པོ་དང་འདྲེ་བར་མ་མོས་ཤིང་བསེ་རུ་བཞིན་
དུག་ཅིག་པུར་གནས་པར་མོས་པའི་ཕྱིར་རོ། །དབེན་པ་ལ་དགའ་བ་ཞེས་བྱ་བས་སེམས་དབེན་པ་ལ་དགའ་བ་བསྟན་ཏེ། དེ་དག་འདོད་པ་དང་ཁོང་ཁོ་བའི་རྣམ་པར་རྟོག་པ་རིང་དུ་སྤོང་བར་མོས་པས་སོ། །རབ་ཏུ་དཔེན་པར་གནས་མལ་འཆའ་བ་ཞེས་བྱ་བས་གནས་དབེན་པ་ལ་
དགའ་བ་བསྟན་ཏེ། ཤིང་དྲུང་དང་རི་སུལ་ན་གནས་འཆའ་བའི་ཕྱིར་རོ། །རྐྱེན་ཀྱང་རྣམ་པ་བཞི་སྟེ། ཡོན་ཏན་དང། གནས་པ་དང། རྟོགས་པ་དང། འབྱུང་བའོ། །ཡོན་ཏན་ཡང་རྣམ་པ་གསུམ་སྟེ། ཉོན་མོངས་པ་ཆུང་ཞིང་དགེ་བ་ཆེ་བ་དང། རྫུ་འཕྲུལ་ཆེ་བ་དང་
འབྲས་བུའི་ས་ལ་མངོན་པར་དགའ་བའོ། །བསྙེན་པར་དཀའ་ཞེས་བྱ་བས་ཉོན་མོངས་པ་ཆུང་ཞིང་དགེ་བ་ཆེ་བར་བསྟན་ཏེ། མཐུ་དང་གཟི་བརྗིད་དང་བྱིན་རྣམས་ལ་གཞན་གྱིས་ཟིལ་གྱིས་མི་ནོན་ཞིང་དེས་གཞན་ཟིལ་གྱིས་གནོན་པའི་ཕྱིར་རོ། །མཐུ་ཆེ་བ་ཞེས་བྱ་བས་རྫུ་འཕྲུལ་ཆེ་བ་བསྟན་ཏེ་
སྟོད་ནས་མེ་འབར་བ་དང། སྨད་ནས་ཆུ་གྲང་མོའི་རྒྱུན་འབབ་པ་དང། གཅིག་ཏུ་གྱུར་ནས་མང་པོར་འགྱུར་བ་དང། མང་པོར་གྱུར་ནས་གཅིག་ཏུ་འགྱུར་བ་ལ་སོགས་པའི་མཐུ་དང་ལྡན་པའི་ཕྱིར་རོ། །སྦྱིན་པའི་གནས་ཀྱིས་ལ་མངོན་པར་དགའ་བ་ཞེས་བྱ་བས་འབྲས་བུའི་ས་ལ་དགའ་བ་བསྟན་
ཏེ༑ སྦྱིན་པའི་གནས་ཀྱིས་ནི་འགྲོ་བ་ཐམས་ཅད་ཀྱི་བསོད་ནམས་ཀྱི་ཞིང་དུ་གྱུར་པའི་དུས་དགྲ་བཅོམ་པའི་འབྲས་བུ་ཐོབ་པ་སྟེ། མངོན་པར་དགའ་བ་ནི་འབྲས་བུ་དེ་ཐོབ་པར་བྱ་བར་སེམས་མོས་པའོ། །རྒྱུན་མི་འཆད་པར་རྟེན་ཅིང་འབྲེལ་བར་འབྱུང་བལ་ལྟ་བ་ཞེས་བྱ་བས་གནས་པ་བསྟན་ཏེ་
