གང་འདི་ལ་མཁས་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། གང་འདི་ལ་མཁས་པ་ཞེས་བྱ་བ་ནི་གོང་དུ་བཤད་པ་རྣམས་ལའོ། །ཆོས་ཐམས་ཅད་ལ་མཁས་པ་བསྟན་པའི་ཕྱིར། དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཆོས་ཐམས་ཅད་ལ་མཁས་
པ་གང་ཞེ་ན་ཞེས་དྲིས་སོ། །དྲིས་པའི་ལན་གདབ་པའི་ཕྱིར་ཆོས་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་མདོར་ན་འདུས་བྱས་དང་འདུས་མ་བྱས་ཞེས་བྱ་བ་གསུངས་ཏེ། འདུས་བྱས་དང་འདུས་མ་བྱས་སུ་ཆོས་ཐམས་ཅད་འདུས་པས་འདུས་པའི་དོན་གྱིས་དེ་གཉིས་ལ་ཆོས་ཐམས་ཅད་ཅེས་བྱའོ། །དེ་ལ་
གང་རྒྱུ་དང་རྐྱེན་གྱིས་བསྐྱེད་པའི་ཟག་པ་དང་བཅས་པའི་ཆོས་དང། ཟག་པ་མེད་པའི་ཆོས་ནི་འདུས་བྱས་སོ། །གང་རྒྱུ་དང་རྐྱེན་གྱིས་མ་བསྐྱེད་པ་དེ་བཞིན་ཉིད་དང་ཡང་དག་པའི་མཐའ་དང་སྟོང་པ་ཉིད་དང་ཆོས་ཀྱི་དབྱིངས་ནི་འདུས་མ་བྱས་སོ། །དེ་ལ་བྱང་ཆུབ་སེམས་དཔའ་འདུས་བྱས་དང་
འདུས་མ་བྱས་ལ་མཁས་པར་བྱ་སྟེ་ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱིས་འདུས་བྱས་དང་འདུས་མ་བྱས་ཀྱི་ཆོས་རྣམས་ཀྱི་མཚན་ཉིད་ཁོང་དུ་ཆུད་པར་བྱ་སྟེ་ཞེས་བྱ་བའི་དོན་ཏོ། །གང་དེ་དག་ལུས་ཀྱི་འདུ་བྱེད་དགེ་བ་དང། ངག་གི་འདུ་བྱེད་དགེ་བ་དང། ཡིད་ཀྱི་འདུ་བྱེད་དགེ་བ་
ཞེས་བྱ་བ་ལ། སྲོག་མི་གཅོད་པ་དང། མ་བྱིན་པར་མི་ལེན་པ་དང། འདོད་པས་ལོག་པར་མི་གཡེམ་པ་ནི་ལུས་ཀྱི་འདུ་བྱེད་དགེ་བའོ། །བརྫུན་དང་ཕྲ་མ་དང་ངག་རྩུབ་པོ་དང་ཚིག་ཀྱལ་པའི་ཚིག་མི་སྨྲ་བ་ནི་ངག་གི་འདུ་བྱེད་དགེ་བའོ། །བརྣབ་སེམས་དང་གནོད་སེམས་དང་ལོག་
པར་ལྟ་བ་སྤངས་པ་ནི་ཡིད་ཀྱི་འདུ་བྱེད་དགེ་བ་ཞེས་བྱའོ། །དེ་དག་ཐམས་ཅད་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་བསྔོ་བ་ཞེས་བྱ་བ་ལ། དེ་དག་ཅེས་བྱ་བ་ནི་གོང་དུ་སྨོས་མ་ཐག་པའི་འདུ་བྱེད་རྣམ་གསུམ་མངོན་པར་འདུས་བྱས་པའོ། །བླ་ན་མེད་པ་ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་དེ་སྤོང་
བ་དང༑ ཡེ་ཤེས་དང་སྒྲུབ་པ་དང་དུས་དང་ཐོབ་པ་དང་ལྡན་པ་དང་སྤྱོད་པ་ལ་སོགས་པས་ཉན་ཐོས་ཀྱི་བྱང་ཆུབ་ལས་ཁྱད་ཞུགས་པའི་ཕྱིར་བླ་ན་མེད་པ་ཞེས་བྱའོ། །དེ་ལ་སྤོང་པ་ཁྱད་ཞུགས་པ་ནི་ཉན་ཐོས་ཀྱི་བྱང་ཆུབ་ཀྱིས་ཉོན་མོངས་པའི་སྒྲིབ་པ་ཙམ་སྤོང་བར་ཟད་ཀྱི། འདིས་ནི་ཉོན་མོངས་
