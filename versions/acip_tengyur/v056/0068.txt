༄༅། །པའི་སངས་རྒྱས་རྣམས་ཀྱིས་གང་གསུངས་པ་དང། མ་འོངས་པའི་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་རྣམས་ཀྱིས་གསུང་བར་འགྱུར་བ་གང་ཡིན་པ་དང། ཡང་ད་ལྟར་བྱུང་བའི་སངས་རྒྱས་བཅོམ་ལྡན་འདས་རྣམས་ཡང་དང་ཡང་དུ་གསུངས་པ་གང་ཡིན་པའི་ཞེས་བྱའོ། །
དེས་འདི་སྐད་གསུང་པར་འགྱུར་ཏེ། དུས་གསུམ་གྱི་དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱིས་མཚན་ཡང་དག་པར་བརྗོད་པ་གསུངས་པ་དེའི་ཕྱིར་ཆོས་ཀྱི་རྣམ་གྲངས་འདི་ནི་ནམ་མཁའ་དང་འདྲ་བར་བསྟན་པ་ཡིན་ཏེ། དེ་སྐད་དུ་ཡང་གསུངས་པ། དེ་བཞིན་གཤེགས་པ་རྣམས་བྱུང་ཡང་རུང་མ་བྱུང་ཡང་རུང་ཆོས་ཉིད་ནི་ཀུན་དུ་གནས་པའོ་ཞེས་
བསྟན་པར་འགྱུར་རོ་ཞེས་སོ། །ཡང་ཅི་འདྲ་བ་ཞེ་ན། རྒྱུད་ཆེན་སྒྱུ་འཕྲུལ་དྲ་བ་ལས། །ཞེས་པ་ལ་སོགས་པ་ནི་སྒྱུ་འཕྲུལ་དྲ་བའི་རྒྱུད་ཆེན་པོ་འདི་ལ་དཔག་ཏུ་མེད་པའི་རྡོ་རྗེ་འཛིན་པ་ཆེན་པོས་གསང་སྔགས་ཆེན་པོ་འཛིན་པ་རྣམས་དགའ་ནས་ཡང་དག་པར་གླུ་བླངས་པ་གང་ཡིན་པ་དེ་བདག་གིས་གཟུང་ཞེས་པ་དང་འབྲེལ་ལོ། །
མགོན་པོ་རྫོགས་སངས་རྒྱས་ཀུན་གྱི། །ཞེས་པ་ལ་སོགས་པ་ནི། ཀྱེ་མགོན་པོ་ཇི་ལྟ་བ་བཞིན་དུ་རྫོགས་པའི་སངས་རྒྱས་ཀུན་གྱི་གསང་བ་འཛིན་པར་བདག་འགྱུར་བ་དེ་བཞིན་དུ་བཤད་པར་གསོལ་ཞེས་གོང་མ་དང་འབྲེལ་ཏོ། །འཛིན་ཞེས་བྱ་བ་འདི་ནི་ཆོས་ཀྱི་སྐད་དོ། །ངེས་པར་འབྱུང་བའི་བར་དུ་འདི། བདག་
གིས་བསམ་པ་བརྟན་པོས་གཟུང། །ཞེས་པ་ནི་བདག་གིས་མཚན་ཡང་དག་པར་བརྗོད་པ་འདིའི་ངེས་པར་འབྱུང་བ་མ་ཐོབ་ཀྱི་བར་དུ་གཟུང་ངོ། །ངེས་པར་འབྱུང་ཞེས་པ་ནི་མྱ་ངན་ལས་འདས་པའོ། །ཅི་འདྲ་བར་གཟུང་ཞེ་ན། བསམ་པ་བརྟན་པོས་སོ། །བསམ་པ་ནི་མནོ་བསམ་མོ། །བརྟན་པོ་ནི་མི་གཡོ་བའོ། །
དེ་དང་ལྡན་པ་གང་ཡིན་པ་དེ་ནི་བསམ་པ་བརྟན་པའོ། །ཇི་ལྟ་བ་བཞིན་དུ་བདག་གིས་དེ་རྟོགས་ནས་སེམས་ཅན་ཐམས་ཅད་ཀྱི་བསམ་པའི་ཁྱད་པར་ཇི་ལྟ་བ་བཞིན་དུ་བཤད་པར་འཚལ་ལོ་ཞེས་འོག་མ་དང་འབྲེལ་ལོ། །གང་གི་དོན་དུ་ཞེ་ན། གསུངས་པ། ཉོན་མོངས་མ་ལུས་བསལ་བ་དང། །མི་ཤེས་མ་ལུས་སྤངས་པའི་ཕྱིར། །
ཞེས་པ་སྟེ། དེ་ལ་ཉོན་མོངས་པ་མ་ལུས་པ་ནི་ཉོན་མོངས་པ་དང་ཉེ་བའི་ཉོན་མོངས་པ་གཟུང་ངོ། །དེ་རྣམས་ཟད་པའི་དོན་ནི་ཉོན་མོངས་མ་ལུས་བསལ་བའོ། །མ་ལུས་པའི་དངོས་པོ་མི་ཤེས་པ་ནི་མི་ཤེས་མ་ལུས་ཏེ། །དེ་གཞོམ་པར་བྱ་བ་ནི་མི་ཤེས་མ་ལུས་སྤང་བའི་ཕྱིར་ཏེ། དེ་ཡི་དོན་དུ་བཤད་པར་འཚལ། །ཞེས་
