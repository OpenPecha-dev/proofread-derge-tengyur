ཕྱག་འཚལ་བ་དང། མི་སྙེད་འབྱོར་བ་དང་བློས་སྤྲུལ་པའི་མཆོད་པས་མཆོད་ལ། སྡིག་རྦ་བཤགས་པ་ལ་སོགས་པ་བྱ་སྟེ་དངོས་གྲུབ་བླངས་ནས་རོ་གཅིག་ཏུ་གཞུག་པར་བྱའོ། །དེའི་རྗེས་ལ་སྙིང་པོ་ཅན་གྱི་མཐར་སྔགས་འདི་
དང། གཟུངས་འདི་ཉེ་བར་བཀོད་ལ་རྩེ་གཅིག་ཏུ་བཟླས་སོ། །ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སཱ་ཧཱ། ན་མོ་རཏྣ་ཏྲ་ཡཱ་ཡ། ན་མཿ་ཨཱརྱཱ་ཝ་ལོ་ཀི་ཏེ་ཤྭ་རཱ་ཡ། བོ་དཧི་སཏཱ་ཡ། མ་ཧཱ་སཏཱ་ཡ། མ་ཧཱ་ཀཱ་རུ་ཎི་ཀཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་ཏཱ་རེ་ཏུཏྟཱ་རེ་ཏུ་རེ་སརྦ་དུ་ཥྚཱཾ། པྲ་དུཥྚཱཾ། མ་མ་ཀྲྀ་ཏེ། ཛཾ་བྷཱ་ཡ། སྟྲཾ་བྷཱཾ་ཡ། མོ་
ཧཱ་ཡ། བནྡྷཱ་ཡ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཕཊ་ཕཊ་ཕཊ། སརྦ་དུཥྚ་སྟཾ་བྷ་ནི་ཏཱ་རེ་སཱ་ཧཱ། སྙིང་པོའི་སྔགས་ལ་ནི་འབུམ་ཕྲག་བཅུ་བསྙེན་ཞིང། གཟུངས་སྔགས་ལ་ནི་འབུམ་ཕྲག་གཅིག་གམ་གསུམ་ལ་སོགས་པ་ཅི་རིགས་པར་ཤེས་པར་བྱའོ། །དེ་ནས་ལས་ལ་སྦྱར་བ་ཡང་དེ་ཉིད་དོ། །སྟོབས་བསྐྱེད་པའི་ཕྱིར་
གཏོར་མ་སྦྱིན་པ་ནི། བཅོམ་ལྡན་འདས་མའི་ང་རྒྱལ་ལ་གནས་པས། ཨུཏྤལས་མཚན་པའི་སྣོད་དུ་བཟའ་བ་བཟང་པོ་དང་ཚོད་མ་དང་བཅས་པ་སྦྱང་ཏེ། བགེགས་བསལ་ཞིང་ཡི་གེ་བཞིས་དྲི་མ་བསང་བ་དང། བདུད་རྩིར་བསྒྱུར་བ་དང། སྤེལ་བ་དང། མངོན་པར་དགྱེས་
པའི་རྒྱུར་ལན་གསུམ་དུ་བྱིན་གྱིས་བརླབས་ལ། པདྨ་ཁ་ཕྱེ་བའི་ཕྱག་རྒྱས་དབུལ་བར་བྱའོ། །ཨོཾ་ཨ་ཀཱ་རོ་མུ་ཁཾ་སརྦ་དྷ་རྨྨཱཎཱཾ་དྱ་ནུཏྤ་ནྣ་ཏཱཏ་ཨོཾ་ཨཱ་ཧཱུཾ་ཏཱཾ་ཕཊ་སཱ་ཧཱ། ཨོཾ་ན་མཿ་སརྦ་བུདྡྷཱ་བོ་དཧི་སཏཱ་ནཱཾ། ཨ་པྲ་ཏི་ཧ་ཏ་ཤཱ་ས་ནཱ་ནཾ། ཧེ་ཧེ་བཧག་ཝ་ཏི་བཛྲ་ཏཱ་ར་མ་ཧཱ་སཱ་ཧཱ། སརྦ་
བུདྡྷཱ་ཝ་ལོ་ཀི་ཏེ་ཨི་དཾ་བ་ལིཾ་གྲྀ་ཧཎ་ཨ་མྲཾ་ཏ་ཁཱ་ཧི་ཁཱ་ཧི། ཏིཥྛ་ཏྀཥྛ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཱ་ཧཱ། བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། གརྫ་གརྫ། བི་སྤོ་ཊ་ཡ། བི་སྥོ་ཊ་ཡ། སརྦ་བིགྷྣན་བི་ནཱ་ཡ་ཀ་མ་ཧཱ་ག་ཎ་པ་ཏི་ཇཱཨ་ཝི་ཏཱནྟ་ཀ་རཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཱ་ཧཱ། རྣམ་དག་རྡོ་རྗེ་སྒྲོལ་མ་ཆེ། །
