།རྒྱ་གར་སྐད་དུ། ཥྨ་ཤཱ་ནེཥྚ། བོད་སྐད་དུ། རོ་བསྲེག་བྱ་བ། བཅོམ་ལྡན་འདས་འཇིག་རྟེན་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །མངོན་སུམ་རྡོ་རྗེ་འཛིན་མཛད་པ། །རྡོ་རྗེ་སློབ་དཔོན་ཕྱག་འཚལ་ནས། །དེ་ཡི་ཡོངས་
སུ་བོར་བའི་རོ། །འདུས་བྱས་ཆོ་ག་བཤད་པར་བྱ། །སངས་རྒྱས་བྱང་ཆུབ་སེམས་དཔའ་ཡིས། །མ་ཤེས་པར་ནི་བསྲེགས་གྱུར་ན། །མཆོད་རྟེན་བཤིག་པ་ལྟ་བུར་འགྱུར། །གང་ཡང་བསོད་ནམས་མ་ཡིན་ཞེས། །མཁས་པ་རྣམས་ཀྱིས་བརྒྱུད་
