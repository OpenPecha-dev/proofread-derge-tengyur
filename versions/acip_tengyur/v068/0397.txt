ཨོཾ་བཛྲ་ཧཱུཾ། ཨོཾ་བཛྲ་ཛི་ཧཱ། ཨོཾ་བཛྲ་ཙིཏྟ་ཨ། དེ་ནས་མེ་ལྟར་འབར་བ་བསྐྱེད། །དེ་ལ་བྱིན་གྱིས་བརླབ་པ་དང། །སྦྱི་ནས་དབང་ནི་བསྐུར་བར་བྱ། །ཆོ་ག་བསྒོ་བར་བྱ་བའོ། །ཨོཾ་གྲིཧྣ་བཛྲ་ས་མ་ཡ་ཧཱུཾ་བཾ། ཨོཾ་བཛྲ་ཛྭ་ལ་ཨ་ན་ལཱརྐ་ཧཱུཾ། ཨ་བཧི་ཥིཉྩ་མཱཾ། ཨོཾ་ཊུཾ། དེ་ནས་རྡོ་རྗེ་ཐོ་བ་
དང༑ །ཁྲོ་བོ་རྡོ་རྗེ་མེ་དང་ནི། རྡོ་རྗེ་ལྕགས་སྒྲོག་ཕྱག་རྒྱ་སྔགས། །གློ་བུར་བགེགས་ནི་བསལ་བར་བྱ། །ཨོཾ་བཛྲ་ཛྭ་ལ་ཨ་ན་ལཱརྐ་ཧཱུཾ། ཨོཾ་བཛྲ་ཛྭ་ལ་ཨ་ན་ལ་ཧ་ན་ད་ཧ་པ་ཙ་མ་ཐ་བྷཉྫ་ར་ཎ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནེ་ཏྲི་བནྡྷ་སརྦ་བིགྷྣན། རང་བཞིན་སྒེག་ནི་ཁྲོ་བརྒྱད་ཅིང། །རྡོ་རྗེའི་ཅན་གྱི་འོག་
