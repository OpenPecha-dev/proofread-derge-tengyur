ཡང་གསང་བའི་བཤད་པ་ནི། ཨ་ཀ་ཙ་ཊ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། ཤེས་རབ་འཆར་བའི་དུས་སུ་རྣམ་པར་ཤེས་པའི་ཚོགས་བརྒྱད་གནས་གྱུར་པས་དུར་ཁྲོད་བརྒྱད་ཀྱི་མཚན་ཉིད་བསྟན་ཏོ། །ཇི་སྐད་དུ། རྣམ་ཤེས་བརྒྱད་པོ་རྣམ་དག་པས། །དུར་ཁྲོད་བརྒྱད་ཀྱིས་ཡོངས་སུ་བསྐོར། །
གསང་བའི་གསང་བ་ལ་ཡང། །ཨ་ཀ་ཙ་ཊ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི། །རེ་ལེར་གེའི་རྣལ་འབྱོར་ལས་འབྱུང་བ་དང། ཨོཾ་ཊྲཾ་ཧྲཾ་ཁཾ་ཧཱུཾ་ལཱཾ་མཱཾ་པཱཾ་ཏཱཾ། ཆོས་ཀྱི་འཁོར་ལོ་ལ་ཞེས་བྱ་བ་ནི། ཕུང་པོ་དང་ཁམས་ལ་སོགས་པའི་ཆོས་དུ་མ་རྣམས། རྒྱུར་བསྡུས་པའམ་སྤྲོས་པ་ཡོངས་སུ་སྤང་བར་བྱ་བའི་
ཕྱིར། ཧཱུཾ་གཅིག་པུ་དགོད་པར་བྱ་བའི་དོན་ཏོ། །ནང་གི་བཤད་པ་ལ། རྣམ་པར་སྣང་མཛད་ལ་སོགས་པའི་དེ་བཞིན་གཤེགས་པ་ལྔ་དང། ལྷ་མོ་བཞིའི་ས་བོན་བཞི་ཕུང་པོ་དང་ཁམས་རྣམ་པར་དག་པར་བྱ་བའི་ཕྱིར་ཆོས་ཀྱི་འཁོར་ལོ་ལ་དགོད་ཅེས་བྱ་བ་བསྟན་ཏོ། །གསང་བའི་
བཤད་པ་ནི། ཤེས་རབ་དང་འབྲེལ་བའི་དུས་སུ་ཨོཾ་ཏྲཾ་ཞེས་བྱ་བ་ལ་སོགས་པའི་རྫས་ཕུང་པོ་འབྱུང་བའི་རྩའི་འཁོར་ལོ་ཞེས་བྱ་བ་ནི། །རྩ་དབུས་མ་ལ་བཀོད་ནས་བསྒོམ་ཞེས་བྱ་བའི་དོན་ཏོ། །གསང་བའི་གསང་བ་བཤད་པ་ནི། །ཨོཾ་ཏྲཾ་ཞེས་བྱ་བ་ལ་སོགས་པའི་ཡི་གེ་དགུས་པོ་ལ་དང་ཀཀྐོ་
ལའི་དུས་སུ་དེ་བཞིན་གཤེགས་པ་དང་ལྷ་མོ་ཉེ་བར་དམིགས་པ་དེ་ཁོ་ན་དགོད་པའོ། །ཆོས་ཀྱི་འཁོར་ལོ་ཞེས་བྱ་བ་ནི་ཆོས་ཉིད་ལ་སྦྱར་བར་བྱ་སྟེ། །དེ་བཞིན་གཤེགས་པ་རྣམས་ཀྱང་ཡི་གེའི་ངོ་བོ་ཡིན་པའི་ཕྱིར་རོ། །ཀྵ་ར་ནི་གཡོ་བའི། །ཨ་ནི་དགག་པའི་དོན་ཏོ། མི་གཡོ་ཞིང་མི་
ཤིགས་པས་ན་ཨ་ཀྵ་ར་སྟེ། དེ་བཞིན་ཉིད་བསྟན་ཏོ། །ཇི་སྲིད་དུ། བོ་ལ་ཀཀྐོ་ལའི་སྦྱོར་བ་ལས། །རེག་པས་སྲ་བའི་བག་ཆགས་ནི། །སྲ་བ་གཏི་མུག་ཆོས་ཀྱི་བདག་གཏི་མུག་རྣམ་པར་སྣང་མཛད་འདོད། །ཅེས་བྱ་བ་ལ་སོགས་པ་འབྱུང་ངོ། །སླར་ཡང་དེ་ཉིད་དུ་གསུངས་པ་
