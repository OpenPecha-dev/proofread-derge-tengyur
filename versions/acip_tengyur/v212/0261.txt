༄༅། །བྱེ་བྲག་བྲིས་པའོ། །གླ་རྩི་དང་ནི། དྲིའི་གོས་ཅན། །རི་དགས་ལྟེ་བ་རི་དགས་མྱོས། །མྱོས་པའི་མཚན་མ་དྲི་ཡི་ཐོད། ་གུར་ཀུམ། དཔའ་བོ། གྷྲྀ་སྲི་ཎཾ། །མནྡ། ཧ་རི། ཙནྡན་དང། །ཁ་ཆེ་མཆོག་དང། ཁ་ཆེ་སྐྱེས། །ཙནྡན་དམར་དང་ཡང་
དག་འཁུམས། །སྣང་བྱེད། ཉི་མ། དམར་མིང་ཅན། །ཛཱ་གུ་ཌཾ་དང། པཱི་ཏ་ནཾ། །སཱི་ཏཱ་ཀཱ་དང། བེ་རི་དང། །སེར་སྣ་ཟུར་དང། དྲི་ཞིམ་དང། །ལུས་དམར། ཨ་བི། མེའི་རྩེ་མོ། །མེ་ཏོག་དོན་ཅན། སརྦ་ཊཾ། ་སྤྲིན་གྱི་སྙིང་པོ་རས་བལ་དང། །ཁ་བའི་ཕྱེ་མ་འོད་དཀར་
དང༑ །རྡུལ་གྱི་སྙིང་པོ་ཟླ་བའི་ཐལ། །ཁ་བའི་མིང་ཅན་རྟོགས་བྱེད་དང། །ཤིང་གི་སྙིང་པོ་རོའི་གེ་སར། །མ་ལ་ཡ་སྐྱེས་དྲིའི་སྙིང་པོ། །དཔལ་གྱི་དུམ་བུ་ཙནྡན་དང། །བཟང་དཔལ་མཱ་ཎ་ཡོ་རུ་དང། །བྱ་མེ་ཀཾ་ག་སྐྱེད་པ་དང། །གོས་ཅན། མདོག་ཅན། ཙནྡན་དཀར། །
དྲི་ཞིམ་བཏགས་དང། སྙིང་པོ་སེར། །ཁ་བ་ཡཱ་མྱཾ་ཞེས་པ་དང། །དྲི་ཞིམ་དཀར་པོ་ཞེས་པ་ཡང། ་ཏིལ་འདབ་ཅན་དང། ས་མཆོག་དང། །འཕྲོག་བྱེད་ཙནྡན་ཞེས་པ་དབང། །རྣ་བ་ནག་མོ་དུས་རྗེས་འབྲང། །སྐྱེ་ཅན་དང་ནི། ཁྱབ་འཇུག་སྐྱེ། ་ཨ་ག་རུ་དང་རྒྱལ་མཉེས་
བྱེད༑ །དམར་པོ་སྤྲིན་སྐྱེས་ཛོངྒ་ཀཾ་བ་ཎི་ཀ་དང། དྲི་ཡི་ཤིང། །བྱ་རོག་གདོང་དང་ཙནྡན་སེར། །དུས་སྐྱེས། དོན་ལྡན། ཛོདྒྲ་ཎི། །ནགས་ཀྱི་ཙནྡན་མི་བདག་དགའ། །ཤིང་ནག། མདོག་སྦྱིན། མཛེས་སྦྱིན་བྱེད། ་ཨ་གུ་རུ་ནག་ཨ་གུ་རུ། །བཀྲ་ཤིས་ཕྲེང་དང་
དྲི་དང་ནི། །ཙནྡན་དམར་དང་འདབ་མའི་ལུས། །རཉྫ་ན་དང་སའི་ཙནྡན། །སྣུམ་ལྡན་དང་ནི་བསམ་མི་ཁྱབ། ་སི་སནྟུ་རཾ། རྣ་བ་འཛིན། །ཀོ་ལ་ཀ་དང་སྣུམ་འབྲས་དང། །སྦུབས་སྐྱེས་ཛ་ཏི་ཕ་ལ་མཚུངས། །ཞོ་དང་འོ་མ་མར་གྱི་མཚན། །དཔལ་གྱི་གོས་དང་ཐང་ཆུའོ། །སྦྱར་
སྤོས་དང་ནི་པཱ་ཡབ། །རཀྟ་ཤཱིརྵ་ཀ་ཞེས་སོ། །གནོད་སྦྱིན་སྤོས་དང་སྤོས་དཀར་དང། །སྲ་རྩི་བྱེད་དང་ཀུན་ལྡན་དང། །གཟུགས་མང་ནོར་བུ་སྤྲོ་བ་དང། །པྲི་ཎ་ཀ་དང། མིའི་ལྕུག་མ། །ཨཱ་ར་ལ་དང་སྡོང་པོ་ཅན། །གྲགས་ལྡན་གདོང་བཀྲིས་གཡོ་བའི་གདོང། ་ཏུ་རུ་ཥྐ་དང་
