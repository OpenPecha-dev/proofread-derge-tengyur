སྤྲིད་འབྱུང་སྨན། །ཨུ་པ་ལ་དཧི་འཐོབ་པ་དང། །བློ་གྲོས་ཤེས་རབ་མཚན་ཉིད་མ། །ཀཱ་ལ་ཀནྡྷ་ཏ་མཱ་ལ། །ཙིནྡུ་ཀ་དང་འཚོ་བྱེད་ཤིང། །བྷཱིཀྵྞ་གནྡྷ་ཤུ་དག་དང། །སྐེ་ཚེ་མ་ཏ་ཤླི་གྲུའོ། །ཏྲྀ་ཎཾ་གོ་དྷཱ་ཙི་ཏྲ་ཀ་རྒྱ་སྐྱེགས་དང་ནི་རྩངས་པའོ། །བ་རི་བྱ་དྷ་འདོད་
ཆུང་དང། །ཨུཏྟ་ལའི་ཤིང་ལའོ། །བྲཧྨ། །བནྡྷུ་ལྷག་པར་འདོར། །ངེས་པར་སྟོན་དང་གཉིས་སྐྱེས་མཆོག་མ་ཧཽ་ར་ཥ་དུག་བཅའ་སྒ། །ཤྲྀངྒ་བ་ར། རོ་བཅུད་དོ། །ས་མུནྣདྡྷ་ཀུན་ཏུ་འབྱུང། །མཁས་པར་རློམ་དང་ཚིམ་པ་ལ། །ཡོ་ཛ་ན་གནྦྷ། གླང་རྩི། །རྒྱས་པས་སྤྲོས་དང་
