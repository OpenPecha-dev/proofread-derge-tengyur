།ཡ་ཡི་ཌྷྭི། ཡ་ཡེ། ཡ་ཡི་བ་ཧེ། ཡ་ཡི་མ་ཧེ། བཱ། བ་བཽ། བ་བ་ཏུཿ་བ་བུཿ་བ་བཱ་ཐ། བ་བི་ཐ། སྠཱ། གཉིས་བརྗོད་དོ། །ཥི་ཊ་ལས་ཕ་རོལ་གྱི་ཨ་གྷོ་ཥའོ་ཞེས་པས་
ཥིཊ་ལས་ཕ་རོལ་གྱི་སྒྲ་མེད་པ་བསྲུང་བཞིན་པར་འགྱུར་རོ། །ཥི་ཊའི་དབྱིའོ་ཞེས་པའི་དོན་ཏོ། །གཉིས་པ་བཞི་པ་དག་གི་དང་པོ་གསུམ་པ་དག་གོ་ཞེས་པས་ཐ་ཡིག་ཏ་ཡིག་གོ། །ཏསྠཽ། ཏ་སྠ་ཏུཿ་ཏསྠུ་ཏསྠཱ་ཐ། ཏསྠི་ཐ། གླཻ། མཚམས་སྦྱོར་གྱི་ཡི་གེའི་མཐའ་
རྣམས་ཀྱི་བི་ཀ་ར་ཎ་མ་ཡིན་པ་ལ་ཨཱ་ཡིག་གོ་ཞེས་པས་ཨཱ་ཉིད་དོ། །ཀ་སྡེའི་ཙེ་སྡེའི། ཛ་གླཽ། ཛ་གླ་ཏུཿ་ཛ་གླུཿ་དངོས་པོ་ལ། ཛ་གླཽ། མལཽ། མ་མལཽ། མ་མླ་ཏུཿ་མ་མལུཿ་ཨཱཏྨ་ནེའི་ཚིག་ལ། མམླི། ཌུ་དཱཉ། ད་དཽ། ད་ད་ཏུཿ་ད་དུཿ་ད་དེ། ཌུ་དྷཱཉ། །ཌྷཽ། ད་དྷ་
ཏུཿ་ད་དཧུཿ་ད་དཧེ། ཙཀྵིང། པ་རོཀྵཱ་ལ་ཡང་ནའོ་ཞེས་པས་ཡང་ན་ཁྱཱཉ་དུ་བསྒྱུར་རོ། །གཉིས་བརྗོད་དོ། །ཀ་སྡེའི་ཙ་སྡེའོ། །གཉིས་པའི་དང་པོའོ། །ཨཱ་ཙ་ཁྱཽ། ཨཱ་ཙ་ཁྱ་ཏུཿ་ཨཱ་ཙ་ཁྱུཿ་ཨཱ་ཙ་ཁྱེ། ཨཱ་ཙ་ཁྱཱ་ཏེ། ཨཱ་ཙ་ཁྱཱི་རེ། ཉ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ཀྱི་ཕྱིར་
གཉི་གའི་ཚིག་གོ། །ཨ་ཙ་ཙི་ཀྵེ། ཨཱ་ཙ་ཙ་ཀྵཱ་ཏེ། ཨཱ་ཙ་ཙཀྵི་རེ། ཨེ་ཡིག་མཐའ་རྣམས། ཛི་རྒྱལ་བ་ལའོ། །ཛིའི་སན་པ་རོཀྵཱ་དག་ལ་གིའོ་ཞེས་པས་གི་རུ་བསྒྱུར་རོ། །གཉིས་བརྗོད་དོ། །ཨའི་ཨུ་པ་དྷ་ཞེས་སོགས་ཀྱིས་འཕེལ་བའོ། །ཨཻ་ཨཱ་ཡའོ་ཞེས་པས་
ཨ་ཡ་རུ་བསྒྱུར་རོ། །ཛི་གཱ་ཡ། པ་རོ་ཀྵཱ་ལའང་ངོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །སྔ་མ་བརྩེགས་པ་མིན་པའི་ཨི་ཡིག་གི་ཨ་སྟེ་ཡི་གེ་གཅིག་མིན་པའི་འོ་ཞེས་པས་ཡ་ཡིག་གོ། །ཛི་གྱ་ཏུཿ་ཛི་གྱུཿ་ཐལ་ལ་ཨེ་ཊ་ཡང་ན་བ་ཉིད་དོ། །ཡོན་ཏན་དང་ཨ་ཡ་རུ་བསྒྱུར་
བ་དག་གོ། །ཛིག་ཡི་ཐ། ཛེ་གེ་ཐ། ཛེ་གྱ་ཐུཿ་ཛི་གྱ། ཨ་ཊ་ཕྱི་མ་ལ་ཡང་ན་འཕེལ་བའོ། །ཛི་གཱ་ཡ། ཛི་ག་ལ། ཛི་གྱི་བ། ཛི་གླི་མ། ཨཱཏྨ་ནེའི་ཚིག་ལ། ཨཱཏྨ་ནེ་ལ་ཐམས་ཅད་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །བི་ཛི་གྱེ། བི་ཛི་གྱཱ་ཏེ། བ་ཛི་གྱི་རེ་༄༅།
