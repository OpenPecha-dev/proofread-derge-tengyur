ཛུ་ཧོ་མི། ཛུ་ཧོ་བཿ། ཛུ་ཧུ་མཿ། ཉི་བྷཱི་འཇིག་པ་ལའོ། །ཉི་ཨིག་ན། ཉི་རྗེས་སུ་འབྲེལ་པ་ལ་འདོད་པ་བློ་མཆོད་པའི་དོན་རྣམས་ཀྟའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨན་འཇིག་གོ། །གཉིས་བརྗོད་དོ། །ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུའོ། །གཉིས་པ་བཞི་པ་
དག་གི་དང་པོ་གསུམ་པ་དག་གོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་བ་ཡིག་གི་བ་ཡིག་གོ། །ནཱ་མིན་མཐའི་ཡོན་ཏན་ནོ། །བི་བཧེ་ཏི། བཧིའི་ཡང་ན་བཤད་པར་བྱའོ་ཞེས་པས་གསལ་བྱེད་དང་པོ་ཡོན་ཏན་ཅན་བྱིངས་ཐམས་ཅད་པ་ཡོན་ཏན་མེད་པ་ལ་རྣམ་པར་བརྟག་པས་ཨཱི་ཉིད་དོ། །བི་བྷཱི་ཏཿ་སྔ་
མ་བརྩེགས་པ་མ་ཡིན་པའི་ཨིའི་ཡི་གེའི་ཡ་སྟེ་ཡི་གེ་གཅིག་མིན་པའི་འོ་ཞེས་པས་ཡ་ཉིད་དོ། །བི་བྷྱེ་ཏི། བི་བཧེ་ཥིཿ། བི་བཧི་ཐཿ། བི་བཧི་ཐ། བི་བཧི་མི། བི་བཧི་བཿ། བི་བྷཱ་བཿ། བི་བཧི་མཿ། བི་བྷཱི་མཿ། ཧྲཱི། ངོ་ཚ་བ་ལའོ། །ཨ་ན་འཇིག་གོ། །གཉིས་བརྗོད་དོ། །ཨ་བྷྱཱ་སའི་གསལ་བྱེད་
དང་པོ་ལྷག་མར་བྱའོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་གསལ་བྱེད་དང་པོ་ལྷག་མར་བྱ་བ་སྟེ་བསྲུང་བར་བྱ་བར་འགྱུར་གྱི་དང་དབྱི་བར་བྱ་བ་མ་ཡིན་ནོ་ཞེས་པའི་དོན་ཏོ། །ཐུང་ངུའོ། །ནཱ་མིན་མཐའ་ཡོན་ཏན་ནོ། །ཛི་ཧྲེ་ཏི། ཛི་ཧཱི་ཏིཿ་སྔ་མ་བརྩེགས་པ་མ་ཡིན་པའི་ཞེས་པའི་ཚིག་ལས་སྔ་མ་བརྩེགས་
པའི་ཡ་ཉིད་དུ་མི་འགྱུར་རོ། །ཨི་ཡ་རུ་བསྒྱུར་རོ། །ཛི་ཧྭི་ཡ་ཏི། ཛི་ཧྲི་ཥི། ཛི་ཧྲཱི་ཐཿ། ཛི་ཧྲི་ཐ། ཛི་ཧྲི་མི། ཛི་ཧྲི་བིཿ། ཛི་ཧྲི་མཿ། པྲྀ་སྐྱོང་བ་དང་རྫོགས་པར་བྱེད་པ་དག་ལའོ། །ཨན་འཇིག་གོ། །གཉིས་བརྗོད་དོ། །རྀ་ཡིག་གི་ཨ་ཡིག་གོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་རྀ་ཡིག་ཨཱ་ཡིག་
གོ༑ །ཨརྟི་པི་བརྟི་དག་གི་འང་ངོ་ཞེས་པས་ཨི་ཡིག་གི་ཨི་ཉིད་དོ། །ནཱ་མིན་མཐའི་ཡོན་ཏན་ནོ། །པི་བརྟྟི། པི་པྲྀ་ཏཿ། བི་པྲ་ཏི། བི་པརྵི། བི་པྲྀ་ཐཿ། བི་པྲྀ་ཐ། བི་པརྨྨི། བི་ཕྲྀ་བཿ། བི་པྲྀ་མིཿ། ཁ་ཅིག་ཏུ་པྲཱྀ་ཞེས་རིང་པོའི་མཐར་སེམས་ཏེ། དེ་འདོད་པ་དེ་རྣམས་ཀྱང་གང་ལ་ཡོན་
ཏན་ཡོད་པ་དེ་ལ་ཁྱད་པར་མེད་དོ། །གཞན་ལ་ནི་མཆུར་གྱུར་ཨུ་པ་དྷཱའི་ཡང་ཨུར་རོ་ཞེས་པས་ཨུར་དུ་བསྒྱུར་རོ། །ནཱ་མི་ནཱའི་ར་པ་དག་ལ་གསལ་བྱེད་ལ་སྟེ་ཀུར་ཚུར་དག་མ་ཡིན་པའི་འོ་ཞེས་པས་རིང་པོའོ། །པི་པཱུརྟྟ། པི་བུ་ར་ཏི། པི་པ་རི། པི་པཱུརྠ། བི་པུརྠ། པི་བུརྨྨི། བི་པུརྦྦ
