།ཨཱ་ཙ་ཌྷྭེ། ཨཱ་ཙཀྵེ། ཨཱ་ཙཀྵ་ཧེ། ཨཱ་ཙཀྵྨ་ཧེ། ཨཱིར་འགྲོ་བ་དག་ལ་སྟེ་མགུལ་བ་ལ་ཡང་ངོ། །ཨན་འཇིག་གོ། །ཨིརྟྟེ། ཨཱི་རཱ་ཏེ། ཨཱི་ར་ཏེ། ཨཱི་ར་ཥེ། ཨཱི་རཱ་ཐ། ཨཱིརྡྷེ། ཨཱི་རེ། ཨཱིརྦ་ཧེ། ཨཱིརྨྨ་ཧེ། སམ་སྔོན་དུའོ། །ས་མཱིརྟྟེ། ཚིག་དབུས་
མའི་གནས་ཉིད་ཀྱི་ཕྱིར་རེ་པའི་རྣམ་བཅད་དུ་མི་འགྱུར་རོ། །ཨཱི་ཌ་བསྟོད་པ་ལའོ། །ཨཱན་འཇིག་གོ། །ཊ་སྡེ་ཉིད་དོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཥི་ཊི་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་ཊ་ཡིག་གི་ཌ་ཡིག་གོ། །ཨཱིཌེ། ཨཱི་ཌཱ་ཏེ། ཨཱི་ཌཱ་ཏི། ཨཱི་ཊ་ཛ་ན་དག་གི་ས་དང་དཧྭ་ལའང་ངོ་ཞེས་
པས་ཨཱི་ཊ་ཨཱ་ག་མའོ། །ཨི་ཥ་ཉིད་དོ། །ཨཱི་ཌི་ཥེ། ཨཱི་ཌཱ་ཐེ། ཨིཊ་ཨཱ་ག་མའོ། །ཨི་ཌི་དཧེ། ཨཱི་ཌི། ཨཱི་ཌ་ཧེ། ཨཱཌྨ་ཧེ། ཨཱི་ཥ། དབང་ཕྱུག་ཉིད་ལའོ། །ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཊ་སྡེ་ཉིད་ཀྱང་ངོ། །ཨཱིཥྚེ། བི་ཥཱ་ཏཻ། ཨཱི་ཤཱ་ཏེ། ཨཱི་ཤ་ལས་ས་
ལའོ་ཞེས་པས་ཨཱིཊ་ཨཱ་ག་མའོ། །ཨཱི་ཤི་ཥེ། ཨཱི་ཤཱ་ཐེ། ཨཱིཌ་ཛ་ན་དག་གི་ས་དང་དཧྭ་ལའང་ངོ་ཞེས་ཀློག་པ་ཉིད་ལས་ཡང་ཡིག་གིས་ཨཱི་ཤ་ལ་རྗེས་སུ་འགུགས་པར་བྱའོ། །དེས་ན་ཨཱི་ཤ་ལ་ཡང་དཧྭེ་ཕ་རོལ་ལས་ཨི་ཊར་འགྱུར་རོ། །ཨཱི་ཤི་དཧྭེ། ཨཱི་ཤི། ཨཱི་ཤྭ་ཧེ། ཨཱི་ཤྨ་ཧེ། ཨཱ་
ས༑ ཉེ་བར་བསྟོད་པ་ལའོ། །ཨན་འཇིག་གོ། །ཨཱ་སྟེ། ཨཱ་སཱ་ཏི། ཨཱས་ཏེ། ཨཱསྶེ། ཨཱ་སཱ་ཐེ། དཧུ་ཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་ས་ཡིག་གི་ད་ཡིག་གོ། །དེའི་རིགས་ཅན་རྣམས་ཡང་དག་པར་སྦྱོར་བའི་ཡི་གེ་རྣམས་ཀྱི་གཅིག་གམ་གཅིག་མ་ཡིན་པའི་བརྗོད་
པ་སོ་སོའི་དབྱེ་བ་མ་ཡིན་པ་ལས་ས་ཡིག་གི་གསུམ་པ་ཉིད་ལ་ཡང་སྐྱོན་མེད་དོ། །དབྱིའོ་ཞེས་ཁ་ཅིག་གོ། །ཨཱ་དཧྭེ། ཨཱ་སེ། ཨཱ་སྭ་ཧེ། ཨཱསྨ་ཧེ། ཨུ་བཱ་སྟེ། ཨུ་བཧུ་བཱ་སྟེ། ཨུ་དཱ་ཏེ། ཨཱནྨ་ཏེ། ཨ་དྷྱ་ཏེ། སྭ་སྟེ། ཨཱང་ཤཱ་སུ། འདོད་པ་ལའོ། །ཨཱ་ཤཱ་སྟེ། ཨཱ་ཤཱ་སཱ་སྟེ། ཨཱ་ཤཱ་ས་ཏེ། ཨཱ་ཤཱསྶེ་
ཨཱ་ཤཱ་སཱ་ཐེ། སྔ་མ་བཞིན་སའི་གསུམ་པའོ། །ཨཱ་ཤཱ་དཧྭེ། དབྱིའོ་ཞེས་ཁ་ཅིག་གོ། །ཨཱ་ཤཱ་སེ། ཨཱ་ཤཱས་ཧེ། ཨཱ་ཤཱསྨ་ཧེ། བ་ས། ཀུན་ཏུ་འགེབས་པ་ལའོ། །བ་སྟེ། བ་སཱ་ཧེ། བ་ས་ཏེ། བསྶེ། བ་སཱ་ཐེ། བ་ས་དཧྭེ། བ་སེ། བ་སྭ་ཧེ། བསྨ་ཧེ། ཀ་སི། འགྲོ་བ་དང་རྣོ་བར་བྱེད་པ་དག་
