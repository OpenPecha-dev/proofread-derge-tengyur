རྔ་སྒྲའི་རྒྱུན་འབྱུང་མི་འཇིགས་སྦྱིན་པ་ནི། །ཉོན་མོངས་གཡུལ་དུ་འཇུག་ཚེ་ལྷ་མིན་གཡུལ་རྒྱལ་རྩེད་མོ་སེལ་བ་བཞིན། །དེ་བཞིན་འཇིག་རྟེན་དག་ན་བསམ་གཏན་གཟུགས་མེད་ལ་སོགས་རྒྱུས་བྱུང་བ། །སེམས་ཅན་ཉོན་མོངས་སྡུག་བསྔལ་རབ་འཇོམས་ཞི་བ་བླ་མེད་ལམ་ཚུལ་བརྗོད། །
ཅིའི་ཕྱིར་འདིར་ཆོས་ཀྱི་རྔ་ཉིད་ཀྱི་དབང་དུ་བྱས་ཀྱི། ལྷའི་སིལ་སྙན་གྱི་རྣམ་པ་དེ་ལས་གཞན་པ་དག་ཀྱང་མ་ཡིན་ཏེ། དེ་དག་ལྷ་རྣམས་ཀྱི་སྔོན་བྱས་པའི་ལས་ཀྱི་དབང་གིས་མོད་ཁོ་ནར་ལྷའི་རྣ་བར་ཡིད་འཕྲོག་པའི་སྒྲའི་རྗེས་སུ་འབྱུང་བར་འགྱུར་རོ་ཞེ་ན། དེ་བཞིན་གཤེགས་པའི་དབྱངས་
དང་དེ་དག་ཆོས་མི་མཐུན་པའི་ཡོན་ཏན་རྣམ་པ་བཞིའོ། །དེ་དག་ཀྱང་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཉི་ཚེ་བ་ཉིད་དང། ཕན་པ་མ་ཡིན་པ་ཉིད་དང། བདེ་བ་མ་ཡིན་པ་ཉིད་དང། ངེས་པར་འབྱིན་པར་བྱེད་པ་མ་ཡིན་པ་ཉིད་དོ། །ཆོས་ཀྱི་རྔ་ནི་བག་མེད་པའི་ལྷའི་ཚོགས་མཐའ་
