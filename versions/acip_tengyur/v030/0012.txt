ཅེས་བྱ་བའི་བར་ནི་སྒྲ་ཇི་བཞིན་པའོ། །དེ་ནས་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་མཆོག་ཅེས་བྱ་བ་ནས། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་གྱི་ལེའུ་ཞེས་བྱ་བའི་བར་ནི་དྲང་བའི་དོན་དང་ངེས་པའི་དོན་ཏོ། །ལེའུ་བཞི་བའི་བཤད་པའོ། །དེ་ནས་ཞེས་བྱ་བ་ནས་

ཞེས་བྱ་བའི་བར་ནི་དྲང་བའི་དོན་དང་ངེས་པའི་དོན་ཏོ། །དེ་ནས་ཞེས་བྱ་བ་ནས། བྱང་ཆུབ་སྤྱོད་པའི་གནས་འདི་ཡིན། །ཞེས་བྱ་བའི་བར་དུ་ནི་སྒྲ་ཇི་བཞིན་པའོ། །དེ་ནས་ཡང་བརྗོད་ཀྱིས་མི་ལང་བའི་ཞེས་བྱ་བ་ནས། བརྒྱལ་བར་གྱུར་ཏོ། །ཞེས་བྱ་བའི་བར་ནི་དྲང་བའི་དོན་དང་ངེས་པའི་དོན་ཏོ།།
དེ་ནས་བཅོམ་ལྡན་འདས་ཞེས་བྱ་བ་ནས། བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བྱ་བའི་བར་ནི་སྒྲ་ཇི་བཞིན་པའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཞེས་བྱ་བ་ནས། རང་རང་གི་སྟན་ལ་གནས་པར་གྱུར་ཏོ། །ཞེས་བྱ་བའི་བར་ནི་དྲང་པའི་དོན་དང་ངེས་པའི་དོན་ནོ། །དེ་ནས་ཞེས་བྱ་བ་ནས། སྤྱོད་པ་མཆོག་གི་
