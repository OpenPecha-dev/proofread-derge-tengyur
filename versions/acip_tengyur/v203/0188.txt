དྲུག་ཡང་བྲི། ཀ་ར་ཀ་སོགས་ལྷོ་བགྲོད་དྲུག་ལ་བསྣན་པས་འཕོ་བའི་ཆུ་ཚོད་དག་པར་འགྱུར། ཆུ་ཚོད་དུ་འདས་ལྟ་བ་དེ། །རྒྱ་མཚོ་ཆུ་དང་ཟླ་བ་ལ། །ལུག་དང་སྲང་ལ་དྲུག་སྦྱིན་བྱ། །གླང་དང་བུ་མོ་རང་སར་གཞག །འཁྲིག་པ་སེང་གེ་བརྒྱད་དོར་བྱ། །ཀ་ར་ཀ་གཉིས་ལ་བཅོ་བརྒྱད་དབྲི། །སྡིག་ཉ་གཉིས་
ལ་བཅུ་སྦྱིན་ཞིང་། །གཞུ་དང་ཆུ་སྲིན་བུམ་པ་ནི། །གསུམ་ལ་ཉི་མ་བྱིན་པ་ལ། །ལུག་དང་སྲང་ལ་དྲུག་སྦྱིན་བྱ། ཆ་བདུན་གྲིབ་མར་བཅས་པས་བགོས་པས་ཐོབ་པའི་ནོར་ནི་སྔ་དྲོ་ཆུ་ཚོད་འདས་པ་ཡིན། ཕྱི་དྲོ་ལུས་པ་ཉིད་དུ་འགྱུར་རོ། །འདི་དུས་སྦྱོར་བཅུ་གཉིས་ལ་སོགས་ལ་ངོས་འཛིན་ན། དེའི་དུས་
ཀྱི་ཆུ་ཚོད་ནི་འོངས་པ་དྲུག་ཅུས་བསྒྱུར་བར་བྱ། དེས་ནི་དབྱུ་གུ་ཡང་ཆུ་སྲང་གི་ཕུང་པོར་འགྱུར་རོ། །དེ་ལ་ལུག་ལ་ཉིས་བརྒྱ་ལྟ་བུ་ཁྱིམ་རང་རང་གིས་བགོའོ། །ལུག་སོགས་དྲུག་ལ་ཡས་ཀྱིས་དབྲི་ཞིང་། སྲང་སོགས་དྲུག་ལ་མས་ཀྱིས་དབྲིའོ། །མ་ལྡང་བ་དེ་ཏ་ཏྐལ་ཡིན་ནོ། །དེ་ནི་ཉིན་ཞག་
གཅིག་གི་བཅུ་གཉིས་ཆའོ། །འཕེལ་བ་དེ་ལས་གནས་ལྔ་བྱེད་ན། ལྷག་སུམ་ཅུས་བསྒྱུར་བ་ལ་ཁྱིམ་ཆུ་སྲང་གིས་བགོས་པས། ནོར་ནི་ཆ་ཤས་ཡིན་བས་ཏཏྐལ་འོག་ཏུ་བྲིའོ། །ལྷག་དྲུག་ཅུས་བསྒྱུར་ཞིང་། །ཐོབ་པ་དབྱུ་གུ་ཡིན་ཞིང་། དེའི་ལྷག་དྲུག་ཅུས་བསྒྱུར་བས་ཐོབ་པ་ཆུ་སྲང་ཡིན། དེའི་ལྷག་དྲུག་
གིས་བསྒྱུར་བས་ཐོབ་པ་དབུགས་ཡིན་ནོ། །གནས་ལྔ་ཞེས་ནི་དེ་ལ་ཟེར། དུས་དྲུག་དག་གམ་ལྟ་བ་དང་། །དུམ་བུ་དྲུག་གི་དོན་དུ་ཡང་། །ཏདྐཱ་ལ་ཆུ་སྲང་སུམ་ཅུས་བསྒྱུར། །དེ་ལ་འོག་གི་ཅ་ཤས་གཞུག །དེ་འང་དྲུག་ཅུས་བསྒྱུར་ཞིང་ཆུ་ཚོད་གཞུག །སྟོང་དང་བརྒྱད་བརྒྱས་དེ་ལ་བགོ་བར་བྱ། །ནོར་ནི་
དུས་སྦྱོར་གཅིག་གི་རང་བཞིན་ཏེ། །གོང་གི་དེ་དང་མཉམ་ན་དག་པ་ཡིན། །དག་པར་མ་གྱུར་མཉམ་པའང་མིན་ནོ། །ཆ་ཤས་དྲུག་ཅུས་བསྒྱུར་ཆུ་ཚོད་སྟེང་དུ་གཞུག །དེ་ནི་དུམ་བུ་ལྔའི་དགོད་བྱ་ཡིན་ནོ། །དེ་ལ་དགུ་བརྒྱས་བགོས་ནོར་དུས་སྦྱོར་ཕྱེད་པའོ། །དྲུག་བརྒྱས་བགོས་ནོར་སུམ་ཆ་ཡིན། །
ཉིས་བརྒྱས་བགོས་ནོར་དགུ་ཆ་ཡིན། བརྒྱ་དང་བཅུ་ཡིས་བགོས་ནོར་ནི། །བཅུ་གཉིས་ཆ་རུ་འགྱུར་བ་ལགས། །དྲུག་ཅུས་བགོས་བའི་ནོར་ནི་སུམ་ཅུའོ། །དེ་ལྟར་དུམ་བུ་དྲུག་སྟེ་འདི་རྣམས་ཐམས་ཅད་ལ་ཕྱེད་པས་ལྷག་ལ་རེ་རེ་སྟེང་དུ་སྦྱིན་པར་བྱའོ། །འདི་རྣམས་རྒྱུད་ཀྱི་བཅུད་ལྟ་བུ་ཡིན་ཞིང་།
