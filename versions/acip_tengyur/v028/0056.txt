བསམ། །དུར་ཁྲོད་རྗེས་གསུམ་རྡོ་རྗེའི་ཆུས། །སྦྱར་ཏེ་སྔགས་བཅས་སྦྱིན་པར་བྱ། །དེ་རྗེས་སྐེ་ཚེ་མིང་དང་བཅས། །དབང་སྡུད་ཡིད་ཀྱིས་སྦྱིན་སྲེག་བྱ། །བདུན་ཕྲག་ཞག་ནི་གཅིག་དང་གཉིས། །དེ་བཞིན་བདུན་གྱི་མི་མོ་དང། །མི་ཕོ་ལྷ་མོ་སོགས་དབང་བསྐུར། །ཨོཾ་ཏཱ་རེ་
ཏུཏྟཱ་རེ་ཏུ་རེལ་པྦྷ་ལམྦྷ་སོ་སཱ་ཏིཀྵ་སི་ཨུ་ཀྭོ་སི་བ་ཤཾ་ཀ་ར་ཡ་ཨ་མུ་ཀཾ་སྱ་དྷྱརྨ་གྷ་ཌ་ནི་ར་ཧ་ཏྲ་ཙ་མ་ན་བ་བྷྱ་སིད་ཧ་ད་ཧ་སྠི་དྷྱ་མ་མ་སྫ་ཀ་དྷ་ཧ་ལཾ་པ་ན་བི་རུ་ཏི་པ་ལ་ཎཱ་པ་ཙ་ཏི་རཏྣེ་ཏཾ་བ་ལན་དརྱ་མ་ནེ་ཧི་ཀུ་ཏོ་ནི་ཀུ་ཀུ་ཏྲོ་རོ་ཏྲོ་ཏྲི་ཡ་ཏི་བ་ས་ནེ་ཧི་བ་ལན་བ་དྷ་ཕུ་ར་ས་ཀྲི་ཏཾ། ཨོཾ་ཙི་ཏྲི་ཙི་ཏྲི་བི་ལོ་ལི་ལ་བི་ལོ་ལ་
སཱ་ཧཱ། ཨཱ་མུ་ཀཾ་སཱ་དྷ་ནཾ་པ་ར་པ་ཏཱམྨེ་ད་ས་མ་ན་ཡེ་སཱ་ཧཱ། ཨོཾ་ཀུ་རུ་ཀུལླེ་ཧྲཱི་སཱ་ཧཱ། འབུམ་ཕྲག་རྣལ་འབྱོར་རྩེ་གཅིག་གིས། །མེ་ཏོག་བདུག་པ་བཟའ་བཏུང་དག་བྱིན་ན་བརྒྱ་བྱིན་ཡང་དབང་བྱེད། །མེ་ཏོག་དམར་པོའི་སྦྱིན་སྲེག་གཅེས། །ཡུངས་ཀར་སྐྱེས་བུ་མཆོག་
ལ་བསྔགས། །འཁར་བའི་སྣོད་དུ་སྦྲང་རྩི་མར། །སྔགས་དང་བཅས་གུས་དུས་གསུམ་དུ། །མཆོད་པ་བཅས་པས་དོན་རྣམས་འགྲུབ། །ཁམ་ཕོར་རམ་ནི་དུང་ཕོར་དུ། །སྦྲང་རྩི་སོགས་ལ་བརྒྱ་རྩ་བརྒྱད། །བཟླས་སོ་མིག་བྱུགས་དགྲ་པོ་ཡང། །བུ་སྡུག་བཞིན་དུ་
