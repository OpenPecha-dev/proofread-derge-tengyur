༄༅། །ཤེས་པ་སྐྱེ་བ་མེད་པ་ནི། །ཀུན་ཤེས་ཉིད་ཀྱི་སྐད་ཅིག་མ། །ཞེས་བསྟན་པ་ཡིན་ནོ། །དེ་ཉིད་ནི་རབ་འབོར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་ཆོས་གནག་ཡང་ཐོབ་པར་མི་བྱེད། མཐོང་བར་མི་བྱེད། ཐམས་ཅད་ཤེས་པ་ཉིད་ཀྱི་དབང་དུ་བྱས་ཏེ་རྫོགས་པར་གསལ་བར་
བསྟན་པ་འདི་དང། རབ་འབྱོར་དེ་ལྟར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་འདོད་པའི་ཁམས་ལས་ཤིན་ཏུ་འདའ་བ་མ་ཡིན་ཏེ། བཟློག་པའི་རིམ་པས་དེ་མ་ཐག་ཏུ་ལམ་ཤེས་པའི་དབང་དུ་བྱས་པ་རྫོགས་བར་བསྟན་པ་འདི་དང། རབ་འབྱོར་གཞན་ཡང་དེ་ལྟར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་
ཕྱིན་པ་ནི་སངས་རྒྱས་ཀྱི་ཆོས་རྣམས་སྦྱིན་བར་མི་བྱེད། འདུས་བྱས་འདོར་བར་མི་བྱེད། འདུས་མ་བྱས་འཐོབ་པར་མི་བྱེད་དེ། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་དབང་དུ་བྱས་པ་རྫོགས་པར་གསལ་བར་བསྟན་པ་ཡིན་ནོ། །དེ་ལྟར་འདི་དང་ཡང་འདི་ཞེས་བྱ་བ་ནི་ཚིག་ཕྲད་ཀྱིས་
ཤིན་ཏུ་ཁྱད་པར་བྱས་ནས་གསུམ་གྱི་དབང་དུ་བྱས་པ་ཁྱད་པར་དུ་བསྟན་ཏོ། །དེ་ལ་དད་པས་བ་སྤུ་ལངས་པའི་སྒྲ་ནི་ཅ་ཅོའི་སྒྲ་ཡིན་ནོ། །ལྷག་མའི་ཚིག་གི་དོན་ནི་འགྲེལ་པས་བཤད་དོ། །ཡང་ཅིའི་ཕྱིར་ལམ་ཤེས་པ་ཉིད་བཞིན་དུ་ཐམས་ཅད་ཤེས་པ་ལ་བསྒོམ་པའི་ལམ་ལོགས་ཤིག་
ན་མེད་ཅེ་ན། ཉན་ཐོས་དང། རང་སངས་རྒྱས་ཀྱི་ཕུན་སུམ་ཚོགས་པ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཤེས་པ་དང། མཐོང་བས་བལྟས་ནས་འདའ་བར་བྱའོ་ཞེས་གསུངས་པས་འདི་ནི་གཞན་གཙོ་བོར་བྱེད་པ་ཡིན་ནོ་ཞེས་བྱ་བར་དགོངས་སོ། །གང་གཞན་ཡང་རྒྱུན་ཏུ་ཞུགས་པ་
ནས་རང་སངས་རྒྱས་ཀྱི་བར་གྱི་ཤེས་པ་དང། སྤོང་བ་གང་ཡིན་པ་དེ་ནི་མི་སྐྱེ་བའི་ཆོས་ལ་བཟོད་པའི་བྱང་ཆུབ་སེམས་དཔའི་བཟོད་པ་ཡིན་ནོ་ཞེས་གསུངས་སོ། །དེ་ལྟར་ན་གང། ཇི་སྐད་དུ། ཤེས་པས་སྲིད་ལ་མི་གནས་ཤིང། །སྙིང་རྗེས་ཞི་ལ་མི་གནས་དང། །ཐབས་
མ་ཡིན་པས་རིང་བ་དང། །ཐབས་ཀྱིས་རིང་བ་མ་ཡིན་ཉིད། །མི་མཐུན་གཉེན་པོའི་ཕྱོགས་དག་དང། སྦྱོར་དང་དེ་ཡི་མཉམ་ཉིད་དང། །མཐོང་བའི་ལམ་ཞེས་བྱ་བ་ཡིན། །དེ་འདྲར་ཀུན་ཤེས་ཉིད་དུ་འདོད། །ཅེས་བསྟན་པ་ཡིན་ནོ། །དེ་འདིར་རྗེས་སུ་འབྲེལ་
