སྨན་མས་གཏོང་ཞིང་བསལ་བ་དང་། མར་དང་འབྲུ་མར་ལ་སོགས་པའི་བཀྲུ་སྨན་དྲག་པོ་མས་གཏོང་བ་དང་། ཁོང་བརྟ་བའི་འཇམ་སྨན་མས་གཏོང་བ་དེ་ལྟ་བུས་ཕོ་བའི་མེ་དྲོད་ཀུན་སྐྱེད་དེ་བདེ་བར་འགྱུར་རོ། །དེ་ལྟར་བདེ་བར་གྱུར་ན་སེམས་ཅན་གྱི་བློ་དང་དབང་པོ་གསལ་
བ་དང་བཞིན་དང་ལུས་མདོག་གསལ་ཏེ་དྲི་མ་མེད་པ་དང་། ཚེ་རིང་ཞིང་རོ་ཙ་བར་འགྱུར་རོ། །དེ་ལ་འཕྲལ་གྱི་གློ་བུར་གྱི་ནད་ཅེས་བྱ་བ་འདོད་ཆགས་དང་ཞེ་སྡང་དང་འཇིགས་པ་ལ་སོགས་པ་དང་། གདོན་དང་དུག་དང་རླུང་དང་ནད་དང་མེ་དང་སྨས་པ་དང་། ཆག་གྲུགས་ལ་
སོགས་པ་འདི་ལྟ་བུ་གློ་བུར་དུ་འབྱུང་བའི་ནད་འདི་དག་ཀྱང་སེམས་ཀྱི་བློ་འཁྲུལ་པའི་ཉེས་པ་ཡིན་ཏེ། འཁྲུལ་པའི་ཉས་པ་དེ་ལྟ་བུ་བསལ་བ་དང་དབང་པོ་རྣམས་ཞི་བར་བྱ་ཞིང་། ཡུལ་གྱི་རྣམས་དང་དུས་དང་སྦྱར་ལ་བདག་གིས་ཚོད་ཤེས་པར་བྱ་སྟེ། སྐྱེས་བུ་དམ་
པའི་དཔྱོད་པའི་རྗེས་སུ་འབྲང་བར་བྱས་ན་འཕྲལ་གྱི་གློ་བུར་གྱི་ནད་མི་འབྱུང་ལ། རྒ་བ་དང་བཀྲེས་པ་དང་ལྟོགས་པའི་ནད་འདི་ཡང་མི་འབྱུང་ངོ་། །དེ་བས་ན་ཆོ་གའི་མདོ་འདི་ལྟར་བསྟན་པས་ནད་དུ་མ་གྱུར་པ་ནི་ནད་དུ་མི་འགྱུར་ཞིང་ནད་མི་འབྱུང་བར་བྱའོ། །ནད་
གྱུར་པ་ཡང་ཆོ་གའི་ཐབས་འདི་འདྲ་བར་བྱས་ན་ནད་ཞི་བར་འགྱུར་རོ། །དགུན་ཟླ་ར་བ་དང་། དགུན་ཟླ་ཐ་{𰻔ཆུངས་,ཆུང་}དང་། དགུན་ཟླ་འབྲིང་པོ་དང་། དཔྱིད་ཟླ་ར་བ་ཡར་གྱི་ངོ་དང་། ཟླ་བ་བཞིའི་ཚེ་ན་གྲང་ལྷགས་སུ་འགྱུར་ཏེ་བད་ཀན་འབྱུང་ཞིང་དགུན་ཐོག་ཐག་བད་ཀན་
བསགས་ནས། དཔྱིད་ཟླ་འབྲིང་པོ་དང་། དཔྱིད་ཟླ་ཐ་{𰻕ཆུངས་,ཆུང་}གཉིས་ལ་བད་ཀན་སྨིན་ནས་བད་ཀན་གྱི་ནད་དུ་འགྱུར་རོ། །དེའི་ཕྱིར་དཔྱིད་སྐྱུག་སྨན་དང་བཀྲུ་སྨན་གྱིས་བད་ཀན་གྱི་ནད་བཀྲུའོ། །ཡང་སོས་ཀའི་ཟླ་བ་དབྱར་ཟླ་ར་བ་དང་། དབྱར་ཟླ་འབྲིང་པོ་གཉིས་ལ་
རླུང་ནད་བསགས་པ་ནི་དབྱར་ཟླ་ཐ་{𰻖ཆུངས་,ཆུང་}དང་སྟོན་ཟླ་ར་བ་གཉིས་ལ་རླུང་ནད་དུ་འགྱུར་རོ། །དབྱར་ཟླ་གཉིས་ལ་མཁྲིས་པའི་ནད་བསགས་པ་ནི་སྟོན་ཟླ་ར་བ་དང་སྟོན་ཟླ་ཐ་{𰻗ཆུངས་,ཆུང་}གཉིས་ལ་མཁྲིས་པའི་ནད་ཚ་བ་ལྡན་པར་འགྱུར་ཏེ། རླུང་དང་བད་ཀན་དང་མཁྲིས་པའི་ནད་
