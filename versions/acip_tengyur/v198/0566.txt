ནད་སྲིན་ཅན་ལ། །ཉུངས་མར་དྲོན་པོ་བླུགས་པར་བྱ། །རྣ་བའི་འབྲས་ལ་ཁོང་འབྲས་ཀྱི། །གསོ་བ་དང་ནི་སྐྱུག་སྨན་གཏང། །སྣད་པའི་ཁོང་འབྲས་གསོ་བ་ནི། །མཁྲིས་གྱུར་རྣ་གཟེར་གསོ་བར་བྱ། །འབྲུམ་པ་ཐོར་བུ་སྣ་ལྟར་བྱ། །རྣ་བ་གས་པ་མ་སྨིན་ལ། །རྣ་བའི་ཁོང་འབྲས་

དག་རྣ་ཤལ་འཕེལ་ཞིང་རྒྱས་བྱེད་ཡིན། །རྣ་ཤལ་རབ་ཟད་བཅས་པས་ནི། །ལྷག་མ་སྦྱར་ལ་གསོ་ཞིང་བསྐྱེད། །དེ་བཞིན་རྒྱུད་འདྲ་འཚོ་ཙམ་བྱ། །འབྱོར་བ་ལ་ཡང་ཆོ་ག་འདི། །གནོད་གྱུར་བསིལ་སྨན་བསྐུ་བ་དང། །སྲིན་བུ་པད་པས་ཁྲག་དབྱུང་བྱ། །འཛམ་བུ་ཨ་མྲའི་ལོ་མ་

ཏིལ་མར་དང། །ཏ་ལའི་ལོ་མ་ཨརྐ་དང། །ཨ་ཤྭ་གནྡྷ་བྱེའུ་རུག་པ། །སོ་མ་ར་ཛ་འབྲས་བུ་དང། །རྒྱམ་ཚ་ལང་ཀ་ལིར་བཅས་བསྐོལ། །ཏིལ་མར་བསྐུ་བར་བྱ་བ་དང། །རྣ་སྨན་རྣོན་པོ་གཏང་བར་བྱ། །དབུག་ཉེས་པ་ལ་ཨ་སྨན་ཏ། །འཛམ་བུ་ཨ་མྲའི་ལོ་མ་ནི། །བསྐོལ་བའི་
