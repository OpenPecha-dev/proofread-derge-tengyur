༄༅། །བསྐོལ་བ་ཡི་ཁུ་བ་དང། །དེ་དང་འོ་མ་བསྐོལ་ཏེ་བསྒྲངས། །སྦྲང་རྩི་ཁར་ར་བཅས་པ་བླུད། །ཡང་ན་དེ་ཡིས་ཐུག་པ་བསྐོལ། །ལེགས་བསྐོལ་སྦྲང་རྩིར་བཅས་པ་སྦྱིན། །ཟུར་པ་དང་ནི་བ་ལེ་ཀ། །ཁ་ར་བཅའ་སྒ་བྲི་ཧ་ཏི།།བཏགས་བིལ་ཁུ་བ་རས་ཀྱིས་བཙགས་
མར་དང་བསྲེས་ཏེ་བླུད་པར་བྱ། །ཁ་ར་དང་ནི་ཛི་བ་ཀ་མུངྒ་ཤ་པརྞི་དང། །བྱི་ཚེར་བ་ནི་ལྡེ་གུར་སྦྱར། །མར་དང་འོ་མ་བརྒྱད་བསྐོལ། །དེ་བཏུང་བཟའ་དང་བལྡག་པ་དང། །སྦྱར་བས་མཁྲིས་པའི་ལུད་པ་སེལ། །འདི་དགེ་ཕྱེ་མའི་ལྡེ་གུ་འམ། །བསྐོལ་བའི་ཁུ་བ་བླུད་བར་བྱ། །
བད་ཀན་ལུད་ཅན་གྱིས་དང་པོར། །ཐང་ཤིང་བསྲེགས་ནས་གཟགས་པའི་སྣུམ། །བ་ཤ་ཀ་དང་ནས་ཚིག་འཁུས། །ཐལ་བའི་ཕྱེ་མ་བཏབ་པ་བཏང། །སྣུམ་བཅས་བཀྲུས་སྨན་གྱིས་སྟེང་དང། །འོག་དང་སྤྱི་བོར་ཅི་རིགས་པར། །ཉམས་སྟོབས་ལྡན་བཀྲུ་སྨན་རྣོ་གཏང། །ཐུག་པ་སོགས་ཀྱང་འདི་ལ་
བླུད། །ནས་དང་མུངྒ་རྒྱ་སྲན་ཟས། །རྩུབ་པ་དྲོན་ལེ་ཤམ་གྱིས་ཚ། །ཀ་ས་མརྡ་བརྡ་ཀ། །བྱ་ཀྲི་ཡ་བཀྵ་ར་དང། །པི་པི་ལིང་དང་བཅས་པ་སྦྱིན། །སྐམས་ཁུང་ནས་ཁུ་སྣུམ། །ཏིལ་དང་ཡུངས་ཀར་བིལ་བའི་སྣུམ། །སྦྱར་བའི་ཐུག་པ་སྦྱིན་པར་བྱ། །སྐོམ་ན་རྩ་བ་བཅུ་བསོལ་ཆུ། །ཆུ་ཚན་ཆང་
མང་སྦྲང་ཆུ་བླུད། །དོང་ག་མ་ནུའི་རྩ་བ་དང། །པ་ཏོལ་ཡི་རྩ་བར་བཅས། །ནུབ་གཅིག་ཆུར་སྤངས་སྦྲང་རྩི་བཏབ། །ཟས་དུས་གསུམ་མམ་ནང་པར་བླུད། །པི་པི་ལིང་དང་དེ་རྩ་བ། །བཅའ་སྒ་དང་ནི་བ་རུང། །རྨ་བྱ་ཁྱིམ་བྱའི་མཇུག་སྒྲོ་ནི། །སྣག་ཚར་བྱས་དང་ནས་ཚིག་ཐལ། །བི་ཤ་གླ་སྒང་པི་
པི་ལིང། །དུར་བྱིད་སྦྲང་རྩིར་སྦྱར་བ་སྟེ། །ཚིགས་བཅད་ཕྱེད་ཀྱིས་རྫོགས་ལྡེ་གུ། །གསུམ་པོ་བད་ཀན་ལུད་པ་སེལ། །སྦྲང་རྩི་ན་ལེ་ཤམ་སྦྱར་བལྡག། །ཨ་ཀ་རུ་སྦྲང་རྩི་ཉིད་ཀྱིས། །ཀྲྞྜ་ཀ་རི་བརྟ་ཀ། །བྲིང་ག་ར་ཛའི་ཁུ་བ་དང། །ཀ་ས་མརྡ་རྟ་སྤངས་བཙིར། །ཁུ་བ་བྱེ་རུ་གནས་པོའི་ཁུ། །སྦྲང་
རྩི་སོ་སོར་སྦྱར་བ་ལྡག། །ཐང་ཤིང་ཟུར་པར་སྣ་དང། །བྱི་ཚེར་ཀར་ཀ་ཏ་སྲིང་གི། །གླ་སྒང་བཅའ་སྒ་པི་པི་ལིང། །འབྲས་ཡོས་ཁ་ར་དཀར་པོ་དང། །སྐྱུ་རུ་ར་ཡི་པགས་པ་རྣམས། །སྦྲང་རྩི་ཏིལ་མར་སྦྱར་ལྡེ་གུ། །གསུམ་པོར་རླུང་འདྲེས་བད་ཀན་སེལ། །སེའུ་འབྲུ་སྲང་གཉིས་
