།ཤྭེ་ད་དུ་ཤ་དང། སིནྡུ་ཀ་དང། སིནྡུ་བ་རི་དང། ཨརྫ་ན་ས་ཏ་བོ་བར་ཁྱ་དང། སྟི་ར་དང། ཨརྦ་ས་ཏ་ཀ་དང། དེ་དག་ནི་སིནྡུ་བ་རའི་མིང་ངོ། །བ་ཛི་གན་དང། བ་ཛི་བ་ཀརྞ་ནི་དང། ཏུ་ར་ཀི་དང། བ་ར་ཧ་ཀརྞི་དང། བ་ར་དྭ་ཏ་དང། ཀཾ་བཧུ་ཀ་དང། ཨ་ཤ་བ་རོ་ཧ་ཀི་
དང༑ དེ་དག་ནི་ཨ་ཤ་གནྡྷིའི་མིང་ངོ། །མ་དུ་ཡནྟི་དང། རི་ར་ཛ་ནི་དང། མྲི་གན་དནྟི་དང། ཨ་ལི་བུ་ས་དང། མེ་དུ་ག་ལ་དང། གནྟ་བ་ཏི་དང། ཁ་ར་ས་ན་དང། དེ་དག་ནི་མ་དུ་ཡནྡའི་མིང་ངོ། །མརྐ་བ་དང། བཾ་ཀ་ཛ་ཏ་དང། བྲིག་ཧ་དང། ཀེ་ཤ་རཉྫ་དང། མ་ཧཱ་ནཱི་ལ་དང་
བྲིང་གར་ཛ་དང། སུརྱ་བརྡ་དང། རི་བི་པྲི་ཡ་དང། དེ་དག་ནི་བྲིང་ག་ར་ཛིའི་མིང་ངོ། །ཤི་རི་ཤི་ཀ་དང། དིནྡྷི་ནི་ཀ་དང། སོ་ཕ་ར་ལ་དང། རུ་ཕ་ལ་དང། ཀི་ནི་ཡ་ས་ཤཱི་ཤིང་དང། བ་རི་རི་ཤི་ཀ་དང། དེ་དག་ནི་ཤི་རི་ཤིའི་མིང་ངོ། །ཨ་པ་ར་ཙ་ཏ་དང། བའི་ཀྵུ་ཀནྟ་དང། ནཱི་ལ་པུ་ཤི་
དང༑ སཱ་ཏི་མཱ་དང། ཚ་ཏི་ཀ་དང། དེ་དག་ནི་བའུ་ཤཱུ་ཀྲནྠའི་མིང་ངོ། །ཨ་ག་སྟི་གོ་དང། མུ་ནིན་ན་མ་དང། མུན་དུ་ཤི་ཀྲུ་དང། དེ་དག་ནི་ཨ་གསྟའི་མིང་ངོ། །བདྶ་ཏ་ནི་དང། པུ་ཏ་ཤི་ནི་དང། དྷ་མ་ནྱ་ཚ་དང། ན་ཕ་ལ་དང། བཱ་ཌ་ཀར་བས་དང། ཨགྣི་ཀ་དང། དེ་དག་ནི་ཙ་གྲང་
ཀའི་མིང་ངོ། །མནྡཱ་ར་དང། པ་རི་བྷ་ཏྲ་དང། ཀཎྚི་ཀི་ཀ་ཤུ་ཀ་དང། བ་ར་ན་དང། སྟེ་རི་ནི་དང། ཀཎྜ་ཀ་ཀརྞ་ཛ་དང། ཏེ་རི་བིཉྫ་ཀ་དང། དེ་དག་ནི་མནྡ་ལའི་མིང་ངོ། །ཀ་ལི་ཏ་རུ་དང། ཀ་ན་ཀ་པུཥྦ་དང། དེ་དག་ནི་ཀུང་ཀའི་མིང་ངོ། །ཨ་ར་རི་ཀ་དང། ཧཏྟ་ལི་དང། ཁ་དར་བལླི་
དང༑ དཧི་ན་ཀི་དང། སི་ཏ་ཀ་དང། མོ་ཧ་ནི་དང། བ་ཏི་པཏྟྲི་ཀ་དང། དེ་དག་ནི་ད་རི་ཀའི་མིང་ངོ། །ཤོན་བ་ཏི་དང། ཨཻ་ར་ཐ་ཏི་དང། ཤ་མ་དང། ཁ་ཊ་དམ་བླ་མི་ཀ་དང། རི་ཏི་བང་གུ་དང། མཧཱ་ཤ་མ་དང། བྲིག་ཏ་ད་རུ་དང། དེ་དག་ནི་ཨེ་རེ་བ་ཏིའི་མིང་ངོ།། །།ནི་ནི་
ཀ་དང། བྲ་ཧྨ་སོམ་ད་དང། ཨགྣི་ཡ་ག་དང། ཨ་ར་མ་ཤི་ཏ་ལ་དང། དྲེ་བ་གནྡྷ་དང། ཀུརྡུ་ཏ་མརྟ་ཀ་དང། དེ་དག་ནི་བྲ་ཧྨ་སོ་མའི་མིང་ངོ། །བ་ཏྲི་ཀ་དང། བགྟྲི་ཀ་དང། ན་བ་ནི་ཏཾ་དང། དུཏྟུ་བ་སོ་དང། དུཏྟི་ལི་ཀ་དང། པུ་ཌུ་དངྐ་དང། ཚིག་ཏྲ་བི་ཏྲི་ཀ་དང། དེ་དག་ནི་དུདྡུ་ལི་ཀའི་
