དང་། གདོན་ནད་ཀྱི་ཆོ་ག་དང་། སྨྱོ་བ་དང་། བརྗེད་བྱེད་ཀྱི་ནད་དང་། མིག་ནད་དང་། སྣ་ནད་དང་། ཁ་ནད་དང་། རྣ་ནད་དང་། གླད་ནད་ལ་སོགས་པའི་དཔྱད་དང་། རྨ་དང་། མཚོན་གྱིས་ཕོག་པ་སྣ་ཚོགས་དང་། རྐང་ལག་དང་འབུར་བ་ལ་
སོགས་པ་གསོ་བའི་དཔྱད་རྣམ་པ་བཞི་བཅུ་ཡོད་དོ། །དེ་ལ་ཚེ་རིང་པར་འདོད་ན་ཉིན་རེ་ཞིང་རྟག་ཏུ་ཇི་ལྟར་བྱ་བ་དང་། ལོ་གཅིག་ཅིང་དུས་ཚིགས་ཐ་དད་པ་དྲུག་ཡོད་པ་ལ་ཁ་ཟས་ལ་སོགས་པའི་ཆོ་ག་དང་། དཔྱད་ཇི་ལྟར་བྱ་བ་དང་། རྟག་པར་བདེ་ཞིང་ནད་མེད་པར་བྱ་བ་དང་།
སྐོམ་ཅི་དང་ཅི་བཏུང་བ་དང་། ཁ་ཟས་སོ་སོའི་བྱེ་བྲག་ཤེས་པར་བྱ་བ་དང་། ཁ་ཟས་དང་། སྐོམ་ལ་སོགས་པའི་ནང་དུ་དུག་བཏབ་པ་ཐམས་ཅད་ལ་ཚོར་ཞིང་ངོས་ཟིན་པར་བྱ་བའི་ཐབས་བསྟན་ཏེ། ཟས་བསྡམ་པར་བྱ་བ་བཤད་པ་དང་། ཟས་རན་པར་བཟའ་བའི་ཚོད་
གཟུང་བ་བསྟན་པ་དང་། སྨན་སྣ་སོ་སོའི་མཐུ་དང་ཡོན་ཏན་དང་ཞུ་བའི་རྗེས་བསྟན་པ་དང་། སྨན་གྱི་ཁུ་བ་མདར་བ་དང་། སྐྱུར་བ་དང་། ཁ་བ་ལ་སོགས་པའི་བྱེ་བྲག་རོ་ལ་གནས་པ་བསྟན་པ་དང་། བད་ཀན་དང་རླུང་དང་མཁྲིས་པ་ལ་སོགས་པའི་ནད་ཤེས་པར་བྱ་བ་
དང་། ནད་དེ་དག་གི་སོ་སོའི་བྱེ་བྲག་དབྱེ་བ་དང་། ནད་སོ་སོའི་གསོ་བའི་ཐབས་གཉིས་པ་བསྟན་པ་དང་། རིད་པ་དང་རྐེ་ཞིང་ཉམ་ཆུང་བ་བརྟ་བའི་ཐབས་དང་། སྣུམ་བག་དཔྱད་ཀྱི་ཆོ་ག་བྱ་བ་དང་། སྐྱུག་སྨན་དང་བཀྲུ་སྨན་ལ་སོགས་པའི་ནད་སྦྱང་བར་བྱ་བ་དང་། དུགས་དང་
ལུམས་བྱ་བ་དང་། མས་བཀྲུ་ཞིང་བསལ་བ་ལ་སོགས་པས་ལུས་བརྟན་པར་བྱ་བ་དང་། སྣ་ནས་གཏང་བ་དང་། སྨན་གྱི་དུད་པ་བཏང་ཞིང་བརྔུབ་པ་དང་། བད་ཀན་རྒྱས་སམ་སོ་འགུལ་ལམ་ན་བ་ལྟ་བུ་ལ་ཏིལ་མར་རམ་སྨན་གྱི་ཁུ་བ་ཁར་བླུགས་ཤིང་དོར་བ་དང་། མིག་སྨན་
བླུགས་ཤིང་མིག་ནད་བཀྲུ་བ་དང་། མིག་ནད་བཀྲུས་ནས་མིག་གསོ་བའི་སྨན་གཏང་བ་དང་། མདའ་ལ་སོགས་པའི་མཚོན་གྱིས་སྨས་ཏེ་མདེའུ་ཁོང་དུ་ལུས་པ་དང་། ཚེར་མ་ཟུག་པའམ། གྲེ་བར་སྣད་པའམ། གྲ་མ་ཆག་པ་དབྱུང་བའི་ཐབས་དང་། གཙགས་བུ་དང་།
