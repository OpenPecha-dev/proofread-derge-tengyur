མང་བའི་དཔྱད་བྱའོ། །རྡུལ་མི་འབྱུང་ཞིང་ཟད་པའི་ནད་ལ་ནི་ལུས་གསིག་ཅིང་བསྐྱོད་དེ་རྡུལ་དབྱུང་བར་བྱ་བ་དང་། ལུས་ལ་མར་གྱིས་བསྐུ་བ་དང་། ལུས་བྱིབས་ཤིང་དུགས་བྱ་བ་དང་ཆང་བཏུང་བར་བྱའོ། །ཡང་ལུས་ཀྱི་སྲོག་ཟུངས་བདུན་པོ་ཀུན་
ལ་ལུས་ཀྱི་མེའི་དྲོད་རང་རང་གི་གནས་ན་རན་པར་འདུག་པ་དེ་ལྟ་བུ་མེའི་དྲོད་སྙོམས་པ་ནི། ལུས་ཀྱི་སྲོག་ཟུངས་ལ་བདེ་ཞིང་བརྟ་སྟེ་ཕན་པ་ཡིན་ནོ། །དེ་ལས་ལུས་ཀྱི་མེའི་དྲོད་ཧ་ཅང་འབར་ཆེས་ཏེ་ཆེ་བར་གྱུར་ན། ལུས་ཀྱི་སྲོག་ཟུངས་ཐམས་ཅད་ཚིག་ཅིང་ཟད་པར་བྱེད་དེ་གནོད་དོ། །དེ་
ལ་དང་པོར་ཟས་ཀྱི་བཅུད་དངས་མ་རྒྱས་ཤིང་རྟག་ན། དེ་བཞིན་སྲོག་ཟུངས་བདུན་པོ་གཅིག་ནས་གཅིག་ཏུ་ལེགས་པར་བརྟས་ཏེ་བདེ་བ་ཡིན་ནོ། །དེ་བཞིན་དུ་དང་པོར་ཟས་ཀྱི་བཅུད་དངས་མ་ཉམས་ཏེ། གནོད་པར་གྱུར་ན་སྲོག་ཟུངས་བདུན་པོ་གཅིག་ནས་གཅིག་ཏུ་གནོད་ཅིང་
ཟད་པར་བྱེད་དོ། །དེ་ལྟ་བས་ན་ཟས་ཀྱི་བཅུད་དངས་མར་གྱུར་ཏེ་ཉམས་པ་དེས་ནི། རླུང་ལས་མཁྲིས་པ་དང་བད་ཀན་གྱི་ནད་གང་ཡང་རུང་སྟེ་སློང་བར་བྱེད་དོ། །རླུང་དང་མཁྲིས་པ་དང་བད་ཀན་གྱི་ནད་གཡོས་པས་སྲོག་ཟུངས་བདུན་པོ་རྣམས་ལ་གནོད་པར་འགྱུར་རོ། །སྲོག་
ཟུངས་བདུན་པོ་མི་བདེ་བ་ལས་བཤང་གཅི་དྲི་མ་རྣམས་གཉིས་ལ་སོགས་པ་ལ་གནོད་པའི་ནད་དུ་འགྱུར་རོ། །ལུས་ཀྱི་དྲི་མའི་གནོད་པའི་ནད་དུ་འགྱུར་རོ། །ལུས་ཀྱི་དྲི་མའི་ནད་ཀྱང་ཐུར་དུ་འབྱུང་བའི་དྲི་མ་རྣམ་གཉིས་ཀྱང་མགོ་ལ། རྣ་བ་དང་། སྣ་དང་། མིག་དང་། ཁ་ནས་འབྱུང་
བའི་དྲི་མ་གཏོགས་པར་བདུན་ཏེ་འདི་དགུ་དང་ལུས་ལས་རྡུལ་འབྱུང་བ་རྣམས་དྲི་མ་སོ་སོ་རང་རང་གི་གནས་ནས་འབྱུང་བ་དེ་དག་ལ་ནད་དུ་འགྱུར་རོ། །དེ་ལ་སྲོག་ཟུངས་བདུན་པོ་ཁུ་ཆུ་ལ་སོགས་པ་ཡན་ཆད་ཀྱི་ནང་ན་ཟས་ཟོས་པ་ལས་ནང་ན་བཅུད་དངས་མར་གྱུར་པ་དེ་ནི་དེ་ལ་སྲོག་
ཟུངས་བདུན་པོ་རྣམས་ཀྱི་ནང་ན་མཆོག་ཡིན་ནོ། །མདངས་ཞེས་བྱ་བ་སྙིང་གི་ནང་ན་ཆུའི་ཐིགས་པ་ལྟར་དངས་ཤིང་གསལ་བར་འདུག་མོད་ཀྱི་ལུས་ལ་ཁྱབ་པ་དང་། ལུས་ལ་བརྟན་པར་གནས་པ་ནི་དེ་ཡིན་ནོ། །དེ་ཇི་ལྟ་ཞེ་ན། མདངས་ཞེས་བྱ་བ་སྙིང་གི་ནང་ན་ཡོད་པའི་ཐིགས་
