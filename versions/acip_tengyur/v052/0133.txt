༄༅། །དོ་ཧ་ཀོ་ཥ་ཧྲྀ་ད་ཡ་ཨརྠ་གའི་ཏི་ཊཱི་ཀཱ་ནཱ་མ། བོད་སྐད་དུ། དོ་ཧ་མཛོད་ཀྱི་སྙིང་པོ་དོན་གྱི་གླུའི་འགྲེལ་པ་ཞེས་བྱ་བ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །སྤྲུལ་པའི་བྱང་ཆུབ་སེམས་དཔའ་རིན་ཆེན་བློ་གྲོས་དང། །དཔལ་ལྡན་རྣལ་འབྱོར་
དབང་ཕྱུག་ཆེན་པོ་མདའ་བསྣུན་གྱི། །ཇི་བཞིན་དགོངས་པ་ལོན་པ་དཔལ་ལྡན་ས་ར་ཧའི། །ཞབས་ཀྱི་རྡུལ་ལ་སྒོ་གསུམ་དང་བས་ཕྱག་འཚལ་འདུད། །སྙིང་པོའི་དོན་ནི་བློ་ཡི་ཡུལ་དུ་མ་གྱུར་པས། །བདག་འདྲས་ཚིག་ཏུ་བརྗོད་ཀྱང་དོན་དང་ཤིན་དུ་རིང། །ཚིག་དང་དོན་གྱི་
འབྲེལ་པ་མེད་དེ་འོ་ན་ཡང། །དྲན་རིག་སྒྲོན་མེ་འདི་ལ་བརྟེན་པའི་དོན་འགྲེལ་འདི། །མུ་སྟེགས་ལ་སོགས་གཞན་གྱི་གཞུང་ལ་མི་སྨོད་དོ། །རྟོགས་ན་དེ་དག་སྙིང་པོའི་དོན་ལས་མ་གཡོས་པས། །འོན་ཏེ་ལོག་པར་རྟོགས་པས་གྲོལ་བར་མི་འགྱུར་ཏེ། །དོན་ལ་དགོས་པ་
མེད་པས་ཞིབ་ཏུ་མ་གསུངས་སོ། །གཞན་གྱི་གཞུང་གི་རྩ་བའི་རྗེས་ལ་འགྲེལ་པས་བཤད། །བདག་འདྲའི་ལུགས་ཀྱིས་བཤད་པའི་རྗེས་ལ་རྩ་བ་བྲི། །དེ་ཡང་སྡེ་སྣོད་རྣམས་ཀྱི་ལུང་ཚིག་མི་འདྲེན་ལ། །སྔགས་དང་རྒྱུད་རྣམས་ལུང་ཚིག་བྲིས་པས་རྫོགས་པ་མེད། །
དཔལ་ལྡན་རི་ཁྲོད་ཞབས་ཀྱི་ཐུགས་དགོངས་འབའ་ཞིག་ལ། །བརྟེན་ནས་གསུང་གི་བདུད་རྩིའི་ཐིག་པའི་བརྗེད་བྱང་འདི། །བདག་དང་བདག་འདྲ་མོས་པ་རྣམས་ཀྱི་དོན་དུ་ནི། །དོན་གྱི་གདམས་ངག་འབའ་ཞིག་བསྡུས་ནས་བྲི་བར་བྱ།། །།དེ་ལ་རེ་ཞིག་དོ་ཧ་ཀོ་ཥ་གའི་ཏི་འདི་ལ་
དོན་རྣམ་པ་གཉིས་སོ། །སྙིང་པོའི་དོན་མ་རྟོགས་པ་ལ་བརྒལ་བ་དང། རྟོགས་པ་གསལ་བར་བྱེད་པའོ། །མ་རྟོགས་པ་ལ་བརྒལ་བ་ནི། ཕྱི་རོལ་མུ་སྟེགས་པ་དང། ནང་པ་སངས་རྒྱས་པའོ། །དེ་ཡང་མུ་སྟེགས་བདག་ཏུ་ལྟ་བ་ལ། །རྟག་པ་དང་བདེ་བ་དང། ཁྱབ་པ་
དང༑ གཅིག་པུ་དང་བཞི་ཁས་ལན་ཏེ། མི་གསལ་བའི་མཚན་མ་དང། གསལ་བའི་མཚན་མ་ཞི་ལ་བཞི་པ་ལ་བཞེད་དེ། དེ་ལས་འགྲོ་བའི་དོན་དུ་རིགས་རྣམ་པ་བཞིར་གཟིགས་ནས། སྟོན་པ་རྣམ་པ་བཞིར་སྤྲུལ་པ། ཚངས་པ་དང། དབང་ཕྱུག་ཆེན་པོ་དང་
