ཞིང་བྲལ་བར་འགྱུར་རོ། །རྣམ་པར་འགྱུར་བའི་ཚོགས་ཐམས་ཅད་ཀྱང་སྐྱེ་བའི་རིམ་པ་ལས་བཟློག་པའི་སྒོ་ནས་དེ་ཉིད་དུ་རབ་ཏུ་ཞི་བའི་ཕྱིར་མི་གསལ་བའི་ངོ་བོར་གྱུར་པ་དེའི་ཚེ་སྐྱེས་བུ་ཉག་གཅིག་གནས་ཏེ་དེའི་ཕྱིར་གྲོལ་བ་ཞེས་བྱའོ། །དེ་ནི་རྣམ་པར་འགྱུར་བ་ཞར་ཟིན་ཀྱང་ཞ་བར་མི་འགྱུར་
བ་ཉིད་ཀྱིས་རྟག་ཏུ་ཡན་གར་བའི་ངོ་བོར་གནས་པའི་ཕྱིར་རྟག་པ་ཞེས་བྱ་བའོ། །རང་བཞིན་ཉིད་བྱེད་པ་པོ་ཡིན་ཞིང་རྣམ་པར་འགྱུར་བ་རྣམས་ལས་ཀྱང་ཁ་ཅིག་བྱེད་པ་པོ་ཡིན་པའི་ཕྱིར་ལ། བདག་ནི་བྱ་བ་ལ་སྙོམ་ལས་ཅན་ཉིད་དུ་གྱུར་པའི་ཕྱིར་ན་བྱེད་པ་པོ་མ་ཡིན་པ་ཉིད་ཅེས་བརྗོད་དོ། །ཉེ་བར་
ལོངས་སྤྱོད་པ་པོ་ཉིད་ནི་ཇི་སྐད་སྨྲས་པའི་ཚུལ་གྱིས་བསྟན་ཟིན་པ་ཉིད་དོ། །རྡུལ་དང་མུན་པ་དང་སྙིང་སྟོབས་ཀྱི་རང་བཞིན་མེད་པའི་ཕྱིར་ན་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཁྱབ་པར་བྱེད་པ་ཉིད་མ་ཡིན་པའི་ཕྱིར་ནི་བྱ་བ་མེད་པ་ཉིད་དེ། དེ་ལྟ་བུ་ནི་སྐྱེས་བུའི་ཁྱད་པར་དག་གོ། །དེ་ལ་རང་བཞིན་ཉིད་བྱེད་པ་
པོ་ཡིན་ཞིང་རྣམ་པར་འགྱུར་བ་རྣམས་ལས་ཀྱང་ཁ་ཅིག་བྱེད་པ་པོ་ཡིན་པའི་ཕྱིར་ལ་ཞེས་གང་སྨྲས་ན། རྣམ་པར་འགྱུར་བ་གང་དག་གཅིག་ནི་བྱེད་པ་པོ་ཉིད་ཡིན་ལ་གང་དག་གཅིག་ནི་བྱེད་པ་པོ་ཉིད་མ་ཡིན་ཞེ་ན། འདི་ལ་བྱེ་བྲག་ཏུ་བཤད་པར་བྱ་བའི་ཕྱིར་ཅུང་ཟད་ཅིག་བརྗོད་པར་བྱ་སྟེ། དེ་ལ་རྡུལ་
དང་མུན་པ་དང་སྙིང་སྟོབས་གསུམ་ནི་ཡོན་ཏན་གསུམ་མོ། །དེ་ལ་རྡུལ་ནི་གཡོ་བ་དང་འཇུག་པའི་བདག་ཉིད་ཅན་ནོ། །མུན་པ་ནི་ལྕི་བ་དང་གཡོགས་པའི་བདག་ཉིད་ཅན་ནོ། །སྙིང་སྟོབས་ནི་ཡང་བ་དང་རབ་ཏུ་གསལ་བའི་བདག་ཉིད་ཅན་ནོ། །བདེ་བ་དང་སྡུག་བསྔལ་བ་དང་གཏི་
མུག་ཅེས་བྱ་བ་ནི་འདི་དག་ཁོ་ནའི་རྣམ་གྲངས་སོ། །འདི་དག་ཆ་མཉམ་པའི་གནས་སྐབས་ནི་གཙོ་བོ་སྟེ། འདིར་ཡོན་ཏན་དག་གཙོ་བོར་གྱུར་ཅིང་རབ་ཏུ་ཞི་བར་གྱུར་པའི་ཕྱིར་རོ། །དེ་དག་རྣམ་པར་མི་འགྱུར་བའི་གནས་སྐབས་ནི་རང་བཞིན་ཅན་ནོ། །རང་བཞིན་ལས་ཆེན་པོ་སྟེ་ཆེན་པོ་ཞེས་
