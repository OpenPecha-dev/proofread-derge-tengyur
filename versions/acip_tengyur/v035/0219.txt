གཞན་དག་ཀྱང་ནི་སྒྲོལ་བར་བྱེད། །རང་དོན་ཙམ་ལ་གཞོལ་བས་ནི། །བྱོལ་སོང་རྣམས་ཀྱང་ཉོན་མོངས་བྱེད། །འགྲོ་བའི་དོན་ནི་བྱེད་པ་པོ། །འགྲོ་བ་དེ་གཏོར་གངྒཱ་ན། །གྲང་དང་རླུང་སོགས་སྡུག་བསྔལ་རྣམས། །རང་དོན་ཞུགས་ལས་ཀྱང་བཟོད་ན། །འགྲོ་
བའི་དོན་ལ་ཞུགས་རྣམས་ལྟ། །དེ་མི་བཟོད་པ་ཇི་ལྟ་བུ། །དམྱལ་བ་ལ་སོགས་སྡུག་བསྔལ་ཡང། །སྙིང་རྗེ་ཅན་གྱིས་བཟོད་བྱས་ན། །གྲང་དང་རླུང་སོགས་སྡུག་བསྔལ་དག་ད་ལྟར་གྱི་ལ་བསམ་ཅི་དགོས། །དཀའ་ཐུབ་བརྟག་པ་མི་བྱ་ཞིང། །སྨྱུང་བ་ཡང་ནི་བྱ་བ་
མེད། །འདིར་ནི་ཁྲུས་དང་གཙང་སྦྲ་མེད། །གྲོང་པའི་ཆོས་ནི་སྤང་བར་བྱ། །སེན་མོ་རྐང་དང་རུས་པ་སོ། །ཕ་ཡི་ཁུ་བའི་རྒྱུ་ལས་སྐྱེས། །ཤ་དང་ཁྲག་དང་སྐྲ་ལ་སོགས། །མ་ཡི་ཁྲག་ལས་ཡང་དག་བྱུང། །འདི་ལྟར་མི་གཙང་ལས་བྱུང་ཞིང། །མི་གཙང་བས་གང་རིལ་པོ་
གང༑ །དེ་ལྟ་བུ་ཡི་ལུས་འདི་ཉིད། །གངྒཱར་བཀྲུས་ཀྱང་ཇི་ལྟར་འདག་མི་གཙང་ཕུར་མ་ཆུ་དག་གིས། །ཡང་ཡང་བཀྲུས་ཀྱང་དག་མི་འགྱུར། །དེ་བཞིན་མི་གཙང་བས་གང་བའི། །ཕུར་མ་འདི་ཡང་འདག་མི་འགྱུར། །གངྒཱའི་ཀླུང་ལ་ཁྱི་དག་གིས། །རྐྱལ་ཞིང་འཕྱོ་ཡང་དག་
མི་འགྱུར། །དེ་ཕྱིར་ཆོས་འདོད་སྐྱེས་བུ་དག་འབབ་སྟེགས་ཁྲུས་བྱེད་འབྲས་བུ་མེད། །ཁྲུས་ཀྱིས་གལ་ཏེ་དག་འགྱུར་ན། །ཉ་པ་རྣམས་ཀྱང་དོན་བྱས་འགྱུར། །ཉིན་དང་མཚན་དུ་གནས་པ་ཡི། །ཉ་ལ་སོགས་པ་སྨོས་ཅི་དགོས། །ཁྲུས་ཀྱིས་སྡིག་པ་ཟད་པ་ཡང། །ངེས་
པར་ཡོད་པ་མ་ཡིན་ཏེ། །ཇི་ལྟར་འབབ་སྟེགས་བསྟེན་པ་ལ། །འདོད་ཆགས་ལ་སོགས་འཕེལ་བར་མཐོང། །འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་དང། །ང་རྒྱལ་ལྟ་བ་ཐེ་ཚོམ་རྣམས། །སྡིག་པའི་རྩ་བར་རབ་ཏུ་གྲགས། །འདི་དག་སྦྱང་བྱ་ཁྲུས་ཀྱིས་མིན། །བདག་དང་
