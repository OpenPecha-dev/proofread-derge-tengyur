ཨོཾ་ད་ཧ་ད་ཧ་མ་ཧཱ་བཧཻ་ར་བི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པ་ཙ་པ་ཙ་བཱ་ཡུ་བེ་གེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྷ་ཀྵ་བྷ་ཀྵ་བ་ས་རུ་དཧི་ར་ཨནྟྲ་མཱ་ལ་ཨ་བ་ལ་མབི་ནེ་སུ་ར་བྷཀྵཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་གྲྀཧྞ་གྲྀཧྞ་ས་ཕ་པཱ་ཏཱ་ལ་ག་ཏ་བཧུ་ཇཾ་ག་སརྦམྦཱ་ཏ་རྫ་ཡ་ཏ་རྫ་ཡ་ཤྱཱ་མ་དེ་བའི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱ་ཀཌྜྱ་ཨཱ་ཀཌྜྱ་སུ་བཱ་དྲེ་ཧཱུཾ་ཧཱུཾ་
ཕཊ༑ ཨོཾ་ཧྲིཾ་ཧྲིཾ་ཧ་ཡ་ཀར་ཎེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཇྞཽཾ་ཇྞཽཾ་ཁ་གཱ་ནེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵྨཱཾ་ཀྵྨཱཾ་ཙཀྲ་བེ་གེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱཾ་ཧཱཾ་ཁཎྜ་རོ་ཧེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱིཾ་ཧཱིཾ་ཤཽཎ་ཌི་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱུཾ་ཧཱུཾ་ཙཀྲ་བར་མི་ཎཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀི་ལི་ཀི་ལི་སུ་བའི་རེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སི་ལི་སི་ལི་མ་ཧཱ་བ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཙི་ལི་ཙི་ལི་ཙཀྲ་བར་ཏི་ནཱི་ཡེ་
ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དཧི་ལི་དཧི་ལི་མ་ཧཱ་བའི་རྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀཱ་ཀཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨུ་ལཱུ་ཀཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤཱ་ནཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སཱུ་ཀ་རཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཱ་དྷཱི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དའུ་ཏཱི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཾཥྚཱི་ཎཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐཱ་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཌཱ་ཀི་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
ལཱ་མ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁཎྜ་རོ་ཧེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་རཱུ་པི་ཎཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ན་མོ་བྷ་ག་བ་ཏེ་བའི་རེ་ཤཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་ཀལྤ་ཨགྣི་སནྣི་བྷཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཛ་ཊཱ་མུ་ཀུ་ཊོདྐ་ཊཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དཾ་ཥྚྲ་ཀ་རཱ་ལོ་གྲ་བྷཱི་ཥ་ཎ་མུ་ཁཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ས་ཏ་སྲ་པྲ་བཧུ་ཛ་བྷཱ་སུ་རཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པ་ར་ཤུ་
པཱ་ཤོ་དྱ་ཏ་ཤཱུ་ལ་ཁ་ཊཱཾ་གདྷཱ་རི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྱཱ་གྷྲཱ་ཛི་ན་ཨཾ་བ་ར་དྷཱ་རཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་དཧུ་མབྲན་དྷ་ཀཱ་ར་བ་པུ་ཥཱ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། མཆོད་པའི་སྔགས་སོ། །ཨོཾ་ན་མོ་བྷ་ག་བ་ཏི་བཛྲ་བཱ་རཱ་དའི་བཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱ་རྱ་ཡེ་པ་རཱ་ཛི་ཏ་ཏྲཻ་ལོ་ཀྱ་མཱ་ཏེ་མ་ཧཱ་བི་དཡེ་ཤྭ་རི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སརྦ་བྷཱ་ཏ་བྷ་ཡཱ་བ་ཧེ་མ་ཧཱ་བཛྲེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ས་ནེ་ཨ་ཛི་ཏེ་ཨ་པ་རཱ་ཛི་ཏེ་བ་ཤྱཾ་ཀ་རི་ནེ་ཏྲ་བྷ་མ་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤོ་ཥ་ཎི་རོ་ཥ་ཎི་ཀྲོ་དྷ་ཀ་རཱ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཏྲཱ་ས་ནི་མཱ་ར་ཎི་པྲ་བཧེ་ད་ནི་པ་རཱ་ཛ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བིཛ་ཡེ་ཛམྦྷ་ནི་སྟམྦྷ་ནི་མོ་ཧ་ནི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཱ་རཱ་ཧཱི་མ་ཧཱ་ཡོ་གི་ནཱི་ཀཱ་མེ་སྭ་རི་ཁ་གེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སརྦ་ཏྲ་ཀ་ཡི་ནཱ་མ། ལྷ་མོའི་མཆོད་པའི་སྔགས་
སོ༑ །ཨོཾ་བཛྲ་བཱ་རཱ་ཧཱི་པྲོ་ཏཾ་གེ་པྲོ་ཏཾ་གེ། །ཧ་ན་ཧ་ན་པྲཱ་ཎཱཾ་ཀིཾ་ཀི་ནི། ཁིཾ་ཁི་ནི། དཧུ་ན་དཧུ་ན། བཛྲ་ཧ་སྟེ་ཤོ་ཥ་ཡ། བཛྲ་ཁ་ཊཱཾ་ག་ཀ་པཱ་ལ་དྷཱ་རི་ཎི། ཨ་ཧཱ་བི་ཤི་ཏ་མཱཾ་སཾ་ཨ་ཤ་ནི་མ་ནུཥྱ་ནྟྲ་པྲ་པྲྀ་ཏ་ཏེ་ས་ནྣི་རྷྱཾ། ན་ར་ཤི་རོ་མཱ་ལཱ་གྲནྠི་ཏ་དྷཱ་རི་ཎི། སུམྦྷ་ནི་སུམྦྷ། ཧ་ན་ཧ་ན་པྲཱ་ཎཾ། སརྦ་པཱ་པཾ་ས་ཏཱ་ནཱཾ། སརྦ་
