༄༅། །དྲེཾ་དྷཾ་དྲེཾ་དྷཾ། གྲཾ་གྲཾ། ཧེ་ཧེ་ཧ་ཧ། བྷཱི་མེ། ས་ཧ་ས་ཧ། བའི་རེ་ཧཱ་ཧཱ་ཧཱ་ཧཱ་ཧོ་ཧུཾ་ཧཱུཾ། ཏྲཻ་ལོ་ཀྱ་བི་ནཱ་ཤ་ནི། ཤ་ཏ་ས་ཧ་སྲ་ཀོ་ཊི་ཏ་ཐཱ་ག་ཏ་པ་རི་ཝཱ་རི་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧ་རཱུ་པེ་ཁཿ་ག་ཛ་རཱུ་པེ་གཿ་ཏྲཻ་ལོ་ཀྱ་ད་ཏེ། མ་ཧཱ་ས་མུ་དྲ་མེ་ཁ་ལེ། གྲ་ས་གྲ་ས་ཧཱུཾ་ཕཊ། པའི་ར་
དྭཻ་ཏེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱ་ཧཱ་མ་ཧཱ་པ་ཤུ་མོ་ཧ་ནི་མ་ཧཱ་ཡོ་གའི་ཤྭ་རི་ཏྭཾ། ཌཱ་ཀི་ནཱི་ལོ་ཀཱ་ནཱཾ་བནྡྷ་ན། སྱ་པྲ་ཏྱ་ཡ་ཀ་རི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། བྷཱུ་ཏ་ཏྲ་ས་ནི་མ་ཧཱ་བའི་རེ་པ་ར་མ་སིདྷྱེ་བིདྱེ་ཤྭ་རི་ཕཊ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཧཱུཾ་ཕཊ་སཱ་ཧཱ། བཅོམ་ལྡན་འདས་མའི་རྩ་བའི་སྔགས་སོ།། །།ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པི་
ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙུ་མབི་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པ་རཱ་བིཨ་ཏཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་བཱ་ལི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ནུ་ཝ་རྟྟཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་མེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡོ་གའི་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷ་དྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པཱ་ལི་ནིའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
ཀཾ་ཀཱ་ལ་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ། ཨོཾ་བཛྲ་རཱ་ཛ་བྲྟྠི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁ་ཎྜ་རོ་ཧཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤྨ་ཤཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་དྲ་པའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུ་རུ་ཀུལླི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རུ་ད་ན་ཧཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ན་ཊཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོམ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་བཧཻ་ར་བའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁནྡཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ཊཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྶྫྲ་རུ་དྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་མེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་ཤྭ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁ་ཎྜ་
རོ་ཧེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པདྨ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རྟན་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཙ་ཎྜཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁ་ཎྜ་ཀ་པཱ་ལི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙཎྜཱ་ཀྵའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་ཀཾ་ཀ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་བྷཱ་ཝ་ཏི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཾ་ཀ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་མ་ཧཱ་ནཱ་སེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་ཀ་ཊ་དཾ་ཥྚཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བའི་ར་མ་ཧཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་རཱ་བའི་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁརྦ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་མི་ཏཱ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཾ་ལེ་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྲ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དྲཱུ་མ་ཙྪཱ་ཡེ་ཧུཾ་ཧཱུཾ་
