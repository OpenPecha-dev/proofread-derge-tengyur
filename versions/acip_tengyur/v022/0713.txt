ལྷ་དེ་རྣམས་གཡོན་དང་ཅིག་ཤོས་ཀྱི་ཕྱག་དག་གིས། ས་མ་ཡ་སྟྭཾ་ས་མ་ཡ་ཧོཞེས་པ་བརྗོད་ཅིང་གཡོན་དང་ཅིག་ཤོས་སུ་བསྐོར་ཞིང་བསྐོར་བས་མཆོད་བྱ་དང་མཆོད་བྱེད་དང་མཆོད་པ་རྣམས་དབྱེར་
མེད་པར་ལྷག་པར་མོས་པ་སྔོན་དུ་འགྲོ་བའི་གཏོར་མས་མཉེས་པར་བྱའོ། །ཨིནྡྲ། ཡ་མ། བ་རུ་ཎ། ཀུ་བེ་ར། ཨཱི་ཤ། ཨགྣི། རཱཀྵ་ས། པཱ་ཡུ་ཙནྡྲ། སཱུརྱ། བྲཧྨ། ཨཱ་དི། བེ་མ་ཙི་ཏྲ་པྲ་ཐི་བི་དེ་བ་ཏཱ། ས་པ་རི་ཝཱ་ར་ཨཱ་ཀརྵ་ཡ་ཛཿ་ཧཱུཾ་བཾ་ཧོསཾ་ཏྲྀ་པ་ཏཿ་སརྦ་དེ་བ་ཏཱ། སཱ་དྷ་ཀ་ཤཱིནྟཾ་
ཀུ་རུ་སརྦ་ཀེ་རྱཾ་ཀུ་རུ་སིདྡྷིར་བྷ་བ་ཏུ་ཡ་ཐཱ་སུ་ཁཾ། ཨོཾ་ཀ་ར་ཀ་ར། ཀུ་རུ་ཀུ་རུ། བནྡྷ་བནྡྷ། ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ། ཀྵོ་བྷ་ཡ་ཀྵོ་བྷ་ཡ། ཧྲཽཾ་ཧྲཽཾ་ཧྲཿ་ཧྲཿ་ཕེཾ་ཕེཾ། ཕཊ་ཕཊ། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། བྷཀྵ་བྷཀྵ། བ་ས་རུ་དཧི་ར་ཨནྟྲ་མཱ་ལཱ་བ་ལ་མབི་ནི་གྲྀཧྞ་གྲྀཧྞ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བཧུ་ཛཾ་ག་སརྦམྦཱ། ཏརྫ་
ཡ་ཏ་རྫ་ཨཱ་ཀཌྜྷྱ་ཧྲིཾ་ཧྲིཾ། ཛྞཽཾ་ཛྞཽཾ། ཧཾ་ཧཾ། ཧིཾ་ཧིཾ། ཧཱུཾ་ཧཱུཾ། ཀི་ལི་ཀི་ལི། སི་ལི་སི་ལི། ཧི་ལི་ཧི་ལི། དཧི་ལི་དཧི་ལི། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཱ་ཧཱ། ཞེས་པ་འདིས་ལན་ཅིག་བརྗོད་པས་བཅོམ་ལྡན་འདས་ཤེས་རབ་དང་བཅས་པ་དང། མཁའ་འགྲོ་མ་ལ་སོགས་པ་གཤིན་རྗེ་
འཇོམས་པའི་མཐར་ཐུག་པ་རྣམས་ཀྱིས་ཇི་ལྟར་རིགས་པར་བསྐོར་བར་བྱའོ། །ཨོཾ་བཛྲ་ཨ་རལླི་ཧོཿ་ཛཧཱུཾ་བཾ་ཧོབཛྲ་ཌཱ་ཀི་ནཱི་ས་མ་ཡ་སྟཾ་དྲྀ་ཤྱ་ཧོཞེས་པ་འདིས་ལན་གཅིག་དང་གཉིས་དང་གསུམ་དང་བཞི་དང་ལྔར་བརྗོད་པས་དབུལ་བར་བྱའོ། །དེ་ནས་དུར་ཁྲོད་བརྒྱད་དང་ཕྱོགས་
སྐྱོང་བ་ལ་སོགས་པ་རྣམས་ཀྱིས་སྔར་བཞིན་བསྐོར་བར་བྱའོ། །ཨོཾ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། སརྦ་ཡཀྵ་རཱཀྵ་ས་བྷཱུ་ཏ། པྲེ་ཏ། པི་ཤཱ་ཙ། ཨུན་མཾ་ད་ཨ་བསྨ་ར་ཌཱ་ཀ་ཌཱ་ཀི་ནྱ་ད་ཡ་ཨི་མཾ། བ་ལིཾ་གྲྀཧྞནྟུ་ས་མ་ཡཾ་རཀྵ་ནྟུ་སརྦ་སིདྡྷི་མམེ་པྲ་ཡ་ཙྪ་ནྟུ་ཡ་ཐཻ་བཾ་ཡ་ཐེ་ཥྚཾ་བྷུཉྫ་ཐ། པི་བ་ཐ། ཛི་གྷྲ་ཐ་མཱ་ཏི་ཀྲ་མ་
ཐ་མ་མ་སརྦ་ཀཱ་རྟ་ཡ་སཚུ་ཁཾ་པྲྀདྡྷ་ཡེ་ས་ཧཱ་ཡི་ཀ་བྷ་བནྟུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སཱ་ཧཱ་ཞེས་པས་དབུལ་བར་བྱའོ། །དེ་ནས་རྐང་པ་བརྒྱད་པའི་སྔགས་ཀྱིས་ཡང་དག་པར་བསྟོད་དེ་མངོན་པར་འདོད་པའི་དངོས་གྲུབ་གསོལ་བ་བཏབ་ནས་ཡི་གེ་བརྒྱ་བ་བཟླས་ལ་ཙྪོ་མའི་ལག་པས་བརྡ་བྱས་ལ། ཨོཾ་
