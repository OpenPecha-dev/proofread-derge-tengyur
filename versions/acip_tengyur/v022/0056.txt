ཨོཾ་བཛྲ་ཀ་ཧཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཱ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨོ་ར་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨོ་ར་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཧུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཧུ་ཀི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷ་ཎཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷཎྜི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀིཾ་ཀི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཀིཾ་ཀི་ཎི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷུ་རྒྷུ་རཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷུ་རྒྷུ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་ཀོ་ལོ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་ཀོ་ལི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཁཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤཾ་ཁ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷོ་ཥ་ཝ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཥ་ཝ་ཏི་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་པརྵ་འདི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པརཥ་ད་ཧུཾ་ཧཱུཾ་ཕཊ།། །།ཨོཾ་བཛྲ་ག་རུ་ཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་རུ་ཌ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཾ་སཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཾ་ས་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙི་འརི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙི་ཏྲི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཀཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏི་ཏཱ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏིཏྟཱ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཡཱུ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཡཱུ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་མྲ་ཙཱུ་ཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་མྲ་ཙཱུ་ཌི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གུ་ད་
བུ་ལི་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་ད་བུ་ལི་ཀ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀོ་མ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀོ་མ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ར་ཝ་རྟཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ར་ཝ་རྟི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲྀ་ཧ་ཏཀཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲྀ་ཧཏྐཱ་ཀི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་ཌཱི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་ག་ཌི་ནི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པི་ཉྩ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པི་ཉྩ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མནྟི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མནྟྲི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་ར་སཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་ར་སི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྲྀ་དྷཱྲི་
ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྲྀ་དྷྲཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུ་ལླཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུ་ལླཱུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཊ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཊ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀྵྛ་ཙ་ཊི་ཀི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀྵྛ་ཊ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཀྲ་བཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
