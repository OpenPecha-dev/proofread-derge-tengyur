༄༅། །ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ་མི་ཀ་ཊ་དཾཥྚྲ་ཎི་མ་ཧཱ་ན་སེཧཱ་ན་སེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵོ་བྷ་ཡ་ཀྵོ་བྷ་ཡ་སུ་ར་བཻ་རི་ཎ་བའི་རམ་ཏི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲཽཾ་ཧྲཽཾ་ཨ་མི་ཏཱ་བྷ་ཁརྦ་རཱ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲཿ་ཧྲཿ་བཛྲ་པྲ་བྷ་བ་ལཾ་ཀེ་ཤྭ་རཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། །ཨོཾ་ཕེཾ་ཕེཾ་བཛྲ་ཀཱ་ཡ་དྲུ་མཙྪ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཕཊ་ཕཊ་ཨོཾ་ཀུ་རི་ཀ་ཨཻ་ར་ཝ་ཏཱི་
ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ད་ཧ་ད་ཧ་བཛྲ་ཛཱ་ཊི་ལ་མ་ཧཱ་བཧཻ་ར་བའི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པ་ཙ་ཕ་ཙམ་ཧཱ་བའི་རེ་བཱ་ཡུ་བེ་གེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྷཀྵ་བྷཀྵ་རུ་དཧི་རཱནྟྲ་མ་ལཱ་བ་ལ་མབི་ནི་བཛྲ་ཧཱུཾ་ཀ་ར་སུར་བྷཀྵ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་གྲྀ་ཧྞ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བཧུ་ཛཾ་གཾ་སརྦ་ཏ་རྫྫ་ཡ་སུ་བྷ་དྲེ་ཤྱཱ་མ་དེ་ཝཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱ་
ཀཌྜྷྱ་ཨཱ་གཌྜྷྱ་བཛྲ་བྷ་དྲ་སུ་བྷ་དྲེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲིཾ་ཧྲིཾ་མ་ཧཱ་བཧཻ་ར་བ་ཧ་ཡ་ཀ་རྞེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཛྣཽཾ་ཛྣཽཾ་བི་རཱུ་པཱ་ཀྵ་ཁ་གཱ་ན་ནེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵྨཾ་ཀྵྨཾ་མ་ཧཱ་བ་ལ་ཙཀྲ་བེ་གེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱཾ་ཧཱཾ་ཧཱཾ་རཏྣ་བཛྲ་ཁཎྜ་རོ་ཧེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱིཾ་ཧཱིཾ་ཧ་ཡ་གྲཱི་བ་ཤཽཎྜཱི་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱུཾ་ཧཱུཾ་ཨཱ་ཀཱ་ཤ་གརྦྷ་ཙཀྲ་བརྨྨི་ཎི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
ཀི་ལི་ཀི་ལི་ཤྲཱི་ཧེ་རུ་ཀ་སུ་བའི་རི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སི་ལི་སི་ལི་པདྨ་ན་རྟྟེ་ཤྭ་ར་མ་ཧཱ་བ་ལེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧི་ལི་ཧི་ལི་བཻ་རོ་ཙ་ཀྲ་ཝ་རྟྟི་ནི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དཧི་ལི་དཧི་ལི་བཛྲ་སཏ་མ་ཧཱ་བའི་ར་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀཱ་ཀ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨུ་ལུ་ཀཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤྭ་ནཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤུ་ཀ་རཱ་སྱེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
ཡ་མ་ད་དྷཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དའུ་ཏཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཾ་ཥྚི་ཎི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐ་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཌཱ་ཀི་ནཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ལཱ་མཱ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོ་ཁཎྜ་རོ་ཧེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་རཱུ་པི་ཎཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བོ་དཧི་ཙིཏྟ་བཛྲ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ན་མོབྷ་ག་ཝ་ཏེ་ཝཱི་རེ་ཤཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་མ་ཧཱ་ཀ་ལྤཱ་གནི་སནྣི་བྷཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཛཱ་ཊ་མུ་ཀུ་ཊོཏྐ་ཊཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དཾཥྚཱ་ཀ་རཱ་ལོ་གྲ་བཧི་ཥ་ཎ་མུ་ཁཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ས་ཧ་སྲ་བཧུ་ཛ་བྷ་སུ་རཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པར་ཤུ་པཱ་ཤོ་དྱཱ་ཏ་ཤཱུ་ལ་ཁ་ཊཱཾ་ག་དྷ་རི་ཎེ་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྱཱ་གྷྲ་ཛི་ནཱཾ་བ་ར་དྷ་རཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་དཧུ་མྦྲཱ་ནྡྷ་ཀ་ར་ཝ་པུ་ཥཱ་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ། དེ་
དག་ནི་མཆོད་པའི་སྔགས་སོ། །ཁཎྜ་ཀ་པཱ་སོ་སེན་མོར། །སྐྲ་སྤུ་མ་ཧཱ་ཀངྐ་ལ། །པགས་དང་དྲི་མར་ཀངྐ་ལ། །ཤ་ཨལ་མཆེ་བ་རྣམ་པར་གཙིགས། །སུ་ར་བཻ་རི་རྒྱུས་པ་ལ། །འོད་དཔག་མེད་པ་རུས་པ་ལའོ། །མཆིན་པ་ལ་ནི་རྡོ་རྗེ་འོད། །སྙིང་ལ་རྡོ་རྗེ་སྐུ་གནས་ཏེ། །མིག་དག་ལ་ནི་
