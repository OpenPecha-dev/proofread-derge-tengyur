།རྒྱ་གར་སྐད་དུ། ཤིཀྵྱ་ནུ་གྲ་ཧ་བི་དཧི་ནཱ་མ། བོད་སྐད་དུ། སློབ་མ་རྗེས་སུ་གཟུང་བའི་ཆོ་ག་ཞེས་བྱ་བ། རྡོ་རྗེ་ཕག་མོ་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་གསང་བ་བཤད་པར་བྱ། །བླ་
མའི་བཀའ་ཡི་རྗེས་འབྲངས་ནས། །སྣ་ཚོགས་སྒྲུབ་ཐབས་རྟོགས་དོན་དུ། །སློབ་མ་འབེབས་པའི་ཆོ་ག་མཆོག་དང་པོ་རེ་ཞིག་སློབ་དཔོན་གྱིས། །སློབ་མ་རྗེས་སུ་གཟུང་དོན་དུ། །འརི་ཧེ་རུ་ཀ་ཞེས་པའི་ཡི་གེ་བཞིའི་དོན་དང་ལྡན་པའི་རྣལ་འབྱོར་པས། རང་གི་ཡིད་དང་རྗེས་སུ་མཐུན་པའི་གནས་
