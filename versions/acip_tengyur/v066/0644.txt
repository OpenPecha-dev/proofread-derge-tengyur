རོ་ལངས་ཀྱིས་བརྩལ་བ་དང། བཀྲེས་པ་དང་སྐོམ་པས་འཆི་བ་དག་ནི། དུས་མ་ཡིན་པར་འཆི་བ་ཆེན་པོ་དགུ་ཡིན་ལ། དུས་མ་ཡིན་པར་འཆི་བ་ཕྲ་མོ་དཔག་ཏུ་མེད་དོ་ཞེས་འཕགས་པ་དེ་བཞིན་གཤེགས་པ་བདུན་གྱི་སྔོན་གྱི་སྨོན་ལམ་གྱི་ཁྱད་པར་རྒྱས་པའི་མདོ་ལས་གསུངས་ཏེ་
དེ་ལས་འདིར་ནི་དྲུག་པ་དང་བརྒྱང་པ་གཅིག་ཏུ་བྱས་ཏེ་བརྒྱང་དུ་བཞག་སྟེ། གདུག་པ་ཅན་ཉིད་དུ་འདི་དག་རབ་ཏུ་འདྲ་བས་སོ། །ལྷག་མ་ནི་གོ་སླའོ། །ཡི་དྭགས་དང་འདྲེས་མ་འཇིགས་པར་བགྱི་ཞེས་བྱ་བ་ལ། ཡི་དྭགས་ནི་ཡི་དྭགས་ཀྱི་རིགས་སུམ་ཅུ་རྩ་དྲུག་ལ་བསམས་ལ་འདྲེ་ནི་གང་ཤ་
ཟ་རྣམས་ཏེ། ལྷག་མ་ནི་གོ་སླའོ། །དེ་ནས་འབྱུང་པོ་ཐམས་ཅད་ཀྱི་བདག་པོ་དབང་བྱེད་ཅེས་བྱ་བ་ལ་སོགས་པ་ནི། སྔགས་བསྒྲུབ་པ་དག་ལ་ནི་ལྷའི་བུའི་བདུད་དང། ལོག་འདྲེན་དག་ཕལ་ཆེར་བར་དུ་གཅོད་ལ། འདི་ནི་དེ་གཉི་གའི་བདག་པོར་གྱུར་པས་ན། འདིར་བསྲུང་བ་ལས་བྱ་བར་དམ་
བཅས་པའོ། །ཚིག་གི་དོན་ནི་གོ་སླའོ། །དཀྱིལ་འཁོར་བཞི་པའི་འགྲེལ་བའོ།། །།ད་ནི་དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པས་གཟའ་ཆེན་པོའི་དཀྱིལ་འཁོར་སྟོན་ཏེ། གཟའ་ཆེན་པོ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །སྲིན་པོ་བྱང་སྟོང་སྒྲ་གཅན་བྲི། །ཞེས་བྱ་བ་ནི། སྲིན་པོའི་མཚམས་
སམ། ཡང་ན་བྱང་གི་སྟོང་དེ་ནུབ་བྱང་གི་མཚམས་སུ་གཟའ་ལྷ་མིན་དབང་པོ་སྒྲ་གཅན་ཞེས་བྱ་བ་འདི་བྲི་སྟེ། དེ་ནི་ཕྱོགས་དེ་གཉིས་ལ་བརྟེན། །ཞེས་གྲག་གོ། །དཀྱིལ་འཁོར་གྱི་ཕྱི་རིམ་ལྷའི་སྣམ་བུ་ཉིད་ཁོར་ཡུག་ཏུ་དགོངས་ལ། དེར་སྨིན་དྲུག་ལ་སོགས་པའི་རྒྱུ་སྐད་ཐམས་ཅད་སྔོན་བཞིན་དུ་
བྲི་བར་བྱའོ། །ལྷག་མ་ནི་གོ་སླའོ། །ཨེ་བཛྲ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། སྔགས་དང་པོ་ནི་སློབ་མ་ལ་གསང་བ་བསྒོ་བའོ། །གཉིས་པ་ནི་ཡོལ་བའི་ནང་དུ་དྲང་བའོ། །གསུམ་པ་ནི་མེ་ཏོག་དོར་བའོ། །རྒྱུ་སྐར་དང་ཞེས་བྱ་བ་ནི་སྨིན་དྲུག་ལ་སོགས་པའོ། །སྐད་ཅིག་དང་ཞེས་བྱ་བ་ནི། དེ་ལ་བརྟེན་པའི་
ལྷ་སྟེ། ཡུད་ཙམ་དང་ཐང་ཅིག་ཀྱང་དེ་བཞིན་ནོ། །ཀར་ན་ཞེས་བྱ་བ་ནི་ཞག་གདགས་པའོ། །ཟླ་ཚིགས་ཞེས་བྱ་བ་ནི་ཟླ་བ་རིལ་པོའོ། །སྦྱོར་བ་ཞེས་བྱ་བ་ནི་དུས་མཚམས་སྦྱོར་བ་བཞི་པོ་གང་དུ་ཁམས་རབ་ཏུ་འཁྲུག་པ་སྐྱེ་བའོ། །ཁྱིམ་ཞེས་བྱ་བ་ནི་སྲང་ལ་སོགས་པའོ། །ཉི་ཚོད་ཅེས་བྱ་བ་ནི་ཉི་མ་ལྡོག་
