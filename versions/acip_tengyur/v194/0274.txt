པཱི་ཡཱུ་ཥཿ། ཧ་ནཱུ་ཥཿ། ཨངྒཱུ་ཥཿ། པྲཱྀ་ལས་ཀུ་ཥ་ནའོ། །པྲཱྀ་སྐྱོང་བ་ལའོ། །འདི་ལས་ཀུ་ཥ་ན་ཕ་རོལ། ན་དབྱངས་དོན་ཏོ། །ཀ་ཡིག་ཀི་ཏ་ལའོ་ཞེས་པའི་རྟགས་དོན། མཆུ་འབྱུང་ལས་ཨུཏའོ་ཞེས་པས་ཨུ་ཉིད་དོ། །རྀཀའི་ཨན་ལ་ར་ལ་ཞེས་པས་ར་ཉིད་དོ། །པུ་རུ་ཥཿ། ཀྲྀ། ཏྲྀ་དག་ལས་
བདུད་རྩི། ཨཱི་ཥནའོ། སྟག། གླང་པོ་སྲུང་བ། ཀྲཱྀ་རྣམ་པར་གཡེང་བ་ལའོ། །ཏྲཱྀ་རྐྱལ་པ་ལའོ། །འདི་དག་ལས་ཨཱིཥ་ན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་དབྱངས་དོན་ཏོ། །ཨི་ཀའི་ཨཏ་ཨེང་དང་ར་ཕ་རོལ་ཉིད་དོ། །ཀ་རཱི་ཥཿ། ཏ་རཱི་ཥཿ། ཤི་རཱི་ཥ་ལ་སོགས་པ་རྣམས་སྐྱེ་བུ། སོ། འདི་རྣམས་
ལས་ཨཱི་ཥན་ཕ་རོལ་ཤྲཱྀ་འཚེ་བ་ལའོ། །ཨི་ཉིད་དང་ཕ་རོལ། ཤི་རཱི་ཥཿ། པྲྀ་སྐྱོང་བ་ལའོ། །ཨུ་ཉིད་དང་ར་ཕ་རོལ། པུ་རཱི་ཥཿ། ཨ་མ་ནད་ལའོ། །འདི་ཨམྦ་རུ་བསྒྱུར། རེ་ཕའི་ཨ་ཡིག་བརྗོད་དོན། ཨམྦ་རཱི་ཥཿ། ལྕི་བ། རྒྱ་མཚོ། བྷྲཥྚཾ། རྀ་ཛི་འགྲོ་བ་ལའོ། །ཚིག་ཕྲད་ལས་ཨཏ་ཨེང་མེད་
དོ། །རྀ་ཛཱི་ཥཿ། དེ་བཞིན་དུ་གཞན་ཡང་བལྟ་བར་བྱའོ། །ཨ་བི་ལས་ཤིང་གི་རིགས། ཊི་ཥ་ཙའོ། །ཨ་བ་དང་རཀྵ་སྐྱོང་བ་བཤང་བ། ལའོ། །འདི་ལས་ཊི་ཥ་ཙ་ཕ་རོལ། ཊ་ཡིག་དཱི་བ་ཀྱི་དོན་ཏོ། །ཙ་ཡིག་དབྱངས་དོན། ཨ་བི་ཥཿ། ཊིཏ་ཉིད་ཀྱི་ཕྱིར་དཱིབའོ། །བ་ཡིག་དབྱངས་དོན། དཱི་ཡིག་ལས་
ཨི་ཉ། ཡིག་ཕྱུང་ནས་ང་ཡིག་ནི་དཱི་ཨཱུང་ཞེས་པའི་རྟགས་དོན་ཏོ། །ཨ་བི་ཥཱི། ན་དཱི། ཀི། ལྭི། ཥ་ལ་སོགས་པ་རྣམས་སོ། །འདི་རྣམས་ལས་ཊི་ཥཙ་ཕ་རོལ་ཏུ་ཚིག་ཕྲད་དུ་བྱས་པའོ། །ཀི་ལ་རྩེ་བ་ལའོ། །རུ་ཧ་རབ་ཏུ་རྒྱ་མཚོ། སྐྱེ་བ་ལའོ། །ཏ་ནུ་རྒྱས་པའོ། །ཏྲཱྀ་རྐྱལ་པ་ལའོ། །བྱ་ཐ་སྡུག་
བསྔལ་བ་ལའོ། །ན་ཉ་སྔོན་དུ། དེ་ལ་ཀི་ལ་ལ་བུཀ་ཨཱ་ག་མའོ། །ཀི་ལྭི་ཥཾ། ཚིག་ཕྲད་ལས་ཨོ་ཉིད་དོ། །ཧོ་རི་ཥ། སྔ་མ་བཞིན་དུ་དཱི་བའོ། །རོ་ཧི་ཥཱི། ཏ་ནི་ཥཿ། ཨིཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨ་ཉིད་དང་ར་ཕ་རོལ་ཉིད་དོ། །ཏ་རི་ཥཿ། ཨ་བྱ་ཐི་ཥཿ། ཨ་བྱ་ཐི་ཥི། དེ་བཞིན་དུ་གཞན་
ཡང་བལྟ་བར་བྱའོ། །བྲྀ། ཏྲཱྀ། བ་དི། ཧ་ནི། མཱ་ནི། ཀ་མི། ཨ་ཤི། ཀ་སི་སྡིག་པ། ལས་སོ། །བྲྀ་ཉ་སྒྲིབ་པ་ལའོ། {𰵐རི་དགས,རི་དྭགས}། ཏྲཱྀ་རྒྱལ་བ་ལའོ། །བ་ད་{𰵑རི་དགས་,རི་དྭགས་}མོ། རྒྱ་མཚོ། སྨྲ་བ་ལའོ། །ཧ་ན་འཚེ་བ་ལའོ། །མཱ་ན་མཆོད་པ་ལའོ། ཀ་ཆུ་ཀླུང་གི་བདག་པོ། ཉི་མ། མུ་མཚན་མོ། མཛེས་པའོ། །
