རྒྱུ་མེད་ན། རྒྱུ་མཚན་ཅན་ཡང་མེད་དོ་ཞེས་པས་ཊའི་ཏ། ཥིཏ་ལས་དཱིཥ་ཞེས་པས་དཱིཥའོ། །ང་ཡིག་དའི་ཨུང་ཞེས་པའི་རྟགས་དོན། ཥ་ཡིག་དའིས་དང་དའི་ཥ་ཐུན་མོང་བ་གཟུང་བའི་དོན་ཏེ། དའི་ཨཱམ་ཞེས་པས་ཨ་ཡིག་དབྱིའོ། །སུའིོ། །དའི་ཨས་ཀྱི་རིང་ཕོ་ལས་སོ་ཞེས་
པས་སུ་དབྱིའོ། །དྷཱ་ཏྲཱི།་ །ཡ་མ། དཱ་ཏྲཱི།་ ཟོར་བ།  ཨིཀའི་ཨཏ་ཨེང། ནེ་ཏྲཾ། བལ་དང་པོའི་ཨི་ཊ་ཞེས་པས་ཨིཏ་མི་འགྱུར་ཏེ། མང་པོ་ལས་སྤྱིར་བཏང་བའི་ཕྱིར་རོ། །པཏྟྲཾ།་ །ལོ་མ།  པཱ་ཏྲཾ།་ ལྷུང་བཟེད།  ཤཱསྟྲཾ།་ བསྟན་བཅོས།  དེ་ལ་སོགས་པ་གཞན་ཡང་ཤེས་པར་བྱའོ། །ཨུ་ཥི་སཱུ་མའུ་རྣམས་ལས་ཀིཏའོ། །ཨུ་ཥ་སྲེག་པ་
ལའོ༑ །ཥཱུ་དེད་པ་ལའོ། མའུང་འཆིང་བ་ལའོ། །མིག། འདི་རྣམས་ལས་ཥྚྲན་ཕ་རོལ་ཏུ་འགྱུར་ཏེ། ཀིཏ་བཞིན་ནོ། །དིས་ན་ཨཏ་ཨེང་དུ་མི་འགྱུར་རོ། སྟུའི་ཤྩུ་དང་ཥྚུ་དག་གི་དེ་དག་གོ་ཞེས་པས་ཏ་ཡིག་གི་ཊ་ཡིག་གོ། །ཨུཥྚྲ།་ རྔ་མོ།  སཱུ་ཏྲཾ།་ སྲད་བུའམ་སྣལ་མ།  མའུ་ཏྲཾ།་ གཅིན།།
ཨ་མི། ནཀྵི། ཀ་ཌི་རྣམས་ལས་ཨ་ཏྲནའོ། །ཨ་མནད་ལའོ། །ན་ཀྵ་འགྲོ་བ་ལའོ། །ཀ་ཌ་རྒྱགས་པ་ལའོ། །འདི་རྣམས་ལས་ཨ་ཏྲན་ཕ་རོལ་དང། ན་ཡིག་དབྱངས་དོན། ཨ་མ་ཏྲཾ།་ སྣོད།  ནཀྵ་ཏྲཾ།་ རྒྱུ་སྐར།  ཀ་ཌ་ཏྲཾ།་ ཆུང་མ།  ཀ་པི་རི་ཀ་ལ་སོགས་བསལ་ཉིད་དོ། །ཀ་ལ་ཏྲཾ།་ ཆུང་མ།།
བྲྀཉ་ལས་ཨ་ཏྲ་ཙའོ། །བྲྀཉ་སྒྲིབ་པ་ལའོ། །འདི་ལས་ཨ་ཏྲ་ཙ་ཕ་རོལ། ཙ་དབྱངས་དོན། །ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨཉིད་དོ། །ཌཱཔའོ། །བ་ར་ཏྲཱ།་ འབྲང་བ།  ཨ་མི། ཙི། མི་དི་ལས་ཏྲ་ཀའོ། །ཨ་མ་ནད་ལའོ། ཙི་སོགས་པ་ལའོ། །མི་དཱ་
སྣུམ་པ་ལའོ། །ཨཱ་ཡིག་ཨཱ་དིཏ་ལས་སོ་ཞེས་པའི་རྟགས་དོན། འདི་དག་ལས་ཏྲཀ་ཕ་རོལ། ཀ་ཡིག་ནི་ཀིཏྲ་ལས་སོ་ཞེས་པས་ཨཏ་ཨེང་དགག་པའི་དོན། མ་བ་དག་གིའང་མའི་ནའོ་ཞེས་པས་མའི་ནའོ། །ཨནྟཾ།་ སྒྱུ་མ།  ཙི་ཏྲཾ།་ བཀྲ་བ་འམ་རྨད་དུ་བྱུང་བ།  མི་དིའི་ང་ཡིག་གི་ཏ་ཡིག་གོ། །མིཏྟྲཾ།་ བཤེས་གཉེན་ནམ་མཛའ་བཤེས།  པཱུཉ་ལས་ཐུང་དུ་ཡང་ངོ། །པཱུཉ་དག་བྱེད་
ལའོ༑ །འདི་ལས་ཏྲཀ་ཕ་རོལ་ཏེ་ཐུང་དུ་ཡང་ངོ། །པུ་ཏྲཿ།་ བུ།  བ་ཧ་དང་ལ་ལ་སོགས་པ་རྣམས་ལས་ཨི་ཏྭ་ཨུ་ཏྲ་དག་གོ། །བ་ཧ་ལ་སོགས་དང་ལ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་གྲངས་ཇི་ལྟ་བ་བཞིན་དུ་ཨི་ཏྲ་དང་ཨུ་ཏྲ་ཞེས་པ་འདི་
