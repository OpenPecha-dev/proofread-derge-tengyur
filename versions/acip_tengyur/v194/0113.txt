ག་ཏིཿ། ག་ཏཱི། ག་ཏ་ཡཿ། ག་ཏིཾ། ག་ཏཱི། ག་ཏཱི། ག་ཏྱཱ། ག་ཏི་བྷྱཱཾ། ག་ཏི་བཧིཿ། ག་ཏྱཻ། ག་ཏ་ཡེ། ག་ཏི་བྷྱཱཾ། ག་ཏི་བྷྱ། ག་ཏྱཱ་ག་ཏེཿ། ག་ཏི་བྷྱཱཾ། ག་ཏི་བྷྱ། ག་ཏྱཱ། ག་ཏེཿ། ག་ཏྱོ། ག་ཏཱི་ནཱཾ། ག་ཏྱཱཾ། ག་ཏཽ། ག་ཏྱོ། ག་ཏི་ཥུ། ཧེ་ག་ཏེ། ཧེ་ག་ཏཱི། ཧེ་ག་ཏ་ཡཿ། དཧེ་ནུ་ཞེས་པ་

ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །དེ་ལ་ངིའི་ཨཱམའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཨེཙའི་ཨཡ་ཨཝ་ཨཱ་ཡ་ཨཱ་ཝའོ། །ང་སི་ངས་དག་ལའོ། །ཐུང་ངུ་ཨཱ་པ་ལས་ནུཊའོ། །ནཱམ་ལ་ཏིསྲྀ་ཙ་ཏ་སྲྀ་དག་མ་ཡིན་པ་ལའོ། །ཨེངའི་ཨཊ་འང་ངོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །ཀུན་དུ་བོད་པ་ལ་
ཨིཊ་ཨུཊའི་ཨེངའོ། །ཨེང་ཐུང་ངུ་ལས་ཀུན་དུ་བོད་པ་ལ་ཏ་མ་ཡིན་པའིོ། །རྣམ་པར་གཅད་པའོ། །དཧེ་ནུཿ། དཧེ་ནཱུ། དཧེ་ན་ཝཿ། དཧེ་ནུཾ། དཧེ་ནཱུ། དཧེ་ནཱུ། དཧེ་ནཱ་དཧེ་ནུ་བྷྱཱཾ། དཧེ་ནུ་བྷྱི། དཧེ་ནྭཻ། དཧེ་ན་ཝེ། དཧེ་ནུ་བྷྱཱཾ། དཧེ་ནུ་བྷྱ། དཧེ་ནཱ། དཧེ་ནོཿ། དཧེ་ནུ་བྷྱཱཾ། དཧེ་ནུ་བྷྱ། དཧེ་ནཱ། དཧེ་ནོཿ་
དཧེ་ནྭོཿ། དཧེ་ནཱུ་ནཱཾ། དཧེ་ནཱཾ། དཧེ་ནཽ། དཧེ་ནྭོཿ། དཧེ་ནུ་ཥུ། ཧེ་དཧེ་ནོ། ཧེ་དཧེ་ནཱུ། ཧེ་དཧེ་ན་ཝཿ། དེ་ལྟར། རཛྫུ། ར་ཛྫུ། རཛྫ་ཝཿ། ན་ནཾ་དྲྀ་ཞེས་པ་འདིའི། རྀཏ་ཨུ་ཤ་ནས་པུ་རུ་དཾ་ཤས་ཨ་ནེ་ཧས་རྣམས་ཀྱིའང་སུ་ལ་ཨ་ནངའོ་ཞེས་པ་ཨ་ནང་དུ་བསྒྱུར་རོ། །ཀུན་དུ་བོད་པ་མ་ཡིན་པའི་སུ་ལའོ། །སུའིོ། །སུཔའི་
རང་བཞིན་གྱིའོ། །རྀཏའི་ངི་སུཊ་ལ་ཨ་ཏའོ་ཞེས་པ་ཨ་ཉིད་དོ། །རྀཀའི་ཨཎ་ལས་ར་ལ་དག་གོ་ཞེས་པ་ཕ་རོལ་དུ་ར་ཉིད་དོ། །དང་པོ་དག་གི་ཨཙ་ལའོ། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ། །རྀཏའི་ཨུ་ཏའོ་ཞེས་པ་རི་ཡིག་གི་ཨུ་ཡིག་ཉིད་དོ། །རིགའི་ཨན་ལས་ར་ལ་དག་གོ། །ར་ལས་སའིོ་ཞེས་པ་ས་
