༄༅། །ཤྣུ་བྱིངས་བྷཱུ་རྣམས་ཀྱི་ཨཱི་ཨཱུ་དག་གི་ཨི་ཡ་ཨུཝ་དག་གོ་ཞེས་པ་ཨི་ཡ་རུ་བསྒྱུར་རོ། །དེ་ཨི་ཨ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །ཨེཙ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་ནུཊ་ཡང་ནའོ། །ཚིག་གཅིག་ལ་རཱཥ་ལས་
ནའི་ཎའོ། །ངིའི་ཨཱམ་དེ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །རྣམ་པར་གཅད་པའོ། །ཤྲཱི། ཤྲི་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཾ། ཤྲི་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཱ། ཤྲཱི་བྷྱཱཾ། ཤྲི་བཧིཿ། ཤྲི་ཡཻ། ཤྲི་ཡེ། ཤྲི་བྷྱཱཾ། ཤྲཱི་བྷྱ། ཤྲི་ཡཱ། ཤྲི་ཡཿ། ཤྲཱི་བྷྱཱཾ། ཤཱི་བྷྱ། ཤྲི་ཡཿ་
ཤྲི་ཡཱ། ཤྲི་ཡོཿ། ཤྲཱི་ནཱཾ། ཤྲི་ཡཱཾ། ཤྲི་ཡཱཾ། ཤྲི་ཡི། ཤྲི་ཡོཿ། ཤྲཱི་ཥུ། ཧེ་ཤྲཱི། ཧེ་ཤྲི་ཡཽ། ཧེ་ཤྲཱི་ཡཿ། དེ་ལྟར། སྟྲཱི། སྟྲི་ཡཽ། སྟྲི་ཡཿ། སུ་དྷཱི་ཞེས་པ་འདིའི་ཤྲཱིའི་སྒྲ་བཞིན་ནོ། །སུ་དྷཱི། སུདྷི་ཡཽ། སུ་དཧི་ཡཿ། སུ་དཧི་ཡཾ། སུ་དཧི་ཡཽ། སུ་དཧི་ཡཿ། བྷྲཱུ་ཞེས་ལ་འདིའི་རྣམ་པར་གཅད་པའོ། །ཤྣུ་
བྱིངས་བྷྲཱུ་རྣམས་ཀྱི་ཨིཡ་ཨུཝ་དག་གོ་ཞེས་པ་ཨུ་ཝརུ་བསྒྱུར་རོ། །དེ་ཨི་ཡ་ཨུཝའི་ཡང་ནའོ་ཞེས་པ་སྔོན་དུ་ཨཊ་ཡང་ནའོ། །ཨེཙ་ལའོ། །ཨཀའི་ཨཀ་ལ་རིང་པོའོ། །དེ་ཨིཡ་ཨུ་ཝའི་ཡང་ནའོ་ཞེས་པ་ནུཊ་ཡང་ནའོ། །ཚིག་གཅིག་ལ་ར་ཥ་ལས་ནའི་ཎའོ། །དེ་ལ་ངིའི་ཨཱམའོ། །ཨཀའི་ཨཀ་ལ་
རིང་པོའོ། །རྣམ་པར་གཅད་པའོ། །བྷཱུ། བཧྲུ་ཝཽ། བཧྲུ་ཝཿ། བཧྲུ་ཝཾ། བཧྲུ་ཝཽ། བཧྲུ་ཝཿ། བཧྲུ་ཝཱ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བཧིཿ། བཧྲུ་ཝཻ། བཧྲུ་ཝེ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷྱ། བཧྲུ་ཝཱ། བཧྲུ་ཝཿ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷྱ། བཧྲུ་ཝཱ། བཧྲུ་ཝ། བཧྲུ་ཝོཿ། བྷྲཱུ་ཎཾ། བཧྲུ་ཝཱཾ། བཧྲུ་ཝཱཾ། བཧྲུ་ཝི། བཧྲུ་ཝོཿ་
བྷྲཱུ་ཥུ། ཧེ་བྷྲཱུ། ཧེ་བཧྲུ་ཝཽ། ཧེ་བཧྲུ་ཝཿ། འདི་ལྟར། བཧུཿ། བཧུ་ཝཽ། བཧུ་ཝཿ། སྭ་ཡཾ་བཧུའི་སྒྲ་ནི་བྷྲཱུའི་སྒྲ་བཞིན་ནོ། །ཀཱི་ལཱ་ལ་བཱ་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། །ཨེཙ་ལའོ། །དང་པོ་དག་གི་ཨཙ་ལའོ། །ཨམ་ལ་སྔ་མའོ། །ཤས་ལ་སོགས་པ་ཨཙ་དང་པོ་ཅན་རྣམས་ལ་ཨཱཏའིའོ་ཞེས་པ་ཨཱ་ཡིག་གི་དབྱིའོ། །རྣམ་པར་
