༄༅། །སཾ་བསྐྲི་ཏ་ལ་བྷཀྵ་ལའོ། །འདུས་བྱས་ཀྱི་དོན་ལ་བདུན་པའི་མཐའ་ལས་ཨཎ་འགྱུར་ཏེ། དེ་གལ་ཏེ་བཅོས་པའི་བཟའ་བ་ཡིན་ནའོ། །བྷཱཥྚེ་སཾ་སྐྲི་ཏཱ་བྷཱཥྚཱ་མོད་ཀཱ། བཟའ་བ་ཞེས་པ་ཅི་ཞེ་ན། ཕ་ལ་ཀེ་སཾ་སྐྲྀ་ཏཱ་མཱ་ལཱ། ཁ་ར་པི་ཥཱ་དཾ། ཁམ་གྱི་ཟས་ནི་བཟའ་བའོ། །ཤཱུ་ལ་དང་ཨུ་
ཁ་ལས་ཡ་ཏའོ། །བདུན་པའི་མཐའ་ལས་བཙོས་པའི་བཟའ་བའི་དོན་ལ་འགྱུར་ཏེ། མདུང་གིས་བཅོས་པ་ནི། ཤཱུལྱ་ཤའོ། །ཨུ་ཁཱ་ལས་བཅོས་པ་ནི། ཨུ་ཁྱཾ། དདྷི་ལས་ཋཀའོ། །ད་དཧིའི་སྒྲ་བདུན་པའི་མཐའ་ལས་བཅོས་པའི་བཟའ་བའི་དོན་ལ་ཋཀ་འགྱུར་ཏེ། ཞོ་ལ་བཙོས་པའི་བཟའ་བ་ནི། དཱ་དཧི་
ཀཾ། དདྷི་ཀཱི་ཡཾ། ཞོ་ལ་བཙོས་པ་དང། ཞོ་ཡིས་བཙོས་པ་ལ་ནི་དེ་ཞེས་པས་ཋཀ་འཆད་པར་འགྱུར་བས་གྲུབ་པ་ལ་བརྗོད་པ་ནི་ཨཎ་དགག་པའི་དོན་ཏོ། །གཞན་དུ་ནི་ཞོ་ཡིས་གཞན་གྱུར་པ་དང་དངོས་པོ་གཞན་གྱིས་བཅོས་པའི་དེ་ཡིས་མི་འགྲུབ་བོ་ཞེས་སྒྲུབ་པའི་དོན་དུ་རིག་གོ། །ཨུད་ཤྭི་ཏ་ལས་
ཡང་ནའོ་བདུན་པའི་མཐའ་ལས་བཅོས་པའི་བཟའ་བའི་དོན་ལ་ཋ་ཀ་ཡང་ན་འགྱུར་ཏེ། ཨཽད་ཤིཏྐཾ། ཨཽད་ཤིཏཾ། ཀྵཱི་ར་ལས་དྷཉའོ། །བདུན་པའི་མཐའ་ལས་བཅོས་པའི་བཟའ་བ་ལ་ཌྷཉ་འགྱུར་ཏེ། འོ་མ་ལས་བཙོས་པའི་བཟའ་བ་ནི། ཀྵཻ་རེ་ཡཾ། ཀྵཻ་རེ་ཡཱི། མིང་ལ་དེ་ཡིས་ཟླ་བ་ཉ་བ་ལའོ། །ཟླ་བ་ཉ་བ་བརྗོད་
པ་དང་པོའི་མཐའ་ལས་དེ་འདིར་ཟླ་བ་ཉ་བའི་དོན་བརྗོད་པ་ལ་ཨཎ་འགྱུར་ཏེ། པཽཥཱི། པཽརྞ་མཱ་སཱི། འདིར་པཽཥོ་མཱ་སཔོ་ཥོ་རྡྷརྨཱ་སཿ། པཽ་ཥ་སམྦཏྶ་རཿ། དེ་བཞིན་དུ་མགྷ། མིང་ལ་ཞེས་པ་ཅི་ཞེ་ན། པཽརྞ་མཱ་སཱི། འདིར་ད་ཤྭ་རཱ་ཏྲེ། ཡང་ན་བཧུ་ཏ་ཀ་མཱ་སྟེ། ཟླ་བ་ཉ་ལ་ཞེས་པ་ཅི་ཞེ་ན། པཽཥཱི། འདིར་ཟླ་བ་ལ་ཚེས་
འདི་ལ་ཉ་གང་བ་ནི་ཕཽཎྞ་མཱ་སཱི། དེ་ཉིད་ཀྱི་ཕྱིར་ཚིག་ཕྲད་ལས་ཨཎའོ། །ཨ་གྲ་ཧཱ་ཡ་ཎི་ཨ་ཤཏྠ་ལས་ཋཀའོ། །དེ་འདི་པཽརྞ་མཱ་སཱི་ཞེས་པའི་དོན་འདི་ལ་འགྱུར་ཏེ། ཨ་གྲ་ཧཱ་ཡ་ཎི་ཀོ་མཱ་སཿ།ཨཱ་ཤཏྠི་ཀཿ། ཨ་ཤཏྠ་ནི་རྒྱུ་སྐར་ཟླ་བ་དང་ལྡན་པས་མཚོན་པའི་ཟླ་བ་ཉ་བ་ལ་ཨ་ཤཏྠའི་སྒྲ་བརྗོད་དོ། །ཕཱལྒུ་ནཱི་
ཤྲ་བ་ཎཱ། ཀཱརྟི་ཀཱི་ཙཻ་ཏྲཱི་རྣམས་ཡང་ནའོ། །དེ་འདིའི་ཟླ་བ་ཉ་བ་ཞེས་པ་འདིའི་དོན་ལ་ཋཀ་ཡང་ན་འགྱུར་ཏེ། དབོ་ཅན་གྱི་ཟླ་བ་ནི་ཕཱལ་གུ་ནཿ། གྲོ་བཞིན་ཅན་གྱི་ཟླ་བ་ནི་ཀཱརྟྟི་ཀཿ། ནག་པ་ཅན་གྱི་ཟླ་བ་ནི་ཙཻ་ཏྲཿ། འདིའི་ལྷ་ཡིན་པ་ལས་སོ། །ལྷ་བརྗོད་པ་དང་པོའི་མཐའ་ལས་དེ་འདིའི་ཞེས་པའི་དོན་ལ་ཨཎ་འགྱུར་ཏེ་
