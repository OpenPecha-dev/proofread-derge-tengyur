༄༅། །རིང་པོའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །ཨ་དསའི་ཨ་ཉིད་ལ་ད་ལས་ཨུ་དའི་མའོ་ཞེས་པ་ཨུ་ཉིད་དོ། །མ་ཉིད་ཀྱང་ངོ། །རྣམ་པར་གཅད་པའོ། །ཨ་ས། ཨ་མའུ། ཨ་མའུཿ། ཨ་མུཾ། ཨ་མའུ། ཨ་མའུན། ཨ་མ་ཡཱ། ཨ་མཱ་བྷྱཱཾ། ཨ་མའུ་བྷཱི། ཨ་མུ་སྱཻ། ཨ་མའུ་བྷྱཱཾ། ཨ་མའུ་བྷྱ། ཨ་མའུ་སྱཱ་
ཨ་མའུ་བྷྱཱཾ། ཨ་མའུ་བྷྱཱ། ཨ་མའུ་སྱཱ། ཨ་མུ་ཡོཿ། ཨ་མའུ་ཥཱཾ། ཨ་མའུ་ཥྱཾ། ཨ་མའུ་ཡོཿ། ཨ་མའུ་ཥུ། མ་ནིང་ལ་ཨ་དས་ཞེས་པ་འདིའི་ཏྱད་རྣམས་ཀྱི་ཏས་ལ་སོགས་པ་རྣམས་ཀྱིའང་དྭིའི་བར་གྱི་ཨའོ། །ཨཏའི་ཨཏ་ཨེང་ལའོ། །སུ་ཨམ་དག་གི་མ་ནིང་ལས་སོ། །ཕྱིས་པ་མིན་པ་ལའོ། །ཨི་ཀའི་ཨཙ་ལ་སུཔ་ལའོ། །མ་
ནིང་ལས་སོ། །ཛས་ཤས་དག་གི་ཤིའོ། །ཡམ་མིན་པའི་ཤི་ལའོ། །ཤི་སུཊ་ལའོ། །གསུམ་པ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཨ་དཿ། ཨ་མུ་ནཱི། ཨ་མའུ་ནི། ཨ་དཿ། ཨ་མུ་ནཱི། ཨ་མའུ་ནི། ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་བཞིན་ནོ། །ཨི་དཾ་ཞེས་པ་འདིའི་ཏྱད་རྣམས་ཀྱི་ཏས་ལ་སོགས་པ་རྣམས་ལའང་དའྭིའི་བར་
