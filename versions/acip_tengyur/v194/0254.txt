།རཥ་ལས་ནའི་ཎའོ། །ཧ་རི་ཎཿ།་ རི་དགས།  ཨ་བི་ན།་ མཐོང་བ། པྲྀ་ཛི་ན་ཨ་ཛི་ནའོ། །བྲྀ་ཛའི་འདོད་པ་ལའོ། །ཨཛ་བྲ་ཛ་བ་ཛ་འགྲོ་བ་ལའོ། །འདི་དག་ལས་ཨི་ན་ཙ་ཕ་རོལ་ཚིག་ཕྲད་ལས་པྲྀ་ཛའི་ཨཏ་ཨེང་དུ་མི་འགྱུར་རོ། །བྲྀ་ཛི་ནཾ།་ འཁྱོགས་པོ།  ཡུ་གྷཉ་ལས་ཀྱང་རྣམས་མ་ཡིན་པ་ལ་ཨ་ཛའི་བའིའོ་ཞེས་པས་བའིའི་དངོས་པོར་
མི་འགྱུར་རོ། །ཨ་ཛི་ནཿ།་ །པགས་པའི་གཡང་གཞི།  བའི་བ་ཏི་དག་ལས་ཏ་ན་ནའོ། །ཨི་ཎ་བའི། བཱ་འགྲོ་བ་ལའོ། །པཏླྀ་ལམ་ལ་འགྲོ་བ་ལའོ། །འདི་དག་ལ་ཏ་ན་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ན་ཡིག་དབྱངས་དོན། ཨི་ཀའི་ཨཏ་ཨེང་ཞེས་པས་ཨེ་ཉིད་ཌཱ་པ་བ་ཏ་ནཱ།་ གཡོག། པཏྟ་ནི།་ གྲོང་ཁྱེར།  དྲུ་
དཀྵི་དག་ལས་ཨི་ན་ནའོ། དྲུ་འགྲོ་བ་ལའོ། །དཀྵ་འཕེལ་བ་ལའོ། །འདི་དག་ལས་ཨི་ནན་ཕ་རོལ། ན་ཡིག་དབྱངས་དོན། །ཨི་ཀའི་ཨ་ཏ་ཨེང་ཞེས་པས་ཨོ་ཉིད། །ཨེ་ཙའི་ཨ་ཡ་ཨ་བ་ཨཱ་ཡ་བ་རྣམས་སོ་ཞེས་པས་ཨཱ་བར་བསྒྱུར། །རཥ་ལས་ཞེས། པས་
ཎ་ཉིད་དོ། །དྲི་བི་ཎཾ།་ རྫས།  དཀྵིཎཾ།་ ཡོན། སྦྱིན་པ།  བི་སི་ན། ཨི་རི་ཎ། ཏུ་ཧི་ན། མ་ཧི་ན་རྣམས་སོ། །ཊུ་པ་ས་བོན་འདེབས་པ་ལའོ། །ཨིར་འགྲོ་བ་ལའོ། །ཏུ་ད་གཟིར་བ་ལའོ། །ཨརྷ་མཧ་མཆོད་པ་ལའོ། །འདི་རྣམས་ལ་ཨི་ནན་ཕ་རོལ། །བ་པ་ཏིའི་ཨཏ་ཀྱི་ཨི་ཉིད་དོ། །བི་པི་ནཾ།་ འཐིབས་པོ།  ཨི་རིའི་ཐུང་གནུ་དང་ཎ་ཉིད་
ཨི་རི་ཎཾ། མའུ་ཁ་ལཾ། ཏུ་ཏིའི་དའི་ཧ་དང་ཨཏ་ཨེང་མེད། ཏུ་ཧི་ནཾ།་ བ་མོ།  མ་ཧི་ནཾ།་ མཆོད་པ།  ར་སི། རུ་ཙི། རུ། བྲྀཉ་རྣམས་ལས་ཡུཙའོ། །རས་མདོའི་བྱིངས་སོ། །རུ་ཙིར་འབར་བ་ལའོ། །རུ་སྒྲ་ལའོ། །བྲྀཉ་སྒྲིབ་པ་ལའོ། །འདི་རྣམས་ལ་ཡུཙ་རོལ་ཙ་ཡིག། དབྱངས་དོན། ས་མེད་
པའི་ཡུ་བུའི་ཨ་ན་ཨ་ཀ་དག་གོ་ཞེས་པས་ཨ་ནར་བསྒྱུར་རོ། །ཨཙ་ལ་སོགས་པ་དང་ཨཏ་ལས་སོ་ཞེས་པས་ཌཱཔའོ། །ར་ས་ནཱ།་ སྐེ་རགས།  ཡང་བ་མཐའ་ཉེའི་འོ་ཞེས་པས་ཨོ་ཉིད་དོ། །རོཙ་ནཱ།་ འདི་ནང་ངམ་བ་ལང་གི་མཁྲིས་པ།  བྱ་བའི་དོན་གྱི་ཨི་ཀའི་ཨཏ་ཨེང་ངོ་ཞེས་པས་ཨོ་ཉིད་དོ། །ཨེ་ཙའི་ཨ་ཡ་ཨ་བ་ཨཱ་ཡ་ཨཱ་བ་རྣམས་སོ་
ཞེས་པས་ཨཱ་བར་བསྒྱུར། །རཥ་ལས་ཞེས་པ་ཎ་ཉིད་དོ།  །ཉ་མ། ར་བ་ནཾ། ཨཏ་ཨེང་དང་ཕ་རོལ་ཉིད་ཌཱ་པའོ། །ཎ་ཉིད་དོ། །བ་ར་ཎཱ།་ ཆུ་ཀླུང།  ཨུནྡིའི་ན་དབྱི་བའང་ངོ། །ཨུཎྜི་ཀླས་པ་ལའོ། །འདི་ལས་ཡུ་ཙ་ཕ་རོལ། །ས་མེད་པའི་ཡུ་བུའི་ཨ་ན་ཨ་ཀ་དག་དན་
