༄༅། །པ༑ བརྩམ་དུ་རུང་བ་བསྟན་པ་སྩལ་ལགས་ཏེ། འདི་ལྟར་ཅང་མི་གསུང་བས་དེ་དེ་བཞིན་དུ་འཛིན་ཏོ། །དེ་ནི་གཙུག་ལག་ཁང་གི་གཞི་བསྟན་པ་སྦྱིན་པའི་ཆོ་ག་དང་ལས་སོ། །དགེ་འདུན་གྱི་དབྱེན་བྱེད་པའི་དགེ་འདུན་ལྷག་མའི་སྐབས་ནས། དགེ་འདུན་གྱི་དབྱེན་
བྱེད་པ་ལ་ཟློས་པའི་ཕྱིར་བསྒོ་བར་བྱའོ་ཞེས་གསུངས་པ་ནི་འདི་ལྟར་བྱ་སྟེ། དགེ་སློང་གང་དགེ་འདུན་མཐུན་པ་དབྱེ་བའི་ཕྱིར་རྟུལ་བར་བྱེད་ཅིང་དབྱེན་བྱེད་པར་གྱུར་པའི་རྩོད་པ་ཡང་དག་པར་བླངས་ནས་རབ་ཏུ་བཟུང་སྟེ་འདུག་ན། དགེ་སློང་དེ་ལ་དགེ་སློང་རྣམས་ཀྱིས་དང་པོར་
བཞམས་ཏེ་བསྒོ་བར་བྱའོ། །འདི་སྐད་ཅེས་བསྒོ་བར་བྱ་སྟེ། ཚེ་དང་ལྡན་པ་ལྷ་སྦྱིན་ཁྱོད་དགེ་འདུན་མཐུན་པ་དབྱེ་བའི་ཕྱིར་རྟུལ་བར་མ་བྱེད་ཅིག་དབྱེན་བྱེད་པར་གྱུར་པའི་རྩོད་པ་ཡང་དག་པར་བླངས་ནས་རབ་ཏུ་བཟུང་སྟེ་མ་འདུག་ཅིག་ཚེ་དང་ལྡན་པ་ལྷ་སྦྱིན་ཁྱོད་དགེ་འདུན་

་དེ་ལ་གསོལ་བ་དང་བཞིའི་ལས་ཀྱིས་བསྒོ་བར་བྱའོ། །འདི་ལྟར་བསྒོ་བར་བྱ་སྟེ། གནས་མལ་བཤམ་པ་བྱས་ནས་གཎྜཱི་བརྡུངས་ཏེ་དྲིས་པའི་ཚིག་གིས་དགེ་སློང་རྣམས་ལ་ཡང་དག་པར་བསྒོ་ནས་དགེ་འདུན་ཐམས་ཅད་འདུག་ཅིང་མཐུན་པར་གྱུར་པ་ལ་དགེ་སློང་ཅིག་གིས་གསོལ་བ་བྱས་ཏེ་ལས་
བྱའོ༑ །འདི་ལྟར་བྱ་སྟེ། འདི་སྐད་ཅེས་བརྗོད་པར་བྱའོ། །དགེ་འདུན་བཙུན་པ་རྣམས་གསན་དུ་གསོལ། དགེ་སློང་ལྷ་སྦྱིན་འདིས་དགེ་འདུན་མཐུན་པ་དབྱེ་བའི་སླད་དུ་རྟུལ་ཏེ་དབྱེན་བགྱིད་པར་འགྱུར་བའི་རྩོད་པ་ཡང་དག་པར་བླངས་ནས་རབ་ཏུ་བཟུང་སྟེ་མཆིས་ནས་འདི་ལ་དགེ་སློང་མང་པོས་
