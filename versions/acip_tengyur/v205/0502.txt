གྱི་ནང་དུ་སྟེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་རིན་པོ་ཆེ་ཆེན་པོའི་རྒྱལ་པོའི་པད་མ་དེ་ལ་འདུག་མ་ཐག་ཏུ་དེ་ནས་རིན་པོ་ཆེའི་རྒྱལ་པོའི་འཁོར་པད་མ་ཇི་སྙེད་བྱུང་བ་དེ་སྙེད་ཀྱི་བྱང་ཆུབ་སེམས་དཔས་དེ་བསྐོར་ནས་པད་མ་དེ་དག་ལ་འཁོད་པར་སྣང་སྟེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་ཉིད་ལ་བལྟ་
ཞིང་དེ་དག་རེ་རེ་ལ་ཡང་ཏིང་ངེ་འཛིན་སྟོང་ཕྲག་བཅུ་བཅུ་ལ་སྙོམས་པར་འཇུག་གོ། །མཚན་མ་གང་བྱུང་བར་གྱུར་ན་དབང་བསྐུར་ཞེ་ན། འོད་ཟེར་བྱུང་བ་སྟེ། འོད་ཟེར་དེ་ཡང་ལས་གསུམ་དང་ལྡན་པ་ཡིན་ཏེ། ཕན་འདོགས་པའི་ལས་ཅན་དང་། སྦྲོན་པའི་ལས་ཅན་དང་། འགུགས་པའི་
ལས་ཅན་ནོ། །དེ་ལ་ཕན་འདོགས་པའི་ལས་ཅན་ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའ་དེ་རིན་པོ་ཆེ་ཆེན་པོའི་རྒྱལ་པོའི་པད་མའི་སྟན་དེ་ལ་འདུག་མ་ཐག་ཏུ་དེའི་ལུས་ཀྱི་ཡན་ལག་ཐམས་ཅད་ཀྱི་ཡན་ལག་རེ་རེ་ནས་ཀྱང་འོད་ཟེར་གྲངས་མེད་པ་བརྒྱ་ཕྲག་བཅུ་བཅུ་དང་། སྟེང་གི་ཡན་ལག་གི་དམ་པ་
མགོ་ལས་སྟོང་གསུམ་གྲངས་མེད་པ་བརྒྱ་ཕྲག་བཅུར་ཚང་བའི་རྡུལ་ཕྲ་རབ་ཀྱི་རྡུལ་སྙེད་ཀྱི་འོད་ཟེར་བྱུང་བས། ཕྱོགས་བཅུའི་ངན་སོང་གསུམ་དང་མི་དང་ལྷ་དང་ལྷ་མ་ཡིན་དང་ཉན་ཐོས་དང་རང་སངས་རྒྱས་དང་། བྱང་ཆུབ་སེམས་དཔའ་དང་བདུད་དང་དེ་བཞིན་གཤེགས་པའི་འཁོར་གྱི་
དཀྱིལ་འཁོར་ཐམས་ཅད་སྣང་བར་བྱས་ནས་སྡུག་བསྔལ་ཞི་བ་དང་། ཆོས་ཀྱི་སྣང་བའི་སྒོ་དང་། ཏིང་ངེ་འཛིན་གྱི་སྒོའི་ཚུལ་དང་། ཤེས་རབ་དང་ཐབས་ལ་མཁས་པའི་ཚུལ་ཉེ་བར་སྒྲུབ་པ་དང་། མཆོད་པ་དང་། རིམ་གྲོ་བྱ་བ་འོད་ཤིན་ཏུ་གསལ་བ་ཞེས་བྱ་བ་རབ་ཏུ་འབྱེད་པས་འཇིག་རྟེན་གྱི་
ཁམས་ཐམས་ཅད་མ་ལུས་པར་གཡོ་བ་དང་། ངན་སོང་ཐམས་ཅད་རྒྱུན་ཆད་པ་དང་། ཆོས་ཀྱི་དབྱིངས་ཐམས་ཅད་སྣང་བས་རྒྱས་པ་དང་། འཇིག་རྟེན་གྱི་ཁམས་ཐམས་ཅད་ཡོངས་སུ་དག་པ་དང་། ཞིང་ཐམས་ཅད་དུ་སངས་རྒྱས་ཀྱི་སྒྲ་གྲག་པ་དང་། སྤྱོད་པ་མཐུན་པའི་བྱང་ཆུབ་སེམས་དཔའ་འདུ་བ་
དང་། འཇིག་རྟེན་གྱི་ཁམས་ཐམས་ཅད་ཀྱི་ལྷ་དང་མིའི་རོལ་མོ་གླུ་དང་རོལ་མོའི་སྒྲ་དང་། སིལ་སྙན་གྱི་སྒྲ་འབྱུང་བ་དང་། སེམས་ཅན་ཐམས་ཅད་བདེ་བ་སྐྱེ་བ་དང་། ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཐམས་ཅད་ལ་མཆོད་པ་དང་། རིམ་གྲོ་བྱ་བ་བསམ་གྱིས་མི་ཁྱབ་པར་བྱེད་པ་འབྱུང་བར་འགྱུར་
