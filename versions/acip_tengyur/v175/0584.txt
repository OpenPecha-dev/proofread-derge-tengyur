༄༅། །པ་ཉིད་ཀྱི་ཕྱིར། རི་བོང་ཅན་ཟླ་བ་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་གནོད་པ་ཅན་གྱི་གཏན་ཚིགས་འདི། ཐུན་མོང་ཡིན་པ་ཉིད་མི་འགྱུར། །གལ་ཏེ་ཐམས་ཅད་ཟླ་བའི་སྒྲར་རུང་བར་འགྱུར་བ་དེའི་ཚེ། མཐུན་པའི་ཕྱོགས་མེད་པའི་ཕྱིར། ཐུན་མོང་མ་ཡིན་པར་འགྱུར་རོ། །དེ་ཉིད་ཀྱི་
ཕྱིར་དེ་འགོག་སྟེ། སྒྲའི་རུང་བ་ཉིད་འགོག་པ། རྗེས་དཔག་ལས་འགྱུར་གྱི། མངོན་སུམ་ནི་མ་ཡིན་ནོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། སྒྲ་དོན་ལ་དབང་ཕྱུག་མེད་ཕྱིར། སྒྲའི་རུང་བ་ཉིད་དམ་རུང་བ་ཉིད་མ་ཡིན་པ་ལ་དབང་པོ་འཇུག་པར་མི་འགྱུར་ན། གང་གིས་ན་མངོན་སུམ་ལས་ངེས་པ་དེ་འགོག་
པར་འགྱུར། དེ་ལྟར་ན་རྗེས་སུ་དཔག་པ་མེད་པ་ལས་འགྱུར་བ་ཡིན་ནོ། །དེ་ལ་ཡང་རང་བཞིན་ཉིད་ཀྱིས་རུང་བ་ཉིད་འདི་ལ་སྒྲ་ཐམས་ཅད་སྦྱོར་བ་བྱེད་པས་མཐུན་པའི་ཕྱོགས་ཟླ་བ་མ་ཡིན་པ་ཅུང་ཟད་ཀྱང་ཡོད་པ་མ་ཡིན་ན། གང་ལ་རྟགས་ཡོད་པ་རྗེས་སུ་འགྲོ་བ་ཅན་དུ་འགྱུར། དེ་ལ་རུང་བ་ཉིད་
འགོག་པ་ལ། ཐུན་མོང་མ་ཡིན་ཉིད། གཏན་ཚིགས་ཞེས་བྱ་བ་རྗེས་སུ་དཔག་པ་མེད་པ་ཡིན་ནོ། །དེ་ལྟར་ནི་འགྱུར་མོད་ཀྱི། ཐུན་མོང་མ་ཡིན་པ་ཉིད་དུ་མི་འགྱུར་ཏེ། མི་མཐུན་པའི་ཕྱོགས་ལ་ཡོད་པའི་ཕྱིར་འགལ་བ་ཉིད་དུ་འགྱུར་རོ་ཞེས་ན། བདེན་ཏེ་འོན་ཀྱང་གཞན་བུམ་པ་ལ་
སོགས་པ་མི་མཐུན་པའི་ཕྱོགས་ཉིད་དུ་མི་འགྱུར་བའི་ཕྱིར། གཞན་གྱི་བསམ་པས་མི་མཐུན་པའི་ཕྱོགས་ལ་ཡོད་པ་མ་ཡིན་པ་དེ་ལྟར་ན་ཐུན་མོང་མ་ཡིན་པར་བརྗོད་དོ། །འདོད་པའི་སྒྲས་རུང་བ་ཉིད་སྤོང་བ། གང་ལ་ཡོད་པ་ཉིད། ཇི་སྲིད་དུ་དངོས་པོ་ཐམས་ཅད་འབྱུང་བའི་ངང་ཚུལ་ཅན་
དུ་ཡོད་པ་དེ། རྗེས་སུ་འགྲོ་ཅན་མིན། དེ་ལ་གང་ལས་ཡིན་འདིར་ཡོད་པ་ཉིད་ཅེས་བྱ་བའི་རྒྱུ། དཔེ་ནི། སློབ་དཔོན་དེ་ལྟར་འབྲས་ལྡན་འདོད། །གཏན་ཚིགས་ཐམས་ཅད་ཐུན་མོང་མ་ཡིན་པ་ཉིད་དུ་འདོད་གལ་ཏེ་གང་ལས་སྒྲ་ཐམས་ཅད་དོན་ཐམས་ཅད་ལ་གྲུབ་པ་ཡིན་ན། གང་གིས་ན་སྒྲ་ལས་
བྱུང་བའི་གྲགས་པ་ཞེས་བརྗོད་ཅེ་ན། འདིར་འདི་ལྟར། སྒྲ་ནི་བརྡ་ལ་བརྟེན་པ་ཅན། །བརྡ་ཡང་འདོད་ཙམ་རྟེན་ཅན་ཡིན། །ཞེས་བཤད་པ་དེའི་ཕྱིར། སྒྲ་ཡིས་གྲུབ་ན་མི་འགྲུབ་མིན། །དེ་ལྟར་སྒྲ་བྱུང་གྲགས་པའི་ཚིག་ཅེས་བཤད་པ་ཡིན་ནོ། །ཡང་ན་དངོས་པོ་སྟོབས་ཀྱིས་ཞུགས་པའི་
