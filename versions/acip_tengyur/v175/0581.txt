ཚད་མ་དང་གནོད་པ་བྱེད་པ་ཡིན་ནོ་ཞེས་རྒོལ་བ་དེ་མི་འདོད་པ་དེའི་ཚེ། ཅིས་ཀྱང་གནོད་པ་བྱེད་པར་འགྱུར། མི་འགྱུར་བ་ཉིད་ཡིན་ནོ། །དེ་ལྟར་ན་འདོད་པ་དེའི་སྐྱོན་དེ་ཐོབ་པར་འགྱུར་གྱི། རིགས་པ་ལས་མ་ཡིན་ནོ། །ལུང་དེ་མི་འདོད་ན་ནི་གནོད་པ་བྱེད་པ་ཡིན་
པར་མི་འགྱུར་རོ། །གལ་ཏེ་སྔར་དེས་བསྟན་བཅོས་ཁས་ལེན་ཅིང་ཕྱིས་དེ་མི་འདོད་པར་འགྱུར་བ་སྔ་མ་དེས་ཕྱི་མ་ལ་གནོད་པའི་ཕྱིར་རྒོལ་བ་འདིའི་རང་གི་ཚིག་དང་འགལ་བ་ཡིན་ནོ་ཞེ་ན། འོ་ན་དེ་ལྟ་ཡིན་ན། ཁོ་བོ་ཅག་གིས་སྔར་ཇི་སྐད་བཤད་པ་བཞིན། རང་གི་ཚིག་བསྟན་བཅོས་
འགལ་ཐ་དད་མེད་འགྱུར་བ། དེ་ལྟར་ན་གགས་བྱེད་པ་ཉིད་ཡིན་པར་འགྱུར་རོ། །གལ་ཏེ་སྔར་དེས་བསྟན་བཅོས་ཁས་ལེན་ཅིང་ཕྱིས་རང་གི་ཚིག་དང་མཚུངས་པ་མ་ཡིན་པ་དེ་བས་ན་དེ་གནོད་པ་བྱེད་པ་ཡིན་ནོ་ཞེ་ན། ཉེས་པ་ཟད་པར་ཤེས་པ་མེད་ན། འདི་གཉིས་ལ་ཡང་ཚིག་ཙམ་
ཉིད་དུ་བྱེ་བྲག་ཅི་ཡང་ཡོད་པ་མ་ཡིན་པའམ། ཤེས་ན་དེའི་ཚེ་དོན་དེ་ཉིད་ཚད་མས་མཐོང་བ་ཡིན་པ་དེ་ལྟར་ན། ཡིད་ཆེས་པའི་ཚིག་དེས་གནོད་པ་ཡིན་ན། རྗེས་སུ་དཔག་པ་ཉིད་ཀྱིས་གནོད་པ་ཡིན་ནོ། །དེ་བས་ན་བསྟན་བཅོས་ཀྱིས་གནོད་པ་ཞེས་བྱ་བ་ཐ་དད་པར་བརྗོད་པར་
བྱ་བ་མ་ཡིན་པས་གང་དེ་ནི་ཅིར་ཡང་མི་རུང་ངོ་། །གཞན་ཡང་རྒོལ་བས་ཁས་བླངས་པའི་ཕྱིར། གང་གི་ཚེ་བསྟན་བཅོས་ཚད་མ་ཉིད་ཡིན་པ་དེའི་ཚེ། སྐྱེས་བུའི་འདོད་པས་བྱས་འདི་ཡི། །ཚད་མ་ཉིད་ནི་ཡོངས་སུ་རྫོགས། །སྐྱེས་བུའི་འདོད་པ་ཉིད་ནི་དངོས་པོ་ལ་རག་ལས་པ་མ་ཡིན་
པའི་ཕྱིར། དེའི་དབང་གིས་ཚད་མ་ཉིད་དུ་ཡོད་པ་མ་ཡིན་ནོ་སྙམ་དུ་དགོངས་པ་བཞད་གད་དུ་མཛད་པའོ། །གང་གི་ཕྱིར་འདི་ལྟར། བསྟན་བཅོས་གནོད་པ་མ་ཡིན་ནོ། །ཞེས་གནས་པ་ཡིན་ན། དེའི་ཕྱིར། མངོན་སུམ་དང་རྗེས་སུ་དཔག་པ་དག་གི་རབ་གྲགས་དོན་དག་ལ། གང་
གི་ཚེ་ཆོས་ཅན་དང་རྟགས་ཡིན་པར་དཔྱོད་པ་དེའི་ཚེ་བསྟན་བཅོས་དོར་ནའང་། བསྟན་བཅོས་ལ་རྟགས་ཡིན་པར་དཔྱོད་པ་དེའི་ཚེ། བསྟན་བཅོས་དོར་ནའང་བསྟན་བཅོས་ལ་བརྟེན་པ་མེད་པར་ཡང་ཉེས་པ་མེད། འདོད་པའི་དོན་འགྲུབ་པ་ལ་གེགས་བྱེད་པ་མེད་པའི་ཕྱིར་རོ། །ཤིན་ཏུ་ལྐོག་གྱུར།
