། ། ། ། ། ། ། ། ། ། ། ། །  ག་ལ་ཨ་ཋ་གྷནྷས་བྲིས་ཞེས་པ། གུར་གུམ་གླ་རྩི་ག་བུར་གི་ཧཾ་ཙནྡན་དཀར་དམར། ཨ་ཀ་རུ། ཨ་ཀ་ར། གོ་སྙོད་མ་རྣམས་ལ། ཨ་ཋ་གྷནྷ་ས་གཟེར་རོ། །འཁོར་ལོ་གཉིས་བྲིས་གཅིག་མགུལ་དུ་གདགས་གཅིག་ཆུ་ལ་བཀྲུས་ནས་གྲོག་ཕྱིར་ཕྱུང་ལ་བཀྲུས་པའི་ཆུ་དེ་རང་ཉིད་འཐུངས་པས་དུག་ལས་རྣམ་པར་རྒྱལ་ལོ། །དོན་གཉིས་ འཁོར་ལོ།།   ༈    འགྲོན་པོ་སླར་ལོག་ཡོང་ཞིང་ཡུལ་གཞན་པར་མི་འདོད་བའི་ཐབས་སོ། །བརྒྱ་བཞིའི་ གནོད་པ་ཐམས་ཅད་ཐུབ། གཅིག་ཆུས་བཀྲུས་ནས་འཐུངས་ན་ཕན་ནོ། །    ༈   མགུལ་དུ་གདགས་ན་འདྲེ་གདོན་མཐའ་དག་ལས་རྣམ་པར་རྒྱལ་     ༈    །གོང་འདྲ་སྙིང་གར་གདགས་ན་མདའ་མི་ཕོག་པའི་གདམས་པའོ། །སོ་བརྒྱད་འཁོར་ལོ།    ༈     ། །།།བུད་མེད་ཁྱོ་ལ་བློ་མི་གཏོད་པ་ལ་གདགས་ན་བློ་གཏོད་པར་འགྱུར་རོ། །དོན་ལྔ་འཁོར་ལོ།    ༈    །གོང་འདྲ་མགུལ་དུ་གདགས་ན། རླུང་གི་ནད་ཀྱིས་གཟུགས་པོའི་མགོ་མཇུག་ཐམས་ཅད་དུ་སྲིན་བུ་རྒྱུགས་པ་ལྟར་ཁྱུགས་པ་ལ་ཕན་ནོ། །   །    ༈   སེར་བ་སོགས་ལོ་ཐོག་ལ་གནོད་པ་མི་  །བརྒྱ་དང་བརྒྱད་ཅུའི་འཁོར་ལོ། །    ༈   སོགས་ཀྱི་བོང་མཛོད་ནང་དུ་བཅུག་ན་རུལ་བ་སོགས་དྲི་ དྲི། མི་མན་མ་པའི་ཐབས་སོ། །ཡང་འདི་གདགས་ན་ནག་ཅན་བཙོན་ནང་ནས་འཐོན་ནོ། འབུམ་འཁོར་ལོ།།    ༈    གདགས་ནས་མདུན་ཕྱོགས་སུ་འོངས་པ་བྱམ་ན་བུད་མེད་བུ་མི་ཆགས་པར་བཤད་དོ། །བརྒྱ་ཉི་ཤུ་འཁོར་ལོ།    ༈    །སུ་ཧྲུག་ལ་ཁབ་ཀྱིས་འཁོར་ལོ་བྲིས་ནས་ཁྱོའམ་སྐྱེས་སྨན་ཕྱི་དང་མཐུན་ནང་དུ་མི་མཐུན་ན་སོ་སོར་གཉིས་ བྲིས་ནས་གཡག་གི་རྭ་འམ་ཟེ་བར་གདགས་ན་ཁ་ཚ་ཟེར་བའི་ཕྱུགས་ནད་ལ་ཕན་ནོ། །སོ་གཉིས་འཁོར་ལོ།    ༈    །གོང་འདྲ་ཕྲུ་གུ་ཉིན་མཚན་དུ་དུ་བར་མགུལ་དུ་གདགས་ན་ཕན་ནོ། །བཞི་བཅུ་འཁོར་ལོ་ །།ནང་གི་ཟུར་བཞིར་གདགས་ན། མི་ཤི་ནས་འདྲེར་སྐྱེས་ནས་གནོད་པ་བྱེད་པ་མཐའ་དག་ཞི་བར་འགྱུར་རོ། །དོན་གཉིས་འཁོར་ལོ།    ༈    །ཤིང་རྡོ་ཏིའི་མྱུ་གུས་གི་ཧཾ་ཙནྡན་དམར་པོ་ཕོ་རོག་ཁྲག་གིས་བོང་བུའི་པགས་པ་ལ་བྲིས་རྐང་པ་ལ་གདགས་ ན། གཞན་གྱིས་མི་མཐོང་ངོ། སྒྲིབ་ཤིང་ཡིན་ནོ།།    ༈   རྒྱལ་དུ་ཉལ་ནས་གང་ན་སར། ཟངས་ ལྕགས་ཀྱི་སྣོད་གང་རུང་གི་ན་དུ་འཁོར་ལོ་གཅིག་མེ་ལ་བསྲེགས་ནས་ཆུ་བསྲེས་ཏེ་ཁོང་དུ་བཏང་ན། སྲིན་ནད་ཀྱི་རིགས་མཐའ་དག་ལ་ཕན་ནོ། །གཅིག་གིས་མ་ཕན་སླར་གང་དགོས་བསྐྱར་རོ།།   ༈    །དོན་བརྒྱད་འཁོར་ ། མཚོན་གྱི་རྨས་བྱུང་ན། རྨ་གང་བྱུང་ངུ་འཁོར་ལོ་འདི་གདགས་ན་རུལ་མི་འོང་ཞིང། །དྲི་ངན་པ་མི་འབྱུང་ངོ་ལྔ་བཅུ་དགུའི་འཁོར་ལོ།    ༈    །བཅུ་དགུ་འཁོར་ལོ་ མྱོང་གྲུབ་ལགས་སོ། །ཁུར་པོ་ཁུར་ནས་དུབ་མི་ཡོང་བ་ལ་ཕན་ནོ། །ལྔ་བཅུ་བཞིའི་འཁོར་ལོ།    ༈    །བརྒྱ་ཉེར་གསུམ་འཁོར་ལོ་ །།  སྨན་ཁུག་གམ། སྒྲོ་བའི་ནང་དུ་གདགས་ན་ནོར་རྫས་འབད་མེད་དུ་འབྱུང་ངོ། །རེ་ལྔ་འཁོར་ལོ།    ༈    གཉིད་དུ་སོང་དུས་སྡུར་པ་མི་འབྱུང་བའི་གདམས་པ་ཨི་ཐི༑  པར་འགྱུར་རོ། །ཉིས་བརྒྱ་གསུམ་འཁོར་ལོ།  མགུལ་དུ་གདགས་ན་ཉན་པར་བཤད་དོ། །བརྒྱ་བདུན་ཅུ་འཁོར་ལོ། །   ༈   གཅིག་ཆོམ་ནས་མི་ནོར་ཤོར་ན་སླར་མི་གཅིག་ལ་མགུལ་  དགྲ་དེའི་ཡུལ་དུ་བཏ་ངན་མི་ནོར་རྣམས་ལོག་པར་འགྱུར་རོ། །དོན་གཉིས་འཁོར་ལོ། །   ༈    །གྲོ་ག་ལ་གོང་འདྲ་བྲིས་མགུལ་དུ་གདགས་ན་གསོན་འདྲེ་མོའི་གནོད་པ་མི་འབྱུང་བར་ངེས་སོ་  །བརྒྱ་དྲུག་ཅུའི་འཁོར་ལོ།།   ༈    །གོང་འདྲ་མགུལ་དུ་གདགས། རྫ་འདམ་ལ་འཁོར་ལོ་འདི་བཞིན་བྲིས་ནས་རྫ་འདམ་རིམ་བཞིན་བཅག་པས་དཀྱིལ་དུ་ཁྱི་སྨྱོན་གང་གིས་འཇུས་པའི་ཁ་དོག་དང་མཐུན་པའི་སྤུ་འོངས་པས། དེས་ཁྱི་སྨྱོན་གྱི་དུག་ལས་གྲོལ་ལོ། །གལ་ཏེ་མ་བྱུང་ན་སླར་བསྐྱར་ པས་ཕན་ནོ།།༈    །བརྒྱ་ལྔ་བཅུའི་འཁོར་ལོ་ །།    ༈   །བརྒྱ་ཉི་ཤུའུ་འཁོར་ལོ།    ༈    བཟའ་མི་མི་མཐུན་བ་ལ་མཐུན་པར་འགྱུར་བ་ཡིན༑ གཞན་མི་མཐུན་པ་མཐའ་  །ལྔ་བཅུ་ང་དྲུག་འཁོར་ལོ། །    ༈   གདགས་ན་གཉིས་ཀྱི་ནང་དུ་དངངས་སྐྲག་འོང་བ་ལ་ཕན་ནོ། །འཁོར་ལོ་འདི་གདགས་ གདམས་པའོ། །སྟོང་ཕྲག་འཁོར་ལོ།    ༈    །བརྒྱ་བཅུའི་འཁོར་ལོ་  བུད་མེད་བུ་ཡང་ཡང་ཤར་འགྲོ་བར། གཟའ་ཉི་མའི་ཉིན་ཚེས་ཉེར་དགུ་ལ་བྲིས་ནས་གདགས་ན་བུ་ཆགས་པར་འགྱུར་རོ།    ༈    ཚད་པའི་རིགས་མཐའ་དག་ལ་ཕན་ནོ། །བརྒྱ་བཅུའི་འཁོར་ལོ་ །།  ཁ་ཟས་རྙེད་པར་བཤད་དོ། །ཉེར་བརྒྱད་འཁོར་ལོ།    ༈    ཁྲལ་འུ་ལག་ཉུང་བའི་གདམས་པའོ། །དྲུག་ཅུ་འཁོར་ལོ་  །བརྒྱད་བརྒྱ་གོ་གཉིས་འཁོར་ལོ།    ༈    ཞོ་སྲུབས་  །སུམ་ཅུ་   ༈     ལྷ་དང་བླ་མ་དཔོན་པོའི་ཐུགས་རྗེས་འཛིན། ལག་པ་ལ་གདགས་ན་ཚད་པའི་རིགས་ལ་ཕན་ནོ། །སུ་ཧྲུག་ལ་ཁྲག་གིས་བྲིས་ནས་གདགས་ན་མཛེ་ནད་ལ་ཕན་ནོ། །བརྒྱ་བཅུ་འཁོར་ལོ།    ༈   འདི་བཞིན་བྲིས་ལ་ཐུམ་སྐོགས་དར་དཀར་དམར་སེར་ལྗང་བཞིས་གཏུམས་ལ་ཕྱོགས་བཞིར་བཙུགས་ན། ཆོམ་རྐུན་ལས་རྣམ་ གདགས་ན། ཁ་མཆུ་ལས་རྒྱལ་ལོ།    ༈    བད་ཀན་རླུང་གིས་བེ་སྣབས་ཁྲུ་བ་ལ་ཕན་ནོ། །ཉིས་བརྒྱ་འཁོར་ལོ་ །    ༈    །འཁོར་ལོ་འདི་མགུལ་དུ་། གཅིག་གཏིང་གི་ལོགས་ལ་འཁོར་ལོ་ལྟར་བྲིས་ནས་ཆུས་བཀྲུས་ལ་ཁུ་བ་འཐུངས་ན། བུ་སྐྱེ་མ་ཐུབ་པ་ལ་ཕན་ནོ། བཞི་བཅུ་བཞིའི་འཁོར་ལོ།    ༈    མགུལ་དུ་གདགས་ན་ལམ་བར་ལ་རྐང་པ་མིན་ཁྱིས་མི་འཆའ་བའི་གདམས་པའོ་ བ་ལ་མཆོག་ཏུ་ཕན་པར་བཤད་དོ། །དྲུག་ཅུ་རེ་གཉིས་འཁོར་ལོ། །   ༈    གདགས་ན་ཕོ་བ་བད་རླུང་གི་ནད་རྣམས་ལ་ཕན་པར་ གྱིས་བྲིས་ནས་གདགས་ན་ཁྲག་རླུང་གིས་མགོ་ན་བ་ལ་མཆོག་ཏུ་བཤད་དོ། །བཞི་བཅུ་འཁོར་ལོ།།    ༈    ནས་རྐེད་པ་ལ་གདགས་ན། བཤལ་བ་བཅད་པའི་མཆོག་གོ་ཞེ་བརྒྱད་འཁོར་ལོ།    ༈    འཁྲུ་བ་མི་ཕེབས་པར་ལྟེ་བ་ན་བ་ལ་ཕན་ནོ། །བརྒྱ་ མགུལ་དུ་གདགས་ན་སི་བའི་རིགས་ལ་ཕན་ནོ། །བརྒྱ་བརྒྱད་འཁོ་རློ།    ༈    །དགུང་ཅུ་གཉིས་  ཆོས་སྐད་དང་རིགས་ལུང་སོགས་རྒྱལ་བ་དང། བློ་གསལ་བའི་གདམས་པའོ། །སྟོང་ཕྲག་བཞིའི་འཁོར་ལོ།    ༈    །གྲོ་ག་ལ་ཨ་ཋ་གྷནྷས་བྲིས་ནས་ཁང་  སྐྲད་རྙིང་གི་དུད་པ་བཏང་ན། འདྲེ་གདོན་གྱི་གནོད་པས་མི་ཚུགས་སོ།། བརྒྱ་ཉེར་གཉིས་འཁོར་ལོ།།    ༈   ན་གཟའ་ནད་ཕུར་བུས་གཟུང་བ་ལ་ཕན་ནོ། །སུམ་ ན་ནད་རིགས་མཐའ་དག་ལ་སྔར་ནས་ཀྱི་ནད་གཞི་མེད་པར་འགྱུར། ཕྱིས་ནད་རིགས་མི་འབྱུང་བའི་གདམས་པ་ཨི་ཐི། བརྒྱ་དགུ་བཅུ་འཁོར་ལོ།    ༈    །གྲོ་ག་ལ་ཨ་ཋ་གྷནྷ་ས་བྲིས་ནས། མགུལ་དུ་གདགས་ན། མཐུ་གཏད་ཀྱིས་གླགས་མི་རྙེད་དོ། །བརྒྱ་དང་ ཤོག་བུར་བྲིས་ཏེ་མགུལ་དུ་གདགས་ན། སོ་ནད་ཀྱི་རིགས་མཐའ་དག་ལ་ཕན་ནོ། །    ༈    གཉིད་ཀྱི་ནང་དུ་གད་མོ་འོང་བ་ལ་ཕན་     ༈    །གྲོ་ག་ལ་ཨ་ཋ་འཁོར་ལོ་གཉིས་བཅོས་ནས། གཅིག་མགུལ་དུ་གདགས་ན། གཅིག་མེ་ལ་བསྲེགས་ནས་དུད་པ་བཏང་ན། བད་རླུང་གིས་གཟེར་བ་ལ་ཕན་ནོ།    ༈    །འདི་རྫས་གོང་འདྲ་ནས་སོ་ན་ན་སོའི་བར་དུ་སྟེམས་ན་སོ་ནད་ །    ༈    ནད་པ་ལ་རི་ཐང་མཚམས་སམ་གདོན་སོགས་ཀྱིས་སྨྱོ་ལབ་བྱེད་  ཉེར་དགུའི་འཁོར་ལོ། །    ༈   ན་འདྲེའི་གནོད་པ་ཐམས་ཅད་ཀྱིས་མི་ཚུགས་པར་ ། བརྒྱད་ཅུ་གྱ་གཅིག་འཁོར་ལོ།།    ༈    །འཁོར་ལོ་འདི་མགུལ་དུ་གདགས་ན། བསམ་པ་ངན་པ་མི་འཆང་བར་འགྱུར་ཞིང་སེམས་ ། །སོ་གཉིས་འཁོར་ལོ༑    ༈    །འཁོར་ལོ་འདི་གཅིག་མགུལ་དུ་གདགས་གཅིག་ཆུས་བཀྲུས་ནས་ཁུ་བ་འཐུངས་པ་དང། གྲོ་ག་དང་གུ་གུལ་མེ་ལ་བསྲེགས་ནས་དུད་པ་བཏང་ངོ། །དེས་མི་འགྲོ་ན་ ན་དམར་བཤལ་གྱི་ནད་ལ་ཕན་ནོ། །ཁྲག་སྟོད་དུ་རླུང་གིས་བུས་ནས་མགོ་བོ་ན་བ་ལ་ཕན་པར་བཤད་དོ། །སོ་བདུན་འཁོར་ལོ།    ༈    རང་གི་བུད་མེད་བློ་ཕྱིར་ཁྱེར་ནས་ཁྱོ་གཞན་མ་ཚོལ་ན་ཕན་ནོ། །ལྔ་བཅུ་ལྔ་  །  ༈    །  ཀ་ནཱ་མཱ་དི་བི་ཛ་ཧཱ་ར། ༄༅། །རྒྱ་གར་སྐད་དུ། བྲཧྨ་བེ་ད་སཱ་རཱཾ་ག་དྷ་ར་ཙ་ར་ཀ་ནཱ་མ། བོད་སྐད་དུ། ཚངས་པའི་རིག་བྱེད་རྒྱ་མཚོའི་༄༅།  །ཚངས་པའི་རིག་བྱེད་རྒྱ་མཚོའི་ཡན་ལག་འཛིན་པ་གཡོ་བ་ཅན་སོགས་བཞུགས་སོ།  །ཡན་ལག་འཛིན་པ་གཡོ་བ་ཅན་ཞེས་བྱ་བ། བཅོམ་ལྡན་འདས་མ་ཚངས་པའི་བུ་མོ་དབྱངས་ཅན་མ་ལ་ཕྱག་འཚལ་ལོ། །ཚངས་པ་ཆེན་པོའི་ཚེའི་རིག་བྱེད་རྒྱ་མཚོ་ལྟ་བུ་ལས་སྙིང་པོར་ཕྱུང་བ་འདི་ལ། ནད་ཀྱི་བརྟག་པ། ངོས་འཛིན། གསོ་ཐབས་སོགས་རིམ་པར་བསྟན་པ་ལས། ཐོག་མར་རྩ་ལ་བརྟག་པ་ནི། ནད་པ་སྐྱེས་པའི་ལག་པ་གཡས་དང་བུད་མེད་ཀྱི་གཡོན། མཐེ་བོང་རུས་འབུར་འོག་ནས་སོར་གང་བོར་བའི་ས་ནས་མཛུབ་མོ་གསུམ་གྱིས་མནན་པའི་དང་པོའི་འོག་མཁྲིས་པའི་ས། གཉིས་པ་བད་ཀན། གསུམ་པ་རླུང་གི་ས་ཡིན་ལ། དེ་ལ་ལེགས་པར་བརྟག་སྟེ། དེ་འང་མཁྲིས་པའི་རྩ་ནི་བྱ་རོག་ཁྲུང་ཁྲུང། སྦལ་པ་གསུམ་གྱི་འགྲོས་དང་འདྲ་སྟེ་འཕར་ཞིང་མཆོང་བའི་ཚུལ་ཅན་ནོ། །བད་ཀན་གྱི་རྩ་ནི་དང་པ། རྨ་བྱ། ཕུག་རོན་གསུམ་གྱི་འགྲོས་དང་འདྲ་སྟེ་སྦོམ་ཞིང་དལ་བའི་ཚུལ་ཅན་ནོ། །རླུང་གི་རྩ་ནི་སྦྲུལ་དང་ཛ་ལུ་ཀའི་འགྲོས་ལྟ་བུ་སྟེ་འཁྱོག་ལ་བེལ་བའི་ཚུལ་ཅན་ནོ། །འདུས་པའི་ནད་ཀྱི་རྩ་ནི་མཐིང་རིལ་གྱི་འགྲོས་དང་འདྲ་སྟེ་མྱུར་ལ་དྲག་པར་འཕར་ཞིང་སྡོད་ལ། དེ་རྣམས་མ་ནོར་བར་ཞིབ་ཏུ་བརྟག་པ་གཅེས་སོ། །ནད་དེ་རྣམས་སློང་བའི་དུས་དང་རྐྱེན་ནི། དེ་འང་མཁྲིས་པའི་ནད་ནི་དུས་ཐ་སྐར། སྨིན་དྲུག་ས་ག་སྣྲོན་གྱི་ཟླ་བ་རྣམས་ལ་ལྡང་ལ། རྐྱེན་ཟས་རོ་ཁ་བ་ཟོས་པ། མེ་དང་ཉི་མ་ཚ་སར་བསྡད་ཅིང་ཕྱིན་པ། དྲག་ཤུལ། སེམས་ཁྲོས་པ། ཆང་བཏུངས་ནས་མེ་དང་ཉི་མར་བསྡད་པ་སོགས་ཀྱིས་སློང། བད་ཀན་ལྡང་བའི་དུས་ནི། དབོ་དང་ནག་པའི་ཟླ་བ་ཡིན་ལ། ཟས་
