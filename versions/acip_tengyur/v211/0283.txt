སྒྲ་ལ་ཙ་ཏུ་ར་ན་ཊུ་རཱ་མ་ཞེས་པས་སུ་ཊི་ལ་ཨ་མ་གཞུག་གོ་ཨི་ཀོ་ཡ་ཎ་ཙི་ཞེས་པས་ཡ་ཎ་བྱའོ། །སཱ་བ་ཎ་ཊུ་ཧ་སུ་ར་ནུ་མ་ལ་གཞུག་གོ་་སུ་ཕྱིས་ལ་ཡང་དག་པར་སྦྱོར་བའི་མཐའི་ན་དབྱིའོ། ་དཽཿ་  ྾  ཊུ་ཧ་ཨི་ཏི་ཧ་ཡིག་ལ་ཏ་དྷ་དི་ཏི་དེ་ཤཱ་རྦ་ལཱ་དོ་ཨ་ཡིག་གོ་་སུ་པི་
ཁ་རི་ཙ་ར་ཛྷ་ལ་ད་ལ་ཏ་ཡིག་གོ་ཙ་ཡཿ་ཤ་རི་དྭི་ཏཱི་ཡ་ཞེས་པས་སྡེ་ཚན་གྱི་གཉིས་པ་ཡང་རུང་ངོ། ་ཡང་དག་པར་རྟོགས་པ་ལ་ཨཾ་སཽ་སཾ་བུ་དྷ་བ་ཞེས་པས་སཱ་བ་ཨཾ་གཞུག་གོ་ཨི་ཀོ་ཡ་ཎ་ཙི། ་སཱ་བ་ན་ཊུ་ཧ་ཞེས་པས་ནུ་ནའོ། །སུ་ཕྱིས་ལ་ཡང་དག་པར་སྦྱོར་
བའི་མཐའི་ན་དབྱིའོ། །ཨན་ཊཱ་ཎ། ཨན་ཊཱ་ཧཽ། ཨན་ཊཱ་ཧཿ་ཨན་ཊཱ་ཏཾ་ཨན་ཊཱ་ཧཽ་ཨ་ན་ཊུ་ཧཿ་ཨན་ཊཱ་ཧ་ཨ་ན་ཊུ་ཌྦྷྱཱ་ཨ་ན་ཊུ་བཧིཿ་ཨ་ན་ཊུ་ཏེ་ཨན་ཊུ་ཌྦྷྱཾ་ཨ་ན་ཊུ་ཌྦྷྱ་ཨན་ཊུ་ཏཿ། ཨན་ཊུ་ཌྦྷྱཱཾ། ཨན་ཊུ་བྷྱ་ཨན་ཊུ་ཧཿ། ཨན་ཊུ་ཧོཿ་ཨན་
ཊུ་ཧཱཾ། ཨན་ཊུ་ཧི་ཨན་ཊུ་ཧོཿ། ཨ་ན་ཊུ་ཊྶུ་ཨ་ན་ཊུ་ཐྶུ་ཧེ་ཨན་ནྭན་ཧི་ཨ་ནན་ཊྶ་ཧེ་ཏེ་ཨ་ན་ཊྶཱ་ཧཿ། མ་དཧུ་ལི་ཏའི་སྒྲ་ལས་ཁྱད་པར་ནི། ཧོ་ཌྷ་ཞེས་པས་ཚིག་མཐའ་ལ་བ་ལཱ་ཏོ་ཙ་ཏ་དྷ་དི་ཉི་དེ་ཤ་ལས་ཧ་ཡིག་ལ་ཌྷུ་ཉིད་དོ། །ཏསྱ་ཛྷ་ལོ་ཛ་སི་ལཾ་ཞེས་པས་ཊཱ་ཡིག་གོ་་ཏ་སྱ་བི་བ་རཱ་
མ་ཞེས་པས་ཊ་ཡིག་ཀྱང་ངོ། ་བུ་བི་ཁ་རི་ཙ་ར་ཛྷེ་ལ་ཞེས་པས་ཌྷ་ཡིག་གོ་་མ་དཧུ་ལ་ཊི་མ་དཧུ་ལི་ཊ་མ་དཧུ་ལི་ཧོ་མ་དཧུ་ལི་ཧཿ་མ་དཧུ་ལི་ཧཾ་མ་དཧུ་ལི་ཧེ། མ་དཧུ་ལི་ཧཿ་མ་དཧུ་ལི་ཧཱ་མ་དཧུ་ལི་ཌྦྷྱཱཾ། མ་དཧུ་ལིཌྦྷི་མ་དཧུ་ལི་ཧེ་མ་དཧུ་ལ་ཌྦྷྱཱཾ་མ་དཧུ་ལ་ཌྦྷྱ་མ་དཧུ་ལ་ཧིཿ་མ་དཧུ་ལཌྦྷྱཱཾ་མ་དཧུ་ལ་ཌྦྷྱ་མ་དཧུ་
ལི་ཧ་མ་དཧུ་ལི་ཧོཿ། མ་དཧུ་ལི་ཧཱི་མ་དཧུ་ལི་ཧི་མ་དཧུ་ལི་ཧོཿ་མ་དཧུ་ལཊྶུ། ཧེ་མ་དཧུ་ལིཊ་ཧེ་མ་དཧུ་ལི་ཊ་ཧེ་མ་དཧུ་ལི་ཧཽ་ཧེ་མ་དཧུ་ལི་ཧཿ། །མྲི་ཏྲ་དྲུ་ཧའི་སྒྲ་ལའང་དྲུཧ་མུ་སྦུ་ཧ་མི་ཏི་ས་ཧ་ཡིག་ལ་ཚིག་མཐོང་བ་ལ་ད་ཙ་བའི་གྷ་ཡིག་གོ་་ཏ་སྱ་ཛྷ་ལོ་ཛ་སི་ཞེས་པས་ག་ཡིག་གོ་་ཏ་སྱ་བའི་བ་རཱ་མ་ཞེས་པས་
