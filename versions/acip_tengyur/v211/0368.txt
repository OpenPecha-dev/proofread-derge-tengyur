ལས་སོ། །བྲཙྪ་ལས་སོ། །བྷ་ཛི་བཧུ་ཛི་བྷཉྫི་ཡ་ཛིཿ་ཡུ་ཛིཿ་རཉྫི་ནི་ཛི་བྀ་ཛྀ་སནྫི་སྲི་ཛིཿ་རུ་ཛི་སནྫཱི་ཏྱ་ཛི་མསྫི་བྲསྫི་རྣམས་ལས་སོ། །ཨ་དི་ཧ་དིཿ་ས་ད་སྐནྡ་ཤ་དིཿ་ཏུ་དིཿ་ཙྪི་དིཿ་ཀྵུ་དིཿ་ནུ་དི་བིཿ་དི་བ་དིཿ་བིན་ཁི་དིཿ་བི་དྱ་སཱ་དི་རྣམས་
