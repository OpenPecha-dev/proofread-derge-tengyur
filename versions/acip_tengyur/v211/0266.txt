སུ་ལ་སོགས་པ་རྣམས་སོ། །དེ་ལ་ཛ་ལ་མུ་ཙའི་སྒྲ་བཞིན་བྱའོ། །ཀུ་ཉིད་ཀྱིས་ཛའི་ག་ཉིད་ནི་ཁྱད་པར་རོ། །བྷཱུ་བཧུ་ཀཿ་བྷཱུ་བྷུག་བྷཱུ་བྷུཀྐ་བྷཱུ་བྷུགྒ་བྷཱུ་བཧུ་ཛཽཿ་བྷཱུ་བཧུ་ཛཿ་བྷཱུ་བཧུ་ཛན་བྷཱུ་བཧུ་ཛོཿ་བྷཱུ་བཧུ་ཛཿ་བྷཱུ་བཧུ་ཛཿ་བྷཱུ་བཧུ་གྦྷྱཱཾ་བྷཱུ་བཧུ་བཧིཿ་ཞེས་པ་ལ་སོགས་
པ་རྣམས་སོ། །དེ་བཞིན་དུཿ་ཧུ་ཏ་བཧུ་ཛཿ་བ་ལི་བཧུ་ཛཿ་ཀྲ་ཏུ་བཧུ་ཛཿ་ཨ་ཁུ་བཧུ་ཛཿ་ཝ་ཎི་བཧུ་ཛཿ་བཧི་བཧུ་ཛཿ་བིཥ་བུཙ་ལ་སོགས་པ་རྣམས་སོ། །བི་ཤྭཿ་སྲྀཛྷཱའི་སྒྲ་ནི། བྲཤྩ་ལ་སོགས་པ་ཉིད་ལས་ཥ་ཉིད་ཐོབ་པ་ལས། ནྱངྐུ་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པའི་ཚིག་ཕྲད་
