།འདི་ནི་ཉན་ཐོས་ཀྱི་རིགས་སོ་ཞེས་བྱ་བ་འདི་རུང་བ་མ་ཡིན་ནོ། འདིར་ལན་ནི་བསྟན་བཅོས་ལས། བརྟེན་པའི་ཆོས་ཀྱི་བྱེ་བྲག་གིས། །དེ་ཡིས་དབྱེ་བ་ཡོངས་སུ་བརྗོད། །ཅེས་བྱ་བ་གསུངས་སོ། །བརྟེན་པ་ནི་སངས་རྒྱས་ལ་སོགས་པའི་ཆོས་ཐ་དད་པ་ཡིན་ཏེ། དེ་བས་ན་དེའི་
དབྱེ་བས་རིགས་ཀྱི་དབྱེ་བ་ཡིན་ནོ། །ནམ་མཁའ་ཐ་དད་པ་མེད་ཀྱང་ལམ་གྱི་དབྱེ་བས་ཐ་དད་དེ། དབྱེ་བའི་ལམ་དང། བྱའི་ལམ་དང།ཉི་མའི་ལམ་བཞིན་ནོ། །དེ་ལྟར་འདི་ཐམས་ཅད་མདོར་བསྡུས་ནས་རབ་འབྱོར་གྱིས་ཞུས་པ་སྟེ། དེ་སྐད་ཅེས་གསོལ་བ་དང་ཞེས་བྱ་བ་ལ་སོགས་པས་ནི་དེ་ཉིད་
སྟོན་ཏོ། །བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཞེས་བགྱི་བ་དེ་ལ་ཞེས་བྱ་བ་ལ། ཐེག་པ་ཆེན་པོ་ཐམས་ཅད་དུ་ཞེས་བྱ་བ་ལྷག་མ་ཡིན་ནོ། །དེ་ལ་ཞེས་བྱ་བ་ནི་ཚིག་གཉིས་ལའོ། །བྱང་ཆུབ་སེམས་དཔའ་ཞེས་བགྱི་བ་ཞེས་བྱ་བ་ནི་གང་ལ་བྱང་ཆུབ་ཀྱི་སྒྲའི་ཚིག་གི་དོན་དུ་བརྗོད་ཅེས་བྱ་བ་དང་
ཚིག་གི་དོན་གང་ལགས་ཞེས་བྱ་བར་རྗེས་སུ་སྦྱར་རོ། །འདིར་རིགས་ཀྱི་དོན་གྱི་དངོས་པོ་གང་ཡིན་ཞེས་བྱ་བའི་དོན་ཏོ། །གང་ལ་བྱང་ཆུབ་སེམས་དཔའི་རིགས་ཡོད་པ་གང་ཡིན་པ་དེ་ནི་བྱང་ཆུབ་སེམས་དཔའོ། །དེ་ཡང་བདེ་བར་གཤེགས་པའི་བྱང་ཆུབ་ལ་ཡིན་གྱི་རིགས་མེད་པ་ནི་བྱང་ཆུབ་ཀྱི་རྟེན་
མ་ཡིན་ནོ། །བྱང་ཆུབ་སེམས་དཔའི་སྒྲ་དེ་ཉིད་ཀྱིས་བྱང་ཆུབ་ཀྱི་རིགས་བརྗོད་པར་བྱ་བའི་ཕྱིར་འཕགས་པ་རབ་འབྱོར་གྱིས་འདི་ཞུས་སོ། །བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ་ནི། རབ་འབྱོར་ཚིག་དོན་མེད་པ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཚིག་དོན་ཞེས་སྨོས་ཏེ། བྱང་ཆུབ་ཀྱི་རིགས་གང་ཡིན་པ་དེ་ཉིད་དོ། །
དེས་ན་བྱང་ཆུབ་སེམས་དཔའི་ཚིག་གི་དོན་ཞེས་བྱ་བ་གསུངས་ཏེ། དེ་ནི་ཚིག་གི་དོན་མེད་པ་དང། རིགས་ཀྱི་དངོས་པོར་མེད་པ་སྟེ་ཆོས་དབྱིངས་ཞེས་བྱ་བའི་དོན་ཏོ། །བྱང་ཆུབ་སེམས་དཔའི་འདིའི་ཐ་སྙད་ཀྱི་ཡུལ་ནི་གང་གི་ཕྱིར་གང་ཕུང་པོ་ལ་སོགས་པའི་སྟོང་པ་ཉིད་དང། ཆོས་ཉིད་དེ་ནི་དེའི་
ཆོས་ཀྱི་དབྱིངས་ཏེ། དེ་ཉིད་དག་ཅིང་མཐར་ཐུག་པ་ནི་བྱང་ཆུབ་ཅེས་བརྗོད་དོ། །དེ་བས་ན་རིགས་ཡིན་ན་ཡང་དངོས་པོར་ཡོད་པ་ནི་མ་ཡིན་ཏེ་དངོས་པོ་མེད་པའི་མཚན་ཉིད་ཀྱི་སྟོང་པ་ཉིད་དོ། །ཡང་ན་བྱང་ཆུབ་ཀྱི་རིགས་ཡིན་ཡང་རྟེན་ནི་མ་ཡིན་ཏེ་སྔ་ཕྱི་ཐ་མི་དད་པས་སོ། །རྟེན་དང་བརྟེན་པའི་དབྱེ་
