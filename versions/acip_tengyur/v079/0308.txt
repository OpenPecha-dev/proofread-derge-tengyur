བྱས་པས་ལན་གསུམ་དུ། །ཆུ་ལ་བྱི་ཟུལ་བྱས་ནས་ནི། །ཁྱོར་བ་དག་ནི་བརྒྱ་རྩ་བརྒྱད། །ཛཾ་བྷ་ལ་ལ་སྦྱིན་པར་བྱ། །རྡོ་རྗེ་འཛིན་པའི་དཀྱིལ་འཁོར་ལའང། །ཆུ་ནི་སྙིམ་པ་གསུམ་དག་ཉིད། །རེ་རེ་ཞིང་ནི་རང་སྔགས་ཀྱིས། །མཁས་པ་རྣམས་ཀྱིས་དབུལ་བར་བྱ། །དེ་ནས་སྙིམ་
པར་གསུམ་གྱིས་ནི། །ཡི་གེ་བརྒྱ་པས་ཚིམ་པར་བསྔོ། །ཡི་དགས་རྣམས་ལ་ཆུ་སྙིམ་ལྔ། །འཇིག་རྟེན་དབང་པོའི་སྔགས་ཀྱིས་ནི། །དེ་ནས་ལངས་ཏེ་བ་བརྒྱད་ནི། །ཁྱུ་མཆོག་མཐོན་པོ་དང་ལྡན་པ། །འཁོར་ལོའི་ཚུལ་དུ་བཞག་ནས་ནི། །དབུས་སུ་ཁྱུ་མཆོག་མཐོན་པོ་གཞག་དེ་ནས་
མཆོད་ཡོན་དག་གི་སྣོད། །བའི་རྣམ་ལྔས་དགང་བར་བྱ། །དའུར་བའི་ཆུན་པོས་བསངས་ནས་ནི། །མཁའ་འགྲ་དྲག་མོ་རོ་ལངས་མ། །དེ་བཞིན་གཞན་ཡང་གཏུམ་མོ་དང། །སེང་མོ་སྟག་མོ་འུག་པ་མོ། །དེ་བཞིན་གཞན་ཡང་ཅེ་སྤྱང་མའི། །ས་བོན་དེ་དག་ལ་དགོད་བྱ། །རྡོ་རྗེ་སེམས་དཔའི་ས་བོན་གྱིས། །
དབུས་སུ་ཁྱུ་མཆོག་བྱིན་རླབས་བྱ། །དེ་ལ་ས་བོན་ནི་འདི་རྣམས་ཡིན་ཏེ། །ཡེ་ཤེས་མཁའ་འགྲོ་མ་རྣམས་ཀྱི་ས་བོན་ནི་ཡི་གེ་ཧཱུཾ། །དྲག་མོའི་ས་བོན་ནི་ཡི་གེ་ཀྵུཾ། །རོ་ལངས་མའི་ས་བོན་ནི་ཡི་གེ་ཡུཾ། །གཏུམ་མོའི་ས་བོན་ནི་ཡི་གེ་ཧཱུཾ། སེང་མོའི་ས་བོན་ནི་ཡི་གེ་སྨྲྱུཾ། སྟག་མོའི་ས་བོན་ནི་ཡི་གེ་ཀྵྨྲྱུཾ་འུག་པ་མོའི་ས་བོན་ནི་ཡི་གེ་
ཡྨྲྱུཾ་ཅེ་སྤྱང་མའི་ས་བོན་ནི་ཡི་གེ་ཀྵྨྲྱུཾ། ས་བོན་བརྒྱད་པོ་དེ་དག་དང། །དགུ་པ་ཡེ་ཤེས་ས་བོན་ནི། །དེ་དག་སྙིང་གར་བཀོད་ནས་ནི། །དེ་ནས་གཏོར་མ་སྦྱིན་པར་བྱ། །འབྲས་ཆན་ཆང་བུ་རྣམས་ཀྱིས་ཀྱང། །སྣོད་ནི་དགུ་དག་དགང་བར་བྱ། །འོ་ཐུག་ལན་ཚྭ་དང་ལྡན་པ། །དེ་རྣམས་ཀྱི་ནི་ནང་དུ་བ། །དེ་རྣམས་ལོགས་
