ཡས་པ་རྣམས་སོ། །བགྱི་ཞེས་བྱ་བ་ནི་བསྒྲུབ་བོ་ཞེས་བྱ་བའོ། །དེ་ལྟར་བསགས་པའི་བསོད་ནམས་ཡོངས་སུ་བསྔོ་བ་བསྟན་ནས། ད་ནི་བཟང་པོ་སྤྱོད་པའི་སྨོན་ལམ་བཏོན་པ་ཡིས། །ཞེས་བྱ་བ་ལ་སོགས་པ་ཚིགས་སུ་བཅད་པ་གཅིག་གིས་བསྟན་པའི་ཡོངས་སུ་བསྔོ་བ་
མདོར་བསྡུས་ཏེ་སྟོན་ཏོ། །བཟང་པོ་སྤྱོད་པ་ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའ་ཀུན་དུ་བཟང་པོའོ། །དེའི་སྤྱོད་པ་ནི་བཟང་པོ་སྤྱོད་པ་སྟེ་ཡོན་ཏན་སྣ་ཚོགས་རྒྱ་ཆེ་བ་དུ་མ་དང་ལྡན་པའོ། །དེ་ཉིད་ཀྱི་སྨོན་པའི་བྱེ་བྲག་གི་རང་བཞིན་ཡིན་པས་སྨོན་ལམ་ཞེས་བྱའོ། །དེ་བཞིན་དུ་དེ་སྟོན་
པའི་སྒྲའི་རང་བཞིན་ལ་ཡང་སྨོན་ལམ་ཞེས་བརྗོད་པར་འགྱུར་ཏེ། འདུལ་བ་བཞིན་ནོ། །དེ་བཏོན་པ་ཞེས་བྱ་བ་ནི་གླེགས་བམ་དུ་བྱས་པ་ལ་སོགས་པ་བཏོན་པའོ། །དགེ་བ་ཅུང་ཟད་བདག་གིས་བསགས་པ་གང། །ཞེས་བྱ་བ་ནི་བདག་གིས་དགེ་བ་བསོད་ནམས་བསགས་པ་
གང་ཡིན་པ་སྟེ། ཀུན་དུ་བསགས་ཤིང་ཚོགས་སུ་བྱས་པ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེས་ནི་ཞེས་བྱ་བ་དགེ་བས་ཞེས་བྱ་བར་སྦྱར་རོ། །འགྲོ་བའི་ཞེས་བྱ་བ་ནི་སེམས་ཅན་གྱི་ཚོགས་ཟད་པར་བགྱིའོ། །སྨོན་ལམ་ཞེས་བྱ་བ་ནི་སྨོན་པའི་ཁྱད་པར་ཏེ། བཟང་པོ་སྤྱོད་པ་ཞེས་བྱ་
བ་འདི་ཉིད་དོ། །དགེ་བ་ཞེས་བྱ་བ་ནི་དཀར་པོ་སྟེ་དེ་ཁ་ན་མ་ཐོ་བ་མེད་པ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །སྐད་ཅིག་ཏུ་ཡང་ཐམས་ཅད་འབྱོར་པར་ཤོག་ཅེས་བྱ་བ་ནི་སེམས་ཅན་རེ་རེའི་སྐད་ཅིག་སྐད་ཅིག་ལ་འབྱོར་པ་གྲུབ་པར་གྱུར་ཅིག་ཅེས་བྱ་བ་ཡིན་ཏེ། ཐམས་ཅད་ཅེས་བྱ་བ་ནི་
མ་ལུས་པའོ། །བཟང་སྤྱོད་ཡོངས་སུ་བསྔོས་པའི་བསོད་ནམས་ནི། །ཞེས་བྱ་བ་ལ་སོགས་པ་འདི་མན་ཆད་ཀྱིས་ནི་བཏོན་པ་དང། བསྒོམ་པས་བསྡུས་པའི་སྨོན་ལམ་བྱེ་བྲག་མེད་པར་ཡོངས་སུ་བསྔོ་བ་བསྟོན་ཏེ། ཀུན་དུ་བཟང་པོའི་སྤྱོད་པ་ཐོག་མ་དང་ཐ་མ་མེད་པ་ཟབ་ཅིང་རྒྱ་
