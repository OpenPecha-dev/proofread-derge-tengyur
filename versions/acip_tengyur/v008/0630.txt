རྒྱ་གར་གྱི་མཁན་པོ་རཏྣ་ཤྲཱི་མི་ཏྲ་དང། དགེ་སློང་ཤཱཀྱ་ཡེ་ཤེས་ཀྱིས་བསྒྱུར་བ།།  ༄༅། །བརྡ་ངེས་པར་གཟུང་བ་བཞུགས། ། །དཔལ་དགྱེས་པའི་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །བརྡ་ངེས་པར་གཟུང་བར་བྱ་སྟེ། དང་པོ་སྒྲ་
ཇི་ལྟ་བ་བཞིན་དུ་ཤེས་པར་བྱ་ཞིང། དེ་ནས་ཡང་མན་དག་གི་རིམ་པ་བཞིན་དུ་ཤེས་པར་བྱའོ། །བརྡ་ཞེས་བྱ་བ་ནི་ལུས་དང་ངག་གི་སྤྱོད་པའོ། །ལུས་ནི་རྡོ་རྗེ་དང་པདྨ་སྟེ། དེའི་སྤྱོད་པའོ། །ངག་ནི་ཡི་གེ་སྟེ། དེས་མཁའ་གསང་བྱིན་གྱིས་བརླབས་པ་ནི། དེའི་སྤྱོད་པ་
སྟེ༑ དེ་ལྟ་བུའི་བརྡ་ཤེས་ན་སྤུན་དང་སྲིང་མོར་ངེས་སོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །སྤུན་ཞེས་བྱ་བ་ནི་དམ་ཚིག་གཅིག་པའི་སྐྱེས་བུའོ། །སྲིང་མོ་ཞེས་བྱ་བ་ནི་དམ་ཚིག་གཅིག་པའི་བུད་མེད་དོ། །ཡང་ན་སྤུན་དང་སྲིང་མོ་ཞེས་བྱ་བ་ནི་དཔའ་བོ་དང་རྣལ་འབྱོར་མའོ། །ཡང་ན་སྤུན་
དང་སྲིང་མོ་ཞེས་བྱ་བ་ནི་ཐབས་དང་ཤེས་རབ་བོ། །ཡང་ན་སྤུན་དང་སྲིང་མོ་ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་དང་པདྨ་སྟེ། དེ་ཤེས་པ་ནི་ཐེ་ཚོམ་མེད་པའོ། །སོར་མོ་གཅིག་སྟོན་པ་ཞེས་བྱ་བ་ནི་གཡོན་གྱི་མཐེའུ་ཆུང་སྟོན་པ་སྟེ། ཟབ་མོའི་དོན་དང་ལྡན་རྣམ་ཞེས་བྱ་བའོ། །གཉིས་ཀྱིས་ཞེས་
བྱ་བ་ནི་བདག་ངེས་པའི་དོན་དང་ལྡན་པ་ཞེས་བྱ་བའོ། །གཡོན་པའི་ཞེས་བྱ་བ་ནི་ཤེས་རབ་སྟེ་པདྨའོ། །མཐེ་བོ་ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་པདྨ་བཅངས་པ་སྟེ་སྙོམས་པར་ཞུགས་པའོ། །བསྙུན་གྱི་ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་དང་པདྨ་ལས་སྐྱེས་པའི་བདེ་བ་དང་ལྡན་པ་ཞེས་བྱའོ། །སྲིན་
ལག་ཅེས་བྱ་བ་ནི་རྡོ་རྗེ་སྟེ། པདྨ་རྒྱས་པར་བྱེད་པའི་ཕྱིར་རོ། །སྟེར་བ་ལ་ཞེས་བྱ་བ་ནི་དེ་བསྟན་པ་ལའོ། །མཐེ་ཆུང་ཞེས་བྱ་བ་ནི་འོག་གི་ཆ་ཡིན་པའི་ཕྱིར་པདྨ་སྟེ། དེ་བསྟན་པར་བྱ་ཞེའོ། །གུང་མོ་ཞེས་བྱ་བ་ནི་ཉན་ཐོས་ལ་སོགས་པའི་མྱ་ངན་ལས་འདས་པའི་མཐའ་
དང་འཁོར་བའི་མཐའ་གཉིས་དང་བྲལ་བར་བྱེད་པའི་ཕྱག་རྒྱ་ཆེན་པོའི་རིག་པའོ། །མཛུབ་མོ་ཞེས་བྱ་བ་ནི་བསྡེབས་པར་བྱེད་པའི་ནུས་པ་དང་ལྡན་པའི་ཕྱིར་རྡོ་རྗེ་སྟེ། དེ་བསྟན་པར་བྱ་ཞེས་བྱ་བའོ། །སྲིན་ལག་སྟོན་པ་ལ་ཞེས་བྱ་བ་ནི་གུས་པ་དང་ལྡན་མ་ཞེས་བྱ་བའོ། །མགྲིན་
