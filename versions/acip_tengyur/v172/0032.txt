༄༅། །བར༑ །ཇི་ལྟར་སྨན་དཔྱད་ལས་བཤད་བཞིན། །སྡུག་བསྔལ་གྱིས་ཀྱང་མི་ཕན་པ། །བསལ་བར་བྱ་བ་ཅི་ཞིག་འགལ། །ཆོས་རྣམས་སྔོན་དུ་ཡིད་འགྲོ་ཞིང། །ཡིད་ནི་གཙོ་བོ་ཞེས་གྲགས་པས། །སྡུག་བསྔལ་གྱིས་ཀྱང་ཕན་ཡིད་ཀྱིས། །ཕན་པར་བྱེད་པ་ཅིས་མི་

རྒྱས་ཀྱིས། །གང་ལས་བསྟན་པའི་ཐེག་ཆེན་དེ། །མི་ཤེས་ལྡོངས་རྣམས་མི་བཟོད་དོ། །ཡོན་ཏན་མཁའ་ལྟར་བསམ་ཡས་པས། །རྒྱལ་པོའི་ཡོན་ཏན་བསམ་ཡས་གསུངས། །གང་ཕྱིར་སངས་རྒྱས་བདག་ཉིད་ཆེ། །ཐེག་ཆེན་ལས་བཤད་འདིར་བཟོད་ཀྱི། །འཕགས་པ་
