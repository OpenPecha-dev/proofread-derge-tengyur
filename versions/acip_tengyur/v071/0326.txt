མཱ་ར་ཎ་ཞེས་བྱ་བའི་བར་གྱིས་ནི་དེ་ཁོ་ན་ཉིད་ཀྱི་ཡེ་ཤེས་བཞི་བསྐུལ་བ་བསྟན་ཏོ། །བཛྲ་བི་དཱ་ར་ཎཱ་ར་ཡ་སཱཧཱ་ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། ཀཱི་ལི་ཀཱི་ལ་ཡ་ཞེས་བྱ་བའི་བར་གྱིས་ནི་དེ་ཁོ་ན་ཉིད་ཀྱི་རྒྱུའི་རང་བཞིན་བསྐུལ་བ་བསྟན་ཏོ། །ཙུ་རུ་ཙུ་རུ་ཞེས་བྱ་བ་ནས་བརྩམས་
ཏོ༑ །བཛྲ་པྲ་བྷཉྫ་ནཱ་ཡ་སཱཧཱ། ཞེས་བྱ་བའི་བར་གྱིས་ནི་ཁྲོ་བོ་བཞིའི་སྙིང་པོའི་སྒོ་ནས་བསྐུལ་བ་བསྟན་ཏོ། །མ་ཏིསྠ་ར་བཛྲ་ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། ཨ་མུ་ཀ་མཱ་ར་ཡ་ཕཊ་ཅེས་བྱ་བའི་བར་གྱིས་ནི་དེ་དག་ཉིད་ཀྱི་གསུང་གི་སྒོ་ནས་ཕྲིན་ལས་མཛོད་ཅིག་ཅེས་བསྐུལ་བ་བསྟན་
ཏོ༑ ན་མཿ་ས་མནྟ་བཛྲ་ཎཱཾ་ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། བཛྲ་ཨདྐུ་ཤ་ཛཱ་ལཱ་ཡ་སཱཧཱ། །ཞེས་བྱ་བའི་བར་གྱིས་ནི་ཁྲོ་བོ་རྣམས་ཀྱི་སྐུ་གསུང་ཐུགས་དག་གིས་བཅོལ་བའི་ཕྲིན་ལས་མཛད་ཅིག་ཅེས་བསྐུལ་བ་བསྟན་ཏོ། །ན་མོ་རཏྣ་ཏྲ་ཡཱ་ཡ་ཞེས་པ་ནས་བརྩམས་ཏེ། བྷིནྡ་བྷིནྡ་བཛྲ་ཧཱུཾ་
ཕཊ་ཅེས་བྱ་བའི་བར་གྱིས་ནི་བཅོམ་ལྡན་འདས་ཀྱི་སྙིང་པོའི་སྒོ་ནས་བསྐུལ་བ་བསྟན་ཏོ། །ན་མཙཎྜ་བཛྲ་ཀྲོ་དྷཱ་ཡ་ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། ཨ་མྲྀ་ཏེ་ཧཱུཾ་ཞེས་བྱ་བའི་བར་གྱིས་ནི་དེ་ཁོ་ན་ཉིད་ཀྱི་ཉེ་བའི་སྙིང་པོའི་སྒོ་ནས་བསྐུལ་བ་བསྟན་ཏོ། །དེ་དག་གིས་ནི་ཕྲིན་ལས་ལ་བསྐུལ་བའི་
བསྡུས་པའི་དོན་བསྟན་ཏོ། །སྡིག་པ་ཐམས་ཅད་བྱང་བྱས་ནས། །ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། དཔལ་ཀུན་གྱིས་ནི་ལེགས་པར་བརྒྱན། །ཞེས་བྱ་བའི་བར་གྱིས་ནི་གཟུངས་འདིའི་དཀྱིལ་འཁོར་སྒྲུབ་པར་བསྟན་ཏོ། །སེམས་ཅན་དབང་པོ་ཉམས་པ་དང། །ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ་
རྨི་ལམ་སྡིག་པ་མཐོང་ན་ཡང། །ཞེས་བྱ་བའི་བར་གྱིས་ནི་ཇི་ལྟ་བ་བཞིན་དུ་བསྒྲུབས་པས་ཚེ་འདི་དག་དང་གཞན་ལ་གནོད་པར་གྱུར་པའི་མི་མཐུན་པའི་ཕྱོགས་འདི་ལྟ་བུ་དག་ཡོད་ན་ཡང་སྤང་པར་འགྱུར་རོ་ཞེས་བྱ་བར་བསྟན་ཏོ། །དེས་ནི་རབ་བཀྲུས་གཙང་མ་ཡིན། །ཞེས་བྱ་བ་ནས་
བརྩམས་ཏེ། མདོ་སྡེ་འདི་ནི་ཉན་བྱེད་ན། །ཞེས་བྱ་བའི་བར་གྱིས་ནི་མདོ་སྡེ་འདི་ཉན་པའི་ཐབས་བསྟན་ཏོ། །མདོ་སྡེ་འདི་ཡི་གཟི་བརྗིད་ཀྱིས། །ཞེས་བྱ་བ་ནས་བརྩམས་ཏེ། སྡིག་པ་ཀུན་ལས་རྣམ་པར་ཐར། །ཞེས་བྱ་བའི་བར་གྱིས་ནི་མཉན་པའི་ཕན་ཡོན་བསྟན་ཏོ། །ནོར་བུ་
