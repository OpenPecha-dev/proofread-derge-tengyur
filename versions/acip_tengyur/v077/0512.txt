།གདན་གྱི་ཕྱག་རྒྱའི་སྔགས་ནི། ཨོཾ་པདྨོ་བྷ་བ་ནི་ཥ་ན་ཎཱ་ཡ་སརྦ་དེ་བཏྟ་སཱཧཱ། དེ་ནས་འཕང་བར་བྱ་སྟེ། ཕན་ཚུན་དུ་ཁུ་ཚུར་བཅིངས་ནས་མཐེབ་ཆེན་རབ་ཏུ་བརྐྱང་སྟེ། ཁུ་ཚུར་གཡོན་པ་སྙིང་གར་བཞག་སྟེ། ཁུ་ཚུར་གཡས་པ་མཚོན་ཆར་འཕེན་པའི་ཚུལ་དུ་བྱའོ། །ཨོཾ་ནཱ་ཤ་ཡ་
སརྦ་དུཥྚཱ་ན་ད་ཧ་པ་ཙ་དུཥྚཱ་ན་བྷསྨཱི་ཀུ་རུ་ཧཱུཾ་ཕཊ། དེ་ནས་སྔགས་འདིས་ཕྱོགས་བཅིང་བར་བྱ་སྟེ། ཕན་ཚུན་དུ་ཁུ་ཚུར་སོ་སོར་བཅིངས་ཏེ། ལག་པ་གཡོན་པའི་སྡིགས་མཛུབ་རབ་ཏུ་བརྐྱང་སྟེ་གཡས་པའི་དཔུང་པའི་རྩ་བར་གཞག་པར་བྱའོ། །གཡས་པའི་མཐེབ་ཆེན་གྱིས་མཐེའུ་ཆུང་གི་སེན་མོ་
མནན་ཏེ་སོར་མོ་ལྷག་མ་རྣམས་རབ་ཏུ་བརྐྱང་ནས་ཕྱོགས་བཅུར་བསྐོར་བར་བྱའོ། །ཨོཾ་བཛྲ་ཀྲོ་དྷ་མ་ཧཱ་ཙཎྜ། བནྡྷ་བནྡྷ། ཤ་དི་ཤཱཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ། དེ་ནས་འདི་བརྗོད་ནས་རྡོ་རྗེ་གསོར་བར་བྱ་སྟེ། ཁྲོ་བོའི་དངོས་གྲུབ་དངོས་གྲུབ་ཆེ། །དངོས་གྲུབ་རྒྱ་ཆེན་དམ་ཚིག་བསྟན། །ལྷ་རྣམས་ཀུན་
གྱི་དངོས་གྲུབ་དང། །བླ་མེད་དངོས་གྲུབ་བདག་མྱུར་མཛོད། །ཁྲོ་བོའི་རྒྱལ་པོ་རྩ་བའི་ཕྱག་རྒྱ་བཅིངས་ནས་ཧཱུཾ་བཛྲ་ཕཊ། བལྟ་ཞིང་དམ་ཚིག་སྦྱིན་པར་བྱའོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ལ་མཆོད་པར་བྱས་ནས་ཁྲོ་བོ་དབབ་པའི་ཕྱག་རྒྱ་བཅིངས་ནས་རྡོ་རྗེ་ཁྲོ་བོ་རང་གི་ལུས་ལ་རབ་ཏུ་
གཞུག་པར་བྱའོ། །ཨོཾ་པྲ་བི་ཤ་ཀྲོ་དྷ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཨཱཛ་ལཱ་མཱ་ལཱ་ཀུ་ལ་བཧི་ཤ་ཎ་བཛྲ་ཨཱརང་གི་ལུས་ཀྱི་ནང་དུ་བསྒོམས་ཏེ་གོ་ཆ་ལ་སོགས་པ་བྱས་ནས་ཐམས་ཅད་འདིར་རྫོགས་པར་བྱ་བའི་དོན་དུ་སྙིམ་པས་ཐལ་མོ་སྦྱར་ནས་རྫོགས་པར་བྱ་བའི་སྔགས་བརྗོད་པར་བྱའོ། །ཨོཾ་སིདྡྷ་བཛྲ་ཨ་པཱུ་ར་ཡ་
ཨ་པཱུ་ར་ཡ་ཧཱུཾ་ཕཊ། འདིས་ལྷག་ཆད་ཐམས་ཅད་ཡོངས་སུ་ཁ་བསྐོང་བར་འགྱུར་རོ། །དེ་ནས་ཁྲོ་བོའི་ང་རྒྱལ་བསྐྱེད་པས་དངོས་གྲུབ་ཐམས་ཅད་འགུགས་པར་བྱེད་དོ། །སྔགས་འདིས་ཕན་ཚུན་དུ་ཁུ་ཚུར་བཅིངས་ཏེ་མཐེའུ་ཆུང་གཉིས་པོ་འཁྱུད་པར་བྱའོ། །མཛུབ་མོ་རབ་ཏུ་བརྐྱང་སྟེ་བཀུག་
ནས་ཅུང་ཟད་བསྐྱོད་པར་བྱའོ། །ཨོཾ་བཛྲ་རོ་ཥ་མ་ཧཱ་ཀྲོ་དྷ་སིདྷྱཱ་ཀ་ར་ཧཱུཾ་ཛདེ་ནས་རང་གི་སྙིང་གར་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་ཡི་གེ་རྣམས་བུང་བ་ནག་པོའི་ཕྲེང་མེ་འབར་བ་བཞིན་དུ་བསམ་པར་བྱའོ། །ཧཱུཾ་བཛྲ་ཕཊ། རང་གི་བ་སྤུའི་བུ་ག་ནས་ཁྲོ་བོའི་སྐུ་སྤྲོ་བ་དང་བསྡུ་བར་བྱེད་པར་བསམ་པར་བྱའོ། །དེའི་ཡི་
