༄༅། །ཉིད་བདུད་ལས་རྒྱལ་བ་ཡིན། །ཞེས་པ་དང། ཡང། རང་སེམས་སོ་སོར་རིག་པ་ནི། །བྱང་ཆུབ་སེམས་མཆོག་རྡོ་རྗེ་ཡིན། །དེ་ཉིད་དེ་ལས་དེ་སངས་རྒྱས། །རང་བྱིན་བརླབ་མཆོག་བསྒོམ་ཕྱིར་རོ། །ཞེས་སོ། །འདིས་ནི་དབུ་ཞབས་ཀྱི་ཐེག་པ་རྣམ་པར་འབྱེད་པ་སྟེ། ཁྱུང་
དང་བྱ་ལག་གི་དཔེས་བཤད་དོ། །ད་ནི་ཕན་ཡོན་གྱི་ཆེ་བ་བསྟན་པའི་ཕྱིར། །བྱང་ཆུབ་སེམས་འདི་རྡོ་རྗེ་སྟེ། །སངས་རྒྱས་ཀུན་གྱི་བདག་ཉིད་ཡིན། །དེ་བས་ཐམས་ཅད་བདག་སྦྱོར་བས། །སངས་རྒྱས་ཐམས་ཅད་ཐོབ་པར་འགྱུར། །ཞེས་ཏེ། །ཟབ་པ་དང་རྒྱ་ཆེ་བའི་སྒོ་ནས་
བྱང་ཆུབ་ཀྱི་སེམས་རྣམ་པ་བཞི་པོ་འདི་ནི་རྡོ་རྗེ་ལྟ་བུ་སྟེ། སངས་རྒྱས་ཀུན་གྱི་བདག་ཉིད་ཡིན་ནོ། །དེ་ལྟ་བས་ན་ལྟ་བ་དང་ཏིང་ངེ་འཛིན་དང་སྤྱོད་པ་ལ་སོགས་པའི་དོན་རྣམས་ཤེས་ཤིང་ཉམས་སུ་བླངས་ན་སྦྱོར་བའི་སྟོབས་ཀྱིས་ཐོབ་པར་འགྱུར་རོ། །དེ་ཡང་རྩ་བའི་རྒྱུད་
ལས༑ ལྟ་སྤྱོད་ཏིང་འཛིན་བསྒྲུབ་པ་དང། །དཀྱིལ་འཁོར་དབང་བསྐུར་དམ་ཚིག་ཚོགས། །ཕྲིན་ལས་ཐམས་ཅད་བདག་སྦྱོར་བས། །སངས་རྒྱས་ཐམས་ཅད་ཐོབ་པར་འགྱུར། །ཞེས་པ་དང། རྟོག་པ་བསྡུས་པ་ལས་ཀྱང། །གང་གི་རྟོག་པ་མཆོག་འདི་སྒྲུབ་པར་བྱེད་པ་ལ། །བདུད་ཉིད་ཉེ་
བར་མི་འགྲོ་མི་བརྟེན་ཉིད་ཀྱང་མིན། །རྟོག་པའི་ཚོགས་བཅས་བདག་གིས་གཞོམ་མཛད་དེ། །མྱུར་བ་མཆོག་གི་མཆོག་ཉིད་འགྲུབ་ཕྱིར་བརྟེན་པར་མི་བྱེད་དེ། །གང་གིས་བདག་གི་སྙིང་པོ་བདག་ལ་རྟག་ཏུ་འགྲུབ་བྱེད་ཅིང། །མཆོག་ཆེན་དངོས་གྲུབ་དང་ནི་རྟོག་པ་བདེ་ཆེན་དང། །
ཆགས་དང་ཤིན་ཏུ་དགའ་དང་བརྟན་དང་རྒྱས། །ཡིག་ཆེན་ཡང་མི་བརྟེན་པར་བདག་གིས་འགྱུར་བར་བྱེད། །ཅེས་སོ། །དེས་ནི་ལེའུ་གཉིས་པའི་སྨན་དང་ལེའུ་སོ་སོའི་ཁྱད་པར་གྱི་ཕན་ཡོན་བསྡུས་ཏེ། རུས་སྦལ་གྱི་དཔེས་བཤད་དོ། །ད་ནི་མཉམ་སྤྱོར་གྱི་སྒོ་ནས་འབྲས་བུ་ཕྲིན་
ལས་དང་བཅས་པ་བསྟན་པའི་ཕྱིར། །བདག་ཉིད་སངས་རྒྱས་ཐམས་ཅད་དང། དཔའ་བོ་ཐམས་ཅད་བདག་ཡིན་ནོ། །དེ་བས་བདག་ཉིད་ལྷར་སྦྱོར་བས། །བདག་ཉིད་རབ་ཏུ་བསྒྲུབ་པར་བྱ། །ཞེས་ཏེ། འབྲེལ་བའི་ཚུལ་དུ་མཇུག་བསྡུ་བའི་ཕྱིར། །ཚིག་དང་པོ་གཉིས་ཀྱིས་ནི་
