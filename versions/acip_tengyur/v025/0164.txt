རྡོ་རྗེ་དང། རལ་གྲི་དང། དྲིལ་བུ་དང། འཁོར་ལོ་དང། ཞགས་པ་བསྣམས་པ། རྡོ་རྗེ་བདུད་རྩི་ཅོད་པན་དུ་བྱས་པའོ། །མངོན་སྤྱོད་ལ་སྒྲོལ་མ་ཉིད་རབ་ཏུ་བརྟག་པར་བྱ་སྟེ། རྣམ་པར་འགྱུར་ཞིང་ཁ་དོག་ནག་པོའོ། །ཞལ་གསུམ་ནི་ནག་པོ་དང། དཀར་པོ་དང། ཉི་མ་ལྟ་བུའོ། །
ཕྱག་དྲུག་ནི་རལ་གྲི་དང། འཁོར་ལོ་དང། དྲིལ་བུ་དང། ཞག་པ་དང། ལྕགས་ཀྱུ་དང། རྡོ་རྗེ་འཛིན་པ། རྡོ་རྗེ་བདུད་རྩིའི་ཅོད་པན་ནོ། །རྒྱས་པ་ལ་གར་མཁན་མའམ། རྒྱལ་རིགས་མ་ནི་གོས་དཀར་མོ་གཟུགས་ཏེ་ཁ་དོག་སེར་པོའོ། །ཞལ་གསུམ་ལ་སེར་པོ་དང། དཀར་པོ་དང། ཉི་
མའི་མདོག་གོ། །ཕྱག་དྲུག་ན་རྡོ་རྗེ་དང། འཁོར་ལོ་དང། རལ་གྲི་དང། དྲིལ་བུ་དང། ཞགས་པ་དང། ལྕགས་ཀྱུ་རྣམས་འཛིན་པའོ། །རྡོ་རྗེ་བདུད་རྩི་པུ་སྟིས་བརྒྱན་ཅིང་མཚན་པར་ཤེས་པར་བྱའོ། །དེ་ལྟར་ལས་ཀྱི་དབྱེ་བས་གཞན་ཡང་བསྒོམ་པར་བྱའོ། །མཆོད་པ་རྣམ་པ་ལྔས་
མངོན་པར་མཆོད་ནས་སྙིམ་པས་མེ་ཏོག་འཕགས་ཞེས་པ་སྟེ། དེ་ཉིད་གསུངས་པ། མེ་ཏོག་འཕགས་ཞེས་པ་ལ་སོགས་པའོ། །དེ་ལ་བྷ་གའི་དབུས་སུ་མེ་ཏོག་ནི་ཁུ་བ་སྟེ་གཞུག་པར་བྱའོ། །དེ་ལྟར་སྦྱིན་སྲེག་གི་ས་གཞི་ལེགས་པར་བྱས་པར་འགྱུར་རོ། །དཀྱིལ་འཁོར་གྱི་ཆོ་ག་
ཡང་ཐུན་མོང་བ་སྟེ་བུ་མོ་གཞན་ཡང་ཤེས་པར་བྱའོ། །དཀྱིལ་འཁོར་གྱི་ཁོངས་སུ་གཏོགས་པའི་སྦྱིན་སྲེག་གི་ཆོ་ག་དེ་བསྟན་པའི་ཆེད་དུ་གསང་བའི་ཆོས་ཀྱི་དཀྱིལ་འཁོར་ཡང་བསྡུས་པ་སྟེ། འདིས་ནི་འདིའོ། །དཀྱིལ་འཁོར་ནི་རྣམ་པ་གཉིས་ཏེ་ཕྱི་ནི་རྡུལ་ཚོན་གྱི་དཀྱིལ་འཁོར་རོ། །གསང་
བའི་ཆོས་ཀྱི་དཀྱིལ་འཁོར་གྱི་ངེས་པའོ། །དེ་ལྟར་རྣལ་འབྱོར་གྱི་གནས་ཀྱི་སར་གྱུར་ནས། ས་ཕྱོགས་དག་ཏུ་ཐབ་ཁུང་བྱ། །ཞེས་པ་སྟེ། དེ་ཉིད་གསུངས་པ། དེ་ནས་ཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བྲི་ཞེས་པ་ནི་རྣམ་པར་དགོད་པའོ། །དེ་ཡང་ཚད་དང་ཚད་ཀྱི་རྣམ་པ་སྟེ་ཆུ་དང་ཞེང་གི་
མཚན་ཉིད་དག་གོ། །དེ་ལ་ཚད་དང་རྣམ་པ་གསུངས་པ། ཞི་བའི་ཞེས་པའོ། །ཞི་བའི་ཐབ་ཁུང་ནི་ཟླུམ་པོའི་རྣམ་པ་ཁྲུ་གང་གི་ཚད་དུ་ཐིག་གདབ་པར་བྱའོ། །ཐིག་གདབ་ནས་ཟབས་སུ་བརྐོ་བ་ནི་ཁྲུ་ཕྱེད་ས་ལ་ཞེས་པ་ནི་འོག་གི་ཟབས་སུ་བརྐ་བའོ། །ཁ་དོག་དཀར་པོས་བྱུག་པར་བྱ། །
