སྐྲའི་ཁྱོན་གཅིག་ལས་མི་འདའ་ཞིང་སྐྲའི་ཁྱོན་ཐམས་ཅད་དུ་སྐྱེ་བ་སྐྱ་བ་གཞོན་ནུ་རོལ་པ་ཕོ་བྲང་ནང་དུ་གནས་པ་མངོན་པར་འབྱུང་བ་དཀའ་བ་སྤྱོད་པ་བྱང་ཆུབ་སྙིང་པོར་ཉེ་བར་འགྲོ་བ་བདུད་འདུལ་བ་མངོན་པར་རྫོགས་པར་བྱང་ཆུབ་པ་བསྐུལ་བ་ཆོས་ཀྱི་འཁོར་ལོ་ཆེན་པོ་བསྐོར་བ་མྱ་ངན་
ལས་འདའ་བ་ཆེན་པོ་ཉེ་བར་བསྟན་པའི་ཕྱིར་དང། སངས་རྒྱས་ཀྱི་ཞིང་ཆེན་པོ་མཐུ་སྟོབས་ཡེ་ཤེས་ཀྱི་རྗེས་སུ་འགྲོ་བའི་ཕྱིར་དང། ཆོས་ཀྱི་དབྱིངས་ཡངས་པ་ནམ་མཁའི་མཐས་གཏུགས་པ་ཕྱི་མའི་མཐའི་མུར་ཐུག་པར་བསྐལ་པའི་གྲངས་ཐམས་ཅད་དུ་སངས་རྒྱས་འབྱུང་བའི་གྲངས་ཆད་པ་མེད་པའི་ཕྱིར་
དང༑ ཡེ་ཤེས་ཆེན་པོ་མངོན་པར་བསྒྲུབ་པའི་ཕྱིར་སྨོན་ལམ་ཆེན་པོ་བཅུ་པ་མངོན་པར་སྒྲུབ་པའོ། །སྨོན་ལམ་ཆེན་པོ་དེ་རྣམས་ནི་འབད་རྩོལ་ཆེན་པོ་བཅུའོ། །དེ་ལ་སོགས་པ་སྨོན་ལམ་རེ་རེ་ལ་འཁོར་ནི་སོ་སོར་ཡོངས་སུ་སྦྱོང་བར་བྱེད་པ་གྲངས་མེད་ཁྲག་ཁྲིག་སྟོང་ཕྲག་བརྒྱ་དང་གང་གིས་བྱང་ཆུབ་སེམས་
དཔའ་མངོན་པར་སྒྲུབ་པར་བྱེད་པ་ཐོབ་པ་འདི་ནི་སྨོན་ལམ་གྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །སྟོབས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་གཉིས་ཏེ། འདི་ལྟར། སོ་སོར་རྟོག་པའི་སྟོབས་དང། བསྒོམ་པའི་སྟོབས་སོ། །ཡང་སྟོབས་བཅུ་གཉིས་ཏེ། འདི་ལྟར། བསམ་པའི་སྟོབས་དང། ལྷག་པའི་བསམ་པའི་སྟོབས་དང་
གཟུངས་ཀྱི་སྟོབས་དང། ཏིང་ངེ་འཛིན་གྱི་སྟོབས་དང། ཕུན་སུམ་ཚོགས་པའི་སྟོབས་དང། དབང་གི་སྟོབས་དང། སྤོབས་པའི་སྟོབས་དང། སྨོན་ལམ་གྱི་སྟོབས་དང། ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྟོབས་དང། བྱམས་པ་ཆེན་པོའི་སྟོབས་དང། སྙིང་རྗེ་ཆེན་པོའི་སྟོབས་དང། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་
ཀྱི་བྱིན་གྱི་རླབས་ཀྱི་སྟོབས་སོ་ཞེས་པ་ནི་སྟོབས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །དེ་བཞིན་གཤེགས་པའི་ངག་མ་ལུས་པ་དགོངས་པ་དང་བཅས་པ་ངེས་པར་རྟོགས་པ་ནི་ཡེ་ཤེས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའོ། །པཎྜི་ཏ་ཆེན་པོ་སྟོབས་བཅུ་དཔལ་བཤེས་གཉེན་གྱིས་བསྡུས་པ་འདུས་བྱས་དང་འདུས་མ་བྱས་རྣམ་པར་
ངེས་པ་ལས་ཕ་རོལ་ཏུ་ཕྱིན་པ་རྣམ་པར་ངེས་པ་ཞེས་བྱ་བ་ལེའུ་ཉི་ཤུ་རྩ་གཉིས་པའོ།། །།བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་སུམ་ཅུ་རྩ་བདུན་ནི་ཕུང་པོ་བདུན་ཏེ། འདི་ལྟར་དྲན་པ་ཉེ་བར་གཞག་པ་དང། ཡང་དག་པའི་སྤོང་བ་དང། རྫུ་འཕྲུལ་གྱི་རྐང་པ་དང། དབང་པོ་དང། སྟོབས་དང། བྱང་ཆུབ་ཀྱི་ཡན་ལག་དང་
