༄༅། །སྒྲུབ་པར་བྱེད་པ་སྐུར་པ་འདེབས་པའི་རྣམ་པར་རྟོག་པ་ཐམས་ཅད་དུ་སྟེ། རྣམ་པ་ཐམས་ཅད་དུ་སླར་འགོག་པར་བྱེད་པ་སྟོང་པ་ཉིད་ཀྱིས་ནི་མི་སྟོང་ཞེས་བྱ་བ་ལ་སོགས་པ་དེ་ལྟ་བུའི་རང་བཞིན་གྱིས་སེལ་བ་ཞེས་བྱ་བ་ཟློག་པར་བྱེད་པ་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་ཏེ་
བཀའ་སྩལ་ཏོ། །ཚིག་འདི་འབའ་ཞིག་གིས་སྐུར་པ་འདེབས་པའི་རྣམ་པར་རྟོག་པ་འགོག་པར་བརྗོད་པ་ནི་མ་ཟད་དེ། དེ་བཞིན་དུ། སངས་རྒྱས་སྒྱུ་འདྲ་དེ་བཞིན་རྨི་ལམ་འདྲ། །ཞེས་བྱ་བའི་ཚིག་གིས་ཀྱང་སྐུར་པ་འདེབས་པའི་རྣམ་པར་རྟོག་པ་སྤོང་བར་ཤེས་པར་བྱའོ། །དེ་བཞིན་
གྱི་སྒྲ་ནི་སྔ་མ་དང་སྦྱར་བའི་དོན་ཏོ། །སྒྱུ་འདྲ་ཞེས་བྱ་བ་ནི་སྒྱུ་མས་དཔེར་བྱས་པ་གང་ལ་ཡོད་པ་ནི། སྒྱུ་འདྲ་སྟེ་སྒྱུ་མ་ལྟ་བུའོ། །སུའི་ཡིན་ཞེ་ན། སངས་རྒྱས་སོ། །སངས་རྒྱས་དེ་ནི་སྒྱུ་འདྲ་ཞེས་བྱ་བས་ཀྱང་སྟེ། ཀྱང་གི་སྒྲ་ནི་བསགས་པའི་དོན་ཏེ། རྨི་ལམ་དང་ཡང་འདྲ་བ་ཞེས་བྱ་བའི་
ཐ་ཚིག་གོ། །གང་གི་ཕྱིར་སངས་རྒྱས་ཀྱི་སྒྲས་ནི་འདིར་གཉིས་མེད་པའི་ཡེ་ཤེས་བརྗོད་ལ། དེ་ལ་ཡང་སོ་སོའི་སྐྱེ་བོ་ལ་སོགས་པའི་ཤེས་པ་ལ་ཡང་རང་བཞིན་གྱིས་ཡོད་མོད་ཀྱི། འོན་ཀྱང་མ་རིག་པའི་སྒྱུ་མས་བསྒྲིབས་པའི་རང་བཞིན་ཅན་ནི་བྱིས་པ་རྣམས་ལ་མི་སྣང་བར་ཟད་དོ་
ཞེས་དེ་ལྟར་འདྲེ་བར་བྱེད་བས་སྟོན་པར་བྱེད་པ་ནི། ཚུལ་འདི་ཉིད་ནི་མཁས་རྣམས་ཀྱིས། །ངག་གཞན་དག་ལ་འང་ཤེས་པར་བྱ། །ཞེས་བྱ་བའོ། །བཤད་པའི་ཚུལ་འདི་ཉིད་ནི་ཤེས་པར་བྱའོ། །སུ་དག་གིས་ཤེ་ན། མཁས་རྣམས་ཀྱིས་ཏེ་རྣམ་པར་ཤེས་པ་རྣམས་ཀྱིས་སོ། །གང་
དུ་ཞེ་ན། ངག་གཞན་དག་ལ་ཡང་སྟེ་འདི་ལྟ་སྟེ། ཆོས་ཐམས་ཅད་ནི་སྒྱུ་མ་ལྟ་བུའོ་ཞེས་བྱ་བ་ལ་སོགས་པ་རྣམས་སུ་ཡང་ངོ། །དེར་ཡང་སྐུར་པར་རྣམ་པར་རྟོག་པ་སེལ་བ་ཁོ་ནར་རྟོགས་པར་བྱའོ། །ཡང་བརྗོད་མ་ཐག་པ་འདི་ཇི་ལྟར་རྟོགས་པར་བྱ་ཞེ་ན། བརྗོད་པ་ནི་གང་གི་
ཕྱིར། གཞི་མཐུན་པ་ཡིས་རྒྱལ་བ་ནི། །སྒྱུ་མ་ལྟ་བུར་རབ་བརྗོད་ཅིང། །ཞེས་བྱ་བའོ། །གལ་ཏེ་ཡང་ཐམས་ཅད་དུ་གཉིས་མེད་ཀྱི་ཡེ་ཤེས་མེད་པ་ཉིད་ཡིན་ན་དེའི་ཚེ་གཞི་མཐུན་པ་ཡིས་བསྟན་པ་མི་འཐད་དོ། །གཏན་ཚིགས་ཅི་སྟེ་ཞེ་ན། སྒྱུ་མ་ལ་སོགས་པ་རྣམས་ནི་
