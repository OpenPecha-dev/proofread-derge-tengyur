༄༅། །དང་འདྲ་སྟེ། དེས་བྲིས་ཞེས་བྱ་བའི་དོན་ཏོ། །བྱིས་པ་ཐམས་ཅད་ནི་བླང་བ་དང་དོར་བ་མི་ཤེས་པས། ཅིའི་ཕྱིར་བྱིས་པའི་སྐྱེ་བོ་ཡུལ་རྣམས་ཀྱིས་བཅོམ་ཞེ་ན། རང་གི་ཉེས་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། གང་གི་ཚེ་སེམས་ཡུལ་གྱི་རྣམ་
པའི་བག་ཆགས་ཡིན་པར་ཁོང་དུ་མ་ཆུད་པ་དེའི་ཚེ། རང་བདག་ཉིད་ཀྱི་ཉེས་པས་བཅོམ་ཞེས་བྱ་བའི་དོན་ཏོ། །སྒྲ་བརྗོད་པས་ཀྱང་ཡུལ་རྣམས་སྣང་བ་མིན་ནམ་ཞེ་ན། ཉིན་དང་མཚན་མོ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཉིན་དང་མཚན་མོ་མཚུངས་པ་ནི་འདྲ་བའོ། །
ལོང་བས་སྔོན་མཐོང་བའི་དོན་ནམ་ནངས་ཞེས་བྱ་བའི་སྒྲ་ཐོས་ནས། སྣང་བ་དང་ནི་བཅས་པ་སྟེ། །ཞེས་བྱ་བ་ནི་ཇི་ལྟར་མཐོང་བ་དང་ལྡན་པའི་སྐྱེས་བུས་དོན་མཐོང་བ་དེ་བཞིན་དུ་ལོང་བས་ཞེས་བྱ་བའི་དོན་ཏོ། །ད་ནི་ཡུལ་དང་དུས་ལ་སོགས་པ་ཡང་གཞན་དུ་སྣང་བ་འཁྲུལ་པ་
ཉིད་དུ་བསྟན་པའི་ཕྱིར། ཡུལ་དུས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཇི་ལྟར་རྨི་ལམ་ན་ཡུལ་དང་དུས་ལ་སོགས་པའི་དངོས་པོ་གཞན་དུ་སྣང་བ་ཉིད་ཀྱི་ཕྱིར་འཁྲུལ་པའི་ཤེས་པ་བསྐྱེད་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་སད་པའི་གནས་སྐབས་ལ་ཡང་སྐྱེས་ཞེས་བྱ་བའི་དོན་ཏོ། །འདི་ལྟར་དུབ་པ་ནི་ལམ་
རིང་པོས་སྐྱོ་བར་རོ། །ལམ་ཉེ་བ་ལ་རིང་བ་རུ་ནི་སེམས་པ་ཡིན་ཏེ། དེ་བཞིན་ཡིད་འོང་དང་བྲལ་བས། །ཞེས་བྱ་བ་སྨོས་ཏེ། དེ་བཞིན་ནི་གང་གི་ཕྱིར་གྱི་དོན་ཏོ། །གང་གི་ཕྱིར་འདོད་ལྡན་ནི་འདོད་པ་ལ་མངོན་པར་ཞེན་པའི་སྐྱེས་བུ་ཉིན་གཅིག་སྟེ། ཆུ་ཚོད་གྲངས་དང་མཉམ་པ་ལ་ནི་ཆུ་ཚོད་སུམ་
ཅུའི་ཚད་ནི་རང་བཞིན་ཡིན་ན་ཡང་ལོ་སྐོར་བརྒྱ་དང་འདྲ་བར་སེམས་ཏེ། ལོ་བརྒྱར་གྱུར་པར་རོ། །དུས་གང་ལ་ཞེ་ན། ཡིད་དུ་འོང་བ་དང་བྲལ་བའི་སྟེ། མངོན་པར་འདོད་པའི་དངོས་པོ་དང་བྲལ་བ་ཐོབ་པས་མ་འབྲེལ་ཞེས་བྱ་བའི་དོན་ཏོ། །གང་གི་ཚེ་ཕྱི་རོལ་གྱི་དོན་མེད་ན་བྱང་ཆུབ་སེམས་དཔའ་
རྣམས་སྦྱིན་པ་ལ་སོགས་པ་ལ་འཇུག་པ་དང་སེམས་ཅན་གྱི་དོན་ཇི་ལྟར་བྱེད་ཅེ་ན། དེ་ཉིད་མཐོང་བ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། དེ་ཁོ་ན་ཉིད་དོན་དམ་པ་མཐོང་བའི་བྱང་ཆུབ་སེམས་དཔའ་དེ་ཉིད་དང་དེ་འདྲ་བ་ནི་བརྟགས་པའི་དོན་གྱི་སྦྱིན་པ་ལ་སོགས་པ་ལ་ཀུན་རྫོབ་ཀྱི་ཐ་སྙད་དུ་འཇུག་གི་དེ་བཞིན་
