༄༅། །གསུངས་སོ། །ཤེས་རབ་ཀྱང་བསམ་གཏན་བཞི་པ་བསྒོམས་པ་ལས་བྱུང་ལ། བསམ་གཏན་བཞི་པ་ཡང་ལུས་དབེན་པ་དང། སེམས་དབེན་པ་ལས་ཐོབ་པར་འགྱུར་ཏེ། དེ་བས་ན་ལུས་དབེན་པར་བྱ་དགོས་པའི་ཕྱིར། དགོན་པའམ་ཤིང་དྲུང་ངམ་ཞེས་
བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །སེམས་དབེན་པར་བྱ་དགོས་པས། བདེ་བ་ཡང་སྤངས་སྡུག་བསྔལ་ཡང་སྤངས་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། དེ་དག་གི་དོན་ནི་གོང་དུ་བཤད་པར་ཟད་དོ། །མདོར་ན་དྲི་མ་རྣམ་པ་བརྒྱད་པ་དང་བྲལ་བའི་བསམ་གཏན་བཞི་པ་བསྒོམས་པ་ལས་
ནི་དགེ་བ་ཟག་པ་ཅན་དང། ཟག་པ་མེད་པ་ལ་སོགས་པ་ཡོན་ཏན་ཕུན་སུམ་ཚོགས་པ་སྐྱེ་བའི་རྒྱུ་བྱེད་དེ། དེ་བས་ན་དེ་བསྒོམས་པ་ལས་ཤེས་རབ་ཀྱང་རབ་ཏུ་བསྒོམ་པར་འགྱུར་བའི་ཕྱིར་བསམ་གཏན་བཞི་བསྒོམ་པ་ནི། ཤེས་རབ་སོ་སོར་རབ་ཏུ་རྙེད་པར་འགྱུར་བ་ཞེས་བྱའོ། །དེ་དྲན་པ་ཉེ་
བར་གཞག་པ་བཞི་ལས་བརྩམས་ཏེ། ལམ་གྱི་ཡན་ལག་བརྒྱད་ཀྱི་བར་དུ་བྱང་ཆུབ་ཀྱི་ཕྱོགས་སུམ་ཅུ་རྩ་བདུན་རིམ་གྱིས་བསྟན་པའི་ཕྱིར། དགེ་སློང་དག་དྲན་པ་ཉེ་བར་གཞག་པ་བཞི་གང་ཞེ་ན་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །དེ་ལ་བྱང་ཆུབ་ཀྱི་ཕྱོགས་སུམ་ཅུ་རྩ་བདུན་བདུན་བསྟན་
པ་ཡིན་ཏེ། དངོས་པོ་ཡོངས་སུ་བརྟག་པའི་ལམ་དང། འབད་པ་དང་བཅས་པའི་ལམ་དང། ཏིང་ངེ་འཛིན་ལས་སུ་རུང་བར་བྱས་པའི་ལམ་དང། མངོན་པར་རྟོགས་པ་ལ་རབ་ཏུ་སྦྱོར་བའི་ལམ་དང། མངོན་པར་རྟོགས་པ་དང་འདྲེས་པའི་ལམ་དང། མངོན་པར་རྟོགས་པའི་
ལམ་དང། རྣམ་པར་དག་པ་ངེས་པར་འབྱུང་བའི་ལམ་མོ། །དེ་ལ་དངོས་པོ་ཡོངས་སུ་བརྟག་པའི་ལམ་ནི་དྲན་པ་ཉེ་བར་གཞག་པ་བཞི་རྣམས་ཏེ། དང་པོ་ཉིད་དུ་དེ་དག་གི་ལུས་དང། ཚོར་བ་དང། ཆོས་དང། སེམས་ལ་མི་གཙང་བ་དང། སྡུག་བསྔལ་བ་དང། མི་རྟག་པ་
དང། བདག་མེད་པའི་དངོས་པོར་རྟོག་པའི་ཕྱིར་རོ། །འབད་པ་དང་བཅས་པའི་ལམ་ནི། ཡང་དག་པར་སྤོང་བ་བཞི་རྣམས་ཏེ། དེས་དེ་ལྟར་དངོས་པོ་ཐམས་ཅད་བརྟགས་ནས་དེའི་འོག་ཏུ་སྒྲིབ་པ་སྤོང་བར་བྱ་བའི་ཕྱིར། བརྩོན་འགྲུས་རྩོམ་མོ།
