ཡི། །རྒྱ་མཚོ་ལས་ནི་སྒྲོལ་བའི་ཕྱིར། །རྒྱལ་བ་མཉན་ཡོད་གྲོང་ཁྱེར་ནི། །ཉམས་དགའ་བར་ནི་ཡུན་རིང་བཞུགས། །བདེན་པའི་ཚིག་ནི་འདི་དག་གིས། །བདག་ལ་རྒྱལ་བའི་བཀྲ་ཤིས་ཤོག །ཨ་མྲ་ཤིང་གི་དྲུང་དུ་ནི། །ཡ་མ་ཟུང་གི་ཆོ་འཕྲུལ་དག །རྒྱལ་

ཚིག་ནི་འདི་དག་གིས། །བདག་ལ་རྒྱལ་བའི་བཀྲ་ཤིས་ཤོག །ཚངས་སོགས་དཀར་ཞིང་ཆས་དཀར་བས། །ཡོངས་སུ་བསྐོར་བའི་དེ་བཞིན་གཤེགས། །ལྷ་ལས་བབ་པ་ཆེན་པོ་ཡི། །ངོ་མཚར་བ་དག་བསྟན་པ་མཛད། །བདེན་པའི་ཚིག་ནི་འདི་དག་གིས། །བདག་ལ་
