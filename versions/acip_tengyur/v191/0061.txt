བྱེ་བྲག་བྱེ་བྲག་ཅན་རྟོག་ན། །དེ་མེད་རྒྱུ་མཚན་མེད་པ་ཉིད། །དེ་ནི་གྲུབ་པ་མ་ཡིན་ནོ། །བུམ་པའི་སྔ་ན་མེད་པ་ཞེས། །བྱ་བ་ལ་སོགས་ཚིག་འདི་ཡང། །རྟོག་པ་ཅན་གྱིས་སྤྲུལ་པ་སྟེ། །བརྟགས་པའི་དཔའ་བོ་ཉིད་སོགས་བཞིན། །བྱེ་བྲག་བྱེ་བྲག་ཅན་ཉིད་འདི། །
གང་ཡིན་དངོས་པོ་ལ་བརྟེན་ཏེ། །འབྲེལ་པ་གཞན་ཞིག་ཡོད་པར་ནི། །དེ་ལ་ངེས་པར་བརྟག་པར་བྱ། །དེ་ལ་མེད་པ་མེད་པ་ཞེས། །འདིར་ནི་ཤེས་པ་ཙམ་བརྗོད་པ། །བྱེ་བྲག་ལ་གནས་སྤྱི་ཡིན་ཏེ། །རང་གི་རྟེན་རྣམས་ལ་གནས་པའོ། །དེ་རྣམས་རྗེས་འཇུག་འདོད་མིན་གྱི། །གལ་ཏེ་
གཞན་ཡོད་འདོད་མིན་ན། །ཚིག་གི་དོན་ལས་ཐ་དད་པས། །མེད་པའི་བློ་ནི་སྐྱེ་བ་ཡིན། །འདོད་པས་སྦྱར་བའི་གཟུགས་སོགས་པའི། །དོན་ལ་རིགས་ནི་ཡོད་མིན་ཏེ། །གསལ་བ་མེད་པ་ཉིད་ཀྱི་ཕྱིར། །དེ་ནི་འཁྲུལ་པ་ཉིད་དུ་བསྟན། །འདས་པ་དང་ནི་མ་བྱུང་བའི། །རྟག་པའི་སྤྱི་ཡི་
ཡུལ་ཅན་གྱིས། །ཤེས་ཡོད་ཅེ་ན་འདི་ནི་བྱིན། །འབའ་ཞིག་མི་འཛིན་མ་ཡིན་ནམ། །འབའ་ཞིག་དམིགས་པ་ཡིན་ན་ནི། །གསལ་བ་རྣམས་ཀྱིར་འདིར་མི་འགྱུར། །སྤྱི་ཡི་གསལ་བྱེད་དེ་མིན་ཏེ། །གངས་ཅན་འབིགས་བྱེད་ཉིད་བཞིན་ནོ། །འདི་ནི་དེ་རྣམས་བསྐྱེད་པ་ལ། །གཞན་
དབང་ཡིན་པས་འབྲེལ་པ་མེད། །རྟག་ཕྱིར་གཅིག་པུ་འཛིན་པའི་ཕྱིར། །ཤེས་པ་གཞན་དབང་ཉིད་ཀྱང་མིན། །རང་རྟེན་དབང་པོར་སྦྱོར་བ་ལས། །སོགས་ལ་ལྟོས་པ་མེད་ཕྱིར་རོ། །ཡོད་པ་རྟག་ཏུ་དམིགས་ཉིད་དམ། །ཡང་ན་ནམ་ཡང་མ་ཡིན་ཏེ། །རང་གི་བདག་ཉིད་ཤེས་བསྐྱེད་པར།།
རུང་བའམ་ཡང་ན་མི་རུང་ཉིད། །གལ་ཏེ་ལན་ཅིག་དེ་ལྟའི་དངོས། །ཡིན་ན་རྟག་ཏུ་དེ་ཉིད་འགྱུར། །དེ་ནི་རུང་བ་འམ་མི་རུང་བའི། །ངོ་བོར་རང་བཞིན་གྱིས་གང་གནས། །དེ་རྣམས་རྟག་ཕྱིར་གཡོ་མིན་ཏེ། །འཇུག་པར་འགྱུར་བ་གང་ཞིག་ཡིན། །བ་ལང་ཉིད་ནི་བ་ལང་ལས། །དོན་གཞན་
ཐ་དང་བློ་ཡུལ་ཕྱིར། །དེ་ནི་གཟུགས་དང་རེག་སོགས་བཞིན། །བརྗོད་པ་ནག་པའི་རྟ་བཞིན་ནོ། །ཞེས་འདིར་འཁྲུལ་པ་སྨྲ་བ་སྟེ། །འཚེད་པ་པོ་ཉིད་སོགས་དེ་བཞིན། །འདི་ཡི་ཚུལ་གྱིས་གཞན་ལ་ཡང། །དམན་པའི་གཏན་ཚིགས་སུན་ཀུན་སྟོན། །གཞན་ཡང་སྤྱི་ནི་དུ་མ་ལ། །འཇུག་
