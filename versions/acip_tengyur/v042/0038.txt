སྒྲུབ་ཅེས་རང་སེམས་བསྐྱེད་པར་བྱ། །བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་པའོ། །སྦྱིན་པ་ལ་སོགས་བྱ་བས་སྐབས་བྱས་ཏེ། །འདས་པའི་རྒྱལ་བའི་སྲས་རྣམས་ཀུན་གྱི་ལམ། །གཤེགས་དང་གཤེགས་འགྱུར་འཇིག་རྟེན་འདས་པ་ཡི། །དཔལ་འབྲས་ལམ་མཆོག་དེ་ལ་བརྟེན་པར་བགྱི། །ལམ་
ལ་བརྟེན་པའོ། །རྒྱལ་དང་བདག་ཉིད་ལས་སྐྱེས་དེ་རྣམས་ལ། །ཀུན་ནས་ཀུན་དུ་བདག་ཉིད་བདག་གིས་དབུལ། །སེམས་དཔའ་མཆོག་གིས་བདག་ཉིད་ཡོངས་གཟུང་བས། །དད་པས་ཁྱོད་ལ་འབངས་སུ་མཆི་བར་བགྱི། །བདག་ཉིད་དབུལ་བའོ། །གློ་བུར་ཉོན་མོངས་ཚོགས་ཆེན་གྱིས་
གདུངས་པ། །བསྐྱོད་པའི་སྐྱེ་བོ་མཐའ་མེད་དགྲོལ་བའི་ཕྱིར། །ཆོས་ཀྱི་བདུད་རྩི་སྔགས་རྣམས་བཤད་པ་རུ། །སྨན་མཆོག་མགོན་པོ་ལ་ནི་བདག་བསྐུལ་ལོ། །བསྐུལ་བའོ། །མཆོག་ཏུ་མི་དགེ་བསམ་པས་རབ་བཅོམ་པའི། །བདུད་ཀྱི་སྐྱེ་རྒ་ན་དང་འཆི་བ་ལས། །མྱ་ངན་འདས་བཞེད་
