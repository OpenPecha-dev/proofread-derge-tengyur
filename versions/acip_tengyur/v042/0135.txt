རྒྱལ་བ་རྣམས་ཀྱིས་བཀའ་སྩལ་ཏོ་ཞེས་འབྲེལ་ཏོ། །དེའི་གདམས་ངག་གསུངས་པ་རིགས་ཀྱི་བུ་ཞེས་པ་ལ་སོགས་པའོ། །གདོད་ཆོས་འཆད་དོ་ཞེས་པ་ནི་དོན་དང་ལྡན་པ་ཉིད་ཀྱིས་དེ་ལའོ། །དེ་ནས་དགའ་བའི་མདངས་ཀྱིས་ངོ་མཚར་བ་ནི། དེ་ནས་བྱང་ཆུབ་ཅེས་
པ་ལ་སོགས་པའོ། །འདིར་མཉམ་པ་ཉིད་ཀྱི་དོན་དེ་ཡིན་ཡང་རྣམ་པར་ཤེས་པས་གཟུགས་ལ་སོགས་པ་སྣ་ཚོགས་པ་སྤྲོས་པ་ཀུན་རྫོབ་སྟེ། དོན་དམ་པ་དེ་དང་མི་མཐུན་པ་ཉིད་ཀྱི་ཕྱིར། བཤད་པ་མཚར་ཞེས་པ་ཡིན་ནོ། །ལེའུ་དགུ་བའི་དོན་རྫོགས་སོ།། །།ད་ནི་
དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་སྙིང་པོ་བསྐུལ་བ་ལ་སོགས་པ་བསྟན་པའི་ཕྱིར། དེ་ནས་ཞེས་པ་ལ་སོགས་པ་སྨོས་ཏེ། དེ་ལ་དམ་ཚིག་ཆེན་པོ་ཞེས་པ་ནི་རྡོ་རྗེ་འཛིན་པ་སྟེ། དེ་ལ་འདུན་པ་དང་བཅས་ཏེ་གསོལ་ཏོ་ཞེའོ། །དེའི་སྔགས་ཀྱི་སྙིང་པོ་ནི་ཧཱུཾ་ལ་སོགས་པ་སྟེ་ཞུས་པའོ། །
ཞུས་པའི་དེ་མ་ཐག་གསུངས་པ་དེ་ནས་ཞེས་པ་ལ་སོགས་པའོ། །ལུས་དང་ངག་དང་སེམས་རྣམས་ནི་རྣམ་པར་མི་རྟོག་ཅེས་པ། གཟུང་འཛིན་གྱི་རྣམ་པར་རྟོག་པ་མེད་པ་ཡིན་ཏེ། མི་དམིགས་པ་དང་མི་གནས་པའི་ཕྱིར་རོ། །གལ་ཏེའོ་ན་ནི་ཅི་ཡང་མེད་པ་ཡིན་ནོ་ཞེ་ན་ནི་མ་ཡིན་
ཏེ༑ ཡེ་ཤེས་ཙམ་གནས་པའི་ཕྱིར་རྡོ་རྗེར་བསྒོམ་ཞེས་སོ། །ཡང་གསུངས་པ་ནི་དེ་ནས་ཞེས་པ་རྣམས་ཏེ། དེ་ལ་དག་པ་ནི་དྲི་མ་དང་བྲལ་བ་ཉིད་ཀྱི་ཕྱིར་རོ། །ཕ་རོལ་ཏུ་ཕྱིན་པའི་སྔགས་ཀྱི་ཚུལ་ནི་སྣང་མཛད་ཆེན་པོའི་རྣམ་པར་གང་གྱུར་པ་དེ་སྙོམས་པར་ཞུགས་ནས་ཞེས་ཏེ་
སྣང་བ་ལས་འདས་པའི་སྔགས་ཚིག་རྣམས་ཏེ། དེ་ནི་ཐབས་དང་ཤེས་རབ་ཀྱི་བདག་ཉིད་དུ་མངོན་ནོ། །གང་ཞི་བ་ལ་སོགས་པའི་སྒོ་ནས་བསྐུལ་བས། དེ་ནས་དེའི་ལས་བྱེད་པ་ནི་རིག་པ་སྟེ་དེ་རྣམས་དང་ལྷན་ཅིག་སྤྱོད་པ་ཡོད་ཀྱང། སྔགས་ཀྱི་སྤྱོད་པ་མེད་ན་མི་འགྲུབ་པས་
