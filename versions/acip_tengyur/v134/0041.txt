ནི་རིན་མི་བླང། །ལེགས་སྦྱིན་ཆོས་གོས་བདག་ཅག་རུང། །ཞལ་ཏ་ཕོ་ཉ་དེ་ལ་བརྗོད། །དེ་ཡིས་གང་ཚེ་ཆོས་གོས་རིན། །ཞལ་ཏ་བྱེད་ལ་གཏད་པ་ན། །དགེ་སློང་ཆོས་གོས་དོན་གཉེར་དེས། །ལན་དྲུག་བར་དུ་དྲན་པར་བྱ། །དེར་ནི་མངོན་པར་གྲུབ་ན་ནི། །དེར་ནི་བླང་བར་རུང་བ་ཉིད། །
དེ་ལས་ལྷག་པར་བརྩོན་བྱེད་ན། །ལྟུང་བའི་ཚང་ཚིང་ནང་དུ་ལྟུང། །འདི་ལ་ལན་དྲུག་བསྐུལ་ནས་ནི། །རང་ཉིད་འོངས་ནས་སྟེར་བྱེད་ན། །དགེ་སློང་གིས་ནི་འདི་སྐད་བརྗོད། །བདག་གིས་འདི་སྤངས་དགོས་པ་མེད། །དེ་ལྟ་ན་ཡང་གལ་ཏེ་དེ། །ཤིན་ཏུ་ཕྱག་སོགས་སྔོན་འགྲོ་བས། །སྟེར་ན་

བཞུགས་ནས་ནི། །ཕྱག་ཞབས་ལེགས་པར་བཀྲབ་དང། །བཟའ་དང་བཏུང་བ་དགྱེས་བཞིན་གསོལ། །ལན་དྲུག་བསྐུལ་བའི་སྒོ་དེར་ཡང། །མ་ཐོགས་བར་ནི་རྗོད་བྱེད་ན། །ལྟུང་མེད་ཐོགས་ནས་རྗོད་བྱེད་པ། །ལྟུང་བར་འགྱུར་བ་ཁོ་ནའོ། །གལ་ཏེ་ཕོ་ཉ་མི་མ་ཡིན། །སྦྱིན་བདག་དང་
