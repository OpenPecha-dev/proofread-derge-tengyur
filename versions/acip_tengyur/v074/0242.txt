གེ་འབྲུ་བཅུ་བཟླས་བྱ་སྟེ། །སྔགས་ཕྲེང་ཁ་ནས་ཁར་ཞུགས་ཡེ་ཤེས་དགུག་ལྟེ་བ་ནས་འཐོན་དེར་ཞུགས་བདག་ཉིད་ཀྱི། །སྙིང་གར་བསླན་པ་འཁོར་ལོ་བཞིན་དུ་བསམ། །དེ་ལྟར་ཞག་གྲངས་ཉི་ཤུ་རྩ་གཅིག་ཏུ། །ཐུན་གསུམ་ཉིད་དུ་བསྒོམ་པ་གང་ཡིན་དེས། །བཀའ་དང་བསྟན་བཅོས་
ཀུན་གྱི་ཚིག་དང་དོན། །བལྟས་བཙམ་གྱིས་ཁོང་དུ་ཆུད་པར་འགྱུར། །གཞན་ཡང་ཀུན་ལ་མི་འཇིགས་སྤོབས་པ་འཐོབ། །སྙན་དངགས་བསྟན་བཅོས་ཐོགས་མེད་རྩོམ་ནུས་ཤིང། །དེ་ཡི་སྨྲ་བ་སུས་ཀྱང་བཟློག་མི་ནུས། །དེ་ཕྱིར་འདི་འདྲ་གར་ཡང་རྙེད་པར་དཀའ། །འཕགས་པ་མི་གཡོ་
བའི་སྒྲུབ་ཐབས་རྫོགས་སོ།། རྒྱ་གར་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་ཆེན་པོ་དཱ་ན་ཤཱི་ལས་རང་འགྱུར་དུ་མཛད་པའོ།།  །རྒྱ་གར་སྐད་དུ། སུ་སིདྡྷི་ཀ་ར་སཾ་གྲ་ཧ་སཱདྷ་ནཾ། བོད་སྐད་དུ། ལེགས་པར་གྲུབ་པར་བྱེད་པའི་སྒྲུབ་པའི་ཐབས་བསྡུས་པ། སངས་རྒྱས་
དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་སྒྲུབ་པ་པོས་འགྲོ་བ་མཐའ་དག་ལ་ཕན་བ་བསྒྲུབ་པ་དང། དངོས་གྲུབ་བླ་ན་མེད་པ་འདོད་པ་རྣམས་ལ་ལེགས་པར་གྲུབ་པར་བྱེད་པ་ཞེས་བྱ་བ། རིགས་གསུམ་སྤྱིའི་སྒྲུབ་པའི་ཆོ་ག་རབ་ཏུ་གསལ་བར་བརྗོད་པར་བྱའོ། །དེ་ལ་རེ་ཞིག་
དང་པོ་སློབ་དཔོན་གྱི་ཡོན་ཏན་དང་རྗེས་སུ་མཐུན་པའི་སློབ་དཔོན་གྱིས་སྒྲུབ་པ་པོ་བདག་ཉིད་མངོན་པར་དབང་བསྐུར་ཞིང་རྗེས་སུ་གནང་བ། རྒྱུད་ལས་གསུངས་པའི་སྒྲུབ་པ་པོའི་མཚན་ཉིད་དང་ལྡན་པའི་གསང་སྔགས་ཀྱི་མཚན་ཉིད་ལེགས་པར་ཚོགས་པ། དངོས་གྲུབ་ལས་ཀྱི་བྱེ་བྲག་ལ་མཁས་
པ་མཐུན་པའི་དུས་ཀྱི་དབྱེ་བ་ཤེས་པ། མཆོད་པ་རྣམ་པ་ལྔའི་ཆོ་ག་རིག་པ། རང་གི་མཚན་ཉིད་དང་མཐུན་པའི་སྔགས་གཡོག་དང་ལྡན་པས་སྒྲུབ་པའི་ཡན་ལག་གིས་ཡོ་བྱད་ཐམས་ཅད་བསྡུ་བར་བྱའོ། །དེ་ནས་གསང་སྔགས་ཡོངས་སུ་བཟུང་བའི་ཆོ་གས་གང་ལ་མངོན་པར་དད་པའི་གསང་
