རྒྱུའི་དངོས་མི་ཤེས་དེ་དག་ལ། །སྟོན་ཏེ་མཁས་པ་རྣམས་ལ་ནི། །གཏན་ཚིགས་འབའ་ཞིག་བརྗོད་པར་ཟད། །ཅེས་བྱ་བ་སྨོས་སོ། །དེ་དང་དེ་དག་ནི་ངེས་པར་ཁས་བླངས་པས་ལྷག་མ་ཚར་གཅད་པའི་གནས་ཀྱི་ཁྱད་པར་གྱི་ཡུལ་དུ་གཞག་གོ། །ཁྱད་པར་གྱི་
ཡུལ་དེ་ཡང་སྒྲུབ་པར་བྱེད་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་སོ། །འོ་ན་ངེས་པ་ཁས་བླངས་ནས་བརྗོད་པ་ཡིན་ཞེས་བྱ་བ་རྣམ་བཤད་མཁན་གྱིས་བརྗོད་པ་སློབ་དཔོན་གྱིས་འདིར་ཅི་ཞིག་ཏུ་བརྗོད། བདེན་ཏེ་ཅུང་ཟད་ཀྱང་བརྗོད་པ་མེད་དེ་སློབ་དཔོན་གྱིས་རྐང་མིག་གིས་སྨྲས་པ་ཉིད་རྗེས་སུ་ཤེས་
པའི་དོན་དུ་ཡང་རྗེས་སུ་བརྗོད་དོ། །སྒྲ་དང་དོན་དག་སྨྲ་བ་ནི་ཡང་བརྗོད་པ་ཡིན་ནོ་ཞེས་བྱ་བར་སྨྲ་བ་ནི་སྐུར་པ་འདེབས་པ་ནི། རྗེས་སུ་སྨྲས་པ་ལས་གཞན་ཞེས་བྱ་བ་སྨོས་སོ། །རྗེས་སུ་སྨྲས་པ་ནི་མཇུག་བསྡུ་བའོ། །རྗེས་སུ་སྨྲ་བ་ནི་ཡང་བརྗོད་པས་བསྟན་པར་འགྱུར་བ་མ་ཡིན་ཏེ། སྒྲ་དག་གིས་དོན་གྱི་བྱེ་བྲག་
བྱེད་ན་འཐད་པའི་ཕྱིར་རོ། །བསྒྲུབ་བྱ་བསྟན་པ་གང་ལས་དམ་བཅའ་བ་གྲུབ་ཟིན་པ་སྟོན་པ་ནི་མཇུག་བསྡུ་བ་ཡིན་ནོ་ཞེས་བཤད་དེ་ཡང་སྒྲ་དུ་མ་ཉིད་མཐོང་བས་དུས་མི་རིང་བར་ཡང་ངེས་པའི་ནུས་པ་ངེས་པར་ཡང་བྱ་བའོ་ཞེས་ཀྱང་རིག་པར་བྱ་བ་ཉིད་དོ། །འོན་ཏེ་དེ་ལྟ་ན་གལ་ཏེ་ཡང་བརྗོད་
པས་ཐོབ་པ་ཉིད་མེད་པ་ཡིན་ན་ནི་དོན་འདི་གསལ་པ་ཅིའི་ཕྱིར་རྩོལ་ཞེ་ན། བདེན་ཏེ་འདི་ལྟ་བུ་འདི་ཉིད་རྟོག་གེ་བ་ཕུལ་དུ་ཕྱུང་བ་ལ་ཇི་ལྟ་བ་ཡིན་ཞེས་བྱ་བ་ལ་སོགས་པ་འདྲི་བར་བྱའོ། །ཁོ་བོ་ཅག་ལ་ཕ་རོལ་གནས་སེམས་ཤིང་དཔྱོད་པ་རྣམས་ཀྱིས་ཅི་ཞིག་བྱ་སྟེ། ཞར་ལ་འབྱུང་བས་ཆོག་
གོ། འདི་ལ་ཡང་མདོ་གཉིས་པ་ཡོད་པ་ཡིན་ཏེ། དོན་སྤངས་ནས་སྒྲ་རང་ཉིད་ལ་འཇུག་པ་ཡིན་ནོ་ཞེའོ། །དེ་སློབ་དཔོན་གྱི་ཉེ་བར་བཀོད་པ་ནི་མཚོན་པ་ཙམ་ཡིན་པའི་ཕྱིར་རོ། །གོ་བར་བྱ་བ་དང་གོ་བར་བྱེད་པའི་དོན་ལ་བཟླས་པ་ནས་བརྗོད་པ་ལ་སོགས་པ་འདི་དག་ཀྱང་སྤང་བར་བྱ་ཞིང་
བསལ་བར་འདོད་པས་འདིར་ཞེས་བྱ་བ་ལ་སོགས་པ་སྐྱོན་རྩོལ་བ་ཡིན་ནོ། །འདི་སྐད་སྟོན་པ་ཡིན་ཏེ། གང་ལ་སྒྲ་དང་དོན་ལ་ཉེ་བར་ཐ་དད་པ་དེ་ལ་དོན་ཡང་བཤད་པ་ཉིད་ཀྱིས་རྟོགས་པ་ཡིན་ནོ། །གང་ལ་སྒྲ་དོན་དང་ཉེ་བར་ཐ་དད་པ་དེ་ལ་སྒྲ་ཡང་བརྗོད་པ་འདི་ཡང་སྐྱོན་ཅུང་ཟད་ཙམ་མེད་དེ། ཅི་འདི་འདྲ་
