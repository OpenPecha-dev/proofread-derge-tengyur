བར་དུ་བྱ་རྒོད་ཀྱི་དྲི་ལྟ་བུ་དང་། ཐ་མ་ག་པུར་གྱི་དྲི་ལྟ་བུ་འབྱུང་བར་འགྱུར་ཏེ། རྡོ་རྗེ་སློབ་དཔོན་གྱིས་མཚོན་པར་བྱའོ། །དེ་བཞིན་དུ་གླུའི་མཚན་ཉིད་ཀྱིས་ངག་གི་གདངས་ཐོག་མ་དང་ཐ་མ་གཉིས་སུ་ངང་པའི་རྒྱལ་པོ་དང་། བུང་བའི་རྒྱལ་པོ་ལྟ་བུའི་སྒྲས་སྣ་དྲང་ཞིང་མཐའ་བརྟེན་ཏོ། །བར་དུ་ནི་ཁྱུ་མཆོག་
ལྟ་བུའི་སྒྲ་སྟེ། རའི་སྒྲ་ལྟ་བུའི་རྒྱས་པར་བྱ་བའི་དོན་ཏེ་སྐད་ཅིག་དེ་ཉིད་ལ་དཀྱིལ་འཁོར་བའི་ཉེ་འཁོར་དུ་ཡང་རྟགས་དེ་དག་འབྱུང་བར་འགྱུར་རོ། །དེ་ལྟ་བས་ན། ངང་པ་དང་ནི་བུང་བའི་སྒྲ། །བ་ལང་གཙེ་ཡི་སྒྲ་ལྟ་བུ། །གླུ་ཡི་མཐའ་ནས་མཉན་པར་བྱ། །གླུ་ཡི་བྱིན་རླབས་མཚན་ཉིད་

གལ་ཏེ་ཡོངས་སུ་མ་སྦྱངས་པར་འཇིག་རྟེན་པའི་བྱ་བ་ལ་འཇུག་པ་དེ་ལྟ་ན་ནི་འཇིག་རྟེན་པའི་ཐ་སྙད་ཀྱང་མེད་དེ་ཀླ་ཀློ་ཡོངས་སུ་དར་བར་འགྱུར་རོ། །དེ་ལྟ་བས་ན་རྒྱུད་དེ་དང་དེ་ལས་ཀྱང་། ཇི་ལྟར་ཕྱི་རོལ་དེ་བཞིན་ནང་། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །དེ་བས། ཕྱག་རྒྱ་རྟགས་དང་མཚན་མ་སྟེ། །
འདི་ཡི་རིགས་ནི་མཚོན་པར་བྱ། །རིགས་འཆོལ་བསྒོམ་པའི་སྦྱོར་བ་ལས། །སྒྲུབ་པ་པོ་ལ་དངོས་གྲུབ་མེད། །ཅེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། གང་གི་རིགས་གང་ཡིན་པ་དེ་དང་དེའི་རིགས་ཀྱི་ཕྱག་རྒྱས་གདབ་པ་ནི་དེ་བཞིན་གཤེགས་པའི་ངེས་པའོ། །འདིའི་དོན་ཡང་རྩ་བའི་རྒྱུད་ལས་

པའི་མགོན་པོ་དཔག་མེད་རིགས། །ཕྲག་དོག་ཕྱག་རྒྱས་མཁའ་འགྲོ་མ། །ཕྲག་དོག་དོན་ཡོད་གྲུབ་པའི་རིགས། །དེ་བཞིན་མཚམས་སོགས་ལྷ་མོ་ཡང་། །ས་དང་ཆུ་དང་མེ་རླུང་ཕྱིར། །རིམ་པ་བཞིན་དུ་ཕྱག་རྒྱས་གདབ། །ཅེས་གསུངས་པས་དོན་ཁོང་དུ་ཆུད་པར་བྱའོ། །དེ་བཞིན་
