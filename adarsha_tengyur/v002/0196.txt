པས་ཀུན་གྱི་ལྷག་མ་སྤང་། །ཞེས་གསུངས་ཏེ། ཚེས་བཅུ་དྲུག་པ་མེད་པའི་ཕྱིར། དེ་སྐད་ཅེས་བྱ་བ་ལ། གང་ཡང་བཅུ་དྲུག་ཅེས་གྲགས་པ་ནི་མར་ངོའི་ཚེས་གཅིག་ཡིན་པའི་ཕྱིར་རོ། །དེ་ཅིའི་སླད་དུ་ཞེ་ན། དོན་གྱི་བྱ་བ་མི་བྱེད་ཕྱིར། །ཞེས་བྱ་བས་གསལ་བར་རབ་ཏུ་བསྟན་པ་ཡིན་ནོ། །དེ་ཡང་
བསྡུས་པའི་རྒྱུད་འདི་ཉིད་ལས། །ཆ་ཤས་བཅོ་ལྔའི་བདག་ཉིད་ཅན། །བྱང་ཆུབ་སེམས་རྣམས་ཟླ་བར་འགྱུར། །རྣལ་འབྱོར་མ་རྣམས་དེ་ཡི་ཆ། །བདེ་བ་ཆེན་པོ་ཨཱ་ལིའི་གཟུགས། །ཞེས་གསུངས་སོ། །དེ་བཞིན་དུ། སྟོང་པ་ཆེན་པོ་ཡི་གེ་ལྔ། །ཞེས་པའི་དུས་སུ་ཡི་གེ་ལྔ་པོའི་
དབྱིབས་ལྟ་བུར་གནས་པ་ཐིག་ལེ་དང་རྣམ་པར་གཅད་པར་བྱ་བ་དག་གིས་ཀུན་དུ་མཛེས་པར་འགྱུར་བའི་ཕྱིར་རོ། །རྣལ་འབྱོར་བྱིས་པའི་བློ་ཅན་དག་འདི་སྐད་ཅེས་ཅི་སྟེ། ཨཱ་ལི་ཡང་བཅུ་དྲུག་ཏུ་གསུངས་ལ་ཐིག་ལེ་དང་དགའ་བ་འཕོ་བ་ཡང་བཅུ་དྲུག་ཏུ་གསུངས་པ་མ་ཡིན་ནམ། དེ་ཅིའི་
ཕྱིར་ཞེ་ན། བཅུ་དྲུག་པ་དང་སུམ་ཅུ་རྩ་གཉིས་པ་སྤང་ཞེས་སེམས་པ་དེ་ནི་མ་ཡིན་ཏེ། འདི་ལྟར་དེ་བཞིན་གཤེགས་པས་བྱིས་པའི་སྐྱེ་བོ་རྣམས་ལ་དགོངས་ནས། སྐུ་གསུང་ཐུགས་ཡེ་ཤེས་ཀྱི་དབྱེ་བས་ཕྱེ་སྟེ། ངོ་བོ་ཉིད་དང་། ཆོས་དང་ལོངས་སྤྱོད་རྫོགས་པ་དང་སྤྲུལ་པའི་སྐུ་སྟེ། སྐུ་བཞི་
རྣམས་ལ་དགོངས་ནས་དགའ་བ་བཅུ་དྲུག་ཏུ་གསུངས་སོ། །དེ་བཞིན་དུ་ཨཱ་ལི་བཅུ་དྲུག་དང་། ཐིག་ལེ་བཅུ་དྲུག་དག་ནི་དངོས་སུ་ཡོད་པ་མ་ཡིན་ཏེ། ཆ་གཉིས་གསལ་བར་བྱེད་པའི་ཕྱིར་སྒྲ་གཅན་ནམ། མ་ནིང་ལྟ་བུ་ཡིན་ནོ། །དེ་བས་ན་སུམ་ཅུ་རྩ་གཉིས་དང་། བཅུ་དྲུག་པའི་འཁྲུལ་པ་སྤང་
སྟེ། གཞན་དུ་ན་དོགས་པ་མི་བྱའོ། །རྡོ་རྗེ་སྙིང་པོས་གསོལ་པ། རྣལ་འབྱོར་མ་ཀུན་ལས་བྱུང་བ། །ལྷན་ཅིག་སྐྱེས་དགའི་རང་བཞིན་ཅན། །འགྲིབ་མེད་བཏུང་མཆོག་ནམ་མཁའ་ཉིད། །ག་པུར་ངེས་པར་ཅིས་མི་སྤང་། །ཞེས་པ་གསོལ་ཏེ། དེ་ལྟ་བས་ན་རྣལ་འབྱོར་མ་བཅོ་ལྔའི་
བདག་ཉིད་ཅན་གྱི་ཁུ་བ་བྱང་ཆུབ་ཀྱི་སེམས་བཏུང་བའི་མཆོག་ནམ་མཁའ་ལྟ་བུའི་བདེ་བ་རྒྱུན་མི་འཆད་པ་ལུས་ངག་ཡིད་གསུམ་བསིལ་བར་བྱེད་པའི་སླད་དུ་ག་པུར་ལྟ་བུ་འདི་ཅིས་ཀྱང་མི་སྤང་བར་བསྒྲུབ་པར་བགྱིའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །བཅོམ་ལྡན་འདས་ཀྱིས་ཚུལ་དང་མཐུན་པའི་ལན་གྱིས་
