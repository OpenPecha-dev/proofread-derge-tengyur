རྟེན་གསུམ་སྟེ་སེམས་ཅན་ནོ། །དེ་ཁོ་ན་འཇིག་རྟེན་གསུམ་ཉིད་དེ་རང་གི་དོན་ལ་ཡ་ཎའོ། །མཚུངས་པ་ཁོ་ན་མཚུངས་པ་ཉིད་ཅེས་པ་ཇི་ལྟ་བ་བཞིན་ནོ། །ཁུ་བ་དང་བདེ་བ་དག་གི་བར་ནི་ཤེས་རབ་དང་ཐབས་ཀྱི་རྣམ་པར་དབྱེ་བ་སྟེ། སྙོམས་པར་ཞུགས་པའི་ལྷ་དག་གི་
ལྟར་ཡིན་གྱི། འདི་དོན་དམ་པ་ནི་མ་ཡིན་ནོ། །དོན་དམ་པའི་ཤེས་རབ་དང་ཐབས་ཉིད་ནི་སྟོང་པ་ཉིད་དང་སྙིང་རྗེ་དག་ཁོ་ནའི་འོ། །རང་བྱིན་བརླབ་པ་ནི་དབང་བསྐུར་བ་ལ་སོགས་པ་ལ་རང་ཉིད་བྱིན་གྱིས་རློབ་པར་བྱེད་པ་དག་ཡིན་པ་སྟེ། ཀུན་མཁྱེན་ཡེ་ཤེས་ཀྱང་དེའི་རང་བཞིན་
ཁོ་ནའོ། །འོད་གསལ་བ་ཞེས་པ་སྤྲོས་པ་ཐམས་ཅད་ཀྱི་མུན་པ་དང་བྲལ་བ་ཉིད་ཀྱི་ཕྱིར་ཕུལ་དུ་བྱུང་བར་གསལ་བའི་ངང་ཚུལ་ཅན་ཏེ། ཤེལ་དག་པ་ཇི་ལྟ་བ་བཞིན་ནོ། །ལྔ་ཞེས་པ་ལ་སོགས་པ་སྟེ། དེ་བཞིན་གཤེགས་པ་ལྔའི་བདག་ཉིད་མེ་ལོང་ལྟ་བུ་ལ་སོགས་པ་ཡེ་ཤེས་ལྔའི་རང་
བཞིན་དེ་ཁོ་ན་ཉིད་ནི་དྲུག་པ་རྡོ་རྗེ་སེམས་དཔའི་ངོ་བོ་སྟེ། དེའི་དབུས་ན་དེའི་བདག་ཉིད་ཀྱིས་གནས་པ་རོལ་པར་བྱེད་པས་ན་ལྷ་བྱང་ཆུབ་ཀྱི་སེམས་ལྷན་ཅིག་སྐྱེས་པའི་བདེ་བའོ། །དེ་བརྟག་པ་གཉིས་པར་ཡང་གསུངས་པ། བྱང་ཆུབ་སེམས་ནི་ལྷ་ཉིད་དོ། །ཞེས་སོ། །ཀླུ་སྒྲུབ་ཞབས་
ཀྱིས་ཀྱང་། སེམས་ཀྱི་རྡོ་རྗེ་རྣམ་གྲོལ་ལས། །ལྷ་གཞན་མངོན་པར་བརྗོད་དུ་མེད། །ཅེས་སོ། །མི་གསལ་བ་ནི་དོན་དམ་པར་ཏེ། སྟོང་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །གསལ་བའི་གཟུགས་ཅན་ཞེས་པ་ཀུན་རྫོབ་ཏུའོ། །འཕེལ་བ་ཞེས་པ་ལ་སོགས་པ་ལ། རི་དྷུ་འཕེལ་བ་ལའོ། །འཕེལ་བར་
བྱེད་པ་འཕེལ་བ་དང་། དེའི་ཙམ་སྟེ་མཐའ་དག་ཉིད་མཆོག་སྟེ་ཕུལ་དུ་བྱུང་བ་གང་གི་ཡིན་པ་དེ་དེ་སྐད་དོ། །མཆོག་གི་རྫུ་འཕྲུལ་མ་ལུས་པ་དང་ལྡན་པ་ཞེས་པའི་དོན་ཏོ། །ཕྲ་བ་སྟེ་དོན་དམ་པར་ཟབ་པ་ཉིད་ཀྱི་ཕྱིར་ཀུ་ཤའི་རྩེ་མོ་ལྟ་བུའི་བློ་ཅན་གྱི་སྤྱོད་ཡུལ་ལོ། །ཀུན་རྫོབ་ཏུ་ནི་ཐིག་
ལེའི་གཟུགས་ཅན་ནོ། །གང་གིས་ཉེ་བར་འཇལ་བ་ཡིད་ཡིན་པ་དེ་ནི་ཡིད་དཔེ་ཅན་ཞེས་པ་གཙུག་ལག་སྟེ་ཡིད་ཀྱི་རང་བཞིན་ཞེས་པའི་དོན་ཏོ། །སྙིང་གི་དབུས་ན་ཞེས་ལྷན་ཅིག་སྐྱེས་པའི་གཏུམ་མོའི་རྣལ་འབྱོར་གོམས་པར་གྱུར་པ་རྣམས་ཀྱི་སྙིང་པོ་ཉིད་ཀྱི་ཕྱིར་སྙིང་ཞེས་བྱ་བ་
