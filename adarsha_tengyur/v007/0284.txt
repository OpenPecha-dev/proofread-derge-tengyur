བྱིན་བརླབ་ཅིང་། །སྐུ་དང་གསུང་དང་ཐུགས་ཉིད་ཀྱང་། །ཡེ་ཤེས་འཁོར་ལོ་གཞུག་པ་དང་། །བདུད་རྩི་མྱང་བ་ཉིད་དང་ནི། །མཆོད་པ་ཆེ་དང་བསྟོད་པ་ཡང་། །རྣལ་འབྱོར་ཆེན་པོར་འདོད་པའོ། །ཞེས་སོ། །འདིར་ནི་དཔལ་འདུས་པར། བཅོམ་ལྡན་འདས་དེ་བཞིན་
གཤེགས་པ་མི་བསྐྱོད་པ་ལ་སོགས་པ་རྣམས་ཐམས་ཅད་ཡོད་དེ། དེ་བཞིན་གཤེགས་པ་རྣམས་ཐམས་ཅད་རིག་མ་བྱེ་བ་ཁྲག་ཁྲིག་བརྒྱ་སྟོང་དུ་མ་དང་ལྷན་ཅིག་སེམས་ཅན་གྱི་དོན་བྱ་བ་ལ་གར་རབ་ཏུ་སྟོན་ཅིང་། ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་ཀྱི་མཐས་གཏུགས་
པའི་འཇིག་རྟེན་གྱི་ཁམས་ནས་ཐམས་ཅད་དུ་འདོད་པ་ལྔའི་བདེ་བ་རྣམས་ཀྱིས་རྩེ་ཞིང་དགྱེས་ལ་རབ་ཏུ་རྒྱུའོ། །དེ་རྣམས་ཀྱིས་གསང་སྔགས་སྤྱོད་པ་ལ་མངོན་པར་བརྩོན་པར་མི་གཟིགས་སོ་ཞེས་པ་ལ་སོགས་པ་རྒྱས་པར་བསྟན་པའི་ཕྱིར། བཙན་ཐབས་ཀྱི་སྒྲུབ་
ཐབས་ཀྱི་ཆོ་ག་ཉེ་བར་སྟོན་ཏོ། །མཐར་ཐུག་པ་ན་ཡང་ཇི་ལྟ་བ་བཞིན་དུ་ཟླ་བ་དྲུག་རྣམས་རིག་མའི་བརྟུལ་ཞུགས་སྤྱད་པས་སྙིང་གའི་འོད་ཟེར་གྱིས་སྤྲོས་པའི་ཁྲོ་བོ་རྣམས་ཀྱིས་དང་རང་ཉིད་ཀྱང་ཐལ་མོ་སྦྱར་བས་ཕྱོགས་བཅུ་རྣམས་སུ་སངས་རྒྱས་རྣམས་ལ་དངོས་གྲུབ་སྒྲུབ་པའི་
དོན་དུ་གསོལ་བ་བཏབ་སྟེ། ཏིང་ངེ་འཛིན་དུའམ་རྨི་ལམ་དུ་རྗེས་སུ་གནང་བས་སམ་མ་བཀག་པས། མཆོད་པ་དང་ཉེ་བར་བསྒྲུབ་པ་ལོངས་སྤྱད་པའི་དོན་དང་དངོས་གྲུབ་སྟེར་བའི་དོན་དུ། དེ་ལྟ་ཁོ་ནར་འདི་རྣམས་ལ་སྲོད་ལ་སྙན་གསན་ཕབ་སྟེ། ནང་པར་སྒྲུབ་
པའི་དཀྱིལ་འཁོར་མངོན་པར་སྤྲུལ་ནས་དེ་ལྟ་ཁོ་ནར་རྗེས་སུ་བསྒྲུབ་བོ། །དེ་ལྟ་ན་ཡང་མི་གཟིགས་ན་ཕྱོགས་བཅུ་ན་བཞུགས་པའི་སངས་རྒྱས་རྣམས་བློས་དམིགས་ཏེ་ཡང་དག་པར་མཆོད་ནས་བཟོད་པར་གསོལ་ལ་ཟླ་བ་དྲུག་པའི་རིག་མའི་བརྟུལ་ཞུགས་སྤྱད་ཅིང་
དེ་ནས་དེ་ལྟ་ཁོ་ནར་རྗེས་སུ་བསྒྲུབ་བོ། །དེ་ལྟ་ན་ཡང་མི་གཟིགས་ན་སླར་ཡང་དེ་ལྟ་ཁོ་ནར་ཟླ་བ་དྲུག་པའི་རིག་མའི་བརྟུལ་ཞུགས་སྤྱད་ནས་དེ་ལྟ་ཁོ་ནར་རྣམ་པར་བསྒྲུབ་བོ། །དེ་ལྟ་ན་ཡང་མ་གྲུབ་ན་ཉེས་པ་མེད་པའི་རྣལ་འབྱོར་པས་བཙན་ཐབས་ཀྱི་བསྒྲུབ་པ་བྱ་
