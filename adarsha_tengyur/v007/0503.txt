སུཾ། བཾ་མཾ། བཻ་ཙཾ། བཾ་མཾ་མོ། །འདིར་སོ་སོར་ས་བོན་གཉིས་ལ་ཨོཾ་དང་ཧྲཱིཿཡིག་སྤེལ་བ་སྟེ་དཔེར་ན། ཨོཾ་ཧྲཱིཿཨཻཾ་ཆེ་གེ་ཧྲཱིཿཨོཾ་ཞེས་པ་ལ་སོགས་པའོ། །དེའི་ཕྱི་རོལ་ཁོར་ཡུག་གཉིས་ཀྱི་དབུས་སུ་དེ་བཞིན་དུ། ཛྷཾ་ཉཾ། ཊ་ཋ་ཌཾ་ཌྷ་ཎཾ། ཏ་ཐ་དཾ་ངྷཾ་ནཾ། པཾ་ཕཾ་བཾ་བྷཾ་མཾ་ངོ་། དེའི་ཕྱི་རོལ་གཞལ་ཡས་
ཁང་གྲུ་བཞི་པའི་ཤར་སྒོའི་ནང་དུ། ཧཱུྃ་ཕཊ་ཡཾ་ཀཱ་ཡ་ཆེ་གེ་ཧཱུྃ་ཕཊ། དེ་བཞིན་དུ་བྱང་ལ་སོགས་པར། རཾ་ཨཱུ་རཾ་དང་། ལ་ཤྭཱུ་ལ་དང་། ཝཾ་སཱུ་ཝཾ་སྟེ། མེ་ལ་སོགས་པར། ཤཾ་ཡཾ་ཤཾ།  ཥ་ཡ་ཥ། སཾ་ཡཾ་ས། ཧཾ་ཡཾ་ཧཾ་ངོ་། །བདུན་པོ་རྣམས་ལ་ཡང་ས་བོན་གསུམ་པོ་སོ་སོ་ལ་ཧཱུྃ་ཕཊ་སྤེལ་ཞིང་དབུས་སུ་མིང་ཡང་ངོ་། །
སྒོའི་སྟེང་དུ་རྡོ་རྗེ་ཕྱེད་པ་རྩེ་ལྔ་པ་རྣམས་ཏེ། ཟུར་གྱི་སྟེང་དུ་ཟླ་བ་ཉི་མ་རྡོ་རྗེ་ཕྱེད་པ་རྣམས་སོ། །དེ་ལྟ་བུའི་རྡོ་རྗེ་བརྒྱད་པོ་རྣམས་ཀྱི་རྭ་དབུས་མ་རྣམས་ལ་སྔ་མ་ལྟ་བུའི་སྐོར་དག་གིས། ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་ཝཱི་རེ་ཤཱ་ཡ་ཞེས་པ་ལ་སོགས་པ་མཆོད་པའི་སྔགས་བརྒྱད་རྣམས་སོ། །དེ་ལས་ཕྱི་རོལ་ཁོར་ཡུག་གཉིས་ཀྱི་
དབུས་སུ་ཤར་ལ་སོགས་པར་གཡོན་སྐོར་གྱིས། ན་མཿས་མནྟ་ཀཱ་ཡ་ཝཱ་ཀ་ཙིཏྟ་བཛྲ་ཧེ་རུ་ཀཱ་ཎཱཾ་ཨོཾ་ཧཱུྃ་བཛྲ་ཧེ་རུ་ཀ་ཙ་ཀྲ་ཝརྟི། སརྦ་ལོ ཀི་ཀ་ལོ་ཀོཏྟ་རཱ་ནི་ཡནྟྲ་མནྟྲ་ཏནྟྲ་ཙུརྞ་པྲ་ཡོ་གཱ་དཱི་ནི། ཡེ་ན་ཀེ་ན་ཙི་ཏ་ཀྲཾ་ཏཱ་ནཱི་སརྦཱ་ཎི། ཨཱ་ཀཌྜྷྱ། ཨཱ་ཀཌྜྷྱ། ཏརྫ་ཡ། ཏརྫ་ཡ། གྲྀ་ཧྞཱ་གྲྀ་ཧྞཱ། བྷཀྵ་བྷཀྵ། ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ།ཧ་ན་ཧ་

རིན་པོ་ཆེའམ་རལ་གྲིའི་ཕྲེང་བའོ། །དེ་ནས་འོད་ཟེར་འབར་བའི་ཕྲེང་བ་བྲིའོ། །རིགས་ངན་མོའམ་བྲམ་ཟེ་མོའམ། རྒྱལ་པོའི་བུ་མོའི་མེ་ཏོག་དང་གླང་པོ་ཆེའི་འགྲམ་པའི་ཆང་དང་སྙོམས་འཇུག་ལས་སྐྱེས་པའི་ཁུ་བ་དང་གི་ཝང་དང་མར་ཁུ་ཆེན་པོ་ལས་བྱས་པའི་དུད་པ་བསྲེ་བར་བྱས་ནས་
