ཞེས་བྱ་བ་འདིས་ནི། བྱམས་པའི་རླན་མེད་ཅེས་བྱ་བའི་ཚིགས་བཅད་དེའི་གཉེན་པོ་བསྟན་ཏོ། །སྔོན་ཆད་ཕན་བཏགས་དྲིན་དུ་མི་གཟོ་བ། །ཞེས་བྱ་བ་འདིས་ནི། ལན་ཅིག་ཙམ་ཞིག་སྨ་བབ་གྱུར་པ་ཡིས། །ཞེས་བྱ་བའི་ཚིགས་བཅད་ཕྱེད་ཀྱི་གཉེན་པོ་བསྟན་ཏོ། །
བསླང་བ་དྲེགས་ན་ཞེས་བྱ་བ་ནི་བསླང་བ་ཉིད་དྲེགས་པ་ནི་བསླང་དྲེགས་པ་སྟེ། དེས་ནི་བྱམས་པ་མེད་པར་བྱེད་དོ་ཞེས་གཟུང་བར་བྱའོ། །གནས་པའི་སྐྱོན་ཞེས་བྱ་བ་ནི་ལྷན་ཅིག་ཏུ་འདུག་པ་ལ་སོགས་པའི་སྐྱོན་ཡིན་ཏེ། དེས་ན་དེའི་ཉེས་པ་ཡིན་པའི་ཕྱིར་ད་ནི་རེ་ཞིག་འགྲོའོ་
ཞེས་བྱ་བའི་དོན་ཏོ། །བར་ཆད་མང་ལ་གནོད་པ་དང་དགྲ་མང་བས་ཞེས་བྱ་བ་ནི་སེམས་ཅན་དང་སེམས་ཅན་མ་ཡིན་པའི་བར་ཆད་གང་ཡིན་པའི་རྒྱུ་བསྟན་ཏོ། །མཆི་བའི་སྐབས་དང་ལྡན་ན་ཞེས་བྱ་བ་ནི་རྐྱེན་ཚོགས་ནའོ། །གལ་ཏེ་ཞེས་བྱ་བ་ནི་དགེ་ཤིས་པའོ། །མངོན་པར་ཤེས་
པ་ལྔ་ནི་མངོན་པར་ཤེས་པ་དྲུག་ལས་ཟག་པ་ཟད་པའི་མངོན་པར་ཤེས་པ་མ་གཏོགས་པའོ། །སྙིང་བརྩེ་བས་རྗེས་སུ་འབྲང་བའི་ཚུལ་གང་ལ་ཡོད་པ་དེ་ནི་སྙིང་བརྩེ་བའི་དྲན་པ་བྱུང་བར་གྱུར་པ་སྟེ། སྙིང་བརྩེ་བ་ནི་སྙིང་རྗེ་སྟེ། མི་དགེ་བའི་རྒྱུ་ལས་བཟློག་པར་འདོད་པ་ཉིད་ནི་སྙིང་རྗེའོ། །
རྒྱུ་བརྟག་མི་བཟོད་པ་ཞེས་བྱ་བ་ལས་རྒྱུ་བསྟན་པར་མི་ནུས་པའི་ལྟ་བ་གང་ཡིན་པ་དེ་ནི་རྒྱུ་བརྟག་པ་མི་བཟོད་པའི་ལྟ་བའོ། །འདིར་རྒྱུ་ནི་གང་ཡང་མེད་པར་འདོད་པ་སྟེ། མེད་པར་འདོད་པ་ཉི་ཚེ་ཙམ་དུ་ཟད་པར་ལྟ་བའོ། །པདྨའི་སྡོང་བུ་ནི་འཇམ་པའོ། །ལོ་མ་ནི་འཇམ་པའོ། །ཁ་དོག་ནི་
སྔོན་པོ་ལ་སོགས་པའོ། །ཟེ་བ་ནི་འཇམ་པའོ། །སྙིང་པོ་ནི་དབྱིབས་སོ། །དེ་བཞིན་དུ་སེམས་ཅན་ལ་རྒྱུ་མེད་པར་ལྟ་བ་ནི་བྱའི་འདབ་མ་ཞེས་བྱ་བས་བསྟན་པ་ཡིན་ནོ། །འགྲོ་བ་འདི་དག་ངོ་བོ་ཉིད་ཀྱིས་ངེས་ཏེ། ཅིའི་ཕྱིར་ཞེ་ན། ཕྱི་དང་ནང་གི་དངོས་པོ་རྣམས་ལ་བྱེད་པ་པོ་མེད་པར་
