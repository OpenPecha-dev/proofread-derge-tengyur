རྩི་མྱོང་དང་སྤྲོ་བསྡུ་དང་། །བཟླས་པའི་རིམ་པའི་མན་ངག་ལས། །ཤེས་པས་དངོས་གྲུབ་འགྲུབ་པར་འགྱུར། །ཨོཾ་ཀཱ་ལ་རཱུ་པ་ཧཱུྃ། ཨོཾ་ཡ་མ་རཱ་ཛ། ས་དོ་མེ་ཡ། ཡ་མེ་དོ་རུ། ཎ་ཡོ་ད་ཡ། ཡ་ད་ཡོ་ནི། ར་ཡཀྵེ་ཡ། ཡཀྵེ་ཡཙྪ། ནི་རཱ་མ་ཡ། ཧཱུཾ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཧྲཱིཿཥྚྲྀཿཝི་ཀྲྀ་ཏཱ་ན་ན་

རྗེ་གཤེད་དུ་འགྱུར་བར་ཤོག །གཤིན་རྗེ་གཤེད་ནག་པོའི་སྒྲུབ་ཐབས། སློབ་དཔོན་མཁས་པ་ཆེན་པོ་དཔལ་མར་མེ་མཛད་ཀྱིས་མཛད་པ་རྫོགས་སོ།།      །།རྣལ་འབྱོར་པ་པྲཛྙཱ་ཤྲཱི་ཛྙཱ་ན་ཀཱིརྟིས་རང་འགྱུར་དུ་མཛད་པའོ།།      །།༄༅༅། །རྒྱ་གར་སྐད་དུ། བཻ་རོ་ཙ་ན་ཡ་མཱ་རི་ནོ་པི་ཀ་ནཱ་མ། བོད་སྐད་དུ། 
གཤིན་རྗེ་གཤེད་རྣམ་པར་སྣང་མཛད་ཀྱི་མངོན་པར་རྟོགས་པ་ཞེས་བྱ་བ། རྡོ་རྗེ་སེམས་དཔའ་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་རྡོ་རྗེ་རྟག་པའི་དཔལ། །མེ་ལོང་ཡེ་ཤེས་བྱང་ཆུབ་སེམས། །འོད་གསལ་རྡོ་རྗེ་ཀུན་ཞི་བ། །གཏི་མུག་རྡོ་རྗེ་ལ་ཕྱག་འཚལ། །ས་རྣམས་ཕ་རོལ་བགྲོད་པའི་ལམ། །
རྟག་པའི་སྒྲུབ་ཐབས་བཤད་པར་བྱ། །དང་པོར་སྔགས་པས་སྙིང་རྗེ་དྲན་པས་རང་ལུས་རྫོགས་པ་དང་། །སྙིང་གར་ས་བོན་གསལ་བ་ལས་ནི་དམ་པ་རྣམས་བསམས་ནས། །སངས་རྒྱས་ཆོས་དང་དགེ་འདུན་གསུམ་གྱི་ཡོན་ཏན་བསམ་བྱས་ནས། །བསོད་རྣམས་རྒྱ་ཆེར་རབ་ཏུ་རྒྱས་པའི་ཚོགས་རབ་
