རྟོག་པའོ། །སེམས་ཕྱིན་ཅི་ལོག་ཅེས་བྱ་བ་ནི་སེམས་ཕྱིན་ཅི་ལོག་བཞི་པོ་དག་ལ་ལྟ་བས་མངོན་པར་ཞེན་པའི་སེམས་འདོད་ཆགས་ལ་སོགས་པས་ཀུན་ནས་ཉོན་མོངས་པར་གྱུར་པའོ། །ལྟ་བ་ཕྱིན་ཅི་ལོག་ཅེས་བྱ་བ་ནི་འདུ་ཤེས་ཀྱིས་ཡོངས་སུ་རྟོག་པ་དེ་ཉིད་ལ་དེ་བཞིན་
དུ་བཟོད་པ་དང་། འདོད་པ་དང་། རྣམ་པར་འཇིག་པ་དང་། མངོན་པར་ཞེན་པའོ། །ཕྱིན་ཅི་མ་ལོག་པའི་དོན་ཅེས་བྱ་བ་ནི་ཕྱིན་ཅི་ལོག་གི་དོན་དེ་ལས་བཟློག་པ་ཡིན་ནོ། །ཀུན་ནས་ཉོན་མོངས་པའི་དོན་ཅེས་བྱ་བ་ནི་འཛིན་པའི་དོན་དེ་དག་ཉིད་ལ་ཕྱིན་ཅི་ལོག་གི་དོན་སྔོན་དུ་འགྲོ་བ་གང་ཡིན་པ་སྟེ།
དེ་ཡང་རྣམ་པ་གསུམ་མོ། །ཉོན་མོངས་པའི་ཀུན་ནས་ཉོན་མོངས་པ་ཞེས་བྱ་བ་ནི་ཉོན་མོངས་པ་དང་ཉེ་བའི་ཉོན་མོངས་པ་རྣམས་སོ། །དེ་དག་གི་རབ་ཏུ་དབྱེ་བ་ནི་རྣམ་པར་གཞག་སྟེ། ངོ་བོ་ཉིད་དང་། རབ་ཏུ་དབྱེ་བ་དང་། རྒྱུ་དང་། གནས་སྐབས་དང་། སྒོ་དང་། ཆེ་བ་དང་། ཕྱིན་ཅི་ལོག་
དང་། རྣམ་གྲངས་དང་། ཉེས་དམིགས་ཀྱིས་སོ། །དེ་ལ་ངོ་བོ་ཉིད་གང་ཞེ་ན། ཆོས་གང་སྐྱེ་བ་ན་བདག་ཉིད་ཀྱང་རབ་ཏུ་མ་ཞི་བའི་མཚན་ཉིད་དུ་སྐྱེ་ལ་དེ་སྐྱེས་པས་འདུ་བྱེད་ཀྱི་རྒྱུད་ཀྱང་རབ་ཏུ་མ་ཞི་བ་ཁོ་ནར་འཇུག་པའོ། །རབ་ཏུ་དབྱེ་བ་གང་ཞེ་ན། ཉོན་མོངས་པ་དྲུག་དང་། ཉེ་བའི་ཉོན་མོངས་
པ་མང་པོ་དག་གི་མཚན་ཉིད་རབ་ཏུ་དབྱེ་བའོ། །རྒྱུ་གང་ཞེ་ན། རྣམ་པ་དྲུག་སྟེ། རྟེན་དང་། དམིགས་པ་དང་། འདུ་འཛི་དང་། བཤད་པ་དང་། གོམས་པ་དང་། ཡིད་ལ་བྱེད་པ་ལས་ཉོན་མོངས་པ་སྐྱེ་བའོ། །གནས་སྐབས་གང་ཞེ་ན། རྣམ་པ་བདུན་ཏེ། བག་ལ་ཉལ་བ་དང་། ཀུན་
ནས་དཀྲིས་པ་དང་། ཀུན་བརྟགས་པ་དང་། ལྷན་ཅིག་སྐྱེས་པ་དང་། ཆུང་ངུ་དང་། འབྲིང་དང་། ཆེན་པོའི་གནས་སྐབས་སོ། །སྒོ་གང་ཞེ་ན། རྣམ་པ་གཉིས་ཏེ། ཀུན་ནས་དཀྲིས་པ་དང་། བག་ལ་ཉལ་གྱི་སྒོའོ། ཆེ་བ་གང་ཞེ་ན། རྣམ་པ་དྲུག་སྟེ། ལྷུང་བ་དང་། སྐྱེ་བ་དང་། རྒྱུད་དང་།
གཞི་དང་། ལས་ཀུན་ནས་སློང་བ་དང་། མཐར་གྱུར་པའོ། །ཕྱིན་ཅི་ལོག་གང་ཞེ་ན། རྣམ་པ་བདུན་ཏེ། ཕྱིན་ཅི་ལོག་བཞི་དང་། དེ་དག་ལ་འདུ་ཤེས་དང་། ལྟ་བ་དང་། སེམས་ཕྱིན་ཅི་ལོག་ཏུ་གྱུར་པའོ། །རྣམ་གྲངས་གང་ཞེ་ན། རྣམ་གྲངས་མང་པོ་འདི་ལྟ་སྟེ། ཀུན་དུ་སྦྱོར་བ་དང་། འཆི་བ་
