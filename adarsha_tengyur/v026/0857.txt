རྡོ་རྗེའི་ཕྱག་རྒྱ་པདྨོར་བྱས་པ་ལས། །ཞེས་པས་རྡོ་རྗེ་ཉི་མའི་ཕྱག་རྒྱ་བསྟན་ཏོ། །རྡོ་རྗེའི་ཕྱག་རྒྱ་ཡོངས་སུ་བཅིངས་ནས་ཞེས་པ་ལ་སོགས་པས་རྟ་མཆོག་གི་ཕྱག་རྒྱ་བསྟན་ཏོ། །རྡོ་རྗེ་ཐམས་ཅད་ཀྱི་ཕྱག་རྒྱ་དེ་ཤེས་པས་ཅི་ཞིག་བྱེད་ཅེ་ན། སྒྲུབ་པར་འདོད་པ་དང་། 
དངོས་གྲུབ་ལེན་པའི་གང་ཟག་གིས་རྡོ་རྗེ་ཐལ་མོ་བསྡམས་པ་དང་། གུང་མོ་གཉིས་ནང་དུ་བཅུག་པ་དང་། མཐེ་བོ་མཐེ་ཆུང་བརྐྱང་བས་ཡེ་ཤེས་འབེབས་པ་དང་། རྡོ་རྗེའི་ཕྱག་རྒྱ་ཡོངས་སུ་སྒྱུར་བས་ལྷའི་སྐུར་བསྒོམ་པ་དང་། གུང་མོ་ལྕགས་ཀྱུ་བྱས་པས་ཡེ་ཤེས་སེམས་དཔའ་
སྤྱན་དྲང་བ་ནི། རྡོ་རྗེ་སེམས་དཔའི་དངོས་གྲུབ་དང་བསྒྲུབ་པའི་ཡན་ལག་ཏུ་འགྱུར་རོ། །དེ་བཞིན་དུ་དེ་བཞིན་གཤེགས་པ་ལ་སོགས་པ་ལྔ་ལ་ཡེ་ཤེས་དབབ་པ་དང་། ཡེ་ཤེས་སེམས་དཔའ་སྤྱན་དྲངས་ནས་ལྷར་བསྒོམ་པར་བལྟ་བར་བྱའོ། །དེ་ཡང་དང་པོའི་རྡོ་རྗེ་སེམས་
དཔའ་དང་དབུས་མའི་ཧེ་རུ་ཀ་དང་། ཡེ་ཤེས་དབབ་པ་ཡེ་ཤེས་སེམས་དཔའ་ཞེས་གསུངས་སོ། །བསྟོད་པ་དང་བཅས་པའི་ཡུམ་གྱི་ཕྱག་རྒྱ་ནི། ཨཱ་ལི་ཀཱ་ལི་མཉམ་བྱས་ནས། །ཁ་ཡི་ཐད་ཀར་འདུད་པར་བྱ། །ཞེས་པ་དབྱངས་ཡིག་བཅུ་དྲུག་དང་། གསལ་བྱེད་སུམ་
ཅུ་རྩ་བཞི་ལས། དཀྱིལ་འཁོར་དྲུག་ལྷ་གང་རུང་བསྒོམས་པ་ལ་སྙིང་གར་ཉི་ཟླ་ཉིས་བརྩེགས་བསམ་པ་ནི། ཨཱ་ལི་ཀཱ་ལི་མཉམ་སྦྱར་ནས། །ཞེས་བྱའོ། །དེ་ལ་གཟུགས་ཀྱི་རྣམ་རྟོག་ཡིག་འབྲུ་དང་། སྒྲའི་རྣམ་རྟོག་གཉིས་ལས་སྒྲའི་རྣམ་རྟོག་གི་དབང་དུ་བྱས་ནས། ཨཱ་
ལི་ཀཱ་ལི་ནི་འབྱུང་རྔུབ་པོ། །མཉམ་པར་གནས་པ་ནི་ལྟེ་བའི་དབུས་ཀྱི་གནས་པའི་རླུང་ངོ་། །ཁ་ཡི་ཐད་ཀར་འདུད་པར་བྱ། །ཞེས་པ་ལ། ལྟེ་བ་སྙིང་ག་ལྐོག་མ་བྱེད་པ་གསུམ་གྱིས་འགྲིབ་པའི་གླིང་བུར་བརྡབས་ནས་སྒྲའི་གནས་བརྒྱད་ཀྱི་བྱེད་པས་ཕྱེ་སྟེ། དེའི་འོག་ཏུ་བཞི་པས་
གཟུང་དུ་རུང་བའི་སྒྲའི་རི་མོ་རྣམས་དཔག་བསམ་གྱི་ཤིང་གི་སྡོང་པོ་ལྟར་རྩ་བར་དགོད་པར་བྱ་བ་ཡིན་ལ་དབྱངས་ཡིག་བཅུ་དྲུག་གི་གུ་གུ་ལྕུག་མ་ལྟར་བྱས་ལ་གསལ་བྱེད་ཀྱི་སྟེང་དུ་ཡི་གེར་བཞག་པས་མགོ་བརྩེགས་མར་གྱུར་པ་ལྕུག་མ་ལས་མྱུ་གུ་སྐྱེས་པ་བཞིན་ནོ། །དེ་ལས་མེ་
