བསྒོམ་བྱ། །འདི་ལྟར་རྟག་ཏུ་བསྒོམས་པ་ཡིས། བདག་བྱིན་བརླབ་པ་འགྲུབ་པར་འགྱུར། །ཞེས་བྱ་བ་སྨོས་ཏེ། དེ་ལ་མདོར་བསྡུས་ཞེས་བྱ་བ་ནི་གོ་རིམས་བཞིན་དུའོ། །སྙིང་གི་དབུས་གྱུར་ཞེས་བྱ་བ་ནི་ཤེས་རབ་ཀྱི་པདྨའི་ནང་དུའོ། །ཟླ་བ་ནི་ཟླ་བའི་དཀྱིལ་འཁོར་རོ། །མི་ཤིགས་པ་ཞེས་
བྱ་བ་ནི་ཐིག་ལེ་དང་། ཟླ་ཚེས་དང་། དབྱངས་ཡིག་དྲུག་པས་བརྒྱན་པའི་ཡི་གེ་ཧཱུྃ་ངོ་། །འོད་ཟེར་སྣ་ཚོགས་ཀུན་དུ་འཕྲོ། །ཞེས་བྱ་བ་ནི་འོད་ཟེར་སྣ་ལྔ་དང་ལྡན་པའོ། །དེ་ཡིས་རྡོ་རྗེ་རྣམ་པར་བསྒོམ། །ཞེས་བྱ་བ་ནི་ཧཱུྃ་ཡོངས་སུ་གྱུར་པ་ལས་བྱུང་བའི་རྡོ་རྗེ་སྔོན་པོ་ནས་འབྲུ་ཙམ་བསམ་
པའོ། །རྡོ་རྗེས་བདག་མེད་མཁའ་འགྲོ་མ། །བསྒོམས་ནས་ཞེས་བྱ་བ་ནི་རྡོ་རྗེས་མཚན་པའི་དོན་ཏོ། །ཆོ་གས་བསྐུལ་ནས་སུ་ཞེས་བྱ་བ་ནི་དབང་པོ་གཉིས་ཀྱི་སྦྱོར་བས་དབབ་པའོ། །མཁའ་འགྲོ་མ་ཡི་སྣ་རྩེར་གནས། །ཞེས་བྱ་བ་ནི་གཉི་གའི་སྣའི་རྩེ་མོའི་མཚམས་སུའོ། །ཁམས་གསུམ་
རྒྱུ་དང་མི་རྒྱུ་བ། །ཞེས་བྱ་བ་ནི་རྟེན་དང་བརྟེན་པའི་དཀྱིལ་འཁོར་རོ། །དེར་ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་ལྟེ་བའི་ནང་དུའོ། །ཡང་ན་སྙིང་གའི་དབུས་གྱུར་ཞེས་བྱ་བ་ནི་རང་གི་སྙིང་གར་རོ། །ཟླ་བ་དང་མི་ཤིགས་པ་ནི་བཤད་ཟིན་ཏོ། །ཕྲ་མོ་ཞེས་བྱ་བ་ནི་རྡོ་རྗེ་ཡོངས་སུ་གྱུར་པའི་ཐིག་ལེ་སྟེ། མཁའ་འགྲོ་བདག་
མེད་མ་ཞེས་བྱ་བ་ནི་བདག་མེད་མ་ཉིད་སྟོན་པའོ། །ཆོ་གས་བསྐུལ་ནས་སུ་ཞེས་བྱ་བ་ནི་སྔར་བཤད་པ་བཞིན་དུ་སྙིང་ག་ནས་བསྐུལ་བའོ། །གང་དུ་ཞེ་ན། མཁའ་འགྲོ་མ་ཡི་སྣ་རྩེར་གནས། །ཞེས་བྱ་བ་སྟེ། པདྨའི་སྣའི་རྩེ་མོར་རོ། །ལྷག་མ་ནི་སྔར་བཤད་པ་བཞིན་དུ་སྦྱར་རོ། །ཆོ་གའི་གོ་
རིམས་རིམ་པ་ཡིས། །དེ་ནས་མཚན་མ་མཚོན་པར་བྱ། །ཞེས་བྱ་བ་ལ། ཆོ་གའི་གོ་རིམས་ནི་གོམས་པ་ཕུལ་དུ་བྱུང་བའོ། །རིམ་པ་ཞེས་བྱ་བ་ནི་མཚན་མ་རྣམས་རིམ་གྱིས་སྐྱེ་བའོ། །གཞན་དག་ན་རེ་རིམ་པའི་མཚན་ཉིད་ཐ་དད་པ་སྟེ་དུས་གཅིག་ཏུ་སྐྱེ་བར་འགྱུར་རོ་ཞེས་ཟེར་རོ། །དང་པོ་
སྤྲིན་གྱི་རྣམ་པ་སྟེ། །ཞེས་བྱ་བ་ནི་ཕྲ་མོའི་འོད་ཟེར་དང་པོ་སྤྲིན་དང་འདྲ་བར་མཐུག་པ་སྟེ། ཧེ་རུ་ཀའི་འོ། །གཉིས་པ་དུ་བ་ལྟ་བུ་ཡིན། །ཞེས་བྱ་བ་ནི་འོད་ཟེར་གཉིས་པ་མཁའ་ལ་ཆགས་པའི་དུད་སྤྲིན་ལྟར་སྣང་བ་སྟེ་རྟག་པའི་འོ། །གསུམ་པ་སྲིན་བུ་མེ་ཁྱེར་འདྲ། །ཞེས་བྱ་བ་མཁའ་
