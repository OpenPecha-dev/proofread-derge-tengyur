རུ་ཡང་དེ་ཉིད་ཀྱི་ལྟ་བས་ཐམས་ཅད་བྱ་ཞེས་པ་སྟེ་ངེས་པའོ། །ཉི་ཤུ་གསུམ་པའོ།།      །།དེ་བཞིན་དུ་ལྷ་རྣམས་དག་པར་འདོད་པས་ཐིག་ལེའི་བསྒྲུབ་པ་གསུངས་པ། ལྷ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། རྡོ་རྗེ་བཅིངས་པ་ཞེས་པ་ནི་སྔོན་དུ་གསུངས་པའི་མན་ངག་ཉིད་དོ། །རྟོག་པ་དང་པོའི་རབ་
ཏུ་བྱེད་པར་ཡང་གསུངས་པའོ། །དེ་བཞིན་དུ་རྟོག་པ་བཅུ་གཅིག་པ་ལས་ཀྱང་བསྟན་པར་བྱའོ། །དེ་བཞིན་དུ་རྡོ་རྗེ་མེ་ཏོག་ཅེས་པ་ནི་མཆོག་ཏུ་མི་འཛག་པའི་བྱང་ཆུབ་ཀྱི་མེ་ཏོག་གོ། །དེ་ཉིད་ཀྱིས་ན་སྒྲུབ་པ་པོ་དེ་ཡང་། ཀུ་ཐར་ཚིནྡྷ་དང་བསྲེས། །ཞེས་པ་ནི་མི་རྒྱུའོ། །ཉི་མ་གཟས་ཟིན་པ་ཞེས་པ་
ནི་སྡུད་པའི་དུས་སུ་སྒྲ་གཅན་གྱི་རྩ་ལ་འཇུག་པའོ། །མི་བསྐྱོད་པ་དང་སྟ་རེ་ཞེས་པ་ནི་གང་གི་ཕྱིར་མི་འཁྲུགས་པས་ནི་མི་བསྐྱོད་པ་སྟེ་མི་གཡོ་བའི་ཏིང་ངེ་འཛིན་ཞེས་པའི་དོན་ཏོ། །སྟ་རེ་ཞེས་པ་ནི་ཀུན་གཞི་རྣམ་པར་ཤེས་པ་སྟ་རེ་རྣོན་པོ་བཞིན་དུ་གྱུར་པ་དེ་ཉིད་ཀྱིས་རྣམ་པར་རྟོག་པ་ཐམས་ཅད་གཅོད་
པར་བྱེད་པའོ། །རྐང་པས་མནན་ཏེ་ཞེས་པ་ནི་རྐང་པའི་སྒྲས་ནི་དགའ་བ་བཞིའི་ལྷག་མ་ལྷན་ཅིག་སྐྱེས་པའི་དགའ་བའི་བཟུང་སྟེ། དེས་མནན་པ་ནི་སྟོང་པའི་སྟ་རེའོ། །དེའི་ཕྱིར་སྔགས་བཟླས་པ་ཞེས་པ་ནི་ཨོཾ་ལ་སོགས་པ་དུས་ཀྱི་ཡི་གེ་རྣམས་སོ། །དེས་ན་བརྗོད་པའི་དུས་ཀྱི་ཞེས་པ་སྟེ་ངེས་
པའོ། །ཐིག་ལེ་ཞེས་པ་ནི་སྦྱོར་བ་འདིས་སྟོང་པ་ཉིད་ལ་ཞུགས་པའོ། །གང་ལ་ཕྱག་བྱས་པ་ཞེས་པ་ནི་དངོས་པོ་གང་དང་གང་བྱུང་བ་དེ་དང་དེ་ཐམས་ཅད་དེ་བཞིན་ཉིད་དུ་ཞུགས་པས་ན་དེ་རྣམས་འཇིག་པའོ། །ཕྱི་རུ་ཡང་ལྷ་རྣམས་དག་པས་ནི་དེ་ལ་བརྟེན་པའི་ཕྱིར་ཞེས་དགོངས་སོ། །ཉི་ཤུ་
བཞི་པའོ།།  །།ད་ནི་སྤྲིན་རྣམས་ལས་ཆར་དབབ་པའི་ཆོ་ག་གསུངས་པ། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། ཆོ་ག་བཞིན་ཞེས་པ་ནི། དུར་ཁྲོད་ཀྱི་ས་དང་སོལ་བ་དང་དུག་དང་། སྐྱེ་ཚེ་ལ་སོགས་པ་དང་ལྡན་པས་བྱ་བའོ། །དཀྱིལ་འཁོར་བྲིས་ལ་ཞེས་པ་ནི་སྔོན་དུ་གསུངས་པའི་དགྱེས་
པའི་རྡོ་རྗེའི་དཀྱིལ་འཁོར་ཏེ། དེར་བདག་པོ་ནས་བརྩམས་ཏེ་དཀྱིལ་འཁོར་བ་ཡང་ངོ་། །དེ་བཞིན་དུ་མཐའ་ཡས་མནན་པའི་དགྱེས་པའི་རྡོ་རྗེའི་གཟུགས་ཞེས་པ་ནི་སྒྲུབ་ཐབས་ཀྱི་རིམ་པས་བསྒོམ་པར་བྱའོ། །ཨ་རྫ་ཀ་ཞེས་པ་ནི་པ་ན་སའོ། །ལྷག་མ་ནི་གོ་སླའོ། །ཉི་ཤུ་ལྔ་པའོ།།      །།
