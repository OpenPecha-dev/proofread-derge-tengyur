ཡང་རིག་མར་བཅས་པ་ཞེས་པ་ནི་སྙིང་རྗེ་དང་ལྷན་ཅིག་གོ། །དེའི་ཚེ་སྐུལ་བར་བྱེད་ཅེས་པ་ནི་ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་ཐིམ་པས་ཡང་དཀྱིལ་འཁོར་བཞིའི་རྣལ་འབྱོར་མས་སད་པའི་ངོ་མཚར་གྱིས་ལངས་པའོ། །གླུ་བླངས་པས་ཞེས་པ་ནི་རང་བྱུང་གི་སྒྲས་སོ། །
དེ་ཡང་རིག་མ་སྣ་ཚོགས་ཞེས་པ་ནི་གཙུག་ཏོར་ལ་སོགས་པའི་འཁོར་ལོ་དྲུག་ལ་གནས་པའི་རྣལ་འབྱོར་མ་རྣམས་ཀྱི་ནང་རྣམས་སུ། སྒྲའི་དབྱངས་རང་བྱུང་ཉིད་ལངས་པའོ། །དེ་བཞིན་དུ་སྙིང་རྗེ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་འཁོར་ལོ་དྲུག་རྣམས་ཡི་གེ་བརྒྱ་བཞི་བཅུ་རྩ་བདུན་དུ་འགྱུར་ཏེ།
དེ་ལ་གཙུག་ཏོར་དུ་ལྔའོ། །དཔྲལ་བར་བཅུ་བཞིའོ། །དེ་བཞིན་དུ་མགྲིན་པར་སུམ་ཅུའོ། །སྙིང་གར་བརྒྱད་དོ། །དེ་བཞིན་དུ་ལྟེ་བར་དྲུག་ཅུའོ། །གསང་བར་སུམ་ཅུའོ། །དེ་བཞིན་དུ་འདིར་གསུངས་པ། ཨུ་ཐ་བྷ་རཱ་ངྷོ་ཞེས་པ་ལ་ཨུ་ནི་སྤྱི་བོའི་སྣལ་མ་སྟེ། ཐ་ནི་དབུས་
ཀྱི་ཤར་ཉིད་དུ། །བྷ་ནི་དབུས་ཀྱི་ལྷོ་རུའོ། །དེ་བཞིན་བྱང་དུ་རཱ་ཉིད་དོ། །ནུབ་ཏུ་ངྷོ་ཞེས་བྱ་བའོ། །འདིར་ནི་རྟག་ཏུ་ནམ་མཁའ་ལྔ། །སྒྲ་ལྔ་ཡི་གེ་ལྔ་རྣམས་ཉིད། །རིགས་ནི་ལྔ་རུ་གསུངས་པ་ཡིན། །དེ་ལྟར་རྩ་ནི་ལྔ་རྣམས་ལ། །འཁོར་ལོ་རྣམ་པ་ལྔ་ནི་གནས། །རྩ་ཡི་དངོས་
པོའི་རང་བཞིན་ཉིད། །ཡི་གེ་ཡི་ནི་ནང་དུ་འདུས། །རྩ་ནི་ལྔ་པོ་རེ་རེ་ལ། །འཁོར་ལོ་ལྔ་ནི་རིམ་པ་བཞིན། །འཁོར་ལོའི་རྣམ་པར་གནས་པ་ནི། །ཡི་གེ ལྔ་ཡི་རྣམ་དག་ཉིད། །དེ་ལྟར་དེ་བླ་མེད་ཕུང་པོའོ། །འཁོར་ལོའི་རྣམ་པ་ཉིད་དུ་གནས། །འདིར་ནི་ཕྲ་མོའི་རྣལ་འབྱོར་
བརྗོད། །བཅོམ་ལྡན་འདས་ནི་ཡི་གེར་བཞུགས། །ཀ་རུ་ཎ་མ་ཎ་པུཀྐ་སཱི་མ་ཧུ་པ་རི་ཏཱ་ཧི་མ་ཞེས་པ་ལ། ཀ། ཐོད་པའི་དབུས་སུ་རྩ་ནི་གང་། རུ། འདོད་པའི་རྣལ་འབྱོར་ལ་ནི་གཞོལ། ཎ། གུས་པ་དང་ནི་འདུལ་བའི་གཟུགས། མ། བྱང་ཆུབ་ཀྱི་ནི་བདེ་བས་བརྒྱན། ཎ། སྟོན་པའི་ཚིག་
ནི་རྗེས་སུ་འགྲོ། །པུ། སྔོན་གྱི་ཆོས་ནི་སྟོན་པའི་ཕྱིར། སྐ། ལས་ཉིད་བག་ཆགས་སྔོན་འགྲོ་བ། །སཱི། རྩ་ནི་བཅུ་བཞི་རྣམས་ནི་འགྲུབ། མ། དབུ་མ་ཡང་ནི་གཙུག་གཏོར་དུ། །ཧུ་ནི་ནམ་མཁའི་མིང་ཅན་ཏེ། པ། ལྷག་མ་ཡང་ནི་རྣམ་པ་བཞི། །རི་ནི་མགྲིན་པའི་འཁོར་ལོ་སྟེ། ཏཱ། དེ་
