ཆེན་བཅུ་དྲུག་ཕྲག་གཉིས་དགོད། དེ་ན་ཀཱ་ལི་ཀ་ཆེན་སུམ་ཅུ་རྩ་གཉིས་ཀྱི་སྟེང་དུ། ཡ་ར་ལ་ཝ། ཤ་ཥ་ས་ཧ་དང་། ངྷ་ཌ་གཉིས་ཏེ་བཞི་བཅུ་རྩ་གཉིས་དགོད། །དེ་ཉི་མར་གྱུར་ཏེ། དངོས་སུ་ཨཱ་ལི་ཟླ་བ་སྟེ་ཐབས་ཀྱི་བདག་ཉིད་ཀཱ་ལི་ཉི་མ་སྟེ་ཤེས་རབ་ཀྱི་བདག་ཉིད་དོ། །དེ་ནས་ཉི་མ་མནན་ནས་ནི། །
ཞེས་པ་ལ། ཟླ་བའི་འོད་ཉི་མའི་འོད་ཀྱིས་མི་སྣང་ཞིང་གཅིག་ཏུ་གྱུར་པས་ཉི་མ་ཉིད་དུ་སྣང་ངོ་། །བདེ་བ་ཆེན་པོ་གཉིས་ཨཱ་ལི་ཀཱ་ལི་གཉིས་ལས། ཟླ་བ་དང་ཉི་མ་གཉིས་གཅིག་ཏུ་གྱུར་པའི་དོན། དེ་གཉིས་སུ་མེད་པ་ལས་འོག་མའི་གཽ་རི་ལ་སོགས་པར་གྱུར་པའི་དོན། དེ་ལ་ཟླ་བ་ནི་མེ་ལོང་ལྟ་བུའི་ཡེ་

སྦྱར་རོ། །དེ་གཉིས་ནི་མཉམ་པ་ཉིད་ཀྱི་ཡེ་ཤེས་སོ། །རང་གི་ལྷ་ཡི་ས་བོན་དགོད། །ཅེས་བྱ་བ་ལ། ལྷ་མོ་རྣམས་ཀྱི་ཡིག་འབྲུ་དང་ཕྱག་མཚན་ཟངས་གྲི་དགོད་པའི་དོན། དེ་ནི་སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་ཀྱི་བདག་ཉིད་དོ། །སྐུ་ཡོངས་སུ་རྫོགས་པ་ནི། བྱ་བ་གྲུབ་པའི་ཡེ་ཤེས་ཀྱི་བདག་ཉིད་དེ། 
སྐུ་ལྔ་ཉིད་དུ་གྱུར་པ་སྟེ། །ཞེས་བྱ་བ་ནི། ཟླ་བ་དང་ཉི་མ་དང་ཡིག་འབྲུ་དང་། ཕྱག་མཚན་དང་། སྐུ་ཡོངས་སུ་རྫོགས་པ་དང་། ལྔ་ལས་གྱུར་པའི་དོན་དེ་ལྟ་བུའི་ཆོ་ག་མཁས་པས་བཤད་དོ། །ཨཱ་ལི་ཀཱ་ལི་མཉམ་སྦྱོར་ཏེ། །ཞེས་བྱ་བ་ནི། ཨཱ་ལི་ཀཱ་ལི་གཉིས་ལས་སྐུ་རྫོགས་པའི་ཚུལ་དུ་བྱུང་བས། 
རྡོ་རྗེ་སེམས་དཔའི་བདག་ཉིད་དེ། སྐུ་རྫོགས་པའི་ཚུལ་དུ་བྱུང་བ་ཡིན་ལ། ཡི་གེ་ལས་མ་བྱུང་ཞེས་བྱ་བར་སྦྱར་རོ། །དེ་ནི་དངོས་ཧཱུཾ་ལས་གཙོ་བོ་བསྐྱེད་པ་དང་། ཕཊ་ལས་འཁོར་བསྐྱེད་པ་ལ་སོགས་པ་མི་དགོས་ཞེས་བྱ་བའི་དོན། སེམས་ཅན་གཟུགས་བརྙན་ལས་བྱུང་བ། །དཀྱིལ་
འཁོར་གཙོ་བོར་རྣམ་པར་བསྒོམ། །སྲས་ཀྱི་ཞལ་དང་མཚན་ལ་སོགས། །ཟླ་བ་ནོར་བུ་རིན་ཆེན་འོད། །ཅེས་པ་ལ། འདིར་སློབ་དཔོན་དག་མི་མཐུན་པར་སྟོན་ཏེ། པཎྜི་ཏ་རིན་ཆེན་འབྱུང་གནས་ཞི་བའི་ཞལ་ནས་ནི། སེམས་ཅན་གྱི་སེམས་ཀྱིས་སེམས་རྡོ་རྗེའི་དཀྱིལ་འཁོར་གྱི་གཙོ་བོའི་བདག་
