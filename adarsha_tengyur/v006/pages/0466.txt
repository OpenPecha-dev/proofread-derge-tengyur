རུ། ནད་སེལ་བ་ཡི་སྨན་པ་མོ། །ག་གདོང་འགེགས་ལྷ་མོ་ཁྱིམ་པ་མོ། །དེ། དེ་ལྟར་ལྟེ་བའི་གཡོན་ལོགས་ཀྱི། །འཁོར་ལོ་འདི་ནི་སྤྲུལ་པའོ། །རྣལ་འབྱོར་མའི་ཆ་ལུགས་མང་པོ་ནི། །ལུས་ཀྱི་དངོས་པོ་ལས་ནི་བྱུང་། །རྣལ་འབྱོར་མ་ཡི་རྒྱུད་འདི་ཉིད། །དུས་ཀྱི་རྩ་ལས་ཡང་
དག་འབྱུང་། །འདི་ཉིད་ཀྱི་ནི་གཡས་ལོགས་ལ། །ཐུགས་ཀྱི་འཁོར་ལོའི་རྩ་བརྒྱད་གནས། །ཏ་ཞེས་པ་ནི་མ་ཉིད་དེ། །ཨི་ནི་དེ་བཞིན་བུ་མོའོ། །ན་ཞེས་པ་ནི་ཆ་མོའོ། །དེ་ནི་དེ་བཞིན་སྲིང་མོའོ། །མ་ཞེས་པ་ནི་སྒྱུག་མོའོ། །བ་ནི་དེ་བཞིན་གཉེན་མོའོ། །ཛི་ནི་ཕ་ཡི་སྲིང་
མོའོ། །ཨ་ཡི་ཁྱད་བྱེད་ཡི་གེ་གཉིས། །དེ་ལ་ཨ་ནི་ནམ་མཁར་འགྱུར། །ཨི་ཞེས་པ་ནི་རླུང་དང་ལྡན། །རྩ་བཅུ་པོ་ནི་རླུང་གི་གཟུགས། །ཨི་ཌ་ལ་སོགས་ལྟེ་བའི་མིང་། །ཡི་གེ་རྣམས་ནི་རླུང་གི་མིང་། །དེ་ལྟར་རྩ་ནི་དྲུག་རྣམས་ཀྱི། །ཡི་གེའི་དོན་དང་རྗེས་སུ་མཐུན། །ཡི་གེས་རྣལ་
འབྱོར་རྣམས་ནི་བརྗོད། །མིང་ནི་དམ་ཚིག་གི་ནི་སྐད། །དེ་བཞིན་ལྟེ་བའི་འཁོར་ལོ་ཡི། །རྣལ་འབྱོར་མ་ཡི་གནས་ཀྱི་མ། །དྲུག་ཅུ་རྩ་བཞི་ཡི་ནི་གྲངས། །མྱུ་གུ་ལ་སོགས་ཡི་གེ་རྣམས། །མུ་ཞེས་པ་ནི་ཁ་ཡི་གནས། །གྷོ་སྣའི་བུ་ག་གཡོན་པ་ལ། །གྷུ་ནི་སྣ་ཁུང་གཡས་སུ་
འདོད། །དེ་ལྟར་གུ་གུ་གཉིས་ནི་མིག །དེ་བཞིན་རྣ་གཉིས་ལ་མྱུ་གུ། །གཅི་བའི་བུ་ག་དང་བཤང་སྒོ། །གཉིས་ལ་ལུ་གྷུ་གཉིས་ནི་བརྗོད། །དེ་བཞིན་འདིར་ནི་དྲ་ཞེས་པ། །ཁུ་བའི་བུ་གའི་ནང་དུའོ། །ལྟེ་བའི་བུ་གའི་གནས་རྟག་ཏུ། །ཥྛ་ཞེས་པ་ནི་གཅིག་པུར་བརྗོད། །དེ་ལྟར་དགོངས་
བཤད་གཞན་མ་ཡིན། །ས་ནི་བཅུ་གཉིས་དག་ལ་གནས། །དེ་ཉིད་ཀྱིས་ན་དེ་དག་ཅེས་བྱ་བ་ལ་རྣལ་འབྱོར་པས་མཐོང་ཞེས་བྱ་བ་ནི་སྤྱོད་ཡུལ་ལོ། །ས་བཅུ་གཉིས་ཀྱི་ཚུལ་གྱིས་གྲུབ་པའི་ཚིག་གསུངས་པའི་ཕྱིར་རོ། །འདི་རྣམས་ཀྱི་མཚན་རྣམ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་ཁམས་རྣམས་
གནས་སུ་བརྗོད། ཏ་ཞེས་པ་ནི་མཐིལ་ནི་ལྔ། །ཏྱ་ནི་སོར་མོ་ཉི་ཤུའོ། །ར་ནི་མཚན་མ་ཞེས་སུ་གསུངས། །ཌ་ད་ཞེས་པ་སྒོ་ང་གཉིས། །པུ་སུ་གཉིས་ནི་ཕྲག་བརྒྱ་པ། །རྐེད་པར་མ་ཡོ་ཡི་གེ་གཉིས། །བྷི་ངྷི་སྤྱི་བོ་མགྲིན་པར་རོ། །ལུ་ཞེས་པ་ནི་སྙིང་གར་གནས། །སྠི་ཞེས་པ་ནི་གནས་
