དྲི་མས་མ་གོས་པ་ལྟར། ནང་གི་ཆུ་སྐྱེས་ཁ་ཕྱེ་བ་ལྟར་བསྟན་ལ། ཤེས་རབ་མས་འདི་བརྗོད་དེ། ཨེ་མའོ། །བདེ་བ་ཐམས་ཅད་དང་ལྡན་པ། །བདག་གི་པདྨ་རྨད་བྱུང་ལ། །གང་ཞིག་ཆོ་གས་མཉེས་བྱེད་པ། །དེ་ཡི་མདུན་དུ་བདག་ནི་འདུག །ཡང་དག་སངས་རྒྱས་མཉེས་བྱེད་ཕྱིར། །
ཇི་ལྟར་པདྨའི་དོན་གྱིས་ཤིག །དེ་ཉིད་བདེ་ཆེན་རྒྱལ་པོ་འདི། །འདི་ཡི་གནས་ན་རྟག་ཏུ་བཞུགས། །བདེ་ཆེན་དམར་ཐིག་བསྙེན་པར་གྱིས། །ཁ་ཡི་ཆུ་སྐྱེས་ཀྱིས་ནིའོ། །བྷཉྫ་མོ་ཀྵ་ཧོ། །དེ་ནས་སྒྲུབ་པ་པོས་མན་ངག་ཇི་ལྟ་བ་བཞིན་དུ་ས་བོན་དང་ཕྱག་མཚན་གྱི་སྦྱོར་བས་བདག་མེད་མའི་
གཟུགས་བསམ་མོ། །དེའི་སྙིང་ག་དང་། མགྲིན་པ་དང་། སྤྱི་བོར་ཧཱུྃ་ཨཱཿཨོཾ་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་བསམ་པར་བྱའོ། །དེ་ནས་རྡོ་རྗེ་དང་པདྨ་དག་པར་བྱའོ། །ཨོཾ་པདྨ་སུ་ཁ་དྷ་ར་མ་ཧཱ་རཱ་ག་སུ་ཁཾ་ད། ཙ་ཏུར་ཨཱ་ནནྡ་བྷ་ག་བི་ཤྭ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཀརྱཾ་ཀུ་རུ་ཥྭ་མེ། ཨོཾ་བཛྲ་མ་ཧ་དྭེ་ཁ། ཙ་ཏུ་ར་ཨཱ་ནནྡ་ད་ཡ་
ཀཿཁ་ག་མུ་ཁེ་ཀཱ་ར་སོ་ནཱཾཐཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཀརྱཾ་ཀུ་རུ་ཥྭ་མེ། །འདི་དག་གིས་དག་པར་བྱ་སྟེ། ཤིན་ཏུ་བདེ་བ་མཚོན་ཞིང་བསྐྱེད་པའི་དོན་དུ་བལྟ་བ་དང་། དགོད་པ་དང་། ལག་བཅངས་དང་། འཁྱུད་པའི་སྦྱོར་བ་དང་ལྡན་པར་བྱས་ལ། བདག་མེད་མའི་པདྨའི་དབུས་སུ་རྩ་གསུམ་གྱི་བདག་ཉིད་ལྕེ་འམ་ལག་
པའི་སྦྱོར་བས་བཙལ་བར་བྱ་སྟེ། ཇི་སྲིད་བདེ་བ་མ་རྒྱས་ཀྱི་བར་དུ་བསྒྲུབ་པར་བྱའོ། །དེ་ལ་བལྟ་བ་ལ་སོགས་པ་ནི་ཚོགས་དང་། སྦྱོར་ལམ་དང་། མཐོང་བ་དང་། བསྒོམ་པ་དང་ཁྱད་པར་མཐར་ཕྱིན་པའི་ལམ་གྱི་བར་དུ་བར་མ་ཆད་པར་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སུ་གྱུར་པ་ནི་བླ་
མའི་མན་ངག་བརྒྱུད་པའི་རིམ་པ་ལས་ཤེས་པར་བྱའོ། །ནོར་བུའི་རྩེ་མོར་ཕྱིན་པའི་བྱང་ཆུབ་ཀྱི་སེམས་དེ་ནི་ལྷན་ཅིག་སྐྱེས་པའི་བདེ་བ་ཆེན་པོའི་བདག་ཉིད་དབང་པོ་ཐམས་ཅད་ཀྱིས་རྣམ་པར་གྲོལ་བ་ཡུལ་མེད་པའི་དམིགས་པ་དང་བཅས་པའི་ཕྱིར་དོན་དམ་པའི་ལྷན་ཅིག་སྐྱེས་པ་མཚོན་
པར་བྱེད་པ་སྟེ། དཔེའི་ལྷན་ཅིག་སྐྱེས་པའི་གཟུགས་སོ། །བླ་མའི་མན་ངག་དང་གོམས་པ་དང་། མནན་པའི་སྦྱོར་བ་དང་། རླུང་གི་རྣམ་པར་རྒྱལ་བ་དང་ཕྱི་རོལ་གྱི་གཟུངས་མ་ལ་རླུང་ཁ་སྦྱོར་དུ་བྱ་སྟེ་བསྟན་པར་བྱས་ནས་དོན་དམ་པ་ལྷན་ཅིག་སྐྱེས་པ་མཐོང་བར་འགྱུར་རོ། །གལ་
