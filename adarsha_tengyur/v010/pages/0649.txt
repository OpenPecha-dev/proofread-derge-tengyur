རབ་མཆོག་འབྲས་བུ་སྦྱིན། །ཇི་ལྟར་དུག་ཤེས་པ་ཡིས་དུག་གསོལ་བ། །དེ་ལ་ནམ་ཡང་སྐྱོན་ནི་བྱེད་མི་འབྱུང་། །ཐ་མར་ཡང་ནི་རབ་མཆོག་འབྲས་བུ་སྦྱིན། །མཛེ་ནད་ཚབས་པོ་ཆེ་ཡང་མེད་པར་བྱེད། །དེ་ལྟའི་འགྲོས་སུ་བཅས་པའོ། །གང་ཞིག་བདག་གིས་འདི་རུ་
བསྟན། །ཡོན་ཏན་དུ་མ་དགེ་བའི་ལམ། །འགྲོ་བ་ཡི་ནི་ཐབས་ཡིན་ནོ། །ཉམས་སུ་མྱོང་བའི་མཐར་ཕྱིན་པ། །དེ་ཉིད་ཤེས་པ་བརྟེན་པ་ཡིན། །དེ་དང་དྲང་དོན་བྱེ་བྲག་ཏུ། །གསང་བ་ཆེན་པོ་ཡེ་ཤེས་མཆོག །རྣལ་འབྱོར་རྣལ་འབྱོར་མ་རྒྱུད་དག །རྡོ་རྗེ་སེམས་དཔས་རྒྱས་བཏབ་པ། །དེ་བས་
གང་ཞིག་རྒྱུད་རྒྱས་བཏབ། །དེ་ལ་རང་གིས་སྒྲུབ་མི་བྱེད། །གལ་ཏེ་དེ་ནི་ལོག་པར་འགྲོ། །ཤེས་པ་མ་ཡིན་གྲུབ་མ་ཡིན། །འོན་ཀྱང་དེ་ལ་ལོག་པ་སྟེ། །མྱུར་དུ་དམྱལ་བར་འགྲོ་བར་འགྱུར། །དེ་བས་ཀུན་དུ་འབད་པ་ཡིས། །སྒྲུབ་ཐབས་ལ་སོགས་བླ་མའི་བཀའ། །སྤྱད་པ་སྒྲུབ་པའི་དབང་
སྔགས་པ། །ཐེ་ཚོམ་མེད་པར་དངོས་གྲུབ་ཐོབ། །འདི་ནི་བདག་གིས་ཡང་དག་ལམ་ཉིད་བསྟན། །གསང་བའི་གསང་མཆོག་སྙིང་པོ་ཀུན་དགའ་བ། །བླ་མའི་རིམ་པ་གཤེགས་པ་སྔོན་མེད་པ། །རབ་ཀྱི་མན་ངག་རྙེད་པ་རྨད་བྱུང་ངོ་། །འདི་ནི་དཔལ་ཨུ་རྒྱན་རྡོ་རྗེ་ཆེན་པོའི་གནས་ནས་
བྱུང་བ། དཔལ་དེ་ཁོ་ན་ཉིད་ཀྱི་བདུད་རྩི་མན་ངག་དཔལ་ཨིནྡྲ་བྷཱུ་ཏི་རང་ལས་བྱུང་བའོ། །སློབ་དཔོན་ཆེན་པོ་རྒྱལ་པོ་ཨིནྡྲ་བྷཱུ་ཏིའི་ཞལ་སྔ་ནས། དཔལ་དེ་ཁོ་ན་ཉིད་ཀྱི་བདུད་རྩིའི་མན་ངག །མཆོག་གི་དངོས་གྲུབ་བརྙེས་ནས་མཛད་པ་རྫོགས་སོ།།      །།རྒྱ་གར་གྱི་མཁན་པོ་པྲཛྙཱ་
ཤྲཱི་གུཔྟ་དང་། བོད་ཀྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་རྒྱལ་མཚན་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།།      །།༄༅༅། །རྒྱ་གར་སྐད་དུ། རཏྣ་མཱ་ལཱ་ཨུ་པ་དེ་ཤ། བོད་སྐད་དུ། དབང་རིན་ཆེན་ཕྲེང་བའི་མན་ངག ། དཔལ་བདེ་བ་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །བདེ་
བ་ཆེན་པོ་འཁོར་ལོའི་བར། །ལངས་ནས་སྤྲུལ་པའི་འཁོར་ལོ་གཟིགས། །ཟླ་བ་ཉི་མའི་ལམ་དེ་ཡི། །ཨ་བ་འདུད་ལ་ཕྱག་འཚལ་ལོ། །རིན་ཆེན་རྡོ་རྗེ་ཞབས་བཀའ་དྲིན། །ཐམས་ཅད་བདག་རྩའི་འཁོར་ལོ་མཁྱེན། །དེ་དབུས་རྩ་གསུམ་དེ་ཁོ་ན། །ཤེས་རབ་དཔལ་བདག་གསལ་
