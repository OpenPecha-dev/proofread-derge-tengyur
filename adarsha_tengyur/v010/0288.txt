གཏོར་མ་གཏོང་བའི་དུས་སུ་རང་གང་འདོད་པའི་བསྐུལ་ཚིག་དང་། དམོད་བཅོལ་དང་། འོ་དོད་བོད། སུ་བ་བྱ། མཱ་ར་ཡ་དང་བཅས་པས་གཏོར་མ་གཏང་། དཔལ་ཀྱེའི་རྡོ་རྗེའི་གཏོར་མའི་ཆོ་ག་ཞེས་བྱ་བ་རྫོགས་སོ།།      །།༄༅༅། །རྒྱ་གར་སྐད་དུ། ཧེ་བཛྲ་བ་ལིཾ་ཏ་བི་དྷི་ནཱ་མ། བོད་སྐད་དུ། 
ཀྱེའི་རྡོ་རྗེ་ཞེས་བྱ་བའི་གཏོར་མའི་ཆོ་ག །ཀྱེའི་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །ནམ་མཁའ་ཀུན་དུ་རྒྱུ་གྲུབ་པའི། །མཁའ་འགྲོ་རྣམས་ལ་རབ་ཏུ་འདུད། །དེ་དང་དེ་ཡི་བྱ་བ་ནི། །གཏོར་མའི་ཆོ་ག་ཅུང་ཞིག་བཤད། །དེ་ལ་གཏོར་མ་རྣམས་གསུམ་སྟེ། །དཀྱིལ་འཁོར་ཆེན་པོའི་ལྷ་རྣམས་དང་། །དེ་སྤྲུལ་
མཁའ་འགྲོ་མ་རྣམས་དང་། །མགྲོན་གྱུར་ཕྱོགས་སྐྱོང་རྣམས་ལའོ། །དཀྱིལ་འཁོར་ཆེན་པོའི་ལྷ་ལ་ནི། །རེ་ཞིག་བྱིན་གྱིས་བརླབ་པ་ཡང་། །བདག་ཉིད་འདོད་པའི་ལྷར་མོས་པས། །སྣོད་དག་མཆོག་ཏུ་གྱུར་པ་རུ། །གཏོར་མའི་རྫས་ནི་བདུད་རྩིར་བསམ། །དེ་བསྟན་དེ་ཉིད་གསུམ་སྔགས་
བརྗོད། །ཡང་ན་འདོད་ཡོན་ལྔར་གྱུར་པར། །བསམ་ཞིང་བྱིན་གྱིས་བརླབས་སྔགས་བརྗོད། །དེ་རྗེས་མདུན་དུ་དཀྱིལ་འཁོར་བའི། །ལྷ་རྣམས་སྐད་ཅིག་རྡོར་འཛིན་བསམ། །དེ་རྣམས་དབུལ་བའི་རིམ་པ་ནི། །ཨོཾ་མ་ཧཱ་བྷཱུ་ཏ་ཁཱ་ཧི། གྲྀཧྞཱ་གྲྀཧྞཱ་བྷུཉྫ་བྷུཉྫ་ཁ་ཁ་ཁཱ་ཧི། ཨི་དཾ་བ་ལིཾ་ཏ་ཁཱ་ཧི། ཨ་
མྲྀ་ཏ་རཱ་ས་ཁཱ་ཧི། ཀཱ་མ་གུ་ཎི་ས་མཱ་ར་ས་ཁཱ་ཧི། ཞེས་གླུ་བླང་སོ་སོའི་མཚན་དང་སྦྱར་བ་ནི། ཨོཾ་ཧེ་བཛྲ་ནེེ་ར་ཨཱཏྨཱ་ཏ་ཨི་དཾ་མ་ཧཱ་བ་ལིཾ་ཏ་ཁཱ་ཧི། དེ་བཞིན་དུ་ཨོཾ་བཛྲ་གཽ་རཱི་ཨི་དཾ་མ་ཧཱ་བ་ལིཾ་ཏ་ཁཱ་ཧི། ཞེས་བྱ་བ་ལ་སོགས་པ་བརྗོད་དོ། །དེ་ལྟར་ཚིམ་ཞིང་མཉེས་གྱུར་ནས། །བདག་ཉིད་བསྒྲུབ་པ་རྐྱེན་སྦྱོར་བའི། །

ལ་རབ་བསྐུལ་ལ། །དེ་རྣམས་དབུལ་བའི་རིམ་པ་ནི། །ལྷག་མ་བསྡུས་པའི་རིམ་པ་ལས། །སྔགས་འདི་གླུ་རུ་བཅས་པ་བརྗོད། །ཨོཾ་ཨུ་ཙིཥྚ་བ་ལིཾ་ཏ་བཛྲ་ཌཱ་ཀི་ནཱི་སཾ་པ་ནི་ན། བཛྲ་ཌཱ་ཀི་ནཱི། རཏྣ་ཌཱ་ཀི་ནཱི། པདྨ་ཌཱ་ཀི་ནཱི། ཨི་དཾ་ཨུ་ཙིཥྚ་བ་ལིཾ་ཏ་ཁ་ཁ་ཁཱ་ཧི། གྲྀཧྞཱ་གྲྀཧྞཱ། བྷུཉྫ་བྷུཉྫ། པྲ་ཏཱིཙྪ་ཁཱ་ཧི། ཀརྨ་
