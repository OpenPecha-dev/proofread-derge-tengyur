བྱང་ཆུབ་ཀྱི་ཚོགས་དང་མཐུན་པའི་ཆོས་བཞི་སྟེ། དེ་དག་གི་གཞིའི་ཕྱིར་བཞི་ཡིན་པས་མི་ཉུང་མི་མང་ངོ་ཞེས་སྦྱར་རོ། །བྱང་ཆུབ་ཀྱི་ཚོགས་དང་མཐུན་པའི་ཆོས་བཞི་པོ་དག་གང་ཞེ་ན། གདམས་ངག་དང་རྗེས་སུ་བསྟན་པ་འདོད་པ་ཉིད་དང་། དེ་འཐོབ་པ་
དང་། དེའི་སྣོད་ཉིད་དང་། དེ་སྒྲུབ་པའོ། །བཞི་པོ་དེ་དག་ནི་བྱང་ཆུབ་ཀྱི་ཚོགས་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཞེས་བྱ་བའི་རྗེས་སུ་མཐུན་པ་དེ་སྒྲུབ་པར་བྱེད་པའོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེ་ལ་བྱང་ཆུབ་ཀྱི་སེམས་ཡོངས་སུ་མི་གཏོང་བ་ནི་གདམས་ངག་དང་རྗེས་སུ་བསྟན་པ་ཉིད་ཀྱི་ཕྱིར་ཡིན་ཏེ། 
འདི་ལྟར་བྱང་ཆུབ་ཀྱི་སེམས་མ་བཏང་ན་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་དགེ་བའི་བཤེས་གཉེན་ལ་གདམས་ངག་དང་རྗེས་སུ་བསྟན་པ་ནོད་པར་འདོད་དེ། དཔེར་ན་འཕགས་པ་ནོར་བཟངས་ལྟ་བུའོ། །དེ་འཐོབ་པ་ཞེས་བྱ་བ་ནི་གདམས་ངག་དང་རྗེས་སུ་བསྟན་པ་འཐོབ་པ་སྟེ། དེ་ཡང་དགེ་
བའི་བཤེས་གཉེན་ལ་བརྟེན་ནས་དེ་གཉིས་འཐོབ་པའི་ཕྱིར་རོ། །དེའི་སྣོད་ཉིད་ཅེས་བྱ་བ་ནི་དེ་གཉིས་ཀྱི་སྣོད་ཉིད་དེ། གདམས་ངག་རྗེས་སུ་བསྟན་པའི་སྣོད་ཅེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེའི་གཞི་ནི་བཟོད་པ་དང་དེས་པའོ། །དེའི་སྒྲུབ་པ་ཞེས་བྱ་བ་ནི་གདམས་ངག་དང་རྗེས་སུ་བསྟན་པ་དེ་གཉིས་
སྒྲུབ་ཅེས་བྱ་བའི་ཐ་ཚིག་སྟེ། དེའི་གཞི་ནི་དགོན་པ་ལ་གནས་པ་ལ་བྱའོ། །དེ་ནི་ཞི་གནས་དང་ལྷག་མཐོང་དང་ཚུལ་ཁྲིམས་ཡང་དག་པར་འགྲུབ་པའི་རྒྱུ་ཡིན་ནོ། །གྲོང་ནི་འཕྲལ་གྱི་བྱ་བ་མང་བའི་ཕྱིར་མ་ཡིན་ནོ། །ཡང་བྱང་ཆུབ་ཀྱི་ཚོགས་དང་མཐུན་པའི་ཆོས་བཞི་སྟེ། དེ་དག་གི་གཞིའི་
ཕྱིར་བཞི་ཞེས་བྱ་བར་སྦྱར་རོ། །བཞི་པོ་དག་གང་ཞེ་ན། དོན་དུ་གཉེར་བ་ཉིད་དང་། ཐབས་ཤེས་པ་ཉིད་དང་། སྒྲུབ་པ་སྟེ། དེ་ལ་སྒྲུབ་པ་ནི་རྣམ་པ་གཉིས་ཏེ། སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱེད་པ་དང་། སངས་རྒྱས་ཀྱི་ཆོས་ཡོངས་སུ་སྨིན་པར་བྱེད་པའོ། །དེ་ལ་དོན་དུ་གཉེར་བ་
ཉིད་ཀྱི་གཞི་ནི་བྱང་ཆུབ་ཀྱི་སེམས་ཡོངས་སུ་མི་གཏོང་བའོ། །ཉན་ཐོས་དང་རང་སངས་རྒྱས་བྱང་ཆུབ་ཀྱི་སེམས་བཏང་བ་དག་ནི་སངས་རྒྱས་ཀྱི་ཡོན་ཏན་དོན་དུ་གཉེར་བར་མི་འགྱུར་རོ། །ཐབས་ཤེས་པ་ཉིད་ཅེས་བྱ་བ་ནི་སངས་རྒྱས་ཉིད་ཐོབ་པར་བྱེད་པའི་ཐབས་ཤེས་པ་ཉིད་ཅེས་བྱ་བའི་ཐ་ཚིག་གོ། །
