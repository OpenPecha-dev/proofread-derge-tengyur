དང་ན་བའི་རྐྱེན་སྨན་རྣམས་སོ། །དེའི་རྟེན་ཞེས་བྱ་བ་ནི་གནས་ཀྱི་རྟེན་ནོ། །ཡོངས་སུ་དོར་བའི་སེམས་ཀྱིས་གསོད་པ་མཐར་ཕྱིན་པར་བྱེད་ལ། གཞན་ནི་རྩུབ་པའི་སེམས་ཀྱིས་རྫོགས་པར་བྱེད་དེ། དེའི་ཕྱིར་གསོད་པ་དང་གནོད་སེམས་དང་ཚིག་རྩུབ་པོ་
རྣམས་ནི་གཅིག་ཏུ་ཞེ་སྡང་གིས་མཐར་ཕྱིན་པར་བྱེད་དོ། །ཆེར་རྨོངས་པར་གྱུར་ན་ཞེས་བྱ་བ་ནི། གང་ཞིག་རྒྱུ་དང་འབྲས་བུས་འབྲེལ་པའི་ཆོས་ཉིད་ལ་རྨོངས་པས་དེ་ནི་རྫོགས་པར་བྱེད་དེ། གཏི་མུག་ཁོ་ནས་ལོག་པར་ལྟ་བ་མཐར་ཕྱིན་བྱེད་དོ། །ལྷག་མ་གསུམ་
གྱིས་རྫོགས་པར་འདོད་ཅེས་བྱ་བ་ལ། ཆགས་པ་ལས་སྐྱེས་པའི་བརྫུན་དུ་སྨྲ་བ་ནི་ཆགས་པ་ཁོ་ན་མཐར་ཕྱིན་པར་བྱེད་ལ། དེ་བཞིན་དུ་ཞེ་སྡང་ལས་སྐྱེས་པ་དང་། གཏི་མུག་ལས་སྐྱེས་པ་ཡང་ཞེ་སྡང་དང་གཏི་མུག་གིས་མཐར་ཕྱིན་པར་བྱེད་དོ། །དེ་བཞིན་དུ་གཞན་
གཉིས་ལ་ཡང་བརྗོད་པར་བྱའོ། །གཞན་དག་ན་རེ་བརྣབ་སེམས་ལ་སོགས་པ་ཆགས་པ་དང་ཞེ་སྡང་དག་གིས་མཐར་ཕྱིན་པར་བརྗོད་པར་བྱ་བ་མ་ཡིན་ཏེ། དེ་མཐར་ཕྱིན་པའི་དུས་སུ་དེ་ལ་ཐ་དད་པའི་ཆགས་པ་དང་ཞེ་སྡང་མེད་པའི་ཕྱིར་དེ་ཉིད་ཀྱིས་མཐར་ཕྱིན་པ་མ་ཡིན་ནོ་ཞེས་ཟེར་རོ། །
དེའི་ཕྱིར་གཏི་མུག་གིས་བརྣབ་སེམས་ལ་སོགས་པ་མཐར་ཕྱིན་པར་བརྗོད་པར་བྱ་སྟེ། གཏི་མུག་ལས་དེ་སྐྱེས་པའི་ཕྱིར་རོ། །འདིར་ལན་སྨྲས་པ། གང་གི་ཕྱིར་དེ་ཉིད་ཀྱིས་དེ་བསྟན་པར་མི་འགྱུར་ཏེ། རྟེན་མ་ཡིན་པས་འདི་དེ་ལྟར་བརྟགས་པས་དགེ་བའི་རྩ་
བ་རྣམས་ལས་གཞན་པ་སྲིད་པ་ཡོད་དོ་ཞེས་ཅིས་མི་འདོད། དེ་ལྟ་ན་འོ་ན་ལས་ཀྱི་ལམ་ཐམས་ཅད་ཀྱང་ངེས་པར་གཏི་མུག་དང་ལྷན་ཅིག་འབྱུང་བའི་ཕྱིར་གཏི་མུག་ཁོ་ནས་རྫོགས་པར་ཐལ་བར་འགྱུར་རོ་ཞེ་ན་མི་འགྱུར་རོ། །ཆགས་པ་དང་ཞེ་སྡང་དག་གི་དེར་
གཙོ་བོ་ཡིན་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་ལོག་པར་ལྟ་བ་ཡང་གཏི་མུག་གིས་ཡོངས་སུ་རྫོགས་པ་མ་ཡིན་ཏེ། དེར་དེ་གཙོ་བོ་ཡིན་པའི་ཕྱིར། གཙོ་བོ་ཉིད་དེ་ལས་གཞན་པའི་དགེ་བའི་རྩ་བ་ཡོད་པའི་ཕྱིར། ལོག་པར་ལྟ་བ་དགེ་བའི་རྩ་བའི་བདག་ཉིད་མ་ཡིན་
