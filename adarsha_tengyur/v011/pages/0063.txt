རྒྱའི་དངོས་གྲུབ་འགྱུར། ༡༡༥ །སྲོག་རྩོལ་རབ་ཏུ་བྱ་སྟེ་སྙིང་ག་དེ་བཞིན་མགོ་བོ་དག་ནི་ཇི་སྲིད་མེ་ཡིས་གདུང་འགྱུར་བ། །དེ་ལས་སྟེང་དག་ཏུ་ནི་བཟང་པོའི་པདྨར་བརྒྱལ་བར་འགྱུར་ཏེ་མ་བསྡམས་པའམ་སྟོབས་དག་གིས། །གཙུག་ཏོར་ཕུག་ཅིང་
ཕུང་པོ་དག་དང་ཁམས་རྣམས་རྣམ་པར་སྤངས་ནས་འཆི་བ་ཉིད་དུ་འགྲོ་འགྱུར་ཏེ། །ཕྱག་རྒྱ་འགྲོགས་པ་རབ་ཉམས་སྐྱེས་བུ་རྣམས་ཀྱི་སྐྱེ་བ་འདི་ཉིད་ལ་ནི་བདེ་འབྲས་སྟེར་བ་མེད། ༡༡༦ །ཕྱག་རྒྱ་བསྒོམ་པའི་དོན་དུ་གསུངས་ཏེ་ཉིན་མཚན་དུས་སུ་ཆགས་པ་ཟད་པའི་དོན་ནི་མིན་
པ་ཉིད། །གསུང་གི་རྡོ་རྗེ་མཆོད་པའི་དོན་ཏེ་ངེས་པར་སྔགས་པ་རྣམས་ལ་མྱོས་བྱེད་གསུངས་པ་ཉིད་མ་ཡིན། །ཁ་ཟས་ཐམས་ཅད་བདེ་བའི་དོན་ཏེ་ཉིན་ཞག་སོ་སོའི་དུས་སུ་མ་ཞུའི་རྒྱུར་ཡང་མ་གསུངས་སོ། །དཔལ་ལྡན་སྤྱོད་པ་དངོས་གྲུབ་རྒྱུར་ཏེ་མཚན་མ་དུར་ཁྲོད་
འཁྱམ་པ་དག་ཀྱང་རྩེད་མོའི་དོན་མ་ཡིན། ༡༡༧ །མི་བསྐྱོད་པ་ནི་ཅུང་ཟད་དྲོ་བ་ཁ་ནད་ཞི་བར་བྱེད་དེ་སོ་ཡི་ཟུག་རྔུ་ཉིད་ཀྱི་ཡང་། །ཐོ་རངས་མི་བསྐྱོད་རྔུབས་པ་དག་ནི་མགོ་ཡི་ནད་འཕྲོག་དེ་བཞིན་ཉིད་དུ་ཆུ་ནི་རྔུབས་པས་སོ། །རྣ་བ་དག་དང་མིག་ཏུ་གཅིན་ནི་དྲོ་དང་གྲང་མོ་རབ་བླུགས་
གཉིས་ཀ་དག་གི་ནད་འཕྲོག་གོ། །འབྱུང་པོས་གཟིར་ལ་ཚ་བ་གསུམ་བཅས་མི་བསྐྱོད་རྔུབས་ན་བདེ་བ་སྟེར་ཏེ་ཟོས་པ་ཉིད་ལ་ཡང་། ༡༡༨ །བཤང་བ་གཅིན་དང་ཁུ་བ་ཁྲག་ནི་མི་ཡི་ཤ་དང་བཅས་པ་ཟོས་པས་ཚེ་ནི་སྟེར་བར་འགྱུར། །བསམ་གཏན་བཅས་པར་མེ་ཏོག་རྔུབས་
ན་སྐྲ་དཀར་དང་བཅས་རྒ་བ་དང་ནི་ལུས་སྐྱེས་རྣམས་ཀྱང་འཕྲོག །སྒྲོན་མ་ལྔ་པོ་སྦྲང་བུའི་སྐྱུགས་པ་དག་དང་བསྲེས་པ་ཟོས་པས་མཐའ་དག་ནད་ནི་འཕྲོག་པ་སྟེ། །བུད་མེད་མེ་ཏོག་ཁུ་བ་དག་དང་བསྲེས་པ་ལོ་གཅིག་ཟོས་ན་རྒ་བ་ངེས་པར་སེལ་བར་བྱེད། ༡༡༧ །རླུང་འཇོམས་ལན་ཚྭ་སྐྱུར་
བ་དག་ཏུ་རབ་ཏུ་འགྱུར་ཏེ་མངར་དང་བསྐ་བ་དག་ནི་མཁྲིས་པའི་དགྲ། །བད་ཀན་འཇོམས་པ་ཁ་བ་ཀུན་དང་དེ་བཞིན་ཚ་བ་དག་ཀྱང་སྨན་རྣམས་སམ་ནི་ཁུ་བ་འམ། །བད་ཀན་འཇོམས་པ་ར་ཡི་འོ་མ་ཚ་བ་གསུམ་བཅས་དག་སྟེ་མ་ཧེའི་འོ་མ་མཁྲིས་པའི་དགྲ། །རླུང་འཇོམས་དག་
