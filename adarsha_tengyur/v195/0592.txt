ཨལ་འགྱུར་ཞིང་བྱ་དྷ་དང་ཛས་དག་ལས་ཀྱང་ངོ་། །མ་དཿ། བྱ་དྷཿ། ཛ་པཿ། ཉེར་བསྒྱུར་མེད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། པྲ་མཱ་དཿ། ཨཱ་བྱཱ་དྷཿ། ཨུ་པ་ཛཱ་བཿ། སྭན་ཧས་དག་གི་ཡང་ནའོ། །ཉེར་བསྒྱུར་མེད་པར་གྱུར་པ་ལ་སྭན་དང་ཧས་དག་ལས་ཨལ་ཡང་ན་འགྱུར་རོ། །སྭ་ནཿ། སྭ་ནཿ། ཧ་སཿ།  
ཧ་སཿ། ཉེར་བསྒྱུར་མེད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། པྲ་སྭ་ནཿ། པྲ་ཧཱ་སཿ། སཾ་ནི་ཨུ་པ་བི་རྣམས་ལའང་ཡམ་ལས་སོ། །ཉེར་བསྒྱུར་མེད་པར་གྱུར་པ་ལ་ཡམ་ལས་ཨལ་ཡང་ན་འགྱུར་ཞིང་། སཾ་དང་ནི་དང་ཨུ་པ་དང་བི་རྣམས་ཉེ་བའི་ཚིག་ལའང་ངོ་། །ཡ་མཿ། ཡཱ་མཿ། སཾ་ཡ་མཿ། སཾ་ཡཱ་མཿ། ནི་ཡ་མཿ། 
ནི་ཡཱ་མཿ། ཨུ་པ་ཡ་མཿ། ཨུ་པ་ཡཱ་མཿ། བི་ཡ་མཿ། བི་ཡཱ་མཿ། ནི་ལ་གད་ནད་པ་ཋ་སྭན་རྣམས་ཀྱིའོ། །ནིའི་སྒྲ་ཉེ་བའི་ཚིག་ལ་གད་དང་ནད་དང་པ་ཐ་དང་སྭན་ལས་ཨལ་ཡང་ན་འགྱུར་རོ། །ནི་ག་དཿ། ནི་གཱ་དཿ། ནི་ན་དཿ། ནི་ནཱ་དཿ། ནི་པ་ཐཱཿ། ནི་པཱ་ཐཿ། ནི་སྭ་ནཿ། ནི་སྭཱ་ནཿ། ཉེར་བསྒྱུར་མེད་པ་
ཞེས་པའདིར་སྤྲེལ་བར་བྱ་སྟེ་རྣམ་པ་གཞན་དུ་ན་སྭན་ཧས་དག་གི་ཞེས་སྲན་སྨོས་པ་དོན་མེད་པར་འགྱུར་བའི་ཕྱིར་རོ། །ཀྭཎའི་སྟེ་པི་ཝང་ལའོ། །ནི་ལ་ཞེས་པ་འཇུག་ཅིང་ཉེར་བསྒྱུར་མེད་པ་ཞེས་པ་དང་ཡང་ན་ཞེས་པའང་ངོ་། །ནི་ཉེ་བའི་ཚིག་དང་ཉེར་བསྒྱུར་མེད་པ་ལའང་ངོ་། །ཀྭཎ་ཏི་ལས་པི་ཝང་གི་ཡུལ་
