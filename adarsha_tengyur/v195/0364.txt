བརྗོད་ཀྱི་ན་བ་ད་ར་རྣམས་ཀྱི་མ་ཡིན་ནོ། །ཙ་སྡེའི་ཊ་སྡེ་ཞེས་པས་དའི་ཌའོ། །ཨཽཎྜི་ད་ཏི། ཨཽནྫི་ཛཏ། ཨཱ་ནཌྜི་ཌཏ། རུལ་བར་བྱེད་པ་པོ་ལ་སྦྱོར་རོ། །ལུགས་བྱེད་པ་པོ་ལ་སྦྱོར་རོ། །འཕུར་བར་བྱེད་པ་པོ་ལ་སྦྱོར་རོ། །ཟ་བ་པོ་ལ་སྦྱོར་རོ། །མཆོད་པ་པོ་ལ་སྦྱོོར་རོོ་ཞེས་
པས་ཨིནྲའོ། །དིའོ། །ཤྲི་དྲུ་སྲུ་ཀ་མི་ཀཱ་རི་ཏའི་མཐའ་ལས་ཞེས་པས་ཙཎའོ། །གཉིས་བརྗོད་དོ། །ཀཱ་རི་ཏ་དབྱིའོ། །དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ་ཞེས་པས་ཨུ་ཡིག་འཕེལ་བའོ། །ཨ་ཡིག་གི་ཨ་ཡིག་གོ་ཞེས་པས་ཨཱ་ཡིག་གོ། །ན་བ་ད་ར་ཞེས་པ་ཅི་ཞེ་ན། ཨཱི་
ཙིཀྵིཥ་ཏི། ཨཱི་ཀྵ་མཐོང་བ་ལའོ། །སནྲ་དང་ཨི་ཊའོ། །ཀྵི་གཉིས་བརྗོད་དོ། །ཀ་སྡེའི་ཙ་སྡེའོ། །བརྩེགས་པ་དང་པོ་ཞེས་པ་ཅི་ཞེ་ན། ཨཱ་རི་རི་ཥ་རི། རྀ་འགྲོ་བ་ལའོ། །འགྲོ་བར་འདོད་དོ་ཞེས་སནྲའོ། །སྨིང་སུང་རྀ་ཨཉྫ་ཞེས་པས་ཨིཊའོ། །རི་གཉིས་བརྗོད་དོ། །ཡ་མིན་པ་ཞེས་
པ་ཅི་ཞེ་ན། ཨཱར་ཪྻ་ཏེ། རྀ་འགྲོ་བ་ལའོ། །འཁྱོག་པོར་འགྲོའོ་ཞེས་པས་ཡ་སྒྲ་སྟེ་བཤད་པ་ལས་སོ། །ཙ་ཀྲྀ་ཡི་ཏ་ལ་ཞེས་པ་ཡོན་ཏན་ནོ། །རྨ་གཉིས་བརྗོད་དོ། །ཨ་བྷྱཱ་སའི་གསལ་བྱེད་དང་པོ་ཞེས་པས་ཡ་དབྱིའོ། །ཨཱ་ག་མ་མེད་པའི་རིང་པོའོ་ཞེས་པས་རིང་པོའོ། །སྔ་མ་ཨ་བྷྱཱ་
སའོ། །གཉིས་བརྗོད་དོ་ཞེས་པས་རྗེས་འཇུག་གོ། །དེའང་དོན་གྱི་དབང་ལས་བདུན་པའི་མཐའ་སྦྲེལ་ལོ། །བྱིངས་གཉིས་བརྗོད་པ་བྱས་པ་ལ་སྔ་མའི་ཆ་ལ་བྷ་སའི་མིང་ཅན་དུ་འགྱུར་རོ། །པ་པཱ་ཙ། པ་པཱ་ཐ་ཨ་ཊ། གཉིས་བརྗོད་དོ། །ཨ་བྷྱཱ་སའི་མིང་ཡོད་པ་ན་ཨ་བྷྱཱ་སའི་གསལ་
བྱེད་དང་པོ་ཞེས་པས་ཙ་ཡིག་དང་ཐ་ཡིག་དབྱིའོ། །ཨུ་པ་དྷཱའི་ཨ་ཞེས་པས་རིང་པོའོ། །ཤི་ཤྲ་ཡ། དུ་དཱ་ཝ། ཤིཉ་བསྟེན་པ། དྲུ་འགྲོ་བ་ལའོ། །པ་རོ྅ཀྵའི་ཨཊའོ། །གཉིས་བརྗོད་དོ། །འཕེལ་བའོ། །ཨ་བྷྱཱ་སའི་མིང་གིས་གང་དུ་དགོས་ཤེ་ན། ཨ་བྷྱཱ་སའི་གསལ་བྱེད་དང་པོ་
ཞེས་ལ་སོགས་པ་རྣམས་སུའོ། །གཉིས་པ་ཨ་བྷྱ་སྟའོ། །གཉིས་བརྗོད་དོ་ཞེས་པ་རྗེས་སུ་འཇུག་གོ། །བྱིངས་གཉིས་བརྗོད་པ་བྱས་པའི་གཉིས་པ་གང་ཡིན་པ་དེ་ཨ་བྷྱ་སྟའི་མིང་ཅན་དུ་འགྱུར་རོ། །ད་ད་ཏི། ད་ད་ཏུ། ད་དྷ་ཏི། ད་དྷ་ཏུ། ཌུ་དཱཉ། སྦྱིན་པ་ལའོ། །ཌུ་དྷཱཉ་འཛིན་པ་ལའོ། །
