ཏི་བྷུ་རྣམས་ལས་ཞེས་པས་སི་ཙ་འཇིག་གོ། །སི་ཙ་འཇིག་པ་ཞེས་པ་ཅི་ཞེ་ན། བྱ་ཏྱ་བྷ་བིཥྚི། ཀྲྀ་སྲི་པྲྀ་བྷྲི་སྟུ་དྲུ་ཤུ་སྲུ་རྣམས་ཉིད་ལས་པ་རོ྅ཀྵཱ་ལའོ། །ཌུ་ཀྲྀཉ་བྱེད་པ་ལའོ། །སྲྀ་འགྲོ་བ་ལའོ། །ཌུ་བྷྲིཉ་འཛིན་པ་ལའོ། །བྲྀ་ཞེས་པ་བྲྀང་དང་བྲྀཉ་དག་གོ། སྟུཉ་བསྟོད་པ་ལའོ། །དྲུ་འགྲོ་
བ་ལའོ། །ཤྲུ་རྟོགས་པ་ལའོ། །སྲུ་འགྲོ་བ་ལའོ། །འདི་རྣམས་ལས་པ་རོ྅ཀྵ་ལ་ཨིཊ་ཨཱ་ག་མ་མི་འགྱུར་རོ། །ཙ་ཀྲྀབ། ཙ་ཀྲྀམ། དེ་བཞིན་དུ། ས་སྲྀབ། ས་སྲྀམ། བ་བྷྲྀབ། བ་བྷྲྀམ། བ་བྲིབ། བ་བྲིམ། ཏུཥྚུབ། ཏུཥྚུམ། དུ་དྲུམ། ཤུ་ཤྲུམ། སྲུ་སྲུ་མ། ཀྲྀ་ལ་སོགས་པ་རྣམས་ཀྱི་དགག་པ་གྲུབ་
པ་ལ་སྦྱོར་བ་བརྩམས་པ་ནི་ནུས་པའི་དོན་དུའོ། །ཀྲྀ་རྣམས་ཉིད་ལས་པ་རོ྅ཀྵཱ་ལ་ཨིཊ་མེད་པ་དེས་གཞན་རྣམས་ཀྱི་ཨིཊ་བཀག་པ་ལས་ཀྱང་པ་རོ྅ཀྵ་ལ་ཨིཊ་ཨཱ་ག་མའོ། །པེ་ཙི་བ། ཀིམ། གསལ་བྱེད་གཅིག་པའི་དབུས་ཞེས་པས་ཨེ་ཉིད་དང་ཨ་བྷྱས་དབྱི་བ་དག་གོ། །དགག་ཚིག་ངེས་པར་བསྟན་
པ་མི་རྟག་པ་སྟེ། དེས་ན་བྲྀཉ་ལས་ཐལ་ལ་དགག་པ་མི་ནུས་ཏེ། པ་བ་རི་ཐ། རྀ་ཡིག་ལས་ཐལ་ལའོ། །རྀ་ཡིག་མཐའ་ཅན་ལས་ཐལ་ལ་ཨིཊ་ཨཱ་ག་མ་མི་འགྱུར་རོ། །སྔ་མའི་ངེས་པ་ཐོབ་པ་ལའོ། །དདྷྭརྠ། ཛཧརྠ། དྷྲྀ་གནད་པ་ལའོ། །ཧྲྀཉ་འཕྲོག་པ་ལའོ། །ཤྭསྟྭཱ་ནཱི་ལ་རྟག་ཏུ་ཨིཊ་མེད་པའི་ངེས་པ་ཉེ་བར་
བསྟན་པ་ལ་ཨ་ཡིག་ལྡན་པའི་བྱིངས་ལས་ཐལ་ལ་ཨིཊ་ཡང་ན་བཤད་པར་བྱའོ། །ཉེ་བར་བསྟན་པ་ནི་མདོ་བཀླགས་པ་དང་བྱིངས་བཀླགས་པའང་ངོ་། །པ་པཀྠ། པེ་ཙེ་ཐ། སཱསྟ་ནཱི་ལ་ཨིཊ་མེད་པ་ཞེས་པ་ཅི་ཞེ་ན། ཛ་གྷ་སི་ཐ། ཨ་དིའི་པ་རོ྅ཀྵཱ་ལ་ཡང་ནའོ་ཞེས་པས་གྷསླྀའོ། །ཤྭསྟ་ནཱི་ལ་གྷས་ལ་ཨིཊ་དང་
བཅས་པ་ཉིད་ཀྱི་ཕྱིར་རྟག་ཏུ་ཨིཊ་མེད་པ་ཉེ་བར་བསྟན་པ་མེད་དོ། །རྟག་ཏུ་སྨོས་པ་ཅི་ཞེ་ན། ཨཱནཉྫི་ཐ། ཨནྫུ་གསལ་བ་ལའོ། །ཐལའོ། །ཨིཊའོ། །དང་པོ་ཨའི་ཐམས་ཅད་དུ་ཞེས་པས་རིང་པོའོ། །དེ་ལས་ན་ཨཱ་ག་མའོ། །ཨནྫི་ལས་ཡང་ན་ཨིཊ་ཉིད་ཀྱི་ཕྱིར་ཕྱོགས་གཅིག་ལ་ཨིཊ་མེད་པ་ཉིད་ཡོད་ཅེས་རྣམ་
པར་བརྟག་པར་འགྱུར་རོ་བ་ལ་རྟག་ཏུ་སྨོས་པས་མི་འགྱུར་རོ། །ཉེ་བར་བསྟན་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཙཀཪྵི་ཐ། ཀྲྀཥ་འབྲི་བ་ལའོ། །ཡོན་ཏན་བྱས་པར་གྱུར་པ་ན་འདི་ཨ་ཡིག་དང་ལྡན་པ་ནི་ཉེ་བར་བསྟན་པ་མ་ཡིན་ནོ། །དེ་བཞིན་དུ་དབྱངས་མཐའ་རྣམས་ལས་དང་པྲོ་ཛི་ཏྲྀ་ཤ་དག་ལས་ཀྱང་ངོ་། །ཡ་ཡཱ་ཐ། ཡ་ཡི་ཐ།
