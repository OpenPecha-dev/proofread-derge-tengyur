པ་དྷཱ་ཕྱིས་པའི་ཞེས་པས་གྷའོ། །གཞི་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨུ་པ་གྷཱ་ཏཿ། ས་དང་སྣ་དང་པཱ་དང་བྱ་དྷ་དང་ཧ་ནི་དང་ཡུ་དྷི་རྣམས་ལས་ཀའི་རྐྱེན་བཤད་པར་བྱ་སྟེ། འབྲུ་རྣམས་འདི་ལ་གནས་པར་བྱེད་དོ་ཞེས། པྲསྠ། འདིས་གཙང་བར་བྱེད་དོ་ཞེས། པྲསྣཿ། འདི་ལས་འཐུང་ངོ་ཞེས། པྲ་པཱ་བྱིངས་ཐམས་
ཅད་པ་མ་ཡིན་པ་ལ་ཞེས་པས་ཨཱ་དབྱིའོ། །འདི་ཀུན་དུ་གནོད་པར་བྱེད་དོ་ཞེས། ཨཱ་བ་དྷིཿ། གྲ་ཧི་ཛྱཱ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །བྱ་བ་འདིས་འཇོམས་པར་བྱེད་དོ་ཞེས། བིགྷྣ། སྔ་མ་བཞིན་དུ་ཨུ་པ་དྷཱ་དབྱི་བ་དང་གྷ་ཉིད་ཀྱང་ངོ་། །འདི་ཀུན་དུ་འཁྲུག་པར་བྱེད་དོ་ཞེས། ཨཱ་ཡུ་དྷ། ཌུ་རྗེས་སུ་འབྲེལ་པ་ལས་
ཨ་ཐུའོ། །ཊུའི་སྒྲ་རྗེས་སུ་འབྲེལ་བའི་བྱིངས་ལས་ཨ་ཐུའི་རྐྱེན་འགྱུར་རོ། །བེ་ས་ཐུཿ། ཀྵ་བ་ཐུཿ། གཡོ་བར་བྱེད། སྒྲོགས་པར་བྱེད་ཅེས་ཚིག་རྣམ་པར་སྦྱར་རོ། །ཊུ་པེ་སྲི་ཀ་པི་གཡོ་བ་ལའོ། །ཊུ་ཀྵུཥ་སྒྲ་ལའོ། །ཊའི་སྒྲ་ནི་འདིའི་དོན་ནོ། །ཌུ་རྗེས་སུ་འབྲེལ་པ་ལས་དེས་ཚར་བ་ལ་ཏྲི་མ་ཀའོ། །ཌུའི་སྒྲ་
རྗེས་སུ་འབྲེལ་པའི་བྱིངས་ལས་བྱིངས་དོན་དེས་ཚར་བའི་དོན་ལ་ཏྲི་མ་ཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །སྨིན་པས་ཚར་བ་ལ། པཀྟྲི་མཾ། དེ་བཞིན་དུ་བྱེད་པས་ཚར་བ་ནི། ཀྲྀ་ཏྲི་མཾ། ཌུབ་ཙཥ་འཚེད་པ་ལའོ། །ཌུ་ཀྲྀཉ་བྱེད་པ་ལའོ། །ཌུའི་སྒྲ་ནི་འདིའི་དོན་ཏོ། །ཡཱ་ཙི་བིཙྪ་པྲ་ཙྪི་ཡ་ཛི་སྭ་པི་རཀྵི་ཡཏ་རྣམས་ཀྱི་ནངའོ། །
ཡཱ་ཙི་དང་བཙྪ་དང་པྲཙྪི་དང་ཡཛི་དང་སྭ་པི་དང་རཀྵི་དང་ཡཏྱ་ཞེས་པ་འདི་རྣམས་ལས་ནང་ཕ་རོལ་ཏུ་འགྱུར་རོ། །སློང་བར་བྱེད། ཡཙྙཱཿ། དེ་བཞིན་དུ། བིཤྣཿ། པྲཤྣཿ། ཡཛྙཿ། སྭཔྣཿ། རཀྵྞཿ། ཡཏྣཿ། སྒྲ་ལྡན་དང་ཏི་དག་ལ་ཞེས་པས་ཨིཊ་མེད་དོ། །ཚ་བ་དག་གི་ལྔ་པ་ལ་ཞེས་པས་ཚ་ཡིག་གི་ཤའོ། །
ཉེར་བསྒྱུར་ལ་དཱ་ལས་ཀིའོ། །ཉེར་བསྒྱུར་ཉེ་བའི་ཚིག་ལ་དཱའི་མིང་ཅན་ལས་ཀི་རུ་འགྱུར་རོ། །སྦྱིན་པར་བྱ་བ་དང་པོ་ལས་ཐོབ་པ་ནི། ཨཱ་དིཿ། སྒྲུབ་པར་བྱེད་པ་ནི། བིདྷིཿ། ཨནྟར་གྱི་སྒྲ་ཉེ་བའི་ཚིག་ལས་ཀྱང་ཀི་འདོད་དེ། གཞན་འཛིན་ནོ། །ཨཊྟ་དྷིཿ། ལས་ལ་གཞི་ལའང་ངོ་། །ལས་ཉེ་བའི་ཚིག་
ལའང་དཱའི་མིང་ཅན་ལས་གཞིའི་དོན་ལ་ཀིའི་རྐྱེན་དུ་གྱུར་ཏོ། །འདི་ལ་ཆུ་བརྟེན་ཏོ་ཞེས། ཛ་ལ་དྷིཿ། ཤར་དྷིཿ། བུད་མེད་ལ་ཀྟིའོ། །བྱིངས་ལས་བུད་མེད་ཀྱི་ཡུལ་ལ་ཀྟིའི་རྐྱེན་འགྱུར་རོ། །གྷཉ་དང་ཨལ་དག་འགེགས་པའོ། །ཀྲྀ་ཏིཿ། ཙི་ཏྲིཿ། སྟུ་ཏིཿ། སཱ་ཏི་ཧེ་ཏི་ཡུ་ཏི་ཛཱུ་ཏི་རྣམས་ཀྱང་ངོ་། །སྒྲ་འདི་རྣམས་
