ཡིན་པ། མཱ་རྔངྒི་ཀཿ། དེ་ཕན་གྱི་མཐའི་མྲྀ་དཾ་གིའི་སྒྲ་ནི། རང་བཞིན་ལས་རྔའི་རྡུང་བ་པོ་ལ་འཇུག་གོ། །ངེས་སྦྱོར་ལས་སྒོ་ལ་ངེས་པར་སྦྱོར་བ་ནི། དཽ་ཝཱ་རི་ཀཿ། ཡ་ཝ་ཚིག་དང་པོ་དག་གི་མིན་ཞིང་ཞེས་པས་ཨཽ་ཡིག་ཨཱ་ག་མའོ། །ཨ་གཱ་རའི་སྒྲའི་མཐའ་ལ་ངེས་
སྦྱོར་ལས་ཨི་ཀ་ཎའི་རྐྱེན་བཤད་པར་བྱའོ། །ལྷའི་ཁང་པ་ལ་ངེས་པར་སྦྱོར་བ་དཻ་བཱ་གཱ་རི་ཀཿ། དེ་བཞིན་དུ། ཀཽཥྛ་གཱ་རི་ཀཿ། དེས་ཉོས་པ་ལ་བདུན་ཅུས་ཉོས་པ། སཱཔྟ་ཏི་ཀམ། བརྒྱད་ཅུས་མྱོས་པ། ཨཱ་ཤཱི་ཏི་ཀམ། རྐྱེན་འདི་ནི་བྱེད་པ་ཉིད་ལས་བཤད་པར་བྱའི་བྱེད་པ་པོ་ལ་ནི་མ་ཡིན་ནོ། །
དེས་ན་འདིར་མི་འགྱུར་ཏེ། ལྷས་བྱིན་གྱིས་ཉོས་པ་ནི། དཻ་བ་དིཏྟ་ཀམྲ། ཤཏའི་སྒྲ་ལས་ཉོས་པའི་དོན་ལ་ཡ་དང་ཨི་ཀ་ཞེས་པ་འདི་དག་བཤད་པར་བྱའོ། །བ་ཏྱས་ཉོས་པ་ནི། ཤ་ཏྱམ྄་ཤ་ཏི་ཀམ྄། ས་ཧ་སྲ་དང་བ་ས་ན་དག་ལས་ཨཎ་བཤད་པར་བྱའོ། །སྟོང་གིས་ཉོས་པ་སཱ་ཧ་སྲཾ་ཝཱ་ས་ནཾ། 
སོགས་པ་སྨོས་པས་དོན་གཞན་ལ་ཡང་ཨི་ཀཎ་སྟེ་དེ་ལ་འདིའི་ཚད་ཅེས་པའི་དོན་ལ་སྟེ། སྲ་སྠ། འདིའི་ཚད་ཡིན་པ་ནི། སྲཱ་སྠི་ཀོ་རཱ་ཤིཿ། དེས་རྐོས་སོ་ལ་སྟེ། ཏོག་ཙེས་བརྐོས་ལ། ཀཽ་དྭོ་ལི་ཀཿ། དེས་འཚོའོ་ལ། བཻ་ཏ་ནི་ཀཿ། ཨཱ་ཡུ་དྷི་ཀཿ། སྐྱེད་དོ་ལའོ། །མཆོད་སྦྱིན་པ་
རྣམས་བསྐྱེད་དོ། །ཝཱ་ད་རི་ཀཿ། སྒྲ་ལས་བྱེད་པ་པོའི་དོན་ལའོ། །སྒྲ་བྱེད་པ་པོ་ནི། ཤཱབྡི་ཀཿ། བཻ་ཡཱ་ཀ་ར་ཎཿ། དེ་བཞིན་དུ་གཞན་ཡང་སྟེ། དྷརྨའི་སྒྲ་ལས་འདིར་སྤྱོད་དོ་ཞེས་པའི་དོན་ལའོ། །ཆོས་སྤྱོད་ནི། དྷཱརྨི་ཀཿ། ཨོ་ཛས་དང་། པ་ཧས་དང་། ཨམ་བྷ་ས་ཞེས་པ་འདི་
རྣམས་ལས་དེའི་ས་འདི་ལ་འཇུག་གོ་ཞེས་པའི་དོན་ལའོ། །གཟི་བརྗིད་ཀྱིས་འཇུག་པ་ནི། ཨཽ་ཛ་སི་ཀཿཤཱུ་རཿ། བཱ་ཧ་སི་ཀ་ཙཽ་རཿ། ཨཱམྦྷ་སི་ཀོ་མ་ཏ་སྱཿ། དྷུ་རའི་སྒྲ་ལས་འབབ་བོ་ཞེས་པའི་དོན་ལ་འདིར་ཡ་དང་ཨེ་ཡ་ཎ་ཞེས་པ་འདི་དག་བཤད་པར་བྱའོ། །དྷུ་ར་འབབ་པ་ནི། དྷུཪྻཿ། དྷཽ་རེ་
ཡཿ། སརྦ་དྷུ་རའི་སྒྲ་ལས་དེ་ཉིད་ཀྱི་དོན་ལ་ཨཱཎའི་རྐྱེན་བཤད་པར་བྱ་སྟེ། དྷུ་ར་ཐམས་ཅད་འབབ་པ་ནི། སརྦ་དྷུ་རཱི་ཎཿ། ཨསྟི་དང་ནསྟིའི་སྒྲ་དག་ལས་འདིར་དེའི་བློ་གྲོས་ཞེས་པའི་དོན་ལ་ཨི་ཀ་ཎའོ། །འཇིག་རྟེན་ཕ་རོལ་ཡོད་དོ་ཞེས་པ་འདིའི་བློ་གྲོས་ཡིན་པ་ནི། ཨཱསྟི་ཀཿ། དེ་བཞིན་དུ།
