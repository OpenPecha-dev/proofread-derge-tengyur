ཡོད་པ། དི་དང་སིའོ། །ཨནའོ། །དི་སི་དག་གི་ཞེས་པ་ཅི་ཞེ་ན། དོགྷྣི། དུ་ཧ་འཇོ་བ་ལའོ། །ཏིའོ། །ཡོན་ཏན་ནོ། །ད་དང་པོའི་གྷའོ། །གྷ་ཌྷ་དྷ་བྷ་ཞེས་པས་ཏའི་དྷའོ། །དུཊ་རྣམས་ཀྱི་གསུམ་པའོ། །དི་དང་ལྷན་ཅིག་སྤྱོད་པའི་སི་གཟུང་ངོ་། །དེས་ན་འདིར་མི་འགྱུར་རོ། །དོཀྵི། དུཧ་འཇོ་བ་ལའོ། །བརྟ་མཱ་
ནཱའི་སིའོ། །ཨནའོ། །ཨད་སོགས་ལས་འཇིག་གོ། །ད་དང་པོའི་གྷའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཞེས་པས་གྷའི་ཀའོ། །ཥ་ཉིད་དོ། །བྱིངས་མིན་པའི་ཡའི་ཨན་མེད་པ་ལའོ། །གསལ་བྱེད་ལས་ཞེས་པ་ཁོ་ནའོ། །གསལ་བྱེད་ལས་ཕ་རོལ་གྱི་བྱིངས་མིན་པ་དང་འབྲེལ་པའི་ཡ་ཡིག་དབྱི་བར་འགྱུར་ཏེ། ཨན་མེད་པ་སྟེ་ཨན་
མེད་པ་ཞེས་པ་འདིས་བྱིངས་ཐམས་ཅད་པ་མིན་པ་མཚོན་པར་བྱའོ། །པཱ་པ་ཅི་ཏཱ། པཱ་པ་ཙིཥྱ་ཏེ། པཱ་པ་ཙྱའི་སྒྲ་ལས་ཤྭསྟ་ནཱིའིཏཱའོ། །བྷ་པིཥྱནྟིའི་སྱ་ཏེའོ། །ཨིཊའོ། །ཨའི་ཡང་དབྱིའོ། །ཞེས་པ་འདིས་ཨ་ཡིག་དབྱིའོ། །བེ་བྷི་དི་ཏཱ། བེ་བྷི་དིཥྱ་ཏེ། བེ་བྷི་ཏྱའི་སྒྲ་ལས་ཏཱ་དང་སྱ་ཏེའོ། །གཞན་སྔ་མ་བཞིན་ནོ། །གསལ་
བྱེད་ལས་ཞེས་པ་ཅི་ཞེ་ན། ལོ་ལུ་ཡི་ཏཱ། ཤིན་ཏུ་གཅོད་དོ་ལ་ཡ་སྒྲའོ། །གཉིས་བརྗོད། ཏཱའོ། ཨིཊའོ། །ཨའི་འང་དབྱིའོ། །ཨན་མེད་པ་ཞེས་པ་ཅི་ཞེ་ན། པཱ་པ་ཏྱཱཏ། བྱིངས་མིན་པ་ཞེས་པ་ཅི་ཞེ་ན། མ་དྷྱ་ཏཱ། མ་དྷྱ་འཆིང་བ་ལའོ། །ཕལ་པ་དང་གཙོ་བོ་དག་གི་གཙོ་བོ་ལ་བྱ་བ་ཡང་དག་པར་རྟོགས་པར་བྱའོ་ཞེས་པའི་ཚིག་

ཏྲྀ་ཥད་འདོད་དོ་ལ་ཡིན་ནོ། །ཏྲྀ་ཥད་བཞིན་དུ་སྤྱོད་དོ་ལ་ཨཱ་ཡིའོ། །ཏ་དང་ཨིཊའོ། །ཨིའང་དབྱིའོ། །བྱིངས་མིན་པ་ཞེས་པ་ལྡོག་གོ། །ཨ་ཡིག་གི་དབྱི་བར་འགྱུར་ཏེ། ཨན་མེད་པ་ལ་སྟེ་ཨན་མིན་ནོ་ཞེས་པ་འདིས་འདིར་ཡང་བྱིངས་ཐམས་ཅད་པ་མིན་པ་མཚོན་ནོ། །ཙི་ཀཱིཪྵི་ཏཱ། ཙི་ཀཱིཪྵི་ཥྱཱ་ཏི། ཙིཀཱིརྟིའི་སྒྲ་དང་ཏ་ལས་ཏཱ་
དང་སྱ་ཏིའོ། །ཨིཊའོ། །ཨ་ཡིག་དབྱིའོ། །ག་ཎ་ཡ་ཏི། མྲི་ག་ཡ་ཏི། བྲྀཧ་ཡ་ཏི། ག་ཎ་པ་གྲང་བ་ལའོ། མྲྀག་ཚོལ་བ་ལའོ། །སྤྲྀཧ་འདོད་པ་ལའོ། །ཨཏ་མཐའ་རྣམས་ཀྱི་ཙུར་སོགས་ལས་ཀྱང་ངོ་ཞེས་པས་ཨིནའོ། །ཨའི་འང་དབྱིའོ། །དེའི་དབྱངས་བསྒྱུར་བ་ནི་རྒྱུ་མཚན་གཞན་ཞེས་པས་གནས་ཅན་དང་འདྲ་བའི་དངོས་པོ་
