ཐོབ། དྷིའི་ཨི་ཡར་གྱུར་པ་དང་སྦྱར་བ་སོགས་གོང་བཞིན་ནོ། །ཨ་དྷྱ་པ་ཡ་ཏིའོ། །བི་ནི་འགྲོ་བ་དང་རྙེད་པ་དག་གོ། །ལིང་ནི་འབྱར་བའོ། །ཉི་བྷི་ནི་འཇིགས་པའོ། །འདི་རྣམས་ཀྱི་ཕ་རོལ་ན་ཨིན་ཡོད་ན་ཨཱ་ཉིད་ཐོབ་པར་ཡང་གསུངས་ཏེ། བཱི་ལ་ཨིན་སྦྱར་བས་ཨིས་ཨཱ་ཐོབ། 
ཨིན་དང་སྦྱར། གུ་ཎ་ཨེ་ཨ་ཡ། ཡ་དང་ཨན་གྱི་ཨ་སྦྱར། པཱ་པ་ཡ་ཏིའོ། །ལིང་ལ་ཡིན་སྦྱར་བས་གོང་བཞིན། བཱི་ལཱ་པ་ཡ་ཏིའོ། །བྷྲི་ལ་ཡིན་སྦྱར་བས་གོང་བཞིན། བྷྲ་བ་ཡ་ཏེའོ། །ཡང་ན་ཕྱོགས་གཅིག་ལ་ཨིན་སྦྱར་བས། གུ་ཎ་ཨེ་ཨ་ཡ། ཡ་དང་ཨན་སྦྱར་རོ། །པཀྵ་
བཱ་ཡ་ཡ་ཏིའོ། །ལཱིང་ལ་ཨིན་སྦྱར་བས་གོང་བཞིན། བི་ལཱ་ཡ་ཡ་ཏིའོ། །བྷྲི་ལཱ་ཨིན་སྦྱར་བས། བྷེ་ཨ་ཤོག་གི་མདོས་བར་དུ་ཤ་འབྱུང་སྟེ། གོང་མ་བཞིན། བྷཱི་ཥ་ཡ་ཏེའོ། །སི་དྷ་ནི་ཡང་དག་པར་སྒྲུབ་པའི་དྷཱ་ཏུ་འདིའི་ཨེས་ཨཱ་ཉིད་ཐོབ་པར་ཡང་གསུངས་ཏེ། བཱི་ལ་ཨིན་སྦྱར་
བས། ཨིས་ཨཱ་ཐོབ། བར་དུ་པ་ཐོབ། ཨིན་དང་སྦྱར་གུ་ཎ་ཨེ་ཨ་ཡ། ཡ་དང་ཨན་གྱི་ཨ་སྦྱར། བཱ་པ་ཡ་ཏིའོ། །ལཱིང་ལ་ཡིན་སྦྱར་བས་གོང་བཞིན། བི་ལཱ་བ་ཡ་ཏིའོ། །བྷི་ལཱ་ཡིན་སྦྱར་བས་གོང་བཞིན། བྷཱ་པ་ཡ་ཏེའོ། །ཡང་ན་ཕྱོགས་གཅིག་ལ་ཨིན་སྦྱར་བས། གུ་ཎ་
ཨེ་ཨཡ། ཡ་དང་ཨན་སྦྱར་རོ། །པཀྵ་བཱ་ཡ་ཡ་ཏིའི། ལཱིང་ལ་ཨིན་སྦྱར་བས་གོང་བཞིན། བཱི་ལཱ་ཡ་ཡ་ཏིའོ། །བྷྲི་ལ་ཨིན་སྦྱར་བས། བྷེ་ཨ་ཤོག་གི་མདོས་བར་དུ་ཥ་འབྱུང་སྟེ། གོང་མ་བཞིན། བྷྲི་ཥ་ཡ་ཏེའོ། །སི་དྷཱ་ནི་ཡང་དག་པར་བསྒྲུབས་པའི་དྷཱ་ཏུ་འདིའི་ཨེས་ཨ་
ཉིད་ཐོབ་པར་གསུངས་ཏེ། འཇིག་རྟེན་ཕ་རོལ་མ་ཡིན་པའི་དོན་ལའོ། །པཱ་ཡ་སཾ་སཱ་དྷ་ཡརྟི། འོ་མའི་ཐུག་པ་བྱེད་པ། སི་དྷུ་ལ་ཨིན་སྦྱར་བས་སའི། ཨིས་ཨཱ་ཐོབ། དྷ་དང་ཨི་སྦྱར། གུ་ཎ་ཨེ་ཨཡ། ཡ་དང་ཨན་གྱི་ཨ་སྦྱར། སཱ་དྷཱ་ཡ་ཏི་བྷ་ཀྟུ་མའོ། །འཇིག་རྟེན་
ཕ་རོལ་མ་ཡིན་པ་ཞེས་ཅིའི་ཕྱིར་སྨོས་ཤེ་ན། པཱ་ཡ་སཾ། འོ་ཐུག་སི་དྷ་ལ་ཨིན་སྦྱར། ཡང་ཨ་མ་ཐོབ་སྟེ་གུ་ཎའོ། །སེ་དྷ་ཡ་ཏིའོ། །དྭ་སུཾ་སྦྱིན་པ་ནི་ཕ་རོལ་གྱི་དོན་ནོ། །སྲྀ་ཛི་དྲྀ་ཤོ་རཱ་ག་མོ་ཀཱ་རསྶྭ་རཱརྦ་རོ་དྷུ་ཊི་གུ་ཎ་བྲྀ་དྷི་སྠཱ་ནེ། སྲྀ་ཛ་ནི་རྣམ་པར་གཏོང་བའོ། །དྲྀ་ཤིར་ནི་ལྟ་བའོ། །
