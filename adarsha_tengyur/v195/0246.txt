ཛ་ཏི། སཾ་སྠཱ་ཏུཾ་བྲ་ཛ་ཏི། ཞེས་པའི་དོན་ཏོ། །ཨ་ཤྲ་ཡཱ་ཡ་བྲ་ཛ་ཏི། ཨཱ་ཤྲ་ཡི་ཏུཾ་བྲ་ཛ་ཏི། ཞེས་པའི་དོན་ཏོ། །ཏུ་མའི་དོན་ལས་ཞེས་པ་ཅི་ཞེ་ན། པ་ཙ་ནཾ་པཱ་ཀཿ། ཏྱ་ཛ་ནཾ་ཏྱ་གཿ། དངོས་པོ་བརྗོད་པ་ཅི་ཞེ་ན། ཀཱ་ར་ཀོ་བྲ་ཛ་ཏི། བྱ་བ་ལ་བྱ་བའི་དོན་ལ་ཝུ་ཎ་ཏུམ྄་ཎམ྄་དག་གོ་
ཞེས་པས་ཝུཎའོ། །ལྷན་ཅིག་སྦྱོར་བ་ལ་གསུམ་པའོ། །ལྷན་ཅིག་གི་སྦྱོར་བ་ནི་ལྷན་ཅིག་སྦྱོར་བའོ། །ལྷན་ཅིག་པའི་དོན་གྱི་སྦྱོར་བ་ལ་རྟགས་ལས་གསུམ་པར་འགྱུར་རོ། །པུ་ཏྲེ་ཎ་ས་ཧ་ཨཱ་ག་ཏཿ། པི་ཏཱ་པུ་ཏྲེ་ཎ་ས་ཧ་སྠུ་ལཿ། པུ་ཏྲེ་ཎ་ས་ཧ་གོ་མཱ་ན། འདིར་གཙོ་བོ་མ་ཡིན་པ་ལས་གསུམ་པ་
བཤད་པར་བྱའོ། །དེས་ན་པི་ཏྲྀ་སྒྲ་ལས་མི་འགྱུར་རོ། །ལྷན་ཅིག་ཅེས་པ་ནི་འདིར་དོན་འཛིན་པ་སྟེ། དེས་ན། པུ་ཏྲེ་ཎ་པ་ཀཾ་པ་ཋ་ཏི། པུ་ཏྲེ་ཎ་སཱ་མཾ་ཏྲཱ་ཡ་ཏི། པུ་ཏྲེ་ཎ་སཱ་དྷྭཾ་པཱུ་ཛ་ཡ་ཏི། ཏུ་ལྱའི་དོན་གྱི་སྒྲའི་རྟགས་ལས་དྲུག་པ་དང་གསུམ་པ་དག་བཤད་པར་བྱ་སྟེ། ཉེ་བར་འཇལ་བའི་མཉམ་པའོ་ཞེས་པའི་ཏུ་ལྱའི་
དོན་དོར་བ་ཉིད་ལའོ། །ཏུ་ལྱོ་དེ་བ་དཏྟ་སྱ། ཏུ་ལྱོ་དེ་བ་དཏྟེ་ན། ས་དྲྀ་ཤོ་དེ་བ་དཏྟ་སྱ། ས་དྲྀ་ཤོ་དེ་བ་དཏྟེ་ན། ཉེ་བར་འཇལ་བའི་ཏུ་མྱ་འོས་སྦྱོར་བ་ལ་དྲུག་པའང་ངོ་། །ཏུ་ལཱ་ཏུ་ཏྲ་སྱ་ནཱ་ཏྲི་ཨུ་པ་མཱ་ཀྲྀཥྞ་སྱ་ན་བི་དྱ་ཏེ། རྒྱུའི་དོན་ལའོ། །རྒྱུའང་འདི་ཡིན་ལ་དོན་ཡང་ཡིན་པ་ནི་རྒྱུའི་དོན་ཏོ། །འདིར་
གསུམ་པ་རྗེས་སུ་འཇུག་སྟེ། མདོ་ཕྱི་མ་གསུམ་རྣམས་ལའང་ངོ་། །འབྲས་བུ་སྒྲུབ་པ་དང་ལྡན་པའི་དངོས་པོ་ནི་རྒྱུའོ། །རྒྱུ་ལས་གསུམ་པར་འགྱུར་རོ། །བི་དྱ་ཡཱ་ཡ་ཤཿ། ཀ་ནྱ་ཡཱ་ཤོ་ཀཿ། རྒྱུ་ལའོ་ཞེས་མ་བྱས་པ་ནི་རྣམ་པ་སྣ་ཚོགས་པའི་དོན་ཏོ། །སྨད་པ་ལ་ཡན་ལག་ལའོ། །ཡན་
ལག་ལ་སྨད་ཅིང་སྨོད་པར་བྱེད་པ་ལ་གསུམ་པར་འགྱུར་རོ། །ཨཀྵྞཱ་ཀཱ་ཎཿ། ཤི་ར་སཱ་ཁལླི་ཊ། ནཱ་ས་ཡཱ་ཙི་པ་ཊཿ། སྨད་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། མུ་ཁེ་གཽ་རཿ། འདིར་གཙོ་བོ་མ་ཡིན་པ་ལས་གསུམ་པ་བཤད་པར་བྱའོ། །དེས་ན་འདིར་མི་འགྱུར་ཏེ། ཨཀྵི་ཀཱ་ཎ་དེ་བ་དཏྟ་སྱ། འདིར་ནི་ཨཀྵི་
གཙོ་བོའོ། །ཁྱད་པར་དུ་བྱེད་པ་ལའོ། །གང་གིས་ཁྱད་པར་དུ་བྱེད་པ་དེ་ཁྱད་པར་དུ་བྱེད་པའོ། །ཁྱད་པར་དུ་བྱེད་པ་ལ་གསུམ་པར་འགྱུར་རོ། །ཛ་ཊཱ་བྷི་སྟཱ་པ་ས་དྲྀཥྚཿ། ཁཌྒེ་ན་བྷ་ཊོ་དྲྀཥྚཿ། བྱེད་པ་པོ་ལའང་ངོ་། །བྱེད་པ་པོ་ལ་གསུམ་པར་འགྱུར་རོ། །རུ་དྲེ་ཎ་དགྡྷཾ་ཏྲྀ་པུ་རཾ། བིཥྞུ་ནཱ་བདྡྷོ་
