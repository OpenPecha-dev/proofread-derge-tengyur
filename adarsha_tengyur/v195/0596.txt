ཀྟིའི་རྐྱེན་ཚིག་ཕྲད་དེ། སཱ་ཏིཿ། ཥོ་མཐའི་ཞེས་པ་འདིའི་དྱ་ཏི་སྱ་ཏི་ཞེས་པས་ཨི་ཉིད་ཐོབ་པ་ལ། ཏིའི་རྐྱེན་ཚིག་ཕྲད་དོ། །ཧེ་ཏིཿ། ཧི་འགྲོ་བ་ལའོ་ཞེས་པ་འདི་ཡོན་ཏན་ཡང་ནའོ། །ཡུ་ཏིཿ། ཛུཏིཿ། ཡུ་བསྲེ་བ་ལའོ་ཞེས་པ་འདི་མདོའི་རིང་པོ་ཚིག་ཕྲད་དོ། །ཨུ་ཏི་དང་ཀཱིརྟི་དག་གི་
བཀླགས་པ་དོན་མེད་དེ། ཨུཏའི་སྒྲ་ཤྲི་བི་ཨ་བི་མ་བི་ཞེས་པ་འདིར་གྲུབ་བོ། །ཀཱིརྟིའི་སྒྲ་ཡང་ཀཱིརྟི་ཨིཥ་དག་གི་ཀྟིའང་ངོ་ཞེས་པ་འདིརའོ། །བ་ཙི་གཱ་པསྠ་རྣམས་ལས་དངོས་པོ་ལའོ། །བུད་མེད་ལ་ཞེས་པ་འཇུག་སྟེ། ཀྲྀ་ཏྱ་ཡུཊ་རྣམས་ཞེས་པ་འདི་ལས་གོང་དུའོ། །པ་ཙི་དང་གཱ་དང་པཱ་དང་སྠཱ་རྣམས་ལས་
དངོས་པོའི་དོན་ལ་ཀྟིའི་རྐྱེན་འགྱུར་རོ། །ཨཱ་ཏ་ལས་ཀྱང་ངོ་ཉེར་བསྒྱུར་ལ་ཞེས་པས་ཨང་ཐོབ་པ་ལའོ། །པ་ཙི་ལས་ཀྱང་ཥ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་སོ། །པཀྟིཿ། ཨུངྒཱི་ཏིཿ། པྲ་པཱ་ཏིཿ། སཾ་སྠི་ཏཿ། དཱ་མཱ་གཱ་ཡ་ཏི་ཞེས་པས་ཨཱི་ཉིད་དོ། །ཏིཥྛ་ཏི་ལས་དྱ་ཏི་སྱ་ཏི་ཞེས་པས་ཨི་ཉིད་དོ། །ཀྲྀཏྱ་དང་ཡུཊ་རྣམས་
གཞན་ལ་ཡང་ངོ་ཞེས་པའི་ཚིག་ལས་ཨང་ཡང་ཇི་ལྟར་མཐོང་བ་བཞིན་འགྱུར་ཏེ། ཨ་བསྠཿ། ཨཱསྠ། བྲཛ་ཡ་ཛ་དག་གི་ཀྱཔའོ། །པྲ་ཛི་དང་ཡ་ཛི་དག་ལས་ཀྱཔ་འགྱུར་ཏེ། ཀྟིའི་རྐྱེན་འགེགས་པའོ། །བྲ་ཛྱ། ཨི་ཛྱ། སྭ་པི་བ་ཛི་ཡ་ཛ་སོགས་རྣམས་ཀྱི་ཞེས་པས་སམྤྲ་སཱ་ར་ཎའོ། །པ་རྗེས་སུ་འབྲེལ་པ་བྱེད་
པར་ནི་ཕྱི་མ་ལ་ཏ་ཡིག་ཨཱ་ག་མའི་དོན་ཏོ། །ས་མ་ཛ། ཨཱས། ནི་ཥད། ནི་པ་ཏི། ཤིང་། སུ། བིད། ཨ་ཊི། ཙ་རི།མ་ནི། བྷྲྀཉ། ཨིཎ་རྣམས་ཀྱི་མིང་ལའོ། །ཨ་ཛི་ལ་སམ་སྔོན་དུའོ། །ཨཱས་ཥད་པ་ཏ་དག་གི་སྔོན་དུ་ནིའོ། །ཤིང་ཉལ་བ་ལའོ། །ཥུཉ་བཀྲུ་བ་ལའོ། །བིད་ཤེས་པ་ལའོ་ཞེས་
པ་འདི་རྣམས་དང་དེ་བཞིན་དུ། ཨཊི་དང་ཙ་རི་དང་། མ་ནི་དང་བྷྲྀཉ་དང་ཨིཎ་ཞེས་པ་འདི་རྣམས་ལས་ཀྱང་མིང་གི་ཡུལ་ལ་ཀྱཔ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ས་མ་ཛྱཱ། ཨཱ་སྱཱ། ནིཥ་དྱཱ། ནིཔདྱཱ། ཤཡྱ། ཡ་ལ་ཨཱི་ནི་ཨཡའོ་ཞེས་པས་ཨཡའོ། །སུརྱཱ། བིདྱ། ཨ་ཊཱ་ཊྱཱ། ཨཊི་ལས་ཙེ་ཀྲྀ་ཡི་ཏའི་
མཐའ་ལས་ཡའི་རྐྱེན་བཤད་པར་བྱའོ། །དེས་ན། ཨ་ཊཱ་ཊྱའི་སྒྲ་ལས་ཀྱཔའོ། །ཨའི་ཡང་དབྱིའོ། །ཡའི་ཨན་མིན་པ་ལའོ་ཞེས་པས་ཡ་དབྱིའོ། །པ་རི་ཙཪྻཱ། པ་རི་སྔོན་མ་ཉིད་ལས་ཙ་རིའོ། །མནྱཱ། བྷྲྀ་ཏྱཱ། ཨི་ཏྱཱ། འོ་ན། བྷྲྀཏྲིཿ། མ་ན་ནཾ་མ་ཏིཿ། ཞེས་པ་ཅི་ལྟར་ཞེ་ན། བྷྲྀ་ཏི་ལ་ལས་
