ས་སྭ་ཏུཿ། ད་དྷྭ་ར་ཏུཿ། ད་དྷྭ་ཏིཿ། སྭྲྀ་སྒྲ་ལའོ། །དྷྭྲྀ་གནོད་པ་ལའོ། །རིཏ་ཅེས་པ་ཅི་ཞེ་ན། ཙིཀྵི་ཡ་ཏུཿ། ཙིཀྵི་ཡུཿ། ཀྵི་ཟད་པ་ལའོ། །བྱིངས་ཀྱི་ཨིའི་ཡི་གེ་དང་ཨུའི་ཡི་གེ་དག་མཐའི་དབྱངས་དང་པོ་ལ་ཨི་ཡ་ཨུ་བ་དག་ཅེས་པས་ཨི་ཡའོ། །བརྩེགས་པ་དང་པོ་ཞེས་པ་ཅི་ཞེ་ན། ཙ་ཀྲ་ཏུཿ། ཡོན་ཏན་

ཡི་གེ་ཨ་ཡིག་གོ། །དང་པོའི་ཨའི་ཐམས་ཅད་དུའོ་ཞེས་པས་རིང་པོའོ། །རྀ་ཡིག་ལའང་ངོ་ཞེས་པས་ན་ཨཱ་ག་མའོ། །འདིས་ཡོན་ཏན་ནོ། །པ་རོ྅ཀྵ་ལ་ཞེས་པ་ཅི་ཞེ་ན། རྀཙྪིཏཱ། ཤཱིངའི་བྱིངས་ཐམས་ཅད་པ་ལའོ། །ཤིང་ཉལ་བ་ལའོ་ཞེས་པ་འདིའི་བྱིངས་ཐམས་ཅད་པ་ཕ་རོལ་ལ་ཡོན་ཏན་དུ་འགྱུར་
