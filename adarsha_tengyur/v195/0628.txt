དྷྭ་ནི་པ་ཎི་རྣམས་ཀྱི་ལྔ་པའི་ཨུ་པ་དྷཱའི་ཞེས་པས་རིང་པོའོ། །ཨ་ནཱ་ཡཱས་ཡི་སྟེ་འབད་པ་མེད་པས་གང་འགྲུབ་པ་གང་ཡིན་པ་དེ་ནི་ཡཱཎྜའོ། །དེ་ཡང་ཐམས་ཅད་ནི་མ་ཡིན་ཏེ། འོན་ཀྱང་གང་བཙོས་པ་མིན་ཞིང་ཕྱེ་མ་མིན་པ་ཆུ་ལ་རེག་ཙམ་ལས་རྣམ་པར་ཕྱེ་བའི་ཁུ་བ་དེའོ། །སྲུབ་པ་ལ་སོགས་པ་རྣམས་ལ་
ཞེས་པ་ཅི་ཞེ་ན། ཀྵུ་བྷི་ཏཿ། བཱ་ཧི་ཏཿ། སྭ་ནི་ཏཿ། དྷྭ་ནི་ཏཿ། ཡ་ཎི་ཏཿ། ཀ་ཥི་ཏཿ། གྷུ་ཥི་ཏཿ། འོ་ན་ཀྵུ་བྷྣཿ། རྒྱ་མཚོ་ཅི་ལྟར་ཡིན་ཞེ་ན། བརྗོད་པར་བྱ་སྟེ། ཀྵུ་བྷའི་སྒྲ་དང་པོར་སྲུབ་པའི་ཡུལ་ལ་འདུས་བྱས་ནས་ཕྱི་ནས་སྲུབ་པ་ལྟ་བུ་ནི། ཀྵུ་བྡྷ། ཞེས་ཉེ་བར་འཇལ་བ་ལས་རྒྱ་མཚོ་ལ་འཇུག་པར་འགྱུར་རོ། །བཱ་ཧི་ཨིན་མཐའ་
ཅན་པར་མི་འགྱུར་ཏེ། ཨིན་མཐའ་ཅན་མ་ཡིན་པའི་ཀྵུ་བྷི་སོགས་དང་ལྷན་ཅིག་སྤྱོད་པའི་ཕྱིར་རོ། །ལགྣ་མླིཥྚ་བི་རི་བྡྷཱ་རྣམས་ནུས་པ་མི་གསལ་བ་དབྱངས་རྣམས་ལའོ། །སྒྲ་འདི་རྣམས་གྲངས་བཞིན་དུ་ནུས་པ་དང་མི་གསལ་བ་དང་དབྱངས་ཀྱི་དོན་རྣམས་ལ་ཚིག་ཕྲད་དོ། །ལགྣཿ། ལ་གེ་ཆགས་པ་ལའོ། །ཀྟའི་རྐྱེན་ལ་
ཨིཊ་མེད་པ་དང་། ན་ཉིད་ཀྱི་ཚིག་ཕྲད་དོ། །མླིཥྚཾ། མླེཙྪ་མི་གསལ་བར་བརྗོད་པ་ལའོ། །དངོས་པོ་ལ་མ་ནིང་ལ་ཀྟའོ། །ཨིཊ་མེད་པ་དང་། ཨེ་ཡིག་གི་ཨི་ཉིད་ཀྱང་ཚིག་ཕྲད་དོ། །བི་རིབྡྷཿ། རེ་བྷྲྀ་སྒྲ་ལའོ། །བི་སྔོན་དུའོ། །སྔར་བཞིན་ཚིག་ཕྲད་དོ། །ནུས་པ་མི་གསལ་བ་དབྱངས་རྣམས་ལ་ཞེས་པ་ཅི་ཞེ་ན། ལ་གི་ཏཾ་མླེཙྪཱ་
ཏི། བི་རེ་བྷི་ཏཾ། པ་རི་བྲྀཌྷ་དྲྀཌྷ་དག་ཇོ་བོ་དང་བ་ལ་བཏ་དག་ལའོ། །སྒྲ་འདི་དག་གྲངས་ཇི་ལྟ་བ་བཞིན་དུ། ཇོ་བོ་དང་བ་ལ་བཏ་ཀྱི་དོན་ལ་ཚིག་ཕྲད་དོ། །པ་རི་བྲྀཌྷཿ། བྲྀཏ་བྲྀཧི་འཕེལ་བ་ལའོ། །པ་རྀ་སྔོན་དུའོ། །ཀྟའི་རྐྱེན་ཕ་རོལ་ལ་ཨིཊ་མེད་ཅིང་ན་དབྱི་བའང་ཚིག་ཕྲད་དོ། །ཨིཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ད་དབྱི་བར་མི་འགྱུར་
བའོ། །དྲྀཌྷཿ། དྲྀ་ཧི་འཕེལ་བ་ལའོ། །སྔར་བཞིན་དུ་ཚིག་ཕྲད་དོ། །སྦོམ་པོའི་དོན་ལ་ཡང་དྲྀཌྷའི་སྒྲ་ཚིག་ཕྲད་དེ། དྲྀཌྷཿསྠུ་ལ། སྦོམ་པོའོ། །ཇོ་བོ་དང་བ་ལ་བཏ་དག་ལ་ཞེས་པ་ཅི་ཞེ་ན། བ་རི་བྲྀཧི་ཏ། བྲྀཧི་ཏཾ། སཾ་ནི་བི་རྣམས་ལ་ཨརྡི་ལས་སོ། །ནིཥྛཱ་ལ་ཞེས་པ་ཁོ་ནའོ། །སཾ་དང་ནི་དང་བི་ཞེས་པ་འདི་རྣམས་ལས་ཕ་རོལ་གྱི་
ཨརྡ་འགྲོ་བ་སྟེ་སློང་བ་ལའང་ཞེས་པ་འདིའི་ནིཥྛཱ་ལ་ཨིཊ་མི་འགྱུར་རོ། །ས་མཪྞྞཿ། ནྱནྞྞཿ། བྱརྞྞཿ། ཨརྡ་འགྲོ་བ་ལའོ། །སཾ་དང་ནི་དང་བི་རྣམས་སྔོན་དུའོ། །ཀྟའི་རྐྱེན་ནོ། །དེ་ལས་དའི་ཞེས་པས་ན་ཉིད་དོ། །སྔ་མའི་ན་ཡིག་གི་ཎ་ཉིད་དུ་བྱས་པ་ལ་ཏ་སྡེའི་ཙ་ཊའི་སྡེབ་སྦྱོར་བ་ལ་ཞེས་པས་ཎའོ། །ས་ཾནི་བི་རྣམས་ལས་ཞེས་པ་ཅི་ཞེ་ན།
