དྷི་རཱ་ཤི་ཥི། ཧ་ན་འཚེ་བ་དང་འགྲོ་བ། འདིའི་ཕ་རོལ་ན་ཨཱ་ཤཱི་ཥ་ཡོད་ན་ཧ་ན་ལ་བ་དྷི་གཞག་པའོ། །ཧ་ན་ལ་ཡཏ་སྦྱར་བས་ཝ་དྷ་ཐོབ་སྟེ་ཡ་དང་སྦྱར། ཝ་དྷྱ་ཏའོ། །ཧ་ན་ལ་ཡཱ་ཏམ་སྦྱར་བས་སྔ་མ་བཞིན། ཝ་དྷྱཱ་ཏམའོ། །ཨ་དྱ་ཏ་ནྱཱཉྩ། ཧ་ན་ཏེའི་ཕ་རོལ་
ན་ཨ་དྱ་ནཱི་ཡོད་ན་ཝ་དྷི་གཞག་པ་སྟེ། ཧ་ན་ལ་དི་སྦྱར་བས། ཝ་དྷ་ཐོབ། གོང་དུ་ཨ་སྦྱར། སི་ཙ་དང་། ཨི་ཊ་སྦྱར་ཏེ། ཨི་ཏ་ཤྭ་ཨི་ཏི་སི་ཙ་ལོ་པ་བྱས་ནས། ཨཱི་དང་། ཨཱི་ས་མ་ན་དཱིརྒྷའོ། །ཨ་ཝ་དྷི་ཏའོ། །ཧ་ན་ལ་ཏམ་སྦྱར་བས། ཝ་དྷཱ་ཐོབ། ཨིཊ་དང་སི་ཙ་ཐོབ་སྟེ་ཨི་དེ་ཙ་
ཕྱི། སས་ཥ། ཏས་ཊ་ཐོབ་སྟེ་སྦྱར། ཨ་ཝ་དྷིཥྚ་མའོ། །བདག་གི་ཚིག་ཅན་ཕ་རོལ་ན་ཡོད་ན་ཝ་དྷི་གཞག་པར་ཡང་གསུངས་ཏེ། ཧ་ན་ལ་ཉ་སྦྱར་བས་གོང་བཞིན་དུ་ཨ་སྦྱར། ཧནྟེ་ཨཱ་སུརྦ་ཏུ་ཙའི་མདོས་ཨ་སྦྱར་ཏེ་ས་མཱ་ན་དཱིརྒྷའོ། །ཧར་ལ་ཨཱ་ཏམ་སྦྱར་བ་སོགས་སྔ་མ་བཞིན། ཨཱ་
ཝ་དྷི་ཥཱ་ཏའོ། ཡང་ན་ཧ་ན་ལ་ཏ་སྦྱར་བས་གོང་དུ་ཨ་སྦྱར། ཨཱིཥྞི་བ་ཤིན་གྱི་མདོས་ན་ཕྱི། དྷུ་ཏི་པ་རེ་ཧྲ་སྭ་ཤྩའི་མདོས་ཨི་དང་སི་ཙ་ཕྱི། ཨ་ཧ་ཏའོ། །ཧནྟ་ལ་ཏཱམ་སྦྱར་བས་སི་ཙ་སྦྱར་བ་སྟེ། ཨི་དང་ཙ་ཕྱི། གཞན་ནི་སྔ་མ་བཞིན་ནོ། །ཨཱ་ཧ་ནཱ་ཏམའོ། །ཨི་ཎོ་གཱ། ཨི་ཎ་ག་ཏ་ཞེས་པ་
འདིས་ག་ཐོབ་པ་སྟེ། ཕ་རོལ་ན་ཨ་དྱ་ཏ་ནཱི་ཡོད་ནའོ། །ཨི་ཎི་ལ་དི་སྦྱར་བས་ག་ཐོབ། སི་ཙ་སྦྱར་ཏེ་ཨིན་ཏྠ་དེས་ཕྱི། གོང་དུ་ཨ་སྦྱར་རོ། །ཨ་གཱ་ཏའོ། །ཨིན་ཏམ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་གཱ་ཏཱ་མའོ། །ཨི་ཀ་ག་ཏ་འདིའི་ཕ་རོལ་ན་ཨ་དྱ་ཏ་ནཱི་ཡོད་ན་གཞག་པར་གསུངས་ཏེ། ཨིཀ་ལ་
དི་སྦྱར་བས་གཱ་ཐོབ། གོང་དུ་ཨ་སྦྱར། ཨི་བརྞ་ཡཱ་མཱ་ས་བརྞེའོ། །སི་ཙ་སྦྱར་ཏེ་ཕྱི། ཨ་དྷྱ་གཱ་ཏའོ། །ཨིཀ་ལ་ཏཱམ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་དྱ་གཱ་ཏཱ་མའོ། །ཨིངཿ པ་རོ྅ཀྵཱ་ཡཱཾ། ཨིང་བརྗོད་པ་ཞེས་པ་འདི་ལ་གཱ་གཞག་པ་སྟེ། ཕ་རོལ་ན་པ་རོཀྵཱ་ཡོད་ནའོ། །ཨིང་ལ་པ་རོཀྵཱ་ཨེ་
སྦྱར་བས། གཱ་ཐོབ། ཨ་དྷི་གོང་དུ་སྦྱར། གཱ་གཉིས་པ་བྱ་སྟེ། གས་ཛའོ། །ཨ་ཀ་སྱ་ཨ་ལོ་པའི་མདོས་གའི་ཨ་ཕྱི་སྟེ་ཨེ་དང་སྦྱར། ཨ་དྷི་ཛ་གེའོ། །ཨིང་ལ་ཨ་ཏེ་སྦྱར་བ་སོགས་སྔ་མ་བཞིན། ཨ་དྷི་ཛ་གཱ་ཏེའོ། །ཨིང་གི་ཕ་རོལ་ན་ཨཱ་དྱ་ཏཱ་ནི་དང་། ཀྲཱི་ཡ་ཏི་བ་ཏི་ཡོད་ན་བསྲིང་བ་
