ཀྲྀ་བཱ་ས་ཛི་མི་སྭ་དི་སྭ་དྷྱ་ཤུ་རྣམས་ལས་ཨུཎའོ། །བྱུང་བ་ལ། བྷ་སི་ཏ་བྷ་སྨཱ། ཙ་ར་ཏི་ཙཾརྨཱ། བྷས་གཙོ་བ་ལའོ། །ཙར་འགྲོ་བའི་དོན་ཏོ། །མནའོ། །སྒྲ་མེད་པ་དག་ལའོ་ཞེས་པས་ཨིཊ་མེད་དོ། །འབྱུང་འགྱུར་ལ་ག་མི་ལ་སོགས་པ་རྣམས་སོ། །ག་མི་ལ་སོགས་པའི་སྒྲ་རྣམས་འབྱུང་འགྱུར་གྱི་དུས་
བསྒྲུབ་པར་འགྱུར་རོ། །སྔ་མས་བྱུང་བ་དང་ད་ལྟར་བ་ཐོབ་པ་ལའོ། །འགྲོ་བར་འགྱུར་ཞེས་ག་མི་ལས་ཨི་ནིའོ་ཞེས་པས་ཨི་ནིའོ། །འོང་བར་འགྱུར་རོ་ཞེས། ཨཱ་གཱ་མཱི། ཨཱང་ལ་ནི་ཏའོ་ཞེས་པས་ཨི་ནིའོ། །ཨའི་ཨུ་པ་དྷཱ་རིང་པོའོ། །འབྱུང་བར་འགྱུར་རོ་ཞེས་བྷཱ་བཱི། བྷཱུ་ལས་ལས་ཀྱང་ངོ་ཞེས་པས་ཨི་
ནིའོ། །དེ་ལ་ནི་ཏ་ཞེས་པ་འཇུག་སྟེ་དེས་ན་འཕེལ་བའོ། །ག་མཱི། ཨཱ་གཱ་མཱི། བྷ་བཱི། པྲ་ཡཱ་མཱི། བྲ་ཏི་རེ་དྷཱི། པྲ་བོ་དྷཱི་ཞེས་པ་རྣམས་ག་མཱི་སོགས་སོ། །བྱ་བ་ལ་བྱ་བའི་དོན་ལ་བུཎ་ཏུམ་དག་གོ། །བྱ་བ་གང་གིས་བྱ་བའི་དོན་ལ་བྱ་བའི་དོན་སྦྱོར་བ་དེ་ནི་བྱ་བའི་དོན་ཏོ། །འབྱུང་འགྱུར་ལ་ཞེས་པ་
འཇུག་སྟེ། གྷཉ་བསྒྲགས་པ་གོང་དུའོ། །བྱ་བ་ལ་བྱ་བའི་དོན་ཉེ་བའི་ཚིག་ལ་བྱིངས་ལས་འབྱུང་འགྱུར་གྱི་དུས་ལ། བུཎ་དང་ཏུམ་དག་འགྱུར་རོ། །སཱ་ཙ་ཀོ་བྲ་ཛི་ཏ། སཱ་ཀྟུཾ་པྲ་ཛ་ཏི། ཀ་ར་ཀོ་བྲ་ཛ་ཏི། ཀཪྟུ་ཡྦྲ་ཛ་ཏི། བྱ་བ་ལ་ཞེས་པ་ཅི་ཞེ་ན། བྷིཀྵི་ཥྱེ་ཨིཏྱ་སྱ་ཛ་ཊཱ། འདིར་ཛ་ཊཱའི་སྒྲ་ཉེ་བའི་ཚིག་ཏུ་
གྱུར་པའི་བྷིཀྵའི་བྱ་བ་བརྗོད་པའི་རྐྱེན་མེད་དོ། །བྷིཀྵའི་དོན་ཉིད་ཀྱི་ཕྱིར་རལ་པ་སོགས་ཀྱི་བྱ་བའི་དོན་ཉིད་ཡོད་པ་ཁོ་ནའོ། །བྱ་བའི་དོན་ལ་ཞེས་པ་ཅི་ཞེ་ན། པ་ཏི་ཥྱི་ཏ་དྷཱ་བ་ཏ་སྟེ་དཎྜཿ། འདིར་དྷཱ་བ་བྱ་བ་མིན་པ་ཉེ་བའི་ཚིག་ཏུ་འགྱུར་བ་དེ་ནི་དཎྜ་དང་པཱ་ཏའི་དོན་མ་ཡིན་ཏེ་པ་ཏི་ལས་རྐྱེན་མེད་དོ། །འོ་ན་གལ་ཏེ་བུཎའི་
རྐྱེན་གྲུབ་ཟིན་པས་བུཎ་སྨོས་པ་དེ་ནི་དོན་མེད་དེ་ཏུམའི་ས་འགེགས་པར་འགྱུར་རོ་ཞེ་ན། མི་འགྱུར་ཏེ་དོན་ཐ་དད་པའི་ཕྱིར་རོ། །གལ་ཏེ་མི་ཟད་པ་བྱ་བའི་དངོས་པོ་ལ་འགྱུར་རོ་ཞེས་པས་དངོས་པོ་ལ་ཏུམའི་རྐྱེན་དང་བྱེད་པ་པོ་ལ་ནི་བུཎ་ཡིན་ན། འོ་ན་ཤནྟྲྀང་ཨཱན་དག་གིས་གནོད་པར་འགྱུར་རོ་ཞེ་ན། གཟུགས་
