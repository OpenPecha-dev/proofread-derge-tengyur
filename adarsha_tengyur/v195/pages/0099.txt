གཉིས་པ་བྱས་རྀ་ཕྱི། མདོ་འདིས་རཱྀ་ཐོབ། ན་རཱྀ་ནྲྀ་ཏྱ་ཏེའོ། །གྲ་ཧ་ལ་ཡ་སྦྱར་བས། གོང་དུ་གྲ་གཉིས་པ་བྱས། རྀ་ཕྱི་གས་ཛ་ཐོབ། བར་དུ་རྀ་འབྱུང་བའོ། །ཛ་རཱྀ་གྲྀ་ཧྱ་ཏེའོ། །ནྲྀ་ཏི་ལུས་བསྐྱོད་པའོ། །གྲ་ཧ་ནི་ཉེ་བར་ལེན་པའོ། །རྀ་རེ་ཕཾ་ཙ་ལུ་ཀི་རྀ་མ་ཏོའི་དྷཱ་ཏུ་ལ་ཙེ་ཀྲཱི་ཡི་ཏ་ཕྱི་བ་
ཡོད་ན། ཨ་བྷྱ་སའི་མདུན་དུ་རཱི་དང་རྀ་དང་། ར་འདི་དག་ཐོབ་པ་འབྱུང་བ་སྟེ། ནྲྀ་ལ་ཙེ་ཀྲཱི་ཡི་ཏའི་ཡ་སྦྱར་ཏེ་ཕྱི་བས། གོང་དུ་ནྲྀ་གཉིས་པ་བྱས། ར་ཕྱི་བར་དུ་རྀ་ཐོབ་པ་ཨར་དུ་གྱུར་ཏེ་སྦྱར་ན་རཱི་ན་རྟིའོ། །ནྲྀ་ལ་ཡ་སྦྱར་བས་རྀ་ཐོབ་པ་གོང་བཞིན། ན་རྀ་ནརྟིའོ། །ནྲྀ་ལ་ཡ་སྦྱར་ཏེ་
ཕྱི་བས་ར་ཐོབ་པ་གོང་བཞིན། ན་ར་ནརྟིའོ། །ཨ་ལོ་པེ་ས་མཱ་ན་སྱ་སཾ། དྷལླ་གུ་ནཱི་ཙ་ཎྱ་པ་རེ། མཉམ་པའི་ཡི་གེའི་མིང་ཅན་ཕྱི་བར་མ་གྱུར་པ་ཡོད་ལ་ཕ་རོལ་ན་ཡང་པ་ར་ཡི་གེ་ཡོད་ན་ཨ་བྷྱ་ས་ལ་སན་ལྟ་བའི་བྱ་བ་ནི་གང་སན་གྱི་བྱ་བར་བརྗོད་པ་དེ་འབྱུང་བར་འགྱུར་བ་སྟེ། ཨིན་དང་
ཙཎ་དག་ཕ་རོལ་ན་ཡོད་པར་གྱུར་ན་འོ། །ཀྲི་ལ་དི་སྦྱར། བར་དུ་ཙཎ་སྦྱར་བས་གོང་དུ་ཀྲི་གཉིས་པ་བྱས། ཀས་ཙ་ཐོབ། མདོ་འདིས་ཙས་ཨི་ཐོབ། དཱིརྒྷ་ལ་སོགས་པས་ཨི་བསྲིངས་པའོ། །གོང་དུ་ཨ་ཡར་སྦྱར། ཨ་ཙཱི་ཀ་ར་ཏའོ། །པ་ཋ་ལ་དི་དང་ཙན་སྦྱར་བ་སོགས་གོང་བཞིན། ཨ་
པཱི་པ་ཋ་ཊའོ། །ཌུ་ཀྲཱི་ཉ་ནི་བྱེད་པའོ། །པ་ཋ་ནི་གསལ་བར་སྨྲ་བའོ། །སན་གྱི་བྱ་བ་ལྟ་བུ་ནི་སན་ནྱ་བརྞ་སྱ་ཡ་ཞེས་པའི་མདོས། ཨི་ཉིད་དོ། །དཱིརྒྷ་ལ་གྷོཿ། མདོ་སྔ་མ་རྗེས་སུ་འཇུག་སྟེ། མཉམ་པའི་ཡི་གེ་ཕྱི་བར་མ་གྱུར་ཅིང་། ཕ་རོལ་ན་ཙ་ཎ་དང་ཨིན་ཡོད་པ་དང་། གང་ཡང་དེ་ན་
ཡོད་ན། ཨ་བྷྱ་ས་ཐུང་བ་རིང་བར་འགྱུར་བ་སྟེ། པ་ཙ་ལ་དི་སྦྱར། ཨིན་དང་ཙ་ཎ་སྦྱར་བས། ཨི་ཐོབ་པ་མདོ་འདིས་བསྲིངས་པའོ། །ཨ་པཱི་པ་ཙ་ཏའོ། །ཌུ་པ་ཙ་ཥ་འཚེད་པའོ། །ཨ་ཏྭ་རཱ་དཱི་ནཱཾ་ཙ། ཏྭ་ར་སོགས་པའི་དྷཱ་ཏུ་རྣམས་ཀྱིས་ཨ་བྷྱ་སའི་མཐས་གང་ཡིན་པ་དེའི་ཕ་རོལ་ན་
ཨིན་དང་ཙན་ཡོད་ན། ཨ་ཐོབ་པར་འགྱུར་བ་སྟེ། ཏྭ་ར་ལ་ཨིན་དང་ཙན་སྦྱར་བས་རའི་ཨི་ཕྱི། ཙའི་ཨ་དང་སྦྱར། གོང་དུ་ཏྭ་གཉིས་པ་བྱ་སྟེ། མདོ་འདིས་ཏྭས་ཏའོ། །གོང་དུ་ཨ་ཡང་སྦྱར། ཨ་ཏ་ཏྭ་ར་ཏའོ། །ཉི་ཏྭ་ར་ནི་མྱུར་བའོ། །ག་ཎེ་རཱི་ཙ་ག་ཎ་ནི་གྲངས་ཏེ། འདིའི་དྷཱ་ཏུའི་
