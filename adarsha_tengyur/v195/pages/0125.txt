བྷེ་ཏིའོ། །ཧྲཱི་ལ་ཏི་དང་ཨན་སྦྱར། ཧྲཱི་གཉིས་པ་སོགས་སྔ་མ་བཞིན། ཛི་ཧྲེ་ཏིའོ། །ཧུ་ནི་སྦྱིན་པ་དང་བཟའ་བའོ། །ཉི་བྷི་ནི་འཇིགས་པའོ། །ཧྲི་ནི་ངོ་ཚ་བའོ། །ཝཱ་ཙེ་ཀྲཱི་ཡི་ཏ་སྱ། ཙེ་ཀྲཱི་ཡི་ཏའི་ཡ་ཕྱི་བའམ། བྷཱུ་ལ་ཡ་སྦྱར་བ་ཕྱི་ནས། པྲ་ཏྱ་ཡ་ལུས་ཏམ། པྲ་ཏྱ་ཡ་ལ་ཀྵ་ཎའི་མདོས་
བྷཱུ་གཉིས་པ་བྱས། བྷས་བ་ཐོབ། ཨིཊ་སྦྱར། ཨུ་ཨོར་གྱུར། ཨོ་ཨཝ་དང་སྦྱར་རོ། །བོ་བྷ་ཝཱི་ཏིའོ། །ར་ཊ་ལ་ཡ་སྦྱར་ནས་ཕྱི་བ་སོགས་སྔ་མ་བཞིན། རཱ་ར་ཊཱི་ཏིའོ། །ཡང་ན་བྷཱུ་ལ་ཡ་སྦྱར་བས། ས་མ་ཕྱི་སྟེ། བྷུ་གཉིས་པ་ལ་སོགས་པའོ། །བ་བོ་དྷུ་ཡ་ཏེའོ། །ར་ཊ་ལ་ཡ་སྦྱར་
བས་སྔ་མ་བཞིན། དཱིརྒྷོ་ནཱ་ག་མནྟ་སྱའི་མདོས་ཨ་བྷྱ་ས་བསྲིངས་པའོ། །རཱ་ར་ཊྱ་ཏེའོ། །ཨིཎྶྠ་ད་པ་ཝ་ཏི་བྷཱུ་བྷྱཿསི་ཙཿཔ་རསྨཻ། ཨིན་ནི་འགྲོ་བ་དང་། སྠཱ་ནི་འགྲོ་བ་ལ་མི་འཇུག་པ་དང་། དཱའི་མིང་ཅན་དང་། པཱ་ནི་བཏུང་བ་དང་། བྷུ་ནི་ཡོད་པ་ཞེས་བྱ་བ་འདི་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་སི་ཙི་
ཕྱི་བ་སྟེ། ཕ་རོལ་ན་གཞན་གྱི་ཚིག་ཅན་ཡོད་ནའོ། །ཨིན་ལ་དི་སྦྱར། བར་དུ་སི་ཙ་སྦྱར། བ་ཕྱི། ཨིཎ་གྱི་གྱི་གཱ་ཐོབ། གོང་དུ་ཨ་སྦྱར། ཨ་གཱ་ཏའོ། །སྠཱ་ལ་དི་སྦྱར་བས་བར་གྱི་སི་ཙ་ཕྱི། སྠཱ་ཐོབ། གོང་དུ་ཨ་སྦྱར། ཨསྠཱ་ཏའོ། །དཱ་ལ་དི་སྦྱར་བས་སི་ཙ་ཕྱི། ཨ་དཱ་ཏའོ། །དྷཱ་ལ་དི་
སྦྱར། སི་ཙ་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨ་དྷཱ་ཏའོ། །པཱ་ལ་དཱི་སྦྱར། སི་ཙ་ཕྱི། ཨ་གོང་དུ་སྦྱར་རོ། །ཨ་པཱ་ཏའོ། །བྷུ་ལ་དི་སྦྱར། སི་ཙ་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨ་བྷུ་ཏའོ། །གྷཱ་ནི་དྲི་ཉེ་བར་ལེན་པའོ། །ཤོ་ནི་སྲབ་མོར་བྱེད་པའོ། །
ཙྪོ་ནི་གཅོད་པའོ། །ཪྷེ་ཊ་ནི་བརྡུང་བ་སྟེ། འདི་རྣམས་ཀྱི་ཕ་རོལ་ན་སི་ཙ་ཕྱི་བར་བྱ་བ་ཡང་གསུངས་ཏེ། གྷྲ་ལ་དི་སྦྱར། བར་དུ་སི་ཙ་སྦྱར་ཏེ་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨ་གྷྲ་ཏའོ། །ཤོ་ལ་དི་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་ཤཱད་ཙྪོ་ལ་དི་སྦྱར། སི་ཙ་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨཙྪ་ཏའོ། དྷེ་ཊ་ལ་དི་
སྦྱར། སི་ཙ་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨེས་ཨཱ་ཐོབ་པའོ། །ཨ་དྷཱ་ཏའོ། །ཡང་ལ་གྷྲ་ལ་དི་སྦྱར། སི་ཙ་དང་ཨཱི་སྦྱར། ཨི་དང་ཨིས་མཱ་ནི་དཱིརྒྷས་ཨཱ་ག་མ་ཐོབ་པ་དང་སྦྱར། གོང་དུ་ཨ་སྦྱར། ཨ་གྷྲཱ་སཱི་ཏའོ། །ཤོ་ལ་གོང་མ་བཞིན། མ་ཤྲཱ་སཱི་ཏའོ། །ཙྪ་ལ་གོང་མ་བཞིན་སྦྱར།
