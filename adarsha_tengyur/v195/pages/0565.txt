ཁ་ཊྭ་མཱ་ནི་ནཱི། ཁྲི་ནས་ཁེངས་པ་ཅན་མ་ཞེས་པའོ། །བྲཧྨ་བནྡྷུ་མཱ་ནི་ནི། ཎི་ནིའོ། །རྀ་དང་ན་མཐའ་རྣམས་ལས་ནི་ཨཱིའོ། །བདག་ཉིད་ལ་ཁ་ཤའང་ངོ་། །སེམས་པར་བྱེད་པ་དང་ཁེངས་པར་བྱེད་པ་ནི་བདག་ཉིད་ལ་འཇུག་བཞིན་པའི་མན་ལས་ཎི་ནིའི་རྐྱེན་འགྱུར་ཞིང་ཁཤའི་རྐྱེན་འང་ངོ་། །བལྟ་ན་སྡུག་
པའི་བདག་ཉིད་དུ་སེམས་པ་མོ། །དཪྴ་ནཱི་ཡ་མཱ་ནཱི། པཎྜི་ཏ་མཱ་ནཱི། དཪྴ་ནཱི་ཡཾ་མ་ནྱཿ། པཎྜི་ཏཾ་མནྱཿ། ཁཤའོ། །དིབ་སོགས་ལས་ཡནའོ། །ཐུང་བ་དང་ཨ་ཧུ་ཥ་དག་གི་ཞེས་པས་ལ་ཡིག་ཨཱ་ག་མའོ། །བདག་ཉིད་ལ་ཞེས་པ་ཅི་ཞེ་ན། པཎྜི་ཏཾ། དེ་བ་དཏྟོ་ཡ་ཛྙ་དཏྟཾ་མནྱ་ཏེ། བྱེད་པ་ལ་ཡ་ཛ་
ལས་འདས་པ་ལའོ། །བྱེད་པ་ཉེ་བའི་ཚིག་ལ་འདས་པའི་དུས་ལ་ཡཛ་ལས་ཎི་ནིའི་རྐྱེན་འགྱུར་རོ། །ཨགྣིཥྚོ་མ་ཡཱ་ཛཱི། པུ་ཏྲོ་སྱ་ཛ་ནི་ཏཱ། མེའི་མཆོད་སྦྱིན་བྱས་ཟིན་ཏོ་ཞེས་ལ་ཎི་ནིའོ། །འདས་པ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨགྣིཥྚོ་མེ་ན་ཡ་ཛ་ཏེ། ལས་ལ་ཧན་ལས་སྨད་པ་ལའོ། །འདས་པ་ལ་ཞེས་པ་
འཇུག་སྟེ། བརྟ་མཱ་ན་ལ་དང་པོའི་ཞེས་པ་འདི་ལས་གོང་དུའོ། །ལས་ཉེ་བའི་ཚིག་ལ་ཧན་ལས་འདས་པའི་དུས་ལས་ཎི་ནིའི་རྐྱེན་འགྱུར་རོ། །པི་ཏྲྀ་བྱ་གྷཱ་ཏཱི། མཱ་ཏུ་ལ་གྷཱ་ཏཱི། ཁུ་བོ་བཅོམ་པར་བྱས་ཟིན་ཏོ། །ཞང་པོ་བཅོམ་ཟིན་ཏོ་ཞེས་ལ་ཎི་ནིའོ། །སྨད་པ་ལ་ནི་བཤད་པར་བྱའོ། །དེས་ན་འདིར་མི་
འགྱུར་ཏེ། ཙཽ་རཾ་ཧ་ཏ་བཱནྲ། ཧན་ལས་དེའི་ཞེས་དང་ནིཥྛཱའོ་ཞེས་པས་ཀྟ་དང་ཀྟ་བནྟུའོ། །བ་ན་ཏི་ཏ་ནོ་ཏི་སོགས་ཞེས་པས་ལྔ་པ་དབྱིའོ། །བྲཧྨ་བྷྲཱུ་ཎ་བྲྀ་ཏྲ་རྣམས་ལ་ཀྭིཔྲའོ། །བྲཧྨ་དང་བྷྲཱུཎ་དང་བྲྀ་ཏྲ་ཞེས་པ་འདི་རྣམས་ཀྱི་ལས་ཉེ་བའི་ཚིག་ལ་ཧནྟི་ལས་འདས་པའི་དུས་རྟོགས་བཞིན་པ་ལ་ཀྭིཔ་འགྱུར་
རོ། །ཚངས་པ་བཅོམ་ཟིན་ཏོ། །བྲཧྨ་ཧཱ། དེ་བཞིན་དུ། བྷྲུ་ཎ་ཧཱ། བྲྀ་ཏྲ་ཧཱ། ཀྭིཔའོ། །མདོ་འདིས་འཆད་པར་འགྱུར་བའི་མདོ་གསུམ་ལའང་ཀྭིཔ་ཞེས་པ་འདིས་གྲུབ་པ་ཉིད་ཀྱི་ཕྱིར་སྤྲོས་པའི་དོན་ཏོ་ཞེས་གསུང་ངོ་། །སུ་པུ་ཎྱ་པཱ་པ་ཀརྨ་མནྟྲ་པ་ད་རྣམས་ལ་ཀྲྀཉ་ལས་སོ། །འདི་
རྣམས་ལས་ཉེ་བའི་ཚིག་རྣམས་ལ་འདས་པའི་དུས་ལ་འཇུག་བཞིན་པའི་ཀྲྀཉ་ལས་ཀྭིཔ་འགྱུར་རོ། །ལས་ལ་ཞེས་པའོ། །པུའི་སྒྲའི་ཁྱད་པར་མ་ཡིན་ཏེ་མི་སྲིད་པའི་ཕྱིར་རོ། །མཛེས་པར་བྱས་ཟིན་ཏོ། །སུ་ཀྲྀཏ། དེ་བཞིན་དུ། པུ་ཎྱ་ཀྲྀཏ། པཱ་བ་ཀྲྀཏ། ཀརྨ་ཀྲྀཏ། མནྟྲ་ཀྲྀཏ། པད་ཀྲྀཏ།
