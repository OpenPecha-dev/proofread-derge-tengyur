སྐྱེས་པ་ནི། ཛ་ལ་ཛཾ། མཚོ་ལས་སྐྱེས་པ་ནི། སཱ་ར་སི་ཛཾ། སྐྱོ་བ་སྐྱེས་པ་ནི། ཁ་ད་ཛཾ། བྲམ་ཟེ་རྣམས་སྐྱེས་པ་ནི། བྲཧྨ་ཎ་ཛཾ། འདིས་ཌའོ། །གཞན་ལའང་ངོ་། །གཞན་ལ་ཡང་ཉེ་བའི་ཚིག་ལ་འདས་པའི་དུས་ལ་ཛ་ནི་ལས་ཌའི་རྐྱེན་འགྱུར་རོ། །མ་སྐྱེས་པར་གྱུར་པ་ནི། ཨ་ཛཿ། སྐྱེས་པའི་རྗེས་
སུ་སྐྱེས་པ་ནི། པུ་མ་ནུ་ཛཿ། བུད་མེད་ཀྱི་རྗེས་སུ་སྐྱེས་པ་ནི། སྡྱཱ་ནུ་ཛཿ། རྗེས་སུ་སྐྱེས་པ་ནི། ཨ་ནུ་ཛཿ། གཉིས་སྐྱེས་པ་ནི། དྭི་ཛཿ། འདིས་ཌའོ། །ནིཥྛཱའོ། །བྱིངས་ལས་འདས་དུས་ལ་ནིཥྛཱའི་རྐྱེན་འགྱུར་རོ། །ཀྲྀ་ཏཿ། ཀྲྀ་ཏ་བཱནྲ། ཧྲྀ་ཏཿ། ཧྲྀ་ཏ་བཱནཿ། སུ་ཡ་ཛ་དག་ལས་དྭ་ནི་པའོ། །ཥུཉ་དང་
ཡཛ་དག་ལས་འདས་པའི་དུས་ལ་དྭ་ནིཔ་རྐྱེན་དུ་འགྱུར་རོ། །ཀུན་བཤད་པའི་སྐབས་ཡིན་པས་དེ་ཉིད་ཀྱི་ཨིན་མོད་སྨོས་པ་ལས་སོ། །གཞན་ལ་ཡང་ཨའི་ཨུ་པ་དྷཱ་ཞེས་པ་འདིར་འཕེལ་བར་མི་འགྱུར་ཏེ། ཁྲུས་བྱས་ཟིན། སུ་ཏྭཱ། མཆོད་པ་བྱས་ཟིན། ཡ་ཛཱ། འདིས་དྭ་ནི་པའོ། །ང་ཡིག་ནི་ང་ལ་
ཡོན་ཏན་མེད་དོ་ཞེས་པའི་དོན་ཏོ། །བྱིངས་ཀྱི་མཐའ་ལ་ཏའོ། །ཛཱིཪྻ་ཏི་ལས་ཨནྟྲྀ་ནའོ། །ཛྲཱིཥ་ལང་ཚོ་ཉམས་པ་ལའོ་ཞེས་པའི་ལས་འདས་པའི་དུས་ལ་ཨནྟྲྀན་རྐྱེན་དུ་འགྱུར་རོ། །ན་ཡིག་ནི་རྣམ་པ་སྣ་ཚོགས་པའི་དོན་ཏོ། །ཛ་རན། ཛྲིཥ་ལང་ཚོ་ཉམས་པ་ལའོ། །ལང་ཚོ་ཉམས་ཟིན་ཞེས་ལ་འདིས་ཨནྟྲྀ་
ནའོ། །རྀ་ཡིག་ནི་ཨུཏ་དང་རྀ་ཡིག་ཅེས་པའི་དོན་ཏོ། །ཡོན་ཏན་ནོ། །ཛྲྰིཥྲ་ལས་ཞེས་མ་བྱས་པ་ནི་རྣམ་པ་སྣ་ཚོགས་པའི་དོན་ཏོ། །ཇོ་བོ་དྲག་འབྱོར་གྱིས་མཛད་པའི་འགྲེལ་པ་སློབ་མ་ལ་ཕན་པར་ཀྲྀཏའི་སྐབས་རྐང་པ་གསུམ་པ་རྫོགས་སོ།།      །།ཀཱནྶུ་ཀཱན་དག་པ་རོཉཀྵ་བཞིན་འང་ངོ་། །
བྱིངས་ལས་འདས་པའི་དུས་ལ་ཀྭོནྤུ་དང་ཀཱན་དག་རྐྱེན་དུ་འགྱུར་ཞིང་དེ་དག་ཀྱང་པ་རོཉཀྵ་བཞིན་དུ་འགྱུར་རོ། །པ་རོ྅ཀྵ་ཕ་རོལ་ལ་གང་བཤད་པ་དེ་འདི་དག་ལ་འགྱུར་རོ། །བེ་ཙི་བཱན། བེ་ཙ་ནཿ། ཌུ་པ་ཙ་ཥ་འཚེད་པ་ལའོ། །འདིས་ཀྭནྶུ་དང་ཀཱན་དག་གོ། །པ་རོ྅ཀྵ་བཞིན་དུ་འགྱུར་བས། ཙ་ཎ་པ་
རོ྅ཀྵ་ཞེས་པས་གཉིས་བརྗོད་དོ། །གསལ་བྱེད་གཅིག་པུ་པའི་དབུས་ཞེས་པས་ཨེ་ཉིད་དང་ཨ་བྷྱས་དབྱི་བ་དག་གོ། །ཨརྟི་ཨིཎ་གྷས་དབྱངས་གཅིག་པ་ཞེས་པས་ཨིཊའོ། །ཛཀྵི་བཱན། བྱ་ཏི་ཛཀྵ་ཎཿ། ཨཏ་ལ་ཀྭནྶུ་དང་ཀཱན་དག་གོ། །པ་རོ྅ཀྵ་ལ་ཡང་ནའོ་ཞེས་པས་གྷསླྀའོ། །གཉིས་བརྗོད་
