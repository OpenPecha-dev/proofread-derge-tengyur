ཡ་དང་ཨུ་བ་དག་འབྱུང་བ་སྟེ། མི་འདྲ་བའི་དབྱངས་ཕ་རོལ་ན་ཡོད་ནའོ། །རྀ་ལ་དེ་སྦྱར། བར་དུ་ཨན་སྦྱར་ཏེ་ཕྱི་བས་རི་གཉིས་པ་བྱས། ཨ་ཀཱ་རཱ་རྀ་ཝརྞ་མདོས་རྀས་ཨ་ཐོབ། ཧ་ཏི་པ་ཏིདྷོ་ཙའི་མདོས་ཨས་ཨི་ཐོབ། མདོ་འདིས་ཨི་མ་ཐོབ། དྷཱ་ཏུའི་རྀ་ཨར་དུ་གྱུར་ཏེ་
སྦྱར། ཨ་ཡརྟིའོ། །ནོརྦི་ཀ་ར་ཎ་སྱ། བི་ཀཱ་ར་ཎའི་ནུ་གང་ཡིན་པའི་ཡི་གེ་ཨུ་དེ་ལ་ཨུ་ཝ་གཞག་པར་འབྱུང་བ་སྟེ། དབྱངས་ཐོག་མ་ན་གནས་པའི་ཨ་གུ་ཎེ་ཕ་རོལ་ན་ཡོད་ནའོ། །ཨབླྀ་ལ་ཨན་ཏེ་སྦྱར་བས། པྲ་གོང་དུ་སྦྱར་ཏེ། ས་མ་ན་དཱིརྒ་ནུ་སྭ་རའི་མདོས། བར་དུ་ནུ་ཐོབ་སྟེ། ཨུས་
ཨུ་ཝ་ཐོབ་སྟེ་སྦྱར། བྲཱཔྣུ་ཝནྟིའོ། །ཨ་པ་ལི་ནི་ཁྱབ་པའོ། །ཤཀླི་ལ་ཨན་ཏི་སྦྱར་བས་གོང་བཞིན། ཤཀྣུ་ཝན་ཏིའོ། །ཤཀ་ལི་ནི་ནུས་པའོ། །ཡུ་ལི་ཝརྞ་སྱཱ་སཾ་ཡོ་ག་པཱུརྦ་སྱ་ནེ་ཀཱཀྵ་ར་སྱ། ཙི་ལ་ཨ་ཏུས་སྦྱར། ཙི་གཉིས་པ་བྱས་པས་དབྱངས་དུ་མར་གྱུར་ཏེ། དྷཱ་ཏུ་ཙིའི་ཨིས་ཡ་
ཐོབ་སྟེ། ཡ་དང་ཙ་དང་སྦྱར་རོ། །ཙི་ཙྱ་ཏྲཿ ཙིཉ་ནི་སྡུད་པའོ། །ནཱི་ལ་ཨ་ཏུས་སྦྱར་བས་གོང་བཞིན། ནི་ནྱ་ཏུཿའོ། །ཎཱིཉ་ནི་ཐོབ་པའོ། །ཨི་ཎ་ཤྩ། ཨི་ཎ་ནི་འགྲོ་བ་སྟེ། འདི་ལ་ཡ་ཉིད་འབྱུང་བ་ནི་དབྱངས་ཐོག་མ་ན་ཡོད་པའི་ཨ་གུ་ཎ་ཕ་རོལ་ན་ཡོད་ནའོ། །ཨིཎ་ལ་ཨནྟི་སྦྱར། ཎ་
ཕྱི། བར་དུ་ཨན་སྦྱར་ཕྱི། ཨིའི་ཡ་ཐོབ། ཡནྟིའོ། །ཨ་ཎི་ལ་ཨན་ཏུ་སྦྱར་བས་སྔ་མ་བཞིན། ཡནྟུའོ། །ནོརྦ་ཀ་རོ་བི་ཀ་ར་ཎ་སྱ། གང་བི་ཀ་ར་ཎའི་ནུའི་ཐོག་མ་ན་མི་ལྡན་པ་ཡོད་ན་ཨུ་དེ་ལ་ཝ་ཉིད་འབྱུང་བ་སྟེ། སུ་ནུ་ལ་ཨནྟི་སྦྱར་བས། མདོ་འདིས་ནུའི་ཨུས་ཝ་ཐོབ་སྟེ་
སྦྱར་བས། སུ་ནྭནྟིའོ། །ཙི་ནུ་ལ་ཨན་ཏི་སྦྱར་བས་གོང་བཞིན། ཙི་ནྭནྟིའོ། །ཙིཉ་ནི་སྡུད་པའོ། །ཛུ་ཧོ་ཏེ། སཪྺ་དྷཱ་ཏུ་ཀེ །ཧུ་ནི་སྦྱིན་པ་དང་བཟའ་བ་དག་སྟེ། འདིའི་ཕ་རོལ་ན་སཪྺ་དྷ་ཏུ་ཀ་དབྱངས་ཐོག་མ་ན་གནས་པ་ཡོད་ན་ཝ་ཉིད་འབྱུང་བའོ། །ཧུ་ལ་ཨན་ཏི་སྦྱར། བར་དུ་ཨན་
སྦྱར་ཏེ་ཛུ་ཧོ་ཏིས་ཕྱི། ཧ་གཉིས་པ་བྱས། ཨུས་ཝ་ཐོབ་ལ་སྱོ། །ཨ་བྷྱ་ཏའི་མདོས་ན་ཕྱིའོ། །ཛུ་ཧྭ་ཏིའོ། །ཧུ་ལ་ཨན་ཏུ་དང་། ཨན་སྦྱར་བས་སྔ་མ་བཞིན། ན་ནི་མ་ཕྱི་བའོ། །ཛུ་ཧྭནྟུའོ། །བྷོ་བོ་བནྟུ་པ་རོཀྵཱ་དྷྱ་ཏ་ནྱོཿ། བྷུ་ནི་ཡོད་པ་ཞེས་བྱ་བ་སྟེ། འདིའི་དབྱངས་ཕ་རོལ་ན་དབྱངས་
