མང་པོས་བསྟན་པ་སྐྱོང་གྱུར་ཅིག ༢༨ །སྒྲ་ཚད་རིག་པའི་མདའ་གཞུའི་འཁྲུལ་འཁོར་གྱིས། །སྨྲ་ངན་བདེན་པ་བྲལ་བའི་ཕྱོགས་ཀུན་ལས། །རྣམ་རྒྱལ་གྲགས་པའི་སཱི་ཏ་འཕྲོག་བྱེད་པའི། །རྩོད་མཁས་མང་པོའི་ལོག་རྟོགས་ཆོམ་གྱུར་ཅིག ༢ །སྡེ་སྣོད་གསུམ་གྱིས་
བརྗོད་པར་བྱ་བའི་དོན། །བཅད་ལྷུག་སྤེལ་མའི་རྗོད་བྱེད་མེ་ན་ཀས། །སྒྲ་དོན་རྒྱན་གྱི་གླིང་བུ་འབུམ་བསྒྱུར་བའི། །རྩོམ་མཁས་མང་པོའི་བསྟན་པ་གསལ་གྱུར་ཅིག ༣༠ །བསྟན་པའི་རྩ་ལག་དགེ་འདུན་འདུས་པའི་སྡེས། །བཤད་དང་སྒྲུབ་པས་ས་སྟེང་ཡོམ་ཡོམ་དུ། །བགྱིད་པ་དེ་ལ་སྐྲག་
ནས་མེ་ཏོག་གཞུ། །སྟོང་པའི་བརྫུ་ལ་ཞུགས་སྙམ་བྱེད་གྱུར་ཅིག ༣༡ །ཐུབ་ཆེན་རང་གི་ལུང་རྟོགས་དམ་པའི་ཆོས། །དར་ཞིང་རྒྱས་པའི་གནས་སུ་རབ་བསྔགས་པ། །བོད་ཆེན་སྡེ་བཞི་དགེ་བཅུའི་འབྱོར་པ་ཅན། །ས་ཕྱོགས་འདིས་མཚོན་ཡངས་པའི་རྒྱལ་ཁམས་སུ། ༢༢ །བསྟན་ལ་དགའ་
བའི་ལྷ་རྣམས་ནམ་མཁའ་ལ། །དུས་སུ་ལྷགས་ནས་སྤྲིན་གྱི་འཐོར་ཡབ་བཟུང་། །སྭ་སྟི་དབྱངས་སུ་ལེན་ཅིང་ཆུ་གཙང་གི། །ཨནྡྷ་རྙིལ་གྱི་སིལ་མ་འབེབས་གྱུར་ཅིག ༢༣ །དེ་ཚེ་གསེར་ལྡན་དགའ་མ་རང་གི་ལུས། །སྔོ་བསངས་མཛེས་པའི་མེ་ཏོག་རྒྱན་དང་བཅས། །རྩ་ལྡུམ་ལོ་འདབ་མྱུ་གུ་
འབྲས་བུའི་ཚོགས། །སྨིན་པའི་དགའ་སྟོན་ཅི་ཡང་ངོམ་གྱུར་ཅིག ༣༤ །ཡངས་པའི་འཇིག་རྟེན་ཁམས་ཀྱི་ལུས་ཅན་རྣམས། །ཀུན་འབྱུང་ལས་ཀྱི་སྤྲུལ་པ་སྒྱུ་མའི་ལུས། །ཉམས་དང་དུས་མིན་འཇིག་བྱེད་ནད་ཀྱི་རིགས། །མུན་ནང་སྒྲོན་མེ་བཏེག་བཞིན་ཡལ་གྱུར་ཅིག ༣༥།ཟག་བཅས་
སྡུག་བསྔལ་ཉིད་དུ་བསྟན་པའི་འཕྲིན། །ལོ་ཟླ་ཞག་དུས་རུང་བ་མིན་པའི་སྐེག །སྲི་དང་གཞན་གྱི་བྱད་ཁ་སྦོད་གཏོད་དག །འབྲས་བུ་འབྱིན་པའི་ནུས་པ་བྲལ་གྱུར་ཅིག ༣༦ །ལས་དང་སྨོན་ལམ་ལོག་པའི་ཆུ་ཀླུང་གིས། །དབང་མེད་བདས་ཏེ་ལེགས་པའི་གེགས་བྱེད་ཅིང་། །ཉེས་པར་སྦྱོར་
བའི་གདོན་བགེགས་འབྱུང་པོ་རྣམས། །བྱམས་པ་ཆེན་པོས་འཁྱུད་བཞིན་གནས་གྱུར་ཅིག ༢༢ །ཁྲོས་པས་གཙུབས་པའི་མིག་བསྒྱུར་སྨིན་མ་འཁྱོག །ཁྲག་གིས་བསྒོས་པའི་མཚོན་ཆར་འབེབས་པ་ཡིས། །ཕན་ཚུན་གཅོད་དང་འབིགས་པའི་མི་སྲུན་ཚོགས། །སེམས་རྒྱུད་སྙིང་རྗེ་བས་
