བས་ང་སངས་རྒྱས། །ཞེས་པའི་བར་དུ་ནི་ཡོན་ཏན་གྱི་ཁྱད་པར་བསྟན་པ་སྟེ། དེ་ལ་ཚིག་ཕྱེད་དང་གཉིས་ཀྱིས་ང་འདོད་ཆགས་དང་བྲལ་བས་འཇིགས་པ་མེད་པའི་ཡོན་ཏན་བསྟན་ཏོ། །ཚིག་འོག་མ་གཅིག་གིས་ནི་ཕྱི་རོལ་གྱི་རྒྱན་དང་ཡང་དག་པའི་ཆོས་ཀྱི་ཡོན་ཏན་
བསྟན་ཏེ། སེར་སྣ་མེད་ཅེས་བྱ་བ་ནི་ཐེག་པ་ཆེན་པོ་བཅབས་ཏེ། ཐེག་པ་ཆུང་ངུ་སྟོན་ཅིང་སེམས་ཅན་བསླུ་བའམ། སྔར་ཐེག་པ་ཆེན་པོ་སྦྱིན་པར་བྱ་བ་ལས་ཕྱིས་མ་སྦྱིན་པ་མ་ཡིན་པས་སེར་སྣ་མེད་དོ། །སེམས་ཅན་གྱིས་ཐེག་པ་ཆེན་པོ་རིག་ཅིང་ཆུད་ཀྱིས་དོགས་པའི་ཐུགས་མི་མངའ་
བས་ཕྲག་དོག་མེད་དོ། །སྡིག་པའི་ཆོས་ནི་གང་ཉོན་མོངས་པའི་ལས་དང་བཅས་པ་སྡུག་བསྔལ་བ་འཁོར་བའི་ཆོས་ཡོངས་སུ་བཅད་པས་ཉེས་པའི་སྐྱོན་གང་ཡང་མེད་དེ། ཡང་དག་པའི་ཡེ་ཤེས་རྫོགས་ནས་ཕྱོགས་བཅུར་སེང་གེའི་སྒྲ་བསྒྲགས་ཏེ་འཇིགས་པ་མི་མངའ་བས། འགྲོ་བ་ཀུན་
རིག་ང་སངས་རྒྱས། །ཞེས་བྱའོ། །མདོ་ལས། ཇི་ལྟར་མཚན་རྣམས་ཀྱིས་ནི་ང་བརྒྱན་ཏེ། །ཞེས་པ་ནས། ཆོས་ཀྱི་རང་བཞིན་ཕྱག་རྒྱ་འདི་བཤད་དོ། །ཞེས་པའི་བར་དུ་ནི་ཕྱི་རོལ་གྱི་མཚན་གྱིས་བརྒྱན་པ་བསྟན་པ་སྟེ། མཚན་བཟང་པོས་བརྒྱན་པའི་སྐུའི་འོད་དང་། ཡེ་ཤེས་ཀྱི་འོད་
ཀྱིས་སྣོད་ཀྱི་འཇིག་རྟེན་དང་བཅུད་ཀྱི་འཇིག་རྟེན་སྣང་བར་བྱས་ནས་སྲོག་ཆགས་ཀུན་གྱིས་ང་ལ་བཀུར་སྟི་བྱས་སོ། །ནང་ན་ཉོན་མོངས་པ་མེད་པས་ཕྱི་རོལ་གྱི་མཚན་གྱིས་བརྒྱན་ཏེ་ཆོས་ཀྱི་རང་བཞིན་གྱི་ཕྱག་རྒྱ་འདི་བཤད་དོ། །ཕྱག་རྒྱ་ནི་སྟོང་པ་གཉིས་ཀྱི་ཐེག་པ་ཆེན་པོའི་ཕྱག་རྒྱ་སྟེ། ཕྱག་རྒྱ་
འདིས་བཏབ་པ་ནི་ཐེག་པ་ཆེན་པོའི་ཆོས་ཡང་དག་པ་ཞེས་བྱའོ། །མདོ་ལས། དེ་ལྟར་ངས་བསམས་ཤཱ་རིའི་བུ། །ཞེས་པ་ནས། བདག་འོད་རང་བྱུང་འཇིག་རྟེན་རིག་པར་རོ། །ཞེས་པའི་བར་དུ་ནི་གོ་བར་བྱ་བ་བསྟན་པ་སྟེ་ཐེག་པ་ཆེན་པོ་ཡང་དག་པའི་ཡེ་ཤེས་ངས་ཇི་ལྟར་རིག་པ་བཞིན་
དུ་སེམས་ཅན་ཐམས་ཅད་ལ་ཡང་བསྟན་ཞེས་པའི་དོན་ཏོ། །མདོ་ལས། ཇི་ལྟར་མཐོང་ཞིང་ཇི་ལྟར་བསམས་པ་དང་། །ཞེས་པ་ནས། དུར་ས་དེ་དག་ཕྱི་ཕྱིར་སྐྱེ་བར་འགྱུར། །ཞེས་པའི་བར་དུ་ཚིགས་བཅད་གཅིག་གིས་ཆུད་པར་བྱེད་པ་བསྟན་པ་སྟེ། སྔོན་གྱི་སྨོན་ལམ་ཡོངས་སུ་རྫོགས་ནས་
