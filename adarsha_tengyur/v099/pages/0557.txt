སྔ་མ་ལས་མཚན་ཉིད་མི་འདྲ་བ་ཁྱད་པར་གང་གིས་སྨྲ་བ་པོ་ཞེས་བརྗོད་པ་འབྱུང་ཞིང་དེའི་རྒྱུ་ནི་སྨྲ་བ་པོ་ཡིན་པས་སྨྲ་པ་པོ་ཚིག་སྨྲ་བར་བྱེད་དོ་ཞེ་ན་དེ་ཡང་མི་རུང་སྟེ། གང་ཕྱིར་སྨྲ་བའི་སྔ་རོལ་མེད། །གང་ཞིག་ཅི་ཞིག་སྨྲ་བར་འགྱུར། །ཞེས་བྱ་བ་སྨྲས་
ཏེ། ཇི་ལྟར་ན་སྨྲ་བ་པོ་ཚིག་སྨྲ་བར་བྱེད་ཅིང་ཞེས་བྱ་བ་དེ་ལ་རང་གི་བདག་ཉིད་ལ་བྱ་བ་བྱེད་པ་མི་འཐད་ཅེ་ན། འདིར་མདོ་སྡེ་པ་དག་སྨྲ་བ་པོ་ཚིག་སྨྲ་བར་བྱེད་པས་བྱ་བ་གང་དང་ལྡན་པ་དེ་བྱེད་པར་འདོད་པའི་གཞུང་བསྟན་པའི་ཕྱིར། གང་གི་ཕྱིར་འདི་ཡང་ཞེས་བྱ་བ་ལ་
སོགས་པ་སྨྲས་སོ། །འདིར་ཡང་འདུན་པའི་རྒྱུ་ཞེས་བྱ་བ་ལ་འདུན་པ་ཞེས་བྱ་བ་ནི་བྱེད་འདོད་པ་ཉིད་དོ། །འདུན་པ་དེ་ཉིད་རྒྱུ་ཡིན་པས་འདུན་པའི་རྒྱུའོ། །གང་གི་རྒྱུ་ཡིན་ཞེ་ན། རྩོལ་བས་ཏེ་དེའི་ཕྱིར་འདུན་པའི་རྒྱུ་ལས་བྱུང་བའི་རྩོལ་བ་ཞེས་བྱ་བ་སྨྲས་ཏེ། 
རྩོལ་བ་ནི་གཡོ་བའི་མཚན་ཉིད་དོ། །རྩོལ་བ་དེས་གནས་དང་བྱེད་པ་སྐྱེད་པས་ན་རྩོལ་བས་སྐྱེད་པའི་གནས་དང་བྱེད་པ་ཞེས་བྱ་བ་སྨྲས་སོ། །གནས་དང་བྱེད་པ་དག་གི་རྒྱུ་ལས་ངག་གི་འདུ་བྱེད་ཀྱི་ཚོགས་ཡི་གེ་དང་ངག་དང་ཚིག་ཅེས་བྱ་བ་འབྱུང་བ་ནི་ཚིག་
སྨྲ་བ་ཡིན་ནོ་ཞེས་བྱ་བར་སྦྱར་རོ། །དེ་ལ་ངག་གི་འདུ་བྱེད་ཀྱི་ཚོགས་ཞེས་བྱ་བ་ནི་ཡི་གེ་དང་ངག་དང་ཚིག་ཅེས་བྱ་བའི་ཚོགས་ཏེ་ཚིག་སྨྲ་བར་བྱེད་པ་ཞེས་བྱ་བ་ཡིན་ནོ། །ཡི་གེ་ཞེས་བྱ་བ་ནི་ཨ་ལ་སོགས་པའོ། །ངག་ཅེས་བྱ་བ་ནི་ཡི་གེ་དག་བསྡུས་པ་སྟེ། ལྷས་
བྱིན་ཞེས་བརྗོད་པ་ལ་སོགས་པའོ། །ཚིག་ཅེས་བྱ་བ་ནི་ངག་དག་བསྡུས་པ་སྟེ། ལྷས་བྱིན་བུམ་པ་བྱེད་དོ་ཞེས་བརྗོད་པ་ལ་སོགས་པའོ། །ཇི་ལྟར་ན་གནས་ཀྱི་རྒྱུ་ལས་ངག་གི་འདུ་བྱེད་ཀྱི་ཚོགས་ཡི་གེ་དང་ངག་དང་ཚིག་ཅེས་བྱ་བ་དག་འབྱུང་ཞེ་ན། ཡི་གེ་རྣམས་
ཀྱི་གནས་བརྒྱད་དེ། །བྲང་དང་ལྐོག་མ་མགོ་དང་ནི། །ལྕེ་ཡི་དྲུང་དང་སོ་དང་སྣ། །རྐན་དང་དེ་བཞིན་མཆུ་གཉིས་སོ། །ཞེས་བྱ་བ་བཤད་པ་ལྟ་བུའོ། །ཇི་ལྟར་ན་བྱེད་པའི་རྒྱུ་ལས་ངག་གི་འདུ་བྱེད་ཀྱི་ཚོགས་ཡི་གེ་དང་ངག་དང་ཚིག་ཅེས་བྱ་བ་འབྱུང་ཞེ་ན།
