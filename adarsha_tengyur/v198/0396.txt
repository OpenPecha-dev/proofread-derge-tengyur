འོ་མ་དང་། །ལྷན་ཅིག་བླུད་ཅིང་འོ་མར་བཟའ། །ཡང་ན་དེ་དང་བསྐོལ་བ་ཡི། །མར་གྱིས་ཚད་འཁྲུ་མྱུར་དུ་སེལ། །རྒྱ་སྐྱེགས་བཅའ་སྒ་པི་པི་ལིང་། །སྐྱེར་པའི་ཤུན་པ་པུ་ཙེ་ཤེལ། །དུག་མོ་ཉུང་དང་བཅས་བསྐོལ་མར། །ཐུག་སླ་ཐུག་པ་དང་སྦྱར་བས། །ནད་གཞི་གསུམ་ལས་
བྱུང་བ་ཡི། །མི་ཟད་འཁྲུ་བའང་མྱུར་དུ་སེལ། །ས་ནག་དུང་དང་ཤིང་མངར་དང་། །སྦྲང་རྩིར་ཁྲག་འབྲས་ཆུ་བླུད། །ཁྲག་གིས་འཁྲུ་བ་སེལ་བར་བྱེད། །པྲི་ཡང་ཀུ་དང་སྦྲང་རྩིའང་ཕན། །ཏིལ་ནག་བཏགས་པའི་ལྡེ་གུ་དང་། །ཤ་ཁར་ནི་ཆ་ལྔ་ནི། །ར་ཡི་འོ་མ་སྦྱར་བ་བླུད། །
འཕྲལ་དུ་ཁྲག་ཏུ་འཁྲུ་བ་སེལ། །ཁ་ར་སྦྲང་རྩི་ཙནྡན་བཅས། །འབྲས་ཀྱི་ཆུ་དང་བླུད་ན་ནི། །ཁོང་ཚ་སྐོམ་དང་གཅིན་ནད་དང་། །ཁྲག་ཏུ་འཁྲུ་ལས་ཐར་བར་འགྱུར། །གཞང་ཁ་ཚ་ཞིང་འདྲུལ་བ་ལ། །བསིལ་བའི་སྨན་གྱིས་བླུགས་ཤིང་བསྐུ། །གང་ཚེ་ཡང་དང་ཡང་དུ་

ཨུ་དུམ་ཝཱ་ར་ནྱ་གྲོ་དྷ། །ཨ་ཤྭདྠ་ཡི་མྱུ་གུ་ནི། །བརྡུངས་ཏེ་ཞག་གཅིག་ཆུ་ནང་སྦངས། །མཉེས་ཏེ་བཙགས་པར་མར་བསྐོལ་ལ། །ཁ་ར་དེ་ཡི་ཕྱེད་བཏབ་ཅིང་། །སྦྲང་རྩི་བཞི་ཆ་བཏབ་སྟེ་བླུད། །གང་ཞིག་ཁྲག་ནི་ཐུར་དུའམ། །གྱེན་དུ་འབྱུང་བར་བྱེད་ལ་སྦྱིན། །བད་ཀན་
