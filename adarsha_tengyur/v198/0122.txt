དེ་ལ་ལྕེ་མྱང་ཚ་མངར་བཅས། །རོ་ཙ་ཡིད་འོང་ཉེས་གསུམ་སེལ། །ཡང་ཞིང་ཚ་ལ་མིག་ལ་ཕན། །ཚ་བ་མི་སྐྱེད་མེ་དྲོད་སྐྱེད། །རྩལ་རུ་ཚ་ཡང་ཡིད་འོང་ཞིང་། །སྒྲེགས་པ་སྦྱོང་ཞིང་དྲི་ཞིམ་ལ། །ཞུ་རྗེས་ཚ་ཞིང་བཤང་སྲི་སེལ། །དྲོད་སྐྱེད་ཡི་ག་འབྱེད་བྱེད་ཡིན། །
ཁ་རུ་ཚ་ནི་སྟོད་བད་ཀན། །སྨད་ཀྱི་རླུང་འབྱིན་དྲོད་སྐྱེད་ལ། །བཤང་འགགས་ལྟོ་སྦོས་རྟུག་སྐམ་དང་། །གཟེར་ཟུག་ལུས་ལྕི་སེལ་བ་ཡིན། །རྒྱ་མཚོའི་ལན་ཚྭ་ཞུ་རྗེས་མངར། །ལྕི་ཞིང་བད་ཀན་རྣམ་འཕེལ་བྱེད། །བཀོད་མའི་ཚ་ནི་ཅུང་ཟད་ཁ། །ཚ་དང་བ་ཚ་རྣོ་བྱེད་
གཤེར། །ཁ་རུ་ཚ་ནག་ཡོན་ཏན་དྲི། །མ་གཏོགས་ཡོན་ཏན་བི་དར་མཚུངས། །མཚོ་ཚ་ཡང་སྟེ་ས་སྐམ་ཚ། །བད་ཀན་སྐྱེད་ཅིང་འཇུ་དཀའ་ལྕི། །ལན་ཚྭ་རྣམས་ཀྱི་སྦྱོར་བ་ལ། །རྒྱ་མཚོ་ལ་སོགས་རབ་ཏུ་སྦྱར། །ནས་ཚིག་འཁུས་པའི་ལན་ཚྭ་ནི། །སྐྲན་དང་
སྙིང་ནད་སྐྱ་རྦབ་དང་། །ཕོ་བ་མཆེར་པ་ལྐོག་ནད་དང་། །ལུད་པ་གཞང་འབྲུམ་དབུགས་མི་བདེ། །ལྟོ་སྦོས་བད་ཀན་སེལ་བ་ཡིན། །སྭ་རྫི་ཀ་སོགས་ཐལ་ཚ་ཀུན། །རབ་རྣོ་ཚ་ཞིང་སྲིན་སེལ་ཡང་། །ཁྲག་མཁྲིས་སྐྱེད་བྱེད་སྨིན་བྱེད་ལ། །གཅོད་ཅིང་མི་ཞིམ་རྣམ་
པར་འབྱེད། །ནས་ཚིག་འཁུས་ཚ་ཁ་བས་ན། །མདོག་མདངས་ཁུ་བ་སྐྲ་མིག་གནོད། །ཤིང་ཀུན་རླུང་དང་བད་ཀན་སེལ། །གཟེར་འཇོམས་མཁྲིས་པ་སྐྱེད་བྱེད་དེ། །ཞུ་རྗེས་རོ་ཚ་ཡི་ག་འབྱེད། །དྲོད་སྐྱེད་ཟས་འཇུ་ཡང་བ་ཡིན། །ཨ་རུ་ར་བསྐ་ཞུ་རྗེས་
མངར། །རྩུབ་ཅིང་ལན་ཚྭའི་རོ་མེད་ཡང་། །དྲོད་ཆེ་ཟས་འཇུ་ཡིད་གཞུངས་བྱེད། །ན་ཚོད་ལ་འཇོག་དམ་པ་ཡིན། །དྲོ་ཞིང་འཁྲུལ་ཚེ་ལ་ཕན། །བློ་དང་དབང་གསལ་སྟོབས་རབ་སྐྱེད། །མཛེ་དང་བཞིན་ལུས་མདོག་མི་སྡུག །སྐད་འགགས་རིམས་རྙིང་
མི་བཟད་རིམས། །གླད་པ་མིག་ནད་སྐྱ་རྦབ་དང་། །མིག་སེར་ཕོ་བའི་ནད་རྣམས་དང་། །སྐེམ་བཅས་འོར་དང་གཅིན་ནད་སེལ། །ཚད་འཁྲུ་རྨོངས་བག་མྱོས་སློན་སྲིན། །མཆིལ་མང་ལུད་པ་དབུགས་མི་བདེ། །གཞང་འབྲུམ་མཆེར་པ་ལྟོ་སྦོས་དུག། །
