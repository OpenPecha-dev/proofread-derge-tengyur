སྭཱ་ཧཱ། སེམས་ཅན་ཕོངས་པ་བསྐྱབ་པ་དང་། །ངན་སོང་གསུམ་གྱི་མགོན་བྱས་དང་། །དཀོན་མཆོག་གསུམ་གྱི་རྟེན་བྱིན་དང་། །འཁོར་བའི་སྐྱབས་བགྱི་བྱང་ཆུབ་བསྔོ། །ཨོཾ་ཀུ་ཤ་ལ་བ་རི་ན་ཡེ་སྭཱ་ཧཱ། སེམས་ཅན་ཕོངས་པ་བསྐྱབ་པ་དང་། །ངན་སོང་གསུམ་གྱི་མགོན་བྱས་དང་། །དཀོན་
མཆོག་གསུམ་གྱི་རྟེན་བྱིན་དང་། །འཁོར་བའི་སྐྱབས་བགྱིས་བདག་ལུས་དབུལ། །ཨོཾ་ཨཱཏྨ་ནཾ་ནིརྱ་ཏ་ཡཱ་མི་སྭཱ་ཧཱ། ངན་འགྲོ་རྣམས་ལས་བསྒྲལ་བ་དང་། །ཉོན་མོངས་རྣམས་ལས་སྒྲོལ་བ་དང་། །སྲིད་པའི་མི་བདེ་དབུགས་བྱིན་དང་། །མྱ་ངན་འདས་པའི་སེམས་བསྐྱེད་དོ། །ཨོཾ་བོ་དྷི་ཙིཏྟ་ཨུཏྤཱ་ད་ཡཱ་
མི་སྭཱ་ཧཱ། ངན་འགྲོ་རྣམས་ལས་བསྒྲལ་བ་དང་། །ཉོན་མོངས་རྣམས་ལས་སྒྲོལ་བ་དང་། །སྲིད་པའི་མི་བདེ་དབུགས་འབྱིན་དང་། །མྱ་ངན་འདས་པར་སྐྱབས་སུ་མཆི། །ཨོཾ་ཏྲི་ཤ་ར་ཎཾ་གཙྪ་ཡཱ་མི་སྭཱཧཱ། སྔོན་གྱི་རྒྱལ་བའི་སྲས་རྣམས་ཀྱིས། །སྨོན་ལམ་ཇི་ལྟར་མཛད་པ་བཞིན། །སྦྱིན་དང་ཚུལ་
ཁྲིམས་བཟོད་བརྩོན་འགྲུས། །བསམ་གཏན་ཤེས་རབ་བདག་བསྐྱེད་ཤོག །ཨོཾ་པྲ་ནེ་ཏ་ན་ཡེ་སྭཱ་ཧཱ། དེ་ལྟར་འཁོར་ལོ་དྲུག་སྤྱན་དྲང་ཞིང་ཚོགས་བསགས་ནས་སྦྱང་བ་བཤད་པ་ནི། དང་པོ་མཆོད་པའི་ཡན་ལག་རྫས་སྦྱང་བ་བཤད་པ་ནི་སྔར་བསྟན་པའི་བདུད་རྩི་ཐབ་སྦྱོར་གྱི་སྔགས་དང་ཕྱག་རྒྱས་
མཆོད་པའི་བགེགས་བསྐྲད་ལ། བདུད་རྩི་ཐབ་སྦྱོར་གྱི་སྔགས་བཟླས་པའི་སྤོས་ཆུ་ལག་མཐིལ་དུ་བླུགས་ལ་རྫས་རྣམས་ལ་བལྟ་ཞིང་ཀུ་ཤའི་པིར་ཐོགས་ལ་ཨོཾ་ཛྭ་ལ་ཛྭ་ལ་བན་དུ་རི་ཧཱུཾ་ཕཊ་ཅེས་བྱ་བའི་ཆུས་བསང་བ་དང་། ཡུངས་ཀར་གྱིས་བྲབ་པར་བྱའོ། །དེའི་རྗེས་ལ་གནས་སྦྱང་བ་ནི། ས་གཞིའི་
དབུས་སུ་ཡི་གེ་རཾ་བསམས་ལ་འོད་ཟེར་གྱི་སྤྲོ་བསྡུ་བྱས་པས་ལས་དང་ཉོན་མོངས་པའི་ས་གཞི་སྦྱངས་ལ། དེ་ནས་སྟོང་པ་སྒྱུ་མ་ལྟ་བུ་ལས། ཡི་གེ་ཨ་གསེར་གྱི་མདོག་ཅན་བསམས་ལ་དེའི་འོད་ཟེར་འཕྲོ་འདུ་ལས་གསེར་གྱི་ས་གཞི་རྡོ་རྗེ་རྩེ་གསུམ་པས་མཚན་པ་དཔངས་སུ་དཔག་ཚད་འབུམ་
ཕྲག་བརྒྱད་ཡོད་པ། རྒྱར་སྟོང་གསུམ་གྱི་ཚད་ཡོད་པ་བསམས་ལ། བདུད་རྩི་འཁྱིལ་པའི་ཕྱག་རྒྱས་ལ་བཞག་ནས་ཨོཾ་ཨ་ཙ་ལེ་བི་ར་སྭཱ་ཧཱ། ཞེས་བྱ་བས་ས་གཞི་རྡོ་རྗེར་བྱིན་གྱིས་བརླབ་བོ། །དེ་ནས་སའི་ལྷ་མོའི་སྔགས་དང་ཕྱག་རྒྱས་ཕྱིས་འཆད་པར་འགྱུར་བས་ཀྱང་བྱིན་གྱིས་བརླབ་
