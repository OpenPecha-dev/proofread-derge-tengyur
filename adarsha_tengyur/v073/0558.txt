བླ་མ་དང་རྡོ་རྗེ་སྤུན་ལ་ཞེ་སྡང་གིས་མི་སྨད། མཐུ་རྒྱས་མི་བྱའོ། །དེ་ལྟར་སྒྲུབ་པའི་ཡན་ལག་བསྟེན་ནས། ལུས་ཀྱི་ཀུན་དུ་སྤྱོད་པ་ནི་ཐུན་གསུམ་དུ་ཁྲུས་བྱས་ཏེ། ཨོཾ་ཧཱུཾ་ཧ་ན་བཛྲ་ཎི་ཧ་ས་ལན་འགའ་བཟླས་ནས་རོ་སྟོད་དང་རོ་སྨད་ལ་བསྐུ་བ་དང་། ཨོཾ་བཛྲ་ཧ་ར་ཧཱུཾ་ཞེས་བྱ་བའི་སྔགས་ཀྱིས་ཆུའི་བགེགས་བསལ་
ལ། ན་མོ་བཛྲཱ་ཡ་ཨོཾ་ཧཱུཾ་དྷཱུ་ན་དྷཱུ་ན་མ་ཐ་མ་ཐ་བིདྷྭཱ་ན་སྭཱ་ཧཱ། ཞེས་པས་འཁྲུད་ཅིང་བདུད་རྩི་འཁྱིལ་པའི་སྔགས་རིངས་ཀྱིས་བགེགས་བསྐྲད་ལ་ཁྲུས་བྱས་ནས། ཨོཾ་ཨ་མྲྀ་ཏེ་ཧཱུཾ་ཕཊ་ཅེས་པས་ཁྲུས་ཟིན་པའི་རྟིང་ལ་ཆུ་སྙིམ་པ་གསུམ་སྤྱི་བོར་བླུགས་སོ། །ཨོཾ་སུ་སི་ཏི་ཀ་རི་སྭཱ་ཧཱ། ཞེས་ཟློས་ཤིང་། ཁྱིམ་པ་ཡིན་
ན་སྤྱི་གཙུག་བཅིང་། རབ་བྱུང་ཡིན་ན་ཞྭ་དང་གོས་གྱོན། དེ་ནས་ཉེ་བའི་དབང་པོ་ལྔ་ཁ་དང་ལག་པ་དང་རྐང་པ་དང་བཤང་བ་དང་གཅི་བའི་ལམ་དང་ཉེ་བའི་དབང་པོ་ལྔ་ཆུས་གཏོར་ལ། ཨོཾ་ཀི་ལི་ཀི་ལི་རཀྵ་རཀྵ་བཛྲ་ཧཱུཾ་ཕཊ་ཅེས་བྱ་བ་ཟློས་ཤིང་བསྙེན་པའི་ཁང་པར་འགྲོ་བར་བྱའོ། །རྡོ་རྗེའི་རིགས་ལ་སྟོད་
གཡོགས་སྨད་གཡོགས་སྔོན་པོའམ། དཀར་པོ་ལ་ཁྲོ་བོ་བདུད་ཙི་ཐབ་སྦྱོར་བཟླས་ཏེ་བགོ་བ་དང་། སངས་རྒྱས་སྤྱན་བཟླས་ཏེ་ཟེ་ཁ་བཅིང་བ་དང་། ཨོཾ་བཛྲ་ཀྲོ་དྷ་མ་ཧཱ་བ་ལ་ཧ་ན་ད་ཧ་པ་ཙ་བིདྷྭན་ས་ཡ་ཨུཙྪཥྨ་ཀྲོ་དྷ་ཧཱུཾ་ཕཊ་ཅེས་པས་རབ་བྱུང་ཡིན་ན་ཤམ་ཐབས་བགོའོ། །ཁྱིམ་པ་ཡིན་ན་རྐང་སྣམ་བགོའོ། །རྡོ་
རྗེ་དབྱིངས་ཀྱི་དབང་ཕྱུག་མའི་ཕྱག་རྒྱ་དང་། ནམཿས་མནྟ་བུདྡྷཱ་ནཱཾ་ཧཱུཾ་དྲུཾ་ཞེས་པས་ཅོད་པན་བཅིང་ངོ་། །ན་མཿས་མནྟ་བུདྡྷཱ་ནཱཾ་དྷིཾ་དྷིཾ་རིཾ་རིཾ་བཛྲི་ཛ་སྭཱ་ཧཱ། ཞེས་པས་རྡོ་རྗེ་དྲིལ་བུ་བཟུང་ངོ་། །དེ་སྐད་དུ་ཡང་། གོས་དང་ཁྲུས་དང་རྒྱན་དག་གིས། །བདག་ཉིད་བརྒྱན་པར་མཐོང་ནས་ནི། །དེ་ཚེ་བསྙེན་པའི་སྒྲུབ་པོ་
ཡིས། །ལས་རྣམས་རབ་ཏུ་བརྩམ་པར་བྱ། །ཞེས་གསུངས་སོ། །དེ་ནས་བདག་ཉིད་ལས་ཐམས་ཅད་པ་བདུད་རྩི་ཐབ་སྦྱོར་དུ་བསམ་པར་བྱ་སྟེ། ན་མོ་རཏྣ་ཏྲ་ཡཱ་ཡ། མ་ཧཱ་ཡཀྵ་སེ་ནཱ་པ་ཏ་ཡེ། ན་མོ་བཛྲ་ཀྲོ་དྷ་ཡ་ཏདྱ་ཐཱ་ཧུ་ལུ་ཧུ་ལུ་ཏིཥྛ་ཏིཥྛ་བནྡྷ་བནྡྷ་ཧ་ན་ཧ་ན་ཨ་མྲྀ་ཏེ་ཧཱུཾ་ཕཊ་ཅེས་བྱ་བ་དང་མཐེ་བོང་གཉིས་
ཀྱིས་མཐེའུ་ཆུང་མནན་ཏེ། བསྣོལ་བ་དཔུང་བར་གཞག་སྟེ། སྙིང་གའི་ཟླ་བའི་དཀྱིལ་འཁོར་ལ་ཧཱུཾ་བསྒོམས་ཏེ་སྥ་ར་ཎ་ཧཱུཾ་ཞེས་པས་རང་དང་འདྲ་བའི་ཁྲོ་བོ་དུ་མ་སྤྲོས་ཏེ་སེམས་ཅན་གྱི་དོན་བྱས་ནས། སཾ་ཧ་ར་ཎ་ཧཱུཾ་ཞེས་པས་བསྡུས་ནས་སྤྱན་གསུམ་པ་སྟག་གི་པགས་པའི་ཤམ་ཐབས་ཅན་ཀླུ་བརྒྱད་ཀྱིས་
