ཞེས་བྱར་བཤད། །ཡི་གེ་ཧ་ནི་པདྨར་བཤད། །དེ་དག་གཉིས་ཀྱི་ས་བོན་ནི། །སྱ་ཞེས་བྱ་བར་བཤད་པ་ཡིན། །ཞེས་གསུངས་པ་ཡིན་ཏེ། ཡི་གེ་ར་ནི་རྡོ་རྗེ་རྩེ་མོ་འདྲ་བའི་ཕྱིར་རོ། །ཧ་ནི་འདབ་ཆགས་ཀྱི་མཆུ་འདྲ་བའི་ཕྱིར་རོ། །ས་དང་ཡ་ཀུན་དུ་སྦྱོར་བ་ནི་གཉི་
གའི་ས་བོན་གྱི་གཟུགས་ཅན་ཡིན་པའི་ཕྱིར་རོ། །དེ་ཡང་ས་ཡན་དུ་སྲ་བ་ན་ཞེས་བྱ་བའི་ཡི་གེའི་ཐ་མ་གཟུང་བའི་ཕྱིར་ཏེ། དེ་ནི་དོན་དམ་པའོ། །ཇི་ལྟར་བཤད་པར་བྱ་བ་གསལ་བར་བསྟན་པ་ནི་མདོར་བསྡུས་པ་སྟེ་ཞེས་གསུངས་པ་ཡིན་ཏེ། ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད་ལས་
ནི་ཤིན་ཏུ་རྒྱས་པར་སྨོས་པ་ཡིན་ནོ་ཞེས་བྱ་བར་དགོངས་སོ། །དོན་དེ་ཉིད་གསལ་བར་བསྟན་པ་ནི་རྒྱས་པ་མིན་ཞེས་བྱ་བ་གསུངས་པ་ཡིན་ནོ། །ཡང་ན་མདོར་བསྡུས་པ་མ་ཡིན་གྱི་རྒྱས་པ་ཞེས་བྱ་བར་སྦྱར་རོ། །འདིར་དགོས་པ་ནི་འདི་ཡིན་ཏེ། ནམ་མཁའ་དང་མཉམ་པའི་རྒྱུད་ནི་གཞུང་
རྒྱས་པ་ཡིན་ལ། འདིར་ནི་དོན་རྒྱས་པ་ཡིན་ནོ་ཞེས་བྱ་བ་སྟེ། ཁམས་གསུམ་པའི་བདག་ཉིད་དུ་འཁོར་ལོ་གསུམ་བཤད་པར་བྱ་བ་ཡིན་ནོ། །ད་ནི་ཚིག་བཞིས་གསང་བ་ཉིད་ཀྱི་ཁྱད་པར་བསྟན་པར་བྱ་སྟེ། དཔལ་གྱི་སྒྲ་ནི་མཆོད་པའི་ཚིག་གོ། །ཧེ་རུ་ཀ་ཞེས་བྱ་བ་ནི་རྒྱུད་འདི་ར་
བཅོམ་ལྡན་འདས་བདག་པོའི་མཚན་ཏེ། དཔལ་གཙོ་བོར་གྱུར་པའི་ཧེ་རུ་ཀ་ནི་དཔལ་ཧེ་རུ་ཀའོ། །དེའི་ཡང་དག་སྦྱོར་བ་ནི་བཅོམ་ལྡན་འདས་ཡུམ་རྡོ་རྗེ་ཕག་མོ་དང་ལྷན་ཅིག་རོ་གཅིག་པའི་ཚུལ་གྱིས་ཀུན་དུ་འཁྱུད་པའོ། །ཁམས་གསུམ་པའི་བདག་ཉིད་དུ་བཅོམ་ལྡན་འདས་ཧེ་རུ་ཀ་ཞེས་
བྱ་བ་ཡི་གེ་གསུམ་གྱིས་སྟོན་པ་ཉིད་ཡིན་ནོ། །ཆོས་ཐམས་ཅད་ནི་རྒྱུ་དང་བྲལ་བའོ། །ཆོས་ཐམས་ཅད་ནི་རྟོག་པ་དང་བྲལ་བའོ། །ཆོས་ཐམས་ཅད་ནི་གང་ན་ཡང་མི་གནས་པའོ་ཞེས་པ་ནི་མི་དམིགས་པའི་ཡི་གེ་གསུམ་གྱི་བདག་ཉིད་ཡིན་ནོ། །ཡང་ན་དཔལ་ཧེ་རུ་ཀ་ཞེས་བྱ་བ་ནི་ཁྱད་
པར་དུ་བྱ་བའི་ཚིག་ཡིན་ལ། གསང་བ་ཞེས་བྱ་བ་ནི་ཁྱད་པར་དུ་བྱེད་པའི་ཚིག་ཡིན་ཏེ། དེ་ཉིད་ཀྱིས་ནི་འདོད་པའི་དོན་ཀུན་སྒྲུབ་པར་བྱེད་པ་ཡིན་ནོ་ཞེས་བཅོམ་ལྡན་འདས་ཀྱིས་ཁྱད་པར་དུ་གསུངས་པ་ཡིན་ཏེ། ཐམས་ཅད་ཀྱི་འདོད་པ་ནི་འདོད་པ་ཀུན་ཏེ་དེ་ཉིད་ནི་དོན་དེ་དགོས་པའོ། །དེའི་
