རྒྱས་ལ་ཕྱག་བྱ་བ་དང་། ཁྲོ་བའི་བསྟོད་པ་བྱ་བ་དང་། བགེགས་ལ་བཀའ་བསྒོ་བ་དང་། ལྷག་པ་དང་མ་ཚང་བ་ཁ་བསྐང་བའི་ཕྱིར་བསྐོར་བ་ལན་བདུན་བྱ་བ་དང་། ལྟ་བས་བགེགས་བསལ་བ་དང་། དཔའ་བོ་དང་དཔའ་མོའི་སྟང་སྟབས་བྱ་བ་དང་། གནོད་མཛེས་ཀྱི་འཁོར་
ལོས་སྲུང་བའི་འཁོར་ལོ་བྱ་བ་དང་། ཕུར་བུས་བཏབ་ནས་དྲིའི་ཐིག་ལེ་བྱ་བ་དང་། ལྷ་ལྷག་པར་གནས་པར་བྱ་བ་དང་། རྩ་བའི་རྒྱུད་ནས་བཤད་པའི་བུམ་པ་དགུའམ། བཤད་པའི་རྒྱུད་ནས་བྱུང་བའི་བུམ་པ་དྲུག་ལ་ལྷག་པར་གནས་པར་བྱ་བ་དང་། ནང་གི་སྦྱིན་སྲེག་
ཚོགས་ཀྱི་འཁོར་ལོ་དང་། ཕྱི་རོལ་གྱི་སྦྱིན་སྲེག་གང་ཡང་རུང་བས་ལྷག་པར་གནས་པར་བྱ་བ་དང་། ཁ་སྦྱོར་དང་མཁའ་འགྲོ་ལྟར་གཏོར་མ་ལྷག་པར་གནས་པར་བྱ་བ་དང་། དེ་ཉིད་བསྡུས་པའི་རྒྱུད་དང་། རྡོ་རྗེ་མཁའ་འགྲོ་ལྟར་སློབ་མ་ལྷག་པར་གནས་པའི་ཆོ་
ག་བྱའོ། །འདིའི་ཆོ་ག་རྒྱ་ཆེར་ནི་ཡིད་འོང་ཕྲེང་བར་བལྟ་བར་བྱའོ། །དེ་ལྟར་ལྷག་པར་གནས་པ་བྱས་ནས་ཕྱི་དེ་ནང་པར་ཉི་མ་འཆར་ཀར་དཀྱིལ་འཁོར་ནམ་མཁར་བཏེག་ལ་ཐིག་གདབ་པར་བྱའོ། །དེ་ལྟར་བདག་ཉིད་རབ་བསྲུང་ཞིང་། །སྔགས་དང་ཕྱག་
རྒྱས་རྣམ་བརྒྱན་ནས། །དངོས་གྲུབ་ཐམས་ཅད་རབ་སྟེར་བའི། །དཀྱིལ་འཁོར་འཇིགས་པ་བྲི་བར་བྱ། །ཞེས་བྱ་བས་ནི་དཀྱིལ་འཁོར་ཆེན་པོའི་སློབ་དཔོན་གྱི་ལས་མདོར་བསྟན་པ་ཡིན་ནོ། །དེ་ལྟར་བདག་ཉིད་རབ་བསྲུང་ཞེས་བྱ་བ་ནི་སྟ་གོན་གྱི་ཆོ་ག་རྫོགས་ནས་ཕྱི་དེའི་
ཉིན་པར་ཚད་མེད་པའི་རྣལ་འབྱོར་དང་། བྱིན་གྱིས་བརླབས་པའི་རྣལ་འབྱོར་དང་། བསྲུང་བའི་རྣལ་འབྱོར་དང་། ཚོགས་སོགས་པའི་རྣལ་འབྱོར་དང་། སྟོང་པ་ཉིད་ཀྱི་རྣལ་འབྱོར་བྱས་ནས་དང་པོ་སྦྱོར་བའི་དང་དཀྱིལ་འཁོར་རྒྱལ་མཆོག་གི་ལྷར་བསྐྱེད་
དེ་ལས་ཀྱི་རྒྱལ་མཆོག་གི་དུས་སུ་ཆོ་ག་དྲུག་གི་སྔགས་ཀྱིས་ལྷར་བསྐྱེད་པ་དང་། ཡེ་ཤེས་སེམས་དཔའ་གཞུག་པ་དང་། དབང་བསྐུར་ཞིང་ཏིང་ངེ་འཛིན་བསྒོམས་ནས་ཕྱག་རྒྱ་དྲུག་གིས་རྣམ་པར་བརྒྱན་པས་དངོས་གྲུབ་ཐམས་ཅད་སྟེར་བའི་དཀྱིལ་འཁོར་ཐིག་
