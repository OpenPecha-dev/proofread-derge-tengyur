སྔགས་ཀྱིས་གཙང་བར་བྱས་པའི་དྲི་དང་བཅས་པ་གཏོར་བ་སྟེ། རྡོ་རྗེའི་ཆུས་ཡང་དག་པར་གཏོར་རོ་ཞེས་བྱ་བའི་དོན་ཏོ། །བདག་ཉིད་ཀུན་ནས་སྒོ་ཡིས་ནི། །ཞེས་བྱ་བ་ཁྱད་པར་དུ་བྱེད་དེ། །དཔལ་ཧེ་རུ་ཀའི་བདག་ཉིད་གང་ཡིན་པའི་བཅོམ་ལྡན་འདས་ཀྱི་ཞལ་བཞི་
ཉིད་ཡིན་པའི་ཕྱིར་རོ། །མར་མེ་བརྒྱ་ནི་དབུལ་བར་བྱ། །ཞེས་བྱ་བ་ནི་དོན་ཅི་ལྟ་བ་དེ་ཉིད་དོ། །ཡང་ན་རྣམ་པ་སྣ་ཚོགས་པའི་བདག་ཉིད་ཀྱི་ཁྱད་པར་དུ་བརྗོད་དོ། །གཞན་ན་སྒྲོན་མ་ལྔ་གང་ཡིན་པས་རྔ་མོ་དང་རྟ་ལ་སོགས་པའི་ཤ་ཡང་གཟུང་བ་ཉིད་ཀྱི་ཕྱིར་རོ། །ནམ་མཁའི་མཁའ་
འགྲོ་མ་རྣམས་ནི། །ཡིད་ཀྱིས་ཐམས་ཅད་སྟེང་དུ་དགོད། །ཅེས་པ་ནི་ཐུགས་ཀྱི་འཁོར་ལོའི་མཁའ་འགྲོ་མ་ཐམས་ཅད་དགོད་པར་བྱ་བ་སྟེ། སེམས་ནི་ནམ་མཁའ་ལ་འགྲོ་བའི་ངོ་བོ་ཉིད་ཀྱི་ཕྱིར་རོ། །ས་སྟེང་མཁའ་འགྲོ་མ་གང་ཡིན། །དཀྱིལ་འཁོར་ཀུན་དུ་དགོད་པར་བྱ། །
ཞེས་པ་ནི་ཐམས་ཅད་དུ་དཀྱིལ་འཁོར་ལ་ཐམས་ཅད་ཀྱིས་སམ། ས་སྟེང་གི་འཁོར་ལོ་གང་བ་དེ་ལ་སྤྱོད་པ་ཡང་དགོད་པར་བྱ་བ་སྟེ། གསུང་གི་འཁོར་ལོ་ནི་དབུས་སུ་གནས་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །གང་ཡངས་འོག་མཁའ་འགྲོ་མ། །དེ་ནི་ས་འོག་རྣམས་སུ་དགོད། །ཅེས་པ་ནི་
ས་འོག་ན་གནས་པ་ཡང་སྟོབས་ཀྱིས་སྐུའི་འཁོར་ལོ་དགོད་པར་བྱ་ཞེས་རྟོགས་པར་བྱའོ། །མ་མོ་ཐམས་ཅད་ཕྱོགས་དང་ནི། །ཕྱོགས་མཚམས་རྣམས་སུ་དགོད་པར་བྱ། །ཞེས་པ་ནི་ཁྭ་གདོང་མ་ལ་སོགས་པའི་རྣལ་འབྱོར་མ་བརྒྱད་དགོད་པར་བསྟན་པའོ། །དྲི་དང་ཞེས་བྱ་བ་
ལ་སོགས་པ་ནི་མཆོད་པ་རྣམ་པ་ལྔའི་ཆོ་ག་སྟོན་ཏོ། །མར་མེ་བརྒྱ་ཡང་དབུལ་བྱ་ཞིང་། །ཞེས་པ་ནི་སྔོན་དུ་བསྟན་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །འདིར་ནི་མཆོད་པ་རྣམ་པ་ལྔའི་ཁ་བསྐང་བའི་དོན་དུ་གསུངས་པའོ། །མཆོད་བྱ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་དོན་ཇི་ལྟ་བ་ཉིད་ཀྱི་ཕྱིར་མ་བཤད་
པ་ཡིན་ནོ། །རབ་མཉམ་གཞག་ཅེས་པ་ནི་ཐམས་ཅད་དུ་མཆོད་པའི་ཆོ་ག་རྣམ་པར་དག་པ་རྡོ་རྗེ་ཡིན་ནོ་ཞེས་པའི་དོན་ཏོ། །དཔལ་ཧེ་རུ་ཀ་མངོན་པར་བརྗོད་པའི་འཁོར་ལོ་མཆོད་པའི་ཆོ་གའི་ལེའུ་གཉིས་པའི་འགྲེལ་པའོ།།      །།དེ་ནས་ཞེས་པ་ནི་མངོན་དུ་བསྟན་པ་ཉིད་ཀྱི་
