གསུམ་དེ་བཞིན། །ཞི་དང་རྣམ་རྟོག་འཇིགས་པ་དང་། །འཇིགས་པ་བར་དང་ཤིན་ཏུ་འཇིགས། །སྲེད་དང་སྲེད་པ་བར་མ་དང་། །ཤིན་ཏུ་སྲེད་དང་ཉེར་ལེན་པ། །མི་དགེ་བཀྲེས་དང་སྐོམ་ཉིད་དང་། །ཚོར་དང་ཚོར་བ་བར་མ་དང་། །ཤིན་ཏུ་ཚོར་བའི་སྐད་ཅིག་དང་། །རིག་པ་པོ་དང་རིག་འཛིན་
བཞི། །སོ་སོར་རྟོག་དང་ངོ་ཚ་ཤེས། །སྙིང་རྗེ་དང་ནི་བརྩེ་བ་གསུམ། །དོགས་དང་བསམ་པ་ཉིད་དང་ནི། །ཕྲག་དོག་ཅེས་ནི་བྱ་བར་གྲགས། །རང་བཞིན་སུམ་ཅུ་རྩ་གསུམ་གང་། །ལུས་ཅན་རྣམས་ཀྱི་རང་རིག་སྟེ། །ཀུན་རྫོབ་གསལ་བའི་གཟུགས་ཀྱིས་ནི། །མཚན་མོའི་
མིང་དུ་ངེས་པར་བཤད། །དེ་བཞིན་བུད་མེད་མིང་དུ་བརྗོད། །འཇམ་པོ་ལྟ་བུའང་དེ་བཞིན་ནོ། །གཡོན་པའི་མིང་ཡང་དེ་ཉིད་དེ། །ཟླ་བའི་དཀྱིལ་འཁོར་པདྨ་ཅན། །བརྟན་པར་བྱ་བའི་རྒྱུར་གྱུར་པ། །དང་པོའི་ཡི་གེ་ཐིག་ལེར་བཅས། །ཟླ་བའི་འོད་ཟེར་ལྟར་སྣང་ཞིང་། །སྣང་བའི་
ཡེ་ཤེས་ཡང་དག་འབྱུང་། །ཤིན་ཏུ་སྟོང་པ་ཐབས་ལྡན་པ། །སྣང་བ་མཆེད་པ་ཞེས་བརྗོད་དེ། །དེ་བཞིན་དུ་ནི་ཀུན་བརྟགས་དང་། །དེ་བཞིན་སེམས་ལས་བྱུང་བར་བཤད། །ཆགས་དང་ཞེན་དང་དེ་བཞིན་དགའ། །དགའ་བ་བར་མ་ཤིན་ཏུ་དགའ། །རངས་དང་རབ་ཏུ་མགུ་བ་དང་། །ངོ་
མཚར་དེ་བཞིན་དགོད་པ་དང་། །ཚིམ་དང་དེ་བཞིན་འཁྱུད་པ་དང་། །འོ་བྱེད་དེ་བཞིན་འཇིབ་པ་དང་། །བརྟན་དང་བརྩོན་དང་ང་རྒྱལ་དང་། །བྱ་དང་འཕྲོག་དང་སྟོབས་དང་ནི། །སྤྲོ་དང་ལྷན་ཅིག་སྐྱེས་ཉིད་དང་། །དེ་བཞིན་ལྷན་སྐྱེས་བར་མ་དང་། །ཤིན་ཏུ་ལྷན་ཅིག་སྐྱེས་དང་གྲག །རྣམ་པར་སྒེག་
དང་ཞེ་འགྲས་ཉིད། །དགེ་དང་ཚིག་གསལ་བདེན་པ་དང་། །མི་བདེན་པ་དང་དེ་བཞིན་ངེས། །ཉེ་བར་མི་ལེན་སྦྱིན་པའོ། །སྐུལ་དང་དེ་བཞིན་དཔའ་བོ་དང་། །ངོ་ཚ་མེད་དང་སྒྱུ་དང་གདུག །མི་སྲུན་པ་དང་གྱ་གྱུ་ཉིད། །མཚན་ཉིད་བཞི་བཅུ་གང་ཡིན་པ། །ཤིན་ཏུ་སྟོང་པའི་སྐད་ཅིག་སྟེ། །
ཉིན་མོ་སྐྱེས་པའི་མིང་དང་ནི། །རྩུབ་མོ་གཡས་པར་བསྟན་པ་ཡིན། །ཉི་མའི་དཀྱིལ་འཁོར་མིང་དང་ནི། །དེ་བཞིན་དུ་ནི་རྡོ་རྗེའི་མིང་། །ཐིག་ལེ་གཉིས་ཀྱིས་བརྒྱན་པ་ཡི། །ཆ་རུ་དེ་ཉིད་ཤེས་པར་བྱ། །ཉི་མའི་འོད་ཟེར་འདྲ་བ་ཡང་། །སྣང་བ་མཆེད་པའི་སྦྱོར་ལས་བྱུང་། །སྣང་བ་ཐོབ་པ་
