མེ་ཆུང་ངུ་ཁྱད་དུ་གསད་པར་མི་བྱ་ཞིང་མེ་ཆུང་ངུ་ཟིལ་གྱིས་མནན་པར་མི་བྱ་བ་དང་། རྒྱལ་པོ་ཆེན་པོ་དགེ་སློང་གཞོན་ནུ་ཁྱད་དུ་གསད་པར་མི་བྱ་ཞིང་། གཞོན་ནུ་ཟིལ་གྱིས་མནན་པར་མི་བྱའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། རྒྱལ་པོ་ཆེན་པོ་དགེ་སློང་གཞོན་ནུ་ཡིན་ཡང་དེ་ནི་རྫུ་འཕྲུལ་ཆེ་ཞིང་
མཐུ་ཆེ་བ་ཡིན་པའི་ཕྱིར་རོ་ཞེས་བྱ་བ་ལ་སོགས་པ་གཞོན་ནུ་ལས་དཔེ་བླངས་པའི་མདོ་སྡེ་སྟེ། དེ་ཉིད་གཞུང་འདི་ལས་དགེ་འདུན་གྱི་དབྱེན་གྱི་གཞིའི་སྐབས་ནས་འོག་ནས་འབྱུང་ངོ་། །བཏུལ་ཞེས་བྱ་བ་ནི་འདུལ་བར་བྱས་པ་སྟེ། དཀོན་མཆོག་གསུམ་ལ་དད་པར་བྱས་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ས་མཚམས་
ཀྱི་བར་ཞེས་བྱ་བ་ནི་མཚམས་ཀྱི་བར་རོ། །སེལ་བ་དང་བཅས་པ་ཞེས་བྱ་བ་ནི་དཔུང་དང་བཅས་པའོ། །བྱ་མ་རྟ་ཞེས་བྱ་བ་ནི་ཉན་རྣ་སྟེ་བྱ་ར་བ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །བཏང་སྙོམས་ཞེས་བྱ་བ་ནི་བརྩལ་བ་ཆུང་ངུའོ། །མ་བསམས་པར་ཞེས་བྱ་བ་ནི་སོ་སོར་མ་བརྟགས་པར་རོ། །སྐྲ་ཡེགས་ཞེས་བྱ་
བ་ནི་སྐྲ་འགྲེང་བ་དང་། སྐྲ་འཕྱར་བར་གྱུར་པའོ། །ཀུ་ཅོ་འདོན་ཞེས་བྱ་བ་ནི་ངུ་བའི་སྒྲ་འབྱིན་པའོ། །བག་ཡངས་སུ་ཞེས་བྱ་བ་ནི་བག་ཚ་བ་མེད་པར་རོ། །གསོན་པོར་བཟུང་སྟེ་ཞེས་བྱ་བ་ནི་སྲོག་དང་ལྡན་བཞིན་དུ་བཟུང་སྟེའོ། །ལམ་པོ་ཆེ་ཞེས་བྱ་བ་ནི་ཤིང་རྟའི་ལམ་མོ། །སྲང་ཞེས་བྱ་བ་ནི་
ལམ་དོག་པའམ་ཚོང་འདུས་ཀྱི་དབུས་སོ། །རྗེས་སུ་སྒྲོགས་ཤིང་ཞེས་བྱ་བ་ནི་གང་དག་བྱ་བ་མ་ཡིན་པ་བྱེད་པ་དེ་དག་ལ་ཆད་པ་འདི་ལྟ་བུ་བྱའོ་ཞེས་ཐམས་ཅད་དུ་གོ་བར་བྱེད་ཅིང་ངོ་། །གསད་པ་ཞེས་བྱ་བ་ནི་གསད་པའི་གནས་ཏེ་དུར་ཁྲོད་ལ་སོགས་པའོ། །རྒུད་པ་ཞེས་བྱ་བ་ནི་དཔལ་མེད་པའོ། །
འབྱོར་པ་ཞེས་བྱ་བ་ནི་ཕུན་སུམ་ཚོགས་པའོ། །ཅིའི་ཕྱིར་འབྱོར་པ་ཀུན་དུ་སྐྱོ་བར་འགྱུར་བའི་གནས་ཡིན་ཞེ་ན། རྒྱལ་པོ་དང་རྐུན་པོ་ལ་སོགས་པ་དང་ཐུན་མོང་ཡིན་པས་གནོད་པ་དང་། མི་བརྟན་པ་ཉིད་དུ་ཀུན་དུ་སྐྱོ་བར་འགྱུར་བ་དང་། རང་གི་ལས་ཀྱིས་འཕངས་པས་གཞན་ལ་བསྒྱུར་དུ་མེད་པའི་ཕྱིར་
ལས་ལ་ཀུན་དུ་སྐྱོ་བར་འགྱུར་བའི་ཕྱིར་དེ་དག་ཀྱང་སྙེམས་པ་རྣམས་དང་། ཞུམ་པ་རྣམས་ཀུན་དུ་སྐྱོ་བ་བསྐྱེད་པའི་ཕྱིར་རོ། །གང་ཟག་དགེ་འདུན་གྱི་དབྱེན་བྱས་པ་རབ་ཏུ་དབྱུང་བར་མི་བྱ་ཞིང་དེ་ལ་སྡོམ་པ་སྐྱེ་བའི་ཆོས་ཉིད་མེད་པ་དང་། དེ་ལྟ་བུ་རབ་ཏུ་དབྱུང་བའི་རྟགས་ཅན་བསྙིལ་བར་བྱ་བ་བསྟན་པའི་
