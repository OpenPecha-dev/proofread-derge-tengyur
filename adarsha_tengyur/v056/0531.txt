རྡོ་རྗེ་ཞེས་བྱ་བའི་མཐེ་བོས་ཐུམ་པོ་བཅངས་པའི་མཚམས་སུ་གཡོགས་པ་ནི་རྡོ་རྗེ་ཏི་རཱི་ནི་རིའི་ཞེས་བྱ་བ་སྟེ། ཧཱུཾ་ཀ་ར་ཉིད་ཀྱི་དམ་ཚིག་གི་ཕྱག་རྒྱའོ། །ཧཱུཾ་ནི་ངག་གི་ཕྱག་རྒྱའོ་ཞེས་པ་ཧཱུཾ་ཀ་རའི་ངག་གི་ཕྱག་རྒྱ་ནི་ཧཱུཾ་ཞེས་བྱ་བ་ཉིད་དོ། །ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱ་ནི་ཞེས་པ་རབ་ཏུ་ཁྲོས་པའོ། །ཁྲོ་བོའི་
རྡོ་རྗེས་ལ་བཅིང་བ་ནི། །ཞེས་པ་རྡོ་རྗེ་བཅིང་བ་བྱས་ལ་བརྐྱང་སྟེ་ས་ལ་རེག་པར་བྱས་པས་བགེགས་དང་གནོད་པར་བྱེད་པ་མི་འབྱུང་སྟེ་ས་བཅིངས་པར་འགྱུར་རོ། །དེ་ལ་འདིར་ནི་རྡོ་རྗེ་འདོད་པ་ཁྲོ་བོའི་ཕྱག་རྒྱའོ་ཞེས་པ་ཕྱག་ན་རྡོ་རྗེ་ཉིད་འདོད་པ་ཁྲོ་བོའི་ཏིང་ངེ་འཛིན་ལ་གནས་པའི་ཕྱག་རྒྱ་སྟེ། མཛུབ་མོ་རྡོ་རྗེ་ཞེས་བྱ་བ་ནི་
མཛུབ་མོ་གཉིས་སོ་སོར་བསྒྲེང་བ་སྟེ། འདིས་ནི་སེམས་ཅན་ཐམས་ཅད་ལ་དབང་དུ་བྱེད་དོ། །རྡོ་རྗེའི་དམ་ཚིག་ནི་ཞེས་པ་སྤྱིའི་དམ་ཚིག་ཡིན་པའི་ཕྱག་རྒྱའོ། །ལྷ་ཆེན་པོ་ལ་སོགས་པ་འཇིག་རྟེན་པའི་ལྷ་རྣམས་དགུག་པའི་ཕྱག་རྒྱ་བསྟན་པ། རྡོ་རྗེའི་ཕྱག་རྒྱ་གཉིས་བཅིངས་ལ། །ཞེས་པ་ལག་པ་གཉིས་ཁུ་ཚུར་
མཛུབ་མོ་བསྒྲེང་བ་ལས་མཛུབ་མོ་གཉིས་བཀུག་སྟེ་ཕན་ཚུན་སྦྲེལ་ནས་གཅིག་ལ་གཅིག་དྲིལ་ཞིང་དགྲོལ་ལོ། །ཉི་མ་ལ་སོགས་པ་བར་སྣང་གི་ལྷ་རྣམས་དགུག་པའི་ཕྱག་རྒྱ་ནི། རྡོ་རྗེ་བཅིང་བ་བརྟན་བྱས་ལ། །ཞེས་པ་དག་གོ། །རྡོ་རྗེ་བཅིང་བ་ལས་མཛུབ་མོ་གཉིས་སོ་སོར་བཀུག་སྟེ་མཐེ་བོའི་སེན་མོའི་
སྟེང་དུ་བསྣན་ལ་མཐེ་བོང་གཉིས་གཤིབས་པས་གུང་མོ་བཙིར་རོ། །ནམ་མཁའ་ལ་སྤྱོད་པའི་ལྷ་རྣམས་ཀྱི་ཕྱག་རྒྱ་གསུངས་པ། རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་བཅིངས་ལ། །ཞེས་པ་དག་གོ། །རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་རྒྱབ་སྤྲད་ལ་མཐེའུ་ཆུང་དང་མཛུབ་མོ་སོ་སོར་སྦྲེལ། གཡོན་པའི་མཐིལ་གྱེན་དུ་བསྟན་ཏེ་བཟློག་པའོ། །
ས་པ་རྣམས་ཀྱི་ཕྱག་རྒྱ་བསྟན་པ། རྡོ་རྗེ་བཅིང་བ་བརྟན་བྱས་ནས། །ཞེས་པ་དག་གོ། །རྡོ་རྗེ་བཅིང་བ་བསྡམས་པ་ས་ལ་རེག་པར་བྱས་པ་ནི་ས་པའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་སྟེ་དགུག་པ་དང་གཞུག་པ་ལ་སོགས་པའི་ལས་ཐམས་ཅད་བྱེད་པ་སྟེ་ཕྱག་རྒྱ་འདི་ལ་སུའང་འགལ་བར་མི་བྱེད་ཅེས་པའི་དོན་ཏོ། །ས་འོག་མ་པ་
རྣམས་དགུག་པའི་ཕྱག་རྒྱ་བསྟན་པ། མཛུབ་མོ་གཉིས་ནི་བརྟན་པོ་རུ། །ཞེས་པ་དག་གོ། །རྡོ་རྗེ་བཅིངས་པ་ལས་མཛུབ་མོ་གཉིས་སྦྲེལ་ཏེ་བསྡམས་ལ་མཐེ་བོང་གཉིས་ཀྱིས་རང་རང་གི་གུང་མོའི་རྩེ་མོ་མནན་པའོ། །ཕྱག་རྒྱ་གོང་མ་རྣམས་རིམ་བཞིན་དུ་སྦྱར་བ་བསྟན་པའོ། །རྡོ་རྗེ་དཀྲི་བ་དང་ཞེས་པ་དག་གོ། །
