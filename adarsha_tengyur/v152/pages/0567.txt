མ་ཡིན་པས་བཀྲུ་བ་ལ་སོགས་པ་མ་བྱས་ན་ཉེས་བྱས་སུ་གསུངས་པའི་ཕྱིར་རོ། །སྦྱོར་བ་གཅིག་གིས་ལུག་བལ་རུང་བ་ཞྭ་རྒྱུ་ལ་སོགས་པའི་ཚད་ལས་ལྷག་པའི་ཆུན་པོ་གཅིག་གམ་གཉིས་སམ་མང་པོ་ཞིག་འཁྲུ་བ་དང་འཚེད་པ་དང་རྨེལ་བ་དང་བར་སྐབས་སུ་བཅད་ན་ཡང་
ལྟུང་བའི་དངོས་གཞི་གཅིག་རྣམ་པར་གཞག་གོ། །སྦྱོར་བའི་ལྟུང་བ་ནི་ཉེས་བྱས་ཏེ་ཇི་སྲིད་དུ་བལ་རྨེལ་དུ་འཇུག་པའི་སྤང་བའི་ལྟུང་བྱེད་ཀྱི་སྦྱོར་བ་བརྩམས་ནས་དངོས་གཞི་ལ་མ་ཐུག་གི་བར་དུ་ཉེས་བྱས་སོ། །དངོས་གཞི་ལ་སྦྱོར་བ་ཐ་དད་པ་ཇི་སྙེད་བརྩམས་པ་རེ་རེ་ལ་ཉེས་བྱས་རེ་རེའོ། །ཕྱི་རོལ་པའི་རབ་ཏུ་
བྱུང་བ་དང་མ་ཆགས་པ་དང་ཉམས་པ་ལ་ཉེས་བྱས་སོ། །བལ་རུང་བ་མ་ཡིན་པ་ལ་ཉེས་བྱས་སོ། །ཞྭ་རྒྱུ་ལ་སོགས་པའི་ཚད་དང་དེ་བས་ཆུང་བ་ལ་ཉེས་བྱས་སོ། །ལུག་བལ་མ་ཡིན་པར་ར་བ་ལ་ལ་སོགས་པ་ལ་ཉེས་བྱས་སོ། །ཚིག་གིས་མ་བསྒོ་བར་མངགས་པ་ལ་སོགས་པ་ནི་ཉེས་བྱས་
སོ། །དགེ་སློང་མས་མ་བྱས་པ་ལ་ཉེས་བྱས་སོ། །ཉེ་དུ་ལ་ཉེ་དུ་མ་ཡིན་པར་འདུ་ཤེས་པ་དང་ཉེ་དུ་ལ་ཡིད་གཉིས་ཟ་བ་ལ་ཉེས་བྱས་སོ། །རུང་བར་བྱས་པ་དང་། སྤངས་པ་དང་དགེ་འདུན་གྱི་ལ་ཉེས་བྱས་སོ། །ཉེ་དུ་དང་ཉེ་དུ་མ་ཡིན་པའི་མཚན་ཉིད་ནི་སྔར་འཁྲུར་འཇུག་པའི་སྤང་བའི་སྐབས་སུ་
བསྟན་པ་བཞིན་ནོ། །སྦྱོར་བའི་སྦྱོར་བའི་ལྟུང་བ་ནི་ཉེས་བྱས་ཕྲ་མོ་སྟེ། སྦྱོར་བ་ཇི་སྙེད་པ་ལ་སྦྱོར་བའི་སྦྱོར་བ་ཡང་དེ་སྙེད་དོ། །སྦྱོར་ཉེས་བྱས་ཀྱི་སྦྱོར་བའི་སྦྱོར་བ་ཡང་ཉེས་བྱས་ཏེ་ཉེས་བྱས་ཕྲ་མོའོ། །དེ་ཡང་དང་པོ་ཡིད་ལ་སེམས་པ་ནི་སྦྱོར་བའི་སྦྱོར་བའི་ཉེས་བྱས་ཕྲ་མོའོ། །དེ་ནས་ལུས་དང་
ངག་གིས་སྦྱོར་བ་ནི་སྦྱོར་བའི་ཉེས་བྱས་དངོས་སོ། །དེ་ནས་མཐར་ཐུག་པ་ནི་སྤང་བའི་ལྟུང་བྱེད་ཀྱི་དངོས་གཞིའོ་ཞེས་བྱ་བར་ཤེས་པར་བྱའོ། །ཁྱིམ་པའི་རྟགས་ཅན་མོ་ལ་ཉེས་བྱས་ཕྲ་མོའོ། །ལྟུང་བ་མེད་པ་ནི་དགེ་སློང་མ་ཉེ་དུ་ལ་ཉེ་ཉིད་དུ་འདུ་ཤེས་པ་ལ་ལྟུང་བ་མེད་དོ། །ལས་དང་པོ་པ་དང་། སྨྱོས་
པ་དང་། སེམས་འཁྲུགས་པ་དང་ཚོར་བ་དག་གིས་ཉེན་པ་ལ་ལྟུང་བ་མེད་དོ། །བསླབ་པར་བྱ་བ་ནི་དགེ་སློང་གིས་ཡུལ་ཐམས་ཅད་ལ་གནས་སྐབས་ཐམས་ཅད་དུ་བལ་རྨེལ་དུ་འཇུག་པའི་སྤང་བར་འགྱུར་བ་རྣམ་པ་ཐམས་ཅད་མི་བསྟེན་པར་བསླབ་པར་བྱའོ། །ཕྱིར་བཅོས་པ་ནི་སྔ་མ་བཞིན་ནོ། །
