ཨོཾ་སརྦྦ་ཡོ་ག་ཙིཏྟ་ཨུཏྤ་ད་ཡ་མི་ཞེས་བྱ་བ་འདིས། བྱང་ཆུབ་སེམས་ནི་བླ་མེད་པ། །དམ་པ་བདག་གིས་བསྐྱེད་ནས་སུ། །དེ་ཡི་སྙིང་གར་སྙིང་པོས་ནི། །རྡོ་རྗེ་རབ་ཏུ་གཞག་པར་བྱ། །སུ་ར་ཏེ་ས་མ་ཡ་སྟྭཾ་ཧོ་བཛྲ་སིདྡྷི་ཡ་ཐཱ་སུ་ཁཾ། དེ་ནས་དེ་འགྲོ་བ་འདུལ་བར་
བྱིན་གྱིས་བརླབས་ཏེ། འགྲོ་བ་འདུལ་བའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་དང་། རྡོ་རྗེ་པདྨ་ལ་སོགས་པའི་ཕྱག་རྒྱ་བཞིས་བྱིན་གྱིས་བརླབས་ལ། དེ་ནས་ཕྱི་རོལ་ན་གནས་པའི་བུམ་པའི་ཆུས་དབང་བསྐུར་ཏེ། དྲི་དང་མེ་ཏོག་ལ་སོགས་པ་རྣམས་ཀྱིས་མཆོད་ཡོན་གྱི་མཆོག་བླངས་ཏེ། རྡོ་རྗེ་
བརྟུལ་ཞུགས་ཀྱིས་དཀྱིལ་འཁོར་ཆེན་པོར་འཇུག་པར་བྱའོ། །དེ་ལ་དང་པོ་རེ་ཞིག་སློབ་དཔོན་གྱིས། ཨོཾ་བཛྲ་པདྨ་ས་མ་ཡ་སྟྭཾ་ཞེས་བརྗོད་ཅིང་རྡོ་རྗེ་པདྨའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་རང་ཉིད་བཅིངས་ལ་སློབ་མ་ཡང་འཆིང་དུ་གཞུག་པར་བྱའོ། །དེ་ཉིད་ཀྱི་གུང་མོའི་སོར་མོ་གཉིས་ཀྱིས་མེ་ཏོག་གི་ཕྲེང་
བ་འཛིན་དུ་བཅུག་ལ། ཨོཾ་པདྨ་ས་མ་ཡ་ཧཱུཾ་ཞེས་བྱ་བའི་སྙིང་པོ་འདི་དང་། རྟ་མགྲིན་གྱི་ལྕགས་ཀྱུའི་ལས་ཀྱི་ཕྱག་རྒྱས་ཤར་ཕྱོགས་ཀྱི་སྒོ་ནས་བཅུག་ལ་དགུག་པར་བྱའོ། །ལྷོ་ཕྱོགས་ཀྱི་སྒོ་ནས་དེ་བཞིན་དུ་བཅུག་ལ་དོན་ཡོད་ཞགས་པའི་ལས་ཀྱི་ཕྱག་རྒྱས་གཞུག་པར་བྱའོ། །ཡང་ནུབ་
ཕྱོགས་ཀྱི་སྒོར་བཅུག་ལ་པདྨའི་ལྕགས་སྒྲོག་གི་ལས་ཀྱི་ཕྱག་རྒྱས་བཅིང་བར་བྱའོ། །དེ་ནས་ཡང་བྱང་ཕྱོགས་ཀྱི་སྒོ་ནས་བཅུག་ལ་པདྨ་འབེབས་པའི་ལས་ཀྱི་ཕྱག་རྒྱས་དབང་དུ་བྱའོ། །ཡང་ཤར་ཕྱོགས་ཀྱི་སྒོ་ནས་དེ་བཞིན་དུ་བཅུག་ལ་འདི་སྐད་ཅེས་བརྗོད་པར་བྱ་སྟེ། དེང་ཁྱོད་དེ་བཞིན་
གཤེགས་པ་ཐམས་ཅད་ཀྱི་པདྨའི་རིགས་སུ་ཞུགས་ཀྱིས། ངས་ཁྱོད་ལ་ཡེ་ཤེས་གང་གིས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་གྲུབ་པ་ཡང་ཐོབ་ན། དངོས་གྲུབ་གཞན་ལྟ་སྨོས་ཀྱང་ཅི་དགོས་ཏེ། དེ་ལྟ་བུའི་ཡེ་ཤེས་བསྐྱེད་པར་བྱས། ཁྱོད་ཀྱིས་དཀྱིལ་འཁོར་ཆེན་པོ་མ་མཐོང་བ་རྣམས་
ལ་སྨྲ་བར་མ་བྱེད་ཅིག །ཁྱོད་དམ་ཚིག་ཉམས་པར་གྱུར་ཏ་རེ། དེ་ནས་སློབ་དཔོན་རང་ཉིད་ཀྱིས་རྡོ་རྗེ་པདྨའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་ཁ་ཐུར་དུ་བསྟན་པ་དང་། གྱེན་དུ་བསྟན་པ་བཅིངས་ལ་པདྨ་སློབ་མའི་སྤྱི་བོར་བཞག་ལ་འདི་སྐད་ཅེས་བརྗོད་པར་བྱའོ། །འདི་ནི་ཁྱོད་ཀྱི་རྡོ་རྗེ་དམ་ཚིག་ཡིན་
