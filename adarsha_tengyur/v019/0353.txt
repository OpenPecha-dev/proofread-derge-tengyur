ལས་བསྟན་པ་ཡིན་ནོ། །ཟླ་བ་གཉིས་ནི་ཚེག་དྲག་བཅས། །ཞེས་བྱ་བ་ལ། ཟླ་བ་གཉིས་ནི་ཐིག་ལེ་གཉིས་ཏེ་དེ་ཉིད་ཚེག་དྲག་གོ། །འདིས་ནི་དབྱངས་ཡིག་བཅུ་དྲུག་པ་བསྟན་ཏེ། འདིའི་ལས་ནི་འོག་ནས་བཤད་པར་བྱའོ། །འདི་ནི་ཞེས་བྱ་བ་ནི་དབྱངས་ཡིག་བཅོ་ལྔ་པ་སྟེ། ཡི་
གེའི་མཆོག་ཅེས་བྱ་བ་དང་སྦྱར་རོ། །ཅི་འདྲ་ཞེ་ན། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ།  སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཕྱག་རྒྱ་ལ་ཡང་དག་གནས་ཞེས་བྱ་བར་སྦྱར་ཏེ། སངས་རྒྱས་ཀུན་གྱི་ཕྱག་རྒྱ་ནི་བདུད་རྩི་འབབ་པའི་རྩ་ཨ་བ་དྷཱུ་ཏཱི་ཞེས་བྱའོ། །སངས་རྒྱས་རི་མོ་ཞེས་བྱ་
བ་ལ་རི་མོ་ནི་མེ་འབར་བའི་ཚུལ་གྱིས་སྟེང་དུ་འགྲོ་བ་སྟེ་ཨཾ་ངོ་། །སངས་རྒྱས་ནི་བདུད་རྩིའི་ཚུལ་གྱིས་འོག་ཏུ་འབར་བ་སྟེ་ཧཾ་ངོ་། །དེ་ཉིད་ཀྱི་ཕྱིར་གསུངས་པ། འདི་ལྟར་ཨཱ་ལི་ཀཱ་ལི་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། །ཨཱ་ལི་ཀཱ་ལི་ཞེས་བྱ་བ་ནི་སྤྲུལ་པའི་འཁོར་ལོར་གནས་པའི་ཡི་གེ་ཨཾ་གི་གཡས་
གཡོན་གྱི་རྩ་སྟེ།  ཐབས་དང་ཤེས་རབ་ཀྱི་རང་བཞིན་ཅན་ནོ། །བཅུད་ལེན་ཞེས་བྱ་བ་ལ་བཅུད་ནི་ཁྲག་ལ་སོགས་པའོ། །ལེན་པ་ནི་སྒོའོ། །གཙོ་བོ་ཞེས་བྱ་བ་ནི་ཨ་བ་དྷཱུ་ཏཱིའོ། །ཐིམ་པ་ནི་ཞུགས་པའོ། །ཅི་ཞིག་ཅེ་ན། རྡོ་རྗེ་སེམས་སོགས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། རྡོ་རྗེ་སེམས་
སོགས་ནི་ཡི་གེ་ཧཾ་ངོ་། །འོག་ཏུ་ནི་ཁ་ཐུར་དུ་བལྟས་པའོ། །ཡི་གེ་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་ཀྱི་ཆུ་རྒྱུན་ནོ། །ཡང་ན་ཨཱ་ལི་ཀཱ་ལི་ཞེས་བྱ་བ་ནི་ཡི་གེ་རྣམས་ཏེ། དེ་ཉིད་བཅུད་ཀྱི་གནས་དང་རྡོ་རྗེ་སེམས་དཔའ་ལ་སོགས་པའི་རང་བཞིན་ནོ། །གཙོ་བོ་ནི་སྒོ་ཞེས་བྱ་བར་སྦྱར་ཏེ། 
ཨཾ་ག་ཐིག་ལེའི་རི་མོའི་མེ་ལྕེའི་སྒོར་རོ། །སྦྱར་ཞེས་བྱ་བ་ནི་འོད་ཟེར་གྱི་འགྲོ་འོང་མཉམ་པར་སྦྱར་བའོ། །ཟླ་བ་གཉིས་དང་ཡང་དག་ལྡན། །ཞེས་བྱ་བ་ནི་ཟླ་ཚེས་དང་ཐིག་ལེ་གཉིས་དང་མཉམ་པར་སྦྱར་བའོ། །དབང་དུ་བྱེད་ཅེས་བྱ་བ་ནི། ཁམས་གསུམ་ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས་ཀྱི་དབང་
དུ་བྱེད་པའོ། །ཁྱབ་པ་ཉིད་དུ་བསྒོམ་པར་བྱ། །ཞེས་བྱ་བ་ནི། བསྒྲུབ་བྱའི་ལུས་ལ་བཀུག་ནས་བསྐོར་བའོ། །ད་ནི་དབྱངས་ཡིག་བཅུ་དྲུག་པའི་ལས་བསྟན་པའི་ཕྱིར། ཨ་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། ཨ་ཞེས་བྱ་བ་ནི་ཟླ་ཚེས་དང་ཐིག་ལེ་དང་བཅས་པའི་དབྱངས་ཡིག་བཅོ་
