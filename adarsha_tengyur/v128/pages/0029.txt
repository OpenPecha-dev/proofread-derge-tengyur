རྣམས་དང་བྲལ་བ་མ་ཡིན་ན་ཡང་། མི་ཁོམ་པ་རྣམས་སུ་སྐྱེ་བར་མི་འགྱུར་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་ཞུགས་པའི་གང་ཟག་ལ་དགོངས་ནས། གང་ལ་འཇིག་རྟེན་ཡང་དག་ལྟ། །ཆེན་པོ་ཡོད་པར་གྱུར་པ་དེ། །ཚེ་རབས་སྟོང་ཕྲག་རྣམས་སུ་ཡང་། །
ངན་འགྲོར་འགྲོ་བར་མི་འགྱུར་རོ། །ཞེས་གསུངས་པ་ལྟ་བུའོ། །དེ་ཡང་གང་གི་ཚེ་ཡོངས་སུ་སྨིན་པར་འགྱུར་བའི་དགེ་བའི་ཆོས་ཆེན་པོ་དག་ལ་རིམ་གྱིས་ཞུགས་པར་གྱུར་པ་དེའི་ཚེ། མི་ཁོམ་པ་རྣམས་སུ་སྐྱེ་བར་མི་འགྱུར་བ་ལ་བྱའི། །གཞན་དག་ལ་ཞུགས་པ་ནི་མ་ཡིན་ཏེ། དེ་
ནི་ཞུགས་པའི་གང་ཟག་གི་རྟགས་གཉིས་པ་ཡིན་ནོ། །གཞན་ཡང་ཞུགས་པའི་གང་ཟག་ནི་སངས་རྒྱས་སམ། ཆོས་སམ། དགེ་འདུན་གྱི་ཡོན་ཏན་དག་ཐོས་སམ། དྲན་ནས་དགེ་བ་ངེས་པར་འབྱུང་བ་དང་ལྡན་པའི་དད་པ་རྒྱ་ཆེན་པོའི་སེམས་ཐོབ་པར་འགྱུར་ཞིང་། དམིགས་པ་དེས་
ཡང་དང་ཡང་དུ་དད་པས་བརླན་པའི་སེམས་ཀྱིས་མཆི་མ་ཁྲུག་ཅེས་བྱེད་པ་དང་། སྤུ་ཟིང་ཞེས་བྱེད་པ་འཐོབ་པར་འགྱུར་ཏེ། དེ་ནི་ཞུགས་པའི་གང་ཟག་གི་རྟགས་གསུམ་པ་ཡིན་ནོ། །གཞན་ཡང་ཞུགས་པའི་གང་ཟག་ནི། འདི་ལྟ་སྟེ། ཁ་ན་མ་ཐོ་བ་ཀུན་འབྱུང་བའི་གནས་ཐམས་ཅད་
ལ། རང་བཞིན་གྱིས་ངོ་ཚ་ཤེས་པ་དང་། ཁྲེལ་ཡོད་པ་ཤས་ཆེན་པོ་དང་ལྡན་པ་ཡིན་ཏེ། དེ་ནི་ཞུགས་པའི་གང་ཟག་གི་རྟགས་བཞི་པ་ཡིན་ནོ། །གཞན་ཡང་ཞུགས་པའི་གང་ཟག་ནི། ལུང་ནོད་པ་དང་། ཁ་ཏོན་བྱ་བ་དང་། ཡོངས་སུ་ཞུ་བ་དང་། རྣལ་འབྱོར་དུ་བྱ་བ་དང་། 
ཡིད་ལ་བྱ་བ་དག་ལ་འདུན་ཅིང་འདུན་པ་དྲག་པོ་དང་ལྡན་ལ། དགེ་བ་གང་ཡིན་ཞེས་ཚོལ་བར་བྱེད་དེ། དེ་ནི་ཞུགས་པའི་གང་ཟག་གི་རྟགས་ལྔ་པ་ཡིན་ནོ། །གཞན་ཡང་ཞུགས་པའི་གང་ཟག་ནི་འདི་ལྟ་སྟེ། ཡང་དག་པར་བསྒྲུབ་པར་བྱ་བའི་ཕྱིར། ལས་ཀྱི་མཐའ་ཁ་ན་མ་ཐོ་བ་མེད་པ་
ཐམས་ཅད་དང་། དགེ་བའི་ཕྱོགས་ཀྱི་སྦྱོར་བ་ཡང་དག་པར་བླངས་པ་ཐམས་ཅད་ལ། དམ་དུ་རྩོམ་པ་དང་། བརྟན་པར་རྩོམ་པ་དང་། ཤིན་ཏུ་ངེས་པར་རྩོམ་པ་ཡིན་ཏེ། དེ་ནི་ཞུགས་པའི་གང་ཟག་གི་རྟགས་དྲུག་པ་ཡིན་ནོ། །གཞན་ཡང་ཞུགས་པའི་གང་ཟག་ནི་རང་བཞིན་གྱིས་འདོད་
