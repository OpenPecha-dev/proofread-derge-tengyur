བྱིན་གྱིས་བརླབ་པའོ། །དེས་རིམ་གྱིས་རྟོགས་པའི་ཕྱིར་རང་བྱིན་གྱིས་བརླབ་པའི་རིམ་པ་སྟེ། དེས་ན་རང་གི་བདག་ཉིད་ལ་བྱ་བ་འགལ་བའི་ཕྱིར་ཐམས་ཅད་དམིགས་པ་མེད་པས་ཞེས་དགོངས་པའོ། །འདིས་ནི་དཀྱིལ་འཁོར་འཁོར་ལོ་ཐབས་དང་པོས་བསྐྱེད་པར་བྱ་སྟེ། རྡོ་རྗེ་པདྨ་མཉམ་
པར་སྦྱོར་བས་ལྟེ་བའི་དབུས་སུ་སོན་པ་མི་སྤང་བའོ། །རང་བྱིན་གྱིས་བརླབ་པའི་རིམ་པས་ཀྱང་སྤང་བ་སྟེ། ཁྱབ་པའི་ངོ་བོ་ལྷན་ཅིག་སྐྱེས་པའི་ཡེ་ཤེས་ཀྱི་རང་བཞིན་ཞེས་པའི་དོན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར། སྒྲིབ་མེད་སྒྲིབ་པའི་རང་བཞིན་ཅན། །སྒྲིབ་མེད་ནི་སྒྲིབ་པ་དང་བྲལ་བ་སྟེ་དོན་དམ་པའི་བདེན་
པའོ། །དེ་ཉིད་ཀུན་རྫོབ་སྟེ། སྒྲིབ་པ་ནི་སྒྲིབ་པ་དང་ལྡན་པ་ཀུན་རྫོབ་ཀྱི་བདེན་པའོ། །ཨེ་ཝཾ་གྱི་གཟུགས་ཀྱི་རང་བཞིན་གྱི་ཡེ་ཤེས་བདེན་པ་གཉིས་དབྱེར་མེད་པ་ཉིད་དེ། མི་ཟད་པའི་རང་བཞིན་མི་ཟད་པ་ཡང་དག་པའི་ཡེ་ཤེས་བཞི་པ་ལྷན་ཅིག་སྐྱེས་པ་ཞེས་པའི་དོན་ནོ། །གཉིས་ཀའི་རང་
བཞིན་ཉིད་གསུངས་པ། ཀུན་རྫོབ་ཀུནྡ་ལྟ་བུ་སྟེ། །དོན་དམ་བདེ་བའི་གཟུགས་ཅན་ནོ། །བུད་མེད་ཀཀྐོ་ལ་བདེ་བ་ཅན། །ཨེ་ཝཾ་ཡི་གེའི་རང་བཞིན་ནོ། །ཀུནྡ་ལྟ་བུ་ཞེས་པ་བདུད་རྩི་ཁུ་བའི་རང་བཞིན་སྤངས་པ་མ་ཡིན་པ་ཀུན་རྫོབ་བོ། །བདེ་བའི་གཟུགས་ཅན་ཞེས་པ་ལྷན་ཅིག་
སྐྱེས་པའི་དགའ་བའི་རང་བཞིན་སྤངས་པའི་དོན་དམ་པའོ། །འདི་དག་ཀྱང་བུད་མེད་ཀཀྐོ་ལ་ཞེས་པ་བུད་མེད་ཀྱི་གཟུགས་ཅན་ཀཀྐོ་ལ་སྟེ། བུད་མེད་ཀྱི་གཟུགས་དང་ཀཀྐོ་ལ་ཆོས་འབྱུང་དེ་ལ། བུད་མེད་ཀྱི་གཟུགས་ཀཀྐོ་ལ་དེ་ཉིད་ཇི་ལྟ་བུ་ཞེ་ན། གསུངས་པ། ཨེ་ཝཾ་ཡི་གེའི་རང་
བཞིན་ནོ། །དེ་ལ་གང་གིས་ཆོས་འབྱུང་དེ་ཉིད་ཤེས་རབ་དང་ཐབས་དབྱེར་མེད་པ་ཞེས་པ་སྟེ། དེ་ཉིད་ཨེ་ཝཾ་ཡི་གེའི་རང་བཞིན་བདེ་བ་ཅན་གཞིར་གྱུར་པའོ། །འདི་ཇི་ལྟར་ཞེ་ན། གསུངས་པ། བདེ་བ་བསྲུང་བ་ཉིད་ཀྱི་ཕྱིར། །བདེ་བ་ཅན་ཞེས་བསྒྲགས་པ་སྟེ། །སུ་ཁུ་བ་རྡོ་རྗེ་འཆང་དང་ཁཾ་
ཆོས་འབྱུང་བདག་མེད་མ་སྟེ། རྡོ་རྗེ་འཆང་དང་དབྱེར་མེད་པའི་ཡེ་ཤེས་ནི་བདེ་བའོ། །དེ་བསྲུང་བ་ཉིད་ཀྱི་ཕྱིར་གཡོ་བ་མེད་པའི་རང་བཞིན་གྱི་རྒྱུ་ཉིད་ཀྱི་ཕྱིར་རོ། །བདེ་བ་ཅན་ཞེས་བསྒྲགས་པ་དེ། །ཞེས་པ་སྒྲའི་ཡུལ་དུ་བྱས་པ་དང་། ཆ་དང་བཅས་པ་ཉིད་ཆ་དང་བྲལ་ཞེས་པའི་དོན་ནོ། །
