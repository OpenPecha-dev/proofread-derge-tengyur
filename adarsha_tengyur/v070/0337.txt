གཉིས་སོ། །དེ་དག་ཉིད་རྒྱལ་བ་སྟེ། གནོད་པ་ལ་སོགས་པས་བརྫི་བར་མི་ནུས་པའོ། །རབ་ཏུ་གསང་བའི་སྔགས་འདི་ནི། །ཞེས་པ་ནི། བཛྲ་ཀྲོ་དྷ་བི་ཛ་ཡ། བཛྲ་ཀྲོ་དྷ་ནཱི་ལ་དཎྜཱ་ཡ། བཛྲ་ཀྲོ་དྷ་ཡ་མཱནྟ་ཀ་ཧཱུཾ། བཛྲ་ཀྲོ་དྷ་ཨ་ཙ་ལ་ཨརྱ་བནྡྷ་བནྡྷ། བཛྲ་ཀྲོ་དྷ་ཧྱ་གྲྀ་བ། བཛྲ་ཀྲོ་དྷ་ཨ་པ་ར། 
ཙྪིནྡ་ཙྪིནྡ། བཛྲ་ཀྲོ་དྷ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི། བཛྲ་ཀྲོ་དྷ་ཏྲེེ་ལོ་ཀྱ་བི་ཇ་ཡ། སུཾ་བྷ་ནི་སུཾ་བྷ། ཨུནྡྷི་ཥ་ཙཀྲ་བརྟི། ཨ་མྲྀ་ཏཱ་ལཱི་ཡ། བཛྲ་ཀྲོ་དྷ་པ་ཏྲ་ཡ་པ་ཏཱ་ཡ། བཛྲ་ཙཎྜ་བཛྲ། བཛྲ་ན་ལཱརྐ་བཛྲ། བཛྲོ་ཥྞཱི་ཥཱ་བཛྲ། བཛྲ་ཀུཎྜ་ལི་བཛྲ། བཛྲ་བི་ཤ་ན་བཛྲ། བཛྲ་མ་ཧཱ་པཱ་ལཱ་བཛྲ། བཛྲ་ཀ་
