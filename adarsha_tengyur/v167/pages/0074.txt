ལ་རྣམ་པ་བཞི་སྟེ། ཆུའི་རྒྱུན་དུ་རྐུ་བ་དང་། རྒྱུན་ལས་གྱེན་དུ་རྐུ་བ་དང་། ཐད་ཀར་རྐུ་བ་དང་། བཅིངས་པའི་རྒྱུན་དུ་རྐུ་བའོ། །ཐད་ཀར་རྐུ་བ་ནི་མི་སྣང་བའི་ཚད་ཀྱིས་བཟུང་ངོ་། །གྱེན་དུ་རྐུ་བ་ནི་གྲུ་གང་འཕགས་པས་ཕམ་གྱི། མ་ངེས་པ་ནི་ཆུའི་ནང་དུ་ནུབ་པས་ཕམ་མོ། །ཤིང་ཞེས་པ་ནི་
ལོ་མ་ལ་སོགས་པ་བཟང་པོ་སྐྱེ་བའི་ཤིང་དང་། ཡང་ན་ཤིང་དག་གོ། །ཤོ་གམ་དང་ཞེས་པ་ནི་བདག་གི་ཤོ་གམ་འདས་ན་རིན་ཐང་བརྩི། གཞན་གྱི་བདག་གིས་བཟླས་ཏེ་བླངས་ན་ཡང་རིན་ཐང་བརྩི་བདག་ཉིད་ཀྱི་ནོར་མ་ཡིན་ཏེ། དཀོན་མཆོག་དང་ཕ་མ་ལ་སོགས་པ་ནི་ཉེས་བྱས་སུ་འགྱུར་
རོ། །དེ་བཞིན་གྲུ་ཡི་བཙས་རྣམས་དང་། །ཞེས་པ་ནི་གོང་མའི་ཤོ་གམ་བཞིན་དུ་སྒྲོལ་གྲུ་འདས་པའམ་དབྱར་དབབ་པའི་ཕྱིར་གྲུ་ལ་བདག་གྱེན་དུའམ་ཆུ་འབྱམ་དང་ཐད་ཀར་ཡང་རུང་སྟེ། ཞོན་ལ་མ་བྱིན་པ་ལ་རིན་ཐང་བརྩིའོ། །སྲོག་ཆགས་རྐང་པ་མེད་པ་དང་། །ཞེས་པ་ནི་དངོས་སུ་སྦྲུལ་
ལྟ་བུ་སྟེ། དེ་ཡང་རྒྱལ་པོའི་གཉེར་བ་ལ་ནན་ཏུར་དག་བྱ་བའི་དགོས་པ་དང་། འཚེ་བའི་ཕྱིར་ཏེ་ནེའུ་ལེ་དང་སྦྲུལ་འཐབ་པ་དག་མཛའ་བར་བྱས་ནས་འཚོ་བ་ཚོལ་ཏེ་དེ་ལྟ་བུ་བརྐུས་ན་རིན་ཐང་བརྩིའོ། །རྐང་གཉིས་ཞེས་པ་ནི་མི་དང་བྱ་སྟེ། དེ་ལ་བྱ་ནི་ཁྲ་འམ་ལོངས་སྤྱོད་ཀྱི་ཕྱིར་ཁྲུང་ཁྲུང་དང་ངང་པ་ལ་
སོགས་པའི་གསོན་མ་དག་གོ། །མི་ནི་མཚན་མ་དང་། དུས་དང་གནས་སུ་ཕྱིན་པས་བསམ་མོ། །བྱ་ནི་རྐུ་བའི་ཕྱིར་སྒྲོགས་པའམ་གནས་ས་ནས་བདག་གི་གནས་སུ་སྤུར་བས་ཕམ། རྐང་མངས་རྐུ་བ་དང་ཞེས་པ་ནི་རྐང་པ་བཞི་པ་ཡན་ཆད་ལ་བྱ་སྟེ། སྦྲང་སེར་དང་སྲིན་བུ་པད་པ་དང་སྡིག་པ་དག་
གོ། །དེ་ལ་སྦྲང་སེར་ནི་རྒྱ་མཚོར་དགོས་སོ། །སྲིན་བུ་ནི་སྨན་པ་ལ་དགོས་སོ། །སྡིག་པ་ནི་རྒྱལ་པོའི་ནན་ཏུར་དུ་དགོས་སོ། །བརྟུལ་ཞུགས་ཅན་གྱིས་ཞེས་པ་ནི་བསླབ་ཚིགས་བཅུ་ལ་ཞུགས་པའོ། །མིའི་ནོར་ཞེས་པ་ནི་མདོར་བསྡུས་ཏེ་མིའི་ནོར་དུ་གཏོགས་སོ་ཅོག་གིས་སྡུམ་པ་བཞི་པར་
འགྱུར་རོ། །དབྱིག་ལས་ཞེས་པ་ནི་རིན་པོ་ཆེ་ལ་སོགས་པའི་ནོར་ཆེན་པོའོ། །སྩོགས་པ་ཞེས་པ་ནི་གསེར་དང་དངུལ་དང་གཡུ་དང་མུ་ཏིག་དང་དར་ཟབ་ལ་སོགས་པ་ཆུང་ངུ་ཡན་ཆད་ཀྱང་བསྡུའོ། །གང་ཡང་རུང་བ་ཞེས་པ་ནི་མདོར་བསྡུས་པའི་ཚིག་གོ། །རྐུ་བའི་སེམས་ཀྱིས་གང་བརྐུས་
