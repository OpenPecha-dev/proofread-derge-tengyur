པ་ལ། །ཡི་གེ་མ་གཉིས་དབུས་སུ་དགོད། །དེ་ནས་རིམ་པ་གསུམ་པ་ལ། །ཡི་གེ་ར་གཉིས་བསྒྲིལ་ཏེ་དགོད། །དེས་ནི་འཁོར་ལོ་འདྲ་བར་འགྱུར། །དེ་བཞིན་ནུབ་ཀྱི་ཕྱོགས་དག་ཏུ། །དང་པོར་ཨ་གཉིས་བསྐྱེད་བྱས་ཏེ། །གཉིས་པར་མེ་གཉིས་དབུས་སུ་བྲི། །གསུམ་
པར་དོ་གཉིས་བསྒྲིལ་བར་བྱ། །དེ་ནས་བྱང་གི་ཕྱོགས་དག་ཏུ། །དང་པོར་ཡ་སྟེ་གཉིས་པར་ད། །སྔ་མ་བཞིན་དུ་བྲི་བར་བྱ། །གསུམ་པར་ཡོ་ཞེས་བྱ་བ་དང་། །སྔ་མ་བཞིན་དུ་བྲི་བར་བྱ། །ཤར་ཕྱོགས་སུ་ཡང་དེ་བཞིན་དུ། །དང་པོར་ཡ་སྟེ་གཉིས་པར་ཀྵེ། །གསུམ་པར་ཡ་ཞེས་བྱ་བ་ཡང་། །སྔ་མ་བཞིན་དུ་བྲི་བར་བྱ། །ལྷོ་
ནུབ་མཚམས་ཀྱི་འཁོར་ལོའི་རྩེར། །ཡི་གེ་ཛ་ས་བསྒྲིལ་ཏེ་དགོད། །དེ་བཞིན་ནུབ་བྱང་རྩེ་མོ་ལ། །ཡི་གེ་རུ་ན་དགོད་པར་བྱ། །བྱང་ཤར་འཁོར་ལོའི་རྩེ་མོ་ལ། །ཡི་གེ་ནི་རཱ་དེ་བཞིན་དགོད། །དེ་བཞིན་ཤར་ལྷོའི་རྩེ་མོ་རུ། །ཡི་གེ་ཙྪ་ནི་བྲི་བར་བྱ། །ལྷོ་ནུབ་མཚམས་ཀྱི་ནང་གི་གྲྭར། །ལྟེ་བའི་ཆ་ལ་
བརྟེན་ནས་ནི། །ཡི་གེ་ཡ་གཉིས་བར་དག་ཏུ། །འཁོར་བར་བྱེད་པའི་བསྐུལ་ཚིག་དག །ཕྱིར་ལ་འཁྱིལ་བ་ལྟ་བུར་བྲི། །ནུབ་བྱང་མཚམས་སུའང་དེ་བཞིན་དུ། །ཡི་གེ་ཡ་གཉིས་བར་དག་ཏུ། །ཁྲག་འབྱུང་བ་ཡི་བསྐུལ་ཚིག་དག །ཕྱིར་ལ་འཁྱིལ་བ་ལྟ་བུར་བྲི། །བྱང་ཤར་དུ་ཡང་དེ་བཞིན་དུ། །
བརྒྱལ་བར་བྱེད་པའི་བསྐུལ་ཚིག་དག །ཕྱིར་ལ་འཁྱིལ་བ་ལྟ་བུར་བྲི། །ཤར་ལྷོ་མཚམས་སུའང་དེ་བཞིན་དུ། །གསོད་པར་བྱེད་པའི་བསྐུལ་ཚིག་བྲི། །ལྷོ་སྟོད་ནས་ནི་ནུབ་སྨད་དུ། །ཡི་གེ་འཁོར་བའི་ལམ་རྣམས་ལ། །ན་བུན་ཁ་དོག་འདྲ་བ་ཡི། །ལམ་གྱི་ཁ་དོག་བྲི་བར་བྱ། །དབུས་ནས་ཕྱིར་ལ་
བསྐོར་ནས་ནི། །ནུབ་སྟོད་ནས་ནི་བྱང་སྨད་དུ། །ཁྲག་དང་འདྲ་བའི་ལམ་བྲིའོ། །བྱང་སྟོད་ནས་ནི་ཤར་སྨད་དུ། །ཡི་གེ་འཁོར་བའི་ལམ་རྣམས་ལ། །ཁ་དོག་སེར་པོ་སྤྲིན་འདྲ་བྲི། །ཤར་སྟོད་ནས་ནི་ལྷོ་སྨད་དུ། །ཡི་གེ་འཁོར་བའི་ལམ་རྣམས་ལ། །དུར་ཁྲོད་སོལ་བས་ནག་པོར་བྲི། །དབྱངས་
དང་མིང་གི་ཡི་གེ་དག །འཁོར་ལོའི་ཚུལ་དུ་ཕྱི་རོལ་དུ། །བྲིས་ན་ཡང་ནི་རུང་བར་བཤད། །མ་བྲིས་ན་ཡང་ཉེས་པ་མེད། །གཞི་ནི་དུར་ཁྲོད་རས་ལ་བྱ། །དུག་དང་ཁྲག་དང་སྐེ་ཚེ་དང་། །གསེར་ཤིང་ཞེས་བྱའི་ཁུ་བ་དང་། །དུར་ཁྲོད་སོལ་བ་དག་གིས་ཀྱང་། །རས་ལ་བསྒོ་ཞིང་སྣག་ཚ་བྱ། །
