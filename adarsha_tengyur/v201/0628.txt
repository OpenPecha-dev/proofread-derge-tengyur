སུག་སྨེལ་དང་ཆང་དུ་སྦྱར་བ་བླུད་པའམ། སུག་སྨེལ་དང་སྐྱུ་རུ་རའི་ཁུ་བར་སྦྱར་བ་བླུད་པར་བྱའོ། །བཞད་རུས་གཟེ་མ་སུག་སྨེལ་དང་། །ཚ་གསུམ་སྦྲང་རྩི་གཅིན་བཅས་བླུད། །ཅེས་བྱ་བ་ལ། ཡང་ན་བྱ་བཞད་ཀྱི་རུས་པ་དང་གཟེ་མ་ལ་སོགས་པའི་སྨན་སྦྲང་རྩི་དང་གཅིན་སྦྱར་བ་བླུད་ན་
ཡང་ཕན་ནོ། །ཡང་ན་ཀཎྜ་ཀ་རིའི་ཁུ། །སྦྲང་རྩིར་སྦྱར་བ་བླུད་པར་བྱ། །ཞེས་བྱ་བ་ལ། ཀཎྜ་ཀ་རི་བསྐོལ་བའི་ཁུ་བ་སྦྲང་རྩི་དང་སྦྱར་བ་བླུད་ན་ཡང་གཅིན་འགགས་པའི་ནད་སེལ་ཏོ། །ཤི་ཏི་བ་རའི་ས་བོན་ནི། །ཞིབ་བཏགས་དར་བར་སྦྱར་བ་བླུད། །ཅེས་བྱ་བ་ལ། ཡང་ན་ཤི་ཏི་བར་ཞེས་
བྱ་བའི་ས་བོན་གྱི་ཕྱེ་མ་དང་དར་བ་བླུད་ན་ཕན་ནོ། །བ་ད་སཔྟ་པརྞི་དང་། །སླེ་ཏྲེས་དོང་གྲ་དུག་མོ་ཡུངས། །ཀེ་པུ་ཀ་དང་སུག་སྨེལ་དང་། །ཀ་རཉྫ་བསྐོལ་སྦྲང་རྩིར་བཅས། །དེས་སྦྱར་ཐུག་པའམ་བྱུ་རུ་ཡི། །ཕྱེ་མ་འབྲས་ཀྱི་ཆུ་དང་བླུད། །ཅེས་བྱ་བ་ལ། ཡང་ཤིང་བ་ད་ལ་སོགས་
པ་བསྐོལ་བའི་ཁུ་བ་སྦྲང་རྩི་དང་བཅས་པ་བླུད་པའམ། བ་ད་ལ་སོགས་པའི་སྨན་དེ་དག་དང་སྦྱར་བའི་ཐུག་བའམ་བྱུ་རུ་བཏགས་པའི་ཕྱེ་མ་འབྲས་ཀྱི་ཕྱེ་དང་སྦྱར་ཏེ་འཐུངས་ན་ཡང་ཕན་ནོ། །པ་ཐ་ལི་ཡི་ཐལ་བའི་ཆུ། །ལན་བདུན་བཙགས་ཏེ་ཏིལ་མར་བཅས། ཞེས་བྱ་བ་ལ། ཡང་ན་ཤིང་
པ་ཏ་ལི་བསྲེགས་པའི་ཐལ་བའི་ཆུ་ལན་བདུན་གྱི་བར་དུ་བཙགས་ཤིང་གཟགས་པར་བྱས་པ་ཏིལ་མར་དང་བཅས་པ་བླུད་ན་ཕན་ནོ། །ནས་ཚིག་འཁུས་ཐལ་པ་ཐ་ལི། །པ་རི་པ་ཏྲ་ཏིལ་ཐལ་ཆུ། །ཤིང་ཚ་སུག་སྨེལ་ན་ལེ་ཤམ། །ཆང་དང་སྦྱར་བ་བླུད་པར་བྱ། །ཞེས་བྱ་བ་ལ། ནས་ཚིག་འཁུས་
པའི་ཐལ་ཆུ་དང་། པ་ཐ་ལིའི་ཐལ་ཆུ་དང་། པ་རི་པ་ཏྲའི་ཐལ་ཆུ་དང་ཏིལ་བསྲེགས་པའི་ཐལ་ཆུའི་ནང་དུ་ཤིང་ཚ་ལ་སོགས་པའི་ཕྱེ་མ་དང་ཆང་དུ་སྦྱར་བ་བླུད་ན་ཡང་ཕན་ནོ། །འདི་དག་རེ་རེ་ལྡེ་གུར་སྦྱར། །བུ་རམ་སོ་ཡིས་བཅད་ཅིང་བཟའ། །ཞེས་བྱ་བ་ལ། ནས་ཚིག་འཁུས་པའི་ཐལ་བ་ལ་
སོགས་པ་འདི་དག་ལྡེ་གུར་སྦྱར་བ་དང་། བུ་རམ་སོས་བཅད་ཅིང་ཟོས་ན་ཡང་ཕན་ནོ། །འདུས་པའི་བདག་ཉིད་ཅན་ཀུན་ལ། །འདི་དག་གནས་སྐབས་ཇི་བཞིན་ཕན། །ཞེས་བྱ་བ་ལ། འདུས་པ་ལས་གྱུར་པའི་གཅིན་སྲི་བ་ལ་ནི་འདི་དག་གནས་སྐབས་ཇི་ལྟ་བ་བཞིན། གང་དང་གང་ཤས་ཆེ་བའི་
