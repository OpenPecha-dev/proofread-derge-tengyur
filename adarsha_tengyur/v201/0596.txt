བའི་ཁུ་བ་བསྒྲངས་པའམ། །ཞེས་བྱ་བ་ལ། ཤལ་མ་ལིའི་འདབ་མ་དང་རྩ་བ་བསྐོལ་བའི་ཁུ་བ་བསྒྲངས་པ་ལ་སྦྲང་རྩི་བཏབ་པ་ཡང་དེ་དང་འདྲའོ། །རྩ་མཁྲིས་གླ་སྒང་དུག་མོ་ཡུངས། །སྐྱེར་ཁཎྜ་དང་བཅས་པའམ། །ཡང་ན་སྐྱེར་པ་བ་ལེ་ཀ །བིལ་བའི་རྐང་དང་བྱི་ཚེར་བ། །ཏིལ་དང་སེང་
ཕྲོམ་མོན་ཆ་ར། །བཙོད་དང་པདྨ་ཨུཏྤ་ལ། །དོང་གྲ་ད་ཏ་ཀའི་མེ་ཏོག །སེའུ་འབྲུ་ཨུཏྤལ་ཤུན་པར་བཅས། །ཚིགས་བཅད་ཕྱེད་ཀྱིས་བཤད་སྦྱོར་བ། །སྦྲང་རྩིར་བཅས་པ་འབྲས་ཆུ་བླུད། །ཅེས་བྱ་བ་ལ། རྩ་མཁྲིས་ལ་སོགས་པ་ཚིགས་སུ་བཅད་པ་ཕྱེད་ཕྱེད་ཀྱིས་བསྟན་པའི་སྦྱོར་
བ་བཞི་པོ་ལ་སྦྲང་རྩི་བཏབ་པ་འབྲས་བཀྲུས་པའི་ཆུ་བླུད་ན་ཡང་མཁྲིས་པ་ལས་གྱུར་པའི་འཁྲུ་བ་སེལ་བར་བྱེད་དོ། །ས་མང་གི་ནི་ནམ་སྒ་རིའོ། །མོ་ཛ་ར་ས་ནི་མོན་ཛ་ར་ཛ་སྟེ་ཤལ་མ་ལིའི་ཐང་ཆུའོ། །སྐྱེར་པ་སེང་ཕྲོམ་དུག་མོ་ཡུངས། །སུག་སྨེལ་བསྐོལ་བས་སྨིན་འཁྲུ་སེལ། །ཞེས་བྱ་བ་ལ། སྐྱེར་བ་ལ་
སོགས་པ་བསྐོལ་བའི་ཁུ་བས་ཡང་སྨིན་པའི་འཁྲུ་བ་སེལ་ཏོ། །སེང་ཕྲོམ་དང་ནི་ཨ་བརྟ། །པྲི་ཡང་ཀུ་སོགས་སྡེ་ཚན་དང་། །བསྐོལ་བ་སོ་སོར་བླུད་དེ་བཞིན། །ཞེས་བྱ་བ་ལ། སེང་ཕྲོམ་ལ་སོགས་པའི་སྡེ་ཚན་དང་། ཨ་བརྟ་ལ་སོགས་པའི་སྡེ་ཚན་དང་། པྲི་ཡང་ཀུ་ལ་སོགས་པའི་སྡེ་ཚན་དང་། སྐྱུག་སྨན་
ལ་སོགས་པའི་སྨན་སྡེ་ཚན་བསྡུས་པའི་སྐབས་ནས་བཤད་པ་བསྐོལ་ཏེ་བླུད་པ་ཡང་སྔ་མ་བཞིན་དུ་སྨིན་པའི་འཁྲུ་བ་སེལ་བ་ཡིན་ནོ། །པུ་ཙེ་ཤེལ་དང་ཤིང་མངར་དང་། །སེའུ་འབྲུའི་མྱུ་གུ་པྲི་ཡང་ཀུ། །ཞོ་དང་འུ་སུའི་འབྲས་བུར་བཅས། །ཐུག་པ་ལྡེ་གུས་སྦྱར་དུགས་བྱ། །ཞེས་བྱ་བ་ལ། པུ་ཙེ་
ཤེལ་ལ་སོགས་པ་ལས་ཐུག་པ་ལ་སོགས་པ་བྱའོ། །སེའུ་འབྲུའི་མྱུ་གུ་དང་པོར་སྐྱེས་མ་ཐག་པའོ། །དེ་བཞིན་བིལ་བ་ཀ་པིད་ཐ། །ཨ་མྲ་འདམ་བུའི་རྐང་ལས་སྦྱར། །ཞེས་བྱ་བ་ལ། དེ་བཞིན་དུ་ཐུག་པ་དང་ལྡེ་གུ་དང་སྦྱར་ཐུག་ཁ་ལ་ཞེས་བྱ་བ་ལ་སོགས་པ་བིལ་བའི་རྐང་དང་ཀ་པིད་
ཐའི་རྐང་དང་ཨ་མྲའི་རྐང་དང་འདམ་བུའི་རྐང་དང་སྦྱར་བར་བྱའོ། །སྨིན་པའི་ཁུ་བ་རའི་འོ་མ། །མཆོག་ཏུ་ཕན་ཏེ་དེས་ཀྱང་ནི། །སྟོབས་ལྡན་ནད་གཞི་ཆེ་བའི་ཕྱིར། །མ་ཞི་ན་ནི་བཀྲུ་སྨན་གཏང་། །ཞེས་བྱ་བ་ལ། སྨིན་པའི་མཁྲིས་པའི་འཁྲུ་བ་ལ་ནི་རའི་འོ་མ་མཆོག་ཏུ་ཕན་པ་ཡིན་
