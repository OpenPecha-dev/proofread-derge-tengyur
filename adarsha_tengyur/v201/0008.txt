བ་བཅས་པར་ཁོལ། །ཆུ་བུར་སྦོམ་སྐ་ལྡེ་གུ་འདྲ། །དེ་ཚེ་ཕབ་ལ་གྲངས་པ་དང་། །ལྕགས་སྣོད་བཅུག་སྟེ་ནས་ནང་གཞུག །འདི་ནི་ཐལ་སྨན་འབྲིང་པོ་ཡིན། །ཞེས་བྱ་བ་ལ། དུས་གང་གི་ཚེ་ཐལ་སྨན་དེ་ལྦུ་བ་དང་བཅས་པར་ཁོལ་ཞིང་ཆུ་བུར་དང་བཅས་པ་དང་། སླ་བག་
ལས་སྐ་ཞིང་ལྡེ་གུ་འདྲ་བར་གྱུར་ལ། གཟར་བུ་ལ་ཆགས་པ་དེའི་ཚེ་ཐབ་ལས་ཕབ་ལ། གྲངས་པ་དང་། ལྕགས་ཀྱི་སྣོད་དུ་བཅུག་སྟེ། ནས་ཀྱི་ནང་དུ་ལེགས་པར་སྦ་ཞིང་གཞུག་པར་བྱ་སྟེ། འདི་ནི་ཐལ་སྨན་འབྲིང་པོ་ཞེས་བྱའོ། །འཇམ་པོ་མ་བཏགས་པར་བཅུག་ལ། །
གསུར་ལ་སླར་ནི་གདོན་པར་བྱ། །ཞེས་བྱ་བ་ལ། ཐལ་སྨན་འཇམ་པོ་ནི་རྡོ་ཐལ་གྱི་ཐལ་བ་དང་། རྡེའུ་ལ་སོགས་པ་མ་བཏགས་པར་བསྲེགས་པ་ལ་ཐལ་ཆུར་དེ་བཅུག་ཅིང་གསུར་ལ། དེའི་འོག་ཏུ་སླར་ཕྱིར་གདོན་པར་བྱའོ། །རྣོན་པོར་སྔ་མ་བཞིན་བཅུག་ལ། །ལངྐ་ལིངྐ་
དནྟི་དང་། །ཤུ་དག་བོང་ང་ཙི་ཏྲ་ཀ །སྭརྩ་ཀ་དང་ཁྲོན་བུ་དང་། །ཤིང་ཀུན་ཀ་རཉྫའི་ལོ་མ། །ཏ་ལའི་ལོ་མ་ཁ་རུ་ཚའི། །ཕྱེ་བཅུག །ཅེས་བྱ་བ་ལ། ཐལ་སྨན་རྣོན་པོ་སྦྱར་བ་ལ་ནི་སྔར་བཤད་པའི་ཐལ་སྨན་འབྲིང་པོ་སྦྱར་བ་བཞིན་དུ་ལྡེ་གུའི་སྦྱོར་བ་བཅུག་པ་དང་། དེ་བཞིན་
དུ་ལངྐ་ལིངྐ་ནས། ཁ་རུ་ཚའི་བར་ཞིབ་ཏུ་བཏགས་པའི་ཕྱེ་མ་གཞུག་པར་བྱའོ། །ཞག་བདུན་དེ་བཞག་བྱ། ཞེས་བྱ་བ་ལ། ཐལ་སྨན་དེ་གསུམ་ཆར་ཡང་ཞག་བདུན་བཞག་ལ། དེའི་འོག་ཏུ་གཏང་བར་བྱའོ། །རྣོན་པོ་དེ་ནི་རླུང་བད་ཀན། །ཚིལ་སྐྱེས་ཐོར་བུ་སོགས་ལ་
