གོ་ཏང་པ་ལ་ཁུ་བ་དང་། །མངར་བའི་སྡེ་བསྐོལ་འོ་མ་ཁུ། །ཡུ་ཤ་སེའུ་འབྲུར་བཅས་པ་བླུད། །ཅེས་བྱ་བ་ལ། དེ་བཞིན་དུ་ཁ་ར་དཀར་པོ་སྦྱིན་པར་བྱ་བ་དང་། དེ་བཞིན་དུ་རྒུན་འབྲུ་ལ་སོགས་པའི་ཁུ་བ་དང་། དེ་བཞིན་དུ་མངར་བའི་སྡེ་ཚན་
དང་། བསྐོལ་བའི་འོ་མ་དང་། དེ་བཞིན་ཤ་ཁུ་དང་། ཡུ་ཤ་སེའུ་འབྲུ་དང་བཅས་པ་བླུད་པར་བྱའོ། །སཱ་ལུ་རྩ་སྣར་དྲུག་ཅུ་པ། །ནས་དང་འཚོ་བྱེད་སྨན་མར་དང་། །དགེ་བ་ཆེན་པོ་ཏིག་ཏ་ཆེ། །སྲང་དྲུག་འོ་མ་ཙི་ཏྲར་བཅས། །པི་པི་ལིང་དང་བྲག་ཞུན་ནི། །བཅུད་ཀྱིས་
ལེན་གྱི་ཆོ་ག་སྦྱིན། །ཞེས་བྱ་བ་ལ། འབྲུ་ལས་འབྲས་སཱ་ལུ་རྩ་སྣར་ཞེས་བྱ་བ་དང་། དྲུག་ཅུ་པ་དང་ནས་ལས་གཡོས་པ་དག་སྦྱིན་པར་བྱ་ཞིང་། དེ་བཞིན་དུ་འཚོ་བྱེད་གྱི་སྨན་མར་དང་། དགེ་བ་ཆེན་པོའི་སྨན་མར་དང་། ཏིག་ཏ་ཆེན་པོའི་སྨན་མར་དང་། རོ་དྲུག་པའི་
སྨན་མར་དག་ཀྱང་ནད་གཞི་དང་། ལུས་ཟུངས་ཀྱི་ཚུལ་ལས་དཔགས་ཏེ་སྦྱིན་པར་བྱ་བ་དང་། དེ་བཞིན་དུ་འོ་མ་ཙི་ཏྲ་ཀ་དང་བཅས་པ་དག་བཅུད་ཀྱིས་ལེན་གྱི་ཆོ་གས་སྦྱིན་པར་བྱ་བ་དང་། ཡང་ན་པི་པི་ལིང་དང་། བྲག་ཞུན་དག་བཅུད་ལེན་གྱི་ཆོ་གར་ཇི་སྐད་བཤད་པ་བཞིན་དུ་སྦྱིན་པར་
བྱའོ། །ཡང་ན་འབྲས་བུ་གསུམ་མར་དང་། །སྦྲང་རྩི་ཁ་རར་བཅས་པ་སྦྱིན། །ཞེས་བྱ་བ་ལ། ཡང་ན་མྱོས་པ་དང་བརྒྱལ་བའི་ནད་དེ་དག་ལ་འབྲས་བུ་གསུམ་མར་དང་། སྦྲང་རྩི་ཁ་རར་བཅས་པ་སྦྱིན་པར་བྱའོ། །ཤུགས་དྲག་པ་ལ་ཁ་དང་སྣ། །བཀག་པར་གྱུར་པ་
ཕན་པ་ཡིན། །ཞེས་བྱ་བ་ལ། མྱོས་པ་དང་བརྒྱལ་བའི་ཤུགས་དྲག་ཅིང་རྒྱུན་རིང་བ་ལ་ཁ་དང་སྣའི་བུ་ག་ལག་པ་ལ་སོགས་པས་བཀག་ཅིང་རེ་ཞིག་དབུགས་བཀག་པ་དག་ཕན་ནོ། །ཡང་ན་བུད་མེད་ནུ་ཞོ་ནི། །བླུད་པའམ་སྣ་ནས་གཏང་བར་བྱ། །ཞེས་བྱ་བ་ལ། ཡང་ན་ནད་
དེ་དག་ལ་བུད་མེད་ཀྱི་ནུ་ཞོ་སྣར་བླུགས་པའི་སྣ་སྨན་གཏང་བར་བྱའོ། །པདྨའི་རྩ་སྡོང་པི་པི་ལིང་། །ཨ་རུ་ར་འམ་གླ་སྒང་ངམ། །བྱི་ཚེར་སྦྲང་རྩིར་སྦྱར་བ་སྦྱིན། །ཞེས་བྱ་བ་ལ། པདྨའི་རྩ་བ་དང་། སྡོང་བུ་དང་པི་པི་ལིང་དང་། ཨ་རུ་ར་སྦྲང་རྩི་དང་སྦྱར་ལ་སྦྱིན་པར་བྱའོ། །གླ་སྒང་
