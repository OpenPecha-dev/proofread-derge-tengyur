བ་དང་། །འབྲས་བུ་ལོ་མར་བཅས་པ་ཡི། །ཁུ་བ་བྲེ་བཞིའི་ཚད་ལ་ནི། །མར་བྲེ་གང་དང་བ་ལ་དང་། །ཚ་བ་གསུམ་དང་བྱི་ཏང་ཀ ། ཤུ་དག་དང་ནི་སེའུ་འབྲུ་དང་། །ཁ་རུ་ཚ་དང་ནས་ཚིག་ཐལ། །པི་པི་ལིང་གི་རྩ་བ་དང་། །སྐྱུར་རུ་ར་དང་མ་ནུ་དང་། །བྲི་ཤི་བ་དང་
བྲི་ཧ་ཏི། །ཨ་རུ་ར་དང་ཡ་བ་ཏི། །ཀྲུ་ཏྲུག་ཏྲེས་དང་རྒུན་འབྲུ་དང་། །དབྱི་མོ་བྱི་ཚེར་བར་ཤ་པུ། །སྟར་བུ་ཀར་ཀ་ཏ་སྲིང་གི །ཏམ་ལ་གི་ག་བྲའི་རྩ། །ར་སྣ་གཟེ་མའི་ལྡེ་གུ་དང་། །བསྐོལ་བའི་མར་དེས་ལུད་ཀུན་དང་། །སྐྱིགས་བུ་དབུགས་མི་བདེ་ལའང་ཕན། །ཀཎྜ་ཀ་
རིའི་སྨན་མར་འདི། །བད་ཀན་ནད་ནི་སེལ་བྱེད་ཡིན། །ཞེས་བྱ་བ་ལ། གཎྜ་ཀ་རིའི་རྩ་བ་དང་འབྲས་བུ་དང་ལོ་མར་བཅས་པ་བསྐོལ་བའི་ཁུ་བ་བྲེ་བཞིའི་ནང་དུ་མར་བྲེ་གང་བཅུག་ལ། བ་ལ་དང་། ཚ་བ་གསུམ་ལ་སོགས་པའི་སྨན་ལྡེ་གུའི་ཚད་ཀྱིས་སྤྱིར་མར་གྱི་བཞི་ཆ་གཅིག་
བཅུག་ལ་བསྐོལ་བའི་སྨན་མར་དེས་ནི་ལུད་པ་ལ་སོགས་པ་ལྔ་པོ་ཀུན་དང་། སྐྱིགས་བུ་ལྔ་དང་དབུགས་མི་བདེ་བ་ལྔ་པོ་སྔར་བཤད་པ་ལ་ཕན་པ་ཡིན་ནོ། །འདིར་ཆུ་སྲང་ཉིས་བརྒྱ་ཉི་ཤུ་རྩ་བརྒྱད་ཀྱི་ནང་དུ་མར་སྲང་སུམ་ཅུ་རྩ་གཉིས་བཅུག་ལ། ཚ་བ་གསུམ་ལ་སོགས་པའི་ལྡེ་
གུ་མར་གྱི་བཞི་ཆ་གཞུག་པར་བྱའོ། །མ་ནུ་ནི་པུཥྐ་ར་མཱུ་ལའོ། །བྲི་ཤི་ལྕི་བ་ནི་བ་སྦྲུ་བ་རྩ་བ་དཀར་པོ་ཅན་ནོ། །ཡང་བ་ནི་ལ་ལ་ཕུད་དོ། །ཀྲུ་ཏྲུག་ཏྲེས་ནི་ཙི་ཏྲ་ཀའོ། །དབྱི་མོ་ནི་རྩ་བ་ལ་བྱའོ། །བྱི་ཚེར་བ་ནི་དུར་ལ་ཝ་པའོ། །བར་ཤ་བུ་ནི་བ་སྤྲུ་བ་མེ་ཏོག་དམར་པོ་ཅན་ནོ། །
སྟར་བུ་ནི་ཨཾ་ལ་ཙེ་ཏ་སེའོ། །ཏ་མ་ལ་གི་ནི་ཨ་ཛ་ཏའོ། །ག་བྲའི་རྩ་བ་ནི་བར་གྱིའོ། །གཟེམ་ནི་གོ་ཀྵ་རའོ། །བྱ་ཀྲི་སྲང་བརྒྱད་བརྡུངས་ཆུ་སྲང་། །ཉིས་སྟོང་བཞི་བཅུ་བརྒྱད་ནང་བསྐོལ། །བཙགས་པར་ཚ་བ་གསུམ་ར་སྣ། །སླེ་ཏྲེས་ཀྲུ་ཏྲུག་ཕྱེ་མ་དང་། །ཀར་ཀ་ཏ་སྲིང་ག་
བྲའི་རྩ། །གླ་སྒང་པི་པི་ལིང་རྩ་བ། །བྱི་ཚེར་བ་རྣམས་སྲང་ཕྱེད་ཕྱེད། །ཞུན་མར་སྲང་ནི་བཅུ་དྲུག་དང་། །ཁར་དཀར་པོ་སྲང་བཞི་བཅུ། །སླར་ཡང་དེ་ནི་བསྐོལ་བར་བྱ། །གཟར་བུ་ལ་ཆགས་བསྒྲངས་ན་ནི། །པི་པི་ལིང་དང་སྨིག་རྐང་དང་། །སྦྲང་རྩི་རྙིང་པ་དག་ལས་ནི། །
