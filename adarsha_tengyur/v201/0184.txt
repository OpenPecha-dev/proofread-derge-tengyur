གི་རྣམ་གྲངས་སུ་མངོན་པར་རྨུགས་པར་བྱེད་པ་དང་། མདངས་འཕྲོག་པར་བྱེད་པ་ཞེས་བཤད་ཅིང་བཏགས་པ་ཡིན་ནོ། །ནད་ཀྱིས་བཀག་ཅིང་མེ་དྲོད་ཉམས། །མཚན་མ་ཐམས་ཅད་རྫོགས་གྱུར་པ། །གསོར་མི་རུང་སྟེ་དེ་ལས་གཞན། །གསོ་དཀའ་འམ་དབང་
པོ་མ་ཚང་འགྱུར། །ཞེས་བྱ་བ་ལ། རླུང་དང་མཁྲིས་པ་དང་བད་ཀན་དང་བཤང་བ་དང་གཅི་བ་ལ་སོགས་པའི་ནད་ཀྱི་སྐྱོན་གྱིས་འབྱུང་བའི་རྒྱུན་བཀག་ཅིང་མེ་དྲོད་ཉམས་པར་གྱུར་པ་དང་། དེ་བཞིན་དུ་ཇི་སྐད་སྨོས་པའི་མཚན་ཉིད་ཐམས་ཅད་རྫོགས་པར་གྱུར་ན་
གསོར་མི་རུང་སྟེ། གསོ་བར་མི་ནུས་པ་ཡིན་ནོ། །དེ་ལས་གཞན་པ་འདུས་པའི་རིམས་དེ་ནད་ཀྱི་ཚབས་ཆུང་ཞིང་མཚན་ཉིད་མ་རྫོགས་ན། དེ་གསོ་དཀའ་བར་འགྱུར་བའམ། དབང་པོ་མ་ཚང་བ་སྟེ་ལུས་ངག་གི་ལས་ཀྱི་རྟེན་ཉམས་པར་འགྱུར་རོ། །གཞན་ཡང་
འདུས་པ་ལས་བྱུང་བ། །གང་དུ་མཁྲིས་པ་ལོགས་ཤིག་གནས། །པགས་པའམ་ནང་གྲོལ་ཚ་བ་དག །འདེབས་པའམ་ཚ་བ་སྔ་ཕྱིས་འབྱུང་། །ཞེས་བྱ་བ་ལ། འདུས་པ་ལས་གྱུར་པའི་རིམས་སྔ་མ་ལས་མཚན་ཉིད་གཞན་དུ་འབྱུང་བར་གྱུར་པ་སྟེ། འདུས་
པའི་རིམས་གང་ལས་མཁྲིས་པ་ལོགས་ཤིག་ན་གནས་ཤིང་། ལན་འགའ་ནི་པགས་པར་གནས་པ་འམ། ལན་འགའ་ནི་ནང་གྲོལ་ན་གནས་ནས་ཚ་བ་སྐྱེད་པར་བྱེད་པ་སྔར་བྱུང་བའམ། ཕྱིས་འབྱུང་བའི་མཚན་ཉིད་ཅན་ནི་གསོ་དཀའ་བ་ཡིན་ནོ། །དེ་བཞིན་བད་
ཀན་རླུང་གྲང་བ། །ཞེས་བྱ་བ་ལ། ཡང་དེ་བཞིན་དུ་རླུང་དང་བད་ཀན་ལོགས་ཤིག་ཏུ་ཐ་དད་པར་གནས་ཤིང་སྔ་ཕྱིར་གྲང་བར་བྱེད་པ་ནི་གསོ་དཀའ་བ་མ་ཡིན་ནོ། །དེ་གཉིས་ཚ་སྔ་གསོ་དཀའ་ཡིན། །ཞེས་བྱ་བ་ལ། འདུས་པའི་ནད་དེ་གཉིས་ལས་ཚ་བ་སྔ་བའི་ནད་
ནི་གསོ་དཀའ་ཞིང་བཅོས་པར་དཀའ་བ་ཡིན་ནོ། །གྲང་སྔ་དེར་ནི་མཁྲིས་པ་ཡིས། །བད་ཀན་གཤེར་བ་སྐམ་པ་དང་། །གྲང་བ་ཞི་བྱས་ཁ་སྐྱུར་བརྒྱལ། །མྱོས་དང་སྐོམ་དད་སྐྱེ་བར་འགྱུར། །ཞེས་བྱ་བ་ལ། གྲང་བ་སྔར་བྱུང་བའི་འདུས་པའི་རིམས་དེར་ནི་མཁྲིས་
