སྐྱེ་མེད་ལས་འདས་སོ། །སྣང་བ་ཆོས་ཅན་འདི་ཉིད་སྣ་ཚོགས་སུ་སྣང་སྟེ། དེ་ཉིད་རྒྱུ་རྐྱེན་ལས་སྐྱེས་པའི་ཕྱིར་སྟོང་པའོ། །སྟོང་པ་གཏན་མེད་པ་མ་ཡིན་ཏེ། ཉམས་སུ་མྱོང་བ་དང་ལྡན་པས་སོ། །དེ་དབྱེར་མེད་པ་ནི་ཡེ་ཤེས་སོ། །དེ་ཡང་སྒྲོ་འདོགས་གཅོད་པའི་དུས་སུ་ཚད་མ་

ཅན་དང་སངས་རྒྱས་ལ་ཁྱད་མེད་དེ་སེམས་ཅན་ལས་མ་རིག་པས་བསྒྲིབས་སོ། །སྠཱི་ཏོ་ནས་བརྟི་ནཾ་གྱིས་དུས་གསུམ་གྱི་སངས་རྒྱས་ཀྱི་ཐུགས་ཀ་ན་སྐུ་དང་ཡེ་ཤེས་ཀྱི་ཚུལ་གྱིས་བཞུགས་པར་བསྟན་ནོ། །སྐུ་དང་པོ་བུདྡྷ་དང་འཁོར་ལོ་དང་། ཡེ་ཤེས་སེམས་དཔའ་དང་། ཟླ་བ་དང་ཨའོ། །
དེ་ཉམས་སུ་ལེན་ན། ཨོཾ་བཛྲ་ཏཱིཀྵྞ་ནས་ཡ་ཏེ་ན་མཿ སྐུ་དྲུག་དང་དཀྱིལ་འཁོར་དྲུག་དང་། དེ་ལ་བརྟེན་ནས་དབང་དང་རབ་གནས་དང་སྦྱིན་སྲེག་དང་ཚ་ཚ་དང་བཟླས་པ་དང་སྒོམ་པའི་ཚུལ་གྱིས་ཉམས་སུ་བླང་ངོ་། །དེ་ལ་ཨ་ར་པ་ཙ་ནའི་དོན་འདི་ལྟར་ནས་ཚུལ་གྱི་མཆོག་གིས་བསྟན་ཏེ། 
འདིར་ལྷ་དང་སྦྱར་བ་ཡང་ཡིན་ལ་ད་རེས་ཆོས་ཉིད་བྱང་ཆུབ་སེམས་དེ་ཨ་ཡིས་མཚོན་ཏེ་སྐྱེ་མེད་ལས་བྱུང་བའོ།    །སྐྱེ་མེད་དེ་ལ་དབྱེ་བ་དགག་བྱའི་སྒོ་ནས་ཕྱེ་ན་རྣམ་པར་ཐར་པའི་སྒོ་བཞི་སྟེ་ཁོང་ནས་གསལ་བར་བྱེད་ཀྱིས་བསྟན་ནོ། །དེ་ལྟར་རྟོགས་ན་དུག་ལྔ་ཡེ་ཤེས་ལྔར་གྱུར་པ་ནི་
