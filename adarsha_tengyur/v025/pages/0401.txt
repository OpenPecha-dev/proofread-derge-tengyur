རྒྱུན་མི་འཆད་པས་དུས་རིང་པོ་རབ་ཏུ་སྦྱར་ཞིང་བསྒོམ་པའོ། །དུས་བཞིར་རབ་ཏུ་སྦྱོར་བ་ཡིས། །ཞེས་བྱ་བ་ནི་རྣལ་འབྱོར་དང་མཆོད་པའི་དུས་སོ། །གསང་བའི་མཆོད་པ་བསྟན་པའི་ཕྱིར། རྒྱན་རྣམས་ཐམས་ཅད་ཅེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། བུ་མོ་གཞུག་
པར་བྱའོ་ཞེས་བྱ་བ་ནི་ལྷག་མའོ། །རྡོ་རྗེའི་སྐུ་དང་། རྡོ་རྗེའི་གསུང་དང་། རྡོ་རྗེའི་ཐུགས་དག་ནི་སྐུ་གསུམ་མཆོག་གོ། །དེ་ལྟ་བས་ན་སྐུ་གསུམ་མཆོག་སྟེར་བ་སྟེ། རྡོ་རྗེ་སེམས་དཔའོ། །ཇི་ལྟར་ཞེས་བྱ་བ་ནི་ཇི་ལྟ་བ་བཞིན་དུ་དངོས་གྲུབ་ཐོབ་པར་འགྱུར་བའོ། །རབ་ཏུ་སྣང་བར་བྱེད་ཅེས་བྱ་བ་
ནི་རང་བཞིན་འོད་ཀྱིས་སོ། །འདིར་འདའ་བར་བྱ་བ་མ་ཡིན་པའི་དམ་ཚིག་གཉིས་བསྟན་པའི་ཕྱིར། དབང་པོ་གཉིས་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། རྫས་ཞེས་བྱ་བ་ལ། བཟའོ་ཞེས་བྱ་བ་ནི་ལྷག་མའོ། །སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ཉིད་གོ་འཕང་ཡིན་ཏེ་རབ་ཏུ་གནས་པའོ། །
ལྷག་མ་ནི་འདི་ཉིད་ཀྱི་བཤད་པ་སྟེ། ཉེ་བར་འཚེ་བ་ཐམས་ཅད་དང་བྲལ་བས་ན་ཞི་བ་སྟེ། ངེས་པར་ལེགས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཉན་ཐོས་དང་རང་སངས་རྒྱས་དག་ལ་ཡང་འདི་ལྟ་བུའི་ཞི་བ་ཡོད་པས་དེ་ཉིད་ཅེས་བྱ་བ་ནི་དེ་ལས་ཁྱད་པར་དུ་བྱེད་དེ། སྤྲོས་པ་ཐམས་ཅད་དང་བྲལ་བའི་ཡེ་
ཤེས་ཞེས་བྱ་བའི་དོན་ཏོ། །བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་ཡང་དེ་ལྟ་བུའི་ཡེ་ཤེས་ཡོད་མོད་ཀྱི། ད་ནི་སངས་རྒྱས་ཀྱི་ཡེ་ཤེས་ཀྱི་བླ་མ་དང་བཅས་པ་ཡིན་པས་དེ་ལས་བྱེ་བྲག་ཏུ་བྱ་བའི་ཕྱིར་བླ་ན་མེད་པ་ཞེས་བྱ་བ་གསུངས་ཏེ། འདི་ལས་བླ་མར་གྱུར་པ་མེད་པས་ན་བླ་ན་མེད་པའོ། །འདིར་ཚིགས་
བཅད་དུ་འགྱུར་བ་ཞེས་བྱ་བ་ནི་ཇི་སྐད་དུ་གསུངས་པའི་རྒྱུད་ཀྱི་དོན་བསྡུ་བའི་ཚིགས་སུ་བཅད་པའོ། །གཟུགས་རྣམས་ཞེས་བྱ་བ་ནི་ལྷ་རྣམས་ཀྱི་སྐུ་མདོག་དང་དབྱིབས་རྣམས་སོ། །ཕྱག་རྒྱ་ཞེས་བྱ་བ་ནི་ཕྱག་རྒྱས་བཏབ་པའོ། །ཡན་ལག་དྲུག་ཏུ་དགོད་པ་དང་། རྡོ་རྗེ་པདྨོ་སྦྱོར་བ་ལ་
སོགས་པའོ། །དེ་བཞིན་མཚན་མ་ཞེས་བྱ་བ་ནི་ཐོད་པ་དང་ཁ་ཊྭཱཾ་ག་ལ་སོགས་པའོ། །རྫས་རྣམས་ནི་དམ་ཚིག་གི་རྫས་རྣམས་སོ། །བསམ་གཏན་ཞེས་བྱ་བ་ལ། བསམ་པར་བྱ་བས་ན་བསམ་གཏན་ཏེ། བཟླས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །བསྒོམ་པ་ཞེས་བྱ་བ་ནི་དཀྱིལ་འཁོར་
