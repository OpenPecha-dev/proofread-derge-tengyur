པའི་སྒྲ་ལས་ཕྱག་རྒྱ་ཆེན་པོའི་སྟེ། སྔ་མའི་ངོ་བོ་དུ་བ་ལ་སོགས་པའི་གཟུགས་བརྙན་ཆོས་ཀྱི་ཕྱག་རྒྱ་ལ་སོགས་དམིགས་པས་ཉོན་མོངས་པ་ཆུང་ངུ་ཟད་པ་ལས་མི་འགྱུར་བའི་བདེ་བ་ཆེན་པོ་འཕེལ་བའོ། །ཕྱི་ནང་གི་ངོ་བོ་སྣ་ཚོགས་གཟུགས་བརྙན་ཞེས་བྱ་བ་ཕྱག་རྒྱ་ཆེན་པོ་གོམས་
པས། སྒྲིབ་པ་བག་ཆགས་དང་བཅས་པ་ཆུང་ངུའི་ཆུང་ངུ་ཟད་པ་ལས་མི་འགྱུར་བའི་བདེ་བ་ཆེན་པོའི་ཆེན་པོ་འཕེལ་བའོ། །འོ་ན་མི་འགྱུར་བའི་བདེ་བ་ནི་འཕེལ་བ་དང་འགྲིབ་པ་སྤངས་པ་ཡིན་ན་ཅི་ལྟར་འཕེལ་བར་འགྱུར་ཞེ་ན་གསུངས་པ། འཕེལ་བའི་འཕེལ་བ་མེད། འཕེལ་བའི་ཆོས་ཅན་
གྱིས་གཅིག་དང་དུ་མ་ལ་སོགས་པའི་ངོ་བོ་དང་བྲལ་བ་རྣམས་རང་བཞིན་སྟོང་པ་ཉིད་ཀྱི་ཕྱིར་ཟད་པ་ན་ཆོས་མི་འཕེལ་བའོ། །དེ་བཞིན་དུ་ཟད་པ་ལ་སོགས་པའི་ཟད་པ་ཉིད་ལ་སོགས་པ་ཡང་གསལ་བའོ། །འོ་ན་དེ་ལྟ་ན་སྐྱེ་བ་དང་ཟད་པ་མེད་པའི་ཕྱིར་གསུངས་པ། རང་བཞིན་སྟོང་པ་ཉིད་ཀྱི་ཆོས་
ལམ་ལ་སོགས་པ་རྣམས་སྐྱེ་བ་དང་འཇིག་པའང་སྒྱུ་མ་ལྟ་བུ་ཁོ་ན་ཡིན་གྱི་དེ་ཁོ་ན་ཉིད་དུ་མ་ཡིན་ནོ། །དེ་བཞིན་ནོར་བུའི་སྒྲ་ལས་ནུབ་པ་དང་འཆར་བ་ལ་སོགས་པའི་མཚན་ཉིད་ཅན་ཡང་འཇིག་རྟེན་ཀུན་རྫོབ་ཏུ་སྒྱུ་མ་ལྟ་བུ་ཁོ་ནའོ། །དེ་གསུངས་པ། འཇིག་པར་མི་འགྱུར་བ་ཞེས་པ་ལ་སོགས་པ་སྟེ།  ཀུན་རྫོབ་ཏུ་དོན་བྱ་བ་བྱེད་པའི་ཕྱིར་དངོས་པོའི་རང་བཞིན་ཏེ། དེ་ཁོ་ན་ཉིད་དུ་ན་རང་བཞིན་སྟོང་པ་ཉིད་ཀྱི་དངོས་པོ་མེད་པའི་མཚན་ཉིད་ཅན་ནོ། །ལས་ཀྱི་ཕྱག་རྒྱ་ལ་སོགས་པ་རང་གི་ཤེས་རབ་ལ་འཁྱུད་པ་བྱང་ཆུབ་ཀྱི་སེམས་གང་གི་ཚེ་རྡོ་རྗེ་ནོར་བུའི་པདྨར་སོན་པ་དེའི་ཚེ་རྡོ་རྗེ་ནོར་བུ་དེར་
ཟླ་བ་སོན་པ་ནི་ལས་ཀྱི་ཕྱག་རྒྱས་འགྱུར་བའི་བདེ་བའི་གནས་སྐབས་དང་། ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱའི་ས་གཡོ་བའི་གནས་སྐབས་ཏེ། གཡོ་བ་ཞེས་པ་འཛག་པའི་ཆོས་ཡིན་ལ་དེས་འགྱུར་བ་ཡང་དག་པར་གཟུང་བའོ། །བདེ་བའི་གནས་སྐབས་འདི་དག་མངོན་དུ་བྱས་པ་ནི། འདོད་
པ་དང་གཟུགས་ཀྱི་ཁམས་ཀྱི་དབང་ཕྱུག་གི་མཚན་ཉིད་ཅན་འཇིག་རྟེན་པའི་དངོས་གྲུབ་སྒྲུབ་པར་བྱེད་པ་དག་གོ། །ཕྱག་རྒྱ་ཆེན་པོས་མི་གཡོ་བ་དང་མི་འགྱུར་བ་བསྒོམ་པ་ནི་འཇིག་རྟེན་ལས་འདས་པའི་དངོས་གྲུབ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་མཚན་ཉིད་ཅན་སྒྲུབ་པར་བྱེད་པའོ། །དབུ་
