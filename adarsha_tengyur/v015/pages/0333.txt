མཐུན་པ་སྟེ། མདོག་དམར་བ་དང་སྡུག་བསྔལ་སྐྱེད་པ་དང་། རེག་བྱ་ཚ་བའི་ཕྱིར་ཁྲག་གོ། །རྡོ་རྗེ་ཟླ་བ་དྲི་མེད་འོད། །ཅེས་པ་ནི་ཁ་དོག་དཀར་བ་དང་། བདེ་བ་སྐྱེད་པ་དང་། བསིལ་བ་སྟེ་བྱང་ཆུབ་ཀྱི་སེམས་སོ། །ཆགས་བྲལ་ལ་སོགས་ཆགས་པ་ཆེ། །ཞེས་པ་ནི་གནས་
སྐབས་ལྔའི་རང་བཞིན་ཏེ། ཡི་གེ་དང་། ཕྱག་མཚན་ནོ། །ཁ་དོག་སྣ་ཚོགས་འབར་བའི་འོད། །ཅེས་པ་ནི་འོད་ཟེར་སྤྲོ་བ་དང་བསྡུ་བའོ། །རྡོ་རྗེ་སྐྱིལ་ཀྲུང་རྫོགས་སངས་རྒྱས། །ཞེས་པ་ནི་བུ་སྐྱེས་པ་སྟེ། མངལ་ནས་སྐྱེད་པའི་དོན་ཏོ། །བྱང་ཆུབ་པ་ལྔའི་དབང་དུ་བྱས་པ། རྡོ་རྗེ་ཉི་མ་སྣང་བ་ཆེ། །ཞེས་
པ་ནི་ཚོགས་གཉིས་སྔོན་དུ་སོང་བས་བསྲུང་བ་རྡོ་རྗེ་ར་བ་དང་། དེར་ཆོས་ཀྱི་འབྱུང་གནས་དང་། གཞལ་ཡས་ཁང་གདན་དང་བཅས་པའི་ནང་དུ་ཀཱ་ལི་ཚར་གཉིས་ལས་གྲུབ་པའི་ཉི་མའི་དཀྱིལ་འཁོར་ལ་ཡང་ཀཱ་ལིས་བརྒྱན་པ་སྟེ། ཇི་ལྟར་བྱི་རུ་ལ་མུ་མེན་གྱི་འཕྲ་བཀོད་པ་ལྟ་བུར་བསྒོམ་པ་ནི་མཉམ་པ་
ཉིད་ཀྱི་ཡེ་ཤེས་ཀྱི་རང་བཞིན་ནོ། །རྡོ་རྗེ་ཟླ་བ་དྲི་མེད་འོད། །ཅེས་པ་ནི་ཨཱ་ལི་ཟུང་ལས་བྱུང་བའི་ཟླ་བའི་དཀྱིལ་འཁོར་རྒྱས་པ་ལ་དབྱངས་ཀྱིས་བརྒྱན་པ་ནི་ཇི་ལྟར་དུང་ལ་མུ་ཏིག་བཀོད་པ་ལྟ་བུ་སྟེ་མེ་ལོང་ལྟ་བུའི་ཡེ་ཤེས་ཀྱི་ངོ་བོའོ། །དེར་ཡི་གེ་ཧཱུྃ་ནག་པོ་བསམ་པ་ནི་སོ་སོར་རྟོག་པའི་ཡེ་ཤེས་ཀྱི་ངོ་
བོའོ། །དེ་གཉིས་ལས་འོད་ཟེར་ཕར་སྤྲོ་ཞིང་བསྡུས་པ་ལས་རྡོ་རྗེ་ཧཱུྃ་དང་ལྡན་པ་ནི་བྱ་བ་ནན་ཏན་གྱི་རང་བཞིན་ནོ། །ཆགས་བྲལ་ཞེས་པ་ནི་སྤྲུལ་པའི་སྐུ་སྟེ། ཡུམ་མེད་པའི་ཡབ་བམ། ཡབ་མེད་པའི་ཡུམ་བསྒོམ་པ་སྟེ་ཁྲོ་བོའོ། །ཡང་ན་འཇམ་པའི་རྡོ་རྗེ་ཞལ་གཅིག་ཕྱག་བཞི་པ་གཡས་
ན་རལ་གྲི་དང་། མདའ། གཡོན་ཨུཏྤལ་དང་གཞུ་འཛིན་པའོ། །སོགས་ཞེས་པ་ནི་ལོངས་སྤྱོད་པའི་སྐུ་སྟེ། ཆགས་པ་ཆེ་ཞེས་པ་ནི་ཡབ་ཡུམ་ཁ་སྦྱོར་ཁྲོ་བོ་རྣམ་པར་བསྒོམ་པའོ། །ཁ་དོག་སྣ་ཚོགས་ཞེས་པ་ནི་དཀར་པོ་ལ་སོགས་པ་གང་འདོད་པའོ། །འབར་བའི་འོད་ཅེས་པ་ནི་དཀར་པོ་
ཡང་ཤིན་ཏུ་དཀར་བ་ཞེས་བྱ་བ་ལ་སོགས་པའོ། །རྡོ་རྗེ་སྐྱིལ་ཀྲུང་རྫོགས་སངས་རྒྱས། །ཞེས་པ་ནི་བཞུགས་སྟབས་ཏེ་ཇི་ལྟ་བུའི་ཡང་ཞེས་པའི་དོན་ཏོ། །དེ་དག་ནི་བསྡུས་པ་ལ་དགའ་བའི་དབང་དུ་མཛད་ནས་བསྐྱེད་པའི་རིམ་པ་གསུངས་པའོ། །ཡང་ན་ཀློག་པའི་དབང་དུ་བྱས་ནས་ཚོགས་སོགས་
