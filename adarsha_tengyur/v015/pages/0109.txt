པ་རླུང་རྣམས་མི་རྒྱུ་བ་སྐྱེས་པའོ། །སླར་ཡང་། ཤེས་རབ་ཐབས་ཀྱི་བདག་ཉིད་མི་འགྱུར་བདེ་བའི་དབང་ལས་ཡེ་ཤེས་སྐུ་ནི་ཏིང་ངེ་འཛིན། །ཞེས་གསུངས་སོ། །དེ་ཉིད་ཀྱི་ཕྱིར་དབང་ལྡན་ཏེ་རྣམ་པར་རྟོག་པའི་རླུང་མཐའ་དག་མཆོག་ཏུ་མི་འགྱུར་བའི་དེ་ཉིད་བདེ་བ་མི་ཕྱེད་པ་དབང་
དུ་གྱུར་པའི་ཕྱིར་རོ། །ཆེ་བ་པོ་ལྷན་སྐྱེས་ཀྱི་གཟུགས་ཀྱིས་འཇིག་རྟེན་གསུམ་དུ་ཁྱབ་པའི་ཕྱིར་རོ། །འདི་ལྟ་བུ་ནི་འཇིག་རྟེན་ལས་འདས་པའི་དབང་སྟེ། དབང་བསྐུར་ནས་ནི་འདི་ཉིད་བསྟན། །ཞེས་པས་འཇིག་རྟེན་པའི་དབང་སྔོན་དུ་འགྲོ་བ་ནི་སྦྱོར་བ་ཡན་ལག་དྲུག་གི་རྒྱུ་ཉིད་ཁོ་ནའོ། །སྦྱོར་བ་ཡན་ལག་
དྲུག་པོ་དེ་ཡང་དུ་བ་ལ་སོགས་པའི་རིམ་པས་བསྒོམ་པར་བྱའོ། །དེ་ལྟར་ཡང་། འདི་ཉིད་ཁོ་ནའི་སྒྲུབ་པའི་ཐབས། །རབ་ཏུ་སྣང་བ་བསམ་མི་ཁྱབ། །དུ་བ་ལ་སོགས་མཚན་མ་དེ། །ཤེས་རབ་གཟུགས་ནི་ནམ་མཁའ་མཉམ། །ཡོད་དང་མེད་ལས་རྣམ་པར་འདས། །རང་གི་སེམས་ཀྱི་རྐྱེན་
གྱི་དོན། །རྡུལ་ཕྲན་ལས་སྐྱེས་ཚོགས་པ་དག །ཀུན་ནས་ཡོངས་སུ་དོར་བར་བར་བྱ། །དུ་བ་སྨིག་རྒྱུ་མེ་ཁྱེར་དང་། །སྒྲོན་མ་འབར་མ་ཟླ་ཉི་མ། །མུན་ཅན་ཆ་དང་ཐིག་ལེ་ཆེ། །སྣ་ཚོགས་གཟུགས་ནི་འོད་གསལ་དག །ཅེས་གསུངས་སོ། །བཀབ་དང་མ་བཀབ་མིག་དག་གིས། །སྟོང་
པར་འདིས་ནི་རྗེས་མ་བརྟགས། །རྨི་ལམ་ལྟ་བུར་གཟུགས་མཐོང་སྟེ། །གཟུགས་དེ་རྟག་ཏུ་བསྒོམ་པར་བྱ། །ཞེས་ཀྱང་གསུངས་སོ། །བཤད་པ་ཡང་འདི་ཉིད་ཁོ་ནའི་སྟེ་ཉོན་མོངས་པ་དང་ཤེས་བྱའི་སྒྲིབ་པ་མ་ལུས་པ་མེད་པའི་བུད་ཤིང་གི་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་ཀྱི་མེའོ། །སྒྲུབ་
པའི་ཐབས་གསལ་བའི་ངོ་བོར་རྣལ་འབྱོར་པས་དུ་བ་ལ་སོགས་པར་བྱའོ། །ཇི་ལྟ་བུར་གྱུར་པས་ཤེ་ན། རྩ་བའི་རྒྱུད་དང་། བསྡུས་པའི་རྒྱུད་དང་། འཕགས་པ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སོགས་པ་ངེས་པའི་དོན་གྱི་མདོ་སྡེ་ཐམས་ཅད་དུ་གྲགས་པས་སོ། །སྟོང་པ་ཉིད་ལས་གློ་
བུར་དུ་རབ་ཏུ་སྣང་བའི་ཕྱིར་རབ་ཏུ་སྣང་བའོ། །རྣམ་པར་རྟོག་པ་ཐམས་ཅད་ལས་འདས་པའི་ཕྱིར་བསམ་གྱིས་མི་ཁྱབ་པའོ། །དེ་ཁོ་ན་ཉིད་ཀྱི་སྔོན་གྱི་ཚུལ་གྱི་ཕྱིར་མཚན་མའོ། །གཟུང་འཛིན་དང་བྲལ་བའི་རང་བཞིན་སྟོང་པའི་བློ་ནི་ཤེས་རབ་བོ། །དེའི་གཟུགས་ནི་རྣལ་འབྱོར་མངོན་སུམ་
