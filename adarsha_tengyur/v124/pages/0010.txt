མཚན་ཉིད་དུ་མི་འགྱུར་ཏེ། དེས་དེ་ཉིད་དང་གཞན་ལས་རྣམ་པར་གྲོལ་བ་སྟོང་པ་ཉིད་ཀྱི་མཚན་ཉིད་ཡོངས་སུ་བསྟན་པ་ཡིན་ནོ། །རྣམ་གྲངས་ཇི་ལྟར་རིག་པར་བྱ་ཞེ་ན། སྟོང་པ་ཉིད་ནི་མདོར་བསྡུ་ན། །དེ་བཞིན་ཉིད་དང་ཡང་དག་མཐའ། །མཚན་མ་མེད་དང་དོན་དམ་དང་། །ཆོས་ཀྱི་དབྱིངས་ནི་རྣམ་གྲངས་སོ། །རྣམ་
གྲངས་ཀྱི་དོན་ཇི་ལྟར་ཤེས་པར་བྱ་ཞེ་ན། གཞན་མིན་ཕྱིན་ཅི་ལོག་མ་ཡིན། །དེ་འགོག་འཕགས་པའི་སྤྱོད་ཡུལ་དང་། །འཕགས་པའི་ཆོས་ཀྱི་རྒྱུ་ཡི་ཕྱིར། །རྣམ་གྲངས་དོན་དེ་གོ་རིམས་བཞིན། །གཞན་མ་ཡིན་པའི་ཕྱིར་དེ་བཞིན་ཉིད་དེ། རྟག་ཏུ་དེ་བཞིན་པའི་ཕྱིར་རོ། །ཕྱིན་ཅི་ལོག་མ་ཡིན་པའི་ཕྱིར་ཡང་དག་པའི་
མཐའ་སྟེ། ཕྱིན་ཅི་ལོག་གི་གཞི་མ་ཡིན་པའི་ཕྱིར་རོ། །མཚན་མ་འགོག་པའི་ཕྱིར་མཚན་མ་མེད་པ་སྟེ། མཚན་མ་ཐམས་ཅད་མེད་པའི་ཕྱིར་རོ། །འཕགས་པའི་ཡེ་ཤེས་ཀྱི་སྤྱོད་ཡུལ་ཡིན་པས་དོན་དམ་པ་སྟེ། ཡེ་ཤེས་དམ་པའི་ཡུལ་ཡིན་པའི་ཕྱིར་རོ། །འཕགས་པའི་ཆོས་ཀྱི་རྒྱུ་ཡིན་པས་ཆོས་ཀྱི་དབྱིངས་
ཏེ། འཕགས་པའི་ཆོས་རྣམས་ནི་དེ་ལ་དམིགས་པས་འབྱུང་བའི་ཕྱིར་རོ། །འདིར་དབྱིངས་ཀྱི་དོན་ནི་རྒྱུའི་དོན་ཏོ། །སྟོང་པ་ཉིད་ཀྱི་རབ་ཏུ་དབྱེ་བ་ཇི་ལྟར་ཤེས་པར་བྱ་ཞེ་ན། ཀུན་ནས་ཉོན་མོངས་རྣམ་པར་དག །ཅེས་བྱ་བ་སྟེ། དེའི་རབ་ཏུ་དབྱེ་བ་སྟེ། གང་གི་ཚེ་ཀུན་ནས་ཉོན་མོངས་པ་ལ། གང་གི་ཚེ་རྣམ་པར་དག་ཅེ་ན། 
དེ་ནི་དྲི་བཅས་དྲི་མ་མེད། །གང་གི་ཚེ་དྲི་མ་དང་བཅས་པར་གྱུར་པ་དེའི་ཚེ་ཀུན་ནས་ཉོན་མོངས་པ་གནས་པའོ། །གང་གི་ཚེ་དྲི་མ་སྤངས་པ་དེའི་ཚེ་རྣམ་པར་དག་པའོ། །གལ་ཏེ་དྲི་མ་དང་བཅས་པར་གྱུར་ལ་དྲི་མ་མེད་པར་ཡང་འགྱུར་ན་འགྱུར་བའི་ཆོས་ཅན་ཡིན་པས་ཇི་ལྟར་མི་རྟག་པར་མི་འགྱུར་ཞེ་ན། དེ་ནི་འདི་
ལྟར། ཆུ་ཁམས་གསེར་དང་ནམ་མཁའ་རྣམས། །དག་པ་བཞིན་དུ་དག་པར་འདོད། །གློ་བུར་གྱི་དྲི་མ་དང་བྲལ་བ་ལ་བྱའི་དེའི་ངོ་བོ་ཉིད་གཞན་དུ་འགྱུར་བ་ནི་མ་ཡིན་ནོ། །གཞན་ཡང་རབ་ཏུ་དབྱེ་བ་ནི་སྟོང་པ་ཉིད་རྣམ་པ་བཅུ་དྲུག་པོ་འདི་ལྟ་སྟེ། ནང་སྟོང་པ་ཉིད་དང་། ཕྱི་སྟོང་པ་ཉིད་དང་། ཕྱི་ནང་སྟོང་པ་ཉིད་དང་། ཆེན་པོ་
སྟོང་པ་ཉིད་དང་། སྟོང་པ་ཉིད་སྟོང་པ་ཉིད་དང་། དོན་དམ་པ་སྟོང་པ་ཉིད་དང་། འདུས་བྱས་སྟོང་པ་ཉིད་དང་། འདུས་མ་བྱས་སྟོང་པ་ཉིད་དང་། མཐའ་ལས་འདས་པ་སྟོང་པ་ཉིད་དང་། ཐོག་མ་དང་ཐ་མ་མེད་པ་སྟོང་པ་ཉིད་དང་། དོར་བ་མེད་པ་སྟོང་པ་ཉིད་དང་། རང་བཞིན་སྟོང་པ་ཉིད་དང་། ཆོས་ཐམས་ཅད་སྟོང་པ་ཉིད་
