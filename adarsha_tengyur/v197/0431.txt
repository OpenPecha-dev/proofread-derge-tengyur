སེ་བ་ཎཾ་སཱི་བ་ཎཾ་སྱུ་བིརྦེ་ད་རཿསྤུ་ཊ་ནཾ་བྷི་དཱ། །ཨཱ་ཀྲོ་ཤཱ་ན་མ་བྷི་ཥཾ་གཿསེ་བོ་དྭོ་བེ་ད་ནཱ་ན་ནཱ། །ས་མཱུརྩྪ་ན་མ་བྷི་བྱཔྟི་པྟི་ཪྻཱཙྙཱ་བྷི་ཀྵཱརྛ་ནཱ྅རྡྷ་ནཱ། །བརྡྷ་ནཾ་ཙྪེ་ད་ནེ྅ཐ་དྭེ་ཨཱ་ནནྡ་ན་

ས་བྷཱ་ཛ་ནེ། །ཨཱ་པྲྀཙྪ་ཎ་མ་ཐཱམྣཱ་ཡཿསཾ་པྲ་དཱ་ཡཿཀྵ་ཡེ་ཀྵི་ཡཱ། །གྲ་ཧེ་གྲཱ་ཧོ་བ་ཤཿཀཱནྟཽ་རཀྵྞུསྟྲཱ་ཎེ་ར་ཎཿཀྭ་ཎེ། རྒྱུ་དང་མི་གསལ་སྨྲའོ། །བྱ་དྷོ་བེ་དྷི་པ་ཙཱ་པཱ་ཀེ་ཧ་བོ་ཧུ་ཏཽ་བ་རོ་བྲྀ་ཏཽ། །ཨོ་ཥཿཔློ་ཥེ་ན་ཡོ་ནཱ་
དང་། །ཁམས་བདེ་མི་བདེ་འདྲི་བའོ། །དེ་ནས་མན་ངག་སྡོམ་པའོ། །ཟད་པ་དང་ནི་རྫོགས་པའོ། །ལེན་པ་དང་ནི་བླངས་པའོ། །མདངས་ཅན་དང་ནི་འཚེར་བའོ། །བསྲུང་བ་དང་ནི་སྐྱབས་པའོ། །འབུགས་པ་འབིགས་པ་ཚོས་པ་སྨིན། །འབད་པ་དང་ནི་སྦྲོན་པའོ། །མཆོག་དང་རབ་དང་སྲེག་
ཡེ་ཛྱཱ་ནི་རྫཱིརྞྞཽ་བྷྲ་མོ་བྷྲ་མཽ། །སྥཱ་ཏི་རྦྲྀདྡྷཽ་པྲ་ཐཱ་ཁྱཱ་ཏཽ་སྤྲྀཥྚིཿཔྲྀཀྟཽ་སྲ་བཿསྲཱ་བེ། །མྲྀ་དྷཱ་ས་མྲྀ་དྷྭོ་སྥ་ར་ཎཾ་སྥུ་ར་ཎེ་པྲ་མི་ཏཽ་པྲ་མཱ། །པྲ་སཱུ་ཏི་པྲ་ས་རེ་ཤྩྱོ་ཏེ་པྲཱ་གྷཱ་རཿཀླ་མ་ཐཿཀླ་མེ། །ཨུཏྐཪྵོ྅་
དང་ཚིག །མདུན་མ་གྲོས་དང་རྒས་འཁོགས་པ། །ནོར་འཁྲུལ་རྒྱས་དང་འཕེལ་བའོ། །རབ་ཏུ་གྲགས་དང་གྲགས་པའོ། །རེག་པ་དང་ནི་འབྱར་བའོ། །འབབ་ལྟུང་འཕེལ་བ་ཡང་དག་འཕེལ། །འཕྲོ་དང་སྤྲོ་བ་ཚད་མ་ཚ། །རབ་སྐྱེས་འབུང་དང་ལྟུང་དང་ལྟུང་། །ངལ་བ་དང་ནི་དུ་བ་པའོ། །ཤིན་ཏུ་བ་དང་
