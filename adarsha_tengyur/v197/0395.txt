ཀ་ཡོ་བི་ཤྭཾ་ན་ག་རཾ་བི་ཤྭ་བྷེ་ཥ་ཛཾ། །ཨཱ་ར་ཎཱ་ལ་ཀ་སཽ་བཱི་ར་ཀུལྨཱ་ཥ་བཱི་ཥུ་ཏཱ་ནི་ཙ། །ཨ་བནྟི་སོ་ཥ་བཱ་ནྱཱམླ་ཀུཉྫ་ལཱ་ནི་ཙ་ཀཱཾ་ཛི་ཀཾ །ས་ཧ་ཤྲ་བེ་དྷཱི་ཛ་ཏུ་ཀཾ་བཱ་ཧླཱི་ཀཾ་ཧིངྒུ་རཱ་མ་ཋཾ། །ཏཏྤ་ཏྲཱི་ཀཱ་ར་བཱི་པྲྀཥྛཱི་བཱ་ཥྤི་ཀཱ་ཀ་ར་བཱི་པྲྀ་ཐུཿ། །
༄༅། །ཐམས་ཅད་ལ། །གྲོང་སྐྱེས་མ་ནིང་བི་ཤྭ་སྨན། །རྩབས་མོ་སྐྱུར་མོ་ཀུལྨཱ་ཥ། །ཐུག་སྐྱུར་སྐྱུར་རྩི་ཨམ་ལ་དང་། །སྐྱུར་སྦྱར་དང་ནི་ཀཱཾ་ཛི་ཀཾ །སྟོང་འབིགས་ཤིང་ཀུན་མེ་ལྡན་དང་། །ཧིངྒུ་དང་ནི་ཤིང་ཀུན་ནོ། །དེ་ཡི་ལོ་མ་གཱ་ར་བཱི། །ཟི་ར་ནག་པོ་བཱཥྤི་ཀཱ། །ཀ་ར་བཱི་
ནི་ཤཱ་ཧྭཱ་ཀཱཉྩ་ནཱི་པཱི་ཏཱ་ཧ་རི་དྲཱ་བ་ར་བརྞྞི་ནཱི། །སཱ་མུ་དྲཾ་ཡཏྟ་ལ་བ་ཎ་མ་ཀྵཱི་བཾ་བ་སི་རཉྩ་ཏ་ཏ། །སཻནྡྷ་བོ྅སྟྲཱི་སཱི་ཏ་ཤི་བཾ་མཱ་ཎི་མནྠ་ཉྩ་སིནྡྷུ་ཛེ། །རཽ་མ་ཀེ་བ་སུ་ཀ་པཱ་ཀྵཾ་བི་ཌཉྩ་ཀྲྀ་ཏ་ཀེ་དྭ་ཡཾ། །སཽ་བརྩ་ལེ྅ཀྵ་རཱུ་ཙ་ཀེ་
དང་རྒྱ་ཆེན་ནོ། །མཚན་མོའི་མིང་ཅན་ཡུང་བ་སེར། །ཧ་རི་དྲཱ་དང་ཡུང་བའོ། །རྒྱ་མཚོ་གང་ཞིག་ལན་ཚྭ་ཚ། །དཔྱིད་དུས་ཆགས་པ་དེ་དང་དེ། །མཚོ་སྐྱེས་མོ་མིན་དཀར་ཞིང་ཞི། །མཱ་ཎི་མནྠ་སིནྡྷུ་སྐྱེས། །རོ་མ་ཀེ་དང་སྦྱར་ཚའོ། །སྨིན་ཚ་ཁ་རུ་སྦྱར་ཚ་གཉིས། །ཚ་ནི་མི་ཟད་རཱུ་ཙ་ཀེ། །མར་
ཏི་ལ་ཀཾ་ཏ་ཏྲ་མེ་ཙ་ཀེ །མཏྶྱཎྜཱི་ཕཱ་ཎི་ཏཾ་ཁཎྜ་བི་ཀཱ་རཽ་ཤརྐ་རཱ་སི་ཏཱ། །ཀཱུ་ཙི་ཀཱ་ཀྵཱི་ར་བི་ཀྲྀ་ཏིཿསྱཱ་དྲཱ་སཱ་ལཱ་ཏུ་མཱ་རྫི་ཏཱ། །སྱཱཏྟེ་མ་ནནྟུ་ནིཥྛཱ་ན་ཏྲི་ལིཾ་གཱ་པཱ་སི་ཏཱ་བ་དྷེཿ། །ཤུ་ལཱ་ཀྲྀ་ཏཾ་བྷ་ཊི་ཏྲཉྩ་ཤཱུ་ལྱ་མུ་ཁྱཾ་ཏུ་པཻ་ཋ་རཾ། །པྲ་ཎཱི་ཏ་མུ་པ་སཾ་
ཁུ་ལས་བྱུང་མེ་ཙ་ཀེ །ཀ་ར་སྦུ་བ་ཧྭགས་དང་ནི། །ཤརྐ་ར་དང་ཀ་ར་དཀར། །ཕྲུམ་དང་འ་ལས་སྦྱར་བའོ། །ར་ས་མ་ནི་མཉེས་པའོ། །ཚོད་མ་དང་ནི་ནིཥྛཱ་ནི། །རྟགས་གསུམ་བཱ་སི་ཡན་ཆོད་དོ། །རྩེ་བྱས་བསྲེགས་ཤ་ཤཱུ་ལྱའོ། །བཙོས་པའི་ཤ་ནི་པཻ་ཋ་རཾ། །གྱ་ནོམ་པ་དང་ཕུན་
