རཱི་པྲྀ་ཥཱ། །པྲ་ཏྱཀ་ཤེ་ནཱི་སུ་ཏ་ཤེ་ཎཱི་རཎྜཱ་མཱུཥི་ཀ་པརྞྞྱ་པི། །ཨ་བཱ་མཱ་རྒཿཤེ་ཁ་རི་ཀོ་དྷཱ་མཱརྒྒ་བ་མ་ཡཱུ་ར་ཀཽ། །བ་ཏྱཀྤརྞྞཱི་ཀཱི་ས་པརྞྞཱི་ཀཱི་ཎི་ཧཱི་ཁ་ར་མཾ་ཛ་རཱི། །བཾ་ཛི་ཀཱ་པྲ་ཧྨ་ཎཱི་པདྨཱ་བྷཱརྒྒཱི་པྲ་ཧྨ་ཎ་ཡཥྚི་ཀཱཿ། །ཨཾ་གཱ་ར་བ་ལྨཱི་པཱ་ལེ་ཡ་ཤཱ་ཀ་བརྦྦ་
དང་ནི་ཆར་འབབ་དང་། །པྲ་ཏྱཀ་ཤེ་ནཱི་སུ་ཏ་ཤེ་ཎཱི། །རཎྜ་བྱི་བའི་རྣ་བའོ། །ཨ་པཱ་མཱརྒ་ཐོད་ཅན་དང་། །དྷཱ་མཱརྒ་དང་རྨ་བྱ་ཅན། །པྲ་ཏྱ་ཀ་ལོ་མ་ཀཱི་སའི་འདབ། །ཀཱི་ཎི་ཧི་དང་བོང་བུའི་སྐེ། །ཕནྫི་ཀ་དང་ཚངས་མ་དང་། །པདྨ་དང་ནི་བྷཱརྒི་དང་། །ཚངས་པའི་མཆོད་སྡོང་མེ་མདག་ལོ། །བ་ལེ་ཡ་
ར་བརྡྷ་ཀཱ། །མཾ་ཛིཥྛཱ་བི་ཀ་སཱ་ཛིངྒཱི་ས་མཾ་གཱ་ཀཱ་ལ་མེ་ཥི་ཀཱ། །མཎྜུ་ཀ་པརྞྞཱི་བྷཎྜཱི་རཱི་བྷཎྜཱི་ཡོ་ཛ་ན་པརྞྱ་པི། །ཡཱ་སོ་ཡ་བཱ་སོ་དུཿསྤཪྴོ་དྷ་ནྭ་ཡཱ་སཿཀུ་ཎཱ་ས་ཀཿ། །རོ་ད་ནཱི་ཀཙྪུ་རཱཉནནྟཱ་ས་མུ་དྲཱནྟཱ་དུ་རཱ་ལ་བྷཱ། །བྷྲྀཤྣི་པརྞྞཱི་པྲྀ་ཐ་
ཤཱ་ཀ་བརྦ་ར་བརྡྷ་ཀཱ། །བཙོད་དང་ལེ་བརྒན་བི་ཀ་སཱ། །ཛིངྒཱི་ཡན་ལག་མཚུངས་པ་དང་། །སྤལ་པའི་འདབ་མ་བྷཎྜཱི་རཱི། །བྷཎྜཱི་དཔག་ཚད་ལོ་མ་ལའང་། །གཟེ་མ་ར་མགོ་ཡ་བཱ་སོ། །རེག་བྱ་སྡུག་དང་ཆུ་མེད་སྐྱེ། །ཀ་ཎ་ས་ཀ་དུ་བྱེད་ཤུ། །མཐའ་མེད་མཚོ་མཐའོ་རག་པར་དཀའ། །དྷྲྀཤྣི་ལོ་མ་ཐ་དད་
ཀྤརྞྞཱི་ཙི་ཏྲ་པརྞྞྱཾ་གྷྲི་བལླ་ཀཱ། །ཀྲོཥྚུ་བེནྣཱ་སིཾ་ཧ་པུཙྪཱི་ཀ་ལ་ཤིརྡྷཱ་བ་ནི་རྒུ་ཧཱ། །ནི་དིགྡྷི་ཀཱ་སྤྲྀ་ཤཱི་བྱཱ་གྷྲཱི་བྲྀ་ཧ་ཏཱི་ཀཎྜ་ཀཱ་རི་ཀཱ། །པྲ་ཙོ་ད་ནཱི་ཀུ་ལཱི་ཀྵུ་དྲཱ་དུཿསྤཪྴོ་རཱ་ཥྚྲཱི་ཀ་ཏྱ་བི། །ནཱི་ལཱི་ཀཱ་ལཱ་ཀཱི་ཏ་ཀི་ཀཱ་གྲཱ་མཱི་ནཱ་མ་དྷུ་པརྞྞི་ཀཱ། །
ལོ། །སྣ་ཆོགས་འདབ་རྐང་བལླི་ཀཱ། །ཀྲོཥྚུ་བེལླཱ་སེང་གེ་མཇུག །ཀ་ལ་ཤརྡྷི་བ་ནི་རྒུ་ཧཱ། །ནི་དི་ག་བི་ཀཱ་སྤྲི་ཤཱི་དང་། །སྟག་མོ་དང་ནི་བྲི་ཧ་ཏཱི། །ཀཱཎྜ་ཀ་རི་པྲ་ཙོ་ད་ནཱི། །ཀུ་ལི་ཆུང་བ་རེག་དཀའ་དང་། །ཡུལ་སྐྱེས་ཀྱང་ངོ་རམས་ཤིང་ནག །ཀླཱི་ཏ་ཀི་ཀཱ་གྲོང་སྐྱེས་དང་། །སྦྲང་འདབ་ཚོས་བྱེད་
