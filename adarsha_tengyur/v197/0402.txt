ཙ་ཏུཥྚ་ཡཾ། །སུ་བརྞྞཱ་བིསྟཽ་ཧེ་མྣོ་ཀྵེ་ཀུ་རུ་བིསྟསྟུ་ཏཏྤ་ལེ། །ཏུ་ལཱ་སྟི་ཡཱཾ་པ་ལ་ཤ་ཏཾ་བྷཱ་ར་སྱཱདྦིཾ་ཤ་ཏིསྟུ་ལཱཿ། །ཨཱ་ཙི་ཏཾ་ད་ཤ་བྷ་རཱཿསྱུཿསཱ་ཀ་ཊོ་བྷ་ར་ཨཱ་ཙི་ཏཿ། །ཀཪྵཱ་པ་ཎཿཀཪྵིཀཿསྱཱཏྐཱ་ཪྵི་ཀེ་ཏཱ་མྲྀ་ཀེ་པ་ཎཿ། །ཨསྟྲི་ཡཱ་མཱ་ཌྷ་ཀདྲོ་ཎཽ་
ངོ་མོ་མིན་ནོ། །གསེར་འཇམ་ཞོ་ཆ་གསེར་མིག་གོ། །ཀུ་རུ་བསྟི་དེ་ཡི་སྲང་། །མཉམ་བྱེད་མོ་མིན་སྲང་བརྒྱ་ལ། །ཁུར་ཡིན་ཉི་ཤུ་ཐམ་པ་ནི། །ཨཱ་བི་ཏཾ་ཟེར། །སྲང་ངོ་ཨཱ་ཙི་ཏཾ་ཁུར་བཅུ། །ཤིང་རྟ་ཁུར་རྟ་ཨཱ་ཙི་ཏ། །ཀར་ཥཱ་པ་ཎ་དོང་ཚེ་ཡིན། །ཟངས་མའི་དོང་ཚེ་ཀཱཪྵི་ཀཻ །མོ་མིན་བྲེ་བོ་ཨཱ་ཌྷ་ཀ །བྲེ་དང་ཁཱ་རི་
ཁཱི་རཱི་བ་ཧོ་ནི་ཀུཉྩ་ཀཿ། །ཀུ་ཌ་བཿཔྲསྠ་ཨི་ཏྱཱ་དྱཱ་པ་རི་མཱ་ཎཱརྠ་ཀཱཿཔྲྀ་ཐ་ཀ །པཱ་དསྟུ་རཱི་ཡོ་བྷཱ་གཿསྱཱ་དཾ་ཤ་བྷཱ་གོ་ཏུ་བཎྚ་ཀེ །དྲབྱཾ་བིཏྟ་སྭཱ་ཏེ་ཡཾ་རྀཀྠ་མྲཀྠ་དྷ་ནཾ་བ་ཙ་སུ། །ཧི་ར་ཎྱཾ་དྲ་བ་ན་དྱུསྣ་མརྠ་རེ་བི་བྷ་བཱ་ཨ་པི། །སྱཱཏྐོ་
བཱ་ཧ་ནི། །ནི་ཀུཉྩ་ཀ་ཞེས་པའོ། །ཀུ་ཌ་བ་ནི་བྲེ་ལ་སོགས། །ཡོངས་སུ་འཇལ་བྱེད་སོ་སོའོ། །རྐང་པ་དང་ནི་གཅིག་ཆ་ཡིན། །ཆ་དང་ཆ་ཤས་འགོད་པའོ། །རྫས་དང་ཡོ་བྱད་ནོར་དང་ནི། །འབྱོར་པ་དངོས་པོ་དཀོར་དང་རྫས། །དབྱིག་དང་ཟང་ཟིང་ལོངས་སྤྱོད་དོན། །ནོར་དང་དངོས་པོ་ཉིད་ཀྱང་ངོ་། །མཛོད་
ཥཤྩ་ཧི་ར་ཎྱཉྩ་ཧེ་མ་རུ་པྱ་ཀྲྀ་ཏཱ་ཀྲྀ་ཏེ། །ཏཱ་བྷྱཱ་མ་ཡ་དུ་ནྱ་ཏུཏྐུ་པྱཾ་ཏ་དྡྭ་ཡ་མཱ་ཧ་ཏཾ། །གཱ་རུཏྨ་ཏཾ་མ་རཀ་ཏ་མསྨ་གརྦྷཾ་ཧ་རིནྨ་ཎིཿ། །སོ་ཎ་རཏྣཾ་ལོ་ཧི་ཏ་ཀཿཔདྨ་རཱ་གོཉཐ་མོཀྟི་ཀཾ །མུཀྟ྅ཐ་བི་དྲུ་མཿཔུཾ་སི་
དང་རིན་ཆེན་གསེར་གཟུགས་དང་། །སྒྲོམ་དང་སྣོད་ཅེས་བྱ་བའོ། །དེ་དང་གང་ཞིག་གཞན་མཛོད་དང་། །དངུལ་དང་དེ་གཉིས་མ་ཧ་ཏཾ། །མཁའ་ལྡིང་གཤོག་པ་མར་གད་དང་། །རྡོ་ཡི་སྙིང་པོ་ནོར་བུ་ལྗང་། །རིན་ཆེན་དམར་པོ་ལོ་ཧི་ཏ། །པདྨ་རཱ་ག་དེ་ནས་ནི། །མུ་ཏིག་དང་ནི་མུཀྟའོ། །དེ་ནས་
