པ་ལ། ནུཀ྄་སྟེར་བ་སོགས་ཆོ་ག་ཐམས་ཅད་སྔ་མ་དང་འདྲ་བར་བསྒྲུབས་པས། པ་རསྨཻ་པ་ད་རྣམས་ལ། ཨཱ་ནཾ་ཛ། ཨཱ་ནཾ་ཛ་ཏུཿ ཨཱ་ནཾ་ཛུཿ ཨཱ་ནཾ་ཛི་ཐ། ཨཱ་ནཾ་ཛ་ཐུཿ ཨཱ་ནཾ་ཛ། ཨཱ་ནཾ་ཛ། ཨཱ་ནཾ་ཛི་བ། ཨཱ་ནཾ་ཛི་མ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ཨཱ་ནཾ་ཛེ།
ཨཱ་ནཾ་ཛཱ་ཏེ། ཨཱ་ནཾ་ཛི་རེ། ཨཱ་ནཾ་ཛི་ཥེ། ཨཱ་ནཾ་ཛཱ་ཐེ། ཨཱ་ནཾ་ཛི་དྷྭེ། ཨཱ་ནཾ་ཛེ། ཨཱ་ནཾ་ཛི་བ་ཧེ། ཨཱ་ནཾ་ཛི་མ་ཏེ། ཉེ་བའི་ཚིག་དང་སྦྱར་བའི་དཔེ་ནི། ཨཱ་ནཾ་ཛཱཀྵི་ཎཱི་ཀཛྫ་ལེ་ན་ཀཱ་ཙིཏྲ། ཁ་ཅིག་མིག་སྨན་གྱིས་མིག་ལ་བྱུགས་སོ། ༈ དེ་བཞིན་དུ་ཨརྩྲ་མཆོད་པ་བྱེད་པ་ལའང་བདག་དོན་དང་གཞན་དོན་གཉིས་ཀ་འཇུག །བྱིངས་
མཐའི་ཨ་དེའི་དོན། ནུཀ྄་སྟེར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས་པ་རསྨཻ་པ་ད་རྣམས་ལ། ཨཱ་ནརྩ། ཨཱ་ནརྩ་ཏུཿ ཨཱ་ནརྩུཿ། ཨཱ་ནརྩི་ཐ། ཨཱ་ནརྩྠ། ཁ་ཅིག་མིག་སྨན་གྱིས་མིག་ལ་བྱུགས་སོ། །ཨཱ་ནརྩ་ཐུཿ ཨཱ་ནརྩ། ཨཱ་ནརྩ། ཨཱ་ནརྩི་བ། ཨཱ་ནརྩི་མ། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ཨཱ་
ནརྩེ། ཨཱ་ནརྩཱ་ཏེ། ཨཱ་ནརྩི་རེ། ཨཱ་ནརྩི་ཥེ། ཨཱ་ནརྩཱ་ཐེ། ཨཱ་ནརྩི་དྷྭེ། ཨཱ་ནརྩེ། ཨཱ་ནརྩི་བ་ཧེ། ཨཱ་ནརྩི་མ་ཧེ། ༈ ཙིཉ་གསོག་པ་ལའོ་ཞེས་པ་ལ་པ་རསྨཻ་པ་ད་རྣམས་སྦྱོར་བ་ལས། རྐྱེན་ས་དང་ལྐོག་གྱུར་གྱི་རྣམ་དབྱེ་ཕ་རོལ་ལ་བྱིངས་ཙིཉྲ་ནི་ཡང་ན་ཀི་རུ་བསྒྱུར་བར་
བཤད་དེ། ཀི་དང་ཙི་འདམ་ཁ་ཅན་དུ་བྱས་ཀི་རུ་བསྒྱུར་བའི་ཕྱོགས་ལ་ཉེར་བསྒྱུར་ནིརྲ་སྔོན་དུ་བཞག །དེ་ནས་འཕེལ་བ་ཡོན་ཏན་ཨིཡྲ་སོགས་སོ་སོའི་ཐོབ་ལྟར་བསྒྲུབས་པས། ནིཤྩི་ཀཱ་ཡ། ནིཤྩི་ཀྱ་ཏུཿ ནིཤྩི་ཀྱུཿ ནིཤྩི་ཀ་ཡི་ཐ། ནིཤྩི་ཀེ་ཐ། ནིཤྩི་ཀྱ་ཐུཿ ནིཤྩི་
ཀྱ། ནིཤྩི་ཀཱ་ཡ། ནིཤྩི་ཀ་ཡ། ནིཤྩི་ཀྱི་བ། ནིཤྩི་ཀྱི་མ། ཀི་རུ་མ་བསྒྱུར་བའི་ཕྱོགས་ལ། ཙི་ཙཱ་ཡ། ཙི་ཙྱ་ཏུཿ སོགས་དང་། ཉེ་བའི་ཚིག་དང་སྦྱར་བའི་དཔེ་ནི། ནིཤྩི་ཀཱ་ཡ་ཙི་རཾ་རཱ་ཛཱ། གྲཧ་ཉེ་བར་ལེན་པ་ལའང་པ་རསྨཻ་པ་ད་རྣམས་སྦྱོར་བ་ལས། གཉིས་
བརྗོད། ཀུ་ཧ་དག་གི་ཙུ། ཀ་འགྲོ་བ་སོགས་ཀྱི་སཾ་པྲ་སཱ་ར་ཎར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཛ་གྲཱ་ཧ། ཛ་གྲྀ་ཧ་ཏུཿ ཛ་གྲྀ་ཧུཿ ཛ་གྲྀ་ཧི་ཐ། ཛ་གྲ་ཧི་ཐ། ཛ་གྲྀ་ཧ་ཐུཿ རིང་དུ་གནས་པར་གྱུར་ཅིག་རྒྱལ་པོ། ཛ་གྲྀ་ཧ། ཛ་གྲཱ་ཧ། ཛ་གྲ་ཧ། ཛ་གྲྀ་ཧི་བ། ཛ་གྲྀ་ཧི་ན།
