མཚོན་ན། ཨ་བ་ཤྱ་བྷཱ་བྱཿ བྷོ་ཀྟུ་ཀཱ་མཿ ཤྲོ་ཏུ་མ་ནཱཿ སི་ཧཱི་ཏཿ ས་ཧི་ཏཿ སཾ་ཏ་ཏཾ། ས་ཏ་ཏྲཾ། མཱཾ་ས་བ་ཅ་ནཾ། མཱཾ་སྤ་ཅ་ནཾ། མཱཾ་ས་པཱ་ཀཿ མཱཾ་སྤཱ་ཀ་སོགས་སོ། ༈ །རྀ་དུ་བ་ངྷཱ་ཏྐྱ་བ། རྀ་ཨུ་བ་ངྷཱ་ལས་ཀྱ་བའོ། །བྱིངས་རྀ་ཡིག་ཨུ་པ་ངྷཱ་ཙ་ན་ལས་དངོས་པོ་དང་ལས་
ཀྱི་དོན་ལ་རྐྱེན་ཀྱ་པྲ་འགྱུར་ཏེ། ཀྲྀ་ཏཱི་ངེས་པར་བདེན་ནམ་ཡོད། ལོངས་སྤྱོད་འདོད། ཐོས་པ་ཡིད་ལ་བྱེད། གྲོགས་བཟང་པོའམ་ལྷན་ཅིག་པ། དེ་ལེགས་པ། ཤ་འཚེད་པར་བྱེད། གཅོད་པ་ལ་ཀྱ་པྲ ཤ་ཆོས། སྦྱར་བས། ཀྲྀ་ཏྱཾ། ཡང་ཥྚུ་ཉྲ་སོགས་ལའང་རྐྱེན་ཀྱ་པྲ་འཇུག་པར་བཤད་དེ། །ཥྚུ་ཉྲ་དང་ཀྲྀ་ལ་ཀྱ་པྲ་
ཐུང་ངུ་སོགས་ཀྱི་པྲ་འགྲོ་བ་ཏུ་ཀྲ་བྱིན་པས། སྟུ་ཏྱཾ། ཀྲྀ་ཏྱཾ་སོགས་སོ། ༈ །སྟྭ་ཡཱཾ་ཡ་ཛ་ངྷཱ་བེ། ཡ་ཛ་རྣམས་ཀྱི་དངོས་པོ་ལ་བུད་མེད་ལའོ། །བྱིངས་ཡ་ཛུ་ལ་སོགས་པ་རྣམས་ལས་དངོས་པོའི་དོན་ལ་རྐྱེན་ གཅད་པ། ཀྱ་པྲ་འགྱུར་ཞིང་རྟགས་དེ་ཡང་བུད་མེད་ཀྱི་རྟགས་སུ་འགྱུར་བ་ཉིད་དེ། པྲ་སྔོན་མའི་ཡ་ཛ་ལ་
ཀྱ་པྲ། ཡ་ཛྲ་སོགས་ཀྱི་བསྟོད་པ་བྱའོ། །སླར་ཡང་དག་པར་བསྒྱུར། བུད་མེད་ལ་ཨཱཔྲ་སྦྱར་བས། པྲེ་ཛྱཱ། དེ་བཞིན་དུ་སམྲ་སྔོན་མའི་ཨཛྲ་འགྲོ་བ་ལ། ཨཱཔྲ་ ས་མ་ཛྱཱ། བྲཧྨ་སྔོན་མའི་ཧ་ནྲ་ལ་ཀྱ་པྲ་ཏ་གྷ་ཀ་འགྲོ་བ་ལ་ཨ་ནུ་དཱ་ཏྟ་ཕྱིས། ཐུང་ངུ་སོགས་ཀྱིས་ཏུ་ཀྲ་བྱིན་པས། བྲཧྨ་གྷ་ཏ་སོགས་སོ། ༈ །
ཀྟིཿ ཀྟིའོ། །བྱིངས་ལས་དངོས་པོ་སོགས་ཀྱི་དོན་ལའང་ཀྟི་འགྱུར་བའི་དེ་ཡང་རང་བཞིན་གྱི་མོ་རྟགས་ཏེ། རབ་ཏུ་མཆོད། །བྷཱུ་ཏིཿ ཤུངྔྷིཿབཧྲ་ ཨཱུ་ངྷིཿཀི་ པཾཀྟིཿཔིཙ་ ཡང་དག་པར་འགྲོ་མ། བིཏྟིཿསོགས་སོ། ༈ །ཀརྟ་རིཀྟེ་ཤྩ། བྱེད་པ་པོ་ལའང་ཀྟིའོ། །བྱིངས་ལས་བྱེད་པ་པོའི་དོན་ལའང་ཚངས་འཇོམས། རྐྱེན་
ཀྟི་འགྱུར་ཏེ། པྲ་ཀྲྀ་ཏིཿ པི་ངྷྀ་ཏི་སོགས་སོ། ༈ །ཨི་ཀ྄་ཤྟི་པཽ་ངྷཱ་ཏུ་ནི་ཛ་ཤཿ ཨི་ཀྲ་ཤྟི་པྲ་དག་ལས་བྱིངས་ལ་རྣམ་འགྱུར་ འབྱོར་བ། དགབ། ཐོབ་པ། འཕྲེང་བ། མེད་དོ། །བྱིངས་ལས་རིག་པ། དངོས་པོ་སོགས་ཀྱི་དོན་ལ་རྐྱེན་ཨིཀ྄་དང་ཤྟིཔྲ་དག་འགྱུར་ལ་རྐྱེན་དེ་གཉིས་ཕ་རོལ་ལས་བྱིངས་ལ་རྣམ་འགྱུར་
ཡང་མེད་དེ། རང་བཞིན། ཁྱབ་པར་འཛིན་པ། པ་ཙ་ལ། ཨིཀ྄་དང་ཤྟིཔ྄། ཤ྄་ཡིག་ཨ་པྲ་བི་ཀཱ་ར་ཎའི་དོན་ཏེ། བ་ཙིཿ འཚེད་དོ། །པ་ཅཏིཿ འཆེད་བར་བྱེད། སོགས་སོ། ༈ །ཥིངྦྷི་དཱ་མ་དྲ། ཥ་འགྲོ་བ་དང་བྷི་དྲ་རྣམས་ཀྱེ་ཨངྲ་ངོ་། །བྱིངས་ཥྲ་ཡིག་རྗེས་འབྲེལ་འགྲོ་བ་རྣམས་དང་བྷི་དྲ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་རྐྱེན་ཨངྲ་
