ར་ཀཿ། །རཱ་ཛཱ་བྷཊྚ་ར་ཀོ་དེ་བསྟཏྶུ་ཏཱ་བྷ་ཏྲྀ་དཱ་རི་ཀ །དེ་བཱི་ཀྲྀ་ཏཱ་བྷི་ཥེ་ཀཱ་ཡཱ་མི་ཏ་ར་སུ་ཙ་བྷཊྚི་ནཱི། །ཨ་བྲཧྨ་ཎྱ་མེ་བ་དྷྱོཀྟཽ་རཱ་ཛ་ཤྱ་ལསྟུ་རཱསྟི་ཡཿ། །ཨམྦ་མཱ་ཏཱ་ཐ་བཱ་ལ་ཤྱདྦཱ་སཱུ་རཱཪྻ་སྟུ་མཱ་རི་ཥཿ། །ཨནྟི་ཀཱ་བྷ་གི་ནཱི་ཛྱེཥྛཱ་
ཀ་ཞེས་སོ། །རྒྱལ་པོ་བྷཊྚཱ་ར་ཀོ་ལྷ། །དེའི་བུ་མོ་བྷ་ཏྲྀ་དཱ་རི་ཀཱ། །ལྷ་མོ་གང་ལ་དབང་བསྐུར་བ། །གཞན་རྣམས་ལ་ནི་བྷཊྚི་ནཱི། །ཨ་བྲཧྨ་ཎྱ་གསོད་མི་བྱེད། །རཱ་ཛ་ཤྱ་ལ་མངའ་རིས་སོ། །ཨམྦ་མའོ་དེ་ནས་ནི། །བཱ་སུ་ར་ནི་བྱིས་པའོ། །འཕགས་པ་དག་ནི་མཱ་རི་ཥ། །སྲིང་མོ་ཆེ་བ་ཨནྟི་ཀཱ །ནིཥྛཱ་
ནིཥྛཱ་ནི་ནིརྦ་ཧ་ཎཾ་ས་མེ། །ཧཎྜེ་ཧཉྫེ་ཧ་ལཱ་ཧྭཱ་ནཾ་ནཱི་ཙཾ་ཙེ་ཊཱི་སཱ་ཁཱིམྤྲ་ཏི། །ཨཾ་ག་ཧཱ་རོཉངྒ་བི་ཀྵེ་པོ་བྱཉྫཀཱ་བྷི་ན་ཡཽ་ས་མོ། །ནི་བྲྀཏྟེ་ཏྭངྒཱ་ས་ཏྭཱ་བྷྱཱཾ་དྭེ་ཏྲྀ་ཥྭཾ་གི་ཀ་ས་ཏྭི་ཀེ། །ཤྲྀ་གཱ་ར་བཱི་ར་ཀ་རུ་ཎཱདྦྷུ་ཏ་
ཟིན་པ་ཡིན་ཏེ་མཚུངས། །ཧཎྜེ་ཧཉྫེ་ཧ་ལ་འབོད། །དམན་དང་གཡོག་མོ་གྲོགས་མོ་སླར། །ཨཾ་ག་ཧཱ་རོ་ཡན་ལག་བསྐྱོད། །བྱཉྫ་ཀ་དང་ཚུལ་དག་མཚུངས། །ཟིན་པའི་ཡན་ལག་དག་ནི་གཉིས། །གསུམ་ནི་ཨཾ་གི་ཀ་དང་ནི། །ས་ཏྭི་ཀེ་ཞེས་མི་དྲན་པའོ། །སྒེག་དང་དཔའ་དང་སྙིང་རྗེ་དང་། །རྔམ་དང་
ཧཱ་སྱ་བྷ་ཡཱ་ན་ཀཿ། །པཱི་བྷཏྶ་རཽ་དྲེ་ཙ་ར་སཱཿཤྲྀཾ་གཱ་རཿཤུ་ཙི་རུ་ཛྭ་ལཿ། །ཨུཏྶཱ་ཧ་བརྡྷ་ནོ་བཱི་རཿཀཱ་རུ་ཎྱཾ་ཀ་རུ་ཎ་གྷྲྀ་ཎཱ། །ཀྲྀ་པཱ་ད་ཡ་བྱ་ནུ་ཀམྦྷཉབྱ་ནུ་ཀྲོ་ཤོཉཔྱ་ཐོ་ཧ་སཿ། །ཧཱ་སོ་ཧཱ་སྱཉྩ་བཱི་བྷཏྶཾ་བི་ཀྲྀ་ཏཾ་ཏྲི་ཥྭི་དཾ་
དགོད་དང་འཇིགས་སུ་རུང་། །མི་སྡུག་པ་དང་དྲག་ཤུལ་ཉམས། །སྒེག་པ་གཙང་དང་འབར་བ་དང་། །སྤྲོ་བ་འཕེལ་བ་དཔའ་བ་ཡིན། །སྙིང་རྗེ་ཐུགས་རྗེ་བརྩེ་བ་དང་། །དབང་བརྩེ་བ་བྱེད་གཏོང་བརྩེ་བ་དང་རྗེས་སུ་བརྩེ། །རྗེས་སུ་ཁྲོ་བོ་ཉིད་ཀྱང་ངོ་། །དེ་ནས་དགོད་དང་བཞད་གད་མོ། །མི་སྡུག་མི་ལེགས་དག་ནི་
