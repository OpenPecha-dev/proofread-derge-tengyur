ལི་ཏཱརྠ་ཀཾ །ཨ་ཧི་ཏཱ་ན་པྲ་ཏྱ་བྷཱི་ཏ་སྱ་ར་ཎེ་ཡ་ན་མ་ཏི་ཀྲ་མཿ། །བཻ་ཏཱ་ལི་ཀཱ་བོ་དྷ་ཀ་རཱཤྩཱཀྲི་ཀཱ་གྷཱཎྚི་ཀཱརྠ་ཀཱཿ། །སྱུརྨྨཱ་ག་དྷཱསྟུ་མ་དྷུ་ཀཱ་བནྡི་ནཿསྟུ་ཏི་པཱ་ཋ་ཀཱཿ། །སཾ་སཔྟ་ཀཱསྟུ་ས་མ་ཡཱཏྶཾ་གྲཱ་མཱ་ད་ནི་བརྟྟི་ནཿ། །རེ་
འཇུག་པའི་དོན། །ཉམ་མི་ང་དང་འཇིགས་པ་དང་། །གཡུལ་དུ་འཇུག་ཚེ་མངོན་པར་དཔའ། །བཻ་ཏཱ་ལི་ཀ་གོ་བྱེད་པ། །འཁོར་ལོ་ཅན་དང་དྲིལ་བུ་ཅན། །སློང་མོ་པ་དང་རྒྱལ་རིགས་མཁན། །སྙན་པ་ཕྱག་བྱེད་བསྟོད་པ་ཀློག །གཡུལ་འཕྲོད་འཁྲུག་དུས་གཡུལ་ངོར་ཞུགས། །རྡུལ་ཚུབས་གཉིས་
ཎུརྡྭ་ཡོསྟྲི་ཡཱཾ་དྷཱུ་ལིཿ་པཱཾ་ཤུརྣཱ་ན་དྭ་ཡཽར་ཛཿ། །ཙུརྞྞཿ་ཀྵོ་དཿ་ས་མུཏྤིཉྫ་པཉྫི་ལཽ་བྷྲྀ་ཤ་མཱ་ཀུ་ལེ། །པ་ཏཱ་ཀཱ་བཻ་ཛ་ཡནྟཱི་སྱཱཏྐེ་ཏ་ནཾ་དྷྭ་ཛ་མསྟྲི་ཡཱཾ། །སཱ་བཱི་རཱ་ཤནྶ་ནཾ་ཡུདྡྷྭ་བྷཱུ་མི་ཡཱ་ཏི་བྷ་ཡ་པྲ་དཱ། །ཨ་ཧཾ་
པོ་མོ་ནི་རྡུལ། །རྡུལ་ཚུབས་གཉིས་ཏེ་ཚུབ་མ་དང་། །ཕྱེ་མ་དང་ནི་ཀྵོ་དའོ། །ཉེ་བར་འཁྲུགས་དང་འགྱེད་པ་དང་། །འཛིངས་པ་དང་ནི་ཕྱོགས་མེད་དང་། །བ་དན་རྒྱལ་མཚན་ཏོག་དང་ནི། །དྷྭ་ཛ་དང་ནི་མོ་མིན་ནོ། །ཀུན་ཏུ་བསྔགས་པའི་དཔའ་བོ་དེ། །འཁྲུག་པར་ཤིན་ཏུ་འཇིགས་རབ་སྟེར། །ང་
པཱུརྦྦ་མ་ཧཾ་པཱུརྦྦ་མི་ཏྱ་ཧཾ་པཱུརྦྦི་ཀཱསྟྲི་ཡཱཾ། །ཨ་ཧོ་པུ་རུ་ཥི་ཀཱ་དརྤྱཱ་དྱཱ་སྱཱཏྶཱཾ་བྷཱ་བ་ནཱཏྨ་ནི། །ཨ་ཧ་མ་ཧ་མི་ཀཱ་ཏུ་སཱ་སྱཱཏྤ་རསྤ་རཾ་ཡོ་བྷ་བ་ཏྱ་ཧཾ་ཀཱ་རཿ། །དྲ་བི་ཎནྟ་རཿ་ས་ཧོ་བ་ལ་སཽརྻྰ་ཎི་སྠཱ་མ་ཤུཥྨཉྩ། །སཀྟིཿ་པ་རཱ་
སྔ་ང་སྔ་ཞེས་སྒྲོགས་པ། །བདག་སྔ་བ་ནི་མོ་མ་ཡིན། །མཚར་ཅན་སྐྱེས་བུ་དྲེགས་པ་ཅན། །ང་རྒྱལ་བདག་ཉིད་ཅན་ཞེས་སོ། །དང་ཞེས་སྨྲ་ཕན་ཚུན་འགྲན། །གང་ཞིག་ང་རྒྱལ་བྱེད་པའོ། །འཚམ་མེད་ཐལ་འགྲོ་རྟུལ་ཕོད་དང་། །སྟོབས་ལྡན་དཔའ་དང་ཉམ་ང་མེད། །ནུས་ལྡན་ཕ་
