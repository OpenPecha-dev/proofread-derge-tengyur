སྭརྒྒ་ནཱ་ཀ་ཏྲི་དི་བ་ཏྲི་ད་པཱ་ལ་ཡཱཿ། །སུ་ར་ལོ་ཀོ་དྱོ་དི་པོ་དྭེ་སྟྲི་ཡཽ་ཀླཱི་བེ་ཏྲི་པིཥྚ་ཡ། ༦ ༈ །ཨ་མ་རཱ་ནི་རྫ་རཱ་དེ་བཱ་སྟྲི་ད་སཱ་བི་བུདྡྷཱཿསུརཱཿ། །སུ་པརྦཱ་ཎཔྶུ་མ་ན་སསྟྲི་དི་བེ་པཱ་དི་བཽ་ཀ་སཿ། ༧ ༈ །ཨཱ་དི་ཏེ་ཡཱ་དི་བི་ཥ་དོ་ལེ་ཁཱ་
བཟང་དང་། །བདེ་ལྡན་སུམ་རྩེ་སྐབས་གསུམ་གནས། །ལྷ་ཡི་འཇིག་རྟེན་རོལ་ས་མཁའ། །གཉིས་མོ་མ་ནིང་གཞི་གསུམ་པ། །འཆི་མེད་རྒས་མེད་སྐབས་གསུམ་ལྷ། །རྣམ་པར་སད་དང་བཅུད་ལྡན་དང་། །ཚིགས་བཟང་ཡིད་བཟང་སུམ་རྩེན་དབང་། །མཐོ་རིས་སྣང་བྱེད་མི་སྦྱིན་བུ། །མཁའ་སྤྱོད་ལེགས་
ཨ་དི་ཏི་ནནྡ་ནཱཿ། སྐྱེ་མེད། །ཨཱ་དི་ཏྱཱ་རྀ་བྷ་བོཉསྭ་སྣཱ་ཨ་མདྷྱཱ་ཨ་མྲྀ་ཏཱནྡྷ་སཿ། ༨ ༈ །བིཧྣི་མུ་ཁཱཿཀྲ་ཏུ་བྷུ་ཛོ་ལྷའི་སྤྱི་མིང་། གཱིརྦཱ་ཎཱ་དཱ་ན་བཱ་ར་ཡཿ། །བྲྀནྡཱ་ར་ཀཱ་དེ་བ་ཏཱ་ནི་པུ་སི་བཱ་དེ་བ་ཏཱ་སྟྲི་ཡཱཾ། ༩ ༈ །ཨཱ་དི་ཏྱཱ་བི་ཤྭ་བ་ས་བསྟུ་ཥི་ཏཱ་བྷཱ་སྭ་རཱ་ཎི་ལཱཿ། །
བྲིས་མི་སྦྱིན་དགའ། །ཨཱ་དིའི་བུ་དང་རྀ་ཡིག་བུ། །གཉིད་མེད་མི་འཆི་བདུད་རྩི་ཟས། །མེ་ཞལ་དང་ནི་མཆོད་སྦྱིན་ཟས། །ཚིག་གི་མདའ་ཅན་ལྷ་མིན་དགྲ། །ཚོགས་བཟང་བ་དང་ལྷ་རྣམས་ཉིད། །སྐྱེས་པའམ་ལྷ་མོ་བུད་མེད་ལའོ། །ཉི་མ་སྣ་ཚོགས་ནོར་ལྷ་དང་། །དགའ་ལྡན་དང་ནི་འོད་གསལ་རླུང་། །
མ་ཧཱ་རཱ་ཛི་ཀ་སཱ་དྷྱཱཤྩ་ཏུ་དྲཱ་ཤྩ་ག་ན་དེ་བ་ཏཱཿ། ༡༠ ༈ །བི་དྱཱ་དྷ་རོ་ཉཔྶ་རོ་ཡཀྵོ་རཱཀྵ་ས་གནྡྷ་རྦྷ་ཀིནྣ་རཱཿ། །པི་ཤཱ་ཙོ་གུ་ཧྱ་ཀཿ་སིདྡྷོ་བྷཱུ་ཏེ་྅མཱི་དེ་བ་ཡོ་ན་ཡཿ། ༡༡ ༈ །ཨ་སུ་རཱ་དཻ་ཏྱ་དཻ་ཏེ་ ལྷའི་རིགས་ཀྱི་ཡ་དྷ་ནུ་ཛེནྡྲ་རི་དཱ་ན་པཱཿ། །ཤུ་ཀྲ་ཤི་ཥྱཱ་དི་
ཕྲེང་ཆེན་གྲུབ་པ་དྲག་པོ་སྟེ། །དེ་རྣམས་ཚོགས་པའི་ལྷ་རྣམས་སོ། །རིག་འཛིན་ཆུ་སྐྱེས་གནོད་སྦྱིན་དང་། །སྲིན་པོ་དྲི་ཟ་མིའམ་ཅི། །ཤ་ཟ་གསང་བ་གྲུབ་པ་དང་། །འབྱུང་པོ་འདི་རྣམས་ལྷ་ལས་བྱུང་། །ལྷ་མིན་གྱི། ལྷ་མིན་སྦྱིན་སྐྱེས་བཅུད་མི་ལྡན། །གཞུ་སྐྱེས་སྦྱིན་བྱུང་དབང་པོའི་དགྲ། །པ་སངས་སློབ་མ་སྦྱིན་
