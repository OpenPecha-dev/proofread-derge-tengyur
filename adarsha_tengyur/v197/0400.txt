མཱི་ནཱ་སཱ་ཡཻ་བ་པྲ་ཏི་བརྠཾ་པྲ་སཱུ་ཡ་ཏེ། །ཨཱུ་དྷསྟུ་ཀླཱི་བ་མཱ་པཱི་ནཾ་ས་མཽ་ཤི་བ་ཀ་ཀཱི་ལ་ཀཽ །ན་པུཾ་སི་དཱ་མ་སནྡཱ་ནཾ་པ་ཤུ་རཛྫུ་སྟུ་དཱ་མ་ནཱི། །བཻ་ཤཱ་ཁཱ་མནྠ་མནྠཱ་ན་མནྠཱ་ནོ་མནྠ་དཎྜ་ཀེ །ཀུ་ཋ་རོ་དཎྜ་བེཥྐམྦྷོ་ནཱི་མརྒྒ་རཱི་ས་མེ། །ཨུཥྚེ་ཀུ་མེ་
སྐྱེ་བའོ། །ལོ་བཏུད་སྐྱེ་དང་སླར་སྐྱེ་མ། །ཨཱུ་དྷ་མ་ནིང་མཱ་པཱི་ནཾ། །བརྟོད་པའི་ཤིང་དང་ཕུར་པའོ། །ཕོ་མིན་ཐག་པ་འདོགས་ཐག་གོ། །ཕྱུགས་ཐག་རྐང་ཐག་དཱ་མ་ནཱི། །སྲུབ་མ་སྲུབ་ཤིང་མནྠཱ་ན། །སྲུབ་མ་དང་ནི་ཞོ་སྐྱའོ། །སྲུབ་མའི་ཀ་བ་ཀུ་ཋརོ། །ཞོ་རྫ་སྲུབ་པའི་རྫ་དག་མཚུངས། །རྔ་མོ་རིགས་ཅན་
ལ་ཀ་མ་ཡ་མཱ་ཧཱདྒཱཿཀ་ར་བྷཿ་ཤི་ཤུཿ། །ཀ་ར་བྷཱཿསྱུཿཤྲྀངྑ་ལ་ཀཱ་དཱ་ར་བཻཿ་བཱ་ར་བནྡྷ་ནཻཿ། །ཨ་ཛཱཙྪཱ་གཱིསྟུ་བྷ་ཙྪ་ག་བསྟ་ཙྪ་ག་ལ་ཀཱ་ཨ་ཛཿ། །མེཎྜོ་ར་བྷྲོ་ར་ཎོརྞཱ་ཡུ་མེ་ཥ་ཏྲྀཥྞ་ཡ་ཨེ་ཌ་ཀཱཿ། །ཨུཥྚོ་ར་བྷྲཱ་ཛ་བྲྀནྡེ་སྱཱ་དཽཥྚ་ཀཽ་རབྷྲཱ་ཀོ་ཛ་ཀཾ །
རང་བཞིན་དང་། །ལུས་ཆེན་རྡེའུ་རྡེ་ཕྲུག་གོ། །རྡེའུ་འདོགས་ཐག་ཤྲྀངྑ་ལ། །ཤིང་ལ་རྐང་པའི་འདོགས་པའོ། །ར་དང་ཙྪཱ་གི་ཚེ་ཚེ་དང་། །ར་དང་སྦ་འཕྱང་མ་སྐྱེས་པ། །ལུག་དང་ལྟོ་འཕྱང་བལ་ལྡན་དང་། །གཡང་དཀར་དང་ནི་ཨེ་ཌ་ཀཱ །རྔ་མོ་བྲང་འགྲོ་ཚོགས་དང་ཁྱུ། །རྔ་ཁྱུ་གསུས་འཕྱང་ཀོ་ཛ་ཀཾ །
ཙཀྲཱི་བནྟསྟུ་བཱ་ལེ་ཡཱ་ར་ས་བྷཱ་གརད་བྷཱཿཁ་རཱཿ། །བཻ་ད་ཧ་ཀཿ་སཱརྠ་བཱ་ཧོ་ནཻ་ག་མཱ་བཱ་ནི་ཇོ་བཱ་ཎི་ཀ །པ་ཎྱཱ་ཛཱི་བོ་ཧྱཱ་པ་ཎི་ཀཿཀྲ་ཡ་བི་ཀྲ་ཡི་ཀཤྩ་སཿ། །བི་ཀྲེ་ཏཱ་སྱཱ་དྭི་ཀྲ་ཡི་ཀཿཀྲཱ་ཡ་ཀཿཀྲ་ཡི་ཀཿས་མཽ། །བཱ་ནི་ཛྱནྟུ་བ་ཎི་ཛྱཱ་
བོང་བུ་འཁོར་ལོ་ཅན་སྟོབས་ལྡན། །སྣ་ཆེན་སྐད་ཆེན་ཁ་རའོ། །གཞན་སྐྱེས་དེད་དཔོན་ངེས་འགྲོ་དང་། །ཚོང་དཔོན་ཚོང་བ་ཟོང་འཚོང་དང་། །ཟོང་པ་དང་ནི་ཉོ་འཚོང་ངོ་། །འཚོང་བ་བི་ཀྲ་ཡི་ཀའོ། །རྣམ་པར་རྗེ་དང་རྣམ་པར་འཚོང་། །ཚོང་པ་དང་ནི་ཚོང་བྱེད་མཚུངས། །བཙོང་བྱེ་བ་ཎི་ཛྱ་ཡིན་ནོ། །
