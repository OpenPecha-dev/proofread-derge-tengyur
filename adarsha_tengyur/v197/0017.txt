རྣམས་ཀྱི་ཨ་ནུ་དཱཏྟ་དབྱི་བ་ཉིད་དོ། །ནམ྄་ལའོ། །ཨཔྲ་ལ་རཉྫ྄་དཾ་ཤྲ། །ཥཉྫ྄་ཥྭཉྫ྄རྣམས་ཀྱིའོ། །བིད་ལས་ཏིཔ྄་ལ་སོགས་པ་དགུ་རྣམས་ནི་ཎཔྲ་ལ་སོགས་པར་ཡང་ནའོ། །སི་ལ་སའོ། །ཛ་ཧི། ཨེདྷི། ཤཱ་དྷྲིའོ། །ཨསྟི་ལས་ཨཱིཊ྄འོ། །ཌྷའི་
ཌྷ་ལ་དབྱི་ཞིང་རིང་པོ་ཡང་ངོ་། །ས྄་ཀ྄་སོགས་ཀྱི་ཡང་ངོ་། །གཞན་དུ་ས྄་ཛ྄འོ། །དིཔ྄་ལ་སའི་ཏ་སྟེ་སྲིཔ྄་ལ་ཡང་ནའོ། །བ་འགྲོ་བ་ལ་མྲྀ་ཇིའི་འཕེལ་བའོ། །ནཱི་ཛ་རྣམས་ཀྱི་ཡོན་ཏན་ནོ། །སྔོན་མའི་རྀ་པྲྀ་དག་གི་ཨིའོ། །ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཀུན་ནས་བཤད་
པའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།བྱེད་པ་པོ་ལ་ཀྲྀཏྲའོ། །ཏྲི་བུཎ྄་དག་གོ །ཀྲྀཏྲ་ལའོ། །ཨཱུཏྲ་འགྲོ་བ་ལས་ཡང་ནའོ། །སཧྲ་བཧྲ་དག་གི་ཨ་ཡིག་ནི་ཨོཏྲའོ། །ཡུ་བུ་དག་གི་ཨ་ན་ཨ་ཀ་དག་གོ །ནཱ་མིའི་ཨུ་པ་དྷཱ་ལས་ཀའོ། །པ་ཙི་ན་ནཱི་གྲ་ཧཱ་སོགས་ལ་ཨ་ཡུ་ཎི་

ཨི་ཁ་ཁིའོ། །ཁ་འགྲོ་བ་ལ་སྔོན་མའི་ཚིག་གོ །ཨེ་ཛ྄་རྣམས་ཀྱི་ཁ་ཤྲའོ། །བྱེད་པ་ལ་ཁྱུཊ྄འོ། །བྷཛ྄་རྣམས་ཀྱི་བིཎ྄འོ། །བིའིའོ། །ཤས྄་སོགས་དབྱངས་ལ་བ་ཧའི་བ་ཨཽའོ། །ས་ཧྲའི་སཱཌྷ་ལ་ཥ་འང་ངོ་། །བྱིངས་ཀྱི་མའི་ནའོ། །བྱུང་བ་མ་ཡིན་པ་ལས་ཀྲྀ་བྷཱུ་ཨསྲ་སྦྱོར་བ་ལ་
ཙྭིའོ། །ཙྭི་ལ་རིང་པོ་སྟེ་ཨའི་ཡང་ཨཱིའོ། །ཙྭི་ལ་ས་དབྱི་བ་འང་ངོ་། །ཨཱཏྲ་ལ་མ་ཎི་པྲ་ཀྭ་ནི་པྲ་བ་ནི་པྲའོ། །སྠཱ་རྣམས་ཀྱི་ཨཱིའོ། །ཀྭི་པྲའོ། །ཐུང་དུའི་ཀྲྀཏྲ་ཀྱི་པ་འགྲོ་བ་ལ་ཏུཀྲའོ། །ཀྭི་པའོ། །ཉེ་བར་འཇལ་བའི་བྱ་བ་ལ་དྲྀ་ཤྲ་ལ་ཊཀྲ་སཀྲ་དག་ཀྱང་ངོ་། །སརྦ་སོགས་ཀྱི་ཨཱའོ། །ཀིམ྄་
ཨི་དམྲ་དག་གི་ཀཱི་ཨཱི་ཤྲ་དག་གོ །འདས་པ་ལ་ཎི་ནིའོ། །ཀྭི་པྲ་བ་ནི་པྲ་ཌ་རྣམས་སོ། །ཀྟ་ཀྟ་བ་ཏུའོ། །ལས་མེད་པ་ལས་འགྲོ་བའི་དོན་དང་བྱེད་པ་པོ་ལ་ཀྟའོ། །ཨིཊྲ་དང་བཅས་པའི་ཀྟ་ལ་ཡང་ནའོ། །ཀ་འགྲོ་བ་ལའོ། །ཨཱཏ྄་ཨཱིཏ྄་འགྲོ་བ་ལའོ། །ཉི་ཨིཏྲ་འགྲོ་བ་ལ་ད་
