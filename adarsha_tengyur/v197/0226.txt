ལ་སྔོན་པའི་ཨི་ནི་ཨི་ཡྲ་ཞེས་པར་བསྒྱུར་བར་བཤད་པས། ཨི་ཡརྟི། ཨི་ཡྲྀཏཿ ཨི་ཡྲ་ནྟི། ལིང ཨི་ཡྲྀ་ཡཱཏྲ། ཨི་ཡརྟུ། ལོཊ ཨི་ཡྲྀ་ཏཱཏྲ། ཨཻ་ཡ་རཏྲ། ཨཊ དིབྲ ཨིཡཿསོགས་སོ། །ཨིཎྲ་འགྲོ་བ་ལ། ཨེ་ཏི། ཨི་ཏཿ ཡནྟི། ཨི་ཡཱཏ། ཨི་
ཡཱཏཱཾ། ཨིཡུཿ། ཨཻ་ཏུ། ཨི་ཏཱཏྲ། ཨིནྟུ། ཨེཏྲ། ཨེཏཱཾ། ཨཱ་ཡ་ནྲ་སོགས་དང་། དངོས་པོ་དང་ལས་ཀྱི་དོན་ལ། བཞི་པོ་ལ་ཡཀྲ་འོས། ཡཀྲ་བྱིན། ཡ་ལ་འོས་རིང་པོར་བསྒྱུར་བས། ཨཱི་ཡ་ཏེ་སོགས་དང་། ལྐོག་གྱུར་ལིཊ་ལ། ཨཱི་
ཡཱ་ལ། ཨཱི་ཡཏུཿ ཨཱི་ཡུཿསོགས་སོ། །ཨི་ཥུ་འདོད་པ་ལ། ལྐོག་གྱུར་ལིཊྲ་རྣམས་སྦྱར། གཉིས་བརྗོད། སླར་ཡང་ཨི་ཕྱི་མ་གཉིས་བརྗོད་པ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨི་ཡེ་ཥ། ཨཱི་ཥ་ཏུ། ཨཱི་ཥུཿསོགས་སོ། །ངྷཱ་ཏཱུ་ནཱ་མ་ནནྟ་ཏྭཱ་ཏྲ་བི་ཤྭཱརྦྭ་ཏྭཱཉྩ་
སརྦཐཱ། ཨ་བྷི་ངྷཱ་ཏུ་མ་ཤ་ཀྱ་མི་ཏྱལ་མཱ་ཁྱཱ་ཏ་ཁྱཱ་པ་ནེ། དེ་ལྟར་བྱིངས་རྣམས་ཀྱང་ནི་མཐའ་ཡས་ཉིད། །སྣ་ཚོགས་དོན་ཉིད་ཀྱང་ནི་ཐམས་ཅད་དུ། །བྱིང་རྣམས་འདི་ཞེས་མངོན་གསལ་མི་ནུས་ཀྱང་། །ཀུན་ནས་བཤད་པ་རྫོགས་པར་བཤད་ཟིན་ཏོ། །ཨི་ཏི་སྭ་ར་སྭ་སྟི་
བྱཱ་ཀཱ་ར་ཎ་སྱ་སཱུ་ཏྲཱ་དཱ་ཁྱཱ་ཏ་པྲ་ཀྲི་ཡ་ས་མཱ་པྟཱཿ། ཞེས་པ་བརྡ་སྤྲོད་པ་དབྱངས་ཅན་གྱི་མདོ་ལས་ཀུན་ནས་བཤད་པའི་རབ་ཏུ་བྱེད་པ་རྣམས་རྫོགས་སོ།། །།དེ་ནས་བྱིངས་ལས་མིང་དུ་བསྒྲུབ་པཀྲྀ་ཏ་སོགས་ཀྱི་ཆོ་ག་རྣམས་བཤད་པར་བྱ་སྟེ། ༈ །ཀྲྀཏྐ་རྟརི། བྱེད་པ་པོ་ལ་
ཀྲྀ་ཏྲའོ། །འཆད་འགྱུར་ཏྲི་ལ་སོགས་པའི་རྐྱེན་རྣམས་ལ་ཀྲྀ་ཏྲ་རྐྱེན་ཞེས་བྱ་ལ། རྐྱེན་དེ་རྣམས་གཙོ་བོར་བྱེད་པའི་དོན་ལ་འཇུག་པ་ཤས་ཆེའོ། ༈ །ཏྲྀ་བུ་ཎཽ། ཏྲྀ་བུཎཱ་དག་གོ །བྱིངས་ལས་བྱེད་པའི་དོན་གྱི་མིང་དུ་བསྒྲུབ་པ་ལ་རྐྱེན་ཏྲྀ་དང་བུཎ་དག་འགྱུར་ཏེ། ༈ ཌུ་བ་ཙཥ་འཚེད་པ་
ལ་ཏྲ། ཙུའི་ཀུའོས་ཀར་བསྒྱུར་བས། པཀྟྲྀ་ཞེས་འགྲུབ་པ་ལས། དེ་ནི་མིང་དུ་འགྲུབ་པ་ཡིན་ལ། སླར་དེ་ཚིག་ཏུ་བསྒྲུབ་པའི་ཕྱིར་སི་སོགས་རྣམ་དབྱེ་རྣམས་སྟེར་དགོས་པ་ལས་ཆོ་ག་རྣམས་གོང་མིང་ལེར་བཤད་པ་ལྟར་གྱིས་བསྒྲུབས་པས། སི བཀྟཱ། འཚོད་པར་བྱེད།
