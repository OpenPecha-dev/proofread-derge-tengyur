དྷེ྅པི་ས་མུཙྪྲ་ཡཿ། །བི་ཥ་ཡོ་ཡ་སྱ་ཡོཛྙཱ་ཏསྟ་ཏྲ་ཤཔྡཱ་དི་ཀེ་ཥྭ་བི། །ནིརྻཱ་སེ་པི་ཀ་ཥཱ་ཡོཉསྟྲཱི་ས་བྷཱ་ཡཱཉྩ་པྲ་ཏི་ཤྲ་ཡཿ། །པྲཱ་ཡོ་བྷཱུ་མྣྱནྟ་ག་མེ་ནེ་མ་ནྱུ་དཻ་ནྱེ་ཀྲ་ཏཽ་ཀྲུ་དྷི། །ར་ཧ་སྱོ་པསྠཱ་ཡོརྒུ་ཧྱཾ་ས་ཏྱཾ་ཤ་པ་ཐ་ཏ་ཐྱ་ཡོཿ། །བཱིརྻཱཾ་
༄༅། །ས་མུཙྪྲ་ཡ་འགལ་མེད་ཀྱང་། །བི་ཥ་ཡ་ནི་ཤེས་བྱ་ཡི། །གང་གི་དེ་ལ་སྒྲ་སོགས་རྣམས། །དུར་ཁ་བསྐ་བ་ཀ་ཥཱ་ཡ། །འདུ་ཁང་པྲ་ཏི་ཤྲ་ཡའོ། །པྲཱ་ཡོ་ཕལ་ཆེར་མཐར་འབྱིན་དང་། །མ་ནྱུ་མཆོད་སྦྱིན་ཁྲོ་བ་ལ། །གསང་དང་ཉེར་གནས་གུ་ཧྱཾའོ། །ས་ཏྱཾ་དག་ཚང་བདེན་པའོ། །བཱིརྻཾ་
བ་ལ་པྲ་བྷཱ་བོ་ཙ་དྲ་བྱམྦྷ་བྱེ་གུ་ཎཱ་ཤྲ་ཡེ། །དྷིཥྚྱཾ་སྠཱ་ནེ་གྲྀ་ཧེ་བྷེགྣཽ་བྷཱ་གྱཾ་ཀརྨྨ་ཤུ་བྷཱ་ཤུ་བྷཾ། །ཀ་སེ། གྲོ་མ་ལ་འཇུག་པ་ཡོད་ཀྱང་འདིར་རྟ། རུ་ཧེ་མནོ་གཱཾ་གེ་ཡཾ་བི་ཤ་ལྱཱ་དནྟ་ཀཱ་པི་ཙ། །བྲྀ་ཥཱ་ཀ་པཱ་ཡཱི་ཤྲཱི་གཽརྻ་ར་བྷི་ཁྱཱ་ནཱ་མ་སོ་བྷ་ཡོཿ། །ཨཱ་རམྦྷོ་ནིཥྐྲྀ་ཏིཿཤིཀྵཱ་པཱུ་ཛ་ནཾ་
དཔའ་སྟོབས་ནུས་མཐུའོ། །དྲ་བྱཾ་སྐལ་བ་ཡོན་ཏན་གནས། །དྷིཥྚྱཾ་གནས་ཁང་གླང་པོ་མེ། །བྷཱ་གྱཾ་ལས་དང་དགེ་མི་དགེ་རྟ་དང་གསེར་ནི་གཱཾ་གེ་ཡཾ། །བི་ཤ་ལྱཱ་ནི་སོ་ལྡན་ལའང་། །བྲྀ་ཥྛ་ཀ་ཡཱ་དཔལ་གཽ་རཱི། །ཨ་བྷི་ཁྱཱ་ནི་མིང་དང་མཛེས། །རྩོམ་དང་ངེས་བྱེད་སློབ་པ་དང་། །མཆོད་

དང་བསྒྲུབ་བྱ་བགྲོས་པ་ཟེར་ཐབས་ལས་སེམས། །གསོ་དཔྱད་དགུ་ལ་ཀྲི་ཡཱའོ། །ཚཱ་ཡ་ཉི་མའི་ཆུང་མ་མཛེས། །གཟུགས་བརྙན་ཆ་བ་མེད་རྣམས་ལའོ། །ཀ་ཀྵྱཱ་རེ་མིག་བསིལ་ཁང་སོགས། །རྐེད་རྒྱན་གླང་ཆེན་དབུས་སུ་འཆིང་། །ཀྲྀ་ཏྱཱ་བྱ་བ་ལྷ་དག་གསུམ། །དབྱེ་བ་ནོར་ལ་སོགས་རྣམས་ཀྱིས། །
