ཕ་ལ་ལ་སོགས་པ་བཞི་ནི་སྐྱུ་རུ་རའི་ཤིང་གི་མིང་སྟེ། ཏི་ཥྱ་ཕ་ལ་ནི་རྒྱལ་འབྲས་སོ། །ཨཱམླ་ཀཱི། སྐྱུ་རུ་རའོ། །ཨ་མྲྀ་ཏཱ་བདུད་རྩི་ཅན་ནོ། །བ་ཡཿསྠཱ། ན་ཚོད་གནས་བྱེད་དེ། ན་ཚོད་དང་ལང་ཚོ་བརྟན་པར་བྱེད་པས་སོ། །དེ་ནས་བཱི་བྷི་ཏ་ག་ལ་སོགས་པ་དྲུག་ནི་བ་རུ་རའི་མིང་སྟེ། བི་བྷཱི་ཏ་ཀ་ནི་
བ་རུ་རའོ། །བྷཱུ་ཏ་བཱ་ས། འབྱུང་པོའི་གནས་སོ། །ཀ་ལི་དྲུ་མ་རྩོད་པའི་ཤིང་ངོ་། །དེ་ནས་ཨ་བྷ་ཡ་ལ་སོགས་པ་ཚིགས་སུ་བཅད་པ་གཅིག་གིས་ཨ་རུ་རའི་མིང་སྟེ། ཨ་བྷ་ཡ་ནི་འཇིགས་མེད་དེ། ནད་ཀྱི་འཇིགས་པ་མེད་པས་སོ། །ཨ་བྱ་ཐ་ནི་ནད་མེད་དོ། །བ་ཐྱཱ་ནི་ཕན་པ་ཅན་ནོ། །བ་ཡཿསྠཱ་ན་ཚོད་གནས་
བྱེད་དོ། །པཱུ་ཏ་ན། ཚི་ཅན་ནོ། །ཨ་མྲྀ་ཏཱ། བདུད་རྩི་ཅན་ནོ། །ཧ་རི་ཏ་ཀཱི། ཨ་རུར་སྟེ་འཕྲོག་བྱེད་ཅེས་བྱའོ། །ཧེ་མ་བ་ཏཱི་གསེར་ལྡན་ནོ། །ཙེ་ཏ་ཀཱི་སེམས་བྱེད་དོ། །ཤྲེ་ཡ་སཱི་དཔལ་ལྡན་ནམ་དགེ་ལེགས་ཅན་ནོ། །ཤི་བ་ཞི་བྱེད་དོ། །དེ་ནས་པི་ཏ་དྲུ་ལ་སོགས་པ་གསུམ་ནི་ཐང་ཆུ་ཤིང་གི་མིང་སྟེ།
བཱི་ཏ་དྲུ། ཤིང་སེར་དང་། ས་ར་ལ། བཅུད་འཛིན་དང་། པཱུ་ཏི་ཀཱཥྛ། ཐང་ཆུ་ཤིང་ངོ་། །དེ་ནས་དྲུ་མོ་ཏྤལ་ལ་སོགས་པ་གསུམ་ནི། ཨུཏྤལ་ཤིང་གི་མིང་སྟེ། དྲུ་མོཏྤ་ལ། ཨུཏྤ་ལ་ཤིང་དང་། ཀརྞི་ཀཱ་ར། ནི་ཀནྨི་ཡཱ་ར་ཞེས་གྲགས་པ་པདྨ་དང་། ཚད་མཉམ་པ་མེ་ཏོག་སེར་པོ་ཅན་ནོ། །དེ་ནས་ཀཎྜ་
ཕ་ལ་དང་པ་ནས་གཉིས་ཚེར་འབྲས་ཅན་གྱི་ཤིང་སྟེ། འདིའི་འབྲས་བུ་ལ་ཚེར་མ་ཡོད་པས་སོ། །ནི་ཙུ་ལ་ལ་སོགས་པ་གསུམ་ནི་ཆུ་སྐྱེས་ཤིང་གི་མིང་སྟེ། དེ་ནས་དྲུག་ནི། སྔོན་གྱི་ལོ་ཙཱ་བ་རྣམས་ཀྱིས་དོང་ཀར་བསྒྱུར་སྣང་། ཀཱ་ཀོ་དུམྦཱ་ར་སྟེ། བྱ་རོག་ཨུདུམྦཱ་རའི་མིང་ངོ་། །དེ་ནས་ཨ་རིཥྚ་ལ་སོགས་པ་དྲུག་
ནིམྦའི་ཤིང་གི་མིང་ངོ་། །དེ་ནས་ལྔ་ནི་ཨ་ག་རུའི་མིང་སྟེ། པིཙྪི་ལ་འཇུག་མ་ཅན་དང་། ཨ་གུ་རུ། ཨ་ག་རུར་གྲགས་པ་དང་། ཤིནྴཱ་པ་དང་། ཀ་པི་ལ། སེར་སྐྱ་དང་། བྷསྨ་གརྦྷ། ཐལ་བའི་སྙིང་པོ་ཅན་རྣམས་སོ། །དེ་ལ་པིཙྪི་ལ་ནི་ཤིང་ཤལྨ་ལི་ལ་ཡང་བཤད་དོ། །དེ་ནས་གསུམ་ནི་ཤིང་ཤྲི་ཥའི་
མིང་ངོ་། །དེ་ནས་ཙམྤེ་ཡ་དང་། ཙམྤ་ཀ་དང་། ཧེ་མ་པུཥྤ། གསེར་གྱི་མེ་ཏོག་རྣམས་ཙམྦ་ཀའི་མིང་སྟེ། འདིའི་མེ་ཏོག་གསེར་གྱི་མདོག་ཏུ་ཡོད་པས་སོ། །བཉྫུ་ལ་ནི་ཨ་ཤོ་ཀ་སྟེ། དྲག་པོ་ཆུང་མ་མི་འདོད་པས་ཤིང་ཨ་ཤོ་ཀའི་འོག་ཏུ་དཀའ་ཐུབ་སྤྱད་པས་གཽ་རཱི་བསམ་པ་ཡོངས་སུ་རྫོགས་པས་
