རུ་ཏྭཱན། ༈ ཨེ་ཏཏྐིཾ་ཡཏྟདྦྷྱཿ ཨེ་ཏད། ཀིམ། ཡད། ཏད་རྣམས་ལས་སོ། །ཨེ་ཏཏྲ་ལ་སོགས་པ་འདི་རྣམས་ལས་ཚད་སྟོན་པའི་དོན་ལ་རྐྱེན་བ་ཏུ་འགྱུར་ཏེ། ཡདྲ་དང་ཏདྲ་ལ་རྐྱེན་བ་ཏུ་བྱིན། ༈ ཡ་ཏྟ་དོ་རཱ། ཡདྲ་ཏདྲ་དག་གི་ཨཱའོ། །རྐྱེན་བ་ཏུ་ཕ་རོལ་ལ་ཡདྲ་དང་ཏདྲ་དག་གི་ཊི་ནི་
བ་ཏུ་ ནུམྲ་ སི་ ཨཱ་རུ་འགྱུར་ཏེ། ཊི་ཨཱ། སི། ནུམྲ་བྱིན་པས། ཡཱ་བཱནྲ། ཏཱ་བཱནྲ། ༈ ཀི་མཿཀིརྻཤྩ། ཀིམྲ་ནི་ཀི་སྟེ་ཡ་ཡང་ངོ་། །རྐྱེན་བ་ཏུ་ཕ་རོལ་ལ་ཀིམ་གྱི་སྒྲ་ནི་ཀི་རུ་འགྱུར་ཞིང་བ་ཏུའི་བ་དེ་ཡང་ཡ་ཡིག་ཏུ་འགྱུར་ཏེ། ཀི་ཡཱནྲ། ༈ ཨཱ་ཨི་ཤྩཻཏ་དོ་བཱ། ཨེ་ཏཏྲ་ཀྱི་ཨཱ་དང་ཨི་ཡང་ནའོ། །
རྐྱེན་བ་ཏུ་ཕ་རོལ་ལ་ཨེ་ཏཏྲ་ཀྱི་ཊི་ཨཱུ་རུ་འགྱུར་ཞིང་ཡང་ན་ཨེ་ཏཏ་ཡོངས་ གང་ཙམ་ དེ་ཙམ། རྫོགས་ཨི་རུ་འགྱུར་ལ་རྐྱེན་གྱི་བ་ཡང་ཡར་འགྱུར་ཏེ། ཨེ་ཏཱ་བཱནྲ། ཨི་ཡཱན། ༈ ཏུནྡཱ་དེ་རི་ལཿ ཏུནྡ་སོགས་ལ་ཨི་ལའོ། །ཏུནྡ་ལ་ ཇི་ཙམ། སོགས་པ་རྣམས་ལ་ཡོད་པའི་དོན་ལ་འདོན་ལ་རྐྱེན་ཨི་ལ་འགྱུར་ཏེ་
ཏུནྡི་ལཿ ཨུ་ད་རི་ལཿསོགས་སོ། ༈ །ཨཽནྣ་ཏྱེ་དནྟཱ་དུ་རཿ དནྟ་ལས་རིང་བ་ལ་ཨུ་རའོ། །དནྟ་ལས་རིང་བའམ་སྦོམ་པའི་དོན་ཅན་དང་ལྡན་པ་ སི་ནུམྲ་ དེ་ཙམ། སི་ནུམྲ་ལ་རྐྱེན་ཨུར་འགྱུར་ཏེ། ཨུནྣཱཿདནྟཱཿཨསྱ། ཞེས་པའི་དོན་ལ། དནྟུ་ཨུར་རཿསི། ༈ ཤྲདྡྷཱ་དེརཱུཿ ཤྲདྡྷ་ལ་སོགས་པ་ལ་
སྟེ་བ་ཅན། ལུའོ། །ལྟོ་བ་ཅན། ཤྲདྡྷཱ་ལ་སོགས་པ་རྣམས་ལ་ཡོད་པའི་དོན་ལ་ལུ་རྐྱེན་དུ་འགྱུར་ཏེ། ཤྲདྡྷཱ་ལུཿ སི ནི་དྲཱ་ལུཿ ཀྲྀ་པཱ་ལུཿ སོགས་སོ། ༈ །ཨས྄་ཀྱི་མཐའ་ཅན་དང་མཱ་ཡཱ་དང་མེ་དྷཱ་དང་སྲཀྲ་རྣམས་ཛས ཛས྄ སི་རིང་པོ་ཡོད་པའམ་སོ་སྦོམ་པོ་ཡོད་པའམ་སོ་མཐོན་པོ་ཡོད་པ་ཅན། ལས་ཡོད་པའི་དོན་ལ་རྐྱེན་བིནྲ་འགྱུར་
བི་ཏྱ་ཏེ་ཡསྱ་བཅས་ཚིག་སྦྱོར། བར་བཤད་དེ། ཏ་པསྭཱི། མཱ་ཡཱ་བཱི། སི ནུམྲ། མེ་དྷཱ་བཱི། སི་ནུམྲ་ སྲ་གྤཱི། སི་ནུམྲ། དད་ཅན། གཉིད་ཅན། བརྩེ་བ་ཅན་ནམ་བརྩེ་ལྡན། ༈ བཱ་ཙོགྨི་ནིཿ བཱ་ཙ་ལས་གྨི་ནིའོ། །བཱཙྲ་ལས་ཡོད་པའི་དོན་ལ་གྨི་ནི་རྐྱེན་དུ་འགྱུར་ཏེ། བཱཙྲ་ལ། གྨི་ནི། རྐྱེན་གྱི་ནའི་ཨི་དང་གྲ་རྗེས་འབྲེལ། ཙུའི་
ཀུ་དང་། ཙ་པའི་ཨ་བ་ཛ་བ་ཨས་ཏ་པསྲ་བིནྲ་སི་ནུམྲ་དཀའ་ཐུབ་ཅན་ནམ་སྡོམ་ལྡན། སྒྱུ་མ་ཅན། བློ་གྲོས་ཅན། འཕྲེང་ལྡན་ནམ་འཕྲེང་བ་ཅན། སོགས་ཀྱིས་བསྒྲུབས་པས། སྨ་བ་ཅན། བཱགྨཱི། །ཨཱ་ལཱ་ཊཽ་ཀུཏྶི་ཏ་བྷཱ་ཥཱི་ཎཱི། སྨད་པ་བརྗོད་པ་ལ་ཨཱ་ལ་ཨཱཊྲ་དག་གོ །པཱཙ་ལས་སྨད་པའི་ཚིག་བརྗོད་པའི་དོན་ལ་རྐྱེན་ཨཱ་ལ་དང་ཨཱཊྲ་
