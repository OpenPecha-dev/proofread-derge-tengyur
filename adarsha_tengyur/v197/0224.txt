ཧའི་ཌྷ་འོས། ཧྲ་ཌྷྲ། ཏ་ཐ་དག་གི་དྷའོས་ཏ་དྷ། ཥྚུ་སོགས་ཀྱིས་དྷྲ་ཌྷྲ། པ་འགྲོ་བའི་ཡོན་ཏན་བྱས་པས། ལེཌྷྲ། ཌྷྲི་ཞེས་གནས་པ་ལ། ༈ ཌྷི་ཌྷོ་ལོ་པོ་དཱི་རྒྷཤྩ། ཌྷའི་ཌྷ་ལ་དབྱི་ཞིང་རིང་པོ་ཡང་ངོ་། །ཚུ་རོལ་གྱི་མཐའི་ཌྷ་ཡིག་ནི་ཌྷ་ཡིག་ཕ་རོལ་ལ་དབྱི་ཞིང་།
དབྱངས་སྔོན་མ་དེ་ཐུང་ངུ་ཡིན་ན་ཡང་རིང་པོར་འགྱུར་ཏེ། ཌྷ་སྔོན་མ་ཕྱིས་པས། ལེ་ཌྷི། དེ་བཞིན་དུ་བསྒྲུབས་པས། ལཱི་ཌྷཿ ལི་ཧི་ནྟི། ལེ་ཀྵི། ལཱིཌྷཿ ལཱིཌྷ། ལི་ཧྭ། ལིཧྨ། ༈ ཆོག་འབྱུང་ལིང་ལ། ལི་ཧྱཱཏྲ། ལི་ཧྱཱཏཱཾ། ལི་ཧྱུཿསོགས་སོ། ༈ །ཤིས་བྱེད་

ཡི་གེ་ཕ་རོལ་ལ་ལ་བྱིངས་མཐའ་སྲ་ཡིག་དང་ཀྲ་ཡི་ག་སོགས་བརྩེགས་པ་ཡོད་ནའང་སྲ་ཀུ་མཐའ་མེད་དབྱེ་བར་བྱ་སྟེ། ཥ་སར་འོག་དེ་ཕྱིརས་པ་བཀྲ་ཡང་ཥར་ལོག། ཥྚུ་ལ་སོགས་ཀྱིས་བསྒྲུབ་པས། བྱཱ་ཙ་ཥྚེ། བྱཱ་ཙ་ཀྵཱ་ཏེ། བྱཱཙ་ཀྵ་ཏེ། བྱ་ཙཀྵེ། བྱཱ་ཙ་ཀྵཱ་ཐེ། བྱཱ་ཙ་གྔྷྭེ། བྱཱ་ཙ་ཀྵེ།
བྱཱ་ཙ་ཀྵཧེ། བྱ་ཙ་ཀྵེ་ཧེ། ༈ ཆོས་འབྱུང་ལིང་ལ། བྱཱ་ཙ་ཀྵི་ཏ། བྱ་ཙ་ཀྵི་ཏ། བྱཱ་ཙ་ཀྵཱི་ཡཱཏཱཾ། བྱཱ་ཙ་ཀྵཱི་ཡཱ་ཏཱཾ། བྱཱཙ་ཀྵི་རནྲ་སོགས་དང་། ༈ ཤིས་བྱེད་ལེཊ་ལ། བྱཱ་ཙ་ཥྚཱཾ། བྱཱ་ཙ་ཀྵཱ་ཏཱཾ། བྱ་ཙ་ཀྵ་ཏཾ་སོགས་སོ་༈ ཙཀཱ་སྲྀ་གནང་བྱེད་ལ། དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་བྱཱ་ཙ་ཥྟ། ཨ་བྱ་ཙ་ཀྵ་ཏ། ཨ་བྱ་
