ཏུ་ཤ་ཏ་བརྠི་ཀཱ། །ས་ཧ་སྲ་བཱིརྻ་བྷཱརྒྒ་བྱཽ་རུ་ཧཱ་ནནྟཱ་ཐ་སཱ་སི་ཏཱ། །གོ་ལོ་མཱི་ཤ་ཏྲ་བཱིརྻཱ་ཙ་གཎྜཱ་ལཱི་ས་ཀུ་ལཱ་ཀྵ་ཀཿ། །ཀུ་ར་བིནྡོ་མེ་གྷ་ནཱ་མཱ་མུསྟཱ་མུསྟ་ཀ་མསྟྲི་ཡཱཾ། །སྱཱདྦྷ་དྲ་མུསྟི་ཀོ་གུཾ་དྲཱ་ཙཱུ་ཌཱ་ལཱ་ཙ་ཀྲ་ལོཙྩ་ཊཱ། །བཾ་ཤེ་ཏྭཀྶཱ་
དཱུརྦྦཱ་དང་ནི་ཚིགས་བརྒྱ་པ། །སྟོང་ལྡན་བརྩོན་འགྲུས་བྷྲྀ་གུའི་བུ། །སྐྱེ་བ་མཐའ་ཡས་སཱ་སི་ཏཱ། །གླང་སྤུ་བརྩོན་འགྲུས་བརྒྱ་ལྡན་དང་། །གཎྜ་ལི་དང་བྱ་ཡི་མིག །ཀུ་རུ་བིན་ད་སྤྲིན་གྱི་མིང་། །མུསྟཱ་མོ་འདུག་མོ་མིན་ནོ། །མོན་འདུག་བཟང་པོ་གུན་དྲཱའོ། །ཙཱུ་ཌཱ་ལཱ་དང་འཁོར་ལོ་སྤྲོད། །འོད་མའི་པགས་
ར་ཀརྨ་ར་ཏྭ་ཙི་ཤཱ་ར་ཏྲྀ་ཎ་དྷྭ་ཛཱཿ། །ཤ་ཏ་པརྦཱ་ཡ་བ་ཕ་ལོ་བེ་བུ་མསྐ་ར་ཏེ་ཛ་སཱཿ། །པེ་ཎ་བཿཀཱི་ཙ་ཀཱསྟེ་སྱུརྻེསྭ་ནི་ཏྱ་ནི་ལོདྡྷ་ཏཱཿ། །གྲནྠིརྣྣཱ་པརྦྦ་པ་རཱུ་ཥཱི་གུཾ་དྲསྟེ་ཛ་ན་ཀཿཤ་རཿ། །ན་
པའི་སྙིང་པོ་དང་། །ལས་ཤིང་ཤུན་པ་མཁྲེགས་པ་དང་། །རྩ་ཡི་རྒྱལ་མཚན་ཚིགས་པ་བརྒྱ། །ནས་ཀྱི་འབྲས་བུ་སྨྱུག་རྒོད་དོ། །གླིང་ཤིང་སྤྲིན་ཅན་སྤུབས་ཅན་དང་། །སྨྱུག་མ་དང་ནི་རླུང་འབུད་ཅན། །ཚིགས་པ་དང་ནི་མཚམས་ཅན་དང་། །སྨྱུག་མ་མདངས་ལྡན་མདའ་སྨྱུག་གོ །འདམ་
ཌསྟུ་དྷ་མ་ནཿཔོ་ཊ་ག་ལོ་ཐ་ཀཱ་ཤ་མསྟྲི་ཡཱཾ། །ཨིཀྵུ་གནྡྷཱ་པོ་ཊ་ག་ལ་པུཾ་སི་བྷཱུམྣི་ཏུ་བལྭ་ཛཱཿ། །ར་སཱ་ལ་ཨིཀྵུ་སྟདྦྷེ་དཱཿཔུཎྜ་ཀནྟཱ་ར་ཀཱ་ད་ཡཱ། །སྱཱདྦཱི་ར་ཎཾ་བཱི་ར་ཏ་རཾ་མཱུ་ལ་སྱོ་ཤཱི་ར་མསྟྲི་ཡཱཾ། །ཨ་བྷ་ཡནྣ་ལ་དཾ་སེ་བྱ་མ་མྲྀ་ཎཱ་ཛ་
བུ་ན་ཌ་པོ་ཊ་གལ། །དེ་ནས་ཀཱ་ཤ་མོ་མ་ཡིན། །བུར་ཤིང་དྲི་དང་པོཊ་ག་ལ། །ཕོ་འོ་མང་ཚིག་བལྦ་ཛཱཿ། །རོ་ལྡན་དང་ནི་བུར་ཤིང་ངོ་། །དེ་ཡི་དབྱེ་བ་པུཎྜ་དང་། །ཀཱནྟཱ་ར་ཀ་ལ་སོགས་སོ། །བཱི་ར་ཎཾ་དང་བཱི་རྟ་རཾ། །རྩ་བའི་ཨུ་ཤཱི་ར་མོ་མིན། །འཇིགས་མེད་ན་ལ་དཾ་དང་ནི། །བསྟེན་བྱ་པད་
