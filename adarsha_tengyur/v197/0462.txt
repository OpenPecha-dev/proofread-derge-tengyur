ཧཱ་ར་ཎྱེ་དུརྒ་པ་ཐེ་ཀཱནྟཱ་རཿཔུནྣ་བུཾ་ས་ཀཾ །མཏྶ་རོ྅་ནྱ་ཤུ་བྷ་དྭེ་ཥེ་ཏཏྦཏྐྲྀ་པ་ཎ་ཡོ་སྟྲི་ཥུ། །དེ་བཱདྡྷྲྀ་ཏེ་བ་རཿཤྲེཥྛེ་ཏྲི་ཥུ་ཀླཱི་བཾ་མ་ནཱ་ཀྤྲི་ཡེ། །བཾ་ཤཱཾ་ཀུ་རེ་ཀ་རཱི་རོ྅སྟྲའི་ཏ་རུ་བྷེ་དེ་གྷ་ཏྲེ་ཙ་ནཱ། །ནཱ་ཙ་མཱུ་ཛ་ག་མ་ཧ་སྟ་
ཆེན་བགྲོད་དཀའ་ལམ་དང་ནི། །སྐྱེས་བུ་མ་ནིང་ཀཱནྟཱ་ར། །མཏྶ་རོ་ནི་གཞན་དགེ་ལ། །ཞེ་སྡང་དེ་བཞིན་སེར་སྣ་གསུམ། །བ་ར་ལྷ་ཡིས་སྤེལ་བ་དང་། །མཆོག་གསུམ་མ་ནི་ཅུང་ཟད་དགའ། །བཾ་ཤཱཾ་ཀུ་རེ་སྨྱུག་མའི་མྱུག །མོ་མིན་ཤིང་དབྱེ་བུམ་པའོ། །དམག་དཔུང་དང་ནི་དམག་
སཱུ་ཏྲེ་པྲ་ཏི་ཤ་རོཉསྟྲི་ཡཱཾ། །ཡ་མཱ་ནི་ལེནྡྲ་ཙནྡྲཱ་རྐ་བིཥྞུ་སིཾ་ཧཱཾ་ཤུ་བཱ་ཛི་ཥུ། །ཤུ་ཀཱ་ཧི་ཀེ་ཥུ་བྷི་ཀེ་ཥུ་ཧ་རིདྣཱ་ཀ་བི་ལེ་ཏྲི་ཥུ། །ཤརྐ་རཱ་ཀརྦ་རཱཾ་ཤེ་བི་ཡཱ་ཏྲཱ་སྱཱངྱཱ་པ་ནེ་ག་ཏཽ། །ཨི་རཱ་བྷུ་བཱཀྶུ་རཱཔྶུ་སྱཱནྟནྡྲཱི་ནི་དྲཱ་པྲ་མཱི་ལ་
གི་གཞིས། །ལག་སྐུད་པྲ་ཏི་ཤ་རའོ། །གཤིན་རྗེ་རླུང་དང་དབང་པོ་དང་། །ཟླ་ཉི་ཁྱབ་འཇུག་སེང་གེའོ། །རྟ་དང་ནེ་ཙོ་སྦྲུལ་སྤྲེའུ། །སྦལ་པ་རྣམས་ལ་ཧ་རི་ཕོ། །སྐྱ་བོ་ལ་ནི་རྟགས་གསུམ་མོ། །ཤརྐ་རཱ་ནི་གྱོ་དུ་མ་མོ། །ཡཱ་ཏྱཱ་ལྟད་མོ་འགྲོ་བའོ། །ཨི་རཱ་ས་དང་ངག་ཆང་ཆུ། །ཏནྡྲཱི་གཉིད་དང་
ཡོཿ། །དྷཱ་ཏྲཱི་སྱཱ་དུ་པ་མཱཏཱ་པི་ཀྵི་ཏི་ར་པྱཱ་མ་ལ་ཀྱ་པི། །ཀྵུ་དྲཱ་བྱངྒཱ་ན་ཊཱི་བཻ་ཤྱ་ས་ད་གྷཱ་ཀཎྜ་ཀཱ་རི་ཀཱ། །ཏྲི་ཥུ་ཀྲཱུ་རེ྅དྷ་མེ྅་ལྤེ་པི་ཀྵུ་ཏྲཾ་མཱ་ཏྲ་པ་རི་ཙྪ་དེ། །ཨལྤེ་ཙ་བ་རི་མཱ་ཎེ་སཽ་མཱ་ཏྲཾ་ཀཱརྟྶྣྱེ྅པ་དྷཱ་ར་ཎེ། །ཨཱ་ལེ་ཁྱཱཤྩརྻ་ཡོཤྩི་ཏྲཾ་ཀ་ལ་
རྨུགས་པའོ། །དྷཱ་ཏྲཱི་ཉེ་བའི་མ་མ་དང་། །ས་དང་སྐྱུ་རུ་ར་ལ་ཡང་། །ཀྵུ་དྲཱ་ཡོན་པོ་གར་མཁན་མ། །སྨད་འཚོང་དང་ནི་སྦྲང་རྩི་ལ། །ཀཎྜ་ཀཱ་རི་ཀའོ་གསུམ། །ཐང་ན་སྨོད་ཉུང་ཀྵུ་དྲའོ། །མཱ་ཏྲ་ཡོ་བྱད་ཉུང་དང་ཚད། །མཱ་ཏྲཾ་ཀུན་དང་ངེས་བཟུང་ལ། །ཙ་ཏྲཾ་ཀུན་བྲིས་ངོ་མཚར་ཅན། །ཀ་ལ་ཏྲཾ་ནི་ཆུང་
