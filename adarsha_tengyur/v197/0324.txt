བ་ཙམྤ་ཙ་དཱ་རུ་ཧ་རི་དྲ྅པ་རྫ་ནཱི་ཏྱ་པི། །བ་ཙོ་གྲ་གནྡྷཱ་ཥདྒྲནྠཱ་གོ་ལོ་མཱི་ས་ཏ་བརྦྦ་ཀཱ། །ཤུ་ཀླཱ་ཧེ་མ་བ་ཏཱི་བཻདྱ་མཱ་ཏྲྀ་སིཾ་ཧྱཽ་ཏུ་བཱ་ས་ཀཱ། །བྲྀ་ཥོ྅ཊ་ཏུ་ཥཿསིཾ་ཧཱ་སྱཽ་བཱ་ས་ཀོ་བཱ་ཛི་དནྟ་ཀཿ། །ཨཱ་སྥོ་ཏཱ་
ཤིང་། །པ་ཙཾ་པ་ཙ་ཤིང་སེར་དང་། །པརྫ་ནཱི་ཞེས་པ་ཡང་ངོ་། །ཤུ་དག་དྲི་མཆོག་མདུད་པ་དྲུག །གོ་ལོ་མཱི་དང་ཚིགས་བརྒྱ་པ། །དཀར་པོ་གངས་དང་ལྡན་པའོ། །སྨན་པའི་མ་དང་སེང་གེ་དང་། །ནཱ་ས་ཀ་དང་ཁྱུ་མཆོག་དང་། །ཨ་ཊ་རུ་ཥ་སིཾ་ཧཱ་སྱ། །བ་ཤ་ཀ་དང་རྟ་ཡི་སོ། །ཨཱ་སྥོ་ཏཱ་དང་
གི་རི་ཀརྞྞཱི་སྱཱདྤིཥྞུ་ཀྲཱནྟཱ྅པཱ་ར་ཛི་ཏཱ། །ཨིཀྵུ་གནྡྷ་ཏུ་ཀཱཎྜེ་ཀྵུ་ཀོ་ཀི་ལཱཀྵེ་ཀྵུ་ར་ཀྵུ་རཱཿ། །སཱ་ལེ་ཡཿསྱཱཙྪཱི་ཏ་ཤི་བཱཤྪ་ཏྲཱ་མ་དྷུ་རཱི་ཀཱ་མི་ཤཱི། །པི་སྟྲེ་ཡཱ་བྱ་ཐ་ཤཱི་ཧུཎྜོ་བཛྲ་དྲུཿསྣཀྶྣ་ཧཱི་གུ་ཌཱ། །ས་མནྟ་དུ་གྡྷཱ་ཐོ་བེལླ་མ་མོ་
རི་ཡི་ཕྲེང་། །ཁྱབ་འཇུག་གནོན་དང་གཞན་ལས་རྒྱལ། །བུར་ཤིང་དྲི་ལྡན་ཀཱཎྜེ་ཀྵུ། །ཁུ་བྱུག་མིག་དང་ཀྵུར་ཀྵུ་རཱ། །ཚཱི་ཏ་ཤི་བ་སཱ་ལེ་ཡ། །གདུགས་དང་མངར་ལྡན་མི་སཱི་དང་། །པི་སྟེ་ཡཱ་དང་དེ་ནས་ནི། །ཤཱི་ཁཎྜ་དང་རྡོ་རྗེའི་ཤིང་། །སྲུག་སྣ་དང་ནི་ནཱི་གུ་ཌ། །ཀུན་ནས་འོ་མ་ལྡན་
གྷཱ་ཙི་ཏྲ་ཏཎྜུ་ལཱ། །ཏཎྜུ་ལཤྩཿཀྲྀ་མི་གྷྣ་ཤྩ་བི་ཌཾ་གཾ་བུནྣ་པུཾ་ས་ཀཾ། །བ་ལཱ་བཱ་ཌྷྱཱ་ལ་ཀོ་གྷཎྚཱ་ར་བཱ་ཏུ་ཤ་ཎ་པུཥྤི་ཀཱ། །མྲྀ་དྭཱི་ཀཱ་གོསྟ་ནཱི་དྲཱ་ཀྵ་སྭཱ་དྭཱི་མ་དྷུ་རེ་སེ་ཏི་ཙ། །སརྦྦཱ་ནུ་བྷུ་ཏིཿས་ར་ཧཱ་ཏྲི་པུ་ཊཱ་ཏྲི་བྲྀ་ཏཱ་ཏྲི་
བའོ། །དེ་ནས་བྱི་དང་གདོན་མོད། །སྣ་ཚོགས་འབྲས་དང་ཏཎྜུ་ལ། །སྲིན་བུ་གསོད་དང་བི་ཌཾ་ག །སྐྱེས་བུ་དང་ནི་མ་ནིང་ངོ་། །བ་ལཱ་བ་ཌྷྱཱ་ལ་ཀ་དང་། །ཤ་ཎའི་མེ་ཏོག་དྲིལ་བུའི་སྒྲ། །དགུན་འབྲུམ་པ་ཡི་རྒྱུ་མ་དང་། །དྲཱཀྵ་རོ་ལྡན་སྦྲང་གི་རོ། །ཀུན་མཛེས་འབྱུང་དང་ས་ར་ཧཱ། །གསུམ་སྦྱར་ཏྲི་བྲྀད་ཏྲི་
