གསལ་བ་དྲངས་ཏེ་སྒྲུབ་ཐབས་ཀྱི་བང་མཛོད་དུ་བྲིས་པ་དེ་ལྟར་ཡང་། སེམས་ནི་ཁམས་གསུམ་བདག་ཉིད་ཅན། །གཞལ་ཡས་ཁང་བརྟེན་བསམས་ནས་སུ། །ས་ཡའི་འོད་ཀྱིས་ཀུན་ཁྱབ་ནས། །སངས་རྒྱས་སྤྲིན་ནི་སྤྲོ་བ་ཅན། །ཐིག་ལེའི་སྦྱོར་བས་བསྡུས་
ནས་ཀྱང་། །ཕྲ་བའི་སྦྱོར་བ་རྣམ་པར་བསྒོམ། །རིགས་དང་མཐུན་པའི་ཧཱུྃ་ལས་བྱུང་། །ཕྱག་མཚན་ནང་དུ་དཀྱིལ་འཁོར་ཏེ། །ནམ་མཁའ་དང་མཚུངས་འོད་རྒྱས་པ། །རྒྱལ་བ་རིན་ཆེན་རྟེན་པ་ཆེ། །བདེ་བ་ཆེན་པོའི་ཕྲ་བ་ཡང་། །ཕྲ་རབ་རྡུལ་གྱི་ཚུལ་འདུས་པ། །མ་ལུས་
སེམས་ཅན་དམ་ཚིག་ནི། །མཁའ་འགྲོ་ཉིད་ཀྱང་རབ་ཏུ་གནས། །སྤྲོས་ཤིང་ཡེ་ཤེས་ཚོགས་ཕྲེང་བ། །རྣམ་པ་སྣ་ཚོགས་དུ་མའི་སྐུ། །སྲོག་རྩོལ་གྱིས་ནི་མནན་པ་ཡིས། །བརྟན་པའི་སེམས་ཀྱི་རང་བཞིན་འཇུག །གཡོ་དང་འདར་བ་ཉིད་དང་ནི། །མཆོད་དང་དབབ་པ་དེ་བཞིན་
འཕར། །རྟགས་ནི་མངོན་སུམ་ཉིད་དུ་འགྱུར། །ངེས་པར་མཚན་མོ་བདུན་གྱིས་ཏེ། །ཁྱོད་ལ་འཁོར་ལོ་འདི་རྒྱས་སྟོན། །རྡོ་རྗེ་སྙིང་པོ་ཆེ་བའི་ཆེ། །ཞེས་སོ། །སོ་སོར་སྡུད་པ་ལ་སོགས་པ་རྣམས་ཀྱིས་འབྲས་བུ་ཡང་གསུངས་ཏེ་གསང་བ་འདུས་པའི་རྒྱུད་ཕྱི་མ་བླ་མ་ལས། སོ་སོར་

དཔའ་ཀུན་ནས་འཇུག །རྗེས་སུ་དྲན་པའི་སྦྱོར་བ་ལས། །འོད་ཟེར་དཀྱིལ་འཁོར་སྐྱེ་བར་འགྱུར། །ཏིང་ངེ་འཛིན་དབང་ཉིད་ཙམ་གྱིས། །སྒྲིབ་པ་རྣམས་དང་བྲལ་བར་འགྱུར། །ཞེས་སོ། །རྣལ་འབྱོར་ཆེན་པོ་ལས་སྐྱོན་བྱ་བ་གང་བྱེད་ཅེ་ན་དེ་གསུངས་པ་སྐད་ཅིག་ལ་སོགས་པའོ། །
ཡུད་ཙམ་གྱིས་དཀྱིལ་འཁོར་འཁོར་ལོ་མོས་ནས་ལྟེ་བའི་འཁོར་ལོ་ཧཱུྃ་ཡིག་དམར་རམ། ནཱ་ད་བལྟས་ནས་ཡི་གེ་གསུམ་བརྗོད་ཅིང་། ནཱ་ད་དེ་ལ་གདམས་ངག་ཇི་ལྟ་བ་བཞིན་དུ་འཐོན་པའི་རླུང་གི་འཁོར་ལོ་ལྔ་རབ་ཏུ་སྤྲོས་ནས་འགྲོ་བའི་དོན་བྱས་ཏེ་ཡང་བརྗོད་པའི་སྔགས་དང་བཅས་པས།
