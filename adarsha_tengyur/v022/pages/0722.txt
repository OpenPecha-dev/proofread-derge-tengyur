ཁྱོད་ལ་མཛད། །ཅེས་བྱ་བས་སྦྱིན་པར་བྱའོ། །དེ་ནས་རྗེས་སུ་གནང་བའི་དབང་ལྔ་སྦྱིན་པར་བྱ་སྟེ། འཇིག་རྟེན་ཀུན་ལས་ཐམས་ཅད་དུ། །སེམས་ཅན་ཀུན་ལ་ཕན་པའི་ཕྱིར། །ཇི་ལྟར་སྣ་ཚོགས་འདུལ་འགྱུར་བ། །ཆོས་ཀྱི་འཁོར་ལོ་རབ་ཏུ་བསྐོར། །དེ་བཞིན་དུ་རྡོ་རྗེ་དང་ཁྲོ་བོ་
དང་པདྨ་དང་ནོར་བུའི་འཁོར་ལོ་རབ་ཏུ་བསྐོར་ཞེས་བྱ་བའི་ཚིགས་སུ་བཅད་པ་ལྔས་རྗེས་སུ་གནང་བ་ལྔ་དབང་བསྐུར་བ་ལྔ་སྦྱིན་པར་བྱའོ། །དེ་ནས་སློབ་མ་ལ་མཆོད་པ་ལྔས་མཆོད་པར་བྱའོ། །མགོ་བོ་ལ་མེ་ཏོག་སྣ་ལ་བདུག་པ་མིག་ལ་མར་མེ་སྙིང་གར་བྱུག་པ། ཡང་ན་གནས་གསུམ་དུ་
དྲིའི་ཐིག་ལེ་བྱས་ལ། བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་སུམ་ཅུ་རྩ་བདུན་ལྷའི་དེ་ཉིད་དང་། གནས་ཉི་ཤུ་རྩ་བཞི་ལ་སོགས་པ་དཀྱིལ་འཁོར་གྱི་དེ་ཉིད་དང་། རྩ་དང་། ཁམས་དང་རང་བཞིན་ལུས་ཀྱི་དཀྱིལ་འཁོར་གྱི་དེ་ཉིད་དང་། བསྙེན་པ་དང་སྟ་གོན་དང་། དཀྱིལ་འཁོར་དང་སྦྱིན་སྲེག་དང་བུམ་པ་དང་
གཏོར་མ་དང་བདུད་རྩི་ལྔ་སྦྱང་བ་དང་། བསྒོམ་པ་དང་བཟླས་པ་དང་བཅས་པའི་ཕྲིན་ལས་བཞི་སློབ་དཔོན་གྱི་ལས་རྣམས་བསླབ་པར་བྱའོ། །དེ་ནས་ཡོན་དབུལ་བར་བྱ་སྟེ། གླང་པོ་རྟ་དང་ཤིང་རྟ་ནོར། །ནོར་བུ་བྱུ་རུ་ལ་སོགས་པ། །ཡོན་གྱི་མཆོག་ནི་བླངས་པ་རྣམས། །ཕྱག་བྱས་ནས་ནི་དབུལ་
བར་བྱ། །དེང་ནས་བརྩམས་ཏེ་བདག་བྲན་དུ། །ཁྱོད་ལ་བདག་ནི་འབུལ་ལགས་ཀྱིས། །མ་འོངས་མངོན་པར་འཚང་རྒྱ་བར། །ལུང་བསྟན་བྱ་དང་དབུགས་དབྱུང་མཛོད། །ཅེས་བརྗོད་དོ། །དེ་ནས་བདག་ཉིད་དེ་བཞིན་གཤེགས་པར་བསམས་ལ་གཟེངས་བསྟོད་པ་འདི་དང་ལུང་བསྟན་
པར་བྱ་སྟེ། གཡོན་པ་སྙིང་གར་ཁུ་ཚུར་བཅངས་པ་དང་ཆོས་གོས་འཛིན་པའི་ཚུལ་དུ་བྱས་ལ། ཨོཾ་ང་ཡིས་འདི་ལྟར་ལུང་བསྟན་ཏེ། །རྡོ་རྗེ་སེམས་དཔའ་དེ་བཞིན་གཤེགས། །སྲིད་པའི་མཆོག་ནི་གྲུབ་པ་ཡིན། །སྲིད་པའི་ངན་འགྲོ་རྣམས་ལས་བཏོན། །ཀྱེ་དེ་བཞིན་གཤེགས་པ་རྡོ་རྗེ་ཆེ་
གེ་མོ་ཞེས་བྱ་བ་དང་། བྷུར་བྷུ་བ་སཿ། འདི་ཡི་དབང་བསྐུར་གྱུར་པ་ཡིས། །སྔགས་པས་འགྲོ་བ་ཀུན་དང་བདག །བླ་ན་མེད་པའི་བྱང་ཆུབ་ཏུ། །བདེ་བར་གཤེགས་ཀྱིས་ལུང་བསྟན་མཛོད། །ངོ་བོ་ཉིད་ཀྱིས་སེམས་ཅན་རྣམས། །ཇི་ལྟ་ཇི་ལྟར་འདུལ་འགྱུར་བ། །དེ་ལྟ་དེ་ལྟར་ཆགས་
