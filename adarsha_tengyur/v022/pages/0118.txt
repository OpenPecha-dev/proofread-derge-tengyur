དབྱངས་ཀྱི་རྗེས་འགྲོར་བཤད། །ཁྱོད་ཀྱིས་ཀྱང་ནི་རྟག་ཏུ་བཅང་། །བྱང་ཆུབ་མཆོག་ཏུ་རྒྱལ་བ་བཞེད། །སྲིད་པ་ངོ་བོ་ཉིད་ཀྱིས་དག །ངོ་བོ་ཉིད་ཀྱིས་སྲིད་བྲལ་བྱས། །རང་བཞིན་དག་པའི་སེམས་དཔའ་ནི། །སྲིད་པ་དམ་པ་བྱེད་པར་འགྱུར། །རྡོ་རྗེ་དྲིལ་བུ་སྤྱི་
བོར་ནི། །རྣམ་པར་སྣང་མཛད་སྐུ་ཡི་ཙཀྲ་བསྒོམ། །ཨོཾ་ཤྲཱི་ཧེ་རུ་ཀ་ཨ་དྷི་པ་ཏི་སྟྭཾ། ཨ་བྷི་ཤིཉྩ་མི། ཨོཾ་ཤྲཱི་ཧེ་རུ་ཀ་ནཱ་མ་ཨ་བྷི་ཥི་ཀ་ཏ། ཤྲཱི་སྒྲ་དང་བཅས་པ་འདིས། །རིགས་ལས་དྲངས་ཏེ་མིང་དབང་བསྐུར། །ཆུ་དང་ཅོད་པན་རྡོ་རྗེ་དང་། །དྲིལ་བུ་དང་ནི་མིང་
དབང་བསྐུར། །བདེ་གཤེགས་ལྔ་ཡི་བདག་ཉིད་དེ། །ཕུང་པོ་ལྔ་དང་ཉོན་མོངས་དག །རིགས་པའི་དབང་ལྔའོ།།      །།སྲིད་པའི་བདེ་བ་ཤིན་ཏུ་ཆུང་། །སྡུག་བསྔལ་ཆེན་པོ་ཞི་བྱེད་པ། །ཤིན་ཏུ་བདེ་བ་དམ་པའི་ཐབས། །དམ་ཚིག་རྟག་ཏུ་བསྐྱང་བར་
