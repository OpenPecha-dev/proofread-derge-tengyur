སོགས་པ་དང་དེ་ལ་དམིགས་པའི་སྔགས་དང་ཏིང་ངེ་འཛིན་དང་ལས་རྣམས་ཀུན་རྫོབ་ཏུ་ཡོད་དེ་ཞེས་བསྟན་ཏོ། །འདི་སྐད་བཤད་པ་ཡིན་ཏེ་རིམ་པ་གཉིས་ཀྱི་སྐུ་ལ་དམིགས་ནས་དེ་དག་ཐམས་ཅད་བསྒྲུབ་པར་བྱའོ། །དོན་དམ་པར་སྐུ་ནི་སྐུ་མ་ཡིན་ནོ། །དེ་བཟློག་པར་བྱ་བ་ཞེས་པ་
ནས་རྟེན་དང་བརྟེན་པ་དགག་པའོ་ཞེས་བྱ་བའི་བར་དུ་ནི་དོན་དམ་པའི་ཕྱོགས་ལ་བརྟེན་ཏེ། ལུས་ལ་སོགས་པ་ཡོད་པ་མ་ཡིན་པ་དང་། འདོད་པའི་ཁམས་ལ་སོགས་པ་ལ་མི་གནས་པ་དང་། སེམས་ནི་ལུས་ལ་མི་གནས་སོ་ཞེས་པ་ལ་སོགས་པ་ཕན་ཚུན་རྟེན་དང་བརྟེན་པ་ཉིད་མེད་པ་
སྟེ། ངོ་བོ་མེད་པའི་ཕྱིར་ནམ་མཁའ་བཞིན་ནོ་ཞེས་བསྟན་ཏོ། །དེ་ནས་ཞེས་བྱ་བ་ནས་བདག་ཉིད་ལ་གནས་སོ་ཞེས་བྱ་བའི་བར་དུ་ནི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་བདག་པོ་རྡོ་རྗེ་འཆང་གང་དུ་གནས་ཞེས་ཞུས་པ་ལ་ཡིད་ཀྱི་རང་བཞིན་གྱི་ལུས་ལ་གནས་སོ་ཞེས་ལན་ནོ། །
ཡང་ལུས་ལ་སོགས་པ་སོ་སོར་སྣང་བའི་སེམས་གང་དུ་གནས་ཞེས་ཞུས་པ་ལ། ནམ་མཁའ་ལྟ་བུ་འོད་གསལ་བ་ལ་གནས་པ་ཞེས་ལན་བསྟན་པ་དང་། འོད་གསལ་བ་ནི་གང་དུ་གནས་ཞེས་ཞུས་པ་ལ་གང་དུའང་མི་གནས་བདག་ཉིད་ལ་གནས་ཏེ་ཆོས་ཉིད་ཡིན་པས་ན་
འོད་གསལ་བ་ནི་ཐམས་ཅད་ལ་གནས་སོ་ཞེས་བསྟན་ཏོ། །ཇི་སྐད་དུ། བུ་རམ་མངར་བ་མེ་དྲོ་བ། །ཇི་ལྟར་རང་བཞིན་ཡིན་པ་སྟེ། །ཆོས་ཀུན་ལ་ནི་སྟོང་པ་ཉིད། །དེ་ལྟ་བུ་ནི་རང་བཞིན་འདོད། །རྣམ་པ་དུ་མས་ཐ་དད་པ། །དངོས་པོ་རྣམས་ནི་གོས་དང་འདྲ། །སྟོང་པ་ཉིད་མེས་
བསྲེགས་གྱུར་ན། །ཐལ་བའི་དབྱེ་བ་གང་ཡང་མེད། །ལྟ་བ་ཀུན་གྱིས་བྱས་པ་རྣམས། །སྟོང་པ་ཉིད་ནི་མཐའ་ཕྱིར་བསྟན། །ཞེས་འབྱུང་བས་སོ། །དེ་ནས་ཞེས་བྱ་བ་ནས་མི་སྨྲ་བར་གྱུར་པའོ་ཞེས་པའི་བར་དུ་ནི་ཤིན་ཏུ་ཟབ་པའི་དོན་ཐོས་ནས་རྒྱུ་དང་འབྲས་བུ་དག་ལ་ངོ་མཚར་
བ་རྨད་དུ་གྱུར་པ་ཉིད་ཐོབ་ནས་དེ་བསམ་པ་དང་བསྒོམ་པའི་ནུས་པ་རྩེ་གཅིག་པ་ཉིད་ཀྱིས་དོན་དེ་ཉིད་དམིགས་པས་ཅང་མི་སྨྲ་བར་གྱུར་ཏོ་ཞེས་བསྟན་ཏོ། །རྒྱུ་བ་དང་ཞེས་བྱ་བ་ལ་སོགས་པས་དངོས་པོ་ཐམས་ཅད་ཀྱི་རང་བཞིན་གྱི་རྡོ་རྗེ་འཆང་དང་ཞི་བ་ལ་སོགས་པའི་ལས་ཀྱི་ཚོགས་
