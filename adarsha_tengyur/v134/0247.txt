ལ་མཁས་པས་ན་ཆོས་མངོན་པ་སྟེ། རྒྱུ་དགུ་པོ་དེ་དག་གིས་ན་སྡེ་སྣོད་ནི་གསུམ་དུ་འདོད་དོ། །དེ་དག་གིས་ཀྱང་འཁོར་བ་ལས་རྣམ་པར་གྲོལ་བར་བྱེད་པའི་ཕྱིར་རོ། །ཡང་ཇི་ལྟར་ན་དེ་དག་གིས་རྣམ་པར་གྲོལ་བར་བྱེད་ཅེ་ན། བསྒོ་བ་དང་། རྟོགས་པ་དང་། འཇིལ་བ་དང་། རབ་ཏུ་
འབྱེད་པས་རྣམ་པར་གྲོལ་བར་བྱེད་པ་ཉིད་དོ། །ཐོས་པས་ནི་སེམས་ལ་བསྒོ་བ་ཉིད་དོ། །བསམ་པས་ནི་རྟོགས་པ་ཉིད་དོ། །ཞི་བའི་དོན་ཏུ་སྒོམ་པའི་ཕྱིར་འཇིལ་བ་ཉིད་དོ། །ལྷག་མཐོང་གིས་ནི་རབ་ཏུ་འབྱེད་པ་ཉིད་དོ། །མདོར་བསྡུ་ན་མདོ་སྡེ་དང་། འདུལ་བ་དང་། ཆོས་མངོན་པ་དེ་དག་སོ་སོ་ཞིང་དོན་རྣམ་པ་
བཞི་སྟེ། བྱང་ཆུབ་སེམས་དཔའ་དེ་རྣམས་རྟོགས་ན་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཐོབ་པར་འགྱུར་རོ། །ཉན་ཐོས་ཀྱིས་ནི་ཚིགས་སུ་བཅད་པ་གཅིག་གི་དོན་ཤེས་པས་ཀྱང་ཟག་པ་ཟད་པ་འཐོབ་བོ། །དེ་ལྟར་སོ་སོ་ཞིང་དོན་རྣམ་པ་བཞི་ཡིན་ཞེ་ན། གནས་དང་མཚན་ཉིད་དང་། ཆོས་དང་དོན་མདོར་སྟོན་
པས་ན། མདོ་སྡེའོ། །དེ་ལ་གནས་ནི་ཡུལ་གང་དུ་གང་གིས་གང་ལ་བསྟན་པའོ། །མཚན་ཉིད་ནི་ཀུན་རྫོབ་ཀྱི་བདེན་པའི་མཚན་ཉིད་དང་། དོན་དམ་པའི་བདེན་པའི་མཚན་ཉིད་དོ། །ཆོས་ནི་ཕུང་པོ་དང་། སྐྱེ་མཆེད་དང་། ཁམས་དང་། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་དང་། ཟས་དང་། བསམ་གཏན་
དང་། ཚད་མེད་པ་དང་། རྣམ་པར་ཐར་པ་བརྒྱད་དང་། ཟིལ་གྱིས་གནོན་པའི་སྐྱེ་མཆེད་དང་། ཟད་པར་གྱི་སྐྱེ་མཆེད་དང་། བྱང་ཆུབ་ཀྱི་ཕྱོགས་དང་། སོ་སོ་ཡང་དག་པར་རིག་པ་ལ་སོགས་པའོ། །དོན་ནི་རྗེས་སུ་བརྗོད་པའོ། །མངོན་དུ་ཕྱོགས་པའི་ཕྱིར་དང་། ཡང་དང་ཡང་དུ་སྟོན་པའི་ཕྱིར་དང་། 
ཟིལ་གྱིས་གནོན་པའི་ཕྱིར་དང་། མངོན་པར་རྟོགས་པའི་ཕྱིར་ཆོས་མངོན་པར་རིག་པར་བྱ་སྟེ། མི་གནས་པའི་མྱ་ངན་ལས་འདས་པ་ལ་མངོན་དུ་ཕྱོགས་པའི་ཆོས་ནི་ཆོས་མངོན་པའོ། །བདེན་པ་དང་། བྱང་ཆུབ་ཀྱི་ཕྱོགས་དང་། རྣམ་པར་ཐར་པའི་སྒོ་ལ་སོགས་པ་ཡང་དང་ཡང་དུ་སྟོན་པའི་ཆོས་ནི་
ཆོས་མངོན་པ་སྟེ། ཆོས་རེ་རེ་ཞིང་ཡང་ངེས་པའི་ཚིག་དང་། རང་གི་མཚན་ཉིད་དང་། སྤྱིའི་མཚན་ཉིད་ལ་སོགས་པ་རབ་ཏུ་དབྱེ་བ་མང་པོས་སྟོན་པའི་ཕྱིར་རོ། །ཟིལ་གྱིས་གནོན་པས་ན་ཆོས་མངོན་པ་སྟེ། ཕས་ཀྱི་རྒོལ་བར་འདོད་པའི་དབང་དུ་བྱས་ནས་ཟིལ་གྱིས་གནོན་པ་ཉིད་དོ། །མདོ་
