ཆོས་ཀྱི་ཁམས་དང་སྐྱེ་མཆེད་ཀྱི་ཕྱོགས་གཅིག་ཀྱང་ཡིན་ནོ། །འདུས་མ་བྱས་ནི་ཆོས་ཀྱི་ཁམས་དང་སྐྱེ་མཆེད་ཀྱི་ཕྱོགས་གཅིག་གོ། །ཡང་ཤེས་བྱའི་ཆོས་རྣམས་ནི། མོས་པའི་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། རིགས་པའི་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། རྣམ་པར་འཕྲོ་བ་
མེད་པའི་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། སོ་སོ་རང་གིས་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། གཞན་གྱི་བདག་ཉིད་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། འོག་མ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། གོང་མ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། རྣམ་པར་སུན་འབྱིན་པའི་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། ཀུན་ནས་མི་ལྡང་
བ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། མི་སྐྱེ་བ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། ཤེས་པ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། མཐར་ཐུག་པ་ཤེས་པའི་སྤྱོད་ཡུལ་དང་། དོན་ཆེན་པོ་ཤེས་པའི་སྤྱོད་ཡུལ་གྱི་ཕྱིར་ཡང་ངོ་། །བདག་ལ་ཤེས་པ་པོ་དང་། མཐོང་བ་པོར་མངོན་པར་ཞེན་པ་སྤང་བའི་
ཕྱིར་རོ། །ཇི་ལྟར་ན་རྣམ་པར་ཤེས་པར་བྱ་བ་ཡིན། རྣམ་པར་ཤེས་པར་བྱ་བ་རྣམས་ནི་དུ་ཞིག །ཅིའི་ཕྱིར་རྣམ་པར་ཤེས་པར་བྱ་བར་བརྟག་ཅེ་ན། རྣམ་པར་མི་རྟོག་པ་དང་། རྣམ་པར་རྟོག་པ་དང་། རྒྱུ་དང་། འཇུག་པ་དང་། རྒྱུ་མཚན་དང་། རྒྱུ་མཚན་ལས་བྱུང་བ་དང་། མི་
མཐུན་པ་དང་། གཉེན་པོ་དང་། ཕྲ་བའི་རབ་ཏུ་དབྱེ་བས་ཀྱང་རྣམ་པར་ཤེས་པར་བྱ་བར་བལྟའོ། །ཐམས་ཅད་ཀྱང་རྣམ་པར་ཤེས་པར་བྱ་བ་དག་གོ། །བདག་ལ་མཐོང་བ་པོ་ལ་སོགས་པར་མངོན་པར་ཞེན་པ་སྤང་བའི་ཕྱིར་རོ། །ཇི་ལྟར་ན་མངོན་པར་ཤེས་པར་བྱ་བ་ཡིན། མངོན་
པར་ཤེས་པར་བྱ་བ་རྣམས་ནི་དུ་ཞིག །ཅིའི་ཕྱིར་མངོན་པར་ཤེས་པར་བྱ་བར་བརྟག་ཅེ་ན། ཀུན་ཏུ་འགྲོ་བ་དང་། རྗེས་སུ་ཐོས་པ་དང་། སྤྱོད་པ་ལ་འཇུག་པ་དང་། འོང་བ་དང་། འགྲོ་བ་དང་། ངེས་པར་འབྱུང་བའི་ཕྱིར་ཡང་ངོ་། །ཐམས་ཅད་ཀྱང་མངོན་པར་ཤེས་པར་བྱ་བ་དག་གོ། །
བདག་ལ་མཐུ་དང་བཅས་པར་མངོན་པར་ཞེན་པ་སྤང་བའི་ཕྱིར་རོ། །ཇི་ལྟར་ན་གཟུགས་ཅན་ཡིན། གཟུགས་ཅན་རྣམས་ནི་དུ་ཞིག །ཅིའི་ཕྱིར་གཟུགས་ཅན་དུ་བརྟག་ཅེ་ན། གཟུགས་དེའི་བདག་ཉིད་དང་། འབྱུང་བ་རྟེན་དུ་གྱུར་པ་དང་། དགའ་བའི་ཀུན་འབྱུང་བ་དང་། ཕྱོགས་དང་བཅས་
