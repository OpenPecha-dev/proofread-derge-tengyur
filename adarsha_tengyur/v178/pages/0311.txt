སྨྲས་པ་ནི། མེད་ན་མི་འབྱུང་མཚན་ཉིད་ཅན། །རྟགས་ནི་རྗེས་དཔག་རྟེན་ཡིན་ནོ། །བསྒྲུབ་བྱས་གཏན་ཚིགས་ཁྱབ་པ་ནི། །བསྟན་ཕྱིར་དེ་ཡང་གསལ་བར་བཤད། །བསྒྲུབ་བྱས་ཁྱབ་པའི་གཏན་ཚིགས་ནི་གོ་བར་བྱེད་པ་ཡིན་ལ། གཏན་ཚིགས་ཀྱི་ཁྱབ་པ་དེ་ཡང་གང་ཅུང་ཟད་ཀྱང་
ཅུང་ཟད་སྐྱེ་བའི་བདག་ཉིད་ཅན་དེ་ཐམས་ཅད་ནི་འགོག་པའི་ཆོས་ཅན་ཡིན་ནོ་ཞེས་བཅོམ་ལྡན་འདས་ཀྱིས་རྣམ་པ་དུ་མར་བསྟན་པར་མཛད་དེ། དེས་ན་རྗེས་སུ་དཔག་པ་ཡང་གསུང་བ་ཉིད་ཀྱིས་ཚད་མའི་དེ་ཁོ་ན་ཉིད་ནི་བཅོམ་ལྡན་འདས་ཀྱིས་དང་པོ་བསྟན་པ་ཡིན་ལ། དེ་ཡང་ཚད་མ་ཡིན་པའི་ཕྱིར་ཏེ། 
དེས་ན་དང་པོར་སྨོས་པའི་ཚད་མ་ཉིད་ནི་བཅོམ་ལྡན་འདས་ཀྱིས་ཡོན་ཏན་དམ་པ་ཡིན་ནོ། །གང་ཕྱིར་གཉིས་ལྡན་ཚད་མའི་དེ་ཉིད་ནི། །བཅོམ་ལྡན་དེ་བཞིན་གཤེགས་པས་བསྟན་པ་མཛད་པ། །དེ་ཕྱིར་དེ་ལས་ཚད་མ་གཞན་ཡོད་མིན། །དེ་བཞིན་གཤེགས་པ་འདི་ལས་ཐམས་ཅད་གྲུབ། །
ཚད་མ་རྣམ་འགྲེལ་གྱི་རྒྱན་ལས། ཚད་མ་གྲུབ་པ་ཞེས་བྱ་བའི་ལེའུ་སྟེ་དང་པོའོ།།      །།དེ་ལྟར་སྤྱིའི་མཚན་ཉིད་བརྗོད་ནས་ཁྱད་པར་གྱི་མཚན་ཉིད་བཤད་པ། མངོན་སུམ་དང་ནི་རྗེས་དཔག་ཚད་མ་ཡིན་ནོ། །ཚད་མ་དེ་ཡང་མི་སླུ་བའི་ཕྱིར་རོ། །མི་སླུ་བ་ཡང་དོན་ལས་སྐྱེས་པས་
དོན་ལ་མི་འཁྲུལ་པའི་ཕྱིར་རོ། །གལ་ཏེ་དོན་ལ་མི་འཁྲུལ་པ་དེ་ཉིད་ཇི་ལྟར་རྟོགས་པར་བྱ་ཞེ་ན། མངོན་སུམ་དང་རྗེས་སུ་དཔག་པ་ཉིད་ཀྱིས་ཏེ། མངོན་སུམ་ཡིན་པའི་ཕྱིར་དང་། རྗེས་སུ་དཔག་པ་ཡིན་པའི་ཕྱིར་ཚད་མ་མ་ཡིན་ཏེ། གཞན་སྒྲ་ལས་བྱུང་བ་ལ་སོགས་པ་ཡིན་པའི་ཕྱིར་ནི་མ་ཡིན་
ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །གལ་ཏེ་སྒྲ་ལས་བྱུང་བ་ལ་སོགས་པ་དོན་དང་འབྲེལ་པ་མེད་པ་ཉིད་དོ་ཞེས་བྱ་བ་གང་ལས་ཤེ་ན། འདི་ལྟར་མཚན་ཉིད་གཉིས་ནི་གཞལ་བྱ་ཡིན་ལ། ཚད་མ་ནི་གཞལ་བྱ་ལ་འཁྲུལ་པ་མེད་པའི་ཕྱིར་རོ། །གཞལ་བྱ་མེད་ན་ནི་གཞལ་བྱ་ལ་མི་འཁྲུལ་བར་མི་འགྱུར་རོ། །
