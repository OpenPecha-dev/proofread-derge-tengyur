ལོ་བརྒྱད་ན་སྟ་གོན་དུ་གནས་ཤིང་བརྟུལ་ཞུགས་ལྔ་དང་ཉེ་བའི་བརྟུལ་ཞུགས་ལྔ་བླངས་པ་དང་། དེ་ལས་གཞན་པའི་བརྟུལ་ཞུགས་ལྔ་མཉན་པ་དང་། ཨ་བོ་ཧི་སྠ་དང་། མེ་ལ་བྱ་བ་རྣམས་དང་། བཟའ་བ་རྣམས་དང་། བླ་མ་རྣམས་ལ་བསྙེན་བཀུར་བྱས་པ་དང་། དུས་གསུམ་དུ་
འཐོར་འཐུང་བྱ་བ་དང་། བཟླས་པ་དང་། མཆོད་སྦྱིན་དང་། ཉེ་རེག་དང་། རླུང་འབྱུང་བ་དང་། གཞུག་པ་དང་། རིག་བྱེད་རྫོགས་པར་བཀླགས་པ་དང་། བག་མ་བླངས་པ་དང་། བཞི་པ་བྱས་པ་དང་། སྦྱིན་སྲེག་དང་། མེ་བསྲུངས་པ་དང་། མེ་ལ་སྦྱིན་སྲེག་བྱེད་པ་བསྟན་པ་
དང་། ཟླ་བ་གཅིག་ལོན་ནས་མཆོད་སྦྱིན་བྱས་པ་དང་། བྱོལ་སོང་བཅིངས་པ་དང་། ཟླ་བ་བཞི་པའི་ཉི་མ་ལྡོག་པའི་དང་པོ་ལ་ཉེར་གནས་ཀྱི་ལས་དང་། བརྒྱད་པ་གསུམ་པ་བྱས་པ་སྟེ། ལོ་མ་དང་། འོ་མ་དང་། ཤ་དང་། ཁུར་བས་མཆོད་སྦྱིན་བྱས་པ་ཞེས་བྱ་བ་དེ་ལྟ་བུ་བྲམ་ཟེའི་
དབང་དུ་བྱས་པའི་ཤུགས་བཞི་བཅུ་རྩ་བརྒྱད་པོ་འདི་དག་དང་། བསྟན་བཅོས་ལས་འབྱུང་བ་བཞིན་དུ་འབྲུ་དང་། བྱོལ་སོང་དང་། མར་དང་། ཆུང་མ་དང་འབྲེལ་པའི་ལས་རྣམས་བྱས་པའི་ཡོན་ཏན་གྱི་ལམ་ལས་གཞན་གྱིས་གྲོལ་བ་འཐོབ་པར་རིགས་པ་མ་ཡིན་ནོ་ཞེས་ལུང་ལས་འབྱུང་
ངོ་། །རིག་བྱེད་ལས་བསྟན་པའི་སོ་བ་དང་། བྱོལ་སོང་དང་། མར་ལ་སོགས་པས་མཆོད་སྦྱིན་བྱ་བ་དང་། སྟེགས་བུའི་དཀྱིལ་དུ་ཆུང་མ་བླང་ཞིང་འབྲེལ་པའི་བྱ་བ་ལས་གཞན་པའི་ཐར་པ་འཐོབ་པའི་ཐབས་ནི་མེད་དོ། །ཇི་ལྟར་ཤེས་ཤེ་ན། རིག་བྱེད་ཀྱི་ལུང་གི་ལུགས་ཚད་མར་
བྱེད་པ་ཡིན་ཏེ། རིག་བྱེད་ལས་ནི་དཀར་མོ་དང་འབྲུ་རྣམས་ཀྱིས་མཆོད་སྦྱིན་བྱས་པར་འགྱུར་རོ། །དེ་བཞིན་དུ་གང་ལ་ལོ་གསུམ་ཐུབ་པའི་སོ་བ་ཁྱིམ་ན་སྦས་ནས་ཡོད་པའམ། དེ་བས་ལྷག་པར་ཡོད་པ་ནི་ལས་བྱེད་རྣམས་ཀྱི་འཚོ་བ་དང་བཅས་པ་ཡིན་པས་དེ་ནི་ཉི་མའི་མཆོད་
སྦྱིན་བྱེད་པར་འོས་པ་ཡིན་ནོ་ཞེས་འདོན་ཏོ། །བྱོལ་སོང་བཅིངས་པ་ཡང་བསྟན་ཏེ། །བར་གྱི་དུས་ལ་བབ་པ་ན། །བྱོལ་སོང་དྲུག་གམ་བདུན་སྦྱར་བྱ། །རྟ་ཡིས་གཙང་མའི་ཚིག་ཡིན་ནོ། །ཉུང་ལ་བྱོལ་སོང་གསུམ་གྱི་སོ། །རེ་རེ་ལ་ཡང་གྲོང་གི་མི། །བྱོལ་སོང་བཅུ་དྲུག་བཅུ་
