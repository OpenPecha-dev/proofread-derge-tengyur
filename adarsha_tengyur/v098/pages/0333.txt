བརྗོད་པར་བྱའོ། །ཐེག་པ་ཆེན་པོ་ནི་སངས་རྒྱས་ཀྱིས་གསུངས་པ་ཉིད་ཡིན་ཏེ། ཆོས་ཀྱི་ཕྱག་རྒྱ་དང་མི་འགལ་བའི་ཕྱིར་དང་། གང་འཕགས་པའི་བདེན་པའི་མདོ་སྡེ་ལ་ཡང་འཇུག །ཉོན་མོངས་པ་འདུལ་བ་ལ་ཡང་ཡང་དག་པར་མཐོང་། རྟེན་
ཅིང་འབྲེལ་བར་འབྱུང་བའི་ཆོས་ཉིད་དང་ཡང་མི་འགལ་བ་ཉིད་ནི་སངས་རྒྱས་ཀྱི་གསུང་ཡིན་ནོ་ཞེས་བྱ་བ་དེ་ཐམས་ཅད་ཐེག་པ་ཆེན་པོ་ལ་གྲུབ་པའི་ཕྱིར་རོ། །ཡང་ན་སྡེ་པ་བཅོ་བརྒྱད་པོ་རྣམས་ཀུན་གྱི་ཡང་རང་རང་གི་མདོ་སྡེ་ལ་སོགས་པ་ལ་འཇུག་པ་ཡིན་གྱི། ཕན་ཚུན་ནི་
འགལ་བ་ཆེར་ཡོད་པ་ཡིན་ནོ། །ཐེག་པ་ཆེན་པོ་ཡང་གཞན་ལ་ཕན་པའི་དབང་དུ་བྱས་པའི་ཟབ་པ་དང་རྒྱ་ཆེ་བའི་ཚུལ་གྱིས་ཉན་ཐོས་ཀྱི་མདོ་སྡེ་ལ་སོགས་པའི་སྡེ་སྣོད་ལ་འགའ་ཞིག་མི་འཇུག་ཏུ་ཟིན་ཀྱང་། ཐེག་པ་ཆེན་པོའི་མདོ་སྡེ་ལ་ཡང་འཇུག །བྱང་ཆུབ་སེམས་དཔའི་བསླབ་
པའི་གནས་བདུན་བརྒྱ་ལ་ཡང་སྣང་། སྟོང་པ་ཉིད་དུ་བསྟན་པའི་ཆོས་ཉིད་དང་ཡང་མི་འགལ་བའི་ཕྱིར་རོ། །དེ་ལྟར་ན་ཆོས་ཀྱི་ཕྱག་རྒྱ་དང་འགལ་བ་མེད་དོ། །འདིའི་ཕྱིར་ཡང་ཐེག་པ་ཆེན་པོ་སངས་རྒྱས་ཀྱིས་གསུངས་པ་ཡིན་ཏེ། རྩ་བའི་སྡུད་པར་བྱེད་པ་པོ་ཀུན་དུ་བཟང་པོ་
དང་། འཇམ་དཔལ་དང་། གསང་བའི་བདག་པོ་དང་། བྱམས་པ་ལ་སོགས་པ་རྣམས་ཀྱིས་བསྡུས་པའི་ཕྱིར་རོ། །བདག་ཅག་གི་རྩ་བའི་སྡུད་པར་བྱེད་པ་ཉན་ཐོས་ནི་མ་ཡིན་ཏེ། ཐེག་པ་ཆེན་པོའི་གསུང་རབ་ནི་དེ་དག་གི་ཡུལ་མ་ཡིན་པའི་ཕྱིར་རོ། །ཇི་སྐད་དུ། ཤིང་ཤ་པའི་
ཚལ་གྱི་མདོ་ལས། ཀུན་དགའ་བོ་ཤིང་ཤ་པའི་ཚལ་འདི་ལ་འདབ་མ་ཇི་སྙེད་པ་ཡོད་པ་བས་ལྷག་པར་མང་བའི་ཆོས་ངས་རྟོགས་པར་གྱུར་ན་ཡང་ཁྱེད་ལ་ནི་ངས་མ་བསྟན་ཏེ། དེ་སྙེད་ཉིད་ཀྱིས་དོན་བྱ་བ་བྱས་པའི་ཕྱིར་དེ་དག་ཉིད་ཀྱིས་ནི་ཁྱེད་སྐྱོ་བར་ཡང་མི་འགྱུར། འདོད་
ཆགས་དང་བྲལ་བར་ཡང་མི་འགྱུར་རོ་ཞེས་བྱ་བ་དང་། དེ་བཞིན་དུ། བླ་མ་ཀུན་དགའ་པ་ཏ་ལི། །སྡུག་བསྔལ་ཕུང་པོ་ས་པ་དང་། །འཆར་བྱེད་སྟོང་པ་འཆར་བ་དང་། །གང་པོ་སྤངས་པ་རམ་པ་དང་། །མྱ་ངན་འདས་དང་ཡུལ་འཁོར་སྐྱོང་། །ཞེས་བྱ་གཞན་ཡང་འགྲོ་བའི་
