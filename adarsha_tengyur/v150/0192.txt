ཕོ་ཉའི་ལག་ཏུ་ཞེས་བྱ་བ་ནི་སྐྱེས་པ་དང་། བུད་མེད་དང་། མ་ནིང་གི་ལག་ཏུའོ། །གོས་ཀྱི་རིན་དག་ཅེས་བྱ་བ་ནི། གསེར་རམ་དངུལ་དག་བསྐུར་བར་གྱུར་ལ་ཞེས་བྱ་བ་ནི་བསྲིང་བར་གྱུར་པ་ལའོ། །དེ་ནས་ཕོ་ཉ་དེ་གོས་ཀྱི་རིན་དེ་དག་ཁྱེར་ནས། དགེ་སློང་ག་ལ་བ་དེར་སོང་སྟེ་ཕྱིན་
ནས། དགེ་སློང་དེ་ལ་འདི་སྐད་ཅེས་འཕགས་པ་མཁྱེན་པར་མཛོད་ཅིག །ཁྱོད་ལ་རྒྱལ་པོའམ། བློན་པོ་ཆེན་པོའམ། བྲམ་ཟེའམ། ཁྱིམ་བདག་གམ། ཁྱིམ་བདག་གི་ཆུང་མའམ། གྲོང་རྡལ་གྱི་མིའམ། ཡུལ་མིའམ། ནོར་ཅན་ནམ། ཚོང་དཔོན་ནམ། དེད་དཔོན་ཆེ་
གེ་མོས་གོས་ཀྱི་རིན་འདི་དག་བསྐུར་གྱིས། འཕགས་པས་ཐུགས་བརྩེ་བའི་སླད་དུ་འདི་དག་བཞེས་ཤིག་ཅེས་ཟེར་ན་ཞེས་བྱ་བ་ནི་ཕོ་ཉས་དགེ་སློང་དག་ལ་སྦྱིན་བདག་གིས་གོས་ཀྱི་རིན་བསྐུར་བ་དག་ཕུལ་ནས་ཕྲིན་དེ་སྐད་ཅེས་སྨྲས་ནའོ། །དགེ་སློང་དེས་ཕོ་ཉ་དེ་ལ་འདི་སྐད་ཅེས་
ཚེ་དང་ལྡན་པ་ཕོ་ཉ་དགེ་སློང་དག་གི་ནི་གོས་ཀྱི་རིན་དག་ལེན་དུ་མི་རུང་སྟེ། བདག་ཅག་གོས་རུང་བ་དུས་སུ་རྙེད་ན་ནི་ལེན་ཏོ་ཞེས་བརྗོད་པར་བྱའོ་ཞེས་བྱ་བ་ནི། དགེ་སློང་དག་དངོས་པོ་མི་རུང་བར་མ་བྱས་པ་དག་མི་ལེན་པ་དང་། གོས་རུང་བ་དུས་སུ་རྙེད་པ་ལེན་དུ་རུང་བའི་
ཚུལ་གོ་བར་བྱས་ཏེ་གོས་ཀྱི་རིན་དག་ནི་གསེར་དང་དངུལ་ལ་སོགས་པ་མི་རུང་བ་རུང་བར་མ་བྱས་པ་དག་གོ། །གོས་རུང་བ་ནི་སྤྱད་བཟོད་པ་དག་གོ། །དུས་སུ་རྙེད་པ་ནི་གོས་བགེད་པའི་དུས་དང་། སློང་བར་རིགས་པའི་དུས་དང་། བསྟབས་པའི་དུས་སུ་རྙེད་པའོ། །ལེན་ཏོ་
ཞེས་བྱ་བ་ནི་བདག་གིར་བྱེད་དོ་ཞེས་བྱ་བ་སྟེ། དེ་སྐད་ཅེས་བརྗོད་པར་བྱའོ། །གལ་ཏེ་ཕོ་ཉ་དེ་དགེ་སློང་དེ་ལ་འདི་སྐད་ཅེས་འཕགས་པ་རྣམས་ཀྱི་ཞལ་ཏ་བགྱིད་པ། འཕགས་པ་རྣམས་ཀྱི་ཞལ་ཏ་ཉམས་སུ་ལེན་པ་གང་ལགས་པ་འགའ་མཆིས་སམ་ཞེས་ཟེར་ན་ཞེས་བྱ་
བ་ནི་ཕོ་ཉས་དགེ་སློང་རྣམས་ཀྱི་རུང་བར་བྱེད་པ་སུ་ཡོད་ཅེས་བརྟག་པ་སྟེ། ཞལ་ཏ་བགྱིད་པ་ཞེས་བྱ་བ་དག་ལ་ཞལ་ཏ་ཉམས་སུ་ལེན་པ་ཞེས་བྱ་བ་ནི་བསྣན་པའི་ཚིག་སྟེ། གཞི་གཅིག་ཏུ་བསྡུའོ། །དགེ་སློང་གིས་གོས་འདོད་པས་ཕོ་ཉ་འདི་ནི་དགེ་སློང་རྣམས་ཀྱི་ཞལ་ཏ་བྱེད་པ་སྟེ།
