འགྲུས་ཡིན། །བྱིང་རྒོད་ཞི་བ་བསམ་གཏན་ཏེ། །ཡང་དག་དོན་སྟོན་ཤེས་རབ་ཡིན། །སེམས་ཅན་ཀུན་ལ་རོ་གཅིག་བློ། །སྙིང་རྗེ་ཞེས་ནི་བརྗོད་པ་ཡིན། །སྦྱིན་པས་ལོངས་སྤྱོད་ཐོབ་པར་འགྱུར། །ཚུལ་ཁྲིམས་ཀྱིས་ནི་བདེ་འགྲོ་སྟེ། །བཟོད་པ་ཡིས་ནི་གཟུགས་བཟང་བའོ། །བརྩོན་འགྲུས་
ཀྱིས་ནི་གཟི་མདངས་ཆེ། །བསམ་གཏན་གྱིས་ནི་ཤེས་རབ་གྲོལ། །སྙིང་རྗེས་དོན་ཀུན་སྒྲུབ་པར་བྱེད། །བདུན་པོ་དེ་རྣམས་མ་ལུས་པར། །ཅིག་ཅར་ཕ་རོལ་ཕྱིན་པ་ཡི། །བསམ་མི་ཁྱབ་པའི་ཡེ་ཤེས་ཡུལ། །འཇིག་རྟེན་མགོན་ཉིད་ཐོབ་པར་འགྱུར། །དེ་ལྟར་ལས་དང་འབྲས་བུ་
དག །མ་འཁྲུལ་ཚུལ་ནི་ཤེས་བྱས་ནས། །རྟག་པར་འགྲོ་ལ་ཕན་གདགས་བྱ། །དེ་ཉིད་བདག་ལའང་ཕན་པར་འགྱུར། །ཆོས་རྣམས་ངེས་པ་ཞེས་བྱ་བ། །བྱིས་པའི་སྐྱེ་བོས་རྟོགས་བྱའི་ཕྱིར། །གཞུང་རྒྱས་གཞན་ལས་བདག་གིས་ནི། །མདོ་རུ་བསྡུས་ཏེ་བྲིས་
