བྱ་ཞེ་ན་ཞེས་བྱ་བའི་དོན་ཏོ། །བརྟན་པ་རྣམས་ཀྱིས་སྔར་སྨྲ་བའོ། །ཞེས་བྱ་བ་ལ། གང་ཟག་བླ་མ་དང་། མཉམ་པ་དང་། དམན་པ་རྣམས་དང་ཕྲད་ན་དང་པོ་ཉིད་དུ་བདེ་བར་འོངས་སམ་ཞེས་བྱ་བ་ལ་སོགས་པའི་ཚིག་གསེང་པོར་སྨྲ་བ་ནི་ངག་གི་སྒོ་ནས་གཞན་དགའ་བར་བྱེད་པའོ། །ཅིའི་ཕྱིར་ཞེ་ན།
ཞེས་བྱ་བ་ནི་བཞིན་འཛུམ་པ་དང་། ཚིག་གསོང་བོར་སྨྲ་བ་ལ་སོགས་པའི་སྒོ་ནས་འཇིག་རྟེན་དགའ་བ་དང་མགུ་བར་བྱེད་པ་དེའང་ཅིའི་ཕྱིར་དེ་ལྟར་བྱ་ཞེ་ན་ཞེས་བྱ་བའི་དོན་ཏོ། །སེམས་ཅན་རྣམས་ནི་སྣོད་འགྱུར་ཕྱིར། །ཞེས་བྱ་བ་ལ། འདུལ་བའི་སེམས་ཅན་རྣམས་དམ་པའི་ཆོས་ཀྱི་སྣོད་དུ་བྱ་
བའི་ཕྱིར། ལུས་དང་ངག་གི་སྒོ་ནས་སེམས་ཅན་དགའ་བ་བསྐྱེད་པ་ར་བྱའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །དོན་གང་གིས་སྣོད་དུ་འགྱུར་བའི་ཕྱིར་ཞེ་ན་ཞེས་བྱ་བ་ནི་སེམས་ཅན་རྣམས་སྣོད་དུ་བྱ་བའི་ཕྱིར་ཞེས་བཤད་ན། ཆོས་གང་གི་སྣོད་དུ་བྱ་བའི་ཕྱིར་དགའ་བ་དང་མགུ་བ་སྐྱེད་ཅེ་ན་ཞེས་བྱ་བའི་དོན་ཏོ། །
དམ་ཆོས་བསྒྲུབ་པར་བྱ་བའི་ཕྱིར། །ཞེས་བྱ་བ་ལ། ཐེག་པ་ཆེན་པོའི་དོན་རྟོགས་ཤིང་། དེ་བསྒྲུབ་ཏུ་གཞུག་པའི་སྣོད་དུ་འགྱུར་བར་བྱ་བའི་ཕྱིར་ལུས་དང་ངག་གི་སྒོ་ནས་གཞན་ལ་དགའ་བར་བྱེད་དོ། །འདི་ཡན་ཆད་ཀྱི་ཐ་སྙད་ཤེས་པའི་སྒོ་ནས་འཇིག་རྟེན་ཤེས་པར་བསྟན་ཏོ། །ཇི་ལྟར་ན་བདེན་
པ་རྣམས་ཤེས་ཤེ་ན་ཞེས་བྱ་བ་ནི་བདེན་པ་བཞི་ཤེས་ན་ཡང་འཇིག་རྟེན་ཤེས་པ་བརྟན་པ་ལ། ཇི་ལྟར་ཤེས་ན་བདེན་པ་བཞི་ཤེས་པ་ཡིན་ཞེ་ན་ཞེས་པའི་དོན་ཏེ། དེ་ཡང་འབྱུང་བ་ཤེས་པ་དང་། འགོག་པ་ཤེས་ན་བདེན་པ་བཞི་ཤེས་པར་གཞག་གོ། །བདེན་པ་རྣམ་པ་གཉིས་ལས་ནི། །འཇིག་
རྟེན་རྣམས་ཀྱང་རྟག་འབྱུང་དང་། །ཞེས་བྱ་བ་ལ། འཇིག་རྟེན་འབྱུང་བ་ཤེས་པ་ནི་སྡུག་བསྔལ་གྱི་བདེན་པ་དང་། ཀུན་འབྱུང་གི་བདེན་པ་གཉིས་ལས་འཇིག་རྟེན་འབྱུང་བར་ཤེས་ནས་འབྱུང་བ་ཤེས་པ་ཡིན་ནོ། །གཉིས་པོ་དེ་ཡིས་འགོག་ཤེས་པ། །ཞེས་བྱ་བ་ལ། འགོག་པའི་བདེན་པ་
དང་། ལམ་གྱི་བདེན་པ་གཉིས་ལས་འཇིག་རྟེན་འགོག་པར་ཤེས་པ་ནི་འཇིག་རྟེན་འགོག་པར་ཤེས་པའོ། །དེའི་ཕྱིར་འཇིག་རྟེན་ཤེས་ཞེས་བྱ་བ་ལ། བདེན་པ་གཉིས་ལས་ནི་འཇིག་རྟེན་འབྱུང་བར་ཤེས། བདེན་པ་གཉིས་ཀྱིས་ནི་འཇིག་རྟེན་འགོག་པར་ཤེས་པས་ན་འཇིག་རྟེན་ཤེས་པ་ཞེས་
