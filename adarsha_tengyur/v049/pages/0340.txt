སྤཪྵ་པ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་དྷ་རྨ་ཡ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་ཀྵེ་ཏྲ་ཤ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་ཨཱ་ལོ་ཀ་ཀྵ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་ཨཾ་ཀུ་ཤ་རྀ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་པཱ་ཤ་རཱྀ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་བནྡྷ་ལྀ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་གྷཎྜ་ལཱྀ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ལྷ་ཉི་ཤུ་རྩ་
བཞིའི་སྙིང་པོའི་སྔགས་སོ། །ཡི་གེ་རེ་རེ་འབུམ་ཕྲག་གང་ཡིན་ཡབ་ཡུམ་མཉེས་བྱེད་ཡིན་བཤད་གཞན། །པདྨ་དང་ནི་འཁོར་ལོའི་ལྷ་དང་ཁྱམས་བཅས་གོ་རིམས་བཞིན། །འབུམ་ཕྲག་གསུམ་དང་གཉིས་དང་གཅིག་བཟླས་མངོན་སུམ་གཞལ་ཡས་ཁང་། །མཐོང་བའམ་རྨི་
ལམ་བརྟན་དང་གསལ་སྣང་མཆོག་ཐོབ་ཏིང་འཛིན་མཐུ་གྲུབ་བརྟན་པའི་ལྟས། །དམིགས་པས་དུབ་པར་གྱུར་པ་དང་། །རྩེ་གཅིག་བསམ་གཏན་གོ་འཕང་གཉེར། །རིམ་པ་གཞན་ཡང་གོམས་འདོད་པས། །དེ་ནས་རྣམ་པར་མི་རྟོག་བསྒོམ། །སྤྱི་ཡི་སྙིང་ག་ལྟེ་བར་ནི། །ཟླ་
བ་པདྨ་ཉི་སྟེང་དུ། །ཧཾ་དང་ཧཱུཾ་དང་ཨཾ་ཞེས་ཀྱི། ཡི་གེ་བདེ་གསལ་མི་རྟོག་རྒྱུ། །སྣོད་བཅུད་རླུང་རྒྱུན་བརྟན་གཡོ་མཁའ། །ཉི་མ་ཟླ་བ་སྒྲ་གཅན་འཛིན། །རྨི་ལམ་ནམ་མཁའི་སྦྱོར་བ་ཡིས། །བློ་ཆེན་རྣལ་འབྱོར་བསྒྲུབ་པར་བྱ། །སྤྲོ་དང་སྡུད་དང་གཟུང་བ་དང་། །དམིགས་
བྲལ་དང་ནི་རྣམ་པར་དག །སྣ་ཚོགས་མངོན་པར་སྤྲུལ་པ་ནི། །རིམ་སོགས་བླ་མ་རྒྱས་པར་འཆད། །ཐུན་མཚམས་རྟག་ཏུ་གཏོར་མ་སྔོན་འགྲོ་ཇི་སྙེད་འབྱོར་པའི་ཚོགས་འཁོར་ལོས། །ལེགས་མཆོད་ཐལ་སྦྱར་དངོས་གྲུབ་གསོལ་བཏབ་དམ་ཚིག་མ་གྱུར་ཀུན་བཟོད་གསོལ། །
དེ་རྗེས་མདུད་པས་རྡོ་རྗེའི་གར་བསྐོར་སེ་གོལ་བརྡབས་པས་ཡེ་ཤེས་ཀྱི། །དཀྱིལ་འཁོར་མཁར་བསྐྱོད་ཤིན་ཏུ་ལེགས་ཞེས་དབུགས་ཕྱུང་མི་སྣང་གྱུར་པར་བསམ། །དེ་ལྟར་སྔགས་པ་བློ་གྲོས་ཅན་བརྟུལ་ཞུགས་ཟབ་མོ་སྤྱོད་པར་ནུས་པས་ཇི་སྐད་བཤད་པའི་ཆོ་ག་འདི་དག་ཉིན་
རེ་བཞིན་ཐུན་གསུམ་གསུམ་བསྙེན་པར་བྱེད་ཅིང་། བླ་མ་དང་ལྷག་པའི་ལྷ་ལ་མངོན་པར་དད་པ་རྒྱུན་མི་འཆད་པས་སྦྱོར་བར་བྱེད་ན། དེས་ཅི་ནས་ཀྱང་ཟླ་བ་བདུན་ལས་མི་རིང་བར་བཅོམ་ལྡན་འདས་དཀྱིལ་འཁོར་ཆེན་པོའི་ལྷ་ཉི་ཤུ་རྩ་ལྔའི་བདག་ཉིད་ཀྱི་མཚན་མ་གདོན་མི་ཟ་བར་མཐོང་བར་
