སྐྱེ་བའི་ཚེ། །རང་གི་དྲན་པའི་ཕྱིར་དུ་འདི། །བྱས་པས་དེ་ཚེ་དྲན་པར་ཤོག །ནང་གི་སེམས་ནི་དུལ་གྱུར་ན། །ཕྱི་ཡི་དགྲ་འདིས་གནོད་མི་ནུས། །རང་གི་ནང་སེམས་གཡོ་བ་ན། །ཕྱི་ཡི་དགྲ་འདིས་རྐྱེན་བྱས་ནས། །ནང་གི་དགྲ་འདིས་རང་རྒྱུད་བསྲེག །དེས་ན་ནང་གི་དགྲ་འདི་གཞོམ། །དུག་
ཅན་སྣོད་ཀྱིས་ཟས་བརླག་བཞིན། །ནང་གི་དྲི་མས་དགེ་ཚོགས་བརླག །གཉེན་པོ་ཆུ་ཡིས་ཁྲུས་བྱས་ནས། །དགྲ་ལས་བདེ་བ་ཆེན་པོ་འཐོབ། །གང་ན་གནས་ཚེ་གང་གི་ཚེ། །གང་གི་དྲི་མ་ལྡང་བ་དང་། །ནང་གི་དགྲ་འདི་འབྱུང་བའི་ཚེ། །ཐབས་ནི་སྣ་ཚོགས་གསུངས་པས་
སྤང་། །དགྲ་གཞོམ་དྲི་མ་བཀྲུ་བའི་ཕྱིར། །དེ་ཚེ་འབད་པ་ཆེན་པོ་བྱ། །གང་གིས་རང་གི་རྒྱུད་འདི་ལ། །ནང་དུ་བལྟས་ཏེ་རང་སེམས་བརྟག །དེ་ཚེ་ཡུལ་ལ་ཤེས་པ་སྐྱེ། །ཤེས་པ་སྐྱེ་བའི་ཡུལ་རྣམས་གསུམ། །ཉོན་མོངས་བར་མ་དགེ་བ་སྐྱེ། །རྒྱུ་རྐྱེན་ཡུལ་གྱི་བྱེ་བྲག་
གིས། །ཉོན་མོངས་སྐྱེ་བ་དག་ལ་གསུམ། །ཆེན་པོ་འབྲིང་དང་ཆུང་ངུའོ། །གཉེན་པོའི་རིམ་པའང་དེ་བཞིན་ཏེ། །ཆེན་པོ་སྐྱེ་བའི་ཡུལ་དེ་ལ། །བརྟེན་ཏེ་ཚོགས་ཆེན་འབད་པས་སྐྱེ། །འབྲིང་གི་ཡུལ་ལ་འབྲིང་ཡིན་ཏེ། །ཆུང་ངུ་ལ་ནི་ཆུང་ངུའོ། །ཆེན་པོ་ཀུན་ལ་སྐྱེས་ན་
མཆོག །བར་མ་ལུང་མ་བསྟན་པ་ཡང་། །དྲན་པས་ཟིན་པའི་སེམས་དེ་ཡིས། །ཀུན་སློང་རྗེས་ཀྱིས་དགེ་བར་བསྒྱུར། །ཐབས་སྒོ་སྣ་ཚོགས་པ་ཡིས་སོ། །དགེ་བ་ཆེ་འབྲིང་ཆུང་ངུ་དག །ངང་གིས་སྐྱེས་པའི་ཡུལ་དེ་ལ། །སྦྱོར་བ་དངོས་དང་རྗེས་དག་ལས། །ཁྱད་འཕགས་
བསོད་ནམས་ཀུན་བསྔོ་བྱ། །ཤེས་ན་གང་ལའང་ཚོགས་འཕེལ་འགྱུར། །ཐོག་མེད་དུས་ནས་ཇི་སྲིད་དུ། །བདག་ནི་དུག་གསུམ་ཞེན་པས་འཁོར། །ཇི་སྲིད་སྐྱེ་ཤི་ཁ་བརྒྱུད་པས། །དེ་སྲིད་ལས་རྣམས་ཆུད་རེ་ཟོས། །ཇི་སྲིད་དགེ་སེམས་འཕེལ་འགྱུར་བ། །དེ་སྲིད་སངས་
རྒྱས་མངོན་སངས་རྒྱས། །བདག་ནི་གཏི་མུག་མུན་པས་བསྒྲིབས། །མུན་ནག་གློག་འགྱུ་སྣང་བ་ཙམ། །ཅི་ཞིག་ལྟར་སྟེས་དྲན་རྙེད་ཚེ། །དེ་ཚེ་འབད་པས་བསྒྲུབ་པར་བྱ། །འཆི་བར་ངེས་པ་ཆ་ཡོད་ཅིང་། །གནས་པའི་དབང་སྟེར་མེད་མོད་ཀྱི། །བདག་ནི་རེ་ཞིག་མི་འཆི་སྙམ། །
