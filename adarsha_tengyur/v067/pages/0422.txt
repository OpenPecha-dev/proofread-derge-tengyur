ལྟ་བ་བཞིན་དུ་ཞི་བའི་ལས་བྱས་པར་འགྱུར་ན་ལན་གཉིས་པའི་སྟོབས་ཀྱིས་ངེས་པ་ཉིད་དུ་འགྲུབ་པར་འགྱུར་བ་ཉིད་དོ། །དེ་དག་གིས་ཞི་བའི་སྦྱིན་སྲེག་རྫོགས་པའོ། །འབྲས་བུར་ལྡན་པའི་ནགས་ཚལ་ལམ། །ཡང་ན་སྐྱེད་མོ་ཚ་ལ་མཆོག་གམ། །རྒྱལ་རིགས་ཁྱིམ་
དུ་རྒྱས་པ་ནི། །དགའ་བའི་སེམས་ཀྱིས་བྱ་བ་སྟེ། །ཁྲུ་དོ་ཡང་ན་ཁྲུ་བཞི་འམ། །དེ་ལྟར་མཆོག་ནི་ཁྲུ་བརྒྱད་དེ། །ཐབ་ཁུང་ལེགས་པར་བརྐོས་ནས་ནི། །རྒྱས་པའི་ལས་ནི་བྱ་བ་སྟེ། །དཀྱིལ་འཁོར་སྟེགས་བུ་བརྩིག་བྱ་བ། །ཆུང་ངུ་ལ་ནི་ཁྲུ་བརྒྱད་དེ། །བར་མ་
ལ་ནི་བཅུ་གཉིས་ཉིད། །མཆོག་ལ་བཅུ་དྲུག་ཡང་ན་ནི། །ཉི་ཤུ་རྩ་བཞིར་ཤེས་པར་བྱ། །མུ་ཁྱུད་སོར་བརྒྱད་བཅུ་དྲུག་གམ། །སུམ་ཅུ་གཉིས་སུ་བྱ་བ་ཡིན། །ཞེས་གསུངས་པ་ནི་རྒྱས་པའི་སྦྱིན་སྲེག་གསུངས་ཏེ། དེ་ལ་གོང་དུ་གསུངས་པའི་གནས་
ཇི་ལྟ་བ་རྣམས་སུ་སྟེགས་བུ་ཐ་མ་ནི་ཁྲུ་བརྒྱད་པ་བརྩིག་པ་དང་། བར་མ་ནི་བཅུ་གཉིས་བརྩིག་པ་དང་། རབ་ལ་ནི་བཅུ་དྲུག་གམ་ཉི་ཤུ་རྩ་བཞིར་བརྩིགས་ནས་དེ་ཇི་ལྟ་བ་བཞིན་དུ་ཐབ་ཁྲུ་དོ་པ་དང་། བཞི་པ་དང་། བརྒྱད་པ་ཟུར་བཞི་པའམ། རིན་པོ་ཆེ་ལྟ་བུ་
ཟུར་ལྔ་པའམ། བརྒྱད་པར་བྱས་ལ་ཟབས་སུ་ནི་ཕྱེད་བརྐོ་བར་བྱ་ཞིང་། དེའི་ཕྱི་རོལ་གྱི་སྟེང་དུ་མུ་ཁྱུད་སོར་བརྒྱད་པ་དང་། བཅུ་དྲུག་པ་དང་སུམ་ཅུ་རྩ་གཉིས་པ་ནི་གོང་དུ་གསུངས་པའི་རབ་འབྲིང་ལ་སྦྱར་ཏེ་བྱ་བ་ཉིད་དོ། །ཕྱི་ཡི་དཀྱིལ་འཁོར་དེ་ཡི་ཕྱེད། །པདྨ་
སྣ་ཚོགས་བརྒྱན་པར་བྱ། །དཀྱིལ་འཁོར་དབུས་སུ་རིན་ཆེན་དང་། །པདྨ་སྣ་ཚོགས་གཞག་པར་བྱ། །དེ་ཡི་དབུས་སུ་བསྒོ་བ་ནི། །ལེགས་པར་བལྟས་ཏེ་ཕྱེད་ཀྱི་ཚད། །ཁྲུ་བཞི་པའམ་ཁྲུ་བརྒྱད་པ། །ཡང་ན་དཔལ་སྟེར་རིན་ཆེན་འདྲ། །མདོར་ན་
འཇིམ་པ་སེར་པོ་ཡིས། །ཁུང་མཆོག་དེ་ནི་བསྒོ་བར་བྱ། །ཞེས་གསུངས་པ་ནི་དཀྱིལ་འཁོར་བྲི་བར་གསུངས་པ་སྟེ། དེ་ལ་དཀྱིལ་འཁོར་གྱི་ནང་དུ་ནི་པདྨའི་སྟེང་དུ་རིན་པོ་ཆེ་འབར་བ་འོད་ཟེར་སེར་པོ་དང་ལྡན་པ་བྲི་བར་བྱ་བ་དང་། དེའི་མུ་རན་ལ་རིན་པོ་ཆེ་
