བའི་དོན་ཏོ། །སྐུ་རྡོ་རྗེ་ནི་སྐུའི་དཀྱིལ་འཁོར་དང་། གསུང་རྡོ་རྗེ་ནི་གསུང་གི་དཀྱིལ་འཁོར་དང་། ཐུགས་རྡོ་རྗེ་ནི་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ཏེ་ངེས་པའི་དོན་ཏོ། །མན་ངག་ཇི་ལྟ་བ་བཞིན་དུ་བསྒོམ་པ་ནི་བསྒྲུབ་པ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ད་ནི་སྙིང་རྗེ་ཆེན་པོ་མངོན་སྤྱོད་དུ་གྱུར་པ་སྔོན་དུ་འགྲོ་བས་
ཁྱད་པར་གྱི་གནས་ལ་བརྟེན་པ་བཤད་པར་བྱ་སྟེ། དགོན་པ་ཆེན་པོའི་ས་ཕྱོགས་ནི། །མེ་ཏོག་འབྲས་བུ་སོགས་ཀྱིས་བརྒྱན། །རི་བོ་དབེན་པ་དག་ཏུ་ནི། །དངོས་གྲུབ་ཐམས་ཅད་བསྡུས་པས་འགྲུབ། །ཅེས་སྨོས་པ་ལ། འདིར་དང་པོར་སྙིང་རྗེ་ཆེན་པོའི་སེམས་བསྐྱེད་ལ། ཕྱིས་གནས་
ལ་བརྟེན་པར་བྱ་སྟེ། དགོན་པ་ཆེན་པོའི་ས་ཕྱོགས་ནི་ཁྱད་པར་གྱི་གནས་ཡིན་ཏེ། རྫིང་བུ་ལ་སོགས་པས་ཡིད་འཕྲོག་པའི་ཕྱིར་རོ། །མེ་ཏོག་འབྲས་བུ་སོགས་ཀྱིས་བརྒྱན་པ་ནི་རི་བོའི་ཁྱད་པར་རོ། །དབེན་པ་ནི་གཞན་ཡང་དབེན་པའི་ས་ལའོ། །འཇིག་རྟེན་པ་དང་འཇིག་རྟེན་ལས་འདས་
པའི་དངོས་གྲུབ་ཐམས་ཅད་བསྡུས་པ་ནི་གང་དུ་བསྡུས་པར་གྱུར་པ་དེ་ནི་དངོས་གྲུབ་ཐམས་ཅད་བསྡུས་པ་སྟེ་སངས་རྒྱས་ཉིད་དོ། །སྒྲུབ་པ་ནི་བསྒྲུབ་པར་བྱ་བ་སྟེ་དྲང་བའི་དོན་ཏོ། །ནམ་མཁར་རྟོགས་པའི་སྦྱོར་བ་ཡིས། །ཕུང་པོ་དགོན་པ་ཆེན་པོར་འདོད། །གནས་མེད་གནས་ཀྱི་སྦྱོར་བ་
ཡིས། །ས་ཡི་ཕྱོགས་ནི་གནས་ཆེན་པོ། །དཔག་བསམ་ཤིང་ལྟར་ལོངས་སྤྱོད་ནི། །སྣ་ཚོགས་དངོས་གྲུབ་མེ་ཏོག་ཅན། །སྤྲུལ་པའི་འབྲས་བུ་ཀུན་མཛེས་པའི། །ཡང་དག་བྱང་ཆུབ་རི་བོ་ཡིན། །རང་བཞིན་མེད་པ་དབེན་པ་ལ། །སངས་རྒྱས་མཛེས་ཏེ་བསྒྲུབ་པར་བྱ། །ཞེས་ངེས་
པའི་དོན་ཏོ། །དགོན་པ་ཆེན་པོ་ནི་ཆོས་འབྱུང་གི་ཕྱག་རྒྱ་སྟེ། དེའི་ས་ཕྱོགས་ནི་འདོད་པའི་གདུགས་སོ། །འབྲས་བུ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་འཛག་པ་ཟླ་བ་མའི་རྩའོ། །མེ་ཏོག་ནི་རང་བྱུང་གི་མེ་ཏོག་འཛག་པ་རུས་སྦལ་མའི་རྩའོ། །སོགས་པའི་སྒྲས་ནི་བ་ཤང་གཅི་དང་ལྡན་པའི་ལྷས་
བྱིན་མ་དང་། ནོར་རྒྱལ་མའི་རྩ་གཟུང་ངོ་། །དེ་རྣམས་ཀྱིས་བརྒྱན་པའི་བྷ་ག་ལའོ། །རི་བོ་ནི་སོར་བཞིའི་ཚད་ཙམ་གྱི་མོའི་དབང་པོའི་རྩའོ། །དེ་ཉིད་ཆགས་པ་ལ་སོགས་པ་སྤྱོད་པར་བྱེད་པས་ན་དབེན་པའོ། དེ་ལ་དངོས་གྲུབ་ཐམས་ཅད་བསྡུས་པ་ནི་ཐིག་ལེ་ཙམ་སྒྲུབ་པ་སྟེ། དགོངས་
