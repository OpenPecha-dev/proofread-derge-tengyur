རྒྱུད་འདིར་ཕྱག་རྒྱ་བཞི་བསྟན་པ་ལས། ཕྱག་རྒྱ་ཆེན་པོའི་དཀྱིལ་འཁོར་བསྟན་ཟིན་ནས། ད་ནི་དམ་ཚིག་གི་ཕྱག་རྒྱ་རྣམས་ཀྱི་དམ་ཚིག་གི་དོན་དུ་དཀྱིལ་འཁོར་བཤད་པར་བྱ་དགོས་ཏེ། དེ་བས་ན་དེ་ནས་བཅོམ་ལྡན་འདས་ཞེས་བྱ་བ་ལ་སོགས་པས་སྟོན་ཏོ། །དེ་ནས་
ཞེས་བྱ་བ་ནི་གནས་སྐབས་ཀྱི་དོན་ཏོ། །བཅོམ་ལྡན་འདས་ཞེས་བྱ་བ་ནི་སྐལ་བ་ཕུན་སུམ་ཚོགས་པའི་ཕྱིར་རོ། །དེ་བཞིན་གཤེགས་པ་ཡང་ཡིན་ལ་ཐམས་ཅད་ཀྱང་ཡིན་པས་ན་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་དོ། །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱང་ཡིན་ལ་རྡོ་རྗེ་ཡང་ཡིན་པས་ན་དེ་བཞིན་གཤེགས་པ་
ཐམས་ཅད་ཀྱི་རྡོ་རྗེའོ། །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་རྡོ་རྗེའི་གཟུངས་ཡིན་པས་ན། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་རྡོ་རྗེའི་གཟུངས་ཏེ། རྡོ་རྗེ་ཞེས་བྱ་བའི་སྒྲ་ནི་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་དང་། གཟུངས་ཞེས་བྱ་བ་གཉི་ག་དང་སྦྱར་བར་བྱའོ། །གཟུངས་ཞེས་བྱ་བའི་སྒྲ་ནི་སྔགས་ཀྱི་གཟུངས་ཡོངས་
སུ་གཟུང་ངོ་། །དེ་རྣམས་ཀྱི་དམ་ཚིག་ནི་སྙིང་པོ་དང་ཕྱག་རྒྱ་དང་དཀྱིལ་འཁོར་ལ་སོགས་པའོ། །དེ་བསམ་པའི་ཁྱད་པར་གང་ལས་བྱུང་བ་དེ་བྱིན་གྱིས་བརླབ་པའི་ཕྱིར་རྡོ་རྗེ་ལྟར་མི་ཕྱེད་པའི་ཏིང་ངེ་འཛིན་གང་ཡིན་པ་དེ་ལ་སྙོམས་པར་ཞུགས་པར་མཛད་དོ། །སྙོམས་པར་ཞུགས་མ་ཐག་ཏུ་ཞེས་བྱ་བ་ནི་སྙོམས་པར་
ཞུགས་པའི་རྗེས་ཐོགས་ཉིད་ལ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཡང་དེ་ཉིད་ཀྱི་ཚེ་ཞེས་བྱ་བ་ནི་དེ་ནས་སྐད་ཅིག་མ་དེ་ཉིད་ལའོ། །དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཐུགས་ལས་ཞེས་བྱ་བ་ནི་དེ་བཞིན་གཤེགས་པ་བཞིའི་ཐུགས་ལས་སོ། །དྲིས་པས་དཀྱིལ་འཁོར་སླར་བཀྱེ་བས། དེ་བཞིན་གཤེགས་པ་རྣམས་
སླར་གཤེགས་སུ་གསོལ་བ་ཡིན་ན། ཇི་ལྟར་ན་དེ་རྣམས་སླར་ཡང་སྙོམས་པར་ཞུགས་པར་མཛད། གླེང་གཞི་མེད་པར་ཡང་སྔགས་ལ་སོགས་པ་ཡང་ཇི་ལྟར་གསུང་། དེ་ནི་བདེན་མོད་ཀྱི་གང་སླར་གཤེགས་སུ་གསོལ་བ་དེ་ནི་དེ་ལྟར་དཀྱིལ་འཁོར་གྱི་སློབ་དཔོན་དག་དཀྱིལ་འཁོར་གཅིག་གི་དགོས་པ་
མངོན་པར་འདོད་པ་རྣམས་ཀྱི་ཡིན་པར་ཟད་ཀྱི། གང་རྒྱུད་ཀྱི་དོན་སྟོན་པར་མངོན་པར་བཞེད་པའི་བཅོམ་ལྡན་འདས་དེ་རྣམས་ནི་སོ་ན་བཞུགས་བཞིན་དུ། བརྟག་པ་གཞན་རྣམས་དང་དཀྱིལ་འཁོར་གཞན་རྣམས་སུ་ཐུགས་བསྐྱེད་པ་ཙམ་གྱིས་བྱོན་ནས་དགོངས་པར་མཛད་དོ། །དེ་ལྟ་བས་ན་དེ་བཞིན་
