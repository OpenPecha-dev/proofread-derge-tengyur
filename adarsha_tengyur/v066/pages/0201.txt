ཀུ་ཤ་ནེ་ནེར་གྷ་ཏ་ན་མ་དྷཱི་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཛྭ་ལེ་ན་མ་ཧཱ་ཛྭ་ལ་ནི་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཙནྡྲ་སུ་ཙནྡྲི་ཙནྡྲ་བྱཱ་བ་ལོ་ཀི་ནི་སྭཱ་ཧཱ། ཨོཾ་ཨ་མྲྀ་ཏཱ་བྷཱ་ཏོ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ཁ་ཨཱ་ཀ་ཤ་གརྦྷ་ཡ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ནི་ཝ་ར་ཎི་དི་ཤྐཾ་བི་ནི་སྭཱ་ཧཱ། རྡོ་རྗེ་ཧཱུཾ་མཛད་དབང་ལྡན་དུ། །སྐུ་མདོག་

བྱེད་ལྤགས་དམར་རློན་པ་རིམ་པར་ནི། །གཡས་དང་གཡོན་གྱི་ཕྱག་གིས་མདའ་དང་གཞུ། །ཕྱག་གཉིས་ཀྱིས་ནི་ཁྲག་གིས་བཀང་བའི་ཐོད་པ་བསྣམས། །གཡས་བརྐྱང་ཞབས་ཀྱིས་འཇིགས་བྱེད་ཆུང་མར་བཅས་པ་མནན། །འགྲོ་བའི་བྱེད་པོ་ཀེང་རུས་རྒྱན་དང་སྐ་རགས་དང་།
