དབལ་གྱི་ནད་ཀྱིས་བཏབ་པ་ནི་ཁ་དོག་སྣ་ཚོགས་སུ་སྣང་བ་དང་། རིམས་ཀྱིས་འདེབས་པ་དང་། བརྒྱལ་བ་དང་། ཡན་ལག་རྣམས་གྲུགས་པ་དང་། མགོ་འཁོར་བ་དང་། ཁ་སྐམས་པ་དང་། སྙིང་ན་བ་དང་། ལུས་སྟོབས་ཆུང་བ་དང་། འཁྲུ་བ་ནི་འཆི་བར་

དབུགས་མི་བདེ་བ་དང་། རྨོངས་པ་དང་། རྟག་ཏུ་འཁྲུ་བ་དང་། སྐོམ་དད་ཆེ་བ་དང་། མགོ་འཁོར་བ་དང་། ནད་སྣ་ཚོགས་པས་ལུས་ཀྱི་སྐད་དང་། ཁམས་ཀྱི་སྟོབས་དང་མེའི་དྲོད་ཉམས་ཏེ་ཟད་པར་བྱེད་ན་འཆི་བ་ཡིན་ནོ་ཞེས་བཤད་པ་ཡིན་ནོ། །རླུང་གི་ནད་དང་། བརྗེད་
བྱེད་དང་། མཛེ་དང་། ཁྲག་གི་དམུ་རྫིང་དང་། ཟད་བྱེད་ཀྱི་ནད་དང་། སྐྲན་དང་། གཅིན་སྙི་བ་དང་། ནད་འདི་རྣམས་གང་ཡང་རུང་བས་ལུས་ཤིན་ཏུ་ཉམ་ཆུང་བའམ། རིད་པ་ལ་བཏབ་ན་ནི་ནད་ཚབས་ཆུང་ཡང་རིང་དུ་སྤང་བར་བྱའོ། །ནད་གང་གི་སྟོབས་དང་ཤ་ཟད་ལ་
