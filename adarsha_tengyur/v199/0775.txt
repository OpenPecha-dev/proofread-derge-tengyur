དང་། ལྡོང་རོས་དང་། བ་བླ་དང་། སྐྱེར་ཁཎྜ་དང་། ཧ་རེ་ཎུ་དང་། ནག་ཚུར་བྱ་བལ་མ་དང་། རྒྱ་ཚ་དང་། རྒྱམ་ཚ་དང་། དེ་རྣམས་ཀྱི་ཕྱེ་མ་སྦྲང་རྩི་དང་སྦྱར་བས་བསྐུས་ན་ཕོ་མཚན་གྱི་རྨ་འཚོ་བར་བྱེད་དོ། །འབྲས་བུ་གསུམ་རྫེའུ་ཁ་སྦྱར་གྱི་ནང་དུ་བསྲེགས་ཏེ་མར་དང་
སྦྱར་བས་བསྐུས་ན་རྨ་འཚོ་བར་བྱེད་པའི་མཆོག་ཡིན་ནོ། །དེ་དག་ནི་ཕོ་མཚན་གྱི་གསོ་བ་སྤྱིར་བསྟན་པ་ཡིན་ནོ། །ནད་གཞི་རེ་རེ་ལས་གྱུར་པའི་ནད་ནི་སྐྲང་ནད་ཀྱི་གསོ་བ་བཞིན་དུ་སྦྱར་རོ། །སྨན་གྱིས་གསོ་བ་ཕོ་མཚན་གྱི་ནད་ཅི་ནས་ཀྱང་སྨིན་ཅིང་རྣགས་པར་མི་འགྱུར་བའི་
ཐབས་སུ་བྱའོ། །གལ་ཏེ་སྨིན་ན་ནི་རྒྱུས་པ་དང་རྩ་དང་ཤ་རྣམས་ཕལ་ཆེར་འཇིག་པར་འགྱུར་རོ། །གཞང་འབྲུམ་ལྟ་བུའི་ནད་ནི་བཅད་ལ་མེས་བསྲེགས་ཏེ་ལྷག་མ་ནི་ཕོ་མཚན་སྤྱིར་བསྟན་པ་གསོ་བ་གོང་མ་བཞིན་བྱའོ། །ཡུངས་ཀར་ལྟ་བུའི་ནད་ལ་ནི་མཚོན་གྱིས་ཕྲ་བ་བྲིས་
ཏེ། སྨན་གོང་མ་དེ་ཉིད་བསྐོལ་བའི་ཁུ་བས་བླུགས་པ་དང་། ཕྱེ་མས་གདབ་པ་དང་། དེ་ཉིད་ཏིལ་མར་དང་སྦྱར་བས་བསྐུས་ན་རྨ་འཚོ་བར་འགྱུར་རོ། །ཨ་པ་མན་ཐའི་ནད་ལ་ཡང་གསོ་བ་གོང་མ་ལྟར་བྱ་སྟེ་ཁྲག་གཟག་གོ །ཀུམ་བྷི་ཀའི་ནད་སྨིན་པ་ངོས་གཉི་ག་ནས་ཁྲག་
གཏར་ཏེ་རྨ་དག་པར་བྱས་ལ་ཏིནྟུ་ཀ་དང་། འབྲས་བུ་གསུམ་དང་། སེང་ཕྲོམ་དང་། དེ་རྣམས་ཏིལ་མར་དང་སྦྱར་ཏེ་བསྐུ་བ་དང་བྱུག་པ་ལ་སོགས་པ་བྱས་ན་རྨ་འཚོ་བར་འགྱུར་རོ། །ཕོ་མཚན་གྱི་ནད་མེ་དབལ་ལྟ་བུ་ལ་ནི་ཁྲག་གཏར་ཏེ་ཆོ་ག་གོང་མ་ལྟར་བྱའོ། །ཨུ་དམ་
ཞེས་བྱ་བའི་འབྲུམ་བུའི་ནད་ལ་ནི་ལྕགས་ཀྱུ་གུག་པོས་འབྲུམ་བུ་བཅད་དེ་མཁྲིས་པའི་མེ་དབལ་གསོ་བའི་རིམ་པ་བཞིན་དུ་སྨན་གྱི་ཕྱེ་མ་དང་སྨན་བསྐོལ་བའི་ཁུ་བ་དང་དེ་རྣམས་སྦྲང་རྩི་དང་སྦྱར་བ་བཏང་སྟེ་པདྨ་ཅན་ཞེས་བྱ་བ་ལ་ཡང་ཆོ་ག་དེ་ལྟར་མཚུངས་པར་སྦྱར་རོ། །པགས་པ་
སྨིན་པ་དང་། ཚོར་བ་མེད་པ་གཉིས་ལ་ནི་བ་ལའི་མར་དྲོ་འཇམ་ཙམ་གྱིས་བསྐུས་ཤིང་བླུགས་པ་དང་། སྨན་མངར་བའི་སྡེ་ཚན་སྦྱར་བའི་བ་ལའི་མར་གྱིས་དུགས་བྱའོ། །དེ་ལ་གཏུན་བུ་ལྟ་བུའི་ནད་ལ་ནི་ཁྲག་ཕྱུང་སྟེ། བད་ཀན་གྱི་ཐོར་བུའི་གསོ་བ་རྣམས་སྤྱད་དོ། །
