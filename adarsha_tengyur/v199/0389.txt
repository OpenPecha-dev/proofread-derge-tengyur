ལས་གྱུར་པའི་སྐོམ་དད་སེལ་ཏོ། །བད་ཀན་ལས་གྱུར་པའི་སྐོམ་པའི་ནད་ལ་ནི་ཐོག་མར་ནིམ་པ་བསྐོལ་བའི་ཁུ་བ་བླུད་དེ་སྐྱུག་ཏུ་གཞུག་གོ། །ཡང་ན་བིལ་བ་དང་། འུ་སུ་དང་། རྩ་བ་ལྔ་པ་ཆུང་ངུ་དང་། དར་བ་དང་། ཙི་ཏྲ་ཀ་དང་། འདི་
རྣམས་ཆུ་དང་བསྐོལ་བའི་ཁུ་བ་བླུད་དོ། །ཡང་ན་ཡུངས་ཆུ་དང་བསྐོལ་བའི་ཁུ་བའི་ནང་དུ་སྦྲང་རྩི་དང་། ཤ་ཁ་ར་བཏབ་པ་བཏུང་བའམ། ཡང་ན་མུངྒ་དང་། པ་ཏོ་ལ་དང་། ནིམ་པ་རྣམ་གཉིས་ཀྱི་ལོ་མ་བསྐོལ་བའི་ཁུ་བའི་ནང་དུ་ཚ་བ་གསུམ་གྱི་ཕྱེ་མས་བཏབ་
པ་བཏུང་ངོ་། །གཞན་ཡང་ནས་ལས་བྱས་པའི་ཟན་དང་། སྨན་གྱི་ཁུ་བ་མཁུར་བར་བཅང་བ་དང་། སྨན་སྣ་ནས་གཏང་བ་དང་ལྡེ་གུ་ལ་སོགས་པའི་སྨན་ཚ་བའི་བག་དང་བཅས་པ་སྦྱིན་ནོ། །མ་ཞུ་བ་དང་བཅས་པའི་སྐོམ་དད་ཆེ་བའི་ནད་ལ་ནི་གོང་དུ་
སྨོས་པའི་སྦྱོར་བ་རྣམས་བྱ་སྟེ། ཁྱད་པར་དུ་སྐྱུག་སྨན་གྱིས་ཞི་བར་བྱའོ། །དེ་བཞིན་དུ་སྦྱོར་བ་གཞན་ཡང་ཚ་བ་གསུམ་དང་གོ་བྱེ་དང་། ཤུ་དག་དང་། འདི་རྣམས་ཀྱི་ཕྱེ་མ་བལ་པོ་སེའུའི་ཁུ་བའམ། ཆུ་དྲོན་ནམ་ཞོ་ག་ཆུ་གང་ཡང་རུང་བ་དང་སྦྱར་ཏེ་
བཏང་ན། སྐོམ་དད་མ་ཞུ་བ་དང་བཅས་པ་ཞི་བར་བྱེད་དོ། །གལ་ཏེ་ནད་པ་ཟས་སྟོབས་ཆུང་ན་ཟས་ཤིན་ཏུ་སླ་བ་དང་། སྐྱོ་མ་གང་ཡང་རུང་བ་དུས་ཚོད་དང་སྦྱར་ཏེ་གཏང་ངོ་། །ངལ་བས་སྐོམ་དད་སྐྱེས་པ་ལ་ནི་ཤ་ཁུའམ་ཤ་ཁ་ར་དང་སྦྱར་བའི་སྐྱོ་མ་བླུད་
དོ། །ཚ་བས་སྐོམ་དད་སྐྱེ་བ་ལ་ནི་ནས་ཀྱི་ཕྱེའམ། རྒྱ་ཤུག་གི་ཕྱེ་མའི་སྐྱོ་མ་ཁ་ར་དང་སྦྱར་བ་བླུད་དོ། །ཡང་ན་ཏིལ་གྱི་ཚིགས་མ་བཏགས་ཏེ་རང་སྐྱུར་དང་སྦྱར་བས་ལུས་ཐམས་ཅད་བསྐུས་ན་ཚ་བ་འཇིལ་ལོ། །ལུས་ཁྲུས་བྱས་པས་སྐོམ་དད་སྐྱེད་པ་ལ་ནི་ཆང་
ངམ་ཆུས་བསྲེས་པའམ། ཡང་ན་བུ་རམ་ཆུ་བླུད་དོ། །ཆང་འཐུངས་པས་སྐོམ་དད་སྐྱེད་པ་ལ་ནི་ཆུ་དང་ཆང་ཕྱེད་མར་བསྲེས་པ་བླུད་དེ། དེ་ནས་ལུས་ཁྲུས་བྱས་ལ་ཟས་སྐྱུར་དང་ལན་ཚྭ་དང་ལྡན་པར་སྦྱིན་ནོ། །སྣུམ་པ་དང་ལྡན་པའི་ཟས་རྣོ་ཆེས་པ་ཟོས་པས་མེ་དྲོད་སྐྱེས་
