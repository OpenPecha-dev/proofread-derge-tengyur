འབྲས་ཀྱི་རྩེ་མོ་འདྲ་བས་དབུག་ལ་མཚོན་རྩེ་དཀྱིལ་འཁོར་འདྲ་བས་ཕྱི་བར་བྱའོ། །འབྲས་བུ་གསུམ་དང་། ནག་ཚུར་དང་། རྒྱམ་ཚ་རྣམས་སྦྲང་རྩི་དང་སྦྱར་བས་གདབ་བོ། །ཉ་ཕྱིས་ལྟ་བུའི་མིག་ནད་ལ་ནི་བད་ཀན་འཛག་པ་ཞེས་བྱ་བའི་གསོ་བ་ཐམས་ཅད་འདིར་བྱའོ། །
བད་ཀན་བསྒོངས་པ་ཞེས་བྱ་བའི་མིག་ནད་དང་། འབྲས་ཀྱི་ཐིག་ལེ་ལྟ་བུ་ཞེས་བྱ་བའི་མིག་ནད་གཉིས་ནི་གཏར་ག་མ་གཏོགས་པ་ལྷག་མ་བད་ཀན་འཛག་པའི་ཆོ་ག་བཞིན་དུ་བྱའོ། །དེ་བཞིན་དུ་གཉིས་པོ་དེ་ལ་ནི་ཚ་བ་གསུམ་དང་། ཀད་ཕ་ལ་དང་འདི་རྣམས་བཏགས་
པའི་ཕྱེ་མ་ཁ་ལུང་གི་ཁུ་བས་བསྒོས་ཏེ། མིག་སྨན་དུ་གཏང་ངོ་། །སྲི་ཤིང་གི་མེ་ཏོག་ཁ་མ་བྱེ་བ་དང་། རྒྱམ་ཚ་དང་། ཐང་ཤིང་དང་། བཅའ་སྒ་དང་། འདི་རྣམས་བཏགས་པའི་ཕྱེ་མ་ཆང་གི་དངས་མ་དང་མིག་སྨན་དུ་སྦྱར་ན་སྐྲང་པ་དང་། གཡའ་བ་སེལ་བར་བྱེད་དོ། །
མིག་ནད་རྩ་འགྱུར་བ་དང་། བརྩེ་བ་དང་རྩ་འདྲ་བས་གཡོགས་པ་དང་། ལིང་ཏོག་དམར་པོ་དང་། དྲ་བའི་དྲ་བ་ལྟ་བུ་སྟེ་བཞི་པོ་དེ་དག་ནི་ཁྲག་འཛག་པའི་ཆོ་ག་བཞིན་དུ་བྱའོ། །ཁྱད་པར་དུ་རྩ་འགྱུར་བའི་མིག་ནད་ལ་ནི་མར་དང་སྦྲང་རྩི་སྦྱར་བས་བསྐུའོ། །རྩ་བརྩེ་བ་ཞེས་
བྱ་བའི་མིག་ནད་ལ་ནི་སྐྱེར་ཁཎྜ་ཞིབ་ཏུ་བདར་ལ་སྦྲང་རྩི་དང་སྦྱར་བས་མིག་ལ་བསྐུའོ། །ལིང་ཏོག་དམར་པོ་ལ་ནི་སྦྲང་རྩི་དང་། ཁ་ར་དང་ཞོ་ག་ཆུ་དང་སྦྱར་བ་བླུགས་ན་ཕན་ནོ། །དེ་བཞིན་དུ་ལིང་ཏོག་དམར་པོ་ལ་ནི། ཤེལ་དང་། དུང་དང་། ཤིང་མངར་དང་། གུར་ཀུམ་དང་། འདི་
རྣམས་སྦྲང་རྩི་དང་སྦྱར་བའི་མིག་སྨན་གཏང་ངོ་། །ཡང་ན་ལིང་ཏོག་དམར་པོ་ལ་ནི། དུང་དང་། སྦྲང་རྩི་སྦྱར་བའམ། ཡང་ན་ཁ་ར་དང་རྒྱ་མཚོའི་ལྦུ་བ་སྦྱར་བ་བླུགས་ན་ཕན་ནོ། །གོང་དུ་མིག་མཛེར་གྱི་ནད་རྣམ་པ་ལྔ་བཤད་པ་ལས། དུད་ཀ་ལྟ་བུ་དང་། དམར་བ་དང་། ཞོ་འདྲ་བའི་
