བ་དྲང་ཞིང་འབྱིན་པར་བྱེད་དོ། །ཡང་ཆ་བྱད་སྦྲུལ་གྱི་མགོ་བོ་གདེངས་པ་བཞིན་དུ་རྩེ་མོ་གུག་པར་བྱས་པ་འདིས་ནི་རྡེའུ་འབྱིན་པར་བྱེད་དོ། །ཡང་ཆ་བྱད་མདའ་ལྟོང་ལྟ་བུ་སྦོམ་ཞིང་རྩེ་མོ་ཟླུམ་པ་སྲིད་སོར་བཞི་པ་བྱས་པ་དེས་ནི་སོ་དྲང་ཞིང་དབྱུང་བར་སྦྱར་རོ། །ཡང་ཆ་བྱད་
དྲུག་བྱས་པ་དེའི་རྩེ་མོ་ཀུན་རས་བལ་གྱིས་དཀྲིས་ཏེ་དེ་ལས་ཀྱང་སྲིད་དུ་སོར་བཞི་དང་བཅུ་གཉིས་པར་བྱས་པ་འཇམ་པ་གཉིས་ཀྱིས་ནི་གཞང་གི་ནང་དུ་ནད་ཉེ་རིང་གང་ན་ཡོད་ཀྱང་རུང་སྟེ། ཐལ་སྨན་དང་སྣགས་པ་ལས་སྦྱང་ཞིང་བསལ་བའི་ལས་བྱའོ། །ཡང་སྲིད་དུ་སོར་དྲུག་པ་དང་། བདུན་
པར་བྱས་པ་གཉིས་ཀྱིས་སྣའི་བུ་གའི་ནང་དུ་སྣགས་པ་དང་། ཐལ་སྨན་གྱི་ལས་དང་སྦྱར་བ་ལ་སྦྱར་རོ། །ཡང་ཆ་བྱད་སྲིད་དུ་སོར་བརྒྱད་པ་དང་། སོར་དགུ་པ་གཉིས་ཀྱིས་ནི་རྣ་བའི་ནད་སྣགས་པ་དང་ཐལ་སྨན་གྱི་ལས་དང་སྦྱར་བ་ལ་སྦྱར་རོ། །ཡང་ཆ་བྱད་ཨ་ཤྭདྠའི་ལོ་མའི་རྩེ་མོ་ལྟ་
བུའམ། བླུགས་གཟར་གྱི་ཁ་ལྟ་བུའི་ཆ་བྱད་དེས་ནི་རྣ་སྤབས་དང་། དྲི་མ་སྦྱང་ཞིང་བསལ་བ་ལ་སྦྱར་རོ། །ཡང་ཆ་བྱད་སོར་རེ་བ་གསུམ་བྱ་སྟེ། དེས་ནི་རྨའི་ཐན་འཁོར་འཛམ་བུ་ལྟར་དམར་བ་ལ་གཙགས་པ་དང་། ཐལ་སྨན་དང་། མེའི་ལས་རེ་རེ་ལ་སྦྱར་རོ། །ཡང་ཆ་བྱད་ཁར་བ་ལྟ་
བུའི་དབྱིབས་ལ་རྩེ་མོ་དང་དབུས་ཟླུམ་པོ་རྩ་བ་ཕྱེད་པ་ལྟར་བྱས་པ་དེས་ནི་མིག་མཛེར་དང་། རྒྱུ་མ་སྦོམ་པོ་དང་ཕྲ་མོ་ལ་རྨ་བྱུང་བ་ལ་སྦྱར་རོ། །རྒྱ་ཤུག་གི་ཚི་གུ་ཕྱེད་བཀས་ལྟ་བུ་ལ། སོ་དང་བཅས་པའི་ཆ་བྱད་ཀྱིས་ནི་སྣའི་ཐོར་བུ་དང་འབྲུམ་བུ་གཅད་པར་བྱའོ། །ཡང་ཆ་བྱད་སྲིད་
དུ་སོར་བརྒྱད་དང་། རྒྱབ་སྒུར་ཞིང་སོར་ཅུང་ཟད་གུག་པ་གསུམ་བྱས་ཏེ། དེའི་སོ་ཡང་མཐེའུ་ཆུང་དང་གུང་མོའི་སེན་མོ་ལྟ་བུའི་སོ་བྱས་པ་དེ་ནི་ཐལ་སྨན་དང་། སྨན་གྱི་བྱུག་པ་དང་། བསྐུ་བ་ལ་སོགས་པ་ལ་སྦྱར་རོ། །ཕོ་མཚན་དང་མིག་སྨན་ལ་སོགས་པའི་ཆ་བྱད་ནི་རང་རང་གི་ལེའུ་ལས་
བསྟན་པ་ཡིན་ནོ། །དེ་ལ་ཉེ་བའི་ཆ་བྱད་ཅེས་བྱ་བ་རྡོ་ཁབ་ལེན་དང་། རྩ་མུན་ཛ་ལ་སོགས་པའི་ཐག་པ་དང་། རས་དང་རྡོ་དང་། ཐོ་བ་དང་། པགས་པ་དང་། སྲོག་ཆགས་ཀྱི་རྒྱུ་མ་དང་། ལྕེ་དང་རྟ་རྔ་དང་། ཤིང་གི་ཡལ་ག་དང་། སེན་མོ་དང་། སོ་དང་། ཉེ་བའི་སྤྱད་དེ་དག་གིས་ཀྱང་།
