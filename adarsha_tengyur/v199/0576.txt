ལ་ནི། སྐྱུག་སྨན་དང་། བཀྲུ་སྨན་མི་བྱའོ། །དེ་བཞིན་དུ་གང་གི་སྲོག་ཟུངས་ཕྱི་ས་བརྟན་པར་འཛིན་པ་ལ་ཡང་བཀྲུ་སྨན་མི་བཏང་ངོ་། །ཆོ་ག་གྲུབ་པའི་གནས་ལས་སྨན་མས་གཏོང་བའི་ཆོ་གའི་ལེའུ་སྟེ་བཞི་པ་རྫོགས་སོ།། །།དེ་ནས་སྨན་མས་གཏོང་བའི་ཆོ་གས་ནད་ཕྱིན་ཏེ་སོས་
པར་བྱ་བའི་ཆོ་གའི་ལེའུ་བཤད་པར་བྱ་སྟེ། དེ་ལ་ནད་པ་སྣུམ་དང་དུགས་མ་བྱས་པ་དང་ལྟོ་ཤིན་ཏུ་ལྕི་བ་ལམས་གཏོང་བའི་སྨན་གྲང་མོ་དང་། སྣུམ་དང་། ལན་ཚྭ་ཉུང་བ་དང་། སྨན་ཐུན་ཚད་ཆུང་བ་དང་། ཡང་ན་བསྐ་བ་མས་གཏོང་ན་སྨན་གྱི་སྟོབས་ཆུང་བས་ནད་གཞི་འཁྲུགས་ནས་མི་
འབྱུང་སྟེ་སྦྱོར་བ་ལོག་པ་དེས་ནི་རླུང་དང་ཕྱི་ས་དང་གཅིན་འགགས་ཤིང་ཆུ་སོ་དང་ལྟེ་བ་རྣམས་སུ་ན་བ་དང་། ལུས་ཚ་ཞིང་དང་ཀ་མེར་མེར་པོར་བྱེད་པ་དང་། གཞང་སྐྲང་བ་དང་། གཡའ་བ་དང་། འབྲས་བྱུང་བ་དང་། མདོག་འགྱུར་བ་དང་། གནས་ལ་མི་དགའ་བ་དང་། མེ་དྲོད་ཆུང་བར་
འགྱུར་ཏེ། དེ་ལ་ནི་འཁྲུ་བ་གསོ་བའི་ལེའུ་ལས་འབྱུང་བའི་སྨན་བུ་ཏི་ཀ་དང་། པི་པི་ལིང་དང་། བཅའ་སྒ་དང་། ཤུ་དག་དང་། འུ་སུ་དང་། ཨ་རུ་ར་བསྐོལ་བ་འམ། ཡང་ན་བིལ་བ་དང་། འུ་སུ་དང་། གླ་སྒང་དང་། བཅའ་སྒ་དང་། བལ་དང་། དེ་རྣམས་བསྐོལ་བ་འམ། སྦྱོར་
བ་དེ་གཉིས་གང་ཡང་རུང་བ་དྲོན་པོ་འཐུངས་ན་རབ་ཏུ་བསྔགས་སོ། །ཡང་ན་རྟུག་པ་འགགས་ཏེ་ཟུག་གཟེར་ཆེ་བ་ལ་ནི་པོ་སོན་ཆའི་རེང་བུ་བཏང་ཞིང་དུགས་བྱ་བ་དང་། དུས་ཚོད་ཤེས་པར་བྱས་ལ་བཀྲུ་སྨན་ཡང་བཏང་སྟེ། བིལ་བའི་རྩ་བ་དང་། དུར་བྱིད་དང་། ཐང་ཤིང་དང་། ནས་
དང་། རྒྱ་ཤུག་དང་། རྒྱ་སྲན་དང་། འདི་རྣམས་བཏགས་བའི་ཆང་དང་། བ་གཅིན་དང་ལྷན་ཅིག་མས་གཏོང་ན་ཕྱི་ས་འགགས་པ་སེལ་པར་འགྱུར་རོ། །དེ་ལ་ནད་གཞི་སྟོབས་ཆེ་བ་འམ། ཁོང་དུ་རྩུབ་པ་འམ། ལྟོ་སྲ་བ་ལ་སྨན་ཐུན་ཆུང་བ་དང་། ནུས་པ་ཆུང་བའི་སྨན་མས་
གཏོང་ན། ནད་གཞི་རྣམས་སློང་སྟེ། རླུང་རྒྱུ་བའི་རྒྱུ་སྲང་འགགས་ནས། རླུང་ལམ་ལོག་པར་སོང་སྟེ། ལྟོ་སྦོས་ཤིང་གནད་རྣམས་ཀྱང་ཤིན་ཏུ་གཟིར་ཞིང་ན་བར་འགྱུར་རོ། །དེ་བཞིན་དུ་ལུས་ཚ་བ་དང་། གཞང་དང་། ལྟོ་བ་དང་། རླིག་པ་དང་། ཆུ་སོ་རྣམས་སུ་ཟུག་ཅིང་ན་
