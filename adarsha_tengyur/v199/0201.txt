ཡང་ན་ཁོང་པ་ནང་ཚ་སྟེ། དེ་ཡང་ན་ཚ་བ་སྔར་བྱུང་། ཡང་ན་ཚ་བ་ཕྱིས་བྱུང་བ་ཡོད་དེ། དེ་ལ་ཚ་བ་སྔར་བྱུང་བའི་རིགས་ནི་གསོ་དཀའ་བ་ཡིན་ནོ། །ཚུལ་དེ་དང་འདྲ་བར་རླུང་དང་བད་ཀན་གཉིས་ཀྱང་ཡང་ན་ཕྱི་རོལ་དུ་ཕྱུང་སྟེ་ཤ་དང་པགས་པ་
གྲང་ངམ། ཡང་ན་ཁོང་པའི་ནང་དུ་བྱུང་སྟེ་ཁོང་གྲང་བ་དེ་ལྟ་བུ་གྲང་བའི་རིམས་སྔར་བྱུང་ངམ། ཡང་ན་གྲང་བའི་རིམས་མཇུག་ཏུ་ཕྱིས་བྱུང་བ་ཡང་ཡོད་དོ། །གྲང་བའི་རིམས་དང་། རླུང་དང་། བད་ཀན་གྱི་རིམས་དེ་སྔར་བྱུང་བའི་རྗེས་ལ་ཕྱིས་མཁྲིས་པའི་རིམས་ཚ་བས་
བསྣན་ན་བད་ཀན་གཤེར་བ་འབྱུང་བ་དེ་ནི་བསྐམས། རླུང་གྲང་བའི་བག་ཞི་བར་བྱས་པས་ཁའི་ནང་རོ་སྐྱུར་བར་བྲོ། བརྒྱལ། མྱོས། སྐོམ་དད་ཆེ་བར་འགྱུར་རོ། །ཡང་ན་མཁྲིས་པའི་ནད་ཚ་བས་ཐོག་མར་བཏབ་སྟེ་དེའི་རྗེས་ནི་རྨུགས་ཤིང་བད་ཀན་ཁ་ནས་མང་དུ་འབྱུང་ངོ་། །
སློན་ནོ། །ཁོང་སྙོམ། ནད་ཐབས་འདི་ལྟ་བུ་ནི་འདུས་པའི་རིམས་ཡིན་ཏེ། དཀྱེལ་ཆེར་སྐྱེས་པ་མངོན་དུ་རྨུགས་བྱེད་ཅེས་བྱ་བ་གསོ་དཀའ་བའི་ནད་ཐབས་ཡིན་ནོ། །དེ་ལ་གློ་བུར་གྱི་རིམས་ནད་རྣམ་པ་བཞི་སྟེ། དྲག་ཤུལ་ལས་གྱུར་པའི་རིམས་ནད་དང་། མངོན་པར་ཟིན་
པའི་རིམས་ནད་དང་། དམོད་པའི་རིམས་ནད་དང་། བྱད་ཀྱི་རིམས་ནད་དོ། །དེ་ལ་དྲག་ཤུལ་གྱི་རིམས་ནད་ཀྱང་རྣམ་པ་བཞི་ལས་འབྱུང་སྟེ། མཚོན་གྱིས་སྨས་པ་དང་། གཅད་པ་དང་། མེས་ཚིག་པ་ལས་རིམས་སུ་གྱུར་པ་དང་། སྟོབས་རྩལ་གྱི་བག་ལས་སྣད་དེ་
རིམས་སུ་གྱུར་པའོ། །འདི་རྣམས་ལྟ་བུ་ལས་རིམས་སུ་གྱུར་པ་ནི་ཕལ་ཆེར་རླུང་གི་ཁྲག་ནད་དུ་གྱུར་པ་བྱས་ནས་ཁྲག་ལ་གནོད་པར་གྱུར་པ་དང་ན་ལ་སྐྲང་། མདོག་མི་སྡུག་ལ་ན་སྟེ་རིམས་སུ་འགྱུར་རོ། །མངོན་པར་ཟིན་པའི་རིམས་ནད་ཅེས་བྱ་བ་ནི་གདོན་གྱི་བག་མིའི་ལུས་ལ་བབ་
པའམ། གདོན་གྱིས་ཟིན་པ་དང་སྨན་གྱི་དྲི་བསུང་དྲག་པོ་སྣར་ཕོག་པ་ལས་རིམས་སུ་གྱུར་པ་དང་། མདའ་དུག་ལྟ་བུའི་རླངས་པའམ། དྲིས་ཕོག་པ་ལས་རིམས་སུ་གྱུར་པ་དང་། ཁྲོས་པ་ལས་རིམས་སུ་གྱུར་པ་དང་། འཇིགས་པ་དང་། མྱ་ངན་དང་། འདོད་པ་ལས་རིམས་
