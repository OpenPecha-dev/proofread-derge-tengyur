སྐྱུགས། ཁ་ཟས་ཡི་གར་མི་འོང་། བ་སྤུ་དང་ཤའི་པགས་པར་རེག་ན་བརྩེ། ལུད་པ་ལུ་ན་བད་ཀན་དཀར་པོ་དཀར་ལ་སྣུམ་བག་ཏུ་འབྱུང་ངོ་། །ནད་ཐབས་འདི་ལྟ་བུ་ནི་བད་ཀན་གྱིས་བཀར་བའི་ལུད་པ་ལུ་བའི་ནད་ཐབས་ཞེས་
བྱའོ། །དེ་ལ་གློ་འགྲམས་ཏེ་རྡོལ་བའི་ནད་ཐབས་ནི་སྟོབས་ལ་སོགས་པ་དྲག་ཤུལ་གྱི་བག་ལྟ་བུར་བྱས་པ་དང་། སྟོབས་ཀྱི་ཚན་གྱིས་མི་ནུས་བཞིན་དུ་ཡང་སྟོབས་སམ། ཁུར་ལྟ་བུ་ཚན་ཆེན་པོ་ཕྱུང་བས་རླུང་གིས་གློ་བ་རྡོལ་ཏེ་དྲལ་ནས། དེའི་
རྗེས་སུ་མཁྲིས་པ་ཡང་ཞུགས་ཏེ་ནད་དཀྱེལ་དུ་གྱུར་པ་ལས་ལུད་པ་ལུ་ནས་ལུད་པ་དེ་ཁར་བད་ཀན་དང་ཁྲག་འབྱུང་བ་ཁ་དོག་སེར་པོ་ནག་པོ་རྣག་གི་ལྗན་ལྗིན་རིལ་རིལ་པོ་དང་འདྲེས་པའི་བད་ཀན་ཅིག་འབྱུང་ངོ་། །གྲེ་བའི་ནང་བད་ཀན་རྣག་ཁྲག་འོང་བའི་ཚེ་
ན་ལ་བྲང་དང་། ནུ་འཇིང་དང་། ཐོད་ག་ཕྱོགས་སུ་གཟེར་བ་བཞིན་དུ་ཟུག །རུས་པའི་ཚིགས་ཁ་བྱེ་བ་སྙམ་བྱེད་ལ་རིམས་ནད་ཀྱིས་འདེབས། དབུགས་མི་བདེ། སྐོམ་དད་ཆེ། སྐད་འགགས། ལུས་འདར། རྩིབ་ལོགས་སུ་ཟུག་ཅིང་ན་སྟེ།
མཁུན་ཞིང་ཕུག་རོན་སྐད་འབྱིན་ཏོ། །དེ་ལྟར་གློ་འགྲམས་ཏེ་རྡོལ་ན་བགས་ཀྱིས་ཉམས་སྟོབས་འབྲི། ཁ་ཟས་ཁད་ཀྱིས་ཡི་གར་མི་འོང་། ཕོ་བའི་དྲོད་དང་ཉམས་འཚག་དང་། བཞིན་མདོག་དང་འདི་རྣམས་ཀྱང་ཁད་ཀྱིས་འབྲི་བར་འགྱུར་རོ། །ནད་
ཐབས་འདི་ལྟར་ན་ན་གློ་འགྲམས་ཏེ་རྡོལ་བ་ཡིན་ནོ། །དེ་ལ་ལུས་ཀྱི་སྲོག་ཟུངས་བདུན་ཟད་པར་འགྱུར་བའི་གློ་གཅོད་ན་བའི་ནད་ཐབས་ནི་གཅིན་ཁྲག་ཏུ་འོང་། རྒྱབ་དང་རྐེད་པ་ན། མཆོག་ཏུ་རླུང་ནད་ཀྱིས་ནང་སློང་བར་བྱེད་དེ་དེ་བས་ན་ཟད་པར་བྱེད་པའི་
ནད་འདི་ནི་ལུས་ཀྱི་ནད་ལ་རྒྱལ་པོ་ལྟར་བྱེད་པས་ན་ནད་ཀྱི་རྒྱལ་པོ་ཟད་བྱེད་ཅེས་བྱའོ། །དེའི་ནད་ཐབས་ཀྱང་ལུད་པ་ལུ་ཞིང་བད་ཀན་དུ་འབྱུང་ངོ་། །བད་ཀན་དུ་ལུ་བ་ལས་ཀྱང་རུལ་ཏེ་རྣག་བཞིན་དུ་འབྱུང་ལ་མདོག་སེར་མནམ། རྩ་སྔོ་ཞིང་དམར། རྩིབ་
