ཏེ་ཁོང་རྩུབ་པ་དང་། རིད་པར་གྱུར་ཏེ་ཉམ་ཆུང་བ་དང་། བརྒྱལ་བ་དང་སྔར་ལུམས་དྲགས་པ་ལས་སླར་གསོ་ཞིང་དཔྱད་བྱ་དགོས་པ་དང་། གློ་ན་བ་དང་། གློ་རྡོལ་བ་དང་། སྲོག་ཟུངས་བདུན་པོ་ཟད་དེ་ལུས་སྐམ་པོར་སོང་བ་དང་། ལུས་ཁུར་རམ་ལས་ཀྱིས་དུབ་པ་ལྟ་བུ་དང་།
ཆང་ལས་ནད་དུ་གྱུར་པ་ལྟ་བུ་དང་། མིག་རབ་རིབ་ཟ་བ་དང་། དམུ་རྫིང་དུ་གྱུར་ཏེ་ལྟོ་ཆེར་སྐྱེས་པ་དང་། ཤུ་བ་མཆེད་པ་དང་། མཛེ་ནད་དང་། འོར་ནད་སྐྲང་བག་དང་། ཚ་བ་མཁྲིས་པའི་ནད་ཡོད་པ་དང་། འོ་མ་དང་། ཞོ་དང་མར་འཐུངས་པ་དང་། སྦྲང་རྩི་འཐུངས་
པ་དང་། བཀྲུ་སྨན་བཏང་བ་དང་། ལུས་སྣད་པ་དང་། གཞང་འབྲུམ་ཁྲག་ཚད་འོང་བ་དང་། སེམས་འཁྲུགས་པ་དང་། མྱ་ངན་གྱིས་བཏབ་པ་དང་། འཇིགས་ཤིང་བག་ཚ་བ་ལྟ་བུ་ཡོད་པ་དང་། བཀྲེས་པ་དང་། སྐོམ་པའམ་མིག་སེར་གྱིས་བཏབ་པ་འམ། སྐྱ་རྦབ་
ཀྱི་ནད་ཡོད་པ་འམ་གཅིན་སྙི་སྟེ་རྟག་ཏུ་འོང་བ་འདི་ལྟ་བུ་དང་། སྤྱིར་མཁྲིས་པའི་ནད་ཡོད་པས་ཚ་བའི་ནད་ཆེ་བ་འདི་ལྟ་བུ་ལ་ལུམས་བྱར་མི་རུང་། བུད་མེད་སྦྲུམ་མ་དང་། ཟླ་མཚན་ཟག་པ་དང་། བུ་བཙས་པ་ལྟ་བུ་དང་། འདི་ལྟ་བུ་ལ་ལུམས་བྱར་མི་རུང་ངོ་། །གལ་ཏེ་
སྦྲུམ་མ་འམ་ཟླ་མཚན་ཟག་པ་དང་བུ་བཙས་པ་འདི་ལྟ་བུ་ལ་ལུམས་བྱར་མི་རུང་བ་འམ། དུག་གིས་ཕོག་སྟེ་ལུམས་བྱ་དགོས་ན་ཡང་ལུམས་མི་དྲགས་པ་འཇམ་པོར་གྱིས་ཤིག །དེ་ལ་ལུམས་བྱ་བའི་ནད་ཀྱི་དམིགས་ནི་བད་ཀན་ཆེས་ཏེ་དབུགས་མི་བདེ་བ་དང་། ལུད་པ་ལུ་
བ་དང་། ཆམ་པས་བཏབ་པ་དང་། སྐྱིགས་བུས་བཏབ་པ་དང་། ལྟོ་སྦོས་པ་དང་། བཤང་གཅི་གཉིས་ཀ་འགགས་ཏེ་དབྱུང་དུ་མི་རུང་བ་དང་། མགུལ་སུབས་ཤིང་སྐད་འགགས་པ་དང་། རླུང་ནད་ཀྱིས་བག་ཏུ་མི་བདེ་བ་དང་། བད་ཀན་རྒྱས་པ་དང་། བེའུ་སྣབས་
ཡོད་དེ་མཆིན་པ་ན་བ་དང་། ལུས་དགྱེ་སྒུ་མི་ཤེས་ཏེ་ཁོང་རེངས་པ་དང་། ལུས་ལྕི་བ་དང་། རླིག་རླུགས་ཀྱི་ནད་ཡོད་པ་དང་། མཚན་བར་རྡོལ་བའི་ནད་དང་། ད་རྒན་ལུས་ནང་དུ་སྒུ་བ་འམ། ཕྱིར་དགྱེ་བའི་ནད་ཀྱང་རུང་རླུང་ནད་ཀྱིས་ཟུག་ཐབས་སུ་ན་བ་དང་། ཆུ་
