ཅིང་སྲ་ལ་འབུར་འབུར་པོ་མཐོར་འགྱུར། སྐྲན་དེ་མྱུར་དུ་སྨིན་པར་འགྱུར་རོ། །ནད་འདུས་པ་ཚང་བའི་ནད་འདི་ནི་གསོར་མི་རུང་ངོ་། །དེ་ལ་ཁྲག་སྐྲན་ནི་བུད་མེད་འབའ་ཞིག་ལ་འབྱུང་སྟེ། ཟླ་མཚན་ཟག་གམ་གསར་དུ་བུ་བཅས་སམ། འོན་ཏེ་མངལ་གྱི་ནད་དང་གཅིག་
ཡོད་པ་ལས་ཁ་ཟས་རླུང་ནད་དུ་འགྱུར་བའི་བག་ཟོས་པ་དང་། རླུང་ནད་སློང་བར་འགྱུར་བའི་བག་བསྟེན་པས་རླུང་ནད་ལངས་ནས་ཟླ་མཚན་ཟླ་རེ་ཞིང་དུས་སུ་འོང་བ་ནང་དུ་འགགས་ཏེ། ལྟོ་བ་ཆེར་སྐྱེས་ཏེ། རུམ་ན་བུ་ཆགས་པ་འདྲ་བར་སྣང་། སྙིང་ན། ཁ་ཟས་ཀྱི་རོ་ཁ་ཅིག་
ཡིད་ལ་འདོད་པ་སྙམ་བྱེད། ནུ་ཞོ་ཡང་ཆང་ཆུང་སྣང་། ཁོང་རྨྱ་སྟེ། དེ་ལྟ་བུ་ལས་བགས་ཀྱིས་རླུང་ནད་དང་ཕྲད་པ་དང་སྐྲན་དེ་མཁྲིས་པའི་གནས་སུ་ཞེན་ནས་ཁྲག་སྐྲན་ཏེ། རླུང་དང་མཁྲིས་པ་ལས་འབྱུང་བ་ན་ལ་ཁོང་པའི་ནང་སྲ་བ་མེར་མེར་པོའི་བག་དང་ཁོང་ཚབ་དང་། ཚད་པས་
འཁྲུ། སྐོམ། རིམས་ནད་ཀྱིས་འདེབས། འདི་ལྟ་བུའི་ནད་བླ་ཐབས་སུ་ཡང་འགྱུར་རོ། །ཡང་ན་མངལ་གྱི་ནང་ན་ཁྲག་ཡོད་པ་བས་ནད་ཚབས་རབ་ཏུ་ཆེའོ། །མངལ་གྱི་ནང་ནས་ཀྱང་ཟག་ལ་དྲི་མི་ཞིམ་པ་དང་། ཟུག་པའི་ནད་ཐབས་དང་། ན་ལ་གཡོ་ཞིང་འཕར་བ་ཡང་།
དེ་ནི་མངལ་དུ་བུ་ཆགས་པ་ཞེས་མི་བྱའི་སྐྲན་མངལ་དུ་ཆགས་པ་ཡིན་ཏེ། སྐྲན་དེ་ལ་པྲྭ་པྲྭ་བྱེད། སྐྲན་དེ་ཉིད་གོང་བུར་གྱུར་ཏེ་འདུག་ཀྱང་གལ་ཏེ་ཚིགས་རིང་ཞིག་ལོན་ཞིང་འགུལ་ཏེ་པར་པར་བྱེད་ན་ཡང་དེ་ལྟ་བུ་ཡང་རུམ་ན་བུ་ཡོད་ཅེས་མི་བྱའི་མངལ་དུ་སྐྲན་དེ་ཉིད་ཆེར་སྐྱེས་ཞེས་བྱའོ། །
ཡང་ན་སྐྲན་ནོ་ཅོག་རླུང་དང་མཁྲིས་པ་དང་བད་ཀན་སོ་སོའི་ནད་ཀྱི་གནས་ལ་བརྟེན་ཏེ་སྐྲན་དུ་གྱུར་པ་དང་། འདུས་པའི་སྐྲན་དུ་གྱུར་པ་དེ་དག་ནི་གལ་ཏེ་ཡུན་རིང་པོ་ཞིག་ན་སྐྲན་སྨིན་ཏེ་རྡོལ་བར་འགྱུར་རོ། །ཡང་ན་ནི་སྨིན་ཅིང་རྡོལ་དུ་མི་རུང་ངོ་། །ཁོང་འབྲས་ཀྱི་ནད་བྱུང་བ་ནི་སྐྲན་དང་མི་
འདྲ་སྟེ། དེ་ནད་ཁྲག་ངན་པ་ལ་བརྟེན་པས། མྱུར་དུ་སྨིན་ཅིང་རྡོལ་བར་འགྱུར་རོ། །དེ་བས་ན་དེ་ནི་མྱུར་དུ་ཚ་བར་བྱེད་པས་ན་དེའི་ཕྱིར་ཞེས་བྱའོ། །དེ་ལ་སྐྲན་གྱི་ཕྱི་ནང་གི་བར་ན་ཡོད་པའི་ནད་ཐབས་ནི་ཆུ་སོ་དང་། ལྟོ་བ་དང་། སྙིང་དང་། མཆེར་པ་དང་། ཁོང་པའི་དྲོད་ཆུང་ཞིང་མདོག་
