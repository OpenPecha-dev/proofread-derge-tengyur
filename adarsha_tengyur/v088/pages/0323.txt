བཞིན་དུ་རབ་འབྱོར་གྱི་ཡང་སྟེ། དེ་ནས་འདོད་པ་ན་སྤྱོད་པ་ཞེས་བྱ་བ་ནས། ཚིག་ཏུ་བརྗོད་དོ་ཞེས་བྱ་བའི་བར་གྱིས་འགྲོ་བ་མེད་པའི་ངོ་བོ་ཉིད་གསུངས་སོ། །འདི་ལ་སྐྱེ་བ་ཡོད་པ་མ་ཡིན་པས་ན་སྐྱེ་བ་མེད་པ་སྟེ། གང་རྗེས་སུ་སྐྱེ་བ་དང་གང་གི་རྗེས་སུ་སྐྱེ་བ་དང་། གང་གིས་རྗེས་
སུ་སྐྱེ་བ་དེ་དག་མེད་པའི་ཕྱིར་རོ། །དེ་ནས་གནས་བརྟན་ཞེས་བྱ་བ་ནས། མི་དམིགས་སོ་ཞེས་བྱ་བའི་བར་གྱིས་སྐྱེ་བ་མེད་པའི་ངོ་བོ་ཉིད་གསུངས་སོ། །དེ་བཞིན་ཉིད་ནི་མི་དམིགས་དང་། །ཞེས་བྱ་བ་ནི་གཟུགས་ལ་སོགས་པ་དམིགས་སུ་མེད་པ་ཉིད་ཡིན་ན་གཟུགས་ལ་སོགས་པའི་དེ་བཞིན་ཉིད་
དམིགས་པ་ལྟ་ཡང་ག་ལ་འགྱུར། དེ་ནས་ཚེ་དང་ལྡན་པ་ཞེས་བྱ་བ་ནས། རང་སངས་རྒྱས་ཀྱི་སར་ལྟུང་ངོ་ཞེས་བྱ་བའི་བར་གྱིས་དེ་བཞིན་ཉིད་མི་དམིགས་པའི་ངོ་བོ་ཉིད་གསུངས་པའོ། །ངོ་བོ་ཉིད་བཅུ་དྲུག་བདག་ཉིད་ཅེས་བྱ་བ་ནི་བཤད་མ་ཐག་པའི་རྣམ་པ་བཅུ་དྲུག་ནི་ངོ་བོ་ཉིད་དོ། །དེ་ནི་མཚན་ཉིད་བཞི་པ་
སྟེ། བྱེད་པ་སྒྲུབ་པའི་མཚན་ཉིད་གསུམ་པོ་དག་ལས་ལྷག་པའི་ཕྱིར་རོ། །ཇི་ལྟར་འདི་མཚན་ཉིད་དུ་བཞེད་ཅེ་ན། མཚན་གཞི་རུ་ནི་མཚོན་པས་ན། །ཞེས་བྱ་བ་སྨོས་ཏེ། གང་གི་ཕྱིར་འདི་མཚན་གཞི་སྟེ། མཚན་ཉིད་དང་ལྡན་ཞིང་འདི་མཚོན་པར་བྱ་བ་ཡིན་པས་མཚན་གཞིའོ། །དེ་ནི་ངོ་བོ་ཉིད་ཀྱི་མཚན་

བརྩོན་འགྲུས་དང་། །བསམ་པ་ཕུན་ཚོགས་དྲན་པ་དང་། །རྣམ་པར་མི་རྟོག་ཏིང་འཛིན་དང་། །ཆོས་རྣམས་རྣམ་པ་ཐམས་ཅད་དུ། །ཤེས་པའི་ཤེས་རབ་དང་རྣམ་ལྔ། །རྫོགས་པའི་བྱང་ཆུབ་རྣོན་པོ་ཡིས། །རྟོགས་སླ་རྟུལ་པོས་རྟོགས་དཀར་བཞེད། །ཅེས་བྱ་བ་གསུངས་ཏེ། ཚིགས་སུ་བཅད་པ་
དང་པོས་ནི་མཚན་ཉིད་དང་ཅིག་ཤོས་དག་གིས་ནི་རབ་ཏུ་དབྱེ་བའོ། །སྦྱིན་པ་གཏོང་བ་ནས་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་བར་དུ་བསྒོམ་པ་འདི་ལ་མཚན་མར་མི་བྱེད་ཅིང་ཡང་དག་པར་གྲུབ་པའི་ཚོགས་ལ་མཁས་པ་གང་ཡིན་པ་དེ་ནི་སྐབས་འདིར་ཐར་པའི་ཆ་དང་མཐུན་པར་འདོད་དོ་ཞེས་བྱ་བ་ནི་མཚན་
