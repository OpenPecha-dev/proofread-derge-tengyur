ལྟར་ན་རྟ་ཅང་ཤེས་དེ་བློ་གསལ་བ་ཡིན་ཏེ། དེ་ལ་མཚམས་མེད་པར་འགྱུར་རོ་སྙམ་དུ་བསམ་པ་ཡིན་ནོ། །མིའི་ཡང་ཕ་མ་མི་མ་ཡིན་པ་བསད་ན་མཚམས་མེད་པར་མི་འགྱུར་རོ་ཞེས་བྱ་བ་ནི་མཚམས་མེད་པར་མི་འགྱུར་བ་ཁོ་ནའོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ཇི་ལྟ་ཞེ་ན། ཕན་པ་དང་ངོ་ཚ་ཆུང་བ་
ཁོ་ནའི་ཕྱིར་ཏེ། མི་མ་ཡིན་པས་ཀྱང་མི་དང་འཚམ་པའི་ཕམ་པར་བྱ་བར་མི་ནུས་ལ། མི་དེ་ཡང་ཕ་མ་མི་མ་ཡིན་པ་དག་ལ་ངོ་ཚ་དང་། ཁྲེལ་ཤས་ཆེན་པོ་ཡང་དག་པར་མི་གནས་ཏེ། འགྲོ་བ་མི་མཐུན་པ་ཉིད་ཀྱི་བྱམས་པ་ཆུང་བའི་ཕྱིར་རོ། །བཞི་ནི་ལུས་ཀྱི་ལས་ཡིན་ལ་ཞེས་བྱ་བ་ནི་མ་གསོད་པ་ལ་
སོགས་པའོ། །གཅིག་ངག་གི་ལས་ཡིན་ཏེ་ཞེས་བྱ་བ་ནི་དགེ་འདུན་གྱི་དབྱེན་ནོ། །གསུམ་ནི་སྲོག་གཅོད་པ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་མ་བསད་པ་ལ་སོགས་པའོ། །གཅིག་ནི་བརྫུན་དུ་སྨྲ་བ་ཡིན་ཞེས་བྱ་བ་ནི་དགེ་འདུན་གྱི་དབྱེན་ནོ། །གཅིག་ནི་སྲོག་གཅོད་པའི་སྦྱོར་བ་སྟེ་ཞེས་བྱ་བ་ནི་དེ་བཞིན་གཤེགས་
པའི་སྐུ་ལ་ངན་སེམས་ཀྱིས་ཁྲག་འབྱིན་པའོ། །ཅིའི་ཕྱིར་སྲོག་གཅོད་པ་མ་ཡིན་ཞེ་ན། དེ་བཞིན་གཤེགས་པ་རྣམས་ནི་ཉེ་བར་མི་བརྫི་བའི་ཆོས་ཅན་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་གྲངས་སུ་མེད་པའི་ཆོས་ཅན་ཡིན་ནོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །རྒྱུ་ལ་འབྲས་བུ་བཏགས་པའམ་ཞེས་བྱ་བ་ནི་དགེ་འདུན་གྱི་དབྱེན་སེམས་
དང་ལྡན་པ་མ་ཡིན་པའི་མཚན་ཉིད་ཀྱི་རྒྱུ་ནི་བརྫུན་དུ་སྨྲ་བ་ཡིན་ནོ། །རྒྱུ་དེ་ལ་འབྲས་བུ་ཉེ་བར་བཏགས་པ་སྟེ། དགེ་འདུན་གྱི་དབྱེན་ཞེས་བྱ་བར་འབྲས་བུ་སེམས་དང་ལྡན་པ་མ་ཡིན་པའི་ཆོས་ཉེ་བར་འདོགས་པ་ཡིན་ནོ། །འདིས་འབྱེད་པར་བྱེད་པས་ན་ཞེས་བྱ་བ་ནི་བྱེད་པར་སྒྲུབ་པ་ཡོངས་སུ་བཟུང་ནས་
བརྫུན་དུ་སྨྲ་བ་ལ་དགེ་འདུན་གྱི་དབྱེན་ཞེས་བརྗོད་དེ། བརྫུན་དུ་སྨྲ་བས་ནི་དགེ་འདུན་འབྱེད་པར་འགྱུར་རོ། །དགེ་འདུན་གྱི་དབྱེན་ནི་དགེ་འདུན་བྱེ་བ་ཡིན་ནོ་ཞེས་ངོ་བོ་བསྒྲུབ་པར་ཡོངས་སུ་བཟུང་བ་ནི་མ་ཡིན་ནོ། །དགེ་འདུན་དབྱེན་ནི་མི་འཕྲོད་པའོ་ཞེས་བྱ་བ་ནི་བྱེ་བྲག་ཏུ་སྨྲ་བ་རྣམས་ཀྱི་ལྟར་ན་དངོས་ཀྱི་ཚུལ་དུ་རྫས་
སུ་ཡོད་པ་ཡིན་ནོ། །དེའི་ཕྱིར། རང་བཞིན་ལྡན་པ་མ་ཡིན་ཆོས། །ཉོན་མོངས་ཅན་མིན་ལུང་བསྟན་མིན། །ཞེས་བྱ་བ་སྨོས་སོ། །དེ་ཇི་ལྟར་མཚམས་མེད་པར་འགྱུར་ཞེས་བྱ་བ་ནི་དེ་ལྟ་བུའི་མཚན་ཉིད་ཅན་དེ་ནི་མཚམས་མེད་པར་མི་འགྱུར་བ་ཁོ་ནའོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ལྡན་པ་མ་ཡིན་པའི་ཆོས་དེ་
