ཆོས་ལ་བདེ་བར་གནས་པར་བྱ་བའི་ཕྱིར་ཡང་སྤེལ་བར་བྱེད་དོ། །དབང་པོ་རྣོན་པོ་རྣམས་ནི་གནས་གཙང་མར་སྐྱེ་བར་བྱ་བའི་ཕྱིར་དང་། མཐོང་བའི་ཆོས་ལ་བདེ་བར་གནས་པར་བྱ་བ་ཁོ་ནའི་ཕྱིར་སྤེལ་བར་བྱེད་ཀྱི་དེ་དག་ལ་ཡང་ཡོངས་སུ་ཉམས་པ་མི་སྲིད་པའི་ཕྱིར་ཉོན་
མོངས་པས་འཇིགས་པའི་ཕྱིར་ནི་མ་ཡིན་ནོ། །ཡང་ཇི་ལྟར་ན་ཉོན་མོངས་པས་འཇིགས་པས་ཡོངས་སུ་མི་ཉམས་པར་བྱ་བའི་ཕྱིར་སྤེལ་བར་བྱེད་ཅེ་ན། རོ་མྱོང་བ་དང་མཚུངས་པར་ལྡན་པའི་ཏིང་ངེ་འཛིན་ཐག་བསྲིང་བའི་སྒོ་ནས་ཡོངས་སུ་མི་ཉམས་པར་བྱ་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་སྨོས་ཏེ་
འབྲས་བུ་ལས་ཡོངས་སུ་མི་ཉམས་པར་བྱ་བའི་ཕྱིར་རོ། །རྣམ་པ་ལྔ་ཡིན་ཏེ་ཞེས་བྱ་བ་ནི་ཆུང་ངུ་དང་། འབྲིང་དང་། ཆེན་པོ་དང་། ཤིན་ཏུ་ཆེ་བ་དང་། ཤིན་ཏུ་ཆེས་ཆེ་བའོ། །དང་པོ་ལ་ནི་སེམས་གསུམ་ཞེས་བྱ་བ་ནི་གང་ཟག་དེ་ལ་ནུས་པ་དེ་ཙམ་ཞིག་ཡོད་པས་ཏེ། དེ་བཞིན་དུ་འོག་མ་རྣམས་ལ་
ཡང་བརྗོད་པར་བྱའོ། །གཉིས་པ་ལ་ནི་དྲུག་གོ་ཞེས་བྱ་བ་ནི་ཟག་པ་མེད་པ་དང་། ཟག་པ་དང་བཅས་པ་དང་། ཟག་པ་མེད་པ་ཞེས་བྱ་བ་གསུམ་དང་། ཡང་ཟག་པ་མེད་པ་དང་། ཟག་པ་དང་བཅས་པ་དང་། ཟག་པ་མེད་པ་ཞེས་བྱ་བ་གསུམ་སྟེ་དྲུག་ཡིན་ནོ། །གསུམ་པ་ལ་ནི་
དགུའོ། །བཞི་པ་ལ་ནི་བཅུ་གཉིས་སོ། །ལྔ་པ་ལ་ནི་བཅོ་ལྔ་སྟེ་ཞེས་བྱ་བ་ལ། ཡང་དེ་ཁོ་ན་བཞིན་དུ་སྦྱར་བར་བྱའོ། །དེ་དག་གི་འབྲས་བུ་ནི་གྲངས་བཞིན་དུ་ཞེས་བྱ་བ་ནི་སྤེལ་མར་བསྒོམ་པ་དེ་དག་གི་འབྲས་བུ་ནི་གྲངས་བཞིན་དུ་གནས་གཙང་མ་ལྔ་ཡིན་ཏེ། བསྒོམ་པ་ཆུང་ངུའི་
འབྲས་བུ་མི་ཆེ་བ་ཡིན་པ་ནས་བསྒོམ་པ་ཤིན་ཏུ་ཆེས་ཆེ་བའི་འབྲས་བུ་འོག་མིན་ཡིན་པའི་བར་དུའོ། །དད་པའི་དབང་པོ་ལ་སོགས་པ་ལྷག་པའི་ཕྱིར་ཞེས་བྱ་བ་ནི་བཙུན་པ་དཔལ་ལེན་གྱི་གཞུང་ལུགས་ཡིན་ཏེ། བསྒོམ་པ་དང་དད་པ་ཤས་ཆེ་བའི་འབྲས་བུ་མི་ཆེ་བ་ཡིན་པ་ནས་དེ་
བཞིན་དུ་བསྒོམ་པ་ཤེས་རབ་ཤས་ཆེ་བའི་འབྲས་བུ་འོག་མིན་ཡིན་པའི་བར་དུའོ། །ཕྱིར་མི་འོང་བ་གང་ལ་ལ་ཞེས་བྱ་བ་ནི་དད་པས་མོས་པའམ་མཐོང་བས་ཐོབ་པའོ། །ལུས་ཀྱི་མངོན་སུམ་དུ་བྱེད་པའི་ཕྱིར་ཞེས་བྱ་བ་ནི་སེམས་མེད་པའི་ཕྱིར་ལུས་ཁོ་ནས་མངོན་སུམ་དུ་བྱེད་དོ། །
