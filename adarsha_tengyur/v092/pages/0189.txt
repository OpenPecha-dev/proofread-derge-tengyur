པའོ། །དེ་ཅིའི་སླད་དུ་ཞེ་ན། བཅོམ་ལྡན་འདས་སྐྱེས་བུ་གླེན་པ་གང་། རིགས་ཀྱི་བུ་དང་རིགས་ཀྱི་བུ་མོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་འདྲི་བ་དང་། ཀློག་པ་དང་། ལུང་འབོགས་པ་དང་། ཁ་ཏོན་བགྱིད་པ་དང་། སྟོན་པ་དང་། ཚུལ་བཞིན་དུ་ཡིད་ལ་བགྱིད་པ་དེ་
དག་གི་བར་ཆད་བགྱིད་པ་ནི་བདུད་ཀྱིས་བྱིན་གྱིས་བརླབས་པ་སྟེ་ཞེས་བྱ་བ་ནས། སངས་རྒྱས་ཀྱི་ཆོས་རླབས་པོ་ཆེ་དག་ལ། དེ་དག་གི་སེམས་མི་འཇུག་གོ་ཞེས་བྱ་བའི་བར་གྱིས་ནི། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འདི་ལ་སྤྱོད་པའི་བྱང་ཆུབ་སེམས་དཔའ་ལ་བར་ཆད་བྱེད་པའི་མི་རྣམས་སྨད་
པ་བསྟན་པའོ། །རབ་འབྱོར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་འདྲི་བ་དང་། ཀློག་པ་དང་། ལུང་འབོགས་པ་དང་། ཁ་ཏོན་བྱེད་པ་དང་། སྟོན་པ་དང་། ཚུལ་བཞིན་དུ་ཡིད་ལ་བྱེད་པ་དག་ལ། བདུད་ཀྱི་ལས་སུ་འདི་དག་ཀྱང་འབྱུང་མོད་ཀྱི་ཞེས་བྱ་བ་ནས། རྣམ་པ་ཐམས་ཅད་
མཁྱེན་པ་ཉིད་ཡོངས་སུ་རྫོགས་པར་བྱེད་ན། རབ་འབྱོར་དེ་ནི་སངས་རྒྱས་ཀྱི་མཐུ་ཡིན་པར་ཤེས་པར་བྱ་བ་སྟེ་ཞེས་བྱ་བའི་བར་གྱིས་ནི། གོང་དུ་བཤད་པའི་ཚུལ་དུ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འདི་ལ་སྤྱོད་པ་ལ་བར་ཆད་དང་བདུད་ཀྱི་ལས་མང་དུ་འབྱུང་བ་ཐོས་ནས། འཁོར་གྱི་ནང་ནས་ཁ་
ཅིག་སེམས་ཞུམ་པར་འགྱུར་བ་དག་ཀྱང་སྲིད་དེ་དེ་དག་སྤྲོ་བ་བསྐྱེད་པའི་ཕྱིར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སོགས་པ་ཡོངས་སུ་རྫོགས་པར་བྱ་བ་ལ་འབད་ན། སངས་རྒྱས་བཅོམ་ལྡན་འདས་མཐུ་གཏོང་ཞིང་བྱིན་གྱིས་རློབ་སྟེ། བདུད་ཀྱིས་བར་ཆད་བྱ་མི་ནུས་པར་བསྟན་པའོ། །དེས་རིགས་ཀྱི་བུ་
དང་རིགས་ཀྱི་བུ་མོ་དེ་དག་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་འདྲིའོ་ཀློག་གོ་ཞེས་བྱ་བ་ནས་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཡོངས་སུ་རྫོགས་པར་བྱེད་དོ་ཞེས་བྱ་བའི་བར་གྱིས་ནི། དེ་ལྟར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འདི་འདྲི་བ་དང་། ཀློག་པ་ལ་སོགས་པ་ལ་འབད་ན། སངས་རྒྱས་
བཅོམ་ལྡན་འདས་མཐུ་གཏོང་ཞིང་བྱིན་གྱིས་རློབ་པ་དེས། ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་ལ་སོགས་པ་རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ལ་ཐུག་གི་བར་དུ་ཡོངས་སུ་རྫོགས་པར་བྱེད་ནུས་པའི་ཕན་ཡོན་བསྟན་པའོ། །ཤར་ཕྱོགས་ཀྱི་སངས་རྒྱས་བཅོམ་ལྡན་འདས་གང་ཇི་སྙེད་ཅིག་བཞུགས་ཤིང་འཚོ་སྐྱོང་བ་དེ་དག་
