ཏ་བ་ཡུ་རཱ་ག་ཏི་སྭཱ་ཧཱ། ཨོཾ་དྷརྨ་ཀེ་ཏུ་སྭཱ་ཧཱ། ཨོཾ་ཀེ་ཏུ་ཤྭ་རཱ་ཛ་སྭཱ་ཧཱ། ཨོཾ་པདྨ་ལ་པདྨ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་རཏྣ་པྲ་བརྟ་ཡ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་རཏྣ་ཨརྩི་བ་རི་ནི་སྭཱ་ཧཱ། ཨོཾ་ཨ་ཀ་ལ་ཀོ་པྲ་བརྟ་ན་སྭཱ་ཧཱ། ཨོཾ་ཨཱ་ཙར་མུ་ཏ་སྭཱ་ཧཱ། ཨོཾ་དུ་ཏི་པ་ནི་ར་ཛ་ཎེ་སྭཱ་ཧཱ། ཨོཾ་དུ་པེ་པ་
དུ་ན་དུ་པ་ནུ་དེ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་རཏྣ་དྲྀཥྚི་ཤ་ག་ས་ཏྲི་བྲི་ས་རི་སྭཱ་ཧཱ། ཨོཾ་ཀ་མ་ལ་རཱ་ཛ་པདྨ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་པདྨ་བ་ལ་པདྨ་རཱ་ཛ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨཀྵོ་བྷྱཿསྭཱ་ཧཱ། ཨོཾ་ཨ་མོ་གྷ་རཱ་ཛ་པདྨ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བི་ཤི་རཏྣ་ན་སནྡྷི་སྭཱ་ཧཱ། ཨོཾ་ཤྭ་ར་རཏྣ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་ཀརྨཱ་སིདྡྷི་
དྷརྨ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བྷ་བ་ཏེ་སྭཱ་ཧཱ། ཨོཾ་བི་ཡ་མ་ས་བ་བེ་སྭཱ་ཧཱ། ཨོཾ་ཙཎྜ་མ་ཧུ་རོ་ཥ་ཎ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་ས་རི་ཏི་ག་ན་མ་རཱ་སྭཱ་ཧཱ། ཨོཾ་ཏྲཾ་ཨཱ་ཀཱ་ཤ་སརྦ་སཏྭ་སྭཱ་ཧཱ། ཨོཾ་ཀྵི་ཏི་གརྦྷ་ད་ན་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་དུ་རུ་མ་ལེ་ག་ཏོ་བ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ས་མནྟ་བྷ་དྲ་རཏྣ་ད་རེ་སྭཱ་ཧཱ། 
ཨོཾ་མཉྫུ་ཤྲཱི་ཀུ་མཱ་ར་མུཾ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཨ་པཱ་ཡ་བི་ཤ་ཏ་ནེ་རཾ་བ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་མཻ་ཏྲི་མཻ་ཏྲི་མཻ་ཏྲི་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བི་མུཀྟེ་མརྒ་ད་ཤ་མ་ཏི་སྭཱ་ཧཱ། ཨོཾ་རུ་ན་མ་ཡ་ཛ་སྭཱ་ཧཱ། ཨོཾ་གུ་ཎ་ཊི་ཨ་རཏྣ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་ཨཱ་ལོ་ཀ་ཏི་རི་ཏི་རི་སྭཱ་ཧཱ། ཨོཾ་ཨ་མྲྀ་ཏ་ཙ་ར་ཨ་མྲྀ་ཏ་ར་ཤ་ཎཱེ་སྭཱ་ཧཱ། ཨོཾ་
སརྦ་ཏ་ཐཱ་ག་ཏ་སཱུཪྱ་མི་ལ་ཏ་སྭཱ་ཧཱ། ཨོཾ་ཀུ་ལཾ་པདྨ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཨརྩེ་ནན་ཏིག་སྣ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཏི་ར་མི་ཏི་བཛྲ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཏྲ་བ་ལེ་ཏྲ་བ་ཏེ་སྭཱ་ཧཱ། ཨོཾ་རཏྣ་སཾ་བྷ་བ་རཏྣ་དྷ་ར་སྭཱ་ཧཱ། ཨོཾ་ཤུ་རངྒ་མ་ཏི་སྭཱ་ཧཱ། ཨོཾ་ཙིནྟཱ་པྲ་བརྟ་ཤུ་རངྒ་མེ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པདྨ་
བ་ལ་སྭཱ་ཧཱ། ཨོཾ་ཛྙཱ་ན་ཀེ་ཏུ་བི་ད་ར་ཎ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཤོ་ག་མ་ཏོ་ནིརྒ་ཏི་སྭཱ་ཧཱ། ཨོཾ་ཨ་ཀྵོ་ཀ་མོ་ནིརྒ་ཏི་སྭཱ་ཧཱ། ཨོཾ་ཨ་ཀྵ་ཡ་ཕུ་ཕུ་བཾ་ཤ་ནེ་སྭཱ་ཧཱ། ཨོཾ་ཛྙཱ་ན་པྲ་བརྟ་དྷ་ར་སྭཱ་ཧཱ། དེ་ནས་བསྡུ་བས་བྱོན་པའི་ཡུམ་ཆེན་མོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་མ་དང་སྲས་ཕྱོགས་བཅུའི་སངས་རྒྱས་འཁོར་
དང་བཅས་པ་ཐམས་ཅད། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་བནྡ་ནཾ་ཀ་རོ་མི། ཞེས་བརྗོད་ཅིང་། ཨེ་མ་ཧོ་ཀུན་དུ་བཟང་པོའི་བྱང་ཆུབ་སེམས་དཔའི་སྤྱོད་པ་མཆོག །དེ་བཞིན་གཤེགས་པའི་འཁོར་ལོ་ཡི། །དབུས་ན་བདེ་གཤེགས་ཡུམ་མཛེས་པ། །ཆེད་དུ་བརྗོད་པར་བསམ་པར་བྱ། །དེ་ནས་
