བྱ་བ་ལ་སོགས་པ་སྟེ་སྐྱེས་པ་ཙམ་གྱི་རང་བཞིན་གང་ཞིག་རྙེད་པ་ནི་ལྷན་ཅིག་སྐྱེས་པའོ། །མངོན་སུམ་བཅོས་མ་མ་ཡིན་པའི་ཐབས་དང་ཤེས་རབ་དག་རྫོགས་པའི་ཡེ་ཤེས་རྣམ་པར་མི་འགྱུར་བ་ནི་རང་བཞིན་ལྷན་ཅིག་སྐྱེས་པ་ཞེས་བྱ་བར་བརྗོད་དོ། །ཡང་ན་ལྷན་
ཅིག་སྐྱེས་པའི་དེ་ཁོ་ན་ཉིད་ཇི་ལྟར་ཤེས་པར་བྱ་ཞེ་ན། རང་བཞིན་ནི་རང་གི་བདེ་བ་ཙམ་སྟེ་དབང་པོ་ཐམས་ཅད་ལས་གྲོལ་ཞིང་བླང་བ་དང་། དོར་བ་མེད་པ་སྟོང་པ་ཉིད་དང་། རང་བཞིན་གྱིས་རྣམ་པར་དག་པ་ཐོག་མ་དང་ཐ་མ་དང་དབུས་མེད་པའོ། །ལྷན་ཅིག་སྐྱེས་ཞེས་བྱ་བ་ནི་ཀུན་ནས་ཉོན་
མོངས་པ་ཐམས་ཅད་ཀྱི་རྒྱུད་དང་ཡང་དག་པར་བྲལ་བ་སྟོང་པ་ཉིད་དང་། མཚན་མ་མེད་པ་དང་། སྨོན་པ་མེད་པའི་མཚན་ཉིད་ཤེས་རབ་ཀྱི་བདག་ཉིད་ཀྱིས་རིག་པའོ། །རྣམ་པ་ཐམས་ཅད་གཅིག་སྡོམ་པ། །ཞེས་བྱ་བ་ལ། རྣམ་པ་ཐམས་ཅད་ནི་འཁོར་བ་མཐར་ཐུག་པའི་དམིགས་པ་རྣམ་པ་སྣ་
ཚོགས་པའོ། །གཅིག་ནི་བདེ་བ་ཆེན་པོའི་རང་བཞིན་དུའོ། །སྡོམ་པ་ནི་ཁྱབ་པའོ། །ད་ནི་བླ་ན་མེད་པའི་དེ་ཁོ་ན་ཉིད་བསྟན་པར་བྱ་བའི་ཕྱིར་ཐབས་གསུངས་པ། ཐུགས་རྗེ་ཐབས་སྒྱུར་རྣལ་འབྱོར་པའི། །ཞེས་བྱ་བ་སྟེ། མངོན་པར་འདོད་པ་ཉེ་བར་ལེན་པས་སོ། །གང་ཕྱིར་སྙིང་
རྗེ་ཐབས་སུ་བརྗོད། །ཅེས་གསུངས་པའི་ཕྱིར་རོ། །ཕྱག་རྒྱ་རྒྱུ་དང་བྲལ་བ་ཉིད། །ཅེས་བྱ་བ་ལ། སྐྱེ་བ་མེད་པའི་ཡེ་ཤེས་ཀྱི་རང་བཞིན་གྱི་ཤེས་རབ་ནི་ཕྱག་རྒྱའོ། །དེ་ལས་གཞན་དུ་གྱུར་པ་ནི་རྒྱུ་སྟེ་དེ་དང་བྲལ་བའོ། །དེ་དག་ཀྱང་གསུངས་པ། སྤྲོས་པ་ཀུན་དང་བྲལ་བ་ཡི། །ཐབས་དང་
ཤེས་རབ་གཉིས་མེད་པ། །རང་བཞིན་ཡིད་ཀྱིས་བསྒོམས་བྱས་ན། །བླ་མེད་འབྲས་བུ་ཐོབ་པར་འགྱུར། །ཞེས་པའོ། །དོན་དེ་ཉིད་གསུངས་པ། སྟོང་ཉིད་སྙིང་རྗེ་ཐ་དད་མིན། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། རྟོག་པ་མཐའ་དག་དང་བྲལ་བའི་རང་གི་སེམས་ནི་སྟོང་པའོ། །མ་ལུས་པའི་
སེམས་ཅན་ཐམས་ཅད་ཀྱི་སྡུག་བསྔལ་བ་རྣམས་ལས་ཡོངས་སུ་བསྐྱབ་པའི་རྣམ་པ་ནི་སྙིང་རྗེ་སྟེ་སེམས་ལས་བྱུང་བའོ། །སེམས་དང་སེམས་ལས་འབྱུང་བ་ནི་ཐ་དད་པ་ཡིན་ན་ཇི་ལྟར་ན་འདི་གཉིས་ཐ་དད་པ་མེད་པར་བརྗོད་སྙམ་པ་ལ། བྱང་ཆུབ་ཅེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། གཉིས་ཀའི་ཡང་
