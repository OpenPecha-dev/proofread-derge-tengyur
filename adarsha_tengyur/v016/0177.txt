ལོ་གཅིག་གི་དུས་སོ། །ནང་དུ་ཉིན་ཞག་སོ་སོའི་དུས་ནི་དབུགས་དྲུག་བརྒྱ་ལྷག་པའི་སྟོང་ཕྲག་ཉི་ཤུ་རྩ་གཅིག་རྣམས་དང་། ཆུ་སྲང་སུམ་སྟོང་དྲུག་བརྒྱ་རྣམས་དང་། ཆུ་ཚོད་དྲུག་ཅུ་དང་། ཡུད་ཙམ་སུམ་ཅུ་དང་། ཐུན་ཕྱེད་པ་བཅུ་དྲུག་དང་། དུས་སྦྱོར་བཅུ་གཉིས་
དང་། ཐུན་བརྒྱད་དང་། ཐུན་མཚམས་བཞི་དང་། ཉིན་མོ་དང་མཚན་མོ་རྣམས་ཏེ་འདི་རྣམས་ཀྱི་ཚོགས་ནི་ཉི་མ་གཅིག་གི་དུས་སོ། །དེ་ལྟར་ཕྱི་རོལ་དུ་ཉི་མ་རྒྱུ་བས་དུས་ཀྱི་ཁྱད་པར་ཇི་ལྟ་བ་དེ་བཞིན་དུ་ནང་དུ་སྲོག་རབ་ཏུ་རྒྱུ་བས་དུས་ཀྱི་ཁྱད་པར་ཏེ། དུས་ཀྱི་ཁྱད་པར་དེས་ལོ་སོ་སོའི་ཉི་མ་སོ་
སོ་ལ་ཕོ་ཉ་མོ་རྣམས་མཆོད་པར་བྱའོ་ཞེས་པའོ། །ཕོ་ཉ་མོ་ནི་རྡོ་རྗེ་མཁའ་འགྲོ་མ་ལ་སོགས་པ་གཤིན་རྗེ་འཇོམས་མའི་མཐར་ཐུག་པ་སུམ་ཅུ་རྩ་དྲུག་གི་རྣལ་འབྱོར་མ་རྣམས་ཏེ། དེ་རྣམས་མཆོད་པར་བྱའོ་ཞེས་པའོ། །འདིར་ཇི་ལྟར་ཕོ་ཉ་མོ་རྣམས་མཆོད་པ་དེ་བཞིན་དུ་ལྷན་ཅིག་སྐྱེས་གྲུབ་
རྣམས་ཏེ། རིམ་པ་མ་ཡིན་པས་རྡོ་རྗེ་སེམས་དཔའ་ལ་སོགས་པ་ཉི་ཤུ་རྩ་བཞི་དམན་པ་དང་བར་མ་དང་མཆོག་རྣམས་ཏེ་སྐུ་དང་གསུང་དང་ཐུགས་ཀྱི་འཁོར་ལོ་རྣམས་ལ་ས་འོག་དང་ས་སྟེངས་དང་ནམ་མཁའ་ལ་སྤྱོད་པའི་དངོས་གྲུབ་སྟེར་བ་ཉིད་ཀྱི་ཕྱིར་ཞེས་པའོ། །དེ་རྣམས་ཀྱང་དུས་ཀྱི་ཁྱད་
པར་གྱིས་མཆོད་པར་བྱའོ་ཞེས་པ་དེ་བཞིན་གཤེགས་པའི་ངེས་པའོ། །དེ་ལྟར་ཕོ་ཉ་མོ་རྣམས་དང་ལྷན་ཅིག་སྐྱེས་གྲུབ་རྣམས་བདུད་རྩི་ལྔའི་རི་ལུ་དང་། གོ་ཀུ་ད་ཧ་ན་ལ་སོགས་པའི་ཤ་ཁ་ན་མཐོ་བ་དང་བྲལ་བའི་དམ་ཚིག་རྣམས་དང་། ཉེ་བར་སྤྱོད་པ་ལྔ་པོ་རྣམས་ཀྱིས་ཕྱི་རོལ་དུ་ཚོགས་ཀྱི་
འཁོར་ལོ་དང་ནང་དུ་ཡེ་ཤེས་དང་ཐུགས་དང་གསུང་དང་སྐུའི་འཁོར་ལོ་ལ་མཆོད་པར་བྱ་སྟེ་ཉི་མ་རྒྱུ་བ་དང་སྲོག་རྒྱུ་བས་དུས་ཀྱི་ཁྱད་པར་རོ། །དུས་ཀྱི་ཁྱད་པར་དེས་ཕོ་ཉ་མོའི་ཀུན་དུ་སྤྱོད་པའོ། །ཕོ་ཉ་མོའི་ཀུན་ཏུ་སྤྱོད་པས་ལོངས་སྤྱོད་པ་དང་ཐམ་པ་དང་དབང་བྱེད་པ་དང་བདག་པོ་ཉིད་དེ། རིགས་
ལྡན་མ་དང་ཕོ་ཉ་མོ་སུམ་ཅུ་རྩ་དྲུག་པོ་རྣམས་ཀྱི་ཡང་རིམ་པ་ཇི་ལྟ་བས་ད་ལྟར་བྱུང་བའི་དུས་ལ་ལོངས་སྤྱོད་དོ། །མ་འོངས་པའི་དུས་ལ་ཐིམ་མོ། །འདས་པའི་དུས་ལ་དབང་བྱེད་དོ། །ཉི་མ་འཆར་བ་ན་ལོངས་སྤྱོད་དོ། །ཉི་མ་ཕྱེད་པ་ན་དབང་བྱེད་དོ། །ཉི་མ་ནུབ་པའི་གནས་ལ་བདག་པོ་
