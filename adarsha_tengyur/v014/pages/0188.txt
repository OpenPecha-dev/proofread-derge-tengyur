སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨ་ནནྟ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཏཀྵ་ཀཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མ་ཧཱ་པདྨཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཀུ་ཡི་ཀཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤཾ་ཁ་པཱ་ལཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་པདྨཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲཾ་བི་ཛ་ཡཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཤྭཱ་ནཱ་སྱཻ་ཧུྃ་ཕཊ་དྭ་ཧཱ། ཨོཾ་སུ་ཀ་རཱ་སྱཻ་
ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བྱཱ་གྷྲ་སྱཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཛཾ་བུ་ཀཱ་སྱཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ག་རུ་ཌཱ་སྱཻ་ཧུྃ་ལཊ་སྭཱ་ཧཱ། ཨོཾ་ཨུ་ལཱུ་ཀཱ་སྱཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་གྲྀ་དྷ་སྱཻ་ཧུ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཀཱ་ཀཱ་སྱེེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཤུདྡྷེ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཙནྡཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་སཱུརྱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། 
ཨོཾ་བཛྲ་མཾ་ག་ལཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བུདྡྷཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བྲྀཧྤ་ཏཱ་ཡེེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤུ་ཀྲཱ་ཡ་ཧུྃ་ལཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཤ་ནེཤྩ་རཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཀཱརྟ་རི་ཧུྃ་ལཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་རཱ་ཧུ་ལ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཀཱ་ལཱགྣྱེེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷྲུ་བེ་ནི་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་
བཛྲ་ཨཱ་གསྟྱཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་སརྦ་ན་ཀྵ་ཏྲོ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་དྭཱ་ད་ཤ་རཱ་ཤི་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཥོ་ཌ་ཤ་ཀཱ་ལ་བྷྱོ་ཧུྃ་ལཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨ་ནནྡེ་བཱི་ཤྭཱ་རཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མ་ཧཱ་ཀཱ་ལཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་གྷཎྚ་ཀརྞཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་བྷྲྀཾ་གི་ར་ཊེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཀྵཻ་ཏྲ་པཱ་ལི་
བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་དཱུ་ཏི་བྷྱོ་ཧུ་ཕཊ་སྭ་ཧཱ། ཨོཾ་ཧ་རི་ཏི་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་སིདྡྷི་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་གུ་རུ་བུདྡྷ་བོ་དྷི་ས་ཏྭེ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་དྷརྨྨཱ་ཙཀྲཱ་ཡུ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བྷ་དྲ་གཎྜཱི་ཡེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དུནྡུ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བོ་དྷི་བྲི་ཀྵི་ཡེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། དེ་ནས་མཆོད་བསྟོད་ཅི་ནུས་བྱ། །ཁྲོ་རྒྱལ་ལ་
སོགས་གཏོར་མ་སྦྱིན། །དགེ་བ་འདི་ཡིས་རྣམ་གསུམ་སྲིད་པ་ལ་གནས་སེམས་ཅན་རྣམས་ནི་བླ་མེད་བྱང་ཆུབ་འགྱུར་བར་ཤོག་ཐུབ་པ་འཁོར་བཅས་གཙོ་བོའི་བྱིན་གྱིས་བརླབ་ཀྱིས་སྣ་ཚོགས་ཕྱག་རྒྱའི་དངོས་གྲུབ་ཐོབ་གྱུར་པ། །སྲིད་པ་རྣམ་གསུམ་ལ་གནས་སེམས་ཅན་བཅོམ་ལྡན་དུས་འཁོར་སྟོབས་
ཀྱིས་དེ་བཞིན་ཤོག །བསོད་ནམས་དེ་ཡིས་སེམས་ཅན་མི་འགྱུར་མཆོག་གི་བདེ་བར་གྱུར་ཅིག་དེ་ལ་འདུད། །སྤྱི་བོར་རྡོ་རྗེ་རྩེ་ལྔ་རུ། །གཙོ་བོ་ཡབ་ཡུམ་གཤེགས་སུ་གསོལ། །ཐུགས་ཀར་རྩེ་དགུ་དུ་བ་སོགས། །དཔྲལ་བར་རྩེ་རྨོ་བཅུ་དྲུག་བསྒོམ། །རིགས་ལྔ་ཡབ་ཡུམ་བུམ་པར་
