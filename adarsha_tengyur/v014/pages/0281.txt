རྡོ་རྗེ་ཀུན་ལ་གོ་བར་བྱའོ། །མི་བསྐྱོད་པའི་ཧཱ་ས་དང་། པི་ལས་དང་། ར་ཏི་དང་། ལི་ལཱ་དང་། ལ་ལི་ཏ་དང་། སྭ་ཡཾ་བྷུ་དང་། དྭེ་ཥ་དང་། ཧེ་དང་། ཀ་རུ་ཎ་དང་། ཙིཏྟ་དང་། ཧཱུྃ་ཀ་ར་དང་། ཨཀྵོ་བྷྱ་བཛྲ། ཞེས་པ་རྣམས་སོ། །དོན་ཡོད་གྲུབ་པའི། །ཨ་མོ་གྷ་དང་། བ་ར་མཱ་ཤྭཱ་དང་། ས་མ་ཡ་དང་། 
པཱ་ར་མི་ཏ་དང་། ཀརྨ་དང་། ཨ་ནི་ལ་དང་། སྤརྴ་དང་། ཨཪྵ། ཞེས་པ་རྣམས་སོ། །རིན་ཆེན་འབྱུང་ལྡན་གྱི། རཏྣ་དང་། ཙིནྟཱམ་ཎི་དང་། ཁ་དང་། ག་ག་ན་དང་། ཏེ་ཛོ་དང་། མཱརྟ་ཎ་ཌ་དང་། བྷས་ཀ་ར་དང་། མཱ་མ་ཞེས་པ་རྣམས་སོ། །རྣམ་པར་སྣང་མཛད་ཀྱི། བཻ་རོ་ཙ་ན་དང་། ཨོཾ་ཀཱ་ར་དང་། ཤ་ཤཱ་ཏ་
དང་། ཏ་ཐཱ་ག་ཏ་དང་། མཱ་ཡཱ་དང་། བུདྡྷ་དང་། རཱུ་པ་དང་། ཀཱ་ཡ་བཛྲ། ཞེས་པ་རྣམས་སོ། །འོད་དཔག་མེད་ཀྱི། རཱ་ག་དང་། ཨ་བ་ལོ་ཀི་དང་། དྷརྨྨ་དང་། པདྨ་དང་། བཛྲ་ཨཾ་ཀུ་ར་དང་། པདྨ་ཨཾ་ཀུ་ར་དང་། སུ་ར་ཏ་ཨཾ་ཀུ་ར་དང་། ཀ་རུ་ཎ་ཨཾ་ཀུར་དང་། པདྨ་ཨ་ནནྟ་དང་། སུ་ར་ཏ་ཨ་ནནྟ་དང་། ཨ་
མི་ཏ་དང་། བཱ་ཀ་བཛྲ། ཞེས་པ་རྣམས་སོ། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི། བི་གི་དང་། བེ་ལ་ས་ནི་དང་། པ་དཱ་དང་། ཀོ་ཀ་ན་དང་། ཨ་ནནྟ་དང་། པ་དཱ་ཀཱ་དང་། མཱ་ལཱ་དང་། བཛྲ་ཨུ་ད་ཡ། ཞེས་པ་རྣམས་སོ། །རྡོ་རྗེ་དབྱིངས་ཀྱི་དབང་ཕྱུག་མའི། བཛྲ་ཧ་ར་དང་། སཱ་སཱ་དང་། ཛྭ་ལ་དང་། མཱ་ལཱ་
དང་། ཨཱ་ཀཱ་སཱ་དང་། བཛྲ་ཨཱ་ཀཱ་ས་ར་ཏི། ཞེས་པ་རྣམས་སོ། །མཱ་མ་ཀཱིའི། ཨཱ་ཏ་མ་དང་། ཧསྟ་དང་། ཨ་ཝ་ལཱི་དང་། ཀ་རཱ་དང་། བཛྲ་མཱ་མ་ཀཱི་དང་། དྭེ་ཥ་ར་ཏཱི། ཞེས་པ་རྣམས་སོ། །སྒྲོལ་མའི། གརྦྷ་དང་། མཱ་ཡཱ་དང་། བ་ཌ་ཝ་དང་། ཏཱ་རཱ་དང་། བཛྲ་ན་རྟྱེ་ཤྭཱ་རཱི་དང་། ས་མ་ཡ་བཛྲ་དང་། བཛྲ་ར་ཏི་
ཞེས་པ་རྣམས་སོ། །སྤྱན་མའི། ཌཱ་ཀི་ནཱི་དང་། གོོ་རཱི་དང་། ལཱ་སྱ་དང་། ལོ་ཙ་ན་དང་། བཛྲ་མོ་ཧ་ར་ཏི་ཞེས་པ་རྣམས་སོ། །གོས་དཀར་མོའི། རཱ་ག་དང་། གཱི་ཏ་དང་། ཨ་ཏོ་སུ་ཁ་དང་། བཻ་ཏཱ་ལི་དང་། མ་ཏཱི་དང་། ཧ་སྱ་དང་། པཱཎྜ་རི་དང་། བཛྲ་རཱ་ག་ཏི། ཞེས་པ་རྣམས་ཀྱི་དང་པོ་རྡོ་རྗེ་དང་ལྡན་པར་
བྱས་ཏེ། འདི་རྣམས་ཀྱིས་ཡིད་རྣམ་པར་དག་པ་རྡོ་རྗེའི་བརྟུལ་ཞུགས་དང་མིང་གི་དབང་གི་ཆོ་གའི་རིམ་པའོ། །དེ་ནས་དེ་ལྟར་གཡས་སྐོར་དུ་ཁྲིད་ལ་ནུབ་ཀྱི་ཞལ་དུ་ཡེ་ཤེས་དག་པའི་དོན་དུ་སློབ་མས་གསོལ་བ་གདབ་པ་དང་། ཡེ་ཤེས་རྡོ་རྗེའི་སྐུ་ཡེ་ཤེས་སེམས་དཔའ་དང་ཐ་མི་དད་པ་ཡེ་ཤེས་དང་རྣམ་པར་ཤེས་
