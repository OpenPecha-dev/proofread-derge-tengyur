ཨཱཿབཛྲ་ཝསྟྲ་ཡེ་ཧཱུྃ་སྭཱ་ཧཱ། ཞེས་པ་དང་། དེ་ལྟར་ཨོཾ་ཨཱཿབཛྲ་དང་པོར་བརྗོད་ཅིང་ཧཱུཾ་སྭཱ་ཧཱ། མཐའ་མར་སྦྱར་བའི་སྔགས་རྣམས་ཀྱིས་མེ་ཏོག་དང་། བདུག་སྤོས་དང་། མར་མེ་དང་། དྲི་དང་། ལྷ་བཤོས་དང་། རྒྱན་ལ་སོགས་པ་དབུལ་ལོ། ཨོཾ་ཨཱཿསརྦ་ཤོ་དྷ་ན་ཧཱུྃ་ཕཊ། ཅེས་པ་ནི། འབྲས་བུ་ལ་
སོགས་པ་དག་པར་བྱ་བའི་སྔགས་སོ། །ཨོཾ་བཛྲ་སྭཱ་ཧཱ། ཞེས་པའི་འབྲས་བུས་འདྲེན་པའི་སྔགས་སོ། །དེ་ནས་དམ་ཚིག་གི་རྫས་བྱིན་ལ། ཨོཾ་བཛྲ་ཨཱཿ། ཞེས་པའི་སྔགས་ཀྱིས་མེ་ལོང་བསྟན་ནོ། །དེ་ནས་སྦྱིན་སྲེག་གིས་མཉེས་པར་བྱ་བ་ནི། ཡོན་བདག་གི་འབྱོར་པའི་བྱེ་བྲག་གིས་ལུགས་མ་
ལ་སོགས་པའི་གཟུགས་ཀྱི་བྱེ་བྲག་ལ་སྟེ། ཞི་བ་ལ་སོགས་པས་སངས་རྒྱས་ཀྱི་བར་འདོད་པའི་དོན་ནི་དགོས་པའོ། །གསེར་གྱི་ཐུར་མས་དངུལ་ལ་སོགས་པའི་སྣོད་དུ་སྦྲང་རྩི་བཞག་པ་ལས་བླངས་ཏེ། ཨོཾ་ཙཀྵུ་ས་མནྟ་ཙཀྵུར་བི་ཤོདྷ་ནི་སྭཱ་ཧཱ། ཞེས་བརྗོད་ཅིང་ལུགས་མའི་སྤྱན་ལ་བྱུག་ཅིང་དབྱེ་བར་བྱའོ། །
པོ་ཏི་ལ་སོགས་པ་རྣམས་ལ་མེ་ལོང་གི་གཟུགས་བརྙན་ལ་དེ་ལྟར་བྱའོ། །དེ་ནས་ཐབ་ཁུང་གི་མེའམ། མེ་གཞན་ལ་འོ་མ་དང་། འབྲས་མཉམས་པ་དང་། མར་དང་། སྦྲང་དང་། ཀ་ར་རྣམས་ཧོམ་གཟར་རམ། པི་པི་ལིང་གི་ལོ་མས་སྒྱེད་པུའི་སྟེང་གི་སྣོད་དུ་བླུགས་སོ། །གཙོ་བོའི་སྔགས་བརྗོད་
ཅིང་ཚོས་པར་བཙོ་བར་བྱའོ། །ཐུག་པ་དེ་ཉིད། ཨོཾ་དི་བྱཱ་ནཾ་ས་མཱ་དྷཱི་དྷྱ་ན་པྲི་ཎ་ནེ་སྭཱ་ཧཱ། ཞེས་བརྗོད་ཅིང་ལུགས་མ་ལ་སོགས་པ་ལྷའི་གཟུགས་བརྙན་དུ་བསམས་ལ་དྲང་བར་བྱའོ། །ཞལ་བསིལ་ལ་དྲི་ཞིམ་པོའི་སྦྲང་རྩི་དང་བཅས་པ་དབུལ་ལོ། །དེ་ནས་ཨོཾ་ཧཱུྃ་ཏྲཱཾ་ཧྲཱིཿཨ། ཞེས་པའི་དམ་ཚིག་འདི་
ཉན་དུ་གཞུག་གོ། །པོ་ཏི་རྣམས་ལ་འོད་དཔག་མེད་ཤེས་རབ་དང་བཅས་པ་བསྟིམ་པས། བྱང་བུ་རྣམས་མཚན་སུམ་ཅུ་རྩ་གཉིས་དང་། དཔེ་བྱད་བརྒྱད་ཅུའི་རང་བཞིན་དབྱངས་དང་གསལ་བྱེད་དུ་བསྒོམ་མོ། །ལྷ་ཁང་ལ་སོགས་པ་རྣམས་ལ་ནོམ་པར་སྣང་མཛད་ཤེས་རབ་དང་བཅས་པ་ཐིམ་
པས། རང་བཞིན་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་ཐམས་ཅད་དང་། དེའི་ཆོས་དང་། བྱང་ཆུབ་སེམས་དཔའ་དང་། ཉན་ཐོས་ལ་སོགས་པའི་ཏིང་ངེ་འཛིན་སྣ་ཚོགས་ཀྱི་བྲང་ཁང་དུ་བསམ་པར་བྱའོ། །མཆོད་རྟེན་ལ་དེ་ལྟར་རྣམ་པར་སྣང་མཛད་ཤེས་རབ་དང་བཅས་པ་ཐིམ་པས་དེའི་དབུས་སུ་
