ཙ་ར་ཎཱ་ཡ། རིགས་དྲུག་བཀོད་པའི་བསྟོད་པ་དང་། །མཆོག་སྦྱིན་རྡོ་རྗེའི་བསྟོད་པས་བསྟོད། །དེ་ནས་མཚན་ཕྱེད་གཏོར་མ་སྦྱིན། །ཨོཾ་ཨོཾ་བཛྲ་ཙནྡྲ་སརྦ་དྷརྨ་སུ་བི་ཤུདྡྷ་སརྦ་དྷརྨཱ་སུ་བི་ཤོ་དྷ་ནཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཨཱཿབཛྲ་སཱུརྱ་སརྦ་དྷརྨ་པྲ་བོ་དྷ་ཀ་སརྦ་དྷརྨ་པྲ་བོ་དྷཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཧཱུྃ་བཛྲ་ཨ་ན་ལ་
སརྦ་དྷརྨ་པྲ་དཱི་པ་ཀ་སརྦ་དྷརྨ་པྲ་དཱི་པཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཧོ་སརྦ་དྷརྨ་བཛྲ་ཨ་མྲྀ་ཏ་ཀར་སརྦ་དྷརྨ་བཛྲ་ཨ་མྲྀ་ཏཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། དེ་ལྟར་སྔགས་དང་ཕྱག་རྒྱ་ཡིས། །གཏོར་མ་བྱིན་བརླབ་ཟིན་ནས་ནི། །ཕྱག་རྒྱ་སྔགས་ཀྱིས་སྤྱན་དྲངས་བྱ། །ཨོཾ་ཧ་ཧཱ་ཧི་ཧཱི་བཛྲ་བྷེེ་ར་བ་ཨཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ་པྲ་བེ་ཤ་ཡ་པྲ་བེ་ཤ་ཡ། བནྡྷ་ཡ་
བནྡྷ་ཡ། ཏོ་ཥ་ཡ། ཏོ་ཥ་ཡ། ཛཿཧཱུྃ་བཾ་ཧོཿ། སརྦ་ཌཱ་ནཱི་ནཾ་ཧྲྀ་ད་ཡཾ་ཧཱུྃ་ཧཱུྃ་ཕྲེཾ་ཕྲེཾ་ཕཊ། དེ་རྗེས་མགྲོན་རྣམས་བྱོན་པ་དང་། །ལྷར་བསྐྱེད་དབང་བསྐུར་བྱིན་བརླབས་པའི། །ཆོ་གའི་རིམ་པས་རྫོགས་པར་བྱ། །མཆོད་པ་རྣམས་དང་གཏོར་མ་བསྔོ། །ཨོཾ་བྲཧྨ། བིཥྞུ། ནེཪྼི་ཏྱ། བཱ་ཡུ། ཡ་མ། ཨགྣི། ས་མུ་
དྲ་ཨི་ཤོ་ར། ཨིནྡྲ་ཡཀྵེ་བྷྱཿ། ས་པ་རི་བཱ་རེ་བྷྱཿ། ཨི་དཾ་བ་ལིཾ། གནྡྷཾ། པུཥྤཾ། དྷཱུ་པཾ་དཱི་པཾ་ཨ་ཀྵ་ཏཾ་ད་ད་མ་ཧེ་ཏེ་ཙ་ཨེ་ཀཏྱ་ས་པ་རི་བཱ་རཱ་ཤཱི་གྷྲཾ་ཨི་དཾ་བ་ལིཾ་གྲྀཧྣ་ཏུ་ཁ་དནྟུ་པི་བནྟུ། ཛཿཧཱུྃ་བཾ་ཧོཿ། སན་ཏྲྀབྟ། སརྦ་ས་ཏྭ་ནཱཾ་ཤཱནྟིཾ་པུཥྚི་རཀྵཱ་བ་ར་ན་གུབྷིཾ་ཀུརྦནྟུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། བཛྲ་དྷར་ཨཱཛྙཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། སླར་ཡང་
མཆོད་ཅིང་བསྟོད་པ་དང་། ཡི་གེ་བརྒྱས་ཕྲིན་ལས་སོ་སོར་བཤགས། །གཙུག་ཏོར་དུ་ནི་རྩེ་ལྔ་པའི། །རྡོ་རྗེ་རུ་ནི་གྱུར་པ་ཡིན། །ཕྱི་ནས་རྩེ་མོ་ཉིས་འགྱུར་དུ། གྱུར་པ་ཆོས་ཀྱི་འཁོར་ལོའོ། །དེ་ནི་ཉིས་འགྱུར་དུ་བསྒྲེས་པ། །ལྷན་ཅིག་སྐྱེས་པར་འགྱུར་བ་དང་། །དེ་ཡི་ཡང་ནི་གཞན་ཉིས་འགྱུར། །ལོངས་
སྤྱོད་རྫོགས་པའི་འཁོར་ལོ་དང་། །སྤྲུལ་པ་ཡི་ནི་འཁོར་ལོའོ། །དེ་ཡི་ནང་དུ་རྩེ་གཅིག་གོ། །མཉམ་པའི་བདེ་བའི་འབྲས་སྟེར་བ། །གསང་བའི་པདྨ་ལ་གནས་པ། །ཕྱག་རྒྱ་བསྒོམ་པའི་དོན་དུ་གསུངས། །ཉིན་དང་མཚན་མོའི་དུས་སུ་ནི། །འདོད་ཆགས་ཟད་དོན་ཉིད་དུ་ནི། །གསུང་གི་རྡོ་རྗེ་མྱང་དོན་དུ། 
སྔགས་པ་རྣམས་ལ་གསུངས་པ་ཉིད། །ཉེས་པར་བྱ་བ་ཁོ་ན་མིན། །ཉི་མ་སོ་སོའི་དུས་སུ་ནི། །ཟས་ཀུན་བདེ་བའི་དོན་དུ་སྟེ། །མི་འཇུག་པ་ཡི་རྒྱུ་རུ་ནི། །དཔལ་གྱི་སྤྱོད་པ་དངོས་གྲུབ་སྒྱུར། །རོ་བསྲེགས་གནས་སུ་མཚན་མོ་ནི། །འགྲོ་ལ་བརྩེ་བའི་དོན་དུ་མིན། །རྣལ་འབྱོར་པ་གང་སྲོག་གཅོད་པའི། །
