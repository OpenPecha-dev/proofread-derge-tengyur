བཱ་ཡུ་བེ་གཱ། པྲ་ཙཎྜ་རོོ་དྲཱ་ཀྵཱ། སྠཱུ་ལ་ནཱ་སཱ་བྷཱཿས་པ་རི་ཝཱ་རེ་བྷྱ། ཨེ་དཾ་བ་ལིཾ་གནྡྷཾ་པུཥྤཾ་དྷཱུ་པཾ་དཱི་པཾ་ཨཀྵ་ཏཾ་ད་དཱ་མ་ཧེ་ཏེ་ཙ་ཨཱ་བ་ཏྱ་ས་པ་རི་ཝཱ་རེཿ ཤཱི་གྷྲཾ་ཨི་དཾ་བ་ལིཾ་གྲྀ་ཧྞ་ནྟུ་ཁཱ་དནྟ། པི་བནྟུ་ཛཿ་ཧཱུཾ་བཾ་ཧོཿ། སཾ་ཏྲིཔྟཱཿ སརྦ་སཏྭ་ནཱཾ་ཤཱནྟི་པུཥྚི་རཀྵཱ་བར། གུཔྟིཾ་ཀུརྦནྟུ་ཧུཾ་ཧཱུཾ་ཕཊ་བཛྲ་དྷ་ར་ཨཱཛྙཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། ཞེས་
བྱ་བས་ཙཱ་རྩི་ཀཱ་འཁོར་དང་བཅས་པ་ལ་གཏོར་མ་དབུལ་ལོ། །ས་པ་རི་ཝཱ་ར་ཞེས་བྱ་བ་ནས་སྭཱ་ཧཱའི་བར་གྱིས་གཞན་རྣམས་ལ་ཡང་རང་མིང་དང་བཅས་པས་དབུལ་བར་བྱའོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿབཛྲ་བཻཥྞུ་པི་ཤྲཱིཿམཱ་ཡཱ། ཀཱི་ཏཱི། ལཀྵྨྱ། །སུ་པ་རམ། བི་ཛ་ཡཱ། ཤྲཱི་ཛ་ཡ་ན་ཏཱི། ཤྲཱི་ཙཀྲི་བྷྱཿ། ཞེས་
བྱ་བས་ཁྱབ་འཇུག་མ་འཁོར་དང་བཅས་པ་ལ་འབུལ་ལོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿ། བཛྲ་བཱ་རཱ་ཧཱི། ཀཾ་ཀཱ་ལཱི། ཀཱ་ལ་རཱ་ཏྲཱི། པྲ་ཀུ་པི་ཏི་པ་དཱ་ནཱ། ཀཱ་ལ་ཛི་ཧྭཱ། ཀཱ་རཱ་ལཱཾ། ཀཱ་ལཱི། གྷོ་རཱ། པི་རཱུ་པ་བྷྱཿ། ཞེས་བྱ་བས་རྡོ་རྗེ་ཕག་མོ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿབཛྲ་ཀུ་མཱ་རཱི། པདྨཱ།
ཨ་ནཾ་གཱ། ཀུ་མཱ་རཱི། མྲྀ་ག་པ་ཏི་ག་མ་ནཱ། རཏྣ་མཱ་ལཱ། སུ་ནེ་ཏྲཱ། ཀླཱི་ནྣ། བྷ་དྲཱ་བྷྱཿ། ཞེས་བྱ་བས་གཞོན་ནུ་མ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿ་བཛྲ་ཨིནྡྲི། བཛྲཱ་བྷཱ། བཛྲ་གཱ་ཏྲཱ། ཝ་ཪྐ་ན་ཀ་བ་ཏཱི། ཙོ་ར་བ་སཱི། ཙི་ཏྲ་ལེ་ཁཱ། རམ་བྷཱ། ཧ་ལྱཱ། སུ་ཏྲཱ་རཱ་བྷྱ༔། ཞེས་བྱ་བས་
དབང་མོ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿམ་ཧཱ་བྲ་ཧྨཱ་ཎཱི། སཱ་བི་ཏྲ། པདྨ་ནེ་ཏྲཱ། ཁ་ལུ་ཛ་ལ་ཛ་བི་ཏཱི། བུདྡྷཾ། ཝཱ་གཱི་ཤྭཱ་རཱི། དྷཱེ་གཱ་ཡ་ཏཱི། བི་བྱུ་ཏཱ། སྨྲྀ་ཏི་བྷྱཿ། ཞེས་བྱ་བས་ཚངས་མ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། ཨོཾ་ཨཱཿཧུཾ་ཧོཿབཛྲ་མ་ཧེ་ཤྭ་རཱི། གཽ་རཱི། 
གཾ་གཱ། ན་ནི་ཏྱཱ།  སུ་པ་ར་མ་ཏུ་རི་ཏཱ། ཏོ་ཏ་ལཱ། ལཀྵྨ་ཎཱ། པི་ག་ལཱ། ཀྲྀཥྞཱ་བྷྱཿ། ཞེས་བྱ་བས་དྲག་མོ་འཁོར་དང་བཅས་པ་ལ་དབུལ་ལོ། །ཨོཾ་ཨཱཿཧུཾ་ཧོཿ། མ་ཧཱ་ལཀྵྨཱ། ཤྲཱི་ཤྭེ་ཏཱ། ཙནྡྲ་ལེ་ཁཱ། ཤ་ཤཱ་དྷ་ར་བ་དཱ་ནཱ། ཧཾ་ས་བ་ར་ཎཱ། དྷྲྀ་ཏི། པདྨེ་ཤྭཱ། ཏྲཱ་རཱ་ནེ་ཏྲཱ། བི་མ་ལ་ཤ་ཤཱ་དྷ་རཱ་བྷྱཿ།
ཞེས་བྱ་བས་བཀྲ་ཤིས་ཆེན་མོ་འཁོར་དང་བཅས་པ་རྣམས་ལ་གཏོར་མ་དབུལ་བར་བྱའོ། །ས་པ་རི་ཝཱ་རེ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནས་སྭཱ་ཧཱའི་བར་དུ་སྡེ་ཚན་རེ་རེ་ཞིང་ཡང་ཤེས་པར་བྱའོ། །དབྱུ་གུ་དྲུག་ཅུ་རྩ་བཞི་ཡི། །གཏོར་མ་དྲུག་ཅུ་རྩ་བཞི་ཡིས། །ས་སྟེངས་གནས་ན་ཕོ་ཉ་རྣམས། །
