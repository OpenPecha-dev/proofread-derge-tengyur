འདི་ནི་ཇི་ལྟར་སྒྲུབ་བྱེད་མིན། །དེའི་ཚེ་ཕྱོགས་ཀྱང་ཤེས་པ་ཡིན། །ཞེས་བྱ་བ་བསལ་བ་ཡིན་ནོ། །ཁས་བླངས་པས་གྲུབ་པ་ནི་མངོན་སུམ་མ་ཡིན་ན་གང་གིས་དེ་བཀག་པ་ན་དེས་གནོད་པར་འགྱུར་དེའི་ཕྱིར་མི་རིགས་སོ། །མངོན་སུམ་ལ་སོགས་པ་ཡི་ཡུལ། །ཕྱི་རོལ་ཁོ་ནར་གནས་པ་ཡིན། །
དེ་དགག་བྱས་པ་དེ་ཡི་ཕྱིར། །དེ་དག་གི་ཕྱོགས་ལ་གནོད་འགྱུར། །ཞེས་ཟེར་བ་ནི། གལ་ཏེ་དེ་ལྟར་ཞེས་བྱ་བའོ། །འཁྲུལ་པ་དང་ཅིག་ཤོས་དག་གིས་གནོད་པར་བྱ་བ་ལ་སོགས་པའི་ངོ་བོར་འདོད་པ་ཁོ་ན་ཡིན་ནོ་ཞེས་ཟེར་བ་ནི། གལ་ཏེ་སྔ་ཕྱི་ཞེས་བྱ་བའོ། །ཉ་ཕྱིས་ལ་དངུལ་གྱི་ཤེས་
པ་བཞིན་དུ་དངུལ་མ་ཡིན་ན་ཇི་ལྟར་དངུལ་གྱི་ཁྱད་པར་དགག་པ་ཡིན་ཞེས་བརྒལ་ཞིང་བརྟགས་པའོ། །དེའི་འཁྲུལ་པ་ཉིད་ག་ལས་ཞེས་བྱ་བ་ནི་གཞན་ནོ། །ཅི་ལས་ཤེ་ན་འཁྲུལ་ཉིད་ཅེས་བྱ་བའོ། །དོན་དམ་པར་ན་ཕྱི་རོལ་གྱི་འཁྲུལ་པ་མེད་པ་ཅུང་ཟད་ཀྱང་ཡོད་པ་མ་ཡིན་པའི་ཕྱིར་
ཁྱབ་ཆེས་པ་ཡིན་ནོ། །རྒྱུ་དང་ལྡན་པ་ཉིད་ཀྱིས་འཁྲུལ་པ་ནི་མ་ཡིན་གྱི་འོན་ཀྱང་དེ་རྒྱུ་དང་ལྡན་པ་ནི་ཡིན་ཏེ། དེ་ལས་བསྟན་པ་ནི་གལ་ཏེ་རྒྱུར་བཅས་ཞེས་བྱ་བ་སྟེ། མ་འཁྲུལ་པར་འདོད་པ་དེ་དང་རྒྱུ་དང་ལྡན་པ་ཡིན་ན་དེ་ལས་བརྒལ་ཞིང་བརྟགས་པ་མཚུངས་སོ་ཞེས་བྱ་བའི་དོན་ཏོ། །འཁྲུལ་
པ་ཡོད་མོད་ཅེས་བྱ་བ་ནི་གནས་སྐབས་ལས་སོ། །འཁྲུལ་པར་འདོད་པ་དེ་སྒྲུབ་པར་བྱེད་པས་མ་འཁྲུལ་པས་འཁྲུལ་པ་ཐོབ་པར་འགྱུར་རོ་ཞེས་ཟེར་བ་ནི། གལ་ཏེ་མ་འཁྲུལ་པ་ནི་ཞེས་བྱ་བའོ། །ཕྱི་རོལ་དུ་འཁྲུལ་པ་ལ་སྒྲུབ་པའི་ངོ་བོ་འཐོབ་པ་མ་ཡིན་ཏེ་མི་མཐོང་བའི་ཕྱིར་རོ། །དེའི་
ཕྱིར་ཁྱད་པར་དུ་འཁྲུལ་པ་ཡིན་པར་ངེས་པ་ཉིད་དུ་བསྟན་པ་ནི། སྒྲུབ་པའི་ཚུལ་དུ་མཐོང་བ་གང་ཞེས་བྱ་བའོ། །འཁྲུལ་པ་ཉིད་ནི་རྗེས་སུ་བརྗོད་པར་བྱ་བ་ཡིན་གྱི། བསྒྲུབ་པར་བྱ་བ་ནི་མ་ཡིན་ཏེ། ཐལ་ཆེས་པའི་ཕྱིར་རོ་ཞེས་བསྟན་པ་ནི། གང་ཡིན་པ་དེ་ཞེས་བྱ་བའོ། །ཁྱད་པར་དང་བཅས་པའི་
བློ་གང་དུ་རྒྱུ་ཡིན་པ་དེ་འཁྲུལ་པ་ཡིན་ནོ་ཞེས་བྱ་བའི་ཁྱད་པར་གྱིས་ཉེ་བར་བཀོད་པ་ལས་ཡིན་གྱི། རྒྱུ་ངེས་པ་ལས་ནི་སྔར་རྟོགས་པའི་འཁྲུལ་པ་ཉིད་རྟོགས་པ་ནི་མ་ཡིན་ནོ་ཞེས་བསྟན་པ་ནི། འཁྲུལ་པའི་རང་བཞིན་ཤེས་ན་ཡང་། །ཞེས་བྱ་བ་སྟེ། ཁས་བླངས་པ་ཙམ་ལས་ནི་འཁྲུལ་པ་ཉིད་མ་ཡིན་
