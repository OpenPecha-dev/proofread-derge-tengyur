ཡཱ་ཏེ། ཨཱ་དྷཱི་ཡཱཏ། ཨ་དྷཱི་ཥེ། ཨ་དྷཱི་ཡཱ་ཐེ། ཨ་དྷཱི་དྷྭེ། ཨ་དྷཱི་ཡེ། ཨ་དྷཱི་བ་ཧེ། ཨ་དྷཱི་མ་ཧེ། དཱི་དྷཱིང་། འབར་བ་དང་རྩེ་བ་དག་ལའོ། །བི་བཱིང་། བྱར་དགའ་མཚུངས་པ་ལའོ། །འདི་དག་སྡེབ་སྦྱོར་དག་ལའོ་ཞེས་ཁ་ཅིག་གོ། །ཨཱང་སྔོན་དུའོ། །ཨཱ་དཱིདྷཱི་ཏེ། སྔ་མ་བརྩེགས་
བ་མ་ཡིན་པའི་ཨེ་ཡིག་གི་ཡ་སྟེ་ཡི་གེ་གཅིག་མིན་པའི་འོ་ཞེས་པས་ཡ་ཉིད་དོ། །ཨཱི་དཱི་དྷྱཱ་ཏེ། ཨ་དཱི་དྷྱ་ཏེ། ཨཱ་བེ་བཱི་ཏེ། ཨ་བེ་བྱ་ཏེ། ཧྣུང་སེལ་བར་བྱེད་པ་ལའོ། །ཧྣུ་ཏེ། ཧྣུ་བཱ་ཏེ། ཧྣུ་བ་ཧེ། ནི་ཧྣུ་ཏེ། ནི་ཧྣུ་བཱ་ཏེ། ཨཱ་བ་ཧྣུ་ཏེ། ཨཱཏྨ་ནེའི་ཚིག་རྣམས་སོ།། །།དྭི་ཥ་མི་དགའ་བ་ལའོ། །
