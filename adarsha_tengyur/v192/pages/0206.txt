རོ། །དེ་ཉིད་ཀྱི་ཕྱིར་ཤེས་པ་ཉིད་ཤེས་པའི་རྒྱུ་ཡིན་པར་བསྟན་པའི་ཕྱིར་རོ། །རྩོད་པའི་གཞིར་སེམས་དག་ཅེས་བྱ་བ་ལ་སོགས་པའི་སྦྱོར་བ་དང་པོ་ལ། ཐ་དད་ལུས་ལ་གནས་པའི་ཕྱིར། །ཞེས་བྱ་བའི་གཏན་ཚིགས་མ་གྲུབ་པ་ཉིད་ཡིན་ཏེ། རེ་ཞིག་གལ་ཏེ་འཇུག་པ་བརྟེན་པའི་མཚན་ཉིད་དུ་མངོན་
པར་འདོད་ན་ལྷུང་བ་མེད་པའི་ཆོས་ཅན་གྱི་སེམས་དེ་ནི། ལུས་ལ་ཤིན་ཏུ་ཡང་མ་གྲུབ་སྟེ། དེའི་རྒྱུ་དང་འབྲས་བུའི་དངོས་པོ་ཡིན་ན་ཡང་སེམས་ལུས་ཅན་མ་ཡིན་པ་ཉིད་ཀྱིས་མི་ལྟུང་བའི་ཆོས་ཅན་ཡིན་པའི་ཕྱིར་རོ། །ལྟུང་བ་མེད་པའི་ཆོས་ཅན་ཡང་བརྟེན་པར་རིགས་པ་མ་ཡིན་ཏེ། ཅུང་ཟད་ཀྱང་མི་
བྱེད་པའི་ཕྱིར་རོ། །འོ་ན་ཆུ་ལ་སོགས་པ་ཅི་ཞིག་བྱེད་ན་བརྟེན་པར་འགྱུར་ཞེ་ན། ཆུ་ལ་སོགས་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། རང་བཞིན་གྱིས་ལུས་ཅན་རང་གི་ཉེ་བར་ལེན་པའི་ཡུལ་ཡོངས་སུ་སྤངས་ནས་སྐྱེ་བ་ཉེ་བར་ལེན་པའི་ཡུལ་དུ་སྐྱེ་བའི་རྒྱུ་ཉིད་ཡིན་པའི་ཕྱིར།  ལྷུང་བའི་གེགས་བྱེད་
པས་རྟེན་དུ་རྣམ་པར་གཞག་གི །ལུས་ཅན་མ་ཡིན་པ་ནི་དེ་ལྟར་རྣམ་པར་གཞག་པར་ནུས་པ་མ་ཡིན་ནོ། །ཅི་སྟེ་དེའི་བདག་ཉིད་ཀྱི་མཚན་ཉིད་འཇུག་པར་མངོན་པར་འདོད་ན་དེ་ཡང་མ་གྲུབ་པོ། །ཕྱི་རོལ་གྱི་དོན་ལ་མངོན་པར་ཞེན་པ་ཁྱེད་ཅག་གིས་རྣམ་པར་ཤེས་པ་ལུས་ཀྱི་བདག་ཉིད་ཡིན་ནོ་
ཞེས་ནི་བརྗོད་པར་རིགས་པ་མ་ཡིན་ནོ། །རྣམ་པར་ཤེས་པ་ཙམ་དུ་སྨྲ་བ་ངེད་ཅག་ལ་ནི་རིགས་ཏེ། ལུས་ནི་ཀུན་གཞི་རྣམ་པར་ཤེས་པའི་རང་བཞིན་ཡིན་པའི་ཕྱིར་རོ་སྙམ་དུ་བསམས་པ་ཡིན་ནོ། །ཅིའི་ཕྱིར་མི་རིགས་ཤེ་ན། དེའི་བདག་ཉིད་ཡིན་ན་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། འདི་
ལྟར་ལུས་བཟུང་བ་ན་གཞན་གྱི་འདོད་ཆགས་ལ་སོགས་པ་ཡང་འཛིན་པར་འགྱུར་ཏེ། ཐ་མི་དད་པའི་ཕྱིར་རོ། །དུག་གི་ནུས་པས་མ་ངེས་པ་ཡང་མ་ཡིན་ཏེ། དེ་ནི་དེའི་ཚེ་མངོན་སུམ་མ་ཡིན་པའི་ཕྱིར་རོ། །སེམས་པ་ཅན་ནི་མངོན་སུམ་མ་ཡིན་ཏེ། བདག་ཉིད་ཀྱིས་ཀྱང་མི་འཛིན་
པར་ཐལ་བར་འགྱུར་བའི་ཕྱིར་རོ། །གཞན་ཡང་གང་དག་གཉི་གས་ངེས་པ་དང་གཅིག་གིས་ངེས་པ་དེ་དག་ནི་ཐ་དད་པ་ཡིན་ཏེ། དཔེར་ན་ཕོ་གཟེར་དང་ཟློས་གར་བཞིན་ནོ། །ལུས་དང་སེམས་ཀྱང་གཉི་ག་དང་གཅིག་གིས་ངེས་པ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་རང་བཞིན་གྱི་གཏན་ཚིགས་སོ། །རང་ཉིད་ཅེས་བྱ་
