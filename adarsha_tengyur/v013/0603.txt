པ་ཡིན་ན་ཡང་། ཟངས་ཀྱིའོ་ཞེས་ཅི་ལྟར་ཐོབ་སྟེ། དེ་དང་འབྲེལ་པ་མེད་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །དེས་ན་ཡང་ཇི་ལྟར་ཐ་དད་པ་དང་། ཐ་མི་དད་པར་བརྗོད་པར་བྱ་སྟེ། དྲི་མ་ཉམས་པ་ལས་གསེར་དུ་སྣང་བ་ནི་བསམ་གྱིས་མི་ཁྱབ་པའོ། །དེ་ཁོ་ན་བཞིན་
དུ་སྟོང་པ་ཉིད་ཀྱི་སྦྱོར་བས་མི་ཤེས་པའི་དྲི་མ་དོར་བ་ལས། མཐའ་ཡས་པའི་ཡོན་ཏན་གྱི་ཚོགས་ཀྱི་གནས། འཁོར་བ་དང་མྱ་ངན་ལས་འདས་པ་ལ་རབ་ཏུ་མི་གནས་པ། རྨི་ལམ་དང་མཚུངས་པ། རྣམ་པར་མི་རྟོག་པ་རྣམ་པ་ཐམས་ཅད་པའི་ཡེ་ཤེས་སོ། །
དེ་ལྟར་ཡང་ས་ར་ཧའི་ཞབས་ཀྱིས། རྟོག་པ་ཉམས་པ་བློ་ཉམས་པའི། །བློ་ངན་རྗེས་སུ་རྟོག་བྱེད་ལོ། །བློ་བཟང་རྣམས་ལ་འདི་ཉིད་ལས། །མཁའ་ལས་ལྷག་པར་ཅི་ཡང་འཆར། །ཞེས་སོ། །འགྱུར་མེད་བདེ་བ་ཆེས་པ་ནི། རྣམ་པར་མི་རྟོག་
པའི་བདེ་བ་ཆེན་པོ་ལའོ། །ཆགས་བྲལ་ཞེས་པ་ནི། འཁོར་བར་ཆགས་པའི་བག་ཆགས་ཀྱིས་འཕོ་བ་ལ་མངོན་པར་ཞེན་པའོ། །མི་བྱ་ཞེས་པ་ནི། རྗེས་སུ་གནས་པར་མི་བྱ་བ་སྟེ། ཅིའི་ཕྱིར་ན། ཆགས་པའི་བག་ཆགས་ཀྱིས་འཕོ་བ་ནི། འཁོར་
བ་འཕེལ་བའི་རྩ་བ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །འདི་ནི་འཕོས་པ་ལས་ནི་ཞེས་པ་ལ་སོགས་པས་གསུངས་སོ། །བདེ་བ་དོན་དུ་གཉེར་བའི་འཇིག་རྟེན་པས་ཀྱང་འཕོ་བའི་བདེ་བ་འདོད་ཅིང་རེ་བར་བྱ་བ་མ་ཡིན་ན། དེ་ཅི་ལྟར་ན། དེ་ཁོ་ན་ཉིད་ཀྱི་རྣལ་འབྱོར་པས་འདོད་
པར་བྱ་བ་ཡིན་ཞེས་བྱ་བ་འདི་ནི་ཆགས་ཅན་ཞེས་བྱ་བ་ལ་སོགས་པས་གསུངས་པ་ཡིན་ནོ། །དེའི་ཕྱིར་རྣམ་པར་མི་རྟོག་པའི་བྱང་ཆུབ་ཀྱི་སེམས་འགྱུར་བ་མེད་པའི་རྣལ་འབྱོར་ཁོ་ན་ལས། མ་བརྟགས་པ་མཆོག་ཏུ་མི་འགྱུར་བའི་བདེ་བ་ཆེན་པོ་བསྒྲུབ་པར་
བྱ་བ་ཡིན་གྱི། བརྟགས་པའི་ཧཱུཾ་དང་། ཕཊ་ལ་སོགས་པའི་ཡི་གེ་འགྱུར་བའི་རྣལ་འབྱོར་ལས་ནི་མ་ཡིན་ཏེ། འདི་ནི་ཁུ་བ་ཞེས་བྱ་བ་ལ་སོགས་པས་གསུངས་པ་ཡིན་ནོ། །དོན་དམ་པའི་བདེན་པ་ལས་ནི། སྒྲུབ་པར་བྱེད་པ་དང་བསྒྲུབ་པར་བྱ་བའི་རྟོག་པ་ཡང་ཡོད་པ་མ་
