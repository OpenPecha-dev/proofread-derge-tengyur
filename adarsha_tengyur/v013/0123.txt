ལེན་ལ་སྤྲིན་བཞིན་སྒྲ་སྒྲོགས་བྱེད་ཅེས་པའི་དོན་ལ་སྒྲོགས་བཞིན་པ་སྟེ་དབུགས་འབྱུང་བར་བྱེད་བཞིན་པའོ། །བཅིང་བ་ནི་ཨུ་རའི་ཞགས་པ་སྟེ་སྦྲུལ་གྱི་ཞགས་པ་ལ་སོགས་པའོ། །རྩེ་མོ་ནི་ཐོར་ཏོའི་ཆའོ། །སོའི་རྩེ་མོས་འབྱེད་པར་བྱེད་དེ་ནོར་བུས་རབ་ཏུ་སྟོབས་ལ་སོགས་
པ་བྱེད་པའོ་ཞེས་པའོ། །དེ་ཡང་སོ་སོར་ལྡོག་པས་ཁམས་མཐའ་དག་གཡོ་བ་བྱུང་བ་འདིས་ཀྱང་དེའི་ལུས་གཏོང་བར་བྱེད་དོ་ཞེས་པའོ། །མེ་ནི་འདིར་ལུས་ཀྱི་ལྷན་ཅིག་སྐྱེས་པའི་དགའ་བའི་ཐིག་ལེའོ། །མེ་ལྕེ་ནི་ལུས་ཀྱི་དྲོད་དེ་འདིའི་འབར་བ་སྟེ་ཁྱབ་པར་གྱུར་པའོ་ཞེས་པའོ། །ཀུན་ནས་ཏེ་
ཐམས་ཅད་ནས་སོ། །ཕྱོགས་དང་མཚམས་སུ་སྟེ་རིམ་པས་གཡོན་པའི་ཡན་ལག་དང་གཡས་པའི་ཡན་ལག་གི་ལག་པ་དང་རྐང་པ་ལ་སོགས་པ་རྣམས་སུའོ། །ཀུན་དུ་དུ་བའི་མུན་པ་ཆེན་པོ་སྟེ་ཀུན་དུ་རྣ་བ་དང་སྣ་ལ་སོགས་པ་དང་གདོང་ལ་སོགས་པ་རྣམས་སུ་རབ་ཏུ་དགའ་བ་ཤིན་ཏུ་བརྟེན་
པ་ལས་དུ་བ་སྟེ་གྱེན་དུ་གཡོ་བ་ཡིན་ཞིང་དེས་མུན་པ་སྟེ་དབང་པོ་རྣམས་ཡུལ་རྣམས་ལ་འཁྲུལ་བའོ། །ཆེན་པོ་སྟེ་རབ་ཏུ་གྱུར་པའི་མུར་ཐུག་པར་སྐྱེས་པའི་ཀུན་དུ་དུ་བའི་མུན་པ་གང་གིས་ཡིན་པ་ཞེས་པའོ། །མེ་སྟག་སྟེ་ཐིག་ལེའི་ཚོགས་རྣམས་སོ། །གང་ལས་ཤེ་ན། གྲོང་དང་དགོན་པ་
གཅིག་ཏུ་བསྲེགས་པ་ལས་ཏེ་གྲོང་ནི་གཙུག་ཏོར་གྱི་འཁོར་ལོ་དང་། དགོན་པ་ནི་གསང་བའི་པད་མ་སྟེ། དི་དག་གཅིག་སྟེ་བར་མ་ཆད་པ་ཡང་འདི་ཡིན་ལ་བསྲེགས་པ་ཡང་འདིའོ་ཞེས་པ་དེ་དང་འདི་ཡང་གཏུམ་མོ་འབར་བའི་རྣལ་འབྱོར་ལས་སོ་ཞེས་པའོ། །དུས་ཀུན་དུ་ཤིན་ཏུ་དགའ་
བའི་དུས་ཚོད་ཇི་ལྟར་འགྱུར་བར་རོ་ཞེས་པ་བྱ་བའི་ཁྱད་པར་རོ། །འཆི་བ་ནི་བྱང་ཆུབ་གྱི་སེམས་དང་བྲལ་བའི་མཚན་ཉིད་ཅན་ནོ། །ཕྱོགས་ཀུན་གསང་བའི་འཁོར་ལོའི་མཐའི་མཚན་ཉིད་ཅན་ནོ། །རབ་ཏུ་ཐོབ་པ་སྟེ་མངོན་པར་ཁྱབ་པའང་འདི་ཡིན་ལ། འཁོར་ལོ་དྲུག་བདེ་བ་དང་བྲལ་
བའི་མཚན་ཉིད་ཅན་མེ་སྟེ་ལུས་ཀྱི་ལྷན་སྐྱེས་དགའ་བའི་ཐིག་ལེ་ཡང་ངོ་ཞེས་པ་ལས་འཛིན་པའོ། །བསིལ་བ་ཉིད་ནི་གྲང་བ་ཉིད་དེ་མི་ཤེས་པ་དང་མ་རིག་པས་ཉམས་པའི་ཆོས་ཅན་ཞེས་པའི་བར་རོ། །འགྲོ་བ་སྟེ་ཕ་རོལ་དུ་བྱེད་དོ་ཞེས་པའི་དོན་ཏོ། །གདངས་ཅན་དབང་པོ་ནི་ཡེ་ཤེས་
