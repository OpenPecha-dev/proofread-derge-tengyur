མི་འགྲོ་བ་ནི་མ་ཡིན་ཏེ། དེ་བཞིན་གཤེགས་པས་བཀའ་སྩལ་པ་ཇི་ལྟ་བ་དེ་བཞིན་དུ། བདག་གིས་བརྗོད་པར་མི་ནུས་ཏེ། འོན་ཀྱང་རང་གི་བློ་དང་མཐུན་པར་བཤད་པར་བྱ། །ཞེས་བྱ་བ་ནི་སྐྱོན་མེད་དོ། །རྒྱུད་ཀྱི་རྒྱལ་པོ་ལ་ཟབ་པ་དང་རྒྱ་ཆེ་བ་བཤད་པར་མཛད་དེ། མ་ལུས་སྲིད་པ་བདག་ལུས་གང་དུ་
གསལ་བར་ནི། །ཞེས་བྱ་བ་ལ། གང་དུ་ཞེས་བྱ་བ་ནི་བཅོམ་ལྡན་འདས་དུས་ཀྱི་འཁོར་ལོ་སྟེ། མ་ལུས་སྲིད་པ་ཞེས་བྱ་བ་ནི་འདོད་པ་དང་། གཟུགས་དང་གཟུགས་མེད་པའི་ཕྱི་རོལ་དུ་ཇི་ལྟ་བར་བཀའ་སྩལ་པ་དེ་བཞིན་དུ། རང་གི་ལུས་པོ་ལ་བརྗོད་དོ། །གང་གིས་བརྗོད་ཅེ་ན། ཨཱ་ལི་ཐིག་
ལེ་འཛིན་པའི་རྒྱལ་བའི་དབང་པོས་ཞེས་བྱ་བ་ལ། ཨཱ་ལི་ཐིག་ལེ་ཞེས་བྱ་བ་ནི། བྱང་ཆུབ་སེམས་ཀྱི་ཐིག་ལེ་རྣམས་ཏེ། དེ་གང་གིས་འཛིན་པ་དེ་ཨཱ་ལིའི་ཐིག་ལེ་འཛིན་པ་རྒྱལ་བའི་དབང་པོ་ཞེས་བྱ་བ་ནི། རྡོ་རྗེ་སེམས་དཔའ་ཡིན་ལ་དེས་བཀའ་སྩལ་ཏོ། །དེ་བས་ན་རྒྱུད་ཀྱི་རྒྱལ་པོ་འདི་ཆུ་གཏེར་
ཞེས་བྱ་བ་ནི། རྒྱ་མཚོ་ཟབ་པ་དང་རྒྱས་པར་གྱུར་པ་དང་རིན་པོ་ཆེ་ལ་སོགས་པ་དང་། གདུག་པའི་སྲོག་ཆགས་རྣམས་ཀྱི་གནས་སུ་གྱུར་པ་ཇི་ལྟ་བ་དེ་བཞིན་དུ་དཔལ་དུས་ཀྱི་འཁོར་ལོའི་རྒྱུད་ཀྱི་རྒྱལ་པོ་དོན་དང་གཞུང་གིས་ཟབ་པ་དང་རྒྱ་ཆེ་བ་དང་། སངས་རྒྱས་ཀྱི་ཡོན་ཏན་རིན་པོ་ཆེ་རྣམས་དམ་
ཚིག་ཉམས་པའི་སེམས་ཅན་རྣམས་ལ་སྡུག་བསྔལ་གྱི་གནས་སུ་གྱུར་པས་ན་རྒྱ་མཚོ་དང་འདྲ་བའོ། །དེ་བས་ན་རྒྱ་མཚོ་དེ་ལས་བདག་གིས་སྒྲོལ་བར་འདོད་དེ། དེ་ལ་གལ་ཏེ་ཕ་རོལ་ཏུ་ཕྱིན་པར་གྱུར་ན་བཅོམ་ལྡན་འདས་ཤཱཀྱ་ཐུབ་པས་བྱིན་གྱིས་བརླབས་པའི་མཐུ་ཡིན་གྱི། བདག་གི་ཡོན་
ཏན་གྱི་སྟོབས་ཀྱིས་ནི་མ་ཡིན་ནོ། །དེ་ལྟར་རྒྱུད་ཀྱི་ཡོན་ཏན་བསྟན་པར་མཛད་ནས། རྩ་བའི་རྒྱུད་དང་བསྡུས་པའི་རྒྱུད་ཀྱི་ཡོན་ཏན་བསྟན་པར་མཛད་དེ། གསང་བདག་ལ་ནི་དང་པོའི་མཆོག་གི་སངས་རྒྱས་བཤད། །སྟོབས་བཅུ་པ་ཡིས་སྔོན་དུ་ལུང་བསྟན་བསྡུས་རྒྱུད་གང་། །ཞེས་བྱ་བ་ལ། 
གསང་བ་ཞེས་བྱ་བ་ནི་འཇིག་རྟེན་གྱི་ཐ་སྙད་ཀྱིས་གནོད་སྦྱིན་རྣམས་ཏེ། དེ་རྣམས་ཀྱི་གཙོ་བོར་གྱུར་པས་ན་གསང་བའི་བདག་པོ་ཞེས་བྱའོ། །ངེས་པའི་དོན་དུ་གསང་བ་ཞེས་བྱ་བ་ནི། སེམས་ཀྱི་ནང་དུ་མཐོང་བའི་ངོ་བོས་གནས་པར་གྱུར་པ་ཉོན་མོངས་པ་ལ་སོགས་པ་རྣམས་ཏེ། དེ་རྣམས་གང་གིས་
