ནཱ་བྷི་སནྡྷིཿ།།[ཞི་གནས་ལྷག་མཐོང་ལ་སོགས་པའི་མིང་ལ།] ཤ་མ་ཐཿ། བི་པ་ཤྱ་ནཱ། ཡོ་ག །ཡོ་ནི་ཤོ་མ་ནསྐཱ་རཿ།།[མི་དགེ་བ་བཅུའི་མིང་ལ།] ཏྲཱི་ཎི་ཀཱ་ཡ་དུཤྩ་རི་ཏཱ་ནི། ཙ་ཏྭཱ་རི་བཱགྔུཤྩ་རི་ཏཱ་ནི། ཏྲཱི་ཎི་མ་ནོ་དུཤྩ་རི་ཏཱ་ནི།།[དགེ་བ་བཅུའི་ལས་ཀྱི་མིང་ལ།] ཀཱ་ཡ་སུ་ཙ་རི་ཏམ྄། པྲཱ་ཎཱ་ཏི་གྷཱ་ཏཱ་ནཱ་དྭི་ར་ཏིཿ། ཨ་དཱཏྟཱ་དཱ་ནཱ་དྭི་ར་ཏིཿ། 
ལྡེམ་པོར་དགོངས་པ། ཞི་གནས། ལྷག་མཐོང་། རྣལ་འབྱོར། ཚུལ་བཞིན་ཡིད་ལ་བྱེད་པ། ལུས་ཀྱི་ཉེས་པར་སྤྱོད་པ་གསུམ། ངག་གི་ཉེས་པར་སྤྱོད་པ་བཞི། ཡིད་ཀྱི་ཉེས་པར་སྤྱོད་པ་གསུམ། ལུས་ཀྱིས་ལེགས་པར་སྤྱོད་པ། སྲོག་གཅོད་པ་སྤོང་བ། མ་བྱིན་པར་ལེན་པ་སྤོང་
ཀཱ་མ་མི་ཐྱཱ་ཙཱ་རཱ་དྭི་ར་ཏིཿ། བཱཀྶུ་ཙ་རི་ཏཾ། མྲྀ་ཥ་བཱ་དཱཏྤྲ་ཏི་བི་ར་ཏིཿ། པཱ་རུ་ཥཱཏྤྲ་ཏི་བི་ར་ཏིཿ། བ་ཤཱུ་ནྱཱཏྤཱ་ཏི་བི་ར་ཏིཿ། སཾ་བྷིནན་པྲ་ལཱ་པཱཔྟྲ་ཏི་བི་ར་ཏིཿ། མ་ན་སུ་ཙ་རི་ཏཾ། ཨ་བྷི་དྷྱཱ་ཡཱཿཔྲ་ཏི་བི་ར་ཏིཿ། བྱཱ་པཱ་དཱཏྲ་པྲ་ཏི་བི་ར་ཏིཿ མི་ཐྱཱ་དྲྀཥྚེཿཔྲ་ཏི་བི་ར་ཏིཿ།།[བསོད་ནམས་བྱ་བའི་དངོས་པོ་བཞིའི་མིང་ལ།] པུ་ཎྱ་ཀྲྀ་ 
བ། འདོད་པ་ལ་ལོག་པར་གཡེམ་པ་སྤོང་བ། ངག་གི་ལེགས་པར་སྤྱོད་པ། བརྫུན་དུ་སྨྲ་བ་སྤོང་བ། ཚིག་རྩུབ་པོར་སྨྲ་བ་སྤོང་བ། ཕྲ་མར་སྨྲ་བ་སྤོང་བ། ཚིག་བཀྱལ་སྤོང་བ། ཡིད་ཀྱིས་ལེགས་པར་སྤྱོད་པ། བརྣབ་སེམས་སྤོང་བ། གནོད་སེམས་སྤོང་བ། ལོག་པར་ལྟ་བ་སྤོང་བ། བསོད་ནམས་
ཡཱ་བསྟུཿ། དཱ་ན་མ་ཡཾ་པུཎྱ་ཀྲྀ་ཡཱ་བསྟུཿ། ཤཱི་ལ་མ་ཡཾ་པུཎྱ་ཀྲྀ་ཡཱ་བསྟུཿ། བྷཱ་བ་ནཱ་མ་ཡཾ་པུཎྱ་ཀྲྀ་ཡཱ་བསྟུཿ། ཨཽ་པ་དྷི་ཀཾ་པུཎྱ་ཀྲྀ་ཡཱ་བསྟུཿ། གུ་ཎཱཿ།། [དོན་དམ་པའི་མིང་ལ།] བ་ར་མརྠཿ། ཏ་ཏྭཾ། བྷཱུ་ཏ་ཀོ་ཊིཿ། ཏ་ཐ་
བྱ་བའི་དངོས་པོ། སྦྱིན་པ་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ། ཚུལ་ཁྲིམས་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ། བསྒོམས་པ་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ། རྫས་ལས་བྱུང་བའི་བསོད་ནམས་བྱ་བའི་དངོས་པོ། ཡོན་ཏན། དོན་དམ་པ། དེ་ཁོ་ན། ཡང་དག་པའི་མཐའ། དེ་བཞིན་
