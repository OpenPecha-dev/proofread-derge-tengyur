ལྷས་ཀྱང་དེ་སྣོད་མ་ཡིན་པ་ཙམ་དུ་རིག་ནས་བཅོས་ཏེ་བཏང་བ་དང་། དེ་ཚིག་པ་ཟ་ནས་ཕྲད་དགུ་ལ་བཟང་བའི་ལྷ་ལ་ཁ་ཕྱར་བཏང་ངོ་། །དེ་ཤི་ནས་བསྐལ་པ་དྲུག་ཅུར་ནལ་བུར་སྐྱེས་ཤིང་སྐྱེས་མ་ཐག་ཏུ་སྲང་གི་རྡོག་ལམ་དུ་བོར་ནས་ཝ་ཁྱི་ལ་སོགས་པས་
ཟ་བ་དང་། འགྲོ་བ་མང་པོས་ནལ་བུ་སྦྱར་མའི་བུ་ཞེས་ཁ་ཕྱ་བར་གྱུར་ཏོ། །བྱང་ཆུབ་ཀྱི་སེམས་ཀྱིས་བཟུང་བས་སེམས་ཅན་དམྱལ་བར་མ་སྐྱེས་སོ། །བཟང་བའི་ལྷ་ནི་དེ་བཞིན་གཤེགས་པ་མི་སྒུལ་བའི་སངས་རྒྱས་ཀྱི་ཞིང་ན་ཚངས་པར་སྤྱད་པ་སྤྱོད་དེ། བྱང་ཆུབ་སེམས་
དཔའ་དཔལ་འབྱུང་ཞེས་བྱའོ། །སྣང་བ་བསམ་དུ་མེད་པས་ཀྱང་སྔོན་སངས་རྒྱས་བྱེ་བ་ཕྲག་དགུ་བཅུ་རྩ་བཞིའི་དྲུང་དུ་ཚངས་པར་སྤྱད་ཅིང་མཉེས་པར་བྱས་པས། དེ་ལྟ་བུའི་བདག་ཉིད་ཆེན་པོར་གྱུར་ཏོ། །དེ་བས་ན་སྲོག་དང་ཡང་བསྡོས་ཏེ། སྡིག་པའི་ལུས་དང་ངག་དང་ཡིད་ཀྱི་ལས་
བསྡམ་མོ། །དགེ་བའི་བཤེས་གཉེན་ལ་བརྟན་པར་བྱ་སྟེ། དེ་ལས་ཡོན་ཏན་གྱི་ཚོགས་ཐམས་ཅད་འགྲུབ་པར་འགྱུར་རོ། །དེའི་ཚེ་བུདྡྷ་ཛེ་ཏ་ན་བཞུགས། བུདྡྷ་བསོད་སྙོམས་ལ་གཤེགས་པའི་ཚེ། ཁྱེའུ་ཡང་ཛེ་ཏའི་ཚོང་འདུས་ཀྱི་སྲང་གི་ཁང་སྟོང་ཞིག་ན་གཞག་པ་གཟིགས་སོ། །
ཁྱེའུ་ཡང་བསྐལ་པ་གྲངས་མེད་པ་བརྒྱ་ན་སྣང་བ་བསམ་དུ་མེད་པའི་བུདྡྷར་འགྱུར་བར་ལུང་བསྟན་ཏོ། །རིན་པོ་ཆེའི་ཕུང་པོ་ལས། དགེ་སྦྱོང་གི་དྲི་མ་ནི་སུམ་ཅུ་རྩ་གཉིས་ཏེ། གང་ཞེ་ན། འདོད་པའི་ལྷ་རྣམ་པར་རྟོག་པ་དང་། གནོད་སེམས་དང་། འཚེ་བ་དང་། བདག་བསྟོད་པ་
དང་། ཕ་རོལ་ལ་བརྙས་པ་དང་། ཡ་ང་བའི་རྙེད་པ་ཚོལ་བ་དང་དྲི་མ་ནི་རྙེད་པས་རྙེད་པ་འདོད་པ་དང་། དད་པས་བྱིན་པ་ཆུད་གཟོན་པ་དང་། ལྟུང་བ་འཆབ་པ་དང་། ཁྱིམ་པ་ལ་བརྟེན་པ་དང་། རབ་ཏུ་བྱུང་བ་དང་བརྟེན་པ་དང་། འདུ་འཛི་ལ་དགའ་བ་དང་། རྙེད་པ་ཡོངས་སུ་མ་རྫོགས་པར་
རྟོག་པ་དང་། ཕ་རོལ་གྱི་རྙེད་པ་ལ་འདོད་པ་དང་། བདག་གི་རྙེད་པས་ཆོག་མི་ཤེས་པ་དང་། ཕ་རོལ་བསོད་ནམས་རྙེད་པ་ལ་ཕྲག་དོག་དང་། སེར་སྣ་བྱེད་པ་དང་། ཕ་རོལ་གྱི་འཁྲུལ་པ་ཚོལ་བ་དང་། བདག་གི་འཁྲུལ་པ་ལ་སོ་སོར་ཐར་པ་དང་། འདུལ་བས་མ་བསྡམས་པ་དང་།
