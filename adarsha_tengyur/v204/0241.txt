ཡཱ་གླཱ་ནཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ནོཥྞཱི་ཥ་ཤི་ར་སེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཁཽ་ལཱ་ཤིར་སེ་དྷརྨན྄་དེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་མཽ་ལི་ཤི་ར་སེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་བེཥྚི་ཏ་ཤི་ར་སེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཧ་སྟྱཱ་རཱུ་ཌྷཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ནཱ་ཤྭ་རཱུ་ཌྷཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཤི་བི་ཀཱ་
ཕྲག་པ་གཉིས་ཀ་ལ་གཟར་བ་ལ་ཆོས་མི་བཤད། སྐྲ་དོ་ཀེར་ཅན་ལ་ཆོས་མི་བཤད། ཞྭ་གྱོན་པ་ལ་ཆོས་མི་བཤད། མགོ་ཅོད་པན་ཅན་ལ་ཆོས་མི་བཤད། མགོ་དཀྲིས་པ་ལ་ཆོས་མི་བཤད། གླང་པོ་ཆེ་ལ་ཞོན་པ་ལ་ཆོས་མི་བཤད། རྟ་ལ་ཞོན་པ་ལ་ཆོས་མི་བཤད། ཁྱོགས་
རཱུ་ཌྷཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཡཱ་ནཱ་རཱུ་ཌྷཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་པཱ་ད་ཀཱ་རཱུ་ཌྷཱ་ཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་དཎྜ་པཱ་ཎ་ཡེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ནཙྪ་ཏྲ་པཱ་ཎ་ཡེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཤ་སྟྲ་པཱ་ཎ་ཡེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་ཁཌྒ་པཱ་ཎ་ཡེ་དྷརྨནྡེ་ཤ་ཡི་
ན་འདུག་པ་ལ་ཆོས་མི་བཤད། བཞོན་པའི་སྟེང་ན་འདུག་པ་ལ་ཆོས་མི་བཤད། ལྷམ་མཐོན་པོ་གྱོན་པ་ལ་ཆོས་མི་བཤད། ལག་ན་འཁར་བ་ཐོགས་པ་ལ་ཆོས་མི་བཤད། ལག་ན་གདུགས་ཐོགས་པ་ལ་ཆོས་མི་བཤད། ལག་ན་མཚོན་ཐོགས་པ་ལ་ཆོས་མི་བཤད། ལག་ན་རལ་གྲི་ཐོགས་པ་ལ་
ཥྱཱ་མཿ། ནཱ་ཡུ་དྷ་པཱ་ཎ་ཡེ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ན་སནྣདྡྷཱཡ་དྷརྨནྡེ་ཤ་ཡི་ཥྱཱ་མཿ། ནྟ་གླཱ་ན་ཨུཏྠི་ཏ་ཨུཙྕཱ་ར་པྲ་སྲཱ་བཾ་ཀ་རི་ཥྱཱ་མཿ། ནཱ་གླཱ་ན་ཨུ་ད་ཀ་ཨུཙྩཱ་ར་པྲ་སྲཱ་བཾ་ཁེ་ཊཾ་སི་གྷྲཱ་ཎ་ཀཾ་བཱནྟཾ་བི་རིཀྟཾ་ཙོ་ར་ཡི་ཥྱཱ་མཿ། ནཱ་གླཱ་ནཱ་ས་ཧ་རི་ཏ་པྲྀ་ཐི་བཱི་པྲ་དེ་ཤེ་ཨུཙྩཱ་ར་བྲ་སྲཱ་བཾ་ཁེ་ཊཾ་སིཾ་གྷཱ་ཎ་ཀཾ་
ཆོས་མི་བཤད། ལག་ན་དགྲ་ཆ་ཐོགས་པ་ལ་ཆོས་མི་བཤད། གོ་ཆ་གྱོན་པ་ལ་ཆོས་མི་བཤད། མི་ན་བར་འགྲེང་སྟེ་གཤང་གཅི་མི་བྱ། མི་ན་བར་ཆུའི་ནང་དུ་གཤང་གཅི་དང་མཆིལ་མ་དང་སྣབས་དང་སྐྱུགས་པ་དང་ཁ་ནས་བྱུང་བ་མི་དོར། མི་ན་བར་རྩྭ་སྔོན་པོ་ཡོད་པའི་ཕྱོགས་སུ་གཤང་གཅི་དང་མཆིལ་མ་
