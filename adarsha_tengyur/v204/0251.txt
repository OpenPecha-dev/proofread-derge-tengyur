ཨ་ནུ་པཱ་ར་པཊྚ་ཀཾ། ཨ་ཡ་སྦིཎྜཾ། ཤྲྀཾ་ཁ་ལི་ཀཱ། ནེེ་ཏྲི་ཀཾ། ནསྟཿཀ་ར་ཎཾ། ཀ་ཋི་ནཾ། ཀཱནྟཱ་རི་ཀཱ། ཙཪྤ་ཊ་ཀཾ། ཀུ་ཋཱ་རི་ཀཱ། ཨཉྫཱ་ན་ཤ་ལཱ་ཀཱ། ཙཱི་བ་ར་བཾ་ཤཿ། ཁ་ཊྭཱ། མཉྩཿ། པེ་ཋི་ཀཱ། པྲ་ཏི་པཱ་ད་ཀཾ། ཨ་ཛཱ་པ་ད་ཀ་དཎྜཿ། ནཱ་ག་དནྟ་ཀཾ། དརྦི་ཀཱ། སྤི་ཛཾ།
ཕུལ་བའི་བཏུང་ཕོང་། ལྕགས་ཀྱི་གར་བུ། ལྕགས་ཐག །སྨན་གཅེའུ། སྣར་བླུགས་པའི་སྤྱད། སྲ་བརྐྱང་། གཟར་ཐག །དབང་བུ། སྟ་རེ། བསྐུ་བའི་ཐུར་མ། གོས་ཀྱི་གདང་། ཁྲི། ཁྲིའུ། སྐྲའི་རྐང་རྟེན། དབྱུག་གུ་སྐམ་ཁ། ཟུང་ང་། གཟར་བུ། ཐལ་ཕྱག ། 
ཀརྨཱ་ར་བྷཎྜི་ཀཱ། ནཱ་པི་ཏ་བྷཎྜཾ། གྷ་ཊ་བྷེ་ད་ན་ཀཾ། ཀརྤ་རཿ། ཨུ་ད་ཀ་བྷཱཎྜཱ་དྷིཥྛཱ་ནཾ། པཱ་དཱ་དྷིཥྛཱ་ནཾ། [གང་ཟག་བཅུ་གཉིས་ཀྱི་མིང་ལ།] བི་ཧཱ་རོདྡེ་ཤ་ཀཿ། བྷ་ཀྵོདྡེ་ཤ་ཀཱ། ཡ་བཱ་གཱུ་ཙཱ་ར་ཀཿ། ཁཱ་དྱ་ཀ་ཙཱ་ར་ཀཿ། ཕ་ལ་ཙཱ་ར་ཀཿ། ཡཏྐིཉྩི་ཙྩཱ་ར་ཀཿ། བྷཱཎྜ་གོ་བ་ཀཿ། བྷཱཎྜ་བྷཱ་ཛཱ་ཀཿ། བཪྵཱ་ཤཱ་ཊཱི་
མགར་སྤྱད། འདྲེག་མཁན་གྱི་ལག་ཆ། རྫ་གཟོང་། དབྱིབ་མོའམ་གྱོ་མོ། ཆུ་སྣོད་ཀྱི་གཞི། རྐང་རྟེན། གནས་ཁང་བསྐོ་བ། ཟས་ལ་བསྐོ་བ། ཐུག་པ་འགྲིམ་པ། བག་ཆོས་འགྲིམ་པ། ཤིང་ཐོག་འགྲིམ་པ། ཕྲན་ཚེགས་འགྲིམ་པ། སྣོད་སྤྱད་འདྲུབ་པ། སྣོད་སྤྱད་འགེད་པ། དབྱར་
གོ་པ་ཀཿ། ཙཱི་བ་ར་གོ་པ་ཀཿ། ཙཱི་བ་ར་བྷཱ་ཛཱ་ཀཿ། ཨུ་བ་དྷི་བཱ་རི་ཀཿ། པྲེ་ཥ་ཀཿ། བྷཱ་ཛཱ་ན་བཱ་རི་ཀཿ། པཱ་ནཱི་ཡ་བཱ་རི་ཀཿ། པྲཱ་སཱ་དི་བཱ་རི་ཀཿ། བ་རི་ཥཎྜ་བཱ་རི་ཀཿ། ཤ་ཡ་ནཱ་ས་ན་བཱ་རི་ཀཿ། མུཎྜ་ཤ་ཡ་ནཱ་ས་ན་བཱ་རི་ཀཿ། ཙྪཎྜི་ཀཱ་བཱ་རི་ཀཿ། [སྡེ་བཞི་རྣམ་པ་བཅོ་བརྒྱད་དུ་གྱེས་པའི་མིང་ལ།] ཨཱཪྻ་སརྦཱ་
གྱི་རས་ཆེན་སྦེད་པ། གོས་སྦེད་པ། གོས་འགེད་པ། དགེ་སྐོས། མངག་གཞུག་པ། སྡེར་སྤྱད་ཀྱི་ཞལ་ཏ་པ། ཆུའི་ཞལ་ཏ་པ། མཛེས་འཆོས། བང་རིམ་གྱི་ཞལ་ཏ་པ། མལ་སྟན་གྱི་ཞལ་ཏ་པ། མལ་སྟན་གྱི་ཕྱོགས་ཀྱི་ཞལ་ཏ་པ། སྒེའུ་ཆུང་བསྲུང་བ། འཕགས་པ་ཐམས་ཅད་
