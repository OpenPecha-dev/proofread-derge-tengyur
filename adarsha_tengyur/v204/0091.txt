ཛྙི་ས་ཏྭཱ།།[མི་ཁོམ་པ་བརྒྱད་ཀྱི་མིང་ལ།] ཨཥྚཱ་བཀྵ་ཎཱཿ། ན་ར་ཀཱ། ཏིཪྻཉྩཿ། པྲེ་ཏཱ། དཱིརྒྷཱ་ཡུ་ཥོ་དེ་བཱ། པྲ་ཏྱནྟ་ཛ་ན་པ་དཾ། ཨིནྡྲི་ཡ་བཻ་ཀ་ལྱཾ། མེ་ཐྱཱ་དཪྴ་ནཾ། ཏ་ཐཱ་ག་ཏཱ་ནཱ་མ་ནུཏྤཱ་དཾ།།[ལས་ཀྱི་རྣམ་པར་སྨིན་པ་ལ་སོགས་པའི་མིང་ལ།] དྲྀཥྚི་དྷརྨ་བེ་ད་ནཱི་ཡཾ། ཨུ་པ་
པའི་སེམས་ཅན། མི་ཁོམ་བརྒྱད། སེམས་ཅན་དམྱལ་བ། དུད་འགྲོ། ཡི་དྭགས། ལྷ་ཚེ་རིང་པོ། མཐའ་འཁོབ་ཀྱི་མི། དབང་པོ་མ་ཚང་བ། ལོག་པར་ལྟ་བ། དེ་བཞིན་གཤེགས་པ་མ་བྱུང་བ། མཐོང་བའི་ཆོས་ལ་མྱོང་བར་འགྱུར་བ། སྐྱེས་
པ་དྱ་བེ་ད་ནཱི་ཡཾ། ཨ་པ་ར་བཪྻཱ་ཡ་བེ་ད་ནཱི་ཡཾ། ནི་ཡ་ཏ་བེ་ད་ནཱི་ཡཾ།  ཨ་ནི་ཡ་ཏ་བེ་ད་ནཱི་ཡཾ། ཀརྨ་སྭ་ཀཿ། ཀརྨ་དཱ་ཡ་དཿ། ཀརྨྨ་ཡོ་ནིཿ། ཀརྨ་པྲ་ཏི་ས་ར་ཎམ྄། པྲ་ཡོ་གཿ། མོ་ལཾ། པྲྀཥྚཾ། ཨཱ་སེ་བི་ཏཾ། བྷཱ་
ནས་མྱོང་བར་འགྱུར་བ། ལན་གྲངས་གཞན་ལ་མོང་བར་འགྱུར་བ། མྱོང་བར་འགྱུར་བར་ངེས་པ། མྱོང་བར་འགྱུར་བར་མ་ངེས་པ། ལས་བདག་གིར་བྱ་བ། ལས་ཀྱི་བགོ་སྐལ་སྤྱོད་པ། ལས་ཀྱི་སྐྱེ་གནས་པ། ལས་བརྟེན་པར་བྱ་བ། སྦྱོར་བ། དངོས་གཞི། འཇུག །བསྟེན་པའམ་བསྙེན་པ། བསྒོམ་པའམ་གོམས་
བི་ཏཾ། བ་ཧུ་ལཱི་ཀྲྀ་ཏཾ།།[མཚམས་མེད་ལྔའི་མིང་ལ།] མཱ་ཏྲྀ་གྷཱ་ཏཿ། ཨརྷདྦ་དྷཿ། པི་ཏྲྀ་གྷཱ་ཏཿ། སདྒྷ་བྷེ་དཿ། ཏ་ཐཱ་ག་ཏ་སྱཱནྟི་ཀེ་དུཥྚ་ཙིཏྟ་རུ་དྷི་རོཏྤཱ་ད་ནཾ།།[མཚམས་མེད་པ་དང་ཉེ་བ་ལྔའི་མིང་ལ།] མཱ་ཏུ་ར་རྷཾ་ནྟྱཱ་དུ་ཥ་ཎཾ། ནི་ཡ་ཏ་བྷུ་མི་སྠི་ཏ་སྱ་བོ་དྷི་ས་ཏྭ་སྱ་མཱ་ར་ཎཾ། 
པར་བྱས་པ། ལན་མང་དུ་བྱས་པ། མ་བསད་པ། དགྲ་བཅོམ་པ་བསད་པ། ཕ་བསད་པ། དགེ་འདུན་གྱི་དབྱེན་བྱས་པ། དེ་བཞིན་གཤེགས་པ་ལ་ངན་སེམས་ཀྱིས་ཁྲག་ཕྱུང་བ། མ་དགྲ་བཅོམ་མ་སུན་ཕྱུང་བ། བྱང་ཆུབ་སེམས་དཔའ་ངེས་པའི་ས་ལ་གནས་པ་བསད་པ།
