པ་ལ། །རྟག་པར་དགྱེས་པ་ཉིད་དུ་མཛད། །ཞམ་རང་བློན་པོའི་པུ་ཚུགས་ནི། །ཡོན་ཏན་ནོར་གྱི་གཉེར་ལ་གཉེན། །མཁས་ཤིང་སྙིང་རྗེའི་བདག་ཉིད་ཅན། །བྱས་གཟོ་ཕོངས་ཆོས་མི་བགྱིད་པ། །བཟང་པོ་ཁྱོད་ཀྱི་སླད་དུ་ཞོག །མི་གང་དཀའ་བ་སྤྱོད་བྱེད་དང་། །དོར་བར་བྱ་དཀའ་འདོར་བ་
དང་། །བཟོད་པར་དཀའ་བ་བཟོད་བྱེད་དང་། །བཀུར་དཀའ་ཉམས་སུ་ལེན་པ་དང་། །རིང་པོར་བལྟ་ཞིང་ངེས་པ་དང་། །འཕྲལ་བློ་མུར་ཞིང་སྡུག་པ་དང་། །ཁོང་སྲན་ཆེ་ཞིང་དཔའ་བ་དང་། །བསླུ་བར་མི་ཚུགས་གསང་ཐུབ་དང་། །ཆོས་ཀྱིས་རྒྱལ་སྲིད་སྐྱོང་དགའ་དང་། །དམ་ཆོས་མཛད་ལ་མི་འཁོར་
བ། །རིན་གྱིས་འཇལ་ཡང་མི་ཕམ་གྱིས། །ཐུགས་བརྟན་མཛོད་ལ་སྤྱན་སྔར་ཞོག །ཁྱོད་ཀྱིས་བླུན་དང་ཕོངས་ཆོས་ཅན། །གཞུང་ངན་གདོང་གང་བྱེད་པ་དང་། །བརྐམ་ཞིང་བྱས་པ་མི་གཟོ་དང་། །ཁ་གཅམ་གཏུམ་པོ་མི་བཟང་བ། །ཡུལ་ནའང་གནས་སུ་མ་སྩལ་ཅིག །གལ་ཏེ་གཞུང་ལུགས་
བཟང་དགྱེས་ན། །གོང་ནས་གོང་དུ་འབར་བར་འགྱུར། །གལ་ཏེ་ངན་པ་སྤྱོད་མཛད་ན། །འོག་ནས་འོག་ཏུ་ལྟུང་བར་འགྱུར། །ཁྱུ་མཆོག་དང་ནི་ཁྱུས་འགྲོ་བཞིན། །ཁྱོད་ནི་གྱ་གྱུ་ལ་ཞུགས་ན། །གྱ་གྱུ་ཁོ་ནའི་རྗེས་འབྲངས་ཏེ། །སྐྱེ་དགུ་ཁྱོད་ཀྱི་རྗེས་འབྲངས་པས། །རྒྱལ་ཁམས་ངན་པར་འོང་
འགྱུར་རོ། །ཟླ་བ་མར་གྱི་ངོ་བཞིན་དུ། །ཉེས་པའི་ཚོགས་རྣམས་འགྲིབ་པར་མཛོད། །ཟླ་བ་ཡར་གྱི་ངོ་བཞིན་དུ། །རྒྱལ་ཁམས་ལེགས་པར་རྒྱས་པར་མཛོད། །དམ་ཆོས་བསྐྱང་བའི་ཁྲིམས་རྣམས་མཛོད། །དམ་ཆོས་འཇིག་པའི་སྐྱེ་བོ་དང་། །བཀའ་ཁྲིམས་འདྲལ་བའི་ངན་པ་རྣམས། །ལོ་ཐོག་
རྩི་ཐོག་རྒྱས་པ་ལ། །ཐོག་དང་སེར་བ་བཞིན་དུ་ཕོབ། །འོན་དང་ངན་མིག་འགག་པར་འགྱུར། །ལེ་ལོ་ཉམ་བུ་ཤེད་མ་མཛད། །འཆི་བདག་སུ་དང་མི་བཤེས་པ། །གློ་བུར་དག་ཏུ་འབབ་འགྱུར་བས། །སང་དག་བྱ་ཞེས་མ་གཞེས་པར། །དམ་པའི་ཆོས་ལ་སྨྱུར་ཏེ་མཛོད། །འཆི་བདག་གིས་ནི་
དྲངས་པ་ན། །དགེ་དང་སྡིག་པ་མ་གཏོགས་པ། །འགྲོ་བ་ཐམས་ཅད་ཕྱིར་ལོག་ནས། །འགའ་ཡང་ཁྱོད་རྗེས་མི་འབྲང་བས། །མཁྱེན་པར་མཛོད་ལ་ལེགས་པར་སྤྱོད། །སློབ་དཔོན་ཆེན་པོ་ཀླུ་སྒྲུབ་ཀྱིས། །རྒྱལ་པོ་བདེ་སྤྱོད་བཟང་པོ་ལ། །བརྫངས་པའི་འཕྲིན་ཡིག་དག་ལས་ཀྱང་། །
