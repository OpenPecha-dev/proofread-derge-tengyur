ནིཥྞཱ་ཏཿ། བྱཀྟཿ། མེ་དྷཱ་བཱི། བུདྡྷཿ། བུདྡྷི་མཱན྄། པྲ་ཛྙཱ། བོདྡྷཱ། པྲ་ཛྙཱ་བཱན྄། བིདྦན྄། ནི་པུ་ཎཿ། བི་ཛྙཿ། ཨཱ་ཀྲྀཥྚི་མཱན྄། པ་ཊུཿ། ཙ་ཏུ་ར་སྲཿ། དཀྵཿ།།[ཟབ་པའི་རྣམ་གྲངས་ཀྱི་མིང་ལ།] 
བྱང་བ། གསལ་བ། ཡིད་གཞུངས་པ། རྟོགས་པ། བློ་དང་ལྡན། ཤེས་རབ་ཅན། ཁོང་དུ་ཆུད་པ། ཤེས་རབ་དང་ལྡན་པ། མཁས་པའམ་རིག་པ། བརྟགས་པའམ་རྟོགས་པའམ་མཁས་པ། མཛངས་པའམ་ཤེས་པ། ལྡོག་རྐྱེན་པ། གཡེར་པོ། གྲིམས་པ། མཁས་པའམ་ཤེས་ཉེན་ཅན། 
གམྦྷཱི་རོ་གམྦཱི་རཱ་བ་བྷཱ་སཿ། དུརྡྡྲྀ་ཤཿ། དུ་ར་བ་བོ་དྷཿ། དུར་ནུ་བོ་དྷཿ། སཱུཀྵྨ་ནི་པུ་ཎ་པཎྜི་ཏ་བི་ཛྙ་བེ་ད་ནཱི་ཡཿ། ཨ་ཏརྐྱཿ། ཨ་ཏརྐཱ་བ་ཙ་རཿ། ཨ་ནི་དཪྴ་ནཿ། ཨ་བ་གཱ་ཧཿ། ཤི་བཿ། ཨ་པྲ་པཉྩཿ། ནིཿ
ཟབ་པ་ཟབ་པར་སྣང་བ། མཐོང་བར་དཀའ་བ། ཁོང་དུ་ཆུད་པར་དཀའ་བ། ཁོང་དུ་ཆུད་པར་དཀའ་བ། ཞིབ་མོར་བརྟགས་པའི་མཁས་པ་དང་མཛངས་པས་རིག་པ། བརྟག་པར་མི་ནུས་པ། རྟོག་གེའི་སྤྱོད་པ་མ་ཡིན་པ། བསྟན་དུ་མེད་པ། གཏིང་དཔག་དཀའ་བ། ཞི་བ། སྤྲོས་པ་མེད་པ། སྤྲོས་པ་
པྲ་པཉྩཿ། ཨ་བི་པྲ་པཉྩཿ། དུརྦི་གཱ་ཧྱཾ།།[དགའ་བའི་རྣམ་གྲངས་ཀྱི་མིང་ལ།] ཏུཥྚཿ། ཨུ་ད་གྲཿ། ཨཱཏྟ་མ་ནཱཿ། པྲ་མུ་དི་ཏཿ། པྲཱི་ཏི་སོོ་མ་ན་སྱ་ཛཱ་ཏཿ། ཧཪྵ་ཛཱ་ཏཿ། ཧྲྀ་ཥི་ཏ་ཙིཏྟཿ། པྲ་མོ་དྱམ྄། སནྟུཥྚཿ། པ་རི་ཏོ་ཥཿ། ཨཽདྡྷི་ལྱ་
