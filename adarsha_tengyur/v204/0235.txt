ཨ་བ་དྷྱཱ་ནཾ། ཨཱ་ཛྙཱ་བི་ཧེ་ཋ་ནཾ། {མཎྩཿ}། སཾ་སྟ་རཿ། ནིཥྐཪྵ་ཎཾ། ཨ་ནུ་པྲཥནྡྱཱཔཱ་ཏཿ། ཨཱ་ཧཱཪྻ་བཱ་ད་ཀཱ་རོ་ཧཱིཿ། ས་པྲཱ་ཎི་ཀོ་པ་བྷོ་གཿ། དཽ་བཱ་ཏྲ་ཡོ་བཱཙྪ་ད་ན་པ་མཪྻཱ་དཱ་ཏ་བྱཱཿ། ཏྲྀ་ཏཱི་ཡཾ་ད་ཤ་ཀཾ། ཨ་སམྨ་ཏཱ་བ་བཱ་དཿ། ཨསྟ་མི་ཏཱ་བ་བཱ་དཿ། ཨཱ་མི་ཥ་
འཇིག་པ། འཕྱ་བ། བསྒོ་བ་བཅག་པ། ཁྲི། གདིང་བ། སྐྲོད་པ། ཕྱིས་གནོན་པ། རྩ་བ་འབྱུང་བ་ལ་འདུག་པ། སྲོག་ཆགས་དང་བཅས་པ་ལ་སྤྱོད་པ། རིམ་པ་གཉིས་སམ་གསུམ་བརྩིག་པ། བཅུ་པ་གསུམ་པ་ལ། མ་བསྐོས་པར་སྟོན་པ། ཉི་མ་ནུབ་ཀྱི་བར་དུ་སྟོན་པ། ཟས་ཅུང་
ཀིཉྩི་ཀཱ་བ་བཱ་དཿ། ཙཱི་བ་ར་དཱ་ནཾ། ཙཱི་བ་ར་ཀ་ར་ཎཾ། བྷིཀྵུ་ཎཱི་སཱརྠེ་ན་ས་ཧ་ག་མ་ནཾ། ས་བྷི་ཀྵུ་ཎཱི་ཀཱ་ཛ་ལཱ་ཡཱ་ནོ་ཊིཿ། ར་ཧ་སི་ནི་ཥ་དྱཱ། ར་ཧ་སི་སྠཱ་ནཾ། བྷིཀྵུ་ཎཱི་པ་རི་པཱ་ཙི་ཏ་པིཎྜ་པ་ཏྲོ་པ་བྷོ་གཿ། ཙ་ཏུརྠཾ་ད་ཤ་ཀཾ། པ་རམྤ་ར་བྷོ་ཛཱ་ནཾ། ཨེ་ཀཱ་བ་ས་ཐཱ་བཱ་སཿ། 
ཟད་ཙམ་གྱི་ཕྱིར་སྟོན་པ། གོས་སྦྱིན་པ། གོས་འཚེམ་པ། དགེ་སློང་མའི་མགྲོན་པོ་མང་པོ་དང་ལྷན་ཅིག་འགྲོ་བ། དགེ་སློང་མ་དང་ལྷན་ཅིག་གྲུར་འཇུག་པ། དབེན་པ་ན་འདུག་པ། དབེན་པ་ན་འགྲེང་བ། དགེ་སློང་མས་སྦྱོར་དུ་བཅུག་པའི་ཟས་ཟ་བ། བཅུ་པ་བཞི་པ་ལ། ཡང་ཡང་ཟ་བ། གནས་གཅིག་ན་གནས་པ། 
དྭི་ཏྲི་པཱ་ཏྲ་པཱུ་རཱ་ཏི་རིཀྟ་གྲ་ཧ་ཎཾ། ཨ་ཀྲྀ་ཏ་ནི་རིཀྟ་ཁཱ་ད་ནཾ། ཨ་ཀྲྀ་ཏ་ནི་རྀཀྟ་པྲ་བཱ་ར་ཎཾ། གཎ་བྷི་ཛཱ་ནཾ། ཨ་ཀཱ་ལ་བྷོ་ཛཱ་ནཾ། སནྣི་ཧི་ཏ་བརྫ་ནཾ། ཨ་པྲ་ཏི་གྲཱ་ཧི་ཀ་བྷུཀྟིཿ། པཎཱི་ཏ་བི་ཛྙཱ་པ་ནམ྄། ས་པྲཱ་ཎཱི་ཛ་ལོ་པ་བྷོ་གམ྄། ས་བྷོ་ཛཱ་ན་ཀུ་ལ་ནི་
ལྷུང་བཟེད་དོ་གསུམ་ལས་ལྷག་པར་ལེན་པ། ལྷག་མར་མ་བྱས་པའི་བཟའ་བ། ལྷག་མར་མ་བྱས་པ་སྟོབ་པ། འདུས་ཤིང་ཟ་བ། དུས་མ་ཡིན་པར་ཟ་བ་དང་། གསོག་འཇོག་སྤང་བ། སྦྱིན་ལེན་མ་བྱས་པར་ཟ་བ། ཟས་གསོད་པ་སློང་བ། སྲོག་ཆགས་དང་བཅས་པའི་ཆུ་ལ་སྐྱོད་པ། ཉལ་པོ་བྱེད་པོ་བཤོམས་པའི་
