ཨི་ཏི་ལོ་ཀཿ། ལོ་ཀཱནྟ་རི་ཀཿ། སརྦ་ལོ་ཀ་དྷཱ་ཏུ་པྲ་ས་རཿ། བུདྡྷ་བི་ཥ་ཡཿ། བུདྡྷ་ཀྵེ་ཏྲཾ། ས་ཧཱ་ལོ་ཀ་དྷཱ་ཏུཿ། ཨ་བ་མཱུརྡྷཿ། བྱ་བ་ཏསྟཿ། ཏིཪྻགློ་ཀ་དྷཱ་ཏུཿ།།[ཁམས་གསུམ་པའི་རིམ་པའི་མིང་ལ།] ཀཱ་མ་དྷཱ་ཏུཿ། རཱུ་པ་དྷཱ་ཏུཿ། 
པས་ན་འཇིག་རྟེན། འཇིག་རྟེན་གྱི་བར། འཇིག་རྟེན་གྱི་ཁམས་རབ་འབྱམས་ཐམས་ཅད། སངས་རྒྱས་ཀྱི་ཡུལ། སངས་རྒྱས་ཀྱི་ཞིང་། མི་མཇེད་ཀྱི་འཇིག་རྟེན་གྱི་ཁམས། སྤྱིའུ་ཚུགས། སྣྲེལ་གཞི། ཐད་ཀ་བའི་འཇིག་རྟེན་གྱི་ཁམས། འདོད་པའི་ཁམས། གཟུགས་ཀྱི་ཁམས། 
ཨ་རཱུ་པ་དྷཱ་ཏུཿ།[འདོད་པའི་ཁམས་ཀྱི་ལྷའི་མིང་ལ།] བྷཽ་མཱཿ། ཨནྟ་རི་ཀྵ་བཱ་སི་ནཿ། ཙ་ཏུརྨ་ཧཱ་རཱ་ཛ་ཀཱ་ཡི་ཀཱཿ། ཏྲ་ཡ་སྟིཾ་ཤཱཿ། ཡཱ་མཱཿ། ཏུ་ཥི་ཏཱཿ། ནིརྨཱ་ཎ་ར་ཏ་ཡཿ། པ་ར་ནིརྨི་ཏ་བ་ཤ་བརྟི་ནཿ།།[བསམ་གཏན་དང་པོའི་སའི་མིང་ལ།] བྲཧྨ་ཀཱ་ཡི་ཀཱཿ། བྲཧྨ་པཱ་རི་ཥ་དྱཱཿ། བྲཧྨ་པུ་རོ་ཧི་ཏཱཿ། མ་ཧཱ་ 
གཟུགས་མེད་པའི་ཁམས། ས་བླ། བར་སྣང་ལ་གནས་པ། རྒྱལ་ཆེན་བཞིའི་རིས། སུམ་ཅུ་རྩ་གསུམ་པ། འཐབ་བྲལ། དགའ་ལྡན། འཕྲུལ་དགའ། གཞན་འཕྲུལ་དབང་བྱེད། ཚངས་རིས། ཚངས་འཁོར། ཚངས་པ་མདུན་ན་འདོན། ཚངས་
བྲཧྨ་ཎཿ།།[བསམ་གཏན་གཉིས་པའི་སའི་མིང་ལ།] པ་རཱིཏྟཱ་བྷཱཿ། ཨ་པྲ་མཱ་ཎཱ་བྷཱཿ། ཨ་བྷ་སྭ་རཱཿ།།[བསམ་གཏན་གསུམ་པའི་སའི་མིང་ལ།] པ་རཱི་ཏ་ཤུ་བྷཱཿ། ཨ་པྲ་མཱ་ཎ་ཤུ་བྷཱཿ། ཤུ་བྷ་ཀྲྀཏྶྣཱཿ།། [བསམ་གཏན་བཞི་པའི་སའི་མིང་ལ།] ཨ་ན་བྷྲ་ཀཱཿ། པུ་ཎྱ་པྲ་ས་བཱཿ། བྲྀ་ཧཏྥ་ལཿ།།[གནས་གཙང་མའི་སའི་མིང་ལ།] ཨ་བྲྀ་ཧཱཿ། ཨ་ཏ་པཱཿ། སུ་དྲྀ་ཤཱཿ། སུ་དཪྴ་
