མེ་ཏོག་གིས་རྒྱན་གྱི་ལས་བྱེད། ལོ་མས་ནི་དཀྲི་བའི་ལས་བྱེད། འབྲས་བུས་ནི་འགྲངས་པའི་ལས་བྱེད་པ་དང་། འཕགས་པའི་ཆོས་ལ་བདག་གི་སའི་བྱེ་བྲག་གིས་སེམས་ཅན་ལ་ཕན་འདོགས་པའང་ཐབས་ཀྱི་ཐ་དད་པ་ནས་སོ་སོའི་ལམ་ལ་གཞོག་པ་ལ་སོགས་པ་ཆོས་གང་གི་རྒྱུར་ངེས་པ་
དེའི་འབྲས་བུ་སོ་སོར་སྐྱེད་པ་དང་། དབང་པོ་དྲུག་ཤེས་པ་དང་བཅས་པ་ཡུལ་དུ་ངེས་པར་འཇུག་པ་དང་། བྱ་ལ་སོགས་པ་ནི་འཕུར། ཉ་ལ་སོགས་པ་ནི་རྒྱུག་པ་དང་འདི་ལྟ་བུའི་ཚུལ་གྱིས་དངོས་པོ་སོ་སོའི་མཐུ་ཡོད་པས་གང་ལ་གང་རིགས་པའི་བྱ་བ་བྱེད་པ་ཡིན་ནོ། །དེ་དག་གི་རྒྱུར་གྱུར་པ་དང་།
རྐྱེན་དུ་གྱུར་པ་ཞེས་བྱ་བ་ནི་གང་བྱེད་པའི་ཡན་ལག་ཏུ་གཏོགས་པ་དང་། གཞིར་གྱུར་པ་ཡིན་ཏེ། རྒྱུ་དྲུག་དང་རྐྱེན་བཞི་གོང་དུ་སྨོས་པ་རྣམས་ཡིན་ནོ། །བྱ་བ་བྱེད་པའི་རིགས་པའི་མཚན་ཉིད་ཀྱི་ལེའུ་སྟེ་གཉིས་པའོ།།      །།གཏན་ཚིགས་སྒྲུབ་པའི་རིགས་པའི་མཚན་ཉིད་ལ་དམ་བཅས་
ཤིང་བསྟན་ཏེ། སྨྲས་པའི་དོན་སྒྲུབ་པ་དང་ཡང་དག་པར་ཤེས་པར་བྱ་བའི་ཕྱིར་རྒྱུར་གྱུར་པ་དང་རྐྱེན་དུ་གྱུར་པ་དེ་དག་ནི་གཏན་ཚིགས་སྒྲུབ་པའི་རིགས་པའོ་ཞེས་གསུངས་པ་བཤད་དོ། །དེ་ལ་དམ་བཅའ་བ་ཞེས་བྱ་བ་ནི་ཇི་སྐད་དུ་སྨྲས་པ་དེ་བཞིན་དུ་འབྲས་བུ་དང་བཅས་པར་ཁས་བླངས་པའོ། །འབྲས་བུ་གང་
ཞེ་ན། གང་གིས་ཡལ་གཟུགས་འཁྲ་བར་བྱ་བ་ཡིན་ཏེ། ཇི་ལྟར་དམ་བཅས་མ་གྲུབ་ན་ཕ་ལོགས་ཀྱི་སྐྱབས་སུ་འགྲོ་བ་དང་། དེའི་བསྟན་པ་ལ་རབ་ཏུ་འབྱུང་ཞིང་ཚུལ་ཁྲིམས་བསྲུང་བ་དང་། བསམ་གཏན་ལ་མཉམ་པར་གཞག་པ་དང་ཤེས་རབ་བསྒོམ་པའོ། །འདི་ལྟ་བུ་ལ་སོགས་པ་སེམས་ཅན་གྱི་དོན་
དུ་འགྱུར་བས་ཡལ་གཟུགས་འཁྲ་བར་བྱའི་ལྷ་ཁང་གཞིག་པ་དང་ཡུལ་གྲུ་དབྲོག་པ་དང་སྲོག་གི་བར་ཆད་དང་། ལུས་ཀྱི་སྡུག་བསྔལ་དུ་འགྱུར་བ་ལ་སོགས་པ་སྡིག་ཏོ་མི་དགེ་བའི་ཡལ་གཟུགས་མི་བཅའོ། །ཡལ་གཟུགས་འཁྲ་བ་མེད་པར་བཙན་ན་ཡང་འགྲན་ཏོར་ཅན་ཁས་བརྒྱ་པ་དང་འདྲ་བར་འགྱུར་
གྱི་ཅི་རིགས་པར་ཡལ་གཟུགས་འཁྲ་བར་བྱའོ། །ཡལ་གཟུགས་འཁྲས་པ་དེ་ལ་ཕམ་རྒྱལ་དང་། །གཉིས་ཀར་མི་འགྱུར་བའི་བྱེ་བྲག་ཏུ་བྱ་བ་ནི་ཡོན་ཡོད་པ་དང་ཁྲིན་ཅན་དང་བར་མ་དོ་དང་གསུམ་དུ་ཤེས་པར་བྱའོ། །དེ་གང་ཞེ་ན། མཚན་ཉིད་ཀྱིས་ཡོངས་སུ་དག་པའི་ཚད་མས་གྲུབ་ན་ཡོན་ཡོད་པ་
