པ། གོས་ཟས་ཀྱིས་འཚོས་པ། ཁ་དུམ་པ། ཕས་བསྲུངས་པ། མས་བསྲུངས་པ། མིང་པོས་བསྲུངས་པ། ཕུ་ནུ་མོས་བསྲུངས་པ། གྱོས་པོས་བསྲུངས་པ། སྒྱུག་མོས་བསྲུངས་པ། གཉེན་པས་བསྲུངས་པ། རུས་ཆགས་ཀྱིས་
ཏཱ། མཱ་ལཱ་གུ་ཎ་པ་རི་ཀྵིཔྟཿ། ཏཏ྄་ཀྵ་ཎི་ཀཿ། ཨཱ་བཱ་ཧཿ། བི་བཱ་ཧཿ། ཀཱ་ཡ་སཾ་སརྒཿ། པ་རིཥྛ་ཛ་མཱ་ཏྲཾ། དྭ་ཡ་དྭ་ཡ་ས་མཱ་བཏྟིཿ། [དྲུག་སྡེའི་མིང་ལ།] ནནྡཿ། ཨུ་པ་ནནྡ། པུ་ནརྦ་སུ། ཙྪནྡཿ། ཨ་ཤྭ་ཀཿ། ཨུ་ད་ཡཱི།།
བསྲུངས་པ། སྐྱེས་ཀྱི་དུས་བཏབ་པ། ཐང་འགའ་འཕྲད་པ། བག་མར་བླངས་པ། བག་མར་བཏང་བ། མཐང་སྤྲད་པ། མཐང་སྤྲད་པ་ཙམ། གཉིས་ཀྱིས་གཉིས་སྤྲད་པ། དགའ་བོ། ཉེ་དགའ་བོ། ནབ་སོ། འདུན་པ། འགྲོ་མགྱོགས། འཆར་
[མ་མ་བཞིའི་མིང་ལ།] ཨཾ་ག་དྷཱ་ཏྲཱི། ཀྵི་ར་དྷཱ་ཏྲཱི། མ་ལ་དྷཱ་ཏྲཱི། ཀྲི་ཌ་ནི་ཀཱ་དྷཱ་ཏྲཱི། [{སྐད་ཀྱི་མིང་ལ།}] བྱཱ་དྷི། རོ་གཿ། བ་རི་དཱ་ཧཿ། བྲ་ཎཾ། གཎྜ། ཤྲཱ་བི་ཀཱ། ཀུཥྛཾ། བི་སྥོ་ཊཿ། ཀི་ཊི་བྷཿ། དརྡྡྲུཿ། ཀི་ལཱ་སཾ། བི་ཙརྩི་ཀཿ།
