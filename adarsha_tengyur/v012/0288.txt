པའི་སླད་དུ་ལྷ་རྣམས་ཀྱི་སྔགས་ཀྱི་ཚིག་རྣམས་ནི་རྩ་བའི་རྒྱུད་ལས་འདི་ལྟ་སྟེ། ཨོཾ་བུད་དྷཱ་ཡ་ན་མཿ། ཨོཾ་དྷརྨྨཱ་ཡ་ན་མཿ། ཨོཾ་སཾ་གྷཱ་ཡ་ན་མཿ། ཨོཾ་བཛྲ་ས་ཏྭ་ཡ་ན་མཿ། ཨོཾ་པྲཛྙཱ་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་སྭ་བྷཱ་བི་ཀ་ཀཱ་ཡཱ་ཡ་ན་མཿ། ཨོཾ་དྷརྨྨཱ་ཀཱ་ཡཱ་ཡ་ན་མཿ། ཨོཾ་སཾ་བྷོ་ག་ཀཱ་ཡཱ་ཡ་ན་མཿ། ཨོཾ་ནིར་མཱ་ཎ་
ཀཱ་ཡཱ་ཡ་ན་མཿ། ཨོཾ་ཛྙཱ་ན་བཛྲཱ་ཡ་ན་མཿ། ཨོཾ་ཙིཏྟ་བཛྲཱ་ཡ་ན་མཿ། ཨོཾ་བཱ་ཀ་བཛྲཱ་ཡ་ན་མཿ། ཨོཾ་ཀཱ་ཡ་བཛྲཱ་ཡ་ན་མཿ། ཨོཾ་ཙ་ཏུར་བི་མོཀྵེ་བྷྱོ་ན་མཿ། ཨོཾ་ཙ་ཏུར་བྲཧྨ་བི་ཧཱ་རེ་བྷྱོ་ན་མཿ། ཨོཾ་སཔྟ་ཏྲིཾ་ཤད་བོ་དྷི་པཱ་ཀྵི་ཀ་དྷརྨྨ་བྷྱོ་ན་མཿ། ཨོཾ་ཙ་ཏུར་ཤཱི་ཏི་ས་ཧཱ་སྲ་དྷརྨྨ་སྐན་དྷེ་བྷྱོ་ན་མཿ། ཨོཾ་སརྦ་དྷརྨྨ་དེ་ཤ་ཀེ་བྷྱོ་ན་མཿ། ཨོཾ་རཏྣ་ཏྲ་ཡཱ་ཡ་ན་མཿ། 
ཨོཾ་ཤཱཀྱ་མུ་ན་ཡེ་ན་མཿ། ཨོཾ་ཀཱ་ལ་ཙཀྲཱ་ཡ་ན་མཿ། ཨོཾ་དཱ་ན་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་ཤཱི་ལ་བཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་ཀྵཱན་ཏི་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་བཱིརྱ་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་བྷྱཱ་ན་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་པྲཛྙཱ་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་ཨུ་པཱ་ཡ་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་པྲ་ཎི་དྷཱ་ན་པཱ་
ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་པ་ལ་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་ཛྷཱ་ན་པཱ་ར་མི་ཏཱ་ཡཻ་ན་མཿ། ཨོཾ་ཛ་ཡ་གྷ་ཊེ་བྷྱོ་ན་མཿ། ཨོཾ་བི་ཛ་ཡ་གྷ་ཊེ་བྷྱོ་ན་མཿ། ཨོཾ་སརྦ་ཙིཧྣེ་བྷྱོ་ན་མཿ། ཨོཾ་སརྦ་མུ་དཱ་བྷྱོ་ན་མཿ། ཨོཾ་ཅིན་ཏཱ་མ་ཎ་ཡེ་ན་མཿ། ཨོཾ་དྷརྨྨ་ག་ཎ་ཊྱཻ་ན་མཿ། ཨོཾ་དྷརྨྨ་ཤཾ་ཁཱ་ཡ་ན་མཿ། ཨོཾ་ཀལྤ་བྲྀ་ཀྵཱ་ཡ་ན་
མཿ། ཨོཾ་ཨཀྵོ་བྷྱཱ་ཡ་ན་མཿ། ཨོཾ་ཨ་མོ་གྷ་སིད་དྷ་ཡཻ་ན་མཿ། ཨོཾ་རཏྣ་སཾ་བྷ་བཱ་ཡ་ན་མཿ། ཨོཾ་ཨ་མི་ཏཱ་བྷཱ་ཡ་ན་མཿ། ཨོཾ་བཻ་རོ་ཙ་ནཱ་ཡ་ན་མཿ། ཨོཾ་ལོ་ཙ་ནཱ་ཡཻ་ན་མཿ། ཨོཾ་མཱ་མ་ཀྱཻ་ན་མཿ། ཨོཾ་པཱཎྜ་རཱ་ཡཻ་ན་མཿ། ཨོཾ་ཏཱ་རཱ་ཡཻ་ན་མཿ། ཨོཾ་བཛྲ་དྷཱ་ཏཱི་ཤྭཪྻཻ་ན་མཿ། ཨོཾ་བཛྲ་པཱ་ཎ་ཡཻ་ན་མཿ། ཨོཾ་ཁ་
གརྦྷཱ་ཡ་ན་མཿ། ཨོཾ་ཀྵི་ཏི་གརྦྷཱ་ཡ་ན་མཿ། ཨོཾ་ལོ་ཀེ་ཤྭ་རཱ་ཡ་ན་མཿ། ཨོཾ་སརྦ་ནི་བ་ར་ཎ་བི་སྐམ་བྷི་ནེ་ན་མཿ། ཨོཾ་ས་མནྟ་བྷ་དྲཱ་ཡ་ན་མཿ། ཨོཾ་གན་དྷ་བཛྲཱ་ཡཻ་ན་མཿ། ཨོཾ་རཱུ་པ་བཛྲཱ་ཡཻ་ན་མཿ། ཨོཾ་ར་ས་བཛྲ་ཡཻ་ན་མཿ། ཨོཾ་སྤར་ཤ་བཛྲ་ཡཻ་ན་མཿ། ཨོཾ་ཤཔྟ་བཛྲཱ་ཡཻ་ན་མཿ། ཨོཾ་དྷརྨྨ་དྷཱ་ཏུ་
བཛྲཱ་ཡཻ་ན་མཿ། ཨོཾ་ཨུཥྞཱི་ཥཱ་ཡ་ན་མཿ། ཨོཾ་བིགྷྣཱན་ཏ་ཀཱ་ཡ་ན་མཿ། ཨོཾ་པྲཛྙཱནྟ་ཀཱ་ཡ་ན་མཿ། ཨོཾ་པདྨཱནྟ་ཀཱ་ཡ་ན་མཿ། ཨོཾ་ཡ་མཱནྟ་ཀཱ་ཡ་ན་མཿ། ཨོཾ་སྟམ་བྷི་ནྱཻ་ན་མཿ། ཨོཾ་མཱ་ནི་ནྱཻ་ན་མཿ། ཨོཾ་སྟོ་བྷི་ནྱཻ་ན་མཿ། ཨོཾ་ཨ་ཏི་བཱིཪྻཱ་ཡཻ་ན་མཿ། ཨོཾ་ཨ་ཏི་ནཱི་ལཱ་ཡཻ་ན་མཿ། ཨོཾ་སརྦ་དྷཱ་རི་ཎཱི་བྷྱོ་ན་མཿ།
