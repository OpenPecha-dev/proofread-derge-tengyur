གཉིས་གདངས་རྩེ་སྤྲད་ཁ། །མཐེའུ་ཆུང་གྱེན་དུ་བསྒྲེང་བའི་རྒྱས། །འཇིག་རྟེན་པ་ཡི་ལས་ཚོགས་རྣམས། །ཕྱོགས་ཀུན་དངོས་གྲུབ་རྒྱས་བྱེད་པའོ། །ཨོཾ་པདྨ་བ་མུ་ཀརྨ་སིདྡྷི་ཧཱུཾ། ཕྱག་རྒྱ་བདུན་རོལ་བསྒྱུར་བྱས་ཏེ། །མཐོབ་གུང་མཛུབ་སྲིན་གོང་བཞིན་སྦྱར། །དངོས་གྲུབ་མཆོག་རྣམས་
དབང་བྱས་པས། །རྟེན་བྱེད་མི་འབྲལ་ཀུན་སྤྱོད་མཛོད། །ཨོཾ་པདྨ་བ་མུ་ཁེ་ཨ་བེ་ཤ་ཡ་ཧྲཱིཿ། རྡོ་རྗེ་ཁུ་ཚུར་བཅངས་པ་ཡི། །མཐེ་བོ་རྩེ་བཞི་མཆེ་བའི་ཚུལ། །མཆོད་གཏོར་ཕྱོགས་ཀུན་གནས་བསྒྱུར་བས། །ལས་ཀུན་བྱ་བ་མཐར་ཕྱིན་འགྱུར། །ཨོཾ་བཛྲ་ཨེ་ས་ཧཱུཾ། པུ་ས་ཧཱུཾ། ས་མ་ཡ་
བག་ཤ་ཧཱུཾ། ཨོཾ་རུ་ལུ་རུ་ལུ་ཧཱུཾ་བྷྱོ་ཧཱུཾ། ཕྱག་རྒྱའི་ཏིང་འཛིན་སྔགས་བཅས་ལྡན། །གཏོར་མའི་ལས་བསྐུལ་གནས་སུ་བཏང་། །སླར་ལོག་དཀྱིལ་འཁོར་མ་ཉམས་བློས། །བཀྲ་ཤིས་དངོས་གྲུབ་བསྐུལ་བར་བྱ། །ཨོཾ་དངོས་གྲུབ་དབང་ཕྱུག་ཡང་དག་བདེན་པའི་སྟོབས། །ཕུན་སུམ་ཚོགས་པ་
ཀུན་དུ་འབྱུང་བའི་གཞི། །སྟོབས་མཆོག་མི་འཇིགས་གདོན་བྲལ་འཚེ་བ་མེད། །བཅོམ་ལྡན་བཀྲ་ཤིས་མི་འགྱུར་བརྟན་པར་ཤོག །ཨོཾ་སངས་རྒྱས་ཐམས་ཅད་མཉམ་པར་སྦྱོར་བ་ཡི། །མཁའ་འགྲོ་སྒྱུ་མ་བདེ་བ་ཆེན་པོའི་མཆོག །དཀྱིལ་འཁོར་ཕྱག་རྒྱ་གསང་སྔགས་སྦྱོར་བའི་མཐུས། །རིགས་
ཀུན་བཀྲ་ཤིས་མི་འགྱུར་བརྟན་པར་ཤོག །ཨོཾ་ཐམས་ཅད་དངོས་གྲུབ་དོན་ཚོགས་རྒྱ་ཆེན་པོ། །ཐུན་མོང་མངོན་ཤེས་རྣམ་ལྔ་སྒོ་གསལ་ཞིང་། །གསེར་འགྱུར་བུམ་པ་བཅུད་ལེན་བདུད་རྩིའི་ཚོགས། །བཅུ་དྲུག་བཀྲ་ཤིས་མི་འགྱུར་བརྟན་པར་ཤོག །ཨོཾ་མཚན་མཆོག་དཔེ་བྱད་མི་འགྱུར་
བརྟན་ཞིང་བརྗིད། །བདུད་འདུལ་བཀྲ་ཤིས་གདུགས་མཆོག་རྒྱལ་མཚན་དུང་། །འཁོར་ལོ་དཔལ་བེའུ་གསེར་ཉ་བུམ་པ་མཚོན། །ཉེ་བའི་བཀྲ་ཤིས་མི་འགྱུར་བརྟན་པར་ཤོག །ཨོཾ་ཞི་རྒྱས་དབང་དྲག་མནན་བཟློག་ལས་རྣམས་ཀུན། །ཐམས་ཅད་སྐད་ཅིག་མཐུ་ཡིས་གྲུབ་འགྱུར་ཞིང་། །
བདག་གཞན་ཆ་ལུགས་གནས་ཀུན་སྦྱོར་བྱེད་པའི། །ཕྲིན་ལས་བཀྲ་ཤིས་མི་འགྱུར་བརྟན་པར་ཤོག །ཨོཾ་རྒྱལ་བ་ཐམས་ཅད་གཉིས་མེད་མཉམ་པའི་དབྱིངས། །ཡེ་ཤེས་སྐུ་མཆོག་རིགས་དང་སྤྲུལ་པའི་ཚོགས། །མ་འདྲེས་དངོས་གྲུབ་འབྲས་བཟང་དཔེ་དང་བྲལ། །སྙོམས་འཇུག་
