གྱིས་ལེགས་པར་སྦངས་པ་སྟེ། །ཡང་ན་མངར་གསུམ་གྱིས་སྦངས་ཤིང་། །སེར་པོའི་དྲི་ཡིས་བྱུགས་པའོ། །ཏིལ་དང་ཞོ་ཟན་མར་ལྡན་དག །བིལ་བ་ཡི་ནི་འབྲས་བུ་དང་། །སཱ་ལུ་དན་དུ་ལ་དག་དང་། །ཞོ་ཡི་སྤྲིས་མ་དག་དང་ནི། །པདྨ་ནཱ་ག་གེ་སར་སོགས། །རྒྱས་པ་ལ་ནི་སྦྱིན་སྲེག་བྱ། །ཙན་ཙ་
ཀ་དང་ཨ་ཤོ། ཝ་ཀུ་ལ་ལ་སོགས་པ་ཡི། །ཡམ་ཤིང་སོར་ལྔའི་ཚད་ཙམ་ནི། །མཁས་པ་ཡིས་ནི་ཤེས་པར་བྱ། །སྐྱུར་པོ་མེ་ཏོག་དམར་པོ་དང་། །དེ་ལ་སོགས་པ་མཐའ་ཡས་མཆོག །གཞན་ཡང་རང་གི་རྟོག་པ་ལས། །གསུངས་པ་དབང་ལ་སྦྱིན་སྲེག་བྱ། །ནིམ་པ་དང་ནི་
སེང་ལྡེང་དང་། །བ་ལ་ཏ་ཀ་དག་དང་ནི། །མྱོས་བྱེད་ཀྱི་ནི་ཤིང་བུ་དང་། །ཁ་དང་ཚྭ་དང་སྐྱུར་ཤིང་རྣམས། །དྲག་ཏུ་སྟོན་པ་བྱས་ནས་ནི། །རྩ་བའི་ཆ་ལས་བྱུང་བ་ཡི། །སོར་བཅུ་པ་ཡི་ཚད་ཙམ་ལ། །དུག་དང་ཁྲག་དང་མར་ཁུས་བྱུག །པལ་ཀ་ཡི་ནི་མར་ཁུ་དང་། །ཁྲག་དང་ཚྭ་དང་
སྐྱེ་ཚེ་དང་། །ནིམ་པ་ཡི་ནི་ལོ་མ་དང་། །དུར་ཁྲོད་ཐལ་བ་རྐང་རལ་དང་། །རྔ་མོ་བོང་བུའི་རྟུག་པ་དང་། །སྐྲ་དང་ཞག་དང་ཁ་ལ་སོགས། །ཚེར་མ་མི་རུས་ཕྱེ་མ་དང་། །སྤུ་དང་ཤ་སོགས་ཇི་སྙེད་རྫས། །དུག་དང་ཁྲག་དང་ཚྭ་དང་ནི། །སྐྱེ་ཚེ་མྱོས་བྱེད་ཁུ་བ་ཡིས། །རྫས་འདི་དག་ནི་མཁས་པ་
ཡིས། །ལེགས་པ་ཉིད་དུ་བྱུག་པར་བྱ། །སོལ་བ་དང་ནི་ཐལ་བ་དང་། ཕུབ་མས་དགྲ་ཡི་གཟུགས་བྱས་ཏེ། །གཏུབ་ཅིང་གཏུབ་པར་བྱས་ནས་ནི། །མངོན་སྤྱོད་ཕྱིར་ནི་སྦྱིན་སྲེག་བྱ། །མཆོད་ཡོན་ཤིང་ཉིད་དག་དང་ནི། །ཨ་ཤྭདྠ་ལ་སོགས་པའི་ཤིང་། །ཞི་བ་ཡི་ནི་སྦྱིན་སྲེག་བྱ། །རྒྱས་པའི་
སྦྱིན་སྲེག་ཤིང་བུའོ། །སྣུམ་ཞིང་སྐྱུར་ལ་མེ་ཏོག་བཅས། །དབང་ལ་ཤེས་པའི་ཤིང་བུ་ཡིན། །ཚ་ཞིང་ཚེར་བཅས་དྲི་ངའི་ཤིང་། །དུར་ཁྲོད་ཀྱི་ནི་ཤིང་བུ་དང་། །རྔ་མོ་བོང་བུའི་རྟུག་པ་ནི། །མངོན་སྤྱོད་ཀྱི་ནི་ཤིང་བུ་ཡིན། །ཨོཾ་སརྦ་དུཥྚ་ཨུ་པ་ཤཱནྟིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ནི་ཞི་
བའི་སྔགས་སོ། །ཨོཾ་བཛྲ་པུཥྚིཾ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ནི་རྒྱས་པའི་སྔགས་སོ། །ཨོཾ་བཛྲ་སཾ་ཀ་ར་ཡ་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ནི་དབང་གི་སྔགས་སོ། །ཨོཾ་སུཾ་བྷ་ནི་སུཾ་བྷ་ཧཱུཾ། གྲྀཧྣ་གྲྀཧྞ་ཧཱུཾ། གྲྀཧྞཱ་པ་ཡ་གྲྀཧྞཱ་པ་ཡ་ཧཱུཾ། ཨཱ་ན་ཡ་ཧོ་བྷ་ག་བཱན་བིདྱ་རཱ་ཛ་ཧཱུཾ། མཱ་ར་ཡ་ཧཱུཾ་ཕཊ། ཅེས་བྱ་བ་ནི་མངོན་སྤྱོད་ཀྱི་
