མེད་པ་ཉིད་ཐོབ་པར་གྱུར་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ལྟར་རེ་ཞིག་ཕྲག་དོག་ལ་སོགས་པ་མི་མངའ་བའི་ཕྱིར་དཔེ་མེད་པ་ཉིད་བཤད་ནས། ཁྱོད་ནི་ཡོན་ཏན་རྒྱུ་རྣམས་ལ། །ཞེས་བྱ་བའི་རྒྱས་པས་རྒྱུ་གཞན་བཤད་པར་བྱ་སྟེ། ཁྱོད་ནི་ཡོན་ཏན་རྒྱུ་རྣམས་ལ། །མངོན་དགྱེས་འབྲས་བུ་རྣམས་ལ་མིན། །
ཡང་དག་སྒྲུབ་པ་དེ་ཉིད་ཀྱིས། །ཁྱོད་ཀྱི་ཡོན་ཏན་ཕ་མཐར་ཕྱིན། །འཇིག་རྟེན་པ་རྣམས་ནི་ཡོན་ཏན་རྣམས་ཀྱི་འབྲས་བུ་རྣམས་ལ་མངོན་པར་ཞེན་པ་ཡིན་གྱི་དེའི་རྒྱུ་རྣམས་ལ་ནི་མ་ཡིན་ནོ། །བཅོམ་ལྡན་འདས་ནི་བྱང་ཆུབ་སེམས་དཔར་གྱུར་ཅིང་སྲེད་པ་དང་བཅས་པར་གྱུར་པས་ཀྱང་འབྲས་
བུ་ལས་ཉུང་ཡང་རྒྱུ་རྣམས་ལ་མངོན་པར་ཞེན་པ་ནི་དེ་རྣམས་ཚོལ་བ་དང་དགོངས་པས་བགྱིས་སོ། །གང་དག་གི་རྒྱུ་ཡིན་ཞེ་ན། བཤད་པ། ཡོན་ཏན་ནི་བསམ་གཏན་དང་སྙོམས་པར་འཇུག་པ་རྣམས་ཀྱི་ཞེས་བྱ་བའི་དོན་ཏོ། །མིན་ཞེས་པ་ནི་མངོན་པར་ཞེན་པ་དང་སྦྱར་རོ། །གང་
ལ་ཞེ་ན། བཤད་པ། འབྲས་བུ་རྣམས་ལ་ཞེས་བྱ་བ་ནི་བསམ་གཏན་དང་སྙོམས་པར་འཇུག་པའི་བདེ་བ་རྣམས་ལའོ། །ཡང་ན་རྫུ་འཕྲུལ་ལ་སོགས་པ་རྣམས་ལའོ། །དེ་ཉིད་ཀྱིས་ཞེས་བྱ་བ་ནི་རྒྱུ་དེ་དང་ཡོན་ཏན་རྣམས་ཀྱི་རྒྱུ་རྣམས་ལ་མངོན་པར་ཞེན་པ་རྣམས་ཀྱིས་སོ། །ཁྱོད་ནི་
ཡོན་ཏན་ཕ་མཐར་ཕྱིན་པ་ནི་དེ་ཉིད་ཀྱིས་དང་སྦྱར་རོ། །རྣམ་པ་གང་གིས་ཤེ་ན། བཤད་པ། ཡང་དག་པར་བསྒྲུབ་པ་ཞེས་བྱ་བ་ནི་ཡང་དག་པའི་ལམ་གྱིས་དང་།གཞན་ལ་ཕན་པའི་སྒོ་ནས་ཞེས་བྱ་བའི་དོན་ཏོ། །ཁྱོད་ཅེས་བྱ་བ་ནི་བཅོམ་ལྡན་འདས་ལའོ། །མཐར་ཕྱིན་ཞེས་བྱ་བ་ནི་
ཁྱད་པར་རོ། །ཡོན་ཏན་ནི་བསམ་གཏན་ལ་སོགས་པ་སྟེ། ཕྱིན་པ་ནི་ཐོབ་པའོ། །དེ་ལྟ་བུའི་ཚིགས་སུ་བཅད་པ་གཉིས་པོ་འདིས་བྱང་ཆུབ་སེམས་དཔའི་གནས་སྐབས་དང་པོ་ཕྲག་དོག་ལ་སོགས་པ་མ་མཛད་པ་དང་། ཡོན་ཏན་རྣམས་ཀྱི་རྒྱུ་རྣམས་ལ་མངོན་པར་ཞེན་པ་དང་
འབྲས་བུ་ལ་མངོན་པར་ཞེན་པའི་རྒྱུས་བྱང་ཆུབ་བརྙེས་པའི་བཅོམ་ལྡན་འདས་ཀྱི་དཔེ་མེད་པ་འདིར་བཤད་དོ། །གནས་སྐབས་བར་མར་དམན་པ་ལ་མོས་པ་བསལ་བ་བཤད་པ་སྟེ། གང་གི་ཕྱིར་དམན་པ་ལ་མོས་པའི་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་འདི་སྐད་སྨྲ་སྟེ། ཉོན་
