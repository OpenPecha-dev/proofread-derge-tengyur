ཏྲ་ཀཱ་ནཱརྠ་ན། པརྻཱ་སྷཱཧྣ་ཙ་ལཙྪི་ཁཎྜ་ཀ་བྷྲྀ་ཏོ་དཱ་རཱཾཤྩ་བྷཱ་བཱ་ནུ་གཱན། པྲཱ་དཱཾ་སྟྱཾ་ཛི་ན་ཏེ་ན་ནཱུ་ན་མ་བྷ་བནྨཱ་རངྒ་ནཱ་ནཱཾ་མུ་ཁཾ་སངྒོ་བཱཉྩི་ཏ་པཀྵྨ་ལོ་ཙ་ན་ད་ལཾ་ཡཛྷ་ལ་སཱ་མནྠ་རཾ། ཤ་ཡྻ་སྭཔྣ་བཱི་ཙཱ་རི་ཎཱ་བ་ན་བྷུ་བཱ་བྷཱུ་ཏྲཱ་པུ་རཱ་པཀྵི་ཎཱ། བྷིནྣཱ་ཤེཥ་ཀ་ལེ་ཛོ་ག་ཏྱ་བི་ཀ་ལེ་ཛྱཱ་ཡཿཔ་ཐཱ་ཤཾ་སི་ནཱ། བིསྟཱ་རཾ་ག་མི་ཏཾ་བྲ་ཏ་བྷྲྀ་ཏཱཾ་ཡནྣཱ་ཐ་ཀཱ་
བུ་དོན་གཉེར་ལ། གཙུག་ཕུད་བཟུང་ཆུང་མ་ཡང་དང་པོ་དང་རྗེས་མཐུན། རྒྱལ་བ་དེ་ཡིས་བདུད་ཀྱི་བུ་མོ་རྣམས་ཀྱིས་རྫི་མ་མིག་འདབ་མ་གང་དྲ་བ་སྙན་ཚིག་འདར་མལ་གཉིད་རྣམ་དཔྱོད་ནགས་ས་གྱུར་ནས་གྲོང་བྱ་ཡིས། ཐ་དད་མ་ལུས་འགྲོ་ལ་མ་ཚང་མེད་རྒན་པོའི་ལམ་བསྔགས་རྒྱ་ཆེན་བགྲོད་པ་བརྟུལ་ཞུགས་བཟུང་གང་
པིཉྫ་ལཾ། ཏེ་ནཱ་ཛྙཱ་མ་རུ་ཏཱཾ་ཀི་རཱི་ཊ་ཀི་ར་ཎ་ཙྪཱ་ཡཾ་ཡཱསྤ་དཾ་བོ་ག་ཏཱ། ཡཏྤ་རྞྞཾམྦུ་ཕ་ལཱ་ཤི་ནཱི་ཤ་ཤ་ཏ་ནུཾ་ཏྭཾ་ནཱ་ཐ་བི་བྷྲ་ཏཱུ་རཱ། ཏཊྐཱ་ལཱ་ཏི་ཐ་ཡེ་ཀྲྀ་པཱ་རི་ཙ་ཡཱ་དཱ་ཏི་ཥྱ་མཱ་བེ་ད་ན་ཡ་ན། ཨཱཿརཱུ་ཌྷཿསྥུ་ཊ་བི་སྥུ་ལིངྒ་མ་ན་ལནྟེ་ནཱ་ངྷུ་ན་པི་ཀྵི་ཏཱ། བིནྡོ་མཎྜ་ལ་ལཱཉྩ་ནཾ་ཤ་ཤ་མ་ཡཾ་ཡུཥྨ་ཪྻ་ཤོ་ལཀྵྱ་ཏེ། ཡཏྐོ་པཱ་རུ་ཎ་དྲྀཥྚི་ནཱ་ཀ་ལ་
མགོན་དེ་ཡིས་ཤེས་ཅོན་པན་འོད་མཛེས་གོ་འཕང་། གང་ཞིག་ལོ་མ་ཆུ་དང་འབྲས་བུ་ཟ་བའི་རི་བོང་ལུས་ནི་མགོན་ཁྱོད་བཟུང་བ་ན། དེ་དུས་འགྲོན་ལ་བརྩེ་བས་ཡོངས་སུ་འགྲིས་པས་སྦྱིན་བྱར་རིག་ནས་བཞུགས་འགྲོ་མེ་སྟག་མེལ་དེ་ནས་ལྟོས་བ་མེད་ཐིག་ལེའི་དཀྱིལ་འཁོར་མཚན་མ་རི་བོང་རང་བཞིན་ཁྱོད་ཀྱིས་གྲགས་པ་མཚོན་པར་བྱེད། གང་འཁྲོས་སྐྱ་རིངས་
བྷུ་བཱ་ཀྲྀ་ཏྭཱ་བྷཱུ་བོ་བྷངྒ་ནཱི་མུལླ་སི་ཏཱ་སི་ནཱ་བི་ད་ཏ་པྲངྒ་མཱ་སངྒི་ནཱ། ཏྭཱ་མཱ་སཱི་ཏ་ཀྵ་ཎ་མ་པྱ་བི་ཀྵ་ཏ་དྷི་ཡཾ་ཏྱཀྟུ་ཀྵ་མཱ་ན་ཀྵ་མཱ་ནྱི་ཏེ་ན་ནི་རངྐུ་ཤཱཿསྨ་ར་ཤ་རཱཿཔུདྑེ་ཥུ་བྷངྒངྒ་ཏཱ། ཡདྡྭེཥཱི་ཧ་རི་ཎཱ་ཙི་རཱ་ཏ་ན་ངྷ་ནུརྡནྟེསྟཱ་ཡཱཉཱབྷྱརྩི་ཏཿཥཌྟནྟེ་ན་ས་ཏཱ་སི་ཏོནྣ་ཏ་བྲྀ་ཧཛྫི་མཱུ་ཏ་པུཉྫ་ཏྭི་ཥཱ། ཀོ་ཤཱ་ཀཪྵ་ཎ་བཱནྟ་བྷཱུ་རི་རུ་ངྷི་ར་
ས་སྟོང་བྱས་ནས་སྨིན་པ་གཡོ་རི་མོ་རལ་གྲིལ་རྣམ་པར་བཅོམ། ཉིང་ལག་ཤ་ཁྱོད་ལ་སྐད་ཅིག་ཀྱང་བློ་བཏང་བཟོད་དང་མི་བཟོད་སེམས་དེས་ན་ལྕགས་ཀྱུ་མེད་དྲན་བའི་མདང་བཅོམ་པར་གྱུར། གང་སྡང་འཕྲོག་བས་རིང་ནས་གཞུ་སོ་ཁྱོད་ཀྱིས་མངོན་པར་མཆོད་སོ་དྲུག་པ་ཡིས་དཀར་ཞིང་མཐོ་བ་ལྷ་ཡི་ཆར་སྤྲིན་ཕུང་པོ་འོད་འབར། །སྦུབས་ནས་བཞུགས་པས
