འགྱུར་རོ། །དེ་དག་ནི་དགེ་བའི་རྩ་བ་དེས་སངས་རྒྱས་བཅོམ་ལྡན་འདས་རྣམས་དང་འབྲལ་བར་མི་འགྱུར་ཏེ། དེ་ངན་སོང་དུ་སྐྱེ་བར་མི་འགྱུར་བར་ལྷ་དང་མི་རྣམས་སུ་སྐྱེ་བར་འགྱུར་ཏེ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་དང་འབྲལ་བར་མི་འགྱུར་ཞིང་ཕྱིར་མི་ལྡོག་
པའི་ས་ནོན་པར་འགྱུར་རོ། །བྱང་ཆུབ་སེམས་དཔའི་ཐེག་པ་པའི་རིགས་ཀྱི་བུའམ། རིགས་ཀྱི་བུ་མོ་དེ་དག་དགེ་བའི་རྩ་བ་དེ་ཉིད་ཀྱིས་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག་དང་། ནང་སྟོང་པ་ཉིད་དང་། ཕྱི་སྟོང་པ་ཉིད་དང་། ཕྱི་ནང་སྟོང་པ་ཉིད་དང་། དངོས་པོ་མེད་པའི་ངོ་བོ་ཉིད་སྟོང་པ་ཉིད་ཀྱི་བར་དང་འབྲལ་
བར་མི་འགྱུར་རོ། །དྲན་པ་ཉེ་བར་གཞག་པ་དང་། ཡང་དག་པར་སྤོང་བ་དང་། རྫུ་འཕྲུལ་གྱི་རྐང་པ་དང་། དབང་པོ་དང་། སྟོབས་དང་། བྱང་ཆུབ་ཀྱི་ཡན་ལག་དང་། ལམ་དང་། འཕགས་པའི་བདེན་པ་དང་། ཚད་མེད་པ་དང་། བསམ་གཏན་དང་། གཟུགས་མེད་པའི་སྙོམས་པར་འཇུག་
པ་རྣམས་དང་འབྲལ་བར་མི་འགྱུར་རོ། །རྣམ་པར་ཐར་པའི་སྒོ་སྟོང་པ་ཉིད་དང་། མཚན་མ་མེད་པ་དང་། སྨོན་པ་མེད་པ་རྣམས་དང་འབྲལ་བར་མི་འགྱུར་རོ། །རྣམ་པར་ཐར་པ་བརྒྱད་དང་། མཐར་གྱིས་གནས་པའི་སྙོམས་པར་འཇུག་པ་དགུ་རྣམས་དང་ནམ་ཡང་འབྲལ་བར་མི་འགྱུར་རོ། །མངོན་
པར་ཤེས་པ་དང་། ཏིང་ངེ་འཛིན་དང་། གཟུངས་ཀྱི་སྒོ་རྣམས་དང་འབྲལ་བར་མི་འགྱུར་རོ། །དེ་བཞིན་གཤེགས་པའི་སྟོབས་བཅུ་རྣམས་དང་། མི་འཇིགས་པ་བཞི་རྣམས་དང་། སོ་སོ་ཡང་དག་པར་རིག་པ་བཞི་རྣམས་དང་། སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་བཅོ་བརྒྱད་དང་འབྲལ་བར་མི་འགྱུར་རོ། །
རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་དང་འབྲལ་བར་མི་འགྱུར་རོ་ཞེས་བྱ་བ་ནི་དོན་ཆེན་པོ་ཉིད་ལ་སོགས་པའི་ཡོན་ཏན་ཡིན་ནོ།    །ཤཱ་རིའི་བུ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་འདི་དེ་བཞིན་གཤེགས་པ་མྱ་ངན་ལས་འདས་ནས་ལྷོ་ཕྱོགས་ཀྱི་རྒྱུད་དུ་དར་བར་འགྱུར་ཏེ། དེ་ན་དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་
དང་། དགེ་བསྙེན་མ་དག་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཟབ་མོ་འདི་འདྲི་བར་འགྱུར། འཛིན་པར་འགྱུར། འཆང་བར་འགྱུར། ཀློག་པར་འགྱུར། ཀུན་ཆུབ་པར་བྱེད་པར་འགྱུར། ཚུལ་བཞིན་དུ་ཡིད་ལ་བྱེད་པར་འགྱུར་ཏེ། དེ་དག་དགེ་བའི་རྩ་བ་དེས་ངན་སོང་ལོག་པར་ལྟུང་བར་འགྲོ་བར་མི་འགྱུར་བར་དེ་
