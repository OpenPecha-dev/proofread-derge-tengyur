ཆོ་ག་གསུམ་གྱིས་བསྐྱེད། །ལྷ་མོ་སྐུ་མདོག་ཕྱག་མཚན་ནི། །གོང་དུ་བསྟན་པ་བཞིན་དུ་བསམ། །ཕྱག་རྒྱ་བཞི་དང་ལྡན་ཞེས་བྱ་བ་ནི། ལྕགས་ཀྱུ་ལ་སོགས་ཕྱག་རྒྱ་བཞིས། །དགུག་གཞུག་བཅིང་བ་ལ་སོགས་བྱ། །དེ་ཡང་སྔོན་གྱི་དམ་ཚིག་གིས། །བསྐུལ་བར་བྱ་བའི་ཚིགས་
བཅད་བརྗོད། །ཧཱུཾ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །དེ་ནས་བདེ་གཤེགས་སྤྱན་དྲངས་ཏེ། །དབང་བསྐུར་ནས་ནི་རྒྱས་གདབ་བྱ། །དེ་ནས་འཁོར་ལོའི་ལྷ་རྣམས་ལ། །ལྕེ་ལ་ཧཱུཾ་ལས་རྡོ་རྗེ་བསམ། །དེ་ནས་སྦྱིན་སྲེག་དབུལ་བྱ་སྟེ། །མེ་ནི་རབ་ཏུ་སྦར་བས་ཏེ། །མེ་ལྷ་བསྐྱེད་པའི་སྔོན་དུའོ། །དགང་
