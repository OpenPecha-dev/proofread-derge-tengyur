ཅེས་བྱ་བ་ནི་དུར་ཁྲོད་ཀྱི་མགལ་དུམ་དང་ལྡན་པས་བསྒྲུབ་བྱའི་ཕ་མེས་ཀྱི་རོ་བསྲེགས་པའི་སར་བསྲེག་པར་བྱའོ། །འཁོར་ལོ་བྲིས་ལ་ཞེས་བྱ་བ་ནི། ཕྱི་རོལ་ཟུར་གསུམ་འབར་བར་བྱ། །དབུས་སུ་མུ་ཁྱུད་བཞིས་བསྐོར་ཏེ། །ཕྱི་རོལ་ཟུར་གསུམ་དག་ཏུ་ནི། །རྡོ་རྗེ་འབར་བ་བྲི་བར་བྱ། །དབུས་སུ་ནང་གི་སྙིང་
པོ་ནི། །བསྒྲུབ་བྱའི་མིང་དང་ལྡན་པར་བྲི། །ཕྱི་རོལ་གྱི་ནི་མུ་ཁྱུད་ལ། །ཕྱི་ཡི་སྙིང་པོ་བྲི་བར་བྱ། །དེ་ཡི་ཕྱི་རོལ་མུ་ཁྱུད་ནི། །ལིང་ཚེ་བརྒྱད་དུ་གནས་པ་ལ། །འཁོར་ལོའི་སྔགས་སུ་བཀོད་པ་ནི། །སྡེ་ཚན་བརྒྱད་དུ་རབ་ཕྱེ་སྟེ། །ཧཱུཾ་གི་བར་དུ་བྲི་བར་བྱ། །ཕྱི་རོལ་མུ་ཁྱུད་བརྒྱད་ལ་ནི། །རྩ་བའི་སྔགས་ནི་
དགོད་པར་བྱ། །དེ་ཡི་ཕྱི་རོལ་མུ་ཁྱུད་ལ། །ལས་ཀྱི་སྔགས་ནི་དགོད་པར་བྱ། །ཕྱི་རོལ་རྡོ་རྗེ་སྙིང་པོ་ལ། །ནང་གི་སྙིང་པོ་བྲི་བར་བྱ། །དེ་ནས་རྡོ་རྗེ་རྩེ་གཞན་ལ། །དྲག་པོའི་ཧཱུཾ་ནི་བྲི་བར་བྱ། །ཕྱི་རོལ་ཟུར་གསུམ་བར་ཁྱམས་ལ། །གསང་བའི་སྔགས་ནི་དགོད་པར་བྱ། །ཡི་གེ་དགོད་ཅེས་བྱ་བ་ནི། །
ལུགས་དང་ལུགས་མིན་གཉིས་སུ་དགོད། །གསང་བའི་སྔགས་ནི་བྲི་བ་ནི། །འཁོར་ལོ་རྩིབས་བརྒྱད་བྲི་བར་བྱ། །ལྟེ་བར་མེ་ཡི་དཀྱིལ་འཁོར་བྲི། །མེ་ཡི་དབུས་སུ་ནང་གི་ནི། །སྙིང་པོའི་མིང་དང་བཅས་པ་དགོད། །ཕྱི་རོལ་ཟླ་གམ་གསུམ་ལ་ནི། །ཕྱི་ཡི་སྙིང་པོ་དགོད་པར་བྱ། །འཁོར་ལོའི་
རྩིབས་ནི་ཐམས་ཅད་ལ། །རྩ་བའི་སྔགས་ནི་དགོད་པ་སྟེ། །སྡེ་ཚན་བརྒྱད་དུ་དབྱེ་བར་བྱ། །བར་དུ་བསྒྲུབ་པའི་མིང་ཡིག་གཞུག །ཕྱི་རོལ་འཁོར་ལོའི་མུ་ཁྱུད་ལ། །གསང་བའི་སྔགས་ནི་བྲི་བར་བྱ། །དེ་ཡང་གཡས་སྐོར་གཡོན་སྐོར་དུ། །འཁོར་ལོ་གཉིས་ནི་བྲི་བར་བྱ། །སྟོབས་ཀྱི་པར་
བུ་ཞེས་བྱ་ནི། །ཕྱག་མཚན་གཉིས་ནི་བཀོད་པ་སྟེ། །སྟེང་དང་འོག་ཏུ་ཡང་དག་གནས། །ལིངྒ་ཆོ་གས་གཞུག་པར་བྱ། །ཞེས་བྱ་བ་ནི་ཡིད་འཕྲོག་མ་ལ་སོགས་པའི་མཁའ་འགྲོ་མ་དགུ་སྟ་གོན་ལ་གནས་པར་བྱའོ། །ལྷོ་ནུབ་ཏུ་ཁ་བལྟས་ཏེ་ཞེས་བྱ་བ་ནི། འཁོར་ལོ་བྲི་བའི་དུས་སུའོ། །ལིངྒ་མིང་གི་
ཡི་གེ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །བུམ་པར་འཁོར་ལོ་གཞུག་པར་བྱ། །ཞེས་བྱ་བ་ནི་ཕྱིར་མི་ལྡོག་པའི་ཆོ་ག་སྟེ་སྲུང་བའི་འཁོར་ལོའོ་ཞེས་བཀའ་སྩལ་ཏོ་ཞེས་བྱ་བ་ནི་པར་བུ་བཅའ་བའི་རིམ་པ་བསྡུས་ཏེ་བསྟན་པའོ། །དེ་ནས་གཞན་ཡང་རབ་བཤད་པ། །ཞེས་བྱ་བ་ནི་སྐུའི་ཆོ་ག་ལ་
