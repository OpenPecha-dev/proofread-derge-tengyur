སྦྱང་བའི་ཕྱིར་ཏེ། གནོད་པ་བྱེད་པའི་ནུ་ཞོ་ལས་བྱུང་བའི་ནད་ཉེ་བར་ཞི་བར་བྱ་བའི་དོན་ཏོ། །གདོན་གསོ་བ་ནི་ལྷ་དང་། ལྷ་མ་ཡིན་དང་། དྲི་ཟ་དང་། གནོད་སྦྱིན་དང་། སྲིན་པོ་དང་། ཡི་དྭ་གས་དང་། ཤ་ཟ་དང་། ཀླུ་ལ་སོགས་པའི་གདོན་གྱིས་སེམས་
ཟིན་པ་ལ་ཞི་བའི་ལས་དང་། གཏོར་མ་སྦྱིན་པ་ལ་སོགས་པ་གདོན་ཉེ་བར་ཞི་བའི་དོན་དུ་བྱའོ། །ལུས་ཀྱི་སྟོད་ཀྱི་ཡན་ལག་གསོ་བ་ནི། ནམ་ཚོང་ཡན་ཆད་དུ་གཏོགས་པའི་ནད་ཀྱི་རྣམ་པར་གྱུར་པ་སྟེ། མིག་དང་། རྣ་བ་དང་། སྣ་དང་། ཁ་ལ་སོགས་པའི་
ནད་ཉེ་བར་ཞི་བར་བྱ་བའི་དོན་དུ་བྱའོ། །མཚོན་གྱིས་སྨས་པ་གསོ་བ་ནི། རྩྭ་དང་། ཤིང་དང་། རྡོ་དང་། ས་རྡུལ་དང་། ལྕགས་དང་། བོང་བ་དང་། རུས་པ་དང་། སྐྲ་དང་། སེན་མོ་སྣ་ཚོགས་ཀྱི་རྣག་འཛག་པ་དང་། རྨ་མི་བཟད་པ་དང་། ནང་དུ་ནུབ་
པའི་ཟུག་རྔུ་དབྱུང་བའི་དོན་དུ་བྱའོ། །མཆེ་བའི་གསོ་བ་ནི། སྦྲུལ་དང་སྲིན་བུ་ལ་སོགས་པས་ཟིན་པ་བསལ་བ་དང་། དུག་སྣ་ཚོགས་ཀྱིས་བཅོམ་པ་ཉེ་བར་ཞི་བའི་ཕྱིར་བྱ་བའོ། །རྒས་པ་ལ་གསོ་བ་ནི་ལང་ཚོ་ལ་གནས་པར་བྱ་བ་དང་།ཚེ་དང་བློ་འཕེལ་བ་དང་། 
ནད་ཉེ་བར་ཞི་བའི་དོན་དུ་བྱའོ། །རོ་ཙ་བར་བྱ་བའི་ཕྱིར་གསོ་བ་ནི། ཁུ་བ་ཆུང་བ་དང་། གདུག་པ་དང་། ཟད་པ་དང་། བསྐམས་པར་གྱུར་པ་རྣམས་ས་བོན་འཕེལ་ཞིང་རབ་ཏུ་དང་བ་བསྐྱེད་པའི་ཕྱིར་དང་། རབ་ཏུ་དགའ་བ་བསྐྱེད་པའི་ཕྱིར་བྱ་བའོ། །དེ་ལྟར་ན་ཡན་
ལག་རྣམ་པ་བརྒྱད་དུ་བཤད་པ་ཡིན་ནོ། །གང་གི་དོན་དུ་གསོ་བར་བྱ་བའི་ཉེས་པ་དེ་བསྟན་པར་བྱ་བའི་ཕྱིར། འདི་ལྟར། རླུང་དང་མཁྲིས་པ་བད་ཀན་ཡང་། །ཉེས་པ་རྣམ་གསུམ་མདོར་བསྡུས་ཡིན། །ཞེས་སྨྲས་པ་ཡིན་ཏེ། དེ་ལ་དང་པོར་རླུང་སྨོས་པའི་རྒྱུ་བཤད་
པར་བྱ་སྟེ། ཁྱབ་དང་མྱུར་དུ་བྱེད་ཕྱིར་དང་། །སྟོབས་ལྡན་གཞན་དག་འཁྲུག་བྱེད་ཅིང་། །རང་རྒྱུད་གྲུབ་དང་ནད་མང་ཕྱིར། །ཀུན་ལས་རླུང་ནི་སྟོབས་ལྡན་ཡིན། །ཞེས་བཤད་པ་ཡིན་ནོ། །དེའི་འོག་ཏུ་ནི་ཕལ་ཆེར་རྒྱུ་འདི་དག་ཉིད་ཀྱིས་མཁྲིས་པ་སྨོས་པ་ཡིན་ལ།
