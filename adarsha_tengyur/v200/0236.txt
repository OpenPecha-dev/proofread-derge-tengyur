སྤྱི་ཡི་མཚན་མ་ནི། །རྣམ་པར་འགགས་སམ་ཤས་ཆེར་འབྱུང་། །ན་ཞིང་རླུང་ནི་མི་གསལ་རྒྱུ། །མཁྲེགས་ཤིང་ལུས་ལྕི་མགོ་བོ་འཁོར། །ཞེས་བྱ་བ་ལ། མ་ཞུ་བ་སྤྱིའི་མཚན་མའམ། རྟགས་སུ་གྱུར་པ་ནི། བཤང་བ་དང་གཅི་བ་དང་། འཕྱེན་ལ་སོགས་པ་སྲ་
ཞིང་རྣམ་པར་འགགས་པར་གྱུར་པའམ། ཡང་ན་བཤང་གཅི་ལ་སོགས་པ་ཤས་ཆེར་འབྱུང་བར་འགྱུར་བ་དང་། ན་ཞིང་རྨྱ་བར་འགྱུར་བ་དང་། རླུང་མི་གསལ་བར་རྒྱུ་ཞིང་ཤེས་པར་དཀའ་བ་འབྱུང་བར་འགྱུར་བ་དང་། བཤང་བ་མཁྲེགས་པ་དང་། ལུས་ལྕི་བ་དང་། མགོ་

ཆུ་མང་དང་། །བསྲེས་པའི་ཟས་ནི་འཇུ་མི་འགྱུར། །མྱ་ངན་ཁྲོ་དང་བཀྲེས་པ་ཡིས། །ཉེ་བར་གདུངས་ཤིང་ཟས་མི་འཇུ། །ཞེས་བྱ་བ་ལ། ཟས་ཀྱི་ཚོད་ལྷག་པར་གྱུར་པར་ཟོས་པ་གཅིག་པུ་མ་ཞུ་བར་གྱུར་པར་མ་ཟད་ཀྱི། མི་འདོད་བཞིན་དུ་ཟོས་པ་དང་། འགག་པར་བྱེད་པ་སྟེ། 
རྩ་སྒོའི་སྦུབས་ཀྱི་གགས་བྱེད་པ་དང་། ཚིག་པ་སྟེ་བཙོས་པ་ཐལ་དྲགས་པ་དང་། མ་ཚོས་པ་སྟེ་ལེགས་པར་ཚོས་ཤིང་སྨིན་པར་མ་གྱུར་པ་དང་། ལྕི་བ་དང་འཇུ་དཀའ་བ་དང་། རྩུབ་སྟེ་སྣུམ་མེད་པ་དང་། མི་གཙང་བ་སྟེ་མ་དག་པ་དང་། གྲང་བ་སྟེ་ཁེངས་པ་དང་། ཚ་བ་
