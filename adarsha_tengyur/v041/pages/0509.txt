མོ་དང་ཡང་དག་པར་སྦྱོར་བའི་སྒོ་ནས་ནི་རྣམ་པར་ནང་མཛད་བསྒྲུབ་པར་བྱ་བའོ། །དེ་བཞིན་རྒྱལ་རིགས་དང་། རྗེའུ་རིགས་དང་། དམངས་རིགས་ཀྱི་བུ་མོ་རྣམས་པར་ཡང་དག་པར་སྦྱོར་བའི་སྒོས་ནས། གོ་རིམས་བཞིན་དུ་རིན་ཆེན་འབྱུང་ལྡན་དང་། འོད་དཔག་
ཏུ་མེད་པ་དང་། དོན་ཡོད་པར་གྲུབ་རྣམས་ཀྱིས་དངོས་གྲུབ་སྒྲུབ་པར་ཤེས་པར་བྱའོ། །རྡོ་རྗེ་འཆང་ཆེན་པོ་ནི་རྡོ་རྗེའི་རིགས་གཙོ་བོར་གྱུར་བའི་སྒོ་ནས་བདག་པོ་ཉིད་ཡིན་པའི་ཕྱིར་ཐམས་ཅད་དང་ཡང་དག་པར་སྦྱོར་བ་ཉིད་ཀྱི་སྒོ་ནས་བསྒྲུབ་པར་ཤེས་པར་བྱའོ། དེ་བཞིན་དུ་རིགས་ཀྱི་
བྱེ་བྲག་གི་སྒོ་ནས་ལྷ་ཐམས་ཅད་བསྒྲུབ་པར་རྟོགས་བར་བྱའོ། རྡོ་རྗེ་ཆང་ཆེན་པོ་ལ་སོགས་པའི་ཆོས་ཀྱི་རང་བཞིན་གྱི་བདག་ཉིད་ཡིན་པས་ན་རྡོ་རྗེ་ཆོས་བདག་སྟེ་སྒྲུབ་པ་པོས་སོ། དེ་ཉིད་ནི་བསྒྲུབ་པར་བྱ་ཞེས་བྱ་བ་ལ་སོགས་པས་བཤད་དོ། །བསྐྱེད་པའི་རིམ་པ་སྒོམ་པ་པའི་དབང་
དུ་བྱས་ནས། བཤད་མ་ཐག་པའི་བསྒྲུབ་པའི་དུས་ངེས་པ་ནི། རྡོ་རྗེ་ཉི་མ་ཞེས་བྱ་བ་ལ་སོགས་གསུང་ཏེ། འདུལ་བ་ལ་ཤེས་བཞིན་སྔོན་དུ་འགྲོ་བ་བཞིན་དུ་རྡོ་རྗེ་སྔོན་དུ་འགྲོ་ཅན་དུ་བརྗོད་པར་བྱ་བ་ཡིན་པས་ཉི་མ་ཉིད་རྡོ་རྗེ་ཉིད་མ་སྟེ། དེ་ནུབ་བར་གྱུར་ནས་
བསྒོམ་པ་བྱའོ། །སྐྱ་རིངས་ནི་ཉི་མའི་ཤིང་རྟ་ཁྲིད་པ་སྟེ། དེ་འཆར་བའི་དུས་སུ་ཞག་བདུན་ཚུན་ཆད་ཀྱིས་གྲུབ་པར་འགྱུར་རོ། །རྟག་ཏུ་རྣལ་འབྱོར་གནས་པ་ལ་ནི་དུས་ངེས་པ་མེད་པས་གཞན་དུ་བརྗོད་པར་བྱ་སྟེ། སེམས་ཅན་ཐམས་ཅད་ཀུན་དུ་སྡུས་པར་བྱེད་པའི་སྙིང་རྗེ་ཆེན་པོའི་རང་
བཞིན་ཉིད་དང་། མི་ཤེས་པའི་མུན་པ་འཇོམས་པར་བྱེད་པ་ཉིད་ཡིན་པའི་ཕྱིར། རང་གི་གསང་བའི་རྡོ་རྗེ་ཉི་མའོ། །དེ་ནུབ་ནས་ཏེ་བཙུན་མོའི་དབྱིངས་སུ་ནུབ་ནས་སོ། སྐྱ་རེངས་ནི་བྱང་ཆུབ་སེམས་ཏེ་དེ་འཆར་བ་ནི་འབྱུང་བའོ། །དེ་བས་ནི་དེས་ནི་འདི་སྐད་བཤད་པ་ཡིན་
ཏེ། རྟག་ཏུ་ལྷའི་ང་རྒྱལ་དང་ལྡན་ཞིང་རང་གི་རིགས་ཀྱི་བུ་མོ་དང་སྙོམས་པར་ཞུགས་པས་སྤྲོ་བ་ལ་སོགས་པ་མི་བྱ་ཞིང་ཅུང་ཟད་འབར་བའི་བྱང་ཆུབ་ཀྱི་སེམས་མི་དབྱུང་བར་བསྒོམས་ན། བྱང་ཆུབ་ཀྱི་སེམས་དབྱུང་བ་བརྟན་པ་ཉིད་དུ་གྱུར་ནས་གདོན་མི་ཟ་བར་དེའི་རང་བཞིན་དུ་འགྱུར་རོ་ཞེས་བྱ་
