རྗེས་སུ་སྦྱོར་བ་ན་གཉིད་ཀྱི་བདེ་བ་དང་། ཉལ་བའི་བདེ་བ་དང་། གློས་འབེབས་པའི་བདེ་བ་ལ་ཆགས་ཤིང་སེམས་འཇུག་པ་ནི་བདུད་ཀྱི་ལས་སུ་རིག་པར་བྱའོ། །ཤེས་བཞིན་དང་བཅས་པར་གནས་པ་ནི་བྱིས་པའམ་འགྲོ་བ་བཟང་མོ་ཁ་དོག་མཆོག་དང་ལྡན་པའི་རྣམས་མཐོང་ནས་
ཚུལ་བཞིན་མ་ཡིན་པ་ཡིད་ལ་བྱེད་ཅིང་དེར་སེམས་འཇུག་པ་དང་། འཇིག་རྟེན་གྱི་འཕྲལ་གྱི་བྱ་བ་མང་པོ་ལ་སེམས་འཇུག་པ་དང་། འདུ་འཛི་མང་པོའི་ནང་དུ་འདྲི་ཞིང་སྡིག་པའི་གྲོགས་པོ་མི་དགེ་བའི་ལས་བྱེད་པ་རྣམས་མཐོང་ནས་སེམས་ལ་མོས་པ་སྐྱེ་བ་ནི་བདུད་ཀྱི་ལས་
སོ། །དེ་བཞིན་དུ་སངས་རྒྱས་དང་ཆོས་དང་དགེ་འདུན་དང་། སྡུག་བསྔལ་དང་། ཀུན་འབྱུང་དང་། འགོག་པ་དང་། ལམ་དང་ཚེ་འདི་དང་ཚེ་རབས་ཕ་རོལ་རྣམས་ལ་ཐེ་ཚོམ་དང་། ཡིད་གཉིས་སྐྱེ་བ་ནི་བདུད་ཀྱི་ལས་སོ། །གང་གི་ཚེ་རྙེད་པ་དང་བཀུར་སྟི་ལ་སེམས་
ཆགས་པ་དང་། སེར་སྣ་དང་། འདོད་ཆེན་དང་། ཆོག་མི་ཤེས་པ་དང་། ཁྲོ་བ་དང་། ཁོན་དུ་འཛིན་པ་དང་། ཚུལ་འཆོས་པ་དང་། ཁ་གསག་ལ་སེམས་འཇུག་ན་བདུད་ཀྱི་ལས་སུ་རིག་པར་བྱའོ། །གཞན་ཡང་རྒྱས་པར་བཤད་ན་དེ་ལ་སོགས་པ་བདུད་ཀྱི་ལས་
ཚད་མེད་པར་ཡོད་དེ། འཇམ་དཔལ་རྣམ་པར་འཕྲུལ་པ་ལས་ཀྱང་། ལྷའི་བུ་ཇི་ཙམ་དུ་ལས་རྣམས་འབྱུང་བ་དེ་ཙམ་དུ་བདུད་ཀྱི་ལས་སོ། །ཇི་ཙམ་དུ་སྨོན་པ་དང་། ཇི་ཙམ་དུ་ལོག་པར་འཛིན་པ་དང་། ཇི་ཙམ་དུ་མཆོག་ཏུ་འཛིན་པ་དེ་ཙམ་དུ་བདུད་ཀྱི་ལས་སོ། །ཇི་
ཙམ་དུ་འདོད་པ་དང་། ཇི་ཙམ་དུ་འདུ་ཤེས་པ་དང་། ཇི་ཙམ་དུ་རློམ་སེམས་པ་དང་། ཇི་ཙམ་དུ་ཡོངས་སུ་རྟོགས་པ་དེ་ཙམ་དུ་བདུད་ཀྱི་ལས་སོ། །ལྷའི་བུ་གཞན་ཡང་བྱང་ཆུབ་ཀྱི་སེམས་ལ་མངོན་པར་ཞེན་པ་ཡང་བདུད་ཀྱི་ལས་སོ། །སྦྱིན་པའི་སེམས་ལ་མངོན་པར་ཞེན་
པ་ཡང་བདུད་ཀྱི་ལས་སོ། །ཚུལ་ཁྲིམས་དང་། བཟོད་པ་དང་། བརྩོན་འགྲུས་དང་། བསམ་གཏན་དང་། ཤེས་རབ་ཀྱི་སེམས་ལ་མངོན་པར་ཞེན་པ་ཡང་བདུད་ཀྱི་ལས་སོ། །སྦྱིན་པ་རློམ་སེམས་སུ་བྱེད་པ་དང་། ཚུལ་ཁྲིམས་མཆོག་ཏུ་འཛིན་པ་དང་། བཟོད་པ་
