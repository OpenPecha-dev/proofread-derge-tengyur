མེ་མིན་ལས་ཀྱང་དུ་བ་འབྱུང་། །ཞེས་བྱ་བའོ། །དེ་ལ་ཞེས་བྱ་བ་ནི་མེ་མ་ཡིན་པ་ལས་བྱུང་བའི་དུ་བ་ལའོ། །རྒྱུ་མེད་པའི་དུ་བ་ལ་ཡང་མཐོང་བའི་སྟོབས་ཀྱི་ཡུལ་ལ་སོགས་པ་མ་ངེས་པར་འགྱུར་བ་ཁོ་ནའོ་ཞེས་བརྗོད་པ་ནི་དངོས་པོ་རྒྱུ་མེད་པ་ལ་ཡང་ཞེས་བྱ་བའོ། །དེར་ཡང་ཞེས་
བྱ་བ་ནི་དུ་བ་ལའོ་དེའི་ཕྱིར་དེ་ལྟར་འདོད་ཆགས་ལ་སོགས་པ་ཡང་ཐམས་ཅད་ཀྱི་རྒྱུ་ཅན་ནམ། རྒྱུ་མེད་པ་ཅན་དུ་འགྱུར་གྱི་ལུས་ཀྱི་རྒྱུ་ཅན་ནི་མ་ཡིན་ནོ། །ངེས་པ་ནི་ཡོངས་སུ་གྱུར་པའི་ཁྱད་པར་རམ། མཐོང་བ་ལས་འགྱུར་བ་ཡིན་ནོ་ཞེས་གནས་སོ། །གལ་ཏེ་ཡང་མ་ངེས་པའི་རྒྱུ་ཅན་ཉིད་ཀྱིས་
རྟག་ཏུ་ཡོད་པར་ཐལ་བ་ལས། མ་མཐོང་བ་ཉིད་ལན་ཡིན་པ་དེའི་ཚེ་རྣམ་པ་ཐམས་ཅད་དུ་མཐོང་བ་གང་ཡིན་པ་དེ་ཁོ་ན་ཡོད་པ་ཡིན་ནོ་ཞེས་གཞན་གྱིས་ཁས་བླང་བར་བྱ་བར་འགྱུར་ལ། དེ་ལྟ་ན་ཡང་གཉིས་སུ་མེད་པ་ཉིད་རྣམ་པར་གནས་པར་འགྱུར་རོ། །དེས་ན་འབད་པ་མེད་པར་ཐ་སྙད་
མཐའ་དག་རྣམ་པར་ཞིག་པར་འགྱུར་རོ་ཞེས་བརྗོད་པ་ནི་མཐོང་བ་ཙམ་ལ་བརྟེན་པ་ནི་ཡོད་པ་ཉིད་དུ་འཛིན་པ་སྟེ། དེ་ལྟར་བྱེད་ན་དེའི་ཚེ་རྣམ་པ་ཐམས་ཅད་ཀྱིར་མཐོང་བ་ནི་གཉིས་སུ་མེད་པ་ཁོ་ན་ཡིན་གྱི་འབྲས་བུ་ལ་སོགས་པ་ནི་མ་ཡིན་ནོ། །དེ་ལྟ་ཡིན་ན་མི་འདོད་པ་ཅི་ཡོད་སྙམ་ན་
འདིར་ནི་ཞེས་སྨོས་ཏེ་གཉིས་མེད་དུ་གྱུར་ནའོ། །འཇུག་པ་དང་ལྡོག་པའི་མཚན་ཉིད་དམ་རྒྱུ་དང་འབྲས་བུ་སྒྲུབ་པར་བྱེད་པའི་རྣམ་པར་རྟོག་པའི་མཚན་ཉིད་ཀྱི་ཐ་སྙད་འཇིག་པར་འགྱུར་རོ། །འོན་ཏེ་འདོད་ཆགས་ལ་སོགས་པ་ལུས་ཀྱི་རྒྱུ་ཅན་ཉིད་དུ་མཐོང་བ་ཡིན་ནོ། །རྗེས་སུ་འགྲོ་བ་
དང་ལྡོག་པའི་ཁྱད་པར་ཅན་གྱི་རང་རིག་པས་ནི་རྒྱུ་ལ་སོགས་པའི་རྣམ་པར་རྟོག་པ་ཐ་སྙད་ཀྱི་མིང་ཅན་གྲུབ་པ་ཡིན་ཏེ། དེས་ནི་རྒྱུ་ལ་སོགས་པ་གྲུབ་པ་ཡིན་ནོ་སྙམ་དུ་དོགས་པ་བསུ་ནས། འདོད་ཆགས་ལ་སོགས་པ་ནི་བདག་དང་བདག་གིར་འཛིན་པས་ཆགས་པའི་རྒྱུ་ཅན་ཡིན་གྱི། ལུས་ཀྱི་རྒྱུ་
ཅན་ནི་མ་ཡིན་ཏེ། དེ་ལྟར་རྟོགས་པའི་ཕྱིར་རོ་ཞེས་བརྗོད་པ་ནི་གལ་ཏེ་ཐ་སྙད་ཅེས་བྱ་བའོ། །གཞན་དུ་ཞེས་བྱ་བ་ནི་དོར་བར་ནུས་པའོ། །གལ་ཏེ་སྐྱེ་བ་དང་པོ་ལ་སོགས་པར་རང་རིག་པ་ལས་བྱུང་བའི་རྣམ་པར་རྟོག་པས་འདོད་ཆགས་ལ་སོགས་པ་ལུས་ཀྱི་རྒྱུ་ཅན་ཉིད་དུ་རྟོགས་པ་ཡིན་གྱི་འདས་པའི་
