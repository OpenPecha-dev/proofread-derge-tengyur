བྲལ་བ་མ་ཡིན་གྱི། འོན་ཀྱང་སྡུག་བསྔལ་དེ་བློ་ལ་གནས་པར་བྱས་པ་ལས་ཡིན་ནོ། །བློ་ལ་གནས་པར་བྱས་པ་དེ་ཡང་གནས་སྐབས་ཀུན་དུ་སྐྱེས་བུ་ལ་ཡོད་པའི་ཕྱིར་སྡུག་བསྔལ་བསྒོམས་པའི་ཉེ་བར་བསྟན་པས་ཅི་བྱ་སྟེ། དེ་དཔྱོད་ཀྱང་
ཤེས་པར་སླ་བའི་ཕྱིར་རོ་ཞེས་སྟོན་པ་ནི་དེ་དག་ཅེས་བྱ་བའི་སྐྱེས་བུ་དེ་དག་ནི་སྡུག་བསྔལ་བསྒོམས་པའི་ཉེ་བར་བསྟན་པས་ཀྱང་སྡུག་བསྔལ་ཉིད་དེ། སྡུག་བསྔལ་གྱིས་སྒྲུབ་པར་བྱེད་པ་དང་། སྡུག་བསྔལ་དང་རྗེས་སུ་འབྲེལ་པ་ཉིད་རྣམ་ཤེས་བྱེད་དེ་ཁོང་དུ་ཆུད་པར་བྱེད་
པ་ཡིན་ན། རང་བཞིན་ཉེ་བར་བསྟན་པ་ལས་སྔར་ཡང་མངོན་སུམ་བྱས་པ་ཉིད་དོ། །དེ་ནས་ཆགས་པ་དང་མ་བྲལ་བ་ཉེ་བར་བསྟན་པས་བྲལ་བར་བྱེད་པ་ཡང་མ་ཡིན་ཏེ། དེ་ཡང་དེ་ཉིད་རྗེས་སུ་བརྗོད་པ་ཙམ་ཡིན་པའི་ཕྱིར་རོ། །སྔར་ནི་ཆགས་པ་དང་བྲལ་བར་གྱུར་པ་དང་། ཆགས་པ་དང་བྲལ་
བའི་ཡུལ་མེད་པའི་ཕྱིར་བསྒོམ་པ་དོན་མེད་པར་བརྗོད་པ་ཡིན་ལ། དེ་ལྟ་ན་ནི་ཆགས་པ་དང་བྲལ་བའི་སྒྲུབ་བྱེད་ཉིད་མེད་པར་བརྗོད་པ་ཡིན་ནོ། །ཁོ་བོ་ཅག་གིས་ནི་འདུ་བྱེད་ཀྱི་སྡུག་བསྔལ་ཉིད་ལ་བསམས་ནས་བླང་བྱ་དང་དོར་བྱ་དང་བྲལ་བའི་ཕྱིར་སྟེའུ་ཙན་དན་མཉམ་པ་ཉིད་འདོད་ཆགས་དང་བྲལ་
བར་འགྱུར་བའི་ཕྱིར་བསྒོམ་པ་ཡིན་ནོ། །འདོད་ཆགས་དང་བྲལ་བའི་རྒྱུ་མ་ཡིན་པ་ཉིད་བརྗོད་པ་འདིས་ནི་ཞེས་བྱ་བའོ། །འདི་ཉིད་གསལ་བར་བྱེད་པ་ནི་སྡུག་བསྔལ་བསྒོམ་པ་ཞེས་བྱ་བའོ། །དེ་ལྟ་མ་ཡིན་ན་ཞེས་བྱ་བ་ནི་མངོན་དུ་བྱེད་པའི་རྒྱུ་མ་ཡིན་ནོ། །དེ་ལྟ་མ་ཡིན་ན་ནི་ཞེས་བྱ་
བ་ནི་སྡུག་བསྔལ་མངོན་སུམ་ཉིད་མ་ཡིན་ནའོ། །དེ་ལྟ་ན་ཡང་ཞེས་བྱ་བ་ནི་སྡུག་བསྔལ་མངོན་སུམ་ཉིད་ཡིན་ཡང་ངོ། །སྡུག་བསྔལ་བསྒོམ་པ་དོན་དང་བཅས་པར་དོགས་པ་བསུ་བ་ནི་གལ་ཏེ་སྡུག་བསྔལ་ཞེས་བྱ་བའོ། །དེ་བཞིན་དུ་བདེ་བ་བསྒོམས་པས་འདོད་ཆགས་པར་འགྱུར་
རོ་ཞེས་ལན་བཏབ་པ་ནི་དེ་ནི་མེད་དེ་ཞེས་བྱ་བའོ། །དེས་ན་ཞེས་བྱ་བ་བདེ་བའི་ཚོགས་རྣམ་པར་ཤེས་པས་སོ། །གལ་ཏེ་འཁོར་བ་པའི་བདེ་བ་མཐའ་དག་ནི་སྡུག་བསྔལ་གྱི་གཞི་ཡིན་པའི་ཕྱིར་ན་སྡུག་བསྔལ་ཉིད་མ་ཡིན་ནམ། འདི་ལྟར་བདེ་བའི་དོན་དུ་སྦྱོར་བ་ནི་དེ་བསྒྲུབ་པ་ལ་སོགས་
