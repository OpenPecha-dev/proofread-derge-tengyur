ལ་ཟླ་བ་ཕྱེད་བཅབས་པ་དེའི་སླད་དུ། ཟླ་བ་ཕྱེད་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་སྔ་མ་དང་འདྲ་བ་ཡང་བྱུང་ལ་བཅབས་པ་དེའི་སླད་དུ་གཞི་ནས་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་བར་མ་དང་
འདྲ་བ་ཡང་བཟླས་ཏེ་བྱུང་ལ། བཅབས་པའི་སླད་དུ་ཡང་གཞི་ནས་བསླང་སྟེ། ཡོངས་སུ་གནས་པ་ཡང་བགྱིས་ལགས་ཏེ། དགེ་འདུན་ལས་ཞག་དྲུག་མགུ་བར་བགྱི་བ་གསོལ་ནས། དགེ་འདུན་གྱིས་དགེ་སློང་འཆར་ཀ་ཆེ་གེ་མོ་ལས་བྱུང་བའི་དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་བྱུང་ལ་ཟླ་བ་ཕྱེད་བཅབས་
པ་དེའི་སླད་དུ་ཟླ་བ་ཕྱེད་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་སྔ་མ་དང་འདྲ་བ་ཡང་བྱུང་ལ་བཅབས་པ་དེའི་སླད་དུ་གཞི་ནས་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་བར་མ་དང་འདྲ་བ་ཡང་བཟླས་ཏེ་བྱུང་ལ་བཅབས་པ་དེའི་སླད་དུ་
ཡང་གཞི་ནས་བསླང་སྟེ་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས་པ་འདི་ལ་ཞག་དྲུག་མགུ་བར་བགྱིས་པ་སྩལ་ཏེ། དགེ་སློང་འཆར་ཀ་འདིས་ཆེ་གེ་མོ་ལས་བྱུང་བའི་དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་བྱུང་ལ་ཟླ་བ་ཕྱེད་བཅབས་པ་དེའི་སླད་དུ་ཟླ་བ་ཕྱེད་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས་ལགས། དགེ་འདུན་
ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་སྔ་མ་དང་འདྲ་བ་ཡང་བྱུང་ལ་བཅབས་པའི་སླད་དུ་གཞི་ནས་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས་ལགས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་བར་མ་དང་འདྲ་བ་ཡང་བཟླས་ཏེ་བྱུང་བ་བཅབས་པའི་སླད་དུ་ཡང་གཞི་ནས་བསླང་སྟེ་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས་ལགས། 
ཞག་དྲུག་མགུ་བ་ཡང་བགྱིས་ལགས་ཏེ་དགེ་འདུན་ལས་དབྱུང་བ་གསོལ་ན། གལ་ཏེ་དགེ་འདུན་གྱི་དུས་ལ་བབ་ཅིང་བཟོད་ན་དགེ་འདུན་གྱིས་གནང་བར་མཛོད་ཅིག་དང་། དགེ་འདུན་གྱིས་དགེ་སློང་འཆར་ཀ་ཆེ་གེ་མོ་ལས་བྱུང་བའི་དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་བྱུང་ལ་ཟླ་བ་ཕྱེད་བཅབས་པ་དེའི་སླད་
དུ་ཟླ་བ་ཕྱེད་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་སྔ་མ་དང་འདྲ་བ་ཡང་བྱུང་ལ་བཅབས་པའི་སླད་དུ་གཞི་ནས་ཡོངས་སུ་གནས་པ་ཡང་བགྱིས། དགེ་འདུན་ལྷག་མའི་ལྟུང་བ་ལྟུང་བ་བར་མ་དང་འདྲ་བ་ཡང་བཟླས་ཏེ་བྱུང་ལ་བཅབས་པའི་སླད་དུ་ཡང་གཞི་
