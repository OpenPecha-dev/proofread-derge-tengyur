ནི་སྨྱིག་མ་མཁན་གྱིས་བྱས་པའི་གདུགས་སོ། །དེ་ནི་སྐྱེ་བོ་ཐམས་ཅད་ཀྱི་དོན་དུ་གདུགས་ཁྲ་བོ་བཀྲ་བ་དག་བྱེད་པས་དེ་འདིར་ཉེས་པ་མེད་པར་བསྟན་པའི་ཕྱིར་སྨྱིག་མ་མཁན་གྱིས་བྱས་པའི་ཞེས་སྨོས་སོ། །ལོ་མ་ལས་བྱས་པའི་གདུགས་ནི་འཕྲལ་གྱི་རང་བཟོར་
བྱས་པ་ཡིན་ཏེ། དེ་གཉིས་གང་འབྱོར་པ་བཅང་བར་བྱའོ། །ཡུ་བ་ནི་ལུས་ཀྱི་ཚད་དོ་ཞེས་བྱ་བ་ནི་གདུགས་ཀྱི་ཡུ་བའི་སྲིད་ཀྱི་ཚད་ནི་ལུས་ཀྱི་སྲིད་ཀྱི་ཚད་དང་མཉམ་པར་བྱའོ། །དེས་གྲོང་དབུས་སུ་འགྲོ་བར་མི་བྱའོ་ཞེས་བྱ་བ་ནི། གདུགས་དེས་གྲོང་གི་དབུས་སུ་འགྲོ་བར་མི་བྱའོ། །གལ་
ཏེ་ལམ་གྱི་དབང་གིས་ནི་གློར་བསྙལ་ཏེའོ་ཞེས་བྱ་བ་ནི་གལ་ཏེ་ལམ་གྱི་དབང་གིས་གྲོང་དབུས་སུ་མི་འགྲོར་མི་རུང་ན། གདུགས་དེ་གློར་བསྙལ་ཏེ་འགྲོ་བར་བྱའོ། །ཆར་འབབ་པ་ཉིད་ནི་བསོད་སྙོམས་ལ་རྒྱུ་བར་བྱའོ་ཞེས་བྱ་བ་ནི། གདུགས་ཀྱིས་གྲོང་དབུས་སོ། །ཡུ་བ་ལ་གཞར་དུ་
མི་རུང་བ་ཉིད་མ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི། གོས་བསྐྱབ་པའི་ཕྱིར་ཡུ་བ་ལ་གཞར་ཞིང་ཡིབ་ཏུ་མི་རུང་བ་མ་ཡིན་ཏེ་རུང་བ་ཁོ་ནའོ། །ཡུ་བ་ནི་བྱི་དོར་བྱའོ། །ཆད་ན་ཁྱིམ་དག་ཏུ་གཞག་གོ་ཞེས་བྱ་བ་ནི། ཆར་པ་ཆད་ན་དེའི་ཚེ་དྲུང་ན་འདུག་པའི་ཁྱིམ་གང་ཡིན་པ་དེར་གདུགས་དེ་བརྟགས་ཏེ་གཞག་
པའོ། །འབྱུང་ཀར་བླང་ངོ་ཞེས་བྱ་བ་ནི། གང་དུ་བཞག་པའི་གྲོང་དེ་ནས་ཕྱིར་འབྱུང་ཀར་གདུགས་དེ་བླང་བའོ། །གཙུག་ལག་ཁང་ན་གནས་པའི་སྐབས་བཤད་ཟིན་ཏོ།།      །།ད་ནི་གནས་གཞན་ན་གནས་པའི་སྐབས་བརྗོད་པར་བྱ་སྟེ། རོལ་མོ་མཁན་གྱི་གནས་དང་། 
སྨད་འཚོང་མའི་གནས་དང་། ཆང་འཚོང་མའི་གནས་དང་། རྒྱལ་པོའི་ཕོ་བྲང་དང་། གདོལ་པའི་ཁྱིམ་གྱི་གནས་དག་བསྟེན་པར་མི་བྱའོ་ཞེས་བྱ་བ་ནི། སྤྱོད་ཡུལ་མ་ཡིན་པའི་གནས་ལྔ་པོ་དེ་དག་ཏུ་འཇུག་པར་མི་བྱའོ། །མི་གཙང་བའི་སྐྱབས་དང་ཉེ་བར་འདུག་པར་
མི་བྱའོ་ཞེས་བྱ་བ་ནི། འཇུག་པའི་ཕྱིར་སོང་ན་ཡང་དེའི་ཉེན་ཀོར་དུ་བསྡད་པར་མི་བྱའོ། །འདིར་གཞུང་ནི། གླེང་གཞི་དག་ལས་ཏེ་ལྟ་བས་ན་དགེ་སློང་གིས་ཆབ་ཁུང་སར་སོང་ན་འདུ་འཛིས་གནས་པར་མི་བྱའོ། །དེའི་ཉེན་ཀོར་དུ་ལུང་དབོག་པར་མི་བྱ། བཀླག་པར་མི་བྱ། ཁ་
