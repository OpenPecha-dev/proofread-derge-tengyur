རུང་སྟེ་སྔར་མགུ་བ་སྤྱད་པ་དེ་ཕྱིས་མཚན་གྱུར་པའི་གནས་སྐབས་ན་དོན་མེད་པ་ཉིད་ཡིན་ནོ། །འདིར་གཞུང་ནི་སྤོ་བའི་གཞིའི་ཞུ་བ་ལས་བཙུན་པ་རེ་ཞིག་དགེ་སློང་སྤོ་བ་ལ་ཐུག་པ་མཚན་གྱུར་ཏེ་དགེ་སློང་མར་གྱུར་ན། ཅི་དེས་གཉི་གའི་དགེ་འདུན་ལ་ཟླ་བ་ཕྱེད་མགུ་བ་སྤྱད་པ་སྤྱད་
པར་བགྱི་འཚལ་ལམ་སྤྱད་པར་བྱ་དགོས་སོ། །བཙུན་པ་རེ་ཞིག་དགེ་སློང་མ་གཉི་གའི་དགེ་འདུན་ལ་མགུ་བ་སྤྱད་པ་མཚན་གྱུར་ཏེ། དགེ་སློང་དུ་གྱུར་ན་ཅི་དེས་དགེ་འདུན་ལ་ཞག་དྲུག་མགུ་བ་སྤྱད་པ་སྤྱད་པར་བགྱི་འཚལ་ལམ། སྤྱད་པར་བྱ་དགོས་སོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །དེ་ལ་དགེ་
སློང་སྤོ་བ་ལ་ཐུག་པ་མཚན་གྱུར་ཏེ། དགེ་སློང་མར་གྱུར་ན་ཅི་དེས་གཉི་གའི་དགེ་འདུན་ལ་ཟླ་བ་ཕྱེད་མགུ་བ་སྤྱད་པ་སྤྱད་པར་བགྱི་འཚལ་ལམ་ཞེས་བྱ་བའི་མདོ་སྔ་མ་ཤོས་ཀྱིས་ནི་ཉེས་པ་བཅབས་པ་ཉམས་པར་བསྟན་ཏེ། འདི་ལྟར་མགུ་བ་སྤྱད་པ་འབའ་ཞིག་བསྟན་གྱི་སྤོ་བ་མ་བསྟན་པའི་
ཕྱིར་རོ། །དགེ་སློང་མ་གཉི་གའི་དགེ་འདུན་ལ་མགུ་བ་སྤྱད་པ་མཚན་གྱུར་ཏེ། དགེ་སློང་དུ་གྱུར་ན་ཅི་དེས་དགེ་འདུན་ལ་ཞག་དྲུག་མགུ་བ་སྤྱད་པ་སྤྱད་པར་བགྱི་འཚལ་ལམ་ཞེས་བྱ་བའི་མདོ་ཕྱི་མ་ཤོས་ཀྱིས་ནི་མགུ་བ་སྤྱད་པ་དོན་མེད་པར་བསྟན་ཏེ། འདི་ལྟར་སྔར་གཉི་གའི་དགེ་འདུན་
ལ་ཟླ་བ་ཕྱེད་མགུ་བ་སྤྱད་དུ་ཟིན་ཀྱང་ཡང་དགེ་འདུན་ལ་ཞག་དྲུག་མགུ་བ་སྤྱད་པར་བྱ་དགོས་པར་བསྟན་པའི་ཕྱིར་རོ། །ཅིའི་ཕྱིར་བྱས་པ་ནི་དེ་ཉིད་ཡིན་ནོ། །ཞིང་མ་ཡིན་པ་ལ་ནི་ཉམས་སོ། །མགུ་བ་སྤྱད་པ་ནི་དོན་མེད་པ་ཉིད་ཡིན་ནོ་ཞེ་ན། འདི་ལྟར་འདི་ལ་ཆོ་ག་རྣམས་གྱུར་ཏོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། མཚན་
གྱུར་པ་འདི་ལ་གང་གི་ཕྱིར་ཆོ་ག་སྤྱད་པ་རྣམས་གྱུར་པ་དེའི་ཕྱིར་དེ་དག་རྣམ་པར་གཞག་པ་མཛད་དོ། །དེའི་ཚེའི་ལུས་བྱེད་པ་པོ་ཉིད་ཡིན་ནོ་ཞེས་བྱ་བ་ནི། གང་གི་ཚེ་གསང་བར་བྱ་བ་ལ་སོགས་པ་ཡུལ་གང་ལས་ལྟུང་བ་འབྱུང་བར་འགྱུར་བའི་ཡུལ་གྱི་ལུས་གང་ཡིན་པ་དེ་དང་ལྟུང་བ་
འབྱིན་པར་བྱེད་པའི་རྒྱུད་ཀྱིས་ལུས་ཀྱི་དངོས་པོ་མཚན་གྱུར་ཏེ། ཕོའམ་མོའི་ལུས་སུ་གནས་པའི་ལུས་དེ་ཉིད་ལྟུང་བ་དེའི་བྱེད་པ་པོ་ཉིད་ཡིན་ཏེ། ཡུལ་གང་ལས་ལྟུང་བ་འབྱུང་བའི་ལུས་དང་རྒྱུད་གང་ལ་ལྟུང་བ་འབྱུང་བའི་ལུས་དེ་གཉིས་གནས་སྐབས་གྱུར་པ་ན་གང་གིས་གནས་སྐབས་
