བྱུང་ངོ་བྱུང་ངོ་ཞེས་སྨྲས་པ་དང་། ཚོང་པ་དོན་མཐུན་དག་གིས་དེ་ཐོས་ནས་བསམས་པ། སེང་གེ་བྱུང་ངོ་སྙམ་ནས་དེ་དག་སྐྲག་སྟེ་ཕྱོགས་བཞིར་བྱེར་རོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཅིའི་ཕྱིར་
ཚོང་པ་དོན་མཐུན་འདི་དག་ཕྱོགས་བཞིར་བཞིར་བྱེར། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོས་སྐབས་དེ་རྒྱས་པར་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་སོང་ལ་ཚོང་པ་དོན་མཐུན་དག་ལ་དེ་བཞིན་གཤེགས་པ་ནི་སེང་གེའི་འཇིགས་པ་དང་བྲལ་བ་ཡིན་གྱི་ཕྱིར་ལོག་
ཤིག་ཅེས་བསྒོ་ཤིག །ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོས་དེ་དག་ལ་དེ་སྐད་བསྒོ་ནས་ཕྱིར་ལོག་པར་གྱུར་ཏོ། །དྲུག་སྡེ་དག་གིས་ཚོང་པ་དོན་མཐུན་གྱི་ཚོགས་སྐྲག་པར་བྱས་ཏེ། ཕྱོགས་བཞིར་བྱེར་བ་བཅོམ་ལྡན་འདས་ཀྱིས་དབུགས་ཕྱུང་བར་གྱུར་པའི་སྐབས་དེ་ལས་བཅོམ་ལྡན་འདས་
ཀྱིས་སྔོན་རི་བོང་དྲུག་ཅིག་མཚེའུ་འགྲམ་ན་ཤིང་བིལ་བའི་ཚལ་གསེབ་ན་འདུག་པ་ན་ཤིང་བིལ་བ་ཞིག་མཚེའུའི་ནང་དུ་རྡིབ་པའི་སྒྲ་ཅལ་ཞེས་ཟེར་བ་རི་བོང་དྲུག་གིས་ཐོས་ནས་དེ་དག་འབྲོས་པར་བརྩམས་པ། ཝ་དང་སྦྲེ་དང་རི་དྭགས་དང་ཕག་དང་། མ་ཧེ་དང་བསེ་དང་གླང་པོ་ཆེ་དང་
དོམ་དང་དྲེད་དང་གུང་དང་། སྟག་དང་། སེང་གེ་དག་གིས་བརྒྱུད་མར་ཐོས་ནས་དེ་དག་ཀྱང་འབྲོས་པར་བརྩམས་པ་ན། རི་བོང་དྲུག་དང་ཉེ་བ་ཞིག་ན་སེང་གེ་རལ་པ་ཅན་ཅོད་པན་ལྟ་བུ་དང་ལྡན་པ་ཞིག་འདུག་པ་དེས་ཕྱིར་ལོག་ཅིང་དབུགས་ཕྱུང་སྟེ། རི་བོང་དྲུག་ནི་དྲུག་སྡེ་དག་ཡིན་ལ། ཝ་
ལ་སོགས་པ་རི་དགས་ཀྱི་ཚོགས་ཆེན་པོ་སྐྲག་སྟེ་བྱེར་བ་དག་ནི་ཚོང་པ་དག་ཡིན་ནོ། །སེང་གེ་རལ་པ་ཅན་ཅོད་པན་ལྟ་བུ་དང་ལྡན་པ་དག་གིས་རི་དྭགས་ཀྱི་ཚོགས་ཆེན་པོ་སྐྲག་པ་ཕྱོགས་བཞིར་བྱེར་བ་དག་དབུགས་ཕྱུང་བར་གྱུར་པ་དེ་ནི་དེ་བཞིན་གཤེགས་པ་ཉིད་ལགས་པའི་བཀའ་མཆིད་ཀྱང་ 
སྩལ་ཏོ། །དེ་ནི་རེ་ཞིག་བྱུང་བ་ཡིན་ཏེ། སངས་རྒྱས་བཅོམ་ལྡན་འདས་རེ་ཞིག་འདུལ་བ་ལ་ཉན་ཐོས་རྣམས་ཀྱིས་བསླབ་པའི་གཞི་འཆའ་བར་ནི་མི་མཛད་དོ། །བཅས་པའི་གླེང་གཞི་ནི་དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་ལ་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་
