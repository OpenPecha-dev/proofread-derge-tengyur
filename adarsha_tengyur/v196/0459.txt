ཊུ་ཡཱི་ཙྼི། ཡཱ་ཡ་ཙཱི་ཏི། ཙ་སྡེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཀིའོ། །ཡཱ་ཡཱཀྟི། ཡཱ་ཡཀྟཿ། ཡཱ་ཡཱ་ཙ་ཏི། ཡཱ་ཡཱ་ཙཱི་ཥི། ཡཱ་ཡཀྵི། ཡཱ་ཡཱཀྠཿ ཡཱ་ཡཱཀྠ། ཡཱ་ཡཱ་ཙཱ་མི། ཡཱ་ཡཱ་ཙྨི། ཡཱ་ཡཱ་ཙྭིཿ། ཡཱ་ཡཱ་ཙྨཿ། བྷ་སྲྀ། བཱ་བྷཱ་སྲི་ཏི། བཱ་
བྷ་སྟི། བཱ་བྷཱསྟཿ། བ་དྷ་ས་ཏི། བཱ་བྷ་སི་ཥི། བཱ་བྷསྶི། བཱ་བྷཱསྠཿ། བཱ་བྷཱ་སཱི་མི། བཱ་བྷཱསྨི། བཱ་བྷཱ་སྭཿ། བཱ་བྷཱསྨཿ། ག་ཧུ། ཛ་གཱ་ཧཱི་ཏི། དྷུཊ་ལ་ཧའི་ཌྷའོ། །ཏ་ཐ་དག་གི་དྷ་ཉིད་དོ། །ཥ་དང་ཊ་སྡེ་ལས་ཏ་སྡེའི་ཊ་སྡེའོ། །ཌྷ་ལ་དྷའི་དབྱིའོ། །ཛཱ་གཱ་ཌྷི། ཛཱ་གཱ་
ཧ་ཏི། ཛཱ་གཱ་ཧི་ཥི། སྔ་མ་བཞིན་དང་པོ་གའི་བཞི་པ་གྷ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །ཛཱ་གྷཱཀྵི། ཛཱ་གཱ་ཌྷཿ། ཛཱ་གཱ་ཌྷ། ཛཱ་གཱ་ཧི་མི། ཛཱ་གཱ་ཧྨི། ཛཱ་གཱ་ཧྭཿ། ཛཱ་གཱ་ཧྨཿ། བིད། ཨ་བྷྱསྟའི་ཡང་ཨུ་པ་དྷཱ་ན་མའི་
བྱིངས་ཐམས་ཅད་པ་ཡོན་ཏན་ཅན་ལ་དབྱངས་ལའོ་ཞེས་སོགས་ལ་ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་བཀག་གོ། །བི་བཱི་དཱི་ཏི། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །བེ་བེཏྟི། བེ་བེཏྟཿ། བེ་བི་ད་ཏི། བེ་བི་དཱི་ཥི། བེ་བེངྶ། བེ་བིཏྠཿ། བེ་བིཏྠ། བེ་བེ་དཱི་མི། བེ་བེདྨི། བེ་

ལེ་ལཱི་ཌྷ། ལེ་ལ་ཧཱི་མི། ལེ་ལི་ཧྭཿ། ལེ་ལི་ཧྨཿ། མི་ཧ། མེ་མི་ཧཱི་ཏི། མེ་མེ་ཌྷི། མེ་མཱི་བྷཿ། མེ་མི་ཧ་ཏི། དི་བ། དེ་དི་བཱི་ཏི། ཡ་བ་དག་གི་གསལ་བྱེད་ལ་ཡ་མ་ཡིན་པ་ལའོ་ཞེས་པས་བ་དབྱིའོ། །དེ་དེ་ཏི། ཚ་བ་དག་གི་ལྔ་པ་ལ་ཡང་ཤ་ཨཱུཊའོ་ཞེས་པས་བ་ཡིག་གི་
ཡོན་ཏན་མེད་པ་ལ་ཨཱུཊའོ། །ཊ་ཡིག་ནི་ཨ་ཡིག་ལས་ཨུཊའི་འཕེལ་བའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །ཨིའི་ཡི་གེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཡ་སྟེ་ཕ་རོལ་དབྱི་བའང་མ་ཡིན་ནོ་ཞེས་པས་ཨི་ཡིག་གི་ཡ་ཉིད་དོ། །དེ་དྱུ་ཏཿ། དེ་དི་བ་ཏི། དེ་དི་བཱི་ཥི། དེ་དྱུ་ཐཿ། དེ་དྱུ་ཐ།
