འགྱུར་བ་དེའི་ཚེ། པྲི་ཡ་ཙ་ཏྭཱ་ར། པྲི་ཡ་ཙ་ཏྭཱ་རོོ། པྲི་ཡ་ཙ་ཏྭཱ་རཿ། ཛ ། པྲི་ཡ་ཙ་ཏྭཱ་རཾ། པྲི་ཡ་ཙ་ཏྭཱ་རོོ། །ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲི་ཡ་ཙ་ཏུ་རཿ། པྲི་ཡ་ཙ་ཏུ་རཱ། པྲི་ཡ་ཙ་ཏུརྦྷྱཱཾ། པྲི་ཡ་ཙ་ཏུརྦྷིཿ། དྲུག་པའི་མང་ཚིག་ལ་ནུ་མེད་དེ། པྲི་ཡ་ཙ་ཏུ་རཱཾ། ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལ་གཉི་ག་ཐུང་བའོ། །
ཧེ་པྲི་ཡ་ཙ་ཏྭཿ། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རོོ། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རཿ། དེ་བཞིན་དུ། བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ་ནི། ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲི་ཡ་ཙ་ཏུཿ། པྲི་ཡ་ཙ་ཏུ་རཱི། པྲི་ཡ་ཙ་ཏྭཱ་རི། སླར་ཡང་། དེ་བཞིན་དུ་ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་དང་མཚུངས་སོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་པྲི་ཡ་ཙ་ཏུཿ།
ཧེ་པྲི་ཡ་ཙ་ཏུ་རཱི། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རི།།      །།ཀ་ཏིའི་སྒྲའི་ནི། ཀ་ཏི་ལ་ཡང་ཛས་ཤས་དག་འཇིག་གོ། །ཀ་ཏི། དྲུག་རྣམས། ཀ་ཏི། དྲུག་རྣམས་ལ། ཀ་ཏི་བྷིཿ།-དྲུག་རྣམས་ཀྱིས། ཀ་ཏི་བྷྱཿ། ཀ་ཏི་བྷྱཿ། ཨཱམ་ལའང་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་སྟེ། ནུ་དང་བཅས་པའི་ཨཱམ་ལ་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཀ་ཏི་ནཱཾ། ཀ་ཏི་ཥུ། དེ་
བཞིན་དུ། བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ་ནི། ཀ་ཏི། ཀ་ཏི། གཟུགས་སོ་སོ་བ་དང་དབྱེ་བ་མེད་ཅིང་། དྲུག་པའི་མང་ཚིག་ལ་ནཱ་མིན་ལས་དབྱངས་ལའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ། །ཀཱི་ཏཱི་ནཱཾ།།      །།པཉྩན་ལ་སོགས་པ་ན་ཡིག་གི་མཐའ་ཅན་རྣམས་ཀྱི། གྲངས་ལས་
ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ཀ་ཏི་ལ་ཡང་ཛས་ཤས་དག་འཇིག་གོ། །བུད་མེད་ལ་ཡང་བུད་མེད་ཉིད་ཁོ་ནའི་དངོས་པོ་མ་ཡིན་པ་ལ་ཨཱིའི་རྐྱེན་མེད་དེ། རྟགས་གསུམ་ཀ་རྣམས་ལ་མཚུངས་པ་ཉིད་ཀྱི་གཟུགས་སོ། །པཉྩ། ཛས། པཉྩ། ཤས། པཉྩ་བྷིཿ། པཉྩ་བྷྱཿ། པཉྩ་བྷྱཿ། གྲངས་
ལས་ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་དང་། ན་མཐའི་འང་ཨུ་བ་དྷཱའིའོ་ཞེས་པས་རིང་པོ་ཉིད་དང་། རྟགས་མཐའི་ན་ཡིག་གིའོ་ཞེས་པས་ན་དབྱི་བའོ། །པཉྩ་ནཱཾ། པཉྩ་སུ།།      །།ཥཥྚའི་སྒྲའི་ནི། ཧ་དང་ཤ་དང་ཥ་དང་ཚ་མཐའ་དང་ཡ་ཛ་སོགས་རྣམས་ཀྱི་ཌའོ་ཞེས་པས་
ཌ་ཉིད་དེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་རྫོགས་པ་ལ་ཊ་ཡིག་གམ་ཌ་ཡིག་དག་གོ། །ཥཊ།—དྲུག་རྣམས། ཥཌ།-དྲུག་རྣམས་ལ། སླར་ཡང་དེ་བཞིན་དུ། སྔ་མ་བཞིན་དུ་ཌ་ཉིད་དོ། །ཥཌྦྷིཿ། ཥདྦྷྱཿ། ཥདྦྷྱཿ། གྲངས་ལས་ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་བྱས་པ་ན་ཥ་ཡིག་གི་ཌ་ཡིག་
