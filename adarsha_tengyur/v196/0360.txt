ཏན་ནོ། །ཛ་ཛཱ་ག་ར་ཏུཿ། ཛ་ཛཱ་ག་རུཿ། རྀ་ཡིག་ལ་ཐལ་ལའོ་ཞེས་པས་ཨིཊ་མེད་པ་ནི་འདིར་དབྱངས་གཅིག་པ་ལས་ཡིན་པའི་ཕྱིར་རོ། །ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཛ་ཛཱ་ག་རི་ཐ། ཛ་ཛཱ་ག་ར་ཐུཿ། ཛ་ཛཱ་ག་ར། ཛ་ཛཱ་ག་རི་བ། ཛ་ཛཱ་ག་རི་མ། ཉི་བྷཱི། ཧྲཱི་ངོ་ཚ་བ་ལའོ། །ཌུ་བྷྲྀ་ཉ། ཧུ་འདི་རྣམས་ལ་
བྷི་ཧྲཱི་བྷྲྀ་ཧཱུ་རྣམས་ཀྱི་ཏི་བཞིན་ཡང་ངོ་། །ཞེས་པས་པ་རོཀྵཱ་ཕ་རོལ་ལས་ཡང་ན་ཨཱམ་འགྱུར་ཞིང་ཏི་བཞིན་ནོ། །དེས་ན་ཛུ་ཧོ་ཏི་སོགས་རྣམས་ཀྱི་བྱིངས་ཐམས་ཅད་པ་ལའོ། །ཞེས་པས་གཉིས་བརྗོད་དོ། །བྷྲྀ་ཉ་ཧཱང་མཱང་རྣམས་ཀྱི་ཨིཏའོ། །ཞེས་པས་བྷྲྀ་ཉའི་ཨ་བྷྱཱ་སའི་ཨི་ཉིད་དང་ཀྲྀ་ཉ་རྗེས་སུ་སྦྱོར་
བ་ལ་སོགས་པའོ། །ཨེ་ཨ་ཡའོ། །བི་བྷ་ཡཱཾ་ཙ་ཀཱ་ར། བི་བྷ་ཡཱ་མཱ་ས། བི་བྷ་ཡཱཾ་བ་བྷཱུ་བ། ཨཻ་ཨཱ་ཡའོ། །བི་བྷཱ་ཡ། ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །བི་བྷྱཱ་ཏུཿ། བི་བྷྱཱུ། རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ། །ཞེས་པ་ལ་སོགས་པས་ཐལ་ལ་ཡང་ན་ཨི་ཊའོ། །བི་བྷ་ཡི་ཐ། བི་
བྷེ་ཐ། བི་བྷྱ་ཐུཿ། བི་བྷྱ། བི་བྷཱ་ཡ། སྲྀ་ལ་སོགས་པའི་ངེས་པ་ལས་ཨི་ཊའོ། །བི་བྷྱི་བ། བི་བྷྱི་མ། ཛི་ཧྲ་ཡཱཾ་ཙ་ཀཱ་ར། ཛི་ཧྲ་ཡཱ་མཱ་ས། ཛི་ཧྲ་ཡཱཾ་བ་བྷཱུ་བ། ཛི་ཧྲཱ་ཡ། དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་པས་ཨི་ཡའོ། །ཛི་ཧྲ་ཡ་ཏུཿ། ཛི་ཧྲི་ཡུཿ། ཛི་ཧྲ་ཡི་ཐ། ཛི་ཧྲི་ཐ། ཛི་ཧྲི་ཡ་ཐུཿ། ཛི་ཧྲི་ཡ། ཛི་
ཧྲཱ་ཡ། བི་བྷ་རཱཾ་ཙ་ཀཱ་ར། བི་བྷ་རཱཾ་ཙ་ཀྲེ། བི་བྷ་རཱ་མཱ་ས། བི་བྷ་རཱཾ་བ་བྷཱུ་བ། བ་བྷཱ་ར། བ་བྷཱ་བ། རྀའི་ཡི་གེ་རའོ། །བ་བྷྲ་ཏུཿ། བ་བྷུཿ། སྲྀ་ལ་སོགས་པའི་མདོ་ཡིས་ཨིཊ་མེད་དོ། །བ་བྷྲ་ཐ། བ་བྷྲ་ཐུཿ། བ་བྷྲ། བ་བྷཱ་ར། བ་བྷྲྀ་བ། བ་བྷྲྀ་མ། བ་བྷྲེ། བ་བྷྲཱ་ཏེ། བ་བྷྲི་རེ། བ་བྷྲི་ཥེ། བ་བྷྲཱ་ཐེ། ནཱ་མི་
