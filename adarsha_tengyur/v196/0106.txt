ཛས་ཤས་དག་གི་ཤིའོ། །ཙ་ཏྭཱ་རི། ཙ་ཏྭཱ་རི། ལྷག་མ་སྐྱེས་པ་དང་མཚུངས་པའོ། །པཉྩ་ནའི་སྒྲ་ལས་ཛས་ཤས་དག་འཇིག་གོ། །རྟགས་མཐའི་ན་ཡིག་གིའོ་ཞེས་པས་ན་དབྱིའོ། །པཉྩ། པཉྩ། པཉྩ་བྷིཿ། པཉྩ་བྷྱཿ། པྲཉྩ་བྷྱཿ། གྲངས་ལས་ཥ་དང་ན་མཐའ་ལས་སོ་ཞེས་པས་གྲངས་ལ་ཥ་
ཡིག་དང་ན་མཐའ་ཅན་ལས་ཨམ་ཕ་རོལ་ལ་ནུ་ཨཱ་ག་མའོ། །ན་མཐའི་འང་ཨུ་པ་དྷཱའིའོ་ཞེས་པས་རིང་པོའོ། །རྟགས་མཐའི་ན་ཡིག་གོ་ཞེས་པས་རང་བཞིན་གྱིས་ན་ཡིག་དབྱིའོ། །པཉྩ་ནཱཾ། པཉྩ་སུ། བུད་མེད་དང་མ་ནིང་དག་ལའང་དེ་བཞིན་ནོ། །ཥ་ཥའི་སྒྲ་ལས་སྔ་མ་བཞིན་དུ་ཛས་ཤས་
དག་འཇིག་གོ། །ཧ་ཤ་ཥ་ཚ་མཐའ་དང་ཡ་ཛ་སོགས་རྣམས་ཀྱི་ཌའོ་ཞེས་པས་ཌ་ཉིད་དོ། །རྫོགས་པ་ལ་ཡང་ནའོ། །ཥཊ།ཥཌ། ཥཊ། ཥཌ། ཥཌྦྷིཿ། ཥཌྦྷྱཿ། ཥཌྦྷྱཿ། སྔ་མ་བཞིན་དུ་ནུ་དང་ཌ་ཉིད་དོ། །ཥཌའི་ཎ་ལ་ནའོ་ཞེས་པས་ཌའི་ཎ་ཉིད་དོ། །ཏ་སྡེའི་ཙ་ཊའི་སྡེ་པ་སྦྱོར་བ་ལ་ཙ་
ཊའི་སྡེ་པ་དག་གོ་ཞེས་པས་ན་ཡིག་གི་ཎ་ཉིད་དོ། །ཥཎྞཱམ། སྒྲ་མེད་པ་ལ་དང་པོའོ། །ཥཊྶུ། བུད་མེད་དང་མ་ནིང་དག་ལའང་དེ་བཞིན་ནོ། །སཔྟནའི་སྒྲ་ནི་པཉྩནའི་སྒྲ་དང་འདྲ་སྟེ། སཔྟ། སཔྟ། སཔྟ་བྷིཿ། སཔྟ་བྷྱཿ། སཔྟ་བྷྱཿ། སཔྟ་ནཱཾ། སཔྟ་སུ། དེ་བཞིན་དུ་བུད་མེད་དང་མ་ནིང་དག་
ལའང་ངོ་། ཨཥྚནའི་སྒྲ་ལས་ས་ལ་སོགས་པ་རྣམས་སོ། །དེ་ལས་ཛས་ཤས་དག་གི་ཨཽའོ་ཞེས་པ་དེ་ལས་ཏེ། ཨཥྚན་ཨ་ཡིག་ཏུ་བྱས་པ་ལས་ཕ་རོལ་གྱི་ཛས་ཤས་དག་གི་གནས་སུ་ཨཽའོ། །ཨཥྚཽ། ཨཥྚཽ། ཨཥྚ་ནའི་ཐམས་ཅད་ལའོ་ཞེས་པས། ཨཥྚ་ནའི་རྣམ་དབྱེ་རྣམས་ཐམས་
ཅད་ལ་ཨཱ་ཉིད་དོ། །ཨཥྚཱ་བྷིཿ། ཨཥྚཱ་བྷྱཿ། ཨཥྚཱ་བྷྱཿ། ཨཥྚ་ནཱཾ། ཨཥྚཱ་སུ། དེ་ལས་ཛས་ཤས་དག་གི་ཨཽའོ། །ཞེས་པས་འདིར་དེ་ལས་སྨོས་པ་ནི། ཨཱ་ཉིད་རྟག་ཏུ་བ་མ་ཡིན་པའི་དོན་ཏོ། །དེས་ན་ཕྱོགས་ལ་ཨཱ་ཉིད་དུ་འགྱུར་བ་ཡིན་ནོ། །ཨཥྚ། ཨཥྚ། ཨཥྚཱ་བྷིཿ། ཨཥྚཱ་བྷྱཿ། ཨཥྚཱ་བྷྱཿ། 
ཨཥྚ་ནཱཾ། ཨཥྚཱ་སུ། བུད་མེད་དང་མ་ནིང་དག་ལའང་དེ་བཞིན་ནོ། །ན་ཝན་དང་། ད་ཤན་དང་། ཨེ་ཀ་ད་ཤན་དང་། དྭ་ད་ཤན་དང་། ཏྲ་ཡོ་ད་ཤན་དང་། ཙ་ཏུར་ད་ཤན་དང་། པཉྩ་ད་ཤན་དང་། ཥ་ཌ་ད་ཤན་དང་། སབྟ་ད་ཤན་དང་། ཨཥྚ་ད་ཤནའི་སྒྲ་རྣམས་པཉྩའི་སྒྲ་དང་འདྲ་བར་དྲང་བར་བྱའོ། །
