ཉིད་དོ། །ཨཥྚཱད་ཤ །བཅོ་བརྒྱད། ན་བིཾ་ཤ་ཏི་ཞེས་པ་ན་ཉ་བསྡུས་པ་སྟེ། གཅིག་གིས་མ་ཚང་བའི་ཉི་ཤུའོ་ཞེས་པ་གསུམ་པའི་ཚིག་སྡུད་དོ། །ཨེ་ཀའི་སྒྲ་ལ་ཏའི་ཡི་གེ་ཨཱ་ག་མའོ། །ན་ནི་འགའ་ཞིག་ཏུའོ་ཞེས་པ་འདིར། འགའ་ཞིག་སྨོས་པ་ལས་ན་ཡིག་མི་འཇིག་སྟེ། ལྔ་པ་ལ་ལྔ་པ་རྣམས་སུ་སྟེ་ཡང་ན་མ་ཡིན་པར་གསུམ་པ་རྣམས་སུའོ་
ཞེས་པས་ཏ་ཡིག་གི་ད་ཡིག་གམ་ན་ཡིག་དག་གོ། །ན་ཉ་འདི་ལས་ནལ་ནུའོ། །ཨེ་ཀཱནྣ་བིཾ་ཤ་ཏིཿ། བཅུ་དགུ། ཨེ་ཀཱནྣ་བིཾ་ཤ་ཏིཿ། བཅུ་དགུ། དེ་ནས་ཡང་ན་གཅིག་གིས་མ་ཚང་བའམ་ཡང་ན་གཅིག་མ་ཚང་བ་སྟེ་དེ་རྣམས་ཀྱིས་འདི་དག་བིཾ་ཤ་ཏི་ལའང་ངོ་ཞེས་པས་ཨཱུ་ན་བིཾ་ཤ་ཏིཿ། བཅུ་དགུ འདི་དག་ལ་སོགས་པ་དགུའི་མཐར་ཐུག་པ་རྣམས་བུད་མེད་ཀྱི་རྟགས་
རྣམས་ཏེ། གཅིག་ཚིག་གི་ཡུལ་རྣམས་སོ། །དེ་བཞིན་དུ་རྟགས་མཐའ་བརྗོད་ཅིང་བར་དུ་སྦྱོར་བ་རྣམས་སུ་ཡང་གྲངས་རྣམས་པ་གང་གི་ཚེ། བོ་ཤ་ཏིཿ། ཉི་ཤུ་ཐམ་པ། ལ་སོགས་པ་རྣམས་ཀྱི་བཞིན་དུ་ཚོགས་པ་རྣམས་རྣམ་པར་འཆད་པ་དེའི་ཚེ། རྣམ་དབྱེ། ཐམས་ཅད་ལ་བརྗོད་པར་འགྱུར་ཏེ་ རྟགས་སོ་སོའི་སྐབས་སུ་སྔར་བུད་མེད་ལ་བསྟན་པ། བྷཱུ་མའི་སྒྲ་དང་འདྲ་བར་དྲང་ངོ་། །ཉི་ཤུ་སྐྱེས་པ་དང་ཉི་ཤུ་
བུད་མེད་དང་། ཉི་ཤུ་མ་ནིང་དང་། ཉི་ཤུ་སྐྱེས་པ་གང་གི་ཡིན་པ་རྣམས་དང་། ཉི་ཤུ་བུད་མེད་གང་གི་ཡིན་པ་རྣམས་དང་། ཉི་ཤུ་མ་ནིང་གང་གི་ཡིན་པ་རྣམས་དང་། ཉི་ཤུ་སྐྱེས་པ་མ་ཡིན་པ་རྣམས་ཀྱིས་བྱས་པ་དང་། ཉི་ཤུ་བུད་མེད་མ་ཡིན་པ་རྣམས་ཀྱིས་དང་། ཉི་ཤུ་མ་ནིང་མ་ཡིན་པ་རྣམས་
ཀྱིས་སོ། །ཐུང་ངུའི་ཡང་ང་ལྡན་ལའོ་ཞེས་པས་ སྔར་བུད་མེད་ལ་བསྟན་པ། ན་དཱི་དང་འདྲ་བར་འགྱུར་བ་སྟེ། ཉི་ཤུ་སྐྱེས་པའི་ཆེད་དང་། ཉི་ཤུ་སྐྱེས་པ་རྣམས་ཀྱི་ཆེད་དུའོ་ཞེས་པ་དང་། ཡང་ན་ཉི་ཤུ་བུད་མེད་ཀྱི་ཆེད་དང་། ཉི་ཤུ་བུད་མེད་རྣམས་ཀྱི་ཆེད་དང་། ཉི་ཤུ་མ་ནིང་རྣམས་ཀྱི་ཆེད་དུའོ་ཞེས་པ་ལ་སོགས་པའི་རབ་ཏུ་སྦྱོར་བ་རྣམས་འདིར་བརྗོད་དོ། །གང་གི་ཚེ་ཉི་ཤུ་ཉིད་
ལ་ཕྱེ་བ་དེའི་ཚེ། ཉི་ཤུ་དང་། གཅིག་དང་། གཉིས་དང་། ཉི་ཤུ་ཞེས་པ་དང་། བཞི་དང་ཉི་ཤུ་རྣམས་ལ་གཅིག་གིས་ལྷག་པའི་ཉི་ཤུའོ་ཞེས་སོགས་སྔར་བཞིན་དུ་བར་གྱི་ཚིག་ཕྱིས་པའི་བསྡུ་བའོ། །ཨེ་ཀཱ་བིཾ་ཤཏི། ཉི་ཤུ་རྩ་གཅིག །གཉིས་པ་ལ་ཨཱ་ཉིད་དེ། དྭཱ་བིཾ་ཤཏི། ཉི་ཤུ་རྩ་གཉིས། ཏྲ་ཡོ་བིཾ་ཤ་ཏི། ཉི་ཤུ་རྩ་གསུམ། ཙ་ཏུ་བིཾ་ཤ་ཏི། ཉི་ཤུ་རྩ་བཞི། པཉྩ་བིཾ་ཤ་ཏི། ཉི་སྐུ་རྩ་ལྔ། གསལ་བྱེད་མཐའི་གང་
སུ་བྷ་དག་ལའོ་ཞེས་བསྟན་པ་རྣམས་དང་། ཧ་དང་ཤ་དང་ཥ་དང་ཚ་མཐའ་དང་ཞེས་པ་ལ་སོགས་པ་ཌ་ཉིད་དོ། །ཥ་ཌ་བིཾ་ཤ་ཏི། ཉི་ཤུ་རྩ་དྲུག །སཔྟ་བིཾ་ཤ་ཏི། ཉི་ཤུ་རྩ་བདུན། སྔ་མ་བཞིན་དུ་ཨཱ་ཉིད་དེ། ཨཱཥྚཱ་བིཾ་ཤ་ཏི། ཉི་ཤུ་རྩ་བརྒྱད། སྔར་དང་འདྲ་བར་ཏྲིང་ཤད་ཅེས་པ་ན་ཉའི་ཚིག་སྡུད་དེ་དེ་ལྟ་
