ཤེས་པ་ལའོ། །ཨ་བེ་དི། ཨ་བེ་དི་ཥཱ་ཏཱཾ། དུ་ཧ། ཨ་དོ་ཧི། ཨཱ་ཏཱ། མ་སོགས་ལ་བྱེད་པ་པོ་བཞིན་ནོ། །ཨ་དྷུཀྵཏཱཾ། ཨ་དྷུཀྵནྟ། དི་ཧ། ཨ་དེ་ཧི། ཨ་དྷིཀྵཱ་ཏཱཾ། ཨ་དྷིཀྵནྟ། ལི་ཧ། ཨ་ལེ་ཧི། ཨ་ལིཀྵཱ་ཏཱཾ། ཨ་ལིཀྵནྟ། ཤུ་ཙ། ཨ་ཤོ་ཙི། ཨ་ཤོ་ཙིཥཱ་ཏཱཾ། ཨ་ཤོ་ཙི་ཥཏ། ཀྲྀ་ཥ། ཨིཙ་དང་ཨཎའོ། །ཨ་ཀཪྵི། ཨཱ་ཏཱ་མ་

ཏ་པ་ཀུན་ནས་གདུང་བ་ལའོ། །དེ་ལ་རྣམ་པར་གཞག་པ་ཡང་ན་བ་སྤྱིར་བཏང་བ་ལས། ཨ་ནུ་ལས་ཏ་པ་ལ་ཨིཙ་མེད་དོ། །ཨ་ནུ་ཏཔྟ། ཨ་ནུ་ཏཔྶཱ་ཏཱཾ། ཨ་ནུའི་ཞེས་པས་ཁྱད་པར་ཅན་དང་འབྲེལ་པ་དྲུག་གི་བར་ཆོད་པ་ལ་ཡང་ངོ་། །ཨ་ནྭ་བཱ་ཏཔྟ། ལས་བྱེད་པ་པོ་ལས་བཞིན་ནོ་ཞེས་པས་ལས་བཞིན་འགྱུར་བ་ལས།
