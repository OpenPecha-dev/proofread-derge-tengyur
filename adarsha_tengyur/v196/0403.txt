བྷྱཱ་སའི་རི་ཡིག་གི་ཨ་ཡིག་གོ། །ཏི་ཏཪྤི་ཥ་ཏི། ཏི་ཏྲིཔྶ་ཏི། དྲྀ་པ། དི་དཪྤཥ་ཏི། དི་དྲིཔྶ་ཏི། མུ་ཧ། དེ་ལ་སན་ལ་བརྗོད་པར་བྱའོ་ཞེས་པས་ཨིཊ་དང་བཅས་པའི་སན་གྱི་ཡང་ན་ཡོན་ཏན་ཉིད་དོ། །མུ་མོ་ཧི་ཥ་ཏི། མུ་མོ་ཧི་ཥ་ཏི། དྲུ་ཧ། མྲུ་ཧ། སྣུ་ཧ། སྣི་ཧ། 
རྣམས་ཀྱི་རྣམ་པར་བཤད་པ་ཡང་བལྟ་བར་བྱ་སྟེ་ཞེས་པས་ཧའི་གྷ་ཉིད་དོ། །ཧའི་ཌྷའོ་ཞེས་པས་ཡང་ན་ཌྷ་ཉིད་དོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཞེས་པས་གྷའི་ཀའོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །མུ་མུ་ཀྵ་ཏི། དྲུ་ཧ། དུ་དྲོ་ཧི་ཥ་
ཏེ། དུ་དྲུ་ཧི་ཥ་ཏེ། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་པས་གསུམ་པ་དང་པོའི་ཆ་ཤས་ཀྱི་དང་པོ་བཞི་པ་ཉིད་དོ། །དུ་དྷྲུཀྵ་ཏི། ཥྞུ་ཧ། སྔ་མ་བཞིན་གྱི་ངེས་པ་ལས་ཥ་ཎ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །སུ་སྣོ་ཧི་ཥ་ཏི། སུ་སྣུ་ཧི་ཥ་ཏེ། སུ་སྣུ་ཀྵ་ཏི། སྣི་ཧ། སི་སྣེ་
ཧེ་ཥ་ཏེ། སི་སྣི་ཧི་ཥ་ཏི། སི་སྣི་ཧི་ཀྵ་ཏི། ཎ་ཤ །ནི་ན་ཤི་ཥ་ཏི། མསྫི་ན་ཤ་དག་གི་དྷུཊ་ལའོ་ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་ཡིག་ཨཱ་ག་མའོ། །ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ནི་ནཾ་ཀྵ་ཏི། ཏྲ་པཱུཀྵ། ཨཱུཊ་རྗེས་སུ་འབྲེལ་པ་ཉིད་
ལས་ཡང་ན་ཨིཊའོ། །ཏི་ཏྲ་པི་ཥ་ཏེ། ཏི་ཏྲཔྶ་ཏེ། ཀྵ་མཱུ་ཥ། ཙིཀྵ་མི་ཥ་ཏེ། ཙིཀྵཾ་ས་ཏ། ཀྵ་མུ་བཟོད་པ་ཞེས་པ་འདིའི་ནི་རྟག་ཏུ་ཨིཊའོ། །ཙིཀྵ་མི་ཥ་ཏེ། མྲྀ་ཛཱུ། མརྫའི་མཱ་ཛིའོ། །མི་མཱརྫི་ཥ་ཏི། བྷྲྀཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །མྲི་མྲཀྵ་ཏི། ཏཀྵུ་ཏྭཀྵཱུ། ཏི་ཏཀྵི་ཥ་ཏི། 
བརྩེགས་པའི་དང་པོ་ཀ་དབྱིའོ། །ཏི་ཏཀྵི་ཏི། ཏི་ཏྭཀྵི་ཥ་ཏི། ཏི་ཏྭཀྵི་ཏི། ཨཀྵ། དབྱངས་དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་གཉིས་བརྗོད་དོ། །ཨ་ཙི་ཀྵི་ཥ་ཏི། ཨ་ཙི་ཀྵ་ཏི། བྲཤྩུ། བྲི་བྲཤྩི་ཥ་ཏི། བྲཤྩི་མསྫ་དག་གི་དྷུཊ་ལའོ་ཞེས་པས་མཐའ་དབྱིའོ། །བྷྲྀཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་
ཉིད་དོ། །བྲི་བྲཀྵ་ཏི། ཏྲྀ་ཧཱུ། སྟྲྀ་ཧཱུ། འཚེ་བའི་དོན་ལའོ། །ཏི་ཏཪྷི་ཥ་ཏི། ཧའི་ཌྷའོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཏི་ཏྲྀཀྵ་ཏི། ཏི་སྟརྟི་ཥ་ཏི། ཏི་སྟྀཀྵ་ཏི། བྲྀ་ཧཱུ། བི་བརྷི་ཥ་ཏི། བི་བྲྀཀྵ་ཏི། གཱ་ཧཱུ། ཛི་གཱ་ཧི་ཥ་ཏི། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་
