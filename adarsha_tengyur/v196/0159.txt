གྱི་རེ་བ་ཨུ་རུ་འགྱུར་ཞིང་། དེའི་ཚེ་ཕ་རོལ་གྱི་ཨི་ཡིག་ནི་ཨིའི་ཡི་གེ་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལ་ཡའོ་ཞེས་པས་ཡ་ཉིད་དོ། །ཨ་ད་མུ་ཡང་། ཨ་ད་མུ་ཡཉྩཽ། ཨ་ད་མུ་ཡཉྩོཿ། ཨ་ད་མུ་ཡཉྩཾ། ཨ་ད་མུ་ཡཉྩཽ། ཨིཉྩིའི་ཨ་དབྱི་ཞིང་སྔ་མའི་ཡང་རིང་པོ་སྟེ། ཨུའི་ཡི་གེ་ལ་
ཝའོ་ཞེས་པས་ཝ་ཉིད་དུ་ཁ་ཅིག་འདོད་དོ། །ཨ་དམྺཱི་ཙཿ། ཨ་ད་མུ་ཨཱི་ཙཿ། ཨ་དམྺཱི་ཙཱ། ཨ་ད་མུ་ཡདྦྷྱཱཾ། ཨ་ད་མུ་ཡདྦྷིཿ། ཞེས་པ་ལ་སོགས་པ་སྟེ། བུད་མེད་ལ་ན་ད་ལ་སོགས་པའི་མདོས་ཨཱིའི་རྐྱེན་ནོ། །ཨ་ད་མུ་ཨཱི་ཙཱི། ཨ་དམྺཱི་ཙི། མ་ནིང་ལ། ཨ་ད་མུ་ཡཀ །ཨ་ད་མུ་ཡག །ཨ་
ད་མུ་ཨཱི་ཙི། ཨ་དམྺཱི་ཙཱི། ཨ་ད་མུ་ཡཉྩི། སོ་སོ་ལ་སྔ་མ་དང་འདྲ་བའོ། །ཨ་མུ་དྲྱདྦྷྱཱཾ། ཨ་དས་ཀྱི་ངའི་མ་ཉིད་དོ། །ཨ་མུ་དྲྱེང་། ཨ་མུ་དྲྱཉྩཽ། ཨ་མུ་དྲྱཉྩཿ། ཨ་མུ་དྲྱཉྩཾ། ཨ་མུ་དྲྱཉྩཽ། ཨ་མུ་དྲི་ཙཱཿ། ཨ་མུ་དྲི་ཙཱ། ཨ་མུ་དྲྱདྦྷྱཱཾ། ཨ་མུ་དྲྱདྦྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་
ལ་ཨ་མུ་དྲཱི་ཙཱི། མ་ནིང་ལ། ཨ་མུ་དྲུཀ །ཨ་མུ་དྲུག །ཨ་མུ་དྲཱི་ཙཱི། ཨ་མུ་དྲིཉྩི། ཞེས་པ་ལ་སོགས་པའོ། །འདིར་གཉི་གའི་ནི། ཨ་དས་ཀྱི་མ་ཉིད་དེ་མ་ལས་ཕ་རོལ་ཨུ་ཉིད་དང་། ཨ་དྲིར་བསྒྱུར་བའི་མ་ཉིད་དེ། མ་ལས་ཕ་རོལ་ཨུ་ཉིད་དོ། །ཨ་མུ་མུ་ཡང་། ཨ་མུ་མུ་ཡཉྩཽ། ཨ་
མུ་མུ་ཡཉྩ། ཨ་མུ་མུ་ཡཉྩཾ། ཨ་མུ་མུ་ཡཉྩཽ། འདིར་ཡང་ཝ་ཉིད་དུ་ཁ་ཅིག་འདོད་པར་བྱེད་དེ། ཨ་མུ་མུ་ཨཱི་ཙཿ། ཨ་དམྺཱི་ཙཿ། བུད་མེད་ལ། ཨ་མུ་མུ་ཨཱི་ཙཱི། ཨ་དམྺཱི་ཙཱི། མ་ནིང་ལ། ཨ་མུ་མུ་ཡཀ །ཨ་མུ་མུ་ཡག །ཨ་མུ་མུ་ཨཱི་ཙཱི། ཨ་མུ་མམྺཱི་ཙཱི། ཨ་ད་མུཉྩི། ཡང་དག་
པར་ན་སོ་སོར་གཉི་ག་དགག་སྟེ། ཨ་ད་དྲྱང་། ཨ་ད་དྲྱཉྩཽ། ཨ་ད་དྲྱཉྩཿ། ཨ་ད་དྲྱཉྩཾ། ཨ་ད་དྲྱཉྩཽ། ཨ་ད་དྲཱི་ཙཱཿ། ཨ་ད་དྲཱི་ཙཱ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། ཨ་ད་དྲཱི་ཙཱི། མ་ནིང་ལ། ཨ་ད་དྲྱཀ །ཨ་ད་དྲྱག །ཨ་ད་དྲཱི་ཙཱི། ཨ་ད་དྲིཉྩཾ། ལ་སོགས་པའོ། །ས་ཧནྩ་ཏི་ཞེས་པས་ཀྭིཔ་
སྟེ། ས་ཧ་དང་། སཾ་དང་། ཏི་རས་རྣམས་ཀྱི་ས་དྷྲི་དང་ས་མི་དང་། ཏི་རི་རྣམས་སོ་ཞེས་པས་ས་ཧའི་སྒྲ་ནི་ས་དྷྲི་ཞེས་པའི་ཚིག་ཕྲད་དོ། །ས་དྷྱད། སདྷྲྱཉྩཽ། ས་དྷྲྱཉྩཿ། སདྷྲྱཉྩཾ། ས་དྷྲྱཉྩཽ། ཨ་དབྱི་བ་དང་རིང་པོའང་སྟེ། ས་དྷྲཱི་ཙཿ། ས་དྷྲཱི་ཙཱ། ས་དྷྲྱདྦྷྱཱཾ། སདྷྲྱེདྡྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །
