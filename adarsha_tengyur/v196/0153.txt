བ་ཧུ་རཱ་ཡེ། བ་ཧུ་རི་ཎེ། ཀུ་ལཱ་ཡ་ཞེས་པ་ལ་སོགས་པའོ།།      །།སུ་ནཽའི་སྒྲ་ནི། འདིའི་གྲུ་བཟང་པོ་ཡིན་པ་ཞེས་འབྲུ་མང་པོ་པ་སྟེ། སྐྱེས་པ་ལ་ནཽའི་སྒྲ་དང་འདྲ་བའོ། །སུ་ནཽ། སུ་ནཱ་ཝཽ། སུ་ནཱ་ཝཿ། ཞེས་པ་ལ་སོགས་པ་སྟེ། བུད་མེད་ལ་ཡང་དེ་བཞིན་ནོ། །
མ་ནིང་ལ་ནི་དབྱངས་ཐུང་ངུ་མ་ནིང་ལའོ་ཞེས་པས་ཐུང་ངུར་འགྱུར་ཞིང་། མཚམས་སྦྱོར་གྱི་ཡི་གེ་རྣམས་ཨིཏ་ཨུཏ་དག་སྟེ་ཐུང་ངུར་བསྒྱུར་བའོ་ཞེས་པས་ཨ་ཡིག་ནཾ་ཨུ་ཡིག་ཏུའོ། །སུ་ནུ། སུ་ནུ་ནཱི། སུ་ནཱ་ནི། སླར་ཡང་། སུ་ནུ། སུ་ནུ་ནཱི། སུ་ནཱུ་ནི། ཊཱ་ལ་སོགས་པ་ལ་ཡང་ན་སྐྱེས་པ་དང་འདྲ་བར་
འགྱུར་ཏེ། སུ་ནཱ་ཝཱ། སུ་ནུ་ནཱ། སུ་ནུ་བྷྱཱཾ། སུ་ནུ་བྷིཿ། སུ་ནཱ་ཝེ། སུ་ནུ་ནེ་ཀུ་ལཱ་ཡ། ཞེས་པ་ལ་སོགས་པའོ།།      །།ཙི་ཏྲ་གོའི་སྒྲ་ནི་འདིའི་བ་ལང་ཁྲ་བོ་རྣམས་ཡིན་ལ་ཞེས་འབྲུ་མང་པོ་པ་སྟེ། གཙོ་བོ་མ་ཡིན་པའི་གོའི་མཐའ་བུད་མེད་ལ་ཨཱིཏ་ལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་ཞེས་པས་ཐུང་ངུར་
བསྒྱུར་བ་སྟེ། ཨོ་ཡིག་ཨུ་ཡིག་ཏུ་འགྱུར་ཞིང་། སྐྱེས་པ་ལ་ཨང་ཤུའི་སྒྲ་དང་འདྲ་བ་སྟེ། ཙི་ཏྲ་གུཿ། ཙི་ཏྲ་གཱུ། ཅི་ཏྲ་ག་ཝཿ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ་ནི་པ་ཊུ་དང་འདྲ་བ་སྟེ། ཙིཏྲ་གྭཱི། ཙི་ཏྲ་གྭྱོོ། ཙི་ཏྲ་གྭྱཿ། ཞེས་པ་ལ་སོགས་པའོ། །མ་ནིང་ལའང་པ་ཊུའི་སྒྲ་དང་འདྲ་བ་སྟེ།
ཙི་ཏྲ་གུ། ཙི་ཏྲ་གུ་ནཱི། ཙི་ཏྲ་གཱུ་ནི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཐུན་མོང་བརྗོད་པར་བྱེད་པའི་དབྱངས་མཐའ་ཅན་གཞན་རྣམས་ལ་ཡང་བརྟག་པར་བྱའོ།།      །།དེ་ནས་ཐུན་མོང་བརྗོད་པར་བྱེད་པའི་རྟགས་གསལ་བྱེད་ཀྱི་མཐའ་ཅན་རྣམས་བཤད་པར་བྱ་སྟེ། ས་ཏྱ་བཱ་ཙའི་སྒྲ་ནི། འདིའི་དག་
བདེན་པ་ཡིན་པ་ཞེས་འབྲུ་མང་པོ་པ་ཅན་ཡིན་ལ། སྔ་མ་བཞིན་དུ་ག་ཉིད་ལ་སོགས་པ་སྟེ། སཏྱ་ཝཱཀ །སཏྱ་ཝཱག །སཏྱ་བཱ་ཙཽ། སཏྱ་གཱ་ཙཿ། ཞེས་པ་ལ་སོགས་པ་དང་། བུད་མེད་ལ་ཡང་དེ་བཞིན་ནོ། །མ་ནིང་ལ་ནི་སི་ཨམ་དག་དབྱིའོ། །ས་ཏྱ་བཱཀ །ས་ཏྱ་བཱག །ཨཽ་ནི་ཨཱིའོ། །ས་ཏྱ་བཱ་ཙཱི། ཛས་
ཤས་དག་གི་ཤི་དང་། དྷུཊ་དང་དབྱངས་ལས་གྷུཊ་ལ་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་སྟེ། མ་ན་དག་གི་དྷུཊ་ལ་རྗེས་སུ་ང་རོའོ་ཞེས་པས་རྗེས་སུ་ང་རོར་འགྱུར་ཞིང་། སྡེ་པ་ལ་སྡེ་པའི་མཐའ་འོ་ཞེས་པས་ཉ་ཡིག་སྟེ། ས་ཏྱ་བཱཉྩི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་བྱཀྟ་བཱཙ་ལ་སོགས་པ་རྣམས་སོ། །སྦྱོར་བར་བྱེད་དོ། ཡུ་
