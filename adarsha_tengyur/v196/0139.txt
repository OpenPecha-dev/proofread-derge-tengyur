ཐབ་པོ། །བ་དྷཱུཿ། བ་དྷཱོོ། བ་དྷཱཾ། བ་དྷཱཽ། བ་དྷཱཿ། བ་དྷྭཱ། བ་དྷཱུ་བྷྱཱཾ། མ་དྷུ་བྷིཿ། བ་དྷྭེ། བ་དྷཱུ་བྷྱཱཾ། བ་དྷཱུ་བྷྱཿ། བ་དྷྭཿ། བ་དྷཱུ་བྷྱཱཾ། བ་དྷཱུ་བྷྱཿ། བ་དྷྭཿ། བ་དྷཱོཿ། བ་དྷཱུ་ནཱཾ། བ་དྷི། བ་དྷྭོཿ། ཥ་ཉིད་དོ། །བ་དྷཱུ་ཥུ། ཀུན་དུ་བོད་པ་ལ་སི་དབྱི་བ་དང་ཐུང་ངུ་ཉིད་
དོ། །ཧེ་བ་དྷུ། ཧེ་བ་དངྷྭོོ། ཧེ་བ་དྷྭཱཿ།།      །།བྷྲཱུའི་སྒྲ་ནི་སྟྲཱིའི་སྒྲ་དང་འདྲ་བར་ཤེས་པར་བྱ་སྟེ། དབྱངས་དང་པོ་ཅན་ལ་ནི་ཨུ་ཝ་རུ་འགྱུར་ཞིང་། བྷྲཱུ་བྱིངས་བཞིན་ནོ་ཞེས་པས་བྱིངས་དང་འདྲ་བར་ཨཱིཏ་ཨཱུཏ་དག་གི་ཨི་ཡ་ཨུ་ཝ་དག་དབྱངས་ལའོ་ཞེས་པ་ཁྱད་པར་རོ། །བྷྲཱུ། བྷྲུ་ཝཾཾ།བྷུ་ཝཿ། བྷྲུ་
ཝཾ། བྷྲུ་ཝོོ། བྷྲུ་ཝཿ། བྷྲུ་ཝཱ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷིཿ། བྷུ་ཝཻ། བྷྲཱུ་བྷྱཱཾ། བྷྲུ་བྷྱཿ། བྷྲུ་ཝཱཿ། བྷྲཱུ་བྷྱཱཾ། བྷྲཱུ་བྷྱཿ། བྷྲུ་ཝཱཿ། བྷྲཱུ་ཝོཿ། བྷྲཱུ་ནཱཾ། བྷྲུ་ཝཱཾ། བྷྲུ་ཝོཿ། ཥ་ཉིད་དོ། །བྷྲུ་ཥུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་བྷྲུ། ཧེ་བྷྲུ་ཝཽ། ཧེ་བྷྲུ་ཝཿ།།      །།གཅོད་པར་བྱེད་པས་ན་ལཱུ་སྟེ། སཾ་པད་ལ་སོགས་པ་
ཉིད་ལས་བུད་མེད་ལ་ཀྭིཔ་འགྱུར་ཞིང་པ་ར་མའང་འདི་ཡིན་ལ་ལཱུའང་ངོ་ཞེས། གཞི་མཐུན་པ་ཡི་ཚིག་དག་ནི། ཞེས་པ་ལ་སོགས་པའི་ཚིག་སྡུད་དོ། །པ་ར་མ་ལཱུཿ། པ་ར་མ་ལུ་ཝཾཾ། པ་ར་མ་ལུ་ཝ། ཞེས་པ་ལ་སོགས་པའོ། །གང་གི་ཚེ་མི་ཟད་པ་ལྡན་པ་དེའི་ཚེ། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་
ལ་སོགས་པས་ཝ་ཉིད་དོ། །པ་ར་མ་ལྭོ། པ་ར་མ་ལྭཿ།།      །།པུ་ནརྦྷཱུའི་སྒྲའི་ཁྱད་པར་ནི། པུ་ནརྦྷཱུཿ། ཨུ་ཝ་ཐོབ་པ་ལ་ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དེ། དེ་ཡང་བྷཱུའི་ཝཪྵ་བྷཱུ་དང་པུ་ནརྦྷཱུ་མ་ཡིན་པའོ་ཞེས་པས་དགག་པ་ཐོབ་པ་ལ་ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པས་ཝ་ཉིད་
དོ། །པུ་ནརྦྷཱོོ། པུ་ནརྦྷྭཿ། པུ་ནརྦྷྭཾཿ། པུ་ནརྦྷཱཽ། པུ་ནརྦྷཱཿ། པུ་ནརྦྷཱ། པུ་ནརྦྷཱུ་བྷྱཱཾ། པུ་ནརྦྷཱུ་བྷིཿ། ཨུ་ཝ་གནས་པ་ཉིད་མེད་པའི་ཕྱིར་ཐུང་ངུའི་ཡང་ང་ལྡན་ལའོ་ཞེས་རྣམ་པར་བརྟག་པས་ན་དཱི་དང་འདྲ་བར་འགྱུར་བ་མེད་དེ། དེས་ན་ཨཱིཏ་ཨཱུཏ་བུད་མེད་བརྗོད་པ་དག་ན་དཱིའོ་ཞེས་པས་ན་དཱིའི་མིང་ཅན་
ཡིན་པས་རྟག་ཏུ་ན་དཱིའི་བྱ་བའོ། །པུ་ནརྦྷྭཻ། པུ་ནརྦྷཱུ་བྷྱཱཾ། པུ་ནརྦྷཱུ་བྷྱཿ། པུ་ནརྦྷྭཱཿ། པུ་ནརྦྷཱུ་བྷྱཱཾ། པུ་ནརྦྷཱུ་བྷྱཿ། པུ་ནརྦྷྭཱཿ། པུ་ནརྦྷཱོ། པུ་ནརྦྷྭཱཾ། པུ་ནརྦྷཱུ་ཎཱཾ། པུ་ནརྦྷྭཱཾ། པུ་ནརྦྷྭོཿ། ཥ་ཉིད་དོ། །པུ་ནརྦྷཱུ་ཥུ། ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལ་ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུ་སྟེ། ཐུང་ངུ་ན་དཱི་ཞེས་པ་ལ་
