ཌུ་ཀྲྀཉ། ཨ་ཀཱ་རི་ཀ་ཊཿ། སྭ་ཡ་མེ་བ། ཨ་ཀཱ་རི་ཥཱ་ཏཱཾ། ཀ་ཊ་སྭ་ཡ་མེ་བ། ཨ་ཀྲྀ་ཥཱ་ཏཱཾ། སྔ་མ་བཞིན་དུ་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཏྨ་ནེའི་ཚིག་ལ་ནི། ཨ་ཀྲྀ་ཏ་ཀ་ཊཿ། སྭ་ཡ་མེ་བ། ཨ་ཀྲྀ་ཥཱ་ཏཱཾ། ཀ་ཊཽ་སྭ་ཡ་མེ་བ། ལཱུཉ། ཨ་ལཱ་བི་ཀེ་དཱ་རཿ། སྭ་ཡ་མེ་བ། ཨ་ལཱ་བི་ཥཱ་ཏཱཾ་ཀེ་དཱ་རོོ་སྭ་ཡ་མེ་བ། ཨ་ལཱ་བི་ཤཱ་
ཏཱཾ། སྔ་མ་བཞིན་དུ་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་ལ། ཨ་ལ་བིཥྚ་ཀེ་དཱ་རཿ། སྭ་ཡ་མེ་བ། ཨ་ལཱ་བི་ཥཱ་ཏཱཾ་ཀེ་དཱ་རཽ་སྭ་ཡ་མེ་བ། དུ་ཧ། ཨ་དོ་ཧི་གཽཿ། སྭ་ཡ་མེ་བ། གང་གི་ཚེ་སྔ་མ་བཞིན་དུ་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་ཡིན་པ་དེའི་ཚེ། ཨ་དུ་གགྡྷ་གཽཿ། སྭ་ཡ་མེ་བ། ཨ་དྷུ་ཀྵ་ཏ། ཌུ་པ་ཙ་ཥ། ཨ་པཱ་
ཙྱོ་དཱ་ནཿ། སྭ་ཡ་མེ་བ། བྷི་དིར། ཨ་བྷེ་དེ་ཀཥྛཿ། སྭ་ཡ་མེ་བ། ཀུ་ཥི་རཾ་ཛ་དག་གི་ལས་བྱེད་པ་པོ་ལ་ཨ་དྱ་ཏ་ནཱི་ལ་ཡང་ན་པ་རསྨཻའི་ཚིག་ལྟ་བར་མི་བའོ། །ཨ་ཀོ་ཥཱི་པཱ་དཿ། སྭ་ཡ་མེ་བ། ཨ་རཾ་ཛི་བ་སྟཾ་སྭ་ཡ་མེ་བ། ཀི་ར་སོགས་དང་། ཤྲནྠི། གྲནྠི། སན་མཐའ་རྣམས་ལས། ལས་བྱེད་པ་པོར་གནས་སོ་
ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ལས་བཞིན་འགྱུར་བའི་དམིགས་བསལ་ལོ། །གླང་པོས་དཀྲིགས་པའི་ཁ་ལོ་བ། །གླང་པོ་ཞོན་ནས་འདྲེན་པར་བྱེད། །ཨ་བ་ཀཱིརྵྚ་ཧསྟི་སྭ་ཡ་མེ་བ། ཨ་བཱ་ཀཱ་རཱིཥྚ། ཨ་བཱ་ཀ་རཱིཥྚ། ཨ་བཱ་ཀ་རིཥྚ། ཨ་ལཱི་ལ་བ་ཏ་སྭ་ཡ་མེ་བ་ཀེ་དཱ་ར། ཙུ་ར།
ཨ་ཙུ་ཙུ་ར་ཏ་གཽཿ། སྭ་ཡ་མེ་བ། ཤྲནྠི་གྲོལ་བར་བྱེད་པ་དང་སོ་སོར་དགའ་བར་བྱེད་པ་དག་ལའོ། །གྲནྠི་སྤེལ་བ་ལའོ། །ཤྲནྠ། གྲནྠ། སྤེལ་མ་ལ་ཡང་ནའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་རྣམ་པར་རྟོག་པ་མཐའ་ཡས་སོ། །ཨ་ཤྲནྠིཥྚ་མེ་ཁ་ལཱ་སྭ་ཡ་མེ་བ། ཨ་གྲིནྠ་མེ་ཁ་ལཱ་སྭ་ཡ་མེ་བ། བྲུཉ། དྲུའི་བ་
ཙིའིའོ། །ཨ་བོ་ཙཏ་དྷྭརྨ་སྭ་ཡ་མེ་བ། ལྷས་བྱིན་གྱི་ངག་དང་གདུགས་ཀྱི་རྣམ་འགྱུར་ལ། དེ་བ་དཏྟཾ་བྱ་ཀྲྀ་ཏ་ཙྪཱཏྟཿ། སྭ་ཡ་མེ་བ། སྣུ་ན་མ་དག་རང་ཉིད་ལ་ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ལས་བཞིན་འགྱུར་བའི་དམིགས་བསལ་ལོ། །པྲཱསྣོ་ཥྚ་གཽཿ། སྭ་ཡ་མེ་བ། ལྷས་བྱིན་གྱིས་
