ཨཱ་དཏྟེ། ཨཱ་ད་དཱ་ཏེ། ཨ་ད་ད་ཏེ། ཊུ་དྷཱཉ། ཊུ་བྷྲྀ་ཉ་འཛིན་པ་དང་གསོ་བ་དག་ལའོ། །ད་དྷཱ་ཏི། སྔ་མ་བཞིན་ཨ་ཡིག་དབྱིའོ། །ད་དྷཱ་ཏིའི་ཏ་ཐ་དག་ལའང་ངོ་ཞེས་པས་དྷཱའི་མཐའི་ཏ་ཐ་དག་དང་ས་ཡིག་དང་དྷཱའི་སྒྲ་ལ་དང་པོ་དའི་བཞི་པ་དྷཱ་ཉིད་དོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤི་ཊ་
མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་དྷའི་ཏའོ། །དྷ་ཏྟཿ། ད་དྷ་ཏི། ད་དྷཱ་སི། དྷཏྠཿ། དྷཏྠ། ད་དྷཱ་མི། ད་དྷྭཿ། ད་དྷ་མཿ། དྷཏྟེ། ད་དྷཱ་ཏེ། ད་དྷ་ཏེ། དྷཏྶེ། ད་དྷཱ་ཐེ། དྷཱུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་དྷའི་ད་ཉིད་དོ། །དྷ་དྷྭེ། ད་དྷེ། ད་དྷཱ་ཧེ། ད་དྷ་མ་ཧེ། པ་རི་ད་དྷ་ཏི། 
པ་རི་དྷཱཏྟེ། སནྡ་དྷཱ་ཏེེ། སན་དྷཏྟེ། ཨཱ་ད་དྷཱ་ཏི། ཨཱ་དྷཱཏྟེ། ཡི་གེ་ཉམས་པས་ཨ་པའི་ཨ་ཏའི་ཡང་ན་དབྱིའོ། །ཨ་པི་ད་དྷཱ་ཏི། ཨ་པི་དྷཏྟེ། པྲ་ཎི་ད་དྷཱ་ཏི། པྲ་ཎི་དྷཏྟེ། བྷྲྀ་ཉ་ཧཱང་མང་རྣམས་ཀྱི་ཨིཏའོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་ཨི་ཉིད་དོ། །ནཱ་མའི་མཐའི་ཡོན་ཏན་ནོ། །བི་བྷརྟྟི། བི་བྷྲྀ་ཏཿ། བི་
བྷྲ་ཏི། བི་བྷཪྵི། བི་བྷྲྀ་ཐཿ། བི་བྷྲྀ་ཐ། བི་བྷྲརྨྨིཿ། བི་བྷྲྀ་བཿ། བི་བྷྲྀ་མཿ། བི་བྷྲྀ་ཏེ། བི་བྷྱཱ་ཏེ། བི་བྷྲ་ཏེ། བི་བྷྲྀ་ཥེ། བི་བྷྲ་ཐེ། བི་བྷྲྀ་དྷྭེ། བི་བྷྲེ། བི་བྷྲྀ་བ་ཧེ། བི་བྷྲྀ་མ་ཧེ། གཉི་ག་ལས་བཤད་པ་རྣམས་སོ། །མཱང་ཚད་ལ་སྟེ་སྒྲ་ལ་ཡང་ངོ་། །གཉི་ག་རྣམས་ཀྱི་དྷ་མིན་པའི་གསལ་བྱེད་
དང་པོ་ལ་ཨི་ཡིག་གོ་ཞེས་པས་དཱ་མ་ཡིན་པའི་གསལ་བྱེད་དང་པོ་ལ་ཨ་ཡིག་གི་ཨཱི་ཡིག་གོ། །མི་མཱི་ཏེ། མི་མཱ་ཏེ། མི་མ་ཏེ། མི་མཱི་ཥེ། མི་མཱ་ཐེ། མི་མཱི་ཧྭེ། མི་མེ། མི་མཱི་བ་ཧེ། མི་མཱི་མ་ཧེ། པྲ་ཎི་མི་མཱི་ཏེ། ཨ་ནུ་མི་མཱི་ཏེ། ཨོ་ཧཱང་འགྲོ་བ་ལའོ། །ཛི་ཧཱི་ཏེ། ཛི་ཧཱ་ཏེ། ཛི་ཧ་
ཏེ། ཛི་ཧཱི་ཥེ། ཛི་ཧཱ་ཐེ། ཛི་ཧཱི་དྷྭེ། ཛི་ཧེ། ཛི་ཧཱི་བ་ཧེ། ཛི་ཧཱི་མ་ཧེ། ཨུཏ་སྔོན་དུའོ། །ཏ་ཡིག་ལ་དང་ཙཊའི་སྡེ་པ་རྣམས་ལ་ཕ་རོལ་གྱི་གཟུགས་སོ་ཞེས་པས་ཏ་ཡིག་གི་ཕ་རོལ་གྱི་གཟུགས་ཛ་ཡིག་གོ། །ཨུཛྫི་ཧི་ཏེ། ནིར་སྔོན་དུའོ། །རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་
བྱའོ་ཞེས་པས་ར་ཡིག་གི་རྣམ་པར་གཅད་བྱའོ། །སྒྲ་ལྡན་དང་དབྱངས་ཕ་རོལ་ལོ་ཞེས་པས་རྣམ་པར་གཅད་བྱའི་རེ་ཕའོ། ནིརྫི་ཧཱི་ཏེ། ཛུ་ཧོ་ཏི་ལ་སོགས་པ་ཉི་ཤུ་རྩ་བཞི་པོ་རྣམས་སོ་ཞེས་པ་ཨཏ་ལ་སོགས་པ་བི་ཀ་ར་ཎ་འཇིག་པའི་ཚོགས་སོ། །དེ་བུ་རྩེ་བ་དང་བསྔོ་བར་འདོད་
