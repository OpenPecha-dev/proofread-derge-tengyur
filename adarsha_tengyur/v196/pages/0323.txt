ཨ་བི་ཀྵཏ། མུ་ཙླྀ། ཨ་མུ་ཙ་ཏ། ཨཱཏྨ་ནེའི་ཚིག་ལ་སི་ཙའོ། །སི་ཙ་ཨཱ་ཤཱིཥ་དག་ལ་ཡང་ན་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །དྷུཊ་ལས་ཀྱང་དྷུཊ་ལའོ་ཞེས་པས་སི་ཙ་དབྱིའོ། །ཨ་མུཀྵཱ་ཏ། ཨ་མུཀྵཱ་ཏཱཾ། ཨ་མུཀྵཏ། ལུ་བླྀ། ཨ་ལུཔྟ། ཨ་ལུཔྶཱ་ཏཱཾ། བི་དླྀ། ཨ་བིད་ཏ། ཨ་བིཏྟ། ཨ་བིཏྶཱ་ཏཱཾ། རུ་དྷིར། ཨིར་
རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་བརྗོད་པར་བྱའོ་ཞེས་པས་པ་རསྨཻའི་ཚིག་ལ་ཡང་ན་ཨཎའོ། །ཨ་རུ་དྷཏ། ཨ་རུ་དྷ་ཏཱཾ། ཨ་རུ་དྷན། ཕྱོགས་གཅིག་ལ་སི་ཙའོ། །གསལ་བྱེད་མཐའ་རྣམས་ཀྱི་ཨིཊ་མེད་པའི་འོ་ཞེས་པས་འཕེལ་བའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཨ་རཽཏྶཱི་ཏ། སིཙ་
