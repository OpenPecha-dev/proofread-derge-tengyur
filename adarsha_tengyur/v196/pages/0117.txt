ཧེ་ཀྲོཥྚ་རཽ། ཧེ་ཀྲོཥྚ་རཿ། བུད་མེད་དང་པོ་ལ་ནི་མདོས་ཨཱི་རྐྱེན་ནོ། །ཀྲོཥྚཱི། སྭ་ཡཾ་བྷ་ཝ་ཏི་ཞེས་པ་ལ། གཞན་རྣམས་ཀྱང་མཐོང་ངོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ཀིཔྲ་འང་ངོ་ཞེས་པས་ཀྭིཔྲ་འང་སྟེ། ཉེ་བའི་ཚིག་གིས་ཚིག་སྡུད་དོ། །པ་ཡིག་ནི་བྱིངས་ཀྱི་པ་རྗེས་སུ་འབྲེལ་པ་ལ་
ཏ་མཐའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ལ། ཝ་ཡིག་ནི་སྦྱར་བ་མ་ཡིན་པའི་ཝི་དབྱིའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ཞིང་། ཀ་ཡིག་ནི་ཀ་ལ་ཡཎ་བཞིན་ཡང་སྟེ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཏོ། །རྟགས་ཀྱི་མིང་ཅན་ལ་སི་ལ་སོགས་པ་རྣམས་ཏེ། སི་ལ་ནི། སྭ་ཡཾ་བྷཱུཿ། དབྱངས་ལ་གཉིས་ཀྱི་ཚིག་ལ་
སོགས་པ་ལ་ནི། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དུ་ཐོབ་མོད། བྷཱུའི་ཝཪྵཱ་བྷཱུ་དང་པུ་ནར་བྷཱུ་མ་ཡིན་པའོ་ཞེས་པས་བྷཱུ་ནི་ཨུ་ཝ་རུ་ཐོབ་བོ། །སྭ་ཡཾ་བྷུ་བཽ། སྭ་ཡཾ་བྷུ་ཝཿ། སྭ་ཡིཾ་བྷུ་ཝམ། སྭ་ཡཾ་བྷུ་ཝཽ། སྭ་ཡཾ་བྷུ་ཝཿ། སྭཱ་ཡཾ་བྷུ་ཝཱ། སྭ་ཡཾ་བྷཱུ་བྷྱཱམ། སྭ་ཡཾ་བྷཱུ་བྷིཿ། སྭ་ཡཾ་བྷུ་ཝེ།
སྭ་ཡཾ་བྷཱུ་བྷྱཱམ། སྭ་ཡཾ་བྷཱུ་བྷྱཿ། སྭ་ཡཾ་བྷུ་ཝཿ། སྭ་ཡཾ་བྷཱུ་བྷྱཱམ། སྭ་ཡཾ་བྷཱུ་བྷྱཿ།སྭ་ཡཾ་བྷཱུ་ཝཿ། སྭ་ཡཾ་བྷཱུ་ཝོཿ། སྭ་ཡཾ་བྷུ་ཝམ། སྭ་ཡཾ་བྷུ་ཝི། སྭ་ཡཾ་བྷུ་ཝོཿ། སྭ་ཡཾ་བྷཱུ་ཥུ། ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་སྭ་ཡཾ་བྷཱུཿ། ཧེ་སྭ་ཡཾ་བྷཱུ་ཝཾཾ། ཧེ་སྭཾ་བྷུ་ཝཿ། དེ་བཞིན་དུ་ཨཱཏྨ་བྷཱུ། མ་ནོ་བྷཱུ། པྲ་ཏི་བྷཱུ་ཞེས་པ་
འདི་ལྟ་བུ་ལ་སོགས་པ་རྣམས་སོ། །ཁ་ལ་པུའི་སྒྲ་ལ་ཁྱད་པར་ནི། ཁ་ལ་པུ་དག་པར་བྱེད་པ་ལའོ་ཞེས་པས་ཀྭིཔ་སྟེ། ཉེ་བའི་ཚིག་གིས་ཚིག་སྡུད་དོ། །སི་ལ་ནི་ཁ་ལ་པཱུཿ། ཨུ་ཝ་ཐོབ་མོད། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཝ་ཉིད་དོ། །ཁ་ལ་སྭཽ། ཁ་ལ་སྭཿ། ཁ་ལ་སྭཾ། ཁ་ལ་
སྭཽ། ཁ་ལ་སྭཿ། ཁ་ལ་སྭཱ། ཁ་ལ་སཱུ་བྷྱཱམ། ཁ་ལ་སཱུ་བྷིཿ། ཁ་ལ་སྭེ། ཁ་ལ་སཱུ་བྷྱཱམ། ཁ་ལ་སཱུ་བྷྱཿ། ཁ་ལ་སྭཿ། ཁ་ལ་སཱུ་བྷྱཱམ། ཁ་ལ་སྭཱུ་བྷྱཿ། ཁ་ལ་སྭཿ། ཁ་ལ་སྭོཿ། ཁ་ལ་སྭཱམ། ཁ་ལ་སྭི། ཁ་ལ་སྭོཿ། ཁ་ལ་པུ་ཥུ། ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་ཁ་ལ་པུཿ། ཧེ་ཁ་ལ་སྭཽ། ཧེ་ཁ་ལ་སྭཿ།
ཝརྵཱ་བྷཱུའི་སྒྲ་ལའང་ཁྱད་པར་ནི། བྷཱུའི་ཝརྵཱ་བྷཱུ་དང་པུ་ནར་བྷཱུ་མ་ཡིན་པ་ཞེས་པས། འདི་ཝཪྵཱ་བྷཱུའི་སྒྲ་ལ་ནི་བཀག་པའི་ཕྱིར། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་དབྱངས་དང་པོ་ལ་ཝ་ཉིད་དོ། །ཝཪྵཱ་བྷཱུཿ། ཝརྵཱ་བྷཽ། ཝཪྵ་བྷཱཿ།།      །།པི་ཏྲྀའི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ཏེ། དང་པོའི་གཅིག་
