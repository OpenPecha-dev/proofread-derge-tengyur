པས་ཀྭིཔ་སྟེ། ཡ་ཎ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱིར་ཡོན་ཏན་ཉིད་མེད་ལ། ཨིཏ་རྗེས་སུ་འབྲེལ་པ་མ་ཡིན་པ་རྣམས་ཀྱི་ཞེས་པས་ཨ་ནུ་ཥཾ་ག་དབྱི་ཞིང་། སྲནྶི་དང་ཧྟྭཾ་ས་དག་གིའང་ངོ་ཞེས་པས་ཤ་ཡིག་གི་ད་ཡིག་སྟེ། རྫོགས་པ་ལ་ཏ་ཡིག་གམ་ད་ཡིག་དག་གོ། །ཨུ་ཁཱ་ཤྲཏ། ཨུ་ཁཱ་ཤྲད། ཨུ་
ཁཱ་ཤྲ་སོ། ཨུ་ཁཱ་ཤྲ་སཿ། དེ་བཞིན་དུ་བུད་མེད་ལའང་སྟེ། མ་ནིང་ལ། ཨུ་ཁཱ་ཤྲ་ཏ། ཨུ་ཁཱ་ར་ཤྲད། ཨུ་ཁཱ་ཤྲ་སཱི། ཨུ་ཁཱ་སྲནྶི། དེ་བཞིན་དུ་བནྡུ་དྷྭ་ཤ་ལ་སོགས་པ་རྣམས་སོ། །པྲ་ཥྚོ་བ་ཧ་ཏི་ཞེས་པ་ལ་བ་ཧ་ལས་ཀྱང་ངོ་ཞེས་པས་བིན་ཏེ། ཨུ་པ་དྷཱ་རིང་པོ་ཉིད་དང་། ཧ་ཡིག་གི་ཨུ་ཉིད་དོ། །
པྲཥྛ་བཱཌ། པྲཥྚ་བཱཌ། པྲཥྛ་བཱ་ཧཽ། པྲཥྛ་བཱ་ཧཿ། པྲཥྛ་བཱ་ཧཾ། པྲཥྛ་བཱ་ཧཽ། གྷུཊ་མ་ཡིན་པའི་དབྱངས་ལ། བཱ་ཧིའི་བཱ་སྒྲའི་ཨཽ་ཉིད་དོ་ཞེས་པས་བཱ་སྒྲའི་ཨཽ་ཉིད་དེ། པྲཥྛཽ་ཧཿ། པྲཥྛཽ་ཧཱ། པྲཥྛ་བཱདྦྷྱཱཾ། པྲཥྛ་བཱཌྦྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། ཨཱིའི་རྐྱེན་ཏེ། པྲཥྛཽ་ཧཱི། 
མ་ནིང་ལ། པྲཥྛ་བཱཊ། པྲཥྛ་བཱཌ། པྲཥྛ་བཱ་ཧཱི། པྲཥྛ་བཱཾ་ཧི།།      །།གཱཾ་དོགྦྷི་ཏི་ཞེས་པས་ཀྭིཔ་སྟེ། ཧ་དང་བཞི་པ་མཐའ་ཅན་ཞེས་པ་ལ་སོགས་པས་དང་པོ་བཞི་པ་ཉིད་དེ། རྫོགས་པ་དང་གསལ་བྱེད་ལའང་མ་བྱས་པ་དང་འདྲ་བར་སྨོས་པའི་ཕྱིར། བྱིངས་ལས་ཧའི་གའོ་ཞེས་པས་ག་ཉིད་དོ། །
གོ་དྷུཀ །གོ་དྷུག །གོ་དུ་ཧཽ། གོ་དུ་ཧཿ། ཞེས་པ་ལ་སོགས་པ་སྟེ། དེ་བཞིན་དུ་བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ། གོ་དྷུཀ །གོ་དྷུག །གོ་དུ་ཧཱི། གོ་དུ་ཧི།།      །།སྭཱ་ན་བྷ་ནོ་ཨ་ན་ཌྭ་ཧའི་ཞེས་པ་སྐྱེས་པ་དང་བུད་མེད་དག་ལ་ཨན་ཌྭ་ཏའི་སྒྲ་དང་འདྲ་བ་སྟེ། སྭ་ན་ཌྭཱན། སྭ་ན་ཌྭཧོ། སྭ་
ན་ཌྭཱཧཿ། ཞེས་པ་ལ་སོགས་པ་སྟེ། བུད་མེད་ལ། སྭ་ན་ཌ་ཧཱི། མ་ནིང་ལ། རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ཅན་ལ་བརྗོད་པའང་མ་ཡིན་ཞེས་བསྒྲེ་བ་ལས་ཝཱ་སྒྲའི་ཨུ་ཉིད་དེ། རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ཅན་རྣམས་ལ་ཨན་ཌོ་ཧི་ཞེས་པས་ང་ཉིད་དོ། །སྭ་ན་ཌུཏ། སྭ་ན་ཌུད། གྷུཊ་མ་
ཡིན་པའི་དབྱངས་ལ་ཝཱ་སྒྲའི་ཨུ་ཉིད་དེ། སྭ་ན་ཌུ་ཧཱི། སྭ་ན་ཌྭཱནྷི། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཐུན་མོང་བའི་རྗོད་པར་བྱེད་པ་གཞན་རྣམས་ལ་ཡང་བརྟག་པར་བྱའོ།།      །།ད་ནི་སརྦའི་མིང་ཅན་རྣམས་བརྗོད་པར་བྱ་སྟེ། སརྦ་དང་། བི་ཤྭ་དང་། ཨུ་བྷ་དང་། ཨུ་བྷ་ཡ་དང་། ཨ་ནྱ་
