ཏེ། མྲུ་ཧ། དྲུ་ཧ། ཥྞུ་ཧ། ཥྞི་ཧ། མོ་ཧི་ཥྱ་ཏི། དྲོ་ཧི་ཥྱ་ཏི། སྣོ་ཧི་ཥྱ་ཏི། སྣ་ཧི་ཥྱ་ཏི། མུ་ཧ། དྲུ་ཧ། སྣུ་ཧ། སྣི་ཧ་རྣམས་ཀྱི་རྣམ་པར་བཤད་པ་ཡང་ན་བལྟ་བར་བྱ་སྟེ། ཧའི་གྷ་ཉིད་དོ། །ཕྱོགས་གཅིག་ལ་ཧའི་ཌྷའོ། །གྷའི་ཕྱོགས་ལ་སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་
མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ། །ཞེས་པས་གྷའི་ཀ་ཉིད་དོ། །ཌྷ་ཉིད་ཀྱི་ཕྱོགས་ལ་ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཌྷའི་ཀའོ། །མོཀྵྱ་ཏི་དང་པོ་བཞི་པ་ཉིད་སྔ་མ་བཞིན་དའི་དྷ་ཉིད་དོ། །དྷྲོ་ཀྵྱ་ཏི། སྣོ་ཀྵྱ་ཏི། སྣེ་ཀྵ་ཏི། ཎ་ཤ །ན་ཤི་ཥྱ་ཏི། མསྫི་ན་ཤ་དག་གི་དྷུཊ་ལའོ། །ཞེས་པས་
དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་ཡིག་ཨཱ་ག་མའོ། །ཚ་ཤ་དག་གི་འང་ངོ་། །ཞེས་པས་ཥ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ། །ཞེས་པས་ཀ་ཉིད་དོ། །རྗེས་སུ་ང་རོའོ། །སྡེ་པ་ལ་སྡེ་པའི་མཐའོ། །ནརྐྵྱ་ཏི། དེ་ལ་འདིར་སྨོས་པ་ལས་ན་ཤའི་ཥ་ཡིག་མཐའི་ཞེས་པས་ཎ་ཉིད་བཀག་གོ། །པྲ་ན་ཀྵྱ་ཏི། 
ར་དྷ་སོགས་བརྒྱད་རྣམས་སོ། །དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཥྛཱ། སྱ་དང་། སི་ཙ་དང་ཨཱ་ཤཱིཥ་དང་ཤྭསྟ་ནཱི་རྣམས་ལ་དངོས་པོ་དང་ལས་ཀྱི་དོན་རྣམས་ལ་དབྱངས་དང་ཧ་ན་དང་གྲ་ཧ་དང་དྲྀ་ཤ་རྣམས་ཀྱི་ཨིཊ་ཨིཙ་བཞིན་ཡང་ནའོ། །ཞེས་པས་ཡང་ན་ཨི་ཙ་བཞིན་ཨིཊ་ཨཱ་ག་མའོ། །
ཨི་ཙ་བཞིན་འགྱུར་བ་ལས་ཨཱ་ཡིག་གི་ཨཱ་ཡིའོ། །སྠཱ་ཡི་ཥྱ་ཏེ། སྠཱ་སྱ་ཏེ། ཡཱ། ཡཱ་ཡི་ཥྱ་ཏེ། ཡཱ་སྱ་ཏེ། ད་རི་དྲཱ། ད་རི་དྲཱུ་ཏིའི་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལའོ། །ཞེས་པས་མཐའ་དབྱིའོ། །ད་རི་དྲི་ཥྱ་ཏེ། ཇི་ཨིཙ་བཞིན་ཨིཊ་འགྱུར་བ་ལས་འཕེལ་བའོ། །ཛ་ཡི་ཥྱ་ཏི། ཤྲིཉ།
ཤྲཱ་ཡི་ཥྱ་ཏེ། ཎཱིཉ། ནཱ་ཡི་ཥྱ་ཏེ། ནེ་ཥྱ་ཏེ། དཱི་དྷཱིང་། བེ་བཱིང་། དཱི་དྷཱི་བེ་བཱི་དག་གི་ཨི་ཡིག་ཡ་ཡིག་དག་ལའོ། །ཞེས་པས་མཐའ་དབྱིའོ། །དཱི་དྷཱི་ཥྱ་ཏི། བེ་བི་ཥྱ་ཏེ། ཤྲུ། ཤྲཱ་བི་ཥྱ་ཏེ། ཤྭོ་ཥྱ་ཏེ། བྷཱུ། བྷཱ་བི་ཥྱ་ཏི། བྷ་བི་ཥྱ་ཏེ། ལཱུཉ། ལཱ་བི་ཥྱ་ཏེ། ལ་བི་ཥྱ་ཏེ། ཌུ་ཀྲྀཉ། ཀ་རི་
ཥྱ་ཏེ། ཀ་རི་ཥྱ་ཏེ། བྲྀང་། བྲྀཉ། རྀཏ། བྲྀང་། བྲྀཉའི་ཡང་ཡང་ན་རིང་པོའོ། །པ་རོཀྵཱ་དང་ཨཱ་ཤཱིཥ་དག་མ་ཡིན་པ་ལ་ཨིཊའོ། །པ་རསྨཻ་མ་ཡིན་པའི་སི་ཙ་ལ་རབ་ཏུ་བརྗོད་དོ། །ཞེས་སྦྱོར་བ་རྣམ་པར་ཕྱེ་བ་ལས་ཀྱང་ཨིཊའི་ཡང་ན་རིང་པོའོ། །ཨི་ཙ་བཞིན་ཨིཊ་འགྱུར་ཞིང་དེ་མི་འགྱུར་བ་ཡང་ཐོས་པ་ཉིད་
