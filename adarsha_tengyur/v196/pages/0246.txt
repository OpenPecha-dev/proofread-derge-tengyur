མཐའ་ཨི་ཡིག་ཨཱ་ག་མའོ། །རྀའི་ཡི་གེ་རའོ། །དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་པས་ཨི་ཡ་རུ་བསྒྱུར་རོ། །བྱཱ་པྲི་ཡ་ཏེ། མྲྀང་སྲོག་གཏོང་བ་ལའོ། །ཨཱ་ཤཱི་ཥ་དང་ཨ་དྱ་ཏ་མཱི་དག་ལས་ཀྱང་མྲྀང་ངོ་ཞེས་ངེས་པ་བརྗོད་པ་ཡིས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །མྲི་
ཡ་ཏེ། ཨ་ནུ་མྲི་ཡ་ཏེ། བྷརྟཱ་ར། དྲིང་མ་གུས་པ་ལའོ་ཨཱ་དྲྀ་ཡ་ཏེ། དྷྲིང་མི་གནས་པ་ལའོ། །དྷྲི་ཡ་ཏེ། ཨཱ་དྷྲི་ཡ་ཏེ། ཛུ་ཥཱི་དགའ་བ་དང་བསྟེན་པ་དག་ལའོ། །ཛུ་ཥ་ཏེ། ཨོ་བི་ཛཱི་འཇིགས་པ་དང་གཡོ་བར་བྱེད་པ་དག་ལའོ། །བི་ཛ་ཏེ། ཨུདྦི་ཛ་ཏེ། ཨོ་ལ་ཛཱི། ཨོ་ལསྫཱི་སྐྱེངས་པ་ལའོ། །
ལ་ཛ་ཏེ། དྷུཊ་རྣམས་ཀྱི་གསུམ་པའོ་ཞེས་པས་སའི་ད་ཉིད་དོ། །ཏ་སྡེའི་ཙ་ཊའི་སྡེ་པ་ཞེས་པ་ལ་སོགས་པས་དའི་ཛ་ཡིག་གོ། །ལཛྫ་ཏེ། ཨཱཏྨ་ནེའི་བཤད་པ་རྣམས་སོ་ཞེས་པ་ཏུད་ལ་སོགས་པའི་ཚོགས་སོ། །རུ་དྷིར་སྒྲིབ་པ་ལའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །རུ་དྷ་སོགས་
ཀྱི་དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་སྒྲའོ་ཞེས་པས་དབྱངས་ལས་ཕ་རོལ་ཏུ་ན་སྒྲའི་རྐྱེན་ཏེ་བི་ཀ་ར་ཎའི་མིང་ཅན་ནོ། །རང་བཞིན་དང་རྐྱེན་དག་གི་དོན་མངོན་པར་བརྗོད་པ་ལ་ན་སྒྲ་ཉིད་ཀྱིས་ཨ་དང་ཡང་ལྷན་ཅིག་པར་བྱས་པའི་ན་ཉིད་ལས་ན་བི་ཀ་ར་ཎའོ། །ར་ཥ་རྀའི་ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་
སོགས་པས་ནའི་ཎ་ཉིད་དོ། །གྷ་ཌྷ་དྷ་བྷ་རྣམས་ལས་དྷཱ་མ་ཡིན་པ་ལ་ཏ་ཐ་དག་གི་དྷའོ་ཞེས་པས་ཏ་དག་གི་དྷ་ཉིད་དོ། །དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་རང་བཞིན་དྷ་ཡིག་གི་གསུམ་པ་ད་ཡིག་གོ། །རུ་ཎ་དྷི། རུ་དྷ་སོགས་ཀྱི་བི་ཀ་ར་ཎའི་མཐའི་དབྱིའོ་ཞེས་པས་
ཡོན་ཏན་མེད་པ་ལ་བི་ཀ་ར་ཎའི་མཐའ་དབྱིའོ། །མ་ན་དག་གི་དྷུཊ་ལ་རྗེས་སུ་ང་རོའོ་ཞེས་པས་ན་ཡིག་གི་རྗེས་སུ་ང་རོའོ། །སྡེ་པ་ལ་སྡེ་པའི་མཐའོ་ཞེས་པས་རྗེས་སུ་ང་རོའི་སྡེ་པའི་མཐའ་ན་ཡིག་གོ། །རུནྡྷ་ཿ། རུནྡྷ་ནྟི། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་དྷ་
ཡིག་གི་ཏ་ཡིག་གོ། །རུ་ཎཏྶི། རུནྡྷཿ། རུནྡྷ། རུ་ཎདྷྨི། རུ་ཎ་དྷཿ། རུནྡྷཱཿ། རུནྡྷེ། རུནྡྷཱ་ཏེ། ཨ་ཡིག་མ་ཡིན་པ་ལས་ཨཱཏྨ་ནེ་ལ་ཡང་ངོ་ཞེས་པས་ན་དབྱིའོ། །རུནྡྷ་ཏེ། རུན་ཏྶེ། རུནྟ་ཐེ། རུནྡྷྭེ། རུནྡྷེ། རུནྡྷྭ་ཧེ། རུནྡྷྨ་ཧེ། བི་རུ་ཎ་དྷི། བི་རུནྡྷེ། ཨུ་པ་རུ་ཎ་དྷི། ཨུ་པ་རུནྡྷེ། བྷི་དིར་རྣམ་པར་འཇོམས་པ་ལའོ། །
