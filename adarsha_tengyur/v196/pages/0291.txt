ཨ་ཛ་ཛན། ཨ་ཛ་ཛཱ་ཏཾ། ཨ་ཛ་ཛཱ་ཏ། ཨ་ཛ་ཛ་ནཾ། ཨ་ཛ་ཛཱ་བ། ཨ་ཛ་ཛཱ་མ། ངེས་པ་མེད་པ་ལའང་ཨཱ་ག་ཏི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྱ་ཏྱ་ཛ་ཛཱ་ཏ། བྱ་ཏྱ་ཛ་ཛྙཱ་ཏཱཾ། བྱ་ཏྱ་ཛ་ཛྙཱ་ཏ། བྱ་ཏྱ་ཛ་ཛཱ་ཐཱཿ། བྱ་ཏྱ་ཛ་ཛྙཱ་ཐཱཾ། ཨཱིཊ་ཛན་དག་
གིས་དང་དྷྭ་ལའང་ངོ་ཞེས་པས་ཨིཊ་ཨཱ་ག་མའོ། །བྱ་ཏྱ་ཛ་ཛྙི་དྷྭཾ། ཎི་ཛིར། ཨ་ནེ་ནེཀ །ཨ་ནེ་ནེག །ཨ་ནེ་ནིཀྟཱ། ཨ་ནེ་ནི་ཛུཿ། ཨ་ནེ་ནེཀ །ཨ་ནེ་ནེག །ཨ་ནེ་ནིཀྟཾ། ཨ་ནེ་ནི་གྷ། ཨ་ནེ་ནེ་ཛཾ། ཨ་ནེ་ནི་ཛྦ། ཨ་ནེ་ནི་ཇྨ། ཨ་ནེ་ནིཀྟ། ཨ་ནེ་ནི་ཛཱ་ཏཱཾ། ཨ་ནེ་ནི་ཛཏ། དི་བཞིན་དུ་བི་་ཛྲིར་འོ། །བི་ཥློ། ཨ་བ་བེཊ།

ལའོ། །ཨ་སཱུ་ཡ་ཏ། ཨ་སཱུ་ཡེ་ཏཱཾ།ཨ་སཱུ་ཡནྟ། ཨ་སཱུ་ཡ་ཐཱཿ། ཨ་སཱུ་ཡེ་ཐཱཾ། ཨ་སཱུ་ཡ་དྷྭཾ། ཨ་སཱུ་ཡི། ཨ་སཱུ་ཡཱ་བ་ཧི། ཨ་སཱུ་ཡཱ་མ་ཧི། ཥུཉ། ཨ་སུ་ནོ་ཧ། ཨ་སུ་ནུ་ཏཱཾ། ཨ་སུ་ནྭན། ཨ་སུ་ནོཿ། ཨ་སུ་ནུ་ཏ། ཨ་སུ་ན་བཾ། ཨ་སུ་ནྭ། ཨ་སུ་ནུ་བ། ཨ་སུནྨ། ཨ་སུ་ནུ་མ། ཨ་སུ་ནུ་ཏ། ཨ་སུ་
ནྭཱ་ཏཱཾ། ཨ་སུ་ནྭ་ཏ། ཨ་སུ་ནུ་ཐཱཿ། ཨ་སུ་ནྭ་ཐཱཾ། ཨ་སུ་ནུ་དྷྭཾ། ཨ་སུ་ནྭི། ཨ་སུ་ནུ་བ་ཧི། ཨ་སུ་ནུ་མ་ཧི། ཨ་སུ་ནྭ་ཧི། ཨ་སུནྨ་ཧི། ཉེར་བསྒྱུར་ལས་སུ་ནོ་ཏིའི་ཞེས་པ་ལ་སོགས་པས་ཥ་ཉིད་དོ། །ཨ་བྷྱ་ཥུ་ཎོ་ཏ། ཨ་བྷྱ་ཥུ་ཎུ་ཏ། ཨ་པླྀ། ཨ་ཡིག་གི་ཨཱ་ཡིག་སྟེ། འཕེལ་བའི་དམིགས་བསལ་ལོ། །
