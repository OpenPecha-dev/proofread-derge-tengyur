བཾ་ཀྟི། ཡོན་ཏན་མེད་པ་ལ་ཨ་ནུ་ཥངྒ་དབྱིའོ། །བ་ནཱི་བཀྟཿ། བ་ནཱི་བ་ཙི་ཏི། བ་ནཱི་བཉྩི་ཥ། བ་ནཱི་བཾ་ཀྵི། བ་ནཱི་བ་ཀཿ། བ་ནཱི་བཀྵ། བ་ནཱི་བཉྩི་མི། བ་ནཱི་བཉྩྨི། བ་ནཱི་བ་ཙྭཿ། བ་ནཱི་བ་ཙྨཿ། སྲནྶུ། ས་ནཱི་སྲཾ་སཱི་ཏི། ས་ནཱི་སྲཥྚི། ས་ནཱི་སྲཥྚཿ། ས་ནཱི་
སྲ་ས་ཏི། ས་ནཱི་སྲཾ་སཱི་ཥི། ས་ཥི་སྲཥྛཿ། ས་ནཱི་སྲཥྛ། ས་ནཱ་སྲ་མ་སཱི་མི། སཱ་ནཱི་སྲཾ་སྨི། ས་ནཱི་སྲ་སྭཿ། ས་ནཱི་སྲ་མཿ། དྷནྶུ། ད་ནཱི་དྷྭཾ་སྟི། ཏི། དཱ་ནཱཾ་དྷྭཾ་སྟི། ད་ནཱ་དྷྭསྟཿ། ད་ནི་དྷྭ་ས་ཏི། བྷྲནཱུ། བ་ནཱི་བྷྲཾ་ཤཱི་ཏི། དྷུ་ཊ་ལ་ཚ་ཤ་དག་གིའང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །
