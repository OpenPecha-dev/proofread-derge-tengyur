ཏི་སྟྱཱ་པ་ཡི་ཥ་ཏི། སཱ་དྷ། སི་སཱ་དྷ་ཡི་ཥ་ཏི། སྠ་གི། ཏི་སྠ་ག་ཡི་ཥ་ཏི། དེ་ནས་ཥ་ཉེ་བར་བསྟན་པ་ལ་སོགས་འདོད་དོ། །ཏིཥྛ་ག་ཡི་ཥ་ཏི། སནྱཱུ། སི་སྱནྡ་ཡི་ཥ་ཏི། སྑྱ་ལ། ཙི་སྑྱ་ལ་ཡི་ཥ་ཏི། སྥཱ་ཡི། སི་སྥཱ་ཡ་ཡི་ཥ་ཏི། སེ་ནི། སྡེ་མངོན་པར་འདོད་དུ་འཇུག་པར་
བྱེད་པ་ནི། ཨ་བྷི་ཥི་ཥཱ་ན་ཡི་ཥ་ཏི། སྠཱ། སེ་ནི། སེ་དྷ་ཏི་ཞེས་པ་ལ་སོགས་པས་ཥ་ཉིད་དོ། །པྲ་ཀ་ཊི། རེ་ལྡེ་རབ་ཏུ་འདོད་དུ་འཇུག་པར་བྱེད་པ་ནི། པྲ་ཙི་ཀ་ཊ་ཡི་ཥ་ཏི། ཀ་པ་ལི་ཁམ་ཟས་འདོད་དུ་འཇུག་པར་བྱེད་པ་ནི། ཙི་ཀ་པ་ལ་ཡི་ཁ་ཏི། དྷ་བ་ལི། དཀར་པོ་འདོད་དུ་འཇུག་པར་བྱེད་
པ་ནི། དི་དྷ་བ་ལ་ཡི་ཥ་ཏི། དྱུ་ཥ་ཏི། རྒྱུ་ལ་ཨིནའོ། །དྱུ་ཏི་སྭཱ་པི་དག་གི་ཨ་བྷྱཱ་སའི་འོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་སཾ་པྲ་སཱ་ར་ཎའོ། །དི་དྱོ་ཏ་ཡི་ཥ་ཏི། སྥོ་རི། ཡི་གེ་རྣམ་པར་འགྱུར་བའི་ཕ་རོལ་གྱི་ཨོ་ཡིག་གི་ཕྱོགས་ལ་ཡང་ན་ཨ་ཉིད་དོ། །པུ་སྥཱ་ར་ཡི་ཥ་ཏི། པུ་སྥོ་ར་ཡི་ཥ་ཏི། ཛྙཱ། པི། ཨི་ཡིག་
དང་རྀ་དྷ་ཞེས་པ་ལ་སོགས་པས་ཡང་ན་ཨིཊའོ། །རྀ་དྷི་ཛྙཱ་པ་དག་གི་ཨཱིར་དང་ཨིཏ་དག་བརྗོད་པར་བྱ་བ་དག་གོ་ཞེས་པས་རྀ་ཡིག་དང་ཨཱཏ་དག་སན་ལ་དབྱངས་ཀྱི་ཨཱིར་དང་ཨིཏ་དག་གོ། །ཨ་བྷྱཱ་ས་དབྱི་བ་འང་ངོ་། །ཀཱ་རི་ཏའི་ནི་ཕ་རོལ་ཉིད་ལས་དབྱིའོ། །ཛྙཱི་པྶ་ཏི། ཛི་ཛྙཱ་པ་ཡི་ཥ་ཏི། ཞེས་པ་
ཨིན་མཐའ་ལས་སན་མདོར་བསྟན་པའོ།།      །།དཔལ་ངག་གི་དབང་ཕྱུག་རིག་འཛིན་ཆེན་མོ་སྒྲོལ་མ་དཀར་མོ་ལ་ཕྱག་འཚལ་ལོ། །བྱིངས་ལས་བྱ་བ་ཡང་དག་པར་མངོན་པར་སྡུད་པ་ལ་ཡ་སྒྲ་སྟེ་ཙེ་ཀྲཱི་ཡི་ཏའོ་ཞེས་པས་བྱ་བ་ཡང་དག་པར་མངོན་པར་སྡུད་པ་ལ་འཇུག་བཞིན་པ་
ལ་གསལ་བྱེད་དང་པོའི་བྱིངས་ལས་ཡ་སྒྲ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །དེའང་ཙེ་ཀྲཱི་ཡི་ཏའི་མིང་ཅན་ནོ། །ཙེ་ཀྲཱི་ཡི་ཏའི་མཐའི་ཡུལ་ལ་ཙཎ་པ་རོཀྵཱ་ཞེས་པ་ལ་སོགས་པས་དང་པོའི་དབྱངས་གཅིག་པའི་གཉིས་བརྗོད་དོ། །ཙེ་ཀྲཱི་ཡི་ཏ་ལ་ཡོན་ཏན་འང་ངོ་ཞེས་པས་ཙེ་ཀྲཱི་ཡི་ཏ་ལ་གང་ཨ་བྷྱཱ་ས་དེའི་ཡོན་ཏན་ནོ། །ནཱ་མིན་ཉིད་
ཀྱི་ཡི་གེའི་མཐའ་ཉིད་ལ་ཨ་བྷྱཱ་སའི་རྣམ་འགྱུར་ལ་ནི་དམིགས་ཀྱིས་བསལ་བས་སྤྱིར་བཏང་བ་ལ་གནོད་པ་མ་ཡིན་ནོ། །ཨཱ་ག་མ་མེད་པའི་རིང་པོའོ་ཞེས་པས་ཙེ་ཀྲཱི་ཡི་ཏ་ལ་གང་ཨ་བྷྱཱ་ས་དེའི་ཨཱ་ག་མ་མེད་པའི་རིང་པོའོ། །ཙེ་ཀྲཱི་ཡི་ཏའི་མཐའ་ལས་སོ་ཞེས་པས་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྱེད་པ་པོ་ལ་ཨན་བི་ཀ་ར་
