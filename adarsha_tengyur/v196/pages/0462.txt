པ་རྣམ་པར་བཤད་པའི་དོན་ལས་ཏིའི་རྐྱེན་ལ་ཨིཊ་མེད་དོ། །བ་རཱི་བྲིཏྟཿ། བ་རི་བྲིཏྟཿ། བརྦིྲཏྟཿ། བ་རཱི་བྲི་ཏ་ཏི། བརྦྲི་ཏ་ཏི། བ་རི་བརྟྶི། བ་རཱི་བྲི་ཏཱི་ཥི། བ་རི་བརྟྶི། བ་རཱི་བྲི་ཏཱི་ཥི། བརྦརྟྶི། བརྦྲི་ཏཱི་ཥི། བ་རཱི་བྲིཏྠཿ། བ་རི་བྲིཏྠཿ། བརྦྲིཏྠཿ། བ་རཱི་བྲིཏྠ། བརྦྲ་ཏྠ། བ་རི་བྲི་ཏཱི་མི།
བ་རི་བརྟྨི། བ་རི་བྲི་ཏི་མི། བརྦ་རྟྨི། བརྦི་ཏཱི་མི། བ་རཱི་བྲི་ཏྭཿ། བ་རི་བྲི་ཏྭཿ། བ་རི་བྲི་ཏྭཿ། བརྦིྲ་ཏྭཿ། བ་རཱི་བྲིཏྨ། བ་རི་བྲིཏྨཿ། བ་རྦིྲཏྨཿ། ནྲྀ་ཏཱི། ན་རི་ནརྟྟི། ན་རཱ་ནྲི་ཏཱི་ཏི། ན་རི་ནིརྟྟ། ན་རཱི་ནྲི་ཏཱི་ནི། ནརྣརྟྟི། ནཪྣྲི་ཏི་ཏི། ཨཱ་ག་མའི་ཕྱོགས་གཅིག་ལ་སྔ་མ་བཞིན་ཏིའི་རྐྱེན་ལ་
ཨིཊ་མེད་དོ། །ན་རཱི་ནྲིཏྟཿ། ན་རི་ནྲིཏྟཿ། ན་རི་ནྲིཏྟཿ། ནརྣྲིཏྟཿ། ན་རཱི་ནྲི་ཏ་ཏི། ན་རི་ནྲི་ཏ་ཏི། ནཪྣྲི་ཏ་ཏི། བྲྀ་དྷ། ཏ་ཐ་དག་གི་དྷའོ། །སྔ་མ་བཞིན་གསུམ་པ་ཉིད་ཀྱང་ངོ་། །བ་རི་བཪྡྷི། བ་རཱི་དྷི་ཏི། བ་རི་བདྡྷི། བ་རཱི་བྲི་དྷཱི་ཏི། བརྦིྲ་དྷི་ཏི། བཱ་རཱི་བྲིདྡྷཿ། བ་རི་བྲིདྡྷཿ། བརྦིྲདྡྷཿ། 
བ་རཱི་བྲི་དྷ་ཏི། བ་རི་བྲི་དྷ་ཏི།བརྦིྲ་དྷ་ཏི། ཤྲྀ་དྷུ། ཤ་རཱི་ཤརྡྷི། ཤ་རཱི་ཤྲི་དྷཱི་ཏི། ཤ་ར་ཤརྡྡྷི། ཤ་རི་ཤྠི་དྷཱི་ཏི། ཤཪྴ་རྡྷི། ཤཪྴི་དྷི་ཏི། ཀྲུ་པཱུ། ཀྲི་པའི་རའི་ལའོ། །ར་ཐོས་པ་ལས་ལ་ཐོས་སོ། །ཙ་ལཱི་ཀལྤྟ། ཙ་ལཱི་ཀླི་པཱི་ཏི། ཙ་ལི་ཀལྤྟ། ཙ་ལཱི་ཀླི་པི་ཏི། ཙ་ལ་ཀལྤི། ཙ་ཀླི་
པཱི་ཏི། ཙ་ལཱ་ཀླིཔྟཿ། ཙ་ལི་ཀླིཔྟཿ། ཙ་ལི་ཀླིཔྟཿ། ཙ་ལི་ཀླི་པ་ཏི། ཙ་ལི་ཀླི་པ་ཏི། ཙ་ལི་ཀླི་པ་ཏི། སྲྀ་ཛི་དྲ་ཤ་དག་གི་ཨ་ཡིག་ཨཱ་ག་མ་ཞེས་བྱ་བ་ལ་སོགས་པས་ཙུ་ཊ་སོགས་ལ་ཡོན་ཏན་ཅན་ལ་དབྱངས་ལས་ཕ་རོལ་དུ་ཨ་ཡིག་ཨཱ་ག་མའོ། །རྀའི་ཡི་གེ་རའོ། །སྲྀ་ཛིའི་དྷུཊ་ལ་སོགས་པ་
ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །དྲྀ་ཤིའི་ཚ་ཤ་དག་གི་འང་ངོ་ཞེས་པས་ཥ་ཉིད་དོ། །སཱ་རཱི་སྲཥྚ། སཱ་རཱི་སྲི་ཛི་ཏི། ས་རི་སྲཥྚ། ས་རི་སྲི་ཛི་ཏི། སཪྶཥྚི། སཪྶྲི་ཛི་ཏི། ས་རི་སྲིཥྚཿ། སཪྶིཥྟཿ། ས་རཱི་སྲི་ཛ་ཏི། ས་རི་སྲི་ཛི་ཏི། སཪྶྲི་ཛཱི་ཏི། ད་རི་དྲཥྚི། ད་རི་ཤཱི་ཏི། ད་རཱི་
དྲིཥྚིཿ། ད་རི་དྲི་ཤཱི་ཏི། དཪྡྲཥྚི། ད་ར་དྲི་ཤཱི་ཏི། ད་རཱི་དྲིཥྚཿ། ད་རི་དྲིཥྚཿ། ད་ར་དྲིཥྚཿ། ད་རཱི་ཥ་ཏི། ད་རི་དྲི་ཤ་ཏི། དཪྡྲ་ཤ་ཏི། མྲྀ་ཛཱུ། མརྫྫའི་མཱརྫྫིའོ། །བྷི་ཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །མ་རཱི་མཱཪྵྚི། ཡོན་ཏན་མེད་པའི་དབྱངས་ལ་ཡང་ན་བརྗོད་པར་བྱའོ། །
