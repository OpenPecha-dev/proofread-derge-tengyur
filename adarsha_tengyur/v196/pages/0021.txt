གཱ་བིདྷཽཿ། མྲྀ་ག་བི་དྷཿ། མྲྀ་གཱ་བི་དྷཱ། དྷུཊ་རྣམས་ཀྱི་གསུམ་པའོ། །མྲྀ་གཱ་བིདྦྷྱཱམ། མྲྀ་གཱ་བིདྦྷྱཱཿ ཞེས་པ་ལ་སོགས་པའོ། །ཨཱཏྨན་གྱི་སྒྲ་ན་མཐའ་ཅན་ལས་སི་ལ་སོགས་པ་རྣམས་སོ། །གྷུ་ཊ་ལའང་སཾ་བུདྡྷི་མ་ཡིན་པ་ལའོ་ཞེས་པས་གྷུཊ་ལ་ཨུ་པ་དྷའི་རིང་པོ་ཉིད་དོ། །རྟགས་
མཐའི་ན་ཡིག་གོ་ཞེས་པས་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་ལའང་ན་ཡིག་དབྱིའོ། །ཨཱཏྨཱ། ཨཱཏྨཱ་ནཽ། ཨཱཏྨཱ་ནཿ། ཨཱཏྨཱ་ནཱཾ། ཨཱཏྨཱ་ནཽ། ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཨ་ནའི་ཨ་དབྱི་བ་དང་སྔ་མའི་ཆོ་ག་ལ་མ་ཕྱིས་པ་བཞིན་འང་ངོ་ཞེས་པ་ཨ་ཡིག་དབྱི་བ་ཐོབ་པ་ལ། ཝ་མ་བརྩེགས་པ་མ་
ཡིན་པ་ལས་ཞེས་པས་བཀག་པའོ། །ཨཱཏྨཱ་ནཿ། ཨཱཏྨཱ་ནཱ། ན་དང་བརྩེགས་པའི་མཐའ་ཕྱིས་པ་དག་ནི་ཆོ་ག་སྔ་མ་ལ་མ་ཕྱིས་པ་བཞིན་འང་ངོ་ཞེས་པས། ན་ཡིག་ཆོ་ག་སྔ་མ་བྱ་བ་ལ་ཕྱིས་ཀྱང་མ་ཕྱིས་པ་དང་འདྲ་བར་འགྱུར་ཏེ། དེས་ན་ཨ་ཡིག་སྒྲ་ལྡན་ལ་རིང་པོའོ་ཞེས་པ་ལ་སོགས་པ་ཉིད་དུ་འགྱུར་བ་
མ་ཡིན་ནོ། །ཨཱཏྨཱ་བྷྱཱམ། ཨཱཏྨཱ་བྷིཿ། ཨཱཏྨཱ་ནེ། ཨཱཏྨ་བྷྱཱམ། ཨཱཏྨ་བྷྱཿ། ཨཱཏྨ་ནཿ། ཨཱཏྨ་བྷྱཱམ། ཨཱཏྨཱ་བྷྱཿ། ཨཱཏྨ་ནཿ། ཨཱཏྨ་ནོཿ། ཨཱཏྨ་ནཱཾ། ཨཱཏྨ་ནི། ཨཱཏྨ་ནོཿ། ཨཱཏྨ་སུ། ཀུན་དུ་བོད་པ་ལ་སཾ་བུདྡྷི་མ་ཡིན་པ་ལའོ་ཞེས་དགག་པའི་ཕྱིར་ཨུ་པ་དྷཱ་རིང་པོ་ཉིད་དུ་མི་འགྱུར་རོ། །སཾ་བུདྡྷི་མ་ཡིན་པ་

གྷུཊ་མ་ཡིན་པའི་དབྱངས་ལ་ནི། ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལ་སོགས་པས་ཨ་དབྱིའོ། །ཏའི་སྡེ་པ་ནི་ཙ་ཊའི་སྡེ་པ་སྦྱོར་བ་ལ་ཙ་ཊའི་སྡེ་པ་དག་གོ་ཞེས་པས་ན་ཡིག་ཉ་ཡིག་གོ། །རཱ་ཛྙཱཿ། རཱ་ཛྙཱ་བྷྱཱཾ། རཱ་ཛབྷིཿ།རཛྙེ། རཱ་ཛ་བྷྱམ། རཱ་ཛ་བྷྱཿ། རཱ་ཛྙཿ། རཱ་ཛ་བྷྱཱམ། རཱ་ཛ་བྷྱཿ། རཱ་ཛྙཿ། 
རཱ་ཛྙཽཿ། རཱ་ཛྙཱ། ཨི་དང་ངི་དག་ལ་ཡང་ནའོ་ཞེས་རྣམ་པར་བརྟག་པས་ཨ་ཡིག་དབྱི་བའོ། །རཱ་ཛྙི། རཱ་ཛ་ནི། རཱ་ཛྙོཿ། རཱ་ཛ་སུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་རཱ་ཛན། ཧེ་རཱ་ཛི་ནཽ། ཧེ་རཱ་ཛ་ནཿ། དེ་བཞིན་དུ། མཱརྡྷན། མཛྫ་ན། ཨུཀྵན། གྲཱ་ཁན། བྲྀཀྐན། ཏཀྵན། པ་ཏྲྀ་མན། ལགྷི་མན། མ་ཧི་མན།
