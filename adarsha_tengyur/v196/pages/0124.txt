མིང་ཅན་ལས་གྷུཊ་ལ་སྔ་མ་བཞིན་ནོ། །བྲཧྨ་ཧཱ། བྲཧྨ་ཧཱ་ཎཽ། བྲཧྨ་ཧ་ཎཿ། བྲཧྨཧ་ཎཾ། བྲཧྨ་ཧ་ཎཽ། གྷུཊ་མ་ཡིན་པའི་དབྱངས་ལ། ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཨན་གྱི་ཨ་དབྱི་བ་ཞེས་པ་ལ་སོགས་པས་ཨ་དབྱི་བ་བྱས་པ་ན་ཧ་ནིའི་ཧིའི་གྷི་སྟེ་ཨུ་པ་དྷ་ཕྱིས་པ་ལའོ་ཞེས་པས་ཧ་གྷ་ཡིག་
གོ། །བྲཧྨ་གྷྣཿ། བྲཧྨ་གྷྣཱ། བྲཧྨ་ཧ་བྷྱཱམ། བྲཧྨ་ཧ་བྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་བྷྲཱུ་ཎ་ཧན་དང་བྲྀཏ་ཧན་ནོ།།      །།ཧ་ལིན་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ཨིན་ཧན་ཞེས་པ་ལ་སོགས་པས་རིང་པོ་ཉིད་དོ། །ཧ་ལཱི། ཧ་ལི་ནཽ། ཧ་ལི་ནཿ། ཧ་ལི་ནཾ། ཧ་ལི་ནཽ།
ཧ་ལི་ནཿ། ཧ་ལི་ནཱ། ན་དབྱིའོ། །ཧ་ལི་བྷྱཱཾ། ཧ་ལི་བྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་ཤ་རཾ་གིན། ལཱངྒ་ལིན། ཀ་པ་རིན། ཤུ་ལིན། མུ་ཁ་ལིནྲ། ཤ་ཤིན། ཀི་སོ་རིན་ནོ།།      །།བིཤྲ་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། གསལ་བྱེད་ལས་ཀྱང་ཞེས་པས་སི་དབྱི་ཞིང་། ཧ་
དང་ཤ་དང་ཥ་དང་ཚ་མཐའ་དང་ཡཛྲ་སོགས་རྣམས་ཀྱི་ཌའོ་ཞེས་པས་ཌ་ཡིག་སྟེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཊ་ཡིག་གམ་ཌ་ཡིག་གོ། །པིཊྲ། བིཌྲ། བི་ཤཽ། བི་ཤཿ། བི་ཤཾ། བི་ཤཽ། བི་ཤཿ། བི་ཤཱ། བིཌྦྷྱཱམ། བིཌྦྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །དྭི་ཥྲའི་སྒྲ་ལས་སི་ལ་སོགས་
པ་རྣམས་ནི། སྔ་མ་བཞིན་དུ་ཌ་ཉིད་དེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཊ་ཡིག་གམ་ཌ་ཡིག་གོ། །དྭིཊ། དྭིཊ། དྭི་ཥཾ། དྭི་ཥཿ། དྭི་ཥཾ། དྭི་ཤཽ། དྭི་ཥཿ། དྭི་ཥཱ། དྭིཌྦྷྱཱམ། དྭིཌྦྷིཿ། ཞེས་པ་ལ་སོགས་པའོ། །བེ་དྷསྲ་ཀྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ཨནྟུ་དང་ཨས་མཐའི་འང་བྱིངས་
མ་ཡིན་པའི་སི་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དང་། ས་ཡིག་ནི་རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱ་སྟེ། གསལ་བྱེད་ལས་ཀྱང་ཞེས་པས་སི་དབྱིའོ། །བེ་དྷཱཿ། བེ་དྷ་སཽ། བེ་དྷ་སཿ། བེ་དྷ་སཾ། བེ་དྷ་སཽ། བེ་དྷ་སཿ། བེ་དྷ་སཱ། ཨིས་ཨུས་དོས་རྣམས་ཀྱི་སྒྲ་ལྡན་ལ་རའོ་ཞེས་པ་སྤྱིར་བཏང་ནས། རེ་
ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱ་སྟེ། ཨ་དང་སྒྲ་ལྡན་དག་གིའང་ངོ་ཞེས་ཨུ་ཉིད་ཡིན་ལ། ཨུའི་ཡི་གེ་ལ་ཨོ་སྟེ་ཕ་རོལ་ཡང་དབྱིའོ། །བེ་དྷོ་བྷྱཱཾ། བེ་དྷོ་བྷཾཿ། བེ་དྷ་སེ། བེ་དྷོ་བྷྱཱཾ། བེ་དྷོ་བྷྱཿ། བེ་དྷ་སཿ། བེ་དྷོ་བྷྱཱཾ། བེ་དྷོ་བྷྱཿ། བེ་དྷ་སཿ། བེ་དྷ་སོཿ། བེ་དྷ་སཱི། བེ་དྷ་སི། བེ་དྷ་སོཿ། སྒྲ་
