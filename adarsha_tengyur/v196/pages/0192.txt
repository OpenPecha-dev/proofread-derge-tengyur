པྲི་ཡཥྚཱ་བྷྱཿ། པྲི་ཡཱཥྚཱཿ། པྲི་ཡཱཥྚཽ། སྔ་མ་དང་འདྲ་བའི་ནུ་མེད་ད་གྲངས་དང་འབྲེལ་པ་ལ་འདི་མ་ཡིན་ནོ་ཞེས་རྣམ་པར་བཤད་པའི་ཕྱིར་རོ། །པྲི་ཡཱཥྚཱཾ། ཨའི་ཡི་གེ་ཨིའི་ཡི་གེ་ལ་ཨེའོ། །པྲི་ཡཱཥྚེ། པྲི་ཡཱཥྚཽ། པྲི་ཡཱཥྚ་སུ། ཀུན་དུ་བོད་པ་ལ། ཧེ་པྲི་ཡཱཥྚཱཿ། ཧེ་པྲི་ཡཱ་ཥྚཽ། ཧེ་པྲི་ཡཱ་ཥྚཽ། དེ་
བཞིན་དུ་བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ་ནི་དེ་ལ་བརྗོད་པ་བཀག་པའི་ཕྱིར་ཨཱ་ཉིད་མེད་དེ། པྲི་ཡཱཥྚ། ཨཽ་དང་གསུམ་པ་རྣམས་ལ་ཨཱ་ཉིད་དོ། །ཨཽ་ཨཱིའོ། །པྲི་ཡཱཥྚེ། དྷུཊ་དང་དབྱངས་ལས་གྷུཊ་ལ་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ། །པྲི་ཡཱཥྚཱ་ནི། སླར་ཡང་དེ་བཞིན་དེ་ཊཱ་ལ་སོགས་པའི་དབྱངས་
ལ་སྐྱེས་པ་དང་མཚུངས་སོ། །ཀུན་དུ་བོད་པ་ལ། པྲི་ཡཱཥྚཱ། ཧེ་པྲི་ཡཱཥྚཱན། ཧེ་པྲི་ཡཱ་ཥྚཽ། ཧེ་པྲི་ཡཱཥྚཱ་ནི། ན་མཐའི་ཕྱོགས་གཅིག་ལ་པཉྩན་གྱི་སྒྲ་དང་འདྲའོ། །དེ་བཞིན་དུ། ན་ཝན་དང་། ད་ཤན་ནོ།།      །།གཅིག་གིས་ལྷག་པ་ཅན་གྱི་བཅུ་ནི་ཨེ་ཀཱ་ད་ཤ་སྟེ། ཚིག་འགའ་ཞིག །ཉམས་པ་དང་བཅས་
པའི་དོན་ལ་ ཚིག་སྡུད་དུ། གསུངས་པ་མཐོང་ཤིང་བསྟན་པའི་ཕྱིར་བར་གྱི་ཚིག་ཕྱིས་པ་ཅན་གྱི་བསྡུ་བའོ། །གཅིག་དང་བཅུ་ཡང་ཡིན་ལ་ཡང་ན་ཟླཱས་དབྱེའི་བསྡུ་བ་སྟེ། ཐུང་ངུའི་རིང་པོ་ཉིད་དོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཨེ་ཀཱ་ད་ཤ །བཅུ་གཅིག །འགྲེལ་པར་གསུངས་པ། བརྒྱ་ནས་གསལ་བར་གྲངས་
རྣམས་ལ་འབྲུ་མང་པོ་པ་སྟེ། གཅིག་གི་ཡང་གསལ་བ་ནི་མ་ཡིན་ནོ་ཞེས་སོ། །དྭ་ད་ཤ་ བཅུ་གཉིས།དེ་བཞིན་ནོ། །ཏྲིའི་ཏྲ་ཡ་རུ་བསྒྱུར་ཏེ། ཤས་ལ་ཡང་ངོ་ཞེས་པས། ཏྲ་ཡོ་ད་ཤ །རེ་ཕ་དང་ས་དག་གི་རྣམ་པར་གཅད་བྱའོ། །ཨ་དང་སྒྲ་ལྡན་དག་གིའང་ངོ་ཞེས་པས་ཚེག་དྲག་ཨུ་ཉིད་དེ།
ཨུའི་ཡི་གེ་ལ་ཨཽའོ། །ཏྲ་ཡ་རུ་བསྒྱུར་བ་ན། གསལ་བྱེད་མཐའི་གང་སུ་བྷ་དག་ལའོ་ཞེས་བསྟན་པ་ལས་སོ། །ཙ་ཏྭཱར་གྱི་ཝཱ་སྒྲའི་ཨུ་ཉིད་དེ། ཙ་ཏུར་ད་ཤ །བཅུ་བཞི། ལྔ་རྣམས་ཀྱི་ལྷག་པའི་བཅུ་ནི་པཉྩ་ད་ཤའོ་ཞེས་ཡང་ནའི་ བར་གྱི་ཚིག་ཕྱིས་པ་དང་ཡང་ན་ཟླས་དབྱེ། ཚིག་སྡུད་བྱས་པ་ལ་སྔ་མ་བཞིན་ནོ་ཞེས་བསྟན་པ་ལས། རྟགས་མཐའི་ན་
ཡིག་གི་གོ་ཞེས་པས་ན་དབྱིའོ། །པཉྩ་ད་ཤ །བཅོ་ལྔ། དྲུག་རྣམས་ཀྱིས་ལྷག་པའི་བཅུ་ནི་ཥཊ་ད་ཤ་ཡང་སྟེ། ཡང་ནའི་ཚིག་སྡུད་ལ་ཥ་ཥ་ཡིག་གི་ཨུ་ཉིད་དང་ད་ཤན་གྱི་སྒྲའི་ད་ཡིག་ནི་ཌ་ཉིད་དོ། །ཥོ་ཌ་ཤ །བཅུ་དྲུག །གྲངས་རྣམས་ནི་འབྲུ་མང་པོ་ཅན་ནོ་ཞེས་པས་ཨཱ་
