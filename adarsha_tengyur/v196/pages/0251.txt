པྲཱིཉ་ཚིམ་པ་ལ་སྟེ་མཛེས་པ་ལ་ཡང་ངོ་། །པྲཱི་ཎཱ་ཏི། པྲཱི་ཏཱི་ཏེ། ཤྲཱིཉ་འཚེད་པ་ལའོ། །ཤྲཱི་ཧཱ་ཏི། ཤྲཱི་ཎཱི་ཏེ། སྐུཉ་ཀུན་དུ་གྲོལ་བ་ལའོ། །སྐུ་ནཱ་ཏི། སྐུ་ནཱི་ཏི། ཥིཉ་ཥུཉ་འཆིང་བ་ལའོ། །སི་ནཱ་ཏི། སི་ནཱི་ཏེ། སུ་ནཱ་ཏི། སུ་ནཱི་ཏེ། ཀུཉ་སྒྲ་ལའོ། །ཀུ་ནྫ་ཏི། ཀུ་ནཱ་ཏེ། མཱིཉ་དྲུཉ་འཚེ་བ་ལའོ། །མཱི་ནཱ་ཏི། མཱི་ནཱི་ཏེ། ཧི་
ནུ་མི་ན་ལ་སོགས་པ་རྣམས་ཀྱི་ཞེས་པས་ཎ་ཉིད་དོ། །པྲ་མཱི་ཎཱ་ཏི། པྲ་མཱི་ཎཱི་ཏེ། དྲུ་ཎཱ་ཏི། དྲཱུ་ཎཱི་ཏེ། པུ་ཉ་དག་བྱེད་ལའོ། །པཱུ་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ་ཞེས་པས་པཱུ་སོགས་རྣམས་ཀྱི་ཐུང་ངུའོ། །པུ་ནཱ་ཏི། པུ་ནཱི་ཏེ། ལཱུཉ་གཅོད་པ་ལའོ། །ལུ་ནཱ་ཏི། ལུ་ནཱི་ཏེ། སྟིཉ་འགེབས་པ་ལའོ། །སྟི་ཎཱ་ཏི། སྟི་ཎཱི་ཏེ། 
ཨ་སྟྲཱི་ཎཱ་ཏི། ཨཱ་སྟྲཱྀ་ཎཱི་ཏེ། བི་སྟྲཱྀ་ཎཱ་ཏི་ཀཎྜ་པ་ཊཾ། བི་སྟྲྀ་ཎཱི་ཏི། ཀྲཱིཉ་འཚེ་བ་ལའོ། །ཀྲྀ་ཎཱ་ཏི། ཀྲྀ་ཎཱི་ཏེ། བྲྀཉ་སྒྲིབ་པ་ལའོ། །པྲྀ་ཎཱ་ཏི། བྲྀ་ཎཱི་ཏེ། དྷུཉ་འགུལ་བ་ལའོ། །དྷུ་ནཱ་ཏི། དྷུ་ནཱི་ཏེ། གྲ་ཧ་ཉེ་བར་ལེན་པའོ། །ཡ་ཛ་སོགས་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །གྲ་ཧི་ཛྱཱ་ཞེས་པ་ལ་སོགས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །ར་ཥ་རྀའི་
ཡི་གེ་རྣམས་ལས་ཞེས་པ་ལ་སོགས་པས་ནའི་ཎ་ཉིད་དོ། །གྲྀཧྞཱ་ཏི། གྲྀཧྞི་ཏེ། ཨུ་པ་གྲྀཧྞཱ་ཏི། ཨུ་པ་གྲྀཧྞཱི་ཏེ། པ་རི་གྲྀཧྞཱི་ཏེ། ཨ་ནུ་གྲྀཧྞཱ་ཏི། ཨ་ནུ་གྲྀཧྞཱི་ཏེ། ནི་གྲྀཧྞཱ་ཏི། ནི་གྲྀཧྞི་ཏེ། ཨ་བ་གྲྀཧྞཱ་ཏི། ཨ་བ་གྲྀཧྞི་ཏེ། རྣམ་པར་བཤད་པ་རྣམས་སོ། །ཤྲཱྀ་སྟྲི་འཚེ་བ་ལའོ། །ལྷག་མ་ལས་བྱེད་པ་པོ་ལ་པ་རསྨཻའི་ཚིག་གོ། །ཤྲྀ་
ཏཱ་ཏི། སྤྲྀ་ཎཱ་ཏི། པྲྀ་སྐྱོང་བར་བྱེད་པ་དང་རྫོགས་པར་བྱེད་པ་ལའོ། །པྲྀ་ཎཱ་ཏི། བླཱྀ་བྲཱྀ་སྒྲིབ་པ་ལའོ། །བླཱྀ་ནཱ་ཏི། བྲཱྀ་ཎཱ་ཏི། བྷྲྀ་བརྒྱད་བཀག་ལའོ། །བྷྲཱྀ་ཎཱ་ཏི། དྲཱྀ་རྣམ་པར་འཇོམས་པ་ལའོ། །དྲཱྀ་ཎཱ་ཏི། ཛྲྀ་ལང་ཚོ་ཉམས་པ་ལའོ། །ཛྲྀ་ཎཱ་ཏི། མྲྀ་ཚུལ་ལའོ། །ཎྲཱྀ་ཎཱ་ཏི། གྲཱཱྀ་སྒྲ་ལའོ། །གྲྀ་ཎཱ་ཏི། གྲྀ་ཛྱཱ་ལང་ཚོ་ཉམས་པ་ལའོ། །གྲ་ཧི་
