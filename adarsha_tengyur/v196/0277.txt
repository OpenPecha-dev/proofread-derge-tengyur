རུ་དི་ཏཱཏ། རུ་དི་ཏཾ། རུ་དི་ཏ། རོ་དཱ་ནི། རོ་དཱ་བ། རོ་དཱ་མ། དེ་བཞིན་དུ། ཉི་ཥྭ་པ། ཤྭ་ས། ཨན་ཡང་ངོ་ཞེས་པ་དེ་རྣམས་ཀྱི་དེ་ལ། ཡང་སྨོས་པ་ལས་ཨནྟིའི་མཐའི་ཡང་ཞེས་པས་ཎ་ཉིད་དོ། །པྲཱ་ཎི་ཏུ། སྭ་པི་ཏུ། ཤོ་པི་ཏུ། ཛཀྵ། ཛཀྵི་ཏུ། ཛཀྵི་ཏཱཏ། ཛཀྵི་ཏཱཾ། ཨ་བྷྱསྟ་
ལས་ཨནྟིའི་ན་དབྱིའོ། །ཛ་ཀྵ་ཏུ། ཛཱ་གྲྀ། ཛཱ་གརྟྟུ། ཛཱ་གྲྀ་ཏཱཏ། ཛཱ་གྲྀ་ཏཱཾ། ཛཱ་གྲ་ཏུ། ཛཱ་གྲྀ་ཧི། ཛཱ་གྲྀ་ཏཱཏ། ཛཱ་གྲྀ་ཏཾ། ཛཱ་གྲྀ་ཏ། ཛཱ་ག་རཱ་ཎི། ཛཱ་ག་རཱ་བ། ཛཱ་ག་རཱ་མ། ད་རི་དྲཱ། ད་རི་དྲཱ་ཏུ། ད་རི་དྲི་ཏཱཏ། ད་རི་དྲི་ཏཱཾ། ད་རི་དྲ་ཏུ། ད་རི་དྲཱ་ཧི། ད་རི་དྲཱི་ཏཱཏ། ད་རི་དྲི་ཏཾ། ད་རི་དྲི་ཏ། ད་རི་དྲཱ་ཎི། ད་རི་
དྲཱ་བ། ད་རི་དྲཱ་མ། ཙ་ཀཱ་སྲྀ། ཙ་ཀཱསྟུ། ཙ་ཀཱ་སྟཱཏ། ཙ་ཀཱནྟཱཾ། ཙ་ཀཱ་ས་ཏུ། དྷུཊ་རྣམས་ཀྱི་བཞི་པ་རྣམས་ལ་གསུམ་པའོ་ཞེས་པས་སའི་དའོ། །ཙ་ཀཱ་དྷི། ཙ་ཀཱསྟྭཱཏ། ཙ་ཀཱསྟཾ། ཙ་ཀཱསྟ། ཙ་ཀཱ་སཱ་ནི། ཙ་ཀཱ་སཱ་བ། ཙ་ཀཱ་སཱ་མ། ཤཱ་སུ། ཤཱསྟུ། ཤིཥྚཱཏ། ཤིཥྚཱམ། ཤཱ་ས་ཏུ།ཤཱསྟི་ལས་ཀྱང་ངོ་
ཞེས་པས་ཧིའི་དྷིའོ། །ཤཱ་སྟིའང་ཤཱ་རུ་འགྱུར་རོ། །ཤཱ་དྷི། ཤིཥྚཱཏ། ཤིཥྚཾ། ཤིཥྚ། ཤཱ་སཱ་ནི། ཤཱ་སཱ་བ། ཤཱ་སཱ་མ། ཙ་ཀྵིང་། ཨཱ་ཙཥྚཱཱཾ། ཨཱ་ཙཀྵཱ་ཏཱཾ། ཨཱ་ཙཀྵ་ཏཱཾ། ཨཱི་ར། ཨཱིརྟྟཾ། ཨཱི་རཱ་ཏཱཾ། ཨཱི་ར་ཏཱཾ། ཨཱི་ཌ། ཨཱིཊྚཾ། ཨཱི་ཌཱ་ཏཱཾ། ཨཱི་ཌ་ཏཱཾ། ཨཱི་ཊ་ཛ་ན་དག་གི་ས་དང་དྷ་ལའང་ངོ་ཞེས་པས་ཨིཊ་
ཨཱ་ག་མའོ། །ཨཱི་ཌི་ཥྭ། ཨཱི་ཌཱ་ཐཱཾ། ཨཱི་ཌི་དྷྭཾ། ཨཱི་ཤ །ཨཱིཥྚཱཾ། ཨཱི་ཤཱ་ཏཱཾ། ཨཱི་ཤ་ཏཱཾ། ཨཱི་ཤ་ལས་ས་ལའོ་ཞེས་པས་ཨིཊ་ཨཱ་ག་མའོ། །ཨཱི་ཤཱི་ཥྭ། ཨཱི་ཤཱ་ཐཱཾ། ཨཱིཊ་ཛན་དག་གི་ས་དང་དྷཱ་ལའང་ངོ་ཞེས་པས་ཨིཊ་ཨཱ་ག་མའོ། །ཨཱི་ཤི་དྷྭཾ། ཨཱ་ས། ཨཱསྟཱཾ། ཨཱ་སཱ་ཏཱཾ། ཨཱ་ས་ཏཱཾ། ཨཱསྶ། ཨཱ་
སཱ་ཐཱཾ། ཨཱ་སའི་གསུམ་པ་དའོ། །ཨཱ་དྷྭཾ། ཨཱང་ལས་ཤཱ་སུ། ཨཱ་ཤཱསྟཱཾ། ཨཱ་ཤཱ་སཱ་ཏཱཾ། ཨཱ་ཤཱ་ས་ཏཱཾ། བ་ས་ཀུན་དུ་འགེབས་པ་ལའོ། །བསྟཱཾ། བ་སཱ་ཏཱཾ། བ་ས་ཏཱཾ། ཥཱུང་སྲོག་ཆགས་མངལ་ནས་སྐྱེ་བ་ལའོ། །སཱུ་ཏཱཾ། སུ་བཱ་ཏཱཾ། སུ་བ་ཏཾ། སཱུ་ཥྭ། སུ་བཱ་ཐཱཾ། སཱུ་དྷྭཾ། སཱུ་དྷི་ལས་པཉྩ་མཱི་ལའོ་
ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །སུ་བཻ། སུ་བཱ་བ་ཧཻ། སུ་བཱ་མ་ཧཻ། ཤཱིང་ཤཱིངའི་བྱིངས་ཐམས་ཅད་པ་ལའོ་ཞེས་པས་ཡོན་ཏན་ནོ། །ཤེ་ཏཱཾ། ཤ་ཡཱ་ཏཱཾ། ཤེ་ཏི་ལས་ཨནྟིའི་དང་པོ་ཨིརའོ། །ཤེ་ར་ཏཱཾ། ཤེ་ཥྭ། ཤ་ཡཱ་ཐཱཾ། ཤེ་དྷྭཾ། ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཤེ་ཡེེ། ཤ་ཡཱ་བ་ཧཻ། ཤ་ཡཱ་
