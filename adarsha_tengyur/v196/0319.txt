ཨ་གྷུ་ཊིཏ། མུ་ད། ཨུ་པ་དྷཱ་ཡང་བའི་ཡོན་ཏན་ནོ། །ཨ་མོ་དིཥྚ། ཨ་མོ་དི་ཥཱ་ཏཱཾ། ཀྲྀ་ཏཱི་གཅོད་པ་ལའོ། །ཀྲྀ་ཏཱི་བསྐོར་བ་ལ་ཡང་ནའོ། །ཨ་ཀརྟྟིཏ། ཨ་ཀརྟིཥྚཾ། ནྲྀ་ཏཱི། ཨ་ན་རྟྟཱིཏ། བྲྀ་ཥ། ཨ་བཪྵི་ཏཱི། གྷྲྀ་ཥུ་ཡང་དག་པར་དགའ་བ་ལའོ། །ཨ་གྷཪྵི་ཏ། པྲྀ་ཧཱུ། ཡང་ན་ཨིཊའོ། །
ཨིཊ་མེད་པའི་ཕྱོགས་ལ་སཎའོ། །ཨ་བྲྀཀྵཏ། ཨ་བརྷཱིཏ། མྲྀ་ཛཱུ། མཱརྫའི་མཱརྫིའོ། །བྷྲྀཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །ཨ་མཱཪྵཱིཏ། ཨ་མཱཪྵ་ཊཱཾ། ཨ་མཱརྐྵུཿ། ཨ་མཱརྫཱི་ཥྚཱཾ། ཨོ་ནྲྀ། དབྱངས་དང་པོ་རྣམས་ཀྱི་དང་པོའི་འཕེལ་བའོ། །ཨཽ་ཎཱིཏ། ཨཽ་ཎི་ཥྚཱཾ། ལོ་ཙྦྀ་བལྟ་བ་ལའོ། །ཨ་ལོ་ཙིཥྚ། ཨ་

རིང་པོའོ། །བྷྲྀ་ཛ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཥ་ཉིད་དོ། །ཨ་སྲཱཀྵིཏ། དྷུཊ་ལས་ཀྱང་དྷུཊ་ལའོ་ཞེས་པས་སི་ཙ་དབྱིའོ། །ཨ་སྲཱཥྚཱཾ། ཨ་སྲཱཀྵུཿ། དིབ་སོགས་ལ་ནི་ཨཱཏྨ་ནེའི་ཚིག་གོ། །སི་ཙ་ཨཱ་ཤཱི་ཥ་དག་ལ་ཡང་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་ལ། ཨ་སྲྀཥྚ། ཨ་སྲྀ་ཀྵཱ་ཏཱཾ། སྤྲྀ་ཤ་ཀུན་ནས་

དང་འཕེལ་བའི་གནས་ལ་དབྱངས་ལས་ཕ་རོལ་དུ་ཡང་ན་ཨ་ཡིག་ཨཱ་ག་མའོ། །ཨ་སྤྲཱཀྵཱིཏ། གསལ་བྱེད་མཐའ་རྣམས་ཀྱི་ཨིཊ་མེད་པ་རྣམས་ཀྱིའོ་ཞེས་པས་འཕེལ་བའོ། །ཨསྤཱརྐྵཱིཏ། ཕྱོགས་གཅིག་ལ་ཤིཊ་མཐའ་ཉིད་ལས་སཎའོ། །ཨ་སྤྲྀཀྵཏ། ཨ་སྤྲྀཀྵ་ཏཱཾ། ཨ་མྲཀྵཱིཏ། ཨ་མཱརྐཱིཏ།
ཨ་མྲཱཥྚཱཾ། ཨ་མྲྀཀྵཏ། ཨ་མྲྀཀྵ་ཏཱཾ། ཨཀྲཱ་ཀྵིཏ། ཨ་ཀཱརྐྵཱིཏ། ཨ་ཀྲཥྚཱཾ།ཨ་ཀཱརྐྵཱཾ། ཨ་ཀྲྀཀྵ་ཏ། ཨ་ཀྲྀཀྵ་ཏཱཾ། ཏུད་སོགས་ཅན་ལས་གཉི་གའི་ཚིག་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ་ཞེས་པས་སི་ཙ་ཨཱ་ཤཱིཥ་དག་ལ་ཡང་ཨཱཏྨ་ནེ་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །ཨ་ཀྲྀཥྚ། ཨ་ཀྲྀཀྵཱ་ཏཱཾ། ཨ་ཀྲཱྀཀྵཏ། ཨཱ་ཀྲཱྀཀྵཱ་
