གྱི་དྷཱེ། དི་གྱི་ཌྷྭེ། ཏྲེེང་། ཏ་ཏྲེ། ཏ་ཏྲི་ཌྷྭེ། ཏ་ཏྲི་དྷེ། བཻྱང་། པ་པྱེ། པ་པྱི་དྷེ། པ་པྱི་ཌྷྭེ། ཌུ་དཱཉ། ད་དཽ། ད་དཏུཿ། ད་དེ། ཌུ་དྷཱ་ཉ། ད་དྷོ། ད་དྷ་ཏུཿ། ད་དྷེ། ཙཀྵཻང་། པ་རོ་ཀྵ་ལ་ཡང་ནའོ། །ཞེས་པས་པ་རོཀྵའི་ཡུལ་ལ་ཡང་ན་ཁྱཱཉ་རུ་བསྒྱུར་རོ། །ཉ་རྗེས་སུ་
འབྲེལ་པ་ཉིད་ལས་གཉི་གའི་ཚིག་གོ། །ཨཱ་ཙ་ཁྱོོ། ཨཱ་ཙ་ཁྱ་ཏུཿ། ཨཱ་ཙ་ཁྱེ། ཨཱ་ཙ་ཁྱི་དྷྭེ། ཨཱ་ཙ་ཁྱི་ཌྷྭེ། ཨཱ་ཙ་ཙཀྵེ། ཨཱ་ཙ་ཙཀྵཱ་ཏེ། གསལ་བྱེད་དང་པོ་ཞེས་པས་ཨིཊ་ཨཱ་ག་མའོ། །ཨ་ཙ་ཙ་ཀྵི་རེ། ཛི་རྒྱལ་བ་ལའོ། །ཛིའི་སན་པ་རོཀྵཱ་དག་ལ་གིའོ་ཞེས་པས་གི་རུ་
བསྒྱུར་རོ། །གིའི་གཉིས་བརྗོད་དོ། །ཨའི་ཨུ་པ་དྷཱ་ཞེས་པ་ལ་སོགས་པས་བྱིངས་ཀྱི་འཕེལ་བའོ། །ཨཻ་ཨཱ་ཡའོ། །ཛི་གཱ་ཡ། ཨི་ཡིག་གི་ཡ་སྟེ་ཞེས་པ་ལ་སོགས་པས་དབྱངས་དང་པོ་ཡོན་ཏན་མེད་པ་ལ་ཡ་ཉིད་དོ། །ཛི་གྱ་ཏུཿ། ཛི་གྱུཿ། རྟག་ཏུ་ཨཱ་ཉིད་རྣམས་ལའོ། །དབྱངས་
མཐའ་རྣམས་ཀྱི་དང་པྲྀ་ཛི་དྲྀ་ཤ་དག་གི་ཡང་ཐལ་ལ་ཨིཊ་ཡང་ནའོ། །ཏྲི་ཙ་ལ་རྟག་ཏུ་ཨིཊ་མེད་པར་འགྱུར་ཡང་ཛ་ཥོ་དབྲི་བྱེ་ཨད་རྣམས་ཀྱི་ཐལ་ལ་རྟག་ཏུ་ཨིཊའོ། །ཞེས་བརྗོད་པ་ལས་ཡང་ན་ཨིཊའོ། །ཛི་ག་ཡི་ཐ། བི་པ་རཱ་དག་ལས་ཛི། ཞེས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །
བི་ཛི་གྱཱ་ཏེ། བི་ཛི་གྱི་རེ། བི་ཛ་གྱཱ་ཐེ། སྔ་མ་བཞིན་ཡང་ན་དྷ་ཡིག་གི་ཌྷ་ཡིག་གོ། །བི་ཛི་གྱི་དྷྭེ། བི་ཛི་གྱི་ཌྷྭེ། བི་ཛི་གྱེ། བི་ཛི་གྱི་བ་ཧེ། བི་ཛི་གྱི་མ་ཧེ། བ་རཱ་ཛི་གྱེ། རི་པི་འགྲོ་བ་ལའོ། །རི་རཱ་ཡ། རིང་པོ་བསྒྲུབ་པ་ལ་ཡང་ཡ་ཉིད་དོ། །དབྱངས་བསྒྱུར་བ་གནས་ཅན་བཞིན་དུ་འགྱུར་བའི་ཕྱིར་
ཚིག་མཐའ་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཡོངས་སུ་བཤད་པའི་ངེས་པ་མེད་པར་ཁས་བླངས་སོ། །དེ་ལས་ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །རིཪྻ་ཏུཿ། པི་པཱ་ཡ། པི་པྱ་ཏུཿ། ཧི་འགྲོ་བ་ལའོ། །འཕེལ་བ་ཡང་ངོ་། །ཧི་ལས་ཙཎ་མ་ཡིན་པ་ལའོ། །ཞེས་བརྗོད་པར་བྱའོ། །
ཞེས་པས་ཧའི་གྷ་ཉིད་དོ། །ཛི་གྷཱ་ཡ། ཛི་གྷྱ་ཏུཿ། ཙིཉ། ཙིའི་ཀི་ཡང་ནའོ། །ཞེས་པས་ཀི་རུ་འགྱུར་རོ། །ཙི་ཀཱ་ཡ། ཙི་ཙཱ་ཡ། ཙིཀྱ་ཏུཿ། ཙི་ཙྱ་ཏུཿ། ཙི་ཀྱེ། ཙི་ཙྱེ། ཙི་ཀྱི་དྷྭེ། ཙི་ཀྱི་ཌྷྭེ་ཥིཉ། སི་ཥཱ་ཡ། སི་ཥྱ་ཏུཿ། ས་ཥྱེ། ཀྵི། བྱིངས་ཀྱི་ཨིའི་ཡི་གེ་དང་ཨུའི་ཡི་གེ་
