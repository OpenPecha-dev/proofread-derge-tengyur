ཎཱ་ཙ་ཀ་ར། ཨོ་ཎཱ་མཱ་སཿ། ཨོ་ཎཱམྦ་བྷཱུ་བ། ཨཱི་ཧ། ཨཱི་ཧཱཾ་ཙ་ཀྲེ། ཨཱི་ཧཱ་མཱ་ས། ཨཱི་ཧཱ་མྦ་བྷཱུ་བ། ངེས་པ་མེད་པ་ལ་ཡང་ཨཱ་ག་ཏི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་ནང་མར་འདོད་པའོ། །བྱ་ཏི་བྱས་པ་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་རྣམས་ལ་ནི་བྷཱུ་
ཨཱས་རབ་ཏུ་སྦྱོར་བ་ཕ་རོལ་ཏུ་ཉེ་བར་སྦྱིན་ཞིང་སྟོན་པར་འགྱུར་རོ། །ཨཱི་ཧཱཾ་བྱ་ཏྱཱ་སེ། ཨཱི་ཧཱཾ་བྱ་ཏྱཱི་བ་བྷཱུ་བེ། ཨཱི་ཀྵ། ཨཱི་ཀྵཱཾ་ཙ་ཀྲེ། ཨཱི་ཀྵཱ་མཱ་སེ། ཨཱི་ཀྵཱཾ་བ་བྷཱུ་བེ། ཨཱི་ཤ །ཨཱི་ཏ། ཨི་རཱ། ཨི་ཛ་འགྲོ་བ་ལའོ། །ཞེས་པ་འདི་རྣམས་ཀྱིའང་དེ་བཞིན་དུ། །ཨཱུ་ཏ། ཨཱུ་ཧཱཾ་ཙ་ཀྲེ། ཨཱུ་ཧཱ་མ་སེ། ཨཱུ་ཧཱཾ་
བ་བྷཱུ་བེ། ཨེ་དྷ། ཨེ་དྷྭཾ་ཙ་ཀྲེ། ཨེ་དྷཱཾ་མཱ་སེ། ཨེ་དྷཱཾ་བ་བྷཱུ་བེ། རྀ་ཛ་བྱི་དོར་ལའོ། །རྀ་ཛཱཉྩ་ཀྲེ། རྀ་ཛཱ་མཱ་ས། རྀ་ཛཾ་བ་བྷཱུ་བེ། ཨིང་འགྲོ་བ་ལའོ། །ཁྱ་ཡ་སྟོན་པ་བཞིན་འགྱུར་བ་ཡིས། ནཱ་མི་ལ་སོགས་པའོ། །ཨཱ་ཡཱཾ་ཙ་ཀྲེ། ཨཱ་ཡཱ་མཱ་ས། ཨཱ་ཡཱཾ་བ་བྷཱུ་བ། རི། ཨཱ་རཱཾ་ཙ་ཀཱ་ར། 
ཨཱ་རཱ་མཱ་ས། ཨཱ་རཱ་མྦ་བྷཱུ་བ། གཞན་དུ་བྱ་ཡ་སྟོན་པ་བཞིན་འགྱུར་བ་མ་ཡིན་ཞིང་ཨ་ཡཱནྟི་མ་ཡིན་ནོ། །ཨི་ཡེ། ཨཱ་ར། ཨཱ་ར་ཏུཿ། ཨུ་ཥ་བི་ད་ཛཱ་གྲྀ་རྣམས་ལས་ཡང་ནའོ། །ཞེས་པའོ། །ཨཱུ་ཥ་བསྲེགས་པ་ལའོ། །བིད་ཤེས་པ་ལའོ། །ཛ་གྲྀ། འདི་རྣམས་ལས་པ་རོཀྵཱ་ཕ་རོལ་ལ་ཨཱ་མ་
ཕ་རོལ་ཏུ་ཡང་འགྱུར་རོ། །ཀྲྀ་ཉ་ལ་སོགས་པ་རྗེས་སུ་སྦྱོར་བའོ། །ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨོ་ཥཱཾ་ཙ་ཀཱ་ར། ཨོ་ཥཱ་མཱ་ས། ཨོ་ཥཱཾ་བ་བྷཱུ་བ། ཐལ་ལ་ཨིཊའོ། །ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨ་བྷྱཱ་སའི་འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཨུ་བ་ཏུ་བསྒྱུར་རོ། །ཨུ་བོ་ཥ། མཚུངས་
པའི་རིང་པོ་ལ་སོགས་པའོ། །ཨཱུ་ཥ་ཏུཿ། ཨཱུ་ཥུཿ། གསལ་བྱེད་དང་པོ་ལ་ཨིཊ་ཨཱ་ག་མའོ། །ཨུ་བོ་ཥི་ཐ། ཨཱ་མ་ལའང་བིད་ཉིད་ཀྱིའོ། །ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་དོ། །བི་དཱཾ་ཙ་ཀཱ་ར། བི་དྷ་མ་ས། བི་དཱཾ་བ་བྷཱུ་བ། བི་བེ་ད། བི་བིད་ཏུཿ། བི་བེ་དུཿ། བི་བེ་དི་ཐ། ནཱ་མི་མཐའི་ཡོན་ཏན་ནོ། །ཛཱ་
ག་རཱཾ་ཙ་ཀཱ་ར། ཛཱ་ག་རཱ་མཱ་ས། ཛཱ་ག་རཱཾ་བ་བྷཱུ་བ། ཛ་ཞེས་པ་འདི་གཅིག་པུའི་གཉིས་བརྗོད་དོ། །ཨའི་ཨུ་པ་དྷཱ་ཞེས་པ་ལ་སོགས་པས་ཨཊ་ལ་འཕེལ་བའོ། །ཕྱི་མ་ལ་ཡང་ན་སྔ་མ་བཞིན་ནོ། །ཛ་ཛཱ་གཱ་ར། ཛཱ་གརྟིའི་ཀཱ་རི་ཏ་ལའོ་ཞེས་སྤྱིར་བཏང་ནས། པ་རོཀྵཱ་ལ་ཡོན་ཏན་མེད་པ་ལའོ་ཞེས་པས་ཡོན་
