ཡི་གེ་དག་མཐའི་དབྱངས་དང་པོ་ལ་ཞེས་པ་ལ་སོགས་པས་ཨི་ཡ་རུ་བསྒྱུར་རོ། །བི་ཡནྟི། བེ་ཥི། བཱི་ཐཿ བཱི་ཐ། བེ་མི། བཱི་བཿ། བཱི་མཿ། བྷཱ་འབར་བ་ལའོ། །བྷཱ་ཏི། བྷཱ་ཏ། བྷཱནྟི། བྷཱ་སེ། བྷཱ་ཐཿ བྷཱ་ཐ། བྷཱ་མི། བྷཱ་བཿ བྷཱ་མཿ ཨཱ་བྷཱ་ཏི། པྲ་བྷཱ་ཏི། པྲ་ཏི་བྷཱ་ཏི། ངེས་པ་
མེད་པ་ལའང་ཨ་ག་ཏི་ཞེས་པ་ལ་སོགས་པས་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །བྱ་ཏི་བྷཱ་ཏེ། བྱ་ཏི་བྷཱ་ཏེ། བྱ་ཏི་བྷཱ་ཏེ། བྱ་ཏི་བྷཱ་སེ། བྱ་ཏི་བྷཱ་ཐེ། བྱ་ཏི་བྷཱ་དྷྭེ། མཚམས་སྦྱོར་གྱི་ཡི་གེ་ལ་ཡང་ངོ་ཞེས་པས་ཨཱ་ཡིག་དབྱིའོ། །བྱ་ཏི་བྷེ། བྱ་ཏི་བྷཱ་བ་ཏེ། བྱ་ཏི་བྷཱ་མ་
ཧེ། ཡཱ་ཐོབ་པ་ལའོ། །ཡཱ་ཏི། ཡཱ་ཏཿ ཡཱནྟི། པྲ་ཡཱ་ཏི། ཨུང་ཡཱ་ཏི། པྲ་ཎི་ཡཱ་ཏི། བཱ་འགྲོ་བ་དང་དྲི་དག་ལའོ། །བཱ་ཏི། པྲ་བཱ་ཏི། པྲ་ཎི་བཱ་ཏི། ཥྞཱ་གཙང་བ་ལའོ། །སྣཱ་ཏི། ཤྲཱ་འཚེད་པ་ལའོ། །ཤྲཱ་ཏི། དྲཱ་སྨཱད་པར་འགྲོ་བ་ལའོ། །དྲཱ་ཏི། ནི་དྲཱ་ཏི། པྲ་ཎི་དྲཱ་ཏི། པཱ་བསྲུང་བ་
ལའོ། །པཱ་ཏི། པཱ་ཏཿ པཱནྟི། རཱ་ལཱ་ལེན་པ་ལའོ། །རཱ་ཏི། ལཱ་ཏི། དཱ་པ་གཅོད་པ་ལའོ། །པཱ་ཡིག་ནི། དཱ་པ་མིན་པའི་དཱ་དྷཱ་དག་དཱའོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ནོ། །དཱ་ཏི། ཁྱཱ་རབ་ཏུ་གཏམ་པ་ལའོ། །ཁྱཱ་ཏི། པྲ་ཁྱཱ་ཏི། བྱཱ་ཁྱཱ་ཏི། པྲཱ་རྫོགས་པ་ལའོ། །པྲཱ་ཏི། མཱང་ཚང་པ་ལའོ། །

ལྔ་རྣམས་ཀྱི་དང་དགུ་རྣམས་ཏེ། བི་དེའི་ཡང་དེ་བཞིན་ནོ། །ཨཊ་ལ་སོགས་པའི་ཚིག་ཕྲད་རྣམས་སུ་བྱ་བ་ལ་ཏི་ལ་སོགས་པ་རྣམས་ཀྱི་ཡང་ན་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་བིད་ལས་ཕ་རོལ་གྱི་ཏཾ་ལ་སོགས་པ་དགུ་པོ་རྣམས། ཨཊ་ལ་སོགས་པ་རྣམས་སུ་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་ཚིག་
