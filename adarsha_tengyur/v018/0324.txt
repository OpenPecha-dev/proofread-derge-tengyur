བྱང་ཕྱོགས་ཀྱི་གཏུམ་མིག་མར་སྣང་ངོ་། །སྤྱོད་པའི་ཤེས་རབ་ཀྱིས་གྲོགས་བྱས་པའི་རྫུ་འཕྲུལ་གྱི་སྟོང་ཉིད་ནུབ་ཕྱོགས་ཀྱི་འོད་ལྡན་མར་སྣང་ངོ་། །སེམས་ཀྱིས་གྲོགས་བྱས་པའི་རྫུ་འཕྲུལ་གྱི་སྟོང་ཉིད་ལྷོ་ཕྱོགས་ཀྱི་སྣ་ཆེན་མར་སྣང་ངོ་། །གནས་ཀྱི་དང་པོ་ལྷ་མོ་རྣམ་དག་གོ། །དད་པས་མ་དད་པ་སྤངས་
ནས་རང་གི་རིགས་འདྲའི་དབང་པོའི་ཆོས་ཉིད་ནི་ཤར་ལྷོའི་བརྟན་པའི་བློ་གྲོས་མར་སྣང་ངོ་། །བརྩོན་འགྲུས་ཀྱི་དབང་པོའི་སྟོང་པ་ཉིད་ལྷོ་ནུབ་ཀྱི་མི་ཐུབ་མར་སྣང་ངོ་། །དྲན་པའི་དབང་པོའི་ཆོས་ཉིད་ནུབ་བྱང་གི་ལངྐའི་དབང་ཕྱུག་མར་སྣང་ངོ་། །ཏིང་ངེ་འཛིན་གྱི་དབང་པོའི་ཆོས་ཉིད་ཤིང་གྲིབ་མར་སྣང་
སྟེ་སྒྱུ་མ་ལྟ་བུ་བརྒྱད་ནི་དཔའ་བོར་སྣང་ངོ་། །ལུས་ཀྱི་གནས་བརྒྱད་ཀྱི་ལྷའི་རྣམ་པར་དག་པར་སྟོན་ཏོ། །གསུང་གི་འཁོར་ལོའི་ཤེས་རབ་ཀྱི་དབང་པོའི་སྟོང་ཉིད་ཤར་ཕྱོགས་ཀྱི་སྲུང་མར་སྣང་ངོ་། །དད་པའི་སྟོབས་ཀྱི་སྟོང་ཉིད་བྱང་ཕྱོགས་ཀྱི་འཇིགས་བྱེད་ཆེན་མོར་སྣང་ངོ་། །བརྩོན་འགྲུས་ཀྱི་
སྟོབས་ཀྱི་སྟོང་ཉིད་ནུབ་ཕྱོགས་ཀྱི་རླུང་གི་ཤུགས་ཅན་མར་སྣང་ངོ་། །དྲན་པའི་སྟོབས་ཀྱི་སྟོང་ཉིད་ནི་ལྷོ་ཕྱོགས་ཀྱི་ཆང་འཐུང་མར་སྣང་ངོ་། །ཏིང་ངེ་འཛིན་གྱི་སྟོབས་ཀྱི་ཆོས་ཉིད་ཤར་ལྷོའི་སྔོ་བསངས་མར་སྣང་ངོ་། །ཤེས་རབ་ཀྱི་སྟོབས་ཀྱི་ཆོས་ཉིད་ལྷོ་ནུབ་ཀྱི་ཤིན་ཏུ་བཟང་མོར་སྣང་ངོ་། །ཏིང་ངེ་
འཛིན་ཡང་དག་པར་འདུས་མ་བྱས་པ་བྱང་ཆུབ་ས་དང་པོའི་ཡན་ལག་གྲོགས་ནུབ་བྱང་གི་རྟ་རྣ་མར་སྣང་ངོ་། །བརྩོན་འགྲུས་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ནི་བྱང་ཤར་གྱི་བྱ་གདོང་མར་སྣང་ངོ་། །སྒྱུ་མ་ལྟ་བུ་བརྒྱད་ནི་དཔའ་བོར་སྣང་ངོ་། །གསུང་གི་འཁོར་ལོ་ལུས་ལ་དགོད་པའི་རྣམ་པར་དག་
པའོ། །སྐུའི་འཁོར་ལོའི་དགའ་བ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ནི་ཤར་གྱི་འཁོར་ལོའི་ཤུགས་ཅན་མར་སྣང་ངོ་། །ཤིན་ཏུ་སྦྱངས་པ་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ནི་བྱང་གི་དུམ་སྐྱེས་མར་སྣང་ངོ་། །ཆོས་ཡང་དག་པར་འབྱེད་པ་ནི་ནུབ་ཀྱི་ཀོ་ས་རིངས་མར་སྣང་ངོ་། །དྲན་པ་ཡང་དག་
བྱང་ཆུབ་ཀྱི་ཡན་ལག་ནི་ལྷོའི་འཁོར་ལོའི་གོ་ཆ་མར་སྣང་ངོ་། །བཏང་སྙོམས་ཡང་དག་བྱང་ཆུབ་ཀྱི་ཡན་ལག་ནི། ཤར་ལྷོའི་ཤིན་ཏུ་དཔའ་མོར་སྣང་ངོ་། །ཡང་དག་པའི་སྟོང་ཉིད་ཀྱི་ལྟ་བ་ནི་ལྷོ་ནུབ་ཀྱི་སྟོབས་ཆེན་མར་སྣང་ངོ་། །ཡང་དག་པའི་རྟོག་པ་ནི་ནུབ་བྱང་གི་འཁོར་ལོས་སྒྱུར་
