མཚོ་ལས་སྐྱེས་པའོ། །ཥྛ་ནི་ནམ་མཁའི་སྙིང་པོའོ། །ཊ་ནི་ག་བུར་བསྟན་པའོ། །ཞེས་གསུངས་པས་ཁམས་བཅུ་དྲུག་སྟོང་པ་ཉིད་རྗེས་སུ་དྲན་པ་བཅུད་ལེན་དུ་འགྱུར་བ་བསྟན་པ་སྟེ། ར་ནི་རང་འབྱུང་གི་སིཧླ་ལོ། །ཧྱ་ནི་སྐྲའི་མར་ཁུའོ། །སྱ་ནི་གསང་བའི་སྦུབས་ཤའོ། །ས་ནི་མི་ཡི་
སྔགས་པའོ། །ར་ནི་གཞི་མཉམ་མའོ། །མེ་མེ་ནི་སིཧླ་ཕལ་བའོ། །ར་ནི་གླ་རྩིའོ། །མེ་ནི་ཤ་ཆེན་པོའོ། །ས་ནི་ཆིལ་ཆེན་པོའོ། །རྦཱཏ་ནི་གོ་རོ་ཙ་ནའོ། །མ་ནི་རྐང་མར་རོ། །ནི་ནི་གཞི་མཉམ་མར་ཁུའོ། །ས་ནི་སོ་དང་སེན་མོའོ། །ད་ནི་གློ་བའོ། །ཥྛི་ནི་མཆིལ་མའོ། །ཊ་ཏ་ནི་བྱང་ཆུབ་
སེམས་ཀྱི་ཐིག་ལེ་ཉིད་དོ། །ཊ་ཧ་ཏ་ལ་གང་ལས་བསྐྱོད་པའི་བྱང་ཆུབ་སེམས་བདེ་བ་ཆེན་པོའི་ཧཾ་ལས་འབབ་པར་འགྱུར་བའོ། །དེ་ཉིད་ཊ་པ་ཡ་ཏུ་ཏ་ས་མ་ཞེས་བྱང་སེམས་བདེ་བ་དབུ་མའི་ལམ་ནས་འབྱུང་འཇུག་བྱེད་པའོ། །ཥྛི་ཐ་ཀ་རིར་ནི་ཞེས་པས་བོར་བ་སྟེ་སྤངས་པའི་དོན། ཥྛི་ཐ་ཐ་
ཀ་ནི་རི་ནི་གནས་གང་དུ་བྱང་སེམས་བརྟན་པར་གནས་པས་མི་འགྱུར་བའི་བདེ་བ་རྡོ་རྗེ་ནོར་བུ་བརྟན་པར་གྱིས་ཤིག་པའོ། །དྷ་ནི་དྷཱ་ར་སྟེ་གཟུངས་བྱ་བ་ལུས་ཀྱི་ཁམས་སྐྱེད་པར་བྱེད་པས་བདེ་བའི་གནས་ཡིན་ནོ། །ད་ཧ་ད་ཧ་ཞེས་པ་ནི་སྲེག་པར་བྱེད་པ་སྟེ་ཙཎྜ་ལཱི་ཡེ་ཤེས་ཀྱི་མེ་འབར་བས་ཉོན་
མོངས་པ་བསྲེགས་པའོ། །ས་མ་ས་མ་ཡ་ནི་དམ་ཚིག་སྟེ། སྒོ་དགུ་ནས་བྱུང་བའི་བདུད་རྩི་སྟེ་ན་ཅིག་པ་ལས་ཁྱད་པར་དུ་ས་དང་སེན་མོའི་དྲི་མ་སྤང་དུ་མེད་པས། ས་ས་མཱ་པཱ་དྷི་ཞེས་པའི་ཏིང་ངེ་འཛིན་གང་སྲོག་རླུང་དབུ་མར་ཚུད་པས་གནས་པར་བྱ་བའི་དོན་ཏོ། །ནི་ནི་མ་ལ་སྣ་ཚོགས་ཅན་
གྱི་སྲི་བ་སྟེ་ལུས་ལ་ཟས་ཁམས་སྣ་ཚོགས་སུ་མར་ཁུ་ལྟར་འཕེལ་བ་སྟེ་ཟག་པ་བཞི་མཉམ་མོ། །ནི་ན་ནི་བྷ་ཡཱ་ཞེས་ཏེ་ལས་ཁམས་ཀྱི་བཅུད་སྣ་ཚོགས་ཀྱིས་གཉུག་མའི་ཤེས་པ་རོ་མཉམ་ཞེས་པའོ། །མ་མ་ཧཱ་བྷ་ན་ཆེན་པོའི་ཆེན་པོ་སྟེ་ཤ་ཆེན་པོའི་རྐང་མར་གྱིས་སྟོབས་བསྐྱེད་པའོ། །མ་
མ་བྷ་དྷུ་བྷ་ར་ན་ནཾ། ཁམས་དངས་མའི་ཆ་ལས་གྲུབ་པའི་གཟུངས་དག་པའི་རིགས་པ་སྐྱེད་ཅེས་པའོ། །རྦཱད་ནི་ར་ཀཙྪ་ཅེས་གླད་རྒྱས་ནས་ཁོག་པའི་བཅུད་གནས་པ་དྲང་བའོ། །རྦཱད་ད་ཉ་ར་ཀཙྪ་ཞེས་ཟླ་བ་རྒྱས་པ་ལྟ་བུའི་བྱང་སེམས་འཕེལ་བའི་གནས་ནས་བསྐྱོད་པས་ཡེ་ཤེས་བསྐྱེད་པའི་
