བ་ལ། མེ་དྲོད་ཀྱི་སྟོབས་སྐྱེས་ཤིང་རྙེད་ནས་ནད་པ་དེ་ལ་སྣུམ་ཆོས་བྱ་ཞིང་འོ་མ་དང་། དུར་བྱིད་སྦྱར་བའི་ལྡེ་གུའམ། ཨེ་རཎྜ་དང་འོ་མར་བསྐོལ་བའམ། པ་ཏོ་ལ་དང་ཀྱི་ལྕེ་བ་དང་འོ་མ་བསྐོལ་བའམ། དོང་ག་དང་འོ་མ་བསྐོལ་བ་དག་གིས་ཡང་དང་ཡང་དུ་བཀྲུ་སྨན་
བཏང་ནས་སྦྱང་བར་བྱ་ལ། བད་ཀན་དང་མཁྲིས་པར་བཅས་པ་ལ་ནི་བ་གཅིན་དང་བཅས་པའི་འོ་མས་བཀྲུ་སྨན་གཏོང་ངོ་། །རླུང་གིས་དམུ་རྫིང་དེ་རླུང་བཅས་པ་ཡིན་ན་ཏིག་ཏ་དང་མར་དང་བཅས་པའོ་མས་ཡང་དང་ཡང་དུ་བཀྲུ་སྨན་གཏང་བར་བྱའོ། །འདི་དག་འོ་མར་བཅས་བཟའ་
བའམ། །བི་ད་རི་སོགས་སྡེ་བསྐོལ་སྦྱིན། །དམུ་རྫིང་ཅན་ལ་འོ་ཐུག་ནི། །སྦྱིན་ཞིང་ཉེ་བར་བསྐུ་བར་བྱ། །ཞེས་བྱ་བ་ལ། འོ་མ་སྨན་དང་བཅས་པའི་སྦྱོར་བ་དག་གང་ཡང་རུང་བ་དང་བཅས་པ་ཁ་ཟས་སྦྱིན་པར་བྱ། ཡང་ན་བི་ད་རི་ལ་སོགས་པའི་སྡེ་ཚན་བསྐོལ་བའི་འོ་མ་དང་
ཁ་ཟས་སྦྱིན་པར་བྱའོ། །ཡང་ན་དམུ་རྫིང་ཅན་དེ་ལ་འོ་ཐུད་སྦྱིན་ཞིང་འོ་ཐུག་དིས་ལྟོ་བ་བསྐུ་བར་བྱའོ། །སླར་ཡང་འོ་མ་འཇམ་རྩི་དང་། །བཀྲུ་སྨན་ཡང་དང་ཡང་དུ་གཏང་། །རིམ་པས་ངེས་པར་གནས་བྱས་ཏེ། །མ་ཡེངས་མཁྲིས་གྱུར་དམུ་རྫིང་བསལ། །ཞེས་བྱ་བ་ལ། འོ་མ་
དང་འཇམ་རྩི་ལ་སོགས་པའི་སྦྱོར་བ་ཡང་དང་ཡང་དུ་བྱ་བ་གནས་པའི་རིམ་པས་གཞན་དུ་མ་ཡེངས་ཤིང་དེ་ལ་གཏོད་པར་གྱུར་པ་དང་། མཁྲིས་པ་ལས་གྱུར་པའི་དམུ་རྫིང་གསོ་ཞིང་བཅོས་པར་བྱའོ། །བད་ཀན་ལས་གྱུར་སྣུམ་ཆོས་བྱ། །དུག་མོ་ཉུང་སོགས་མར་བསྐོལ་གཏང་། །ཉམས་
སྟོབས་ལྡན་པས་དུགས་བྱས་ཏེ། །སུ་དྷཱ་ཀྵི་རིའི་བཀྲུ་སྨན་བཏང་། །ཐུག་སོགས་རིམ་དང་ཐལ་སྨན་ཚ། །ལྡན་པས་བད་ཀན་སེལ་བ་སྦྱིན། །ཞེས་བྱ་བ་ལ། བད་ཀན་ལས་གྱུར་པའི་དམུ་རྫིང་གི་ནད་ལ་ནི་དུག་མོ་ཡུངས་ལ་སོགས་པ་དང་བསྐོལ་བའི་མར་བླུད་ཅིང་སྣུམ་ཆོས་བྱས་ལ། དེ་
ཉམས་སྟོབས་དང་ལྡན་པ་ཡིན་ན་དུགས་བྱས་པའི་འོག་ཏུ་སུ་དྷཱ་ཀྵི་ར་དང་སྦྱར་བའི་བཀྲུ་སྨན་བཏང་ལ། བད་ཀན་སེལ་བའི་ཁ་ཟས་ཁ་བ་དང་ཚ་བ་དང་ཐལ་སྨན་དང་ལྡན་པ་བླུད་པར་བྱའོ། །བ་གཅིན་ཚ་གསུམ་ཏིལ་མར་བཅས། །མས་གཏོང་དྲག་པོ་བྱ་བའང་ཕན། །མུངྒ་ཀ་སོགས་ལྡེ་
