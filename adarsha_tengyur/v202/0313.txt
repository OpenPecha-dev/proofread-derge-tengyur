དུ་བསྐྲ་ཞིང་བྱུག་པར་བྱའོ། །བ་ལང་ལྕི་ཁུ་འོ་མ་མར། །བསྐོལ་བ་ཡི་ནི་མིག་སྨན་ནམ། །བལ་བུའི་ལོ་མ་ཡུག་སྣམ་བསྐོལ། །ཕྱེ་མ་བཏབ་པའི་མིག་སྨན་ཕན། །ཞེས་བྱ་བ་ལ། བའི་ལྕི་བ་བཙིར་བའི་ཁུ་བ་དང་འོ་མ་དང་མར་རམ། ཡང་ན་
བལ་བུའི་ལོ་མ་དང་། བཙག་ཡུག་སྣམ་གྱི་ཕྱེ་མ་བསྐོལ་བའི་མིག་སྨན་ཕན་པ་ཡིན་ནོ། །མེ་ད་སེང་ཕྲོམ་ཨ་ནནྟ། །སྐྱེར་པ་བཙོད་དང་ཤིང་མངར་ནི། །མར་གྱི་བརྒྱད་འགྱུར་འོ་མ་བསྐོལ། །ཏིལ་མར་བསྐོལ་བའི་སྣ་སྨན་ཕན། །ཞེས་བྱ་བ་ལ། མེ་ད་ལ་སོགས་པའི་
སྨན་མར་གྱི་བརྒྱད་འགྱུར་གྱི་འོ་མ་དང་། ཏིལ་མར་བསྐོལ་བའི་སྣ་སྨན་བཏང་ན་ཡང་ཕན་ནོ། །འོ་མ་མར་སར་གྱིས་བརྟ་བྱ། །དེས་མ་ཞི་ན་གཏར་ག་གདབ། །ཅེས་བྱ་བ་ལ། འོ་མ་ལས་བྱུང་བའི་མར་སར་ཡིན་པས་ནའོ་མའི་མར་སར་ཏེ། དེས་བརྟ་བར་བྱ་ལ། དེ་ལྟར་བྱས་སུ་ཟིན་ཀྱང་ཞི་
བར་མ་གྱུར་ན། གཏར་ག་གདབ་ཅིང་ཁྲག་དབྱུང་བར་བྱའོ། །སེམས་ལས་ནད་དང་མྱ་ངན་འཇིགས། །ཟས་རྩུབ་ཙོག་ཙོག་པོར་འདུག་དང་། །སྐྱུག་སྨན་བཀྲུ་སྨན་སྣ་སྨན་དང་། །སྨན་ཐིགས་བླུགས་སོགས་ཉེས་པ་དང་། །ཁ་ཟས་ཕྱེད་ཞུ་སློན་པ་དང་། །བཀྲེས་སྐོམ་ལ་
སོགས་བརྣག་པ་དང་། །མིག་ནད་མ་བཅོས་ཡུན་རིང་ལས། །རབ་རིབ་ནད་ལྡན་བརྟ་བར་བྱ། །ཞེས་བྱ་བ་ལ། སེམས་ལས་ལ་སོགས་པ་བསྟེན་པ་དང་། སྐྱུག་སྨན་ལ་སོགས་པ་ཉེས་པ་དང་། དེ་བཞིན་དུ་ཁ་ཟས་ཕྱེད་ཙམ་མ་ཞུ་བ་ལས་སློན་པ་དང་། བཀྲེས་པ་ལ་
སོགས་པ་བརྣག་པ་ལས་མིག་རབ་རིབ་ཀྱི་ནད་དང་ལྡན་པར་བརྟ་བར་བྱའོ། །དེ་ལ་ནད་གཞི་སོགས་བརྟགས་ཏེ། །རང་གི་སྨན་ནི་ཇི་བཞིན་སྦྱར། །ཞེས་བྱ་བ་ལ། སེམས་ལས་ལ་སོགས་པའི་ནད་ཀྱི་རྒྱུ་དེ་དག་ལས་རབ་རིབ་དང་འདྲ་བར་ཉེས་པར་བརྟགས་ལ་ནད་གཞི་ཇི་
ལྟ་བ་ལས། རང་གི་སྨན་གང་དང་གང་ཡིན་པ་ལས་དེ་སྦྱར་ཞིང་གཏང་བར་བྱའོ། །ཉི་མ་གཟས་འཛིན་མེ་འོད་གློག་ལ་སོགས། །བལྟས་པས་མིག་ནི་ཉམས་པར་གྱུར་པ་དག །དེ་ལ་བསིལ་སྨན་ཚོགས་ཀྱིས་བརྟ་བྱ་ཞིང་། །དེ་ལྟར་གསེར་བདར་མར་གྱི་མིག་སྨན་གཏང་། །
