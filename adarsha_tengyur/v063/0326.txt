མ་གྱུར་པ། མཁས་པས་བསྔགས་པ་བསྲུང་ངོ་། །དེ་ཚུལ་ཁྲིམས་དེ་ལ་རབ་ཏུ་གནས་པས་སེམས་སྨོན་པ་འགྲུབ་པར་འགྱུར་ཏེ། འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་སྨོན་ལམ་བསྒྲུབ་པ་ཞེས་བྱའོ། །དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་སྟོབས་བསྒྲུབ་པ་གང་ཞེ་ན། གལ་ཏེ་ལ་ལ་ཞིག་སངས་རྒྱས་ཀྱི་གཟུགས་
སུ་བསྒྱུར་ནས་བྱང་ཆུབ་སེམས་དཔའི་གན་དུ་འོང་སྟེ། སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ལས་ཟློག་པར་བྱེད། འདུན་པ་ཟློག་པར་བྱེད། དབྲལ་བར་བྱེད་ཀྱང་བྱང་ཆུབ་སེམས་དཔའ་དེ་འདུན་པ་ཟློག་པར་མི་འགྱུར་རོ། །བྱང་ཆུབ་ཉིད་རྗེས་སུ་དྲན་ནོ། །བྱང་ཆུབ་ཀྱི་སྙིང་པོ་རྗེས་སུ་དྲན་ནོ། །བདུད་འདུལ་བ་དང་། 
ཆོས་ཀྱི་འཁོར་ལོ་རབ་ཏུ་བསྐོར་བ་དང་། སྙིང་རྗེ་ཆེན་པོ་མངོན་དུ་བྱེད་དོ། །སེམས་ཅན་གང་བྱང་ཆུབ་སེམས་དཔའ་དེའི་འདུན་པ་ཟློག་པར་བྱེད། རྣམ་པར་ཟློག་པར་བྱེད་པའི་སེམས་ཅན་དེ་ལ་ཡང་བྱང་ཆུབ་སེམས་དཔའ་ཁོང་ཁྲོ་བར་མི་འགྱུར་གྱི། སེམས་ཅན་དེ་ལ་བྱམས་པ་ཆེན་པོ་དང་སྙིང་རྗེ་ཆེན་པོ་བསྐྱེད་
དེ། དད་པ་དང་། བརྩོན་འགྲུས་དང་། དྲན་པ་དང་། ཏིང་ངེ་འཛིན་དང་། ཤེས་རབ་ཀྱི་སྟོབས་ལས་ཀྱང་མི་ཉམས་ཏེ། དེའི་ཕྱིར་མི་ལྡོག་པའི་སྟོབས་དེས་སེམས་ཅན་དཔག་ཏུ་མེད་གྲངས་མེད་པ་དག་སངས་རྒྱས་ཀྱི་བྱང་ཆུབ་ལ་རབ་ཏུ་འགོད། རབ་ཏུ་འཛུད་དེ། འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་སྟོབས་བསྒྲུབ་པ་
ཞེས་བྱའོ། །དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཡེ་ཤེས་བསྒྲུབ་པ་གང་ཞེ་ན། འདི་ནི་བྱང་ཆུབ་སེམས་དཔའ་ནི་འདི་ལ་སྐྱེ་བའམ། འཆི་བའམ་རྒ་བ་ཡང་མེད་དོ། །འདི་དག་ནི་ཆོས་ཙམ་དུ་ཟད་དེ་བདག་དང་བདག་གི་དང་བྲལ་བའོ་ཞེས་ཆོས་ཐམས་ཅད་ཡང་དག་པ་ཇི་ལྟ་བ་བཞིན་དུ་རབ་ཏུ་ཤེས་ཏེ། དེ་ཡེ་
ཤེས་ཀྱི་སྟོབས་དེ་ཉིད་ཀྱིས་སེམས་ཅན་དཔག་ཏུ་མེད་གྲངས་མེད་པ་དག་ཡེ་ཤེས་ཀྱི་སྟོབས་དེ་ལ་རབ་ཏུ་འགོད་དོ། །སེམས་ཅན་ཐམས་ཅད་ཀྱི་ཆོས་ཉིད་དེ་ལས་རྣམ་པར་བཟློག་ནས་འཇིག་རྟེན་གྱི་ཆོས་རྣམས་ལ་རབ་ཏུ་འཇོག་མི་ནུས་ཏེ། འདི་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཡེ་ཤེས་བསྒྲུབ་པའོ། །འདི་
ནི་བསྒྲུབ་པ་རྣམ་པ་བཅུ་སྟེ། འདི་དག་ནི་བྱང་ཆུབ་སེམས་དཔའི་སྟོབས་བཅུ་ཞེས་བྱ་སྟེ། སྟོབས་མཐའ་ཡས་པ་གཞན་དང་གཞན་དག་ཀྱང་རྣམ་པར་སྨིན་པའི་ཚུལ་གྱིས་བྱང་ཆུབ་སེམས་དཔའི་ས་ལེགས་པའི་བློ་གྲོས་ལ་མངོན་པར་འགྱུར་བས་སྟོབས་པོ་ཆེ་ལ་ཞེས་བྱའོ། །བྱང་ཆུབ་སེམས་
