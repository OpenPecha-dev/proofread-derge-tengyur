བསྒྱུར་བའོ། །༄༅༅། །རྒྱ་གར་སྐད་དུ། ཧེ་བཛྲ་ཧོ་མ་བི་དྷི། བོད་སྐད་དུ། དགྱེས་པ་རྡོ་རྗེའི་སྦྱིན་སྲེག་གི་ཆོ་ག །དཔལ་དགྱེས་པ་རྡོ་རྗེ་ལ་ཕྱག་འཚལ་ལོ། །དཔལ་ལྡན་དགྱེས་པའི་རྡོ་རྗེ་དང་། །བདག་མེད་མ་ལ་ཕྱག་བྱས་ཏེ། །རྡོ་རྗེ་སློབ་དཔོན་བཀའ་དྲིན་
གྱིས། །སྦྱིན་སྲེག་ཆོ་ག་བཤད་པར་བྱ། །བདག་གཞན་ཕན་པར་འདོད་པ་ཡིས། །རྣམ་རྟོག་ང་རྒྱལ་སོགས་སྤངས་ཏེ། །ཆོས་ཕྱིར་འབྲང་བའི་བློ་ཡིས་ནི། །སྦྱིན་སྲེག་ཆོ་ག་མཉན་པར་གྱིས། །བླ་མེད་ནང་དང་ཕྱི་ཡི་སྟེ། །སྦྱིན་སྲེག་དབྱེ་བ་རྣམ་པ་གསུམ། །ཡེ་ཤེས་མེ་ལ་ཕུང་པོ་
སོགས། །བུད་ཤིང་བསྲེག་པ་བླ་མེད་ཡིན། །ཁ་ཟས་བཟའ་སོགས་ནང་གིར་བ་ཤད། །ཕྱི་ཡི་བསྲེག་རྫས་སྦྱིན་སྲེག་སྟེ། །དཀྱིལ་འཁོར་འདི་ནི་ལྷག་གནས་དང་། །སྒྲུབ་པའི་དུས་སུ་སྦྱིན་སྲེག་བྱ། །གསང་སྔགས་མ་བཟླས་མི་འགྲུབ་ལ། །སྦྱིན་སྲེག་མ་བྱས་འབྲས་མི་འབྱིན། །ལས་
རྣམས་ཐམས་ཅད་རབ་བསྒྲུབ་ཕྱིར། །བརྩོན་པས་སྦྱིན་སྲེག་ལས་རྣམས་བྱ། །སྡིག་པ་གཞིག་ཕྱིར་ཞི་བ་སྟེ། །བསོད་ནམས་ཡོངས་སུ་སྤེལ་རྒྱས་ཡིན། །ཆགས་པ་བསྒོམ་ཕྱིར་དབང་གྱུར་ཏེ། །རྟོག་པ་བསད་ཕྱིར་མངོན་སྤྱོད་ཡིན། །ཉོན་མོངས་སྡིག་པ་ཞི་བྱེད་ཅིང་། །ཏིང་ངེ་འཛིན་ནི་
རྒྱས་བྱེད་ཕྱིར། །རང་གི་ལྷ་ཡི་སྦྱོར་བས་བདག །ཅི་འདོད་པར་ནི་སེམས་ཀྱིས་བྱ། །དེ་ཕྱིར་ཐམས་ཅད་བཤད་བྱ་སྟེ། །ཐབ་དང་རྫས་ཀྱི་མཚན་ཉིད་དང་། །སྒྲུབ་པ་པོ་དང་ལྷ་དག་དང་། །སྒྲུབ་པའི་ཐབས་དང་རྣམ་པ་ལྔ། །དེ་ལ། སྦྱིན་སྲེག་སྤྱི་ཡི་ཐབས་བཤད་པ། །སེམས་ཅན་
དབྱེ་བ་མཐའ་ཡས་པ། །ལས་ཀྱི་དབྱེ་བས་བཞི་རུ་འགྱུར། །རིགས་ཀྱི་དབྱེ་བས་ལྔ་རུ་སྟེ། །ཞི་བ་ཁ་དོག་དཀར་པོ་སྟེ། །རྒྱས་པ་གསེར་གྱི་མདོག་ཏུའོ། །དབང་ལ་ཁ་དོག་དམར་པོའོ། །མངོན་སྤྱོད་ལ་ནི་ནག་པོའོ། །དགུག་དང་བསྐྲད་པ་ནག་པོའོ། །ཞི་བ་ལ་ནི་ཟླུམ་པོ་སྟེ། །རྒྱས་
