རྟག་པ་དང་། སྒྲ་རྣམས་ནི་ལྡོག་པས་བྱས་པའི་ཐ་དད་པ་ཡོད་པ་ཉིད་དོ། །འདི་ལྟར། དོན་གྱི་རང་གི་ངོ་བོ་གཅིག །བདག་ཉིད་མངོན་སུམ་ཡིན་པ་ལ། །ཚད་མ་རྣམས་ཀྱིས་བརྟག་བྱ་གང་། །མ་མཐོང་ཆ་གཞན་ཅི་ཞིག་ཡོད། །དོན་གྱི་བདག་ཉིད་གཅིག་པོ་དེ་ནི་མངོན་སུམ་
གྱིས་མངོན་སུམ་དུ་བྱས་པ་དེའི་ཕྱིར་འདི་ལ་བརྟག་པར་བྱ་བ་གཞན་ཅི་ཞིག་ཡོད། གལ་ཏེ་ཆོས་ཅན་ཙམ་མངོན་སུམ་གྱིས་གྲུབ་པ་ཡིན་གྱི། མི་རྟག་པ་ཉིད་ལ་སོགས་པའི་ཆོས་ནི་མ་ཡིན་ནོ་ཞེ་ན། མ་ཡིན་ཏེ་མི་རྟག་པ་ཉིད་དང་ལྷན་ཅིག་གཅིག་ཉིད་ཡིན་པའི་ཕྱིར་རོ། །མི་རྟག་པ་ཉིད་ཅེས་བྱ་བ་དེ་ལས་གཞན་
ནི་ཡོད་པ་མ་ཡིན་ནོ། །མི་རྟག་གཞན་ཉིད་ཡིན་ན་སྒྲ། ཆོས་ཅན་དུ་ནི་ཇི་ལྟར་འཛིན། །དུས་གཞན་ཁྱབ་དང་ཅིག་ཤོས་ཤེས། །བྱ་བ་གཉི་ག་ཡང་མི་འཐད། །གལ་ཏེ་དུས་གཞན་ཁྱབ་ན་ནི། །རྟག་པ་ཉིད་དུ་ཡོངས་གཟུང་བས། །ཚད་གཞན་སྟོང་ཕྲག་ཡོད་ན་ཡང་། །དེ་ལས་མི་རྟག་ཉིད་འཛིན་
མིན། །ཇི་ལྟར་ཚད་མ་གཞན་དག་ལས། །མངོན་སུམ་གྱིས་ནི་རྟོགས་པ་ཉིད། །སེལ་གྱུར་མངོན་སུམ་ཚད་མ་གཞན། །ཐམས་ཅད་ཀྱི་ནི་རྩ་བ་ཡིན། །སྡོམ་དང་ཆོས་མཚུངས་པ་སྐྱེད་པར་བྱེད་པ་ཟ་བ་ནི་རྒྱལ་པོའི་བུ་དག་ཁོ་ན་ཡན་གྱི། ཚད་མར་འགྱུར་བའི་རྟོགས་པ་དག་ནི་མ་ཡིན་ནོ། །
ཚད་མ་ཐམས་ཅད་ཀྱི་གཙོ་བོ་ནི་མངོན་སུམ་ཡིན་ན། དེ་ཅི་ནག་པོ་ཆེན་པོའི་འདི་མཁའ་འགྲོ་མ་རྣམས་ཀྱིས་ཟ་བ་འམ། འོན་ཏེ་མངོན་སུམ་གྱིས་རྟག་པ་ཉིད་དམ། མི་རྟག་པ་ཉིད་རྟག་པ་ཉིད་རྟོགས་པ་མ་ཡིན་ཏེ། དེས་ནི་དངོས་པོ་ཐམས་ཅད་ཀྱི་རང་གི་ངོ་བོ་ཙམ་གཅོད་པ་ཡིན་ནོ། །རྟག་པ་ཉིད་
དམ་མི་རྟག་པ་ཉིད་ནི་ཕྱིས་ཀྱི་ཚད་མ་ལ་བརྟེན་པ་ཡིན་ནོ་ཞེ་ན། མི་རྟག་པ་ཉིད་རང་གི་ངོ་བོར་གྱུར་པ་མ་ཡིན་ནམ་ཅི་ལྟར་མ་བཟུང་བ་ཞེས་བྱ་བར་འགྱུར། གང་གི་ཕྱིར་རྟག་པ་འདི་ནི་དུས་ཀྱིས་ཆ་རྒྱ་ཆེར་ཁྱབ་པ་ཉིད་ཡིན་ལ་མི་རྟག་པ་ནི་དེ་ལས་བཟློག་པ་ཉིད་དོ། །དངོས་པོ་ཁྱབ་པར་གཟུང་ན་ནི། །
དེ་ལྟ་ཁྱབ་པར་གཟུང་བར་འགྱུར། །གལ་ཏེ་དེ་ནི་མ་བཟུང་ན། །མི་རྟག་པར་ནི་རིག་པར་འགྱུར། །གང་གི་ཚེ་ཉིད་དངོས་པོ་རྣམས་དུས་གཞན་ཁྱབ་པར་མ་རྟོགས་པ་ཉིད་ཀྱི་ཆོས་མི་རྟག་པ་ཉིད་རྟོགས་པར་གསལ་བ་ཡིན་ནོ། །ཉིད་རྟོགས་པ་ཡིན་གྱི་མི་རྟག་པ་ཉིད་ནི་མ་ཡིན་ནོ་ཞེ་ན། མི་རྟག་པ་
