གྱིས་དེ་བྱིན་བརླབ། །ཞི་བ་སྤྲོ་བྱ་སངས་རྒྱས་གནས། །ནམ་མཁའི་དབྱིངས་ནི་རབ་ཏུ་བརྒྱན། །ཞེས་གསུངས་པས་རབ་ཏུ་བསྒྲུབ་པའི་ཕྱིར་ཕྱི་རོལ་གྱི་ཕྱག་རྒྱ་དང་སྙོམས་པར་འཇུག་པའམ། དེ་མེད་ན་ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱ་དང་སྙོམས་པར་འཇུག་པས་ཡང་དག་པར་བསྐྱོད་
པའི་དགའ་བའི་སྒྲས། ཡེ་ཤེས་ཀྱི་འཁོར་ལོ་སྤྱན་དྲངས་ལ། ཞབས་བསིལ་དང་ཞལ་བསིལ་དང་མཆོད་ཡོན་དབུལ་བ་སྔོན་དུ་འགྲོ་བས་རང་གི་གནས་སུ་བཅུག་ནས་འདོད་ཆགས་ཆེན་པོ་ཞུ་བར་གྱུར་པ་རྡོ་རྗེའི་ལམ་ནས་པདྨར་ཕྱུང་བའི་ཡེ་ཤེས་ཀྱི་ལྷའི་འཁོར་ལོའི་བདག་ཉིད་སྤྲོས་ཏེ། 
དམ་ཚིག་གི་འཁོར་ལོ་ལ་ཇི་ལྟ་བ་བཞིན་དུ་ཞུགས་པར་བསམས་ནས། མིག་དང་སྐུ་ལ་སོགས་པ་བྱིན་གྱིས་བརླབས་ཤིང་དེ་མ་ཐག་ཏུ་རང་གི་སྙིང་གའི་ས་བོན་གྱི་འོད་ཟེར་གྱིས་བཀུག་པའི་དེ་བཞིན་གཤེགས་པ་དང་ལྷ་མོ་རྣམས་ཀྱིས་དབང་བསྐུར་ནས་ཇི་ལྟར་རིགས་པར་རྒྱས་གདབ་སྟེ་
རིགས་མི་ཤེས་པ་རྣམས་ནི་མི་བསྐྱོད་པའམ་རྡོ་རྗེ་སེམས་དཔས་སོ། །དེ་ནས་ལྷ་མོ་སྤྲོས་པ་རྣམས་ཀྱིས་ཡིད་ཀྱི་རང་བཞིན་གྱི་མཆོད་པས་མཆོད་ཅིང་། བསིལ་བའི་རས་དང་རས་ཟུང་བླ་རེ་ལ་དཔྱངས་པ་དང་གོས་དང་། ཡི་གེ་གསུམ་གྱིས་བསྔགས་པའི་བཟའ་བ་དང་བཅའ་བ་དང་བལྡག་
པ་དང་བཏུང་བའི་ཉེ་བར་སྤྱད་པ་ལ་སོགས་པ་དང་། གསེར་དང་བུ་མོ་དང་གླང་པོ་ཆེ་དང་རྟ་དང་བ་ལང་ལ་སོགས་པ་ཕུལ་ལ་གཟུགས་རྡོ་རྗེ་མ་ལ་སོགས་པའི་ངོ་བོ་ཕྱི་རོལ་གྱི་རྣལ་འབྱོར་མ་བཅུ་དྲུག་གམ་བརྒྱད་དམ་བཞིས་རང་རང་གི་མཆོད་པས་མཆོད་ནས་དྲིལ་བུ་དཀྲོལ་ཞིང་མི་བསྐྱོད་རྡོ་རྗེ་
ཞེས་པ་ལ་སོགས་པས་བསྟོད་དེ། རང་དང་ཤེས་རབ་ཀྱི་མེ་ཏོག་དང་བཅས་པའི་བྱང་ཆུབ་ཀྱི་སེམས་པདྨའི་སྣོད་དུ་བཅུག་པ་བདུད་རྩི་དང་བཅས་པ་ཕུལ་བས། དེ་རྣམས་བདུད་རྩི་མྱངས་པས་ཚིམ་པར་གྱུར་ཅིང་བདེ་བ་ཆེན་པོའི་རང་བཞིན་དུ་མོས་པར་བྱས་ལ་འཁོར་ལོའི་དབང་པོའི་སྔགས་
བརྒྱ་རྩ་བརྒྱད་དང་། དེ་ལས་གཞན་དཀྱིལ་འཁོར་བའི་སྔགས་ཉི་ཤུ་རྩ་གཅིག་དང་ཡི་གེ་བརྒྱ་པ་བཟླས་པར་བྱའོ། །འཆད་པར་འགྱུར་བའི་ཆོ་གས་འབྱུང་པོ་ཐམས་ཅད་པའི་གཏོར་མ་གཏང་བར་བྱ་སྟེ། ཡོད་ན་སྒྲུབ་པའི་གྲོགས་པོ་དཀྱིལ་འཁོར་དུ་བཞག་ནས་ཕྱི་རོལ་དུ་ཕྱུང་ལ་ཤར་ལ་
