དང་ཁང་ཆེན་ལ། །ལེགས་པར་འཇུག་སྟེ་རྒྱལ་ལ་ཡང་ནི་དེ་བཞིན་ཉིད། །རྒྱལ་མཚན་ཇི་སྙེད་བུམ་པ་དང་ནི་མཚུངས་པ་སྟེ། །ཐུབ་པའི་དབང་པོའི་ཡེ་ཤེས་བདུད་རྩིས་གང་བ་ཉིད། །བདུད་རྩིའི་འོད་ཅན་དམ་པའི་དཀྱིལ་འཁོར་རང་བཞིན་གྱིས། །ལེགས་པར་བཀྲུས་ནས་གཙུག་
ལག་ཁང་གི་ལུས་སུའོ། །དེ་བསླང་བ་ནི་བྷྲཱུཾ་ལས་བྱུང་བའི་མཆོད་རྟེན་གཟུགས། །རྣམ་པར་བསམས་ནས་གདུགས་དང་གོས་དང་རྒྱན་གྱིས་བརྒྱན། །མེ་ཏོག་གོས་དང་སྐུད་པ་ལྔས་བཅིངས་བཟླས་བྱ་སྟེ། །མཉམ་པར་གཟུག་བྱ་རྡོ་རྗེ་དང་ནི་བཅས་པས་སོ། །ཤར་དུ་གནས་པས་
རྒྱལ་བའི་དབང་པོའི་གསང་སྔགས་དང་། །ལྷོ་རུ་གནས་པས་རྒྱལ་མཚན་རྩེ་མོའི་གསང་སྔགས་ནི། །ནུབ་ཏུ་གནས་བྱས་འོད་དཔག་མེད་པའི་གསང་སྔགས་དང་། །བྱང་ནས་ཀྱང་ནི་ཐུབ་པའི་རྒྱན་གྱི་ཡིད་ཀྱིས་ཏེ། །གཡས་སྐོར་དུ་ནི་ཀུན་ནས་རབ་ཏུ་བསྐོར་འགྲོ་ཞིང་། །མེ་ཏོག་གོས་
དང་སྐུད་པར་བྱས་པའི་ལག་པ་ཡིས། །རབ་ཏུ་གནས་བྱ་མེ་ཏོག་མཆོད་ཡོན་འབྲས་སོགས་ཀྱི། །བུམ་པས་མངོན་པར་དབང་བསྐུར་བྱ་སྟེ་འབད་པས་སོ། །རོལ་མོ་དང་བཅས་དུང་དང་ཟངས་དུང་རྣམས་ཀྱི་ནི། །སྒྲ་དང་བཅས་ཤིང་བུམ་པ་རྣམས་དང་མཉམ་ལྡན་པས། །བུད་མེད་རྣམས་ཀྱི་བཀྲ་
ཤིས་གླུ་དབྱངས་དང་། །ལྷན་ཅིག་པ་ཡི་རྡོ་རྗེ་སློབ་དཔོན་རྣམས་ཀྱིས་སོ། །རྒྱལ་བ་རྣམ་པར་སྣང་མཛད་ཀྱི་སྔགས་ནི། །ཨོཾ་སུཀྵྨེ་སུཀྵྨེ་ས་མ་ས་མ་ཡེ་ཤཱནྟེ་ཤཱནྟེ། དཱནྟེ་དཱནྟེ། ཨ་ས་མཱ་རོ་པེ། ནི་རཱ་ལཾ་བྷེ་ནི་རཱ་ཀུ་ལེ་ནིརྦཱ་ནེ། ཡ་ཤོ་བ་ཏི། མ་ཧཱ་ཏེ་ཇོ་སརྦ་བུངྡྷཱ་དྷིཥྛཱ་ན་ཨ་དྷིཥྛི་ཏེ་སྭཱ་ཧཱ། ཤར་ནས་ལན་ཉི་
ཤུ་རྩ་གཅིག་གོ། །རྒྱལ་མཚན་རྩེ་མོའི་དཔུང་རྒྱན་གྱི་སྔགས་ཆེན་པོ་ནི། ཨོཾ་དྷྭ་ཛ་ཨ་གྲ་ཀེ་ཡཱུ་རི་ཛ་ཡ་ཛ་ཡ། བི་ཛ་ཡ་བི་ཛ་ཡ། ཛ་ཡ་བཱ་ཧི་ནི། ཤངྐ་རི་པྲ་བྷཉྫ་ནི། བྷཉྫ་བྷཉྫ། མོ་ཧ་ཡ་མོ་ཧ་ཡ། བྷ་ག་ཝ་ཏི་ཛ་ཡ་བཱ་ཧི་ནི། ཛ་ཡོ་ཏ་རི། མ་ཐ་མ་ཐ། པྲ་མ་ཐ་པྲ་མ་ཐ། ཧ་ས་ཧ་ས། 
གྲ་ས་གྲ་ས། ཧཱུཾ་ཧཱུཾ། ཧཻ་ཧཻ། ལ་གྷུ་ལ་གྷུ། ལམྤོ་ད་རི། དྲི་ནི་དྲེ། ཙ་ཏུར་ཝཱ་ཀྐྲེ། ཙ་ཏུར་དཾཥྚེ། ཙ་ཏུར་བྷུ་ཛེ། ཨ་སི་མུ་ཁ་ལ་ཙཀྲ་ཏྲི་ཤཱུ་ལ་བཛྲ་ཀ་བ་ཙ་དྷཱ་རི་ཎི། རཀྵ་རཀྵ་མཱཾ། བྷ་ག་ཝ་ཏི། ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། དྷུ་ན་དྷུ་ན། བི་དྷུ་ན་བི་དྷུ་ན། སྥོ་ཊ་ཡ་སྥོ་ཊ་ཡ། ཧཱུཾ་ཧཱུཾ་ཕཊ།
