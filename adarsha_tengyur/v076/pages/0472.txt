བ་རྣམས་བྱིན་གྱིས་ཕྲ་བ་དང་། ཤངས་མཐོ་བ་ཉིད་དང་། ཤངས་གཙང་བ་ཉིད་དང་། སྤྱན་ཡངས་པ་ཉིད་དང་། རྫི་མ་སྟུག་པ་ཉིད་དང་། སྤྱན་དཀར་ནག་འབྱེས་ཤིང་པདྨའི་འདབ་མ་ལྟ་བུ་ཉིད་དང་། ཕྱག་རིང་ཞིང་རྒྱས་པ་ཉིད་དང་། སྨིན་ཚུགས་རིང་བ་ཉིད་དང་། སྨིན་མ་འཇམ་པ་ཉིད་དང་། 
སྨིན་མ་སྣུམ་པ་ཉིད་དང་། སྨིན་མའི་སྤུ་མཉམ་པ་ཉིད་དང་། ཕྱག་རིང་ཞིང་རྒྱས་པ་ཉིད་དང་། སྙན་མཉམ་པ་ཉིད་དང་། སྙན་གྱི་དབང་པོ་མ་ཉམས་པ་ཉིད་དང་། དཔྲལ་བ་ལེགས་པར་དབྱེས་པ་ཉིད་དང་། དཔྲལ་བའི་དབྱེས་ཆེ་བ་ཉིད་དང་། དབུ་ཤིན་ཏུ་རྒྱས་པ་ཉིད་དང་། དབུ་སྐྲ་བུང་བ་ལྟར་གནག་
པ་ཉིད་དང་། དབུ་སྐྲ་སྟུག་པ་ཉིད་དང་། དབུ་སྐྲ་འཇམ་པ་ཉིད་དང་། དབུ་སྐྲ་མ་འཛིངས་པ་ཉིད་དང་། དབུ་སྐྲ་མི་གཤོར་བ་ཉིད་དང་། དབུ་སྐྲ་དྲི་ཞིམ་པ་ཉིད་དང་། ཕྱག་དང་ཞབས་ཀྱི་མཐིལ་ལ་དཔལ་བེའུ་དང་བཀྲ་ཤིས་དང་དགའ་བ་འཁྱིལ་པས་བརྒྱན་པ་ཉིད་ཀྱང་ངོ་།།      །།སྔོན་དང་སྔོན་གྱི་
སྐྱེ་བ་རྣམས་སུ་རྣམ་པ་མང་པོའི་བསོད་ནམས་ཉེ་བར་བསགས་པའི་དགེ་བའི་རྩ་བ་ལས་བཅོམ་ལྡན་འདས་ཀྱིས་དཔེ་བྱད་བརྒྱད་ཅུ་གསུངས་ཤིང་། ཆོས་ཀྱི་དུང་གི་མདོ་སྡེ་ལས་ཀྱང་བཅོམ་ལྡན་འདས་ལ་རི་མོ་བརྒྱད་ཅུས་རྗེས་སུ་མཚན་པར་གསུངས་པ་ནི་འདི་ལྟ་སྟེ། གང་ཡང་འཇམ་དཔལ་
དཔེ་བྱད་བརྒྱད་ཅུ་རྗེས་སུ་གྲུབ་པའི་བསོད་ནམས་ཀྱི་ཕུང་པོ་དེ་ཉིད་ཀྱི་བསོད་ནམས་ཀྱི་ཕུང་པོ་དུ་མ་བྱེ་བ་འབུམ་བརྒྱ་སྟོང་དུ་བསྒྱུར་བ་ལ་ནི་དེ་བཞིན་གཤེགས་པའི་ཕྱག་དང་ཞབས་ཀྱི་མཐིལ་ལ་མཚན་པ་གཅིག་བསྐྱེད་པར་འགྱུར་རོ། །དེ་བཞིན་དུ་སོ་སོར་རི་མོ་རྣམས་རྗེས་སུ་མཚོན་པར་འགྱུར་
རོ། །རི་མོ་བརྒྱད་ཅུས་རྗེས་སུ་མཚན་པ་དེ་བཞིན་གཤེགས་པའི་ཕྱག་དང་ཞབས་ཀྱི་མཐིལ་ལ་མངོན་པར་སྐྱེ་བར་འགྱུར་བ་ནི་འདི་ལྟ་སྟེ། གདུགས་དང་རྒྱལ་མཚན་དང་དཔལ་བེའུ་དང་ཕྲེང་བ་དང་ལྕགས་ཀྱུ་དང་ཅོད་པན་དང་རལ་གྲི་དང་བུམ་པ་དང་གླང་པོ་དང་རྟ་དང་སྟག་དང་ནམ་མཁའ་ལྡིང་
དང་ཆུ་སྲིན་དང་ཉ་དང་རུས་སྦལ་དང་རྨ་བྱ་དང་ཀ་ལ་པིངྐ་དང་ཤང་ཤང་དང་ཙ་ཀོ་ར་དང་ངུར་པ་དང་ནེ་ཙོ་དང་ངང་པ་དང་ཕུག་རོན་དང་རྡོ་རྗེ་དང་ཆུ་གཏེར་ཆེན་པོ་དང་འདོད་འཇོ་དང་བ་ལང་དང་ཀླུ་དང་ར་དང་ཁྱུ་མཆོག་དང་རི་དང་ཤིང་དང་ས་དང་རི་དགས་ནག་པོ་དང་ནོར་བུ་རིན་པོ་ཆེ་དང་དུང་གི་མཆོག་དང་འཁོར་ལོ་དང་
