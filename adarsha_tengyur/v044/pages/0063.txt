སྔགས་ཀྱི་བཟླས་བརྗོད་བརྩམ་པའོ། །ཨོཾ་ཨཱཿཨོཾ་ཀྲི་ཏ་ཨ་ནུཥྛ་ན་ཛྙཱ་ན་རསྟུ་ཧཱུྃ། དེ་ཡིས་དོན་ཡོད་གྲུབ་པའི་ཚོགས། །སྤྲོས་ཏེ་བཀུག་བྱས་ཞུགས་བྱས་ལ། །གནས་བརྟན་ངོ་བོར་ཆགས་བྱས་ན། །ཟླ་བ་ཕྱེད་ན་དེ་དག་འགྲུབ། །སྐུ་དང་གསུང་ཐུགས་རྡོ་རྗེ་ཡི། །རྡོ་རྗེ་ཐོགས་མེད་འདྲ་བ་འོང་། །དེ་ནི་
དཔལ་ལྡན་ཡེ་ཤེས་མཚོ། །སེམས་ཅན་དོན་ཀུན་འབྱུང་བར་འགྱུར། །དེ་བཞིན་སྣང་མཛད་བསྒོམ་བྱས་ཏེ། །སྔགས་ཀྱི་བཟླས་བརྗོད་བྱ་བའོ། །ཨོཾ་ཨཱཿཨོཾ་ཨཱ་དཪྴ་ཛྙཱ་ན་བཛྲ་རསྟུ་ཧཱུྃ། སྔོན་བཞིན་ཞུགས་སོགས་རབ་བྱས་ནས། །ཟླ་ཕྱེད་དེ་དང་འདྲ་བར་འགྲུབ། །ཐམས་ཅད་སྔ་མ་བཞིན་
བྱས་ཏེ། །སྣང་མཛད་མགོ་མགྲིན་ཐུགས་ཀ་རུ། །འཁོར་ལོ་པདྨ་རྡོ་རྗེ་ལ། །ཟླ་བའི་དབུས་ནས་ཨོཾ་གསུམ་གནས། །ཨོཾ་ཨཱཿཨོཾ་ཨོཾ་ཨོཾ་ཏྲི་ག་ཀ་ར་ཡ་རསྟུ་ཧཱུྃ། དེ་ཡིས་རྡོ་རྗེ་གསུམ་གྱི་ཚོགས། །བཀུག་ནས་གནས་གསུམ་དུ་བརྟན་པར། །བྱས་ནས་ཉེ་བའི་མཚན་མ་མཐོང་། །གོང་བཞིན་
