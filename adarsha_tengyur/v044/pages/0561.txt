རྒྱས་ཐམས་ཅད་ཡོན་ཏན་གྱིས་དག་པའི་རྟེན་དང་བརྟེན་པའི་རང་བཞིན་དག་ཡོད་པ་ན་བསྒྲུབ་བྱར་འདོད་པའི་རྒྱལ་བ་ལྷ་མོའི་འདོད་པས་དམ་པར་འཁྱུད་པའི་བྱ་བར་གནས་ཤིང་ཆགས་པའོ། །དེ་ནས་གཞན་དུ་ཡང་འགྱུར་བ་སྟེ། མཆོག་ཏུ་རྣམ་པར་གཡེང་བའི་དགའ་བ་འཇིགས་

དེ་ཡི་ཡེ་ཤེས་རྟོགས་རང་བཞིན། །གསལ་བ་ཡིན་པས་རྟོགས་པ་མེད། །སྣ་ཚོགས་གཉིས་སུ་མེད་རང་བཞིན། །ཡོད་མེད་རྟོགས་པས་རྗོད་བྱེད་ཕྱིར། །ཐུགས་རྗེས་སྤྲོ་བ་དང་བསྡུ་བ། །སྔར་གོམས་པ་ཡི་དབང་གིས་ཡིན། །ལྷུན་གྲུབ་སེམས་ཅན་དོན་ཡིན་
ཏེ། །དེ་ནས་ཡིད་བཞིན་ནོར་བུ་བཞིན། །དེ་ནི་ལྷུན་གྱིས་གྲུབ་པས་ན། །རྟག་ཏུ་མ་ཉམས་པ་ཡི་བློ། །ནམ་མཁའི་མཐས་གཏུགས་འགྲོ་བ་ཡི། །དོན་བྱེད་པ་ལ་རབ་ཏུ་འཇུག །དེ་བས་འདི་ལྟར་སྐུ་དང་ལྡན་པ་ཤེས་རབ་ལ་འཁྱུད་པ་རང་གསལ་རང་བཞིན་ཏེ། མཆོག་
ཏུ་ཐུགས་རྗེས་རབ་ཏུ་རྒྱུན་ཆུད་པ་དང་རྣམ་པར་བྲལ་བའི་སྐུ། །དཔེ་མེད་པ་ཡི་བདེ་གང་རྟེན་དང་བརྟེན་པའི་རང་བཞིན་ནི། །ཐུབ་པའི་དབང་པོའི་ཡན་ལག་བདུན་དང་ལྡན་པ་ལ་ཕྱག་འཚལ། །འདི་ལྡན་ཡན་ལག་བདུན་དང་ལྡན་པའི་བཞི་པ་ནི་བསྒྲུབ་བྱའོ། །འདི་ཉིད་ནི་
