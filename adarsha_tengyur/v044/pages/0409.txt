བསྒོམ་སོགས་ཀུན་ལ་རབ་ཏུ་སྩོལ། །སངས་རྒྱས་ཀྱིས་གསུངས་འབྲས་མེད་རབ་ཐལ་འགྱུར། །རྡོ་རྗེའི་ས་བོན་དྲི་མེད་དགའ་འབབ་པ། །དག་པའི་ཆུ་སྐྱེས་ཡང་དག་མྱུ་གུ་འབྱུང་རྒྱས་ཤིང་། །བཞི་པའི་འབྲུ་ནི་ཡོངས་གསལ་སྨིན་འགྱུར་བའི། །བཞི་པ་དག་
ནི་མཁས་པ་རྣམས་ལ་སྦས། །གསལ་བྱེད་བདུད་རྩི་ལྔ་དང་ཐིག་ལེ་ཟླ་བ་དང་། །སྨིན་དབུས་ཐིག་ལེ་ལས་བྱུང་དཀྱིལ་འཁོར་དང་། །རླུང་གི་རང་བཞིན་ལྕེའུ་ཆུང་སོགས་དེ་ཉིད། །ངོ་བོ་མིན་པ་བདག་ཉིད་ཀྱིས་རྟག་ཏུ། །རྨི་ལམ་མིག་འཕྲུལ་གཟུགས་བརྙན་སྒྱུ་མ་དང་། །སྨིག་རྒྱུ་དྲི་མའི་
གྲོང་ཁྱེར་ཆུ་ཟླ་དང་། །གཞན་གྱི་སྒྲ་ཡིས་མངོན་པར་བརྗོད་པ་ཡིས། །འདིར་བཤད་བསྒྲུབ་བྱ་གཞན་ལ་ཡོད་མ་ཡིན། །ཟབ་མོ་སྟོང་ཉིད་སོ་སོར་སྣང་བ་ཙམ། །ཞི་བ་ཤིན་ཏུ་ཕྲ་བ་སྒྲས་བརྗོད་མིན། །གོས་མེད་དངོས་བྲལ་དྲི་མེད་ལ་སོགས་དང་། །གཞན་སྒྲུབ་ཡོད་པ་འཁྲུལ་པར་མི་
བྱའོ། །རྣམ་པ་བདུན་པོ་བཀོད་པའི་ངོ་བོ་དག་ལས་ནམ་མཁའ་མ་ལུས་པའི། །སློབ་དབྱིངས་ཁྱབ་པར་བྱེད་པའི་བསོད་ནམས་གང་ཞིག་བདག་གིས་ཐོབ་གྱུར་པ། །དེས་ནི་འཇིག་རྟེན་འདི་དག་དཔེ་མེད་པ་ཡི་རིག་མས་འཁྱུད་པར་འགྱུར་བ་ཡི། །མཛེས་པའི་སྐུ་ལས་རྒྱལ་
བའི་སྤྲུལ་པ་དག་གིས་འགྲོ་བའི་དོན་བྱེད་ཤོག །དེ་ཁོ་ན་ཉིད་རིན་པོ་ཆེ་སྣང་བ་སློབ་དཔོན་ངག་གི་དབང་ཕྱུག་གྲགས་པས་མཛད་པ་རྫོགས་སོ།།༄༅༅། །རྒྱ་གར་སྐད་དུ། ཏཏྟྭཱ་རཏྣ་ཨཱ་ལོ་ཀ་བྱཱ་ཁྱཱ་ན། བོད་སྐད་དུ། དེ་ཁོ་ན་ཉིད་རིན་པོ་ཆེ་སྣང་བའི་རྣམ་པར་བཤད་
པ། བཅོམ་ལྡན་འདས་གསུང་གི་དབང་ཕྱུག་ལ་ཕྱག་འཚལ་ལོ། །འོད་དཔག་མེད་པ་རིན་ཆེན་གྱིས། །འོད་ཀྱིས་སྤྱི་གཙུག་མཛེས་པ་ཡི། །འཇིག་རྟེན་དབང་ལ་ཕྱག་འཚལ་ལོ། །བཞི་པའི་སྣང་བར་རབ་བཤད་བྱ། །དཔེ་མེད་ཅེས་བྱ་བ་ལ་སོགས་པ་ལ་གང་གི་ཕྱག་ན་པདྨ་བསྣམས་
པ་དེ་ནི་ཕྱག་ན་པདྨ་སྟེ། བཅོམ་ལྡན་འདས་སྤྱན་རས་གཟིགས་དབང་ཕྱུག་ཡིན་ལ། དེ་རྒྱལ་བར་གྱུར་ཅིག་ཅེས་བྱ་བར་སྦྱར་རོ། །ཁྱད་པར་ཇི་ལྟ་བུ་ཞིག་དང་ལྡན་ཞེ་ན། དཔེ་མེད་ཅེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། དཔེ་མེད་པ་ནི་ཤིན་ཏུ་གྱ་ནོམ་པ་ཉིད་དང་། རྒྱ་ཆེ་བ་ཉིད་དང་། འཁོར་བའི་མཐའི་
