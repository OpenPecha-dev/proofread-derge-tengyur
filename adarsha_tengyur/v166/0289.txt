སྨན་གྱི་གཞི་ལས་ཀུན་དགའ་བོ་དེ་བས་ན་དེའི་ཕྱིར་དགེ་སློང་རྣམས་ཀྱིས་སྨན་རྣམ་པ་བཞི་པོ་དུས་སུ་རུང་བ་དང་། ཐུན་ཚོད་དུ་རུང་བ་དང་། ཞག་བདུན་པ་དང་། འཚོ་བའི་བར་དུ་བཅང་བ་རྣམས་བསྟེན་པར་བྱ་བར་གནང་ངོ་ཞེས་གསུངས་པ་དེ་ལ་དུས་སུ་རུང་བའི་
དབང་དུ་བྱས་ནས། བཤད་པ། ནས་ཆན་སྣུམ་ཁུར་འབྲས་ཆན་ཉ་དང་ཤ །འདི་དག་སྲོག་གནས་བྱ་ཕྱིར་བཟའ་བར་གསུངས། །སྡོང་བུ་འབྲས་བུ་ཨུཏྤལ་ལ་སོགས་རྩ། །དེ་དག་བཅའ་བ་དུས་སུ་རུང་བ་ཡིན། །བཅོམ་ལྡན་འདས་ཀྱིས་ནས་ཆན་དང་། འབྲས་ཆན་དང་། ཤ་
དང་ཉ་ཤ་དང་སྣུམ་ཁུར་ནི་བཟའ་བ་ཡིན་པར་གསུངས་ལ། བཅའ་བ་ཡང་སྡོང་བུའི་བཅའ་བ་མི་བུ་རམ་ཤིང་ལ་སོགས་པའོ། །འབྲས་བུའི་བཅའ་བ་ནི་ཆུ་ཤིང་གི་འབྲས་བུ་ལ་སོགས་པའོ། །རྩ་བའི་བཅའ་བ་ནི་ཨུཏྤལ་ལ་སོགས་པའི་རྩ་བ་དག་དང་། ངོ་བ་ལ་སོགས་པ་སྟེ། འདི་
ནི་དུས་སུ་རུང་བ་ཡིན་པར་ཁོང་དུ་ཆུང་པར་བྱའོ། །དུས་མིན་རུང་བ་ཆུ་ཤིང་འབྲས་བུ་ལ། སོགས་པའི་ལྟུང་བ་ཚགས་བཙགས་རུང་ཆུས་བསླད། དར་བ་དང་རྩབས་ཆུར་ཁུ་སྦྱར་བའི་ཚེ། །བུ་རམ་ཤིང་ཁུ་དང་བ་འབྲུ་ཚ་ཡིན། དུས་མ་ཡིན་པར་རུང་བ་ནི་དུས་མིན་རུང་བ་ཡིན་ཏེ།
ཐུན་ཚོང་དུ་དུང་བ་དང་། ཞག་བདུན་དང་། ཇི་སྲིད་དུ་འཚོ་བའོ། །དེ་ལ་ཐུན་ཚོད་དུ་རུང་བ་ནི་ཆུ་ཤིང་གི་འབྲས་བུ་ལ་སོགས་པ་བཏུང་བ་ཡིན་ཏེ། དེ་ཡང་རུང་ཆུས་བཏབ་པ་དང་ཚགས་ཀྱིས་བཙགས་པ་ཉིད་ཡིན་ནོ། །ཕྲན་ཚོགས་ལས། བཙུན་པ་དུས་སུ་རུང་པའི་བཏུང་བ་ནི་གང་
ལགས། དུས་མ་ལགས་པར་རུང་བ་ནི་གང་ལགས། བཀའ་སྩལ་པ། ཉེ་བ་འཁོར་མ་བཙགས་པ་ནི་དུས་སུ་རུང་བའོ། །ཚོགས་ཀྱིས་ཡོངས་སུ་བཙགས་ཤིང་། ཆུས་གཏོར་བ་ནི་དུས་མ་ཡིན་པར་རུང་བའོ་ཞིག་གསུངས་པ་ཡིན་ནོ། །དེའི་ཕྱིར་བཏུང་བ་ཚགས་ཀྱིས་
བཙགས་པ་རུང་བའི་ཆུས་བསླད་པ་ཞེས་བྱ་བས་བྱེ་བྲག་ཏུ་འབྱེད་དོ། སྨན་གྱི་གཞི་ལས། ཆུ་ཤིང་འབྲས་བུ་ཀུ་ཤུ་གོ་ལ་དང་། ཨ་ཤྭད་ཐ་དང་ཨུ་དུམ་བཱ་ར་དང་། །པ་རུ་ཤ་ཀ་དང་ནི་རྒུན་འཐུམ་དང་། བརྒྱད་པ་འབྲ་གོ་དག་ནི་ཡིན་པར་བཤད། ཅིས་སྡོམ་གྱི་
