སོ་སོ་རང་རིག་ལས་འདི་ནི། །གནོད་པ་ལས་འདི་སྐྱེ་བར་འགྱུར། །རང་གི་བྱེད་པོ་རང་རིག་སྡུད་པ་པོའོ། །བདག་ཉིད་རྒྱལ་པོ་བདག་ཉིད་གཙོ་བོ་ཡིན། །འདོད་ཆགས་ཞེ་སྡང་དེ་བཞིན་ངེས་པར་ན། །གཏི་མུག་དང་ནི་ཕྲག་དོག་ང་རྒྱལ་རྣམས། །རིག་པའི་རྒྱལ་པོ་བདེ་ཆེན་
ལས། །ཉོན་མོངས་ལྔ་པོ་དག་ཏུ་བལྟ། །བཀྲེས་པ་དང་ནི་སྐོམ་པ་ཡིས། །གནོད་པར་མ་གྱུར་གང་དེ་ཚེ། །བཤང་དང་གཅི་སོགས་འཇུག་པ་མེད། །དེ་ཚེ་གྲུབ་པ་དག་ཏུ་བསྒྲགས། །དེ་ཕྱིར་བདེ་ལྡན་ཐོབ་འགྱུར་ཞིང་། །དཔའ་བོ་བདེ་ཆེན་ཐུབ་པ་ཡིས། །དོན་ཆེན་བདེ་བ་
ལ་བརྟེན་ནས། །རྣམ་པ་ཀུན་དུ་དངོས་གྲུབ་དག །ཉེ་བར་རབ་ཏུ་གནས་འགྱུར་ཞིང་། །མངོན་ཤེས་ལྔ་ཡང་འབྱུང་བར་འགྱུར། །ལྷའི་མིག་དང་།  ལྷའི་རྣ་བ་དང་། ལྷའི་སྣ་དང་། ལྷ་ཡི་རོ་དག་དང་། ལྷའི་རེག་བྱ་དང་།  ཡིད་ཀྱི་ལུས་ཀྱིས་གཞན་དུ་གནས་པ་རྣམས་དེས་
ལོངས་སྤྱོད་པར་འགྱུར་རོ། །དཔལ་ཌོཾ་བི་ཧེ་རུ་ཀས་མཛད་པའི་ལྷན་ཅིག་སྐྱེས་པ་གྲུབ་པ་ལས། སྤྱོད་པའི་དེ་ཁོ་ན་ཉིད་བསྟན་པའི་ལེའུ་སྟེ་གསུམ་པའོ།།   །།དཔལ་དགྱེས་པའི་རྡོ་རྗེའི་རྒྱུད་ཀྱི་ལྷན་ཅིག་སྐྱེས་པ་གྲུབ་པ་ལས། ལྷན་ཅིག་སྐྱེས་པ་བསྟན་པ་སློབ་དཔོན་
ཆེན་པོ་དཔལ་ཌོཾ་བི་ཧེ་རུ་ཀའི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།།   །།༄༅༅། །རྒྱ་གར་སྐད་དུ། དོ་ཧ་ཀོ་ཥ་གཱི་ཏི། བོད་སྐད་དུ། དོ་ཧ་མཛོད་ཀྱི་གླུ་འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །དུག་སྦྲུལ་ལྟ་བུའི་སྐལ་མེད་ནི། །ངེས་པར་སྐྱེ་བོ་དམ་
པ་ལ། །སྐྱོན་གྱི་དྲི་མས་དགོད་པའི་ཕྱིར། །མཐོང་བ་ཙམ་གྱིས་འཇིགས་པར་བྱོས། །དེ་ཉིད་མི་ཤེས་བྲམ་ཟེ་ནི། །གྱི་ན་རིགས་ཕྱེད་གཞི་དག་འདོན། །ས་ཆུ་ཀུ་ཤ་དག་བྱེད་དང་། །ཁྱིམ་ན་གནས་ཤིང་མེ་ལ་བསྲེག །དོན་མེད་སྦྱིན་སྲེག་བྱེད་པ་ནི། །དུ་བས་མིག་
ལ་གནོད་པར་བས། །དབྱུ་གུ་དབྱུག་གསུམ་ལེགས་ལྡན་གཟུགས། །ཐ་དད་པ་དང་དད་པས་བསྟན་པ་དག །ཆོས་དང་ཆོས་མིན་ཤེས་པར་མི་མཉམ་ཞིང་། །འགྲོ་བ་རྣམས་ནི་བརྫུན་པ་ཉིད་དུ་གོལ། །ཨེ་རའི་ཐལ་བས་ལུས་ལ་བྱུགས་ནས་སུ། །མགོ་ལ་རལ་པའི་
