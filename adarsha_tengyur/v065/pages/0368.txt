བྷ་ནེ་པྲ་ཏི་བྷ་ན་ཀུ་ཊེ་ཧཱུཾ། ཨོཾ་སརྦ་ཤོ་ཀ་ཏ་མོ་ནི་ར་ཧ་ཏ་མ་ཧི་ཧཱུཾ། ཨོཾ་སརྦ་ནི་བ་ར་ཎ་བིཥྐཾ་བི་ན་ཧཱུཾ། དེའི་སྒོར་ནི། ཨོཾ་ཡ་མཱནྟ་ཀ་ཧཱུཾ། ཨོཾ་པྲཛྙཱཾ་ཏ་ཀ་ཧཱུཾ། ཨོཾ་པདྨཱཾ་ཏ་ཀ་ཧཱུཾ། ཨོཾ་ཨ་མྲྀ་ཏ་ཀུཎྜ་ལི་བིགྷྣཱནྟ་ཀ་ཧཱུཾ། མཚམས་རྣམས་སུ། ཨོཾ་སུམྦྷ་ནི་སུམྦྷ་ཧཱུཾ། གྲྀཧྞ་གྲྀཧྞ་ཧཱུཾ། གྲྀཧྞཱ་པ་ལ་ཡ་གྲྀཧྞཱ་པ་ལ་ཡ་
ཧཱུཾ་ཨཱ་ན་ཡ་ཧོཿབྷ་ག་བཱན་བིདྱཱ་རཱ་ཛ་ཧཱུཾ་ཕཊ། ཁམས་གསུམ་རྣམ་པར་རྒྱལ་བའི་འོ། །ཨོཾ་བཛྲ་ཛྭ་ལ་ཨ་ན་ལཱརྐ་ཧཱུཾ། རྡོ་རྗེ་མེ་ལྟར་འབར་བའི་འོ། །ཨོཾ་ཧེ་རུ་ཀ་བཛྲ་ས་མ་ཡ་ཧྲཱི་སརྦ་དུཥྚ་ས་མ་ཡ་མུཊྚ་པྲ་བྷསྟ་ཙཀྲ་ཧཱུཾ་ཕཊ། རྡོ་རྗེ་ཁྲག་འཐུང་གིའོ། །ཨོཾ་སུམྦྷ་ནི་སུམྦྷ་ཧཱུཾ། རྟ་མཆོག་གིའོ། །ཨོཾ་
བཛྲ་པུཥྤེ། ཨོཾ་བཛྲ་དྷཱུ་པེ། ཨོཾ་བཛྲ་དཱི་པེ། ཨོཾ་བཛྲ་གནྡྷེ། ཨོཾ་བཛྲ་རཱུ་པ། ཨོཾ་བཛྲ་ཤཔྟ། ཨོཾ་བཛྲ་གནྡྷེ། ཨོཾ་བཛྲ་རཱ་སེ། ཨོཾ་བཛྲ་སྤཪྴེ། དཀྱིལ་འཁོར་གྱི་མཚམས་རྣམས་དང་། སྒོའི་མཚམས་ཀྱི་ལེའུ་ཙེ་རྣམས་སུའོ། །རྡོ་རྗེ་ཤུགས་ཀྱིས་ཕྱུང་ལ། ཕྱི་རོལ་གྱི་རྡོ་རྗེ་རིགས་ཀྱི་དཀྱིལ་
འཁོར་དུ་ཚངས་པ་ལ་སོགས་ཨོཾ་དང་སྭཱ་ཧཱ་གཉིས་བར་དུ་མིང་དང་ལྡན་པར་དགོད་པར་བྱ་སྟེ། ཨོཾ་བྲཧྨ་ཎེ་སྭཱ་ཧཱ། ཨོཾ་བིཥྞུ་བེ་སྭཱ་ཧཱ། ཞེས་ཐམས་ཅད་ལ་སྦྱར་རོ། །དེ་ནས་དཀྱིལ་འཁོར་ཡང་དག་པར་མཆོད་ལ། ལྷའི་འཁོར་ལོ་ཕྱག་རྒྱ་བཞིས་རྒྱས་བཏབ་པར་བསྒོམས་ནས། དེའི་
གསང་སྔགས་བརྗོད་ལ་དཀྱིལ་འཁོར་ལན་བདུན་བསྐོར་བ་བྱ་སྟེ། རྡོ་རྗེ་དྲིལ་བུ་དཀྲོལ་ལ་མེ་ཏོག་སྔོན་པོའི་ཕྲེང་བ་བླངས་ཏེ། རྡོ་རྗེ་ཧཱུཾ་མཛད་ཀྱི་སྦྱོར་བས་བསྐོར་བ་ལན་གསུམ་བྱས་ལ་ཉེས་པ་ཞི་བར་བྱ་བའི་ཕྱིར་བལྟས་ལ། བཅོམ་ལྡན་འདས་ལ་ཕུལ་ཏེ། གར་བྱས་ནས་བླངས་
ལ་རྡོ་རྗེ་ཧཱུཾ་མཛད་ཉིད་ཀྱིས་རང་གི་མགོ་ལ་བཅིང་བར་བྱའོ། །དེའི་འོག་རྡོ་རྗེ་སློབ་དཔོན་མཉམ་པར་གཞག་པས་དབུས་སུ་འདུག་ལ་ཡིད་ཀྱིས་སྒོ་བཞི་དབྱེ་བར་བྱ་སྟེ། ཨོཾ་བཛྲ་ཨུདྒྷ་ཊ་ཡ་ས་མ་ཡ་པྲ་བེ་ཤ་ཡ་ཧཱུཾ། འདིའི་ཕྱག་རྒྱ་ནི་རྡོ་རྗེ་མཛུབ་མོ་ལེགས་པར་ནི། །སྦྱར་ལ་བཀན་ཏེ་
བསྡམས་ནས་སུ། །རབ་ཏུ་ཁྲོས་ནས་དབྱེ་བ་ནི། །སྒོ་འབྱེད་པ་ཡི་མཆོག་ཡིན་ནོ། །དེའི་འོག་ཏུ་ཕྱག་བཞི་སྔོན་དུ་འགྲོ་བའི་སྡོམ་པ་གཟུང་བར་བྱས་ལ། གོས་དམར་པོ་ལ་རྡོ་རྗེ་སེམས་དཔའ་དང་གཙུག་ཏོར་ཡོངས་སུ་བཟླས་པའི་ཟེའུ་ཀ་བཅིངས་ཏེ། སྒོ་སྲུང་གིས་གདོང་
