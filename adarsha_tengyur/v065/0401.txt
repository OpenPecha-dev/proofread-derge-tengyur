ལག་པ་གཡས་པར་བྱིན་ཏེ། ཆོས་ཀྱི་འཁོར་ལོ་དང་སྨྲ་བ་གཉིས་ཀྱི་ཆོས་ཀྱི་ཡི་གེ་དང་ལྷན་ཅིག་བྱིན་ལ་འདི་སྐད་ཅེས་བརྗོད་དེ། དེང་ནས་བརྩམས་ཏེ་སེམས་བསྐྱེད་པ་ཙམ་གྱིས་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་ཅིག །དེ་རིང་ཕན་ཆད་འཇིག་རྟེན་ལ། །སྐྱོབ་པ་རྣམས་ཀྱི་
འཁོར་ལོ་བསྐོར། །ཆོས་ཀྱི་དུང་ནི་བླ་མེད་པས། །ཀུན་དུ་མ་ལུས་གང་བར་གྱིས། །སོམ་ཉི་ཡིད་གཉིས་མི་བྱ་ཞིང་། །དོགས་པ་མེད་པའི་ཚུལ་གྱིས་སུ། །གསང་སྔགས་སྤྱོད་པ་ཚུལ་གྱི་མཆོག །འཇིག་རྟེན་འདི་ལ་རབ་ཏུ་སྒྲོགས། །དེ་ལྟར་བྱས་ན་སངས་རྒྱས་ལ། །ཕན་
འདོགས་དྲིན་དུ་གཟོ་ཞེས་བྱ། །རྡོ་རྗེ་འཛིན་པ་དེ་ཀུན་ཀྱང་། །ཀུན་དུ་ཁྱོད་ལ་སྲུང་བར་བྱེད། །རང་གི་མཚན་བརྒྱ་རྩ་བརྒྱད་པས་བསྟོད་ལ། དམ་ཚིག་གི་ཕྱག་རྒྱ་བསྟན་ནས། འདོད་ཆགས་བྲལ་བ་ལྟ་བུ་ཡི། །ཁམས་གསུམ་ན་ནི་སྡིག་མེད་དེ། །དེ་བས་འདོད་ལ་ཆགས་བྲལ་བ། །
ཁྱོད་ཀྱིས་ཀྱང་ནི་མ་བྱེད་ཅིག །གསང་བ་བསྒྲགས་ནས་ཐལ་མོ་དང་བཅས་པས་དང་པོར་འདི་སྐད་བརྗོད་དོ། །རྡོ་རྗེ་སེམས་དཔའ་སེམས་དཔའ་ཆེ། །རྡོ་རྗེ་མགོན་པོ་རབ་སྒྲུབ་མཛད། །རྡོ་རྗེ་དབང་བསྐུར་རྡོ་རྗེ་འོད། །རྡོ་རྗེ་དཔལ་ཁྱོད་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་བཞད་པ་ཆོས་ཆེན་པོ། །
དཔལ་ཀུན་གྱི་ནི་རྒྱལ་པོ་མཆོག །སྤྲོས་མེད་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་ལས་དང་རྡུང་བ་ཆེ། །གནོད་སྦྱིན་གཏུམ་པོ་གདོན་ཆེན་པོ། །རྡོ་རྗེ་ཁུ་ཚུར་ཕྱག་རྒྱ་ཆེ། །སངས་རྒྱས་ཀུན་ལ་ཕྱག་འཚལ་ལོ། །བྱང་ཆུབ་སེམས་དཔའ་བྱང་ཆུབ་ཆེ། །སངས་རྒྱས་ཐམས་ཅད་
དེ་བཞིན་གཤེགས། །རྡོ་རྗེ་ཡེ་ཤེས་ཡེ་ཤེས་ཆེ། །ཐེག་ཆེན་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །དོན་ཀུན་གྱི་ནི་དེ་ཉིད་དོན། །སེམས་ཅན་དོན་ཆེན་ཀུན་རིག་པ། །ཀུན་མཁྱེན་ཀུན་བྱས་ཐམས་ཅད་དེ། །ཀུན་གཟིགས་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་བདག་རབ་རྡོ་རྗེ་མཆོག །རྡོ་རྗེ་
བརྩོན་རབ་རྡོ་རྗེ་འཆང་། །དམ་ཚིག་ཆེན་པོའི་དེ་ཉིད་དོན། །སེམས་ཅན་ཁྱོད་ལ་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་ལྕགས་ཀྱུ་འདོད་པ་ཆེ། །རབ་ཏུ་དགའ་བ་འོད་ཆེན་པོ། །རྡོ་རྗེ་འོད་དང་འོད་ཆེན་པོ། །སངས་རྒྱས་འོད་ལ་ཕྱག་འཚལ་ལོ། །རྡོ་རྗེ་རྒྱལ་མཆོག་རྡོ་རྗེ་དགུ། །རིག་མཆོག་
