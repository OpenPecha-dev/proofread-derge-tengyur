གཟུང་ངོ་། །དེ་དག་གི་རྟེན་ཞེས་ཚིག་རྣམ་པར་སྦྱར་རོ། །དེ་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་ཀྱེ་དགེ་སློང་དག་འདི་ལ་ལས་ཡོད་དོ། །འབྲས་བུ་ཡོད་དོ། །གང་ཆོས་སུ་བརྡར་བཏགས་པ་མ་གཏོགས་པ་ཕུང་པོ་འདི་འདོར་ཞིང་ཕུང་པོ་གཞན་ཉེ་བར་ལེན་པའི་བྱེད་པ་པོ་ནི་མི་དམིགས་སོ། །དེ་ལ་ཆོས་སུ་བརྡར་བཏགས་
པ་ནི་འདི་ཡིན་ཏེ། འདི་ལྟ་སྟེ་འདི་ཡོད་ན་འདི་འབྱུང་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་སོ། །འདིས་ནི་ལས་དང་འབྲས་བུའི་འབྲེལ་པ་བརྟག་པ་བསྟན་ཏོ། །རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་འདི་ཡང་ཕུང་པོ་དང་། ཁམས་དང་། སྐྱེ་མཆེད་རྣམས་སུ་ཡང་བལྟ་བར་བྱ་སྟེ། དེ་དག་ཉིད་རྟེན་ནས་བྱུང་བ་ཡིན་པའི་ཕྱིར་རོ། །
འོན་ཏེ་རྫས་དང་། ཡོན་ཏན་དང་། ལས་ལ་སོགས་པ་ཡང་དག་པའི་དངོས་པོ་ཡང་ཡོད་པ་མ་ཡིན་ནམ། དེ་དག་ཅིའི་ཕྱིར་མ་ཡིན་ཞེ་ན། ཡོན་ཏན་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཡོན་ཏན་དང་། རྫས་དང་། བྱ་བ་དང་། རིགས་དང་། འདུ་བ་དང་ཞེས་ཟླས་དབྱེའོ། །རིགས་ཀྱི་སྒྲས་ཁྱབ་པ་དང་། ཉི་ཚེ་བའི་
སྤྱི་རྣམ་པ་གཉིས་གཟུང་ངོ་། །སོགས་ཀྱི་སྒྲས་བརྟགཔའི་རང་བཞིན་གྱི་ཁྱད་པར་རྣམས་ཏེ། གང་དག་གིས་ཆོས་ཅན་ལས་ཐ་དད་པའི་ཆོས་འགའ་ཞིག་བརྗོད་པ་དཔེར་ན་ཚིག་གི་དོན་དྲུག་པོའི་ཡོད་པ་ཉིད་དམིགས་པ་དང་བཅས་པའི་ཚད་མའི་ཡུལ་ཡིན་ནོ་ཞེས་བྱ་བ་དེ་ལྟ་བུ་ལ་སོགས་པ་དེ་དག་
གཟུང་ངོ་། །ཡོན་ཏན་ལ་སོགས་པ་ཡང་དེ་ཡིན་ལ་ཁྱད་པར་ཏེ་བྱེ་བྲག་ཀྱང་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་ཁྱད་པར་གྱི་བསྡུ་བའོ། །དེ་དག་གིས་སྟོང་པ་ནི་བྲལ་བ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ལ་འདི་ནི་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་ཏེ། བྲམ་ཟེ་ཐམས་ཅད་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་ཕུང་པོ་ལྔ་དང་། སྐྱེ་
མཆེད་བཅུ་གཉིས་དང་། ཁམས་བཅོ་བརྒྱད་ཀྱི་བར་དག་གོ་ཅེས་བྱ་བ་ཡིན་ནོ། །འདིས་ནི་ཚིག་གི་དོན་དྲུག་བརྟག་པ་བསྟན་ཏོ། །འོན་ཏེ་ཁྱད་པར་མེད་ན་ཇི་ལྟར་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་སྒྲ་དང་རྣམ་པར་རྟོག་པ་དག་གིས་ཡུལ་དུ་བྱེད་དེ།  དེ་དག་གི་ཡུལ་དུ་མ་བྱས་ན་བརྗོད་པར་མི་ནུས་
ཤིང་། ཁྱད་པར་མེད་ན་སྒྲ་དང་རྣམ་པར་རྟོག་པ་དག་འཇུག་པ་ཡང་མེད་ན་དེ་ཇི་ལྟར་བཅོམ་ལྡན་འདས་ཀྱིས་བསྟན་ཅེ་ན། སྒྲོ་བཏགས་པས་རྣམ་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། སྒྲོ་བཏགས་པ་ནི་ཕྱི་རོལ་ཉིད་དུ་སྒྲོ་བཏགས་པའི་རྣམ་པའི་རང་བཞིན་སྒྲ་དང་ཤེས་པའི་སྤྱོད་ཡུལ་གང་ལ་ཡོད་
