ཉིད་ཕྱིར། །རྒྱུ་སྐར་ལམ་གྱི་པདྨ་བཞིན། །སྐད་ཅིག་ཡིན་པར་ཐམས་ཅད་མེད། །གང་ཕྱིར་ནུས་པ་དངོས་མཚན་ཉིད། །ནུས་པའི་མཚན་ཉིད་དངོས་ཉིད་འདི། །ཅི་སྟེ་ཁས་ལེན་མི་བྱེད་ན། །གལ་ཏེ་དེ་ལྟར་གཞན་དག་གིས། །དངོས་པོའི་མཚན་ཉིད་
གཞན་སྟོན་གྱིས། །གལ་ཏེ་དངོས་པོའི་མཚན་ཉིད་ནི། །ཡོད་ཉིད་དང་འབྲེལ་འདོད་དེ་མིན། །འདུ་སོགས་མ་གྲུབ་ཕྱིར་དང་ནི། །ཇི་ལྟར་གཞན་ནི་གཞན་མཚན་ཉིད། །དོན་བྱེད་དོན་དུ་གཉེར་བྱེད་པ། །གཡེང་བ་མེད་པའི་བློ་ཅན་གྱི། །དོན་ལ་དཔྱོད་སོགས་བྱ་བ་ཀུན། །རིགས་ཀྱི་
སྨྱོན་པ་ཉིད་ཀྱིས་མིན། །མྱོས་པར་གྱུར་པས་ཀྱང་མ་ཡིན། །དེ་ལ་གང་དག་འགའ་ཞིག་ལ། །ལ་ལར་དུས་འགར་ཇི་ལྟར་ཡང་། །ཉེ་བར་མཁོ་གྱུར་དེ་ཉིད་ལ། །ཡོད་ཉིད་རྣམ་གཞག་རིག་པ་ཡིན། །དེ་ཡི་དངོས་ཅན་དོན་ཉིད་ལ། །སྐད་ཅིག་མ་ཉིད་རབ་སྒྲུབ་བྱེད། །ཐམས་ཅད་སྡུད་
ཕྱིར་ཁྱད་པར་ནི། །དེ་ཉིད་ལ་ནི་བརྗོད་པར་བྱ། །མོ་གཤམ་གྱི་ནི་བུ་ལྟ་བུའི། །ནུས་པ་མ་ལུས་སྟོང་པ་གང་། །ཐམས་ཅད་མཁྱེན་པའི་ཐུགས་ཀྱི་ཡང་། །ནམ་ཡང་རྒྱུ་ཉིད་མ་ཡིན་པ། །དེ་ལ་མཁས་པས་ཡོད་ཉིད་ཀྱི། །རྣམ་གཞག་འདི་ནི་ཡོད་མི་བྱ། །རྒྱུ་མེད་ཕྱིར་རོ་འབྲས་བུ་དང་། །ངོ་བོ་འབྲས་
སོགས་མ་མཐོང་ཕྱིར། །དེ་ལ་སྐད་ཅིག་མ་ཉིད་དུ། །བསྒྲུབས་པས་དོན་ནི་ཅུང་ཟད་མེད། །དེ་ལ་བརྒལ་ཞིང་བརྟག་པ་ཡང་། །བྱེད་པའང་འབྲས་བུ་མེད་པ་ཡིན། །འོན་ཏེ་ནམ་མཁའི་པདྨ་སོགས། །རང་གིས་ཤེས་པའི་རྒྱུར་གྱུར་ཕྱིར། །དོན་གྱི་བྱ་བ་བྱེད་ནུས་ཀྱང་། །དེ་དག་དངོས་པོར་
ཡོད་མ་ཡིན། །ནམ་མཁའི་ཆུ་སྐྱེས་པད་སོགས་ལ། །དེར་རྟོགས་བག་ཆགས་ཙམ་ལས་ནི། །འབྱུང་བ་དམིགས་པ་མེད་ཉིད་ནི། །ཤེས་པ་འདི་ནི་རབ་ཏུ་བྱེད། །མེད་པ་རྒྱུ་ཉིད་ཡིན་ན་ནི། །རྒྱུན་མི་འཆད་པར་འབྱུང་བར་འགྱུར། །རྟག་ཏུ་ཁྱད་པར་མེད་ཉིད་ཕྱིར། །དངོས་མེད་ལྟོས་པ་ཅི་
ཡང་མེད། །སྐད་ཅིག་པ་ཡི་དངོས་རྣམས་ཀྱང་། །ཇི་ལྟར་དོན་བྱེད་པ་ཉིད་ཡོད། །ཐོག་མའི་ལྷན་ཅིག་བྱེད་རྣམས་ནི། །ཕན་ཚུན་ཁྱད་པར་བྱེད་མིན་ཏེ། །སྐྱེས་ན་བདག་ཀུན་གྲུབ་ཕྱིར་དང་། །མ་སྐྱེས་རང་བཞིན་མེད་ཕྱིར་རོ། །ཁྱད་པར་མེད་ལས་ཁྱད་པར་ནི། །འབྱུང་ན་འབྲས་བུ་ཅིས་
