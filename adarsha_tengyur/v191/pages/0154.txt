གལ་ཏེ་རིག་བྱེད་བྱེད་པོ་དང་། །འདྲ་བ་དང་ལྟར་ཅུང་ཟད་སྣང་། །དེ་ཕྱིར་ཉེ་བར་འཇལ་བ་དེས། །བྱེད་པོ་ལ་ཡང་ཉེར་འཇལ་འགྱུར། །གལ་ཏེ་བྱེད་པོ་མེད་ཅུང་ཟད། །གཞལ་བྱས་གྲུབ་པར་མི་འགྱུར་ན། །དེས་ན་ངེས་པར་བྱེད་པོ་ནི། །དོན་གྱིས་གོ་བའི་གཞལ་བྱར་འགྱུར། །དེ་མེད་
ཀྱང་ནི་རིག་བྱེད་ལ། །མི་འཐད་ཅུང་ཟད་གང་ཡང་མེད། །དེ་ཡོད་ན་ནི་ཚད་མ་སོགས། །མང་པོ་གྲུབ་པར་མི་འགྱུར་རོ། །དེར་ནི་ལྔ་པོར་མ་ངེས་ཕྱིར། །དངོས་མེད་ཉིད་ཀྱིས་རྟོགས་པར་འགྱུར། །དེས་ན་དེ་ཡོད་རྙེད་དཀའ་སྟེ། །དངོས་མེད་ཚད་མས་གནོད་པའི་ཕྱིར། །ཚད་མ་མ་ཡིན་
ཉིད་ལྡོག་ཕྱིར། །རིག་བྱེད་སྐྱེས་བུས་མ་བྱས་ཉིད། །གང་འདོད་དེ་ཡང་དངོས་མིན་པས། །སྒྲུབ་པ་རྣམས་ཀྱི་བསྒྲུབ་བྱ་མིན། །རྟོག་གེ་རྣམས་ཀྱིས་གང་སྨྲས་ཞེས། །སྐྱེས་བུས་བྱས་པ་ཉིད་སྒྲུབ་པ། །དེ་ནི་སྤང་བར་བྱས་པ་མིན། །རིག་བྱེད་སྐྱེས་བུས་མ་བྱས་གྲུབ། །དངོས་
པོའི་ངོ་བོར་ཕྱོགས་གང་དག །གཙོ་བོ་དང་ནི་རྡུལ་ཕྲན་བཞིན། །དེ་དག་གང་རུང་མ་གྲུབ་ན། །ཅིག་ཤོས་རང་ཉིད་ཀྱིས་མི་གྲུབ། །གཙོ་བོ་ཡི་ནི་རྒྱུ་ཉིད་ཀྱི། །སྒྲུབ་པ་ཡང་ནི་སྤང་བྱས་ཀྱང་། །གཏན་ཚིགས་གཞན་མ་རྣམས་ཀྱིས་ནི། །རྡུལ་ཕྲན་གྱིས་བྱས་འགྲོ་བསྒྲུབ་བྱ། །
དངོས་པོའི་ཕྱོགས་ནི་བསྒྲུབ་དོན་དུ། །སྒྲུབ་པ་གང་ཞིག་བརྗོད་པ་མིན། །དེ་ནི་གཞན་དག་སྤང་བྱས་ན། །མེད་པ་རང་ཉིད་ཀྱིས་ནི་གྲུབ། །སྔ་ཕྱི་ཡི་ནི་མཐའ་དག་ལ། །གཞན་གྱིས་སྒྲུབ་པ་གང་བརྗོད་ཡིན། །དེ་ནི་སྤང་བར་བྱས་ནས་ནི། །རིག་བྱེད་སྨྲ་བའི་བྱ་བ་རྫོགས། །རྟག་
ཉིད་དངོས་པོའི་ངོ་བོ་གང་། །དེ་ནི་མ་བསྒྲུབས་ན་ཡང་ནི། །སྔ་མའི་ཕྱོགས་གཉིས་སུན་བཏོན་ཏེ། །རང་ཉིད་ཀྱིས་ཏེ་གྲུབ་པར་འགྱུར། །ཐོག་མའི་རིག་བྱེད་ཀྱིས་མཐའ་གང་། །སྐྱེས་བུས་བྱས་པའི་མཚན་ཉིད་ཅན། །ཕྱི་མཐའ་འཇིག་པའི་ངོ་བོ་ཡང་། །དེ་དག་མེད་པ་རྟག་པ་ཉིད། །ཐོག་
མར་རིག་བྱེད་གང་བྱས་མིན། །ཕྱིས་ཀྱང་འཇིག་པར་མི་འགྱུར་བ། །དེ་ཉིད་དེ་ཡི་རྟག་པ་ཉིད། །གལ་ཏེ་ཤེས་བྱ་དེར་ཡང་འདོད། །མ་བྱས་ཉིད་དང་འཇིག་མེད་ཀྱིས། །རྟག་པ་ཉིད་དུ་འདོད་པ་ཡིན། །དེ་དག་དངོས་མེད་བདག་ཉིད་ཀྱིས། །རང་གི་སྒྲུབ་པ་ལ་ལྟོས་མིན། །རིག་བྱེད་ངག་དོན་
