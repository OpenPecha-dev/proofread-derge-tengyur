ལས་ནི་ཡོད་པ་ཁོ་ན་ཡིན་ཏེ། ཅིའི་ཕྱིར་ཞེ་ན། ལས་ཀྱི་འབྲས་བུ་ལ་ལོངས་སྤྱོད་པའི་ཟ་བ་པོ་ཡོད་པའི་ཕྱིར་རོ། །འདི་ལ་བཅོམ་ལྡན་འདས་ཀྱིས་དེ་དང་དེ་དག་ཏུ། མ་རིག་བསྒྲིབས་པའི་སེམས་ཅན་དག །སྲེད་པའི་ཀུན་ཏུ་སྦྱོར་དང་ལྡན། །ཞེས་ཀྱང་གསུངས་ལ། གཞན་
ཡང་ཅི་སྟེ་ཁྱོད་རང་ཉིད་ཀྱིས་སྡིག་པའི་ལས་འདི་བྱས་ན། ཁྱོད་རང་ཉིད་ཀྱིས་དེའི་རྣམ་པར་སྨིན་པ་ཉམས་སུ་མྱོང་བར་བྱ་དགོས་སོ། །ཞེས་ཀྱང་གསུངས་པ། དེའི་ཕྱིར། མ་རིག་བསྒྲིབས་པའི་སྐྱེ་བོ་གང་། །སྲེད་ལྡན་དེ་ནི་ཟ་བ་པོ། །མ་རིག་པས་བསྒྲིབས་པའི་
སེམས་ཅན་སྲེད་པའི་ཀུན་ཏུ་སྦྱོར་བ་དང་ལྡན་པ་དང་ཞེས་གང་གསུངས་པ་དང་། ཁྱོད་རང་ཉིད་ཀྱིས་དེའི་རྣམ་པར་སྨིན་པ་ཉམས་སུ་མྱོང་བར་བྱ་དགོས་སོ་ཞེས་གང་གསུངས་པ་དེ་ནི་ལས་ཀྱི་འབྲས་བུ་ལ་ལོངས་སྤྱོད་པའི་ཟ་བ་པོ་ཡིན་ཏེ། དེའི་ཕྱིར་རེ་ཞིག་ལས་ཀྱི་
འབྲས་བུ་ལ་ལོངས་སྤྱོད་པའི་ཟ་བ་པོ་འཐད་པས་མེད་པའི་འབྲས་བུ་ཡང་མེད་ལ་བྱེད་པ་པོ་མེད་པའི་ལས་ཀྱང་མི་འཐད་པས། ཟ་བ་པོ་ཡོད་པའི་ཕྱིར་བྱེད་པ་པོ་དང་ལས་དག་ཀྱང་རབ་ཏུ་གྲུབ་པ་ཡིན་ནོ། །དེ་ཡང་བྱེད་ལས་གཞན་མིན་ཞིང་། །དེ་ཉིད་དེ་ཡང་མ་
ཡིན་ནོ། །ད་ནི་ལས་ཀྱི་འབྲས་བུ་ལ་ལོངས་སྤྱད་པའི་ཟ་བ་པོ་དེ་ཡང་ལས་དེའི་བྱེད་པ་པོ་ལས་གཞན་མ་ཡིན་ཏེ། རང་གི་རྣམ་པར་སྨིན་པ་ཉམས་སུ་མྱོང་བར་འགྱུར་རོ། །ཞེས་གསུངས་པའི་ཕྱིར་རོ། །གཞན་མ་ཡིན་པ་ཡང་མ་ཡིན་ཏེ། ཉེ་བར་ལེན་པ་གཞན་
ཡིན་པའི་ཕྱིར་རོ། །དེ་ལྟ་ན་འཁོར་བ་ཡང་འཐད་ལ་རྟག་པ་དང་ཆད་པའི་སྐྱོན་དུ་ཡང་ཐལ་བར་མི་འགྱུར་རོ། །བཤད་པ། ཅི་ཁྱོད་རྩིག་རྨང་མ་བྲེས་པར་པུ་ཤུ་འདོགས་པར་བྱེད་དམ། ཁྱོད་བྱེད་པ་པོ་དང་ལས་དང་འབྲས་བུ་དག་རབ་ཏུ་མ་གྲུབ་བཞིན་དུ་ལས་ཀྱི་
འབྲས་བུ་ལ་ལོངས་སྤྱོད་པའི་ཟ་བ་པོ་སྒྲུབ་པར་བྱེད་ཀོ། ། །འདི་ལྟར་བྱེད་པ་པོ་དང་ལས་དང་འབྲས་བུ་དག་ཉིད་མི་འཐད་ན་ཟ་བ་པོ་འཐད་པར་ག་ལ་འགྱུར། ཇི་ལྟར་ཞེ་ན། གང་གི་ཕྱིར་ན་ལས་འདི་ནི། །རྐྱེན་ལས་བྱུང་བ་མ་ཡིན་ཞིང་། །རྐྱེན་མིན་ལས་བྱུང་
