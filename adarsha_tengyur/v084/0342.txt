ཞེས་བྱ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོའི་སྤྲོས་པའོ། །དེ་བས་ན་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་སྤྱོད་པའི་ཚེ། གཟུགས་རྟག་པ་དང་མི་རྟག་པར་སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །བདེ་བ་དང་སྡུག་བསྔལ་ཞེས་བྱ་བ་དང་། བདག་དང་
བདག་མེད་ཅེས་བྱ་བ་དང་། ཞི་བ་དང་མ་ཞི་ཞེས་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་མ་ཡིན་ཞེས་བྱ་བར་སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །ཚོར་བ་དང་། འདུ་ཤེས་དང་། འདུ་བྱེད་རྣམས་དང་། རྣམ་པར་ཤེས་པ་རྟག་ཅེས་བྱ་བའམ་མི་རྟག་ཅེས་བྱ་བར་སྤྲོ་བ་འཕྲོ་བར་མི་
བྱེད་དོ། །བདེ་བ་དང་སྡུག་བསྔལ་ཞེས་བྱ་བ་དང་། བདག་དང་བདག་མེད་ཅེས་བྱ་བ་དང་། ཞི་བ་དང་མ་ཞི་ཞེས་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་མ་ཡིན་ཞེས་བྱ་བར་སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །སྐྱེ་མཆེད་དང་། ཁམས་དང་། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་རྟག་པ་དང་མི་
རྟག་ཅེས་བྱ་བ་དང་། བདེ་བ་དང་སྡུག་བསྔལ་ཞེས་བྱ་བ་དང་། བདག་དང་བདག་མེད་ཅེས་བྱ་བ་དང་། ཞི་བ་དང་མ་ཞི་ཞེས་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་མ་ཡིན་ཞེས་བྱ་བར་སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །ཕ་རོལ་ཏུ་ཕྱིན་པ་ཐམས་ཅད་དང་། སྟོང་པ་ཉིད་ཐམས་ཅད་
དང་། བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་སུམ་ཅུ་རྩ་བདུན་རྟག་པ་དང་མི་རྟག་པ་ཞེས་བྱ་བ་དང་། བདེ་བ་དང་སྡུག་བསྔལ་ཞེས་བྱ་བ་དང་། བདག་དང་བདག་མེད་ཅེས་བྱ་བ་དང་། ཞི་བ་དང་མ་ཞི་ཞེས་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་དང་། ཡོངས་སུ་ཤེས་པར་བྱ་བ་མ་ཡིན་ཞེས་བྱ་བར
སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །འཕགས་པའི་བདེན་པ་དང་། བསམ་གཏན་དང་། ཚད་མེད་པ་དང་། གཟུགས་མེད་པའི་སྙོམས་པར་འཇུག་པ་དང་། རྣམ་པར་ཐར་པ་བརྒྱད་དང་། མཐར་གྱིས་གནས་པའི་སྙོམས་པར་འཇུག་པ་དང་། སྟོང་པ་ཉིད་དང་། མཚན་མ་མེད་པ་དང་། སྨོན་པ་མེད་པ་
དང་། མངོན་པར་ཤེས་པ་དང་། ཏིང་ངེ་འཛིན་དང་། གཟུངས་ཀྱི་སྒོ་རྣམས་ལ་བསྒོམ་པར་བྱའོ་ཞེས་བྱ་བར་སྤྲོ་བ་འཕྲོ་བར་མི་བྱེད་དོ། །རྒྱུན་དུ་ཞུགས་པའི་འབྲས་བུ་དང་། ལན་ཅིག་ཕྱིར་འོང་བའི་འབྲས་བུ་དང་། ཕྱིར་མི་འོང་བའི་འབྲས་བུ་དང་། དགྲ་བཅོམ་པ་ཉིད་དང་། རང་བྱང་ཆུབ་ཤེས་
