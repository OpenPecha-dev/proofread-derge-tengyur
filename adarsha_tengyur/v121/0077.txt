ཆགས་དང་བྲལ་བ་གང་ལ་སྙིང་རྗེའི་ས་བོན་ཡོད་པ་དེ་ལ་སངས་རྒྱས་རྣམས་ཀྱིས་གཞན་གྱི་དོན་ལ་བསྐུལ་བས་ན། བྱང་ཆུབ་སེམས་དཔའི་སྤྱད་པ་མངོན་པར་ཤེས་པ་དང་། དབང་དང་། ཏིང་ངེ་འཛིན་རྣམས་ཡོད་པར་འགྱུར་རོ། །སྙིང་རྗེའི་བག་ཆགས་ཀྱིས་གླེང་བ་ན། ཆགས་

ཅད་དེ་བཞིན་གཤེགས་པའི་སྙིང་པོ་ཞེས་བྱ་སྟེ། སྣོད་མ་ཡིན་པའི་སེམས་ཅན་གང་ཡང་མེད་པས་བྱང་ཆུབ་ཏུ་སེམས་བསྐྱེད་པ་ཐམས་ཅད་ཀྱིས་ཐམས་ཅད་ཀྱང་སངས་རྒྱས་སུ་དགོད་པར་བྱ་བས་ན། དེ་ལྟར་སྒྲུབ་ཏུ་གསོལ་ཞེས་བློ་གྲོས་ཆེན་པོས་དྲི་བར་བསམ་པའོ། །སེམས་ཙམ་དུ་
ཡང་ཇི་ལྟར་རིག་པར་བྱ། ཕྱིའི་དོན་དུ་གདོན་མི་ཟ་བས་སེམས་ཅན་རྣམས་ཀྱི་ཐ་སྙད་དུ་འཇུག་སྟེ། ཇི་ལྟར་ཕྱིའི་དོན་མ་མཆིས་པར་དེ་ལྟར་རིག་པའི་ཁྱད་པར་གྱིས་བསྒྲུབ་པར་གསོལ། ཇི་ལྟར་དོན་སུན་འབྱིན་པར་འདི་ཉིད་དུ་གསུངས་པ། གཟུགས་ནི་རྡུལ་ཕྲན་གཞིགས་པ་ན། །
རྡུལ་ཕྲན་ཆ་ནི་ཡོད་པ་མིན། །སྣང་བ་འདི་ཅི་ཞེས་པ་དང་། །གསུངས་པ། ཕྱི་ཡི་དངོས་པོ་མེད་པར་ཡང་། །སེམས་ནི་རྣམ་པ་གཉིས་སུ་སྣང་། །ཞེས་པ་སྟེ་རབ་རིབ་དང་། མགོ་འཁོར་བ་དང་། བརྒྱལ་བ་ལ་སོགས་པའི་དངོས་པོ་ལྟར་རོ། །གང་གི་ཚེ་ན། ཤེས་རབ་མཉམ་
