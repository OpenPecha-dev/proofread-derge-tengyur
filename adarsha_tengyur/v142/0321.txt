ཡང་སྦྱར་བར་བྱའོ། །དེ་དག་ནི་སྐྱེ་བའི་སྐྱེ་ལ་སོགས་ཞེས་བྱ་བ་ནི་དེ་དག་སྐྱེ་བའི་སྐྱེ་ལ་སོགས་པ་ཡང་ཞེས་བྱ་བར་འདིར་ཡང་ཞེས་བྱ་བའི་སྒྲ་མི་མངོན་པར་བསྟན་པར་བལྟ་བར་བྱའོ། །རེ་རེ་ལ་ཡང་མཚན་ཉིད་བཞི་པ་ཞེས་བྱ་བ་རྒྱས་པར་འབྱུང་བ་ནི་མཚན་
ཉིད་རེ་རེ་ལ་མཚན་ཉིད་བཞི་པ་སྟེ། མཚན་ཉིད་བཞི་པོ་ཚན་གཅིག་ཏུ་བྱས་པའི་མཚན་ཉིད་བཞི་པར་འགྱུར་རོ། །མཐའ་ཐུག་པ་མེད་པའི་སྐྱོན་དུ་ཡང་ཞེས་བྱ་བ་ནི་མུ་ཐུག་པ་མེད་པའི་སྐྱོན་དུ་ཡང་སྟེ་སྐྱེ་བ་ལ་ཡང་མི་རྟག་པ་ཉིད་ཀྱི་བར་ཡོད་པ་དང་མི་རྟག་པ་ཉིད་ཀྱི་བར་ལ་
ཡང་ཡོད་པར་ཐལ་བར་འགྱུར་བའི་ཕྱིར་དང་མཐུན་པའི་མཚན་ཉིད་དེ་དག་ལ་ཡང་སྐྱེ་བ་གཞན་ལ་སོགས་པ་ཡོད་པར་ཐལ་བར་འགྱུར་བའི་ཕྱིར་རོ། །དེ་ཆོས་བརྒྱད་དང་གཅིག་ལ་འཇུག་ཅེས་བྱ་བ་ནི་སྐྱེ་བ་ལ་སོགས་པ་དང་སྐྱེ་བའི་སྐྱེ་བ་ལ་སོགས་པའི་ངོ་བོ་ཉིད་ཀྱི་ཆོས་དེ་དག་ནི་རིམ་
བཞིན་དུ་ཆོས་བརྒྱད་དང་གཅིག་ལ་འཇུག་སྟེ་སྐྱེ་བ་ལ་སོགས་པ་ནི་ཆོས་བརྒྱད་ལ་འཇུག་གོ། །སྐྱེ་བའི་སྐྱེ་བ་ལ་སོགས་པ་ནི་ཆོས་གཅིག་ལའོ། །བདག་ཉིད་རྟོགས་པར་དགུ་ཞེས་བྱ་བ་ནི་མཚན་ཉིད་དང་མཐུན་པའི་མཚན་ཉིད་དག་ལ་ལྟོས་ནས་དེ་སྐད་བཤད་ཀྱི། དེ་ལ་
ཅི་རིགས་པར་ཐོབ་པ་ལ་སོགས་པ་དང་འབྱུང་བ་དག་དང་འབྱུང་བ་ལས་གྱུར་པ་དག་མི་འབྱུང་བ་ནི་མ་ཡིན་ཏེ། འདོད་ན་དབང་པོ་མེད་པ་དང་། །སྒྲ་མེད་ཕྲ་རབ་རྡུལ་རྫས་བརྒྱད། །ཅེས་བྱ་བ་དང་། ཐམས་ཅད་འདུས་བྱས་མཚན་ཉིད་དང་མཚན་ཉིད་ཐོབ་པའམ་ཞེས་བྱ་བ་ལ་
སོགས་པ་འབྱུང་བའི་ཕྱིར་རོ། །སྐྱེ་བས་ནི་བདག་ཉིད་མ་གཏོགས་པར་ཞེས་བྱ་བ་རྒྱས་པར་འབྱུང་བ་ནི་རང་གི་བདག་ཉིད་ལ་བྱེད་པ་འགལ་བས་དེའི་ཕྱིར་བདག་ཉིད་མ་གཏོགས་པར་ཆོས་བརྒྱད་སྐྱེད་པར་བྱེད་དོ། །བརྒྱད་གང་ཞེ་ན་གཟུགས་སམ་སེམས་ཀྱི་ཆོས་དེ་དང་གནས་པ་
དང་། རྒ་བ་དང་། མི་རྟག་པ་ཉིད་དང་སྐྱེ་བའི་སྐྱེ་བ་དང་། གནས་པའི་གནས་པ་དང་། རྒ་བའི་རྒ་བ་དང་། མི་རྟག་པ་ཉིད་ཀྱིས་མི་རྟག་པ་ཉིད་སྐྱེད་པར་བྱེད་དོ། །སྐྱེ་བའི་སྐྱེ་བས་ནི་དེ་ཁོ་ན་སྐྱེད་པར་བྱེད་དོ། །དེ་བཞིན་དུ་རྒ་བ་དང་མི་རྟག་པ་ཉིད་དག་ལ་ཡང་སྦྱར་བར་བྱའོ་ཞེས་
