སྤྱན་དྲངས་ཏེ། །ཨོཾ་བ་པ་ཀ་ཨགྣ་ལེ་ཨཱ་གཙྪ་ཡ་ཨཱ་གཙྪ་ཡ། མ་མ་ཤཱནྟིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ལྕགས་ཀྱུའི་ཕྱག་རྒྱ་བཅིངས་ཏེ་སྤྱན་དྲང་ངོ་། །བགེགས་བསྐྲད་བསྟིམས་ཏེ་མཆོད་པ་དབུལ། །རྩ་སྔགས་ཉིད་ལ་མཆོད་སྔགས་བཏགས་ཏེ་དབུལ། །ཨོཾ་བ་པ་ཀ་ཨགྣ་ཡེ།  ཆེ་གེ་
མོ་ཤཱནྟིཾ་ཀུ་རུ་སྭཱ་ཧཱ། ཞེས་སྲེག་རྫས་དབུལ་བར་བྱའོ། །དེ་ཡི་མན་ངག་འོད་ཟེར་ནི། །སུམ་སྐོར་དུ་ནི་བྱ་བ་དང་། །ཉིས་སྐོར་དུ་ནི་བྱ་བ་དང་། །བདག་ཉིད་རབ་ཏུ་ཤེས་པར་བྱ། །བཞི་སྐོར་དག་དང་སློབ་མའོ། །འོད་ཟེར་དཀར་པོ་འཕྲོས་པ་ལས། །ཐུགས་དང་སྙིང་ག་བརྒྱུད་པ་
དང་། །གསུམ་ནས་ངག་ལ་བརྒྱུད་པ་དང་། །དབུ་ནས་མགོ་ལ་བརྒྱུད་པ་ཡི། །ལུས་ངག་ཡིད་ཀྱི་སྒྲིབ་པ་སྦྱང་། །རྩ་བ་གསུམ་འདུས་བསྐང་བར་བྱ། །སྤྱི་བོའི་གཙུག་ནས་བརྩམས་ནས་ནི། །སྡིག་དང་སྲིན་བུ་ཟད་པའི་ཕྱིར། །དམར་པོས་བསྲེག་ཅིང་དཀར་པོས་བཀྲུ། །སྒོ་དགུ་
དག་ནས་བསྡུ་བ་དང་། །ཐམས་ཅད་ཀུན་ནས་བསྡུ་བ་དང་། །སྙིང་གར་བསྡུས་ནས་དབྱུང་བ་དང་། །ཏིལ་གྱིས་གཟུགས་བརྙན་བྱས་ནས་ནི། །སྡིག་པའི་གཟུགས་ལ་དེ་གཞུག་བྱ། །རྫས་རྣམས་བདུད་རྩིས་མཆོད་པས་ནི། །དེ་ནས་སྡིག་པ་དེ་བསྲེག་བྱ། །ཡང་ན་རྫས་ལ་བསྟིམས་ཏེ་
བསྲེག །ནད་རྣམས་ཞི་བར་བྱེད་པ་ནི། །ནད་ཀྱི་མིང་ནས་སྨོས་ནས་ནི། །གོང་མ་དེ་བཞིན་བསྲེག་པའམ། །ནད་ཀྱི་མིང་དང་གཟུགས་བྲིས་ལ། །འོད་ཟེར་དམར་པོས་བསྲེགས་ནས་ནི། །དཀར་པོས་སྦྱང་ཞིང་ཞི་བསམས་ལ། །འབྱུང་བའི་དགྲ་རྣམས་དེ་བཞིན་བསམ། །འབྱུང་
པོ་ཀུན་དང་དགྲ་ཀུན་ཡང་། །འོད་ཟེར་བཀྱེ་བས་ཞི་བར་བསམ། །གཟུགས་དང་མིང་བྲིས་ཞི་བར་བྱ། །སྲེག་རྫས་བདུད་རྩི་འོད་ཀྱིས་མཆོད། །མཆོད་པས་མཉེས་པས་གོང་བཞིན་བསྐོར། །དེ་ལྟ་མིན་ན་འོག་བཞིན་ནོ། །མན་ངག་དེ་ལ་ནན་ཏན་བྱ། །མར་དང་དཀར་གསུམ་
