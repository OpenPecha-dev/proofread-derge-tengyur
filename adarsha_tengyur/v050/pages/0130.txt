རྒྱས་འགྲོ་དོན་བྱས་དོན་གསོལ། །བྱང་ཆུབ་སེམས་དང་གསང་སྔགས་ལྷ། །འཇིག་རྟེན་སྐྱོང་དང་མཐུ་ཆེན་དང་། །ཡང་དག་རྡོ་རྗེའི་སྤྱན་མངའ་བའི། །སེམས་ཅན་བསྟན་ལ་དགའ་རྣམས་ལ། །བདག་མིང་འདི་ཞེས་བྱ་བ་ཡིས། །རྡོ་རྗེ་གཏུམ་པོའི་དཀྱིལ་འཁོར་ནི། །ཡོ་བྱད་ཅི་
འབྱོར་བས་བྲི་ན། །དེང་བདག་སློབ་མར་བཅས་པ་ལ། །ཐུགས་ནི་བརྩེ་བར་དགོངས་ནས་སུ། །ཚོགས་པ་ཐམས་ཅད་དཀྱིལ་འཁོར་དུ། །བྱིན་གྱིས་བརླབ་པ་མཛད་དུ་གསོལ། །དེ་ལྟར་ལན་གསུམ་གསོལ་བཏབ་ནས། །ར་བ་དྲ་བས་མཚམས་བཅད་དོ། །ཧཱུཾ་ཉོན་ཅིག་བགེགས་
དང་ལོག་འདྲེན་ཚོགས། །ང་ནི་བཅོམ་ལྡན་གཏུམ་པོ་ཡིན། །ང་ཡི་བཀའ་ལས་མ་འདའ་ཞིག །ང་ཡི་བཀའ་ལས་འདའ་བྱེད་ན། །རྡོ་རྗེ་ཁྲོ་བོས་མགོ་འགེམས་སོ། །མ་འདུག་སོང་ཤིག་ལོག་འདྲེན་ཚོགས། །ཨོཾ་སུ་བྷ་ནི་སུཾ་བྷ་ཧཱུཾ། གྲྀཧྞ་གྲྀཧྞ་ཧཱུཾ། གྲྀཧྞཱ་པ་ཡ་གྲྀཧྞཱ་པ་ཡ་ཧཱུཾ། ཨཱ་ན་ཡ་

གྲོ་དང་སོ་བ་འབྲས། །ཏིལ་དང་ལན་ཚྭ་བུ་རམ་དང་། །མར་དང་སྦྲང་རྩི་རྣམ་པ་ལྔ། །གསེར་དངུལ་ཟངས་ལྕགས་མུ་ཏིག་དང་། །བྱཱ་གྷྲ་སིང་ཧ་ཧ་ས་དེ་བ་དང་། །ཀཎྜ་ཀ་རི་ནི་དྲ་སྨན་ལྔ་དང་། །དྲི་ཆེན་དྲི་ཆུ་དམར་ཟིལ་མདངས། །མི་རྟ་ཁྱི་བྱ་གླང་པོའི་ཤ །དམར་སེར་འོ་མ་གཞོན་ནུའི་ཆུས། །
ཕྱེད་མཐོ་ཙམ་དུ་དགང་བར་བྱ། །ཁ་ཚར་མ་ཉམས་རས་ཀྱིས་བཅིངས། །འབྲས་བུ་ཅན་གྱིས་ཁ་བརྒྱན་ཏེ། །དྲིས་བདུག་མཎྜལ་ཁར་བཞག་ལ། །མི་དམིགས་བྷྲཱུཾ་བཾ་བདུད་རྩི་ཡི། །རྒྱ་མཚོའི་དབུས་སུ་རི་རབ་ཆེ། །སྤྲིན་སྒྲ་བསྣོལ་བས་སྣ་ཚོགས་སྒྲ། །ལྟེ་བར་བྷྲཱུཾ་ལས་གཞལ་ཡས་
ཁང་། །བཾ་ལས་ཆུ་སྐྱེས་དམ་ཚིག་བསྐྱེད། །ཡེ་ཤེས་འཁོར་ལོ་སྤྱན་དྲངས་ལ། །ཕྱག་རྒྱ་བཞི་ཡིས་གཉིས་མེད་བསྟིམ། །སྐུ་གསུང་ཐུགས་སུ་བྱིན་གྱིས་བརླབ། །སྤྱན་ལ་སོགས་པས་དབང་བསྐུར་རོ། །མཆོད་དང་གཏོར་མས་བསྟོད་བྱས་ལ། །གཟུངས་ཐག་ལྡན་པས་
