བ་ལང་སྦྱོར་ན་གནོད་སྦྱིན་རི། །ཞེས་བྱ་བ་ནི་ནུབ་ཕྱོགས་བ་ལང་སྤྱོད་ཀྱི་རི་མཚམས་ན། གནོད་སྦྱིན་ཁ་དོག་ནག་པ། ཡུལ་དེའི་སྲུང་མར་གྱུར་པ། གཡས་གཡོན་གཉིས་ན་བུམ་པ་བསྣམས་པའོ། །གྲོང་ཁྱེར་ན་ནི་སྡེ་བཟང་པོ། །ཞེས་བྱ་བ་ནི་མ་ག་དྷཱའི་གྲོང་ཁྱེར་ཡངས་པ་ཅན་ན། 
འཁོར་གཡོག་མང་པོ་ཡོད་པ། སྟོབས་དང་ལྡན་པ། ཁ་དོག་སེར་བ། གཡས་ན་རིན་པོ་ཆེ་བསྣམས་པ། གཡོན་བུམ་པ་བསྣམས་པའོ། །གནས་དང་བཅས་ན་ལག་པ་དཔའ། །ཞེས་བྱ་བ་ནི་ཆུ་བོ་གངྒཱའི་འགྲམ་ཤར་ཕྱོགས་ན་རི་བོ་ལ་གནས་པ། ཝ་ཀིམ་ཞེས་བྱ་བར་
གྲགས་པའོ། །དེའི་སྲུང་མར་གྱུར་ཅིང་སྟོབས་དང་ལྡན་པ། ཁ་དོག་དམར་པོ་ཕྱག་ན་དབྱུག་ཏོ་བསྣམས་པའོ། །ཁ་ལ་སྦྱིན་ན་བདེ་བྱེད་པ། །ཞེས་བྱ་བ་ནི་ཆུ་བོ་གངྒཱའི་འགྲམ་ན། མོ་འགེ་རི་ཞེས་བྱ་བའི་ཡུལ་ན། གནོད་སྦྱིན་མདོག་ནག་ལ་བེ་ཅོན་འཛིན་པའོ། །ཀཽ་ཤཱམྦཱི་ན་ཚེགས་མེད་
པ། །ཞེས་བྱ་བ་ནི་རྒྱ་གར་ཡུལ་གྱི་སོ་མཚམས་ན། གནོད་སྦྱིན་ཁ་དོག་ལྗང་ནག་གོ། །གཡས་ན་ཏྲི་ཤཱུ་ལ་གཡོན་ཁ་ཊྭཱཾ་ག་བསྣམས་པའོ། །བཟང་པོ་ཅན་ན་བཟང་པོ་པ། །ཞེས་བྱ་བ་ནི་རི་རབ་ཀྱི་བྱང་ཕྱོགས་སྒྲ་མི་སྙན་ན། གནོད་སྦྱིན་ཁ་དོག་སེར་པོ་གཡས་རལ་གྲི་གཡོན་
བུམ་པ་བསྣམས་པའོ། །སྐྱ་སྣར་ཅན་གྱི་བུ་ན་ནི། །གནོད་སྦྱིན་འབྱུང་པོའི་གདོང་ཞེས་བྱ། །ཞེས་བྱ་བ་ནི་འཛམ་བུ་གླིང་གི་དབུས་ན་མེ་ཏོག་གི་ར་བ་ན། གནོད་སྦྱིན་མེ་ཏོག་ཅན་ཡོད་པའོ། །སྐུ་མདོག་སེར་པོ་ཕྱག་ན་ཨུཏྤ་ལ་སྔོན་པོ་བསྣམས་པའོ། །མཆིང་བུ་ན་ནི་མྱ་ངན་མེད། །ཅེས་བྱ་
བ་ནི་དེའི་ཤར་ཕྱོགས་ན་མེ་ཏོག་ཨ་ཤོ་ཀའི་ར་བ་ལ་གནོད་སྦྱིན་ཁ་དོག་ནག་པོ། ཕྱག་ན་རྡོ་རྗེ་འཛིན་པའོ། །མ་ལ་གནས་ན་ཤ་ཐང་བ། །ཞེས་བྱ་བ་ནི་རི་མ་ལ་ཡའི་ལྷོ་ཕྱོགས་ཙནྡན་གྱི་ནགས་གསེབ་ན་སྐུ་མདོག་དམར་སྐྱར་གནས་པ། ཕྱག་ན་རྩེ་གཅིག་པ་བསྣམས་པའོ། །ཚང་ཚིང་གཅིག་
ན་དོན་གྲུབ་པ། །ཞེས་བྱ་བ་ནི། བལ་ཡུལ་གྱི་ལྷོ་ཕྱོགས་ཀོ་བ་ཡ་ག་ན། གནོད་སྦྱིན་གླང་པོ་ཆེའི་མགོ་ཅན་ཕྱག་བཞི་པ་སྐུ་མདོག་དཀར་པོ། གཡས་གཉིས་ན་དགྲ་སྟ་དང་ཞགས་པ། གཡོན་གཉིས་ན་ཏྲི་ཤྲཱུ་ལ་དང་ལྕགས་ཀྱུའོ། །མ་རྒྱལ་རྒྱལ་ན་དམན་པ་པོ། །ཞེས་བྱ་བ་ནི་དུར་
