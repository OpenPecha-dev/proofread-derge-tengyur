གོ། །ཡཱུ་པཿ། ཀུ་པཿ། སཱུ་པཿ། བཱསྤ་ལ་སོགས་པ་རྣམས་སོ། །བཱ་སྲྀ་སྒྲ་ལའོ། །ཤུ་ཥཱ་སྐེམ་པ་ལའོ། །ཤ་ལ་འགྲོ་བ་ལའོ། །རུ་སྒྲ་ལའོ། །ཤྲཱྀ་འཚེ་བ་ལའོ། །ཥ་ནུ་ངེས་པར་འཇོམས་པ་ལའོ། །ཎི་ཐོབ་པ་ལའོ། །འདི་རྣམས་ལས་པ་ཕ་རོལ་དང་ཚིག་ཕྲད་མཐའོ། །དེ་ལ་བཱ་སའི་ཥ་ཉིད་
དོ། །[མཆོད་སྡོང་། ཁྲོན་པ།] །བཱཥྤ། [ཚོད་མ།] ཤུ་ཥའི་ཨུ་ཡིག་གི་ཨ་ཉིད་དོ། །ཤཥྤཾ། ཤ་ལིའི་ཨཏ་ཀྱི་ཨི་ཉིད་དོ། །ཤིལྤཾ། རཽ་ཏིའི་རིང་པོ་ཉིད་དོ། །རཱུ་པཾ། ཤྲྀ་ཎོ་ཏིའི་ཨཱུ་ཉིད་དོ། །ཤུརྤཿ། པྲ་སྥོ་ཊ་ནི། ཁ་ན་འི་ན་ཡིག་གི་ཥ་ཉིད་དོ། །ཁཥྤཾ། བ་ལཱཏྐཱ་རཾ། ནཱི་པཾ། ཀུ་ཊ་ཛཿ། སརྟི་ལས་
ཨཔ་དང་སུཀ་ཡང་ངོ་། །[རླངས་པ།] སྲྀ་འགྲོ་བ་ལའོ། །འདི་ལས་ཨཔ་ [རྩ་གཞོན་ནུ།] ཕ་རོལ་དང་སུཀ་ཡང་ངོ་། །[རྣོ་ཤེས།] ཨུ་ཡིག་བརྗོད་དོན། ཀ་ [གཟུགས།] ཡིག་ཀི་ཏ་བྱ་བའི་དོན། [སློབ་མ། རབ་ཏུ་འཁྲབ་བྱེད།] བྱ་བའི་དོན་གྱི་ཨི་ཀའི་ཨེང་ཞེས་པས་ཨ་ཉིད་དོ། །ར་ཕ་ [དུག་མོ་ཉུང་།] རོལ་

ཨེང་མེད། ཀུ་ཎ་པཿ། ཨུ་ཥ་པཿ། ཀུ་ཊ་པཿ། ད་ལ་པཿ། བྲ་ཧ་ར་ཎཿ། ཀ་ཙ་པཿ། བི་ཥི་ལས་ཊུ་ཀའོ། །བེཥྚ་བཿ། མཎྜ་བཿ། བི་ཤི་ལས་ཕ་རོལ་གྱི་ཨའི་ཨི་ཡིག་གོ། །བི་ཤི་ [འཐིབས་པོ། ཁྱིམ།] པཿ། །རཱ་ཏི་ལས་ཨི་ཕའོ། །རཱ་ལཱ་ལེན་པ་ [ཀུན་དུ་གདུང་བ།] ལའོ། །འདི་
ལས་ཨི་ཕ་ [ཤི་བའི་རོ།] རོལ་ [མེ་] ཏུ་འགྱུར་ [ཟེའི་བཞི་ཆ།] རོ། །ཨཏ་ཨེང་ཞེས་པས་ [ཡལ་ག །གཉིས་དག་གི་གཅིག་ཉིད་ [དགེ་བ། ཁྱིམ་མམ་འདུན་ཁང་ངམ་ལྷ་ཁང་།] ཨེའོ། །རེ་ཕཿ། [ར་ཡིག །རབ་ཏུ་འཇུག] །གུ་རི་ལས་ཕ་ཀའོ། །གུ་རཱི་བརྩོན་པ་ལའོ། །འདི་ལས་ཕ་ཀ་ཕ་རོལ་ཉིད། ཀ་ཡིག་ཨཏ་ཨེང་
