ཧཱ་དེ་བ་བི་ཤེ་ཏི། ཧ་ལནྟ་ཉིད་ཀྱིས་མི་ཐོབ་པ་ལའོ། །ཛྱེཥྛཱ། ཀ་ནིཥྛཱ། མ་དྷྱཱ། མ་དྷྱཱ་མཱ་ཞེས་སོ། །གང་གི་ཚེ་སོར་མོ་རྣམས་ལ་འཇུག་པ་དེའི་ཚེ་ཨ་ཏ་ལས་ཞེས་པ་ཉིད་ཀྱིས་ཌཱ་པ་འགྲུབ་བོ། །གང་གི་ཚེ་བྷ་གི་ནཱི་ལ་སོགས་པ་རྣམས་ལ་ཡིན་པ་དེའི་ཚེ་ཡང་རྣམ་པར་གཞག་པ་བརྗོད་པའི་ཕྱིར་རོ། །
མཐར་ཡང་བ་ཡས་དང་ཀ་ནིཥྛཱ་སོགས་དམིགས་བསལ་བསྟན་པའི་ཕྱིར་རོ། །ན་ཚོད་ཀྱི་མཐའ་མ་ཡིན་པའི་མཚན་ཉིད་ལས་ངཱིཔའོ། །མཐའ་མ་ཡིན་པའི་ན་ཚོད་སོ་སོར་མི་རྟོགས་སོ། །གང་གི་ཚེ་ཛྱེཥྛ་སྱ་བྷཱཪྻཱ་ཡིན་པ་དེའི་ཚེ་སྐྱེས་པ་སྦྱོར་བའི་མཚན་ཉིད་ལས་ངཱིཔ་ཐོབ་པར་བཀླགས་སོ། །མཱུ་ལ་ལས་ནཉ་ཞེས་
པ་ཡང་བརྗོད་པར་མི་བྱ་སྟེ། ཨ་མུ་ལ་དང་མཱུ་ལའི་མཐའི་མཚན་ཉིད་ལས་ངཱིཔ་མེད་དོ། །ཨཱཛ་ཏི་ཉིད་ལས་འདིར་ཌཱཔ་ཅིའི་ཕྱིར་མི་འགྱུར་ཞེ་ན། པཉྩཱ་ཛཱི། ད་ཤཱ་ཛཱི་ཞེས་པ་གཉིས་པའི་དགག་པ་ལས་ངཱིཔ་བཀག་པ་མེད་པའི་ཕྱིར་རོ། །ཏ་ཕ་རོལ་ཏུ་བྱེད་པ་ཅི་ཞེ་ན། ཤུ་བྷ་མ་ཡཱ། ཨཏ་ལས་ཞེས་པ་འདི་ནི། རིགས་
ལས་བུད་མེད་མ་ཡིན་པའི་ངེས་པ་སོགས་ལས་མཐའ་ཉེ་ལས་ཞེས་པའི་བར་དུ་སྤྱིར་བཏང་བའོ། །ཡང་ལས་སོ། །ཨཎ་ཨིཉ་དག་ཡང་དུ་བསྒྱུར་བ་བཟུང་བའི་ཡངའི་མཐའ་ལས་བུད་མེད་ལ་ཊྱབ་འགྱུར་ཏེ། ཀཱ་རཱི་ཥ་གནྡྷྱ། ཀཽ་མུ་ད་གནྡྷྱ། ཨཽཏྤ་ལ་གནྡྷྱ། བཱ་རཱ་ཧྱཱ། ཨཎ། ཨིཉའི་མཐའ་ཉིད་ལས་ངཱིཔ་ཐོབ་པ་ལ་
བརྗོད་པའོ། །རང་དོན་ལའོ། །ཨིཏ་ལས་ཕ་རོལ་བུད་མེད་ལ་གང་བརྗོད་པ་དེ་རང་དོན་ལ་འཇུག་པའི་ཨངྒ་ལས་རིག་པར་བྱའོ། །འདི་ནི་ཨཏ་རྗེས་སུ་འཇུག་པའི་མཐར་ཐུག་པའི་བར་ལ་སྤྱིར་བཏང་བའོ། །ཊིཏ། ཌྷ་ཨཎ། ཨཧ། ཋཀ །ཋཉ། ནཉ། སྣ་ཉ་ལས་སོ། །ཊིཏ་ལ་སོགས་པའི་སྒྲ་གང་ཡིན་པ་དེའི་མཐའ་
ལས་བུད་མེད་ལ་ངཱིཔ་འགྱུར་ཏེ། མ་དྲ་ཙ་ར་ཀུ་རུ་ཙཱ་རཱི། འདིར་བསླགས་པ་ལ་ཞེས་པས་ངཱིཔ་མི་འགྱུར་རོ། །ཨིཊའི་ཊ་ནི་སྒྲུབ་པ་བསྟན་པ་ལ་སྤྱོད་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ། པ་ཙ་མཱ་ན་ཞེས་སོ། །ལཊའི་ལ་ཡིག་ནི་སྒྲུབ་པ་ལ་སྤྱོད་པའི་དོན་གྱི་ཕྱིར་རོ། །ཌྷ་ཞེས་པ་ནི་ཌྷཀ་དང་ཌྷཉ་དག་གཟུང་ངོ་། །སཽ་པརྞེ་ཡཱི། སཱ་པ་ལེ་ཡཱི། 
ཨཎ་ཞེས་པ་ནི་ཀྲྀཏ་དང་དེ་ཕན་དག་གཟུང་ངོ་། །ཀུམྦྷ་ཀཱ་རཱི། ཨཽ་པ་ག་བཱི། ཨཽཏྶཱི། ཋཀ །ཨཱཀྵི་ཀཱི་ཋཀ །ཨཱཀྵི་ཀཱི། ཋཉ་ལཱ་བ་ཎི་ཀཱི། ཋཀ་ཋཉ་དག་བྱེ་བྲག་ཏུ་ཉེ་བར་ལེན་པ་ནི་ཋན་ལྡོག་པའི་དོན་ཏོ། །ནཉ། སྟྲེེ་ཎཱི། སྣཉ། པཽཾསྣཱི་རང་དོན་ཞེས་པ་ཉིད་དེ། པྲི་ཡ་ཀུ་རུ་ཙ་རཱ་ན་ག་རི་ཊི་ཏ་ལ་སོགས་པ་རྣམས་དང་
