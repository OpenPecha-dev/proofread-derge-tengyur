གཅད་པའོ། །ཁཊྭཱ། ཁཊྭཱེ། ཁ་ཊྭཱཿ། ཁ་ཊྭཱཾ། ཁ་ཊྭཱེ། ཁ་ཊྭཱཿ། ཁ་ཊྭཱ་ཡཱ། ཁ་ཊྭཱེ་ན། ཁ་ཊྭཱ་བྷྱཱཾ། ཁ་ཊྭཱ་བྷིཿ། ཁ་ཊྭཱ་ཡཻ། ཁ་ཊྭཱ་བྷྱཱཾ། ཁ་ཊྭཱ་བྷྱཿ། ཁ་ཊྭཱ་ཡཱཿ། ཁ་ཊྭཱ་བྷྱཱཾ། ཁ་ཊྭཱ་བྷྱཿ། ཁ་ཊྭཱ་ཡཱཿ། ཁ་ཊྭཱ་ཡོཿ། ཁ་ཊྭཱ་ནཱཾ། ཁ་ཊྭཱ་ཡཱཾ། ཁ་ཊྭཱ་ཡོཿ། ཁ་ཊྭཱ་སུ། ཧེ་ཁ་ཊྭཱེ། ཧེ་ཁ་ཊྭཱེ། ཧེ་ཁ་ཊྭཱཿ། དེ་ལྟར། ཤཱ་ལཱ།  
ཤཱ་ལེ། ཤཱ་ལཱཿ། ཀུ་མཱ་རི་ཞེས་པ་འདིའི། ངི་ཨཱཔའི་རིང་པོ་ལས་སོ་ཞེས་པ་སུ་དབྱིའོ། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ་ཞེས་པ་ཡཎ་རུ་བསྒྱུར་བའོ། །ཨམ་ལས་སྔ་མའོ། །དང་པོ་དག་གི་ཨཙ་ལའོ་ཞེས་པའོ། །ཨཱི་ཨཱུ་དག་ལས་ཀྱང་ཨཊའོ་ཞེས་པ་ཕ་རོལ་ཏུ་ང་ཡིག་ཕྱིས་པ་རྣམས་ལ་སྔོན་དུ་ཨཊའོ། །ཨེཙ་ལ་ཞེས་པ་ཨཻ་
ཉིད་དོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །བུད་མེད་ཨཱི་ཨཱུ་དག་ལས་སོ་ཞེས་པ་ནུཊའོ། །ར་ཥ་ལས་ནའི་ཎ་ཚིག་གཅིག་ལའོ། །ངིའི་ཨཱམ་དེ་ལའོ་ཞེས་པ་ངིའི་ཨཱམ་དུ་འགྱུར་རོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་པ་ཁ་ཉིད་དོ། །ངི་ཨུངའིའོ། །ཀུན་དུ་བོད་པ་ལའོ་ཞེས་པ་ཐུང་ངུ་ཉིད་དོ། །ཨེང་
ཐུང་ངུ་ལས་ཏ་མ་ཡིན་པའི་ཀུན་དུ་བོད་པ་ལའོ། །རྣམ་པར་གཅད་པའོ། །ཀུ་མཱ་རཱི། ཀུ་མཱཪྻཽ། ཀུ་མཱཪྻཿ། ཀུ་མཱ་རཱ། ཀུ་མཱཪྻཽ། ཀུ་མཱ་རཱིཿ། ཀུ་མཱཪྻཱ། ཀུ་མཱ་རི་བྷྱཱཾ། ཀུ་མཱ་རཱི་བྷིཿ། ཀུ་མཱཪྻཻ། ཀུ་མཱ་རཱི་བྷྱཱཾ། ཀུ་མཱ་རི་བྷྱཿ། ཀུ་མཱཪྻཿ། ཀུ་མཱ་རཱི་བྷྱཱཾ། ཀུ་མཱ་རཱི་བྷྱཿ། ཀུ་མཱཪྻཿ། ཀུ་མཱཪྻོཿ། ཀུ་མཱ་
རཱི་ཎཱཾ། ཀུ་མཪྻཾ། ཀུ་མཱཪྻོཿ། ཀུ་མཱ་རི་ཥུ། ཧེ་ཀུ་མཱ་རཱི། ཧེ་ཀུ་མཱཪྻཽ། ཧེ་ཀུ་མཱཪྻཿ། དེ་ལྟར། ན་དཱི། ན་དྱཽ། ན་དྱཿ། བདྷཱུ་ཞེས་པ་འདིའི། སུའི་རུ་ཉིད་དེ། རྣམ་པར་གཅད་པ་ལའོ། །ཨིཀའི་ཡཎ་ཨཙ་ལའོ། །ཨམ་ལ་སྔ་མའོ། །དང་པོ་དག་གི་ཨ་ཙ་ལའོ། །ཨཱི་ཨཱུ་ལས་ཀྱང་ཨཏའོ། །ཨེ་ཙ་
ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །བུད་མེད་ཨཱི་ཨཱུ་དག་ལས་སོ་ཞེས་པ་ནུཊའོ། །ངིའི་ཨཱམ་དེ་ལའོ། །ཨཀའི་རིང་པོ་ཨཀ་ལའོ། །ཀུ་ལས་ཀྱང་ཞེས་པའོ། །ཀུན་དུ་བོད་པ་ལ་ངི་ཨཱུའི་ཞེས་པ་ཐུང་ངུའོ། །ཨེང་ཐུང་ངུ་ལས་ཏ་མ་ཡིན་པའི་ཀུན་དུ་བོད་པ་ལའོ། །རྣམ་པར་གཅད་པའོ། །བ་དྷཱུཿ། 
བདྷྭཽ། བ་དྷྭཿ། བ་ནྷྭཾ། བ་དྷྭཽ། བ་དྷཱུཿ། བ་དྷྭཱ། བ་དྷཱུབྷྱཱཾ། བ་དྷཱུ་བྷིཿ། བ་དྷྭཻ། བ་དྷཱུ་བྷྱཱཾ། བ་དྷཱུ་བྷྱཿ། བ་དྷྭཱཿ། བ་བྷཱུ་བྷྱཱཾ། བ་དྷཱུ་བྷྱཿ། བ་དྷྭཿ། བ་དྷོཿ། བ་བྷཱུ་ནཱཾ། བ་དྷྭི། བ་དྷྭོཿ། བ་དྷཱུ་ཥུ། ཧེ་བ་དྷཱུ། ཧེ་བ་དྷྭཽ། ཧེ་བ་དྷྭཿ། དེ་ལྟར། ཀ་དྲཱུ། ཀ་དྲྭཽ། ཀ་དྲྭཿ། ཤྲཱི་ཞེས་པ་འདིའི་རྣམ་པར་གཅད་པའོ། །
