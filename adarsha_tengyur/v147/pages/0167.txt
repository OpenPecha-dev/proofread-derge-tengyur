སྐྱེས་པའི་ཕྱིར། །ཇི་ལྟར་རྒྱ་མཚོ་བཞི་མི་གཡོ། །གཟི་བརྗིད་དྲག་ཅིང་མཐུ་ཆེ་བ། །དགེ་སློང་ཆེན་པོ་འདི་ལྟ་བུ། །འདི་ནི་འཇིག་རྟེན་ཐུགས་བརྩེ་ཅན། །འདི་ནི་འཇིག་རྟེན་སྨན་པ་ཡིན། །འདི་ཡིས་རྒོལ་བ་རབ་འཇོམས་ཤིང་། །མུ་སྟེགས་ཅན་རྣམས་ཆོས་ལྡན་
བྱེད། །འདི་ནི་འཇིག་རྟེན་བླ་མེད་དེ། །འདི་ནི་རྐང་གཉིས་རྣམས་ཀྱི་མཆོག །འདི་ནི་འཇིག་རྟེན་རྣམ་འདྲེན་ཏེ། །སེམས་ཅན་འཁོར་བའི་རྒྱ་མཚོ་ཡི། །ཆུ་བོ་འཇིགས་ཆེན་ལས་སྒྲོལ་འདི། །སྟོན་པ་ཕ་རོལ་ཕྱིན་ཅིང་རྒལ། །འདི་ནི་ཕ་རོལ་སྐམ་སར་གནས། །ཁྱེད་ནི་ཉེར་སོང་དྲི་བྱས་
ན། །དོགས་པ་དེ་ཡིས་སེལ་བར་འགྱུར། །རྐང་གཉིས་མཆོག་གྱུར་མཐོང་ནས་ནི། །རྒྱ་ཤུག་གཟི་བརྗིད་ལྡན་གྱུར་ནས། །ཐལ་མོ་རབ་ཏུ་སྦྱར་བྱས་ནས། །དེ་ལ་འདི་སྐད་ཚིག་གསོལ་ཏོ། །འཁོར་བའི་རྒྱ་མཚོ་ཆེན་པོ་ཡི། །ནང་ན་གནས་པ་བདག་ལ་ནི། །ཐུགས་བརྩེ་བསྐྱེད་པ་
མཛད་ནས་ནི། །མི་མཆོག་བདག་ཅག་རྒལ་བར་མཛོད། །དེ་ནས་སྟོན་པ་ཐུགས་བརྩེ་བ། །ཐུབ་པ་འཇིག་རྟེན་ཀུན་ཕན་མཛད། །རྐང་གཉིས་གཙོ་བོ་སེམས་ཅན་མཆོག །རྒྱ་ཤུག་ལ་ནི་བཀའ་སྩལ་པ། །ཇི་ལྟར་སེམས་ནི་ཀུན་ཉོན་མོངས། །དེ་བཞིན་སེམས་ནི་རྣམ་པར་བྱེད། །
མདུད་པ་ཐམས་ཅད་གྲོལ་བྱེད་པའི། །ཆོས་ནི་རྒྱ་ཤུག་ཁྱོད་ཉོན་ཅིག །འདི་ན་བདག་ལ་བདག་མེད་དེ། །ཕྱིན་ཅི་ལོག་གིས་ཀུན་དུ་བརྟགས། །བདག་གམ་སེམས་ཅན་མེད་པ་སྟེ། །ཆོས་འདི་དག་ནི་རྒྱུ་དང་བཅས། །སྲིད་པའི་ཡན་ལག་བཅུ་གཉིས་དང་། །ཕུང་པོ་དང་ནི་སྐྱེ་མཆེད་ཁམས། །
ཐམས་ཅད་རབ་ཏུ་བསམ་བྱས་ན། །གང་ཟག་རབ་ཏུ་མི་དམིགས་སོ། །ནང་སྟོང་ཉིད་དུ་མཐོང་གྱུར་ཅིང་། །ཕྱི་སྟོང་ཉིད་དུ་མཐོང་གྱུར་ན། །དེ་ཡིས་ཅུང་ཟད་ཀྱང་མི་རྙེད། །སྒོམ་པ་པོ་ཡང་སྟོང་པ་སྟེ། །འཇིག་རྟེན་ཐམས་ཅད་ཉེར་འཚོ་བ། །སྟོང་པ་ཉིད་དུ་བལྟ་བར་གྱིས། །མྱ་ངན་འདས་
པ་གནས་པར་སྐྱབས། །འདི་ཡི་ལམ་ནི་ཉེར་བསྟན་བྱ། །རྒྱ་ཤུག་སེམས་ཅན་བསམ་བྱས་ཏེ། །སྟན་དེ་ཉིད་ལ་གནས་བཞིན་དུ། །འདོད་ཁམས་ཡོངས་སུ་དག་གྱུར་པའི། །ཕྱིར་མི་འོང་བའི་འབྲས་བུ་ཐོབ། །བདེན་པ་མཐོང་བར་གྱུར་པ་ནི། །རྒྱ་ཤུག་ཡིད་རབ་དགའ་བ་ཡིས། །
