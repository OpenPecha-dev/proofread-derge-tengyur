ལེགས་པར་སྦྱར་བའི་ཕྱིར་རོ། །དང་པོ་ཞེས་བྱ་བའི་སྒྲ་ནི་བཤད་ཟིན་ཏོ། །བདེ་བ་ཡང་ཡིན་ལ་མཆོག་ཀྱང་ཡིན་པས་ན་བདེ་བའི་མཆོག་ཡིན་ཏེ། དངོས་གྲུབ་ཐམས་ཅད་མི་འགྱུར་བའི་བདེ་བའོ། །མཆོག་ཀྱང་ཡིན་ལ་དང་པོའི་བདེ་བ་ཡང་ཡིན་པས་ན་
མཆོག་དང་པོའི་སྐྱེས་བུ་སྟེ། རྡོ་རྗེ་འཆང་ཆེན་པོའོ། །དེའི་འཁོར་ལོའོ། །ཡང་འདིའི་དོན་བརྗོད་པར་བྱ་སྟེ། བདག་བསྲུང་བ་དང་། བགེགས་གཞོམ་པ་ལ་སོགས་པ་བྱས་ལ་བསང་གཏོར་བྱ་བའི་ཕྱིར་སྤོས་ཆབ་ཀྱིས་བཀང་བའི་སྣོད་ལ་རྡོ་རྗེ་གནོད་སྦྱིན་ཡོངས་སུ་བཟླས་
ལ་ལྷའི་ཁང་པའི་སྒོར་བཞག་ལ་མཆོད་ཡོན་དང་། མཆོད་པ་རྣམ་པ་ལྔས་ཇི་ལྟ་བ་བཞིན་དུ་མངོན་པར་བཟླའོ། །མཆོད་ཆེན་ཐམས་ཅད་དགུག་པ་དང་། །བདག་ལ་བྱིན་རླབས་སོགས་གྲུབ་པས། །རྟག་ཏུ་རྡོ་རྗེ་དྲིལ་འཆང་ཞིང་། །དངོས་གྲུབ་མཆོག་ནི་བསྒྲུབ་པར་བྱ། །ལུགས་
མའམ་ཡང་ན་བརྐོས་པའམ། །བྱས་པའམ་ནི་བྲིས་པ་ཡི། །དང་པོའི་འཁོར་ལོ་བཞག་ནས་ནི། །དེར་ནི་ལྷར་ཡང་རབ་ཏུ་འགྲུབ། །ཡི་གེ་ཧཱུཾ་ཟློས་ཤིང་དང་པོའི་རྡོ་རྗེ་ཕྱག་རྒྱ་བཅིངས་ལ། ཨ་ནུ་རཱ་ག་ཡ་མི་ཞེས་བརྗོད་ནས་ས་ལ་བཞག་སྟེ། བཅོམ་ལྡན་འདས་ཡུམ་གྱི་དཀྱིལ་
འཁོར་སྤྲུལ་ལ། དེ་ཉིད་རང་གི་སྙིང་གར་བཞག་ན་དཀྱིལ་འཁོར་དེར་རབ་ཏུ་ཞུགས་པར་འགྱུར་རོ། །དེ་ནས་གལ་ཏེ་འབྱོར་པ་ཡོད་ན་མེ་ཏོག་ལ་སོགས་པའི་དཀྱིལ་འཁོར་མཆོད་ལ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཀཱ་ཡ་ཝཱཀ་ཙིཏྟ་པྲ་ཎ་མེ་ན་བཛྲ་བནྡྷ་ནཾ་ཀ་རོ་མི་ཞེས་བྱ་བ་དང་ཕྱོགས་ཐམས་
ཅད་ནས་ཡན་ལག་ལྔས་བཅོམ་ལྡན་འདས་འཁོར་དང་བཅས་པ་ལ་ཕྱག་བྱས་ལ། ཕྱག་བཞི་སྔོན་དུ་འགྲོ་བས་སྡོམ་པ་བཟུང་ལ། སྡིག་པ་དྲལ་བའི་བར་གྱི་ཆོ་ག་རྣམས་རྗེས་སུ་སྤྱད་པར་བྱའོ། །དེ་ནས་རང་གི་སྙིང་གའི་ཟླ་བ་ལ་གནས་པའི་དང་པོའི་རྡོ་རྗེའི་དབུས་སུ་ཡི་གེ་ཧཱུཾ་མངོན་
པར་ཡིད་ཀྱིས་བྲིས་ལ། དེ་ཉིད་ཡིད་ལ་ཟློས་ཤིང་ཆོས་ཐམས་ཅད་བདག་མེད་པར་བསྒོམ་པར་བྱའོ། །དེ་ནས་རང་བཞིན་གྱིས་འོད་གསལ་ཞིང་། ནམ་མཁའ་དང་མཉམ་པའི་སེམས་ཡི་གེ་ཧཱུཾ་དབྱངས་ཀྱི་ཡི་གེ་བཅུ་དྲུག་གིས་བསྐོར་བ་ཟླ་བའི་དཀྱིལ་འཁོར་དུ་གྱུར་པ་
