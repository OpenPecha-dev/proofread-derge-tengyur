ལ་སྐྱབས་འགྲོ་སོགས། །དཀོན་མཆོག་གསུམ་ལ་བདག་སྐྱབས་མཆི། །སྡིག་པ་ཐམས་ཅད་སོ་སོར་བཤགས། །འགྲོ་བའི་དགེ་ལ་རྗེས་ཡི་རང་། །སངས་རྒྱས་བྱང་ཆུབ་ཡིད་ཀྱིས་གཟུང་། །དངོས་མེད་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་། །མཚན་མེད་རྒྱུ་དང་བྲལ་བ་སྟེ། །སྨོན་པ་དང་བྲལ་དངོས་ཀུན་
གྱི། །རྟོག་པ་སྤངས་པ་ཉིད་དུ་འདོད། །ཨོཾ་ཤཱུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་བ་ཨཱཏྨ་ཀོ྅ཧཾ། གྲུ་གསུམ་དཀར་པོ་མཐོ་བའི་དབྱིབས། །སྣ་ཚོགས་ཆུ་སྐྱེས་རྡོ་རྗེའི་དབུས། །མཆོད་རྟེན་ནང་དུ་རྫོགས་སངས་རྒྱས། །རྡོ་རྗེ་སེམས་དཔའ་རིག་བཅས་བསྒོམ། །དེ་ཡི་ཞལ་བཞུགས་དེ་བཞིན་དུ། །རང་སེམས་
རྡོ་རྗེ་ལས་འཐོན་ནས། །ཡི་གེ་གསུམ་བྱུང་ཆུ་སྐྱེས་ལ། །ཧོ་ཡིག་གཉིས་ཀྱིས་རྣམ་པར་བརྒྱན། །དེ་ལས་བྱུང་བའི་འོད་ཟེར་གྱིས། །དེ་ནས་ལུས་ནི་ཞུ་བར་བལྟ། །ཟླ་བ་ཁུ་བའི་འོད་ཟེར་ཞུ། །མེར་མེར་ལ་སོགས་ཡང་དག་འཛིན། །ཕྱོགས་ཀུན་ན་གནས་ལྷ་མོ་རྣམས། །གླུས་བསྐུལ་དེ་ཡི་ཕྱི་ནས་
ནི། །འགྲོ་རྣམས་བདེ་བར་རབ་བསྒྲུབ་པའི། །བདག་ཉིད་བཞེངས་པ་བསམ་པར་བྱ། །རྡོ་རྗེའི་ཐུགས་ཁྱོད་གནས་ཀྱི་དབང་ཕྱུག་སེམས་ཅན་ཁམས། །སྐྱོབ་པ་དག་ནི་དགའ་བ་ཡིད་འོང་དོན་ཆེན་འདོད། །གཉེན་ཆེན་སེམས་ཅན་ཡབ་ཉིད་བདག་ལ་འདོད་པ་ཡིས། །གལ་ཏེ་མགོན་པོས་བདག་ལ་འདོད་
མཛད་འཚོ་བར་འགྱུར། །རྡོ་རྗེའི་སྐུ་ཁྱོད་སེམས་ཅན་མངའ་བརྙེས་བཀའི་འཁོར་ལོ། །སངས་རྒྱས་ཆེད་དུ་དོན་དམ་བྱང་ཆུབ་ཕན་རྗེས་སྟོན། །ཆགས་པས་འདོད་ཆགས་དམ་ཚིག་མཱ་མ་ཀཱི་བདག་ལ། །གལ་ཏེ་མགོན་པོས་བདག་ལ་འདོད་མཛད་འཚོ་བར་འགྱུར། །རྡོ་རྗེ་གསུང་ཁྱོད་ཀུན་ལ་ཕན་མཛད་
བརྩེ་བ་ཅན། །འཇིག་རྟེན་དོན་གྱི་བྱ་མཛད་རྟག་ཏུ་ཡང་དག་འཇུག །འདོད་པས་བདག་ལ་ཤིན་ཏུ་སྤྱོད་མཛད་ཀུན་དུ་བཟང་། །གལ་ཏེ་མགོན་པོས་བདག་ལ་འདོད་མཛད་འཚོ་བར་འགྱུར། །རྡོ་རྗེའི་སྐུ་ཁྱོད་དམ་ཚིག་མཆོག་གི་ཕན་ཆེན་དོན། །རྫོགས་བྱང་རྒྱུད་ལ་འདོད་བཞིན་མཉམ་ཉིད་བརྩེ་བ་ཅན། །
འདོད་པས་བདག་ལ་ཡོན་ཏན་རིན་ཆེན་གཏེར་མང་འབྱུང་། །གལ་ཏེ་མགོན་པོས་བདག་ལ་འདོད་མཛད་འཚོ་བར་འགྱུར། །བཅོམ་ལྡན་དེ་ཉིད་ལ་གཤེགས་ཀྱང་། །ལྷ་མོ་རྣམས་ཀྱིས་གླུ་ཡིས་བསྐུལ། །དངོས་རྣམས་སྒྱུ་མ་ལྟར་རིག་ནས། །རང་གི་སྔགས་དོན་བསྒོམ་པར་བྱ། །སངས་རྒྱས་སྤྲུལ་
