ཀ་སྱ་ཏྭ་ཡཱ་པྲཱཾ་ཤུཿཤ་རཙྩ ནྡྲཱ་ར་དཱ་ཏ་ཡཱ། །ཀཱ་མ་མུཀྟཱ་ལ་བྷེ་བཾ་ཤཱཿསཾ་བྷ་བེ་ན་བི་བྷཱུ་ཥི་ཏཿ། །ཨི་ཏི་ཏེ་ནཱ་ད་རཱཏྤྲྀཥྚཱ་ཛྙཱ་ཏྭཱ་ཏཾ་མུ་ནི་པུ་ཏྲ་ཀཾ། །ཨ་ཛྙཱ་ཏ་ཀཱ་མ་བྲྀཏྟཱ་པི་སཱ་བྷི་ལཱ་ཥཱ་ཛ་གཱ་ད་པཱ། །ཨ་ཧོ་པདྨཱ་བ་ཏཱི་ན་མ་པཱ་
བརྒྱུད་མུ་ཏིག་གི། །འཁྲི་ཤིང་རབ་ཏུ་འཐོབ་པ་དག །སྟོན་ཟླ་ལྟར་དཀར་ཁྱོད་འཁྲུངས་པས། །རྣམ་པར་བརྒྱན་པ་རབ་ཏུ་སྦྲོས། །ཞེས་པ་དེ་ཡིས་གུས་པས་དྲིས། །དེ་ནི་ཐུབ་པའི་བུར་ཤེས་ནས། །འདོད་པའི་སྤྱོད་ཚུལ་མི་ཤེས་ཀྱང་། །མངོན་པར་འདོད་ལྡན་དེ་ཡིས་སྨྲས། །རྐང་པས་པད་འཕྲེང་རབ་བསྐྱེད་
དོ་དྱཏྤདྨ་མཱ་ལི་ཀཱ། །མྲྀ་གི་གརྦྦྷ་ས་མུདྦྷུ་ཏ་ཤཱ ཎྜི་ལྱ་སྱ་སུ་ཏཱ་མུ་ནེཿ། །སྭཱ་ག་ཏཾ་ཏེ་མུ་ནི་སུ་ཏ་པྲི་ཡཾ་མེ་ཏ་བ་དཪྴ་ནཾ། །མ་ནོ་ཧ་ར་ཏ་རཾ་ཙི་ཏྲ་བལྐ་ལཾ་ཀི་མི་དཾ་བྲ་ཏཾ། །བི་བྷཱུ་ཥི་ཏོ་ཡམྨཱ་ཡཱུ་རཻཿཤི་ཁཎྜཻ་རི་བ་པཉྩ་
པས།  དག་ནི་པདྨ་ཅན་ཞེས་བྱ། །རི་དགས་མངལ་སྐྱེས་ཐུབ་པ་ནི། །ཤྭཾ་ཊི་ལྱ་ཡི་བུ་མོ་ཡིན། །ཐུབ་པའི་སྲས་ཁྱོད་ལེགས་པར་འོངས། །བདག་ནི་ཁྱོད་མཐོང་རབ་ཏུ་དགའ། །ཤིན་ཏུ་ཡིད་འཕྲོག་བཀྲ་བ་ཡི། །ཤིང་ཤུན་བརྟུལ་ཞུགས་འདི་ནི་ཅི། །རྨ་བྱའི་གཙུག་ཕུད་ལྔ་དག་གིས། །བརྒྱན་པ་བཞིན་དུ་ཁྱོད་
བྷིཿ། །སུ་རཱཙྩཱ་ཀཱིརྞྞ་ཀུ་སུ་མཿཤོ་བྷ་ཏེ་ཏེ་ཛ་ཊཱ་བྷ་བཿ། །ཨི་ཡ་མཱ་མ་ལ་ཀསྠཱུ་ལ་པྲཱ་ལེ་ཡ་ཕ་ལ་ཀོ་ཛྭ་ལཿ། །དཱིརྒྷཱ་ཀྵ་སཱུ་ཏྲ་མཱ་ལ་བྷི་ཀཎྛེ་བ་ཧུ་གུ་ཎཾ་ཏ་བ། །ཨཱ་སཱུ་ཏྲ་ཡ་ཏི་ཏེ་པཱ་ཎིཤྩི་ཏྲཻཿཀུ་ཤ་པ་བི་ཏྲ་ཀཾཻཿ། །ཀུ་ཊི་མེ་བེ་ཎུ་
ཀྱི་ནི། །རལ་ཚོགས་འདི་དག་ལྷ་ལས་ཀྱང་། །མེ་ཏོག་དག་གིས་རྒྱས་པར་མཛེས། །སྐྱུ་རུ་ར་ནི་རགས་པ་ཡི། །འབྲས་བུ་ཁ་བ་རབ་འབར་འདྲ། བགྲང་འཕྲིང་འཕྲེང་བ་རིང་པོ་ཡིས། །ཁྱོད་ཀྱི་མགྲིན་ལ་ཡོན་ཏན་མང་། །འོད་མའི་དབྱུག་པ་འཁྱོག་པོ་འདིར། །ཡལ་འདབ་གསར་པའི་འཕྲེང་བ་ལ། །ཁྱོད་ཀྱི་ལག་པ་ཀུ་
