དཤ་བྷིརྦྦཪྵོསྟེ་ན་སད་རཏྣ་སཾཉྩ་ཡཿ། །ནི་ཧི་ཏཿསཏྱ་ཤཱི་ལེ་ན་པ་ཎྱཱརྷོ་སཽ་ས་མརྤྤི་ཏཱཾ། །ཨེ་ཏདྦྷཱུ་པཱ་ལ་དུ་ཧི་ཏུཿསནྡེ་ཤཾ་དཱུ་ཏ་བྷཱ་ཥི་ཏཾ། །ཤྲུ་ཏྭཱ་གྲྀ་ཧ་པ་ཏིཿསརྦྦ་རཏྣཱ་ནྱཱ་དཱ་ཡ་སཱ་དཱ་རཿ། །སྭ་ཡཾ་
ལོ་ནི་བཅུ་གཉིས་སུ། །བདེན་པའི་ངང་ཚུལ་དེ་ཡིས་ནི། །རིན་ཆེན་དམ་པའི་ཚོགས་བསྒྲུབས་པ། །རིན་དུ་འོས་པ་དེ་སྦྱིན་མཛོད། །ས་སྐྱོང་བུ་མོ་དག་གི་འཕྲིན། །འདི་དག་ཕོ་ཉས་བཤད་པ་ནི། །ཐོས་པས་ཁྱིམ་བདག་གུས་བཅས་པས། །རིན་ཆེན་ཐམས་ཅད་བླངས་བྱས་ནས། །རང་
རཱ་ཛ་གྲྀ་ཧ་ག་ཏྭཱ་ད་ཏྭཱ་སཽ་ཏྲི་ཎ་ལཱི་ལ་ཡཱ། །སཱུཀྟ་རཏྣཾ་ཏ་དཱ་དཱ་ཡ་པ་པཱ་ཋོཏྐཎྛ་ཏོ་མུ་ཧུཿ། །ཡདྻཏྐིཾ་ཙིཏྶུ་ཁ་མ་ནུ་པ་མཾ་བཱ་ཧྱ་མཱ་བྷྱནྟ་རཾ་བཱ་ཡོ་ཡཿསཏྭ་པྲ་ས་པ་བི་ལ་སདྦཾསྨ་ཡཿསིདྡྷི་ཡོ་གཿ། །ཡེ་ཡེ་
ཉིད་རྒྱལ་པོའི་ཁྱིམ་སོང་སྟེ། །དེ་ལ་རྩྭ་ཡིས་རྩེ་དགའ་བྱིན། །དེ་ཚེ་ལེགས་བཤད་རིན་ཆེན་ཐོབ། །དགའ་བས་ཡང་དང་ཡང་དུ་བརྗོད། །གང་གང་ཕྱིའམ་ནང་གི་བདེ་བ་དཔེ་མེད་ཆུང་ཟད་དག་དང་ནི། །གང་གང་སྙིང་སྟོབས་ཀྱིས་བསྐྲུན་ཡ་མཚན་རྒྱས་པ་གྲུབ་པའི་སྦྱོར་བ་དང་། །གང་གང་

སྲེད་པ་རབ་ཞི་དྲི་མེད་རྒྱུན་དུ་དགའ་བ་ཐོབ་པ་དག །དེ་དེས་བསོད་ནམས་དཔག་བསམ་ཤིང་གི་འབྲས་བུ་སྨིན་ཅིང་རྒྱས་པའི་ཁུར། །རྒྱ་ཆེར་གྱུར་པའི་གཏེར་དེ་ཡིས། །རིན་ཆེན་ཀུན་གྱིས་ལེགས་བཤད་ནི། །བླངས་པ་ཐོས་ནས་མི་ཡི་བདག །ཡ་མཚན་བཅས་པས་དེ་ལ་དྲིས། །ཅི་སླད་
