མསྟྭེ་ཥཱ་མ་མཱཾ་བུ་ཀུཾ་བྷསྟྲྀཥྞཱརྟྟ་ཡོརྫཱི་བི་ཏ་མེ་བ་པི་ཏྲོཿ། །སཽ་ཛ་ནྱ་མཱ་ལམྦྱ་དྷི་ཡཱ་ད་ཡཱརྡྲ་ག་ཏྭཱ་ཏྭ་ཡཱ་ཏཱུརྞྞ་ཏ་ར་པྲ་དེ་ཡཿ། །ཨུཀྟྭེ་ཏི་ཀཎྛཱནྟ་ར་བརྟྟི་ཛཱི་བཿཤྭཱ་ས་པྲ་ཡཱ་ས་སྠ་གི་ཏཱལྤ་བརྞྞཿ། །མཽ་ནི་པ་ཙ་ཀྲེ་ཤྲུ་ཌི་ཏཱ་ན་ན་སྱ་བཻ་ལཀྵུ་
གི་ཆུ་བུམ་ཕྱི་མ་མེད་པ་འདི། །ཕ་མ་སྐོམ་པས་གཟིར་ལ་འཚེ་བར་ནི། །གཞུང་བཟང་ལ་བརྟེན་བརྩེ་བས་བརླན་པའི་བློས། །མྱུར་བར་སོང་སྟེ་ཁྱོད་ཀྱིས་སྦྱིན་པར་བྱ། །ཞེས་པ་མགྲིན་པའི་བར་དུ་སྲོག་རྒྱུ་ཞིང་། །དབུགས་བརྩེགས་ངལ་བས་སྒྲ་བསྒྲིབས་ཆུང་ངུར་བརྗོད། །བཞིན་ནི་ཕྱི་རོལ་ཕྱོགས་པའི་མདའ་དག་ལ། །ཐུབ་པ་
དཱིཀྵྱཱ་མི་བ་སཱ་ཡ་ཀ་སྱ། །ཀུཾ་བྷཾ་ས་མཱ་དཱ་ཡ་ཏ་ཏཿཀྵི་ཏཱི་ཤསྟ་དཱ་ཤྲ་མཾ་པྲཱ་སྱ་བྷྲྀ་ཤཾ་པྲ་ཏཔྟ། །ཏཱ་བེ་ཀ་པུ་ཏྲོསྠ་བི་རཽ་བི་ལོཀྱ་ཏ་དནྟི་ཀ་ནཻ་བ་ཤ་ཤཱ་ཀ་གནྟཱཾ། །པུ་ཏྲེ་ཏི་དཱུརཱཏྤ་ད་ཤཔྟ་དཏྟ་ཀརྞྞཽ་ས་མུཏྐཎྛི་ཏ་མཱ་ན་པཽ་ཏཽ། །སྣེ་ཧཱདྠུ་བཱ་ཎཱ་བུ་པ་
དེས་ནི་སྐྱེངས་པ་སྟོན་བཞིན་མཛད། །དེ་ནས་ས་དབང་མཆོག་ཏུ་རབ་གདུང་བས། །བུམ་པ་བླངས་ནས་དེ་དག་བསྟི་གནས་ཕྱིན། །རྒན་རྒོན་བུ་གཅིག་ལྡན་པ་དེ་དག་མཐོང་། །དེ་དག་གམ་དུ་འགྲོ་བ་ཡོངས་མ་ནུས། །རིང་ནས་རྡོག་སྒྲ་དག་ལ་རྣ་གཏོད་ཅིང་། །མཛའ་བས་རབ་ཏུ་འདོད་ལྡན་བུ་ཞེས་པ། །སྨྲ་བ་དེ་དག་ལ་ནི་ཉེར་

བཏུད་དེ། །དམོད་པས་ཉམས་པ་མི་འདོད་རྒྱལ་པོས་སྨྲས། །གླང་ཆེན་མྱོས་པའི་ཚུལ་གྱིས་ཨ་མྲའི་ཤིང་། །འཇོམས་བཞིན་གང་གིས་གཞོན་ནུ་རབ་བཅོམ་པ། །སྐྱེས་མཆོག་འཁྲི་ཤིང་དག་གི་སྟ་གྲི་རྩུབ། །སྡིག་པའི་སྐལ་ལྡན་བདག་ནི་ཁྱེད་སྲས་མིན། །བདག་གི་མདའ་ཡིས་ལུས་བཅོམ་ཁྱོད་ཀྱི་སྲས། །དེ་ནི་
