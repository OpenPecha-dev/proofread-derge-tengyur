མེ་བཱནྡྷཱ་ཡེ་རཱ་ཛན་པུ་བི་ཙཱཪྻ་བི་དྷི་ཡ་ཏཱཾ། །པཱུ་ཡ་ཏེ་པཱ་བཱ་ཀེ་ནེ་བ་རཱ་ཛ་དཎྜེ་ན་ཀིལྦི་ཥཱི། །ཨི་ཏྱཱ་གྲ་ཧྱེ་ཎ་བ་ཧུ་ཤོ་ཡཱ་ཙ་མཱ་ནཾ་སྭ་ཤཱ་ས་ནཾ། །ཏ་མུ་བཱ་ཙཱ་ནུ་ཛོ་རཱ་ཛོ་པྲཱ་ཏཿཀརྟྟཱསྨི་ཡཏྐྵ་མཾ། །ཨསྨིན་
ཆད་པ་བདག་འཚལ་ལོ། །ཤིན་ཏུ་དཔྱད་ནས་བསྒྲུབ་པར་མཛོད། །དག་བྱེད་ཀྱིས་བཞིན་རྒྱལ་པོ་ཡི། །ཆད་པས་སྡིག་ཅན་གཙང་བར་འགྱུར། །ཞེས་དེ་འབད་པས་ལན་མང་དུ། །རང་གི་ཆད་པ་དོན་གཉེར་བ། །དེ་ལ་ནུ་བོ་རྒྱལ་པོས་སྨྲས། །ནང་པར་བདག་གིས་གང་བཟོད་བྱ། །རེ་ཞིག་
པྲ་ཏཱི་ཀྵ་ཏཱ་བདྦྷ་བཱ་ནུ་དྱཱ་ནུ་ཀཱ་ན་ནེ། །ཨི་ཏྱུཀྟྭཱ་པཽ་ར་ཀཱཪྻཱ་ཡ་ནིཪྻ་ཡཽ་བྲྀ་ཐི་བཱི་པ་ཏིཿ། །ཏ་ཏྲ་པྲ་ཏཱི་ཀྵ་མཱ་ཎེ་ཐ་མུ་ནཽ་ནི་ཡ་མཱ་ནིཤྩ་ལེ། །རཱ་ཛཱ་བིསྨྲྀ་ཏ་ཏཏྐྲྀ་ཏྱཿཥཌྦྷི་ར་བྷྱཱ་ཡ་ཡཽ་དི་ནཻཿ། །ཀྲྀ་ཤཱ་ཀཱ་རནྣི་རཱ་ཧཱ་ར་མ་ཐ་དྲྀཥྚཱ་གྲ་
ཁྱོད་ནི་སྐྱེད་ཚལ་གྱི། །ནགས་འདི་ཉིད་དུ་གནས་པར་མཛོད། །ཅེས་བརྗོད་གྲོང་པའི་བྱ་བ་ལ། །ས་ཡི་བདག་པོ་ཕྱི་རོལ་སོང་། །དེ་ནས་དེར་ནི་ཐུབ་པ་དག །ངེས་པར་གཡོ་མེད་གནས་པའི་ཚེ། །རྒྱལ་པོ་དེ་ཡི་བྱ་བ་བརྗེད། །ཉི་མ་དྲུག་ནས་མངོན་པར་འོངས། །དེ་ནས་རྒྱལ་པོ་ཕུ་བོ་ནི། །ཟས་མེད་རིད་པའི་རྣམ་
ཛཾ་ནྲྀ་པཿ། །ནིནྡ ནྶྭ་བུདྡྷི་སམྨོ་ཧཾ་ཙཀྲ་ཏ་སྱ་པྲ་པཱ་ད་ནཾ། །པྲཱཧྨ་ཎཻཿས་ཧི་ཏ་ཀྲྀ་ཏྭཱ་ས་ཏཏྤཱ་པ་པྲ་མཱཛྫ་ནཾ། །ཥཌིྜ་ན་ཀླེ་ཤ་དཱ་ནེ་ན་ཡ་ཏེ་ཏསྨིནྣ་ཏ་སྱ་ཏཿ། །ཨ་ཧཾ་ས་སཱུཪྻོ་རཱ་ཛཱཪྵིཤྩནྡྲོ་རཱ་ཛཱ་པ་ཡཾ་ཤི་བུ། །ཛཱ་ཏཿསཾ་བཏྨ་རཻཿཥཌྦྷིཿཥི་ཌ་
པ་མཐོང་། །ཀུན་རྨོངས་རང་གི་བློ་སྨོད་ཅིང་། །དེ་ལ་རབ་ཏུ་དང་བར་བྱ། །དེ་ནི་བྲམ་ཟེ་དང་བཅས་པས། །དེ་ཡི་སྡིག་པ་དག་བྱས་ནས། །ཉིན་དྲུག་ཉོན་མོངས་བྱིན་པ་ཡིས། །དེ་སོང་ཚེ་ན་རབ་ཏུ་གདུངས། །ང་ཉིད་རྒྱལ་པོའི་དྲང་སྲོང་ཉི། །བྱིས་པ་འདི་ཡང་རྒྱལ་པོ་ཟླ། །ཉིན་དྲུག་ཐུབ་པ་ཉོན་མོངས་པར། །བྱས་
