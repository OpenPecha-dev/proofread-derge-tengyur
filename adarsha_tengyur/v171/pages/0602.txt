བ་བཉྪི་ཏཾ་ཡཏྟ་ད་པྲ་ཡཏྣོ་བ་ན་ཏཾ་ཀ་ར་གྲེ། །བི་ཤྭཱ་བ་པོཿསིདྡྷ་པ་ཏེསྟྭ་དརྠེ་ཀ་ནྱཱརྠི་ཏཱ་ཏྭཛྫ་ན་ཀེ་ན་སཱ་ཧི། །དྷ་ནྱོསྨི་ཡ་སྱ་ཏ་ན་ཡཱ་ཏྲི་ཛ་གཏྤྲི་ཡེ་ཎ་སཾ་ཡུ་ཛྱ་ཏེ་དྱུ་ཏི་མ་ཏཱ་ཤ་ཤི་ནཱ་ནི་ཤེ་
ཁྱོད་ཀྱི་འདོད་པ་གང་། །དེ་ནི་འབད་མེད་ལག་པའི་རྩེ་ན་གནས། །གྲུབ་བདག་སྣ་ཚོགས་ནོར་ལས་བུ་མོ་དེ། །ཁྱོད་ཀྱི་ཡབ་ཀྱིས་ཁྱོད་ཀྱི་དོན་དུ་བསླངས། །འདོད་ལྡན་ཟླ་བས་མཚན་མོ་ལ་བཞིན་གང་གི་བུ་མོ་ལ། །འགྲོ་གསུམ་དགའ་བས་ཡང་དག་སྦྱོར་བ་བདག་ནི་སྐལ་བ་
བ། །ཨུཀྟྭ་ཏི་སིདྡྷ་པ་ཏི་རཱ་ད་ར་ཧཪྴ་པཱུརྞྞ་ཀ་ནྱཱ་བི་བཱ་ཧ་མ་ཧ་ཏཏ་པ་ར་ཏཱ་མ་བསྟཿ། །པྲཱ་ཏཿས་ཁེ་ས་བྷ་བི་ཏཱ་བི་པུ་ལོཏྤ་བསྟེ་ཀཱནྟཱ་ས་མཱ་ག་མ་སུ་དྷཱ་ས་ཕ་ལོ་བི་ཧཱ་རཿ། །ཡ་སྨིན་
བཟང་། །ཞེས་པ་བརྗོད་ནས་གྲུབ་པའི་བདག་པོ་གུས་དང་དགའ་བས་གང་། །བུ་མོ་ཡིས་ནི་བག་མའི་དགའ་སྟོན་དེ་ཉིད་སྐྱོང་བར་གྱུར། །གྲོགས་པོ་ནང་པར་ཁྱོད་ཀྱི་རྩེ་དགའི་དགའ་སྟོན་ཡངས་པ་དག །མཛེས་མ་དང་འགྲོགས་བདུད་རྩིའི་འབྲས་བུ་ལྡན་པ་དེ་བྱུང་སྟེ། །གང་ལ་
ས་མཱ་ན་གུ་ཎ་སཾ་ག་མ་དཪྴ་ནེ་ན་དྷཱ་ཏུཿཀ་རི་ཥྱ་ཏ་ཛ་ནཿསུ་ཙི་རཱཏྤྲ་ཤཾ་སཾ། །ཨི་ཏི་སུ་ཧྲྀདྦ་ཙ་ནཾ་ས་ཧ་སཻ་བོཙྪ་ལདྦི་པུ་ལ་ཧཪྵ་མ་ཡ་ས་ནི་ཤ་མྱ་ཏཏ། །ནི་ཛ་པ་དཾ་པྲ་ཡ་ཡཽ་ཀ་ལ་ཡནྡྷི་
ཡོན་ཏན་མཚུངས་པ་འགྲོགས་པ་དག་ནི་མཐོང་བས་ན། །སྐྱེ་བོ་ཤིན་ཏུ་ཡུན་རིང་བྱེད་པོ་བསྔགས་པ་བྱེད་པར་འགྱུར། །ཞེས་པ་གྲོགས་པོའི་ཚིག་དེ་ཐོས་གྱུར་ནས། །དགའ་བའི་རང་བཞིན་ཡངས་པ་འཕྲལ་རྒྱས་ཏེ། །ཉིན་མོའི་ལྷག་མ་དུ་མའི་དུས་མཚུངས་པར། །
