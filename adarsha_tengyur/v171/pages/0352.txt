ར་ཡཏ། །ཏཏྤཱུརྞྞ་པུ་ཎྱ་པྲ་ཎི་དྷཱ་ན་ཡོ་གཱ་པྲཱཔྟཿས་རཱ་ཛཱ་ཏི་དི་བེ་ཤྭ་ར་ཏྭཾ། །དྷརྨྨ་ནུ་བདྡྷཾ་བྷ་ག་བནྦི་བྷཱུ་ཏི་མུཀྟྭེ་ཏི་བཱ་ཎཱི་མ་ན་ཡཏྤྲ་ཤཱནྟིཾ། །ཨི་ཏི་ཀྵེ་མེནྡྲ་པི་ར་ཙི་ཏཱ་ཡཱཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་
རབ་ཏུ་བྱས། །དེ་ཡི་བསོད་ནམས་སྨོན་ལམ་རྫོགས་ལྡན་པས། །རྒྱལ་པོ་དེས་ནི་ལྷ་ཡི་དབང་ཉིད་ཐོབ། །འབྱོར་པའི་ཆོས་ཀྱིས་བཅིངས་ཞེས་གསུངས་ནས་ནི། །བཅོམ་ལྡན་གསུང་ནི་རབ་ཏུ་ཞི་བར་གྱུར། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་
ཡཱཾ་ཤཀྲེ་ཙྱཱ་བ་ནཱ་བ་དཱ་ན་མཥྚ་སཔྟ་ཏི་ཏ་མཿཔལླ་ཝཿ།།      །།སྟྲཱི་བྷིརྦི་མོ་ཧི་ཏ་མ་ཏེ རྨྨ་ཏི་རཱཀྵ་སཱི་བྷིརྦྦཏྟ་པྲ་བྲྀཏྟ་མ་ན་སཿསུ་ཁ་བཱཾཉྩ་ཡཻ་བ། །ཀླེ་ཤཿཔ་ཏནྟ་ས་ཧ་སཱ་པུ་རུ་ཥ་སྱ་ཏེ་
བསམ་གྱི་འཁྲི་ཤིང་ལས་བརྒྱ་བྱིན་ལྟུང་བའི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་བདུན་རྩུ་རྩ་བརྒྱད་པའོ།།      །།བུད་མེད་བློ་ཡི་སྲིན་མོ་རྣམས་ཀྱིས་བློ་གྲོས་རྨོངས་བྱས་ཤིང་། །བདེ་བ་འདོད་པ་ཉིད་ཀྱིས་ཡིད་ནི་ནོར་ལ་ཞུགས་གྱུར་པ། །སྐྱེས་བུ་གང་ཞིག་རབ་ཞི་དང་བྲལ་ཞི་བར་མི་བགྲོད་
ཏེ་གཙྪ ནྟི་ནཱ་མ་པ་པི་ནཱ་པྲ་ཤ་མཾ་ཤ་མཾ་ཡེ། །ཤྲཱ་བསྟྱཱཾ་པྲཱཧྨ་ཎཿཔུརྦྦ་ཛཱི་བ་ཤརྨྨཱ་བྷི་དྷོ་བཏ། །བ་ཡ་སོརྦྡྷ་ཤྲུ་ཏཱ་དྷྱཱ་ཡཱི་བྲ ཧྨ་ཙཪྻཾ་ཙཱ་ར་ཡཿ། །ས་ཛ་རཱ་ཤ་བ་ལཤྩ་ཤྲུཿསྣེ་ཧཱངྦ ནྦྷུ་བྷི་ར་ནྭི་ཏཿ། །དྷརྨྨ་མཱརྒ་ནུ་རོ་དྷེ་ན་
པ། །དེ་དེ་ངེས་པར་འཕྲལ་ལ་ཉོན་མོངས་དག་ཏུ་ལྟུང་བར་འགྱུར། །གང་གིས་ན་ཚོད་བྱེད་དག་ནི། །ཐོས་པ་བརྗིད་ཅིང་ཚངས་སྤྱོད་སྤྱད། །བྲམ་ཟེ་འཚོ་བ་བདེ་ཞེས་པ། །མཉན་ཡོད་དུ་ནི་སྔོན་བྱུང་གྱུར། །མཛའ་བས་གཉེན་གྱིས་གསོལ་བཏབ་པའི། །རྒན་པོ་སྨ་ར་ཁྲ་བོ་དེ། །ཆོས་ཀྱི་ལམ་གྱི་ངོ་དག་ཏུ། །
