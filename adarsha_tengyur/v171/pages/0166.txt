པ་པུ་ཏྲཀྩི་རནྡི་བི། །ས་ཨེ་ཥ་པུ་ཎྱོ་བཱ་ན་ཡཿཡ་ཤོ་དཿཤྲེ་ཡ་སཱ་ནི་དྷིཿ། །སུ་ཧྲྀ་དཱ་མ་པི་སཾ་པྲཱཔྟཿཀཱིརྟྱཱ་ཀུ་ཤ་ལ་སེ་ཏུ་ཏཱཾ། །ཀྲྀ་ཀཱིརྣཱ་མ་ཀྵི་ཏི་པ་ཏིཿཀ་ཤྲ་པྱ་སྱཱ་ཀ་རོཏྤུ་རཱ། །ཤཱསྟུཿཤ་རཱི་ར་ནིཏྤཱ་ཎེ་རཏྣ་སྟཱུ་མུ་དཱ་ར་དྷཱིཿ། །
རིས་སུ་ནི་ཡུན་རིང་གྱུར། །བསོད་ནམས་ལྡན་པ་དེ་ཉིད་དང་། །གྲགས་སྦྱིན་དགེ་ལེགས་གཏེར་འདི་སྟེ། །གྲགས་པ་ཡིས་ནི་གྲོགས་རྣམས་ཀྱང་། །དགེ་བའི་ཟམ་སྟེགས་ཉིད་དག་ཐོབ། །ས་བདག་ཀྲྀ་ཀཱི་ཞེས་བྱ་བ། །རྒྱ་ཆེན་བློ་ཡིས་སྟོན་པ་ནི། །འོད་སྲུངས་སྐུ་དག་མྱང་འདས་ལ། །རིན་ཆེན་མཆོད་རྟེན་སྔོན་ཚེ་བྱས། 
ཡ་ཤ་སྭྰི་ར་མེ་ཏ་ན་ཡསྟྲི་ཏཱི་ཡསྟ་སྱ་བྷཱུ་པ་ཏེཿ། །བྷཀྟྱཱ་ཏསྨྲིནྨ་ཎིསྟཱུ་པེ་རཏྣ་ཙྪཏྟྲཾ་ནྱ་བེ་ཤ་ཡཏ། །ཏེ་ནཱ་པི་པཱུརྦྦ་པུཎྱེ་ན་ཛནྨ་ནྱསྨིནྨ་ཧ་ཡ་ཤཱཿ། །ཛཱ་ཏོ་ཡ་ཤོདཤྩྪཏྟྲེ་ཎ་རཏྣ་དཱིཔྟེ་ན་བྷཱུ་ཥི་ཏཿ། །ཨི་ཏྱེ་ནྱ་ཛནྨ་སུ་
ས་ཡི་བདག་པོ་དེ་ཡི་བུ། །གསུམ་པ་གྲགས་ལྡན་དགའ་བ་ཡིས། །ནོར་བུའི་མཆོད་རྟེན་དེ་ལ་ནི། །གུས་པས་རིན་ཆེན་གདུགས་དག་བཀོད། །དེ་དང་བསོད་ནམས་སྔ་མ་ཡིས། སྐྱེ་བ་འདི་ལ་གྲགས་སྦྱིན་ནི། །གྲགས་ཆེན་རིན་ཆེན་འབར་བ་ཡི། །གདུགས་ཀྱིས་བརྒྱན་པ་དག་ཏུ་སྐྱེས། དེ་ལྟར་སྐྱེ་བ་གཞན་ལ་
ཀྲྀ་ཏོད་ཡ་བདྡྷ་མཱུ་ལ་སོནྨེ་ཥ་ཨེ་ཥ་ཤ་ཤི་ཤུ་བྷྲ་ཡ་ཤོཾ་ཤུ་པུཥྤཿ། །དྷརྨྨ་དྲུ་མཿཔྲྀ་ཐུ་ཏ་རཿཔ་ལི་ཏཱ་སྱ་སཱ་དྷོཿཤྲུ་ཏྭེ་ཏི་བིསྨ་ཡ་མ་བཱ་པ་ས་བྷིཀྵུ་སཾ་གྷཿ། །ཨི་ཏི་ཀྵེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཱཾ་བྷོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་
ལེགས་བྱས་རྒྱས་པས་རྩ་བ་བཅིངས། །ཟླ་བ་ལྟར་དཀར་གྲགས་པའི་འོད་ཟེར་རྒྱས་པའི་མེ་ཏོག་ཅན། །དགེ་བ་དེ་ཡིས་ཆོས་ཀྱི་ལྗོན་པ་རབ་རྒྱས་འབྲས་སྨིན་འདི། །དེ་སྐད་ཐོས་ནས་དགེ་སློང་ཚོགས་དེས་ཡ་མཚན་ཐོབ་པར་གྱུར། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་
