ནཾ་ཀྵུབྡྷཱབྡྷོ་སཾ་གྷ་རཀྵི་ཏཿ། །ས་ནཱ་ག་ནི་ལ་ཡཾ་པྲཱབྟ སྟ་ཏྲ་ནཱ་གཻ་རརྠཱ་ནི་ཏཿ། །བི་དྷཱ་ཡ་པཱུརྦྦ་སཾ་བུདྡྷ་དྲྀ་དྡྷ་ཙེ་ཏྱཱ་པི་བནྡ་ནཾ། །དྲྀཥྚ་ཤྭཱ་ས་ར་ད་སྤཪྵ་བི་ཥཱ་ཎཱཾ་ཙིནྟ་ཡ་ཀྲྀ་ཤཿ། །སུ་ཙ་རི་ཏཱ་བྷི་མ་
འདུན་སྲུང་། །རང་གིས་བདག་ཉིད་རབ་ཏུ་འཕངས། །དེ་ནས་ཀླུ་ཡི་གནས་སུ་ཕྱིན། །དེ་ནི་ཀླུ་ཡིས་གསོལ་བཏབ་དེས། །རྫོགས་སངས་རྒྱས་ཀྱི་མཆོད་རྟེན་ལ། །མངོན་པར་ཕྱག་འཚལ་མངོན་བསྒྲུབས་ནས། །བསམ་པས་ཕྲ་དེས་མཐོང་དབུགས་དང་། །སོ་རེག་དུག་ལྡན་དེ་དག་ལ། །ལེགས་པའི་སྤྱོད་ཚུལ་མངོན་
ཏཱནྟེ་ཥཱཾ་བི་ད་དྷ་དྷརྨྨ་དེ་ཤ་ནཱཾ། །ཏཱི་བྲ་ར་ཏི་ས་མུདྦི གྣཿསྭ་དེ་ཤ་ག་མ་ནོཏྶུ་ཀཿ། །ས་ནཱ་གཻརྦ་ཎི་ཛཱ་མེ་བ་ནྱསྟཿཔྲ་བ་ཧ་ཎི་ཀྵ་ཎཱཏ། །ཏེ་པྲ་ཧྲྀཥྚཱཿས་མཱ་སཱ་དྱ་པ་ར་ལོ་ཀཱ་དི་བཱ་ག་ཏཾ། །པྲ་ཏྱཱ་བྲྀཏྟ་པྲ་བ་
འདོད་པའི། །ཆོས་ནི་རབ་ཏུ་བསྟན་པ་མཛད། །ཆགས་བྲལ་དྲག་པོའི་ཡིད་འབྱུང་གིས། །རང་ཡུལ་འགྲོ་བར་འདོད་པ་དེ། །ཀླུ་རྣམས་ཀྱིས་ནི་ཚོང་པ་ཡི། །གཟིངས་ལ་སྐད་ཅིག་ཉིད་ཀྱིས་བཀོད། །འཇིག་རྟེན་གཞན་ལས་འོངས་པ་བཞིན། །དེ་དག་དགའ་བ་ཡང་དག་ཐོབ། །གཟིངས་ནི་ཕྱིར་ལ་ལོག་
ཧ་ཎསྟི་རཾ་པྲཱ་པུརྨྨ་ཧོ་ད་དྷེཿ། །སྭ་གྲྀ་ཧོཏྐ་ཎྛ་ཡ་ཏེ་ཥཱཾ་བྲ་ཛ་ཏཱ་མ་ཏི་ས་ཏྭ་རཾ། །ནི་དྲཱ་ལུརྦྦཱ་ལུ་ཀཱ་ཀུ་ལེ་བི་སྨྲྀ་ཏཿསཾ་གྷ་རཀྵི་ཏཿ། །ག་ཏེ་སཱརྠེ་པྲ་བུདྡྷོ་ཐ་པྲ་བྷཱ་ཏེ་བི་ཛ་ནཱ་དི་ཤཿ། །པ་ཤྱ་ནཾ་སུ་ཧྲྀདྦི་ར་ཧི་ཏཿཔྲ་ད་
གྱུར་ནས། །རྒྱ་མཚོ་ཆེན་པོའི་དོགས་སུ་ཕྱིན། །རང་གི་འདོད་པས་དེ་དག་རྣམས། །ཤིན་ཏུ་རིངས་པར་བགྲོད་པ་ཡིས། །དགེ་འདུན་སྲུང་ནི་བྱེ་མ་ཡི། །ངོགས་སུ་གཉིད་དུ་སོང་བ་བརྗེད། །དེ་ནས་དོན་མཐུན་སོང་ཚེ་སད། །སྔ་དྲོ་མི་མེད་ཕྱོགས་རྣམས་སུ། །ལྟ་ཞིང་གྲོགས་དང་བྲལ་བ་ཡིས། །ཆགས་བྲལ་བློ་
