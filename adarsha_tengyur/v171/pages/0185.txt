བ་བྷྲུརྦྷཱུ ཤྨ་ཤྲུ་ས་པ་ལ་སྟ་ཌིཏྐ་པི་ལ་ལོ་ཙ་ནཿ། །ཨ་ཀཱ་ལེ་ཀཱ་ལ་སཾ་ཀཱ་ཤཱཿཀུ་ཏཿཀྲཱུ་རོ་པ་མཱ་ག་ཏཿ། །བ་ནེ་སཾ་པྲ་ཏི་དིགྦནྡྷཿཀྲྀ་ཏོ་ནེ་ན་དུ་རཱཏྨ་ནཱ། །བ་དྷྣ་ཏྱེ་ཥ་ན་ཡཱ་བནྨཱཾ་ཏཱ་བདྱུགྟི་ཀྲ་མཿཀྵ་མཿ། །ཏི་རོ་བཱནྟེ་བ་ས་ཏྱསྨིནྨ་
པ། །ཁྲོ་བོ་སྨ་ར་སྨིན་མ་སེར། །ལྟ་བྱེད་སེར་སྐྱ་གློག་ལྟ་བུ། །དུས་མིན་དུས་དང་མཚུངས་པ་ཡི། །གདུག་པ་འདི་ནི་ག་ལས་འོངས། །ད་ནི་གདུག་པའི་བདག་ཉིད་འདིས། །ནགས་ཀྱི་ཕྱོགས་ནི་བཅིངས་པ་བྱས། །ཇི་སྲིད་འདིས་བདག་མ་བཅིངས་པ། །དེ་སྲིད་རིགས་པའི་རིམ་པ་བཟོད། །དྲང་སྲོང་ཆེན་པོ་ཤིན་ཤུན་ཅན། །འཇུག་
ཧཪྵིརྦྦལྐ་ལཱ་ཡ་ནཿ། །ཛཱ་ནེ་སོ་སི་ནཱ་མ་སཱ་དྷུ་པ་རི་ཏྲཱ་ཎེ་པྲ་གལྦྷ་ཏེ། །ཏ་སྱཱ་ཤྲ་མ་ཡ་དེ་ཡོ་སཽ་པ་རི་ཙཪྻཱ་ས་རཿས་དཱ། །ལུབྡྷ་ཀཿཔདྨ་ཀོ་ནཱ་མ་ས་མེ་སཾ་རཀྵ་ཎ་ཀྵ་མཿ། །ཨི་ཏི་ནིཤྩི་ཏྱ་མ་ན་སཱ་ག་ཏྭཱ་ལུབྡྷ་ཀཱནྟི་ཀཾ། །སྭ་བྲྀཏྟཱནྟཾ་ནི་བིརྡྱ་
ངོགས་ཉེ་བའི་མཐའ་འདིར་གནས། །བདག་ཡོངས་བསྐྱབ་པ་དམ་པ་ནི། །དེས་ཀྱང་ནུས་པ་མིན་པར་ཤེས། །གང་འདི་དེ་ཡི་བསྟི་གནས་ན། །རྟག་ཏུ་རིམ་གྲོ་ཡོངས་བསྒྲུབ་པ། །རྔོན་པ་པདྨ་ཅན་ཞེས་པས། །དེ་ཡིས་བདག་ནི་བསྲུང་བར་བཟོད། །ཅེས་པ་ཡིད་ཀྱིས་ངེས་བྱས་ན། །རྔོན་པ་ག་ལ་བ་དེར་སོང་། །རང་གི་སྤྱོད་ཚུལ་བཤད་
སྨཻ་བནྡྷ་རཀྵཱ་མ་ཡཱ་ཙ་ཏ། །སཱ་དྷ་ཀ་སྱ་བ་དྷཱ་ཡཻ་བ་ནཱ་གཱ་རཱ་ཛེ་ན་སོརྠི་ཏཿ། །དྷ་ནྭཱི་ཏཾ་དེ་ཤ་མ་བྷྱེ་ཏྱ་མནྟྲ་བཱི་རཾ་བྱ་ལོ་ཀ་ཡཏ། །ཨསྨིནྣ་བ་ས་རེ་མནྟྲ་སཱ་དྷ་ཀོ་པི་ཧུ་ཏ་ན་ལཿ། །ཨཱ་ཀྲྀ ཥྚིཾ་ནཱ་ག་རཱ་ཛ་སྱ་བི་ད་དྷེ་བནྡྷ་ནཏྶུ་
ནས་ནི། །དེ་ལ་འཆི་བ་བསྲུང་བར་ཞུས། །ཀླུ་ཡི་རྒྱལ་པོས་སྒྲུབ་པ་པོ། །གསད་པའི་སླད་དུ་གསོལ་བཏབ་སྟེ། །གཞུ་དང་ལྡན་པས་གནས་དེར་ཕྱིན། །སྔགས་པས་དཔའ་བོ་མཐོང་བར་གྱུར། །གནས་སྐབས་དེར་ནི་སྒྲུབ་པ་པོ། །སྔགས་པ་ཡིས་ཀྱང་སྦྱིན་སྲེག་མེས། །བཅིང་བར་འདོད་པས་ཀླུ་ཡི་ནི། །རྒྱལ་པོ་རབ་ཏུ་དགུག་པར་
