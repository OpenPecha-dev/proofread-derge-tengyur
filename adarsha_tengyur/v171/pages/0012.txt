བརྠ་སཀྟཱ་ནཱ་མརྠཱརྠེ་པཱ་པ་ཙིནྟ་ནཾ། །ཀྵ་ཎེ་ནཱ་ཡཱནྟི་བིཏྟ་ནི་རཀྵི་ཏཱ་ནྱ་པི་སཾཀྵ་ཡཾ། །མུ་ཧུཿཀརྨྨོ རྨྨི་སཾ་རཾ་བྷ་སཾ་བྷ་བདྐྵོ་བྷ་བི་བྷྲ་མཱཿ། །གཙྪ་རྟྱཿཀེ་ན་བཪྻྱནྟེ་ཤཻ་ལ་ཀུ་
ལ་ཆགས་རྣམས་ནོར་དོན་དུ། །སྡིག་པ་བསམ་པ་རིགས་མ་ཡིན། །ནོར་རྣམས་བསྲུང་བར་གྱུར་མོད་ཀྱང་། །སྐད་ཅིག་གིས་ནི་བརླག་པར་འགྱུར། །ཡང་ཡང་ལས་རླབས་རྩོམ་པ་ཡིས། །རབ་སྐྱེད་འཁྲུགས་པའི་རྣམ་འཕྲུལ་ཅན། །དཔལ་རྣམས་རི་གཟར་ཆུ་བོ་བཞིན། །བགྲོད་པ་
ལྱཱ་ཨི་བ་ཤྲི་ཡཿ། །ཏསྨཱནྣ་མེ་སུ་བྷྲཱུ་བྷྲཱ་ཏྲྀ་ཏྲོ་ཧེ་པྲ་བརྟྟ་ཏེ་མ་ནཿ། །བིཏྟ་བྷྲཾ་ཤེ་སྟི་མེ་བྲྀཏྟིརྦྲྀཏྟ་བྷྲཾ་ཤེ་ཏུ་ཀཱ་ག་ཏིཿ། །ཨི་ཏི་ཏ་སྱ་བྲུ་བཱ་ཎ་སྱ་ནཱ་ནཱ་ཡུཀྟི་ནི་དཪྴ་ནཻཿ། །ཤ་ནཻཿསཱ་ནི་དེ་བྷེ་པ་ཏྱུཿཔཱ་ཏ་
དག་ནི་སུ་ཡིས་བཟློག །དེ་སླད་སྨིན་ལེགས་བདག་གི་ཡིད། །སྤུན་ཟླ་གསད་ལ་ཡོངས་མི་འཇུག །ནོར་ཉམས་བདག་ནི་འཚོ་བ་ཡོད། །སྤྱོད་ཚུལ་ཉམས་ན་འགྲོ་བ་གང་། །ཞེས་སྨྲས་བདག་པོ་དེ་ཡིས་ཡིད། །སྣ་ཚོགས་རིགས་པ་ཉེར་བསྟན་ནས། །དལ་གྱིས་སྡིག་པ་དག་ལ་ནི། དེ་ཡིད་མངོན་
ཀཱ་བྷི་མུ་ཁམྨ་ནཿ། །བརྡྷི་ཏཱནྦ་ཧུ་བྷཿསྣེ་ཧཻཿས་ཧ་ཛནྨཱུརྡྡྷ་ཛཱ་ནི་བ། །ཙྪི་ཏྭཱ་ཧ་རནྟི་ས་ཧསཱ་ཀྵུ་ར་དྷ་རཱ་ཁ་རཱཿསྟྲི་ཡཿ། །བ་ཀྲཱ་པ་རཾ་ཀྲཱུ་ར་ཏ་རཱ་ཀྲི་ཡཱ་སུ་པྲ་བརྟྟ་ནཱ་ཡཻ་བ་དྲྀཌྷཱ་བྷི་ཡོ་ག །སཱ་པཱ་ནི་བཱ་
པར་ཕྱོགས་པར་བྱས། །མཛའ་སྣུམ་མང་པོས་བསྐྱེད་པ་ཡིས། །ལྷན་སྐྱེས་སྤྱི་བོར་སྐྱེས་པ་བཞིན། །བུད་མེད་སྤུ་གྲིའི་སོ་རྣོན་གྱིས། །འཕྲལ་ལ་བཅད་ནས་འདོར་བར་བྱེད། །རྨོངས་པས་བཅོམ་པའི་བློ་དང་བུད་མེད་ནི། །འཁྱོག་ཅིང་གཞན་ལ་རབ་གདུག་བྱ་མིན་ལ། །འཇུག་པའི་སླད་དུ་
