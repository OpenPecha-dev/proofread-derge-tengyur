ཧཾ་ཏུཾ། །ཤྲུ་ཏྭེ་ཏི་བཱ་ཀཾྱ་བྱ་ཐི་ཏསྱ་རཱ་ཛྙ་སྟཱི་བྲ་བྱ་ཐཱཾ་ས་ཏྭ་ནི་དྷིརྣི་ཡ་མྱ། །ཏམ་པྲ་བཱིངྐཱི་རྞྞ་ཛ་ཊཱ་ཀ་ལཱ་པཿསྭ་བྷཱ་བ་ནིརྨྨ་ནྱུ་མ་ནཿཀུ་མཱ་རཿ། །ཨ་ལཾ་ཏ་པ་ཤཱ་པཾ་བྷ་ཡེ་ན་དཱ་ཛ་ནྣ་ཧཾ་བི་མ་ནྱུཿཔི་ཏ་རཽ་མ་
གཞོམ་མི་འོས། །ཞེས་པ་རྒྱལ་པོ་གཟིར་བའི་ཚིག་ཐོས་ན། །རལ་པའི་ཚོགས་བཀྲམ་གཞན་ནུ་རང་བཞིན་གྱིས། །ཁྲོ་མེད་ཡིད་ལྡན་སྙིང་སྟོབས་གཏེར་གྱིས་ནི། །དྲག་པོར་གདུང་བ་བཟད་བྱས་དེ་ལ་སྨྲས། །རྒྱལ་པོ་དཀའ་ཐུབ་དམོད་པས་འཇིགས་དོན་མེད། །ཁྲོ་བྲལ་བདག་དང་
མཱནྡྷཽ། །ཨསྨིནྣ་པི་པྲ་ཎ་ཧ་རཱ་པ་ཀཱ་རེ་ཤཱ་པོ་པ་པཾ་ག་ན་ཀ་རི་ཥྱ་ཏ སྟེ། །སུ་ཁཱ་དུཿཁཱ་ཡ་གུ་ཎོ་ད་ཡཱ་ཡཤྩ་དོ་ཥཱ་ཡ་ཤཱ་པཱ་ཡ་བདྷཱ་ཡ་ཀཱ་ལེ། །ཀརྨྨཱ་ཎྱ་སཾ་བྷྲནྟ་ཕ་ལ་ནི་ཛནྟཽ་སྭ་ཡཾ་ཀྲྀ་ཏཱ་ནྱེ་བ་བྷ་བེ་བྷ་བནྟི། །ཡནྨཱརྒེ་ཎ་
བདག་གི་ཕ་མ་དག །ལོང་བ་སྲོག་འཕྲོག་གནོད་པ་འདི་ལ་ཡང་། །ཁྱོད་ལ་དམོད་པ་དག་ནི་འདོར་མི་འགྱུར། །སྲིད་པར་སྲོག་ཆགས་རང་གིས་བྱས་པའི་ལས། །འཁྲུལ་མེད་འབྲས་བུ་ཅན་རྣམས་དུས་སུ་ནི། །བདེ་དང་སྡུག་བསྔལ་ཡོན་ཏན་རྒྱས་པ་དང་། །སྐྱོན་དང་དམོད་པ་གསོད་པའི་སླད་དུ་འགྱུར། །གང་ཞིག་ལམ་
ཤ་ནཻརྦྲ་ཛནྣི་པ་ཏི་ཏཿཔྲཱསྣོཏི་བྷ་གཾ་ཏ་ནོཪྻཿཤྭ་བྷྲ་པ་རི་བྷྲ་མོཏྤ་ཐ་ག་ཏེརྦྦི དྷྭཱ ནྦི་ཤ་ཏྱཀྟ་ཏཿ། །ཡཏྤཱི་ད་ཏྱ་ས་ཀྲྀཏྤྲ་ཡཏྣ་ཙ་ཏུ་རཿཤྲི་སཾ་ཤྲ་ཡཤྩཱ་ལ་སསྟདྦཻ་ཙི་ཏྱ་མ་བྷིཏྟ་ཙི་ཏྲ་ར་ཙ་ནཾ་སརྦྦཱཏྨ་ནཱཾ་ཀརྨྨ་ཎཱཾ། །ཨ་ཡཤྩི་
ནས་དལ་གྱིས་འགྲོ་བ་ལྷུང་ཞིང་ལུས་ནི་རབ་ཏུ་ཉམས་པ་ཐོབ། །གང་ཞིག་ལམ་བྲལ་གཡང་སར་ཡོངས་སུ་འཁྱམ་ཞིང་རྒྱུག་པ་ཉམས་པ་མེད་པར་འཇུག །གང་ཞིག་ལན་གཅིག་མ་ཡིན་མཁས་པ་ཉམས་ཤིང་དཔལ་བསྟེན་ལེ་ལོ་ཅན། །ལས་ཀྱི་རི་མོ་རྩིས་པ་མེད་པར་བདག་ཉིད་ཀུན་གྱིས་བཀོད་དེ་རྣམ་པར་བཀྲ། །བདག་
