སོ་ཙིནྟ་ཡ་ད་ཧོ་ར་མྱ་རཱ་མཱ་རཏྣ་ས་མཱ་ག་མ། །ཀེ་ནེ་ཡཾ་བིརྒྷཱ་ཛ་ན་ནཱི་ཛ་ན་ནཱི་མ་མ་ནིརྨྨི་ཏཱ། །བྷ་གེརྨྨ་ཛན་ཀསྟཱ་བདྻོ་བ་ནཾ་ནི་དྷ་ནཾ་ག་ཏཿ། །ཨི་ཡཾ་ཏྭ་པུ་ཎྱཻཿཀཾ་ཀཱ་ལཱི་ཀཱ་ལ་བྱཱ་ལཱི་གྲྀ་ཧེ་སྠི་ཏཱ། །གཙྪ་
དེས་བསམས་ཀྱེ་མ་དགའ་བ་ཡི། །བུད་མེད་རིན་ཆེན་དང་འགྲོགས་ལ། །གེགས་ནི་སྐྱེད་པར་བྱེད་པ་ཡི། །བདག་གི་མ་འདི་ཅི་ཡི་སྤྲུལ། །རེ་ཞིག་སྐལ་བས་མ་ནི་བདག །ལང་ཚོ་ལྡན་ཞིང་འཚོ་བར་གྱུར། །བསོད་ནམས་མིན་པས་སྤྲུལ་ནག་ནི། །འཇིགས་རུང་མ་ནི་ཁྱིམ་ན་གནས། །གཉིད་དང་
མྱེ་ཏཱཾ་ས་མུལླཾ་གྷྱ་ནི་ཏྲཱ་བི་ར་ཧི་ཏཱཾ་ཡ་དི། །ཏ་ད་སྱཿཀ་ལ་ཧཱ་མ་བཻརྫ་ནི་པི་པྲ་ཏི་བུ་དྷྱ་ཏེ། །པྲཱ་ཡཿཀྵ་པཱ་པ་རི་ཀྵི་ཎཱ་དཱུ་རེ་ཀཱནྟཱ་ཀ་རོ་མི་ཀིཾ། །སུ་ལ་བྷཱ་བྷཱ་གྱ་ཧཱི་ན་སྱ་ཀ་ཐཾ་སཱ་སུ་བྷ་ག་མ་མ། །ཧརྨྱོཏྶ་ག་ག་ཏཱ་བི་
བྲལ་མ་འདི་ལ་ནི། །གལ་ཏེ་བདག་གིས་འགོམས་ཕྱིན་ན། །འདི་ཡི་འཐབ་མོའི་སྨྲ་བརྗོད་ཀྱིས། །སྐྱེ་བོ་རྣམས་ཀྱང་རབ་སད་གྱུར། །མཚན་མོ་ཕལ་ཆེ་ཡོངས་ཟད་ཅིང་། །མཛེས་མ་རིང་ལ་བདག་ཅི་བྱེད། །སྐལ་བ་དམན་པ་བདག་ལ་ནི། །སྐལ་བཟང་མ་དེ་ཇི་ལྟར་མོད། །ཁང་བཟང་ནང་དུ་སོན་ཞིང་
ལོ་ཀ་ན་ར་སནྣ་མྲཱ་ན་ནཽ་བྷོ་རུ་ཧཱ་པྲསྟ་སྱ་ཤྲ་བ་ཎོཏྤ་ལ་སྱན་ཡ་ནཙྪཱ་ཡཾ་དི་ཤཾ་ཏཱི་པ་ཏེ། །ཏཱ་ཌཾ་ཀ་དྱུ་ཏི་བྷིཿཀ་པོ་ལ་ཡུ་ག་ལེ་པ་ཏྲཱ་བ་ལི་བི་བྷྲ་ཏཱིཾ་བྷཱ་གྱཻརྦྷོ་གྱ་བྷུ་བི་པྲ་ཡ་ཏི་སུ་བྷ་གཱ་བྷ་བྱ་སྱ་སཱ་སུནྡ་རཱི། །ཨི་ཏི་
རྣམ་པར་བལྟ་བའི་རོ་ལས་བཞིན་གྱི་ཆུ་སྐྱེས་དུད། །རྣ་བའི་ཨུཏྤལ་ལྷུང་བར་གྱུར་པས་མིག་གི་མཛེས་པ་རྣ་བ་དག་ལ་སྟེར། །རྣ་རྒྱན་འོད་ཟེར་དག་གིས་འགྲམ་པ་ཟུང་ལ་པ་ཏྲའི་འཕྲེང་བ་རྣམ་པར་འཛིན། །མཛེས་མ་དེ་ནི་ལེགས་ལྡན་དག་གིས་སྐལ་བ་རྣམས་ཀྱིས་སྤྱད་བྱའི་ས་རུ་འགྱུར། །དེ་ལྟར་
