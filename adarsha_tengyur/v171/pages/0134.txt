མཱབྡྷྱཾ་ཀྭ་སཱ་བྷི་བྷཱུ་ཏིཿཀྭ་ཙ་དུརྡ་ནེ་ཡཾ། །ན་དཱིཪྻ་ཏེ་མེ་ཧྲྀ་ད་ཡཾ་ན་ཛཱ་ནེ་ཀེ་ནཱ་སྱ་དཏྟཿཀུ་ལི་ཤོ་བ་དེ་ཤཿ། །ཀྭཱ་པཽ་ཛ་ནསྟྭདྦི་བྷ་བ་ནུ་སཱ་རཱི་ཀུ་ལཱ་ནུ་རཱུ་པོ་བྷ་བ་ནིཤྩ་ལེ་ཡཾ། །ཨེ་ཀཻ་བ་སཏྣཱི་པ་རི་བཱ་ར་ཤེ་ཥཿཀྲྀ ཙྪེུ་པི་སཱ་དྷོ་རི་པ་དྷཻཪྻ་
འདི་གར། །འབྱོར་དེ་གང་དུ་གང་དུ་སྐབས་ངན་འདི། །བདག་སྙིང་མི་འགས་འདི་ལ་སུ་ཞིག་གིས། །རྡོ་རྗེའི་མན་ངག་བྱིན་པ་ཡོངས་མི་ཤེས། །ཁྱོད་འབྱོར་རྗེས་འབྲང་སྐྱེ་བོ་དེ་གང་དུ། །ཁྱོད་ཀྱི་ཆུང་མ་གཡོ་མེད་གཅིག་བུ་འདི། །སྡུག་ཀྱང་དམ་པའི་བརྟན་པའི་སྤྱོད་ཚུལ་བཞིན། །རིགས་ཀྱི་རྗེས་མཐུན་འཁོར་གྱི་ལྷག་
བྲྀཏྟིཿ། །ཨི་ཏྱཱ་ཤྲུ་སཾ་བེ་ག་བི་ཤཱིརྞྞཾ་དྷུ་ལཱ་པེ་ནསྟ་སྱ་བ་ཙོ་ནི་ཤ་མྱེ། །ན་ཏྭཱ་ཀུ་མཱ་རསྟ་མུ་བཱ་ཙ་དྷཱི་ར སྟཱུ རྞྞཾ་ཏ་དཾ་ཀཱ་ད་བ་ཏཱིཪྻ་བྷཱུ་མིཾ། །བི་མུཉྩ་པྲི་ཐྭཱི་པུ་ར་ཧཱུ་ཏ་ཤོ་ཀཾ་ཤུ་ཙཱ་བྷི་བྷཱུ་ཏཱ་ན་བྷ་བནྟི་དྷཱི་རཱཿ།     །ཀྵ་ཡཱ་ཡ་ཛཱ་གརྟྟི་ན་ཏོ་ནུ་བྷཱ་བཱ་མེ་ཥ་སྭ་བྷཱ་
མར་གྱུར། །ཞེས་པ་མཆི་མའི་རྒྱུན་གྱིས་ཡི་གེ་ཉམས། །ཆོ་ངེ་སྨྲ་བ་དེ་ཡི་ཚིག་ཐོས་ནས། །དེ་ཡི་ཕང་ལས་མྱུར་དུ་སར་བབས་ཏེ། །གཞོན་ནུ་བརྟན་པས་དེ་ལ་བཏུད་ནས་སྨྲས། །ས་ཡི་བརྒྱ་བྱིན་མྱ་ངན་རྣམ་པར་ཐོང་། །མྱ་ངན་དག་གིས་ཟིལ་མནན་བརྟན་པ་མིན། །མཐོ་དང་དམའ་རྣམས་བརླག་སླད་ངེས་པར་ནི། །མྱོང་བྱའི་
བོ་བྷ་བི་ཏ་བྱ་ཏཱ་ཡཱཿ། །ཨཻ་ཤྭཪྻ་མཱཤྩཪྻ་སུ་ཁཱ་བ་ཏཾ་སཾ་ལཱ་བཎྱ་ལ་ཀྵྨི་ཏི་ལ་ཀཾ་པ་པུཤྩ། །ཀྵ་ཎེ་ན་ཡཱ་ཏྱེ་བ་ཀྲྀ་ཏཱཾ་ཏ་ནཪྨ་ཀརྨྨོརྨྨི་ནིརྨྨ་ཎ་ཧྲྀ་ཏཾ་ན་རཱ་ཎཱཾ། །བྷ་བེ་ད་བྷཱ་བཱ་ནུ་བྷ་བེ་བྷ་བེ་སྨིན་ས་ཏེ་སྭ་བྷཱ་བོ་ཡ་དི་བྷཱ་བ་བརྒཿ། 
རང་བཞིན་འདི་ནི་མེལ་ཚེ་བྱེད། །མི་རྣམས་དབང་ཕྱུག་ངོ་མཚར་བདེ་བ་ཡི། །རྣ་རྒྱན་ཅན་དང་དཔལ་མཛེས་ཐིག་ལེ་ཅན། །ལུས་ཀྱང་མཐར་བྱེད་རྩེ་བའི་ལས་རླབས་ཀྱི། །སྤྲུལ་པས་ཕྲོགས་པ་སྐད་ཅིག་ཉིད་ཀྱིས་འགྲོ། །གལ་ཏེ་དངོས་མེད་ཉམས་མྱོང་སྲིད་པ་འདིར། །དངོས་ཚོགས་རང་བཞིན་བདེན་པར་གྱུར་པ་ན། །
