ཡོ་ར་སྲ་དྷྲྀ་ཥྱཱ་བ་བྷུ་ཏ་མྦ་ལ་ཤཱ་ལི་ནཽ། །དྭེ་ཥ་སྤརྡྷཱ་བྷི་མཱ་ནེ་ན་སུ་པརྞྞཱ་བ་པི་ཀཱ་ཤྱ་པཱ་ཏ། །གྲྀ་ཧི་ཏྭཱ་ཤཱ་ས་ནཾ་རཏྣ་ཏྲ་ཡཾ་ཤ་ར་ཎ་མཱ་པ་ཏུཿ། །ཏཽ་ནཱ་གཱ་བཱཪྻ་ཛཱ་ཏཱི་ཡཽ་དྷྲྀ་ཏ་
འདབ་བཟང་དག་གིས་ནི། །བརྫི་བར་ནུས་པ་མིན་པར་གྱུར། །ཞེ་ལྡང་འགྲན་ཞིང་མངོན་ཁེངས་པ། །འདབ་བཟང་གིས་ཀྱང་འོད་སྲུངས་ལ། །བསྟན་པ་རིན་ཆེན་བཟུང་ནས་ནི། །དཀོན་མཆོག་གསུམ་ལ་སྐྱབས་སུ་སོང་། །ཀླུ་དེ་དག་ནི་འཕགས་པའི་རིགས། །ཡུལ་འཁོར་
རཱཥྚ་པ་རཱུ་ཌྷ་ཀཽ། །ཀྲཱུ་རོ་སུ་པརྞྞཽ་ཏཱ་པེ་ཏཽ་བི་རཱུ་པཱ་ཀྵ་དྷ་ནཱ་དྷི་པཽ། །ཛི་ན་སྱེ་ཏི་བྲཱུ་བ་ཎ་སྱ་པུ་ཎྱཾ་པྲ་བ་ཙ་ནཱ་མྲྀ་ཏཾ། །ཏདྦ་ནཽ་པཱནྟ་ནི་ལ་ཡཿཤྲུ་ཏྭཱ་ཀེ་ནེ་ཡ་ཀོ་མུ་ནིཿ། །བཱིསྨི་ཏ་སྟཏྤྲ་བྷཱ་བེ་ན་ཏ་མེ་བ་ཤ་ར་
སྲུང་དང་འཕགས་སྐྱེས་པོ། །གདུག་པའི་འདབ་བཟང་དེ་དག་ནི། །མིག་མི་བཟང་དང་ནོར་བདག་འདི། །ཞེས་པ་གསུང་བ་རྒྱལ་བ་ཡི། །བསོད་ནམས་གསུང་རབ་བདུད་རྩི་ནི། །ནགས་ཚལ་དེ་ན་གནས་པ་ཡི། །ཐུབ་པ་ཀེ་ནེ་ཡ་ཀས་ཐོས། །དེ་ཡི་མཐུ་ཡིས་ཡ་མཚན་ཏེ། །དེ་ཉིད་ལ་ནི་

སྐྱབས་སུ་སོང་། །ཕྱིར་མི་འོང་ནི་ཐོབ་གྱུར་ནས། །བདེན་པ་མཐོང་ཞིང་དགའ་བར་གྱུར། །དེ་ཡི་ཚ་བོ་ཤི་ལ་ཡང་། །དེ་བཞིན་འདུལ་བ་ལ་བརྟེན་ནས། །བདེན་པ་མཐོང་ཞིང་རྒྱ་ཆེ་བ། །ཕྱིར་མི་འོང་ཀི་འབྲས་ཐོབ་འགྱུར། །དེ་དག་འཁོར་བཅས་བཅོམ་ལྡན་གྱི། །བསྟན་པའི་ཆོས་ལ་རབ་དང་ཞིང་། །
