ཡི་པཱ་རཱ་བ་ཏྱཱ་པྲཱི་ཏྱཱ་ན་དྭེ་ཥ་ཀརྟུ་མརྷ་སི། །ན་ཧི་ཀཱ་ར་ཎ་རཱ་གེ་ཎ་པྲ་བརྟྟནྟེ་བྷ་བདྭེ་དྷཱཿ། །ཡ་ཐཻ་བཱ་ཡ་ཏ་ཐཻ་བཱ་ཧཾ་ཀོ་བི་ཤི་ཁསྟ་བཱ་བ་ཡཽཿ། །སརྦ་བྷཱུ་ཏ་ས་མཿསནྟཿཀྲྀ་སཱནྣཻ་ཀ་ཏྲ་ཀུརྦྦ་ཏེ། །ཨི་ཏྱཀྟེ་ཏེ་
རོན་ལ་ནི་བྱམས་པ་ཡིས། །བདག་ལ་ཞེ་སྡང་བྱ་མི་འོས། །གང་ཕྱིར་ཁྱོད་ཉིད་ལྟ་བུ་ནི། །སྒྱུ་ལྡན་ཆགས་པས་འཇུག་པ་མིན། །ཇི་ལྟར་འདི་ཉིད་དེ་བཞིན་བདག །ཁྱེད་ལ་ངེད་གཉིས་བྱེ་བྲག་ཅི། །དམ་པ་འབྱུང་པོ་ཀུན་ལ་སྙོམས། །གཅིག་ལ་བརྩེ་བ་བྱེད་པ་མིན། །དེས་འདི་སྨྲས་ཚེ་མི་
ན་ནྲྀ་པ་ཏིཪླཱི་ནཾ་ཀ་ཀ་ན་རཱ་བཱི་ཎཱ། །ནེ་བྷ་ཏ་བྱ་མི་ཏཱི་བཱ་ཧ་ཁ་གཾ་པྲཙྪ་དྱ་པཱ་ཎི་ནཱ། །ས་ཏ་ཏཿསྣིགྦྷ་ཛཱི་མཱུ་ཏ་གྷོ་ཥ་གམྦྷཱི་ར་ཡ་མི་བ། །ཨུ་བཱ་ཙ་སརྦྦ་སཏྭཱརྟྟི་པ་རི་ཏྲཱ་ཎ་ཀྲྀ་ཏ་ཀྵ་ཎཿ། །མ་ཀྲྀ་
བདག་གི །ལག་པ་གདུབ་བུའི་སྒྲ་ལྡན་པས། །འཇིགས་པར་མི་བྱ་ཞེས་སྨྲ་བཞིན། །བྱ་ནི་ཕྱག་གིས་རབ་བསྒྲིབས་ཏེ། །དེ་ནས་སེམས་ཅན་གཟིར་བ་ཀུན། །ཡོངས་བསྐྱབ་དུས་ནི་བྱས་པ་ཡི། །ཆར་སྤྲིན་འཇམ་པའི་སྒྲ་དབྱངས་ལྟར། །ཟབ་པའི་ཚིག་གིས་རབ་སྨྲས་པ། །སྐད་
ཐཿཀྵ་ཎ་ཏྲྀ་སྟྱ རྠཱི་བི་ཥ་མཾ་པྲཱ་ཎི་བཻ་ཤ་མ། །ཏུ་མྱ་བྱ་ཐཱ་བི་ཀཱ་རོ་ཡཾ་པྲཱ་ཎ་སྣེ་ཧཿཤ་རཱི་རི་ཎཱཾ། །པ་ར་སྲཱ་ཎཱ་པ་ཧ་རེ་ཎ་ཡཱ་བྲྀ ཏྟིཿཔ་རི་ཀལྦྱ་ཏེ། །ནི་བྲྀཏྟིཾཿཤྲེ་ཡ་སཱཾ་མ་ཧི་པཱ་པ་ཏཱ་པཾ་པྲ་
ཅིག་ཚིམ་པ་དོན་གཉེར་བས། །མི་བཟད་སྲོག་ཆགས་གསོད་མ་བྱེད། །ལུས་ཅན་སྲོག་ལ་ཆགས་པ་ཡི། །གདུང་བའི་རྣམ་འགྱུར་མཚུངས་པ་ཉིད། །གཞན་གྱི་སྲོག་ནི་ཕྲོགས་པ་ཡིས། །གང་ཞིག་ཡོངས་སུ་འཚོ་བྱེད་པ། །དེ་ཡི་དགེ་ལེགས་ལྡོག་འགྱུར་ཏེ། །གང་ཕྱིར་སྡིག་པའི་
