ཡཿ། །པྲ་ཡ་ཡཽ་རཱ་ཛ་ཏཾ་སཱ་ནཱཾ་སཾ་ཏྲཱ་སཱ་ཡཱ་ས་ཧེ་ཏུ་ཏཱཾ། །དཱུ་ར་པྲ་ཡཱ་ཏེ་ད་ཡི་ཏེ་བི་ར་ཧེ་ཎ་མ་ནོ་ཧ་རཱ། །བ་བྷཱུ་བ་བཱ་ལ་ན་ལི་ནཱི་པ་ལཱ་ཤ་ཤ་ཡ་རཱ་ཤ་ཡཱ། །སོཏྐཎྛཱ་ཡཱ་དི་བ་པ་ག་ཎ་ནཱ་རཾ་བྷ་ནི་ཏྱཱ་བྷི་ཡཽ་གེ་སཾ་ཁྱཱ་རེ་ཁཱ་ཤ་རི་ཎི་མ་
པ་ཡིས། །ངང་པའི་རྒྱལ་པོ་སྐྲག་པ་དང་། །ཉོན་མོངས་རྒྱུ་ཉིད་དག་ཏུ་གྱུར། །ཁྱོ་ནི་རིང་དུ་སོང་བ་ན། །འབྲལ་བ་ཡིས་ནི་ཡིད་འཕྲོག་མ། །པདྨ་གཞོན་ནུའི་འདབ་མ་ཡིས། །མལ་སྟན་ལ་ནི་བརྟེན་པར་གྱུར། །སྲེད་ལྡན་ཉིན་པར་བགྲང་བའི་རྩོམ་པ་དག་ལ་རྟག་ཏུ་མངོན་སྦྱོར་ཞིང་། །གྲངས་ཀྱི་རི་མོའི་
བ་ནཽ་ཀམྤ་ལོ་ལཾ་ལི་ཁཾ་བྷྱཿ། །ཏ་སྱཿཔཱ་ཎཽ་ནི་པ་ཏི་ཏ་ར་ཎཏྐཾ་ཀ་ཎེ་ཏཱ་ན་བེ་ན་ཀྵི་པྲཾ་མུཀྟཱ་བ་ལ་ཡ་ཀ་ལ་ནཱ་མ་ཤྲུ་དྷཱ་རཱ་ཙ་ཀཱ་ར། །དྭེ་ཥཿཔུཥྤ་ཤ་རེ་སུ་ཁེ་བ་མུ་ཁ་ཏཱ་དེ་ཧེ་ཡི་ནི་སྣེ་ཧ་ཏཱ་པ་ཏྱཽ་དྷྱཱ་ན་པ་རཱ་ཡ་ཎ་ཏྭ་མ་ནི་
ལམ་ནི་ས་ལ་འདར་ཞིང་གཡོ་བས་འདྲི་བྱེད་པ། །དེ་ཡི་ལག་པ་ཕྲ་བ་ཉིད་ཀྱིས་སྒྲ་ལྡན་གདུ་བུ་ལྷུང་བ་ལ། །མཆི་མའི་རྒྱུན་གྱི་མུ་ཏིག་གདུབ་རྐོར་དག་ནི་མྱུར་དུ་གནས་པར་བྱས། །མེ་ཏོག་མདའ་ཅན་ལ་སྡང་བདེ་ལ་ཕྱིར་ཕྱོགས་ལུས་ལའང་ཆགས་པ་ཉིད་མ་ཡིན། །བདག་པོའི་བསམ་གཏན་སྐྱོང་
ཤཱ་ཏནྣཱ་མ་མནྟྲོ་ཛ་པཿ། །ཤ་ཡྻཱ་བྷཱུ་མི་ཏ་ལཾ་ཏ་ཐཱ་པི་སུ་ཏ་ནོསྟཱ་པཀྵ་ཏིརྣཱ་བྷ་བ་ནཱུ་ནཾ་ནིཤྩ་ལ་ལཱི་ན་རཱ་ག་མ་ན་སཱཾ་མུཀྟཱི་ཙ་ཏཱི་པྲ་བྲ་ཏཻཿ། །ལཱི་ནཱ་སྥ་ཊི་ཀ་བཪྻཾ་ཀཻ་ཧ་རི་ཙན་ད་ན་པཱཎྜ་
ཞིང་ཉིན་མཚན་དུ་ནི་དེ་ཡི་མིང་གི་གསང་སྔགས་ཟློས། །མལ་སྟན་ས་གཞིར་བྱས་ཏེ་དེ་ལྟ་ན་ཡང་གཟུགས་བཟང་གདུང་བ་ཉམས་མ་གྱུར། །འདོད་ཆགས་གཡོ་མེད་ཆགས་པའི་ཡིད་རྣམས་བརྟུལ་ཞུགས་དྲག་པོས་གྲོལ་བ་མིན་ནམ་ཅི། །ཤེལ་གྱི་ཁྲི་སྟན་ལ་འདུག་ཅིང་། །ཧ་རི་ཙན་དན་གྱིས་སྐྱེ་
