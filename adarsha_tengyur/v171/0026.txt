ཙ་རི་ཏཱཾ་རཀྵ་ཎཱདྦོ་དྷེརྣ་པྲཱཔྟཾ་ཏྭ་དྷི་ཀཾ་ཏ་ཏཿ། །པུ་རཱ་བྷ་བངྒྲྀ་ཧ་པ་ཏིརྡྷ་ན་བཱནྣཱ་མ་ཀརྦྦ་ཊེ། །ཤྲཱི་མཱནྣཱ་མ་སུ་ཏསྟ་སྱ་བ་བྷུ་བཱ་སྭསྠ་བི་ཀྲ་ཧཿ། །བཻ་དྱ་སྟིཀྟཱ་མུ་ཁོ་ནཱ་མ་ཏཾ་བ་ཧུ་དྲ་བི་ན་ཤ་ཡཱ། །སྭཱསྠྱནྣི་ནཱ་
དེ་ལས་ལྷག་པ་བྱང་ཆུབ་ཀྱིས། །བསྲུངས་པ་ཡིས་ནི་ཐོབ་མ་གྱུར། །ཁྱིམ་བདག་ནོར་ལྡན་ཞེས་བྱ་བ། །ཀར་པ་ཊར་ནི་སྔོན་བྱུང་སྟེ། །དཔལ་ལྡན་ཞེས་བྱ་དེ་ཡི་བུ། །ལུས་ནི་རྣལ་གནས་མིན་པར་གྱུར། །དེ་ནི་སྨད་པ་ཁ་བའི་བཞིན། །ཞེས་པ་ནོར་མང་རེ་བ་ཡིས། །རྣལ་གནས་བྱས་ཀྱང་
ཡ་ཏཏྤི་ཏྲཱ་དཏྟཾ་ཡཱསྨེ་ན་ཀིཉྩ་ན། །ཀཱ་ལེ་ནཱ་ཏུ་ར་ཏཱཾ་ཡཱ་ཏཿསྭཱསྠཾ་ཏེ་ན་པུ་ནཿཔུ་ནཿ། །ས་ཏཱི་ཏསྟཏྤི་ཏུཿཀིཉྩཏྤྲཱཔྟཾ་ནངྦཱརྟྟི་བྷཱ་ཥི་ཏཾ། །ཨ་མཪྵ་ཛྭ་ར་སནྟ་པྟསྟྲྀཥྞཱ་ཏ་ར་ལི་ཏཿསྭ་བ ན། །སོ་ཙིནྟྲ་ཡདྦྷི་
དེ་ཡི་ཕས། །དེ་ལ་ཅི་ཡང་བྱིན་མ་གྱུར། །དུས་ཀྱིས་གཟིར་བ་ཉིད་གྱུར་ཏེ། །ཡང་ཡང་དེ་ཡིས་བོས་པར་བྱས། །དེ་ཡིས་ཕ་ལས་ཅུང་ཟད་ཀྱང་། །ཁས་བླང་སྨྲས་པ་ཐོབ་མ་གྱུར། །སྨན་པ་ཁྲོ་བའི་རིམས་ཀྱིས་གདུངས། །སྲེད་པས་གཡོ་བའི་ཤུགས་རིངས་ཅན། །ནད་པ་ཡིས་ཀྱང་
ཥགྔུཿཁཱ་དཱ་ཏུ་རེ་ཎཱ་པི་ལཾ་གྷི་ཏཿ། །ཨ་ཧོ་བ་ཏཱ་ཧཾ་དྷཱུརྟེ་ན་བཱ་ཧི་ཏཿས་ར་ལཱ་ཤ་ཡ། །ཀི་ཀ་རོ་མྱ་དྷུ་ནཱ་ཧ་སྟཱཾ་ག་ཏོ་མེ་ནི་དྷི་རཱ་ཏུ་རཿ། །ཀ་ཊུ་ཀཽ་ཥ་དྷི་བདྦཻ་དྱཿཔུརྦྦ་མཱརྟྟ་སྱ་སམྨ་ཏཱཿ། །ཀི་ཙིཏྤི་ཏཱ་
འགོངས་པ་དེས། །སྔུག་བསྔལ་ལས་ནི་རབ་བསམས་པ། །ཀྱེ་མ་བསམ་པ་དྲང་པོ་དག །གཡོ་དང་ལྡན་པ་བདག་གིས་བསྡུས། །བདག་གི་ལག་ནས་ནད་པ་ཡིས། །གཏེར་འདས་ད་ནི་ཅི་ཞིག་བྱ། །ཚ་བའི་སྨན་བཞིན་སྨན་པ་ནི། །ནད་པ་རྣམས་ཀྱིས་སྔོན་དུ་བསྐུར། །གདུང་བ་བསོས་
