ཏ་སྱ་དེ་ཧཱནྟེ་རུ་ཙི་རཱ་ཙཱ་བི་དྷཱ་ཡི་ནཱ། །མ་ཡཱ་ཏཏྟུ་ལྱ་བྷཱ་བཱཔྟྱཻ་པྲ་ཎི་དྷཱ་ན་མ་ཏིཿཀྲྀ་ཏཱ། །ཏེ་ནཱ་ཧཾ་པྲ་ཎི་དྷཱ་ནེ་ན་ཏཏྶ་མ་བྷྱ་ཙྩཱ་ནེ་ན་ཙ། །ས་ཤཱ་ཎ་བསྟྲཿསཾ་ཛཱ་ཏཿཤཱ་ཎ་བཱ་པཱི་ཏི་བི་ཤྲུ་ཏཿ། །ཨི་
ལུས་ཀྱི་མཐར། །མཛེས་པའི་མཆོད་པ་བསྒྲུབས་བྱས་ཏེ། །དེ་མཚུངས་ཉིད་ནི་ཐོབ་པའི་སླད། །བདག་གིས་སྨོན་ལམ་བློ་གྲོས་བྱས། །སྨོན་ལམ་དེ་དང་དེ་ལ་ནི། །ཡང་དག་མཆོད་པ་ཡིས་ཀྱང་བདག །ཤཱ་ནའི་གོས་དང་ལྡན་པར་སྐྱེས། །ཤཱ་ནའི་གོས་ཅན་ཞེས་སུ་གྲགས། །ཞེས་
ཏྱཀྟྭཱ་ས་ཤ་ནཻ་རྒཙྪ ནྣ་བཱཥྤ་མ་ཐུ་རུཾ་པུ་རིཾཱ། །ཨུ་རུ་མུཎྜཱ་བྷི་དྷཾ་ཤེ་ལ་མཱ་རུ་རོ་ཧ་མ་ཧོདྻ་མཿ། །ཏ་ཏྲོ་པ་པི་ཤྱ་པཪྻཾ་ཀ་བནྡྷེ་ནཱ་ཀམྤི་ཏ་ཀྵི་ཏིཿ། །བི་ནཱི་ཡ་བིགྷྣ་པནྣ དྡྷོ་ནཱ་གཽ་ཏུ་མུ་ལ་བཪྵི་ཎཽ། །མ་ཐུ་
བརྗོད་བཅོམ་བརླག་གྲོང་ཁྱེར་དུ། །ཕྱིན་ཅིང་དལ་གྱིས་འགྲོ་བ་དེས། །མགོ་ཆེན་ཞེས་པའི་རི་བོ་ལ། །རྩོམ་པ་ཆེན་པོ་དག་གིས་འཛེག །དེར་ནི་ས་གཞི་འདར་བྱེད་པའི། །སྐྱིལ་ཀྲུང་བཅས་ཏེ་ཉེ་བར་འཁོད། །ཆར་ཆེན་འབེབས་པའི་ཀླུ་དག་ནི། །བགེགས་ལ་ཆས་པ་རྣམ་པར་བཏུལ། །བཅོམ་
རཱ་བཱ་སི་ནཽ་ཤྲེཥྛི་སུ་ཏཽ་ན་ཊ་བྷཊྚཱ་བྷི དྡྷཽ། །ཨཱ་དཱ་ཡ་ས་ཧྱ་ཀྲྀ་ཏྱཱ་ཡ་ས་བི་ཧཱ་རཾ་བི་ནིརྨྨ་མེ། །རཏྣོ་ཛྭ་ལ་སྥ་ཊི་ཀ་ཀཱཉྩ་ན་ར་མྱ་ཧརྨེཿཔཪྻཾ་ཀ་པཱི་ཐ་ར་ཙ་ནཱ་ཤ་ཡ་ནཱ་བྷི་རཱ་མཿ། །སརྒོ་པ་མཿསྭ་བྷག་བྷོ་ག་ཎི་
བརླག་ལ་གནས་ཚོང་དཔོན་བུ། ན་ཊ་བྷ་ཊ་ཞེས་བྱ་བ། །གྲོགས་ཀྱི་སླད་དུ་བཟུང་ནས་ནི། །དེ་ཡི་གཙུག་ལག་ཁང་པར་སྤྲུལ། །རིན་ཅེན་རབ་འབར་ཤེལ་དང་གསེར་གྱི་བ་གམ་དག་དང་ལྡན། །ཁྲི་དང་ཁྲིའུ་མལ་སྟན་དག་གིས་སྤྲས་ཤིང་མངོན་པར་དགའ། །མཐོ་རིས་ལྟ་བུ་སྐལ་བཟང་ལོངས་སྤྱོད་
