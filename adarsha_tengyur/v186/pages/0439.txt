ཆུ་དང་འོ་མ་དེ་ལས་གཞན་པ་སྲ་བ་ལ་སོགས་པའི་རང་བཞིན་དེ་ལས་ཐ་དད་པ་དེའི་ཕྱིར་རོ། །སྲ་བའི་རང་བཞིན་ལས་ཐ་དད་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །རྒྱུ་ལ་རྟེན་པའི་གཅིག་པ་ཉིད་བརྗོད་ནས། འབྲས་བུ་ལ་བརྟེན་པ་ཉིད་བརྗོད་པ་ནི། འབྲས་བུ་ཡང་ལྡོག་པ་གཅིག་པས་
གཅིག་པ་ཉིད་དོ་ཞེས་འབྲེལ་ཏོ། །དོན་དམ་པ་ཉིད་དུ་ཅི་སྟེ་ཐ་མི་དད་པ་མ་ཡིན་ཞེ་ན། དོན་དམ་པར་ཞེས་བྱ་བའོ། །དེ་སྐད་ཅེས་བྱ་བ་ནི་ཐ་མི་དད་པར་རོ། །ཅི་ལྟར་ཆུ་དང་འོ་མ་དག་རང་བཞིན་གང་གིས་བརླན་པའི་རྟོགས་པའི་རྟེན་ཡིན་པའི་རང་བཞིན་དེ་ཉིད་ཀྱིས་རུལ་བའི་འབྲས་བུ་བྱེད་པ་
དེའི་རྟེན་ཡིན་ལ། རང་བཞིན་དེ་ཉིད་ཀྱིས་ན། དེ་དག་འབྲས་བུ་རུལ་བ་ལས་རྟོགས་པ་ཡིན་པ་དེ་བཞིན་དུ། མེ་ཐ་དད་ཀྱང་ཆ་གང་གིས་དེ་མེའི་རྟོགས་པའི་རྒྱུ་ཉིད་ཀྱིས་རང་བཞིན་གྱིས་དེ་ལས་གཞན་ལས་ལྡོག་པས་མར་རྟོགས་པའི་རྟེན་ཡིན་ལ། རང་བཞིན་དེ་ཉིད་ཀྱི་བྱ་བ་བྱེད་
པས་དེའི་རྟེན་ཡིན་ལ། རྟགས་དུ་བ་ལས་རྟོགས་པ་ཡང་རང་བཞིན་དེ་ཉིད་ཀྱིས་ཡིན་ཏེ། གཞན་དུ་ནི་མ་ཡིན་ནོ། །ཅི་ལྟར་ན་རང་བཞིན་དེ་ཉིད་ཀྱིས་དུ་བ་ལས་རྟོགས་པ་ཡིན་ཞེ་ན། མེར་ཤེས་ཡན་ལག་ཅེས་བྱ་བའོ། །མེར་ལྷག་པར་ཞེན་པའི་ཤེས་པ་སྒྲུབ་པར་བྱེད་པ་ཁོ་ན་བུད་ཤིང་གི་ཁྱད་
པར་གྱི་ལྷན་ཅིག་བྱེད་པ་གཞན་ལ་ལྟོས་པ་ལས་དུ་བ་འབྱུང་བའི་ཕྱིར་ན། དུ་བ་མེ་དེ་ལས་འཁྲུལ་པ་མེད་པ་ཅན་ཡིན་ནོ། །དེས་ན་རང་བཞིན་དེ་ཉིད་ཀྱིས་དུ་བ་ལས་མེ་རྟོགས་པ་ཡིན་ནོ། །ཅི་ལྟར་དུ་བ་རང་གི་རྒྱུ་ལ་མི་འཁྲུལ་པ་དེ་བཞིན་དུ་འབྲས་བུ་གཞན་ཡང་དེ་བཞིན་ཏེ། མི་འཁྲུལ་པར་གྲུབ་པ་
ཡིན་ནོ། །འཆད་པ་ནི་དེ་བཞིན་ཞེས་བྱ་བའོ། །ཐ་དད་ཀྱང་ཆུར་རྟོགས་པ་ཐ་མི་དད་པའི་རྟེན་ཉིད་ཅི་ལྟར་ཞེ་ན། དེ་ལས་གཞན་རློན་པ་མ་ཡིན་པ་དག་ལས་ལྡོག་པས་ནའོ། །དེ་ལས་གཞན་དག་ཅེས་བྱ་བ་མེ་མ་ཡིན་པར་རྟོགས་པའི་རྒྱུ་དག་ལས་ལྡོག་པ་ཉིད་ཀྱིས་སོ། །དེ་ཉིད་ཀྱིས་ཞེས་བྱ་བ་མེར་
རྟོགས་པའི་རྒྱུ་ཉིད་ཀྱིས་དུ་བ་ལས་མེ་རྟོགས་པ་ཡིན་ནོ། །ཅིའི་ཕྱིར་ཞེ་ན། གཞན་དུ་ཞེས་བྱ་བའོ། །རྣམ་འགྲེལ་གཉིས་པ་རྣམ་པར་འགྲེལ་པ་ནི། མེར་ཞེས་བྱ་བའོ། །ཅི་ལྟར་དུ་བ་མི་འཁྲུལ་ཏེ། བརྒྱ་བྱིན་གྱི་སྤྱི་བོ་ལ་སོགས་པ་ལས་ཀྱང་འབྱུང་བའི་ཕྱིར་རོ་ཞེ་ན། འདི་ལྟར་ཞེས་བྱ་བའོ། །དུ་བ་
