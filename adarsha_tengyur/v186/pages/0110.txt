བ་དང་ཞེས་བྱ་བའོ། །བྱ་བ་ཙམ་མི་སྲིད་པའི་ཕྱིར་བྱ་བའི་ཁྱད་པར་སྲིད་པ་མ་ཡིན་ཏེ། དེས་ན་ཡུལ་གཞན་དང་གཞན་དུ་སྐད་ཅིག་འདྲ་བ་གཞན་སྐྱེ་བ་ཉིད་བྱ་བ་ཡིན་ནོ་ཞེས་དགོངས་པའོ། །འོན་ཏེ་འདི་ཡང་རྫས་ཁོ་ན་ལ་སྲིད་ཀྱི་ཡོན་ཏན་དག་ལ་ནི་མ་ཡིན་ནོ་ཞེ་ན། དེ་འདྲའི་ཞེས་བྱ་བའོ། །སྒྲ་ལ་དེ་
ལྟར་ཅི་ལྟར་འགྱུར་ཞེ་ན། གཞན་དག་ཅེས་བྱ་བའོ། །འདི་ནི་གཞན་ཡང་འདོད་པ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །རྫས་མེད་པའི་ཕྱིར་ཡོན་ཏན་ཡང་མེད་དེ། དེས་ན་ནི་འབྱོར་བར་འདུ་བས་ནི་འདོད་པ་མ་ཡིན་ནོ་ཞེས་བརྗོད་པ་ནི། གཞན་ཡང་ཞེས་བྱ་བའོ། །ཡོན་ཏན་ཅན་ནི་རྫས་སོ། །འབྱོར་པར་
འདུ་བས་ཁྱེར་བར་ཁས་བླངས་ནས་ཀྱང་ཉེས་པ་བརྗོད་པ་ནི། གལ་ཏེ་ཞེས་བྱ་བའོ། །གཞན་གྱིས་སྨྲས་པ་ནི་གནས་པ་ལ་ཞེས་བྱ་བ་ནས་ནམ་མཁའི་ཡིན་ནོ་ཞེ་ནའི་བར་རོ། །གལ་ཏེ་གནས་པ་ལ་གེགས་མེད་པའི་ཕྱིར་ན། འབྱུང་བ་ཆེན་པོ་ནམ་མཁའ་གནས་པར་རྟོག་ན། དེའི་ཚེ་འདི་མ་
འབྲེལ་ཏེ། ཅིའི་ཕྱིར་ཞེ་ན་ཕན་ཚུན་ཞེས་བྱ་བ་ལ་སོགས་པའོ། །དེ་ཉིད་སྒྲུབ་པ་ནི་གལ་ཏེ་ཞེས་བྱ་བའོ། །དེས་ན་མེད་པ་ཙམ་ཉིད་ནམ་མཁའ་ཡིན་པར་འགྱུར་རོ་ཞེས་བརྗོད་པ་ནི། འབྱུང་ཚེ་གཞན་མེད་པ་ཙམ་ཞེས་བྱ་བའོ། །གཞིའི་ངོ་བོ་ཉིད་ཁས་བླངས་ནས་ཀྱང་ཉེས་པ་བརྗོད་པ་ནི། ཅི་
ལྟར་ནམ་མཁའ་ཞེས་བྱ་བའོ། །ནམ་མཁའ་ནི་རླུང་གི་ཡང་གཞི་མ་ཡིན་ནོ་ཞེས་བརྗོད་པ་ནི་འགྲོ་བ་ཞེས་བྱ་བའོ། །གེགས་བྱེད་པ་དང་སྟེ་ལུས་ཅན་དང་བྲལ་བས་ཡིན་ནོ། །ལུས་ཅན་མ་ཡིན་པས་བྱས་པ་ཡིན་གྱི་ནམ་མཁས་བྱས་པ་ནི་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏེ། གཞན་དག་ཐུག་པ་མེད་པའི་ཉེས་
པ་སྤོང་བ་ནི། འོན་ཏེ་ཞེས་བྱ་བའོ། །འདི་ལ་ནི་ནམ་མཁའ་ལའོ། །ལན་ནི་གཟུགས་མེད་པ་དང་ཞེས་བྱ་བའོ། །དེ་ཉིད་ནི་བདག་གོ། །གཞན་གྱི་ནི་སྒྲའིའོ། །འོན་ཏེ་གང་ཞིག་བདག་མི་འདོད་པ་དེའི་འདོད་པ་ལ་ལྟོས་ཏེ་ནམ་མཁའ་སྟོན་པ་ཡིན་ནོ་ཞེ་ན། གང་ལ་ཞེས་བྱ་བའོ། །དེ་འདྲ་བ་ཞེས་བྱ་བ་
ནི་ནམ་མཁའ་དང་འདྲ་བའོ། །གང་ལ་བདག་གིས་དོན་དེ་བྱེད་པ་མ་ཡིན་པ་དེ་ལ་ནམ་མཁས་ཀྱང་དོན་དེ་བྱེད་པ་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་ཏོ། །གང་ལ་བདག་མེད་པ་དེས་གདོན་མི་ཟ་བར་ནམ་མཁའ་ཁས་བླང་བར་བྱ་བ་ཡང་མ་ཡིན་ཏེ། དེ་ལྟར་མ་ཡིན་པར་ཡང་ནམ་མཁའི་བྱ་བ་འགྲུབ་པའི་
