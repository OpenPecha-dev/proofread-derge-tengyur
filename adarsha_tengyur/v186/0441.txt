བྱེད་པའི་རྒྱུས་བྱས་པ་གང་ཡིན་པ་དེ་ཉིད་ཀྱི་རྣམ་པར་དབྱེ་བ་བྱས་པ་ཡིན་ནོ། །དེ་ལྟར་འབྲས་བུའི་གཏན་ཚིགས་མེད་ན་མི་འབྱུང་བ་ཉིད་ཡིན་མོད། རང་བཞིན་གྱི་གཏན་ཚིགས་གོ་ཅི་ལྟར་ཞེས་དོགས་པ་ནི། རང་བཞིན་ཞེས་བྱ་བའོ། །འདིའི་མེད་ན་མི་འབྱུང་བ་ཡང་རང་ཉིད་ཅི་
ལྟ་བའི་ཚད་མ་ལ་ལྟོས་ཏེ་དེ་རྟོགས་པར་བྱེད་པ་ཡིན་ནོ་ཞེས་ལན་འདེབས་པ་ནི། ཅི་ལྟར་ཞེས་བྱ་བའོ། །དེ་ལྟར་འབྲས་བུ་དང་རང་བཞིན་གྱི་གཏན་ཚིགས་གཉིས་ཀྱི་མེད་ན་མི་འབྱུང་བ་ཉིད་བརྗོད་ནས་སྐབས་ཉིད་དང་སྦྱོར་བ་ནི། གང་གི་ཕྱིར་དེ་ལྟར་རྗེས་སུ་འགྲོ་བ་ཙམ་མམ་ལྡོག་པ་ཙམ་གྱི་རྒྱུ་དང་
འབྲས་བུའི་དངོས་པོ་མ་ཡིན་པ་དེའི་ཕྱིར་རོ། །བདག་དང་སྲོག་ལ་སོགས་པ་ནི་རྒྱུ་དང་འབྲས་བུའི་དངོས་པོ་མ་ཡིན་པའི་ཕྱིར་སྲོག་ལ་སོགས་པ་ནི། བདག་གོ་བར་བྱེད་པ་མ་ཡིན་ནོ། །སློབ་དཔོན་ཕྱོགས་ཀྱི་གླང་པོའིའོ། །ལྡོག་ཅན་ཞེས་བྱ་བ་ནི་སྲོག་ལ་སོགས་པའོ། །མཇུག་སྡུད་པ་
ནི་དེ་ལྟར་ན་རེ་ཞིག་ཅེས་བྱ་བའོ། །ཅིའི་ཕྱིར་ཞེ་ན། ལྡོག་པ་ཅན་ཞེས་བྱ་བའོ། །དེ་ལྟ་ན་ཡང་ཅི་ལྟར་གཏན་ཚིགས་མ་ཡིན་ཞེ་ན། དེ་ཡང་ཞེས་བྱ་བ་སྟེ། མཉན་བྱ་ཉིད་ཀྱང་ངོ་། །དེ་ལྟར་ན་ཡང་མ་ངེས་པ་ཅི་སྟེ་ཡིན་ཞེ་ན། འདི་ལྟར་ཞེས་བྱ་བའོ། །འོ་ན་ཅི་ལྟར་ལྡོག་པ་ཅན་དུ་གསུངས་
ཤེ་ན། མ་མཐོང་བ་ཙམ་གྱི་ཞེས་བྱ་བའོ། །གལ་ཏེ་ཕུང་པོ་གཞན་མེད་ན་ཅི་ལྟར་འདི་བསྒྲུབ་པར་བྱ་བ་དང་། ཅིག་ཤོས་དག་ལ་ཐེ་ཚོམ་ཟ་བའི་རྒྱུ་ཡིན་ཞེ་ན། རྟག་པ་དང་ཞེས་བྱ་བ་སྟེ་མཉན་བྱ་ཉིད་ལ་ལྟོས་ནས་སོ། །བདག་དང་ཞེས་བྱ་བ་ནི་སྲོག་ལ་སོགས་པ་དང་ལྡན་པ་ལ་ལྟོས་ནས་སོ། །
དེ་ལྟ་མ་ཡིན་ན་ཞེས་བྱ་བ་ནི། དངོས་པོ་དང་དངོས་པོ་མེད་པ་དག་གི་འགལ་བ་དང་དངོས་པོ་ཉིད་མ་ཡིན་ནའོ། །གལ་ཏེ་ཕུང་པོ་གཞན་མེད་ཀྱང་ཅི་ལྟར་འདི་ཐེ་ཚོམ་གྱི་རྒྱུ་ཡིན་ཏེ། ཕྱོགས་གཉི་ག་ལ་མི་སྲིད་པའི་ཕྱིར་རོ་ཞེ་ན། ཕྱོགས་གཉི་ག་ལས་ཞེས་བྱ་བའོ། །ཅིའི་ཕྱིར་ཞེ་ན། གལ་
ཏེ་ཞེས་བྱ་བའོ། །གལ་ཏེ་མཉན་པར་བྱ་བ་ཉིད་ཐམས་ཅད་ལས་ལྡོག་ན། དེའི་ཚེ་འདི་མེད་པ་ཁོ་ནར་འགྱུར་རོ། །ཅིའི་ཕྱིར་ཞེ་ན། རྟེན་མེད་པའི་ཕྱིར་རོ། །རྟེན་ནི་རྟག་པའམ་མི་རྟག་པ་ཞིག་ཡིན་ཏེ། དེ་གཉི་ག་ལས་ལྡོག་པ་ནི་རྟེན་མེད་པ་ཡིན་ནོ། །སྒྲ་ཉིད་ཅེས་བྱ་བ་ནི། གལ་ཏེ་ཡང་
