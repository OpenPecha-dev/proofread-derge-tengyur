གཏོར་མ་པྲིཥྚི་ཀ་དང་པྲ་ཏི་སྭ་ར། ཚེས་བཅུ་བཞིའམ་བཅུ་གཉིས་ལ་གཟའ་ལག་གི་ཉི་མ་ལ་མཆོད་པ་བྱའོ། །ཨོཾ་ཧྲཱིཾ་ཤྲཱི་ཧཱུཾ་ཨཿཛཱ་ཡཱནྟི་བིགྷྣཱཾ་བི་ནཱ་ཤཱ་ཡ་ཕཊ་སྭཱ་ཧཱ། ཞིང་སྐྱོང་གིའོ། །ཨོཾ་བེཥྞ་བི་ལོ་ཀཾ་ཨ་བ་ཏ་ར་ཨ་བ་ཏ་ར་ཤཱི་གྷྲཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའོ། །ཨོཾ་ཕེཾ་ཕཊ་བེཥྞ་བི་

ཙཱུ་ཏའི་ཤིང་དང་། ཀླུ་རིགས་ལྡན་སྔོན་པོའོ། །ཚེས་ལྔ་འམ་བཅུ་གསུམ་ལ་གཟའ་བླ་མ་ལ་མཆོད་པ་བྱའོ། །གཏོར་མ་བ་དན་ལྔས་མཚན་པའོ། །ཨོཾ་ཧྲཱིཾ་ཤྲཱིཾ་ཧཱུཾ་ཧཿཨེ་ཀཾ་བྷ་ཀ་བིགྷྣཱཾ་བི་ནཱ་ཤཱ་ཡ་ཕཊ་སྭཱ་ཧཱ། ཞིང་སྐྱོང་གིའོ། །ཨོཾ་ཏ་བཱ་རཱ་ཧྲི་བྷུ་ལོ་ཀཾ་ཨ་ཝ་ཏ་ར་ཨ་བ་ཏ་ར་ཤཱི་གྷྲཾ་ཧཱུ་
ཕཊ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའོ། །ཨོཾ་ཕེཾ་ཕཊ་བཱ་རཱ་ཧྱེ་ན་མཿསྭཱ་ཧཱ། རྩ་བའི་སྔགས་སོ། །ཨོཾ་ཧཱུཾ་བི་ཛ་ཡ་ཙཎྜེ་སྭཱ་ཧཱ། འཇིགས་བྱེད་ཀྱིའོ། །ལྷག་མ་རྣམས་ནི་སྔོན་མ་བཞིན་ནོ། །ནུབ་ཏུ་དབང་ལྡན་མ་གུར་ཀུམ་གྱི་མདོག་ལྟ་བུའོ། །པ་ཡིག་གི་ས་བོན་ལས་སོ། །ཞིང་སྐྱོང་གི་ཆུ་
བདག་གིའོ། །རུ་རུ་གྷཻ་ར་བ་སེར་པོ་ལ་ཉའི་གཏོར་མའོ། །ཨ་ཤྭཏྠའི་ཤིང་དང་། ཀླུ་སྟོབས་ཀྱི་རྒྱུ་སྔོ་བསངས་ལ། ཚེས་དྲུག་གམ་བཅུ་བཞི་ལ་གཟའ་པ་བ་སངས་ཀྱི་ཉི་མ་ལ་མཆོད་པ་བྱའོ། །ཨོཾ་ཧྲཱིཾ་ཤྲཱིཾ་ཧཱུཾ་ཧཿར་ཎ་སྱཾ་བིགྷྣཱཾ་བི་ནཱ་ཤཱ་ཡ་ཕཊ་སྭཱ་ཧཱ། ཞིང་སྐྱོང་གིའོ། །ཨོཾ་པྲ་མ་ཧ་ན་ཏྲཱི་སུ་
ར་ལོ་ཀཾ་ཨ་བ་ཏ་ར་ཨ་བ་ཏ་ར་ཤཱི་གྷྲཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའོ། །ཨོཾ་ཕེཾ་ཕཊ་ཨིནྡྲ་ཎྱེ་ན་མཿསྭཱ་ཧཱ། རྩ་བའི་སྔགས་སོ། །ཨོཾ་ཧཱུཾ་གྷ་ར་ཙཎྜེ་སྭཱ་ཧཱ། འཇིགས་བྱེད་ཀྱིའོ། །ལྷག་མ་རྣམས་ནི་སྔོན་མ་བཞིན་དུའོ། །རླུང་དུ་ཙཱ་མུཎྜཱི་དམར་མོའོ། །ཡ་ཡིག་གི་ས་བོན་ལས་ལྷ་མོ་མཁར་སྡུད་པའི་
འཇིགས་བྱེད་ཁ་དོག་དམར་པོའོ། །རླུང་སྔོ་བསངས་དང་ཨུ་དུམྺཱ་རའི་ཤིང་དང་ཀླུ་ནོར་རྒྱས་ཁ་དོག་དཀར་པོའོ། །ཚེས་བདུན་ནམ་བཅོ་ལྔ་ལ་གཟའ་སྤེན་པ་ལ་མཆོད་པ་བྱའོ། །གཏོར་མ་སིནྡྷུ་རས་བརྒྱན་པའོ། །ཨོཾ་ཧྲཱིཾ་ཤྲཱིཾ་ཧཱུཾ་ཧཿདེ་བི་ཀོ་ཊ་སྱེ་བིགྷྣཱཾ་བི་ནཱ་ཤཱ་ཡ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། ཞིང་སྐྱོང་གིའོ། །ཨོཾ་ཡ་ཙཱ་
