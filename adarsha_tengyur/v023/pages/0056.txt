ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་ཧཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་ཧཱ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨོ་ར་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨོ་ར་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷུ་ཀི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷཎྜཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷཎྜི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀིཾ་ཀི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཀིཾ་ཀི་ཎི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷུརྒྷུ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷུརྒྷུ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་ཀོ་ལི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་ཀོ་ལི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤཾ་ཁཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤཾ་ཁ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷོ་ཥ་ཝ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷོ་ཥ་ཝ་ཏི་ཧུཾ་ཧཱུཾ་ཕཊ།
ཨོཾ་བཛྲ་པཪྵ་དཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཪྵ་ད་ཧུཾ་ཧཱུཾ་ཕཊ།།      །།ཨོཾ་བཛྲ་ག་རུ་ཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་རུ་ཌ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཾ་སཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཾ་ས་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙི་ཏྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙི་ཏྲི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཀཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏིཏྟཱ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏིཏྟཱ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཡཱུ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཡཱུ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་མྲ་ཙཱུ་ཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་མྲ་ཙཱུ་ཌི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གུ་ད་
བུ་ལི་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གུ་དབུ་ལི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀོ་མ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀོ་མ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ར་ཝརྟཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ར་ཝརྟི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲྀ་ཧཏྐཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྲྀ་ཧཏྐཱ་ཀི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་ཌཱི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།
ཨོཾ་བཛྲ་ག་ཌི་ནི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པིཉྩ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པིཉྩ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མནྟྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མནྟྲི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་ར་སཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་ར་སི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྲྀ་དྷཱི་
ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྲྀདྷྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུལླཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུལླཱུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཊ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་ཊ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཥྛ་ཙ་ཊི་ཀི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཥྛ་ཙ་ཊ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙཀྲ་བཱ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
