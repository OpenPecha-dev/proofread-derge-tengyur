ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་ཀཱ་མི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཁོདྦྷ་ཝཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཁོདྦྷ་ཝ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཁ་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་ཁ་མ་ཏི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲཱི་ཡ་ཏ་མཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲཱི་ཡ་ཏ་མ་ཧུཾ་ཧཱུཾ་ཕཊ།  ཨོཾ་བཛྲ་པྲེ་མ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་
ཕཊ། ཨོཾ་བཛྲ་པྲེ་མ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཽ་བྷ་གྱ་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཽ་བྷ་གྱ་མ་ཏི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཽ་བྷ་གཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཽ་བྷཱ་ག་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མེ་ཎུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མེ་ཎུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཏྱུཾ་ན་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
པྲ་ཏྱུཾ་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛཱ་ཏི་རཱུ་པཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛཱ་ཏི་རཱུ་པ་ཧུཾ་ཧཱུཾ་ཕཊ།།      །།ཨོཾ་བཛྲ་བྷ་དྲ་ཀལྤ་བུདྡྷེ་བྷྱོ་ཧུཾ་ཧཱུཾ་ཕཊ།  ཨོཾ་བཛྲ་གཽ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙོོ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཻ་ཏཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གྷསྨ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པུཀྐ་སཱི་ཡེ་
ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཝ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙཎྜ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌོམྦི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་གཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ག་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡཀྵི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡཀྵ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཱུ་ཏི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།  ཨོཾ་བཛྲ་བྷཱུ་
ཏ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲེ་ཏ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲེ་ཏ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ན་ར་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་བཱི་ཙཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་བཱི་ཙི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཏ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཏ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨནྟ་རཱི་ཡེ་
ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨནྟ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུམྦྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུམྦྷ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་སྟྲི་ཡཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་པུ་རུ་ཥ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ལ་སཱུ་ཏྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ལ་སཱུ་ཏྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུ་ཀུ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཀུ་ཀུ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་པ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་པ་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཏ་པ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཏ་པ་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཽ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཽ་ར་བ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་རཽ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་རཽ་ར་བ་ཧུཾ་ཧཱུཾ་ཕཊ།
