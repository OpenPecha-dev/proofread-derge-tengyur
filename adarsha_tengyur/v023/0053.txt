ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་མ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་མ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཎྜ་ར་བ་སི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཎྜ་ར་བ་ས་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་རཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པ་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པ་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤབྡ་བཛྲེ་
ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤབྡ་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ས་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ས་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྤཪྵ་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སྤཪྵ་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨྨ་དྷཱ་ཏུ་བཛྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དྷརྨྨ་དྷཱ་ཏུ་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀྵི་
ཏི་གཎྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀྵི་ཏི་གཎྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁ་གརྦྷ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁ་གརྦྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་པཱ་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཛྲ་པཱ་ཎི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཀ་ནཱ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལོ་ཀ་ནཱ་ཐ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སརྦ་ཨ་བ་ར་ཎ་ནི་ཥྐཾ་བྷི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།
ཨོཾ་བཛྲ་སརྦ་ཨ་བ་ར་ཎ་ནི་ཥྐཾ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་མནྟ་བྷ་དྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་མནྟ་བྷ་དྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཏྣོ་ལྐཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཏྣོ་ལྐི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཻ་རཱཏྨ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཻ་རཱཏྨ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷྲྀ་ཀུ་ཊི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷྲྀ་ཀུ་ཊི་ཀ་ཧུཾ་ཧཱུཾ་
ཕཊ། ཨོཾ་བཛྲ་པརྞ་ཤ་བ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པརྞ་ཤ་བ་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མཱནྟ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཛྭཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲཛྙཱནྟ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པདྨཱནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པདྨཱ་ནྟ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བིགྷྣནྟ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་
ཕཊ། ཨོཾ་བཛྲ་བིགྷཱནྟ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཙ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཙ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱི་ལ་དཎྜཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱི་ལ་དཎྜི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐི་རཱ་ཛཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐི་རཱ་ཛ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་བ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་བ་ལ་ཧུཾ་
ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུཥྞཱི་ཥཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུ་ཥྞཱི་ཥ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུམྦྷ་རཱ་ཛཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུམྦྷ་རཱ་ཛ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྲཱཧྨ་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྲཱཧྨ་ཎ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཏྲི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀྵ་ཏྲི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཻ་ཤཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།
