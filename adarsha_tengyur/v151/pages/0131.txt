བསྐོ་བར་བགྱི་སྟེ། གསལ་བའི་ཡང་གསལ་བ་མིང་འདི་ཞེས་བགྱི་བ་འདི་དགེ་འདུན་ལ་རྩོད་པ་བྱུང་བ་རྣམ་པར་ཞི་བར་བགྱིད་དོ། །འདི་ནི་གསོལ་བའོ། །དགེ་འདུན་བཙུན་པ་རྣམས་གསན་དུ་གསོལ། གསལ་བའི་ཡང་གསལ་བ་མིང་འདི་ཞེས་བགྱི་བ་འདི་དགེ་འདུན་ལ་རྩོད་པ་བྱུང་བ་
རྣམ་པར་ཞི་བར་བགྱིད་པར་སྤྲོ་སྟེ། དེའི་སླད་དུ་དགེ་འདུན་གྱིས་མིང་འདི་ཞེས་བགྱི་བ་འདི་རྩོད་པ་རྣམ་པར་ཞི་བར་བགྱིད་པར་བསྐོ་ན། ཚེ་དང་ལྡན་པ་གང་དག་གསལ་བའི་ཡང་གསལ་བ་མིང་འདི་ཞེས་བགྱི་བ་འདི་རྩོད་པ་རྣམ་པར་ཞི་བར་བགྱིད་པར་བསྐོ་བར་བཟོད་པ་དེ་དག་ནི་ཅང་མ་གསུང་ཤིག །གང་
དག་མི་བཟོད་པ་དེ་དག་ནི་གསུངས་ཤིག །དགེ་འདུན་གྱིས་བཟོད་ཅིང་གནང་ནས། དགེ་འདུན་གྱིས་གསལ་བའི་ཡང་གསལ་བ་མིང་འདི་ཞེས་བགྱི་བ་འདི་རྩོད་པ་རྣམ་པར་ཞི་བར་བགྱིད་པར་བསྐོས་ལགས་ཏེ། འདི་ལྟར་ཅང་མི་གསུང་བས་དེ་དེ་བཞིན་དུ་འཛིན་ཏོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །དེའི་འོག་
ཏུ་དེ་དག་གིས་མ་ནུས་ཤིང་གཞན་མི་བྱེད་ན་གང་ལས་བླངས་པ་དེ་ལ་དབུལ་བའི་གོ་རིམས་བཞིན་རྣམ་པར་གཞག་པ་ནི་འདི་ལྟར་བྱ་སྟེ། གསལ་བ་དང་གསལ་བའི་ཡང་གསལ་བ་གང་དག་ཡིན་པ་དེ་དག་གིས་རྩོད་པ་དེ་ཉེ་བར་ཞི་བར་བྱེད་མ་ནུས་ཤིང་རང་གི་སྡེ་ཚན་ལས། སྡེ་ཚན་གཞན་བསྐོ་བར་ཡང་
མི་བྱེད་ན། དེའི་ཚེ་དང་པོ་རྩོད་པ་དེ་གང་ལས་བླངས་པ་དེ་ཉིད་ལས་ཕྱིར་དབུལ་བར་བྱ་སྟེ། གསལ་བའི་ཡང་གསལ་བ་དག་གིས་ནི་གཟུ་བོ་དག་གམ་དགེ་འདུན་ལ་དབུལ་ལོ། །གཟུ་བོ་དག་གིས་ནི་རྒོལ་བ་དང་ཕྱིར་རྒོལ་བ་དག་ལ་དབུལ་ལོ། །དེ་དག་གིས་ནི་དགེ་འདུན་ལ་དབུལ་ཏེ། བདག་ཅག་ལ་ཁྱེད་ཀྱིས་གཏད་
པ་ནི་རྩོད་པ་འདི་བདག་ཅག་གིས་མངོན་སུམ་གྱིས་འདུལ་བ་སྦྱིན་ཞིང་རྒྱུད་གཅིག་ཏུ་བྱ་བས་ཞི་བར་བྱེད་མ་ནུས་ཀྱིས། ཁྱེད་ཉིད་ལ་ཕྱིར་དབུལ་ལོ་ཞེས་ཕྱིར་ཁུངས་ཁུངས་སུ་ཕུལ་ནས། དགེ་འདུན་བཞི་པོས་ཉམས་སུ་བླང་བར་བྱ་སྟེ། དེའི་འོག་ཏུ་ཕྱིར་འོངས་ནས། དགེ་འདུན་བཞི་པོས་རྩོད་པ་དེ་སྦེད་པ་བསྐོ་བའི་གོ་རིམས་རྣམ་
པར་གཞག་པ་ནི་འདི་ལྟར་བྱ་སྟེ། གསལ་བའི་ཡང་གསལ་བ་དག་ལས་རྩོད་པ་དེ་ཕྱིར་འོངས་ན། དགེ་འདུན་བཞི་པོས་རྩོད་པ་ཞི་བར་མ་ནུས་པ་དེ་མཚམས་གཞན་གྱི་དགེ་འདུན་ལ་ཞི་བར་བྱ་བར་བཅོལ་བའི་ཕྱིར་རྩོད་པ་མཚམས་གཞན་དུ་སྦེད་པའི་དགེ་སློང་དང་པོར་སྤྲོ་བར་བྱ་སྟེ། གསོལ་བ་དང་གཉིས་ཀྱི་ལས་ཀྱིས་
