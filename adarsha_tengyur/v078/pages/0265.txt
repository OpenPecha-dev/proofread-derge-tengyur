གཉིས་མེད་དུ་ཞུ་ནས། སངས་རྒྱས་ཐམས་ཅད་ཀྱི་སྐུ་གསུང་ཐུགས་ཀྱི་ངོ་བོ་བདུད་རྩིའི་ངོ་བོ་དཀར་བ་འོད་དང་བཅས་པར་གྱུར་པར་བསམ་མོ། །དེ་ནས་ན་མོ་བྷ་ག་བ་ཏེ། བཻ་རོ་ཙ་ན་པྲ་བྷ་རཱ་ཛཱ་ཡ། ཏ་ཐཱ་ག་ཏཱ་ཡ། ཨརྷ་ཏེ་སམྱཀྶཾ་བུདྡྷཱ་ཡ། ཏདྱ་ཐཱ། ཨོཾ་སུ་ཀྵྨཻ་སུ་ཀྵྨཻ། སེམས་མ་ཡེ། ཤཱནྟེ། དཱནྟེ། 
ས་མོ་བེ། ཨ་ན་ལཾ་བྷེ། ཏ་རཾ་བྷེ། ཡ་ཤོ་བ་ཏི། མ་ཧཱ་ཏེ་ཛེ། ནིར་ཀུ་ལེ། ནིརྦཱ་ཎི། སརྦ་ཏ་ཐཱ་ག་ཏཱ་ཧྲྀ་ད་ཡ། ཨ་དྷིཥྛཱ་ན། ཨ་དྷིཥྛ་ཏེ་སྭཱ་ཧཱ། ཞེས་ཉི་ཤུ་རྩ་གཅིག་བཟླས་ལ་གདབ་སྟེ། ལག་པས་ལེན་ཙ་ན་འོད་ཀྱི་གོང་བུ་ཨོཾ་དང་ལྡན་པ་བླངས་ཏེ་སྐུ་བར་དུ་བཅུག་ལ། གོང་གི་སྔགས་ཀྱིས་བཏབ་ལ་
རྣམ་པར་སྣང་མཛད་དུ་འབྱུང་ངོ་། །མཇུག །ཨོཾ་ཡེ་དྷརྨཱ་ཧེ་ཏུ་པྲ་བྷཱ་བ་ཧེ་ཏུནྟེ་ཥཱནྟ་ཐཱ་ག་ཏོ་ཧྱ་བ་དཏ། ཏེ་ཥཱཉྩ་ཡོ་ནི་རོདྷ་ཨེ་བཾ་བཱ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎ་ཡེ་སྭཱ་ཧཱ། ཞེས་ལན་བདུན་བརྗོད་པས་རབ་གནས་སུ་འགྱུར་རོ། །མེ་ཏོག་ལ་བཟླས་ཏེ་ཚ་ཚ་ལ་གཏོར་བས་སྐད་ཅིག་ལ་རྣམ་པར་སྣང་མཛད་
གྲངས་མེད་དུ་གྱུར། དེའི་འོད་ཀྱིས་ཕྱོགས་བཅུའི་སངས་རྒྱས་ཐམས་ཅད་བཀུག་སྟེ་སྔོན་བཞིན་དངོས་སམ་བློས་མཆོད་ནས་ཚ་ཚར་གྱུར་པའོ། །གཅིག་བྱས་པས་བྱེ་བ་སྙེད་ཀྱི་བསོད་ནམས་འཐོབ་བོ། །ངལ་བ་གསོ་བ་ལ་གསུམ་སྟེ། སྦྱོར་བ་ལུས་དལ་བར་བསྡད། དངོས་ཆོས་སྤྱོད་
པའི་རྒྱལ་པོ་དང་། ཁྱིམ་བདག་དང་། མཁན་པོ་དང་། སློབ་དཔོན་གྱི་ཆེ་བ་བརྗོད་པ་དང་། དེ་དག་དྲན་པར་བྱ། དགེ་བ་བྱས་པའི་ཡོན་ཏན་དྲན་པར་བྱ། མཇུག་རྒྱུ་འབྲས་ཐེག་པ་ཆེན་པོའི་མདོ་བལྟའོ། །ཕྱི་དྲོའི་ཐུན་ལ་གསུམ་སྟེ། གོང་བཞིན་དུ་ཐུན་གླེགས་བམ་ཀླག་པ་དང་། བསྐོར་བ་
བྱ་བ་དང་། གཙུག་ཏོར་ལ་སོགས་པའི་མདོ་གདོན་པའོ། །སྲོད་ཀྱི་ཐུན་ལ་ལྔ་སྟེ། གོང་བཞིན་དུ་ཁྲུས་བྱ་བ་དང་། གཏོར་མ་གཏང་བ་དང་། མཎྜལ་དང་མཆོད་པ་དབུལ་བ་དང་། བསམ་གཏན་བྱ་བ་དང་ཉལ་བའོ། །ཉལ་བ་ལ་གསུམ་སྟེ། སྦྱོར་བ་ཁམས་དང་སྦྱར་ལ་བསམ་གཏན་བསྡུ་བ་
