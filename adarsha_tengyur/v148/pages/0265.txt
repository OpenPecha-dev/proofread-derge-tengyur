བྱ་བ་ནི་བསྩལ་བ་སྟེ་དེས་ངེས་པར་བྱ་བའོ། །མི་བྱ་བ་དག་བྱེད་པ་དང་། །ཞེས་སྨོས་ཏེ། དགེ་འདུན་གྱི་དབྱེན་ལ་སོགས་པའི་གླེང་གཞི་དང་སྦྱར་བར་བྱའོ། །ཡང་ན་སྲེད་པ་རྣམ་པ་བཞི་བསྐྱེད་པ་ཡིན་ཏེ། ཇི་སྐད་དུ་གསུངས་པ། ཆོས་གོས་ཀྱི་ཕྱིར་སྲེད་པ་བསྐྱེད་ཅིང་རབ་ཏུ་
གནས་པ་དང་། ཇི་སྲིད་དུ་ཟས་དང་གནས་མལ་དང་འཁོར་བའི་འབྱོར་པའི་ཆེད་དུའོ། །ཀྱེ་མ་ཞེས་བྱ་བ་ནི་ཐུགས་བརྩེ་བ་ཡིན་ཏེ། རྗེས་སུ་མི་མཐུན་པའི་ཚུལ་གྱིས་སྤྱོད་ཅིང་བག་མེད་པ་དེ་ལ་ཐུབ་པའི་དབང་པོ་རྣམས་ཐུགས་བརྩེ་བར་མཛད་དོ། །གང་ཡང་དེ་ལྟར་བྱེད་པ་དེའི་ཉེས་དམིགས་བསྟན་པའི་
ཕྱིར། རློམ་བྱེད་པ་ཞེས་བྱ་བ་སྨོས་ཏེ། མཐོ་བ་དང་བསྙེམས་ཤིང་ལོག་པར་སྤྱོད་པ་སྟེ། གང་ཡང་མི་བསྟེན་ཞིང་ང་རྒྱལ་ཆེ་ལ་སྤྱོད་པ་ལོག་པ་དེ་ཐམས་ཅད་ནི་བྱ་བ་མ་ཡིན་པ་བསྟེན་པར་འགྱུར་ཏེ། འདིས་ནི་བྱ་བ་མ་ཡིན་པ་བྱེད་པའི་རྒྱུ་བསྟན་ཏོ། །བག་མེད་པ་ཞེས་བྱ་བ་ནི་བྱ་བ་ལ་ནན་ཏན་མེད་ཅིང་མངོན་པར་
མི་བརྩོན་པ་སྟེ། འདིས་ནི་བྱ་བ་མི་བྱེད་པའི་རྒྱུ་བསྟན་ཏེ། གང་གི་ཕྱིར་བག་མེད་པས་ནི་བྱ་བ་སྤོང་བར་འགྱུར་རོ། །དེ་དག་རྣམས་ཀྱི་ཞེས་བྱ་བ་ནི་དེ་ལྟར་གྱུར་པའི་རྒྱུད་ལ་ནི་ཟག་པ་འཕེལ་བར་འགྱུར་བ་སྟེ། ཟག་པ་ནི་མ་སྐྱེས་པ་སྐྱེ་བ་དང་སྐྱེས་ཟིན་པའི་ཕྱིར་ཞིང་རྒྱས་པར་འགྱུར་བའོ། །ཟག་པ་ནི་ཁམས་
གསུམ་གྱི་ཉོན་མོངས་པ་དང་ཉེ་བའི་ཉོན་མོངས་པའོ། །ཟག་པ་འཕེལ་བའི་སྐྱོན་སྨྲས་པ། །འཕེལ་གྱུར་པ་ཞེས་བྱ་བ་ནི། གང་དེ་ལྟར་ཟག་པ་འཕེལ་བ་སྐྱེ་བ་ནི་ཟག་པ་མེད་པ་རྒྱང་རིང་བ་ན་གནས་པ་ཡིན་ཏེ། མྱ་ངན་ལས་འདས་པ་ཐག་རིང་དུ་གྱུར་པའོ། །འདིར་རྒྱང་མ་ཞེས་བྱ་བའི་སྒྲ་ནི་ཤིན་ཏུ་ཐག་རིང་བའི་
དངོས་པོའི་ཚིག་གོ། །སྡུག་བསྔལ་གྱི་རྒྱུ་ཟག་པ་ཡོད་པར་གྱུར་ན་ནི་སྡུག་བསྔལ་ཁོ་ན་ཉེ་བར་གནས་པར་འགྱུར་ཏེ། དེ་ཞི་བར་ནི་མི་འགྱུར་རོ། །དེ་ལས་ལྡོག་པས་དཀར་པོའི་ཕྱོགས་སྨྲས་པ། གང་ཞིག་ཅེས་བྱ་བ་ལ་སོགས་པ་སྨོས་སོ། །ཤཱ་རིའི་བུ་དང་མཽད་གལ་གྱི་བུ་གཉིས་ཀྱིས་དགེ་འདུན་ཕྱེ་བའི་ཉི་མ་
དེ་ཉིད་ལ་དེའི་བསྡུམས་བྱས་ཏེ། དེའི་དབང་དུ་མཛད་ནས་འདི་གསུངས་སོ། །འདིར་བསྡུས་པའི་དོན་ནི་དྲན་པ་དང་ཤེས་བཞིན་མི་ཉམས་པའི་ཐབས་བསྟན་པ་དང་། ལྷག་མས་དེའི་ཕན་ཡོན་བསྟན་པའོ། །གང་ཞིག་ཅེས་བྱ་བ་ནི་གང་དག་བདག་ཉིད་ཆེན་པོ་དག་གོ། །ལེགས་པར་བརྩམས་པ་ཞེས་བྱ་བ་ནི་
