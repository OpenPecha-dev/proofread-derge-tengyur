མཱ་ར་མཱ་ར། གྲྀཧྞ་གྲྀཧྞ། བནྡྷ་བནྡྷ། ཧ་ན་ཧ་ན། ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ། དི་ན་མེ་ཀེ་ན་མཱ་ར་ཡ་མཱ་ར་ཡ། ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། མ་ཧཱ་ཀཱ་ལ་བ་ལིཿ།།   ༤   །།ཨོཾ་ཨཱཿཧཱུཾ་ཧོཿཨུཥྞཱི་ཥ་སུམྦྷ་རཱ་ཛ་ཡ་མཱནྟ་ཀ །པྲཛྙཱནྟ་ཀ །པདྨཱནྟ་ཀ །བིགྷྣཱནྟ་ཀ །ཨ་ཙ་ལ། ཊཀྐི་རཱ་ཛ། ནཱི་ལ་དཎྜ། མ་ཧཱ་
སོད་སོད། ཟུང་ཟུང་། ཆིངས་ཆིངས། སྣུན་སྣུན། སྲེགས་སྲེགས ཚོས་ཚོས། ཉི་མ་གཅིག་གིས་ནག་པོ་ཆེན་པོའི་གཏོར་མའོ།།      །།གཙུག་ཏོར་གནོད་མཛེས་རྒྱལ་པོ། གཤིན་རྗེ་མཐར་བྱེད། ཤེས་རབ་མཐར་བྱེད། པདྨ་མཐར་བྱེད། བགེགས་མཐར་བྱེད། མི་གཡོ་བ།  འདོད་པའི་རྒྱལ་པོ། དབྱུག་སྔོན། སྟོབས་པོ་ཆེ་

རྣམས། འཁོར་དང་བཅས་པ་རྣམས་འདི་གཏོར་མ། དྲི། མེ་ཏོག །བདུག་པ། མར་མེ་མ་ཉམས་པ་སྦྱིན་གྱིས་དེ་རྣམས་ཀྱང་བྱོན་ནས་འཁོར་བཅས་མྱུར་བར་གཏོར་མ་འདི་གཟུང་དུ་གསོལ་བཞེས་ཤིག་འཐུང་ཤིག །ཡང་དག་པར་ཚིམ་སྟེ། བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཞི་བ་དང་རྒྱས་བ་དང་སྲུང་བ་དང་བསྐྱབ་པ་དང་སྦ་བ་མཛོད། རྡོ་རྗེ་འཆང་གི་བཀའ་བཞིན་དུ།

ཁྲོ་བོ་བཅུའི་གཏོར་མའོ།།      །།འདིར་གནས་པའི་ཞིང་སྐྱོང་རྣམས་དང་གཞན་རྣམས་ལ་ཡང་འདི་དྲི་དང་། མེ་ཏོག །བདུག་པ། མར་མེ་དང་མ་ཉམས་པ་སྦྱིན་གྱི་དེ་རྣམས་ཀྱང་བྱོན་ནས་འཁོར་དང་བཅས་པ་རྣམས་མྱུར་བར་གཏོར་མ་འདི་ཟུང་གསོལ་འཐུང་ཀུན་དུ་ཚིམ་སྟེ་བདག་དང་སེམས་ཅན་ཐམས་ཅད་ལ་ཞི་བ་དང་རྒྱས་པ་དང་བསྲུང་བ་དང་བསྐྱབ་པ་
