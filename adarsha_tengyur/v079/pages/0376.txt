ཅད་ནུས་པ་བརྟན་པས་ན་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བའི་སྙིང་པོ་བསྔགས། ཡེ་དྷ་རྨཱ་ཧེ་ཏུ་པྲ་བྷ་ཝཱ་ཧེ་ཏུནྟེ་ཤཱནྟ་ཐཱ་ག་ཏོ་ཧྱ་ཝ་དཏ། ཏེ་ཥཱཉྩ་ཡོ་ནི་རོ་དྷ་ཨེ་ཝཾ་བཱ་དཱི་མ་ཧཱ་ཤྲ་མ་ཎཿཞེས་བྱ་བ་བཟླས་སོ། །ཀྱང་ཞེས་བྱ་བ་ནི་བསྡུ་བ་སྟེ་ཆོས་ཀྱི་སྦྱིན་པ་ཡང་ཁ་ཏོན་བྱ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ཉིད་
བཤད་པ་དེ་ནས་ཆོས་ཀྱི་སྦྱིན་པ་ནི། །རྒྱ་ཆེན་རབ་ཟབ་མདོ་གདོན་བྱ། །ཞེས་བྱ་བས་བསྟན་ཏོ། །ཆོས་བཤད་པས་ཐམས་ཅད་ཀྱིས་མཉན་པས་ཐམས་ཅད་བྱང་ཆུབ་སེམས་སུ་ལྡན་པར་བསམ་པ་དང་། བདག་གིས་ཀྱང་ཉན་པ་དང་། གསལ་བ་དང་། ཐམས་ཅད་དུ་ཐོས་པ་
དང་། ཐམས་ཅད་ཀྱི་སྐད་དུ་གོ་བ་དང་། ཆོས་ཀྱི་སྒོ་མོ་མཐའ་ཡས་པ་དུས་གཅིག་ལ་སྟོན་པར་བསམས་ལ། ཁ་ཏོན་བྱ་བའམ་ཀླག་པ་ཡང་རུང་སྟེ། ཟབ་པ་དང་དོན་དམ་པ་སྤྲོས་པ་དང་བྲལ་བ་སྟོན་པ་དང་། རྒྱ་ཆེ་བ་ཐམས་ཅད་ཕུན་སུམ་ཚོགས་པ་སྟོན་པའོ། །ནི་ཞེས་བྱ་བ་ནི་
བསྣན་པའམ་གོ་རིམས་བཟློག་པའོ། །ད་ནི་གཤེགས་སུ་གསོལ་བར་བྱ་སྟེ། དེ་ཡང་དེ་རྗེས་མདུན་གྱི་ལྷ་འཇམ་དཔལ་འཁོར་དང་བཅས་པ་དང་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ལ་སོགས་པ་རྣམས་ཏིང་ངེ་འཛིན་གྱི་གསལ་བར་ནི་མཆོད་པ་དང་ནི་བསྐོར་བ་དང་། ཕྱག་བྱ་བ་
དང་། བསྟོད་པ་བྱས་པར་བསམས་ནས་ཡི་གེ་བརྒྱ་པའི་བཤགས་པ། ཨོཾ་བཛྲ་སཏྭ་ས་མ་ཡ། མ་ནུ་པཱ་ལ་ཡ། བཛྲ་སཏྭ་ཏྭེ་ནོ་པ་ཏིཥྛ། དྲྀ་ཌྷོ་མེ་བྷ་ཝ། སུ་ཏོ་ཥྱོ་མེ་བྷ་ཝ། ཨ་ནུ་རཀྟོ་མེ་བྷ་ཝ། སུ་པོ་ཥྱོ་མེ་བྷ་ཝ། སརྦ་སིདྡྷི་མྨེ་པྲ་ཡཙྪ། སརྦ་ཀརྨ་སུ་ཙ་མེ་ཙིཏྟཾ་ཤྲེ་ཡཿཀུ་རུ་ཧཱུཾ། ཧ་ཧ་ཧ་ཧ་ཧོ་བྷ་ག་ཝཱན། 
སརྦ་ཏ་ཐཱ་ག་ཏ། བཛྲ་མཱ་མེ་མུཉྩ། བཛྲཱི་བྷ་བ། མ་ཧཱ་ས་མ་ཡ་སཏྭ་ཨཱཿཞེས་བྱ་བ་བཟླས་ལ། དེ་ནས་གཤེགས་སུ་གསོལ་བ་ནི་ཨོཾ་བཛྲ་མུཿཞེས་བྱའོ། །ཇི་ལྟར་ཞེ་ན་ཐམས་ཅད་རང་རང་གི་གནས་སུ་གཤེགས་པ་ནི་དེས་ནས་སྙན་ངག་དང་བཅས་པས་གཤེགས་པར་བསམ་མོ། །འོ་ན་
བདག་ཉིད་ལྷའི་འཁོར་དང་བཅས་པ་ཅི་ལྟ་བུ་ཞེ་ན། བསྐྱེད་པའི་ལྷ་མོ་བཞི་པོ་ནི་སྙིང་གའི་འཁོར་ལོ་ཡི་གེ་རང་རང་གི་ས་བོན་དུ་གྱུར་པར་དང་། པ་དང་། ཙ་དང་། ན་ལ་བསྡུའོ། །འཁོར་ལོ་རྩིབས་བརྒྱད་ཀྱི་ཡི་གེ་བརྒྱད་པོ་ནི་ལྟེ་བའི་ཨོཾ་ལ་བསྡུ་བར་བྱའོ། །དེ་ནས་ཨོཾ་ལས་བསྡུས་པས་ཨོཾ་
