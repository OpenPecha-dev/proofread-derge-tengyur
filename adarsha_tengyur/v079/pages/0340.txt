མ་ཧཱ་བ་ལཱ། ཏ་ཐཱ་དྷ་ནྱཱ་མ་ཧཱ་བྷཱ་གཱ་པདྨ་ཀུཎྜ་ལི་རེ་བ་ཙ། པུཥྤ་དནྟི་མ་ཎི་ཙཱུ་ཌཱ་སྭ་ནྱ་ཀེ་ཤཱི་ཙ་པིཾ་ག་ལཱ། མ་ཧཱ་ཏེ་ཛཱ་ཏ་ཐཱ་དེ་བཱི་དྷ་ནྱཱ་ཙ་བིདྻུནྨཱ་ལི་ནཱི། རཀྵ་སྱཻ་ཀ་ཛ་ཊཱ་ཙཻ་བ་བུདྡྷ་ཀྵཱི་ཏི་ཀ་ནཱ་ཡི་ཀཱཿ། ཀ་པཱ་ལི་ནཱི་མ་ཧཱ་བྷཱ་གཱ་དྷ་ནྱཱ་ལངྒེ་ཤྭ་རཱི་ཏ་ཐཱ། ཨ་ནྱཱཤྩ་བ་ཧ་བོ་
སྟོབས་ཆེན་མ། །དེ་བཞིན་དཔལ་ཡོན་སྐལ་ཆེན་མ། །པདྨ་འཁྱིལ་པ་ཉིད་དང་ཡང་། །མེ་ཏོག་སོ་མ་གཙུག་ནོར་མ། །རང་གཞན་སྐྲ་མ་དམར་སེར་མ། །དེ་བཞིན་ལྷ་མོ་གཟི་བརྗིད་ཅན། །དཔལ་མོ་གློག་གི་ཕྲེང་ལྡན་མ། །སྲིན་མོ་རལ་པ་གཅིག་མ་དང་། །སངས་རྒྱས་ས་ཡི་འདྲེན་པ་མོ། །ཐོད་པ་ཅན་མ་སྐལ་ཆེན་མ། །དེ་བཞིན་དཔལ་མོ་ལངྐའི་དབང་། །གཞན་ཡང་རིག་པ་མང་
བི་དྱཱཿས་ཏྭཱ་ནུ་གྲ་ཧ་ཀཱ་རི་ཀཱཿ། ཏདྱ་ཐཱ། ཀཱ་ལི་ཀ་རཱ་ལི། ཀུམྡྷཱཎྜི། ཤཾ་ཁི་ནི། ཀ་མ་ལཱཀྵི། ཧཱ་རི་ཏཱི། ཧ་རི་ཀེ་ཤི། ཤྲཱི་མ་ཏི་ཧ་རི་པིངྒ་ལེ། ལམྦ། པྲ་ལམྦེ། ཀཱ་ལ་པཱ་ཤེ། ཀ་ལ་ཤོ་ད་རི། ཡ་མ་དཱུ་ཏི། ཡ་མ་རཱཀྵ་སི། བྷཱུ་ཏ་གྲ་ས་ནི། པྲ་ཏཱིཙྪི་མཾ་གནྡྷཾ་པུཥྤཾ་དྷུ་པཾ་དཱི་པཾ་བ་ལིཾ་
པོ་ནི། །སེམས་ཅན་རྗེས་སུ་འཛིན་པ་མོ། །འདི་ལྟ་སྟེ། ནག་མོ་གཅིགས་མ། གྲུལ་བུམ་མ། དུང་ཅན་མ། པདྨའི་མིག་མ། འཕྲོག་མ། སྐྲ་ལྗང་གུ་མ། དཔལ་ལྡན་མ་འཕྲག་སེར་སྐྱ་མ་འཕྱང་མ། རབ་འཕྱང་མ། ཞགས་པ་ནག་མོ། བུམ་པའི་ལྟོ་མ། གཤིན་རྗེ་ཕོ་ཉ་མ། གཤིན་རྗེ་སྲིན་མོ་འབྱུང་པོ་ཟ་བྱེད་མ། གཏོར་མ་དྲི་དང་མེ་ཏོག་དང་བདུག་པ་དང་མར་མེ་ལྡན་པ་

འདི་འབུལ་གྱི་བཞེས་ཤིག་བསྲུང་བ་མཛོད། བདག་དང་སེམས་ཅན་ཐམས་ཅད་འཇིགས་པ་དང་ཉེ་བར་འཚེ་བ་ཐམས་ཅད་ལས་ལོ་བརྒྱ་འཚོ་བར་ཤོག་ཅིག །སྟོན་བརྒྱ་མཐོང་བར་ཤོག་ཅིག །གསང་སྔགས་ཀྱི་ཚིག་རྣམས་གྲུབ་པར་གྱུར་ཅིག །བསྲུང་བ་ལྔའི་གཏོར་མའོ།།      །།ཕྱག་འཚལ་ལོ། །བཅོམ་ལྡན་འདས་མ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་དཔག་མེད་མ། སེར་ཆེན་མ།
