འགྲོ་ལ་ཕན་སེམས་བྱམས་པའོ། །ཕ་རོལ་གནོད་སེལ་སྙིང་རྗེའོ། །སེམས་ཅད་དོན་གྲུབ་དགའ་བ་ནི། །དེ་ནི་དགའ་བ་ཚད་མེད་ཡིན། །གཞན་གྱིས་གནོད་པ་ཚད་མེད་བྱེད། །དེ་ལ་མཉམ་གཞག་བཏང་སྙོམས་སོ། །ཞེས་བཤད་པའོ། །དེ་ལྟར་ཚད་མེད་བཞི་རྫོགས་ནས། །
དེ་བཞིན་ཉིད་ཀྱི་ཏིང་འཛིན་བསྒོམ། །ཞེས་བྱ་བ་ནི་མཚམས་སྦྱོར་བ་སྟེ། དེ་བཞིན་ཉིད་ནི་ཆོས་ཐམས་ཅད་ཀྱི་ཆོས་ཉིད་དོ། །ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་མཚན་བརྒྱ་རྩ་བརྒྱད་བཤད་པ་ནི་དེ་བཞིན་ཉིད་ཀྱི་རྣམ་གྲངས་སོ། །དེ་ཉིད་བཤད་པ་ན། ཕྱི་ནང་སྣོད་བཅུད་ཐམས་ཅད་
ཀུན། །ངོ་བོ་ཉིད་ཀྱིས་རྣམ་པར་སྟོང་། །ཞེས་བྱ་བ་སྟེ། ཕྱི་ནི་རི་དང་ཁང་པ་ལ་སོགས་པ་སེམས་ཀྱི་གནས་ཡིན་པས་སྣོད་ཅེས་བྱའོ། །དེ་ཐམས་ཅད་ངོ་བོ་ཉིད་ཀྱིས་སྟོང་པ་ཉིད་ནི་སྟོང་པ་ཉིད་ལས་མ་འདས་པའོ། །དེ་ཡང་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལས། དེ་ལ་ཕྱི་སྟོང་པ་ཉིད་གང་ཞེ་ན། ཕྱིའི་
ཆོས་གཟུགས་དང་སྒྲ་དང་དྲི་དང་རོ་དང་རེག་དང་ཆོས་རྣམས་ཏེ། མི་རྟག་མི་འཇིག་པའི་ཕྱིར་གཟུགས་ཀྱིས་སྟོང་ངོ་། །སྒྲ་སྒྲས་སྟོང་ངོ་། །དྲི་དྲིས་སྟོང་ངོ་། །རོ་རོས་སྟོང་ངོ་། །རེག་རེག་གིས་སྟོང་ངོ་། །ཆོས་ཆོས་ཀྱིས་སྟོང་ངོ་ཞེས་གསུངས་པའི་ཕྱིར་རོ། །ནང་གི་སེམས་ཅན་དུ་
གཏོགས་པ་རྣམས་ཏེ་སྣོད་ན་ཡོད་པས་བཅུད་དོ། །དེ་ཡང་སྟོང་པ་ཉིད་དེ། ཡང་དེ་ཉིད་ལས་ནང་གི་ཆོས་མིག་དང་། རྣ་བ་དང་། སྣ་དང་། ལྕེ་དང་། ལུས་དང་། ཡིད་དོ། །དེ་ཡང་མི་རྟག་མི་འཇིག་པའི་ཕྱིར་མིག་མིག་གིས་སྟོང་ངོ་། །རྣ་བ་རྣ་བས་སྟོང་ངོ་། །སྣ་སྣས་སྟོང་ངོ་། །ལྕེ་ལྕེས་སྟོང་ངོ་། །
ལུས་ལུས་ཀྱིས་སྟོང་ངོ་། །ཡིད་ཡིད་ཀྱིས་སྟོང་ངོ་ཞེས་གསུངས་པའི་ཕྱིར་རོ། །གཟུགས་རྣམ་པར་རྟོག་པ་ནི་གཟུགས་ལ་སོགས་པ་འབྱུང་བ་དང་འབྱུང་བ་ལས་གྱུར་པའོ། །དེ་ཐམས་ཅད་ནི་རྣམ་པ་རྟོག་པ་ཙམ་སྟེ་ཕྱི་རོལ་ན་མེད་དོ། །ཅིའི་ཕྱིར་ཞེ་ན། གཅིག་དང་དུ་མ་དང་བྲལ་བའི་ཕྱིར་རོ། །གལ་ཏེ་
གཟུགས་ལ་སོགས་པ་དེ་དག་ཡོད་ན་གཅིག་པུ་ཞིག་གམ། དུ་མ་ཞིག་ཅེས་དེ་དང་པོར་བྱའོ། །གལ་ཏེ་གཅིག་པུ་ཞིག་ཡིན་ན་ནི་ཟིན་པ་དང་མ་ཟིན་པ་དང་། རགས་པ་དང་ཕྲ་བ་དང་དམན་པ་དང་ནོམ་པའི་ཆ་ཐ་དད་པ་མི་འཐད་དོ། །དུ་མ་ཡང་མི་འཐད་པ་ཉིད་དེ་ཕྲ་རབ་ཕྱོགས་ཆ་ཐ་དད་པའི་ཕྱིར་རོ། །
