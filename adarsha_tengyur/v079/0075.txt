ཐུགས་སུ་མངོན་སྦྱར་ལ། །རྟག་ཏུ་བཟླས་པ་བྱེད་པ་ནི། །བྱང་ཆུབ་གཟུང་ལ་ཞུགས་པ་ཡིན། །རབ་ཀྱིས་ཆུ་བོའི་རྒྱུན་བཞིན་ནམ། །ཉིན་གསུམ་མཚན་གསུམ་གཉིས་གཅིག་ཏུ། །མ་ཡེངས་པར་ནི་གང་གིས་བཟླས། །འབྲིང་གིས་ཟླ་གཅིག་དུས་
དྲུག་ཏུ། །དུས་བཞི་བར་དུ་མི་བཅད་ཅིང་། །ཐ་མ་སྤྱད་པའི་རྣལ་འབྱོར་པས། །ཟླ་རེའམ་ར་བའི་དུས་དག་ཏུ། །བརྩོན་པས་འབད་དེ་མི་བྱེད་པ། །བྱང་ཆུབ་ས་བོན་རླག་པ་ཡིན། །ཚོགས་དང་གཏོར་མ་དེ་བཞིན་ཏེ། །ཐ་ན་ལོ་ཟླ་མི་བཟླའོ། །ལྟ་ཟབ་སྤྱོད་པ་བརླང་པོ་དང་། །ལྷ་
ཡི་གཟུགས་དང་གྲུབ་རྟགས་དང་། །སྤྱིར་གསང་རྣམ་པ་བཞིར་བཤད་དོ། །སྒྲུབ་པའི་གནས་དུས་གྲོགས་དང་རྫས། །རབ་ཏུ་གསང་ལ་བསྒྲུབ་པར་བྱའོ། །གསང་བར་འོས་པ་མདོར་བསྡུ་ན། །ཕུང་དང་གཏོར་མ་ཚོགས་དག་དང་། །བྷནྡྷ་ཕུར་པ་མར་མེའི་སྣོད། །རྫས་དང་ལག་
ཆ་དཀྱིལ་འཁོར་དང་། །སྐྱེ་བོས་མཐོང་བར་ཡོངས་མི་བྱ། །དྲིལ་བུ་རྔེའུ་ཆུང་ལ་སོགས་སྒྲ། །སྐྱེ་བོས་ཐོས་པར་ཡོངས་མི་བརྡུང་། །གྲོགས་པོའི་སྤྱོད་དང་མཚང་རྣམས་དང་། །སེམས་ཅན་ཀུན་གྱི་ངན་སྤྱོད་རྣམས། །ཤིན་ཏུ་གསང་ན་བྱང་ཆུབ་རྒྱུ། །གཉེར་གཏད་པ་ཡི་
གདམས་པ་སྟེ། །སྔར་བཤད་སློབ་དཔོན་ཆེད་ཀྱིས་སོ། །གསང་སྔགས་སྐྱོན་ཡོད་མ་ཡིན་ཏེ། །སེམས་ཅན་དོན་ཕྱིར་ཤིན་ཏུ་གསང་། །གསང་ཐུབ་དངོས་གྲུབ་མི་ཡལ་བས། །གསང་ཕྱིར་གདམས་ངག་སྣ་ཚོགས་གསུངས། །དམ་ཉམས་དམ་ལས་ལོག་པ་དང་། །དམ་
མེད་དཀྱིལ་འཁོར་མ་མཐོང་ཞིང་། །འདྲེས་དང་མ་འདྲེས་ཐམས་ཅད་ལ། །སེམས་ཀྱིས་བསྟན་པར་མི་བྱ་ཞིང་། །ལུས་ཀྱིས་ཐམས་ཅད་སྦས་ལ་གཞག །ངག་གིས་རྡོ་རྗེ་ལྕེ་ཡོད་ཀྱང་། །བརྗོད་པ་དག་ཏུ་བྱ་བ་མེད། །དེ་དག་རྩ་བ་བཞིར་བཤད་དོ། །ཐུན་མོང་ཡན་ལག་བཤད་བྱ་
བ། །སྒྲོལ་སྦྱོར་མ་བྱིན་ལེན་པ་དང་། །བརྫུན་སྨྲ་ངག་རྩུབ་སྤྱོད་པ་ལྔ། །རང་བཞིན་གྱིས་ནི་སྦྱོར་སྒྲོལ་དང་། །དབང་པོའི་ཡུལ་དུ་སྦྱོར་སྒྲོལ་དང་། །ཏིང་ངེ་འཛིན་གྱི་སྦྱོར་སྒྲོལ་དང་། །ཐབས་ཆེན་པོ་ཡི་སྦྱོར་སྒྲོལ་དང་། །སྔོན་དུ་བྱ་བའི་སྦྱོར་སྒྲོལ་ལྔའོ། །གཉིས་ནི་ཤེས་བྱའི་ཐུན་མོང་
