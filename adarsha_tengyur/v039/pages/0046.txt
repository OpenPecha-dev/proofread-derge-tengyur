གྱི་ལྷ་རྣམས་གནས་པར་འགྱུར་རོ། །དེ་ནས་མཆོད་པ་ལ་སོགས་པ་བྱས་ནས། བདག་ཉིད་འཇུག་པའི་ཆོ་ག་བྱ། །ཞེས་སོ། །འདི་དག་ཀྱང་ལེའུ་བཅུ་དྲུག་པར་བཤད་དོ། །ད་ནི་བསྟོད་པའི་དེ་མ་ཐག་ཏུ་བདུད་རྩིའི་མཆོད་པས་མཉེས་པར་བྱ་བ་གསུངས་པ། བཤང་དང་ཞེས་པ་ལ་སོགས་པའོ། །
བཤང་བ་ནི་དྲི་ཆེན་ནོ། །གཅི་བ་ནི་དྲི་ཆུའོ། །ཁུ་བ་ནི་བྱང་ཆུབ་སེམས་སོ། །ཁྲག་ནི་རང་བྱུང་ངོ་། །སོགས་ཀྱིས་ནི་མ་སྨོས་པ་བསྡུའོ། །ལྷ་རྣམས་ལ་ནི་དབུལ་བར་བྱ། །ཞེས་པ་ནི།  རླུང་ལ་སོགས་པའི་རིམ་པས་དག་པར་བྱས་པ་ཞེས་པའོ། །དེའི་ཡོན་ཏན་གསུངས་པ། དེ་ལྟར་
ཞེས་པ་ལ་སོགས་པའོ། །སངས་རྒྱས་ཞེས་པ་ནི་རིགས་ལྔའོ། །བྱང་ཆུབ་སེམས་དཔའ་ནི་ལྷ་མོ་བཅུའོ། །གྲགས་ཆེན་ཁྲོ་བོ་བཞི་སྟེ་དེས་མཉེས་པར་འགྱུར་རོ་ཞེས་པའོ། །ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ཞེས་པ་ནི་མི་བསྐྱོད་པས་གཙོ་བོ་བྱས་པའི་དཀྱིལ་འཁོར་རོ། །དེ་སྟོན་པར་བྱེད་
པའི་གཞུང་ནི་ལེའུ་སྟེ་བཞི་པའོ། །ལེའུ་བཞི་པའི་འགྲེལ་པའོ།།      །།ད་ནི་སྒྲུབ་པ་པོའི་སྤྱོད་པའི་ཁྱད་པར་བསྟན་པའི་ཕྱིར། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པས་ལེའུ་ལྔ་པ་གསུངས་པ། སྟོན་པ་བཅོམ་ལྡན་འདས་ཞེས་པ་ནི་བཤད་ཟིན་ཏོ། །རྒྱལ་པོ་ཞེས་པ་ནི་མཛེས་པར་
བྱེད་པའོ། །སྐུ་གསུང་ཐུགས་མངའ་བ་ཞེས་པ་ནི་རྡོ་རྗེ་གསུམ་དང་ལྡན་པས་སྤྱོད་པ་སྤྱོད་པ་ནི་སྐྱོན་གྱིས་གོས་པ་མེད་པའི་ཕྱིར་སྟོན་པ་ལ་དེ་སྐད་ཅེས་བྱའོ། །སའི་དབང་ཕྱུག་ཅེས་པ་ནི་ས་ཐམས་ཅད་ཀྱི་བདག་པོར་གྱུར་པ་ལ་བྱའོ། །ཀུན་མཆོག་གིས་ཞེས་པ་ནི། དེ་ཉིད་ཀྱི་ཕྱིར་ན་མཆོག་
སྟེ་དེ་སྐད་ཅེས་བཀའ་སྩལ་ཏོ་ཞེས་བྱ་བར་སྦྱར་རོ། །ཇི་ལྟ་བུ་ཞིག་ཅེ་ན།  སྤྱོད་པ་ཀུན་ཞེས་བྱ་སྟེ་སེམས་སྤྱོད་པའོ། །ཇི་སྐད་དུ། ཐམས་ཅད་སེམས་ལ་གང་སྤྱོད་པ། །ཀུན་དུ་སྤྱོད་པ་བརྟན་པ་ཡིན། །ཞེས་སོ། །དེ་བས་ན་སེམས་ཀྱིས་ཀུན་ལ་སྤྱོད་པའོ། །དེ་ཉིད་མཆོག་གི་ཆོས་ཀྱི་
དོན་ཡང་དེ་ཉིད་པས་ན་དེ་སྐད་དོ། །དེ་ལྟ་བུའི་སྤྱོད་པའི་མཚན་ཉིད་ཀྱང་འོག་ནས་འབྱུང་བས་བཀའ་སྩལ་ཏོ་ཞེས་སོ། །ད་ནི་སྔོན་དུ་སྤྱོད་པའི་ངོ་བོ་རྣམས་རྣམ་པར་དག་པ་ཐོབ་མ་ཐག་ཏུ་འགྲུབ་པར་གསུངས་པ། རྣམ་པར་མི་རྟོག་པའི་ཞེས་པ་ལ་སོགས་པའོ། །རྣམ་པར་མི་རྟོག་པའི་
