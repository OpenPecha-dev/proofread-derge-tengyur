སྨྲས་པ་ནི། འགྲོ་བའི་སྒྲུབ་པའི་ཁྱད་པར་གྱི་བྱེ་བྲག་མེད་པར་ཐམས་ཅད་ཀྱིས་བསོད་ནམས་ཆེན་པོ་ཉིད་ཐོབ་པར་ཤེས་པར་བྱ་བའི་ཕྱིར་རོ། །དེའི་རྒྱུ་བསྟན་པའི་ཕྱིར་མདོའི་སྡེ་ཐམས་ཅད་ཀྱི་རྩ་བ་ནི་འདི་ཡིན་པའི་ཕྱིར་རོ། །ཆོས་འདི་ལྟ་བུ་ལ་ཀུན་དུ་སྤྱོད་པའི་སྦྱོར་བ་དང་ལྡན་པ་ཡོད་པ་དེ་སྲིད་དུ་
བསོད་ནམས་རབ་ཏུ་འཕེལ་བའི་རྒྱུ་ཡིན་ནོ་ཞེས་བྱའོ། །མི་མཐུན་པ་དང་གཉེན་པོའི་བྱེ་བྲག་ཀྱང་མདོ་སྡེ་གཞན་དག་ཏུ་བཞུགས་སུ་ཟིན་ཀྱང་དམ་པའི་ཆོས་ཀྱི་གཟུགས་བརྙན་ལྟ་བུའི་ཆོས་རྣམས་ཀྱི་གཉེན་པོར་གྱུར་པ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ས་འདི་ཡིན་ཏེ། དོན་ཤིན་ཏུ་གསལ་བར་བསྟན་
པས་རབ་ཏུ་ཕྱེ་བའི་ཕྱིར་རོ། །གང་ཡང་སྨྲས་པ། དམ་པའི་ཆོས་ལྟར་བཅོས་པ་རྣམས་འཕེལ་བར་མི་འགྱུར་རོ་ཞེས་བྱ་བའི་བར་དུའོ། །གང་གི་ཚེ་དེ་འཕེལ་བར་འགྱུར་བ་དེའི་ཚེ་ཡང་དག་པའི་དོན་དང་ལྡན་པ་དེ་ནུས་པར་འགྱུར་ཞིང་། ཡང་དག་པའི་མདོ་སྡེ་རྣམས་ཀྱང་ནུས་པར་འགྱུར་རོ། །མདོ་སྡེ་ལ་བརྟེན་
པའི་དམ་པའི་ཆོས་ལྟར་བཅོས་པ་རྣམས་ནི་རྒྱས་པར་འགྱུར་རོ། །བྱང་ཆུབ་སེམས་དཔའི་སའི་གཞིའི་རྣལ་འབྱོར་གྱི་གནས་ཐ་མ་ལས་རབ་ཏུ་གནས་པའི་ལེའུ་སྟེ་ལྔ་པའི་འགྲེལ་པ་རྫོགས་སོ།།      །།དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་སའི་གོ་རིམས་ནི་འདི་ཡིན་ཏེ། རིགས་ལ་གནས་པའི་བྱང་ཆུབ་སེམས་དཔའ་
ནི་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་ཏུ་སེམས་སྐྱེད་པར་བྱེད་དོ། །སེམས་བསྐྱེད་ནས་བདག་དང་གཞན་གྱི་དོན་ལ་སྦྱར་རོ། །བདག་དང་གཞན་གྱི་དོན་ལ་སྦྱོར་ན་ཀུན་ནས་ཉོན་མོངས་པ་མེད་པར་བྱ་བའི་ཐབས་རྙེད་པར་འགྱུར་རོ། །ཀུན་ནས་ཉོན་མོངས་པ་མེད་ན་མི་སྐྱོ་བའི་ཐབས་རྙེད་པར་འགྱུར་རོ། །སྐྱེ་བ་མེད་ན་དགེ་
བའི་རྩ་བ་འཕེལ་བར་བྱ་བའི་ཐབས་རྙེད་པར་འགྱུར་རོ། །དགེ་བའི་རྩ་བ་རྣམས་ཀྱིས་རྒྱས་པར་གྱུར་ན་བླ་ན་མེད་པ་ཡང་དག་པར་རྫོགས་པའི་བྱང་ཆུབ་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པར་འགྱུར་ཏེ། བདག་དང་གཞན་གྱི་དོན་ལ་སྦྱོར་བ་དང་། ཀུན་ནས་ཉོན་མོངས་པ་མེད་པར་བྱ་བའི་ཐབས་དང་མི་སྐྱོ་བའི་ཐབས་དང་དགེ་
བའི་རྩ་བ་འཕེལ་བར་བྱ་བའི་ཐབས་དང་། བྱང་ཆུབ་ཐོབ་པར་བྱ་བ་ལ་བསྒྲུབ་པ་ན་ཐོག་མ་ཁོ་ནར་ཟབ་ཅིང་རྒྱ་ཆེ་བའི་ཆོས་རྣམས་ལ་མོས་པ་རྣམ་པར་འཇོག་གོ། །དེ་ལ་མོས་ན་ཆོས་དེ་དག་ཡོངས་སུ་ཚོལ་ལོ། །ཡོངས་སུ་བཙལ་ནས་གཞན་དག་ལ་ཡང་སྟོན་ཅིང་བདག་ཉིད་ཀྱང་ནན་ཏན་གྱིས་སྒྲུབ་པར་བྱེད་དོ། །བསྒྲུབས་ནས་ཀྱང་གང་
