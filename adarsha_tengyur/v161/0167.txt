བྱའོ། །འདུལ་བ་དང་མ་མོ་ཡང་ངོ་ཞེས་བྱ་བ་ན་འདུལ་བ་འཛིན་པ་དང་མ་མོ་འཛིན་པ་འོང་བ་ལ་གདུགས་དང་རྒྱལ་མཚན་དང་བ་དན་ལ་སོགས་པ་དག་གིས་བསུ་བར་བྱའོ། །ལམ་ཇི་སྲིད་དུ་ཞེ་ན། དེའི་ཕྱིར་པག་ཚད་ཕྱེད་དང་གསུམ་དུའོ། །མ་ནུས་ན་རྒྱང་གྲགས་ལྔར་རོ་ཞེས་
བྱ་བ་སྨོས་ཏེ། ལམ་དེ་ཙམ་དུ་བསུ་བར་མི་ནུས་ན་རྒྱང་གྲགས་ལྔར་བསུ་བར་བྱའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །གསུམ་མམ་དཔག་ཚད་ཕྱེད་དམ་རྒྱང་གྲགས་སམ་རྒྱང་གྲགས་ཕྱེད་དམ་ཁྱམས་སུ་རུང་ངོ་ཞེས་བྱ་བ་ལ། གསུམ་ཞེས་བྱ་བ་ནི་རྒྱང་གྲགས་རྣམས་ཀྱང་ངོ་། །མ་ནུས་ན་ཞེས་བྱ་བ་ནི་ཐམས་ཅད་
ལ་སྦྱར་བར་བྱའོ། །དེ་ལྟར་བསུ་ནས་ཅི་ཞིག་བྱ་ཞེ་ན། དེའི་ཕྱིར་ངག་གི་བཤེས་པའི་ཚིག་གིས་བསུ་ནས་ལྷུང་བཟེད་དང་ཆོས་གོས་བླང་བར་བྱའོ། །ཁྲུས་བྱའོ། །དགེ་འདུན་ལ་ཡང་སྣུམ་བག་གིས་རྙེད་པའོ། །གནས་ལ་སོགས་པ་བསྟབས་པ་སྦྱིན་ནོ། །ཆོས་ཉན་པ་ཡང་ངོ་ཞེས་བྱ་བ་ནི། 
སྦྱིན་ནོ་ཞེས་བྱ་བར་སྦྱར་ཏེ། དེར་ཐམས་ཅད་ཀྱིས་སྤྲོ་བ་བསྐྱེད་པར་བྱ་ཞེས་བྱ་བ་ནི། མང་པོའི་ཚིག་གིས་བསྟན་པ་དེ་རྣམས་ཐམས་ཅད་ཀྱིས་སྨྲ་བར་མི་རིགས་པའི་ཕྱིར་རོ། །དགེ་སློང་ཕྲག་དོག་ཅན་ལ་འདི་དག་སྟོན་པར་འཛིན་པ་ཡིན་ནོ་ཞེས་གོ་བར་བྱའོ་ཞེས་བྱ་བ་ནི། འདི་སྟོན་པའོ་འདི་ནི་
སྟོན་ཏེ་འདི་ལྟ་སྟེ། མདོ་སྡེ་དང་འདུལ་བ་འཛིན་པ་དང་མ་མོ་ཞེས་གསུངས་པ་དེ་འདིས་བསྡུས་པ་ཡིན་ནོ། །གསོ་སྦྱོང་དང་དགག་དབྱེ་མ་བྱས་པས་དེའི་ཉིན་མོ་རང་གི་དེ་ཡོད་པའི་གནས་ནས་འགྲོ་བར་མི་བྱ་སྟེ། གལ་ཏེ་དེ་ཉིད་ཀྱི་ཚེ་རྣམ་པ་དེ་ལྟ་བུར་ཕྱིན་པའམ། འཐབ་ཀྲོལ་ཅན་དག་དང་དབྲལ་
བའམ། སྤྱོད་པ་བཟང་པོ་དག་དང་ཕྲད་པ་དེ་ལྟ་བུར་འགྲོ་བ་ཉིད་མ་ཡིན་ནའོ་ཞེས་བྱ་བ་ལ། དེའི་ཉིན་མོ་ཞེས་བྱ་བ་ནི་གསོ་སྦྱོང་དང་དགག་དབྱེའི་ཉིན་མོ་ལའོ། །རང་གི་ཞེས་བྱ་བ་ནི་དགེ་སློང་རྣམས་དང་ལྷན་ཅིག་མདོ་འདོན་པ་དང་སྤྱོད་པའོ། །ཡང་ན་རང་གི་ཞེས་བྱ་བ་ནི་དགེ་སློང་མའི་དགག་དབྱེ་
དགེ་སློང་དང་དགེ་སློང་མ་རྣམས་ལྷན་ཅིག་ཏུ་མདོ་འདོན་པའི་ཚེ་དགེ་སློང་གི་ཚོགས་པ་གྲུབ་པ་དགེ་སློང་མ་རྣམས་དང་ལྷན་ཅིག་ཏུའོ། །དེ་ཞེས་བྱ་བ་ནི་གསོ་སྦྱོང་དང་དགག་དབྱེ་ལ་ལྟོས་ནས་བདག་ཉིད་ཀྱིས་གསོ་སྦྱོང་ངམ་དགག་དབྱེ་རབ་ཏུ་ཤེས་བཞིན་དུ་རང་དགག་དབྱེ་ཉིད་དུ་ཞེས་བྱ་བའི་དོན་ཏོ། །
