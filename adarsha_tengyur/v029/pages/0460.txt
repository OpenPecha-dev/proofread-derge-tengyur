ནས་དང་། མ་ཤ་དང་།གྲོ་དང་། འབྲས་དང་ལྡན་པ་དང་། ཅུང་ཟད་བདུད་རྩི་ལྔའི་ཆུ་དང་བཅས་པ་རིགས་ཀྱི་བདག་པོའི་སྔགས་ཀྱིས་བཟླས་ལ། འཁོར་ལོའི་རྩིབས་ཀྱི་དབུས་ན་བཞག་ནས་སྤོས་གཏུལ་ཞིང་མཆོད་པར་བྱའོ། །ལས་ཐམས་ཅད་པའི་བུམ་པ་ཡང་རང་གི་
སྔགས་ཀྱིས་བྱིན་གྱིས་བརླབས་ལ་ཤར་ཕྱོགས་ཀྱི་སྒོའི་ཕྱི་རོལ་དུ་བཞག་ལ། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ལ་སྤོས་གཏུལ་བས་ཞུ་བར་བྱའོ། །ཁྱེད་རྣམས་མཆོད་པར་བྱ་ཕྱིར་དང་། །སློབ་མ་རྣམས་ཀྱང་རྗེས་གཟུང་ཕྱིར། །སྙིང་རྗེའི་བདག་ཉིད་བདག་དེ་ཡིས། །དཀྱིལ་
འཁོར་བྲི་བར་འདོད་པའོ། །དེ་ཕྱིར་དད་ལྡན་བདག་ལ་ནི། །བཅོམ་ལྡན་བཀའ་དྲིན་མཛད་དུ་གསོལ། །འཇིག་རྟེན་མགོན་པོ་ཐུགས་རྗེ་ཅན། །ཁྱེད་རྣམས་བདག་ལ་དགོངས་སུ་གསོལ། །དགྲ་བཅོམ་བྱང་ཆུབ་སེམས་དཔའ་དང་། །སྔགས་ཀྱི་ལྷ་མོ་གཞན་གང་རྣམས། །
ལྷ་དང་འཇིག་རྟེན་སྐྱོང་བ་དང་། །མཐུ་ཆེན་གང་ཞིག་འབྱུང་པོ་རྣམས། །གང་ཞིག་ལྷ་ཡི་སྤྱན་ལྡན་པ། །བསྟན་པ་ལ་དགའ་དེ་རྣམས་ཀུན། །བདག་མིང་འདི་ཞེས་བགྱི་བ་ཡིས། །དཀྱིལ་འཁོར་ཆེན་པོ་འདི་ཞེས་པ། །ཅི་ནུས་པ་ཡི་ཉེར་སྤྱོད་པས། །ནང་པར་བྲི་བར་བྱ་བའོ། །
དེ་ཕྱིར་སློབ་མ་ནང་བཅས་པ། །བདག་ནི་སྙིང་རྗེས་བཟུང་ནས་ནི། །ཐམས་ཅད་བཅས་པ་དཀྱིལ་འཁོར་ལ། །བྱིན་གྱིས་བརླབ་པ་མཛད་དུ་གསོལ། །ཞེས་བྱ་བ་ལན་གསུམ་བརྗོད་པར་བྱའོ། །དེའི་འོག་ཏུ་སློབ་མ་རྣམས་བཀུག་ཅིང་མེའི་དཀྱིལ་འཁོར་ལ་གནས་པའི་ཡི་
གེ་རའི་མེས་དེ་རྣམས་རེ་རེ་ཐལ་བར་བྱས་ལ། ཡི་གེ་གསུམ་གྱི་རང་བཞིན་རང་གི་ཁ་ནས་ཞུགས་ཏེ། བདེ་བ་ཆེན་པོའི་ངོ་བོ་དེ་ཉིད་རྡོ་རྗེ་ནས་སྤྲོ་བར་བྱས་ལ། དེ་ཉིད་སྐུ་ལ་གོགས་པས་བྱིན་གྱིས་བརླབས་ལ། ཡི་གེ་ཧཱུཾ་ལས་བྱུང་བའི་རྡོ་རྗེས་བྱིན་གྱིས་བརླབས་ལ། དཀྱིལ་
འཁོར་གྱི་རྩ་བའི་སྒོའི་ངོས་སུ་བཞག་ནས་འདི་ཡིས་ལྷག་པར་གནས་པར་བྱའོ། །སྡིག་པ་བདག་གིས་གང་བྱས་པ། །བྱེད་དུ་བཅུག་དང་རྗེས་ཡི་རང་། །སླར་ཡང་དེ་ཉིད་མི་བྱའི་ཕྱིར། །དེ་དག་ཐམས་ཅད་བཤགས་པར་བགྱི། །སངས་རྒྱས་མགོན་པོའི་བསོད་ནམས་དག །
