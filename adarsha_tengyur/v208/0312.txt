བྷྲཱ་ཏྲི་བྱཿ། [སྤུན་དྲཱའི་]བུ། བྷ་ཏྲཱི་ཡཿ། བྷྲཱ་ཧྲྀ་ལ་ཚ་ཚ་ཨཱི་ཡ། བྱ་ནྶ་ཕ་ཏྣེ། དགྲ་བོ་ལ་བྱན་ལའོ། །བྷྲཱ་ཏྲི་ལས་དགྲ་བོའི་བུ་བརྒྱུད་བརྗོད་པ་ལ་བྱན་འགྱུར་ཏེ། བྷཱ་ཏྲྀ་བྱཿ། རེ་བ་ཏྱཱ་དི་བྷྱ་ཥྛ་ཀ །རེ་བ་ཏཱི་ལ་སོགས་པ་རྣམས་ལས་ཋ་ཀའོ། །རེ་བ་ཏཱི་ལ་སོགས་པ་རྣམས་ལས་བུ་བརྒྱུད་ལ་ཋ་ཀ་འགྱུར་རོ། ཋ་སྱེ་ཀཿ
བྷྲཱ་ཏྲ་ལ་བྱ་ཏ་ཏ་ནི་ཏ་ཡི་དེའི་དུས་ཀྱི་འོས་དོན། ཋའི་ཨི་ཀའོ། །རྐྱེན་གྱི་ཋ་ནི་ཨི་ཀར་འགྱུར་ཏེ། རཻ་བ་ཏི་ཀཿ། ནམ་གྲུའི་བུ། ཨཱ་ བྷྲཱ་ཏྲི་ལ་བྱན་ན་དབྱངས་དོན་སུ། དགྲ་བོའི་སྤུན་བླའི་བུ། ཤྭ་པཱ་ལི་ཀཿ། རྟ་སྐྱོང་བའི་བུ། ཨ་ནྲྀ་ཥྱཱ་ནཾ་ཏ་རྻེ་བི་དཱ་དི་བྷྱོ་྅ཉ། དྲང་སྲོང་མ་ཡིན་པའི་རྒྱུན་མ་ཆད་པ་ལ་བིད་ལ་སོགས་པ་རྣམས་ལས་

ལ་ཧྻཱཿ། ཨ་ཏྲྀ་བྷྲྀ་གུ་ཀུ་ཏྶ་བ་སི་ཥྛ་གོ་ཏ་མཱ་དྒི་རོ་བྷྱ་ཤྩ། ཨ་ཏྲི་ བྷྲྀ་གུ་ ཀུ་ཏྶ་ བསི་ཥྛ་ གོ་ཏ་མ་ ཨ་དྒི་རས་རྣམས་ལས་ཀྱང་ངོ་། ཨ་ཏྲྀ་ ཡ་སྐ་ལ་ཨ་ཉ་ ཉ་འཕེལ་བའི་དོན། ཡ་སྐ་ལ་ ཛས་ མང་ཚིག་ལ་ ཨ་ཉ་ཕྱིས་རྒྱུ་མཚན་མེད་ན་ཞེས་པས་ཟུང་ངུ་། ལ་སོགས་པ་འདི་རྣམས་ལས་ཀྱང་རྣམ་དབྱི་མང་ ཐོབ་པའི་རིགས།

པའི་རྒྱལ་རིགས་བརྗོད་པ་ཅན་ལས་བུ་བརྒྱུད་ལ་ཨ་ཉ་འགྱུར་ཏེ། རྒྱུན་ཤེས་ཀྱི་བུ་རྣམས། ཌྷ་ཀ་གྱིས་པའོ། དན་སྤོང་གི་བུ་རྣམས། སྨད་པའི་བུ་རྣམས། གནས་འཇིག་གི་བུ་རྣམས། རེགས་མཆོག་གི་བུ་རྣམས། ཡན་ལག་སྐྱེས་སམ་ཉི་མའི་བུ་རྣམས། ལྔ་ལེན་པའི་རྒྱལ་པོའི་བུ། པཱ་ཉྩཱ་ལཿ། [པ་ཉྩཱ་ལ་] ཨ་ཉ་ཉ་འཕེལ་བའི་དོན། ཀཱ་ལི་དྒཿ། [ཀཱ་ལིངྒའི་]རྒྱལ་པོའི་བུ།
