བར་སྔོན་གྱི་དྲང་སྲོང་སྤོཊཱ་ཡ་ནས་བཤད་དེ། གོ ཨཱི་ཤ༔ཞེས་པ་ལ། ག་ཝ་ཤ༔འམ། ཡང་ན་ག་བཱི་ཤ༔སོགས་སོ། །འགྲེལ་བར་ལུང་དྲངས་པ། །ཨི་ནྡྲེ་ཙ་ནི་ཏྱཾ། ཨིནྡྲ་ལ་འང་རྟག་ཏུ་འོ། །གོའི་མཐའ་ཨི་ནྡྲ་དཔའ་ང་སྒྲས་ཨ་ཀྵ་ཕ་རོལ་ལ་ཨ་ཝང་དུ་རྟུག་ཏུ་འགྱུར་ཏེ། གོ ཨི་ནྡྲ་ཞེས་པ་ལ།
ག་བེ་ནྡྲ༔། གོ ཨ་ཀྵ་ཞེས་པ་ལ། གཔཱ་ཀྵ༔། ཨུ་ཉ༔། ཨུ་ཉ་ལས་སོ། །ཨུ་ཉ བ་ལར་དབང་ཕྱུག། །སྟེ་ཨུ་ཞེས་པ་འདི་ཨི་ཏི་ཕ་རོལ་ལ་ཝྲ་ཞེས་པར་འགྱུར་བ་དང་རང་བཞིན་དུ་གནས་པ་ཡང་ན་བར་བྱ་སྟེ། །ཨུ་ཏི། བི་ཏི། མ་ཡ་ཨུ་ཉོ་བོ་བཱ། མ་ཡ་ལས་ཨུ་ཉའི་བྲ་ཡང་ནའོ། མ་ཡ་སོ་སོར་བསྡུ་བ་ལས་ཕ་རོལ་
པ་ཡང་གི་དབང་བོ། གྱི་ཨུ་ཉ་ཨ་ཙ་ བ་ལང་མིག་སྟེ་སྐར་ཁུང་ངོ་། ཕ་རོལ་ལ་བྲ་རུ་ཡང་ན་འགྱུར་ཏེ། ཀི་མུ། ཨུ་ཀྟཾ་ཞེས་པ་ལ། །ཀི་མྦུ་ཀྟཾ་འམ་ཡང་ན་ཀི་མུ་ཨུ་ཀྟཾ་སོགས་སོ། །ཨི་ཀོ་྅ས་བརྞེ་ཤཱཀ་ལྱ་སྱ་ཧྲ་སྭ་ཤྩ། ཨུ་འདི། ཨི་ཀ་གི་འདྲ་བ་མ་ཡིན་པ་ལ་ཤཱ་ཀ་ཡྱ་ཀྱི་ཐུང་དུ་ཡང་ན་འོ། །ཨི་ཀ་སོ་སོར་
བསྡུ་བའི་མཐའི་འདྲ་བ་མ་ཡིན་པའི་ཨ་ཙ་ཕ་རོལ་ལ་ཡང་ན་རང་བཞིན་དུ་གནས་ཤིང་རིང་པོ་ཡིན་ན་ཐུང་ངུར་རམ་ཡང་ན། ཅི་བརྗོད། འགྱུར་བར་སྔོན་གྱི་ ཅི་སྨྲ་བར་བྱེད། དྲང་སྲོང་ཤཱ་ཀ་ལྱ་ཀྱི་འདོད་པ་སྟེ། །ཙ་ཀྲཱི ཨ་ཏྲ་ཞེས་པ་ལ། གཟུགས་གསུམ་སྟེ། ཙ་ཀྲི་ཨ་ཏྲ། ཙ་ཀྲཱི་ཨ་ཏྲ། ཙ་ཀྲྱ་ཏྲ་སོགས་
སོ། །རྀ་ཏྱ་ཀ༔། རྀ་ཏ་ལ་ཨ་ཀ་གིའོ། །རྀ་ཏ་ཕ་རོལ་ལ་ཨ་ཀ་མཐའི་མཚམས་སྦྱོར་བྱ་བ་དང་མི་བྱེད་པ་དང་ཐུང་ངུར་བསྒྱུར་བ་སོགས་སྔོན་མ་ལྟར་དུ་བྱ་སྟེ། པྲ་ཧྨྲ་རྀ་ཥི་ཞེས་པ་ལ། པྲ་ཧྨ་རྀ་ཥྀ༔། པྲ་ཧྨ་རྀ་ཥི༔། བྲ་ཧྨ་རྵི༔སོགས་སོ། །ཨི་ཏི་པཱ་ཎི་ནི་ རྒྱལ་བོ་འདི་འམ་གསལ་པ་འདི་འམ་འཁོར་ལོ་ཅན་འདི།
བྱཱ་ཀ་ར་ཎ་སྱ་ས་ནྡྷེ༔་སཱུ་ཏྲཱ་ད་ཙ་ས་ནྡྷ་ཡ༔ས་མཱ་པྟཱ༔། །ཞེས་བ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མཚམས་སྦྱོར་གྱི་མདོ་ལས་ཨ་ཙ་མཚམས་སྦྱོར་རྣམས་རྫོགས་སོ།། །།གསུམ་པ་དམིགས་བསལ་རང་ཚངས་བ་དྲང་སྲོང་། །བཞིན་དུ་གནས་པའི་མཚམས་སྦྱོར་ནི། ཨ་ཙོ་ནྡྱཱ་དི་ཊི། ཨ་ཙའི་མཐའ་ལ་སོགས་པ་
ཊིའོ། །བྱིངས་སམ་མིང་ངམ་ཚིག་གང་གིའང་སྐབས་སུ་མཐའི་ཨ་ཙ་དང་མཐའ་ན་ཧལ་སྲོག་མེད་དང་བཅས་པའི་ཨ་ཙ་ལ་འང་ཊི་ཞེས་དགོས་པའི་ཆེད་དུ་མིང་འདོགས་བྱེད་དོ། །དཱུ་རཱ་དྡྷཱུ་ཏེ་ཙ། རིང་པོར་བོད་པ་ལ་ཡང་ངོ་། །ཐག་རིང་པོ་ནས་བོད་པའི་ཊི་ལ་བླུ་ཏ་ཞེས་མིང་འདོགས་སོ། །བླུ་ཏ་པྲ་
