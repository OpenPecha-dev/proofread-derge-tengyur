བ་འདིའི། ཨིན་སབྷཾ། དེ་བཞིན་དུ། ཨཱི་ཤྭ་ར་སྱ་ས་བྟཱ། ཞེས་པ་ལ། ཨཱི་ཤྭར་ས་བྷཾ། རཱཛྙཿས་བྷཱ། ཞེས་པ་ལ། རཱ་ཛ་ས་བྷཾ། དེ་བཞིན་དུ། མི་ཤཱཙ་སྱ་ས་བྷཱ། ཞེས་པ་ལ། པི་ཤཱཙས་བྷཾ། ཨ་ཤཱ་ལཱ་ཙ། འགྲོན་ཁང་མ་ཡིན་པ་ལའང་ངོ་། །གོང་དུ་མི་མ་ཡིན་བ་ཞེས་བྱུང་ཡང་མདོ་འདིས་འགྲོན་
ཁང་དུ་འཚོགས་པ་ལྟ་བུ་མ་ཡིན་པའི་འཚོགས་དབང་ཕྱུག་གི་ཚོགས། པའི་དོན་ལམ་དབང་ཕྱུག་གི་ཚོགས་པ །རྒྱལ་པའི་ཚོགས་པ། རྒྱལ་བོའི་ཚོགས། ནིང་གི་རྟགས་སུ་འགྱུར་ཏེ། སྣྲཱ་ཎཱཾས་ ཤ་བའི་ཚོགས་གབྷཱ། ཞེས་པ་ ཤ་ཟའི་ཚོགས་ལ། སྡཱི་སབྷཾ། པིབྷཱཥཱ་སེནཱསུ་རཱ་ཚཱ་ཡཱ་ཤཱ་ལཱུ་ནི་ཤཱ་ནཱཾ །སེནཱ་ གུརཱཚཱ་ཡཱ་ ཤཱ་ལཱ་ ནི་ཤཱ་རྣམས་
ཀྱི་འདམ་ཀའོ། །སེ་ནཱ་ལ་སོགས་པ་དེ་རྣམས་བསྡུ་བའི་མཐའ་མ་ནིང་གི་རྟགས་སུ་ཡང་ན་འགྱུར་ཏེ། དྭིཛཱ་ནཱཾ་ས་ནཱ།ཞེས་པ་ལ། དྭི་ཛ་སེ་ན། དྭི་ཛ་སེ་ནཱ། དེ་བཞིན་དུ། ཡ་བཱ་ནཱཾ་སུ་རཱ། བུད་མེད་ཀྱི་ཚོགས། ཞེས་པ་ལ། ཡ་བ་སུརཾ། རཱ། བྲཀྵ་སྱ་ཚཱ་ཡཱ། ཞེས་པ་ལ། པྲྀ་ཀྵ་ཚཱ་ཡཾ། ཡཱ། ག་
བཱཾ་ཤཱ་ལཱ། ཞེས་པ་ལ། གོ་ཤྭ་ལཾ། ལཱ། སུ་ནི་ཤཱ། སུ་ནི་ཤཾ། སེནཱ་ལ་སོགས་པ་དེ་རྣམས་གཉིས་སྐྱེས་ཏེ་བྲམ་ཟེའི་སྡེ། དེའི་སྐྱེས་བུའི་བསྡུ་བའི་ནང་ཚན་ལས་འཛིན་དང་ ནས་རྣམས་ཀྱི་ཆང་། ན་ཉ་སྔ་མ་ཅན་ལམ་ནིང་གི་རྟགས་སུ་མི་འགྱུར་ཏེ། ཤིང་གི་གྲིབ་མ། དྲྀ་ཌྷ་ཤྞ་སོོ་སེ་ནཱ་ བ་ལང་གི་འགྲན་ཁག་ཙ།
ཞེས་པ་ལྟ་བུ་ལ་རྣམ་དབྱེ་ཕྱིས་མ་ནིང་གི་རྟགས་སུ་མི་ ལེགས་པའི་མཚན་མོ། འགྱུར་ཏེ། གོ་བུད་མེད་སོགས་ཀྱིས་ཐུང་ངུ་། དྲྀ་ཌྷ་སེནཿ། དེ་བཞིན་དུ། ན་སེ་ནཱ། ཞེས་པ་ལ། ཨ་སེ་ནཿ། པར་བ་ལླིངྒཾདྭནྡྭ་ཏཏྤུ་ཏུ་ཥ་ཡོཿ། རྟགས་ཕ་རོལ་འདྲ་བར་ཟླས་ བསྟན་བའང་འདི་སྡེ་ཡང་བ་དི། དབྱེ་དེའི་སྐྱེས་
བུ་དག་གིའོ། །དེའི་སྐྱེས་དུའི་བསྡུ་བ་དང་ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འགར་རྟགས་གཉིས་བསྒྲིགས་པའི་སྔ་མ་ཕ་རོལ་ཏུ་ སྡེ་བ་སྙན་པ། སོང་ཞིང་ཕྱི་མ་སྔོན་མར་སོང་ལ་དེའི་སྐབས་རྟགས་དང་རྣམ་དབྱེ་ཕྲད་པའི་བྱ་བ་རྣམས་ཕ་རོལ་ཏུ་སོང་བའི་རྟགས་དེ་ཉིད་རྟགས་གསུམ་པོ་གང་ཡིན་ཀྱང་དེ་དང་འདྲ་
བར་བསྒྲུབ་སྟེ། ཀུཀྐུཊ་ཤྩ་མ་ཡཱུ་རཱི་ཙ། ཁྱིམ་བྱ་དང་རྨ་བྱ། ཞེས་པ་ལ། རྣམ་དབྱེ་བསྡུས་རྟགས་སྔ་ཕྱི་ཡང་ན་བརྗེས་རྣམ་དབྱེ་དང་ཕྲད་པའི་ཆོ་ག་ཕ་རོལ་ཏུ་གང་བྱུང་བ་དེའི་ལུགས་བྱས་པས། མ་ཡཱུརཱི་ཀུ་ཀྐུཊོོ། ཡང་ན། ཀུཀྐུཊ་མ་ཡཱུརྻོོ། དེ་བཞིན་དུ། པི་པི་ལིང་བྱེད་ཀྱི། ཨརྔྷ་སྱ་པིཔྤ་ལཱི།
