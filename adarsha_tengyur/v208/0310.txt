ཊྷ་ཨི་ཡུ། པཱ་ར་སྟྲཻ་ནེ་ཡཿ། ཀུ་ལ་ཊཱ་ཡཱ་བཱ། ཀུ་ལ་ཊཱའི་ཡང་ནའོ། །ཀུ་ལ་ཊཱ་ལ་བུ་བརྒྱུད་ལ་ཌྷ་ཀ་འགྱུར་ཞིང་ཀུ་ལ་ཊཱའི་མཐའ་ཨི་ནང་དུ་ཡང་ན་འགྱུར་ཏེ། ཀཽ་ལ་ཊི་ནེ་ཡཿ། ཀཽ་ལ་ཊེ་ཡཿ། ཙ་ཊ་ཀཱ་ཡཱཿཨཻ་ར་ཀ །ཙ་ཊ་ཀཱ་ལས་ཨཻ་ར་ཀའོ། །ཙ་ཊ་ཀ་དང་། ཙ་ཊ་ཀཱ་གཉིས་ལས་བུ་བརྒྱུད་ལ་
ཨཻ་ར་ཀ་འགྱུར་ཏེ། ཕ་རོལ་གྱི་བུད་མེད་ཀྱི་བུ། ཙ་ཊ་ཀཱ་ཡཱཿཨ་པ་ཏྱཾ། ཞེས་པ་ལ། ཙཱ་ཊཱ་ཀཻ་རཿ། དེ་བཞིན་དུ། ཙ་ཊ་ཀ་སྱཱ་བ་ཏྱཾ། ཞེས་པ་ལ། ཙཱ་ཊ་ཀཻ་རཿ། ཡང་འགྲེལ་ འབྱོན་མའི་བྱ། འབྱོན་མའི་བུ། པར། ཙ་ཊ་ཀཱ་བུད་མེད་ཀྱི་རྐྱེན་ཅན་ལ་བྱིན་པའི་ཨཻ་ར་ཀ་ནི་རྣམ་དབྱེ་མང་ཚིག་ཕ་རོལ་ལ་འཇིག་སྟེ།
ཙཊ་ཀཱཿ། ཛས་ཨྭཻ་ཀ་ཕྱིས་པའོ། །གོ་དྷཱ་ཡཱཿཌྷཱ་ཀ །གོ་ངྷཱ་ལས། ཁུག་རྟའི་བུ། ཁུག་རྟའི་བུ། ཙ་ཊ་ཀ་ལ། ཨཻ་ར་ཀ་ཀ་འཕེལ་བའི་དོན། ཁུག་རྟ་མའི་བུ། ཌྷྲ་ཀའོ། གོ་དྷཱ་ལས་བུ་ ཁུག་རྟ་ཡོའི་བུ། བརྒྱུད་ལ་ཌྷྲ་ཀ་འགྱུར་ཏེ། གཽ་དྷེ་རཿ། ཅིངས་པའམ་དབྱིད་མོའི་བུ། འགྲེལ་པར་ལུང་དྲངས་པ། ཨཱ་ར་གུ་དཱི་ཙཱི་ཤར་པ་རྣམས་ཀྱི་ཨཱ་
ཁུག་རྟ་མོའི་བུ་རྣམས། ར་ཀའོ། གོ་དྷཱ་ལས་བུ་བརྒྱུད་ལ་ཤར་ཕྱོགས་ཀྱི་པཎྜི་ཏ་རྣམས་ཨཱ་ར་ཀ་སྟེར་བར་འདོད་པས། གཽ་དྷཱ་རཿ། རྩངས་པའམ་དབྱིད་མོའི་བུ། ཀྵུ་དྲཱ་བྷྱོ་བཱ། ཕྲ་མོ་རྣམས་ལས་ཡང་ནའོ། །གོདྷཱ་ལ་[ཌྷྲཀ]།འཕེལ་བ་ཌྷ་ཨེ་ཕྲ་བྲ་ཡུ་དག་གི་ལྦལ་ལ་[དབྱི་]འོས་ཡ་ཕྱིས། ཕྲ་མོ་རྣམས་ལས་བུ་བརྒྱུད་ལ་ཌྷྲ་ཀ་ཡང་
ན་འགྱུར་ཏེ། ཡང་ན་ཌྷ་ཀ་འགྱུར་རོ། །ཀོ་ལ་ཊེ་རཿ། འཕྱོན་མའི་བུ། ཀཽ་ལ་ཊེ་ཡཿ། བྷ་ཀ་དྷ་ཨེ་ཡ། ཌྷ་ཀ་ དཱ་སེ་ གོདྷཱ་ལ་ཨཱ་ར་ཀ །འཕེལ་བའི་དོན། རཿ། ཌྷཀ་ དཱ་སེ་ཡཿ། ཌྷྲཀ་ ཀཱ་ཎེ་རཿ། བྷ་ཀ་ ཀཱ་ཎེ་ཡཿ། པི་ཏྲྀ་ཥྭ་པུ་ཤྪ་ཎ། པི་ཏྲྀ་ཥྭ་སྲྀ་ལས་ཚ་ཚ་ཎའོ། པི་ཏྲྀ་སྔོན་མའི་སྭ་སྲྀ་ལས་བུ་བརྒྱུད་ལ་ཆེ་ཎ་འགྱུར་
ཏེ། ཀུ་ལ་ཊ་ལ་ཌྷྲ་ཀ་འཕེལ་བཌྷ་[ཨེ་ཡྲ་བྲ་ཕྲ་]དག་གི་[ཕ་ལས་]དབྱི་འོས་ཡ་ཕྱིས། པཻ་ཧྲྀ་ཥྭ་སྲཱི་ནཡཿ། འཕྱིན་མའི་བུ། ཌྷ་ཀི་ལོ་པཿ། ཌྷ་ཀ་ལ་དབྱིའོ། །པི་ཏྲྀ་སྔ་མའི་སྭ ཟག་མའི་བུ། སི་ལས་བུ་བརྒྱུད་ལ་ བྲན་གྱི་བུ། ཌྷ་ག་འགྱུར་ཞིང་ཌྷ་ཀ་ཕ་རོལ་ལ་སྭ་སྲྀའི་རྀ་ཡང་དབྱི་སྟེ། པཻ་ཏཱྀ་ཥྭ་སེ་ཡཿ། ཕའི་སྲིང་མོའི་བུའམ་ཨ་ནིའི་བུ། མཱ་
པི་[ཏྲི་ཥྠ་]སྲེལ། ཆ་ཎཎ་འཕེལ་བའི་དོན། ཚཱ་ཨཱི་ཡ། ཨི་ཀ་གི་ཨ་ཙལ་ཡཎ་སུ། པའི་སྲིང་མོའི་བུའམ་ཨ་ནེའི་བུ། ཏྲྀ་ཥྭ་སུ་ཤྩ། མཱ་ཏྲྀ་ཥྭ་སྲྀ་ལས་འང་ངོ་། མཱ་ཏྲྀ་སྔོན་མའི་སྭ་སྤྀ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་ཚཎ་དང་ཌྷཀ་འགྱུར་ཞིང་ཌྷཀ་ཕ་རོལ་ལ་རྀའང་དབྱི་སྟེ། པི་ཏཱྀཥྲ་སྲྀལ། ཌྷ་ཀ་མདོ་འདིས་རྀ་ཕྱིས། ཌྷ་ཨེ་ཡ་འཕེལ་བ་སུ། མཱ་
