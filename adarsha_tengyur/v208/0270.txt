ཀུ་ཞེས་བ་བྱུང་བས་དེ་ཕན་གྱི་ག་ཡིན་བས་མི་དབྱི། བྻཱུབྷོར་སྐཿ། དེ་བཞིན་དུ། པྲི་ཡཿསརྦིཿཡ་སྱ། ཞེས་པ་ལ། པྲི་ཡ་སརྤིཥྐཿ། ཨིནཿསྟྲི་ཡཱཾ། ཨིན་ལས་བུད་མེད་ལའོ། །འབྲུ་མང་པོའི་བསྡུ་བའི་མཐའ་ཅན་ལས་བུད་མེད་དུ་བརྗོད་པ་ན་ཀ་བ་རྐྱེན་དུ་འགྱུར་ཏེ། བཧུ་དཎྜི་ནཿཡ་སྨིན་དེ་ཤེ།
ཞེས་པ་ལ། རྣམ་དབྱེ་ ཟང་རྣམ་པར་རྒྱས་བ། རྗེས་འབྲེལ་སོགས་ གང་གི་དགའ་བ་མར་གསར། ཕྱིས། ཀ་བ་བྱིན་ དགའ་བ་མར་གསར། པ་རྗེས་འབྲེལ། མིང་མཐའི་ན་དབྱི་བོས་ན་ཕྱིས། ཌཱ་བ་བྱིན། བཧུ་དཎྜི་ཀཱ། ན་ཀ་པི། ཀ་བ་ལམ་ཡིན་ནོ། སྤྱིར་བཏང་ གང་ལ་དབྱུག་པ་ཅན་མང་པོའི་ཡུལ་
ཀ་ཕ་རོལ་ལ་ཨཎ་གྱི་ཐུང་ངུའི་ཞེས་པས་རྐྱེན་ཀ་ཡིག་ཕ་རོལ་ལ་ཨཎ་སོ་སོར་བསྡུ་བའི་ཨ་ཙ་རིང་པོ་རྣམས་ཐུང་ངུར་འགྱུར་ཡང་དམིགས་བསལ་རྐྱེན་ཀ་པ་ལ་ཐུང་ངུར་འགྱུར་བམ་ཡིན་ཏེ། བཧུལ་ཀྵྨྱཿ དབྱུག་པ་ཅན་མ་མང་པོ། ཡ་སྨིན་དེ་ཤེ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་བསྡུས། ན་དི་
རྀ་ཏ་མཐའ་དག་ལའང་ངོས་ཀ་པ། མདོ་འདིས་ཐུང་ངུར་མི་འགྱུར། བ་ཧུ་ལཀྵྶཱི་ཀཱ། ཤེཥཱ་དྭི་བྷཱ་ཥཱ། ལྷག་མ་ལས་འདམ་ཀའོ། །སྔར་བཤད་ཟིན་པ་བུད་མེད་དུ་བརྗོད་པར་མ་ གང་ལ་ཕུན་སུམ་ཚོགས་བ་མང་པོའི་ཡུལ། ཟད་ལྷག་མ་སྙེ་མདོ་འདི་ནས་བཟུང་བུད་མེད་དུ་བརྗོད་པ་དང་བུད་མེད་དུ་
བརྗོད་པ་མ་ཡིན་པ་ལའང་ཀ་པ་སྟེར་བ་དང་མི་སྟེར་བ་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། དཔལ་མོ་མང་པོ། བ་ཧུད་ཎྜིནཿཡ་སྨིན་དེ་ཤེ། ཞེས་པ་ལ་རྣམ་དབྱེ་སོགས་སྡུས། ཀ་པ་འདམ་ཀ་ཅན་དུ་བྱས་པས། བ་ཧུ་དྞྜིཀཿ། ཡང་ན། བ་ཧུ་དཱཎཱི། ཨཱ་པོ་ཉ་ནྱ་ཏར་སྱཾ། ཨཱ་པ་ཀྱི་ཡང་ནའོ། །སྒྱིར་
བཏང་རྐྱེན་ཀ་པ་ཕ་རོལ་ལ་ཨཎ་ཐུང་ངུར་མི་འགྱུར་བར་བཤད་ཀྱང་། དེའི་དམིགས་བསལ་ གང་ལ་དབྱུག་བ་ཅན་མང་པོའི་ཡུལ། རྐྱེན་ཀ་པ་ཕ་རོལ་ལ་ཨཱ་པ་མཐའ་ཅན་ཐུང་ངུར་ཡང་ དབྱུག་པ་ཅན་མང་པོ། དབྱུག་ཅན་མང་པོ། དཎྜིལ་སུ་ཨིན་ཧ་ན་སོགས་ཀྱིས་ན་རིང་པོ། མིང་མཐའི་ན་ཏ་ལ་སོགས་ཀྱིས་སུ་ཕྱིས་པའོ།།
ན་འགྱུར་ཏེ། བཧུ་མཱ་ལཱ་ཡ་སྱ། གང་གི་ཕྲེང་བ་མང་པོ་མ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། ལྷག་མ་ལས་འདམ་ཀ་འོས། ཀཔ་ཡང་ན་བར་བྱས་བྱིན་པའི་ཕྱོགས་ལ་མདོ་འདིས་ཐུང་ངུར་འགྱུར་རམ་མི་འགྱུར་བ་ཡང་ན་བར་བྱས་པས། བ་ཧུ་མཱ་ལ་ཀཿ། ཕྲེང་བ་མང་པོ། བཧུ་མཱ་ལཱ་
