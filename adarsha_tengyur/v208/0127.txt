པའི་བྱིངས་ཀྱི་ཧ་ཡིག་སྲོག་མེད་ནི། ཛྷ་ལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་གང་རུང་ཕ་རོལ་དུ་བྱུང་བ་དང་ཚིག་མཐའ་ལ་ཡང་གྷ་རུ་འགྱུར་ཏེ། ཧ་གྷ་ར་འགྱུར། དུ་གྷ། སུ། ཨེ་ཀཱ་ཙོ་བ་ཤོ་བྷཥ་ཛྷ་ཥཾ་ཏ་སྱ་སྡྷཱོཿ། ཨ་ཅག་གཅིག་པའི་བ་ཤའི་བྷ་ཥ་ཛྷ་ཥ་མཐའི་ས་དྷྭ་ད་ལའོ། །
བྱིངས་ཨ་ཙ་གཅིག་པ་ཛྷ་ཥ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་མཐའ་ན་ཡོད་པའི་སྔ་མའི་བ་ཤའི་ཡི་གེ་རྣམས་བྷཥ་སོ་སོར་བསྡུ་པའི་ཡི་གེ་རང་དང་གནས་མཐུན་པ་རྣམས་སུ་འགྱུར་ལ། དེ་ཡང་ཛྷཥའི་ཡི་གེ་དེའི་ཕ་རོལ་དུ་ས་ཡིག་དང་དྷྭ་ཞེས་པ་དང་། ཛྷ་ལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོད་པ་ཚིག་མཐའ་ཡིན་པ་

དུ་ཧཾ། དུ་ཧཽ། དུ་ཧཿ། ཊཱལ། དུ་ཧཱ། བྷྱཱཾ་ལ། ད་སྔོན་མའི་ཞེས་པས་ཧགྷ། ཨ་ཅ་གཅིག་པའི་ཞེས་པས། ད་ངྷ། ཛྷ་ཤ་ལ་ཛྷ་ལ་རྣམས་ཀྱི་ཛྷ་ཤ་འོས། གྷ་ག་དྲངས། དྷུ་གྦྷྱཱཾ། དྷུ་གྦྷིཿ། བཞི་པ་རྣམས་ལ། དུ་ཧེ། དྷུ་སྦྷྱཱཾ། དྷུ་གྦྷྱཿ། ལྔ་བ་རྣམས་ལ། དུ་ཧཿ། དྷུགྦྷཾ། དྷུ་གྦྷྱཿ། དྲུག་
པ་རྣམས་ལ། དུ་ཧཿ། དུ་ཧོཿ། དུ་ཧཱཾ ངི་ལ། དུ་ཧི། ཨོ་ས་ལ། དུ་ཧོཿ། སུ་བ་ལ། ཧ་གྷ། ད་དྷ། ཁར་ལ་ཞེས་པས། གྷ་ཀ ཀུན་ནས་བསྒྱུར་པ་དང་ཞེས་པས། ག་ཥ། དྷུ་ཀྵ། བོད་ཚིག་རྣམས་རྣམ་དབྱེ་དང་པོ་དང་འདྲ། ཧེ་གྷུ་ཀ །ཧེ་དཱུ་ག །ཧེ་དུ་ཧཽ། ཧེ་དུ་ཧཿ། བཱ་དྲུ་ཧ་མུ་ཧ་
སྣུ་ཧ་སྣི་ཧཱཾ། དྲུ་ཧ། མུ་ཧ། སྣུ་ཧ། སྣི་ཧ་རྣམས་ཀྱི་ཡང་ནའོ། དྲུ་ཧ་ལ་སོགས་པ་འདི་རྣམས་ཀྱི་གོང་ལི་ཧ་ཞེས་པའི་སྐབས་ལྟར་ཧ་ཌྷ་འགྱུར་པ་དང་བཤད་མ་ཐག་པ་དུ་ཧའི་སྐབས་ལྟར་ཧ་གྷར་འགྱུར་བ་གཉིས་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། དྲུ་ཧལ། སུ། དྷཱུ་ཀ །དྷཱུ་ག །ངྷཱུ་ཊ། དྷྲུ་ཊ། ཞེས་
གཟུགས་བཞི་འབྱུང་ངོ་། རྨོངས་པ། ཨཽ་དང་ཛས་ལ། དུ་ཧཽ། དྲུ་ཧཿ། གཉིས་པ་རྣམས་ལ། དྲུ་ཧཾ། དྲུ་ཧཽ། དྲུ་ཧཿ། ཊཱ་ལ། དྲུ་ཧཱ། བྷྱཱཾ་ལ། དྷྲུ་གྦྷཱུཾ། དྷྲུ་གྦྷྱཱཾ། བྷིས་ལ། དྷྲུ་གྦྷིཿ། དྷྲུཌྦྷིཿ། ངེ་ལ། དྲུ་ཧེ། བྷྱཱ་མ་བྷྱཿལ། དྷྲུ་གྦྷྱཾ། དྷྲུ་གྦྷྱཾ། འ་ཁུ་ས། དྷྲུ་ཌྦྷྱཿ། ང་སི་ལ། དྲུ་ཧཿ། བྷྱཱཾ་བྷྱཿ། འདྲ།
