མིད། གི་རེ་ཤྩ་སེ་ན་ཀསྱ། གི་རི་ལས་མཆོད་པའིའོ། གེ་རིའི་སྦྲ་མཆོད་པའི་དོན་ ངས། ཉེ་བའི་ཡོག་པ་དང་འབྲེལ་བའི་དོན་མི་ཟད་པར་གྱུར་པའི་མཐཱ་ལས་ཊ་ཅ་ཡང་ན་འགྱུར་ཏེ། གེ་རེཿསམཱིཔཾ། ཞེས་པ་ལ་རྣམ་དབྱེ་ཕྱིས། ཨུ་པ་སྔོན། ཊཙ། ཉེ་བའི་ཡོག་པ། ཊ་ཙ་ཡང་ ཉེ་བའི་ཡོག་པ། ན
བ་བྱས་བས། ཨུ་པགི་རཾ། ཨུ་པ་གི་རི། དེ་རྣམས་ཀྱིས་མི་ཟད་པའི་བསྡུ་བ་རྣམས་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་དེའི་སྐྱེས་བུའི་བསྡུ་བ་བཤད་པར་བྱ་སྟེ། དྭི་ཏཱི་ཡཱ་ཤྲི་ཏཱ་ཏཱི་ཏ་པ་ཏི་ཏ་ག་ཏཱ་ཏྱ་སྟ་རི་ཡི་གེ། པྲཱབྡཱ་བནྡིཿ། གཉིས་པ་ལ་བརྟེན་པ་དང་། འདས་པ་དང་། ལྟུང་བ་དང་། སོང་བ་
དང་། བདེན་པ་ ཊཱཙ། རི་ཡི་ཉེ་བ་ལ། དང་། ཐོབ་པ་དང་། གྱུར་པ་རྣམས་ཀྱིའོ། བསྟེན་པ་སོགས་དོན་དེ་དག་དང་སྦྱར་བ་ལ་རྣམ་དབྱེ་གཉིས་པའི་མཐའ་སུ་པ་དང་སུ་པ་དོན་གཅིག་པའི་དེའི་སྐྱེས་བུའི་བསྡུ་བར་འགྱུར་ཏེ། བསྟེན་པའི་དོན་ལ། ཀྲྀཥྵཾ་ཤྲིཏཿ། ཞེས་པ་ལ། རྣམ་དབྱེ་གཉིས་པའི་
རྗེས་བསྡུས། ཀྲྀཥྞ་ཤིཏཿ། དེ་བཞིན་དུ་རྣམ་དབྱེ་བསྡུས་པས། འདས་བ་ལ། ཨོ་དན་བྷུཀྟཿ། ལྷུང་བ་ལ། ཕ་ལ་པ་ཏི་ཏཿ། སོང་བ་ལ། གྲཱ་མག་ཏཿ། བདེན་པ་པ། བུ་ངྔྷཱདྱསྟཿ། ཐོབ་པ་ལ། ཨརྠ་པྲཱཔྟཿ། གྱུར་པ་ལ། སུ་ནག་པ་ལ་བསྟེན་ཁ་བ་ནྣཿ། འགྲེལ་པར། འགྲོ་བའི་
དོན་དང་སྦྱོར་བ་ལ་ཡང་གཉིས་པའི་མཐའ་དོན་གཅིག་བར་འགྱུར་རོ་ཞེས་བཤད་ ཟབ་ཟོས། ཨ་མ། པས། གྲཱ་ ཨ་མ། འབྲམ་བུ་ལྷུང་། མ་གྲཱ་མཱི། གྲོང་དུ་སོང་། ཨེ་མ། སྭ་ཡཾ་ཀྟེན། སངས་རྒྱས་བདེན། སྭ་ཡཾལ་ཀྟའི་འོ། དོན་ཐོབ། སྭ་ཡཾ་དང་སྦྱོར་བའི་ བདེ་བར་གྱུར། ཀྟའི་རྐྱེན་ཅན་ཡང་སུང་པ་མཐའ་དང་སུ་པ་
མཐའ་དོན་གཅིག་པར་འགྱུར་ཏེ། ཨ་མ་ སྤ་ཡཾ་ཀྲྀ་ཏཾ། བདག་གིས་བྱས། བནྷཱ་ཀྵེ་བེ། ཁཤཱའི་ངན་པ་ལའ། ངན་པའི་དོན་གྲོང་དུ་འཕྲོ། གྲོང་དུ་འཕྲོ། ཨེ་མ། བརྗོད་པའི་ཀྟའི་རྐྱེན་ཅན་དང་སྦྱོར་བའི་ཁཊཱ་ཞིས་བའང་སུ་པ་དང་སུ་པ་དོན་གཅིག་པའི་དེའི་སྐྱེས་བུའི་བསྡུ་བ་འཇུག་སྙེ། ཁྲིལ་བསྡད། ཁཊྭཱཾ་ཨཱརཱུ་
