གཟུགས་གཅིག་པས། ཙས་ཛུའ་སོགས་ཀྱིས་ཥར། དོརྦྷྱཱཾ། དོརྦྷིཿ། ངེ་ལ། དོཥྜཿ། དོཥེ། བྷྱམ་བྷྱས་འདྲ། ང་སི་ལ། དོ་ཥྞཿ། དོ་ཥཿ། བྷྱཱ་མ་བྷྱས་འདྲ། ངས་ལ། དོཥྞོཿ། དོཥོཿ། ཨོས་ལ། དོཥྞཱཾ། དོཥཱཾ། ཨཱ་མ་ལ། དོཥྜཾ། དོཥ་ཎི། ངིལ དོཥི། དོཥ་ཎི། ཤོ་ཥི། ཨོས་ལ། དོཥྞོཿ། དོ་ཥོཿ།
སུ་པ་ལ། དོརྵུ། བོད་ཚིག་རྣམས་ལ། ཧེ་དོཿ། ཧེ་ད་ཥཽ། ཧེ་དོ་ཥེཿ། ༈ ཀྵ་ཡིག་མཐའ་ཅན། བི་བི་ཀྵ་ལ་སུ། ཧ་ལ་དཱི་པ་སོགས་ཀྱིས་སུ་ཕྱིས། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་སྔ་མའི་སྲུ་ཀ་དག་གིའང་ངོས། ཀུ་ཕྱིས། ཛྷ་ཤ་ལ་ཛྷལ་རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོས། ཥ་ཀ །གནས་མེད་པ་ལ་ལ་དན་འོས། བིའི་
བི་བི་ཊ། ཨཽ་དང་ཛས་ལ། བི་བི་ཀྵཽ། བི་བི་ཀྵཿ། གཉིས་པ་རྣམས་ལ། བི་བི་ཀྵཾ། བི་བི་ཀྵཽ། བི་བི་ཀྵཿ། ཊཱ་ལ། བ་བི་ཀྵ། བྷྱཱམ་བྷིས་ལ། དང་པོའི་གཅིག་ཚིག་གི་སྐབས་ལྟར་བསྒྲུབས་པས། བི་བི་རྦྷྱཱི། བི་བི་དྦྷིཿ། ངེ་ལ། བི་བི་ཀྵེ། བྷྱཱམ་བྷྱས་འདྲ། ང་སི་ལ། བི་བི་ཀྵཿ། བྷྱཱ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལན་བི་བིཀྵཿ།
བི་བི་ཀྵོཿ། བི་བི་ཀྵཱཾ། ངི་ལ། བ་བི་ཀྵི། ཨོས་ལ། བི་བི་ཀྵོཿ། སུབ་ལ། བི་བི་ཊྶུ། བི་བི་ཊྶུ། བོད་ཚིག་རྣམས་ལ། ཧེ་བི་བི་ཊ། ཧེ་བི་བི་ཊ། ཧེ་བི་བི་ཀྵཽུ། ཧེ་བི་བི་ཀྵཿ། ༈ ཀྵ་ཡིག་མཐའ་ཅན། ཏམྵ་ཡང་ཆོ་ག་ཐམས་ཅད་བི་བི་ཀྵ་དང་འདྲ། གོར་ཀྵ་ཡར་ལུགས་གཅིག་པ་བཀྵ་དང་འདྲ། ཡང་ན། ཡང་དག་པར་
སྦྱོར་བའི་མཐའ་དབྱི་འོས། ཥ་ཕྱིས། གོར་ཀ། གོར་ག། རྣམ་དབྱེ་ཨ་ཅ་ཐོག་མ་ཅན་རྣམས་བི་བི་ཀྵ་དང་འདྲ། བྷྱཱམ་ལ། གོ་ར་གྦྷྱཱཾ། བི་ས་ལ། གོར་གྦྷིཿ། སུ་པ་ལ། གོར་ཀྵུ་སོགས་སོ། ཕྲ་མོ། ཡང་ཀྵ་ཡིག་མཐའ་ཅན། པི་པ་ཀྵ སུ། བབར་སྐྱོང་། སུ་ཕྱིས། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས། ཥ་
