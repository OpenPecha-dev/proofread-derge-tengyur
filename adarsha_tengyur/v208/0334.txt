[ཨདྷ་]རྨཾཀཿ། ཨསྟི་ཎཱ་སྟཾ་དི་ཥྚཾ་མ་ཏི། ཨསྟི་ཎཱ་སྟི་དི་ཥྟ་ལ་བློ་གྲོས་ལའོ། །ཨ་སྟི་ ནཱ་སྟི་ དི་ཥྚ་རྣམས་ལ་དང་པའི་མཐའ་བློ་གྲོས་བརྗོད་པ་ལ་ཋཀ་འགྱུར་ཏེ། ཡང་ ཨཱ་སྟི་ཀཿ། ནཱ་སྟི་ཀཿ། དཻ་ཥྚི་ཀཿ། པྲ་གྒྟི་ཏཱ་བྱ་ཏ། ཕན་པ་ལས་སྔོན་དུ་ཡཏའོ། །དེ་ལ་ཕན་པ་ལའོ་ཞེས་པས་
ཆོས[་ལྷ་ཡིན་པ་]ལ་སྤྱོད་བ། ཕན་པ་ལས་གོང་དུ་དོན་གང་ཡིན་པ་དེ་ལ་རྐྱེན་དམིགས་བསལ་མེད་པ་རྣམས་ལ་ཡཊ་འགྱུར་རོ་ཞེས་པ་སྤྱིར་བཏང་བའོ། །ཏ་དྭ་ཧ་ཏི་ར་ཐ་ཡུ་ག་པྲཱ་སོ་གཾ། ར་ཐ་ཡུ་ག་མེད་པ། སྟོན་པ། བྲཱ་ས་དྒ་ལ་དེ་འདྲེན་ནོ་ལའོ། །ར་ཐ་ལ་སོགས་པ་འདི་རྣམས་ལ་དེ་འདྲེན་ནོ་ལ་གཉིས་
པའི་མཐའ་ལ་[ཡཏ་]འགྱུར་ཏེ། [རཐཾ་བ་]ཧ་ཏི། ཞེས་པའི་དོན་ལ། ཡཏ་ རཐྱཿ། ཤིང་རྟ་འདྲེན་པ། དེ་བཞིན་དུ། ཡཏ་ ཡུ་གྱཿ། གཉའ་ཤིང་འདྲེན་པ། ཡཏ་ བྲཱ་སོ་གྱཿ། གླང་དྲུག་གིས་འདྲེན་པ། དྷུ་རོ་ཡཏ་ཌྷ་ཀཽ། དྷུར་ལས་ཡཏ་ཌྷ་ཀ་དག་གོ། །དྷུར་ལས་དེ་འདྲེན་ནོ་ལ་ཡཏ་དང་ཌྷ་ཀ་དག་འགྱུར་
ཏེ། ཡཏ་ དྷུ་རྻཿ། དྷཽ་ར་ཡཿ། [ཌྷཀ་འཕེལ་བདྷ་ཨེལ]། ཧ་ལི་ཙ། ཧལ་ལའང་ངོ་། །ར་ཡིག་དང་བ་ཡིག་སྲོག་མེད་ཀྱི་ཨུ་བ་དྷཱ་ཨི་ཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོད་ན་ཧལ་ཕ་རོལ་ལ་རིང་བོར་འགྱུར་རོ། །ན་བྷ་ཀུ་རྪུ་རཱཾ། བྷ་ཀུ་ར་ཆུར་རྣམས་ཀྱི་མ་ཡིན་ནོ། །བྷའི་མིང་ཅན་རྣམས་དང་ཤིང་
རྟའི་འཁོར་ལོ་གཉིས་ཀྱིས་ལྕགས་འདིར་འདྲེན་པ། ཀུར་དང་ཚུར་རྣམས་ཀྱི་ཨུ་པ་དྷའི་ཨི་ཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་རྣམས་ཏལ་ཕ་རོལ་ལ་རིང་པོར་འགྱུར་བ་མ་ཡིན་ནོ། །ཁཿས་རྦ་དྷུ་རཱ་ཏ། སཏྲ་དྷུ་ར་ལས་ཁའོ། །སརྦ་སྔ་མའི་དྷུ་ར་ལས་དེ་འདྲེན་ནོལ་ཁ་འགྱུར་ཏེ། སརྦ་དྷུ་རཱི་ཎཿ། ཨེ་ཀ་དྷུ་[རཱ་ལྐུ་ཀ]་
བཱ། ཨེ་ཀ་དྷུ་ར་ལས་ཡང་ན་འཇིག་གོ། །ཨེ་ཀ་དྷུ་ར་ལས་ཁ་ཡང་ནི་འཇིག་སྟེ། ཨེ་ཀ་བྷུ་རཿ། ཨེ་ཀ་དྷུ་རཱི་ཎཿ། ཤ་ཀ་ཊཱ་དཎ། ཤ་ཀ་ཏ་ལས་ཨཎ་འོ། །ཤ་ཀ་ཊ་ལས་དེ་འདྲེན་ནོ་ལ་ཨཎ་འགྱུར་ཏེ། ཤཱ་ཀ་ཊཿ། ཧ་ལ་སཱཾ་རཱ་ཏ་ཋཀ །ལྕགས་འདིང་ཐམས་ཅད་འདྲེན་པ། [ཁ་ཁའི་ཐ་སྨིན]། ཧ་ལ་
སཱེ་ར་ལས་ཋཀ་འོ། །ཧ་ལ་དང་སཱི་ར་ལས[་དེ་འདྲེད་ནོ་]ལ་ཋཀ་འགྱུར་ཏེ། ཁ་ཕྱིས་པ། ལྕགས་འདིང་གཅིག་འདྲེན་པ། [ཁ་ཨའེས། ཋཀ ཧཱ་ལི་ཀཿ]། ཐོང་གཤོལ་འདྲེན་པ། [སྐེ་ར་ལས་ཋཀ་འཕེལ་བ་ཐཨེཀ]། །སཻ་རི་ཀཿ། ཐོང་བ་འདྲེན་པ། ཏ་ཏྲ་ [ཋཀ་ ]འཕེལ་བ། ཤིང་རྟ་[འདྲེན་]པ། སཱ་ཏྱུཿ། དེར་ལེགས་པ་ལའོ། །དེར་ལེགས་པ་ལ་བདུན་
