སུ་བྲྀ་ཤྩི། སུ་བྲྀ་ཤྩོཿ། བོད་ཚིག་རྣམས་ལ། ཧེ་སུ་དྲྀ་ཌ། ཧེ་སུ་བྲྀ་ཊ། ཧེ་སུ་བྲྀ་ཤྩཿ། ༈ ཏ་ཡིག་མཐའ་ཅན། མ་ཧ་ཏ་ལ། སུ། ཨུ་ཀ་འགྲོ་བ་སོགས་ཀྱིས་ནུམ། ས་མཐའ་མ་ཧ་ཏ་དག་གི་ཡང་དག་པར་སྦྱོར་བའི་འོས་ཨུ་པ་བྷྱཱ་རིང་པོ། ཏ་ལ་སོགས་ཀྱིས་སུ་ཕྱིས། ཡང་
དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས། ཏ་བྱིས། མ་ཧཱ་ན། ཨཽ་དང་ཛས་ལ། མ་ཧཱ་ནྟཽ། མ་ཧཱ་ནྟཿ། ཨམ་དང་ཨོ་ལ། མ་ཧཱ་ནྟཾ། མ་ཧཱ་ནྟོ། ཤས་ལ། མ་ཧ་ཏེ། ཆེན་པོ། ཊཱ་ལ། བྷྱཱམ་དང་བྷིས་ལ། ཛྷ་ཤ་ལ་ཛྷ་ལ་རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོས། ཏྲ་དྲ། མ་ཧ་དྡྷྱཾ། མ་ཧ་དྡྷིཿ། ངེ་ལ། མ་ཧ་ཏེ། བྷྱཱ་མ་བྷྱས་འདྲ། ང་སི་ལ།
མ་ཧ་ཏཿ། བྷྱཱམ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། མ་ཧ་ཏཿ། མ་ཧ་ཏོཿ། མ་ཧ་ཏོཿ། མ་ཧ་ཏཱཾ། བདུན་པ་རྣམས་ལ། མ་ཧ་ཏི། མ་ཧ་ཏོཿ། མ་ཧ་ཏྶུ། བོད་ཚིག་ལ། ཧེ་མ་ཧ་ན། ཧེ་མ་ཧཱ་ནྟཽུ། ཧེ་མ་ཧཱ་ནྟཱཿ། བྷ་ཝ་ཏ་ལ་ཡང་ཨུ་པདྦྷཱ་རིང་པོར་མི་འགྱུར། གཞན་ཆོ་ག་ཐམས་ཅད་མ་ཧ་ཏ་དང་འདྲའོ། དྷེ་མ་ཏེ་ཅེས་བ་ལ་ཡང་། ཨ་ཏ་ཨས་
མཐའི་འང་བྱིངས་མ་ཡིན་པའི་འོ་ཞེས་པས་སུའི་སྐབས་སུ་ཨུ་པདྷཱ་རིང་པོ་བྱེད། གཞན་ཐམས་ཅད་བྷ་ཝ་ཏ་དང་འདྲའོ། །ཨ་ཏྲ་བྷ་ཝ་ཏ་དང་། ཏ་ཏྲ་བྷ་ཝ་ཏ་ཅེས་པ་ལའང་མཆོད་པའི་དོན་ལ་སུའི་སྐབས་སུ་ཨུ་པདྷཱ་རིང་པོ་བྱེད། གཞན་མ་ཐམས་ཅད་ངྷིམ་ར་དང་འདྲའོ། །ཛ་ཤ་སུ་ཏ་ཅེས་པའ་བ་ཨ་ཏྲ་བྷ་ཝ་ཏ་དང་
འདྲའོ། །ཨ་བྷ་ཝ་ཏ་དང་བྷ་ག་ཝ་ཏ་གཉིས་ཀྱང་ཆོ་ག་ཐམས་ཅད་དྷི་མ་ཏ་དང་འདྲ། སཾ་བུ་དྡྷི་ལ་འདམ་ཀ་ཅན་ཏེ། དྷི་མ་ཏ་དང་འདྲ་པའི་ཕྱོགས་ལ། ཧེ་བྷ་ག་ཝ་ན། ཧེ་ཨ་གྷ་ཝ་ན། ཡང་ན་འདམ་ཀ། དྷ་ཝ་ཏ་བྷ་ག་ཝ་ཏ་ཨ་ཏ་ཨ་གྷ་ཝ་ཏ་རྣམས་ཀྱི་ཡང་ན་སྟེ་ཨ་ཝ་ཨོ་ཏ་ཡེང་ངོས། ཨ་ཝ་ཨོ་ཏ་དུ་འགྱུར། ཡང་སྤྲས་ཏ་
སྲོ་ག་མེད་རུ་ཨུ་བརྗོད་དོན། སྐལ་ལྡན། ཧེ་བྷ་བོཿ། ཧེ་ཨ་གྷོ་ལ། ༈ ཨུ་བྷེ་ཨ་བྷྱ་སྟཾ། གཉིས་པ་ལ་ཨ་བྲུ་སྟའོ། །ད་ད་ཏ་ཅེས་པ་ལྟ་བུའི་ཡི་གེ་འདྲ་བ་གཉིས་བཏུད་ནས་ཡོད་པ་དེ་ལ་ཨ་བྷྱ་སྟའི་མིང་ཅན་ཞེས་བརྡ་ཆད་བྱེད་དོ། །ད་ད་ཏ་ལ། སུ། ནཱ་བྷ་སྟཱ་ཙྪ་ཏུ། ཨ་བྷྱ་སྟ་ལས་ཤ་ཏུའི་མ་ཡིན་ནོ། །ཨ་བྷྱ་སྟའོ་མིང་
