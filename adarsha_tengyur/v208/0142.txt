རཱ་ཛཿ། བྷྱཱཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། རཱ་ཛཿ། རཱ་ཛོཿ། རཱ་ཛཱཾ། ངི་ལ། རཱ་ཛི། ཨོས་འདྲ། སུ་པ་ལ། ཛ་ཥ། ཥ་ཌ། ཁར་ལ་ཙར་འོས། ཊཱ་ཊཱ། རཱ་ཊཱུ། བོད་ཚིག་རྣམས་ལ། ཧེ་རཱ་ཊ། ཧེ་རཱ་ཌ། ཧེ་རཱ་ཛཽ། ཧེ་རཱ་ཛཿ། བི་བྷཱ་ཛ། དེ་ཝེ་ཛ། བི་ཤྭ་སྲྀ་ཛ། པ་རི་སྲྀ་ཛ། པ་རི་པྲ་ཛ་རྣམས་ཀྱང་རཱ་ཛ་
དང་འདྲའོ། ༈ །བི་ཤྭ་རཱ་ཛ། སུ། ཛ་ཥ། ཥ་ཌ། བི་ཤྭ་རཱ་ཌ། བི་ཤྭ་སྱ་བ་སུ་རཱ་ཊོཿ། པི་ཤྭའི་བ་སུ་རཱ་ཊ་དག་ལའོ། བི་ཤྭ་ཞེས་པ། བསུ་དང་རཱཛ་རཱ་ཊ་ཞེས་པར་འགྱུར་བའི་སྔོན་ན་ཡོད་ན་རྟགས་མ་[བའི་]རིང་བོར་འགྱུར་ཏེ། བི་ཤྭཱ་རཱི་ཏ་ལྟ་བུ་སྟེ། ཧ་ལ་ཐོག་མ་ཅན་རྣམས་ལ་རཱ་ཛ་རཱ་ཥ་ཞེས་པར་འགྱུར་ལ།
དེའི་ཚེ་བི་ཤྭའི སྣ་ཚོགས་གསལ་བ། ཨ་རིང་པོར་འགྱུར་བ་མ་གཏོགས་རཱ་ཛ་རང་གི་སྐབས་སུ་བཤད་པ་དང་ཁྱད་པར་མེད་དོ། བསུ་ཞེས་པ་ཨུ་ཡིག་མཐའ་ཅན་ནི། ཆོ་ག་ཐམས་ཅད་བྷ་ནུའི་སྐབས་དང་འདྲ། དེའི་སྔོན་དུ་བི་ཤྭ་ཞེས་པ་བྱུང་ན། བི་ཤྭའི་བསུ་རཱ་ཊ་དག་ལ་འོས་རིང་པོར་འགྱུར་ཏེ།
དཔེར་ན། བི་ཤྭཱ་བསུཿལྟ་བུའོ། ༈ །ཛ་ཡིག་མཐའ་ཅན་བྷྲྀ་སྫ་ལ། སུ། སྐོཿསཾ་ཡོ་གཱ་དྱོ་ར་ནྟེ་ཙ། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་ལསྔ་མའི་སྲ་ཀྲ་དག་གེའང་ངོ་། །རྟགས་མཐའ་ཡང་དག་པར་སྦྱོར་བ་སྟེ། སྲོག་མེད་གཉིས་བརྩེགས་པའི་མཐའི་སྔ་མ་དེས་ཡིག་གམ་ཀ་ཡིག་ཡིན་ན་ཛྷལ་སོ་
སོར་བསྡུ་བའི་ཡི་གེ་རྣམས་ཕ་རོལ་དང་ཚིག་མཐའ་ལ་དབྱི་བར་བྱ་སྟེ། ས་ཕྱིས། བྲ་ཤྩ་བྷྲ་སྫ་སོགས་ཀྱིས་ཛ་ཥ། གནས་མེད་པ་ལ་ཡང་ན་འོས་ཌ་འམ་ཊ། བྷྲྀ་ཊ། བྷྲྀ་ཊ། ཨཽལ། ཛྷ་ཤལ་ཛྷལ་རྣམས་སོས། ཛྷ་ཤ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཛའི་སྔ་མའི་ཛྷ་ལ་སོ་སོར་བསྡུ་བའི་ནད་ཚན་
ས་ཡིག་དེ་ཁོ་རང་དང་སྐྱེ་གནས་མཐུན་པའི་ད་ཡིག་ཏུ་འགྱུར། ས་ཏུའི་ཤ་ཙུ་ལ་ཤ་ཙུ་འོས། དང་དྲངས། བྷྲྀ་ཛྫཽ། ཛས་ལ། བྷྲྀ་ཛྷྲཿ། ཨམ་དང་ཨཽ་ལ། བྷྲྀ་ཛྫཾ། བབྷྲྀ་ཛྫཽ། ཤས་ལ། བྷྲྀ་ཛྫཿ། ཊཱ་ལ། བྷྲྀ་ཛྫཱ། བྷྱཱཾ་བྷིས་ལ། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་ལ་སྔ་མའི་ས་ཀ་དག་གིས་འོས།
ས་ཕྱིས། ཛ་ཥ། ཥ་ཌ། བྷྲྀ་དྦྷྱཱཾ། བྷྲྀ་དྦྷིཿ། དེ་ལ། བྷྲྀ་ཛྫེ། བྷྱཱཾ་བྷྱས་འདྲ། ང་སི་ལ། བྷྲྀ་ཛྫྲཿ། བྷྱཱཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། བྷྲྀ་ཛྫཿ། བྷྲྀ་ཛྫོཿ། བྷྲྀ་ཛྫཱཾ། བདུན་པ་རྣམས་ལ། བྷྲྀ་ཛྫི། བྷྲྀ་ཛྫྲོཿ། བྷྲྀ་ཊྶུ། བོད་ཚིག་ལ། ཧེ་བྷྲྀ་ཊ། ཧེ་བྷྲྀ་ཊ། ཧེ་བྷྲྀ་ཌ། ཧེ་བྷྲྀ་ཛྫོ། ཧེ་བྷྲྀ་ཛྫཿ། ༈ །ཛ་ཡིག་མཐའ་ཅན།
