རྣམས་བས་ལའོ། །ཉེར་བསྒྱུར་ཨུ་པ་ཨ་རུ་ཨ་དྷི་ཨཱང་རྣམས་སྔ་མའི་བས་དང་སྦྱོར་བ་ལ་བདུན་པའི་དོན་ལ་གཉིས་པ་འགྱུར་ཏེ། ཨུ་པ་བས་ཏི་བཻ་ཀུཎྛཾ་ཧ་རིཿ། ཞེས་པ་ནས། ཨཱ་བས་ཏི་བཻ་ཀུཎྛཾ་ཧ་རིཿ། ཞེས་བའི་བར་གྟ་བུའོ། །འགྲེལ་པར། ཨུ་པ་སོགས་སྣ་མའི་བས་དང་སྦྱོར་
བ་ཡིན་ཀྱང་གནས་པའི་ཡུལ་དེ་བཟའ་བཏུང་མེད་པའི་ས་ཡིན་ན་བདུན་པའི་དོན་ལ་གཉིས་པ་མི་འགྱུར་ཞིང་བདུན་པ་ཉིད་དོ་ཞེས་པ་བྱུང་བས། བ་ནེ འཕྲོག་བྱེད་མཐོ་རིས་ལ་ཀུན་དུ་གནས། ཨུ་པ་བས་ཏི། ཨུ་བས་རྦ་ཏ་སོཿཀཱ་རྻཱ་ངྷི་གུ་བརྻཱ་དི་ཥུ་ཏྲི་ཥུ་དྭི་ཏཱི་ཡཱ་མྲེ་ཌི་ཏཱཾ་ཏེ་ཥུ་ཏ་
ཏོ་ནྱ་ཏྲཱ་པི་དྲྀ་ཤྱ་ཏེ། གཉིས་དང་། ཐམས་ཅད་དེ་དག་གི་བྱ་བ་དང་། འཛིན་པ་དང་། ཡང་ཡང་ལ་སོགས་པར་གསུམ་དུ་དང་། ཟློས་པའི་མཐའ་རྣམས་སུ་དང་། དེ་ལས་གཞན་དུ་ཡང་དག་གཉིས་པ་མཐོང་ངོ་། ནག་སལ་ཉེ་བར་གནས། །ཨུ་པ་ཡ་དང་སརྦ་ཏས་ཀྱི་མཐའ་ཅན་དེ་དག་གི་བྱ་བ་སྟེ།
དེ་གཉིས་དང་འབྲེལ་པའི་མིང་གཞན་དང་འཛིན་པ་སྟེང་ན་པའི་དོན་ངྷིཀ་སྦྱོར་བ་དང་། ཡང་ཡང་ལ་སོགས་པར་གསུམ་པ་ཟློས་པའི་མཐའ་ཞེས་པ། ཨུ་པ་རི། ཨ་དྷི། ཨ་ཧོ་གསུམ་ཡང་ཡང་ཟློས་པའི་མཐའ་རྣམས་དང་། དེ་ལས་གཞན་དུ་ཡང་ཞེས་བ། ཨ་བྷི་ཏཿ། པ་རི་ཏཿ། ས་མ་ཡཱ།
ནི་ཀ་ཥྰ། ཧཱ། རྀ་ཏེ། པྲ་ཏི་རྣམས་སྦྱོར་བ་ལའང་རྣམ་དབྱེ་གཉིས་པ་འགྱུར་ཏེ། དཔེར་བརྗོད་རིམ་པ་བཞིན། ཨུ་པ་ཡ་ཏཿ། གོ་པཱ་ཀྲྀ་ཥྞཾ། སརྦ་ཏཿཀྲྀ་ཥྞཾ། དྷི་ཀ་ཀྲྀ་ཥྞ་བཀྟཾ། ཨུ་པ་རྻུ་པ་རི་ལོ་ཀཾ་ཧ་རིཿ། ཨ་ངྷྱ་ངྷི་ལོ་ཀཾ། ཨ་དྷོ་ཉ་དྷོ་ལོ་ཀཾ། ཨ་བི་ཏཿ བྀཥ། པ་རི་ཏཿཀྲྀཥྞཾ། སམ་ཡཱ་
ཀྲྀ་ཥྞཾ། ནི་ཀ་ཥཱ་ཀྲྀ་ཥྞཾ། ཧཱ་ཀྲྀཥྞཾ། རྀ་ཏོ་ཀྲྀཥྞཾ། པྲ་ཏི་ཀྲྀཥྞཾ། ཨནྟ་རཱ་ཨནྟ་རེ་ཎ་ཡུ་སྟེ། ཨནྟ་ར་ཨནྟ འདྲོག་བ་ནག་བོ་གཉིས། ཐམས་ཅད་དུ་ནག་པོ་བ། ངན་པ་ནག་པོ་མི་མཚན། སྟེང་མ་སྟེང་མའི་འཇིག་རྟེན་ལ་འཕྲོག་བྱེད། ལྷག་པ་ལྷག་པའི་འཇིག་རྟེན། འོག་མ་འོག་མའི་འཇིག་རྟེན། མངོན་པར་ནག་ཨྭོ། ཡོངས་སུ་ནག་པོ་འདུས་བའི་ནག་པོ་ལ།
ཉེ་བའི་ནག་པོ་ལ། ཀྱེ་མ་ནག་པོ། ངེས་པར་ནག་པོ། སོ་སར་ནག་པོ། རེ་ཎ་སྦྱོར་ལའོ། །ཨནྟ་ར་དང་ ནང་། ཨནྟ་རེ་ཎ བར་རམ་མཚམས་སམ་དབྲལ། སྦྱོར་བ་ལའང་རྣམ་དབྱེ་གཉིས་པ་འགྱུར་ཏེ། ཨནྟ་རཱ་གཱ་རྟ་པ་ ཁྱིམ་བདག་གི་ནང་། ཏྱཾ། ཨནྟ་རེ་ཎ་ད་རྨཾ་སུ་ཁནྣ་བྷ་བ་ཏི། ཆོས་ཀྱི་བར་དུ་གཅོད་པ་ལ་བདེ་བར་མི་འགྱུར།
