ཨ་བི་ཛི་ཥ་ཏ། སོགས་སོ། ལིཊ་ལ། གཉིས་བརྗོད་པ་སོགས་ཀྱི། སྐྱེས་བུ་དང་པོ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛཱ་ཏེ། བི་བི་ཛི་རེ། སྐྱེས་བུ་དབུས་མ་གསུམ་ལ། བི་བི་ཥེ། བི་བི་ཛཱ་ཐེ། བི་བི་ཛི་ཌྷྭེ། སྐྱེས་བུ་ཕྱི་མ་གསུམ་ལ། བི་བི་ཛེ། བི་བི་ཛི་བ་ཧེ། བི་བི་ཛི་མ་ཧེ། ལུཊ་ལ་ཨིཊ་བྱིན།
བི་ཛ་ལས་ཨིཊ་འོས། ངིཏ་ཡིན་པས་ཡོན་ཏན་མི་བྱེད་དེ། བི་ཛི་ཏཱ། བི་ཛི་ཏཱ་སེ། སོགས་སོ། དེ་བཞིན་དུ། སྨོན་པའི་ལིང་ལ། བི་ཛི་ཥཱི་ཥྚ། བི་ཛི་ཥཱི་ཡཱ་སྟཱཾ། བི་ཛི་ཥཱི་ར་ན། སོགས་སོ། ལྲྀཊ་ལའང་ཨིཊ་བྱིན། བི་ཛ་ལས་ཨིཊ་འོས། ངིཏ་ཡིན་པས་ཡོན་ཏན་མི་བྱེད། བི་ཛི་ཥྱ་ཏེ། བི་ཛི་ཥྱི་ཏེ།
