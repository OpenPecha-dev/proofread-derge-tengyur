སྦྱར་བས། སྔར་ཚིག་མང་བ་དང་བརྗོད་པ་མ་བདེ་བ་རྣམས་ཚིག་ཉུང་ཞིང་བརྗོད་པ་བདེ་བར་འགྱུར་བ་ནི་ཚིག་སྡུད་ཀྱི་དགོས་པའོ་ཞེས་བཤད་དོ། དེ་རྣམས་ལས་དང་པོ་མི་ཟད་པའི་བསྡུ་བ་བཤད་པར་བྱ་སྟེ། ཧ་རོོ་ཨ་ངྷིཀྲིཏྱ། ཞེས་པ་ཚིག་སྦྱོར་རམ་ངག་ཅེས་བྱ་སྟེ་ཚིག་
སྡུད་བྱེད་པའི་གཞི་ཡིན་ནོ། པྲ་ཐ་མཱནི་རྔྡིཥྚཾ་ས་མཱ་ས་ཨུ་པ་ས་ཇཾ་ནཾ། དང་ངེས་པར་སྦྱོར་བའི་ཚིག་སྡུད་ཉེར་བསྒྱུར་རོ། རྣམ་དབྱེ་དང་པོ་དང་སྦྱོར་བའི་བསྡུ་བ་བྱེད་པའི་ཡུལ་རྣམ་དབྱེ་དང་པོ་ཐོབ་པ་དེ་ལ་ཉེར་བསྒྱུར་ཞེས་མིང་ འཕྲོག་བྱེད་དྱ་བ་ལྷག་པོར། འདོགས་བྱེད་དོ། ཨུ་པ་སརྫ་ནཾ་པཱུཾརྦཾ།
ཉེར་བསྒྱུར་སྔོན་དུའོ། ཚིག་སྡུད་ཀྱི་སྐབས་འདིར་གོང་དུ་ཉེར་བསྒྱུར་ཞེས་མིང་འདོགས་བྱས་པ་དེ་ཚིག་སྦྱོར་གྱི་སྐབས་མཐའ་དབུས་གང་ན་ཡོད་ཀྱང་སྔོན་དུ་སྦྱོར་བར་བྱེད་པས། ཨ་ངྷི་ཧ་རོོཀྲྀཏྱ། ཞེས་གནས་པ་འདི་ལ་པྲཱ་ཏི་པ་དི་ཀ་ཞེས་མིང་འདོགས་བྱེད་དོ། སུ་པོ་ངྷཱ་ཏུ་པྲཱ་ཏི་པ་དི་ཀ་ཡོཿ།
བྱིངས་པྲཱ་ཏི་པ་དི་ཀ་དག་ལས་སུ་པ་ཀྱིའོ། བྱིངས་དང་སྦྱོར་བ་དང་ཚིག་སྡུད་ཀྱི་སྐབས་པྲཱ་ཏི་པ་དི་ཀ་ཞེས་མིང་འདོགས་བྱས་པ་བསྡུ་བྱ་དེར་གནས་པའི་རྣམ་དབྱེ་རྣམས་དབྱི་ཞིང་དེར་མ་ཟད་གཞི་དང་། ཕྲད་དང་། ཁྱད་པར་དང་འདྲ་བའི་སྐད་དོད་སོགས་ཚིག་གི་རྗེས་འབྲང་རྣམས་ཀྱང་དབྱིབས་ཧ་རི་
ལ་སྦྱིན་པའི་དི་དང་ཀྲྀཏྱ་ཕྱིས། ཨངྷི་ཧ་རི། དེ་ལྟར་རྣམ་དབྱེ་རྗེས་འབྲང་དང་བཅས་པ་ཕྱིས་པས་ཚིག་སྡུད་ཀྱི་མིང་ཐོགས་པ་དེ་ལ་སླར་ཚིག་ཏུ་སྒྲུབ་བྱེད་ཀྱི་རྣམ་དབྱེ་རྣམས་འཇུག་པ་ལས་མཚོན་བྱེད་སུ་བྱིན། ཨ་ངྷི་ཧ་རི་སུ། ཨ་བྱ་ཡཱི་བྷཱ་བཤྩ། མི་ཟད་པར་གྱུར་པ་ལའང་ངོ་། མི་ཟད་
པར་ཀྱུར་པ་སྟེ་མི་ཟད་པ་སྔོན་ན་ཡོད་པ་མི་ཟད་པའི་བསྡུ་བ་བྱས་པ་དེ་ལ་རྣམ་དབྱེ་ཐམས་ཅད་འཇིག་ཅིང་བྱ་བ་ཡང་མེད་པས་སུ་ཕྱིས། དེའི་རྗེས་སུ་དོན་སྟོན་པའི་ཚིག་གཞན་སྦྱོར་དགོས་ཏེ། ཨངྷི་ཧ་རི་པྲབྲྀཏྟཱ་ཀ་ཐཱ། དེ་བཞིན་དུ། རཱ་ཡམ་ཏི་ཀྲཱནྟ། རྣམ་དབྱེ་རྗེས་འབང་དང་བཅས་པ་ཕྱིས། ཨ་
ཏི་རེེ། ཞེས་པ་ལ། མི་ཟད་པའི་བསྡུ་བའི་མཐའ་མ་ནིང་དུ་འགྱུར་ཞིང་། མ་ནིང་ལ་ཐུང་ངུར་འགྱུར་བས། ཨ་ཏི་རི། འབྱོར་ལྡན། དེ་བཞིན་དུ། ནཱ་བ་མ་ཏིཀྲཱནྟལ། གྲུ་ཤིན་ཏུ་ནོན་པ། ལྷག་པར་རབ་ཏུ་འཕེལ་བའི་བརྗོད་པ་འཕྲོག་བྱེད། ཆོས་འབྱོར་པ་དང་ལྡན་པ། ཨ་ཏི་ནུ། ཧྲུ་ནུབ་ལ་ཁར། དེ་རྣམས་མི་ཟད་
