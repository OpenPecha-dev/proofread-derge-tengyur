དྱཏ་སོགས་རྣམས་ལ་ཐམས་ཅད་རྣམས་ཀྱི་རྟག་ཏུའོ། །ཟླས་དབྱེའི་སྐབས་འདིར་ཏྱད་སོགས་རྣམས་ལ་རྟགས་གཞན་ཐམས་ཅད་རྟག་ཏུ་དབྱེ་སྟེ། ས་ཙ་དེ་བ་དདྟཤྩ། ཞེས་པ་ལ། རྣམ་དབྱེ་ཚིག་གྲོགས་དང་བཅས་པ་ཕྱིས །ཏོོ། འགྲེལ་པར། དྱད་སོགས་གཉིས་ཚོགས་པ་ལ་ནི་སྔོན་མ་དབྱིའོ་ཞེས་
བཤད་པས། སཙཱ་ཡཤྩ། ཞེས་པ་ལ། རྣམ་དབྱེ་ཕྲད་སོགས་དང་མིང་གོང་མ་ཕྱིས། ཡོོ། གྲཱ་མྱ་པ་ཤུ་སཾ་གྷེ་ཥྭ་སྟཱི། གྲོང་གི་ཕྱུགས་འདུས་ དེ་དང་ལྷས་བྱིན་ཡང་། བ་རྣམས་ལ་གཞོན་ནུ་མ་ཡིན་པ་ལ་བུད་མེད་འོ། །ལྷས་བྱིན་དེ་དག། ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འདིར་གྲོང་དུ་གནས་པའི་ཕྱུགས་འདུས་
པ་བརྗོད་པ་ལ། གཞོན་ དེ་དང་གང་དང་། ནུ་མ་འམ་ཆུང་ངུ་མ་ཡིན་པ་སྟེ་བོར་བའི་གཞན་མ་ དེ་གང་དག། རྟགས་མོ་རྟགས་རྣམས་གནས་ཤིང་རྟགས་གཞན་མ་རྣམས་ཡང་ན་དབྱི་སྟེ། གཱ་བཤྩ་ཨིམཱཤྩ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། གོ་དབྱི་པ་དང་མི་དབྱི་བ་ཡང་ན་བར་བྱས་པས། ཨིམཱཿ། ཡང་ན་ ག་བིམཱཿ།
ག་དེ་བཞིན་དུ། ཨཛཱཤྩ་ཨི་མཱཤྩ། ཞེས་པ་ལ། ཨིམཱཿ། ཡང་ན། ཨཛ་ཨིམཱཿ། གྲོང་དུ་གནས་པ་མ་ཡིན་ན་མི་དབྱེ་སྟེ། རུ་རབ་ཤྩ་ཨིམཱཤྩ། བལ་རྣམས་དང་འདི་རྣམས་དང་། ཨི་མཱ་ནི་ཨི་དཾ་བུད་མེད་དུ་བསྒྲུབས་པའོ། ཞེས་པ་ལ། རུ་རུ་ཨིམཱཿ། ཕྱུགས་མ་ཡིན་ན་ བ་ལང་འདི་རྣམས། བ་ལང་འདི་རྣམས། འདུས་
པའི་དོན་ཡིན་ཀྱང་མི་དབྱི་སྟེ། རེ་རྣམས་འདི་རྣམས། བྲཱཧྨ་ཎཱཤྩ་ཨི་མཱཤྩ། ར་འདི་རྣམས། ཞེས་པ་ལ། བྲཱཧྨ་ཎ་ཨི་མཱཿ། གཞོན་ནུ་ཡིན་ རི་དགས་རུང་རྣམས་དང་འདི་རྣམས་དང་། ན་མི་དབྱི་སྟེ། བཏྶཱཤྩ་ཨི་མཱཤྩ། ཞེས་པ་ལ། བཏྶ་ཨི་མཱཿ། རི་དགས་རུ་ལུ་འདི་རྣམས། དེ་རྣམས་ཀྱིས་ཟླས་དབྱེའམ་གཉིས་གཉིས་
པའི་བསྡུ་བ་རྣམས་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་ཚིག་སྡུད་བྲམ་ཟེ་རྣམས་དང་འདི་རྣམས་དང་། བྲམ་ཟེ་འདི་རྣམས། བཞི་པོའི་ཆོ་གཅི་རིགས་ཐུན་མོང་དུ་བཤད་པ་ལས། རྀག་པཱུར་བྔྲཱཿཔཐཱ་མ་ནཀྵེ། རྀཙ་པུར་ངྷུར་བེ་བྱེ་རྣམས་དང་འདི་རྣམས་དང་། བེའུ་འདི་རྣམས། པ་ཐིན་རྣམས་ཀྱི་ཨཀྵ་མ་ཡིན་པ་ལས་ཨ་པའོ། །ངྷུར་དང་
ཨཀྵ་སྦྱོར་བ་མ་ཡིན་པའི་རྟགས་དེ་རྣམས་ལས་བསྡུ་བའི་མཐའ་ལ་ཨ་པ་ཀྱི་རྐྱེན་འགྱུར་ཏེ། ཨརྔྷ་སྱརཀ། ཅེས་པ་རྣམ་དབྱེ་བསྡུས། ཨ་པ་བྱིན་པ་རྗེས་འབྲེལ། ཨརྔྷཙཿ། རིག་བྱེད་ཀྱི་ཚིག་ཕྱེད། དེ་བཞིན་དུ། བིཥྞོཿཔཱུཿ། ཁྱབ་འཇུག་གི་གྲོང་། ཞེས་པ་ལ། བཾཥྞུཔུརཾ། ཁྱབ་འཇུག་གྲོང་། རཱཛྙཿངྷཱུཿངྷཱུཿརྒྱལ་པོའི་ག་རྩ། ཞེས་པ་ལ། རཱ་
