ངྷོད་ཀ་སྱོ་པསཾ་སྐུ་རུ་ཏེ། རུ་ཛཱ་རྠཱ་ནཱཾ་བྷཱ་བ་ཙ་ནཱ་ནཱ་མ་ཛྭ་རེཿ། ནད་ཀྱི་དོན་རྣམས་ཀྱི་དངོས་པོ་བརྗོད་པ་རྣམས་ཀྱི་ཛྭར་མ་ཡིན་པ་ལའོ། ནད་དང་བརྡེག་པ་ལ་སོགས་པ་འཚེ་བའི་དོན་ཅན་གྱི་བྱིངས་རྣམས་ལས་གྲུབ་པ་རྣམས་ཀྱི་ཛྭར་མ་ཡིན་པ་སྟེ་མ་རྟོགས་པའི་ཚིག་གི་སྦྱོར་བའི་རྟ་གས་ལ་
ཆུ་འཕེལ་གྱི་བྱ་བ་ཡང་དག་པར་འབད་པར་བྱེད། ལས་ཀྱི་དོན་ལ་དྲུག་པ་འགྱུར་ཏེ། ཙོོ་ར་སྱ་རུ་ཛ་ཏི། ཙོོར་སྱ་ནི་ཧནྟི། ཛྭར་སྦྱོར་བ་ལ་གཉིས་པ་ཉིད་དེ། ཙོོརཾ་ཛར་ཡ་ཏི། ཨཱཤིཥི་ནཱཐཿ། ནཱ་ཐའི་སྨོན་པ་ལའོ། །ནཱ་ཐའི་སྒྲ་སྨོན་པ་དང་ཤིས་པ་བརྗོད་པ་དང་སྦྱོར་བ་ལ་དྲུག་པ་འགྱུར་ཏེ། །སརྦིཥོ་ནཱ་ཐ་ཏེ་ད་
དཱ་ཏི། ཛཱ་སི་ནི་པྲ་ཧ་ཎ་ནཱ་ཊ་ཀྲཱ་ཐ་པི་ཥཱ་ཧི་ངྶརྠ་ནཱཾ། ཛཱ་སི་ནི་པྲ་ཧ་ཎ་ནཱ ཆོམ་རྐུན་ནད་ཀྱི་འདེབས། ཆོམ་རྐུན་ངེས་བར་གསོད། ཊ། ཀྲཱ་ཐ་པི་ཥ་རྣམས་ ཆོམ་རྐུན་<དིམས་,རིམས་>ཀྱི་འདེབས། ཀྱི་འཚེ་བའི་དོན་རྣམས་ཀྱིའོ། ཛཱ་སི་ལ་སོགས་པ་འདི་རྣམས་འཚེ་བའི་དོན་བརྗོད་པ་དང་སྦྱོར་བ་ལ་ མགོན་པོས་མར་སྐྱིན་པར་བྱེད།
གཉིས་པའི་དོན་ལ་དྲུག་པ་འགྱུར་ཏེ། ཙོོར་སྱོ་ཛྡཱས་ཡ་ཏི། ཙོོར་སྱ་ནི་པྲ་ཧ་ནྟི། ཙོོར་སྱོ་ནྣ་ཊ་ཡ་ཏི། ཙོོ་ར་སྱ་ཀྲཱ་ཐ་ཡ་ཏི། ཙོོ་ར་སྱ་ཀྲཱ་ཐ་ཡ་ཏི། ཙོོ་ར་སྱ་པི་ན་ཥྚི། འཆེ་བའི་དོན་མ་ཡིན་ན་མི་འགྱུར་ཏེ། ངྷཱ་ནཱཿཔི་ན་ཥྚི། བྱ་བ་ཧྲྀ་པ་ཎོཿསམཪྦ་ཡོ། བི་ཨ་བ་ཧྲྀ་པ་ཎ་དག་གི་རྒྱལ་བ་ལའོ། བི་དང་ཨ་བ་སྡ་མའི་ཧྲྀ་དང་
པ་ཎ་རྒྱལ་བའི་ ཆོམ་རྐུན་འཇོམས་པར་བྱེད། ཆོམ་རྐུན་ངས་པར་རབ་ཏུ་གསོད་པར་བྱེད། ཆོས་རྐུན་འཆིང་བར་བྱེད། ཆོམ་རྐུན་ལི་སྣུན་བར་བྱེད། ཆོམ་རྐུན་ཕྱེ་མར་འཇག་པར་བྱེད། དོན་བརྗོད་པ་དང་སྦྱོར་བ་ལ། ཡོས་འཐག་པར་བྱེད། གཉིས་པའི་དོན་ལ་དྲུག་པར་འགྱུར་ཏེ། ཤ་ཏ་སྱ་བྱ་བ་ཧར་ཏེ། ཤ་ཏ་སྱ་བྱ་བ་ཎ་ཏེ། རྒྱལ་
བ་བརྗོད་པ་མ་ཡིན་ན་དྲུག་པ་མི་འགྱུར་ཏེ། བྲཱཧྨ་ཎཾ་པ་ཎཱ་ཡ་ཏི། དི་བསྟ་ད་རྠ་སྱ། དི་བའི་དེའི་དོན་གྱིའོ། དི་བ་ཞེས་པའང་ཅེའི་སོན་ཏེ་བཤད་མ་ཐག་པ་ལྟར་རྒྱལ་བའི་དོན་ལ་སྦྱོར་བ་ལའང་གཉིས་པའོ་དོན་ལ་དྲུག་པ་འགྱུར་ཏེ། ཤ་ཏ་སྱ་དཱི་བྱ་ཏི། བི་བྷཱ་ཥོ་པ་སརྒེ། བརྒྱ་ཐམས་པའི་རྒྱལ་བར་བྱེད། ཉེར་བསྒྱུར་
ལ་ཡང་ནའོ། ཉེར་བསྒྱུར་སྡ་མའི་ བྲམ་ཟེ་ལ་བསྟོད་པ་བྱེད། །དི་བ་ལས་རྒྱལ་བའི་དོན་བརྗོད་པ་ལ་གཉིས་པའི་དོན་ལ་དྲུག་པ་ཡང་ན་འགྱུར་ཏེ། ཤ་ཏཾ་པྲ་དཱི་བྱ་ཏི། བརྒྱ་ཐམས་པས་རབ་ཏུ་རྒྱལ་བར་བྱེད། ཤ་ཏ་སྱ་པྲ་དཱི་བྱ་ཏི། ཀྲྀ་ཏོཉ་རྠ་པྲ་ བརྒྱ་ཐམས་པའི་རྒྱལ་བར་བྱེད། ཡེ་གི་ཀཱ་ལོ་ཉདྷི་ཀར་ཎེ། དུས་ཀྱི་དོན་རབ་ཏུ་
