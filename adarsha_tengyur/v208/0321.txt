བུ་ན་པ་ད་ཀཿ། ཏ་ད་སྨི་ན་སྡཱི་ཏི་ཤེ་ཏ་ནྣཱ་མྣི། དེ་འདི་ན་ཡོད་པ་ཞེས་པ་ཡུལ་ལ་དེའི་མིང་ལའོ། །དེ་འདི་ན་ཡོད་པ་ཞེས་ཡུལ་གྱི་མིང་ཡང་གོ་བར་བྱེད་པའི་དོན་ལ་བདུན་པའི་མཐའ་ལ་ཨཎ་འགྱུར་ཏེ། ཨུ་དུ་མྦ་རཱ་ཎི་སཾ་ཏི་ཨ་སྨིན་དེ་ཤེ། ཞེས་པའི་དོན་ལ།
ཨཽ་དུ་མཱ་རཿ། ཏེ་ན་ནི་རྦྲྀ་ཏྟཾ། ཚིག་རིག་པ། དེ་ཡིས་རྫོགས་པ་ལའོ། །དེ་ཡིས་རྫོགས་པའི་དོན་ལ་གསུམ་པའི་མཐའ་ལ་ཨཎ་ལ་སོགས་པ་འགྱུར་ཏེ། ཀུ་ཤུ་ཤཱ་མྦེ་ན་ནི་ནྲྀ་ཏྟ། ཞེས་པའི་དོན་ལ། ཀཽ་ཤ་མྦྵི ན་ག་ཏི། ཏ་སྱ་ནི་བཱ་སཿ། དེའི་ འདི་ན་ཨུ་དུ་མཱ་ར་རྣམས་ཡོད་པའི་ཡུལ། གནས་ལའོ། །
ཨུ་དུ་བྱ་ར་ལོད་བ། གནས་སུ་སྡོད་པ་ལ་དེའི་མིང་ལ་དྲུག་པའི་མཐའ་ལ་ཨ་ཎ་ལ་སོགས་པ་འགྱུར་ཏེ། ཤི་བཱ་ནཱི་ནི་བཱ་སཿ། ཞེས་པའི་དོན་ལ། ཤཻ་བཿ། ཨ་དཱུ་ར་བྷ་བ་ཤྩ། ངན་སེལ་གྱིས་[གངག་]མི་རིང་བ་གྲོང་ཁྱེར་[མཛོད]་ལྡན། ཀུ་ཤ་མཱ་ལ་ཨཎ་འཕེལ་བ་ཊི་ཏ་ཌྷ་སོགས་ཀྱིས་དཱི་བ། ཉིད་ལས་ཀྱང་དོ། །མི་རིང་བ་ཉིད་བརྗོད་པ་ལ་དྲུག་
པའི་མཐའ་ལ་ཨཎ་ལ་སོགས་པ་འགྱུར་ཏེ། བི་དིང་ཤཱ་ཎཱ་མ་དུ་ར་བྷ་བཾ། ཞེས་པའི་དོན་ལ། བཻ་དི་ཤཾ། ཨཎ་འཕེ་བ། ཞི་བ་རྣམས་ཀྱི་གནས། ཞི་བ་པ། ཨཎ་བསེལ་བ། ཛ་ན་པ་དེ་ལུ་བ། ལྗོངས་ལ་དབྱིའོ། །གནས་ཀྱི་དོན་ཅན་གྱི་རྐྱེན་རྣམས་ལྗོངས་བརྗོད་པའི་དོན་ཡིན་ན་རྣམ་དབྱེ་མང་ཚིག་ལ་འཇིག་སྟེ།
པཾ་ཙ་ལཱཿ། ཨཎ་ཕྱིས་པའོ། །ཡུལ་རྣམས་ཀྱི་མི་རིང་བ་ཉིད། བ་ར་ཎཱ་དི་བྷྱ་ཤྩ། བ་ར་ཎ་ལ་སོགས་པ་རྣམས་ཡུལ་མི་རིང་བ། ལས་ཀྱང་ངོ་། །བ་ར་ཎ་ལ་སོགས་པ་རྣམས་ཀྱི་མི་རིང་བའི་དོན་གྱི་རྐྱེན་རྣམས་ཀྱང་རྣམ་དབྱེ་མང་ཚིག་ཕ་རོལ་ལ་འཇིག་སྟེ། པ་ར་ཎཿ། རིགས་ཉེ་བ་རྣམས། ཤེ་ཥེ། ལྷག་མ་ལའོ། །གོང་དུ་

དེར་སྐྱེས་པའི་དོན་ལ་བདུན་པའི་མཐའ་ལ་ཨཎ་ལ་སོགས་པ་རྣམས་འགྱུར་ཏེ། བཅོམ་བརླག་ཏུ་སྐྱེས་པ། མ་ཐུ་རཱ་ཡཱཾ་ བ་ལྟ་བྱ་མིག །ཛཱ་ཏཿ། ཞེས་པའི་དོན་ལ། བལྟ་བྱ། ཨཎ་སུ་ཨ་མི། མཉན་བྱ། མྱང་བྱ། རེག་བྱ། བཅོམ་བརླག་ཏུ་སྐྱེས། མཱ་ཐུ་རཿ། ཨཎ་འཕེལ་བ། བྲཱ་བྲྀ་ཥ་ཥྛ་ཀ །པྲཱ་པྲྀ་ཥ་
