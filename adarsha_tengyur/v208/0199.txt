གྱི་སློབ་དཔོན་ཁ་ཅིག་འདོད་དོ་ཞེས་པ་བྱུང་བས། ཤྲཱ་བ་ཡ་ཏི་གཱི་ཏཾ་དེ་བ་དཏྟཾ། གྲཱ་ཧ་ཡ་ཏི་བ་ཊུཾ་དེ་དཾ། དརྴ་བ་ཏི་བྷ་ཀྟཾ་དེ་བཾ། ཡང་འགྲེལ་བར། ཨ་ཡ་ཀྲནྡ་ཤབྡཱ་པ་ཧྭེ་ཉ་རྣམས་ཀྱི་སྦྱོར་བ་ལ་གཉིས་པ་འགྱུར་བ་མ་ཡིན་གྱི་གསུམ་པ་ཉིད་དོ་ཞེས་པ་བྱུང་བས། ཨ་
ཡ་ཡ་ཏི་སཻ་ནི་ཀཻ་ར་རཱི་ན། ཀྲནྡ་ཡ་ཏི་སཻ་ནི་ཀཻ་ར་རཱིན། ཤབྷཱ་ཡ་ཡ་ཏི་སཻ ལྷས་བྱིན་སླུ་ཉན་དུ་འཇུག། སློབ་མ་རིག་བྱེད་ལེན་དུ་འཇུག། ལྷ་མཆོད་གཏོང་དུ་འཇུག། ནི་ཀཻ་ར་རཱིན། ཧྭཱ་ཡ་ཡ་ཏི་ཤཱུ་རཻ་ར་རཱི་ན། ཧྲྀ་ཀྲོར་ནྱ་ཏར་སྱཱཾ། ཧྲཾ་ཀྲཾ་དག་གི་ཡང་ནའོ། །ཧྲཾ་ཉ་དང་ཀྲི་ཉ་གཉིས་ཀྱི་སྒྲ་སྦྱོར་བའི་བྱེད་པ་
དམག་འགྲོ་རུ་འཇུག་དགྲ་པོ་རྣམས་པ། དམག་རྣམས་ཀྱིས་དགྲ་བོ་རྣམས་དུ་རུ་འཇུག། པོ་ལ་གཉིས་པ་ཡང་ན་འགྱུར་ཏེ། ཧཱ་ར་ཡ་ཏི་བྷཱ་རཾ་བྷྀ དམག་རྣམས་ཀྱིས་དགྲ་པོ་རྣམས་སྒྲ་སྒྲོག་དུ་འཇུག། དབའ་བོ་རྣམས་ཀྱིས་འགྲོན་དུ་དགྲ་པོ་རྣམས་ལ། ཏྱཾ། ཧཱ་ར་ཡ་ཏི་བྷཱ་རཾ་བྷྲཾ་ཏྱེ་ན། ཀཱར་ཡ་ཏི་
ཀ་ཊཱཾ་དེ་བ་ད་ཏྟཾ། ཀཱ་ར་ལ་ཏི་ཀ་ཊཾ་དེ་བ་ད་ཏྟེ་ན། ཡང་འགྲེལ་པར་ཨ་བྷི་བཱ་དི་དང་དྲྀ་ཤ་སྦྱོར་བ་ལས་བྱེད་པ་པོ་ལ་ཏང་ཨཱན་སྦྱོར་བའི་ཡུལ་གཉིས་པ་ཡང་ན་འགྱུར་རོ་ཞེས་པ་བྱུང་བས། ཨ་བྷི་ཝཱ་ད་ཡ་ཏེ་ཀུ་རཱུ་ན་བ་ཊུ། ཨ་བྷི་ཝཱ་ད་ཡ་ཏེ་གུ་རཱུ་ན་ བྲན་ཕུར་འགྲོག་ཏུ་འཇུག། བ་ཊུ་ནཱ། དརྴ་ཡ་ཏེ་ ལྷས་བྱེད་ར་ལྡེ་བྱེད་དུ་འཇུག།
རཱ་ཛཱ་བྷྲྀ་ཏྱཾ། དརྴ་ཡ་ཏེ་རཱ་ཛཱ་གྷྲྀ་ཏྱེ་ན། ཨ་དྷི་ཤཱིང་སྠཱ་ཨཱ་པ་རྣམས་ཀྱི་འངངོ་། །ཉེར་བསྒྱུར་ཨདྷི་སྔ་མའི་ཤཱི་ད་སྠཱ་དང་ཨཱ་པ་རྣམས་སྦྱོར་བའི་བདུན་པའི་ བླ་མ་སློབ་མ་ལ་གསོང་པོར་སྨྲ་རུ་འཇུག། དོན་ལ་གཉིས་པ་འགྱུར་ཏེ། ཨ་དྷི་ཤེ་ཏེ་བེ་ཀུ་ཎྛཾ་ཧ་རིཿ། ཨ་དྷི་ཏིཥྛ་ཏི་བཻ་
རྒྱལ་པོའི་བྲན་ལ་ལྟ་རུ་འཇུག། ཀུ་ཎྛཾ་ཧ་རིཿ། ཨདྦྷྱཱ་སྟེ་བཻ་ཀུ་ཎྛཾ་ཧ་རིཿ། ཨ་བྷི་ནི་བི་ཤ་ཤྩ། ཨ་བྷི་ནི་བི་ཤའི་འང་ངོ་། །ཉེར་བསྒྱུར་ཨ་བྷིཾ་དང་ནི་སྔ་མའི་བི་ཤ་དང་སྦྱོར་བའི་འང་བདུན་བའི་དོན་ལ་གཉིས་པ་འགྱུར་ཏེ། ཨ་བྷི་བི་ཤ་ཏེ་སམྨཱ་རྒཾ། ནི་བི་ཤ་ཏེ འཕྲོག་བྱེད་མཐར་ས་ལ་ཉལ། སམྨཱ་རྒཾ།
འགྲེལ་བར་ནི་དང་ཨ་བྷི་སྔ་མ་ཡིན འཕྲོག་བྱེད་མཐོ་རིས་ལ་གནས། ཀྱང་ལ་ལར་གཉིས་པ་མི་འགྱུར་ཏེ་བདུན་པ་ཉིད་དོ་ཞེས་པ་བྱུང་བས། པཱ་པེ་ཨ་བྷི་ནི་བི་ཤཿ། སྡིགས་ལ་ངེས་བར་འགྲོ། ཨུ་བཱ་ནྭ་དྷྱཱ་ངྦ་སཿ། ཨུ་པ་ཨ་ནུ་ཨདྷི། ལེགས་བའི་ལ་ནི་ཡ་མངོན་པར་འགྲོ། ལེགས་པའི་ལམ་ལ་ངེས་པར་འགྲོ། ཨཱང་
