ཨཾ། །ཅན་རྣམས་ལ་བ་ཏིའི་མིང་ཅན་ཞེས་བརྡ་ཆད་བྱེད་དེ་གོང་དུ་ཀུ་ག་ཏི་པྲལ་སོགས་པ་རྣམས་ཀྱང་ངོ་ཞེས་པའི་དོན་དུ་མིང་འདོགས་བྱས་ཏེ། དཔེར་བརྗོད་ན། ཨམ་ཨཱུ་རཱི་ཀྲྀ་ཏྱ། སུ་ ཨམ་ཨུར་རཱི་ཀྲྀ་ཏྱ། སུ་ཨམ་ཨལཾ་ཀྲྀ་ཏྱ། སུ་ཡམ་ཨནྟཿཀྲྀཏཾ་སུ། ང་ཐམ་ན་སྐྲྀ་ཏཾ། ངི་ བུ་རྀསྐྲཏཾ།
ཏིརསྐྲྀ་ཏཾ། དེ་རྣམས་ག་ཏིའི་མིང་ཅན་དང་ཅྭིའི་མཐའ་ལ། ངས་ ཤུཀླཱི་ཀྲྀ་ཏྱ། ཌཱཙ་ལ། བདེ་བར་བྱས། སུ་ཁཱཀྲྀ་ཏཾ། འགྲེལ་བར་ལུང་དྲངས་པ། ཨ་ཚ་གཏྱརྠ་བདེ་ཥུ། ཨ་ཚའི་འགྲོ་བའི་དོན་ སེམས་པར་བྱ་བ། བརྗོད་ བརྒྱན་པར་བྱ་པ་རྣམས་ལའོ། བར་མཚམས་བྱས། སེམས་ཀྱི་བྱས། སྔོན་དུ་བྱས།
མི་སྣང་བར་བྱས། ཨ་ཚ་ཞེས་པ་འགྲོ་བ་དང་བརྗོད་པའི་དོན་དང་སྦྱོར་དཀའ་པོར་བྱས། སུ་ཁ་ལཌཱཙ་ ཙུ་ཅུ་སོགས་ཀྱིས་ཅེ་ཁྱིས་ ཌཱ་འགྲོ་བ་ལ་ཊི་ཕྱིས་ དེའི་དོན་གྱིཌ་ཕྱིས། འདིའི་ས་ཕྱིས། བ་ཡང་དོན་གཅིག་པར་འགྱུར་ཏེ། དི་ ཨ་ཚ་ག་ཏྱ། དི་ཨ་ཚ་བ་ད་ཏི། ཡང་འགྲེལ་པར། ཨུར་སི་དང་མན་སི་ཡང་འགྲོ་
བའི་དོན་དང་སྦྱོར་བལ་དོན་གཅིག་པར་འགྱུར་རོ་ཞེས་བཤད་པས། སུ་ ཨུར་སི་ག་ཏྱ། སུ་ མ་ན་སིག་ཏྱ། ཏི་ར་སོ་ཉནྱ་ཏ་ར་སྱཱཾ། ཏི་རས་ཀྱི་ཡང་ནའོ། ཏི་རས་ཞེས་པའི་ས་ཀ་སྔེ་དང་པ་སྡེ་ཕ་རོལ། རིང་པ་ལ་སོང་། ལ་ཡང་ན་རིང་པོར་སྨྲས། རྣམ་པར་བཅད་བྱར་འགྱུར་ཞིང་ཡང་ན་སྲ་ཉིད་འདྲེན་ཏེ།
ཏིརསྐྲཏཱ། ཏིརཿཀྲྀ་ཏྭཱ། ཏིར་སྤ་ཏི། ཏི་རཿཔ་ཏི། འགྲེལ་པར། ཧ་སྟེ་ལྟོ་འགྲོབ་ཀྲདྱ། བེདཾ་ཛཱི་ ཡིད་འབྲོག་ བི་ཀཱ། པཱ་ཎོ་ཀྲྀ་ཏྱ། རྣམ་དབྱེ་བདུན་པའི་རྗེས་མི་དབྱི་ཡང་ངེས་པའི་ཚིག་སྡུད་ཟེར་རོ་ཞེས་བྱུང་ངོ་། བྲལ་སོགས་པའི་དཔེར་བརྗོད་ནི། སུ་པྲཀྲྀ་ཏྱ། སཾབྷ་བ་ཏི། སུ་ཧྲྀཏཾ། ཨ་བྷི་ཥེ་
མི་སྣང་བར་བྱས། ཀ། པྲཱཅརྻཿ། མེ་སྣང་བའི་བདག་ལོ། པྲཏི་པཀྵཾ། པྲ་ལག་པར་བྱས་ནས། རིག་བྱེད་འཚོ་བ། ལག་པར་བྱས་ནས། ཏི་ནཱ་ཐཿ། ངུཿན་རཿ། དུཿཀྲྀཏཾ། ཡང་འགྲེལ་པར་ལུང་དྲངས་པ། སྭ་ཏཱི་པུ་ཛཱཡཾ། སུ་ཨ་ཏི་དག་མཆོད་པ་ རང་བཞིན། འཛའ་པོའམ་ཡིད་བཙུགས་། དབང་བསླུར་བ།
ལའོ། སུ་དང་ཨ་རབ་ཏུ་སློབ་དཔོན། ཏི་དག་མཆོད་ ཕྱོགས་སོ་སོར། པའི་དོན་བརྗོད་པ་ཡང་དོན་གཅིག་པར་འགྱུར་ སོ་སོར་མགོན་པོ། མི་ངན། ཉེས་བྱས། ཏེ། སུཥྚེ་ཎཤྲཱི། ཞེས་པ་ལ། རྗེས་འབྲང་དང་རྣམ་དབྱེ་ཕྱིས། སུ་ཤྲཱི། གཟུགས་མཛེས་སམ་དཔལ་ལམ་ཕུན་སུམ་ཚོགས་པ། དེ་བཞིན་དུ་
