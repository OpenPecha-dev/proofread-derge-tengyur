བྱ་ཧྣེ། བྱ་ཧ་ནི། བྱ་ཧྣི། སཱ་ཡ་ཐོག་མར་བཞག་པ་ལ། སྭ་ཡཱ་ཧྣེ། སཱ་ཡཱ་ཧ་ནེ། སཱ་ཡཱ་ཧྣི། ཞེས་པ་ལྟ་བུའི་གཟུགས་གསུམ་གསུམ་བྱུང་བའི་རྗེས་མ་རྣམས་ནའི་ཨ་ཏ་དབྱི་འོ་ཞེས་པས་བསྒྲུབས་པའོ། །ཞེས་བ་དེ་རྣམས་ནི་སྐྱེས་པའི་རྟགས་ལས་ཨ་ཐུང་མཐའ་ཅན་རྣམས་སོ།། ༈ །།
སྐྱེས་པའི་རྟགས། ཉི་བྲལ། ཨཱ་རིང་མཐའ་ཅན་ལ། བི་ཤྭ་པཱ། ཞེས་པའི་རྟགས་ལ་དཔེར་བརྗོད་ན། བི་ཤྭ་པཱ་ཞེས་པ་ལ། རྣམ་དབྱེ་དང་པོའི་གཅིག་ཚིག། སུ། དེ་དབྱི་འོས་ཨུ་ཕྱིས། ས་ས་ཇུ་ཥ་སོགས་དང་། །ཁར་སོགས་ཀྱི་མདོས་བསྒྲུབས་པས། བི་ཤྭ་པྭ༔། ཨཽ་དང་ཛས་ལ། དཱི་རྒྷྭ་ཛྫ་སི་
ཙི། རིང་པོ་ལས་ཛས་ཨི་ཙ་ལའོ། །ཨ་ཡིག སྣ་ཚོགས་འཐུང་། རིང་པོའི་མཐའ་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་ཨི་ཙ་སོ་སོར་བསྡུ་བ་དང་པོ་ཅན་རྣམས་དང་ཛས་ལ། དང་པོ་དག་ལ་སྔ་མ་འདྲའོ། ཞེས་པའི་མདོའི་ཆོ་ག་འཇུག་པ་མ་ཡིན་ཏེ། ཨེ་ཙའི་འཕེལ་བ་དང་། ཨ་ཀ་འདྲ་བ་ལ་རིང་བོ་བྱས་པས།
བི་ཤྭ་པཽ། བི་ཤྭ་པཱ༔། གཉིས་པའི་གཅིག་ཚིག་ལ། ཨ་མ་ལ་སྔ་མ་འོས། བི་ཤྭཔཱཾ། གཉིས་ཚིག་ལ། བིཤྭ་པཽ། མང་ཚིག་ཤས་ལ། ཨཱ་ཏོ་དྷཱ་ཏོ༔། བྱིངས་ཀྱི་ཨཱ་ཏ་ཀྱི་འོ། །རྣམ་དབྱེ་གཉིས་པའི་ཤས་ནས་བརྩམས་ཏེ། བདུན་པའི་སུ་པ་ཀྱི་བར་གྱི་རྣམ་དབྱེ་ཨ་ཙ་ཐོག་མ་ཅན་གང་བྱུང་ཡང་།
རྟགས་བྱིངས་ཀྱི་ཨཱ་རིང་མཐའ་ཅན་ཡིན་པའི་ཨཱ་དེ་དབྱི་སྟེ། ཨཱ་ཕྱིས་པས། བི་ཤྭ་པ༔། རྣམ་དབྱེ་ལྷག་མའང་བཤད་མ་ཐག་བའི་ཆོ་ག་ལྟར་གྱིས་བསྒྲུབས་པ། གསུམ་པ་ལ། བི་ཤྭ་པཱ། བི་ཤྭ་པཱ་བྷྱཱཾ། བི་ཤྭ་པཱ་པྷི༔། བཞི་པ་ལ། བི་ཤྭ་པེ། བི་ཤྭ་པཱ་བྷྱཱཾ། བི་ཤྭ་པཱ་བྷྱ༔། ལྔ་བ་ལ། བི་
ཤྭ་པ༔། བི་ཤྭ་པཱ་བྷྱཱཾ། བི་ཤྭ་པཱ་བྷྱ༔། དྲུག་པ་ལ། བི་ཤྭ་པ༔། བི་ཤྭ་པཱཾ། བདུན་པ་ལ། བི་ཤྭ་པོ༔། བི་ཤྭ་པཱ་སུ། བོད་ཚིག་ལ། ཧེའི་ཤྭ་པཱ༔། ཧེབི་ཤྭ་པཱ༔། དེ་བཞིན་དུ། སོམ་པཱ། ཀཱི་ལྦ་ལ་པཱ། ཤཾ་ཁ་དྒཱ། ཧཱ་ཧཱ། སྟྭཱ། སྣཱ་ལ་སོགས་པ་མང་པོ་
དག་ཀྱང་བི་ཤྭ་པཱ་དང་བྱ་དང་བྱ་བ་འདྲ་བའོ། །ཞེས་པ་སྐྱེས་བའི་རྟགས་ཀྱི་ཨཱ་རིང་མཐའ་ཅན་གྱི་བྱ་བ་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་སྐྱེས་པའི་རྟགས་ཨི་ཐུང་མཐའ་ཅན་གྱི་བྱ་བ་རྣམས་བཤད་པར་བྱ་སྟེ། ཧ་རི་ལ། འཕྲོག་བྱེད། ཞི་བ་འཐུང་། ངུར་བཟོ་བ། དྲི་ཟ། རྣམ་དབྱེ་དང་པོའི་གཅིག་ཚིག་
