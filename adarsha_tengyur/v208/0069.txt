རྀ་ཎཱ་རྞ༔། ཨུ་པས་རྒཱ་དྲྀ་ཏི་དྷཱ་ཏཽ། ཉེར་བསྒྱུར་ལས་བྱིངས་རྀ་ཏ་ལའོ། །ཉེར་བསྒྱུར་ཨ་ཡིག་མཐའ་ཅན་ལས་བྱིངས་ཀྱི་ཐོག་མའི་རི་ཏ་ཕ་རོལ་ལ་གཉིས་དག་གིས་གཅིག་འཕེལ་བར་འགྱུར་ཏེ། ཨུ་པ་རི་ཙྪ་ཏི་ཞེས་པ་ལ། ཨུ་པཱ་རྩྪ་ཏི། དེ་བཞིན་དུ། བྲཱ་རྩྪ་ཏི་སོགས་སོ། །
རྀ་ཏ་ཅེས་ཏ་སྒྱུར་བ་ བུ་ལོན་གྱི་བུ་ལོན། ནི་རཱྀ་འགག་པའི་དོན་ན། །བཱ་སུ་པྨ་པི་ཤ་ལ༔། ཨཱ་བི་ཤ་ལིའི་སུབ་ལ་ཡང་ནའོ། །བྲ་བ་ཐོགས་པ་ཨ་ཡིག་མཐའ་ཅན་ལ་སུ་པ་ཀྱི་བྱིངས་ཀྱི་ཐོག་མའི་རྀ་ཏ་འང་ཡང་ན་འཕེལ་བར་འགྱུར་བ་སྔོན་གྱི་དྲང་སྲོང་ཨཱ ཉེ་བར་འགྲོ། བི་ཤ་ལི་འདོད། རབ་ཏུ་འགྲོ་བ་དེ། ཨུ་པ་
རྀ་ཥ་བྷཱི་ཡ་ཏི་ཞེས་པ་ལ། ཨུ་བཱ་རཱ་བྷཱི་ཡ་ཏི། འགྲེལ་བར་སྐབས་འགར་རྀ་ཏ་ལ་བཤད་པའི་ཆོ་ག་རྣམས་ལ་ཏ་ལའང་འཇུག་གོ་ཞེས་བྱུང་བས། པྲ་ལྀ་ཀཱ་རཱི་ཡ་ཏི་ཞེས་པ་ལ། པྲཱལྐཱ་རཱི་ཡ་ཏི། པྲ་ལྐཱ་རཱི་ཡ་ཏི། ཨེ་དི་བ་ར་རཱུ་པཾ། ཨེང་ལ་ཕ་རོལ་གྱི་གཟུགས་སོ། །ཉེར་བསྒྱུར་ཨ་
ཡིག་མཐའ་ཅན་རྣམས་ཀྱི་ཨེང་སོ། དྲང་སྲོང་བཞིན་ཉེ་བར་སྦྱོད། སོར་བསྡུ་བའི་ཡི་གེ་དང་པོའི་བྱིངས་ཕ་རོལ་ལ་གཉིས་དག་གིས་གཅིག་ཕ་རོལ་གྱི་གཟུགས་སུ་འགྱུར་ཏེ། པྲ ཨེ་ཛ་ཏི་ཞེས་པ་ལ། པྲེ་ཛ་ཏི། རབ་ཏུ་ལྀ་འདོད་པར་བྱེད་དུ་འཇུག ཨུ་པ། ཨོ་ཁ་ཏེ་ཞེས་པ་ལ། ཨཱུ་པོ་ཁ་ཏེ། ཨེ་དཱ་དཽ་སུ་བྷ་ཏཽ་བཱ།
ཨེང་དང་པོའི་སུབ་ཀྱི་བྱིངས་ལ་ཡང་ནའོ། །ཉེར་བསྒྱུར་ཨ་ཡིག་མཐའ་ཅན་རྣམས་ཀྱི་ཨེང་དང་པོའི་སུབ་ཀྱི་བྱིངས་ཕ་རོལ་ལ་ཕ་རོལ་གྱི་གཟུགས་སུ་བསྒྱུར་བ་དང་འཕེལ་བ། རབ་ཏུ་གཡོ་བར་བྱེད། ཡང་ན་བར་བྱ་སྟེ། ཨུ་པ་ཨེཌ་ཀཱི་ཡ་ཏི་ཞེས་པ་ལ། ཨུ་པི་ཌ་ཀཱི ཉེ་བར་སྲེག་པར་བྱེད། ཡ་ཏི། ཨུ་པེ་
ཌཀཱི་ཡ་ཏི། པྲ ཨོགྷཱི་ཡ་ཏི་ཞེས་པལ། བྲཽ་གྷཱ་ཡ་ཏི། པྲོ་གྷཱི་ཡ་ཏི། འགྲལ་པར་ལུང་དྲངས་པ། ཤ་ཀཾ་ངྡྷ་དི་ཥུ་པར་རཱུ་པཾ་པཱ་ཙྱཾ། ཤ་ཀཾ་ངྔྷུ་ལ་སོགས་པ་རྣམས་ཕ་རོལ་གྱི་གཟུགས་སུ་བརྗོད་དོ། །ཤ་ཀཾ་ངྷུ་ལ་སོགས་པ་ཕ་རོལ་གྱི་གཟུགས་ལེགས། ཉེ་བར་ལུས་བཞིན་སྦྱོད་པར་བྱེད།
པ་རྣམས་སུ་འགྱུར་ཏེ། ཤ་ཀ ཨནྡྷ་ཞེས་པ་ལ། རབ་ཏུ་ཆུ་རླབས་བཞིན་སྦྱོད། ཤ་ཀ་ནཱྷ༔། ཁྲོན་པ་ཆུ་སྐམས་པ། ཀ་རྐ། ཨ་ནྡཱུ་ཞེས་པ་ལ། ཆེ་བ་ཁ། ཀུ་ལ། ཨ་ཊཱ་ཞེས་པ་ལ། ཀུ་ལ་ཊཱ། དགེ་སློང་མ། རི་ལ་སློང་མ། སཱི་མ་ན་ ཨ་ནྟ་ཞེས་པ་ལ། སཱི་མནྟ༔། སྐྲ་མཚམས། ཧ་ལ། ཨཱི་ཥཱ་ཞེས་པ་ལ། ཧ་
