རྣམས་ལ། པྲཱ་ཙཿ། པྲཱཙོཿ། བྲཱ་ཙཱཾ། ངིཨོས་ལ། པྲཱ་ཙི། པྲཱ་ཙོཿ། སུ་པ་ལ། ཀྭིན་གྱི་ཀུ་འོས། ཙ྄ཀ྄། ཀུན་ནས་བསྒྱུར་བ་དང་རྐྱེན་དག་ལ་འོས། སཥ། ཀ་ཥ་འདུས་པ་ཀྵ། པྲཱ་ཀྵ། བོད་ཚིག་ལ། ཧེ་པྲཱ་ང་། ཧེ་པྲཱ་ཉྩཿ། ༈ བྲ་ཏྱ་ཙ་སོགས་ཀྱང་འདི་དང་འདྲའོ། །ཡང་ཨཉྩ་ལ་
བརྟེན་པའི་མིང་སྔོན་མ་ལ་རྣམ་འགྱུར་འཇུག་ཚུལ་ནི། བིཥྤ་གྡེ་བ་ཡོ་ཤྩ་ཊེ་ར་དྲྱཾ་ཙ་ཏཽ་རྐྱེན། བ་བྲ་ཏྱ་ཡེ། བི་ཥྭ་ཀ་དེ་ཝ་དག་གིའང་ཊིའི་ཨ་དྲི་ཨཉྩ་བ་འགྲོ་པའི་རྐྱེན་ཅན་ལའོ། །ཨཉྩ་བའི་རྐྱེན་ཅན་གྱི་སྔོན་དུ་བིཥྭ་ཀ་ཅེས་པའི་རྟགས་དང་དེ་ཝ་དང་ས་རྦའི་མིང་ཅན་རྣམས་བྱུང་ན་རྟགས་དེ་རྣམས་ཀྱི་ཏྲི་ནི་ཨ་དྲི་
ཞེས་པར་འགྱུར་ཏེ། ཨ་དང་ཨི། ཨ་ཙ་ཞེས་གནས་པ་ལ། ཨ་ད་སོ་ས་རྡཱ་དུ་དོམཿ། ཨ་དས་ཀྱི་སའི་ར་ལ་ཨུ་སྟེ་དའི་འང་མའོ། །ཨ་དས་ཞེས་པའི་ས་ནི་ད་རུང་གྱུར་ལ། ད་ལ་ཨུ་སྟེ་ད་དེ་ཡང་མ་རུ་འགྱུར་ཞིང་། དེ་ཡང་སྔོན་གྱི་སློབ་དཔོན་རྣམས་ལ་ལས་ད་གཉིས་ཀ་མ་དང་ཨུ་རུ་འགྱུར་བར་
འདོད། ལ་ལས་ཕྱི་མ་ཁོ་ན་ཨིང་དཻ་མི་ཟད་པའོ། མུ་ཞེས་པར་འགྱུར་བར་འདོད་དོ། །ཨ་མུ་མུ་ཡ་ཙ། སུ། ཨུཀ་འགྲོ་བ་ཨ་ཙ་རྣམས་ཀྱི་ཞེས་པས། ནུམ། ཧ་ལ་སོགས་ཀྱིས་སུ་ཕྱིས། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱིའོས། ཙ་ཕྱིས། ཀྭིན་གྱི་ཀུ་འོས་ནང་། ཨ་མུ་མུ་ཡང་། ཨཽ་དང་ཛས་ལ། ཨ་
མུ་མུ་ཡཱཉྩཽ། ཨ་མུ་མུ་ཏ་ཉྩཿ། ཨམ་དང་ཨཽ་ལ། ཨ་མུ་མུ་ཡཉྩཾ། ཨ་མུ་མུ་ཡཉྩཽ། ཤས་ལ། ཨ་ཙའི་འོས་ཨ་ཕྱིས། ཨུ་མུ་མུ་ཨི་ཙ་ཤས། ཙ྄་ལ་འོས། ཨི་རིང་པོ། ཨ་མུ་མུ་ཨཱི་ཙོཿ། ཌཱ་ལ། ཨ་མུ་མུ་ཨཱི་ཙཱཾ། བྷྱཱམ་བྷིས་ལ། ཀྭན་གྱི་ཀུ། རྐྱེན། ཛྷ་ཤ་ལ་ཛྷ་ལ་རྣམས་ཀྱི་ཛད་ཤ་རྣམས་སོས། ཀ྄་ག྄།
ཨ་མུ་མུ་ཡ་ཀྵཱཾ། ཨ་མུ་མུ་ཡ་གྷིཿ། ངེ་ལ་ཨ་མུ་མུ་ཨཱི་ཙེ། བྷྱཱམ་བྷྱས་འདྲ། ང་སི་ལ། ཨ་མུ་མུ་ཨཱི་ཙཿ། བྷྱམ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། ཨ་མུ་མུ་ཨཱི་ཙཿ། ཨ་མུ་མུ་ཨཱི་ཙཽཿ། ཨ་མུ་མུ་ཨཱི་ཙཱཾ། ངི་ཨཽས་ལ། ཨ་མུ་མུ་ཨཱི་ཙི། རྐྱེན། ཨ་མུ་མུ་ཨཱི་ཙོཿ། ཀུ་བ་ལ། ཀྭིན་གྱི་ཀུ། ཀུན་ནས་
བསྒྱུར་པ་སོགས་ཀྱིས་སཥ། ཀ་ཥ་འདུས་པཀྵ། ཨ་མུ་མུ་ཡཀྵུ། བོད་ཚིག་རྣམས་ཀྱང་དང་པོ་དང་འདྲའོ། །ཡང་ན་ད་ཕྱི་མ་ཁོ་ན་མུ་རུ་འགྱུར། ཨ་ད་མུ་ཡ་ཙ། ཞེས་པ་ལ། ཆོ་ག་ཐམས་ཅད་བཤད་མ་ཐག་པ་དང་འདྲ། ཡང་ན། བ྄ི་ཥྭ་ཀྭ་དེ་ཝ་དག་གིའང་ཊིའི་ཨ་ རྐྱན་། དྲི་སོགས་ཀྱི་མདོས་
