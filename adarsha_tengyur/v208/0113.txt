པུདྡྷི་རྣམས་ཀྱང་འདི་དང་འདྲའོ། །ཏྲི་ལ་ཛས། ཏྲི་ཚ་ཏུ་རོཿསྟྲི་ཡཾ་ཏི་སྲྀ་ཙ་ཏ་སྲྀ། ཏྲི་ཙ་ཏུར་དག་གི་བུད་མེད་ལ་ཏི་སྲྀ་ཙ་ཏ་སྲྀ་དག་གོ། །གསུམ་གྱི་སྐད་དོད་ཏྲི་དང་། བཞིའི་སྐད་དོད་ཙ་ཏུར་གཉིས་ནི་མོ་རྟགས་ཀྱི་སྐབས་རྣམ་དབྱེ་ཕ་རོལ་ལ་ཏྲི་ནི། ཏི་སྲྀ་དང་། ཙ་ཏུར་ནི། ཙ་ཏ་སྲྀ་ཞེས་
པར་འགྱུར་རོ། བློ། ཏི་སྲྀ། ཛས། ཙ་ཏ་སྲྀ། ཛས། ཨ་ཙིར་རྀ་ཏཿ། རྀ་ཏ་ཀྱི་ཨ་ཙ་ལ་རའོ། །ཏི་སྲྀ་ཙ་ཏ་སྲྀ་ཞེས་རྀ་ཏ་མཐའ་འགྱུར་པའི་རྀ་ཏ་དེ་རྣམ་དབྱེ་ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ལར་སྲོག་མེད་དུ་འགྱུར་རོ། །ཏི་སྲཿ། ཙ་ཏ་སྲཿ། ཤ་ས་ལ། ཏི་སྲཿ། ཙ་ཏ་སྲཿ། ཏི་སྲྀ་བྷྀཿ། ཙ་ཏ་སྲྀ་བྷིཿ། བཞི་པ་ལྔ་པ་
གཉིས་ལ། ཏི་སྲི་བྷྱཿ། ཙ་ཏ་སྲྀ་བྷྱཿ། དྲུག་པའི་ཨཱ་མ་ལ། ར་ཨ་ཙ་ཞེས་པའི་མདོས། ནུཊ། ན་ཏི་སྲྀ་ཙ་ཏ་སྲྀ། ཏི་སྲྀ་ཙ་ཏ་སྲྀ་དག་ལམ་ཡིན་ནོ། །སྤྱིར་བཏང་ནཱཾ་ལ་རྟགས་མཐའ་རིང་པོ་ཐོབ་ཀྱང་དམིགས་བསལ་ཏི་སྲྀ་དང་། ཙ་ཏ་སྲྀ་དག་ནཱཾ་ཕ་རོལ་ལ་རིང་པོར་འགྱུར་པ་མ་ཡིན་ཏེ། ཏི་སྲྀ་ཎཱཾ། ཙ་ཏ་
སྲྀ་ཎཱཾ། བདུན་པ་ལ། ཏི་སྲྀ་ཥུ། ཙ་ཏ་སྲྀ་ཥུ། བོད་ཚིག་ལ། ཧེ་ཏི་སྲཿ། ཧེ་ཙ་ཏ་སྲཿ། པྲི་ཡ་ཏྲི་ཞེས་པ་ལྟ་བུ་མིང་ཐོག་མ་ཅན་ལ་ཏྲི། ཏྲི་སྲྀ་ཞེས་པར་མི་འགྱུར་ལ། རྣམ་དབྱེ་ཡང་མང་ཚིག་ཏུ་མ་ཟད་ཐམས་ཅད་འཇུག་སྟེ། ཆོ་ག་ཡང་སྐྱེས་པའི་རྟགས་ཧ་རིའི་སྐབས་སུ་བཤད་པ་དང་
འདྲ་ལ། ཊཱ་དང་ཤས་ནང་མི་འགྱུར་རོ། །དེ་བཞིན་དུ། པྲི་ཡ་ཙ་ཏུར་ཞེསཔ་ཡང་། དགའ་བ་གསུམ། ཙ་ཏ་སྲིར་མི་འགྱུར་ཏེ། དཔེར་བརྗོད་རྒྱས་པ་འོག་ཏུ་འབྱུང་ངོ་། ཡང་སྐབས་འགར་མིང་དང་པོ་ཅན་ལ་ཡང་། ཏི་སྲྀ། ཙ་ཏ་སྲྀ། ཞེས་པར་འགྱུར་པའང་སྲིད་དོ། །ཞེས་འགྲེལ་པར་
བཤད་དོ། །དྭིའི་སྒྲ་ལ། ཏད་སོགས་ཀྱི་རྣམ་དབྱེ་ལ་ཨ། དགའ་བ་བཞི། ཨ་ལ་བུད་མེད་དུ་བརྗོད་པའི་ཊཱ་པ་སྦྱར་བས་དྭཱ་ཞེས་གྲུབ། དེ་ལ་རྣམ་དབྱེ་རྣམས་ཀྱི་གཉིས་ཚིག་རྣམས་བྱིན། རམཱ་ལྟར་བསྒྲུབས་པས། ཨཽ་གཉིས་ལ། དྭེ། བྷྱཱཾ་གསུམ་ལ། དྭཱ་བྷྱཱཾ། ཨོ་ས་གཉིས་ལ། དྭ་ཡོཿ། དེ་ཡན་
གྱིས་བུད་མེད་ཀྱི་རྟགས་ཨི་ཐུང་མཐའ་ཅན་རྣམས་ཀྱི་བྱ་བ་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་མོ་རྟགས་ཨཱི་རིང་གི་ཆོ་ག་རྣམས་སྟོན་པར་བྱ་སྟེ། གཽ་རཱིལ། སུ། ཧལ་བཱི་ཞེས་པར་སུ་ཕྱིས། གཽ་རཱི། ཨཽ་དང་། ཛ་ས་ལ། གཽ་རྻྻཽ། གཽརྻྻཿ། ཡཱུ་སྦྷྱཱ་ཁྱཽ་ན་དཱཾ། ཨཱི་ཨཱུ་བུད་མེད་བརྗོད་པ་དག་ན་དཱིའོ། ཨཱི་རིང་དང་
