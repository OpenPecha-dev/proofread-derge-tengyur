རྣམས་ལས་སུ་ཏི་སི་ཨ་པྲྀ་ཀྟའི་ཧ་ལའོ། །རྟགས་མཐའ་ལ་ཧ་ལ་ཏེ་གསལ་བྱེད་སྲོག་མེད་དང་། །བུད་མེད་དུ་བསྒྲུབ་པའི་རྐྱེན་དཱི་པ་དང་ཨཱ་པ་ཀྱེ་མཐའ་དང་རིང་པོ་རྣམས་ལས། སུ་དང་། ཏི་སོགས་ཀྱི་ཏི་དང་། སི་རྣམས་ཀྱི་ཨུ་དང་ཨི་ཕྱིས་བས། །ཧ་ལ་ཨ་ཙ་མེད་པའི་ཡི་གེ་གཅིག་པ། །
ཨ་པྲྀ་ཀྟའི་མིང་ཅན་དེ་དབྱི་བར་བྱ་སྟེ། སུ་ཕྱིས། ས་ཁཱ་ན། མིང་མཐའི་ནང་བྱིའོ་ཞེས་པ་མ། ན་ཕྱིས། ས་ཁཱ། ས་ཁྱུར་སཾ་བུ་དྡྷོ། ས་ཁི་ལས་སཾ་བུ་དྡྷི་མ་ཡིན་པ་ལའོ། །ས་ཁིའི་སྐབས་འདིར་ཀུན་དུ་བོད་པའི་དོན་གྱི་གཅིག་ཚིག་སཾ་བུ་དྡྷི་མ་ཡིན་པའི་དང་པོའི་སུ་ནས་གཉིས་པའི་ཨཽ་ཊའི་
བར་གྱི་རྣམ་དབྱེ་ལྔ་པོ་སྔར་ཐམས་ཅད་གནས་པའི་མིང་ཅན་ཞེས་བརྡ་ཆད་ཆད་བྱས་པ་དེ་རྣམས་ལ་སྐབས་འདིར་ཎི་ཏ་སྟེ། ཎ་འགྲོ་བའི་མིང་ཅན་ཞེས་དགོས་པའི་ཆེད་དུ་མིང་འདོགས་བྱེད་དོ། །ས་ཁི་ལ། དང་པོའི་ཨཽ་དང་ཛས། གཉིས་པའི་ཨམ་དང་ཨཽ་རྣམས་སྦྱིན། ཨ་ཙ་ཉྞི་ཏི། ཉྲ་ཎྲ་འགྲོ་བ་ལ་
ཨ་ཅའིའོ། །ཕ་རོལ་དུ་ཉ་ཎ་འགྲོ་བའི་མིང་ཅན་བྱུང་ན་རྟགས་མཐའི་ཨ་ཅ་སྟེ་དབྱངས་དེ་འཕེལ་བར་འགྱུར་བས་སྐབས་འདིར་ཡང་ཨཽ་སོགས་བཞི་ལ་ཎ་ཡིག་འགྲོ་བའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱས་པའི་དགོས་པ་ནི་འདི་ཡིན་པས་རྟགས་མཐའ་འཕེལ་བ། ཨེ་ཙའི་ཨ་ཡ་ཨ་ཝ་སོགས་ཀྱི་མདོས།
ས་ཁཱ་ཡཽ། ས་ཁཱ་ཡ༔། ས་ཁཱ་ཡཾ། ས་ཁཱ་ཡཽ། ས་ཁི་ལ། ཀུན་དུ་བོད་པའི་སུ་སྦྱིན་སཾ་བུ་དྡྷི་མ་ཡིན་པ་ཞེས་པས་སྐབས་འདིར་བཤད་མ་ཐག་པའི་བྱ་མི་འཇུག །གོང་ཧ་རིའི་སྐབས་དང་འདྲ་བས། ཧེ་ས་ཁེ། གཉིས་ཚིག་དང་མིང་ཚིག་བཤད་མ་ཐག་བའི་རྣམ་དབྱེ་དང་པོའི་གཉིས་ཚིག་དང་
མང་ཚིག་ལྟར་བསྒྲུབས་པས། ཧེ་ས་ཁཱ་ཡཽ། ཧེ་ས་ཁཱ་ཡ༔གཉིས་པའི་མང་ཚིག་ཤས་ཀྱི་སྐབས་ཧ་རི་དང་འདྲ་བས། ས་ཁཱིན། གསུམ་པ་རྣམས་ལ། ས་ཁྱཱ། ས་ཁི་བྷི༔། བཞི་པ་ལ། ས་ཁྱེ། ས་ཁི་བྷྱཱཾ། ས་ཁི་བྷྱ༔། ས་ཁི་ལ། ལྔ་བའི། ང་སི་སྦྱིན། ཁྱ་ཏྟཱ་ཏྤར་སྱ། །བེ་ཏི་
དག་ལས་ཕ་རོལ་གྱི་ཨའིའོ། །རྟགས་མཐའ་ཁི་རིང་པོ་དང་ཐུང་ངུ་ཏི་རིང་པོ་དང་ཐུང་ངུ་རྣམས་ཀྱི་ཕ་རོལ་རྣམ་དབྱེ་ལྔ་པ་དང་དྲུག་པའི་ང་སི་དང་ངས་གཉིས་ཀྱི་ད་ཕྱིས་པའི་ཨ་དེ་ནི་ཨུ་རུ་འགྱུར་ཏེ། ས་ཁྱུ༔། ས་ཁི་བྷྱཾ། སི་ཁི་བྷྱ༔། ས་ཁི་ལ་རྣམ་དབྱི་དྲུག་པ་རྣམས། ས་ཁྱུ༔། ས་ཁྱོ༔། ས་ཁཱི་ནཱ།
