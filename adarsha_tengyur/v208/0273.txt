བརྗོད་ནི། བཧུ་ངྷ་ནཾ་ཡ་སྨིན། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་བསྡུས། བ་ཧུདྷནཿ། ཨསྟི་ངྷ་ནཾཡ་སྱ། ཞེས་པ་ལ། ཨསྟི་ངྷ་ནཿ། དེའི་ཡོན་ཏན་ཅན་གྱི་དཔེར་བརྗོད་ནི། ལམྦོོ་ཀཎོོ་ཡ་སྱ། ཞེས་པ་ལ། ལམྦ་ཀརྞཿ། དེ་བཞིན་དུ། མ་ཧཱ་པྲཛྙ་ཡ་སྱ་
ཞེས་པ་ལ། མ་ཧཱ་བྲཛྙཿ། ཨི་ཏི་བ་ཧུ་ གང་ལ་ནོར་མང་བོ། བྲཱི་ཧི་ས་མཱ་ས་སམཱཔྟཱཿ། དེ་རྣམས་ ནོར་མང་པོ། ནོར་ཡོད་པ་གང་གི། ནོར་ཡོད་པ། ཀྱིས་འབྲུ་མང་པོའི་བསྡུ་བ་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་ཟླས་ གང་གི་རྣ་བ་གཉིས་པོ་རིང་པོ་དག། རྣ་བ་རིང་པོ། དབྱེའི་བསྡུ་བ་ ཤེས་རབ་ཆེན་པོ་གང་གི།
བཤད་པར་བྱ་སྟེ། ཤེས་རབ་ཆེན་པོ། ཙཱརྗེ་དྭ་དྭཿ། དང་གི་དོན་ལ་ཟླས་དབྱེའོ། །སུང་མཐའ་མང་པོ་དང་གི་དོན་གྱི་ཚིག་བསྡུ་བ་ནི་ཟླས་དབྱེའི་བསྡུ་བ་ཞེས་བྱའོ། །དེ་ལ་དབྱེ་བ་བཞི་སྟེ། འདུ་བ་དང་དྲལ་བ་དེའི་དོན་དང་། འབྱོར་པ་དང་བྲལ་བདེའི་དོན་དང་། གཅིག་གཅིག་སྦྱོར་བ་དེའི་དོན་
དང་། ཀུན་སྡུད་བདེའི་དོན་དང་བཞིའོ། དེ་ལྟར་བཞི་ལས། གཅིག་ལ་གཅིག་སྦྱོར་བ་དང་། ཀུན་དུ་སྡུད་པ་ནི། བྱ་བྱེད་གང་ཡང་ནུས་པའི་ཕྱིར་ཟླས་དབྱེའི་བསྡུ་བ་འཇུག་ལ། འདུ་བ་དང་བྲལ་བ་དེའི་དོན་དང་། འབྱོར་པ་དང་བྲལ་བདེའི་དོན་གཉིས་དང་འབྲེལ་བའི་དོན་ལ་ནི་བྱ་བྱེད་
སྟོངས་པའི་ཕྱིར་ཟླས་དབྱེའི་བསྡུ་བ་མི་འཇུག་གོ། དྭཾ་དྭེ་གྷི། ཟླས་དབྱེ་ལ་གྷིའོ། །ཟླས་དབྱེའི་བསྡུ་བའི་སྐབས་འདིར་གོང་དུས་ཁི་མ་ཡིན་པ་ལ་ལྷག་མ་གྷིའོ་ཞེས་རྟགས་གང་ལ་གྷིའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱས་པ་དེ་རྣམས་ནི་ཐོག་མར་འཇོག་སྟེ། ཧ་རི་ཤྩ་ཧ་རཤྩ། ཞེས་པ་ལྟ་བུ་གྷིའི་
མིང་སྔོན་དུ་འཇོག་ལ། དེ་ནས་རྣམ་དབྱེ་པྲད་སོགས་བསྡུས །ཨོོ། ཧ་རི་ཧ་རོོ། དེ་བཞིན་དུ། བཊུ་ཤྩ་གུ་པྟཤྩ། ཞེས་པ་ལ། པ་ཊུ་གུ་པྟོོ། འགྲེལ་པར། རྟགས་གྷིའི་མིང་ཅན་གཅིག་དང་བྷིའི་མིང་ཅན་མ་ཡིན་པ་གཅིག་གི་བར་གྱི་རྣམ་དབྱེ་བསྡུ་ འཕྲོག་བྱེ་དང་འཕྲོག་པོ་ཡང་། བ་ཡིན་
ན་གྷིའི་མིང་ཅན་དེ་སྔོན་མར་འཇོག་ལ་རྟགས་མང་པོ་བསྒྲིགས་པ་འཕྲོག་བྱེད་འཕྲོག་པོ་དག། མཁས་བ་དང་སྤས་པ་ལང་། མཁས་པ་སྤས་བདག། ལ་ནི་གཅིག་ཏུ་མ་ངེས་སོ་ཞེས་བཤད་པས། དཔེར་ན། ཧ་རི་ཤྩ་ཧ་ར་ཤྩགུ་རུ་ཤྩ། འཕྲོག་བྱེད་དང་འཕྲོག་པོ་དང་བླ་མ་དང་ཞེས་པ་ལ། ཛས་ ཧ་རི་ཧ་ར་གུར་འཕྲོག་བྱེད་འཕྲོག་པོ་བླ་མ་རྣམས།
