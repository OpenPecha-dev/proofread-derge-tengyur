རྣམས་དང་ཀྲོཥྚུ ཏྲྀ་ཙ་བཞིན་དུ་འགྱུར་བའི་ཕ་རོལ་རྣམ་དབྱེ་དྲུག་པའི་མང་ཚེག་ཨཱ་མ་ལ་ནུ་ཊ་ཨཱ་ག་མ་ངེས་བར་འཇུག་སྟེ། ནུ་ཊ་ཨཱ་ག་མ་བྱིན་པས། ཀྲོཥྚྲཱྀ་ཎཱཾ། རྣམ་དབྱེ་བདུན་པའི་ངི་ལ། ཀྲོཥྚ་རི། ཀྲོཥྚཽ། ཨོས་ལ། ཀྲོཥྚྲོཿ། ཀྲོཥྚྲོཿ། སུ་པ་ལ། ཀྲོཥྚུ་ཥུ། བོད་ཚིག །
ཧེ་ཀྲོཥྚོ། ཧེ་ཀྲོཥྚཱ་རཽ། ཧེ་ཀྲོཥྚཱ་རཿ། གཉིས་པའི་ཤས་ལ་ཡང་ན་ཀྲོཥྚྲཱཾ་ན་ཞེས་པའང་འབྱུང་ངོ་། །དེ་ཡན་གྱིས་སྐྱེས་པའི་རྟགས་ཀྱི་ཨུ་ཐུང་མཐའ་ཅན་སྤྱིར་བཏང་བར་དེ་མིགས་བསལ་རྣམས་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་སྐྱེས་པའི་རྟགས་ལས་ཨཱུ་རིང་མཐའི་ཅན་གྱི་བྱ་བ་རྣམས་སྟོན་པར་བྱེད་དེ། ཧཱུ་ཧཱུ་
ཞེས་པའི་སྒྲ་ལ་དཔེར་བརྗོད་དུ་བྱ་སྟེ། དབྱངས་མི་འདྲ་བའི་འགྱུར་ཁྱད་ཕྲན་བུ་མ་གཏོགས་པཱ་ཏ་པྲ་མཱི་དང་འདྲ་བས། ཧཱུ་ཧཱུལ། རྣམ་དབྱེད་དང་པོ་རྣམས་བྱིན། གོང་ལྟར་གྱི་ཆོ་གས་བསྒྲུབས་བས། ཧཱུ་ཧཱུཿ། ཧཱུ་ཧཱཽ། ཧཱུ་ཧཿ། གཉིས་པ་ལ། ཧཱུ་ཧྭཾ། ཧཱུ་ཧྭཾ། ཧཱུ་ཧྱཿ གསུམ་པ་ལ། ཧཱུ་ཧཱ། ཧཱུ་
ཧཱུ་བྷྱཱཾ། ཧཱུ་ཧཱུ་བྷིཿ། ལྟ་བུ་སྟེ་ཨ་ཅ་ཐོག་མ་ཅན་རྣམས་ལ། ཨུ་ཝར་འགྱུར། ཧ་ལ་ཐོག་མ་ཅན་རྣམས་ལ་བྱ་བ་ལྷག་པོ་མེད་བས་གོ་བདེའོ། །དེ་བ་ནིན་དུ། ཨ་ཏི་ཙ་མཱུ་ལ་སོགས་པ་སྤྱིར་བཏང དྲིས ཨཱུ་རིང་སྐྱེས་པའི་རྟགས་ཐམས་ཅད་ཆོ་ག་འདི་དང་འདྲའོ། །ཨཱུ་རིང་མཐའ་ཅན་ཁ་ལ་པཱུ་ལ་སུ་
བྱིན། ཁ་ལ་པཱུཿ། ཁ་ལ་པཱུ་ལ། ཨཽ། ཨོཿསུ་པི། ཨཱུའི་སུ་བ་ལའོ། །བྱིངས་ཀྱི་ཨཱུ་རྣམ་དབྱེ་དང་པོའི་སུ་ནས། བདུན་པའི་མང་ཚིག་གི་པའི་བར་གྱི་ཨ་ཙ་ ཤིན་ཏུ་གཡོ་བ། ཐོག་མ་ཅན་གང་བྱུང་ཡང་ཨཱུ་དེ་ཝ་ཁོ་ནར་འགྱུར་ཏེ། དེ་ཡང་མཐའ་ཨཱུ་དེའི་སྔོན་དུ་ཡི་གེ་གཉིས་སམ་གསུམ་སོགས་
ཡོད་པ་ཡི་གེ་བརྩེགས་པ་མེད་པ་ཞིག་དགོས་སོ། །ཁ་ལ་བཱཽ། ཁ་ལ་བྲཿ། གཉིས་པ་རྣམས་ལ། ཁ་ལ་པྭཾ། ཁ་ལ་པྭཽ། ཁ་ལ་པྭཿ། གསུམཔ་རྣམས་ལ། ཁ་ལ་པཱ། ཁ་ལ་པཱུ་བྷཱཾ། ཁ་ལ་པཱུ་བྷིཿ། སོགས་ཏེ་ལ་སྒྲུབ་ཚུལ་རིགས་འགྲེའོ། །སུ་ལཱུ་ལ་སོགས་པ་ཡང་འདི་དང་འདྲའོ། །སྭ་བྷྱ་
ཞེས་པ། སུ་དྡྷིའི་སྐབས་དང་འདྲ་བས། སུ་ལ་སྭ་བྷཱུཿ། རྣམ་དབྱེ་ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ལ། ཨུ་ཝར་འགྱུར་བས། སྭ་བྷུ་ཝཽ། སྭ་བྷུ་ཝཿ། གཉིས་པ་ལ། སྭ་བྷུ་ཝཾ། སྭ་བྷུ་ཝཽ། སྭ་བྷུ་ཝཿ། གསུམ་པ་ལ། སྭ་བྷུ་ཝཱ། སྭབྷུ་བྷྱཱཾ། ལེགས་གཅོད། སྭ་བྷཱུ་བྷིཿ། སོགས་སོ། །ཀ་ཊ བདག་ཡོད།
