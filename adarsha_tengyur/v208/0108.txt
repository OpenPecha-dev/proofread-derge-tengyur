གོ་བྷྱཿ། དྲུག་པ་རྣམས་ལ། གོཿ། ག་ཝཱཾཿ། ག་ཝཱཾ། བདུན་པ་རྣམས་ལ། ག་ཝི། ག་ཝོཿ། གོ་ཥུ། བོད་ཚིག །ཧེ་གཽཿ། ཧེ་གཱ་ཝཽ། ཧེ་གཱ་ཝཿ། དེ་བཞིན་དུ། སུ་དྱོ་སོགས་ཀྱང་འདི་དང་འདྲའོ། །དེ་ལྟར་ཨོ་མཐའ་ཅན་རྣམས་བཤད་ཟིན་ཏོ། ༈ །དེ་ནི་ཨེ་ཡིག་མཐའ་ཅན་ལས་རཻའི་སྒྲ་ལ་
དཔེར་བརྗོད་ན། ཪཻལ་སུ། རཱ་ཡོ་ཧ་ལི། རཻའི་ཧ་ལ་ལའོ། །ཨཻ་ཡིག་མཐའ་ཅན་རཻ་སྒྲ་ལྟ་བུའི་ཕ་རོལ་དུ་རྣམ་དབྱེ་ཧ་ལ་ཐོག་མ་ཅན་གང་བྱུང་ཡང་རྟགས་མཐའི་ཨཻ་དེ་ཨཱ་རུ་འགྱུར་རོ། །ཉི་ག །རཱཿ། རཱ་ཡཽ། རཱ་ཡཿ། གཉིས་པ་རྣམས་ལ། རཱ་ཡཾ། རཱ་ཡཽ། རཱ་ཡཿ། གསུམ་པ་རྣམས་ལ། གསེར་རམ་ནོར།
རཱ་ཡཱ། རཱ་བྷྱཱཾ། རཱ་བྷིཿ། བཞི་པ་རྣམས་ལ། རཱ་ཡེ། རཱ་བྷྱཱཾ། རཱ་བྷྱཿ། ལྔ་བ་རྣམས་ལ། རཱ་ཡཿ། རཱ་བྷྱཱ། རཱ་བྷྱཿ། དྲུག་པ་རྣམས་ལ། རཱ་ཡཿ། རཱ་ཡོཿ། རཱ་ཡཱཾ། བདུན་པ་རྣམས་ལ། རཱ་ཡི། རཱ་ཡོཿ། རཱ་སུ། བོད་ཚིག་ལ། ཧེ་རཱཿ། ཧེ་རཱ་ཡཽ། ཧ་རཱ་ཡཿ། ཨཽ་ཡིག་མཐའ་ཅན་གླཽ་ཞེས་པའི་
སྒྲ་ལ་དཔེར་བརྗོད་བྱ་སྟེ། ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ལ་ཨཽ་ཨཱ་ཝར་འགྱུར། ཧལ་ཐོག་མ་ཅན་རྣམས་ལ་བྱ་བ་ལྷག་པོ་མེད་པས། རྣམ་དབྱེ་དང་པོ་རྣམས་ལ། གླཽཿ། གླཱ་ཝཽ། གླཱ་ཝཿ། གཉིས་པ་ལ། གླཱ་ཝཱ། གླཽ་བྷྱཱཾ། གླཱཽ་བྷྱཱཾཿ། སོགས་སོ། །ནཽལ་
སོགས་བ་རྣམས་ཀྱང་འདི་དང་འདྲའོ། ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ད་ཐནྟ་བུ་རླིངྒཱ་སཱ་མཱ་བྷཱཿ། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་ཨ་ཙའི་མཐའ་ཅན་སྐྱེས་པའི་རྟགས་རྣམས་རྫོགས་སོ། །དེ་རྣམས་ཀྱིས་བརྡ་སྤྱོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག སུ།
དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས་སྐྱེས་པའི་རྟགས་ཨ་ཙ་མཐའ་ཅན་རྣམས་ཀྱི་ཆོ་ག་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པ་རགས་བ་རྣམས་རེ་ཞིག་རྫོགས་སོ། ༈ །ད་ནི་བུད་མེད་ཀྱི་རྟགས་ལས་ཨ་ཙ་མཐའ་ཅན་གྱི་ཆོ་ག་རྣམས་
བཤད་པར་བྱ་སྟེ། རམཱ་ལ། རྩ་བམ། སུ། ཧ་ལ་དཱི་བ་ཨཱ་པ་རིང་པོ་སོགས་ཀྱིས་སུ་ཕྱིས། ར་མཱ། ར་མཱ་ལ། ཨཽ། ཨཽ་པ་ཨཱ་པཿ། ཨཽའི་ཨཱ་པ་ལས་སོ། །བུད་མེད་ཀྱི་རྟགས་ཨཱ་པ་ཀྱི་རྟགས་ཨཱ་པ་ཀྱི་མཐའ་ཅན་ལས་རྣམ་དབྱེ་དང་པོའི་གཉིས་ཚིག་ཨཽ་དང་། གཉིས་པའི་གཉིས་ཚིག་ཨཽ་གཉིས་ནི་ཤཱི་
