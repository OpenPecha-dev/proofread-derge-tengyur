ཥོོ་ཥ་ངྷི་བ་ནསྤ་ཏི་བྷྱཿ། སྨན་རྣམས་ལ་བ་ནའི་གདམ་ངའོ། །སྨན་གྱི་སྒྲ་སྔོན་དུ་བྱུང་བ་ནའི་ན་ཎ་རུ་འགྱུར་བ་དང་མི་འགྱུར་བ་གདམ་ཀ་ཅན་ཏེ། མཱུརྦཱཎཱཾ་བ་ནཾ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། མཱུརྦ་བཎཾ། མཱུརྦ་བནཾ། དེ་བཞིན་དུ། ཤིརཱིཥཱ་ཎཱཾ་བ་ནཾ། ཞེས་པ་ལ། ཤིརཱི་ཥ་བཎཾ། ཤིརཱིཥ་བནཾ། དཱུརྦཱ་ཎཱཾ་
བནཾ། ཞེས་པ་ལ། དཱུརྦ་བནཾ། དཱུརྦབ་ཎི། ཨི་རི་ཀཱ་དི་བྷྱོན། ཨི་རི་ཀཱ་སོགས་རྣམས་ལས་མ་ཡིན་ནོ། །སྨན་གྱི་སྒྲུ་ཡིན་ཀྱང་ཨི་རི་ཀཱ་སོགས་ཀྱི་པ་རོལ་ལ་བ་ནའི་ན་ཎར་མི་འགྱུར་ཏེ། ཨི་རི་ཀཱནཱཾ་བ་ནཾ། ཞེས་པ་ལ། ཨི་རི་ཀཱ་བ་ནཾ། པཱ་ནཾ་དེ་ཤེ། པཱ་ནའི་ཡུལ་ལའོ། །ནིརཱི་ཥའི་ནགས། ཚིག་སྔོན་མ་
ངཱུར་བའི་ནགས། ལ་ན་ཎར་ རྩྭ་དཱུར་བའི་ནགས། འགྱུར་བའི་རྒྱུར་མཚན་ཅན་གྱི་ཡི་གེ་ཡོད་པ་ལ་དེ་ཡང་ཡུལ་བརྗོད་པའི་དོན་ཡིན་ན་པཱ་ནའི་ན་ཎར་འགྱུར་ཏེ། ཀྵི་རཔཱནཾ་ཡེཁཱཾ་ཏེ། ཞེས་པལ། ཀྵིར་པཱཎཾ། དེ་ ཀ་ཀོ་ལའི་ནགས། བཞིན་དུ། སུརཱ་པཱནཾ་ཡེ་ཥཱཾ་ཏེ། ཞེས་པ་ལ། སུརཱ་པཱཎཾ། བཱབྷ་བཀར་ཎ་ཡོཿ། དངོས་པོ་
དང་བྱེད་པ་དག་ལས་ཡང་ནའོ། །ཚིག་སྔོན་མན་ཎར་འགྱུར་བའི་རྒྱུ་མཚན་ཅན་གྱི་ཡི་གེ་ཡོད་པ་ལ་དངོས་པོ་དང་བྱེད་པའི་དོན་ཅན་ཡིན་ན་པཱ་ནའི་ན་ཎར་ཡང་ན་ གང་རྣམས་ཀྱི་འོམ་འཐུང་བདེ། འོམ་འཐུང་ག་གང་རྣམས་ཀྱི་ཆང་འཐུང་བདེ། འགྱུར་ཏེ། ཀྵི་རེ་ཎ་པཱ་ཎཾ། ཞེས་ཆང་འཐུང་བ། བ་ལ། ཀྵི་ར་བཱཎཾ། ཀྵི་
ར་པཱནཾ། དེ་བཞིན་དུ། ཀྵི་ར་པཱནཾ་ཀརོ་ཏིཔཱཏྲཾ། ཞེས་པ་ལ། ཀྵིར་པཱཎཾ་པཱཏྲཾ། ཀྵི་ར་པཱ་ནཾ་པཱ་ཏྲཾ། གི་རི་ན་ངྱཱ་དཱིནཱཾ་བཱ། གི་རིན་དཱི་སོགས་རྣམས་ཀྱི་ཡང་ནའོ། གི་རི་སོ་གསལ་ན་དཱིའི་ནཎ་ན་ར་ཡང་ན་འགྱུར་ཏེ། གི་རོོ་ནདཱི། ཞེས་པ་ལ། གི་འོམ་འཐུང་བ། རི་ཎ་དཱི། གི་རི་ན་དཱི། དེ་བཞིན་དུ།
ཙཀྲ་ཀཱ་རཾ་ན་དཱི་ཡ་སྱ། ཞེས་འོམ་འཐུང་བར་བྱེད་མའི་སྣོད། བ་ལ། ཙཀྲ་ཎ་དཱི། ཙ་ཀྲ་འོམ་འཐུང་སྣོད། ན་དཱི། པྲ་ཏི་པ་དེ་ཀཱཾ་ཏ་ནུ་སྔཾབྷཀྟི་ཥུ་ཙ། མིང་ནུ་ནམ་རྣམ་དབྱེ་རྣམས་ལའང་ངོ་། །ཚིག་སྔོན་མ་ལ་ན་ཎར་འགྱུར་བའི་རྒྱུ་མཚན་ཅན་ རི་ལ་ཆུ། གྱི་ཡི་གེ་ཡོད་ན་ཕ་རོལ་གྱི་མིང་ རིལ་ཆུ། གི་ན་དང་ནུ་མ་བྱིན་པའི་
གང་གི་འཁོར་བྱེད་པའི་ཆུ། ན་དང་ཊཱ་ཨི་ནར་འགྱུར་བ་ལྟ་བུ་རྣམ་དབྱེའི་ཆུ་འཁོར། ན་རྣམས་ཎར་ཡང་ན་འགྱུར་ཏེ། མིང་ལ། མ་ཥཤྩྭ་སོོ་བཱ་པཱི་ཙ། སྲན་མ་གང་འདིས་བོན་ཡང་། ཞེས་པ་ལ། མ་ཥ་བཱཔི་ཎོོ། སྲན་ས་བོན། ནུ་མ་ལ། བྲཱི་ཧི་བཱ་པཱཾ་ཡ་སྱ། གང་གི་འབྲུའི་ས་བོན། ཞེས་པ་ལ། ཛཿ་ཤི་ནུ་མ། བྲཱི་ཧི་བཱ་པཱ་ཎི། འབྲུའི་ས་བོན་རྣམས།
