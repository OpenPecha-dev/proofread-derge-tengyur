བས་ན་སྔ་མ་ལས་བཟློག་པ་མཚམས་ཀྱི་ནང་དང་འཐོབ་པའི་ཕྱོགས་དང་། མདུན་ན་འདུག་པ་ལས་དགེ་སློང་གིས་ལྷག་བོར་བྱ་བ་འབྱུང་སྟེ། དེ་ནི་རེ་ཞིག་ཕྱིར་བཏང་བའི་མཚན་ཉིད་ཡིན་ནོ། །བྱ་བ་ལག་ཏུ་མ་བླངས་པ་ལ་ཡང་མི་འབྱུང་ངོ་ཞེས་བྱ་བ་ལས་ནི། ལྷག་བོར་བྱ་བའི་དངོས་པོ་གང་ཡིན་པ་དེ་ལག་
ཏུ་མ་བླངས་པ་ལ་ཡང་ལྷག་བོར་བྱ་བ་མི་འབྱུང་སྟེ། ཐུ་བར་བཞག་པ་ལ་ཡང་མི་འབྱུང་ལ་བྱུག་རིས་སུ་བཞག་པ་ལ་ཡང་མི་འབྱུང་ངོ་། །འདིར་གཞུང་ནི་དགེ་སློང་དེས་ལག་པ་བསྙལ་ཏེ་བྱིན་ལེན་བྱས་ལ་ཞེས་གསུངས་པ་ཡིན་ནོ། །ནམ་མཁའ་ལ་ཡང་མི་འབྱུང་ངོ་ཞེས་བྱ་བ་དེ་ནི་མཚམས་ཀྱི་ཕྱི་རོལ་
ཞེས་བྱ་བ་འདིའི་ཁོངས་སུ་གཏོགས་པ་ཉིད་ཡིན་ནོ། །རུང་བ་མ་ཡིན་པ་དང་འདྲེས་པ་ལ་ནི་མི་བྱ་བ་ཉིད་དོ་ཞེས་བྱ་བ་ནི། ཟས་རུང་བ་མ་ཡིན་པ་དང་། འདྲེས་པ་ལ་ནི་ལྷག་བོར་གྱི་དངོས་པོ་མི་བྱ་བ་ཉིད་ཡིན་ནོ། །ལྷག་བོར་བྱ་བ་དེ་སུས་བྱ་བ་ཉིད་ཡིན་ཞེ་ན། དེའི་ཕྱིར། གནས་མ་བཏང་བས་བྱ་བ་ཉིད་དོ་ཞེས་བྱ་བ་
སྨྲས་ཏེ། དགེ་སློང་གིས་ཟན་ཟ་བའི་གནས་མ་བཏང་བ་དེ་ནི་གནས་མ་བཏང་བ་སྟེ། དེས་ལྷག་བོར་གྱི་དངོས་པོ་དེ་བྱ་བ་ཉིད་ཡིན་ནོ། །ཅི་ཆོག་པར་མ་བྱས་པའི་དགེ་སློང་འབའ་ཞིག་གིས་ལྷག་བོར་བྱ་བ་ཉིད་ཡིན་ནམ་ཞེ་ན་དེའི་ཕྱིར། མི་བཟའ་བར་བརྟག་པ་བྱས་པས་ཀྱང་ངོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། མི་ཟ་བར་
བརྟག་པ་བྱས་པ་ཞེས་བྱ་བ་ནི། གང་བདག་གིས་དེང་འདི་ལ་ཟས་གཞན་མི་བཟའོ་ཞེས་དེ་ལྟར་བརྟག་པ་བྱས་པ་སྟེ། དེས་ཀྱང་ལྷག་བོར་བྱ་བ་ཉིད་ཡིན་ནོ། །ཀྱང་ངོ་ཞེས་བྱ་བའི་སྒྲས་ནི་མི་བཟའ་བར་བརྟག་པ་མ་བྱས་པས་ལྟ་སྨོས་ཀྱང་ཅི་དགོས་ཞེས་བྱ་བར་སྟོན་ཏོ། །ཅི་མ་གསོལ་བས་ཀྱང་བྱའམ་ཞེ་ན། 
དེའི་ཕྱིར། གསོལ་བས་བྱ་བ་ཉིད་ཡིན་ནོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། ཚེ་དང་ལྡན་པ་དགོངས་སུ་གསོལ། བདག་དགེ་སློང་མིང་འདི་ཞེས་བགྱི་བ་འཚལ་མ་འཚལ་ལགས་ཤིང་སྤངས་ཏེ་ཡོངས་སུ་རྫོགས་ཤིང་མཐར་ཐུག་པར་བགྱིད་པ་ལས་བཅའ་བ་དང་བཟའ་བ་རྙེད་ཅིང་འཚལ་བར་འཚལ་ན། ལྷག་བོར་བགྱིས་
ཏེ། བདག་ལ་སྩལ་དུ་གསོལ་ཞེས་གང་ལ་གསོལ་བ་བཏབ་པ་དེས། ལྷག་བོར་བྱ་བ་ཉིད་ཡིན་ནོ། །ལྷག་བོར་བྱེད་པས་ལྷག་བོར་བྱ་བ་ཇི་ལྟར་བྱ་ཞེ་ན། དེའི་ཕྱིར། ཁྱོད་ཀྱིར་གྱུར་གྱིས་བཞུད་ཅིག་ཅེས་བྱ་བའི་ཚིག་གིས་བྱའོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། ཁྱོད་ཀྱིར་གྱུར་གྱིས་བཞུད་ཅིག་ཅེས་བྱ་བའི་ཚིག་བརྗོད་པར་བྱ་
