དྷཱཿསྨཱ་ར་ཎ་ཧ་ར་ཎ་པྲོོ་ཌྷ་ཀེ་ལཱི་བི་ལཱ་སེ་ཧ་པེ་ནེ་བལྥུ་རི༔རེ་ཏ་རུ་ཙི་ནཱ་ནིརྦྷཱ་ར་བྱཱཔྟ་ཀཱ་ཡཿ། །སོ་ཐ་པྲཱཔྟེ་ལི་པི་པ་རི་ཙ་ཡེ་ཧེ་ལ་ཡཱ་དཏྟ་ཧསྟེཿསཾ་སཱ་རེསྨིནྣ་བི་ཙ་ལ་ཏ་ཡཱ་བནྡྷ་ལེ་ཁྱ་ཀྲི་ཡཱ་སུ། །ཛནྨཱ་
མེད་ལ་འགྲོགས་མཆོག་ཏུ་རྩེ་དགའི་རོལ་པ་དྲན་པ་ལས། །དགོད་པའི་འོད་ཟེར་ཤིན་ཏུ་གསལ་བས་རབ་ཏུ་ཁྱབ་པ་བཞིན། །དེ་བས་དེ་ནི་རྩེ་བས་ཡིག་འཕྲེང་ཡོངས་འདྲིས་དག་ལ་ལག་གཏད་ཐོབ། །འཁོར་བ་འདིར་ནི་གཡོ་མེད་འཆིང་བའི་རྩིབས་ཀྱི་རི་མོའི་ལས་དག་ལས། །རང་
བརྟཾ་ནི་ཛ་མི་བ་ལ་ཁན་དཱིརྒྷ་མོ་ཀཱ་ར་མཱ་དཽ་བརྒེ་བརྒེ་བི་ར་ཏི་མ་ཏིཾ་བྷོ་ག་སརྒེ་ནི་བིཥྚཿ། །པྲཱཔྟཿཔྲཛྙཿཀ་ཐ་མ་པི་ག་ལངྦཱ་ལ་བྷཱ་བ་པྲ་མོ་ཧཿཀ་མོཏྶུ་ཀྱཱཏ་པུ་ན་ར་པི་པ་རི་ཡཽ་བ་ནེ་ནཥྚ་སཾ་ཛྙཿ། །
གི་སྐྱེ་བའི་འཁོར་མོ་རིང་དུ་འབྲི་བཞིན་ཡོཾ་ཡིག་ལ་སོགས་པ། །སྡེ་དང་སྡེ་ལ་བློ་གྲོས་རབ་གསལ་ལོངས་སྤྱོད་སྤྲོས་པ་ངེས་པར་འཇུག །ཇི་ཞིག་ལྟར་ཡང་བྱིས་པའི་དངོས་པོའི་རབ་རྨོངས་བྲལ་ནས་ཤེས་རབ་འཐོབ། །སླད་ནས་ཀྱང་ནི་ལང་ཚོ་དར་ལ་འདོད་པས་རབ་བསླང་འདུ་ཤེས་ཉམས། །
ནིསྶཱ་རེ་ཥུ་བྱ་པ་ན་བི་ལ་སད་མེ་གྷ་མཽ་དཱ་མི་ནཱི་ནཱཾ་བ་དྷྭ་ཏྱསྠཱཾ་སྠི་ར་ཏ་ར་དྷི་ཡ་ཡཽ་ཥི་ཏཱཾ་བི་བྷྲ་མེ་ཥུ། །བཱ་ཙི་ཤྲོ་ཏྲཾ་བ་པུ་ར་པི་ཙ་ལཱ་ལིཾ་ག་ནེ་ཥྭཾ་ཛ་ནཱ་ནཱཾ་གྷཱ་ཎཾ་བཀྵཾ་པ་བ་པ་རི་མཱ་ལེ་སྭ་ད་ནཾ་ཏད་ར་སེ་ཙ། །པཱ་ཏྲི་ཀུརྦཱན་
སྡུག་བསྔལ་མཛེས་པའི་སྤྲིན་གློག་བུད་མེད་རྣམས་ཀྱི་རྩེ་དགའ་རོལ་པ་དག །སྙིང་པོ་མེད་ལ་རབ་ཏུ་བརྟན་པའི་བློ་ཡིས་མངོན་པར་འདོད་པ་བཅིངས། །བུད་མེད་རྣམས་ཀྱི་ཚིག་ལ་རྣ་བ་ལུས་ནི་གཡོ་མེད་འཁྱུད་པ་ལ། །ཁ་ཡི་ཆང་གི་དྲི་ལ་སྣ་དང་དེ་ཡི་རོ་ལ་མྱོང་བྱེད་དང་། །བཞིན་ལ་མིག་ཀྱང་སྣོད་
