བཱནྨ་མ། །བྷི་ཀྵུ་སཾ་གྷཽ་པ་ཀཱ་རཱརྟེ་གྲྀ་ཧ་ཨེ་བཱ་ད་ར་པ་རཾ། །ཨི་ཏྱུཀྟཱ་བྷ་ག་བ་བཱཀྱ་མ་ཏི་ཀྲཱནྟུ་མ་ནཱི་ཤྭ་རཿ། །ཀྲི་ཥྱ་མཱ་ཎཿཔྲི་ཡཱ་པྲེསླ་སོ་བྷཱུདྡོ་ལཱ་ཀུ་ལཱ་ཤ་ཡཿ། །པུ་ནཿབུ་ནཤྩ་ཀཱ་རཱ་སྱ་བྷ་ག་བཱན་བྲ་ཏ་དེ་ཤ་
གི་རབ་བྱུང་ཡིན། །དགེ་སློང་དགེ་འདུན་རིམ་གྲོར་ནི། །འོས་པའི་ཁྱིམ་ལ་མཆོག་ཏུ་གུས། །དེ་སྐད་སྨྲས་ནས་བཅོམ་ལྡན་གསུང་། །བསྒོམ་བྱའི་དབང་ཕྱུག་མིན་པ་དང་། །དགའ་མ་ལ་དགས་དྲངས་པས་འདིའི། །བསམ་པ་ཁྱོགས་ལ་ཕྱང་མོ་ཡུག །ཡང་དང་ཡང་དུ་བཅོམ་ལྡན་གྱིས། །འདི་ལ་བརྟུལ་ཞུགས་ཉེར་
ནཱཾ། །ཨུ་པ་ཀཱ་རོདྻ་ཏཱཿསཾ་ཏཤྩིནྟ་ཡནྟཾ་ན་ཡོ་གྱ་ཏཱཾ། །ཡ་དཱ་ནེཙྪ་ཏི་ནནྡསྟཱཾ་པྲ་བྲཥྐཱ་མ་ཛེ་ཏིནྡྲི་ཡཿ། །ཏ་དཱ་སྱ་བྷ་ག་བདྦཱ་ཀྱ་ས་པ་ཏཏ་བ་པུ་ཥི་སྭ་ཡཾ། །མུཎྜཿཀཱ་ཥཱ་ཡ་པརྞྞ་ནཿཔཱ་ཏྲ་བཱཎཿཏ་ཏ་ཀྵ་ཎ། །བ་བྷོོ་ཙ་མཱི་ཀ་ར་རུ་ཙིརྨ་ཧཱ་
བསྟན་མཛད། །ཕན་པར་བརྩོན་པའི་དམ་པ་རྣམས། །མི་འོས་པ་ཉིད་མི་བསམ་མོ། །དབང་པོ་མ་ཐུལ་དགའ་བོ་ནི། །གང་ཚེ་རབ་བྱུང་མི་འདོད་པ། །དེ་ཚེ་དེ་ཡིས་ལུས་ལ་དེ། །བཅོམ་ལྡན་གསུང་གིས་རང་ཉིད་ལྷུང་། །དེར་ནི་དེ་ཚེ་ངུར་སྨྲིག་གོས། །མགོ་ཟླུམ་ལག་ན་ལྷུང་བཟེད་ཅན། །སྐྱེས་བུ་ཆེན་པོའི་མཚན་རྣམས་
པུ་རུ་ཥ་ལཀྵ་ཎཻཿ། །ས་ཤ་ས་ནཱཛྫི་ན་སྱཱ་བྷཱུ་དཱ་ར་ཎྱ་བིཎྜཿཔཱ་ཏི་ཀཿ། །ཨ་གཱ་རཱ་ད་ནཱ་གཱརྱ་ཏཱཾ་པྲ་ཡཱ་ཏཿཔཱཾ་ཤུ་ཀཱུ་ལི་ཀཿ། །པྲི་ཡཱ་མུ་བཱ་ཧ་ས་ཏ་ཏཾ་ཤྱཱ་མཾ་པྲ་བྲ་ཛི་ཏོ་པི་ས། །ཤ་ཤཱ་ཀ་ཨི་བ་སཾ་བཀྟཾ་ཧྲྀ་ད་ཡེ་ལཱཉྪ་
ཀྱིས། །གསེར་གྱི་འོད་ཟེར་ལྟ་བུར་མཛེས། །དགོན་པ་པ་དང་བསོད་སྙོམས་པར། །རྒྱལ་བའི་བཀའ་ཡིས་དེར་གྱུར་ཏོ། །ཁྱིམ་ལས་ཁྱིམ་ནི་མེད་ཉིད་དང་། །ཕྱག་དར་ཁྲོད་པ་ཉིད་དུ་གྱུར། །དེ་ནི་རབ་ཏུ་བྱུང་གྱུར་ཀྱང་། །སྙིང་ལ་ཡང་དག་གཅགས་པ་ཡིས། །སྔོ་བསངས་དགའ་མ་རྟག་ཏུ་འཛིན། །ཟླ་བའི་མཚན་མ་
