པ་ད་མ་ནྱ་དྷཱིཿ། །ཀྵ་ཎ་པྲ་མུ་ཥི་ཏཱ་བྲཱི་ཏི་མ་ལཱ་ནི་ཡཱ་ཏི་གུ་ཎེ་ཥྭ་པི། །ཏ་ཏཿཀ་ད་ཙི་དཱི་ཙ་ར་ན་ནནྡཿཀྭཱ་པི་བྱ་ལོ་ཀ་ཡ་ཏ། །ཀ་རཱ་ལ་ན་ར་ཀཱ་སཀྟ་བྷཱི་མཱ་ཀུམྦྷི་བྷྲྀ་ཏཱ་བྷུ་བཾ། །ཏཾ་བི་ལོ་ཀྱཻ་བ་སཱ་ཀམྤཿགི་མེ་ཏ་དི་ཏི་
གཞན་གྱི་བློ་དེས་བརྗེད་པར་གྱུར། །སྐད་ཅིག་རབ་ཉམས་དགའ་བ་ནི། །ཡོན་ཏན་ལའང་དམན་པར་གྱུར། །དེ་ནས་ནམ་ཞིག་དགའ་བོ་ནི། །རྒྱུ་བས་རབ་འཇིགས་དམྱལ་བའི་ཚོགས། །འཇིགས་རུང་ཟངས་ཆེན་འཛིན་པ་ཡང་། །འགའ་ཞིག་དག་ཏུ་མཐོང་བར་གྱུར། །དེ་མཐོང་ཉིད་ལ་སྡུག་བསྔལ་ཞིང་། །འདར་བར་

གྱུར་པས་འདི་ཅི་ཞེས། །དེ་ཡིས་དེ་ནས་ཚོགས་པ་ཡི། །འཇིགས་རུང་དམྱལ་སྲུང་རྣམས་ལ་དྲིས། །དེ་ལ་དེས་སྨྲས་ཚ་བའི་ས། །ཟངས་ཆེན་རྒྱ་ནི་རྫོགས་པ་འདི། །ཤིན་ཏུ་ཆགས་ལྡན་རྒྱལ་པོའི་བུ། །དགའ་བོ་ལ་ནི་བྱས་པ་ཡིན། །ཆགས་བྲལ་ཉིད་ནི་ད་དུང་ཡང་། །བརྟུལ་ཞུགས་ལོག་པ་དེ་མི་སྟེན། །མཐོ་རིས་བུད་
