ས་མཱ་པ་དེ་སྭ་ཡཾ་ལོ་ཀཱ་ནུ་ཀམྤ་ཡཱ། །བ་ཧནྟཱི་བོ་དྷི་སཏྭཾ་སཱ་གརྦྷ་བྷཱུ་བ་ན་ནནྡ་ནཾ། །ཨིནྡུ་དུགྡྷཱ་བི་བེ་ལེ་བ་བ་བྷཱུ་བ་པཱཎྜ་ར་དྱུ་ཏིཿ། །ཨིཀྵྣཱ་ཀུ་རཱ་ཛ་བཾ་ཤྱེ་ན་ཏེ་ན་གབྷཱ་སྠི་ཏེ་ན་སཱ། །བྷུནྣི་
ཡི་གནས་དག་ལས། །དེ་ཡི་ལྷུམས་སུ་ཡང་དག་བྱོན། །འཇིག་རྟེན་དགའ་བྱེད་བྱང་ཆུབ་སེམས། །དེ་ཡིས་ལྷུམས་སུ་བཟུང་བ་ནི། །འོ་མཚོའི་རླབས་ཀྱི་ཟླ་བ་བཞིན། །དཀར་བའི་འོད་དང་ལྡན་པར་གྱུར། །བུ་རམ་ཤིང་སྐྱེས་རྒྱལ་པོའི་བརྒྱུད། །དེ་ནི་ལྷུམས་ན་གནས་པས་དེ། །གཏེར་
དྷ་ན་བ་ཏཱི་བཱ་བྷཱུརྦྷ་བྱ་ལ་ཀྵ་ཎ་ལཀྵི་ཏཱི། །བ་བྷུ་བ་དོ་ཧ་དསྟ་སྱཱ་དཱ་ན་པུ་ཎྱ་མ་ཡོ་ད་ཡཿ། །ཨཾ་ཀུ་རེ་སྱ་བི་སཾ་བཱ་དི་ས་ཧ་ཀཱ་ར་སྱ་སཽ་ར་བྷཾ། །ཨ་ཐ་ཀཱ་ལེ་ན་སཾ་པུརྞྞཾ་སཱ་ལུམྦི་ནི་བ་ནེསྠི་ཏཱ། །ཨ་སཱུ་ཏ་ཏ་ན་ཡནྡེ་
དང་ལྡན་པའི་ས་གཞི་བཞིན། །དགེ་བའི་མཚན་ཉིད་ཀྱིས་མཚོན་གྱུར། །དེ་ཡི་སྲེད་པ་བྱུང་གྱུར་པ། །སྦྱིན་པའི་བསོད་ནམས་རང་བཞིན་སྐྱེས། །ས་ཧ་ཀ་རའི་དྲི་བཟང་ནི། །སོལ་བ་ལ་ཡང་སླུ་བ་མིན། །དེ་ནས་དུས་ཀྱིས་ཡང་དག་རྫོགས། །ལུམྦི་ནི་ཡི་ནགས་གནས་པའི། །ལྷ་མོ་དེ་ལ་སྲས་བལྟམས་
བཱི་དི་བཱ་ཀ་ར་མི་བཱ་དི་ཏིཿ། །མཱ་ཏུརྒྒ་རྦྷ་མ་ལཱ་སྤྲྀཥྚཾ་ཀུཀྵིཾ་བྷི་ཏྭཱ་ས་ནིརྒྒ་ཏཿ། །ཏཱཉྩ་ཀྲེ་ཐ་སྭ་བྷཱ་བེ་ན་སྭ་སྠཱཾ་གི་བི་ག་ཏ་བྱེ་ཐཱཾ། །ནིརྒྒ་ཙྪནྣེ་བ་རུདྡྷོ་སོོ་བ་ལ་ཛི་ཛྙ་ས་ཡཱ་ཀྵ་ཎཾ། །ཤ་ཀྲེ་ཎ་བཛྲ་སཱ་རཱཾ་ག་སྟ་སྱཾ་ཤཱཀྱ་ཏྭ་མཱ་
ཏེ། །ཨ་དི་ཏི་ལས་ཉིན་བྱེད་བཞིན། །མ་ཡི་མངལ་གྱི་དྲི་མ་ཡིས། །མ་རེག་དེ་ཡི་དཀུ་བརྟོལ་བྱུང་། །དེ་ནས་དེ་ཡི་རང་བཞིན་གྱིས། །གདུང་བྲལ་ལུས་ནི་རྣལ་གནས་གྱུར། །ཕྱིར་བྱུང་ཉིད་ན་བརྒྱ་བྱིན་གྱིས། །སྟོབས་ཤེས་འདོད་ནས་སྐད་ཅིག་ནི། །རྡོ་རྗེ་སྙིང་པོའི་སྐུ་དེ་བཟུང་། །དེ་ནི་ནུས་མེད་ཉིད་དུ་
