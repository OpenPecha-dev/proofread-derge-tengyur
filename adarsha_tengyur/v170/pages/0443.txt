བ་བྷཱུ་བཱ་མི་ལི་ཏེ་ཀྵ་ཎ། །ཤ་ཀྲ་བྲཧྨ་མུ་ཁ་སརྦྦེ་ས་མེ་ཏྱཱ་ཐ་སུ་དྷཱ་བྷུ་ཛཿ། །ཙཀྲི་རེ་བོ་དྷི་སཏྭ་སྱ་སཱལྶོ ཏྶཱ་ཧ་པྲ་པཱུ་ར་ཎཾ། །ཏཻརྡེ་བ་པུ་ཏྲཤྩ་ཏྭཱ་རཿསམཱ་དིཥྚཱ་མ་ཧཱ་ཛ་བཱཿ། །ས་ཧཱ་
ཕྱོགས་ཏེ། །མིག་ནི་ཟུམ་པ་ཉིད་དུ་གྱུར། །དེ་ནས་བརྒྱ་བྱིན་ཚངས་ལ་སོགས། །བདུད་རྩི་སྤྱོད་རྣམས་ཡང་དག་འོངས། །བྱང་ཆུབ་སེམས་དཔའི་སྤྲོ་བ་ནི། །ཤིན་ཏུ་ཆུང་བ་རབ་རྫོགས་བྱས། །དེ་དག་རྣམས་ཀྱིས་ལྷ་ཡི་བུ། །ས་རི་ཆུ་གཏེར་འཛིན་བཟོད་པ། །ཤིན་ཏུ་
ཡཱ་ག་མ་ནེ་ཏསྱ་བྷཱུ་ཤཻ་ལབྷྷིདྷྲྀ་ཏི་ཀྵ་མཱཿ། །ཤ་ཀྲཱ་དི ཥྚེ་ན་ཡཀྵེ་ཎ་པཱཉྩི་ཀ་ཁྱེ་ན་ནིརྨྨི་ཏེཿ། །ས་ཧརྨྱཱ་སཀྟ་སོ་པཱ་ནེ་ར་བ་ཏཱི ཪྻ་བི་ནིཪྻ་ཡཽ། །སྭཔྣཾ་ས་ར་ཐི་མཱ་དཱ་ཡཙྪ ནྡ་ཀཱ་ཁྱི་བོ་དྷི་སཿ། །ཨུཏྶཱ་
མགྱོགས་པར་གྱུར་པ་བཞི། །དེ་ཡི་འགྲོ་བའི་གྲོགས་ལ་བསྒོས། །བརྒྱ་བྱིན་གྱིས་བསྐོས་གནོད་སྦྱིན་ནི། །ལྔས་རྩེན་ཞེས་པས་རྣམ་སྤྲུལ་པ། །ཁང་བཟང་ལ་རེག་ཐེམ་སྐས་ལ། །བབས་ནས་དེ་ནི་ཕྱི་རོལ་བྱུང་། །འདུན་པ་ཞེས་པའི་ཁ་ལོ་པ། །གཉིད་ནི་སད་པར་བྱས་ཏེ་བོས། བསྔགས་

ལྡན་ཞེས་པ་རྟ་ཡི་མཆོག །དེ་ཡིས་སྤྲོ་བ་བཞིན་དུ་བཟུང་། །དཔལ་མོའི་ཟུར་མིག་ལྟར་གཡོ་བའི། །རྣོ་མཛེས་འཕྲོག་བྱེད་དེ་ལ་ནི། །དེ་ཡི་ཕྱག་གིས་སྲབ་སྐྱོགས་ནས། །བཟུང་ཞིང་སྤྱི་བོར་རེག་པར་བྱས། །གང་གི་མཐུ་ཡིས་ཕྱུགས་རྣམས་ཀྱང་། །གཡོ་བ་དག་ནི་རྣམ་པར་གཏོང་། །ཞི་
