ཨུ་བཱ་ཙ་ས་སྨི་ཏ་ཀེ་ཡཾ་ག་མ་ནེ་སཾ་པྲ་ཏི་ཏྭ་ཡཱ། །བི་ཥ་ཡཱ་སྭཱ་ད་སོོ་ཧརྡ་སཾ་མོ་ཧཱ་དྷི་ཙེ་ཏ་སཾ། །ཨ་ཧོ་གྲྀ་ཧ་སུ་ཁ་སྭེ་བ་ནི་རྦེ་ད་བི་མུ་ཁཱ་མ་ཏིཿ། །གུ་ཎཱ་བྷ་ར་ཎ་མེ་བཱ་ཡུརྦི་བེ་ཀཱ་བྷ་ར་ཎ་གུ་ཎཿ། །པྲ་ཤ་མཱ་བྷ་ར་ཎ་ཤྩ་
འཛུམ་དང་བཅས་གསུངས་ད་ལྟ་ཉིད། །མྱུར་བར་འགྲོ་བ་འདི་ཅི་ཞིག །ཡུལ་གྱི་རོ་ལ་གཅགས་པ་ཡི། །ཀུན་རྨོངས་ཀྱིས་གཟིར་སེམས་ལྡན་རྣམས། །ཀྱེ་མ་ཁྱིམ་གྱི་བདེ་ཉིད་ལ། །ཡིད་འབྱུང་ལས་ནི་ཕྱིར་ཕྱོགས་བློ། །ཚེ་ཡི་རྒྱན་ནི་ཡོན་ཏན་ཏེ། །ཡོན་ཏན་རྒྱན་ནི་དབེན་པའོ། །འདི་ཡི་རྒྱན་ནི་རབ་ཞི་སྟེ། །ཞི་
སཽ་བཻ་རཱ་གྱཱ་བྷ་ར་ཎཿཤ་མཿ། །ཏཛྫ་ཌྱཾ་ཏ་ད་ས་ཧྱ་ཤ་ལྱ་ཤ་ལ་ཀཾ་ནྱསྟཾ་སུ་ཧྲྀཙྩེ་ཏ་སི་པྲཛྙེསྟངྒ་ཎི་ཏཾ་བི་ཙཱ་ར་ས་མ་ཡེ་བཻ་ཕ་ལྱ་མེ་བཱ་ཡུ་ཥཿ། །ཡདྦེེ་རཱ་གྱ་པི་བེ་ཀ་ཤུ་ནྱ་མ་ན་ས་མེ་ཏེ་ས་ཤཱུ་ནཱཾ་ཡ་ཐཱ་ཡཱནྟྱ་ཡཱནྟི་ཙ་ཙཀྲ་ནེ་མེ་ཙ་ལ་
བའི་རྒྱན་ནི་ཆགས་བྲལ་ལོ། །སྙིང་སྡུག་སེམས་ལ་བཀོད་པའི་བླུན་པོ་དེ་དང་བཟོད་དཀའི་ཟུག་རྔུའི་དུམ་བུ་སྟེ། །བློ་ལྡན་རྣམས་ཀྱིས་དེ་ནི་རྣམ་པར་དཔྱད་ཚེ་ཚེ་ནི་འབྲས་བྲལ་ཉིད་དུ་བརྩིས། །གང་ཡིད་ཕྱུགས་བཞིན་ཆགས་བྲལ་དབེན་པ་ཡིས་སྟོང་མཚན་ཉིད་མེད་པ་རྣམས་དག་གིས། །སྐད་ཅིག་འདི་དག་འོང་
ནཻརྣི་ལཀྵ་ཎ་ནཱཾ་ཀྵ་ཎཿ། །པུ་ཎྱཾ་ས་ཏྭ་བ་ཏཱཾ་ཤྲུ་ཏམྨ་ཏི་མ་ཏཱཾ་ཤི་ལཾ་ཙ་བི་དྱཱ་བ་ཏཱཾ་སརྦཾ་བྷཱ་གྱ་བ་ཏཱཾ་སུ་ཁཾ་ཤ་མ་བ་ཏཱཾ་ནཻ་བ་ཀྭ་ཙི་དུརླ་བྷི། །དུཥྤྲཱ་བྷ་སྟུ་ས་མཔྟ་བསྟུ་བ་ས་ཏེཿསྭལྤཾ་ཤ་རྐེ་པྱཱ་ཡུ་
དང་འགྲོ་བ་ཡང་ནི་འཁོར་ལོའི་མུ་ཁྱུད་གཡོ་བ་བཞིན། །སྙིང་སྟོབས་ལྡན་པའི་བསོད་ནམས་བློ་ལྡན་གྱིས་ཐོས་རིག་བྱེད་ལྡན་པའི་ཚུལ་ཁྲིམས་དང་། །སྐལ་ལྡན་དག་གི་ཐམས་ཅད་ཞི་ལྡན་རྣམས་ཀྱི་བདེ་བ་འགྱུར་ཡང་རྙེད་དཀའི་མིན། །མཐའ་དག་དངོས་པོའི་རྟེན་གྱུར་ཚེ་ནི་ཤིན་ཏུ་ཆུང་བའི་ཆ་ཡང་རྙེད་
