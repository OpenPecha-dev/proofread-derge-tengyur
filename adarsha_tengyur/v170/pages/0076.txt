སརྦཱ་ཏ་པ་ཏྲ་སྱ་མཱུརྡྷྣི་སརྦྦ་མ་ཧཱི་བྷྲྀ་ཏཾ། །ཡ་སྱེ་ཤྭ་ར་ཤི་ར་སྠཱ་ཡི་ཤུ་བྷྲཾ་གཾ་གཱ་ཛ་ལོཛྫྭ་ལཾ། །བྷྲ་མེ་ཏྱད་གུ་བྷྲཾ་མེ་ཏྱད་བྷྲཾ་ལོ་ཀེ་ཥུ་བྷཱུ་བ་ནཱ་བྷ་ར་ཎཾ་ཡ་ཤཿ། །ཀརྟུཿཀྲ་ཏུ་ས་ཧ་སྲཱ་ཎཱཾ་ས་ཧ་ཝཱ་ཀྵཱ་དྷི་ཀ་ཤྲི་ཡཿ། །ཡ་སྱ་ཥཥྛི་ས་ཧ་
གནས་པའི་སྤྱི་བོར་གྱུར། །ས་ཡི་བདག་པོ་ཀུན་གྱིའོ། །གང་གི་གྲགས་པ་འཇིག་རྟེན་རྒྱན། །དཀར་ཞིང་དབང་ཕྱུག་སྤྱི་བོར་གནས། །དྲི་མེད་གང་གཱའི་ཆུ་བོ་ནི། །རབ་མང་འཇིག་རྟེན་རྣམས་སུ་རྒྱུ། །མཆོད་སྦྱིན་སྟོང་གི་བྱེད་པ་པོ། །མིག་སྟོང་པ་ལས་ལྷག་པའི་དཔལ། །གང་གི་བཙུན་མོ་ལྷ་མཛེས་ནི། །
སྲཱ་ཎི་ཀ་ལ་ཏྲཾ་སུ་དྲྀ་ཤཱ་མ་བྷཱུཏ། །ཀ་དཱ་ཙི་ཏ་མུ་ནི་རཀྵ་བཻ་རཀྵི་ཀྵ་ཡ་ཀྲྀ་ཏ་ཀྵ་ཎཿ། །བི་ཙ་ཙཱ་རཱ་ཤྭ་མཱ་རུ་ཧྱ་ས་ཏ་པོ་བ་ན་བྷཱུ་མི་ཥུ། །ཏ་ཏྲ་རཱ་ཛཱཪྵི་བྷིཿགཻ་ཤྩིཏ་པུ་ཏྲེཥྚི་ཀ་ལ་ཤཾ་དྷྲྀ་ཏཾ། །དཱུ་རཱ་དྷྭ་ཤྲ་མ་སནྟ་པྟཿཔ་པ་ཡཿཔུརྞ་
སྟོང་ཕྲག་དྲུག་ཅུ་བྱུང་བར་གྱུར། །ནམ་ཞིག་དྲང་སྲོང་བསྲུང་བའི་ཕྱིར། །སྲིན་པོ་གཞོམ་པར་དམ་བཅས་ནས། །དཀའ་ཐུབ་ནགས་ཀྱི་ས་ལ་དེ། །རྟ་ལ་ཞོན་ནས་རྣམ་པར་རྒྱུས། །རྒྱལ་པོ་དྲང་སྲོང་འགའ་ཡིས་དེར། །བུ་ཡི་མཆོད་སྦྱིན་བུམ་པ་བཞག །ལམ་རིང་གིས་ངལ་དུབ་པ་དེས། །ཆུ་ཡིས་གང་བ་
མཱ་པ་པཽ། །བི་ཛ་ན་སཱ་དི་ཏཾ་པི་ཏྭཱ་ས་མནྟྲ་ཀ་ལ་ཤཱཏྤ་ཡཿ། །རཱ་ཛ་དྷཱ་ནཱིཾ་ས་མཱ་སཱ་དྱ་གརྦྷཾ་ལེ་བྷེ་བི་བྷཱུ་བྷུ་བཿ། །སྭ་པྣ་མཱ་ཡེནྡྲ་ཛཱ་ལཱ་དི་ཡ་སྱཿཀཽ་ཏུ་ཀ་བི་པྲུ་ཥཿ། །ཛ་ཡ་ཏྱད་བྷུ་ཏ་སཾ་བྷཱ་ར་བྷཱུ་མིཿསཱ་བྷ་བི་ཏ་བྱ་ཏཱ། །
འཐུངས་བར་གྱུར། །སྐྱེ་བོ་མེད་པར་རྙེད་པའི་ཆུ། །སྔགས་ཀྱི་བུམ་པ་ལས་འཐུངས་ནས། །རྒྱལ་པོའི་ཕོ་བྲང་འོངས་པ་ན། །ས་ཡི་བདག་པོས་མངལ་ཐོབ་གྱུར། །རྨི་ལམ་སྒྱུ་མ་མིག་འཕྲུལ་སོགས། །གང་གི་ལྟད་མོའི་ཟེར་མ་ནི། །རྨད་བྱུང་ཡང་དག་ཚོགས་པའི་ས། །བརྣག་པར་བྱ་སྟེ་རྒྱལ་གྱུར་ཅིག །
