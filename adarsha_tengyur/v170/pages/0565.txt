མ་ནི་བརྟྟ་ཏེ། །ཨ་དྦྱཱ་པི་ཏྲ་ནཾ་པཏྣཱི་དྷི་རྡྭེ་ཥ་བི་དྱཱཾ་ས་དཱ་ནི་ཤི། །ཨུནྣ་ཌཱ་ནཱཾ་སྭ་བཾ་ཤཱ་ནཱཾ་དྭཻ་དྷཱཾ་ཏཱ་བནྣ་ཛ་ཡ་ཏེ། །ཡཱ་བཏཀུ་ཋ་ར་དྷཱ་རེ་བ་ཡོ་ཥིདྦི་ཤ་ཏི་ནཱནྟ་རཾ། །བྷཱ་ཏ་རརྠཱ་ན་བཱ་དེ་ན་གུ་རུཾ་བ་རུ་ཥྱ་ཀུཏྶ་
རིག་བྱེད་བསླབས་པ་ཡི། །སྤུན་ཟླ་བྱེ་བའི་བརྒྱུད་དག་ནི། །ངེས་པ་ཉིད་དུ་ཇི་ལྟར་ཟློག །ཇི་སྟེ་སྟ་རེ་རྣོ་ལྟ་བུའི། །བུད་མེད་ནང་དུ་མ་ཞུགས་པ། །དེ་སྲིད་རང་རིགས་སྨིག་མ་ནི། །མཐོན་པོ་རྣམ་གཉིས་མི་འགྱུར་རོ། །ནོར་ལ་རྩོད་པ་སྤུན་ཟླ་དང་། །སྨོད་ཅིང་རྩུབ་པས་བླ་མ་
ཡཱཿ། །མི་ཏྲ་མེ་ཀཱ་བྷི་ལཱ་ཥེ་ཎ་ན་ཡནྟི་དྭེ་ཥ་ཏཱཾ་སྟྲི་ཡཿ། །ཏངྦ་དནྟི་ཧ་སནྟྱ་བི་བྷྲུ་བི་ལཱ་སེ་ན་ཡོ་ཥི་ཏཿ། །ཡཏ་པྲ་ཡཱ་ཏི་སུ་ཧྲྀ་ཏ་སྣེ་ཧ་མཱུ་ལོནྨཱུ་ལ་ན་ཧེ་ཏུ་ཏཱཾ། །ཧི་ཏ་མུཀྟྭེ་ཏི་པུ་ཏྲཱ་ཎཾ་བྷཱུ་ཏ་ཡེ྅བྷི་མ་ཏཾ་བྷ་
དང་། །གཅིག་ལ་འདོད་པས་གྲོགས་པོ་ལ། །བུད་མེད་ཀྱིས་ནི་སྡང་དུ་འཇུག །གྲོགས་ཀྱི་མཛའ་གཅུགས་རྩ་བ་ནས། །འབྱིན་པའི་རྒྱུར་ནི་རབ་འགྱུར་གང་། །དེ་ནི་བུད་མེད་སྨིན་མ་དག །གཡོ་ཞིང་དགོད་ཀྱང་སྨྲ་བར་བྱེད། །བུ་རྣམས་འབྱོར་པས་མངོན་འདོད་པས། །ཕན་པ་འདི་བརྗོད་འབྱོར་
བཿ། །ཨ་ནི་ཏྱ་ཏཱ་པ་རི་མུཀྟཿཀཱ་ལེ་ནི་དྷ་ན་མཱ་ཡ་ཡཽ། །ཨ་བི་བྷཀྟེ་དྷ་ནེ་ཤཀྟྭཱ་དེ་ཤཱནྟ་ར་དྷ་ནཱ་རྫ་ནེ། །ཛྱེཥྛཱ་བ་བྷཱུ་བུཿཔུརྞྞ་སྟུ་གྲྀ་ཧེ་བིཏྟ་མ་ཙིནྟ་ཡཏ། །ཀ་ལེ་ན་གྲྀ་ཧ་མཱབྷཱ་ནཾ་བསྟྲཱ་ཤ་ན་བི་བཱ་དི་
པ་ནི། །མི་རྟག་པ་ལས་ཡོངས་མ་གྲོལ། །དུས་ཀྱིས་ཕ་རོལ་སོང་བར་གྱུར། །ནོར་ནི་དགོས་པ་མིན་པའི་ཚེ། །ཕུ་བོ་རྣམས་ནི་ཡུལ་གཞན་དུ། །ནོར་ཉིད་བསྒྲུབ་ལ་ཆགས་གྱུར་ཅིང་། །གང་པོས་ཁྱིམ་དུ་ནོར་ལ་བསམས། །དུས་ཀྱི་ཁང་པར་འོངས་པར་གྱུར། །གོས་དང་ཟས་ལ་རྩོད་པ་
