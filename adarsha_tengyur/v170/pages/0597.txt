བརྟྟཱ་པྲ་ཎི་བནྡྷ་ནཾ། །བི་ཤཱ་ལཾ་ཛཱ་ལཾ་ས་ཧ་སཱ་སཾ་སཱ་ར་མི་བ་ཏ་ཏྱ་ཛུཿ། །མཏྶྱ་ཀུམྦྷཱི་ར་ན་ཀྲཱ་དི་སཾ་བྷཱ་རཾ་ཏངྒི་རཱཾ་བྷ་སི། །ཏྱཀྟྭཱ་ཏེ་བི་ར་ཏ་ཀོ་ས་བ་བྷཱུ་བུཿཀིལྦི་ཥ་དྭི་ཥཿ། །ཏཻརྣྱ་སྟཾ་བྷ་ག་བཱན་དྲྀཥྚྭཱ་མ་ཧཱ་མ་
པར་དུད་པ་ཡིས། །སྲོག་ཆགས་འཆིང་བ་ཡངས་པའི་རྒྱ། །འཁོར་བ་ལྟ་བུ་བཏང་བར་གྱུར། །དེ་ཡི་གསུང་གིས་ཆུ་ལ་ཉ། །ཆུ་སྲིན་ན་ཀྲ་ལ་སོགས་ཚོགས། །བཏང་ནས་དེ་དག་ཁྲོ་བྲལ་བས། །སྡིག་པ་ཡི་ནི་དགྲ་བོར་གྱུར། །དེ་དག་གིས་བཀོད་ཆུ་སྲིན་ཆེ། །མདུན་དུ་བཅོམ་
ཀ་ར་མ་གྲ་ཏཿ། །ཏཾ་ཛ་གཱ་ད་སྲྀ་ཛདྡནྟ་ཀཱནྟྱེ་བ་ཀ་རུ་ཎ་ན་དིཾ། །ཨ་པི་ཏྭཾ་ཀ་བི་ལཿཔུ་ཏྲ་ཀིནྣ་སྨ་ར་སི་དུཿཀྲྀ་ཏཾ། །པ་ཙོ་དུཤྩ་རི་ཏ་སྱཱ་ཡ་མ་པི་བཱ་ཀོ྅ནུ་བྷཱུ་ཡ་ཏེ། །སཱ་སྱ་ཀ་ལྱཱ་ཎ་མི་ཏྲནྟེ་ཛ་ན་ནེ་ཀྭཱ་དྱ་བརྟྟ་ཏེ། །སརྦྦ་
ལྡན་གྱིས་གཟིགས་ནས། །ཚེམས་ཀྱི་མཛེས་པ་སྙིང་རྗེ་ཡི། །ཆུ་བོ་སྤྲོ་བཞིན་དེ་ལ་སྨྲས། །བུ་ཀྱེ་སེར་སྐྱ་ཁྱོད་ཀྱིས་ནི། །ཉེས་པ་བྱས་པ་མི་དྲན་ནམ། །འདི་ཡང་ངག་གིས་ཉེས་སྤྱད་དག །སྨིན་པ་ཉམས་སུ་མྱོང་བ་ཡིན། །ཁྱོད་ཀྱི་མི་དགེའི་གྲོགས་གྱུར་པ། །མ་དེ་ད་ལྟ་གང་ན་གནས། །ཐམས་

ཅད་མཁྱེན་པས་འདི་བརྗོད་ཚེ། །དེ་ཡིས་སྐྱེ་བ་དྲན་ནས་སྨྲས། །ཁྱབ་བདག་བདག་ནི་སེར་སྐྱ་ཡིན། །རང་གིས་ཉེས་པ་བྱས་པ་དྲན། །ངག་གིས་ཉེས་བྱས་ཡོངས་སྨིན་པ། །འདི་ནི་ཉམས་སུ་མྱོང་བར་གྱུར། །དམྱལ་བ་སྟོན་བྱེད་བདག་གི་མ། །སྔོན་དུ་དམྱལ་བ་དག་ཏུ་སྐྱེས། །ཞེས་པ་སྨྲས་
