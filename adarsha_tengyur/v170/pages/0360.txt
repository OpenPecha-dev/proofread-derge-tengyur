ཡ་དི་བྷ་བི་ཥྱ་ཏི། །ཨི་དྱུཀྟྭ་ནྲྀ་པ་ཏིཿཀ་ནྱཱཾ་ས་རི་ཎཱི་ཡཱ་བྷ་བཏ་མུ་ཁཱི། །ཀྲྀཏཿཀརྨྨོ་རྨི་ནི་རྨཱ་ནེེཿཔྲ་བྷ་བནྟི་ན་བུདྡྷ་ཡཿ། །བྷོ་གི་ནསྟ་སྱ་ཀཱ་ལེ་ན་ཏ་སྱཱཾ་སུ་ནུ་ར་ཛ་ཡཏ། །ཛྱོཾ་ཏིཥྐ་ཙ་རི་ཏེ་ཡ་སྱ་བྲྀཏྟཾ་མུཀྟ་པི་ཏྲྀ་དྲུ་ཧཿ། །ཏ་པོ་
དབང་བསྐུར་བྱ། །དེ་སྐད་བརྗོད་ནས་མི་ཡི་བདག །མཛེས་མ་ལེན་ལ་དགའ་བར་གྱུར། །ལས་ཀྱི་རླབས་ཀྱིས་སྤྲུལ་པ་ཡིས། །ཕྲོགས་པའི་བློ་ལ་མཐུ་ཡོད་མིན། །དུས་ཀྱིས་ལོངས་སྤྱོད་ལྡན་པ་དེའི། །བུ་ནི་དེ་ལས་སྐྱེས་གྱུར་ཏེ། །གང་གི་སྤྱོད་ཚུལ་བརྩིས་པ་ཡིས། །ཕ་ནི་གསོད་པར་བརྗོད་གྱུར་པའོ། །དཀའ་
པ་ན་མྲྀ་ག་གྷཱ་ན་མྲྀ་ག་ཡ་བྱ་ས་ནེ་བ་ནེ། །ཨེ་བཾ་བི་དྷོ་ཧྱ་བྷཱུཏྟ་སྱ་མུ་ནི་ཤ་པཿསུ་ཏཱ་ཀྲྀ་ཏིཿ། །ཨ་ཏྲཱནྟ་རེ་མ་ཧཱནྣ་མ་བཻ་ཤ་ལི་ཀ་ག་ཎཱ་གྲ་ཎཱིཿ། །ཀ་ནྱ་མཱ་མྲ་བ་ནཱཏ་བྲཱ་པ་ཀ་ད་ལཱི་སྐནྡྷ་བི་ནིརྒ་ཏཱཾ། །སཱ་ཏ་སྱ་བྷ་བ་
ཐུབ་ནགས་ཀྱི་རི་དྭགས་རྣམས། །བསད་ཕྱིར་ནགས་སུ་རྔོན་ལ་ཞེན། །དེ་ལ་ཐུབ་པའི་དམོད་པ་ནི། །བུ་ཡི་དབྱིབས་ཅན་འདི་ལྟར་གྱུར། །སྐབས་དེར་ཆེན་པོ་ཞེས་བྱ་བ། །ཡངས་པ་ཅན་གྱི་ཚོགས་ཀྱི་གཙོས། །ཨ་མྲའི་ནགས་ལས་ཆུ་ཤིང་གི། །སྡོང་པོ་ལས་འཁྲུངས་མཛེས་མ་ཐོབ། །དེ་ཡི་ཁྱིམ་དུ་
ནེཀཱནྟཱ་བརྡྷ་མཱ་ནཱ་ཤ་ནཻཿཤ་ནཻཿ། །བི་ད་དྷེ་བི་པུ་ལཱཾ་བྲཱི་ཏིཾ་ད་ན་ཙིནྟཾ་ཙ་ཙེ་ཏ་སི། །པྲ་ཎ་ཡཱ་དཱ་མྲ་པཱ་ལཱི་ཏི་བནྡྷུ་བྷིཿསཱ་ཀྲྀ་ཏཱ་བི་དྷཱ། །སུ་ཎུ་ཧཱི་ན་མི་བ་ཏྱཀྟཱ་བ་ལྱ་ཡཽ་བ་ན་མ་ད་དེ། །ཏདྦི་བ་ཧོདྻ་ཏ་སྱཱ་ཐ་
མཛེས་མ་དེ། །དལ་བུ་དལ་བུས་བསྐྱེད་བསྲིངས་ཏེ། །དགའ་བ་ཆེ་དང་སྦྱིན་པ་ཡི། །བསམ་པ་ཡིས་ཀྱང་སེམས་མ་བཟུང་། །གཉན་གྱི་མཛའ་ལས་དེ་ཡི་མིང་། །ཨ་མྲ་སྐྱོང་མ་ཞེས་སུ་བཏགས། །བུ་མོ་དམན་པ་བཞིན་བཏང་ནས། །ལང་ཚོ་དག་ནི་བཟུང་བར་གྱུར། །དེ་ནས་དེ་གཏོང་བརྩོན་པའི་ཕ། །
