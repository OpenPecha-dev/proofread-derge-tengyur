སྱཱ་བ་ཏཏ་པྲྀཥྛེ་ཤུ་ནཱཾ་སཾ་གྷོ་ཏི་བྷཱི་ཥ་ཎཾཿ། །ཏྲཻ་ལོ་ཀྱ་ཤཱ་བ་པཱ་པོཏྠཿཀླེ་ཤ་རཱ་ཤི་རི་པཱ་ཁེ་ལཿ། །ས་ཏཻརྒྲཱི་བཱ་མུ་ཁཱ་རབྡྷ་མཱཾ་ས་གྲཱ་སོ་གྲ་ཀཪྵ་ཎེཿ། །ཨཱ་ཀྲནྡི་རུ་དྷི་རཀྵི་བཻ་བྷ་ཀྵྱཱ་མཱ་ཎཿཀྵ་ཡཾ་ཡ་ཡཽ། །དི་
རྒྱབ་ཏུ་ནི། །ཤིན་ཏུ་འཇིགས་རུང་ཁྱི་ཚོགས་བྱུང་། །འགྲོ་གསུམ་དམོད་སྡིག་གིས་བསླང་བའི། །ཉོན་མོངས་ཕུང་པོ་མ་ལུས་བཞིན། །ཁ་དང་མགྲིན་པ་ནས་བརྩམས་ཤ། །ཟ་ཞིང་དྲག་སྤྱོད་ཁྲག་གིས་མྱོས། །དེ་དག་རྣམས་ནི་ཆོ་ངེ་ལྡན། །དེ་ནི་ཟོས་ཤིང་ཟད་པར་གྱུར། །ཉིན་མོའི་མཐར་
ནཱནྟེ་པུ་ན་རཱ་ཡཱནྟཾ་ཏདྦི་མཱན་མ་དྲྀ་ཤྱ་ཏ། །ཙ་ཏ་སྲོ྅སྤ་ར་སནྟོཤྩ་པུ་རུ་ཥཿས་ཙ་ཀཱནྣི་མཱན། །ཏཾ་ཤྲོ་ཎ་ཀོ་ཊཱི་ཀརྞོ་ཐ་པ་པྲཙྪ་ཏྱཾ་ཏ་པི་སྨི་ཏཿ། །ས་ཁེ་ཀི་མེ་ཏ་རཱཤྩ་རྱཾ་དྲི་ཤྱ་ཏེ་ཀ་ཐྱ་ཏཱ་མི་ཏི། །ས་ཏེ་ན་པྲྀཥྚཿཔྲོ་བཱ་ཙ་ཙ་ཡ་སྱ་
ནི་སླད་ནས་ཡང་། །གཞལ་ཡས་ཁང་དེ་བྱུང་བ་མཐོང་། །ལྷ་མོ་བཞི་པོ་དེ་དག་དང་། །མཛེས་ལྡན་སྐྱེས་བུ་དེ་ཡང་ངོ་། །དེ་ནས་གྲོ་བཞིན་བྱེ་བ་རྣ། །ཤིན་ཏུ་ཡ་མཚན་ནས་དེ་ལ། །གྲོགས་པོ་ངོ་མཚར་འདི་དག་ནི། །མཐོང་བ་ཅི་ཡིན་སྦྲོས་ཞེས་དྲིས། །དེས་དྲིས་དེ་ཡིས་རབ་སྨྲས་པ། །གྲོགས་

པོ་འདི་ནི་མཉན་པར་གྱིས། །ཁྱེད་ནི་གྲོ་བཞིན་བྱེ་བ་རྣ། །ངས་ཤེས་ལེགས་བྱས་ལྡན་པ་ཡིན། །ནོར་ལས་བྱུང་བའི་གྲོང་དུ་བདག །སྡིག་ཅན་ཕྱུགས་དག་སྐྱོང་བར་གྱུར། །རྟག་ཏུ་བདག་གིས་ཕྱུགས་རྣམས་ཀྱི། །ཤ་ལ་རབ་ཏུ་བཅད་ནས་བཙོངས། །ཀཱཏྱཱ་ཡ་ན་ཞེས་འཕགས་པ། །སྙིང་རྗེའི་
