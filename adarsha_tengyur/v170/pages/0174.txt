བྱ་སྭ་བྷ་བོདྡྷ་བ་མཱུརྑཱ་ཎཱཾ་ཀུ་ཤ་ལཾ་པྲ་ཡཱ་ཏྱ་ཧི་ཏ་ཏཱ་མི་ཏྱེ་ཥ་ལ་ཀྵྨཿཀྲ་མཿ། །ནཱི་ཤཱི་གྷནྟི་མི་རཱནྡྷྱ་མཽ་ཥ་དྷི་བ་ན་སྱཱ་ཏྱནྟ་ཀཱནྟི་པྲ་ད་མེ་ད་ཅྩཽ་ལུ་ཀ་ཀཱུ་ལ་དྲྀཥྚ་ཧ་ཏ་ཡེ་སརྦ་ཏྲ་མེ་ཏྲམྨ་ཧཿ། །པུ་ར་
རང་བཞིན་ལས་བྱུང་དགེ་ཉིད་གྱུར། །རྨོངས་པ་རྣམས་ལ་དགེ་བ་མི་ཕན་ཉིད་འགྱུར་འདི་ནི་རིམ་པ་འདི་ཡིས་མཚོན། །མཚན་ཕྱེད་སྨག་རུམ་མུན་པ་སྨན་གྱི་ནགས་ལ་མཆོག་ཏུ་མཛེས་པ་རབ་ཏུ་སྟེར། །གང་ཡང་ཉི་མའི་གཟི་བྱིན་འུག་པའི་རིགས་ཀྱི་མིག་ནི་ཀུན་ཏུ་ཉམས་པར་བྱེད། །ས་
རཱ་ཛ་གྲྀ་ཧཱ་བྷི་ཁྱེ་བིམྦི་སཱ་ར་སྱ་བྷཱུ་པ་ཏེཿ། །ཨ་བྷཱུཏ་པཽ་ར་སུ་བྷ་དྲཱ་ཁྱཱཿཔ་རི་པཱུརྞ་གྲྀ་ཧ་སྠི་ཏེཿ། །མཽརྑྱཱཏ་མོ་ཧ་པྲ་པལླ་སྱ་སརྦ་དཪྴ་ན་པི་དྭི་ཥཿ། །ཏ་སྱ་ཀྵ་པ་ཎ་ཀེ་ཥྭེ་བ་བ་བྷཱུ་བཱ་བྷྱཱ་བྷི་ཀ་ད་རཿ། །ཏ་སྱ་
བདག་གཟུགས་ཅན་སྙིང་པོ་ཡི། །གྲོང་ཁྱེར་རྒྱལ་པོའི་ཁབ་ཅེས་པ། །རབ་བཟང་ཞེས་པའི་གྲོང་ཁྱེར་བ། །ཡོངས་རྫོགས་ཁྱིམ་ན་གནས་པ་བྱུང་། །ལྟ་བ་ཀུན་ལ་རྣམ་སྡང་ཞིང་། །གཏི་མུག་རབ་ཏུ་རྒྱས་པ་དེས། །རྨོངས་ལས་ཟད་བྱེད་པ་ཉིད་ལ། །མངོན་པར་གུས་པ་ལྷག་པར་གྱུར། །དེ་ཡི་ཆུང་
པ་ཏྱ་བ་ཏྭི་ནཱ་མ་ཛཱ་ཡཱ་ཏི་ཛ་ན་ཤཱ་ལི་ནཱི། །གརྦྷེ་མཱ་དྷ་ཏུ་པཱུརྞེནྡུ་བིམྤཾ་པཽ་རཱནྡ་རི་བ་དི་ཀ། །ཀ་ལནྡ་ཀ་ནི་བཱ་པཱ་ཁྱཱ་བེ་ཎུ་ཀཱ་ན་ན་སཾ་ཤྲ་ཡཿ། །ཀ་དཱ་ཙི་ཏ་བྷ་ག་བཱན་བུདྡྷ་པྲཱཔྟཿཔིཎྜཱ་ཡ་ཏད་གྲྀ་ཧཾ། །པཱུ་ཛཱཾ་ས་
མ་ཆོ་རིགས་ཅན། །བདེན་ལྡན་མ་ཞེས་བྱ་བ་ཡིས། །མངལ་བཟུང་ཟླ་བ་ཉ་བ་ཡི། །དཀྱིལ་འཁོར་ཤར་གྱི་ཕྱོགས་ཀྱི་བཞིན། །ཀ་ལན་ད་ཀའི་གནས་ཞེས་པ། །འོད་མའི་ཚལ་དུ་ཡང་དག་གནས། །སངས་རྒྱས་བཅོམ་ལྡན་བསོད་སྙོམས་ལ། །ནམ་ཞིག་དེ་ཡི་ཁང་པར་གཤེགས། །དེ་ནི་ཆུང་མར་
