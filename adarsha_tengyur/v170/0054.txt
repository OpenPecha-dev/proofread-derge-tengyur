མེ་ཀེ་ནཱ་ཧྣཱ་བྲ་ཡཱ་ཏི་ཡཿ། །ཏ་སྨཻ་ས་མརྦི་ཏི་དྲྀཥྚཱ་རཱཛྙ་བྷ་དྲ་གི་རིཾ་ག་ཛཾ། །ཨ་བྷཱུད་དུཿཔྲ་ས་ཧོ་རཱ་ཛཱ་ཏ་ཏ་སྠི་ཧཱ་ཀྲྀཥྚ་མཱ་ན་སཿ། །བྲ་ཡཱ་ཏེཥྭ་ཐ་བྷཱུ་པེ་ཥུ་བིསྨི་ཏེ་ཁུ་མ་ཁ་ཤྲཱི་ཡཱ། །ས་མརྦི་ཏི་ཡཛྙ་ཕ་ལེ་
འདོན། །ཚངས་པའི་ཤིང་རྟ་ཞེས་པར་བྱིན། །རྒྱལ་པོའི་གླང་ཆེན་བཟང་པོའི་རི། །དེ་ལ་ཡང་དག་གཏད་མཐོང་ནས། །རྒྱལ་པོ་ཤིན་ཏུ་བཟོད་དཀའ་དེར། །ཆགས་པས་དྲངས་པའི་ཡིད་དུ་གྱུར། །དེ་ནས་ས་བདག་མཆོད་སྦྱིན་གྱི། །དཔལ་གྱིས་ཡ་མཚན་གཤེགས་གྱུར་ཅིང་། །ས་ཡི་སྤྱོད་པས་མཆོད་སྦྱིན་འབྲས། །
བྷཱརྒ་བཱ་ཡ་མ་ཧཱི་བཱུ་ཛཱ། །ཏ་མུ་བཱ་ཙ་ས་མ་བྷྱེ་ཏྱ་སྭ་སྟི་བཱ་དབུརཿཔ་རཾ། །མ་རཱི་ཙི་ཤི་ཥྱོ་བཱ་ཧཱི་ཀཿཔྲཱཔྟ་པཱུ་ཛཱ་ས་ནོ་མུ་ནིཿ། །རཱ་ཛནྣ་དྷྱ་ཡ་ན་སྠནྟེ་གུ་རུ རྨེ་གུ་རུ་དཀྵི་ཎཱཾ། །ཨཱི་ཧ་ཏེ་པ་རི་ཙཪྻཱ་རྠཱི་སཱ་མཱ་ནྱ་ཛ་ན་དུརླ་བྷཱ། །ཨེ་ཀསྟྭ་
བྡྲྀའི་བུ་ལ་གཏད་པའི་ཚེ། །མ་རཱི་ཙི་སློབ་ཐུབ་པ་ནི། །མཆོད་དང་སྟན་ཐོབ་བཱ་ཧཱི་ཀ། །ཡང་དག་འོངས་ནས་ཤིས་པའི་ཚིག །མངོན་དུ་བཏང་ནས་དེ་ལ་སྨྲས། །རྒྱལ་པོ་སློབ་གཉེར་གང་གི་མཐར། །བདག་གི་བླ་མ་བླ་མའི་ཡོན། །གཡོག་འཁོར་དོན་དུ་སྐྱེ་བོ་ནི། །ཐུན་མོང་བ་ལས་རྙེད་དཀའ་བཞེད། །སྒྲུབ་བྱེད་ཁྱོད་
མེ་བ་བི་དྷི་ནཱ་ནི་རྨི་ཏོ་དུཪྻ་བྷ་པྲ་དཿ། །བ་ཧ་བོ་ཛཱ་ཏུ་ཛཱ་ཡནྟེ་ན་ལོ་ཀེ་ཀལྤ་པཱ་ད་པཱ། །དེ་བི་པདྨ་བ་ཏི་པུ་ཏྲ་ས་ཧི་ཏ་གུ་ར་བེ་མ་མ། །ཏ་པཿཀྲྀ་ཤ་ཡ་བྲྀདྡྷ་ཡ་དཱི་ཡ་ཏཱཾ་པ་རི་ཅཱ་རི་ཀཱཾ། །ཨི་ཏྱུཀྟེ་མུ་ནི་ནཱ་རཱ་ཛཱ་
ཉིད་གཅིག་པུ་ཡིས། །རྙེད་དཀའ་རབ་ཏུ་སྦྱིན་པ་སྤྲུལ། །དཔག་བསམ་རྐང་འཐུང་འཇིག་རྟེན་ན། །ནམ་ཡང་མང་པོ་སྐྱེ་མི་འགྱུར། །བདག་གི་བླ་མ་དཀའ་ཐུབ་ཀྱིས། །ཉམ་ཆུང་རྒས་པའི་གཡོག་འཁོར་དུ། །པདྨ་ལྡན་པའི་ལྷ་མོ་ནི། །བུ་དང་བཅས་པ་གནང་བར་མཛོད། །ཐུབ་པས་དེ་བརྗོད་རྒྱལ་པོ་ནི། །
