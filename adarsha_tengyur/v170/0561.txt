ཡཿས་ཨེ་བོ་ད་ཡ་ནོ་ནྲྀ་པཿ། །སཾ་ཛྙཱ་ནི་དེ་བ་ཀོ་ཡཤྩ་ཀུ་ཀུ་རཿསུ་དྷ་ནོ་པི་སཿ། །རཱ་ཛྙཱ་གྷོ་ཥེ་ཎ་བི་ཛྙཱ་ཏོ་གྷོ་ཥི་ལ་པ་ར་ནཱ་མཱ་བྷཱུ་ཏ། །ཙ་རི་ཏ་མི་ཏྱུ་ཙི་ཏཾ་བྷ་བ་བྷེ་དི་ནཱ་བྷ་ག་
གང་། །དེ་ཉིད་མི་བདག་འཆར་བྱེད་ཡིན། །གང་ཞིག་བརྡ་ནི་རིག་བྱེད་པའི། །ཁྱི་དེ་ཡང་ནི་ནོར་བཟང་ངོ་། །རྒྱལ་པོས་གདངས་ལས་ཤེས་པས་ན། །གདངས་ཅན་ཞེས་པ་མིང་གཞན་གྱུར། །དེ་ལྟར་ཡིད་ནི་ལེགས་པ་རྣམས་ཀྱི་སྤྱོད་པ་དག །རིགས་
པ་ཏཱ་ཀ་ཐི་ཏཾ་ཀི་ལ་བྷིཀྵ་བཿ། །སུ་ཀྲྀ་ཏ་སཽ་ར་བྷ་སཱ་ར་སུ་དྷཱ་ར་སཾ་སུ་མ་ན་སཱ་ཤྲ་བ་ཎཉྫ་ལི་བྷིཿཔ་པུཿ། །ཨི་ཏི་ཀྵེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཱཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་
པར་སྲིད་པ་འབིགས་བྱེད་བཅོམ་ལྡན་གྱིས་གསུངས་པ། །ལེགས་བྱས་དྲི་བཟང་སྙིང་པོར་ལྡན་པ་བདུད་རྩིའི་རོ། །དགེ་སློང་རྣམས་ཀྱིས་གསན་པའི་སྙིམ་པས་འཐུངས་པར་གྲགས། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་
ལ་ཏཱ་ཡཱཾ་གྷོ་ཥི་ལཱ་བ་དཱ་ནཱཾ་ཥ་ཊཱིཾ་ཤཿཔལླ་ལྦ།།      །།བི་བུདྡྷ་ས་ར་སི་པདྨ་ཤོ་བྷ་ཏེ་སདྐ་ཛནྨཱཿཤུ་ཙི་པ་རི་ས་ར་ཛཱ་ཏཱཾ་སྤྲྀ་ཤྱ་ཏེ་ན་སྠ་ལབྫཾ། །ས་ཧ་ཛ་པ་རི་ཙི་ཏཱ་ནཱཾ་ནི་ཏྱ་མནྟ་རྒ་
གྱི་འཁྲི་ཤིང་ལས་གདངས་ཅན་གྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་འདབ་སྟེ་སུམ་ཅུ་རྩ་དྲུག་པའོ།།      །།ལྷ་ཡི་རྫིང་བུ་དག་ཏུ་འདམ་ལས་སྐྱེས་པའི་ཆུ་སྐྱེས་མཛེས། །ཡོངས་བསྐམས་གཙང་སར་སྐྱེས་པའི་པདྨ་དག་ལ་རེག་མི་བྱེད། །ལྷན་སྐྱེས་ཡོངས་སུ་འདྲིས་ཤིང་རྟག་ཏུ་ནང་ན་གནས་
