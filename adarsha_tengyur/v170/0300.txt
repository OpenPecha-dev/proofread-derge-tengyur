སྟཱཾ་བྷི་ར་ཏྱནྣ་རཱ་གི་ཎཿ། །ཀཎྛེ་ཀྲྀ་ཏཱ་བྷི་ནྲྀ་ཏྱནྟི་མཱ་ལཱ་བྷི་རི་བ་མོ་ཧི་ཏཱཿ། །ཏ་ཏྲཻ་ཀཱནྟ་ལ་ཏཱ་ཀུཉྫ་ནི་ཀུཉྫ་ན་ཧི་ཏ་སྠི་ཏིཿ། །ཨནྡྷཿསཽ་ར་བྷ་མཱ་གྷྲཱ་ཡ་སོ་ཤུམྦཱ་ཏི་མི་ར་པྲ་ཛཿ། །ས་ཧ་
བུད་མེད་ཀྱི་ནི་འཕྲེང་བ་བཞིན། །མགྲིན་པར་བྱས་པས་རྨོངས་བྱས་པའི། །ཤིན་ཏུ་ཆགས་ལྡན་བྲོ་གར་བྱེད། །དེར་ནི་དབེན་པར་འཁྲི་ཤིང་གི། །ཚོགས་ཀྱིས་བསྒྲིབས་པར་གནས་བྱས་པ། །ལོང་བས་སོ་སུམ་བཱ་ཡི་ནི། །དྲི་སར་འཕྲེང་བའི་དྲི་བཟང་ཚོར། །འཕྲལ་
སཻ་བ་བི་ཀཱ་རེ་ཎ་རཱ་གཱ་ཏ་བི་སྨྲྀ་ཏ་སཾ་བྲྀ་ཏིཿ། །ཨ་གཱ་ཡནྨ་ད་ན་ཀྵཱི་བཱ་ག་ཎ་ཡན་ཏི་བྷ་ཡཾ་ཀུ་ཏཿ། །ཏནུ་པ་བཏ་བི་ལཱ་སཻཿཀཱིཪྻ་མཱ་ཎ་པྲི་ཡཱ་སཱཿས་མ་ད་བ་ད་ན་པདྨཱ་མོ་ད་པཾ་བཱ་ད་སཱ་
ལ་ཉིད་དུ་ཆགས་པ་ཡི། །རྣམ་འགྱུར་ལས་ནི་སྦས་པ་བརྗེད། །གླུ་བླངས་འདོད་པས་མྱོས་པ་རྣམས། །འཇིགས་པ་དག་གིས་ག་ལ་རྩི། །དམན་པའི་རླུང་གི་རོལ་པས་བསྐྱོད་པར་གྱུར་པའི་དགའ་མ་ཡིས། །བཞིན་ནི་མྱོས་ལྡན་པདྨའི་དྲི་བཟང་ཚོགས་པའི་སྙིང་པོ་
