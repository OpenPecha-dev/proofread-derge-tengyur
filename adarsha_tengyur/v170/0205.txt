ཥསྟདྱ་སྱ་ཀྵ་ཡ་མ་ཏི་ནིསྦ་ལ་ཏ་ཡཱ་ཤོ་ཙྱཱ་ཡ་ཏ སྨཻ་ན་མཿ། །དཱ་མཱ་བརྟེ་བི་ཥ་ཡ་ཛ་ལ་དྷོ་པུརྞ་ལཱ་བ་ཎྱ་ས་རེ་དརྤྱོཏྤ་པྱདྦི་ཥ་མ་མ་ཀ་རཾ་ཀོདྡྷ་བ་ཀྵོ་བ་བྷྱ་མཱ་ནེ། །ནི་ཏྱཱ་སནྣ་པྲ་ཙ་ལ་བི་ར་ཧ་
དཀའ་སྟེ། །འབྲས་མེད་ཉིད་ཀྱིས་གང་གིས་ཟད་འགྱུར་མྱ་ངན་འོས་པ་དེ་ལ་ཕྱག་འཚལ་ལོ། །ཡུལ་གྱི་ཆུ་གཏེར་གཡོན་མའི་ཀློང་འཁྱིལ་འཇོ་སྒེག་ཆུ་ཡིས་གང་གྱུར་པ། །དྲེགས་པ་མི་བཟད་ཆུ་སྲིན་རབ་ལངས་མཚན་མ་འཁྲུགས་པ་རབ་བསྐྱེད་ཅིང་། །ལྷག་པར་འབྲལ་བའི་རྟ་གདོང་
བྲལློ་པངྦཱ་ཌ་བགྣོོ་པུཾ་སཱཾ་སེ་ཏུསྟ་ར་ཎ་ས་ར་ཎཻ་ཏཱི་བྲ་བཻ་རཱ་གྱ་མེ་བ། །ཏསྨཏ་གྲི་ཧཱ་ཎ་པྲ་བྲ་ཛྱཾ་རཱ་ཛ་བུ་ཏྲ་ཛི་ཏེནྡྲི་ཡཿ། །ཨེ་ཏཱཿསྟྲི་ཡ་ཨི་བ་ཀྵི་པྲཾ་ས་མཱ་ག་སུ་ཁཱཿཤྲི་ཡཿ། །ཀྲྀ་ཡ་ཏཱཾ་ཀུ་ཤ་ལཱ་ཡེེ་བ་བྲཧྨ་
མེ་ནི་རབ་འབར་རྟག་ཏུ་གནས་པ་ལ། །སྐྱེས་བུ་རྣམས་ནི་སྒྲོལ་ལམ་ཆགས་བྲལ་དྲག་པོ་ཁོ་ན་བསྡམས་པའི་སྟེགས། །དེ་སླད་རབ་བྱུང་གཟུང་བར་མཛོད། །རྒྱལ་པོའི་སྲས་ནི་དབང་པོ་ཐུལ། །བུད་མེད་བདེ་དང་དཔལ་འབྱོར་འདི། །མྱུར་དུ་ཡང་དག་འགྲོགས་པ་བཞིན། །དགེ་བ་ཉིད་སླད་ཚངས་པ་ཡི། །སྤྱོད་པ་
ཙཪྻ་པ་རི་གྲ་ཧཿ། །ཏྱ་ཛ་ཏཱ་མེ་ཥ་ནིཿསཱ་ར་གྲྀ་ཧ་སཾ་སཱ་ར་དུརྒྲ་ཧཿ། །ཨི་ཏི་ཤྲུ་ཏྭ་བྷ་ག་བ་ཏཿཀ་རུ་ཎཱ་བྷ་ར་ཎཾ་བ་ཙཿ། །པྲ་ཏྱ་བྷཱ་ཥ་ཏ་ཏནྣནྡཿཔཱུ་ཛྱ་པྲ་ཎ་ཡ་ཡནྟྲི་ཏཿ། །ས་དཱ་བྷ་བ་དུ་པཱ་ཡེེ་བ་པྲ་བྲ་ཛྱ་བྷ་ག་
ཡོངས་སུ་བཟུང་བར་མཛོད། །འཁོར་བ་སྙིང་པོ་མེད་པའི་ཁྱིམ། །གདོན་ངན་འདི་ནི་གཏང་བར་མཛོད། །བཅོམ་ལྡན་གསུང་ནི་སྙིང་རྗེ་ཡིས། །བརྒྱན་པ་འདི་སྐད་ཐོས་གྱུར་ནས། །དོན་གཉེར་མཆོད་འོས་ཀྱིས་གདམས་པའི། །དགའ་བོས་དེ་ལ་ལན་གསོལ་པ། །རྟག་ཏུ་ཁྱོད་ཉིད་བསྙེན་པ་ཉིད། །བཅོམ་ལྡན་བདག་
