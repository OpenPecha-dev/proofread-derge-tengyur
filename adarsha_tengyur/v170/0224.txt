ཙ་ར་ཏྱེ་བ་སྭརྒ་སྟྲཱི་པ་ག་མཱ་ཤ་ཡཱ། །མི་ཐྱཱ་པྲ་ཏཱ་ནཱཾ་ལུབྡྷཱ་ནཱཾ་རཱ་ག་དྭེ་ཤ་ཀ་ཥཱ་ཡི་ནཱཾ། །ཨེ་ཏོ་སུ་ནི་ཏྱ་ཏཔྟཱ་སུ་ཀུམྦྷཱི་ཥྭེ་བཱ་ཀྵ་ཡཿཀྵ་ཡཿ། །ཨི་ཏི་ནནྡཿས་མཱཀྞྱ་ཛཱ་ཏ་རོ་མཱཾ་ཙ་ཀཾ་ཙུ་ཀཿ། །ཏ་ཏྲ་ཙྱུ་ཏཱ་མི་
མེད་འགྲོགས་བསམ་པས། །ཚངས་པར་སྤྱོད་པ་སྤྱོད་པ་ཉིད། །བརྐམ་ཆགས་ལོག་པའི་བརྟུལ་ཞུགས་ཅན། །ཆགས་སྡང་སྙིགས་མ་ཅན་རྣམས་ཀྱི། །རྟག་ཏུ་ཚ་བའི་ཟངས་ཆེན་ནི། །འདི་དག་འཇིག་པ་མེད་པའི་གནས། །དེ་ལྟར་དགའ་བོས་ཡང་དག་ཐོས། །སྤུ་ལོང་གོས་ལྟར་རྒྱས་གྱུར་ཅིང་། །དེར་ནི་རྗེས་སུ་
བ་ཏ་ནུཾ་པཤྩཱཏྟཱ་པཱ་ད་མ་ནྱཱ་ཏ། །ས་མ་བྷྱེ་ཏྱ་ཏྱ ཀྟ་རཱ་ག་སཾ་བཱ་ས་བཱ་ས་ནཿསྭ་ཡཾ། །བ་བྷཱུ་བཱ་ནུཏྟ་ར་བྲཧྨ་ཙཪྻ་བཪྻཔྟ་སཾ་ཡ་མཿ། །གྷ་ན་མོ་ཧི་ཀྵ་ཡཱཏྟ་སྱ་བི་མུཀྟེ་སཾ་ཤ་མནྟ་ཏཿ། །མ་ན་པྲ་སཱ་ད་མ་པེ་
གདུང་བ་ཡིས། །ལུས་པོ་དུམ་བུར་གྱུར་བཞིན་སེམས། །གནས་སུ་མངོན་སོང་འདོད་ཆགས་ཀྱིས། །བག་ཆགས་ཞེན་པ་བཏང་བྱས་ནས། །བླ་མེད་ཚངས་པར་སྤྱོད་པ་ནི། །ཡོངས་སུ་རྫོགས་ཤིང་དུལ་བར་གྱུར། །ནང་དུ་ཐེ་ཚོམ་རྣམ་གྲོལ་ཅིང་། །སྟུག་པོའི་རྨོངས་པ་ཟད་ལས་དེའི། །ཡིད་ནི་རབ་དང་ཐོབ་གྱུར་ཏེ། །
དེ་ཤ་ར་དཱི་བ་ད་དཱི་པ་ཡཿ། །ནིཥྐཱ་མཿཔྲ་ཤ་མ་བྲཱབྷྱ་པ་རཾ་ནིཥྛཱ་མུ་པཱ་ག་ཏཿ། །ཤུདྡྷ་དྷཱིཿས་མ་བྷྱེ་ཏྱ་བྷ་ག་བནྟ་མ་བྷཱ་ཥ་ཏ། །ནཱབྶ་རོ་ཏིརྣ་པུནྡཪྻོ་བྷཱ་ག་བན་ཀྲྀ་ཏྱ་མ་སྟི་མེ། །ཨེ་ཏཱཿཔཪྻནྟ་བིཙྪཱ་ཡཿཔཱ་པ་ཐཱ་བི་ཥ་ཡ་ཤྲི་
སྟོན་ཀའི་ཀླུང་གི་ཆུ་བཞིན་ནོ། །འདོད་པ་མེད་ཅིང་རབ་ཞི་ཐོབ། །མཆོག་ཏུ་ངེས་པར་དེར་ཞུགས་པ། །དག་པའི་བློ་ཡིས་བཅོམ་ལྡན་ལ། །ཡང་དག་མངོན་དུ་ཕྱོགས་ཉེ་སྨྲས། །བཅོམ་ལྡན་བདག་ལ་བྱ་བ་ནི། །ལྷ་མོས་ཡོད་མིན་མཛེས་མས་མིན། །གནོད་པ་དང་བཅས་ཡུལ་གྱི་དཔལ། །འདི་དག་མཇུག་ནི་མཛེས་དང་
