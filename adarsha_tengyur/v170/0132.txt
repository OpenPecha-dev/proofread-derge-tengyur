ཥ་སཱ་ད་སཿ། །ཨ་ཏྲཱནྟ་རེ་དི་ན་སྱཱནྟེ་བྱོ་མཱ་ནནྟ་པ་ཐཱ་དྷྭ་གཿ། །ཨ་བཱ་བཱསྠཱ་ཙ་ལོ་བཱནྟཾ་བ་རི་ཤྲཱནྟ་ཨི་བཱཾ་ཤུ་མཱན། །ཨསྟཾ་ག་ཏེ་ས་ཧ་སྲཱཾ་ཤོ་ར་ཛ་ནཱི་ར་མ་ཎི་ཤ་ནཻཿ། །ཏཱ་རཱ་པ་ཏི་མི་བཱ་ནེཥྚཾ་པྲ་ས་ས་རྦྱ་བྷི་
མཐའ་ན་གནས། །སྐབས་འདིར་ཉིན་མོའི་མཐར་གྱུར་ཏེ། །མཐའ་མེད་མཁའ་ཡི་ལམ་བགྲོད་པས། །ཚ་ཟེར་ཅན་ནི་ཡོངས་དལ་བཞིན། །ནུབ་ཀྱི་རི་དང་ཉེ་བར་སླེབ། །འོད་སྟོང་ནུབ་པར་གྱུར་པའི་ཚེ། །མངོན་པར་འགྲོ་བ་མུན་ཅན་མ། །རྒྱུ་སྐར་བདག་པོ་འདོད་གྱུར་ནས། །དལ་བུས་རབ་ཏུ་འགྲོ་
སཱ་རི་ཀཱ། །ཨ་ཐ་པྲ་ཀཱ་ཤ་བི་བྷ་བཻཿསརྦྦ་ཤཱ་པཱུ་ར་ཎོནྨུ་ཁཿ། །བོ་དྷི་སཏྭ་ཨི་བ་སྭཙྪཿསུ་དྷཱ་དཱི་དྷི་ཏི་ཏུཪྻ་ཡཽ། །སྭཱི་ཏཱ་ཏ་མཿས་མཱུ་ཧ་སྱ་ནིཿཤེ་ཥ་པྲ་པ་མོ་ཙི་ཏ། །མཱ་ན་སོལླཱ་སི་ནཱི་ཛྱོདྶྣཱ་སཏྭ་བྲྀཏྟི་
བ་བཞིན། །དེ་ནས་རབ་གསལ་འབྱོར་པ་ཡིས། །ཕྱོགས་ཀུན་ཡོངས་རྫོགས་གྱེན་དུ་ཕྱོགས། །དྲི་མེད་བྱང་ཆུབ་སེམས་དཔའ་བཞིན། །བདུད་རྩིའི་འོད་དེ་ཤར་བར་གྱུར། །མུན་པའི་ཚོགས་ནི་མ་ལུས་པ། །རབ་ཏུ་ཞི་བར་ནུས་ཤིང་རྒྱས། །ཡིད་ནི་དངས་བྱེད་ཟླ་བའི་འོད། །སྙིང་སྟོབས་སྤྱོད་
རི་བཱ་བ་བྷཽ། །ཏ་མོ་མོ་ཧཾ་ཛ་ཧ་རེནྡུ་རྡི་ཤཱཾ་དི་ན་བི་ཡོ་ག་ཛཾ། །པ་རོ་པ་ཀཱ་རོ་ཧི་དུ་རཱ་རོ་ཧོ་མ་ཧཱཏྨ་ནཱཾ། །སུ་པྲི་ཡ་ཤྩནྡྲ་ཀི་ར་ཎཻཿདཱུཪྻ་མཱ་ཎ་ཏ་ནུཿཀྵ་ཎཾ། །ནི་དྲཱཾ་ཀཱརྱ་ས་མཱུ་
ཚུལ་བཞིན་དུ་མཛེས། །ཉིན་བྲལ་ལས་སྐྱེས་ཕྱོགས་རྣམས་ཀྱིས། །མུན་པའི་རྨོངས་པ་ཟླ་བས་ཕྲོགས། །གང་ཕྱིར་རིང་ན་བཞུགས་པ་ཡི། །བདག་ཉིད་ཆེ་རྣམས་གཞན་ཕན་མཆོག །ཟླ་བའི་ཟེར་གྱིས་ཤིན་ཏུ་དགའ། །སྐད་ཅིག་ལུས་ནི་ཁེངས་གྱུར་ནས། །བྱ་བའི་རྒྱ་མཚོ་རྦ་རླབས་དག །
