ལོག་པར་སྒྲུབ་པ་རྣམས་སོ། །འཇུག་ངོགས་བདེ་བའི་ཆུ་ཀླུང་བདེ། །ཆོས་ལ་གོམས་པའི་སྐྱེ་བོ་བདེ། །ཤེས་རབ་ཐོབ་པར་གྱུར་པ་བདེ། །ངའོ་ང་རྒྱལ་ཟད་པ་བདེ། །ཞེས་བྱ་བ་ལ་འཇུག་ངོགས་བདེ་བའི་ཞེས་བྱ་བ་ནི་འཇུག་པའི་ལམ་བདེ་བའོ། །ཆོས་ལ་གོམས་པ་ཞེས་བྱ་བ་ནི་དགེ་
བའི་ཆོས་ལ་གོམས་པའོ། །ཤེས་རབ་ཐོབ་པར་གྱུར་པ་ཞེས་བྱ་བ་ནི་ཟག་པ་མེད་པའི་ཤེས་རབ་ཐོབ་པར་གྱུར་པའོ། །ངའོ་ང་རྒྱལ་ཟད་པ་ཞེས་བྱ་བ་ནི་ཁམས་གསུམ་པའི་འདོད་ཆགས་དང་བྲལ་བའོ། །ངེས་པར་བྱས་ཤིང་དབང་པོ་ཐུལ་བ་དག །དགོན་པ་ཞི་བ་རྣམས་སུ་རྒས་གྱུར་དང་། །
མང་དུ་ཐོས་པ་ནགས་ཀྱི་ནང་དག་ཏུ། །ལང་ཚོ་ཡོལ་བ་རྣམས་ཀྱི་གནས་པ་བདེ། །ཞེས་བྱ་བ་ལ་ནི་ངེས་པར་བྱས་ཤིང་ཞེས་བྱ་བ་ནི་མངོན་པར་བརྩོན་པར་བྱས་པའོ། །དབང་པོ་ཐུལ་བ་དག་ཅེས་བྱ་བ་ནི་དབང་པོ་རྣམས་དབང་དུ་གྱུར་ཅིང་མཉམ་པར་གཞག་པ་དག་གོ། །དེ་དག་དགོན་པ་ཞི་བ་
རྣམས་སུ་རྒས་པར་གྱུར་པ་རྣམས་དང་། མང་དུ་ཐོས་པ་ནགས་ཀྱི་ནང་དག་ཏུ་ལང་ཚོ་ཡོལ་བ་རྣམས་ཀྱི་གནས་པ་བདེ་བ་ཞེས་བྱ་བ་ནི་དེ་ལྟར་གང་དག་གིས་མངོན་པར་བརྩོན་པ་བྱས་ནས་དབང་པོ་ཐུལ་བ་ནི་དགོན་པ་ཞི་བ་དབེན་པ་གྲོང་ལས་ཐག་རིང་བའི་ཕྱོགས་དག་ཏུ་རྒྱས་པར་གྱུར་པ་རྣམས་དང་།
གང་དག་མང་དུ་ཐོས་པ་ནི་ནགས་ཀྱི་ནང་དག་ཏུ་ལང་ཚོ་གཞོན་པ་ཉིད་ཡོལ་ཞིང་ཟད་པར་བྱས་པ་དེ་རྣམས་ཀྱི་གནས་པ་ཞེས་བྱ་བ་གསོན་པ་ཉིད་ནི་བདེའོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ཤཱཀྱ་སེང་གེ་དེ་ལ་ནི། །སེན་མོ་བཅུ་ཡི་ཐལ་སྦྱར་ཏེ། །འདུལ་བ་རབ་ཏུ་གདོན་པར་བྱ། །འདུལ་བའི་དོན་དུ་
ང་ལ་ཉོན། །ཞེས་བྱ་བ་ལ་སེན་མོ་བཅུ་ཞེས་བྱ་བ་ནི་ལག་སོར་གཉིས་ཀྱི་སེན་མོ་བཅུའོ། །འདུལ་བའི་དོན་དུ་ཞེས་བྱ་བ་ནི་སློབ་མ་རྣམས་གདུལ་བའི་དོན་དུའོ། །ཐོས་ནས་དྲང་སྲོང་ཆེན་པོ་ཡིས། །ཇི་སྐད་གསུངས་བཞིན་བསྒྲུབ་བྱ་ཞིང་། །ཁ་ན་མ་ཐོ་ཕྲ་རྣམས་ལ། །
