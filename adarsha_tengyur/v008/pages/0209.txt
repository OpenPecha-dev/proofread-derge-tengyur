སྟོང་པ་ཉིད་ཀྱི་མྱུ་གུ་སྐྱེ་བ་ཡིན་ནོ། །ཡེ་ཤེས་ཀྱི་རྒྱ་མཚོའི་མན་ངག་བསྟན་ནས། བུང་བས་སྦྲང་རྩི་ལེན་པའི་མན་ངག་བསྟེན་པ་ནི། བནྡུ་ཀའི་མེ་ཏོག་རྒྱས་པ་ལྟ་བུ་ནི་ཕྱག་རྒྱ་བཞི་ཡིན་ལ། སྦྲང་རྩི་ནི་བདེ་བ་རྟོག་པ་མེད་པ་ཡིན་ལ། བདུད་རྩི་ནི་བདེ་བ་སྟོང་པ་ཉིད་དེ། དུག་གི་
ཕྱག་རྒྱ་བཞི་ཡིན་ནོ། །ཀཱ་ལི་ཉིད་ཀྱང་བདུད་རྩི་སྟེ་ཁྲག་ནི་ཐིག་ལེ་ལས་ཀྱང་སྟོང་པ་ཉིད་ཀྱིས་རྒྱས་བཏབ་ན་ཡེ་ཤེས་ཀྱི་བདུད་རྩིར་འགྱུར་ལ། བདེ་བ་རྟོག་པ་མེད་པ་མེ་ཏོག་དང་བདེ་བ་སྟོང་པ་ཉིད་ཀྱི་ཚུལ་འཛིན་པ་ཡིན་ནོ། །གཉིས་ལས་སྐྱེས་པའི་ལུས་ཡིན་ཏེ། །ཞེས་པ་ནི་ཁྲག་དང་ཁུ་བ་ལས་སྐྱེས་པའི་ལུས་
སོ། །བདུད་རྩི་ཆུ་བོ་གོ་ཆ་མཆོག་ནི་ཆགས་ལམ་གྱི་དབང་དུ་བྱས་ནས་འབྱུང་བ་བཞིའི་རྩའི་དབུས་ན་ཨ་ཝ་དྷཱུ་ཏཱི་ནམ་མཁའི་རྩ་ལ་རེག་པ་ནི་ཆུ་ལས་སྐྱེས་པས་ནི་ཆུ་ཡིན་ལ། དེ་ཡང་མི་ཟད་པའི་སྟོང་པ་ཉིད་ཀྱིས་རྒྱས་བཏབ་པ་ནི་བདུད་རྩི་ཡིན་ནོ། །དེ་ཡང་དང་པོར་ལུས་ཀྱི་མངོན་སུམ་གྱིས་རེག་བྱ་ལྔ་ཉམས་སུ་མྱོང་
བར་འགྱུར་ཏེ། དུ་བ་ལྟ་བུར་སྣང་བ་གཡོ་བ་རླུང་ཡིན་ནོ། །གཤེར་བ་ཆུ་ནི་ཆུ་ཡིན་ལ། རེག་བྱ་མཁྲང་བ་ནི་ས་ཡིན་ནོ། །ཡེ་ཤེས་རྡོ་རྗེ་བྱིན་གྱིས་བརླབས་པ་ནི་ཨ་ཝ་དྷཱུ་ཏཱིའི་སྒོ་སྤྲད་པས་རྟོག་མེད་ཀྱི་ཡེ་ཤེས་སྟོང་པ་ཉིད་ཀྱི་རྩ་བྱིན་གྱིས་བརླབས་པ་ནི་གཞག་པའོ། །དེས་ན་རྣམ་པ་ལྔར་ནི་སྐྱེས་ཞེས་བྱའོ། །
མ་ལུས་ཆུ་ཡི་དབུས་གནས་ཤིང་། །ཞེས་པ་ནི་ཨ་ཝ་དྷཱུ་ཏཱིའི་སྒོ་སྤྲད་དེ། བྱང་ཆུབ་སེམས་ཀྱི་བདེ་བ་ལ་བརྟེན་ཏེ་འབྱུང་བ་ལྔའི་རེག་བྱ་དང་། དབུས་མ་ཨ་ཝ་དྷཱུ་ཏཱི་དང་རེག་པས་བདེ་བ་རྟོག་པ་མེད་པ་ལུས་ཀྱི་མངོན་སུམ་གྱིས་ཤེས་པར་འགྱུར་རོ། །དེ་སྐད་དུ་ཡང་མཉམ་སྦྱོར་ལས། སྒྱུ་མ་དག་ནི་
ཐམས་ཅད་ལས། །བུད་མེད་སྒྱུ་མ་ཁྱད་པར་འཕགས། །ཡེ་ཤེས་གསུམ་གྱི་དབྱེ་བ་ནི། གསལ་བར་འདི་ལས་རྟོགས་པར་འགྱུར། །ཞེས་གསུངས་སོ། །ཡུལ་དམིགས་པའི་རྐྱེན་ཨ་ཝ་དྷཱུ་ཏཱིའི་རེག་བྱ་དང་། དབང་པོ་བདག་པོའི་རྐྱེན་ཁུ་བ་དཀར་པོ་དང་། མཚུངས་པ་དེ་མ་ཐག་པའི་རྐྱེན་སློབ་དཔོན་
གྱི་མན་ངག་དང་རྟོག་པ་མེད་པའི་བག་ཆགས་ཀྱི་ས་བོན་ལས་བདེ་བ་སྟོང་པ་ཉིད་དུ་འདྲེན་ནུས་པའི་མཚན་མ་སྐྱེ་བ་ཡིན་ནོ། །དེ་ཡང་མཁས་པ་ཆེན་པོའི་ཞལ་སྔ་ནས། ཐམས་ཅད་ལས་ནི་སེམས་བསྡུས་ནས། །ནང་གི་བདག་ཉིད་གཡོ་མེད་ལས། །གནས་པས་ལུས་ཀྱི་རེག་བྱ་ནི། མྱོང་
