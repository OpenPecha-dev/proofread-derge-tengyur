རྣམས་དང་། གླིང་གསུམ་ལ་དབང་བ་རྣམས་དང་། གླིང་བཞི་ལ་དབང་བ་རྣམས་ཏེ། དེ་ལ་གླིང་གཅིག་ལ་དབང་བ་ལ་ནི་ལྕགས་ཀྱི་འཁོར་ལོ་འབྱུང་བར་འགྱུར་རོ། །གླིང་གཉིས་ལ་དབང་བ་ལ་ནི་ཟངས་ཀྱི་འཁོར་ལོ་འབྱུང་བར་འགྱུར་རོ། །གླིང་གསུམ་ལ་དབང་བ་ལ་ནི་དངུལ་གྱི་
འཁོར་ལོ་འབྱུང་བར་འགྱུར་རོ། །གླིང་བཞི་པ་ལ་དབང་བ་ལ་ནི་གསེར་གྱི་འཁོར་ལོ་འབྱུང་བར་འགྱུར་རོ། །དུས་བརྒྱད་གང་ཞེ་ན། འདི་ལྟ་སྟེ། མངལ་ན་གནས་པའི་དུས་དང་། །བཙས་པའི་དུས་དང་། བྱིས་པའི་དུས་དང་། གཞོན་ནུའི་དུས་དང་། ལང་ཚོའི་དུས་དང་། དར་ལ་བབ་
པའི་དུས་དང་། རྒས་པའི་དུས་དང་། འཁོགས་པའི་དུས་སོ། །དེ་ལ་མངལ་ན་གནས་པའི་དུས་ནི་མེར་མེར་པོ་ལ་སོགས་པའོ། །བཙས་པའི་དུས་ནི་དེ་ཡན་ཆད་དང་། འཁོགས་པའི་དུས་མན་ཆད་དོ། །བྱིས་པའི་དུས་ནི་འགྲོ་བ་དང་རྩེ་མི་ནུས་པ་མན་ཆད་དོ། །བྱིས་པ་གཞོན་ནུའི་དུས་
ནི་དེ་ནུས་པ་ཡིན་ནོ། །ལང་ཚོའི་དུས་ནི་འདོད་པ་སྤྱོད་ནུས་པ་ནས་ལོ་སུམ་ཅུ་མན་ཆད་དོ། །དར་ལ་བབ་པའི་དུས་ནི་ལོ་ལྔ་བཅུ་མན་ཆད་དོ། །རྒས་པའི་དུས་ནི་ལོ་བདུན་ཅུ་མན་ཆད་དོ། །དེ་ཡན་ཆད་ནི་འཁོགས་པའི་དུས་སོ། །མངལ་དུ་འཇུག་པ་བཞི་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཤེས་
བཞིན་དུ་འཇུག་ལ་མི་ཤེས་པར་གནས་ཤིང་འབྱུང་བ་དང་། ཤེས་བཞིན་དུ་འཇུག་ཅིང་གནས་ལ་མི་ཤེས་པར་འབྱུང་བ་དང་། ཤེས་བཞིན་དུ་འཇུག་ཅིང་གནས་ཏེ་འབྱུང་བ་དང་། མི་ཤེས་པར་འཇུག་ཅིང་གནས་ཏེ་འབྱུང་བའོ། །དེ་ལ་དང་པོ་ལ་ནི་འཁོར་ལོས་སྒྱུར་བའིའོ། །གཉིས་པ་
ནི་རང་སངས་རྒྱས་ཀྱིའོ། །གསུམ་པ་ནི་བྱང་ཆུབ་སེམས་དཔའིའོ། །བཞི་པ་ནི་དེ་ལས་གཞན་པའི་སེམས་ཅན་རྣམས་སོ། །འཚོ་བ་རྣམ་པ་དྲུག་གང་ཞེ་ན། འདི་ལྟ་སྟེ། ཞིང་རྨོ་བ་དང་། ཚོང་བྱ་བ་དང་། ཕྱུགས་འཚོ་བ་དང་། རྒྱལ་པོ་ལ་ཞོ་ཤས་འཚོ་བ་དང་། ཡི་གེ་དང་། གྲངས་
དང་། ཤོད་དགོད་པ་དང་། རྩིས་དང་། ལག་རྩིས་དང་། དེ་ལས་གཞན་པའི་བཟོའི་གནས་རྣམས་སོ། །ཀུན་ནས་སྲུང་བ་རྣམ་པ་དྲུག་གང་ཞེ་ན། འདི་ལྟ་སྟེ། གླང་པོ་ཆེའི་ཚོགས་དང་། རྟ་བའི་ཚོགས་དང་། ཤིང་རྟའི་ཚོགས་དང་། དཔུང་བུ་ཆུང་གི་ཚོགས་དང་། ནོར་གྱི་སྟོབས་
