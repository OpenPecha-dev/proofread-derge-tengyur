པ་གང་ཡིན་པའོ། །མ་བྱིན་པར་ལེན་པར་བྱེད་པ་ཞེས་བྱ་བ་ནི་བརྒྱ་ལ་ཡོ་བྱད་མེད་པའི་ཕྱིར་བདག་གིར་བྱེད་པའོ། །མ་བྱིན་པ་ལ་དགའ་བ་ཞེས་བྱ་བ་ནི། རྐུ་བའི་ལས་དེ་ཡང་དག་པར་བླངས་ཏེ་འདུག་པའོ། །མ་བྱིན་པ་དང་། མ་བཏང་བ་དང་། མ་སྤངས་པ་འདོད་པ་ཞེས་བྱ་བ་
ནི་གཞན་གྱིས་བརྐུས་པ་བདག་གིར་བྱེད་པར་འདོད་པའོ། །དེ་ལ་སྦྱིན་པ་པོ་བདག་པོས་སྦྱིན་པའི་ཚུལ་གྱིས་སྔར་བྱིན་པ་མ་བྱིན་པ་ཞེས་བྱའོ། །བདག་པོས་ལེན་པ་པོའི་ཕྱིར་ཡོངས་སུ་བཏང་བ་མ་ཡིན་པ་གང་ཡིན་པ་དེ་ནི་མ་བཏང་བ་ཞེས་བྱ་བའོ། །བདག་པོ་ཉིད་ཀྱིས་སྐྱེ་བོ་
ཐམས་ཅད་ལ་ཡོངས་སུ་ལོངས་སྤྱོད་པའི་ཕྱིར། ཇི་ལྟ་ཇི་ལྟར་འདོད་པ་བཞིན་དུ་ལེན་པར་བྱ་བའི་དོན་དུ། མ་སྤངས་པ་གང་ཡིན་པ་དེ་ནི་མ་སྤངས་པ་ཞེས་བྱའོ། །རྐུ་བ་བདག་ཉིད་ཡོངས་སུ་སྦེད་དོ་ཞེས་བྱ་བ་ནི་མ་བྱིན་པར་ལེན་པར་བྱེད་པ་དང་། མ་བྱིན་པར་ལེན་པ་ལ་དགའ་བའི་
ཕྱིར་རོ། །བརྐམ་པ་ཡོངས་སུ་སྦེད་དོ་ཞེས་བྱ་བ་ནི་མ་བྱིན་པ་དང་། མ་བཏང་བ་དང་། མ་སྤངས་པ་འདོད་པའི་ཕྱིར་རོ། །མ་དག་པ་ཡོངས་སུ་སྦེད་ཅེས་བྱ་བ་ནི་རྩོད་པ་ལ་ཕམ་པར་གྱུར་ཏེ་མ་དག་པའི་ཕྱིར་རོ། །མི་གཙང་བ་ཡོངས་སུ་སྦེད་ཅེས་བྱ་བ་ནི་ཕམ་པར་མ་གྱུར་ཀྱང་། 
ཉེས་པ་དེས་དྲི་མ་ཅན་དུ་གྱུར་པའི་ཕྱིར་རོ། །ཁ་ན་མ་ཐོ་བ་དང་བཅས་པ་ཡོངས་སུ་སྦེད་ཅེས་བྱ་བ་ནི། ཚེ་འདི་དང་ཕྱི་མ་ལ་མི་སྡུག་པའི་འབྲས་བུའི་རྒྱུ་ཡོངས་སུ་འཛིན་པའི་ཕྱིར་རོ། །མ་བྱིན་པར་ལེན་པ་མ་སྤངས་པ་ཡིན་ཞེས་བྱ་བ་ནི་སྲོག་གཅོད་པ་རྣམ་པར་དབྱེ་བ་བཞིན་དུ། འདི་དང་
དེ་ལས་གཞན་པའི་ལས་ཀྱི་ལམ་རྣམས་ལ་ཡང་དེ་དང་འདྲ་བར་བལྟ་བར་བྱའོ། །དོན་མདོར་བསྡུ་ན། གང་ལས་བརྐུས་ན་མ་བྱིན་པར་ལེན་པར་བྱེད་པ་ཡིན་པ་དང་། གནས་གང་ཡིན་པ་དང་། ཐབས་ཇི་ལྟར་བརྐུ་བ་དང་། ཉེས་པ་དེ་ལས་སྐྱོན་དང་ལྡན་པར་འགྱུར་བ་འདི་ནི་བསྡུས་པའི་
དོན་ཡིན་ནོ། །འདི་ཡང་མ་བྱིན་པར་ལེན་པར་བྱེད་པའི་གང་ཟག་གི་མཚན་ཉིད་ཡིན་གྱི། མ་བྱིན་པར་ལེན་པའི་མཚན་ཉིད་མ་ཡིན་པར་རིག་པར་བྱའོ། །ལྷག་མ་རྣམས་ཀྱང་དེ་དང་འདྲའོ། །འདོད་པ་ལ་ལོག་པར་གཡེམ་པར་བྱེད་པ་ཡིན་ཞེས་བྱ་བ་ནི་བསྟན་པའི་ཚིག་གོ། །གང་དེ་དག་
