ཅད་ལི་ཁྲི་ལྟར། །སྣ་ཚོགས་རིན་ཆེན་རྒྱན་གྱིས་མཛེས། །བུམ་པ་ཐོད་པར་བཅས་པ་འཛིན། །རྡོ་རྗེ་སྐྱིལ་ཀྲུང་བཞུགས་པའོ། །བཅུ་དྲུག་ལོན་པ་སྐུ་ཤ་རྒྱས། །འཇམ་ཞིང་ནུ་མ་འབུར་བ་ལ། །པདྨ་རྡོ་རྗེ་རིན་ཆེན་དང་། །འཁོར་ལོ་སྣ་ཚོགས་རྡོ་རྗེ་སྟེ། །གཡས་པ་
རང་གི་ཕྱག་མཚན་འཛིན། །གཡོན་པ་ཐོད་པ་བདུད་རྩིས་གང་། །ཡབ་ལ་སྟོབས་ཤིང་གཉིས་སུ་འཁྲིལ། །ཐབས་དང་ཤེས་རབ་གཉིས་སུ་མེད། །ཡབ་ཡུམ་འོད་ཟེར་ཚོགས་འཁྲུགས་ཤིང་། །སྙོམས་འཇུག་ཟག་མེད་བདེ་བ་རྒྱས། །མཚམས་བཞིར་བདུད་
རྩིའི་བུམ་པ་བཞི། །ཕྱོགས་ཀྱི་ལྷ་ཡི་སྣམ་བུ་ལ། །ཤར་དུ་རྡོ་རྗེ་ཁྲོ་བོ་དང་། །ལྷོ་རུ་རིན་ཆེན་ཁྲོ་བོ་དང་། །ནུབ་ཏུ་པདྨའི་ཁྲོ་བོ་དང་། །བྱང་དུ་ཕྲིན་ལས་ཁྲོ་བོ་དང་། །སྟེང་འོག་རྣམ་སྣང་ཁྲོ་བོ་གཉིས། །ཕྱོགས་བཞིར་ཁྲོ་བོ་གཉིས་གཉིས་ནི། །རིགས་ཀྱི་ཡུམ་
