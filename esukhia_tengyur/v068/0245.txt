བདུད་རྩི་འཁྱིལ་པའིའོ། །ཨོཾ་ཀཱ་ལ་དཎྜ་ཧཱུཾ་ཕཊ། བེ་ཅོན་སྔོན་པོའིའོ། །ཨོཾ་ཧྲཱིཿཧཱུཾ་ཕཊ། རྟ་མགྲིན་གྱིའོ། །ཨོཾ་ཨིནྡྲཱ་ཡ་སྭཱ་ཧཱ། བརྒྱ་བྱིན་འཁོར་བཅས་ཀྱིའོ། །ཨོཾ་བྲཧྨ་ཎེ་སྭཱ་ཧཱ། ཚངས་པའིའོ། །ཨོཾ་ག་ཎ་པ་ཏི་ཧཱུཾ། གཏོགས་འདོད་ཀྱིའོ། །ཨོཾ་བཻཥྞ་བི་ཡེ་སྭཱ་ཧཱ།
དབང་ཕྱུག་ཆེན་པོའིའོ། །ཨོཾ་དེ་བ་ག་ཎེ་བྷྱཿསྭཱ་ཧཱ། ལྷ་ཐམས་ཅད་ཀྱིའོ། །ཨོཾ་བེ་བ་སྭ་ད་མ་ཏྲི་བྷྱཿསྭཱ་ཧཱ། གཤིན་རྗེ་འཁོར་དང་བཅས་པའིའོ། །ཨོཾ་བརྱ་ཎ་ཡེ་ནཱ་ག་ག་ཎ་བྷྱཿསྭཱ་ཧཱ། ཆུ་ལྷ་འཁོར་དང་བཅས་པའིའོ། །ཀུ་བེ་ར་ཡཀྵ་ནི་བྷྱཿསྭཱ་ཧཱ། གནོད་སྦྱིན་འཁོར་བཅས་ཀྱི་སྔགས་སོ། །
ཨོཾ་ཨགྣ་ཡེ་ཀུ་ཙི་ཀུ་ཙི་སྭཱ་ཧཱ། དྲང་སྲོང་རྣམས་ཀྱིའོ། །ཨོཾ་ནཻ་ཪྼྀ་ཏྱ་རཱཀྵ་ས་བྷྱཿསྭཱ་ཧཱ། སྲིན་པོ་འཁོར་བཅས་ཀྱིའོ། །ཨོཾ་བཱ་ཡ་བེ་ཨ་སཱུ་ར་ག་ཎེ་བྷྱཿསྭཱ་ཧཱ། རླུང་ལྷ་འཁོར་དང་བཅས་པའིའོ། །ཨོཾ་རུ་དྲ་ཨཀྵ་བྷཱུ་ཏ་ག་ཎེ་བྷྱཿསྭཱ་ཧཱ། དབང་ལྡན་འཁོར་བཅས་ཀྱིའོ། །ཨོཾ་པྲྀ་སྠི་བྷྱཿསྭཱ་ཧཱ།
སའི་ལྷ་མོའིའོ། །ཨོཾ་གྲ་ཧེ་པྲ་བྷཱ་ཛྱོ་ཏི་ས་ཡ་ནཀྵ་ཏྲ་ནི་རྣཱ་ད་ནི་སྭཱ་ཧཱ། གཟའ་དང་རྒྱུ་སྐར་སྤྱིའི་སྔགས་སོ། །ཨོཾ་མ་ཧོ་ར་ག་ག་ཎེ་བྷྱཿསྭཱ་ཧཱ། ལྟོ་འཕྱེ་རྣམས་ཀྱིའོ། །ཨོཾ་ཀུཾ་བྷཎྜ་པཱུཏ་ཨུནྨ་དཱ་པ་སྨ་ར་ཨ་མ་ཨོསྟཱར་ག་ཎེ་བྷྱཿསྭཱ་ཧཱ། འདྲེ་སྤྱིའིའོ། །ཨོཾ་ག་རུ་ཌ་ག་ཎ་བྷྱཿསྭཱ་ཧཱ། ནམ་མཁའ་ལྡིང་
སྤྱིའིའོ། །ཨོཾ་ཨགྣ་སྭཱ་ཧཱ། མེ་ལྷའིའོ། །ཨོཾ་བ་རུ་ཎཱ་ཡ་སྭཱ་ཧཱ། ཆུ་ལྷའིའོ། །ཨོཾ་བཱ་ཡ་བེ་སྭཱ་ཧཱ། རླུང་ལྷའིའོ། །ཨོཾ་པ་ཊ་ལ་སྭཱ་ཧཱ། ས་བདག་གོ། །ཨོཾ་སརྦ་དེ་བ་ཏ་སྭཱ་ཧཱ། ཁྱིམ་ལྷ་ལ་སོགས་པ་ཐམས་ཅད་ཀྱིའོ། །ཨོཾ་ཧྲི་སྭཱ་ཧཱ། མཁའ་འགྲོ་མའིའོ། །དེ་དག་ནི་སྔགས་སུ་གསུངས་
པའོ། །ཏིང་འཛིན་སྔགས་དང་ཕྱག་རྒྱར་ལྡན། །ཉིན་རེ་ཏིང་འཛིན་གསུམ་པོ་ནི། །བདག་དང་ལྷ་སྦྱོར་བསྒོམ་པར་བྱ། །དེ་ནས་བདག་ཉིད་དེར་གསལ་ནས། །རྒྱས་འདེབས་བཞི་དང་ལྡན་པ་ལ། །དཀྱིལ་འཁོར་སྒོ་ནི་དབྱེ་བྱས་ནས། །དཀྱིལ་འཁོར་ཡོངས་སུ་བསྐྱེད་པར་བྱ། །ཨོཾ་སརྦ་
བིད་བཛྲ་ཙཀྲ་ཧཱུཾ། འཁོར་ལོ་དང་ལྡན་ཕྱག་རྒྱ་བྱ། །དཀྱིལ་འཁོར་བསྐྱེད་སྔགས་ཕྱག་རྒྱའོ། །དེ་ནས་གཙོ་བོར་བདག་སྦྱོར་བས། །དཔུང་པར་སེ་གོལ་གདབ་བྱས་ལ། །ཛཿཧཱུཾ་བཾ་ཧོཿས་སྤྱན་དྲང་ངོ་། །དེ་ནས་ཉེ་རེག་འཐོར་འཐུང་དང་། །ཞབས་བསིལ་བ་ནི་དབུལ་བར་བྱ། །མཆོད་པ་
