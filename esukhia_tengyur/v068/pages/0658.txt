ལ་བཛྲ་བི་མ་ལ་རཽ་ཏྲི་པྲ་ཀྲྀ་ཏི། མ་ཧཱ་ཀྲོ་དྷ་བཛྲ་ཡ་ན་ཛ་ཧ་གྲྀཧྞ་ཏ་ཧཱུཾ་ཕཊ། སྦྱོང་བའི་ལས་ནི་ལྷ་ཚོགས་ཀྱིས། །མདུན་དུ་བྱ་བར་ཡང་དག་བཤད། །དེ་ནས་དཀོན་མཆོག་གསུམ་པོ་སྟེ། །སྦྱར་དང་སྤྱན་དྲང་གདན་དག་དང་། །མཆོད་པ་ལ་སོགས་རྣམ་རྒྱལ་བཞིན། །མགོན་པོ་ཐུགས་རྗེ་ཆེར་ལྡན་པ། །

ཀུན་མཛད། །དང་བའི་ཡིད་ཀྱིས་བསྐྱེད་པ་ཡི། །གཞལ་ཡས་ཉིད་དུ་གཤེགས་སུ་གསོལ། །ཨོཾ་བཛྲ་བེ་གཱ་ཀྲ་མ་ཡ་ཧི་ག་ཛ། མེ་ལྷ་གཉིས་ནི་སྤྱན་དྲང་བའི། །མདུན་དུ་ཀཱི་ལིས་མེ་ཉིད་སྦྱོང་། །དེ་ནས་གཡས་པའི་མཐེ་བོ་ཡིས། །མཐེའུ་ཆུང་མདུང་ཚུགས་གཡབས་ནས་ནི། །དད་པ་དང་ནི་
དམ་ཚིག་གིས། །ཚུར་གཤེགས་ཞེས་ནི་བོས་ནས་ནི། །ཛཿཧཱུཾ་བཾ་ཧོཿཨེ་ཧྱེ་ཧི། མ་ཧཱ་བྷཱུ་ཏ་དེ་བ་རི་ཥི། དྭི་ཛ་ས་ཏ་མ་ནི། ཧི་ཏྭ་ཨ་ཧུ་ཏི་མ་ཧཱ་རཾ། ཨསྨི་སནྣི་ཧི་ཏོ་བྷ་བ། ཨགྣ་ཡེ་ཨཱ་གཙྪ། ཧ་བྱ་ཀ་བྱ་བ་ཧ་ནཱ་ཡ་སྭཱ་ཧཱ། དེ་ནས་བཞུགས་པར་བྱ་བ་ནི། །ལྷ་ཡི་དྲང་སྲོང་ཀུན་དང་ནི། །སྲེག་སྦྱིན་
བཏབ་པར་ཁས་འཆེས་པ། །ཐུགས་རྗེ་དམ་ཚིག་བརྟན་པ་ནི། །པདྨའི་གདན་ལ་བཞུགས་སུ་གསོལ། །ཨགྣ་ཡེ་ས་མ་ཡ་སྟྭཾ་ཏིཥྛ། སྭ་བྷཱ་ཝ་བི་ག་ཏ་བི་ཛ་ལོ་ཧི་ཏ་ཨཱ་ཀཱ་ཤ་དཱི་པ་ཡ་ཤུདྡྷེ་སྭཱ་ཧཱ། ཕྱག་ཞལ་བསིལ་བ་བྱས་ནས་ནི། །གཉིས་མིན་དྲང་སྲོང་མེ་ལྷ་ལ། །ཤག་ཏེའུ་ལ་སོགས་
