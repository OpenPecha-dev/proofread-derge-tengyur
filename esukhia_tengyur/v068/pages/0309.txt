གདགས་པར་བྱ་བ་ནི། ཨོཾ་བཛྲ་སཏྭ་ཨ་བྷི་ཥིཉྩ་མི་བཛྲ་ནཱ་མ་ཨ་བྷི་ཥེ་ཀ་ཏ། ཞེས་བྱས་ལ། ཀྱེ་དཔལ་ཆེ་གེ་མོ་ལ་ཞེས་མིང་གདགས་པར་བྱ་སྟེ། དེ་དག་ནི་རིག་པའི་དབང་ངོ་། །དེ་ནས་མི་བསྐྱོད་པ་ལ་སོགས་པའི་བུམ་པ་བཞི་དང་། རྣམ་པར་རྒྱལ་བའི་བུམ་པ་ལྔར། ཨོཾ་བུདྡྷ་
ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཞེས་བྱས་ལ་བུམ་པའི་དབང་བསྐུར་བར་བྱའོ། །དེ་ནས་རྣམ་པར་སྣང་མཛད་ལ་སོགས་པའི་ཕྱག་རྒྱ་བཅས་ལ། ཨོཾ་བུདྡྷ་མུ་དྲ་
ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཞེས་བྱ་བས་ཕྱག་རྒྱའི་དབང་བསྐུར་བར་བྱའོ། །དེ་ནས་མུ་ཏིག་དང་། གསེར་དང་། པདྨ་རཱ་ག་དང་། མརྒཏ་དང་། སྤུག་ལ་
སོགས་པའམ། གཞན་ཡང་རིགས་དང་མཐུན་པའི་ཕྲེང་བ་རྣམས་རང་གི་རིགས་ཀྱི་སྔགས། ཨོཾ་བཻ་རོ་ཙ་ན་ཨོཾ། ཞེས་པ་ལ་སོགས་པའི་སྙིང་པོ་སློབ་དཔོན་དང་སློབ་མའི་རྡོ་རྗེ་ཁུ་ཚུར་གྱི་སྦུ་གུ་ནས་བཏོན་ལ་བཟླས་ཏེ། ཨོཾ་བུདྡྷ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། །ཨོཾ་བཛྲ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ།
ཨོཾ་རཏྣ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཞེས་བྱ་བས་ཕྲེང་བའི་དབང་བསྐུར་བར་བྱའོ། །དེ་ནས་སློབ་མའི་སྙིང་གར་དྲིའི་ཟླ་བའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་མཆོག་ཏུ་གྱུར་པའི་དྲི་ཡིས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཏིང་ངེ་འཛིན་རྣམ་པ་
ལྔ་མངོན་དུ་བྱས་པ་ལས་ཐིག་ལེ་ལྔ་བསྐུ་ཞིང་། ཨོཾ་བུདྡྷ་དྷཱ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་བཛྲ་དྷཱ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་རཏྣ་དྷཱ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་དྷཱ་ར་ཎི་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་དྷཱ་ར་ནི་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཞེས་བྱ་བས་གཟུངས་ཀྱི་དབང་བསྐུར་བར་བྱའོ། །དེ་ནས་བཀྲ་ཤིས་པའི་རྫས་
ཐམས་ཅད་ཀྱིས་བཀྲ་ཤིས་ཀྱི་ཚིགས་སུ་བཅད་པ་རྣམས་ཀྱིས་བཀྲ་ཤིས་བཏོན་ལ་རིན་པོ་ཆེའི་ཐུར་མའམ་གསེར་ལ་སོགས་པའི་ཐུར་མ་ཐོགས་ལ་མདུན་དུ་འདུག་སྟེ། ཇི་ལྟར་མིག་མཁན་རྒྱལ་པོ་ཡིས། །འཇིག་རྟེན་ལིང་ཏོག་བསལ་བ་ལྟར། །བུ་ཁྱོད་ཀྱི་ནི་མི་ཤེས་པའི། །རབ་རིབ་ཐམས་
