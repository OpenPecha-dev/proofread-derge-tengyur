ཀོ་སྐོར་ཨོཾ་སརྦ་བདི་ཏྲཊ། ཅེས་པའོ། །དཔྲལ་བར་ནི། ཨོཾ་སརྦ་བིད་ཨོཾ། ཞེས་པའོ། །སྤྱི་བོར་ནི། ཨོཾ་སརྦ་བིད་ཧཱུཾ། ཞེས་པའོ། །དཔུང་པ་གཉིས་སུ་ནི། ཨོཾ་སརྦ་བིཏ་སྭ། ཞེས་པའོ། །གྲུ་མོ་གཉིས་སུ་ནི། ཨོཾ་སརྦ་བིད་ཨོཾ། ཞེས་པའོ། །ལོང་བུ་གཉིས་སུ་ནི།
ཨོཾ་སརྦ་བིད་ཏྲཊ། ཅེས་པའོ། །གསང་བ་ལ་ནི། ཨོཾ་སརྦ་བིད་ཕཊ། ཅེས་པའོ། །རྐེད་པར་ནི། ཨོཾ་སརྦ་བིད་ཨོཾ། ཞེས་པའོ། །གཙུག་ཕུད་ལ་ནི། ཨོཾ་བཛྲ་ཨ་དྷིཥྛ་ན་ཛྙཱ་ས་མ་ཡེ་ཧཱུཾ། ཞེས་པའོ། །མཆན་འོག་གཉིས་སུ། ཨོཾ་བཻ་རོ་ཙ་ན་ཨོཾ། ཞེས་པའོ། །བརླ་གཉིས་སུ། ཨོཾ་སརྦ་དུརྒ་
ཏཾ་པ་རི་ཤོ་དྷ་ནེ་ཧཱུཾ། ཞེས་པའོ། །བྱིན་པ་གཉིས་སུ་ནི། ཨོཾ་རཏྣ་ཀེ་ཏུ་ཏྲཾ། ཞེས་པའོ། །པུས་མོ་གཉིས་སུ་ནི། ཨོཾ་ཤཱཀྱ་མུ་ནེ་ཨ། ཞེས་པའོ། །བོལ་གོང་གཉིས་སུ་ནི། ཨོཾ་ཏྲཊ་ཨཱ་གཙྪ། ཞེས་པ་གདགས་པར་བྱའོ། །དེ་ནས་རྩ་བའི་སྔགས་ཀྱིས་མངོན་པར་བསྔགས་པའི་བདུག་
པས་བདུགས་ཤིང་མེ་ཏོག་གི་ཕྲེང་བས་སོ་སོའི་ཆ་བྱད་ཇི་ལྟ་བ་བཞིན་བརྒྱན་ཏེ། འགྲོ་བ་ཐམས་ཅད་ཡོངས་སུ་དག་པའི་དཀྱིལ་འཁོར་ཆེན་པོར་ཡོན་དང་བཅས་པས་དབང་བསྐུར་ལ་སོ་སོའི་རིགས་ཇི་ལྟ་བ་བཞིན་བྱིན་གྱིས་བརླབས་ཏེ་རབ་ཏུ་གནས་པར་བྱའོ། །དེ་ནས་སྲེག་ཁང་གི་ནང་དུ་
གནས་པར་བྱ་སྟེ། དེ་ལ་དགེ་བསྙེན་གྱི་སྲེག་ཁང་གི་ཁང་བུ་བུད་ཤིང་ལ་སོགས་པས་ཁ་བཀབ་ལ་ཞལ་ཞལ་གྱིས་འཐས་པར་བྱས་ཏེ། དེའི་སྟེང་དུ་སེང་ལྡེང་གི་ཤིང་ཟ་མ་ཏོག་གི་དབྱིབས་ལྟ་བུར་བརྩིག་པར་བྱའོ། །དགེ་སློང་ལ་སོགས་པའི་ཁུང་བུའི་གྲྭ་བཞིར་ཀ་བ་ཁྲུ་རེ་བ་བཞི་བཙུགས་
ལ། དེའི་སྟེང་དུ་སོར་མོ་བཅུ་གཉིས་པའི་སྟེགས་བརྩེགས་པར་བྱའོ། །དེ་ཉིད་ཁ་ཁྱེར་དང་ལྡན་པར་བྱ་ཞིང་དེའི་འོག་ཏུ་བླ་རེ་དབུབ་ཅིང་སྟེང་དུ་དཀར་པོའི་ཤིང་མཆོད་རྟེན་གྱི་དབྱིབས་ལྟ་བུར་བརྩིག་པར་བྱའོ། །རྡོ་རྗེ་སློབ་དཔོན་ལ་སོགས་པའི་ཉིས་བརྩེགས་སུ་བརྩིགས་ལ། སྟེང་དུ་ནི་ཙནྡན་
སྦྲུལ་གྱི་སྙིང་པོ་ལ་སོགས་པའི་ཤིང་གཞལ་ཡས་ཁང་གི་དབྱིབས་ལྟ་བུར་བརྩིགས་ལ། བར་ཁང་དུ་དཀྱིལ་འཁོར་བྲིས་ལ་འོག་ཏུ་འཛག་པའི་ཁུང་བུར་རྡུལ་ལ་སོགས་པའི་སྣོད་གཞག་པར་བྱ་བ་ཉིད་དོ། །དེ་ནས་དེའི་དབུས་སུ་གཞག་པར་བྱ་བ་ནི་དབང་ལ་སོགས་པ་རྫོགས་པར་བྱས་པའི་
