གྱིས་རལ་གྲི་མཐུ་ཅན་གྱི་ཁྲོམ་ནས་ལ་བ་རིན་པོ་ཆེ་རིན་ཐང་ཆེན་པོ་རྙེད་ནས་དེས་ཀྱང་སྦོམ་དགའ་མོ་ལ་བྱིན་ནས་དེས་དེ་གྱོན་ཏེ་བསོད་སྙོམས་ལ་ཞུགས་པ་དེ་ལས་བཅས་ཏེ་ཡང་དགེ་སློང་མ་གང་གོས་ལྕི་བ་དང་། རིན་ཐང་ཆེན་པོ་འཆང་བར་བྱེད་ན་སྤང་བའི་ལྟུང་བྱེད་དོ་ཞེས་
གསུངས་པ་ཡིན་ནོ། །འཆང་བར་བྱེད་ན་ཞེས་བྱ་བའི་རྣམ་པར་འབྱེད་པ་ནི་བདག་གིར་བྱེད་ན་ཞེས་གསུངས་པ་ཡིན་ནོ། །དེའི་ཕྱིར་བདག་གིར་བྱེད་པ་ལས་ཀྱང་དངོས་གཞིར་འགྱུར་ན་གྱོན་པ་ལྟ་ལ་སྨོས་ཀྱང་ཅི་དགོས་ཏེ་གྱོན་པའི་རྩ་བ་བཅས་པའི་ཕྱིར་བདག་གིར་བྱེད་པ་ཡང་དགག་པ་མཛད་
དོ། །བརྗེ་བའི་ཕྱིར་བདག་གིར་བྱེད་པ་ལ་ནི་ལྟུང་བ་མེད་དོ། །བཙོང་བ་ལ་སོགས་པའི་ཕྱིར་བདག་གིར་བྱེད་པ་ནི་ཐམས་ཅད་དུ་སྤང་བར་འགྱུར་རོ། །རུང་བར་བྱ་བ་གང་གིས་རིན་པོ་ཆེ་ལ་སོགས་པ་བླང་བར་གསུངས་བའི་རུང་བར་བྱ་བ་<དོས་,དེས་>འདི་ཡང་བྱིན་གྱིས་བརླབས་ཏེ་ལེན་པ་ལ་ནི་
ཉེས་པ་མེད་དོ། །གཞན་གྱིས་གོན་པ་ལ་ཡང་སྤང་བར་བྱ་བའི་དངོས་པོ་མེད་དུ་ཟིན་ཀྱང་ཁ་ན་མ་ཐོ་བ་ནི་ཡོངས་སུ་རྫོགས་པར་འགྱུར་རོ། །ལྗིད་དང་རིན་ཐང་གཉི་གའི་སྐབས་སུ་སོགས་པ་ཞེས་བྱ་བའི་སྒྲ་སྨོས་པ་ནི་བསླབ་པའི་གཞི་འོག་མ་ལ་ཡང་དེ་བཞིན་དུ་ཤེས་པར་བྱ་བའི་ཕྱིར་ཏེ། དེ་ལ་ནི་
ལྗིད་ཀྱི་སོགས་པའི་སྒྲས་ནི་ཉུང་ངུ་དག་བསྡུའོ། །རིན་ཐང་གི་སོགས་པའི་སྒྲས་ནི་སྔ་མ་བཞིན་དུ་མང་པོ་དག་བསྡུའོ། །གོས་ལྕི་བ་བསྟེན་པའི་སྤང་བའོ་ཞེས་བྱ་བ་དེ་བཤད་ཟིན་ཏོ།། །།གང་ལ་གོས་ཡང་བ་བསྟེན་པའི་སྤང་བར་འགྱུར་ཞེ་ན། དེའི་ཕྱིར་སྲང་ལྔ་པའི་མཐར་ཐུག་པའོ་
ཞེས་བྱ་བ་སྨྲས་ཏེ། འདིར་གོས་ཡང་བ་ཞེས་བྱ་བ་ནི། ལྗིད་ཀྱིས་ཡང་བ་ཉི་ཚེ་ལ་བྱ་སྟེ། ལྗིད་སྲང་ལྔ་པ་ལ་སོགས་པའོ། །སོགས་པ་ཞེས་བྱ་བའི་སྒྲ་མདོ་སྔ་མ་ལས་སྨོས་པ་འདིར་ཡང་སྦྱར་བར་བྱ་སྟེ། སོགས་པ་ཞེས་བྱ་བའི་སྒྲས་འདིར་ནི་གྲངས་ཉུང་ངུ་བཞི་དང་གསུམ་དང་གཉིས་
དག་བསྡུ་བར་རིག་པར་བྱའོ། །མཐར་ཐུག་པའོ་ཞེས་སྨོས་པས་ནི་ལྗིད་སྲང་གཅིག་པའམ་གཞལ་དུ་མེད་པའི་གོས་ཀྱི་མཐར་ཐུག་པ་དག་བསྡུ་བར་ཤེས་པར་བྱའོ། །འདིར་གཞུང་ནི་ཡང་བ་ཞེས་བྱ་བ་ནི་སྲང་ལྔ་པའམ་ལྔ་ལས་ཆུང་བའོ་ཞེས་གསུངས་པ་ཡིན་ཏེ། འདིས་ནི་ཀ་ཤི་ཀའི་རས་ཕྲན་
