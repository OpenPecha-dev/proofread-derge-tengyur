ཉིད་ཀྱི་ཚིག་གཉིས་རྫོགས་ཟིན་པ་དེ་ལས། ལས་བརྗོད་པ་འབའ་ཞིག་ཡང་གཞི་ནས་བཟླས་པར་འདོད་ན་དེ་སྐད་ཅེས་བྱའོ། །གལ་ཏེ་གསོལ་བའི་བདག་ཉིད་ཀྱི་ཚིག་དང་། ལས་བརྗོད་པའི་བདག་ཉིད་ཀྱི་ཚིག་གཉི་ག་རྫོགས་ཟིན་པ་དེ་དག་ཡང་གཞི་ནས་བཟླས་པར་འདོད་
ན་ནི་དེའི་ཚེ་གསོལ་བ་བགྱི་ཞིང་རྗེས་སུ་ཡང་བསྒྲག་གོ་ཞེས་ཚིག་གཉི་ག་བརྗོད་པར་བྱའོ། །འདིར་དགེ་འདུན་ལ་ཞེས་སྨོས་པ་ནི་དཔེ་ཙམ་དུ་ཟད་དེ། དེས་ན་གང་ཟག་ལ་ལྷུང་བཟེད་ལ་སོགས་པ་བྱིན་གྱིས་བརླབས་པའི་ཚིག་རྫོགས་ཟིན་པ་དེ་ཡང་གཞི་ནས་བཟླས་པར་འདོད་ན་ཡང་གང་ཟག་
དེ་ལ་བཙུན་པ་ཡང་རྗེས་སུ་བསྒྲག་གོ་ཞེས་བྱ་བའི་ཚིག་གིས་ཚོགས་སྔར་རྫོགས་པ་དེ་རྣམ་པར་དཀྲུག་པར་བྱ་ཞིང་དེའི་འོག་ཏུ་སྔགས་ཀྱི་ཚིག་གཞི་ནས་བཟླས་པར་བྱའོ། །དེ་བས་ན་འདིར་གལ་ཏེ་དེ་ཡང་བྱེད་པར་འདོད་ན་ཚིག་གིས་རྣམ་པར་དཀྲུག་ཅིང་གསོལ་བས་ཞེས་སྨོས་པ་ནི། དགེ་འདུན་
འབའ་ཞིག་ལ་བྱ་བར་མ་ཟད་ཀྱི་གང་ཟག་ལ་ཡང་བྱར་རུང་སྟེ། དེའི་སྐབས་སུ་ཡང་གསོལ་བ་བགྱིའོ་ཞེས་བྱ་བའི་ཚིག་ནི་བརྗོད་མི་དགོས་ཏེ། དེ་ནི་གསོལ་བའི་བདག་ཉིད་ཀྱི་ལས་ཀྱི་སྐབས་ཡིན་པའི་ཕྱིར་རོ། །རྗེས་སུ་བསྒྲག་གོ་ཞེས་བྱ་བའི་ཚིག་ནི་བརྗོད་པར་བྱ་སྟེ། །དེ་ནི་བྱིན་གྱིས་བརླབ་
པའི་སྔགས་དག་ལ་ཡང་རུང་བའི་ཕྱིར་རོ། །བཟློག་པའི་ཕྱིར་བསྒོ་བ་དང་། བསྙིལ་བ་དང་། དེའི་ངོ་བོ་ཉིད་ཚོལ་དུ་གཞུག་པ་དང་། བསླབ་བ་དང་མཐུན་པ་དང་། དེའི་གསོ་སྦྱོང་དང་། མུ་སྟེགས་ཅན་གྱི་གནས་དང་། དེ་ལས་གཞན་པའི་སྤོ་བ་དང་། མགུ་བ་དང་། ཡང་གཞི་ནས་བསླང་
བ་དང་། དྲན་པས་འདུལ་བ་དང་། མ་མོས་པས་འདུལ་བ་སྦྱིན་པ་དང་། བསྙེན་པར་རྫོགས་པར་བྱ་བ་དང་། མཚམས་དགྲོལ་བ་དང་། ནན་ཐུར་གྱི་ལས་དང་། དབྱུང་བ་དག་ལ་ནི་བརྗོད་པ་གསུམ་མོ་ཞེས་བྱ་བ་ནི། ལས་སུམ་ཅུ་བསྟན་ཏེ། དེ་དག་ནི་གསོལ་བ་དང་བཞིའི་ལས་ཡིན་ནོ་ཞེས་
བྱ་བར་སྦྱར་རོ། །དེ་ལས་བཟློག་པའི་ཕྱིར་བསྒོ་བས་ཞེས་བྱ་བ་ནི། དགེ་འདུན་འབྱེད་པ་ཉིད་ལ་སོགས་པའི་ཆོས་ལས་བཟློག་པའི་ཕྱིར་བསྒོ་བའི་ལས་བཅུ་བསྟན་ཏེ། དེ་དག་ཀྱང་གང་ཞེ་ན། འདི་ལྟ་སྟེ་དགེ་འདུན་གྱི་དབྱེན་བྱེད་པ་དང་། དེའི་རྗེས་སུ་ཕྱོགས་པ་དང་། ཁྱིམ་སུན་འབྱིན་པ་དང་།
