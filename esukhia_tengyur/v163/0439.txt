རོ། །དེ་དག་གི་སྔགས་ནི། ཚེ་དང་ལྡན་པ་དགོངས་སུ་གསོལ། བདག་དགེ་སློང་མིང་འདི་ཞེས་བགྱི་བའི་སྨན་ཐུན་ཚོད་དུ་རུང་བ་འདི་དག་བདག་དང་ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྣམས་ཀྱིར་བགྱི་བར་ཐུན་ཚོད་དུ་རུང་བར་བྱིན་གྱིས་བརླབ་བོ། །ལན་གསུམ་དུ་བཟླས། ཐབས་
ཡིན་ནོ། །ལེགས་སོ་ཞེས་གསུངས་པ་དང་། ཞག་བདུན་པ་དང་། འཚོ་བའི་བར་དུ་བཅང་བ་དག་ལ་ཡང་དེ་བཞིན་ནོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །དུས་གང་གི་ཚེ་བྱིན་གྱིས་བརླབ་པར་བྱ་ཞེ་ན། དེའི་ཕྱིར་ཟས་སྔ་མའི་ཚེའོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། གུང་ཚིགས་ཀྱི་མདུན་རོལ་དུ་ཟས་བཟའ་བ་ཐམས་ཅད་ཀྱི་
དུས་གང་ཡིན་པ་དེ་ནི་ཟས་སྔ་མལི་དུས་ཏེ་ཆོ་གའི་གཞི་ཞེས་བྱ་བ་ཡིན་པས་དུས་དེར་བྱིན་གྱིས་བརླབ་པར་བྱའོ། །ཇི་ལྟར་ཞེ་ན། དེའི་ཕྱིར་བྱིན་ལེན་བྱས་ཏེའོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། བྱིན་ལེན་བྱས་ཏེ་བྱིན་གྱིས་བརླབ་པར་བྱའོ། །བྱིན་ལེན་མ་བྱས་པ་དང་སོག་འཇོག་དང་འདྲེས་པ་བསྲུང་བར་
བྱ་བ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི། བྱིན་གྱིས་རློབ་པའི་ཚེ་ན་བྱིན་ལེན་མ་བྱས་པར་བྱིན་གྱིས་བརླབ་ཏུ་ཡང་མི་རུང་ལ། དྲི་མ་ལ་སོགས་པ་དང་འབགས་པས་སོག་འཇོག་དང་འདྲེས་པ་ལ་བྱིན་གྱིས་བརླབ་ཏུ་ཡང་མི་རུང་བས་དེའི་ཕྱིར་དེ་གཉིས་བསྲུང་བར་བྱ་བའོ། །དེའི་ཕྱིར་ལག་པ་གཉིས་བསྙལ་
ནས་སོ་ཞེས་བྱ་བ་ནི། གང་གི་ཕྱིར་བྱིན་ལེན་མ་བྱས་པ་དང་སོག་འཇོག་དང་འདྲེས་པ་བསྲུང་བར་བྱ་བ་ཡིན་པ་དེའི་ཕྱིར་ལག་པ་གཉིས་བསྙལ་ནས་བྱིན་ལེན་བྱས་ཏེ་བྱིན་གྱིས་བརླབ་པར་བྱ་བའོ། །དེས་ན་བྱིན་ལེན་མ་བྱས་པ་དང་སོག་འཇོག་དང་འདྲེས་པ་མེད་པར་འགྱུར་རོ། །ལག་ཉར་
བླངས་པ་དང་སོག་འཇོག་བྱས་པ་དང་བྱིན་ལེན་མ་བྱས་པ་དང་ནང་དུ་ཞག་ལོན་པ་དང་ནང་དུ་བཙོས་པ་དང་དགེ་སློང་གིས་བཙོས་པ་དག་ལ་བྱིན་གྱིས་བརླབ་པ་མི་འཆགས་སོ་ཞེས་བྱ་བ་ལ། འདིར་གཞུང་ནི་ཞུ་བ་ལས། བཙུན་པ་དུས་སུ་རུང་བ་དང་ཐུན་ཚོད་དུ་རུང་བ་དང་ཞག་བདུན་པ་
དང་འཚོའི་བར་དུ་བཅང་བའི་སྨན་གང་དག་སུག་ཉར་བླངས་པ་དང་བྱིན་ལེན་མ་བགྱིས་པ་དང་ནང་དུ་དགུང་ལོན་པ་དང་ནང་དུ་བསྒྲལ་བ་དང་རང་གིས་བསྒྲལ་བ་དང་དགེ་སློང་གིས་བསྒྲལ་བ་དེ་དག་ཡོངས་སུ་ལོངས་སྤྱོད་པར་རུང་ངམ། ཉེ་བ་འཁོར་མི་རུང་ངོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །
