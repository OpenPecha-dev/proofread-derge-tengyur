མི་འབྱུང་བར་བྱ་བའི་ཕྱིར་སྤང་ལེབ་དག་གོ་ཞེས་བྱ་བ་ནི་གཞུག་གོ་ཞེས་བྱ་བར་སྦྱར་རོ། །ཕྱོགས་གང་དུ་ཞེ་ན་དེའི་ཕྱིར་ཀ་བའི་བར་དག་ཏུའོ་ཞེས་བྱ་བ་སྨྲས་སོ། །ཅི་ཐམས་ཅད་དུའམ་ཞེ་ན་དེའི་ཕྱིར་སྣང་བར་བྱ་བའི་ཕྱོགས་མ་གཏོགས་པར་རོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། སྣང་བ་འཇུག་
པར་བྱ་བའི་ཕྱོགས་མ་གཏོགས་པར་རོ། །མེད་ན་སེག་དག་གམ་རེ་ལྡེ་དག་གོ་ཞེས་བྱ་བ་ནི། སྤང་ལེབ་དག་མེད་ན་དེར་སེག་དག་གམ་རེ་ལྡེ་དག་གཞུག་པར་བྱ་བའོ། །དབྱར་གྱི་ཟླ་བ་བཞི་པོ་དག་གི་མཐར་དབྱུང་ངོ་ཞེས་བྱ་བ་ནི། སྤང་ལེབ་ལ་སོགས་པའོ། །འཁོར་གྱི་ཁྱམས་བྱའོ་
ཞེས་བྱ་བ་ནི་ཁྱམས་ཟླུམ་པོ་བྱ་བའོ། །གནས་བསིལ་བར་བྱ་བའི་ཕྱིར་ཕྱི་རོལ་རྩིག་པས་བསྐོར་རོ་ཞེས་བྱ་བ་ནི། གནས་བསིལ་བར་བྱ་བའི་ཕྱིར་ནང་དུ་ཀ་བའི་ཚར་གཟུགས་ཤིང་ཕྱི་རོལ་རྩིག་པས་བསྐོར་བར་བྱ་བ་དང་། དབུས་སུ་རྫིང་བུ་གཞག་པར་བྱའོ། །ཀ་བའི་ཚར་དག་གཟུགས་སོ་
ཞེས་བྱ་བ་ནི་ནང་དུའོ། །སྐར་ཁུང་གདོད་དོ་ཞེས་བྱ་བ་ནི། དེར་སྐར་ཁུང་མང་པོ་དག་གདོད་པར་བྱ་བའོ། །མ་མཐའ་དག་སྟན་གྱི་ཡ་མཐའ་དང་ཐད་ཀར་བབ་པར་བྱའོ་ཞེས་བྱ་བ་ནི། སྐར་ཁུང་དེ་དག་གི་མ་མཐའ་དག་ཁྲི་ལ་སོགས་པའི་སྟན་ཚད་མ་དག་གི་ཡ་མཐའ་དང་ཐད་ཀར་བབ་པར་
འགྱུར་བ་དེ་ལྟར་འཁོར་གྱི་ཁྱམས་དེར་སྐར་ཁུང་མང་པོ་དག་གདོད་པར་བྱ་བའོ། །དྲ་བ་དང་སྒོ་གླེགས་དག་གཞུག་པའོ་ཞེས་བྱ་བ་ནི། སྐར་ཁུང་དག་ལའོ། །ཟས་དང་མེ་དང་དཔོངས་དང་འཆག་སའི་ཁང་པ་དག་བྱའོ་ཞེས་བྱ་བ་ནི། ཟས་ཀྱི་ཁང་པ་དང་མེའི་ཁང་པ་དང་དཔོང་སའི་ཁང་པ་
དང་འཆག་སའི་ཁང་པ་དག་བྱ་བའོ། །རྩིག་ངོས་ཀྱི་ཕྱོགས་འཁོར་བར་གློ་འབུར་དག་བྱའོ་ཞེས་བྱ་བ་ནི། གཙུག་ལག་ཁང་གི་རྩིག་ངོས་ཀྱི་ཕྱོགས་སུ་རྒྱུན་ལུགས་ཀྱི་ཕན་ཆད་དུ་གཙུག་ལག་ཁང་གི་རྩིག་པ་ལ་བརྟེན་ཏེ་གནས་ཀྱི་ཆེད་དུ་ཁང་བུ་གཡོགས་མ་དང་བཅས་པ་དག་བྱའོ། །
བསྟི་གནས་ཞེས་བྱ་བ་ནི་ཁྱམས་ཀྱི་ངོ་བོ་ཉིད་ཡིན་པས་འདི་དང་ཐ་དད་དོ། །ས་ཁྱིམ་རུང་ངོ་ཞེས་བྱ་བ་ནི། དགེ་སློང་དག་གི་གནས་སུ་ས་ཕུག་རུང་བའོ། །དགྲ་གང་ཡང་ངོ་ཞེས་བྱ་བ་ཡང་། དགེ་སློང་དག་གི་གནས་སུ་རུང་བ་སྟེ། འདིར་དགྲ་གང་ཞེས་བྱ་བ་ནི་གཙུག་ལག་
