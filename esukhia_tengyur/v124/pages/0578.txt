འཐོབ་པ་ཆེན་པོ་ནི་སྟོབས་དང་མི་འཇིགས་པ་དང་། སངས་རྒྱས་ཀྱི་ཆོས་མ་འདྲེས་པ་ལ་སོགས་པ་ཡོན་ཏན་དཔག་ཏུ་མེད་པ་གྲངས་མེད་པ་འཐོབ་པའི་ཕྱིར་རོ། །ལམ་ཆེན་པོ་ནི་འཁོར་བ་ཇི་སྲིད་པ་དེ་སྲིད་དུ་བྱང་ཆུབ་མངོན་པར་སྟོན་པས་སངས་རྒྱས་ཀྱི་མཛད་པ་སྒྲུབ་པའི་ཕྱིར་རོ། །འདི་དག་ཉིད་ནི་ཐེག་
པ་ཆེན་པོ་བླ་ན་མེད་པ་སྟེ། དེ་ལས་ཐེག་པ་ཁྱད་པར་ཅན་གཞན་མེད་པའི་ཕྱིར་རོ། །དེ་ཡང་སྒྲུབ་པ་དང་དམིགས་པ་དང་ཡང་དག་པར་འགྲུབ་པ་སྟེ། འདི་ལྟར་དེ་དག་དང་ལྡན་པས་ན་ཐེག་པ་ཆེན་པོ་བླ་ན་མེད་པ་ཞེས་བྱའོ། །སྒྲུབ་པ་ཉིད་བླ་ན་མེད་པ་ཡིན་པས་སྒྲུབ་པ་བླ་ན་མེད་པ་ཉིད་དེ། དེ་
བཞིན་དུ་དམིགས་པ་དང་ཡང་དག་པར་འགྲུབ་པ་བླ་ན་མེད་པ་ཉིད་ཀྱང་རིག་པར་བྱའོ། །བླ་ན་མེད་པ་ཉིད་དེ་དག་ནི་བྱང་ཆུབ་སེམས་དཔའི་ལམ་དང་། དེའི་གནས་དང་དེའི་འབྲས་བུའི་དབང་དུ་བྱས་ཏེ། གོ་རིམས་བཞིན་དུ་རྣམ་པར་གཞག་གོ། །འདི་ལྟ་སྟེ། སངས་རྒྱས་ཀྱི་གསུང་རབ་ཀྱིས་ཀྱང་
སྒྲུབ་པ་དང་དམིགས་པ་དང་དེའི་འབྲས་བུ་དང་འདི་རྣམས་ཙམ་ཞིག་བཤད་པར་ཟད་དོ། །དམིགས་པ་ནི་སྒྲུབ་པའི་སྔོན་དུ་འགྲོ་བས་དམིགས་པ་སྔར་བཤད་པའི་རིགས་པ་ལས། བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ལ་སྒྲུབ་པ་སྙིང་པོར་བྱ་བ་བསྟན་པའི་ཕྱིར་སྒྲུབ་པ་སྔར་བཤད་དོ། །དེ་ལ་ཇི་ལྟར་ན་
སྒྲུབ་པ་བླ་ན་མེད་པ་ཉིད་ཅེས་བྱ་ཞེ་ན། དེའི་ཕྱིར། སྒྲུབ་པ་བླ་ན་མེད་པ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་བཅུ་ལ་སྒྲུབ་པར་རིག་པར་བྱའོ་ཞེས་བྱ་བ་སྨོས་སོ། །ཕ་རོལ་ཏུ་ཕྱིན་པ་རྣམས་སྒྲུབ་པ་ནི་ཕ་རོལ་ཏུ་ཕྱིན་པ་རྣམས་ལ་ནན་ཏན་བྱེད་པའོ། །ཕ་རོལ་ཏུ་ཕྱིན་པ་སྒྲུབ་པ་དེ་ཡང་ཐེག་པ་ཆེན་པོ་ཉིད་དུ་བཤད་
ཀྱི། གཞན་ནི་མ་ཡིན་པས་ན་ཐེག་པ་ཆེན་པོ་ཉིད་བླ་ན་མེད་པར་སྟོན་ཏོ། །སྒྲུབ་པ་དེ་ཡང་རྣམ་པ་གང་གིས་ཕ་རོལ་ཏུ་ཕྱིན་པ་རྣམས་ལ་སྦྱོར་བ་ནི་བླ་ན་མེད་པར་འགྱུར་བའི་རྣམ་པ་དེ་དག་བསྟན་པའི་ཕྱིར། སྒྲུབ་པ་དག་ནི་རྣམ་དྲུག་སྟེ། །ཞེས་བྱ་བ་གསུངས་སོ། །ཕ་རོལ་
ཏུ་ཕྱིན་པ་རྣམས་ལ་སྒྲུབ་པ་ཇི་ལྟར་ན་དམ་པར་འགྱུར་བ་ནས། ཇི་ལྟར་ན་ཁྱད་པར་ཅན་མེད་པའི་བར་དུ་འགྱུར་བ་མི་ཤེས་ནས། དེའི་ཕྱིར་དེ་ལ། དམ་པ་བཅུ་གཉིས་བདག་ཉིད་དེ། །ཞེས་བྱ་བ་སྨོས་སོ། །དམ་པ་ནི་རྣམ་པ་བཅུ་གཉིས་ཏེ། ཆེ་བ་དང་བླ་ན་མེད་ཅེས་བྱ་བའི་
