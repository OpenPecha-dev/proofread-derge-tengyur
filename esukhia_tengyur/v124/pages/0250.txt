སྐྱོ་བ་གང་ཡིན་པ་སྟེ། དེ་ནི་བསྐལ་པ་འདི་སྙེད་ཅིག་ཏུ་གོ་བགོའོ་ཞེས་བསྐལ་པ་བགྲང་བས་བྱང་ཆུབ་ཚོལ་བར་མི་བྱེད་ཀྱི། འོན་ཀྱང་བསམ་གྱིས་མི་ཁྱབ་པ་ཁོ་ནར་གོ་གྱོན་པར་བྱེད་དོ་ཞེས་བྱ་བ་དེ་ལྟ་བུ་ལ་སོགས་པ་གསུངས་པ་ལྟ་བུ་སྟེ། འདི་ནི་བརྩོན་འགྲུས་ཀྱི་རྒྱ་ཆེ་བ་ཉིད་བཤད་
དོ། །རང་དང་གཞན་གྱི་ཉོན་མོངས་ཚོགས་གཞོམ་བྱང་ཆུབ་དམ་པ་འཐོབ་པར་བྱེད། །ཅེས་བྱ་བ་འདིས་ནི་ཟང་ཟིང་མེད་པ་ཉིད་ཀྱི་སྦྱོར་བའི་བརྩོན་འགྲུས་སྟོན་ཏོ། །ཇི་ལྟར་ཟང་ཟིང་མེད་པ་ཡིན་ཞེ་ན། རང་དང་གཞན་གྱི་ཉོན་མོངས་པ་ཡང་དག་པར་གཞོམ་པའི་ཕྱིར་དང་། བླ་ན་མེད་
པའི་བྱང་ཆུབ་ཀྱི་ཕྱིར་བརྩོན་འགྲུས་རྩོམ་པར་བྱེད་དོ། །དེ་ནི་སྲེད་པས་བདག་གིར་མ་བྱས་པའི་ཕྱིར་ཉོན་མོངས་པ་གཏན་དུ་ཡང་དག་པར་འཇོམས་པས་ཟང་ཟིང་མེད་པ་ཡིན་ནོ། །བརྩོན་འགྲུས་གང་ཞིག་བདག་གིས་དངོས་པོ་དང་། ལྟ་བ་སྲེད་པས་བདག་གིར་བྱས་པ་དེ་ནི་ཟང་ཟིང་དང་
བཅས་པ་ཡིན་ནོ། །འདི་ཇི་ལྟར་སྦྱོར་བའི་བརྩོན་འགྲུས་ཡིན་ཞེ་ན། ཇི་སྐད་དུ་འཕགས་པ་བློ་གོས་མི་ཟད་པའི་མདོ་ཁོ་ན་ལས་སྦྱོར་བའི་བརྩོན་འགྲུས་ཀྱི་དབང་དུ་མཛད་དེ། དེ་ལ་བྱང་ཆུབ་སེམས་དཔའི་ཕ་རོལ་གནོན་པ་མི་ཟད་པ་གང་ཞེ་ན། གང་སྟོང་གསུམ་གྱི་སྟོང་ཆེན་པོའི་འཇིག་རྟེན་
གྱི་ཁམས་མེས་ཡོངས་སུ་གང་བ་ཡང་བརྒལ་ནས་དེ་བཞིན་གཤེགས་པ་ལ་བལྟ་བར་སྤྲོ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཕ་རོལ་གནོན་པ་ཡིན་ནོ། །དམ་པའི་ཆོས་མཉན་པའི་རྒྱུ་ཅན་གྱི་སྤྲོ་བ་ནི་བྱང་ཆུབ་སེམས་དཔའི་ཕ་རོལ་གནོན་པ་ཡིན་ནོ། །སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱེད་པའི་རྒྱུ་ཅན་གྱི་སྤྲོ་བ་
ནི་བྱང་ཆུབ་སེམས་དཔའི་ཕ་རོལ་གནོན་པ་མི་ལྷོད་པ། མི་ཞན་པ། བརྟན་པ། མི་བཟློག་པ། དགེ་བའི་བསམ་པ་ཅན། སྙིང་རྗེ་ཆེན་པོ་ལ་བརྟེན་པ་ཞེས་བྱ་བ་དེ་ལྟ་བུ་ལ་སོགས་པ་ཡིན་ཏེ། དེས་ན་ཕ་རོལ་གནོན་པ་ཞེས་བྱའོ་ཞེས་གསུངས་པ་ལྟ་བུ་ཡིན་ནོ། །རྒྱལ་བའི་སྲས་རྣམས་ཏིང་འཛིན་
མང་ལྡན་བསམ་གཏན་རྣམ་པ་ཐམས་ཅད་བསྒྲུབས། །ཞེས་བྱ་བ་འདིས་ནི་ཏིང་ངེ་འཛིན་མང་བ་ཉིད་དང་། རྣམ་པ་ཐམས་ཅད་མཁྱེན་པ་ཉིད་ཀྱི་བསམ་གཏན་གྱི་རྒྱ་ཆེ་བ་ཉིད་སྟོན་ཏེ། ཇི་སྐད་དུ་ཆོས་མངོན་པ་ཀུན་ལས་བཏུས་པ་ལས་འདི་ལྟ་སྟེ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་
