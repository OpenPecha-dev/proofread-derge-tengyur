གཞན་ཡང་དངོས་པོ་སྐྱེས་པ་ཉིད་ཡིན་ན་གཞན་ཞེས་བྱ་བའི་ཐ་སྙད་གདགས་སུ་རུང་ན་འབྲས་བུ་དེ་ནི་མ་སྐྱེས་པས་འབྲས་བུ་དེ་ལས་རྒྱུ་གཞན་ཉིད་དུ་རྟོག་པ་ནི་ཉེ་བར་བཏགས་པ་ཡིན་པས། དེའི་ཕྱིར། མ་སྐྱེས་པ་ཡི་གཞན་ཉིད་ནི། །ཀུན་རྫོབ་ཡིན་པས་དོན་དམ་
མིན། །གཞན་ཡང་སྐྱེ་བ་ཉིད་ཀྱང་ནི། །རིགས་པ་མ་ཡིན་གཟུང་བར་བྱ། །ཞེས་བྱ་བ་སྨྲས་སོ། །གཞན་ཡང་སྐྱེ་བར་སྨྲ་བ་དག་ལས། །ཁ་ཅིག་ནི་རྒྱུ་ལས་འབྲས་བུ་ཡོད་པ་ཉིད་ཡིན་ལ། དེ་སྐྱེས་པ་ཉིད་དམ་མ་སྐྱེས་པ་ཉིད་དུ་ནི་ཐ་སྙད་གདགས་པར་བྱ་བ་མ་ཡིན་པར་
སེམས་སོ། །ཁ་ཅིག་ནི་མ་སྐྱེས་པ་རྒྱུ་དང་རྐྱེན་རྣམས་ཀྱི་མངོན་པར་སྒྲུབ་པར་འདོད་དོ། །ཁ་ཅིག་ནི་སྐྱེས་བཞིན་པ་གཉི་གའི་བདག་ཉིད་ཅན་དུ་འདོད་དོ། །ཁ་ཅིག་ནི་བརྗོད་པར་བྱ་བ་མ་ཡིན་པ་ཉིད་དུ་འདོད་དེ། འདི་ལྟར་དེ་ནི་སྐྱེས་པ་མ་ཡིན་ཏེ། བདག་ཉིད་ཀྱི་དངོས་
པོ་མ་ཐོབ་པའི་ཕྱིར་རོ། །མ་སྐྱེས་པ་ཡང་མ་ཡིན་ཏེ། །བྱ་བ་འཇུག་པའི་ཕྱིར་རོ་ཞེས་ཟེར་རོ། །དེ་ལ་བཤད་པ་རེ་ཞིག །སྐྱེས་པ་བརྗོད་དུ་མེད་པ་ལ། །སྐྱེ་བ་དོན་མེད་ཉིད་དུ་འགྱུར། །ཞེས་བྱ་བ་ནི། གལ་ཏེ་རྒྱུ་ལ་འབྲས་བུ་ཡོད་པ་ཉིད་ཡིན་ཞིང་ཐ་སྙད་གདགས་
པར་བྱ་བ་མ་ཡིན་པས། དེ་ནི་སྐྱེས་པ་ཡིན་པས་སྐྱེས་པ་དེ་ལ་ཡང་སྐྱེ་བ་དོན་མེད་པ་ཉིད་དུ་ཡང་འགྱུར་ལ། །མ་སྐྱེས་པ་ན་ཐ་སྙད་གདགས་པར་བྱ་བ་མ་ཡིན་པ་ནི་སྐྱེས་ན་ཡང་དེ་འབྲས་བུའི་ངོ་བོ་ཉིད་དུ་ཐ་སྙད་གདགས་སུ་མི་རུང་བར་འགྱུར་རོ། །གང་དག་འབྲས་བུ་མ་སྐྱེས་
པ་དེ་དག་ལ་ཡང་། །མ་སྐྱེས་པ་ལ་སྐྱེ་མེད་དེ། །མ་སྐྱེས་ཕྱིར་ནི་མཁའ་མིན་བཞིན། །ཞེས་བྱ་བ་སྨྲས་སོ། །སྐྱེ་བཞིན་སྐྱེ་བར་འདོད་པ་ལ། །བརྒལ་ཞིང་བརྟགས་པ་མཚུངས་པར་འགྱུར། །ཞེས་བྱ་བ་ནི་སྐྱེ་བཞིན་པ་ལ་ཡང་གང་སྐྱེས་པའི་ཆ་དེ་ལ་ནི་ཡང་
སྐྱེ་བ་དོན་མེད་ལ། གང་མ་སྐྱེས་པའི་ཆ་དེ་ནི་མེད་པའི་ཕྱིར། ནམ་མཁའི་མེ་ཏོག་ལ་སོགས་པ་བཞིན་དུ་སྐྱེ་བ་མེད་དོ། །བརྗོད་དུ་མེད་པའང་དེ་མེད་དེ། །བརྗོད་དུ་མེད་ཕྱིར་གཞན་བཞིན་ནོ། །ཞེས་བྱ་བ་ལ་འདིར་གཞན་བཞིན་ནོ་ཞེས་བྱ་བའི་སྒྲས་ནི་སྔོ་བསངས་དང་
