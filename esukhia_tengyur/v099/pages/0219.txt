འགྱུར་རོ་ཞེས་ཟེར་བས། དེས་ན་ཁྱིམ་པ་དག་ཉིན་རེ་བཞིན་ཁ་ཟས་ཀྱི་གཡོས་བྱེད་པ་གང་ཡིན་པ་དེ་ཡང་བུ་དང་། ཆུང་མ་དང་། ལྷ་དང་། མགྲོན་དང་། དགེ་སྦྱོང་ལ་སོགས་པ་དང་། གཏོར་མ་དང་། སློང་བ་ལ་སྦྱིན་པ་ལ་སོགས་པ་དང་། ཐུན་
མོང་ཁོ་ནར་གཡོས་བྱེད་པ་ཡིན་ལ། འདིར་ཡང་འབྲུ་དང་། ཆུ་དང་། ཤིང་དང་། ལོ་མ་དང་། ཡལ་ག་ལ་སོགས་པ་ལ་སེམས་ཅན་གང་དག་བརྟེན་པ་དེ་དག་གསོད་པར་བྱེད་པས་དེར་སློང་མོ་ལེན་པ་དང་། ཟ་བ་བསོད་ནམས་མ་ཡིན་པ་དང་འབྲེལ་པས་ཐར་པ་མེད་
ཅིང་ངན་སོང་དུ་འགྲོ་བར་འགྱུར་བའི་ཕྱིར་དང་། ཕྱོགས་ཀྱི་གོས་ཅན་དག་གི་ཕྱིར་འདུག་གནས་གང་དག་བྱེད་པ་དང་། མུ་སྟེགས་བྱེད་དག་གི་ཕྱིར་ལྷ་ཁང་གང་དག་བྱེད་པ་དེར་ཤིང་དང་ཕ་གུ་དང་། འཇིམ་པ་དང་། ཆུ་ལ་སོགས་པ་ཉེ་བར་སྟོབས་པ་དག་ཏུ་ཡང་སྲོག་ཆགས་
མཐའ་ཡས་མུ་མེད་པ་དག་འཆི་བར་འགྱུར་བས། གལ་ཏེ་ལོངས་སྤྱད་ན་བསོད་ནམས་མ་ཡིན་པ་སྐྱེ་བས་ཕྱོགས་ཀྱི་གོས་ཅན་དག་དང་། མུ་སྟེགས་བྱེད་དག་ལ་ཐར་པ་མེད་ཅིང་ངན་སོང་དུ་འགྲོ་བར་འགྱུར་བ་ལྟ་ན། ཁྱེད་ཅག་གིས་དེར་ཡང་འཇུག་པར་མི་བྱ་བའི་ཕྱིར་རོ། །
གཞན་ཡང་རྫིང་བུ་དང་། ལྟེང་ཀ་དང་། ཁྲོན་པ་དང་། ཆུ་སྦྱིན་པའི་ཁང་པ་དག་ཏུ་ཡང་སྲོག་ཆགས་དག་སྐྱེ་ཞིང་འཆི་བས་དེ་ཡང་བཏང་དགོས་ལ། དེས་ན་ཕྱོགས་ཀྱི་གོས་ཅན་དག་ཆུ་དང་ཁ་ཟས་མིད་པར་བྱེད་པའི་ཆུ་དེ་དག་ཏུ་ཡང་སྲིན་བུའི་རིགས་བརྒྱ་སྟོང་ཕྲག་དུ་མ་དག་སྐྱེ་
ཞིང་འཆི་བར་འགྱུར་བས་དེར་ཡང་བསོད་ནམས་མ་ཡིན་པ་དང་འབྲེལ་པས་ཁྱེད་ལ་ཐར་པ་མེད་ཅིང་ངན་སོང་དུ་འགྲོ་བར་འགྱུར་བའི་ཕྱིར་རོ། །གཞན་ཡང་ཁྱེད་ན་རེ། ས་དང་ཆུ་དང་། མེ་དང་རླུང་དང་། ཤིང་ལ་སོགས་པ་ནི་རེག་པའི་དབང་པོ་དང་ལྡན་པའི་ཕྱིར། དབང་པོ་གཅིག་པ་
སྲོག་དང་ལྡན་པ་ཡིན་ནོ་ཞེས་ཟེར་བས་དེའི་ཕྱིར་ཁྱེད་ཀྱིས་བཟའ་བ་དང་། བཅའ་བ་དག་ཟོས་པ་ན་སྲོག་ཆགས་ཀྱི་ཡན་ལག་ཟོས་པར་འགྱུར་བས་ཁྱེད་ལ་ཐར་པ་མེད་ཅིང་ངན་སོང་དུ་འགྲོ་བར་འགྱུར་བའི་ཕྱིར་རོ། །གཞན་ཡང་ཁྱོད་ཀྱིས་བསྟན་པའི་ཆོས་མ་ཡིན་པ་ནི་ངན་སོང་དུ་འགྲོ་
