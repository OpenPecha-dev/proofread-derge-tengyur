ཆ་ནི་སྦྱིན་སྲེག་འབུམ་ཕྲག་བཅུའོ། །དེ་ལྟར་བཟླས་བརྗོད་དང་བཅས་པའི་སྦྱིན་སྲེག་གིས་འགྲུབ་པར་འགྱུར་ཏེ། གཞན་དུ་ན་དོན་མེད་པ་ཁོ་ནར་འགྱུར་རོ། །དེ་ཡང་ལྷ་གང་དང་གང་ཡིན་པའི་བཟླས་པ་དང་། བགྲང་ཕྲེང་ལ་སོགས་པ་རྒྱུ་མཐུན་པས་འགྲུབ་པ་ནི་ཆོས་ཉིད་ཀྱི་བདེན་པའོ། །གལ་ཏེ་
བསྒྲུབ་བྱའི་མིང་གི་ཡི་གེ་དང་པོའི་ཡི་གེ། །སྒྲུབ་པ་པོའི་མིང་གི་ཡི་གེ་དང་པོ་དེའི་ཡི་གེ་དགྲར་འགྱུར་ཏེ་དེའི་ཚེ་སྒྲུབ་པ་པོ་འཆི་བར་འགྱུར་རོ། །འོན་ཏེ་བྲན་དུ་གྱུར་ན་དེའི་ཚེ་ཉོན་མོངས་པར་འགྱུར་རོ། །འོན་ཏེ་བཤེས་གཉེན་དུ་གྱུར་ན་དེའི་ཚེ་གྲུབ་པར་འགྱུར་རོ། །ལྷའི་མིང་གི་དང་པོ་དབྱངས་ཡིག་
དགྲར་གྱུར་ན་སྒྲུབ་པ་པོ་འཆི་བར་འགྱུར་རོ། །གལ་ཏེ་གསལ་བྱེད་དགྲར་གྱུར་ན་ནད་ཀྱིས་ཐེབས་པར་འགྱུར་རོ། །གཞན་དགྲར་གྱུར་པ་ནི་རྩ་བའི་རྒྱུད་ལས་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ། གང་ཞིག །ཨ་ཀུ་ཧ་ྈྐ་ནི་མགྲིན་གང་། །དབྱངས་དང་གསལ་བྱེད་མཚན་ཉིད་ཅན། །སྟོང་པར་
རླུང་སོགས་ཁམས་རྣམས་ཀྱི། །མཛའ་བོ་ཉིད་ཀྱིས་རྟག་ཏུ་གནས། །ཨི་ཙུ་ཡ་ཤ་ཡང་ནི་རྐན། །དབྱངས་དང་གསལ་བྱེད་མཚན་ཉིད་དེ། །རླུང་གི་ཁམས་ལས་ཡང་དག་བྱུང་། །ཆུ་ལས་སྐྱེས་པའི་དགྲ་རྣམས་སོ། །རྀ་ཊུ་ར་ཥ་སྤྱི་བོ་རུ། །དབྱངས་དང་གསལ་བྱེད་མཚན་ཉིད་དེ། །མེ་
ཁམས་ཡང་དག་བྱུང་བ་རྣམས། །ས་ལས་སྐྱེས་པ་རྣམས་ཀྱི་དགྲ། །ཨུ་པུ་ཝ་ྉྤ་མཆུ་ཅན་གང་། །དབྱངས་དང་གསལ་བྱེད་མཚན་ཉིད་དེ། །ཆུ་ཁམས་ཡང་དག་བྱུང་བ་རྣམས། །མེ་ལས་སྐྱེས་པ་རྣམས་ཀྱི་དགྲ །ལྀ་ཏུ་ལ་ས་སོ་ཅན་རྣམས། །དབྱངས་དང་གསལ་བྱེད་མཚན་ཉིད་དེ། །
ས་ཁམས་ཡང་དག་བྱུང་བ་རྣམས། །རླུང་ལས་སྐྱེས་པ་རྣམས་ཀྱི་དགྲ། །རླུང་གི་གྲོགས་ནི་སྟོང་པ་ཉིད། །ཆུ་ནི་ནུས་པ་མེད་ཕྱིར་བྲན། །ཆུ་ཡི་གྲོགས་པོ་ས་ཉིད་དེ། །མེ་ཡི་ནུས་པ་མེད་ཕྱིར་བྲན། །མེ་ཡི་གྲོགས་པོ་རླུང་ཉིད་དེ། །ས་ཡི་ནུས་པ་མང་ཕྱིར་བྲན། །ས་ཡི་གྲོགས་ནི་ཆུ་ཉིད་དེ། །
རླུང་གི་བྲན་དུ་གྱུར་པའོ། །སྔགས་ཀྱི་དང་པོ་ཡི་གེ་རིགས། །ཕྱག་འཚལ་བ་ནི་མ་གཏོགས་ཏེ། །སེམས་དེ་ཉིད་ནི་སྔགས་རྣམས་ཀྱི། །སྐུ་ནི་སྐྱེད་པར་བྱེད་པའི་རྒྱུ། །ཐིག་ལེ་དང་ནི་ཡང་དག་ལྡན། །སྔགས་ཀྱི་དང་པོའི་ཡི་གེ་གང་། །སྔ་ཕྱི་ལས་ནི་དང་པོ་གཟུང་། །དང་པོ་བརྗོད་
