མོངས་པ་དེ་དག་གིས་ཇི་ལྟར་བྱ་སྟེ། ཉོན་མོངས་པའི་རྒྱུ་ལས་བྱུང་བ་མ་ཡིན་པས། ཕ་རོལ་པོའི་གཏན་ཚིགས་ལས་ལ་རྒྱུ་ཡོད་པའི་ཕྱིར་རོ་ཞེས་བྱ་བ་དེ་མ་གྲུབ་པས་དོན་དམ་པར་ལས་ལ་ངོ་བོ་ཉིད་ཡོད་པ་མ་ཡིན་ནོ་ཞེས་བྱ་བར་བསྟན་ཏོ། །དེ་ཉིད་བྱེ་བྲག་ཏུ་བཤད་པ་བསྟན་པའི་ཕྱིར། ལས་
འདི་ཉོན་མོངས་བདག་ཉིད་ན། །ཞེས་བྱ་བ་ནི། ལས་འདི་ཉོན་མོངས་པའི་རྒྱུ་ལས་བྱུང་བ་ཡིན་ན་ཞེས་བྱ་བའི་ཐ་ཚིག་སྟེ། དཔེར་ན་བུམ་པ་འཇིམ་པའི་བདག་ཉིད་ཡིན་ན་ཞེས་བྱ་བ་ལྟ་བུའོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། དེས་ནི་དེ་ལྟ་ན་ངོ་བོ་ཉིད་མེད་དོ་ཞེས་དགག་པ་བསྟན་ཏོ། །ཅིའི་ཕྱིར་དེ་ལྟ་ན་ངོ་བོ་ཉིད་མེད་ཅེ་ན་
དེའི་ཕྱིར། །ཉོན་མོངས་དེ་དག་ཡང་དག་མིན། །ཞེས་བྱ་བ་ནི་ཉོན་མོངས་པའི་ངོ་བོ་ཉིད་མེད་དོ་ཞེས་བྱ་བའི་ཐ་ཚིག་སྟེ། སྐྱེ་བ་བཀག་པའི་ཕྱིར་དང་། འདོད་ཆགས་དང་། ཆགས་པ་བརྟག་པར་དེའི་ངོ་བོ་ཉིད་བཀག་པའི་ཕྱིར་རམ། ཡང་དག་སྡུག་དང་མི་སྡུག་པའི། །ཕྱིན་ཅི་ལོག་ལ་རྟེན་འབྱུང་
བ། །དེ་དག་ངོ་བོ་ཉིད་མེད་པས། །དེ་ཕྱིར་ཉོན་མོངས་ཡང་དག་མིན། །ཞེས་བྱ་བ་འགོག་པར་འགྱུར་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། ཉོན་མོངས་པ་དག་ངོ་བོ་ཉིད་མེད་པས། ཉོན་མོངས་པའི་བདག་ཉིད་ལས་ཀྱང་ངོ་བོ་ཉིད་མེད་དོ། །དེ་ཉིད་བསྟན་པའི་ཕྱིར། དེའི་ཕྱིར་དེ་ལྟར་གལ་ཏེ་ཉོན་མོངས་
པ་དེ་དག་ཡང་དག་པར་ངོ་བོ་ཉིད་མེད་པ་ཡིན་ན། ལས་དེ་དག་གིས་བྱས་པར་ཇི་ལྟར་འགྱུར་ཏེ། ལས་ཀྱི་རྒྱུ་ཉོན་མོངས་པ་མ་ཡིན་ནོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ་ཞེས་བྱ་བ་སྨྲས་སོ། །དེའི་ཕྱིར་གཏན་ཚིགས་ཀྱི་དོན་མ་གྲུབ་པ་ཉིད་དམ། དོན་འགལ་བ་ཉིད་དོ་ཞེས་བྱ་བ་ནི་མཇུག་བསྡུ་བ་དང་། ཚད་མའི་འབྲས་བུ་
ཡིན་ཏེ། དེ་ལྟར་ཉོན་མོངས་པ་དེ་དག་ཡང་དག་པ་མ་ཡིན་པ་དེའི་ཕྱིར། ཕ་རོལ་པོ་དག་གི་གཏན་ཚིགས་ལས་ཀྱི་རྒྱུ་ཡོད་པའི་ཕྱིར་རོ་ཞེས་བྱ་བ་དེ་དོན་མ་གྲུབ་པ་ཉིད་དམ། འགལ་བ་ཉིད་ཡིན་པས། དོན་དམ་པར་ལས་ལ་ངོ་བོ་ཉིད་ཡོད་པར་མ་གྲུབ་པོ།། །།བམ་པོ་ལྔ་བཅུ་
བརྒྱད་པ། །འདིར་སྨྲས་པ། དོན་དམ་པར་ཉོན་མོངས་པ་རྣམས་ནི་ཡོད་པ་ཁོ་ན་ཡིན་ཏེ། དེ་དག་གི་འབྲས་བུ་ཡོད་པའི་ཕྱིར་རོ། །འདི་ན་གང་མེད་པ་དེ་ལ་ནི་འབྲས་བུ་གཟུང་དུ་མེད་དེ། དཔེར་ན་འོན་པའི་རྣ་བའི་དབང་པོའི་འབྲས་བུ་རྣ་བའི་རྣམ་པར་ཤེས་པ་བཞིན་ནོ། །ཉོན་མོངས་པ་རྣམས་ལ་
