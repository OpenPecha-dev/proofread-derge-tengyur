མཁྲིས་པ་སྐྱེའོ། །བྱ་ན་ཤ་སོ་སོའི་བྱེ་བྲག་གི་ཕན་གནོད་ཅི་འདྲ་བ་དེ་འདྲ་བའི་ཚིལ་ཡང་དེ་དང་འདྲའོ། །བྱ་ན་ཤ་སོ་སོའི་བྱེ་བྲག་ནི་ཡུལ་སྐམ་ས་དང་སེར་ཆེས་ནས་བྱུང་བ་རྣམ་པ་གཉིས་ཏེ། སྤྱིའི་ཡུལ་སྐམ་ས་དང་སེར་ཆེ་ས་ནས་བྱུང་བའི་ཕན་གནོད་བསྟན་པ་བཞིན་དུ་
ཤེས་པར་བྱའོ། །ཏིལ་མར་དང་ཨེ་རཎྜའི་མར་དང་། ཡུངས་མར་དང་། བ་རུ་རའི་མར་དང་ཟར་མའི་མར་དང་། ལེ་བརྒན་རྩིའི་མར་གྱི་ཕན་གནོད་བཤད་པའི་ལེའུ་རྫོགས་སོ།། །།ཆང་སྣ་ཚོགས་ཀྱི་ཕན་གནོད་ཀྱི་བྱེ་བྲག་བཤད་པ་ཆང་ངོ་ཅོག་ནི་ཤ་འཕར། ཡི་ག་
འབྱེ། །དྲོད་ཆེ། རྣོ། དགའ་བར་བྱེད། ཁོང་བརྟ། ཆང་གི་སྤྱིའི་རོ་ནི་མངར་ལ་ཁ། བསྐ། རྣོ། འཐུངས་ན་ཞུ་བའི་རྗེས་ནི་སྐྱུར་ལ་ཅུང་ཟད་ཁོང་འཁྲུ། མགུལ་པ་སྙན་པར་འགྱུར། རན་པར་འཐུངས་ན་ཅི་ལ་ཡང་ཕན་ཏེ་ནད་མེད། འཐུངས་པའི་མོད་ལ་སྤོབས་པ་ཆེ། །བཞིན་
མདོག་འགྱུར། ཁོང་ཡང་། གཉིད་མེད་པ་ནི་གཉིད་འོང་། །གཉིད་ཧ་ཅང་ཆེ་བ་ནི་རན་པར་འཐུངས་ན་གཉིད་ཆུང་བར་འགྱུར། ཆང་གི་ཉན་ནི་མཁྲིས་པ་དང་ཁྲག་སྐྱེ་བའི་ནད་དུ་འགྱུར་ཏེ་གནོད། ཚོད་ཟིན་ཅིང་རན་པར་འཐུངས་ན་ཤ་སྐྱེ་བ་དང་ཤ་ཆེ་ཞིང་སྦོམ་པ་གཉིས་ལ་ཕན།
ཆང་འཐུངས་ན་བཅུད་རྣོ་ཞིང་མགྱོགས་ཏེ་ལུས་ཀུན་ཏུ་ཁྱབ། ཅུང་ཟད་རྩུབ། ལུས་ཀྱི་སྒོ་དགུ་ཆར་གྱི་རྒྱུན་སྲིང་གི་ནད་དང་དྲི་མ་སུན་ཏུ་སྦྱོང་ཞིང་སེལ་ཏ། དྲི་མ་ཀུན་ལེགས་པར་འབྱུང་བར་འགྱུར། རན་པར་འཐུངས་ན་རླུང་དང་བད་ཀན་ཀུན་སེལ་ཏེ་ཕན་ནོ། །ཆང་ལ་ཚོད་མི་ཟིན་པ་
ནི་དུག་དང་འདྲ་སྟེ་གནོད་དོ། །ཆང་མ་བགྲུངས་པ་སར་པ་ནི་ལྕི། འཇུ་དཀའ་ནད་དུ་འགྱུར་ཏེ་མི་དགེའོ། །ཆང་བགྲུངས་པ་རྙིང་པ་ནི་འཇུ་སླ། དྲོད་ཆེ། ཁོང་ཡང་། ནད་ཀུན་ལ་ཕན་ནོ། །ཐུག་པ་འམ། ཚོད་མའམ། ཁུ་བ་འཐུངས་པའི་རྗེས་ལ་ཆད་ནན་ཆེར་འཐུངས་
ན་མི་དགེ་སྟེ། ཅུང་ཟད་བསྡད་ལ་ཕྱིས་བཏུང་ངོ་། །བཀྲུ་སྨན་བཏང་བ་དང་། ལྟོ་སྟོང་བཀྲེས་པ་ཉིད་ལ་ཆང་འཐུངས་ན་མི་དགེ་སྟེ་མི་བཏུང་ངོ་། །ཆང་ཧ་ཅང་ཡང་ཞིམ་ཆེས་པ་རྣོན་པོ་དང་ཆང་ནར་མ་སོན་པ་གཞོན་ནུ་དང་། ཆང་སྣ་སོ་སོ་ཐ་དད་པའི་ཆོ་ག་བཞིན་དུ་ལེགས་པར་མ་
