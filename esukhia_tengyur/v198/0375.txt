གྱུར་པའི། །བླ་བརྣན་འབྱུང་བར་མི་འགྱུར་བྱ། །དེ་གཉིས་ལ་ནི་འོ་མ་མར། །འཇམ་རྩི་བཏང་བས་བརྟ་བ་བསྔགས། །བསྐུ་དང་དྲིལ་ཕྱིས་ཁྲུས་དང་ནི། །བཟའ་བཏུང་རླུང་ནི་སེལ་བ་ཕན། །རིག་པར་ཆང་འཐུང་གང་གིས་ནི། །ནད་ནི་འབྱུང་བར་
འགྱུར་མ་ཡིན། །དེ་ལ་རླུང་ནད་ལ་ཕན་ཆང་། །ཕལ་ཆེར་ཟན་ཆང་བུ་རམ་ཆང་། །མཁྲིས་པ་རྒུན་ཆང་ཆུ་དང་བཅས། །བད་ཀན་ཆང་རྙིང་སྦྲང་རྒུན་ཆང་། །བད་ཀན་ཅན་གྱིས་དང་པོར་བཏུང་། །མཁྲིས་པ་ཅན་གྱིས་ཟོས་རྗེས་བཏུང་། །རླུང་ཅན་གྱིས་ནི་བར་དུ་བཏུང་། །
ནད་གཞི་མཉམ་པས་ཅི་དགར་བཏུང་། །མྱོས་དང་བརྒྱལ་ལ་ཕལ་ཆེར་ནི། །རླུང་དང་མཁྲིས་པ་སེལ་ལས་བསྔགས། །ཐམས་ཅད་ལ་ཡང་ཁྱད་པར་དུ། །མཁྲིས་པ་ཤས་ཆེར་ཤེས་པར་བྱ། །བསྐུ་བྱུག་བསིལ་དང་ནོར་བུ་དང་། །ཆུས་གཏོར་རླུང་ཡབ་ཀྱིས་གཡབ་
བྱེད། །ཁ་ར་རྒུན་འབྲུམ་བུ་རམ་ཤིང་། །འབྲ་གོ་ཏང་པལ་ཁུ་བ་དང་། །མངར་བའི་སྡེ་བསྐོལ་འོ་མ་ཁུ། །ཡུ་ཤ་སེ་འབྲུར་བཅས་པ་བླུད། །སཱ་ལུ་རྩ་སྣར་དྲུག་ཅུ་པ། །ནས་དང་འཚོ་བྱེད་སྨན་མར་དང་། །དགེ་བ་ཆེན་པོ་ཏིག་ཏ་ཆེ། །སྲང་དྲུག་འོ་མ་ཙི་ཏྲར་བཅས། །པི་པི་

རྩ་སྡོང་པི་པི་ལིང་། །ཨ་རུ་ར་འམ་གླ་སྒང་ངམ། །བྱི་ཚེར་སྦྲང་རྩིར་སྦྱར་བ་སྦྱིན། །རྒྱ་ཤུག་ཚི་གུ་ན་ལེ་ཤམ། །ནཱ་ག་གེ་སར་པུ་ཤེལ་ཙེ། །ཆུ་གྲང་དང་ནི་སྦྱར་བ་བླུད། །སྐྱུ་རུ་ར་ཡི་ཁུ་བའམ། །ཨ་རུ་ར་མར་བསྐོལ་བའང་ཕན། །ཇི་སྐད་བཤད་པའི་བྱ་བ་དག །ནད་
གཞིའི་ཤས་ནི་ཇི་བཞིན་སྦྱར། །ལས་ནི་རྣམ་ལྔ་བྱ་བ་དང་། །གཏར་གས་ཁྲག་ནི་དབྱུང་བ་དང་། །སྙིང་སྟོབས་ཆེ་ཞིང་ཤེས་དང་ལྡན། །ཡུལ་ལ་ཆགས་པ་མེད་པ་ཕན། །མྱོས་པ་ཤིན་ཏུ་འཕེལ་གྱུར་དང་། །བརྒྱལ་བར་གྱུར་པ་དག་ལ་ནི། །ཀུན་ཏུ་ཞུགས་པའི་སྨན་རྣོན་
