སྲལ་ཕྲེའུ། །མ་ཧཱ་མེད་ཏ་ལ་དང་། །འབྲ་གོ་པདྨའི་རྩ་བ་དང་། །པདྨའི་སྡོང་བུ་བ་རྫུ་ན། །ཤིང་མངར་ཉེའུ་ཤིང་ཛི་བ་ཀ། །པི་ད་རི་དང་བུ་རམ་ཤིང་། །ཐལ་ཏྲེས་རྣམ་གཉིས་བྲི་ཧ་ཏི། །མུར་བ་གཟེམ་ཟངས་རྩི་བ། །སྲིང་ག་ཏ་དང་ཀ་ཤེ་རུ། །ར་སྣ་སྠིར་ལྕ་བ་དང་། །སུག་སྨེལ་ཟུར་
པ་མ་ནུ་དང་། །བ་སྤུ་བ་དང་ཅུ་གང་དང་། །རམ་ཉེ་བ་དང་བྱི་ཚེར་བ། །མ་དུ་ཀ་དང་སྟར་ག་དང་། །ཨ་བི་ཤུ་ཀ་མུཉྫ་ཀ། །ཞོ་ནི་གཉིས་གཉིས་ཕྱེར་བཏགས་དང་། །མར་བྲེ་གང་དང་འོ་མ་ནི། །བཞི་འགྱུར་དག་དང་བསྐོལ་བའི་མར། །འདི་ནི་རྨ་བྱ་ཆེན་པོ་སྟེ། །རྨ་བྱ་ལས་ནི་ཡོན་ཏན་ལྷག །
ལུས་ཟུངས་དབང་པོ་ཉམས་སྐད་འགགས། །ལུད་པ་དབུགས་མི་བདེ་ཉེན་འཇོམས། །ཟླ་མཚན་ཁུ་བའི་སྐྱོན་ལའང་ཕན། །མོ་གཤམ་ལ་ཡང་བུ་སྐྱེད་བྱེད། །ཅེས་བྱ་བ་ལ། སྨན་བཤད་མ་ཐག་པ་འདི་དག་གི་འདེ་གུ་དང་། །ཛི་བན་ཏི་ལ་སོགས་པ་ཞོ་གཉིས་གཉིས་ཀྱི་འདེ་གུ་དང་། མར་བྲེ་
གང་དང་། །འོ་མ་བཞི་འགྱུར་དང་བཅས་པ་བསྐོལ་བ་འདི་ནི་རྨ་བྱ་ཆེན་པོའི་མར་བས་ཡོན་ཏན་ཆེ་བ་ཡིན་ཏེ། ལུས་ཟུངས་ཉམས་པ་ལ་སོགས་པ་ནད་ཕྲ་མོ་ཡང་སེལ་ཅིང་མོ་གཤམ་ལྟ་བུ་ལ་ཡང་བུ་སྐྱེད་པར་བྱེད་པ་ཡིན་ནོ། །བྱི་བ་
ཁྱིམ་བྱ་ངང་བ་དང་། །རི་བོང་དང་ཡང་དེ་བཞིན་སྦྱར། །ཞེས་བྱ་བ་ལ། བྱི་བ་ལ་སོགས་པོ་རེ་རེ་ཞིང་ཡང་རྨ་བྱའི་སྨན་མར་གྱི་ཚུལ་དུ་སྦྱར་བར་བྱ་བའི་དོན་ནོ། །ནམ་ཚོང་ཡན་ཆད་གཏོགས་པའི་ནད། །ཉིས་བརྒྱ་སུམ་ཅུ་རྩ་གཅིག་ཡིན། །དེ་དག་ཕན་ཚུན་མ་འདྲེས་པར། །རྒྱ་ཆེར་
རབ་ཏུ་བཤད་པ་ཡིན། །ཞེས་བྱ་བ་ལ། ནམ་ཚོང་ཡན་ཆད་དུ་གཏོགས་པའི་ནད་སྤྱིར་གྲངས་ཉིས་བརྒྱ་སུམ་ཅུ་རྩ་གཅིག་ཡོད་པ་ཡིན་ལ། དེ་དག་ཀྱང་ཕན་ཚུན་མཚན་ཉིད་མ་འདྲེས་པར་རྒྱ་ཆེར་རབ་ཏུ་བསྟན་ཅིང་བཤད་པ་ཡིན་ནོ། །དྲང་སྲོང་རྣམས་ཀྱིས་སྐྱེས་བུའི་མགོ། །རྩ་བ་
འོག་ནི་ཡལ་གར་བཤད། །དེ་ཕྱིར་རྩ་བ་ལ་གནོད་པའི། །ནད་ནི་མྱུར་དུ་གསོ་བར་བྱ། །ཞེས་བྱ་བ་ལ། དྲང་སྲོང་རྣམས་ཀྱིས་སྐྱེས་བུའི་མགོ་བོ་ནི་རྩ་བ་ཡིན་ལ། མགོ་བོ་མན་ཆད་ཀྱི་འོག་ཏུ་མ་ལུས་པ་ཐམས་ཅད་ནི་ཡན་ལག་ཡིན་པར་བཤད་དེ། དེ་བས་ན། སྨན་པས་ལུས་ཀྱི་རྩ་
