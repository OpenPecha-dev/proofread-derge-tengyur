སྦྱིན་པར་བྱ། །ཞེས་བྱ་བ་ལ། ས་ཞག་ཟོས་པ་ལས་གྱུར་པའི་སྐྱ་རྦབ་ལ་ནི་དང་པོ་ཁོ་ནར་བཀྲུ་སྨན་དང་སློན་སྨན་རྣོན་པོ་བཏང་བས་ལུས་ལས་བསལ་ཞིང་གདོན་པར་བྱའོ། །དེའི་འོག་ཏུ་ལྟོ་བའི་ནད་སྦྱངས་པའི་སྐྱ་རྦབ་ཅན་གྱི་ནད་པའི་ལུས་ཀྱི་སྟོབས་བསྐྱེད་ཅིང་བརྟ་བར་བྱ་བའི་ཕྱིར་ཞུན་མར་དག་སྦྱིན་པར་བྱའོ།
ཚ་བ་གསུམ་དང་བིལ་བ་དང་། །སེར་པོ་གཉིས་དང་འབྲས་བུ་གསུམ། །བ་སྤྲུ་བ་གཉིས་གླ་སྒང་དང་། །ལྕགས་ཕྱེ་པ་ཐ་བྱི་དང་ག །ཐང་ཤིང་བྲི་ཤྩི་ཀ་ལི་དང་། །ག་བྲའི་རྩ་བ་འོ་མ་མར། །བསྐོལ་བ་འཐུངས་ན་ས་ཞག་ཟོས། །ལས་གྱུར་ནད་ནི་ཐམས་ཅད་སེལ། །ཞེས་བྱ་བ་ལ། ཚ་བ་གསུམ་ལ་སོགས་པའི་
སྨན་འདི་དག་འོ་མ་དང་མར་ཚུལ་བཞིན་དུ་བསྐོལ་བ་བླུད་ན་ས་ཞག་ཟོས་པ་ལས་གྱུར་པའི་ནད་ཐམས་ཅད་སེལ་ཞིང་འཇོམས་པར་བྱེད་དོ། །དེ་བཞིན་ནཱ་ག་གེ་སར་དང་། །ཤིང་མངར་ཀུ་ཤ་པི་པི་ལིང་། །འོ་མར་བསྐོལ་བ་བླུད་ན་ཕན། །ཞེས་བྱ་བ་ལ། དེ་བཞིན་དུ་ནཱ་ག་གེ་སར་ལ་སོགས་པའི་སྨན་དང་འོ་མ་
བསྐོལ་ཏེ་བླུད་ན་ཡང་ས་ཞག་ཟོས་པ་ལས་གྱུར་བའི་ནད་ཐམས་ཅད་སེལ་བར་བྱེད་དོ། །དེ་ལ་སྲེད་པར་གྱུར་ན་ནི། །བྱི་དང་ག་དང་པ་ཐ་དང་། །ནིམ་པ་བསྐོལ་བའི་ཁུ་བ་འམ། །མུར་བ་དང་བསྲེས་ས་ཞག་དག །སྲེད་པ་དགག་ཕྱིར་སྦྱིན་པར་བྱ། །ཞེས་བྱ་བ་ལ། སྐྱ་རྦབ་ཅན་གྱི་ནད་པ་དེས་ཞག་ལ་སྲེད་ཅིང་དེ་
ལ་དགའ་བར་གྱུར་ན་དེའི་སྲེད་པ་དག་ག་ཅིང་དེ་དང་འགལ་འགྲས་པར་བྱ་བའི་ཕྱིར་བྱི་དང་ག་ལ་སོགས་པ་དང་ས་ཞག་ཏུ་བསྲེས་པ་སྦྱིན་པར་བྱའོ། །ཡང་ན་མུར་བ་དང་བསྲེས་པའི་ས་ཞག་སྦྱིན་པར་བྱའོ། །ས་ཞག་བྱེ་བྲག་ཐ་དད་ནད། །རྗེས་སུ་མཐུན་པའི་སྨན་སྦྱིན་བྱ། །ཞེས་བྱ་བ་ལ། ས་ཞག་གི་བྱེ་བྲག་ཐ་དད་
པར་ཟོས་པ་ལས་ནད་ཐ་དད་པར་གྱུར་པ་ལས་ནི་རྗེས་སུ་མཐུན་ཞིང་ནད་དམིགས་ཇི་ལྟ་བ་བཞིན་དུ་སྨན་གྱི་སྦྱོར་བ་སྦྱིན་པར་བྱའོ། །དེ་ནི་འདི་སྐད་སྟོན་པ་ཡིན་ཏེ་རོ་བསྐ་བའི་ས་ཞག་ཟོས་པ་ལས་ནད་དུ་གྱུར་པ་ལ་ནི་རླུང་འཇོམས་པའི་སྨན་སྦྱིན་པར་བྱ་ལ་ཚ་སྒོ་ཅན་གྱི་ས་ཟོས་པ་ལས་ནད་དུ་གྱུར་བ་ལ་ནི་མཁྲིས་པ་སེལ་བའི་སྨན་
སྦྱིན་པར་བྱ་ལ། ས་མངར་པོ་ཟོས་པ་ལས་ནད་དུ་གྱུར་པ་ལ་ནི་བད་ཀན་སེལ་བའི་སྨན་སྦྱིན་པར་བྱའོ། །མིག་སེར་ལ་ནི་མཁྲིས་འཇོམས་སྨན། །སྐྱ་རྦབ་གནོད་པ་མེད་པ་སྦྱིན། །ཞེས་བྱ་བ་ལ། མིག་སེར་གྱི་ནད་ལ་ནི་འཁྲིས་པ་འཇོམས་ཤིང་སེལ་བའི་སྨན་སྦྱིན་པར་བྱ་ལ་དེ་ཡང་སྐྱ་རྦབ་ལ་གནོད་ཅིང་མི་མཐུན་པ་
