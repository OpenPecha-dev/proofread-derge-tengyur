ཚད་ཟོས་ན། །གཟུགས་འཁུམས་སྐྱ་རྦབ་ནད་དང་ནི། །མིག་སེར་གྱི་ནི་ནད་འཇོམས་སོ། །ཞེས་བྱ་བ་ལ། རྒུན་འབྲུམ་བྲེ་གང་དང་། པི་པི་ལིང་བྲེ་གང་དང་། ཁ་ར་སྲང་ལྔ་བཅུ། ཤིང་མངར་ལ་སོགས་པའི་སྨན་སྣ་རེ་རེ་ཞིང་ཞོ་གཉིས་གཉིས་ཏེ། དེ་མ་ལུས་པར་
ཕྱེ་མར་བྱས་ཏེ། སྐྱུ་རུ་བསྐོལ་བའི་ཁུ་བ་བྲེ་བཅུ་དྲུག་གི་ནང་དུ་བཅུག་ལ། མེ་ཆུང་ངུས་ལྡེ་གུར་གྱུར་པ་དང་། དེ་བསྒྲངས་པའི་སྦྲང་རྩི་བྲེ་གང་བཅུག་ལ། དེ་ལས་ཞོ་གཉིས་གཉིས་ཀྱི་ཚད་དུ་ཟོས་ན་གཟུགས་འཁུམས་པ་ལ་སོགས་པའི་ནད་འཇོམས་པར་བྱེད་དོ། །རྩ་བ་ལྔ་པ་ཆུང་
ངུའི་ཆུ། །རྒུན་འབྲུམ་སྐྱུ་རུ་རའི་ཁུ་བ། །བཟའ་དང་བཏུང་བར་སྦྱར་བ་ནི། །སྐྱ་རྦབ་མིག་སེར་ནད་སེལ་ཏོ། །ཞེས་བྱ་བ་ལ། སྐྱ་རྦབ་ཀྱིས་ཉེན་པའི་མི་ལ་རྩ་བ་ལྔ་བསྐོལ་བའི་ཆུ་ལ་སོགས་པ་དང་། བཟའ་བ་དང་བཏུང་བ་དང་སྦྱར་ཏེ་བསྟེན་པ་བསྔགས་སོ། །དེ་ལྟར་སྐྱ་རྦབ་ནད་
ལ་ནི། །སྤྱི་ཡི་སྨན་ནི་བསྟན་པ་ཡིན། །མཁས་པས་ནད་དམིགས་སོ་སོས་ཡི། །སྟོབས་དབང་ལས་ཀྱང་སྦྱར་བ་བྱ། །ཞེས་བྱ་བ་ལ། དེ་ལྟར་སྔར་བཤད་པའི་ཚུལ་གྱིས་སྐྱ་རྦབ་ཀྱི་ནད་བསལ་བར་བྱ་བའི་ཕྱིར་ཐུན་མོང་གི་སྨན་བཏང་བ་ཡིན་ནོ། །སྨན་དེ་གསོ་བ་ལ་མཁས་པའི་
སྨན་པས་རླུང་ལ་སོགས་པ་སོ་སོའི་ནད་ཀྱི་སྟོབས་གང་ཤས་ཆེ་བའི་དབང་དུ་བྱས་ནས་ཀྱང་དེ་ཅི་རིགས་པའི་གསོ་བར་སྦྱར་ཞིང་གཏང་བར་བྱའོ། །རླུང་གྱུར་ཕལ་ཆེར་སྣུམ་ཆོས་བྱ། །མཁྲིས་པར་ཁ་དང་བསིལ་བ་གཏང་། །བད་ཀན་ལས་གྱུར་རྩུབ་ཚ་དྲོ། །འདུས་པར་འདྲེན་མར་སྦྱར་བར་
བྱ། །ཞེས་བྱ་བ་ལ། སྐྱ་རྦབ་ཀྱི་ནད་རླུང་ལས་གྱུར་པ་ལ་ནི་སྔར་སྨས་པའི་སྨན་ལས་ཕལ་ཆེར་སྣུམ་གྱིས་ཤས་ཆེར་གསོ་བར་བྱའོ། །མཁྲིས་པ་ལས་གྱུར་པའི་སྐྱ་རྦབ་ཀྱི་ནད་ལ་ནི། སྨན་ཁ་བ་དང་བསིལ་བ་བཏང་ཞིང་གསོ་བར་བྱའོ། །བད་ཀན་ལས་གྱུར་བའི་སྐྱ་རྦབ་ཀྱི་ནད་ལ་ནི་
སྨན་རྩུབ་པ་དང་། ཚ་བ་དང་། དྲོ་བའི་བག་གཏང་བར་བྱའོ། །འདུས་པ་ལས་གྱུར་བའི་ནད་ལ་ནི་སྨན་དེ་དག་ཕན་ཚུན་འདྲེན་མར་སྦྱར་ཞིང་གཏང་བར་བྱའོ། །ས་ཟོས་ལས་གྱུར་སྦྱང་བ་ནི། །རྣོན་པོས་ལུས་ལས་བསལ་བར་བྱ། །དེ་འོག་ལྟོ་སྦྱངས་སྟོབས་བསྐྱེད་ཕྱིར། །དེ་ལ་ཞུན་མར་
