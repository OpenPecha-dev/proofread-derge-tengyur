མུ་ཙ་ལ་སོགས་པ་རྣམས་སོ། །ཚའི་མཐའ་ཅན་རབ་ཏུ་གྲུབ་པ་མེད་དོ། །ཛའི་མཐའ་ཅན་བྷཱུ་བྷུ་ཛའི་སྒྲ་ནི། ས་གཞི་ལ་སྤྱོད་པར་བྱེད་དོ་ཞེས་པ་ཀྭིཔའང་ངོ་ཞེས་པ་ཀྭིཔའོ། །ཚིག་སྡུད་པ་ལ་སོགས་པ་སྔ་མ་བཞིན་ནོ། །སི་ལ་སོགས་པ་རྣམས་སོ། །གསལ་བྱེད་ལས་ཀྱང་ངོ་
ཞེས་པ་སི་དབྱིའོ། །ཙ་སྡེ་དང་དྲྀ་ཤ་ལ་སོགས་པ་རྣམས་ཀྱི་འང་ངོ་ཞེས་པ་རྫོགས་པ་དང་གསལ་བྱེད་དག་ལ་ཛ་ཡིག་གི་ག་ཡིག་གོ། །རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པའི་ཕྱོགས་ལ་ཀ་ཉིད་དོ། །བྷཱུ་བྷུཀ །བྷཱུ་བྷུག །བྷཱུ་བྷུ་ཛཽ། བྷཱུ་བྷུ་ཛཿ། བྷཱུ་བྷུ་ཛཾ། བྷཱུ་བྷུ་ཛཽ། བྷཱུ་བྷུ་ཛཿ། བྷཱུ་བྷུ་ཛཱ། བྷཱུ་བྷུ་གྦྷྱཱམ།
བྷཱུ་བྷུ་གྦྷིཿ། བྷཱུ་བྷུ་ཛེ། བྷཱུ་བྷུ་གྦྷྱཱམ། བྷཱུ་བྷུ་གྦྷྱཿ། བྷཱུ་བྷུ་ཛཿ། བྷཱུ་བྷུ་གྦྷྱཱམ། བྷཱུ་བྷུ་གྦྷྱཿ། བྷཱུ་བྷུ་ཛཿ། བྷཱུ་བྷུ་ཛོཿ། བྷཱུ་བྷུ་ཛཱཾ། བྷཱུ་བྷུ་ཛི། བྷཱུ་བྷུ་ཛོཿ། བྷཱུ་བྷུཀྵུ། བྷཱུ་བྷུཁྶུ། ཀུན་དུ་བོད་པ་ལ་ཁྱད་པར་མེད་དོ། །དེ་བཞིན་དུ། ཧུ་ཏ་བྷུ་ཛ། བ་ལི་བྷཱུ་ཛ། ཨཱ་ཁུ་བྷུ་ཛ། པ་ཎ་ཛི། རྀ་ཏྭ་ཛི་ལ་སོགས་པ་
རྣམས་སོ། །བི་ཤྭ་སྲི་ཛ། པྲཛ་སྲི་ཛ་ཞེས་པ་ལ་སོགས་པ་རྣམས་ཀྱང་དྲྀ་ཤ་ལ་སོགས་པར་བལྟ་བར་བྱའོ་ཞེས་ཁ་ཅིག་གོ། །སམ་རཱ་ཛའི་སྒྲའི་ཁྱད་པར་ནི་ཧ་དང་ཤ་དང་ཥ་དང་ཚའི་མཐའ་དང་ཡཛ་སོགས་ཀྱི་ཌའོ་ཞེས་པས་རྫོགས་པ་དང་གསལ་བྱེད་དང་པོ་རྣམས་ལཌ་རུ་འགྱུར་རོ། །དེའི་
རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་ཡང་ན་ཊ་ཉིད་དོ། །སམྲཱཊ། སམྲ་ཊ། སམྲཱ་ཛཽ། སཾ་རཱ་ཛཿ། སམྲཱ་ཛཾ། སཾ་རཱ་ཛཽ། སམྲ་ཛཿ། སམྲཱ་ཛཱ། སཾ་རཱ་བྷྱཱཾ། སཾ་རཱཌྦྷིཿཞེས་པ་ལ་སོགས་པའོ། །དེ་བཞིན་དུ་པི་རཱ་ཛ། དེ་བ་རཱ་ཛ། སོ་ཛི་རཱ་ཛ། དྷརྨ་རཱ་ཛ། ཡཀྵ་ར་ཛ་ལ་སོགས་པ་རྣམས་སོ། །ཛྙ་མཐའ་
ཅན་དང་ཉ་མཐའ་ཅན་དང་ཌ་སྡེའི་མཐའ་ཅན་རྣམས་རབ་ཏུ་གྲགས་པ་མེད་དོ། །ཏའི་མཐའ་ཅན་མ་རུཏ་ཀྱི་སྒྲ་སྟེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་དང་པོ་དང་གསུམ་པ་དག་གོ། །མ་རུ་ཏ། མ་རུད། མ་རུ་ཏཽ། མ་རུ་ཏཿ། མ་རུ་ཏཾ། མ་རུ་ཏཽ། མ་རུ་ཏཿ། མ་རུ་ཏཱ། གསལ་བྱེད་དང་པོ་ཅན་ལ་དྷུཊ་རྣམས་
ཀྱི་གསུམ་པའོ་ཞེས་པས་གསུམ་པ་སྟེ། ད་ཡིག་གོ། །མ་རུདྦྷྱཱམ། མ་རུདྦྷིཿཞེས་པ་ལ་སོགས་པའོ། །སུཔ་ལ་ཏ་ཉིད་སྒྲ་མེད་པ་ན་སྒྲ་མེད་པ་ལ་དང་པོ་སྟེ་གཉིས་པའང་ངོ་། །མ་རུ་ཏྶུ། མ་རུ་ཐྶུ་དེ་བཞིན་དུ། བི་བཤྩི་ཏ། ནི་བྲྀ་ཏ། ས་ན་ཏ། ཏ་ནུ་ན་ཡཏ། པ་ར་བྷྲྀ་ཏ། མཱ་ར་ཛིཏ། ཙཥ་ཌ་ཀྲྀཏྱ།
