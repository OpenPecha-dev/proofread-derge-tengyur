མའི་རྒྱུ་མཚན་ལས་ཥཎ་ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །སི་ས་ཧི་ཥ་ཏེ། སེ་བྲྀ། སི་སེ་བི་ཥ་ཏེ། ལོ་ཀྲྀ། ལོ་ཙྭི། ཨ་བ་ལུ་ལོ་ཀི་ཥ་ཏེ། ཌྷཽ་ཀྲྀ། ཌུ་དྷཽ་ཀི་ཥ་ཏེ། ཙུ་བི། ཙུ་ཙུམྦི་ཥ་ཏི། གྲནྠ། གཞུང་བསྒྲིག་པ་ལའོ། །ཛི་གྲནྠི་ཥ་ཏི། ཛཱི་བ། ཛི་ཛཱི་བི་ཥ་ཏེ། ཀྲཱི་ཊཱི། ཙི་ཀྲཱི་ཊི་ཥ་ཏི། ལི་ཥ།
གསལ་བྱེད་ལ་སོགས་པས་བར་ཆོད་པར་གྱུར་པ་ལ་ཡང་ནའོ། །ཞེས་པ་འདིར་སན་ཡང་ནའོ་ཞེས་བརྗོད་པར་བྱའོ། །ཞེས་པས་སན་གྱི་ཡང་ན་ཡོན་ཏན་ཉིད་དོ། །ལི་ལི་ཁི་ཥ་ཏི། ལི་ལེ་ཁི་ཥ་ཏི། ཁིཊ་སྐྲག་པ་ལའོ། །ཙི་ཁི་ཊི་ཥ་ཏི། ཙི་ཁེ་ཊི་ཥ་ཏི། ཥྚི་མ། སྔ་མ་བཞིན་གྱི་ངེས་པ་ལས་ཥཎ་
ལ་ཥ་ཉིད་མ་ཡིན་ནོ། །ཏི་སྟི་མི་ཥ་ཏི། ཏི་སྟེ་མི་ཥ་ཏི། དྱུཏ། ཤུ་བྷ། རུཙ། དྱུ་ཏི། སྭཱ་པི་དག་གི་ཨ་བྷྱཱ་སའིའོ་ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །དི་དྱུ་ཏི་ཥ་ཏེ། དི་དྱོ་ཏི་ཥ་ཏི། ཤུ་ཤུ་བྷི་ཥ་ཏི། ཤུ་ཤོ་བྷི་ཥ་ཏེ། རུ་རུ་ཙི་ཥ་ཏེ། རུ་རོ་ཙི་ཥ་ཏེ། གུ་པཱུ། ཨཱུད་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་

སི་སེ་བི་ཥ་ཏི། སུ་སྱཱུ་ཥ་ཏི། སྔ་མ་བཞིན་གྱི་ངེས་པ་ལས་ཥ་ཉིད་མ་ཡིན་ནོ། །རྀ་དྷུ། དབྱངས་དང་པོ་ལས་གཉིས་པའིའོ་ཞེས་པས་གཉིས་བརྗོད་དོ། །རེ་ཕའི་ནི་སྔམ་བཞིན་མ་ཡིན་ནོ། །ཨརྡི་དྷི་ཥ་ཏི། རྀ་དྷི་ཇྙ་པི་དག་གི་ཨཱིར་དང་ཨཱིཏ་དག་བརྗེད་པར་བྱ་བ་དག་གོ། །ཞེས་པས་རྀ་ཡིག་དང་
པོའི་སན་ལ་དབྱངས་ཀྱི་ཨཱིརའོ། །ཨ་བྷྱཱ་ས་དབྱིའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་དྷ་ཡིག་གི་ཏ་ཡིག་གོ། །ཨཱིརྟྶ་ཏི། གྷྲྀ་ཥཱུ། ཛི་གྷཪྵི་ཥ་ཏི། རུ་དིར། བིད་ཤེས་པ་ལའོ། །མུ་ཥ། རུད་བིད་མུ་ཥ་རྣམས་ཀྱི་སན་ལའོ་ཞེས་པས་ཡོན་ཏན་མེད་པ་ཉིད་
