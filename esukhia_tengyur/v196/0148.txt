ཤུ་ཙཱི་ནི། ཊ་ལ་སོགས་པའི་དབྱངས་ལ། སྐྱེས་པར་བཤད་པ་ཅན་ནི་སྐྱེས་པར་བཤད་པ་བཞིན་ཡང་ནའོ་ཞེས་པས་སྐྱེས་པ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱིར་སྐྱེས་པའི་རྟགས་ཀྱི་བྱ་བ་སྟེ། ནུ་ཨཱ་ག་མ་མི་འགྱུར་ཞིང་། ཐུང་ངུ་ཡང་འགྱུར་བ་མ་ཡིན་ལ། ཕྱོགས་གཅིག་ལ་འགྱུར་ཞིང་། གསུམ་པའི་གཅིག་གི་ཚིག་
ལ་ནུ་ཨ་ག་མ་ཕྱོགས་གཅིག་ལ་མེད་པ་ཁྱད་པར་རོ། །ཤུ་ཙི་ནཱ་ཀུ་ལེ་ན། ཤུ་ཙི་བྷྱཱཾ། ཤུ་ཙི་བྷིཿ། ཤུ་ཙི་ནི་ཀུ་ལཱ་ཡ། ཤུ་ཙི་བྷྱཱཾ། ཤུ་ཙི་བྷྱཿ། ཤུ་ཙི་ནཿ། ཀུ་ལཱ་ཏ། ཤུ་ཙིབྷྱཱ། ཤུ་ཙི་བྷྱཿ། ཤུ་ཙི་ནཿ། ཀུ་ལ་སྱཿ། ཤུ་ཙི་ནོཿ། ཀུ་ལ་ཡོཿ། ཤུ་ཙྱཽཿ། ཀུ་ལ་ཡོ་ཤུ་ཙི་ནཱཾ་ཀུ་ལཱ་ནཱཾ། ཤུ་ཙི་ནི་ཀུ་
ལཽ། ཤུ་ཙཽ་ཀུ་ལེ། ཤུ་ཙི་ནོཿ། ཀུ་ལི་ཡོཿ། ཤུ་ཙྱོཿ། ཀུ་ལ་ཡོཿ། ཤུ་ཙི་ཥུ། ཀུ་ལེ་ཥུ། ཀུན་དུ་བོད་པ་ལ་ནི། ཧེ་ཤུ་ཙི། ཧེ་ཤུ་ཙི་ནཱི། ཧེ་ཤུ་ཙི་ནི། དེ་བཞིན་དུ་སི་ཏི། སུ་བུད་དྷི། དིར་གྷཱ་ཧུ་ཏི་ཞེས་པ་ལ་སོགས་པའོ།། །།གྲཱ་ན་ཡ་ཏི་ཞེས་པ་ལས། སད་སུ་དྭཱིཥ་ཞེས་པ་ལ་སོགས་པས་ཀྭིཔ་
འགྱུར་ཞིང་། ཉེ་བའི་ཚིག་དང་ཚིག་སྡུད་དེ། ར་དང་ཥ་དང་རྀའི་ཡི་གེ་ཞེས་པ་ལ་སོགས་ཕ་དང་། ཡང་གི་སྒྲའི་སྟོབས་ལས་ཎ་ཉིད་དུ་འགྱུར་ལ། ཚིག་ཐ་དད་པ་ལ་ཡང་རྟགས་ཀྱི་མིང་ཅན་ཏེ་སི་ལ་སོགས་པ་རྣམས་སོ། །གྲྭ་མ་ཎི། ཡི་གེ་གཅིག་མ་ཡིན་པ་ཞེས་པ་ལ་སོགས་པས་ཡ་ཉིད་དོ། །གྲཱ་
མ་ཎྱཽ། གྲཱ་མ་ཎྱཿ། གྲཱ་མ་ཎྱཾ། གྲཱ་མ་ཎྱཽ། གྲཱ་མ་ཎྱཿ། གྲཱ་མ་ཎྱ། གྲཱ་མ་ཎཱི་བྷྱཱཾ། གྲཱ་མ་ཎཱི་བྷིཿ། གྲཱ་མ་ཎྱེ། གྲཱ་མ་ཎཱི་བྷྱཱཾ། གྲཱ་མ་ཎཱི་བྷྱཿ། གྲཱ་མ་ཎྱཿ། གྲཱ་མ་ཎཱི་བྷྱཱཾ། གྲཱ་མ་ཎཱི་བྷྱཿ། གྲཱ་མ་ཎྱཿ། གྲཱ་མ་ཎྱཽཿ། གྲཱ་མ་ཎཱི་ནཱཾ། ནི་ལས་ངི་ཨཱམ་མོ། །གྲཱ་མ་ཎྱཾ། གྲཱ་མ་ཎྱོཿ། གྲཱ་མ་ཎཱི་ཥུ། ཀུན་དུ་བོད་པ་
ལ་ནི། ཧེ་གྲཱ་མ་ཎཱིཿ། ཧེ་གྲཱ་མ་ཎྱཽ། ཧེ་གྲཱ་མ་ཎྱཿ། དེ་བཞིན་དུ་བུད་མེད་ལའང་ངོ་། །ཨཱིཏ་ཨཱུཏ་བུད་མེད་བརྗོད་པ་ན་དཱིའོ་ཞེས་ན་དཱིའི་མིང་ཅན་དུ་ནི་མི་འགྱུར་ཏེ། བུད་མེད་བརྗོད་པ་སྨོས་པ་ནི་རྟག་ཏུ་བུད་མེད་ཀྱི་ཡུལ་ཅན་ཉིད་ཀྱི་ཕྱིར་རོ། །མ་ནིང་ལ་ནི་དབྱངས་ཐུང་ངུ་མ་ནིང་ལའོ་ཞེས་པས་ཐུང་ངུ་ཉིད་དེ། མ་ནིང་ལས་སི་ཨམ་
དག་དབྱི་ཞིང་དེ་ལ་བརྗོད་པ་ཡང་མ་ཡིན་ནོ་ཞེས་པས་སི་ཨམ་དག་དབྱིའོ། །གྲཱ་མ་ཎི་ཀུ་ལཾ། ཨཽ་ཨཱི་སྟེ། ནཱ་མིན་དབྱངས་ལའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ། །གྲཱ་མ་ཎི་ནཱི། སླར་ཡང་། གྲཱ་མ་ཎི། གྲཱ་མ་ཎཱི་ནི། གྲཱ་མ་ཎཱི་ནི། ཊཱ་ལ་སོགས་པའི་དབྱངས་ལ་སྐྱེས་པ་དང་འདྲ་བའི་དངོས་པོ་སྟེ། དེས་ན་ཐུང་ངུར་མི་འགྱུར་རོ། །
