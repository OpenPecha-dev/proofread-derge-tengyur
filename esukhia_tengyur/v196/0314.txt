ཨ་ཙ་ཀ་མ་ཏ། ཨ་ཙ་ཀ་མེ་ཏཱཾ། ཨ་ཙ་ཀ་མནྟ། གང་གི་ཚེ་ཨི་ནིང་ཡོད་བ་དེའི་ཚེ་ཀཱ་རིའི་ཏའི་མཐའ་ཉིད་ཀྱི་ཕྱིར་ཙཎའོ། །ཨིན་དང་ཙཎ་ལ་མཚུངས་པ་མ་ཕྱིས་པའི་ཨུ་པ་དྷ་ཐུང་ངུའོ་ཞེས་པས་ཐུང་ངུའོ། །དེ་ནས་གཉིས་བརྗོད་དོ། །མཚུངས་པ་མ་ཕྱིས་པ་ཡང་བ་ལ་ཨིན་ལས་ཙཎ་ཕ་རོལ་
ལ་སན་བཞིན་ནོ་ཞེས་པས་ཨ་བྷྱཱ་སའི་སན་བཞིན་འགྱུར་བ་ལས་ཨ་ཉིད་དོ། །ཡང་བའི་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཀཱ་རི་ཏའི་ཨཱམ་ཨིཊ་བི་ཀ་ར་ཎ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཀཱ་རི་ཏ་དབྱིའོ། །ཨཙཱི་ཀ་མ་ཏ། ཨ་ཙཱི་ཀ་མེ་ཏཱཾ། ཨ་ཙཱི་ཀ་མནྟ། བཱ་བྷྲྀ། ཨ་བཱ་དྷི་ཥྚ། ཨ་བཱ་དྷི་ཥཱ་ཏཱཾ། ཨ་བཱ་
དྷིཥྚ། བྷཱ་སྲྀ། ཨ་བྷཱ་སིཥྚ། གཱ་ཧཱུ། ཨཱུཏ་རྗེས་སུ་འབྲེལ་པ་ཉིད་ལས་ཡང་ན་ཨིཊའོ། །ཨ་གཱ་ཧིཥྚ། ཨ་གཱ་ཧི་ཥཱ་ཏཱཾ། ཨ་གཱ་ཧི་ཥ་ཏ། ཨ་གཱ་ཧིཥྛཱཿ། ཨ་གཱ་ཧི་ཥཱ་ཐཱཾ། ཨ་གཱ་ཧི་ཌྷྭཾ། ཨ་གཱ་ཌྷ། བྱིངས་ཀྱི་གསུམ་པ་དང་པོའི་གྷ་ཌྷ་དྷ་བྷ་མཐའི་ཞེས་པ་ལ་སོགས་པས་དང་པོ་གའི་བཞི་པ་གྷ་ཉིད་དོ། །
ཨ་གྷཱ་ཀྵཱ་ཏཱཾ། ཨ་གྷཱ་ཀྵ་ཏ། ཨ་གྷཱ་ཌྷཱཿ། ཨ་གྷཱ་ཀྵཱ་ཐཱཾ། ཨ་གྷཱ་ཌྷཱཾ། ཨ་ཡ་འགྲོ་བ་ལའོ། །པ་རཱ་སྔོན་དུའོ། །ཉེར་བསྒྱུར་གྱི་ཨ་ཡ་ཏི་ལ་ཞེས་པས་རེ་ཕའི་ལ་ཉིད་དོ། །པ་ལཱ་ཡིཥྜཱ། པ་ལཱ་ཡི་ཌྷྭཾ། པ་ལཱ་ཡི་དྷྭཾ། ཥེ་བྲྀ་འཁྲུད་པ་ལའོ། །ཨ་སེ་བིཥྚ། ཨ་སེ་བི་ཌྷྭཾ། ཨ་སེ་བི་དྷྭཾ། ཏ་ནུ། གསལ་བྱེད་
དང་པོ་ཨིཊ་དང་བཅས་པ་རྣམས་ཀྱི་ཞེས་པ་ལ་སོགས་པས་པར་སྨཻའི་ཚིག་ལ་ཡང་ན་ཨའི་རིང་པོའོ། །ཨ་ཏཱ་ནཱིཏ། ཨ་ཏ་ནཱིཏ། ཨ་ཏཱ་ནིཥྚཱཾ། ཨ་ཏ་ནིཥྚཱཾ། ཨ་ཏ་ནིཥྚ། ཨ་ཏ་ནིཥཱ་ཏཱཾ། ཥ་ནུ། ཨ་སཱ་ནཱིཏ། ཨ་ས་ནཱིཏ། ཨ་སཱ་ནིཥྚཱཾ། ཨ་ས་ནིཥྚཱཾ། ཨ་ས་ནིཥྚ། ཨ་ས་ནི་ཥཱ་ཏཱཾ། ཀྵ་ཎུ། ཨ་ཀྵ་ཎཱི་ཏ། ཨ་ཀྵ་
ཎིཥྚ། ཨ་ཀྵ་ཎི་ཥཱ་ཏཱཾ། ཨ་ཀྵ་ཎི་ཥ་ཏ། རྀ་ཎུ། དབྱངས་དང་པོ་རྣམས་ཀྱི་འཕེལ་བ་ཨཱ་རའོ། །ཨཱརྞྞཱིཏ། ཨརྞྞཱིཥྚཱཾ། ཨརྞྞི་ཥུཿ། ཏྲྀ་ཎུ། ནཱ་མི་ཨུ་པ་དྷཱའི་ཡོན་ཏན་ནོ། །ཨ་ཏརྞྞི་ཏ། ཨ་ཏརྞྞཱི་ཥྚཱ། ཨ་ཏརྞྞི་ཥུཿ། ཨ་ཏིརྞྞི་ཥྚ། ཨ་ཏ་རྞྞི་ཥཱ་ཏཱཾ། ཨ་ཏརྞྞི་ཥཏ། གྷྲྀ་ཎུ། ཨ་གྷྲ་རྞྞཱིཏ། ཨ་གྷྲ་རྞྞི་ཥྚཱཾ། ཨ་གྷྲ་རྞྞི་ཥུཿ། ཨ་
