རྣམས་ཀྱི་སན་ལའོ་ཞེས་པས་སམྤྲ་སཱ་ར་ཎའོ། །ཧའི་ཌྷའོ། །སྔ་མ་བཞིན་དང་པོ་བཞི་པ་ཉིད་དོ། །ཥ་ཌྷ་དག་གི་ས་ལ་ཀའོ་ཞེས་པས་ཀ་ཉིད་དོ། །ཛི་གྷྲིཀྵ་ཏི། ཛིགྷྲཀྵི་ཏེ། ཛུགྷུ་ཀྵ་ཏི། ཛུ་གྷུཀྵ་ཏེ། ཉི་ཥྭ་པ། ཨཱ་པི་ཏ་པི་ཞེས་པ་ལ་སོགས་པས་ཨིཊ་མེད་དོ། །པྲཙྪ། སྨིང་། པཱུང་ཞེས་པ་ལ་སོགས་
པས་ཨིཊའོ། །སྔ་མ་བཞིན་སཾ་པྲ་སཱ་ར་ཎ་བྱས་ཏེ་གཉིས་བརྗོད་དོ། །རྒྱུ་མཚན་ལས་ཥ་ཉིད་དོ། །སུ་ཥུཔྶ་ཏི། པི་པྲྀཙྪི་ཥ་ཏི། ཧྭེཉ། ཨ་བྷྱཱསྟའི་ཡང་ངོ་། །ཞེས་པས་སཾ་པྲ་སཱ་ར་ཎའོ། །དེ་མཐའ་རིང་པོའོ་ཞེས་པའམ། དབྱངས་མཐའ་རྣམས་ཀྱི་སན་ལའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །
དེ་ནས་གཉིས་བརྗོད་དོ། །ཛུ་ཧཱུ་ཥ་ཏི། ཛུ་ཧཱུ་ཥ་ཏེ། ཌུ་མི་ཉ། སན་ལ། མི། མཱི། མཱ། དཱ། ར་བྷ། ལ་བྷ། ཤ་ཀ །པེ་ཏ། པ་ད་རྣམས་ཀྱི་དབྱངས་ཀྱི་ཨི་སའོ་ཞེས་པས་དབྱངས་ཀྱི་ཨིས་སུ་བསྒྱུར་རོ། །ཨ་བྷྱཱས་དབྱི་བ་ཡང་ངོ་། །སའི་ས་ལ་བྱིངས་ཐམས་ཅད་པ་མ་ཡིན་པ་ལ་ཏའོ་ཞེས་
པས་སའི་ཏ་ཡིག་གོ། །མིཏྶ་ཏི། མིཏྶ་ཏེ། དེ་བཞིན་དུ་མཱིཉ་ཞེས་པ་འདིའི་འང་ངོ་། །མཱིངའི། མིཏྶ་ཏེ། མཱ་འཇལ་བ་ལའོ། །མིཏྶ་ཏི། མིང་། མཱང་། མིཏྶ་ཏེ། ཌུ་དཱཉ། དིཏྶ་ཏི། དིཏྶ་ཏེ། དཱ་ཎི། དོ། །དིཏྶ་ཏི། དེང་། དིཏྶ་ཏི། ཌུ་དྷཱ་ཉ། དྷིཏྶ་ཏི། དྷིཏྶ་ཏེ། དྷེཊ། དྷིཏྶི་ཏི། དཱིང་ཞེས་པ་
འདིའི་སན་ལ་དཱིངའོ། །ཞེས་པས་ཨཱའི་དོན་གྱི་ཕྱིའི་ཡན་ལག་ལ་ནང་གི་ཡན་ལག་གྲུབ་པ་མ་ཡིན་ཞེས་པས་དཱའི་མིང་ཅན་གྲུབ་པ་མ་ཡིན་པ་ཉིད་ལས་ཨིས་མེད་དོ། །དི་དཱ་ས་ཏེ། ར་བྷ། བརྩེགས་པའི་དང་པོ་ས་དབྱིའོ། །སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་ཀྱི་དང་པོའོ་ཞེས་པས་དང་པོའོ། །ཨཱ་རིཔྶ་ཏེ།
ཌུ་ལ་བྷ་ཥ། ལིཔྶ་ཏེ། ཤ་ཀླྀ། ཤིཀྵ་ཏི། པཏླྀ། ཨི་ཡིག་དང་རྀ་དྷ་ཞེས་པ་ལ་སོགས་པས་ཕྱོགས་གཅིག་ལ་ཨིཊའོ། །པིཏྶ་ཏི། པི་ས་ཏི། ཥ་ཏི། པ་ད། པིཏྶ་ཏི། པྲ་ཏི་སྔོན་མའི་རཱ་དྷ་འཚེ་བ་ལ་ནི། པྲ་ཏི་རིཏྶ་ཏི། ཨཱ་བླྀ། དབྱངས་དང་པོ་ལས་གཉིས་པའི་གཉིས་བརྗོད་དོ། །ཨཱཔ་ནོ་རིའི་ཨིའི་ཞེས་
