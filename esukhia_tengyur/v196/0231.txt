ཛ་གྷརྨྨི། ཛ་གྷྲྀ་བཿ། ཛ་གྷྲྀ་མཿ། ཧྲྀ་ཐལ་བར་བྱེད་པ་ལའོ། །ཛ་ཧརྟྟི། ཛ་ཧྲྀ་ཏ། ཛ་ཧྲ་ཏི། ཛ་ཧཪྵི། ཛ་ཧྲྀ་ཐཿ། ཛ་ཧྲྀ་ཐ། ཛ་ཧརྨྨི། ཛ་ཧྲྀ་བཿ། ཛ་ཧྲྀ་མཿ། རྀ་སྲི་འགྲོ་བ་ལའོ། །རྀ་ཡིག་གི་ཡ་ཡིག་གོ། །ཨརྟི་པི་བརྟི་དག་གི་འང་ངོ་ཞེས་ཨི་ཉིད་དོ། །ཨ་བྷྱ་སའི་
འདྲ་བའི་ཡི་གེ་མ་ཡིན་པ་ལའོ་ཞེས་པས་ཨི་ཡ་རུ་བསྒྱུར་རོ། །ཨི་ཡརྟྟི། ཨི་ཡྲྀ་ཏཿ། ཨི་ཡ་ཏི། ཨི་ཡཪྵི། ཨི་ཡྲྀ་ཐཿ། ཨི་ཡྲྀ་ཐ། ཨི་ཡརྨི། ཨི་ཡྲྀ་བཿ། ཨི་ཡྲྀ་མཿ། སམ་ལས་གམ་ཞེས་པ་ལ་སོགས་པས་ལས་ཅན་གྱི་རུ་ཙ་སོགས་ཉིད་ཀྱི་ཕྱིར་ཨཱཏྨ་ནེའི་ཚིག་གོ། །
ས་མི་ཡྲྀ་ཏེ། ས་མི་ཡྲཱ་ཏེ། ས་མི་ཡྲྀ་ཏེ། ས་མི་ཡྲྀ་ཥེ། ས་མི་ཡྲཱ་ཐེ། ས་མི་ཡྲྀ་དྷྭེ། ས་མི་ཡྲེ། ས་མི་ཡྲྀ་བ་ཧེ། ས་མི་ཡྲྀ་མ་ཧེ། ས་སརྟྟི། ས་སྲྀ་ཏཿ། ས་སྲྀ་ཏི། སི་སཪྵི། སི་སྲྀ་ཐཿ། སི་སྲྀ་ཐ། སི་སརྨྨི། སི་སྲྀ་བཿ། ས་སྲྀ་མཿ། བྷ་ས་བརྒྱད་བཀག་དང་འབར་བར་བྱེད་པ་དག་ལའོ། །
