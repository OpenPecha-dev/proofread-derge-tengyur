གི་སྦྱོར་བ་རྣམས་འབད་པ་ཆུང་ངུས་ཤེས་པར་འགྱུར་རོ། །དེའི་ཕྱིར་འདི་ནི་བསླབས་ནས་ཤེས། །སྨྲ་མཆོག་རྗེས་སྨྲ་རྣམས་ཀྱིས་བཤད། །འབད་པ་ཆུང་ངུས་ཤེས་དེའི་ཕྱིར། །འདི་ནི་བསླབས་པའི་གནས་ཡིན་ནོ། །ཞེས་སྨྲ་བའི་མཆོག་ཐམས་
ཅད་མཁྱེན་པ་དང་དེའི་རྗེས་སུ་སྨྲ་བའི་སྔོན་གྱི་མཁན་པོ་རྣམས་ཀྱིས་བཤད་དོ། །ད་ནི་དགེ་བའི་རྩ་བ་བསྔོ་བ་བསྟན་ཏེ། དེ་ཡང་མིང་གི་ཡོན་ཏན་གྱི་སྒོ་ནས་བསྔོ་བ་བསྟན་པའི་ཕྱིར། དོན་ལ་འཇུག་བྱེད་སྨྲ་བ་རྣམས་ཀྱི་སྒོ། །བློ་འབྱེད་གཏི་མུག་གཅོད་པས་མཚོན་ཆ་འདྲ། །མི་
འཇིགས་སྙམས་པ་སྦྱིན་བྱེད་དཔུང་དང་གོ །དོན་ནོར་སྨྲ་བས་མི་ཉམས་བརྟན་པའི་མཁར། །སྨྲ་མཆོག་རྗེས་སྨྲ་རྣམས་ཀྱིས་བཤད། །སྒྲ་ཉིད་དང་ནི་དོན་ལ་ཁྲིད་པ་ཡི། །འགྲོ་ཀུན་མཐུ་གཉིས་རབ་ཏུ་རྒྱས་པར་ཤོག །ཅེས་བྱ་བ་སྨོས་ཏེ། དོན་ཇི་ལྟ་བ་བཞིན་དུ་སྟོན་པའི་
འཇུག་པར་བྱེད་པའི་སྒོ་ལྟ་བུ་ཡིན་པས་སྨྲ་བའི་སྒོའོ། །གང་ཟག་རྣམས་ཀྱི་བློ་འབྱེད་ཅིང་གཏི་མུག་གི་དྲ་བ་གཅོད་པས་མཚོན་ཆ་ལྟ་བུ་ཡིན་པས་མཚོན་ཆ་ལྟ་བུའོ། །དོན་དང་ལྡན་པར་སྨྲ་བའི་རྒོལ་བ་རྣམས་ཀྱིས་མི་འཇིགས་ཤིང་སྙེམས་པ་རྙེད་པས་དཔུང་དང་གོ་ཆ་ལྟ་བུའོ། །
རང་གི་བློ་ལ་སྒྲ་དང་དོན་ནོར་བར་སྨྲ་བའི་ཤེས་རབ་འཆལ་པའི་སྐྱོན་གྱིས་མི་ཉམས་པའི་བློ་བརྟན་པ་རྙེད་པར་བྱེད་པས་བརྟན་པའི་མཁར་ལྟ་བུའོ། །འདི་ལྟ་བུའི་རབ་ཏུ་བྱེད་པ་འདི་ཐམས་ཅད་མཁྱེན་པ་དང་། སྔོན་གྱི་མཁན་པོ་རྣམས་ཀྱི་རྗེས་སུ་སྨྲས་ཤིང་བཤད་པས་
འགྲོ་བ་རྣམས་སྒྲ་དང་སྒྲའི་དོན་ལ་མཁས་པར་གྱུར་ཅིག་ཅེས་བྱ་བའི་དོན་ཏོ། །སྨྲ་བའི་སྒོ་མཚོན་ཆ་ལྟ་བུའི་འགྲེལ་པ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་དྲན་པའི་ཡེ་ཤེས་གྲགས་པས་བླ་མ་ནོར་བཟངས་གྲགས་པ་ལས་
ཐོས་ཏེ། སློབ་མ་གཞོན་ནུ་གྲགས་པའི་དོན་དུ་མིང་གི་སྦྱོར་བ་ནི་ཅུང་ཟད་གཞག །ཕྲད་ཀྱི་སྦྱོར་བ་ནི་ཅུང་ཟད་བཤད་ནས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།། །།
