ར་ཛ་ཀཱི་ཞེས་པ་ལ་སོགས་པའོ།། །།ལ་ཀྵྨཱིའི་སྒྲའི་ནི། སཾ་བུདྡྷི་ལས་སི་དབྱི་བ་མེད་ཅིང་། ཨཱི་ཡིག་གི་མཐའ་ཅན་རྟགས་གང་ལས་ཞེས་ཁྱད་པར་དང་བཅས་པ་བཤད་པའི་ཕྱིར། ལཀྵི་ལས་ཨ་བི་ཙི་ཏྲྀ་ཏཾ་ཏྲི་རྣམས་ལས་ཨཱིའོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ལཀྵི་ལས་མའི་མཐའ་ཅན་འང་ངོ་ཞེས་
བས་ཨིའི་རྐྱེན་ཏེ། ཨའི་ཡང་མ་ཡིག་གི་མཐའ་ཅན་དུ་འགྱུར་རོ། །ལཀྵྨཱིཿ། ལཀྵྨཽ། ལཀྵྨེཿ། ཞེས་པ་ལ་སོགས་པའོ། །ཀུན་དུ་བོད་པ་ལ། ཚོགས་པ་བུད་མེད་བརྗོད་པ་ཉིད་ལ་ཆ་ཤས་ཀྱང་བུད་མེད་བརྗོད་པར་ཉེ་བར་བཏགས་པའི་ཕྱིར་ན་དཱིའི་མིང་ཅན་དུ་འགྱུར་ཏེ། དེས་ན་སྔ་མ་བཞིན་དུ་
ཐུང་ངུ་ཉིད་དང་སི་དབྱི་བའང་ངོ་། །ཧེ་ལཀྵྨི། ཧེ་ཀྵྨྱཽ། ཧེ་ལ་ཀྵྨྱཿ། དེ་བཞིན་དུ་དྷཱ་ར་རཱི་དང་ཏཾ་ཏྲཱི་ཞེས་པ་ལ་སོགས་པ་རྣམས་སོ།། །།བྷ་ཛ་ཤྲིཉ་བསྟེན་པ་ལ་སྟེ། ཉ་ཡིག་ནི། ཨིན་དང་ཉ་དང་ཡ་ཛ་སོགས་ལས་གཉིས་སོ་ཞེས་པའི་ཁྱད་པར་གྱི་དོན་ཡིན་ལ། བ་ཙི་ཕྲཙྪི་ཤྲི་ཤྲུ་དྲུ་སྲུ་ཛུ་
རྣམས་ལ་ཀྭིཔ྄་དང་རིང་པོའོ་ཞེས་པས་ཀྭིཔ྄་སྟེ། བྱིངས་ཀྱི་འང་རིང་པོར་འགྱུར་ཞིང་། ཀྭིཔ྄་ཐམས་ཅད་བསྡུས་ཏེ་ཕྱིས་པ་ལས་སི་ལ་སོགས་པ་ཡོད་པ་ལས་སིའི་རྣམ་པར་གཅད་བྱའོ། །ཤྲཱིཿ། དབྱངས་དང་པོ་ཅན་ལ་ནི། ཨཱིཏ་ཨཱུད་དག་གི་ཨི་ཡ་ཨུ་ཝ་དག་དབྱངས་ལའོ་ཞེས་པས་ཨཱི་ཨི་ཡ་རུ་འགྱུར་རོ། །
ཤྲྀ་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཾ། ཤྲི་ཡཽ། ཤྲི་ཡཿ། ཤྲི་ཡཱ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷིཿ། སྟྲྀ་ན་དཱི་དང་འདྲའོ་ཞེས་པ་སྤྱིར་བཏང་བ་ལས། བུད་མེད་བརྗོད་པ་ཨི་ཡ་ཨུ་ཝ་དག་ཨཱམ་ལ་ཡང་ནའོ་ཞེས་པ་སྤྱིར་བཏང་བའི་ཕྱིར། ཐུང་ངུའི་ཡང་ང་ལྡན་ལའོ་ཞེས་པས་ང་ལྡན་གྱི་བརྗོད་པ་ཕ་རོལ་ལ་ཡང་ན་ན་དཱི་དང་འདྲ་
བར་འགྱུར་བའི་ཕྱོགས་ལ་ན་དཱིའི་བྱ་བ་སྟེ། ན་དཱི་ལས་ཨཻ་ཨཱས་ཨཱས་ཨཱཾ་ཞེས་པའོ། །ཤྲི་ཡཻ། ཤྲི་ཡེ། ཤྲི་བྷྱཱཾ། ཤྲཱ་བྷྱཿ། ཤྲཱི་ཡཱཿ། ཤྲི་ཡཿ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ། ཤྲི་ཡཱ་ཤྲི་ཡཿཤྲི་ཡོཿ། ན་དཱི་དང་འདྲ་བར་འགྱུར་བའི་ཕྱོགས་ལ་ན་དཱིའི་བྱ་བ་སྟེ། ཨཱཾ་ལ་ཡང་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་
མའོ། །ཤྲཱ་ཎཱཾ། ཤྲི་ཡཱཾ། ཤྲི་ཡི། ཤྲི་ཡོཿ། ཤྲཱི་ཥུཿ། ཀུན་དུ་བོད་པ་ལ་ནི། སྟྲཱི་ན་དཱི་དང་འདྲའོ་ཞེས་སྨོས་པས་ཨི་ཡ་ཨུ་ཝ་དག་གི་ཨི་ཡིག་དང་ཨུ་ཡིག་དག་ན་དཱིའི་མིང་ཅན་དུ་རབ་ཏུ་གྲུབ་པ་མེད་པར་བཤད་དེ། དེས་ན་སཾ་བུདྡྷི་ལ་ཐུང་ངུར་མི་འགྱུར་ཞིང་། དེ་བཞིན་དུ་སི་དབྱི་བའང་ངོ་། །
