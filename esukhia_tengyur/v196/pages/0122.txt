པས་མ་ཕྱིས་པ་དང་འདྲ་བར་འགྱུར་བའི་ཕྱིར། །ཨ་ཡིག་སྒྲ་ལྡན་ལ་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དང་། བྷིས་ཨཻས་ཡང་ནའོ་ཞེས་པ་དང་། དྷུཊ་མང་པོ་ཉིད་ལ་ནི་ཨེའོ་ཞེས་པས་ཨེ་ཉིད་དུ་འགྱུར་པ་ནི་མ་ཡིན་ནོ། ཨཱཏྨ་བྷྱཱམ། །ཨཱཏྨ་བྷིཿ། ཨཱཏྨནེ། ཨཱ་ཏྨ་བྷྱཱམ། ཨཱ་ཏྨ་བྷྱཿ། ཨཱ་ཏྨ་ནཿ། ཨཱ་ཏྨ་
བྷྱཱ་མ། ཨཱ་ཏྨབྷྱཿ། ཨཱ་ཏྨནཿ། ཨཱ་ཏྨནོཿ། ཨཱ་ཏྨནཱམ། ཝ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཞེས་པས་བཀག་པའི་ཕྱིར། ངི་ལ་ཨར་རོ་ཞེས་པའི་རྣམ་པར་བརྟག་པ་མེད་དོ། །ཨཱཏྨ་ནི། ཨཱཏྨ་ནཽ། །ཨཱཏྨསུ། ཀུན་དུ་བོད་པ་ལ། སཾ་བུདྡྷི་མ་ཡིན་པ་ལ་ཞེས་པས་བཀག་པའི་ཕྱིར། །སཾ་བུདྡྷིལ་
རིང་པོར་འགྱུར་བ་མ་ཡིན་ཞིང་། སཾ་བུདྡྷི་མ་ཡིན་པ་ལ་ཞེས་པས་ན་དབྱི་བའང་བཀག་གོ། །ཧེ་ཨཱཏྨན། ཧེ་ཨཱཏྨ་ནཽ། ཧེ་ཨཱཏྨནཿ། དེ་བཞིན་དུ་ཨ་དྷྭནྲ་དང་པྲ་ཧྨན྄་དང་ཨསྨན་དང་ཤླེ་སྨན་ལ་སོགས་པ་རྣམས་སོ། །རཱ་ཛནྲ་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། །གྷུཊ་ལའང་སཾ་བུདྡྷི་མ་ཡིན་པ་
ལ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །རཱ་ཛཱ། རཱ་ཛཱ་ནཽ། །རཱ་ཛཱ་ན། རཱ་ཛཱ་ནཾ། རཱ་ཛཱ་ནཽ། །བ་མ་བརྩེགས་པ་མ་ཡིན་པ་ལས་ཨན་གྱི་ཨ་དབྱི་བ་ཞེས་པ་ལ་སོགས་པས་ཨ་དབྱི་ཞིང་། །ཏའི་སྡེ་པ་ནི་ཙ་ཊའི་སྡེ་པ་ཞེས་པ་ལ་སོགས་པས་ན་ཡིག་ནི་ཉ་ཡིག་གོ། །རཱཛྙཿ། རཱ་ཛྙ། རཱ་ཛ་བྷྱཱཾ། རཱཛ་བྷིཿ། རཱ་
ཛྙེ། རཱཛ་བྷྱཱམ། རཱཛབྷྱཿ། རཱཛྙཿ། རཱ་ཛ་བྷྱཱམ། རཱ་ཛ་བྷྱཿ། རཱ་ཛྙཱཿ། རཱ་ཛྙོཿ། རཱ་ཛྙཱཾ། ཨཱི་དང་ངི་དག་ལ་ཡང་ནའོ་ཞེས་པའི་རྣམ་པར་བརྟག་པས་ཨ་དབྱིའོ། །རཱ་ཛྙི། རཱ་ཛ་ནི། རཱ་ཛྙོཿ། རཱ་ཛ་སུ། །ཀུན་དུ་བོད་པ་ལ་ནི། །ཧེ་རཱ་ཛན། ཧེ་རཱ་ཛཱ་ནཽ། །ཧེ་རཱ་ཛ་ན། དེ་བཞིན་དུ་ཛ་དྷྭན། ནཀྴན།
མཀྴན། ཏཀྴནྲ། །ཨེ་ཌྷི་མཱན་ཞེས་པ་འདི་ལྟ་བུ་ལ་སོགས་པ་རྣམས་སོ།། །།ཡུ་བན་གྱི་སྒྲ་ལས་སི་ལ་སོགས་པ་རྣམས་ནི། ཡུ་བཱ། ཡུ་བཱ་ནཽ། ཡུ་བཱ་ནཿ། ཡུ་བཱ་ནཾ། ཡུ་བཱ་ནཽ། །གྷུཊ་མ་ཡིན་པའི་དབྱངས་དང་པོ་ལ་ཨིཊ་དང་བཅས་པའི་འང་བནྶའི་བ་སྒྲའི་ཨུ་ཉིད་དོ་ཞེས་པ་སྤྱིར་བཏང་
ནས། ཤྭན། ཡུ་བན། མ་གྷ་བན་རྣམས་ཀྱི་འང་ངོ་ཞེས་པས་ཝ་སྒྲའི་ཨུ་ཉིད་དོ། །མཚུངས་པ་ལས་རིང་པོ་དང་། རིང་པོ་ལས་ཕ་རོལ་དབྱིའོ། །ཡཱུ་ནཿ། ཡཱུ་ནཱ། ཡུ་བ་བྷྱཱམ། ཡུ་བ་བྷིཿ། ཡཱུན། ཡུ་བ་བྷྱཱམ། ཡུ་བ་བྷྱཿ། ཡཱུནཿ། ཡུ་བ་བྷྱཱམ། ཡུ་བ་བྷྱཿ། ཡཱུ་ནཿ། ཡཱུ་ནོཿ།
