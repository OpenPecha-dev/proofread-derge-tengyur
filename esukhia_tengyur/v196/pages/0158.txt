དུས་ལ་བྱས་པ་དང་། པྲཱ་ཙྱཱ་ཡང་ན་ཕྱོགས་ལ་འོངས་པ་དང་། པྲཱ་ཅོ་ཡང་ན་ཡུལ་ལ་འོངས་པ་དང་། པྲཱ་ཙ་དུས་ལ་ཡང་ན་དང་། པྲཱ་ཙཱི་ཕྱོགས་ལ་ར་མ་ཎཱི་ཡྰ། པྲཱད་ཡུལ་ལ་དང་། པྲཱཏ་དུས་ལ་ཡང་ན་ར་མ་ཎཱི་ཡ་ཞེས་པའི་དོན་ཏེ་རྣམ་དབྱེ་རྣམས་འགྱུར་བ་དེའི་ཚེ། ཕྱོགས་ཀྱི་སྒྲས་བདུན་པ་དང་
ལྔ་པ་དང་དང་པོའི་མཐའ་ཅན་ཕྱོགས་དང་། ཡུལ་དང་། དུས་ཉིད་ལ་ཡང་ནའོ་ཞེས་པ་སཱ་ཏིའི་རྐྱེན་ནོ། །དེ་ཡང་འགྲེལ་བཤད་དུ་གསུངས་པ། ཨ་བ་ཏ་ལ་སཱ་ཏིའི་འཇིག་ཅིང་། གང་དུ་བུད་མེད་ཀྱི་རྐྱེན་གྱི་མཐའ་ཅན་དུ་འགྱུར་བ་དེར་ཚིག་ཕྲད་ལ་ཨཏ་ཉིད་ལས་ཨཱིའི་རྐྱེན་འཇིག་ལ། མི་ཟད་པ་
ལས་ཀྱང་ངོ་ཞེས་པས་རྣམ་པར་དབྱེ་བའང་འཇིག་གོ། །པྲཱ་གྡེ་ཤི་བ་བ་ཏེ། པྲཱ་གྡེ་ཤི་བ་ས་ཏི། །པྲཱ་ཀྐཱ་ལེ་ཀྲི་ཏཾ། །པྲཱགྡེ་ཤོ་ཨཱ་ག་ཏཿ། པྲཱ་ཀྐཱ་ལོ་པཱ་ཏཿ། །པྲཱཀྡི་ཀ །ཤོ་བྷ་ནཱ། པྲཱ་གྡེ་ཤཿ། ཀཱ་ལོ་བ་རཱ་མ་ཎཱི་ཡཿ། དེ་བཞིན་དུ་པྲཏྱག་ལ་སོགས་པའི་སྒྲའོ།། །།བི་ཤྭག་ཨནྩ་ཏི་ཞེས་པས་ཀྭི་པ་
སྟེ་ཉེ་བའི་ཚིག་དང་ཚིག་སྡུད་ཡིན་ཅིང་རྟགས་ཀྱི་མིང་ཅན་ནོ། །བི་ཤྭག་དང་དེ་ཝ་དག་གི་ཡང་ཨཉྩ་ཏི་དང་ཀྭི་པ་ལ་མཐའི་དབྱངས་དང་པོའི་ཨ་དྲིའོ་ཞེས་པས་བི་ཤྭག་དང་དེ་ཝ་ཞེས་པ་འདི་དག་གི་མཐའི་དབྱངས་དང་པོ་ཨ་དྲི་རུ་བསྒྱུར་བར་འགྱུར་ཏེ། ག་ཉིད་ལ་སོགས་པ་ནི་སྔ་མ་བཞིན་ནོ། །བི་ཤྭ་དྲྱང་། བིཤྭྡྲྱ་ཉྩཽ།
བི་ཤྭ་དྲྱཉྩཿ། །བི་ཤྭ་དྲྱཉྩཾཿ། བུ་ཤྭ་དྲྱངཉྩཽ། ཨ་ཉྩའི་ཨ་དབྱི་བ་དང་སྔ་མའི་ཡང་རིང་པོ་སྟེ། བི་ཤྭ་དྲཱི་ཙཿ། བི་ཤ་དྲཱི་ཙཱ། །བི་ཤྭ་དྲྱ་དྦྷྱཱཾ། བི་ཤ་དྲྱ་ཌྦྷིཿ། །ཞེས་པ་ལ་སོགས་པའོ། །བུད་མེད་ལ། །བི་ཤྭ་དྲཱི་ཙཱི། །མ་ནིང་ལ། བི་ཤྭ་དྲྱཀ །བི་ཤྭ་དྲྱག། །བི་ཤྭ་དྲཱི་ཙཱི། བི་ཤྭ་དྲཱི་ཉྩཱི། དེ་བཞིན་དུ་དེ་ཝ་དྲྱང་སྟེ།
སརྦ་ནཱ་མ་ལ་ཨཱའོ་ཞེས་པ་སྤྱིར་བཏང་སྟེ་སརྦ་ནཱ་མི་རྣམས་ཀྱི་མཐའི་དབྱངས་དང་པོ་ཨ་དྲི་རུ་བསྒྱུར་བར་འགྱུར་རོ། །སརྦ་དྲྱང་། །བི་ཤྭ་དྲྱང་། །ཡ་དྲྱང་། ཏ་དྲྱང་། ཨེ་ཏ་དྲྱང་། །ཨ་དསའི་ཨ་དྲི་རུ་བསྒྱུར་བ་བྱས་པ་ན། ཨ་དས་ཀྱི་ཚིག་ལ་མའོ་ཞེས་པས་ད་ཡིག་གི་མ་ཉིད་དེ། མ་ལས་ཨུ་ཉིད་དོ་ཞེས་པས་མ་
ལས་ཕ་རོལ་གྱི་རེ་ཕའི་ཨུ་ཉིད་དོ། །ཁ་ཅིག་ནི་ཕ་རོལ་འདོད་པར་བྱེད་དོ། །ཁ་ཅིག་ནི་སྔ་མ་འདོད་པར་བྱེད་ཅིང་། ཁ་ཅིག་གཉི་ག་འདོད་པར་བྱེད་ལ། ཁ་ཅིག་གཉི་ག་མི་འདོད་པར་བྱེད་པ་ལས། ཕ་རོལ་ནི་རེ་ཞིག་ཨ་དྲི་རུ་བསྒྱུར་བ་ན་དའི་མ་ཉིད་དེ། མ་ལས་ཨུ་ཉིད་དོ་ཞེས་པས་མ་ལས་ཕ་རོལ་
