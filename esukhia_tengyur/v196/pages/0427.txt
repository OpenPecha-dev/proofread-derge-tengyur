ལ་ཡ་ཏི། ཧྭ་ལ། ཧྨ་ལ་གཡོ་བར་བྱེད་པ་ལའོ། །པྲ་ཧྭ་ལ་ཡ་ཏི། པྲཧྨ་ལ་ཡ་ཏི། སྨྲྀ་ཀློག་པར་བྱེད་པ་ལའོ། །དགའ་བའི་ཡུལ་དྲན་ནས་རྗོད་པར་བྱེད་དོ། །སྨྲྀ་སེམས་པ་ལའོ། །སྨྲ་ར་ཡ་ཏི། ཨཏྱ་ནུ་བི་སྨ་ར་ཡ་ཏི། ཤྭ་ཙ་དྲན་པར་བྱེད་པ་ལའོ། །ཨ་ནུ་སྔོན་མ་ལས་
ཨཱ་ནུ་སྨཱ་ར་ཡ་ཏི། དྲྀ་འཇིགས་པ་ལའོ། །དྲཱྀ་རྣམ་པར་འཇོམས་པ་ལའོ། །ད་ར་ཡ་ཏི། པོ་ཏཱ་ན། གཞན་ལ། བི་དཱ་ར་ཡ་ཏི་བྷཱུ་མིཿ ཨ་བ་དྷཱ་ར་ཡ་ཏི། ནྲྀ་ཚུལ་ལའོ། །ཚུལ་བཞིན་ལ་རབ་ཏུ་སྦྱོར་བར་བྱེད་པ་ལ། ནཱ་ར་ཡ་ཏི། ཤྲཱ་འཚེད་པ་ལའོ། །ཤྲཱ་ཏི། འགའ་ཞིག་འཚེད་པ་ལ་
རབ་ཏུ་སྦྱོར་བར་བྱེད་པ་ནི། ཤྲ་པ་ཡ་ཏི་བྷཀྟཾ། འདྲེན་པར་བྱེད་པ་ལ། ཤྲཱ་བ་ཡ་ཏི། འཆི་བ་དང་དགའ་བ་དང་རྣོ་བར་བྱེད་པ་རྣམས་ལ་ཛྙའོ། །འཆི་བ་ལ་སོགས་པའི་ཡུལ་ལ། ཛ་ན་ཏི། མཱ་ཏུ་བནྡྷ་ཛྙཱ་པ་ཡ་ཏི། འཆི་བ་དང་དགའ་བ་དང་རྣོ་བར་བྱེད་དོ་ཞེས་པའི་དོན་ནོ། །རྣོ་བ་ནི་སོའོ། །གཞན་
དུ་ཤེས་པར་བྱེད་པ་ཞེས་པའི་དོན་ཏེ། བི་ཛྙཱ་པ་ཡ་ཏི་པྲ་བྷུཾ། ཨཱཛྙ་པ་ཡ་ཏི་བི་ཀྲཱནྟཱན། གཡོ་བ་ལ། ཙ་ལི། ཙ་ལ་ཡ་ཏི་མཎྜ་ལཱ་གྲཾ། འདིར་མ་ཡིན་ཏེ། ཙཱ་ལ་ཡ་ཏེ། བུ་ནོར་དང་བཅས་པ་ལ་རུང་ངོ་། །ནུས་པར་བྱེད་པ་ལ་ཙྪ་དིའོ། །ནུས་པ་དང་སྟོབས་དང་སྲོག་ལ་ཡང་ངོ་། །ཙྪ་ད་ཀུན་ཏུ་
སྒྲིབ་པ་ལའོ། །ཙུ་ར་སོགས་སོ། །རང་དོན་ལ་ཨིནའོ། །ཙྪ་ད་ཡ་ཏི། འདིར་ནི་མ་ཡིན་ཏེ། ཙྪཱ་ད་ཡ་ཏི་གྲྀ་ཧཾ། ལྕེ་སྒྲིལ་བར་བྱེད་པ་ལ། ལ་ཌིའོ། །ལ་ཌ་རྣམ་པར་སྒེག་པ་ལའོ། །ལ་ཌ་ཡ་ཏི་ཛིཧྭཱཾ། ཌ་ལ་དག་གི་གཅིག་ཉིད་དོ། །ལ་ལ་ཡ་ཏི་ཛི་ཧྭཱཾ། འདིར་ནི་མ་ཡིན་ཏེ། ལཱ་ཌ་ཡ་ཏི་བཱ་ལཾ། ཡང་ན་
ལཱ་ལ་ཡ་ཏི་བཱ་ལཾ། དགའ་བ་དང་མི་དགའ་བ་དག་ལ་མ་དིའོ། །མ་དཱི་དགའ་བ་ལའོ། །མ་ད་ཡ་ཏི་མི་ཏྲཾ། མ་ད་ཡ་ཏི་ཤ་ཏྲཱུཾ། འདིར་ནི་མ་ཡིན་ཏེ། པྲཱ་མཱ་ད་ཡ་ཏི། དྷྭ་ན་སྒྲ་ལའོ། །དྷྭ་ན་ཡ་ཏི། གཞན་ལ། དྷྭ་ན་ཡ་ཏི། ཛ་ནཱི་དང་ཛྲཱྀ་ཥ་དང་ཀྣ་སུ་དང་། རཉྫ་རྣམས་ལ་མཐའ་མ་ཡིན་པ་རྣམས་
ཀྱང་ངོ་། །ཛ་ནཱི། ཛ་ན་ཡ་ཏི། ཛ་ན་སྐྱེད་པ་ལའོ་ཞེས་པ་འདིའི་ཡང་། ཛ་ན་ཡ་ཏི། རྣམ་པ་གཞན་དུ་ཇི་ལྟར་ན་གཟུགས་དག་གི་ཐ་མི་དད་པའི་ཕྱིར་རོ། །ཛཱ་ཡ་ཏིའི་རྟག་ཏུ། མ་རྗེས་སུ་འབྲེལ་པ་ཉིད་དོན་མེད་པ་ཅན་དུ་འགྱུར་ཞེ་ན། གཞན་ལ་གྲགས་པ་ཉིད་ལས་ཛ་ན་མ་ཡིན་ཞེས་
