འགྱུར་པ་དེའི་ཚེ། པྲི་ཡ་ཙ་ཏྭཱ་ར། པྲི་ཡ་ཙ་ཏྭཱ་རཽ། པྲི་ཡ་ཙ་ཏྭཱ་རཿ། པྲི་ཡ་ཙ་ཏྭཱ་རཾ། པྲི་ཡ་ཙ་ཏྭཱ་རཽ། །ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །བྲི་ཡ་ཙ་ཏུ་རཿ། པྲི་ཡ་ཙ་ཏུ་རཱ། པྲི་ཡ་ཙ་ཏུ་རྦྷྱཱཾ། པྲི་ཡ་ཙ་ཏུ་རྦྷིཿ། དྲུག་པའི་མང་ཚིག་ལ་ནུ་མེད་དེ། པྲི་ཡ་ཙ་ཏུ་རཱཾ། ཀུན་དུ་བོད་པ་ལ་ནི། སཾ་བུདྡྷི་ལ་གཉི་ག་ཐུང་བའོ། །
ཧེ་པྲི་ཡ་ཙ་ཏྭཱཿ། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རཽ། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རཿ། དེ་ཛས། བཞིན་དུ། བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ་ནི། ཝཱ་སྒྲའི་ཨུ་ཉིད་དོ། །པྲི་ཡ་ཙ་ཏུཿ། པྲི་ཡ་ཙ་ཏུ་རཱི། པྲི་ཡ་ཙ་ཏྭཱ་རི། སླར་ཡང་། དེ་བཞིན་དུ་ཊཱ་ལ་སོགས་པ་ལ་སྐྱེས་པ་དང་མཚུངས་སོ། །ཀུན་དུ་བོད་པ་ལ། ཧེ་པྲི་ཡ་ཙ་ཏུཿ།
ཧེ་པྲི་ཡ་ཙ་ཏུ་རཱི། ཧེ་པྲི་ཡ་ཙ་ཏྭཱ་རི།། །།ཀ་ཏིའི་སྒྲའི་ནི། ཀ་ཏིལ་ཡང་ཛས་ཤས་དག་འཇིག་གོ། །ཀ་ཏི། ཀ་ཏི། ཀ་ཏི་བྷིཿ། ཀ་ཏི་བྷྱཿ། ཀ་ཏི་བྷྱཿ། ཨཱ་མ་ལའང་ནུའོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་སྟེ། ནུ་དང་བཅས་པའི་ཨཱམ་ལ་རིང་པོའོ་ཞེས་པས་རིང་པོ་ཉིད་དོ། །ཀ་ཏི་ནཱཾ། ཀ་ཏི་ཥ། དེ་
བཞིན་དུ། བུད་མེད་ལ་ཡང་ངོ་། །མ་ནིང་ལ་ནི། ཀ་ཏི། ཀ་ཏི། གཟུགས་སོ་སོ་བ་དང་དབྱེ་བ་མེད་ཅིང་། དྲུག་རྣམས། དྲུག་རྣམས་ལ། དྲུག་རྣམས་ཀྱིས། དྲུག་པའི་མང་ཚིག་ལ་ནཱ་མིན་ལས་དབྱངས་ལའོ་ཞེས་པས་ནུ་ཨཱ་ག་མའོ། །ཀཱི་ཏཱི་ནཱཾ།། །།པཉྩན་ལ་སོགས་པ་ན་ཡིག་གི་མཐའ་ཅན་རྣམས་ཀྱི། གྲངས་ལས་
ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པ་སྤྱིར་བཏང་ནས། ཀ་ཏི་ལ་ཡང་ཛས་ཤས་དག་འཇིག་གོ། །བུད་མེད་ལ་ཡང་བུད་མེད་ཉིད་ཁོ་ནའི་དངོས་པོ་མ་ཡིན་པ་ལ་ཨཱིའི་རྐྱེན་མེད་དེ། རྟགས་གསུམ་ཀ་རྣམས་ལ་མཚུངས་པ་ཉིད་ཀྱི་གཟུགས་སོ། །པཉྩ། པཉྩ། པཉྩ་བྷིཿ། པཉྩ་བྷྱཿ། པཉྩ་བྷྱཿ། གྲངས་
ལས་ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་དང་། ན་མཐའི་འང་ཨུ་བ་དྷཱའི་འོ་ཞེས་པས་རིང་པོ་ཉིད་དང་། རྟགས་མཐའི་ན་ཡིག་གིའོ་ཞེས་པས་ན་དབྱི་བའོ། །པཉྩ་ནཱཾ། པཉྩ་སུ།། །།ཥཥྚའི་སྒྲའི་ནི། ཧ་དང་ཤ་དང་ཥ་དང་ཛས། ཚ་ཤེས། མཐའ་དང་ཡ་ཛ་སོགས་རྣམས་ཀྱི་ཌའོ་ཞེས་པས་
ཌ་ཉིད་དེ། རྫོགས་པ་ལ་ཡང་ནའོ་ཞེས་པས་རྫོགས་པ་ལ་ཊ་ཡིག་གམ་ཌ་ཡིག་དག་གོ། །ཥ་ཊ། དྲུག་རྣམས། ཥ་ཌ། དྲུག་རྣམས་ལ། སླར་ཡང་དེ་བཞིན་དུ། སྔ་མ་བཞིན་དུ་ཌ་ཉིད་དོ། །ཥ་ཌྦྷིཿ། ཥཌྦྷྱཿ། ཥཌྦྷྱཿ། གྲངས་ལས་ཥ་ཎའི་མཐའ་ལས་སོ་ཞེས་པས་ནུ་ཨཱ་ག་མ་བྱས་པ་ན་ཥ་ཡིག་གི་ཌ་ཡིག་
