རི་གྲི་དྷཱིཏ། ཨ་ཛ་གྷརྟ། ཨ་ཛ་གྷརྡ། ཨ་ཛ་རྒི་དྷཱི་ཏ། ཨ་ཛཱ་རཱི་གྲི་དྷཾ། ཨ་ཛ་རྒི་དྷཱཾ། ཨ་ཛ་རཱི་གྲི་དྷུཿ། ཨ་ཛ་རི་གྲི་དྷུཿ། ར་ལ་སོགས་པས་སྔ་མ་བཞིན་ནོ། །ཨ་ཛྡ་རཱི་གྷཿ། ཨ་ཛ་རི་གྷཿ། ཨ་ཛ་རྒཿ། ཨ་ཛ་རྒྟཿ། ཨ་ཛ་རཱི་གྷ་རྟ། ཨ་ཛ་རི་
རྒརྟ། ཨ་ཛརྒ་ཏ། ཨ་ཛ་རཱི་གྷདྡ། ཨ་ཛ་རི་གྷརྡ། ཨ་ཛ་རི་གྲི་དྷཱིཿ། ཨ་ཛ་རི་གྲི་དྷཱིཿ། ཨ་ཛ་རྒྲི་དྷཱིཿ། ཨ་ཛ་རཱི་གྲི་དྷཱཾ། ཨ་ཛ་རི་གྲི་དྡྷཾ། ཨ་ཛརྒྲི་དྷཾ། ཨ་ཛ་རཱི་གྲིཌྷ། ཨ་ཛ་རཱི་གྲིཌྷ། ཨ་ཛགྲིཌྷ། ཨ་ཛ་རཱི་གྲི་དྷཾ། ཨ་ཛ་རི་གྲི་དྷཾ། ཨ་ཛ་རཱི་གྲི་དྷཾ། ཨ་
ཛྡརྒྲི་དྷཾ། ཨ་ཛ་རཱི་གྲི་དྷྭ། ཨ་ཛ་རི་གྲི་དྷྭ། ཨ་ཛ་རྒྲི་རྡ། ཨ་ཛ་རཱི་གྲི་དྷྨ། ཨ་ཛ་རི་གྲི་དྷྨ། ཨ་ཛ་རྒྲ་དྷྨ། ཀུ་རྡ། ནཱ་མིནའི་ར་བ་དག་ཅེས་པ་ལ་སོགས་པས་རིང་པོའོ། །ཨ་ཙོ་ཀཱུ་རྟྟ། ཨ་ཙོ་ཀཱུ་རྡ། ཨ་ཙོ་ཀཱུ་རྡ་ཏི། སྒྲ་མེད་པ་རྣམས་ལ་ཤིཊ་མ་ཡིན་པ་རྣམས་
ཀྱི་དང་པོའོ། །ཨ་ཙོ་ཀཱུརྟྟཱཾ། ཨ་ཙོ་ཀཱུ་རྡུཿ། ཨ་ཙོ་ཀཱུཿ། ཨ་ཙོ་ཀཱུ་རྟ། ཨ་ཙོ་ཀཱུ་རྡ། ཨ་ཙོ་ཀུ་རྡཱིཿ། ཨ་ཙོ་ཀུ་རྟྟཾ། ཨ་ཙོ་ཀཱུ་རྟྟ། ཨ་ཙོ་ཀཱུ་རྡཾ། ཨ་ཙོ་ཀཱུ་རྡྭ། ཨ་ཙོ་ཀཱུ་རྨྨ། ཨ་ནཱ་ན་རྟ། ཨ་ནཱ་ན་རྡཱི་ཏ། ཨ་ནཱ་ན་རྟྟཾ། ཨ་ནཱ་ན་རྡུཿ། ཨ་ནཱ་ནཱཿ། ཨ་
ནཱ་ནརྟ། ཨ་ནཱ་ན་རྡ། ཨ་ནཱ་ན་རྡིཿ། ཨ་ནཱ་ན་རྟྟཾ། ཨ་ནཱ་ནརྟྟ། ཨ་ནཱ་ན་རྡཾ། ཨ་ནཱ་ན་རྡྭ། ཨ་ནཱ་ན་རྡྨ། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨ་པཱ་པ་ཙྱ་ཏ། ཨ་ལོ་ལཱུ་ཡ་ཏ། ཨ་བོ་བྷཱུ་ཡ་ཏ།། །།ཨ་དྱ་ཏ་ནཱི། ཨ་ཙེ་ཙ་
ཡིཏ། ཨ་ལོ་བ་ལ་པཱ་ཏི། སྔ་མ་བཞིན་རང་བཞིན་གཟུང་བ་ལ་ཙེ་ཀྲཱི་ཡི་ཏ་ཞིག་པའི་མཐའི་འང་གཟུང་ངོ་ཞེས་པའི་ཚིག་གི་ཨནཱི་ཨུསའོ། །ཨ་བོ་བྷཱུ་ཏ། ཨ་བོ་བྷཱུ་ཏཱཾ། ཨ་བོ་བྷུ་བུཿ། དངོས་པོ་དང་ལས་ལའང་ཨཱཏྨ་ནེའི་ཚིག་གོ། །ཨི་ཙ་ལ་སོགས་པ་
