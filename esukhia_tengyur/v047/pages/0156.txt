སྒྲུབ་ཐབས་ཀྱི་མཎྜལ་གྱི་ཆོ་ག་ཞེས་བྱ་བ། དཔལ་གཤིན་རྗེ་གཤེད་ལ་ཕྱག་འཚལ་ལོ། །བླ་མ་རྡོ་རྗེ་ཡེ་ཤེས་དང་། །རྗེ་བཙུན་མགོན་ལ་ཕྱག་འཚལ་ལོ། །དངོས་གྲུབ་སློབ་དཔོན་མཉེས་བྱའི་ཕྱིར། །རྗེ་བཙུན་དཀྱིལ་འཁོར་དམ་པ་ཡི། །ཆོ་ག་བདག་གིས་བཤད་པར་བྱ། །ཡིད་དུ་འོང་བའི་ས་ཕྱོགས་
སུ། །ས་གཞི་བཟང་པོ་ཡིད་འོང་ལ། །རིན་ཆེན་སྤོས་ཆུས་ལེགས་བཀྲུས་ཏེ། །མཉམ་པའི་ས་ལ་དྲི་སོགས་སྦྱིན། །མཛེས་པའི་གཞི་ལ་ཐིག་ལེ་བཞག །ཏཱ་མ་ལ་ཡི་འོ་མའམ། །རས་བལ་གཙང་མས་དགོད་པར་བྱ། །སེམས་ཅན་ཀུན་ལ་རྗེས་བརྩེ་བས། །ཕྱི་ནང་ཁྲུས་ལ་གནས་ནས་
ཀྱང་། །སྐད་ཅིག་བསྐྱེད་ལ་ཚོགས་བསག་བྱ། །སྲུང་དང་ཆོས་འབྱུང་གཞལ་ཡས་ཁང་། །བསྐྱེད་ལ་དེ་ཡི་ནང་དུ་ནི། །མེ་ཏོག་བཟང་པོ་དགྲམ་པར་བྱ། །ཞེ་སྡང་རྡོ་རྗེ་ལས་བརྩམས་ཏེ། །རལ་གྲི་རྡོ་རྗེ་རབ་ཏུ་བསྐྱེད། །ཚོམ་བུ་རེ་ལ་ལན་བདུན་ཏེ། །བཟླས་པ་པདྨ་ཉི་ཟླའི་སྟེང་། །བདེ་
གཤེགས་སྲས་བཅས་བསམ་པར་བྱ། །ཨོཾ་ཧྲཱིཿཥྚྲྀཿཝི་ཀྲྀ་ཏཱ་ན་ན་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཛི་ན་ཛྀཀ །ཨོཾ་རཏྣ་དྷྲྀཀ །ཨོཾ་ཨཱ་རོ་ལྀཀ །ཨོཾ་པྲཛྙཱ་དྷྲྀཀ །ཨོཾ་མུངྒ་ར་དྷྲྀཀ །ཨོཾ་དཎྜ་དྷྲྀཀ །ཨོཾ་པདྨ་དྷྲྀཀ །ཨོཾ་ཁཌྒ་དྷྲྀཀ །ཨོཾ་མོ་ཧ་ར་ཏི། ཨོཾ་དྭེ་ཥ་ར་ཏི། ཨོཾ་རཱ་ག་ར་ཏི། ཨོཾ་བཛྲ་ར་ཏི། སོ་སོའི་སྐུ་མདོག་ཕྱག་

བདུད་རྩི་མྱང་བ་དང་། །སྤྲོ་བསྡུ་དང་ནི་བཟླས་པ་རྣམས། །རྣལ་འབྱོར་པ་ཡིས་ལེགས་བྱས་ཏེ། །འདོད་པའི་དོན་རྣམས་གསོལ་བ་གདབ། །གཤེགས་དང་བཟོད་པ་གསོལ་བྱས་ཏེ། །དེ་ཉིད་གསུམ་གྱི་ཆོ་ག་ཡིས། །ཡེ་ཤེས་འཁོར་ལོ་གཤེགས་སུ་གསོལ། །དམ་ཚིག་བསྐྱེད་པའི་ལྷ་རྣམས་ནི། །
སྙིང་ག་ཉིད་དུ་རྣམ་པར་བསྟིམ། །གཉིས་དོན་དམ་པ་བདེན་པ་ཡི། །ཆོས་ཀྱི་ཚུལ་ལ་གནས་པར་བྱ། །སེམས་ཅན་ཀུན་གྱི་དོན་གྱི་ཕྱིར། །བྱང་ཆུབ་ཏུ་ནི་བསྔོ་བ་དང་། །སྨོན་ལམ་ཆེན་པོ་གདབ་པར་བྱ། །དེ་ལྟར་ཉིན་མཚན་དུས་དྲུག་ཏུ། །སྙིང་བརྩེ་བརྩོན་པས་དབུལ་བར་བྱ། །གཡེང་བ་རྣམ་པར་
