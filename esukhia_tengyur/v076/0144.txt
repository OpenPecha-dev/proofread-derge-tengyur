སཱུ་ཙི་ཀ་བཛྲ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བྷཉྫ་བྷཉྫ་སརྦ་དུཥྚ་སུ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་བི་ཤྭ་སཱུ་ཙི་ཀ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བྷཉྫ་བྷཉྫ་སརྦ་དུཥྚ་སུ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱ་ལཱི་ཌྷ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷུརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་དུཥྚ་དེ་བ་ཨཱ་དི་མཱ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་ཏྱཱ་ལཱི་ཌྷ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷུརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་
དུཥྚ་ཡ་མ་ཨཱ་དི་མཱ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་བི་ཤེ་ཥ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷུརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་དུཥྚ་ནཱ་གཱན་ཧཱུཾ་ཕཊ། ཨོཾ་མཎྜ་ལ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷུརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་དུཥྚ་ཡཀྵཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ས་མ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་དུཥྚ་ཨནྟ་རིཀྵཱ་མཱ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ཛཱ་ཏ་
པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་བེ་ཤ་ཡ་ཨཱ་བེ་ཤ་ཡ་སརྦ་མཱ་རཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་ཏི་ཛཱ་ཏ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨ་བརྫ་ཡ་ཨ་བརྫ་ཡ་སརྦ་པྲ་སནྣཱན་ཧཱུཾ་ཕཊ། ཨོཾ་བི་ཀ་ཊ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་ཥ་ཡ་བི་ཥ་ཡ་སརྦ་དུཥྚཱན་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་ཏི་བི་ཀ་ཊ་པཱ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བི་གྷུརྞ་ཡ་བི་གྷུརྞ་ཡ་སརྦ་དུཥྚཱན་ཧཱུཾ་ཕཊ། ཨོཾ་སཾ་

བུདྡྷ་བ་ལ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱཀྵེ་པ་ཡ་ཨཀྵེ་པ་ཡ་སརྦ་མཱ་རཱན་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱཿཧཱུཾ་ལ་ལ་ལ་ལ་ལ་ལ་ཧེ་ཧེ་གྲ་ས་གྲ་ས་ཀ་རཱ་ལ་བ་ད་ན་མ་ཧཱ་ཀྲོ་དྷ་ཙ་ལ་ཡ་ཙ་ལ་ཡ་ཧཱུཾ་ཀཿ། ཨོཾ་ཨཱཿཧཱུཾ་ལ་ཨཱ་ཧེ་ཧེ་གྲ་ས་གྲ་ས་ཀ་རཱ་ལ་བ་ད་ན་མ་ཧཱ་ཀྲོ་དྷ་ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ་ཧཱུཾ་གཿ། ཨོཾ་ཨཱཿཧཱུཾ་ལ་ཨཱ་ཧེ་ཧེ་གྲ་ས་གྲ་ས་ཀ་རཱ་
