ཚོན་དགྱེ་བ་དང་། བུམ་པ་དགོད་པ་དང་། དཀྱིལ་འཁོར་བསྒྲུབ་པ་དང་། དཀྱིལ་འཁོར་གྱི་ལྷ་ཐམས་ཅད་བསྡུས་ཏེ་བསྒོམ་པ་དང་། སྐུ་གཟུགས་ལ་སོགས་པའི་རབ་ཏུ་གནས་པ་དང་། བགྲང་ཕྲེང་རབ་ཏུ་གནས་པ་དང་། བགྲང་ཕྲེང་གི་མཚན་ཉིད་དང་། རྒྱལ་མཚན་གཟུགས་པ་དང་། རྫིང་ལ་སོགས་
པའི་རབ་ཏུ་གནས་པ་དང་། ཨ་མྲའི་སྐྱེད་མོས་ཚལ་ལ་སོགས་པའི་རབ་ཏུ་གནས་པ་དང་། དགེ་བསྙེན་གྱི་སྡོམ་པ་ལ་གནས་པ་དང་། དབང་བསྐུར་བ་དང་། རང་གིས་དབང་བླང་བ་དང་། རྩ་བའི་ལྟུང་བ་དང་། སྦོམ་པོའི་ལྟུང་བ་དང་། སྦྱིན་སྲེག་དང་། ཡིད་ཀྱི་སྦྱིན་སྲེག་དང་། ནང་གི་སྦྱིན་སྲེག་དང་།
ཚོགས་ཀྱི་བཟའ་བ་དང་། དཀྱིལ་འཁོར་ཉེ་བར་བསྡུ་བ་དང་། ཡིད་ཀྱི་དཀྱིལ་འཁོར་དང་། གཏོར་མ་དང་། གཟའ་ཡུམ་ལ་སོགས་པས་བདེ་ལེགས་སུ་བྱེད་པ་དང་། བགེགས་བསལ་བ་དང་། སྐུ་གཟུགས་ཀྱི་མཚན་ཉིད་དང་། ཁམས་ཀྱི་ཡོན་ཏན་དང་། ཤིང་ལ་སོགས་པའི་སེལ་དབྱུང་བ་དང་།
མཆོད་རྟེན་གྱི་མཚན་ཉིད་དང་། རྡོ་རྗེ་དང་རྡོ་རྗེ་དྲིལ་བུའི་མཚན་ཉིད་དང་། དྲིལ་བུ་དང་རྡོ་རྗེ་གཟུང་བ་དང་། ཁ་ཊྭཱཾ་ག་དང་ཌ་མ་རུ་ལ་སོགས་པ་སྣ་ཚོགས་པའི་མཚོན་ཆའི་མཚན་ཉིད་དང་། ནགས་ཚལ་གྱི་ལྟད་མོ་དང་། ཁྲུའི་ཚད་དང་། གཎྜཱིའི་མཚན་ཉིད་དང་། མཆོད་རྟེན་ལ་སོགས་པའི་སྲོག་ཤིང་
གཟུགས་པ་དང་། སྒོའི་ཤིང་གཟུགས་པ་དང་། ཀ་བ་གཟུགས་པ་དང་། གཙུག་ལག་ཁང་ལ་སོགས་པའི་སྟེང་དུ་ཤིང་དགོད་པ་དང་། དྲིའི་གཙང་ཁང་ལ་སོགས་པའི་མཚན་ཉིད་དང་། ཁང་པ་བརྩེགས་མའི་མཚན་ཉིད་དང་། རྒྱ་ཕུབས་ལ་སོགས་པའི་མཚན་ཉིད་དང་། བསམ་གཏན་གྱི་ཁང་པ་
ལ་སོགས་པའི་མཚན་ཉིད་དང་། ཤི་བའི་རུས་བུ་དག་པར་བྱ་བའི་ཆོ་ག་སྟེ། ཆོ་ག་དྲུག་ཅུ་རྩ་བཞི་བསྒྲུབ་པར་བྱ་བའོ། །དེ་ལ་དགེ་བའི་ཉིན་པར་གཏོར་མ་སྦྱིན་པ་སྔོན་དུ་འགྲོ་བས་གཙུག་ལག་ཁང་ལ་སོགས་པའི་ས་ཡོངས་སུ་བརྟག་པར་བྱའོ། །དེ་ལ་དཀར་པོ་དང་དམར་པོ་དང་སེར་པོ་དང་ནག་པོའི་
ཁ་དོག་རྣམས་དང་། ཁུ་བ་དང་། རྡུལ་དང་། གཅི་བ་དང་། བཤང་བའི་དྲི་རྣམས་དང་། མངར་བ་དང་། བསྐ་བ་དང་། སྐྱུར་བ་དང་། ཚ་བའི་རོ་རྣམས་རིམ་པ་བཞིན་དུ་བྲམ་ཟེ་ལ་སོགས་པའི་སར་འགྱུར་རོ། །དཀར་པོའི་ས་ལས་ནི་ཞི་བའི་འབྲས་བུ་སྟེར་ཞིང་། དམར་པོའི་ཁ་དོག་ལས་ནི་དབང་
