ཀྲོ་དྷ་ཨཀྵེ་པ་ཡ་ཨཀྵེ་པ་ཡ་སརྦ་བྷཱུ་ཏཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷཻ་ར་བ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཤོ་དྷ་ཡ་ཤོ་དྷ་ཡ་སརྦ་ཏྲཻ་དྷཱ་ཏུ་ཀཱན་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུཎྜ་ལི་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་གྲྀཧྞ་བནྡྷ་ཧ་ན་བི་དྷཱན་ས་ཡ་བཛྲེ་ཎ་མཱ་རཱ་ན་ཧཱུཾ་ཕཊ། ཡ་བྷིནྡ་ཡ་ཧྲྀ་ད་ཡཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནྲྀ་ཏྱ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་གྷ་ཏ་ཡ་གྷ་ཏ་ཡ་
སརྦ་དུཥྚ་ནཱ་གཱན་ཧཱུཾ་ཕཊ། ཨོཾ་སྟྲི་མཱ་ཡཱ་ཀ་ར་ཎ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་ན་ཡ་ཨཱ་ན་ཡ་བི་མརྡ་ཡ་བི་མརྡ་ཡ་སརྦ་ཀིནྣ་ར་ཨཱ་དཱི་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ཁཌྒ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཚིནྡྷ་ཚིནྡྷ་སརྦ་དུཥྚ་བི་ནཱ་ཡ་ཀཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་པྲ་མརྡ་ཡ་པྲ་མརྡ་ཡ་སརྦ་དེ་བཱན་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་མུ་ཁ་ལ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཀུཊྚ་ཀུཊྚ་སརྦ་དུཥྚ་བིགྷཱ་བི་ནཱ་ཡ་ཀཱན་ཧཱུཾ་ཕཊ། ཨོཾ་ཙཀྲ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་ནཱ་ཡ་ཨཱ་ནཱ་ཡ་སརྦ་མཎྜ་ལཱན་ཧཱུཾ་ཕཊ། ཨོཾ་པཱ་ཏཱ་ཀ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་པྲ་སཱ་ར་ཡ་པྲ་སཱ་ར་ཡ་སརྦ་པྲ་ཏཱན་ཧཱུཾ་ཕཊ། ཨོཾ་པདྨ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་
ཨཱ་བེ་ཤ་ཡ་ཨཱ་བེ་ཤ་ཡ་སརྦ་དུཥྚ་ས་ཏྲཱན་ཧཱུཾ་ཕཊ། ཨོཾ་དཎྜ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏཱ་ཊ་ཡ་ཏཱ་ཊ་ཡ་སརྦ་ཤ་ཏྲཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཾ་ཀུ་ཤ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ་སརྦ་དུཥྚ་སཏྭ་ནཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཕ་ལ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་པྲ་ཧླ་ད་ཡ་པྲ་ཧླ་ད་ཡ་སརྦ་ཏ་ཐཱ་ག་ཏཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་
ཙཱ་པ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏརྫ་ཡ་ཏརྫ་ཡ་སརྦ་དུཥྚཱན་ཧཱུཾ་ཕཊ། ཨོཾ་ཤ་ར་ཨཱ་ཀཪྵ་ཎ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ་སརྦ་མཱ་ར་པཪྵ་དོ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤ་ར་ཨུ་དཀྵེ་པ་ཎ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་དེ་ཧི་དེ་ཧི་པ་རཾ་སརྦ་སཏྭན་ནཱཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨུ་ད་བ་ལི་ནྟ་ཨ་བྷི་ནཱ་
ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བནྡྷ་ཡ་བནྡྷ་ཡ་སརྦ་ས་མ་ཡཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་སིཾ་ཧ་པ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་སྟཾ་བྷ་ཡ་སྟཾ་བྷ་ཡ་སརྦ་མ་ཧལླ་ཀ་མ་ཧལླཀ་ཨཱ་དཱིན་ཧཱུཾ་ཕཊ། ཨོཾ་སིཾ་ཧ་བི་ཛྲིམྦྷི་ཏ་པ་ད་བཛྲ་མ་ཧཱ་ཀྲོདྷ་ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ་སརྦ་མ་ཧལླ་ཀ་མ་ཧལླ་ཀ་ཨཱ་དཱི་ན་ཧཱུཾ་ཕཊ། ཨོཾ་སིཾ་ཧ་ཀྲཱནྟ་པ་ད་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་ཀྲཱ་
མ་ཡ་སརྦ་མ་ཧལླ་ཀ་མ་ཧ་ལླ་ཀ་ཨཱ་དཱི་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ག་ར་ཊ་ཨཱ་སཱ་ན་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ་སརྦ་ནཱ་གཱན་ཧཱུཾ་ཕཊ། ཨོཾ་བྷིནྡ་པཱ་ལ་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བྷྲ་མ་ཡ་བྷྲ་མ་ཡ་སརྦ་དུཥྜཱ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་བ་ར་ད་ཨ་བྷི་ནཱ་ཡ་བཛྲ་མ་ཧཱ་ཀྲོ་དྷ་བ་ར་ད་བ་ར་ད་སརྦ་སིདྡྷིཾ་ཀུ་རུཿཧཱུཾ་ཕཊ། ཨོཾ་ཤཀྟི་
