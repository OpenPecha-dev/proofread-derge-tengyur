རྣམས། །འབྱིན་ཅིང་སྡུད་པའི་རྒྱུ་ཡིས་སུ། །དཀྱིལ་འཁོར་འཁོར་ལོའི་གཟུགས་སུ་ནི། །མཁའ་འགྲོ་མ་ཀུན་སྐྱེས་པའོ། །ད་ནི་སྣ་ཚོགས་པའི་དགོངས་པ་མཆེ་བ་རྣམ་པར་གཙིགས་པའི་བཤད་པ་གསུངས་པ། དེ་ནས་བཤད་ཅེས་བྱ་བ་ལ་སོགས་
པ་སྟེ། མཆེ་བ་རྣམ་པར་གཙིགས་པ་ཡི། །ཞེས་བྱ་བ་ནི་དཔའ་བོ་དེའི་མིང་གི་ཡི་གེས་སོ། །ཏིང་འཛིན་སྦྱོར་བ་ལ་གནས་པ། །ཞེས་བྱ་བ་ནི་སྦྱོར་བ་ཡིས་གནས་པ་པོ་ཨ་ཝ་དྷཱུ་ཏཱིའི་དཀྱིལ་འཁོར་གྱི་དུས་སུ་ཏིང་ངེ་འཛིན་ལ་སྙོམས་པར་ཞུགས་པས་སོ། །གཞུ་འགེངས་
པའི་རིམ་པས་ཞེས་བྱ་བ་ནི་ཏིང་ངེ་འཛིན་གང་དེ་རྣལ་འབྱོར་པ་ཡིས་ཡོད་པར་བྱས་པ་སྟེ། རྡོ་རྗེ་ཞེས་བྱ་བ་ནི་དབྱེར་མེད་པའི་ཡེ་ཤེས་ཀྱི་གཟུགས་ཅན་ནོ། །དེ་ལྟར་གཞུ་དགང་བ་གསུངས་པ། སྲོག་གི་ཐིག་ལེར་སོན་པར་བསམ། །ཞེས་བྱ་བ་སྟེ། དབུས་མའི་ནང་དུ་སོན་པར་བྱས་
ཏེ། དེ་ཉིད་རྒྱུད་ཅེས་བྱ་བ་ནི་ལྷན་ཅིག་སྐྱེས་པ་གཅིག་པུའི་མཚན་ཉིད་ཅན་གྱི་གཞུའི་རྒྱུད་དོ། །མཚམས་ཀུན་དུ་ཞེས་བྱ་བ་ནི་རྩ་ཐམས་ཅད་ཀྱི་མཚམས་སུ་གནས་སོ། །སྒྲ་ཐམས་ཅད་དུ་ཞེས་བྱ་བ་ནི་[ཀྵྨྲྻུཾ་]ལ་སོགས་པ་ས་བོན་གྱི་ཡི་གེ་བཞི་བརྗོད་པས་སྐུ་འགོག་པར་
བྱའོ། །ལྷག་མ་རྟོགས་པར་སླའོ། །ད་ནི་རྣལ་འབྱོར་འདིའི་ཡན་ལག་རྒྱུ་དང་འབྲས་བུའི་བདག་ཉིད་ཅན་མཆེ་བ་རྣམ་པར་གཙིགས་པས་བསྲུང་བའི་སྦྱོར་བ་གསུངས་པ། མཆེ་བ་རྣམ་པར་གཙིགས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ་ངེས་པར་འབྱུང་བ་ན་མྱ་ངན་ལས་འདས་
པའོ། །འཇུག་པའི་ཞེས་བྱ་བ་ལ་སོགས་པས་བིའི་ཡི་གེའི་དོན་ཏེ་ཟག་མེད་ཅེས་བྱ་བ་ནི་སྐྱེ་བ་སྣ་ཚོགས་པའི་ནང་དུ་མ་རིག་པ་ལ་སོགས་པས་སྡིག་པ་གང་བྱས་པའི་སེམས་ཅན་ཐམས་ཅད་ཀྱི་འཇུག་པ་སྟེ་མྱ་ངན་ལས་འདས་པའི་སྐུ་བཟུང་བས་སོ། །རང་ལས་བྱུང་བའི་
སྐྱེ་བོ་རྣམས་གཞོམ་པར་བྱ་བ་སྟེ་དབུག་པར་བྱ་བའོ། །དེའི་ཕྱིར་གཞན་གྱིས་གྲོང་དུ་བྱེད། །ཅེས་བྱ་བ་ནི་གའི་ཡི་གེའི་དོན་གྱིས་ལས་མྱ་ངན་ལས་འདས་པའི་སྤྱོད་ཡུལ་ཅན་བྱེད་དུ་འཇུག་པའོ། །སླར་ཡང་སྦྱིན་པ་ཆེན་པོ་ལ་སོགས་པ་ཡང་དག་པར་འཇུག་པར་བྱེད་པའི་ལས་
