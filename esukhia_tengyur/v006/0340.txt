ཡོད་མེད་བྱེད། །ཐ། སྲོག་ནི་རྡོ་རྗེ་ལྟ་བུར་བརྟན། །བཱ། བག་ཆགས་བདེ་བ་ཙཾ་ནི་ཉིད། །རྡོ་རྗེ་རླུང་བུ་བསྟན་པ་ཡིན། །འདི་ནི་སྦྱོར་བའི་ལམ་ཡིན་ནོ། །དེ་ནས་མཐོང་བའི་ལམ་གསུངས་པ། །རི་དགས་མིག་ལ་སོགས་པ་ཡི། །འདིར་རི་དགས་བཞིན་དུ་རྣལ་འབྱོར་མིག །བསྒོམ་པ་རྟོག་པ་ཡི་ནི་སེམས། །
རྒྱ་ཆེ་བར་ནི་ལྟ་བའི་ལུས། །དབུ་མ་ཕྲ་མ་སྤངས་པའོ། །བདེ་དང་སྡུག་པ་རོ་སྨད་ཉིད། །ཆོས་ཀྱི་དབྱིངས་ནི་རྒྱ་ཆེའི་ཕྱིར། །ནུ་མ་མཐོན་པོ་ཞེས་པ་འདིར། །སྲིད་དང་མྱ་ངན་འདས་པ་ཉིད། །ལྟ་བའི་སྡོང་པོ་སྤངས་པ་ཉིད། །དེ་དག་གཉིས་མེད་མཆོག་ཏུ་འགྱུར། །སྐལ་བཟངས་ཞེས་པ་འདི་ནི་གསུངས། །
སུ། །བདེར་གཤེགས་འཆིང་བ་དང་ནི་བྲལ། །བྷ། རང་གི་བག་ཆགས་ཀྱིས་ནི་འཁོར། །གཱ། སེམས་ཅན་རྣམས་ཀྱི་རིགས་ནི་གཟུང་། །དེ་ཡང་བསམ་མེད་གཉིས་སུ་མེད། །སྒྱུ་མ་ཉིད་དེ་སྒྱུ་མའི་སྒྱུ། །འཁྲུལ་བས་སྨིག་རྒྱུའི་ཚོགས་ལ་ནི། །ཆུ་ཡི་རྣམ་པ་ཉིད་དུ་མཐོང་། །འཁྲུལ་བའི་ཕྱིར་ན་ཆུ་ནི་མེད། །
དེ་ལྟར་ཐམས་ཅད་དུ་ནི་དོར། །ཟབ་མོའི་ཆོས་ཀྱི་དབྱིངས་ནི་ཉིད། །བཅོམ་པའི་གཟུགས་ལ་འདས་པའི་སྒྲ། །བསྐལ་པའི་དང་པོར་རྟག་ཏུ་གནས། །དམ་ཚིག་ཅེས་པ་དུས་མི་འགྱུར། །སྤྱད་པས་གཉིས་མེད་རབ་ཏུ་འཐོབ། །དེ་ལྟར་ཤིན་ཏུ་གུས་གནས་སུ། །དེ་ཉིད་བརྗོད་མེད་མཆོག་མདུན་དུ། །སྔགས་
དང་རྒྱུད་སོགས་ཐམས་ཅད་ནི། །སྨིག་རྒྱུ་བཞིན་དུ་རྟག་ཤེས་བྱ། །འདི་དག་གཉིས་མེད་བུ་མོ་ཡང་། །གཉིས་མེད་ཆོས་ཉིད་དུ་ནི་གསུངས། །ལུས་ངག་གསུམ་གྱི་སྲིད་གསུམ་དུ། །རིན་ཐང་མེད་པ་བསྒོམ་པའི་བདག །འདི་ཉིད་ཀྱིས་ནི་སེམས་ཅན་ཀུན། །དབང་བསྐུར་ཉིད་བདག་དེ་ཡི་དངོས། །དེ་
ལྟར་བདག་ཉིད་དངོས་གྲུབ་ནི། །ཆོས་རྣམས་ཀྱིས་ནི་རིག་རིམ་གྱིས། །ཡང་ན་ཇི་ལྟ་ཕྱག་རྒྱ་ནི། །ཟབ་མོའི་ཆོས་ཀྱི་ངེས་པས་སོ། །ད་ནི་རྒྱུ་བ་གསུང་པ་སྟེ། །ལོ་ནི་བཅུ་དྲུག་མ་ལ་སོགས། །ལོ་ནི་བཅུ་དྲུག་མ་ཞེས་པ། །འདི་ནི་དགོངས་བཤད་གཞན་ཉིད་ཀྱིས། །འཕོ་བ་བཅུ་དྲུག་ཏུ་ནི་བརྗོད། །དེ་ཡིས་
རྣལ་འབྱོར་མ་རྒྱུ་འགྱུར། །གནས་ནི་བཅུ་དྲུག་གནས་སུའོ། །འཁོར་ལོ་རྩིབས་ནི་བཅུ་དྲུག་ཏུ། །ཐམས་ཅད་དུ་ནི་སྲོག་རྣམས་རྒྱུ། །དེ་བཞིན་འཁོར་ལོའི་དོན་ནི་གསུངས། །ཐིག་ལེ་ལྔ་པ་ལྔ་ཉིད་དེ། །ཕྲེང་བ་བཞིན་དུ་ཡོངས་སུ་གྲགས། །ཟུར་བཞིའི་ཐིག་ལེ་རྣམས་དང་ནི། །དབུས་ཀྱི་ཐིག་ལེ་
