ལ་ཚིགས་སུ་བཅད་པ་གཉིས་ཡོད་དེ། ཤིན་ཏུ་ཕྲ་བའི་དངོས་ལ་ཡང་། །གང་གི་ཆད་པར་རྣམ་བརྟགས་པར། །རྣམ་པར་མི་མཁས་དེ་ཡི་ནི། །རྐྱེན་ལས་སྐྱེས་པའི་དོན་མ་མཐོང་། །འདི་ལ་གསལ་བྱ་གང་ཡང་མེད། །གཞག་པར་བྱ་བ་ཅི་ཡང་
མེད། །ཡང་དག་ཉིད་ལ་ཡང་དག་བལྟ། །ཡང་དག་མཐོང་ན་རྣམ་པར་གྲོལ། །རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བའི་སྙིང་པོའི་རྣམ་པར་བཤད་པ་སློབ་དཔོན་ཀླུ་སྒྲུབ་ཀྱིས་མཛད་པ་རྫོགས་སོ།། ༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨ་བུ་དྷ་བོ་དྷ་ཀ་ནཱ་མ་པྲ་ཀ་
ར་ཎ། བོད་སྐད་དུ། མ་རྟོགས་པ་རྟོགས་པར་བྱེད་པ་ཞེས་བྱ་བའི་རབ་ཏུ་བྱེད་པ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །སྐྱོན་མེད་ཡོན་ཏན་རྒྱ་མཚོ་ཡི། །འཇམ་པའི་དཔལ་ལ་ཕྱག་འཚལ་ཏེ། །དེ་ཉིད་མ་རྟོགས་རྟོགས་བྱེད་པ། །བདག་གིས་
བརྩེ་བས་བྲི་བར་བྱ། །དེ་ནས་གཞན་གྱིས་བརྒལ་བ་དང་། དགག་པ་དང་། དེའི་རྟོགས་པ་དང་། ཆོས་ཀྱི་དེ་ཁོ་ན་ཉིད་དོ། །གཞན་གྱིས་བརྒལ་བ་ནི་གང་ཞེ་ན། སྨྲས་པ། བདག་ཡོད་དོ། །དེ་ཡང་རྟག་པ་དང་། ཁྱབ་པ་དང་། གཅིག་པུ་དང་། བྱེད་པ་པོ་དང་། ཟ་བ་
པོ་ཡིན་ནོ། །འདིར་ཡང་དགག་པ་ནི་བདག་འདི་གང་ཡིན། རེ་ཞིག་ཕུང་པོ་ནི་བདག་མ་ཡིན་ཏེ། སྐྱེ་བ་དང་འཇིག་པའི་ཆོས་ཅན་ཡིན་པའི་ཕྱིར་རོ། །ཕུང་པོ་བདག་ལས་གཞན་ཡང་མ་ཡིན་ཏེ། འདི་ལྟར་བུམ་པ་ལས་སྣམ་བུ་བཞིན་དུ་ཐ་དད་པ་མི་དམིགས་པའི་ཕྱིར་རོ། །བུམ་
པ་ལ་ཆུ་འམ་ཆུ་ལ་སྣམ་བུ་བཞིན་དུ་རྟེན་དང་བརྟེན་པ་ཡང་མ་ཡིན་ནོ། །འབྲས་བུ་དང་ཤུན་པ་བཞིན་དུ་ཕུང་པོའི་གནས་ལུགས་ཀྱང་མ་ཡིན་ནོ། །ཕུང་པོ་ལ་འདུ་བ་ཡང་མ་ཡིན་ཏེ། ཕུང་པོ་རྣམས་ནི་མང་བ་ཉིད་ཀྱི་ཕྱིར་རོ། །ལྷ་སྦྱིན་བ་ལང་དང་ལྡན་པ་བཞིན་དུ་ཕུང་པོ་དང་
ལྡན་པ་ཡང་མ་ཡིན་ནོ། །དེ་ནས་ཕ་རོལ་རྒོལ་བས། འདི་སྐད་སྨྲས་ཏེ། བདག་མེད་ན་དེའི་ཚེ་སྦྱིན་པ་ལ་སོགས་པའི་ལས་དང་འབྲས་བུ་འབྲེལ་པ་དོན་མེད་པར་འགྱུར་ཏེ། ཇི་ལྟར་བ་ལང་ཤི་ན་རྭ་དང་མཇུག་མ་དང་། རྨིག་པ་ལ་སོགས་པ་འགྲུབ་བར་འགྱུར་བ་བཞིན་ནོ་ཞེ་
