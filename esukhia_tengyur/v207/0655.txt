ཨ་གྲ་ཙརྱ་པྲ་ཎི་དྷཱ་ན། བོད་སྐད་དུ། མཆོག་གི་སྤྱོད་པའི་སྨོན་ལམ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །སྐྱེ་དགུ་ཀུན་ལ་སྨན་དང་བདེ་སླད་དུ། །ཕྱོགས་བཅུའི་འཇིག་རྟེན་སངས་རྒྱས་རྣམས་ལ་འདུད། །དེ་དག་ལ་ནི་
བརྗོད་ཅིང་མཆོད་པར་བགྱི། །འགྲོ་ཀུན་བྱང་ཆུབ་ཕྱོགས་ལ་རེག་གྱུར་ཅིག །བྱམས་པ་དང་ནི་འཇམ་དཔལ་མངའ་བདག་ཏོག །ཆོས་ཀྱི་བློ་གྲོས་དཔའ་བོ་བདུད་རྩི་གསུངས། །དཔའ་བོ་དེ་ལ་ཕྱག་འཚལ་མཁྱེན་པ་ལྟར། །དེ་བཞིན་བདག་ཀྱང་མི་དབང་རྣམས་ལ་འདུད། །
ཐུགས་རྗེར་ལྡན་ཞིང་འཇིག་རྟེན་དོན་མཛད་པ། །རྒྱལ་བ་མི་མཁྱེན་མི་གཟིགས་མི་མངའ་བ། །མི་དབང་ཐུགས་བརྩེ་བྱམས་ལ་གནས་པ་ཡི། །རྒྱལ་བ་དེ་ཀུན་བདག་ལ་དགོངས་སུ་གསོལ། །སྔོན་གྱི་སྲིད་པ་དག་ཏུ་བདག་ཉིད་ཀྱིས། །སྡིག་པའི་ལས་རྣམས་གང་བགྱིས་བདག་
གིས་བཤགས། །དེ་བཞིན་ལུས་ངག་ཡིད་ཀྱིས་བགྱིས་པ་རྣམས། །དེ་དག་ཐམས་ཅད་བདག་གིས་སོ་སོར་བཤགས། །ཤཱཀྱ་ཐུབ་དང་མི་འཁྲུགས་འོད་དཔག་མེད། །དེ་དག་གི་ནི་ཕྲིན་ལས་ཡོངས་རྫོགས་དང་། །ཆགས་དང་བཤགས་པ་མཁྱེན་ཅིང་རྟོགས་གྱུར་ལྟར། །
དེ་བཞིན་བདག་གིས་སྡིག་པ་བཤགས་པར་བགྱི། །ཐམས་ཅད་དགའ་མཛད་སྐྱེས་བུ་དམ་པ་དང་། །བྱང་ཆུབ་བཞུགས་པ་ཀུན་ལ་གསོལ་བར་འཚལ། །ངན་སོང་བཞི་པོ་མྱུར་དུ་ཞི་གྱུར་ཅིག །མི་ཁོམ་འཇོམས་པར་བགྱིད་ཅིང་རྒྱུ་བར་ཤོག །སངས་རྒྱས་བཟང་སྤྱོད་དྲི་
མེད་མངའ་བདག་དང་། །དེ་བཞིན་ལྷུན་པོའི་དཔལ་དང་ཉི་མ་སྒྲོན། །དེ་དག་བསོད་ནམས་ཡི་རང་མཁྱེན་པ་ལྟར། །དེ་བཞིན་བདག་ཀྱང་བསོད་ནམས་རྗེ་ཡི་རང་། །ཕྱོགས་བཅུ་དག་ན་སྦྱིན་པ་སྤྱོད་པ་གང་། །དེ་བཞིན་ཏིང་འཛིན་ཚུལ་ཁྲིམས་སྤྱོད་རྣམས་དང་། །ཤེས་
རབ་བསྒོམ་དང་བསོད་ནམས་སྟོབས་ཀུན་ལ། །བྱང་ཆུབ་ཡོན་ཏན་དོན་དུ་ཡི་རང་ངོ་། །རྒྱལ་བ་ཀུན་གྱི་ཡོན་ཏན་མཐའ་ཡས་དང་། །ཆོས་བསྟན་བྱང་ཆུབ་འཇོག་བགྱིད་ཐམས་ཅད་ལ། །བྱང་ཆུབ་ཡོན་ཏན་དོན་དུ་ཡི་རང་ངོ་། །འགྲོ་ཀུན་བྱང་ཆུབ་མཆོག་ལ་རེག་གྱུར་ཅིག།
