མི་རྟོགས་དང་། །བརྐམ་དང་ཟིལ་ནོན་འདོད་པ་ཆེ། །མི་ཆོག་ཉམས་དང་མ་གུས་དང་། །ལྷོད་དང་ཁྲོ་དང་ཁོན་དུ་འཛིན། །མི་བཟོད་གཤེ་དང་བརྡེག་པ་དང་། །མཚང་འདྲུ་འཕྱ་དང་ལེ་ལོ་དང་། །སྙོམ་ལས་ཉིད་དུས་ཚོད་མིན་དང་། །ཉལ་བ་དུས་ཚོད་
གཉིས་དུས་མིན། །ཚོད་མིན་དུས་ཡོལ་གཡེངས་པ་དང་། །མི་གནས་ཡིད་དུལ་རྨོངས་པ་དང་། །ཞུམ་དང་སྡིག་པའི་ཚིག་སྨྲ་དང་། །ལྟ་དང་རྟོག་པའང་དེ་བཞིན་ནོ། །གང་གི་རྒྱུ་དང་སྐྱེས་པ་དང་། །འབབ་དང་བག་གཉིས་ཉེས་མེད་དང་། །ཉེས་དམིགས་ཕྱིར་
བཅོས་མི་འབྱུང་རྒྱུ། །བཅུ་གཅིག་བསླབ་བྱ་ཕན་ཡོན་གཉིས། །གང་ཟག་ལས་དང་པོ་པ། ལས་ཐོག་མ་པ། བདག་ལ་ཕན་པར་འདོད་པ། བདག་གི་དོན་རྗེས་སུ་ཐོབ་པར་འདོད་པས། གནས་བཞི་པོ་ཀུན་ཤེས་པར་འདོད་ཅིང་། ཀླན་ཀའི་སེམས་མེད་པ་དང་། གུས་པ་
དང་བཅས་ཤིང་། ང་རྒྱལ་གྱི་ཁེངས་པ་མེད་པ་དང་། དགེ་བ་གང་ཡིན་ཞེས་ཚོལ་ཞིང་། བདག་ཉིད་སྒྲོགས་པའི་ཕྱིར་མ་ཡིན་པ་དང་། བདག་དང་གཞན་དག་དགེ་བའི་རྩ་བ་ལ་ངེས་པར་སྦྱར་བར་བྱ་ཞིང་། རྙེད་པ་དང་བཀུར་སྟི་དོན་དུ་གཉེར་བ་མེད་པ་དག་ལ་དྲན་པ་ཉེ་བར་བཞག་
ནས། །མཁན་པོའམ་སློབ་དཔོན་ནམ། བླ་མའམ། བླ་མ་ལྟ་བུ་རྣལ་འབྱོར་ཤེས་པ་གང་ཡིན་པའི་དྲུང་དུ་འགྲོ་བར་བྱའོ། །དེ་ལྟར་དུས་སུ་སོང་ནས་ཀྱང་སྐབས་སུ་བབ་པར་བྱ་སྟེ། སྟན་ཤིན་ཏུ་དམའ་བ་ལ་འདུག་ནས་བླ་གོས་ཕྲག་པ་གཅིག་ལ་གཟར་ཏེ། པུས་མོ་གཡས་པའི་
ལྷ་ང་ས་ལ་བཙུགས་ཏེ། རྗེས་དང་བཅས་གུས་པ་དང་བཅས་པས། བཙུན་པ་བདག་ནི་རྣལ་འབྱོར་དོན་དུ་གཉེར་བ་ལགས་ན། བདག་ལ་རྗེས་སུ་ཐུགས་བརྩེ་བའི་སླད་དུ། རྣལ་འབྱོར་བསྟན་དུ་གསོལ་ཞེས་རྣལ་འབྱོར་གསོལ་བར་བྱའོ། །དེ་ལྟར་གསོལ་བ་བཏབ་པའི་རྣལ་
འབྱོར་པ་རྣལ་འབྱོར་ཤེས་པ་དེས་ཀྱང་། གང་ཟག་ལས་དང་པོ་པ། ལས་ཐོག་མ་པ། རྣལ་འབྱོར་ཡིད་ལ་བྱ་བ་ལ་རབ་ཏུ་བརྩོན་པར་འདོད་པ་དེ་ལ། བཞིན་བཟངས་ཁྱོད་སྐྱེ་དགུ་བག་མེད་པ་དང་ལྡན་པ། ཡུལ་ལ་གཞོལ་བ། ཡུལ་ལ་ལྷག་པར་ཞེན་པར་གྱུར་པ། བག་ཡོད་
