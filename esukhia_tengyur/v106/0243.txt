རྙེད་པ་ལས་བྱུང་བའི་བར་མི་འདོད་པ་བཀག་ནས། ད་ནི་མི་འདོད་པ་ལས་བྱུང་བའི་དགའ་བ་དགག་པའི་ཕྱིར། གལ་ཏེ་དགྲ་ཞིག་ཅེས་སྨོས་ཏེ། ཅི་ཞིག་ཡོད་ཅེས་པ་ནི་དེ་ལ་ཅུང་ཟད་ཅིག་ཀྱང་གནོད་པར་མི་འགྱུར་ཞེས་བྱ་བའི་དོན་ཏོ། །གལ་ཏེ་བདག་གིས་བསྔགས་པར་
མ་བྱས་པ་ཉིད་ཀྱིས་དེ་ལ་མི་འདོད་པ་སྐྱེ་བར་འགྱུར་རོ་སྙམ་པ་ལ། ཁྱོད་ཀྱིས་ཞེས་སྨོས་ཏེ། རྒྱུར་མི་འགྱུར་ཞེས་པ་ནི་སྔོན་གྱི་ལས་མེད་པར་ཞེས་བྱའོ། །ཅི་སྟེ་ཁྱོད་ཀྱི་འདོད་པ་ཁོ་ནས་གཞན་གྱི་སྡུག་བསྔལ་དེའི་རྒྱུར་གྲུབ་པ་དེ་ལྟ་ནའང་གཞན་གྱི་སྡུག་བསྔལ་དེ་ལ་ཁྱོད་དགར་ཅི་ཡོད་དེ། འབྲས་བུ་
མེད་པ་ཉིད་དོ་ཞེས་བྱ་བའི་དོན་ཏོ། །འཚེང་བ་ནི་དགོས་པའོ། །དེ་ལས་མ་གཏོགས་པའི་ཕུང་བ་མེད་པ་ནི་འདི་ཉིད་འབྲས་བུ་མེད་པར་ཕུང་བ་ཉིད་དོ་ཞེས་བྱ་བའོ། །ཅིའི་ཕྱིར་ན་འདི་ཕུང་བ་ཡིན་སྙམ་པ་ལ། ཉོན་མོངས་ཞེས་སྨོས་ཏེ། ཉོན་མོངས་པ་ཉིད་མཆིལ་བ་འདེབས་པར་བྱེད་པས་ཉ་པའོ། །མཆིལ་
བ་དང་འདྲ་བས་ན་མཆིལ་བ་སྟེ། ཉ་གཟུང་བར་བྱ་བའི་དོན་དུ་ཤ་དང་བཅས་པའི་ལྕགས་ཀྱུའོ། །མི་བཟད་ཅེས་པ་ནི་ཤིན་ཏུ་གཟེར་བའོ། །གང་ཉ་པ་དང་འདྲ་བའི་ཉོན་མོངས་པ་དེས་བསོད་ནམས་ཀྱི་ཆུ་བོ་ནས་ཉ་དང་འདྲ་བ་ཁྱོད་བཟུང་ནས་ཁ་ཆུང་ལ་གསུས་པ་ཆེ་བའི་བུམ་པར་བཅུག་ཅིང་འཚེད་
པར་འགྱུར་བའོ། །བསྟོད་པ་ལ་སོགས་པ་ལ་ཡང་ཁེངས་པར་མི་བྱའོ་ཞེས་གདམས་པའི་ཕྱིར། བསྟོད་དང་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། སྐྱེས་བུ་རྣམས་ཀྱི་དོན་དུ་གཉེར་བར་འདོད་པའི་རྣམ་པ་ནི་ལྔ་སྟེ། བསོད་ནམས་དང་། ཚེ་དང་། སྟོབས་དང་། ནད་མེད་པ་དང་། ལུས་བདེ་བ་
རྣམས་སོ། །བསྟོད་པ་ནི་ཡོན་ཏན་བརྗོད་པའོ། །སྟོབས་ནི་ལུས་ལས་བྱུང་བའོ། །ཤེས་གྱུར་ནི་མཁས་པར་གྱུར་ནའོ། །དེ་ལ་ཞེས་པ་ནི་རྣམ་པ་ལྔར་བརྗོད་པ་ལའོ། །ཅི་སྟེ་ཡིད་ཀྱང་བདེ་བར་བྱ་རུ་ཡོད་པ་མ་ཡིན་ནམ་དེ་ཇི་ལྟར་ངེས་པར་གཟུང་བར་བྱ་སྙམ་དུ་དོགས་པ་ལ། ཡིད་བདེ་
ཞེས་སྨོས་ཏེ། གཞན་ཡང་བསྟོད་པ་ལ་སོགས་པ་ནི་སླུ་བྱེད་ཉིད་དོ་ཞེས་བསྟན་པའི་ཕྱིར། གྲགས་པའི་ཞེས་སྨོས་ཏེ། ཚིག་འབྲུ་ནི་བསྟོད་པའི་ཚིག་རྣམས་སོ། །བདེ་བ་ནི་གྲགས་པ་ལ་སོགས་པའི་བདེ་བའོ། །ཡང་བརྟག་པ་གཞན་གྱི་སྒོ་ནས་འདི་དག་རྣམས་ནི་བྱིས་པའི་ཆོས་ཉིད་དུ་བསྟན་པའི་
