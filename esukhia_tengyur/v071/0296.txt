གྱིས་རྟ་བབས་སོ། །ཕྱོགས་ཀྱི་གནས་དང་སྙིང་པོའི་གནས་དེ་བཞིན་དུ་ཤེས་པར་བྱའོ། །འཁོར་ལོ་དང་། ཟུར་ལ་རྒྱལ་མཚན་དང་། དྲིལ་བུ་དང་། རྔ་ཡབ་དང་། དཔག་བསམ་གྱི་ཤིང་དང་། དེ་ནས་དཀྱིལ་འཁོར་ལྟེ་བར་ཕུར་པ་བཏབ་ལ། ཕྱི་རོལ་ནས་ཀུན་ནས་བསྐོར་བར་བྱ་སྟེ།
རྡོ་རྗེའི་ཕྲེང་བ་དང་། དེ་བཞིན་རི་མོ་གཉིས་ཀྱིས་བསྐོར། གསུམ་ཡང་བསྐོར་ལ། ཁ་དོག་ལྔ་པའི་གཞི་ཡང་། རིང་པོ་བཞི་པའི་ཡང་། འཁོར་ལོ་གྲུ་བཞི་མཉམ་པ་ལ་བསྐོར་བ་བྱ། རི་མོ་བཅུ་བདུན། ཆ་བཅུ་དྲུག་དང་ལྡན་པ། གནས་བརྒྱད་ཟླུམ་པོར་བསྐོར་ཏེ། རྡོ་རྗེ་ཕྲེང་
བས་ཀུན་ནས་བསྐོར་བ། ཐམས་ཅད་ཀྱང་སྔོན་དུ་བཤད་པའི་མདོག་དང་ལྡན་པའོ། །དེ་ནས་ཡི་གེ་ཧི་ལས། མིག་གཡས་པ་དང་། གཡོན་པ་གཉིས་ཟླ་བ་དང་ཉི་མའི་དཀྱིལ་འཁོར་དུ་བྱས་ལ། དཀྱིལ་འཁོར་ལ་བལྟས་ལ། ཡེ་ཤེས་ཀྱི་ཐིག་གཤེགས་སུ་གསོལ་ཏེ། དེ་
ནས་བཾ་ལཾ་རཾ་ཧཾ་ཛཾ་གི་ས་བོན་ལས། རིམ་པ་བཞིན་ཡོངས་སུ་རྫོགས་པར་བྱས་ལ། སྐུད་པའི་ཁ་དོག་བཞིན་དུ་དཀར་པོ་ལ་སོགས་པ་རྣམས་ཀྱང་བྱིན་གྱིས་བརླབ་པར་བྱ་སྟེ། རིན་པོ་ཆེ་ལྔ་འམ། ལྕགས་ལྔ་འམ། བ་བླ་ལ་སོགས་པའི་ཁམས་གཞན་ནམ། འབྲས་མཁར་
བ་ལ་སོགས་པའི་འབྲུ་རྣམས་ཀྱིས་ཀྱང་རུང་སྟེ། ཡང་ན་སོ་ཕག་ལ་སོགས་པའི་ས་རྣམས་ཀྱིས་ཀྱང་བྱའོ། །ཨོཾ་ཏྲཾཿཧྲཱིཾཿཨཱཿཧཱུཾ། ས་བོན་འདི་དག་ཡོངས་སུ་རྫོགས་པ་ལས། རང་རང་གི་ཁ་དོག་གི་རིགས་བཞིན་དུ་བྱིན་གྱིས་བརླབ་པར་བྱ་སྟེ། ཨོཾ་བཛྲ་ཙི་ཏྲ་ས་མ་ཡ་ཧཱུཾ། དེ་
བཞིན་དུ་རྡུལ་ཚོན་རྣམས་ལ། ས་བོན་གྱི་འོད་ཟེར་ལས། དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་སྤྱན་དྲངས་ཏེ་བསྟིམས་ལ། ཡི་གེ་ཧྲཱིཿལས་གཟི་བརྗིད་དང་ལྡན་ཞིང་། རྡོ་རྗེ་ཁྲོ་བོ་ཉི་མའི་ལྟ་བས་བལྟས་ལ། ཨོཾ་བརྗོད་དོ། །ཡི་གེ་ཨ་བརྗོད་ཅིང་སྔགས་དང་ཕྱག་རྒྱ་བཅས་ཏེ།
རྡུལ་ཚོན་རྣམས་རབ་ཏུ་འབར་བར་བསམ་མོ། །རྡོ་རྗེ་ཁུ་ཚུར་བཅས་ལ། སྙིང་གའི་ཧཱུཾ་ལས་འོད་ཟེར་དྲི་མ་མེད་པ་འཕྲོས་ཏེ་འབར་བར་བསམ་ཞིང་། སཱུརྱ་ཛྭ་ལ་མཎྜལ་ཧཱུཾ་ཕཊ། དེ་ནས་དཀྱིལ་འཁོར་གྱི་ལྟེ་བ་ལ་བཏབ་པའི་ཕུར་པ་ཡི་གེ་ཧཱུཾ་བརྗོད་ཅིང་། རང་གི་རྡོ་རྗེ་དང་བཅས་
