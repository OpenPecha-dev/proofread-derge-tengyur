པར་བྱའོ། །རང་སྐུ་འཇམ་པའི་རྡོ་རྗེ་ཉིད་དུ་བསྒོམས་ནས། ཛ་ཡིག་ཟླ་བ་ཉི་མའི་སྤྱན། །ཞེས་པ་ནི་མིག་གཡས་གཡོན་ཛ་སེར་པོ་ལས་ཉི་མ་དང་ཟླ་བའི་མིག་ཏུ་བསམས་ལ། བར་མིག་ལྕགས་ཀྱུས་དྲངས་ནས་སུ། །ཞེས་པ་ནི་དེ་ཉིད་ལ་ཛས་མཚན་པ། རྫི་མ་རླེབ་རླེབ་
པོས་ནམ་མཁའི་དེ་བཞིན་གཤེགས་པ་ལ་བལྟ་ཞིང་། ཨོཾ་དཱིཔྟ་དྲྀཥྚི་ཨངྐུ་ཤ་ཛཿཞེས་བརྗོད་ཅིང་ཛ་ལས་འཕྲོས་པའི་འོད་ཟེར་གྱི་ལྕགས་ཀྱུས་རྣམ་པར་སྣང་མཛད་ལ་སོགས་པའི་ཐུགས་ཀ་ན་གནས་པའི་ཡེ་ཤེས་ཀྱི་སྐུད་པ་ལ་ཨོཾ་ཨཱཿཧེ་བྷ་ག་ཝཱན་པྲ་མ་ཐ་ཤཱ་ཤྭ་ཏོ་སཱུ་ཏྲ་པ་ཙཀྲ་སུ་དུ་པ་ཡ་
ཧཱུཾ་ཞེས་གསོལ་བ་བཏབ་པས་དྲངས་ནས། སྐུད་བྱེད་སྐུད་པ་དང་ལྷན་ཅིག །ཅེས་པ་ནི་སྔོན་དེའི་ངོ་བོར་འགྱུར་བའི་སློབ་མས་སྣ་གཅིག་བཟུང་བ་ལ། རང་རང་ཁ་དོག་ནང་དུ་ཐིམ། །ཞེས་པའོ། །ཅུང་ཟད་མ་འབྲེལ་པ་ནི་ཚིགས་སུ་བཅད་པའི་ངོར་བཏབ་པའོ། །ཇི་སྙེད་མི་བསྐྱོད་པ་
ཡི་སྔགས། །ཕན་ཚུན་རྗེས་སུ་འབྲེལ་པ་ཡིས། །ཞེས་པ་ནི་པྲ་ཡཙྪའི་རྗེས་ལ་རཏྣ་སམྦྷ་བ་དང་། ཨ་མི་ཏ་དང་། ཨ་མོ་གྷ་སིདྡྷི་དང་། ཨཀྵོ་བྷྱ་ཞེས་བྱ་བ་བརྗོད་ནས་གསོལ་བ་གདབ་པར་བྱ་སྟེ་གཞུག་ཅེས་པའོ། །དེ་ལྟར་སོ་སོར་གནས་པ་ཀུན། །དྲན་པ་ལ་སོགས་གནས་པར་བཀལ། །
ཞེས་པ་ནི། ཨོཾ་ཨ་ནྱོ་ནྱ་ཨ་ནུ་ག་ཏ་སརྦ་དྷརྨཱ་སྤ་ར་སྤ་ར་ཨ་ནུ་པྲ་བིཥྚ་སརྦྦ་དྷརྨྨཱ་ཧཱུཾ་ཞེས་བརྗོད་ནས་བསྒྲིལ་བར་བྱའོ། །ལས་ཐམས་ཅད་པའི་ཁ་དོག་གཅིག་པར་ཡང་བཤད་དོ། །དེ་ནས་དེ་དག་གསེར་ལ་སོགས་པའི་སྣོད་དུ་དྲི་ཞིམ་པོ་དང་བདུད་རྩི་རྣམ་པ་<ལྔ་ས་,ལྔས་>བྱུགས་ནས། བདུད་རྩི་འཁྱིལ་པའི་སྔགས་
ཀྱིས་བསྲུང་བར་བྱའོ། །ཡིག་གསུམ་བར་བཅུག་རྣལ་འབྱོར་པས། །ཞེས་པ་ནི་ཐིག་སྐུད་ཡི་གེ་གསུམ་གྱི་བར་དུ་ཆུད་པར་མོས་ནས། སྐུད་པ་བཟུང་ནས་བརྡེག་པར་བྱ། །ཞེས་སོ། །ཛ་ཡིག་གསུམ་གྱིས་ཀྱང་བྱས་ནས། །ཞེས་པ་ནི་དེ་བརྗོད་དེ་ཤེས་རབ་མས་ཀྱང་བརྗོད་པར་བྱའོ། །དེ་མེད་
ན་ཀུན་བྱེད་དམ་རྣམ་པར་སྣང་མཛད་དུ་བསྒོམས་པའི་སློབ་མ་ལ་གཏད་ཅིང་དེས་ཀྱང་ཛ་ཛ་ཛ་ཞེས་བརྗོད་པར་བྱའོ། །བློ་གྲོས་ཅན་གྱིས་མཁའ་ཐིག་གདབ། །ཅེས་པ་ནི་དེ་བཞིན་གཤེགས་པ་རྣམས་སེམས་ཅན་གྱི་དོན་བྱེད་པའི་དུས་ལ་བབ་པས་གནས་འདིར་གཤེགས་པའི་ཆེད་དུ་ཐིག་སྐུད་
