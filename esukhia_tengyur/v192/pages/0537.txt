རྒྱུད་ལ་ཡོད་པའི། ད་ལྟ་བའི་སེམས་དང་སེམས་ལས་བྱུང་བས་ཡིན་པས་དེ་གཞན་གྱི་རྒྱུད་ལ་ཡོད་པ་དང་། རང་གི་རྒྱུད་ལ་ཡོད་པ་ཡང་མ་འོངས་པ་ལ་སོགས་པའི་སེམས་ཀྱི་ཆོས་རིག་པར་བྱེད་མི་ནུས་སོ། །རྣལ་འབྱོར་པའི་ཤེས་པ་ཉིད་ནི་བསྒྲུབ་
པར་བྱ་བ་ཡིན་པའི་ཕྱིར་དེས་འཛིན་པ་དང་མི་འཛིན་པའི་བསམ་པ་མི་འཇུག་གོ། །རྗེས་སུ་དཔག་པས་ཀྱང་མ་ཡིན་ཏེ། དེ་ནི་བདེ་བར་གཤེགས་པ་རྣམས་ཀྱིས་རྟགས་རྣམ་པ་གསུམ་ལ་བརྟེན་པར་འདོད་དེ། དེ་ལ་འདིར་བསྒྲུབ་བྱ་སྒྲུབ་པར་བྱེད་པའི་དབང་དུ་བྱས་པའི་
ཕྱིར་རེ་ཞིག་མི་དམིགས་པ་ཉིད་ཀྱི་དབང་དུ་བྱས་པ་མ་ཡིན་ནོ། །འབྲས་བུ་ཡང་མ་ཡིན་ཏེ། རྒྱུ་དང་འབྲས་བུའི་དངོས་པོ་ནི་མངོན་སུམ་དང་མི་དམིགས་པས་བསྒྲུབ་པར་བྱ་བ་ཡིན་པའི་ཕྱིར་རང་བཞིན་གྱིས་ནི་བསྐལ་པའི་ཐམས་ཅད་མཁྱེན་པ་དང་ལྷན་ཅིག་ཏུ་རྒྱུ་དང་འབྲས་བུ་
དངོས་པོ་མི་འགྲུབ་པའི་ཕྱིར་རོ། །རང་བཞིན་གྱི་གཏན་ཚིགས་ཀྱང་ཐམས་ཅད་མཐོང་བ་ཡོད་པར་སྒྲུབ་པར་བྱེད་པ་མ་ཡིན་ཏེ། དེ་མངོན་སུམ་མ་ཡིན་པ་ཉིད་ནི་དེ་དང་ཐ་མི་དད་པའི་རང་བཞིན་གཟུང་བར་མི་ནུས་པའི་ཕྱིར་རོ། །དེ་ལྟ་བས་ན་མཐོང་བ་དང་གྲགས་པའི་
རྟགས་སུ་གྱུར་པ་གང་ཞིག་ཐམས་ཅད་མཁྱེན་པའི་རྗེས་སུ་དཔོག་པར་བྱེད་པའི་ཕྱོགས་གཅིག་གི་ཕྱོགས་ཀྱི་ཆོས་དེ་ནི་ངེས་པར་དཔྱད་ན་ཡོད་པ་མ་ཡིན་ནོ། །གཞན་ཡང་ཐམས་ཅད་མཁྱེན་པ་ཡོད་པར་བསྒྲུབ་པར་བྱ་བ་ལ་གཏན་ཚིགས་ཐམས་ཅད་སྐྱོན་གྱི་རིགས་གསུམ་
ལས་མི་འདའ་སྟེ། མ་གྲུབ་པ་དང་། འགལ་བ་དང་། མ་ངེས་པ་ཉིད་དོ། །འདི་ལྟར་གཏན་ཚིགས་སུ་སྟོན་པ་ནི་དངོས་པོའི་ཆོས་ཞེས་བརྗོད་དམ་དངོས་པོ་མེད་པའི་ཆོས་སམ། འོན་ཏེ་གཉི་གའི་ཆོས་ཞེས་བྱ་བ་རྣམ་པ་གསུམ་མོ། །དེ་ལ་ཆོས་ཅན་ཐམས་ཅད་མཁྱེན་པ་
ལ་དངོས་པོའི་ཆོས་ཕྱིར་རྒོལ་བ་ལ་གྲུབ་པ་སྲིད་པ་མ་ཡིན་ཏེ། ཆོས་ཅན་དེ་ཉིད་བསྒྲུབ་པར་བྱ་བ་ཡིན་པ་ཉིད་ཀྱིས་མ་གྲུབ་པའི་ཕྱིར་རམ་གྲུབ་ན་ཡང་རྩོད་པ་མེད་པར་ཐལ་བར་འགྱུར་བའི་ཕྱིར་རོ། །གང་ཞིག་དེ་ལ་དངོས་པོའི་ཆོས་འདོད་པའི་རྒོལ་བ་དེ་ཇི་ལྟར་དངོས་པོ་ཉིད་དུ་མི་
