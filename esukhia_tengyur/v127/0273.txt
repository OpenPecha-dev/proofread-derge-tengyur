པར་བྱའོ། །ཡང་དག་པར་སྦྱོར་བའི་མཚན་མ་གང་ཞེ་ན། དཀར་པོའི་ཕྱོགས་ཀྱི་གཞིའི་མཚན་མ་ཡིད་ལ་བྱེད་པའི་མཚན་མ་གང་ཡིན་པའོ། །ལོག་པར་སྦྱོར་བའི་མཚན་མ་གང་ཞེ་ན། ནག་པོའི་ཕྱོགས་ཀྱི་གཞིའི་མཚན་མ་ཡིད་ལ་བྱེད་པའི་མཚན་མ་
གང་ཡིན་པའོ། །དེ་བཞིན་དུ་དབང་པོ་རྣམས་ཀྱི་སྒོ་མ་བསྲུངས་པར་གནས་པ་ནས། ཤེས་བཞིན་མེད་པར་གནས་པའི་བར་དུ། སེམས་ནི་འདི་ལྟ་བུར་ཀུན་ནས་ཉོན་མོངས་པ་ཅན་ནོ་སྙམ་དུ་ཡིད་ལ་བྱེད་པའི་མཚན་མ་གང་ཡིན་པའོ། །སྣང་བའི་མཚན་མ་གང་ཞེ་ན། འདི་ལྟར་
འདི་ན་ལ་ལས་མུན་པའི་གཉེན་པོའམ། ཆོས་ཀྱི་སྣང་བ་ལས་ལེགས་པར་ཤིན་ཏུ་མཚན་མར་རབ་ཏུ་བཟུང་ཞིང་། ལེགས་པར་ཡིད་ལ་བྱས་ཏེ། འོག་ཇི་ལྟ་བ་བཞིན་དུ་སྟེང་ཡང་དེ་དང་འདྲའོ་ཞེས་བྱ་བ་ལ་སོགས་པ་མུན་པའི་མཚན་མའི་གཉེན་པོར་འདི་གཞག་གོ། །སོ་སོར་རྟོག་
པའི་མཚན་མ་གང་ཞེ་ན། འདི་ལ་དགེ་སློང་གིས་སོ་སོར་རྟོག་པའི་མཚན་མ་ཉིད་ལེགས་པར་ཤིན་ཏུ་རབ་ཏུ་མཚན་མར་བཟུང་བ་ཡིན་ཏེ། དཔེར་ན། འགྲེང་བས་འདུག་པ་ལ་སོ་སོར་རྟོག་པ་བཞིན་དུ། ད་ལྟར་གྱི་འཛིན་པས་མ་འོངས་པའི་གཟུང་བའི་ཆོས་ལ་སོ་སོར་རྟོག་པ་དང་།
འདུག་པས་ཉལ་བ་ལ་རྟོག་པ་བཞིན་དུ། ད་ལྟར་གྱི་འཛིན་པ་ཉིད་ཀྱིས་འདས་པའི་གཟུང་བའི་ཆོས་ལ་རྟོག་པ་དང་། མདུན་དུ་འགྲོ་བ་ལ་ཕྱི་ནས་འོང་བས་སོ་སོར་རྟོག་པ་བཞིན་དུ་འཛིན་པ་ཕྱི་མ་ཕྱི་མས་འཛིན་པའི་ཆོས་སྔ་མ་སྔ་མ་ལ་སོ་སོར་རྟོག་པ་སྟེ། དེ་ཡང་མདོར་བསྡུ་ན།
གཟུང་བ་དང་། འཛིན་པའི་ཆོས་སོ་སོར་བརྟག་པ་ལས་རྣམ་པ་གཉིས་སུ་བསྟན་ཏོ། །ཏིང་ངེ་འཛིན་བཟང་པོའི་མཚན་མ་གང་ཞེ་ན། འདོད་པའི་འདོད་ཆགས་ཀྱི་གཉེན་པོར་རྣམ་པར་བསྔོས་པ་ལ་སོགས་པ་ཡིད་ལ་བྱེད་པའི་མཚན་མ་གང་ཡིན་པའོ། །ཅིའི་ཕྱིར་དེ་ལ་བཟང་པོ་
ཞེས་བྱ་ཞེ་ན། ཉོན་མོངས་པ་ཐམས་ཅད་ཀྱི་གཙོ་བོ་ནི་འདོད་ཆགས་ཡིན་ཏེ། འདོད་ཆགས་ཐམས་ཅད་ཀྱི་ནང་ན་ཡང་འདོད་པའི་འདོད་ཆགས་ནི་ཐམས་ཅད་ཀྱིས་ལེགས་པར་གཞོམ་བར་དཀའ་བ་ཡིན་ལ། དེ་ནི་དེའི་གཉེན་པོ་ལ་དམིགས་པ་ཡིན་པས་དེའི་ཕྱིར་བཟང་པོ་ཞེས་བྱའོ། །
