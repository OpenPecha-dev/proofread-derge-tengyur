འབྱུང་པོ་དང་། ནད་དང་། ཉམས་པ་དང་། སྡིག་པ་དང་། ངན་སོང་གསུམ་ན་གནས་པ་སྦེད་པའོ། །ཡང་ན་<སྔ་,སྦ་>བ་ནི། སྡུག་བསྔལ་ཐམས་ཅད་ལས་ཐར་ཞིང་གཞན་རྣམས་ཀྱིས་མི་མཐོང་བར་འགྱུར་རོ། །ཡོངས་སུ་བསྐྱབ་པ་དང་ཞེས་བྱ་བ་ནི། སྡུག་བསྔལ་ལས་བསྐྱབ་
པ་སྟེ། ཆད་པ་ཐམས་ཅད་བཅོམ་ཞིང་འཁོར་བའི་སྡུག་བསྔལ་ལས་སྐྱོབ་པའོ། །ཡོངས་སུ་གཟུང་བ་དང་ཞེས་བྱ་བ་ནི་འཇིགས་པ་ཐམས་ཅད་ལས་ཐར་བར་བྱེད་ཅིང་། འཇིགས་པ་ལ་མ་བཏང་སྟེ། མི་དགེ་བ་བཅོམ་ཞིང་། དགེ་བ་གཟུང་བའོ། །ཡོངས་སུ་བསྐྱབ་པ་དང་ཞེས་བྱ་
བ་ནི་སྡིག་པ་ཐམས་ཅད་བཅོམ་ཞིང་འབྱུང་པོ་ལ་སོགས་པ་ཐམས་ཅད་ལས་བསྲུང་བ་སྟེ། མི་དགེ་བ་བཅོམ་ཞིང་དགེ་བ་ལ་བསྐྱང་བའོ། །ཞི་བ་དང་བདེ་ལེགས་སུ་གྱུར་པ་དང་ཞེས་བྱ་བ་ནི། ནད་དང་། འབྱུང་པོ་དང་། མི་དང་མི་མ་ཡིན་པ་ལ་སོགས་པ་ཞི་བར་བྱས་ཤིང་། སྡིག་པ་
ཐམས་ཅད་བཅོམ་ཞིང་འཁོར་བའི་སྡུག་བསྔལ་ལས་ཐར་བའོ། །འདིའི་གཟི་བརྗིད་ཀྱིས་བཀྲ་ཤིས་ཤིང་། ནད་ཐམས་ཅད་བཅོམ་ཞིང་གང་ནས་གང་དུ་ཡང་བདེ་ལེགས་འབྱུང་བའོ། །ཆད་པ་སྤང་བ་དང་ཞེས་བྱ་བ་ནི། ལྷའི་གདོན་ལ་སོགས་པའམ། རྒྱལ་པོའམ། རྐུན་མ་དང་།
ཆོམ་རྐུན་ནམ། ཆུ་དང་མེའམ། དུས་མ་ཡིན་པར་འཆི་བ་བརྒྱད་དམ། མཁའ་འགྲོའམ། མ་མོའམ་གཤིན་རྗེའམ། དེ་ལ་སོགས་པ་ཐམས་ཅད་སྤངས་ཤིང་། བདེ་བར་གནས་པའོ། མཚོན་ཆ་སྤང་བ་དང་ཞེས་བྱ་བ་ནི། ཁྲག་ཁྲུག་ཀུན་གྱི་སྡུག་བསྔལ་
ཐམས་ཅད་སྤངས་པ་སྟེ། དེ་ལ་སོགས་པ་གློ་བུར་གྱི་མཚོན་ཆ་སྤངས་པའོ། །དུག་གཞིལ་བ་དང་ཞེས་བྱ་བ་ནི་མཆེ་བའི་དུག་ལ་སོགས་པ་སྤངས་ཏེ། དེ་རྣམས་ཀྱིས་མི་ཚུགས་ཤིང་དུག་བཅོམ་པའོ། །དུག་གསད་པ་དང་ཞེས་བྱ་བ་ནི། ནང་གི་དུག་གསུམ་ཕྱིའི་དུག་གསུམ་
སྟེ། ནང་གི་དུག་གསུམ་འདོད་ཆགས་ཞེ་སྡང་གཏི་མུག་གོ། །འདུག་པ་དང་། བྱས་པ་དང་། འགྲོ་བའི་དུག་གོ། །དེ་རྣམས་བསད་ཅིང་ཡེ་ཤེས་རྣམས་ཤར་བར་བྱེད་པས་སོ། །མཚམས་གཅད་པ་དང་ཞེས་བྱ་བ་ནི། ཁོ་ར་ཁོར་ཡུག་ཙམ་མཚམས་གཅད་དེ། འདིའི་
