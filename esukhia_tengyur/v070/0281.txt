གཞོན་ནུའོ། །ལྷ་རྣམས་ཀུན་ནི་ཕྱག་ན་རྡོ་རྗེའོ། །དེ་ནས་འཁོར་རྣམས་ངོ་མཚར་དུ་གྱུར་ཏེ། ཞུས་པ་ཁྱོད་ཀྱི་དངོས་གྲུབ་མཆོག་འཚལ་བས། ཇི་ལྟ་བུར་ནི་མཉེས་པར་བགྱི། །ཇི་ལྟ་བུར་ནི་དང་བར་འགྱུར། །བདག་ལ་ཐེ་ཚོམ་ཆེན་པོ་སྐྱེས། །ཞེས་པ་སྨྲས་ཏེ། ཁྱོད་ཅེས་
པ་ནི་འཕགས་པ་སྟེ། གཤིན་རྗེ་གཤེད་དོ། །དེའི་དངོས་གྲུབ་ནི་ཐུན་མོང་དང་ཐུན་མོང་མ་ཡིན་པའི་དངོས་གྲུབ་སྟེ། དེ་ལས་མཆོག་ཏུ་གྱུར་འདོད་པའི་རྣལ་འབྱོར་པས། སྐྱེད་པའི་རིམ་པའི་རྣལ་འབྱོར་གྱིས་ཇི་ལྟར་དངོས་གྲུབ་ལ་ཉེ་བར་བགྱི་བ་དང་། རྫོགས་པའི་རིམ་པའི་རྣལ་འབྱོར་གྱིས་
ལུས་དང་། སེམས་ལས་སུ་རུང་བ་ཤིན་ཏུ་སྦྱངས་པའི་བདག་ཉིད་ཀྱི་དང་བ་ཐམས་ཅད་ཀྱི་ཡེ་ཤེས་ཀྱིས་ཕྱག་རྒྱ་ཆེན་པོའི་དངོས་གྲུབ་སྤྲོས་པའི་རྙོག་པ་མེད་པའི་དང་བའི་བདག་ཉིད་ཀྱི་དངོས་གྲུབ་ལ་ཉེ་བར་ཐབས་ལ་ཇི་ལྟར་བསླབ་པར་བགྱི་ཞེས་ཐབས་དང་། དེའི་རིམ་པ་ལ་ཡིད་གཉིས་སུ་གྱུར་
པས་ཐེ་ཚོམ་གྱི་དྲི་མ་བསལ་དུ་གསོལ་ཞེས་ཞུས་པའོ། །ཁྲོ་བོས་ལྷ་རྣམས་ཚིག་གསན་ནས། །གཤིན་རྗེ་གཤེད་ཀྱིས་བཀའ་སྩལ་པ། །ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། གསན་ཞེས་པ་འཁོར་གྱིས་གསུངས་པ་ཐོས་པའོ། །ཐོས་ཞེས་པ་རིམ་པ་གཉིས་ཀྱི་དོན་དང་། དེའི་བྱ་བ་
ལ་རྨོངས་པར་རིག་པའོ། །སྨྲས་པ་བདག་ཉིད་ཁྱད་པར་གྱུར་པར་བསྟན་པའི་ཕྱིར་བཀའ་སྩལ་པ་སྟེ། བསྐལ་པ་གང་ལ་སེམས་བསྐྱེད་ནས་མངོན་པར་རྫོགས་པར་སངས་རྒྱས་པའི་བསྐལ་པ་གོང་མ་ལ་ཡང་བྱུང་བས་འདིའི་ཐོག་མར་བྱུང་བའོ། །དེས་ནི་ཚངས་པ་ལྟ་བུར་གཙོ་བོར་གསུངས་
པའོ། །ཕྱིས་ཀྱང་འབྱུང་ཞེས་པ་ནི། མྱ་ངན་ལས་མི་འདའ་བ་སྟེ། དུས་དཔག་ཏུ་མེད་པར་སེམས་ཅན་གྱི་དོན་བྱེད་པའོ། །དེས་ན་མཛད་པ་རིང་བར་བསྟན་པའོ། །ད་ལྟར་ཞེས་པ་ནི་བར་ཆད་མེད་པར་ཕྲིན་ལས་ཚད་མེད་པར་མཛད་པའོ། །དེས་ནི་དོན་ལ་བསྙེལ་བ་མེད་པར་གསུངས་སོ། །
བུདྡྷ་ཀུན་གྱི་སྔོན་དུ་བྱུང་། །ཞེས་པ་ནི་ཆོས་ཀྱི་སྐུའི་མཚན་ཉིད་དོ། །དུས་གསུམ་འབྱུང་བའི་སྐུ་ཞེས་པ་ནི་ལོངས་སྤྱོད་རྫོགས་པའི་ངོ་བོ་ལ་བྱ་བར་བཤད་དོ། །གལ་ཏེ་རྒྱུད་གཞན་དག་ཏུ་བཤད་པ་ཉིད་ཀྱིས་ཐེ་ཚོམ་སེལ་བཞིན་ཡང་བཤད་པ་དོན་མེད་པ་མ་ཡིན་ནམ་ཞེ་ན། ལན་དུ། ང་ནི་ལྔ་བརྒྱ་
