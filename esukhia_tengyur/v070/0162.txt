ཏུ་ཡང་སྤྱོད་པ་ན་རིག་སྔགས་དང་རྗེས་སུ་མཐུན་པའི་སྤྱོད་པའོ། །གསང་སྔགས་ཀྱི་གྲངས་དག་ལས་ཀྱང་། །འབུམ་གྲངས་ཇི་སྙེད་བསྟན་གྱུར་པ། །བརྟུལ་ཞུགས་ཡོངས་སུ་རྫོགས་པར་འགྱུར། །ཞེས་པ་ནི། དུས་ནི་ཇི་སྲིད་ལོན་གྱུར་ན། །བརྟུལ་ཞུགས་ཡོངས་སུ་རྫོགས་
པར་འགྱུར། །ཞེས་དྲིས་པའི་ལན་ཏེ། སྔགས་བཟླས་པའི་གྲངས་འབུམ་ལ་སོགས་པ་ཇི་སྙེད་གཞུང་ལས་བསྟན་པ་དེ་སྲིད་ཀྱི་བར་དུ་གོང་དུ་བཤད་པའི་ཚུལ་བཞིན་དུ་བརྟུལ་ཞུགས་དེ་སྤྱད་ན་ཡོངས་སུ་རྫོགས་པར་འགྱུར་ཞེས་པ་སྟེ། དེ་ཡང་དབང་ཆེན་ལ་སོགས་པའི་
དཀྱིལ་འཁོར་ལ་གནས་པའི་བརྟུལ་ཞུགས་ཀྱི་སྔོན་རོལ་དུ་སེམས་ཡོངས་སུ་སྦྱོང་བའི་མཚན་ཉིད་ཅན་ཡན་ལག་བཞི་དང་ལྡན་པའི་བརྟུལ་ཞུགས་དེ་ལ་བྱའོ། །དབང་ཆེན་གྱིས་ནི་དངོས་པོ་ཡིས། །ཞེས་པ་ནས། ཟླ་བ་གཞན་ལ་བཟླས་བརྗོད་བྱ། །ཞེས་པའི་
གནས་ནི་གང་དུ་བགྱི་བ་ལགས། །ཞེས་དྲིས་པའི་ལན་ཏེ། དབང་ཆེན་ལ་སོགས་པའི་དཀྱིལ་འཁོར་ལ་གནས་ནས་རིག་སྔགས་ཀྱི་བརྟུལ་ཞུགས་སྤྱད་པར་བཤད་པའི་ཕྱིར། དབང་ཆེན་གྱི་ནི་དངོས་པོ་ཡིས། །ཞེས་པ་ནས། སྔགས་པས་ཟླ་བ་གཅིག་ཏུ་བཟླས། །ཞེས་
པའི་བར་དུ་གསུངས་ཏེ། དེ་ལ་དབང་ཆེན་གྱི་དཀྱིལ་འཁོར་ལ་གནས་ནས་བདག་ཉིད་རང་གི་ལྷའི་གཟུགས་སུ་བསྒྱུར་ཏེ། ཇི་ལྟར་སའི་རང་བཞིན་སྲ་ཞིང་མཁྲང་བ་ལྟར་བདག་གི་ལུས་ཀྱང་དེ་ལྟ་བུ་སེམས་ཀྱི་ཞར་ལ་བསྒོམ་ཞིང་བཟླས་པ་ཟླ་བ་གཅིག་ཏུ་གནས་པར་བྱ་བ་
བཤད་དེ། །དབང་ཆེན་གྱི་ནི་དངོས་པོ་ཡིས། །དབང་ཆེན་དཀྱིལ་འཁོར་ལ་གནས་ཏེ། །ཞེས་པ་ནི་དབང་ཆེན་གྱི་སྙིང་པོ་ཆོས་ཐམས་ཅད་གཟོད་མ་ནས་མ་སྐྱེས་པར་སྟོན་པའི་ཨ་ཡིག་གི་སྒྲ་དེ་ཁོ་ན་ཉིད་བསྒོམས་ནས་ཡི་གེ་དེ་ཉིད་ལས་དབང་ཆེན་གྱི་དཀྱིལ་འཁོར་དུ་གྱུར་པར་སྔ་མ་
བཞིན་དུ་བསམས་ཏེ། དེའི་སྟེང་དུ་བདག་རང་གིས་ལྷའི་ལུས་སུ་བསྒྱུར་ནས་རྡོ་རྗེའི་ཕྱག་རྒྱས་བཅིངས་ཏེ་དབང་ཆེན་གྱི་རང་བཞིན་ལྟ་བུར་སྲ་ཞིང་བརྟན་པར་བསྒོམ་པར་བྱ་ཞེས་པའོ། །རྡོ་རྗེ་ཅན་གྱི་རྒྱ་བཅིངས་ལ། །ཞེས་པ་ནི་སྲ་ཞིང་བརྟན་པར་རྗེས་སུ་མཐུན་པའི་ཕྱིར་རྡོ་རྗེའི་ཕྱག་
