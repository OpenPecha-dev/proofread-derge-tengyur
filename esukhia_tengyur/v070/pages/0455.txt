སོར་ནི་བཅུ་དྲུག་བྱ། །ཞེས་པ་བོང་ཚད་ལ་བྱ་སྟེ། དབྱུག་ཏོ་ལ་ཡང་ཡིན་ཞེས་པའོ། །ལེགས་པར་བཞོགས་ཤིང་ལེགས་བྲིས་པ། །ཞེས་པ་མགོ་མཇུག་གཉིས་སུ་རི་མོ་དག་བྲི་ཞེས་པའོ། །བཟང་པོ་ཡིད་དུ་འོང་བྱས་ནས། །གསེར་བཟང་པོ་ཁྱིམ་བདག་
གིས་བཀྲི། །ཞེས་པའོ། །དེ་ཡི་རྩེ་ལ་ནོར་བུ་ནི། །ཞེས་པ་གཎྜཱི་དེའི་རྩེ་མོར་ཞེས་པའི་ཆོ་ག་སྔོན་དུ་བཤད་པ་བཞིན་ཞེས་ཞིབ་ཏུ་བཤད་པའི་ཆོ་ག་བཞིན་དུ་བསྒྲུབ་པོ་ཞེས་པའོ། །ནུབ་སྒྲུབ་ཅེས་པ་མཚན་གཅིག་ཏུ་བསྒྲུབ་པོ། །སྔ་མ་བཞིན་དུ་གྲུབ་རྣམས་སུ། །ཞེས་པ་རིག་འཛིན་
གྱི་རྒྱལ་པོར་འགྱུར་རོ་ཞེས་པའོ། །འདོད་གཟུགས་ཅན་ཞེས་པ་འདོད་དགུར་སྒྱུར་བའི་གཟུགས་ཅན་དུ་ཡང་འགྱུར་རོ་ཞེས་པའོ། །དུ་བ་ཚ་ལ་སོགས་པས་ཀྱང་། །ཞེས་པ། དངོས་གྲུབ་རྣམ་གསུམ་ཉིད་ལ་བྱ། །ཞེས་པའོ། །དེ་ཡི་འབྲས་བུ་སྔོན་བཞིན་དུ། །ཞེས་པ་སྔར་བཤད་
པ་བཞིན་དུ་ཐོབ་བོ། །དེ་ལྟར་མདོ་རུ་བཤད་པ་ཡིན། །ཞེས་པ་འདི་ཡན་ཆད་ལྷའི་མཚོན་ཆ་ཞེས་བྱ་བ་བཤད་དོ། །འདི་མན་ཆད་རོ་ལངས་ཀྱི་དངོས་གྲུབ་བཤད་དོ། །དེ་ལས་རྟོག་ལས་འབྱུང་བ་ཡི། །ཞེས་པ་ནས་བསྟན་ཏེ། དངོས་གྲུབ་གཞན་ཡང་བཤད་པར་བྱ། །ཞེས་
པ་རྒྱུད་ཆེན་པོ་ལས་འབྱུང་བའི་དངོས་གྲུབ་གཞན་ཡང་བཤད་པར་བྱའོ་ཞེས་སོ། །དེ་ལྟར་ཁྲོ་བོའི་མཆོག་གིས་ནི། །ཉིད་ཀྱི་མདོ་རུ་བཤད་པའོ། །ཞེས་པ་དེ་ཡང་ཨཱརྱ་མཉྫུ་ཤྲཱི་ཡ་མཱནྟ་ཀ་ཉིད་ཀྱིས་ཇི་ལྟར་མདོར་བཤད་པའི་དངོས་གྲུབ་དེ་དག་བཤད་པར་བྱའོ། །ཡི་དགས་མང་པོ་
ཞེས་བྱ་བའི། །དཀྱིལ་འཁོར་ཐོག་མར་བྲིས་ནས་སུ། །ཞེས་པ། ཡི་དགས་ཕལ་པོ་ཆེའི་དཀྱིལ་འཁོར་དེ། དུར་ཁྲོད་ཐལ་བས་དང་པོར་བྲི་བར་བྱའོ། །རྨ་མེད་པ་ཡི་རོ་བླངས་ནས། །ཞེས་པ་ནད་ཀྱིས་ཤི་བ་དང་རུལ་བ་དང་། མཚོན་གྱིས་བསད་པ་མ་ཡིན་པ་མ་གོད་པ་མ་ཉམས་
པའི་རོ་སར་པ་བླང་ངོ་ཞེས་མཁས་པས་ཤེས་པ་སྔགས་པས་རོ་དེ་ཁྲུས་བྱས་ཞེས་པའོ། །ལེགས་པར་བརྒྱན་ཞེས་པ་མེ་ཏོག་ཕྲེང་བས་བརྒྱན་ཞེས་པའོ། །མི་རོ་དེ་ལ། མེ་ཏོག་སྤོས་ཀྱིས་མཆོད་ནས་སུ། །ཞེས་པའོ། །གཏོར་མ་སྦྱིན་ཞེས་པ་ཤ་དང་ཁྲག་ཏུ་བཅས་པའི་གཏོར་
