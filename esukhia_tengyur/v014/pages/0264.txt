བྷོ་ག་ཀཱ་ཡཱ་ཡ། ནིརྨཱ་ཎ་ཀཱ་ཡཱ་ཡ། ཛྙཱ་ན་བཛྲཱ་ཡ། ཙིཏྟ་བཛྲཱ་ཡ། བཱ་ཀ་བཛྲཱ་ཡ། ཀཱ་ཡ་བཛྲཱ་ཡ། ཙ་ཏུར་བི་མོ་ཀྵེ་བྷྱོ། ཙ་ཏུར་བྲཧྨ་བི་ཧཱ་རེ་བྷྱོ། སཔྟ་ཏྲིཾ་ཤད་བྷྱོ། དྷི་པཀྵི་ཀ་དྷརྨྨེ་བྷྱོ། ཙ་ཏུར་སཱི་ཏི་སཱ་ཧ་པྲ་དྷརྨྨ་སྐནྡྷེ་བྷྱོ། སརྦ་དྷརྨྨཱ་དེ་ཤ་ཀ་བྷྱོ། རཏྣ་ཏྲ་ཡཱ་ཡ། ཤཱཀྱ་མུ་ན་ཡེ། ཀཱ་ལ་ཙཀྲཱ་ཡ། དཱ་ན་
པཱ་ར་མི་ཏཱ་ཡཻ། ཤཱི་ལ་པཱ་ར་མི་ཏཱ་ཡཻ། ཀྵཱནྟི་པཱ་ར་མི་ཏཱ་ཡཻ། བཱིརྱཱ་པཱ་ར་མི་ཏཱ་ཡཻ། དྷྱཱ་ན་པཱ་ར་མི་ཏཱ་ཡཻ། པྲཛྙཱཔཱར་མི་ཏཱ་ཡཻ། ཨུ་པཱ་ཡ་པཱ་ར་མི་ཏཱ་ཡཻ། པྲ་ཎིདྷ་ན་པཱ་ར་མི་ཏཱ་ཡཻ། བ་ལ་པཱ་ར་མི་ཏཱ་ཡཻ། ཛྙཱ་ན་པ་ར་མི་ཏཱ་ཡཻ། ཛ་ཡ་གྷ་ཊེ་བྷྱོ། བི་ཛ་ཡ་གྷ་ཊེ་བྷྱོ། སརྦ་ཙིཏྣོ་བྷྱོ། སརྦ་མུ་དྲཱ་བྷྱོ། ཙིནྟ་མ་ཎི་
ཡཻ། དྷརྨྨ་གྷཎ་ཊྱཻ། དྷརྨྨ་ཤཾ་ཁཱ་ཡ། ཀ་ལྤ་བྲྀཀྵཱ་ཡ། ཨཀྵོ་བྷྱཱ་ཡ། ཨ་མོ་གྷ་སིདྡྷཱ་ཡཻ། རཏྣ་སཾ་བྷ་བཱ་ཡ། ཨ་མི་ཏཱ་བྷཱ་ཡ། བཻ་རོ་ཙ་ནཱ་ཡཻ། ལོ་ཙ་ནྱཻ། མཱ་མ་ཀྱཻ། པཎྜ་རཱ་ཡཻ། ཏཱ་རཱ་ཡཻ། བཛྲ་དྷཱ་ཏྭཱི་ཤྭརྱེ། བཛྲ་པཱ་ཎ་ཡཻ། ཁ་གརྦྷཱ་ཡ། ཀྵི་ཏི་གརྦྷཱ་ཡ། ལོ་ཀེ་ཤྭ་རཱ་ཡ། སརྦ་ནི་བ་ར་ཎི་བྷ་སྐ་མྦྷི་
ནི། ས་མནྟ་བྷ་དྲཱ་ཡ། གནྡྷ་བཛྲཱ་ཡཻ། རཱུ་པ་བཛྲཱ་ཡཻ། རཱ་ས་བཛྲཱ་ཡཻ། སྤཪྴ་བཛྲཱ་ཡཻ། ཤཔྟ་བཛྲཱ་ཡཻ། དྷརྨྨྨཱ་དྷཱ་ཏུ་བཛྲཱ་ཡཻ། ཨུཥྞཱི་ཥཱ་ཡ། བི་གྷ་ནཱནྟ་ཀཱ་ཡ། པྲཛྙཱ་ན་ཏ་ཀཱ་ཡ། པདྨཱ་ན་ཏ་ཀཱ་ཡ། ཡ་མཱནྟ་ཀཱ་ཡ། སྟཾ་བྷི་ཎྱེ། མཱ་ནི་ཎྱེ། སྟཾ་བྷི་ཎྱཻ། ཨ་ཏི་བཱི་དྱཱ་ཡཻ། ཨ་ཏི་ནཱི་ལ་ཡཻ། སརྦ་དྷ་རི་ཎཱི་
བྷྱོ། ཥ་ཊྭཱཾ་གཱ་ཡ། ཞེས་བྱ་བ་ནི་ཐུགས་ཀྱི་དཀྱིལ་འཁོར་ལ་མཆོད་པའི་ཆོ་གའོ། །བཛྲ་ཙརྩི་ཀཱ་ཡཻ། བཛྲ་བཱ་ར་ཧྱཻ། བཛྲ་མ་ཧེ་ཤྭ་རྱཻ། བཛྲ་ཨི་ནྡྱཻ། བཛྲ་བྲཧྨ་ཎྱཻ། བཛྲ་མ་ཧཱ་ལ་ཀྱཻ། བཛྲ་ཀཽ་མཱ་རྱཻ། བཛྲ་བེཥྞ་བྱཻ། ཨཥྚ་ཨཥྚ་ཀེ་ན་ཙ་ཏུཿ། ཥཥྛི་བཛྲ་ཡོ་གི་ནི་བྷྱོ། ཥཌ་ཏྲིཾ་ཤད་ཨཱིཙྪ་བྷྱོ། ཞེས་བྱ་བ་
འདི་ནི་གསུང་གི་དཀྱིལ་འཁོར་ལ་མཆོད་པའི་ཆོ་གའོ། །བཛྲ་བིཥྞ་བཻ། བཛྲ་ནཻ་རེ་ཏྱཱ་ཡ། བཛྲ་ཨགྣ་ཡེ། བཛྲེད་དྷ་ཡེ། བཛྲེ་ནྡྲཱ་ཡ། བཛྲེ་ཤྭ་རཱ་ཡ། བཛྲ་བྲཧྨ་ཎེ། བཛྲ་བི་ནཱ་ཡ་ཀཱ་ཡ། བཛྲ་ཀཱིརྟི་ཀེ་ཡཱ་ཡེ། བཛྲ་བཱ་ཡ་བཻ། བཛྲ་ཡ་མཱ་ཡ། བཛྲ་ཡཀྵེ་བྷྱོ། བ་རུ་ཎྱཻ། བཱ་ཡ་བྱཻ། ཡཱ་མི་ཎྱཻ། ཡཱཀྵི་ཎྱཻ། མ་ཧཱ་
བ་ལཱ་ཡ། ཨ་ཙ་ལཱ་ཡ། ཊཀྐི་རཱ་ཛཱ་ཡ། ནཱི་ལ་དཎྜཱ་ཡ། སུམྦྷ་རཱ་ཛཱ་ཡ། རོ་དཱ་ཀྵི་ཡེ། བཛྲ་ཤྲཱིཾ་ཁེ་ལཱ་ཡཻ། ཙུནྡཱ་ཡཻ། བྷྲྀ་ཀུ་ཊྱཻ། མཱ་རི་ཙྱཻ། པྲ་ཏྱེ་ཀ་མཱ་བྷེ་དེ་ན་ཥཥྚི་ཨུཏྟ་ར་ཏྲིཾ་ཤ་ཏ་བཛྲ་ཏི་ཐི་དེ་བཱི་བྷྱ། ཥཌ་ཏྲིཾ་ཤད་པྲ་ཏཱིཙྪ་བྷྱོ། བཛྲ་ཛ་ཡཱ་ཡ། བཛྲ་ཀ་རྐོ་ཊ་ཀཱ་ཡ། བཛྲ་བཱ་སུ་ཀ་ཡེ། བཛྲ་ཨ་ནནྟཱ་ཡ། བཛྲ་
