རྣམས་ཀྱིས་ཀྱང་བསྐྱང་བར་གྱིས། །རྒྱན་འགྱེད་ཁ་ན་མ་ཐོ་བཅས་པའི་ཟས་དང་ཚིག་ངན་ཀློག་དང་འབྱུང་པོ་ལྷ་མིན་ཆོས་རྣམས་དང་། །བ་ལང་བྱིས་པ་བུད་མེད་སྐྱེས་པ་རྣམས་དང་ལྷ་དང་མི་རྣམས་བླ་མ་ལྔ་ནི་གསད་མི་བྱ། །གྲོགས་དང་རྗེ་བོ་དང་ནི་ལྷ་དང་མི་ཡི་བླ་མ་དགེ་འདུན་དབྱུགས་འབྱིན་རྣམས་ལ་འཁུ་བ་དང་། །དབང་པོ་རྣམས་
ཀྱིས་ཡོངས་སུ་ཆགས་རྣམས་མིན་ཏེ་སྲིད་པའི་བདག་པོའི་བརྟུལ་ཞུགས་ཉི་ཤུ་རྩ་ལྔའོ། །ཞེས་སོ། །དེ་ནས་ཡོལ་བའི་ཕྱི་རོལ་དུ་སློབ་མ་རྣམས་དཀྱིལ་འཁོར་གྱི་གཙོ་བོར་དམིགས་ལ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་ཨ་ནུཏྟར་བོ་དྷི་ཨ་ལཾ་ཀ་ར་བསྟ་པཱུ་ཛ་མེ་གྷ་ས་མུ་དྲ་སྥ་ར་ཎ་ས་མ་ཡེ་ཧཱུྃ། ཞེས་པས་སྨད་གཡོགས་
བཅིང་ངོ་། །ཨོཾ་བཛྲ་རཀྵ་ཧཾ་ཞེས་པས་སྟོད་གཡོགས་ལའོ། །ཨོཾ་བཛྲ་ཨུཥྞཱི་ཥ་ཧཱུྃ་ཕཊ། ཅེས་པས་ཐོད་པ་ལའོ། །ཨོཾ་དྭཱ་ད་ཤ་ཨཾ་གི་ཀ་ནི་རོདྷ་རི་ཎེ་ཧཱུྃ་ཕཊ། ཅེས་པས་མིག་རས་བཅིང་ངོ་། །ཨཱ་ཁཾ་བཱི་ར་ཧཱུཾ། ཞེས་པས་མེ་ཏོག་གི་ཕྲེང་བ་འཛིན་དུ་གཞུག་གོ། །དེ་ནས་སློབ་མས་དཀྱིལ་འཁོར་དུ་འཇུག །
ཡོན་ཕུལ་བ་དང་། རྣམ་པར་རྒྱལ་བའི་དུང་གི་ཆུས་གཏོར་ཏེ། སྙིང་ག་དང་མགྲིན་པ་དང་དཔྲལ་བ་རྣམས་སུ་གདན་གྱི་སྟེང་དུ་ཡི་གེ་གསུམ་གྱི་འོད་ཟེར་གྱིས་ལུས་དྲི་མ་མེད་པར་བྱས་ཏེ། ཧཱུྃ་ཞེས་བརྗོད་ནས་དཀྱིལ་འཁོར་གྱི་ཤར་གྱི་སྒོ་རུ་གནས་པ་ལས་ཀྱི་རྡོ་རྗེས་གཟུང་བ་དང་། དཀྱིལ་འཁོར་དུ་འཇུག་པ་དྲི་
བར་བྱ་བ་ནི། ཀྱེ་ཧོ་བུ་སྡུག་ཁྱོད་སུ་ཡིན་ཞེས་པའི་ལན། བདེ་ཆེན་སྐལ་ལྡན་བདག་ལགས་སོ་ཞེས་པ་བརྗོད་ན། དེ་ཉིད་བརྗོད་དུ་བཞུག་གོ། །ཨོཾ་སརྦ་ཡོ་གཙིཏྟ་ཨུཏ་པཱ་ད་ཡཱ་མི། ཞེས་པས་བྱང་ཆུབ་ཀྱི་སེམས་བསྐྱེད་ལ་མེ་ཏོག་དང་བཅས་པའི་རྡོ་རྗེ་སྙིང་གར་བཞག་ལ། ཨོཾ་སུ་ར་ཏེ་ས་མ་ཡ་སྟྭཾ་ཧོ་སིདྡྷི་བཛྲ་ཡ་ཐཱ་སུ་
ཁཾ། ཞེས་པ་བརྗོད་ཅིང་རྡོ་རྗེ་ལ་འཛིན་དུ་གཞུག་གོ། །དེ་ནས་བརྗོད་པ། དེང་ཁྱོད་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱིས་བྱིན་གྱིས་རློབ་པས་ཁྱོད་ཀྱིས་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་མཆོག་ཏུ་གསང་བའི་དཀྱིལ་འཁོར་ཆེན་པོ་འདི་དཀྱིལ་འཁོར་མི་མཐོང་བ་རྣམས་ལ་སྨྲ་བར་མ་བྱེད་ཅིག་ཅེས་བརྗོད་ལ།
ཨོཾ་བིགྷ་ནཱནྟ་ཀྲྀཏ་ཧཱུྃ། ཞེས་པས་དགུག་པར་བྱའོ། །དེ་ནས་ཡང་། ཨཱ་ཁཾ་བཱི་ར་ཧཱུྃ། ཞེས་པས་ཡོལ་བའི་ནང་དུ་བཅུག་སྟེ། ཨོཾ་མ་ཧཱ་ར་ཏ་སུ་ཏོ་ཥྱ་སུ་སུ་ཁོ་བཛྲ་ས་ཏྭ་ཨཱརྱ་སིདྡྷྱ་མཾ། ཞེས་བརྗོད་ཅིང་བསྐོར་བ་བྱའོ། །དེ་ནས་ཐལ་མོ་སྦྱར་ལ་ཤར་སྒོ་ནས་བརྗོད་པ། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པཱུ་ཛཱ་ཨུ་པསྠཱ་ན་ཡ་ཨཱཏྨ་ནཾ་
