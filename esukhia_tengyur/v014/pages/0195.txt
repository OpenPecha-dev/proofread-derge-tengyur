སྔགས་རྣམས་བརྗོད་པར་བྱ་སྟེ། ཨོཾ་ཀྵཿདུར་དཱན་ཏཱ། བྷི་མན་ཀྵ་ཡཾ་ཀ་ར་བྱཱ་གྷྲ་ཙརྨ་ཀ་ཊཽ་ཧུྃ་ཧུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧཾ་ཀྵཿཏྲིཾ་ཤང་སྭ་ར་སཔྟ་ཏི་ཧྲཿསྭ་དིརྒྷ་བྱཉྫ་ན་སྭ་བྷཱ་བ་དྱཻ་ཏྱ་ཀུ་ལ་ཤ་ཏ་སི་རོ་བྷིར་མུཎྜ་མཱ་ལ་སྐནྡྷ་ཨ་བ་ལཾ་བི་ནི་སྐནྡྷེ་ཧུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཧ་ཧི་ཧྲྀ་ཧུ་ཧླྀ་པཉྩ་སྐནྡྷ་བི་ཤུདྡྷ་སྭ་བྷཱ་བ་
ཤི་ར་སི་ཀ་པཱ་ལ་མཱ་ལེ་ཧུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཅེས་པ་ནི་ཕྱག་རྒྱ་བཞིའི་ཆོ་གའོ། །དེ་ལྟར་ཕྱག་རྒྱ་བཅུ་རྣམས་ཕ་རོལ་ཏུ་ཕྱིན་པ་བཅུའི་དབྱེ་བ་དང་། གནས་བཅུ་དྲུག་གི་དབྱེ་བས་སྟོང་པ་ཉིད་བཅུ་དྲུག་ཅེས་པ་ནི་ཕྱག་རྒྱ་དགོད་པའི་ཆོ་གའི་རིམ་པའོ། །དེ་ནས་གཡོན་དང་གཡས་ཀྱི་ཕྱག་མཚན་གྱི་ས་བོན་རྣམས་
བརྗོད་པར་བྱ་སྟེ། ཨོཾ་ཀྵྼཿཀྲོ་དྷ་བཛྲ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཁཌྒ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཏྲི་ཤཱུ་ལ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཀརྟ་རི་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་བཱ་ཎ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཨཾ་ཀུ་ཤ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཌ་མ་རུ་ཀ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་མུདྒ་ར་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵརཿབཛྲ་ཙཀྲ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་ཀུནྟ་ཧཱུྃ་
ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་དཎྜ་ཧཱུྃ་ཕཊ། ཨོཾ་ཀྵྼཿབཛྲ་པཪྴཱུ་ཧཱུྃ་ཕཊ། ཅེས་པ་རྣམས་ནི་གཡས་ཀྱི་ཕྱག་མཚན་རྣམས་སོ། །དེ་ནས་གཡོན་གྱི་ཕྱག་མཚན་རྣམས་ནི། དང་པོར་ཧྭཱཾ་ཡིག་བརྗོད་པར་བྱ། ཨོཾ་ཧྭཱཾ་བཛྲ་གྷཎྚེ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཁེ་ཊཀ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཁ་ཊྭཱཾ་ག་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཀ་བཱ་ལ་
ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཙ་པཱ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་པ་ཤ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་རཏྣ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་པདྨ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཤཾ་ཁ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཨཱ་དརྴ་ཧོ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲཱ་ཤྲིཾ་ཁ་ལེ་ཧྭཱཾ་ཕྲེཾ་ཕཊ། ཨོཾ་ཧྭཱཾ་བཛྲ་ཙ་ཏུར་བེ་ཏ་སྭ་བྷཱ་བ་བྲཧྨ་ཤི་ར་ཧོ་ཕྲེཾ་ཕཊ། ཅེས་པ་རྣམས་ཕྱག་
གཡོན་པ་རྣམས་སུ་མཚན་མ་དགོད་པའི་རིམ་པའོ། །དེ་ནས་གླང་པོའི་པགས་པའི་སྔགས་རྣམས་བརྗོད་པ་ནི། ཨོཾ་ཀྵ་ཀྵཱ་ཀྵི་ཀྵཱི་ཀྵྼྀ་ཀྵྼཱྀ་ཀྵུ་ཀྵུ་ཀྵླྀ་ཀྵླཱྀ་ཀྵཾ་ཀྵཾ། ཧ་ཧཱ་ཧི་ཧཱི་ཧྼྀ་ཧྼཱྀ་ཧུ་ཧཱུ་ཧླྀ་ཧླཱྀ་ཧཾ་ཧཱཿ། སརྦ་བགྷྣི་བི་ད་རི་ཏ་ག་ཛ་ཙརྨ་པཊྚཱ་ཨརྡྷ་རཀྟ་ཤྲ་བ་ཏ་ཧཱུྃ་ཕྲེཾ། ཕྱག་གཡས་དང་གཡོན་གྱི་སྡིགས་མཛུབ་དང་བཅས་
ནས་བཟུང་བ་ནི། གཡས་ཀྱི་ལག་པ་དང་མགོ་བོའི་ཕྱོགས་ཀྱི་ལག་པ་དང་། གཡོན་གྱི་རྐང་པ་ནས་བཟུང་བར་བསམ་པ་ནི་པགས་པ་བཟུང་བའི་རིམ་པའི་ཆོ་གའོ། །དེ་ནས་ཀླུའི་རྒྱལ་པོ་བཅུ་དགོད་པའི་སྔགས་ནི། ཨོཾ་ཧཱུྃ་བཛྲ་ཛ་ཡ་ནཱ་གེནྡྲ་བཛྲ་ཛ་ཊཱ་མུ་ཀུ་ཊ་བནྡྷེ་ཏིཥྛ་ཏིཥྛ་ཧུྃ་ཧཱུྃ་ཕཊ་ཕཊ། ཨོཾ་ཀྵུཾ་བཛྲ་
