སརྦ་པཱ་ར་མི་ཏཱ་མ་མ་བཛྲ་པ་ཊཾ་བནྡྷ་ཡནྟུ་ཧུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧཱུྃ་ཧོཿ། པི་ཛྙཱ་ན་ཛྙཱ་ན་སྭ་བྷཱ་བཻ་ཀཱ་རུ་ཎཱ་པྲ་ཛྙཱ་ཨཱཏྨ་ཀེ་བཛྲ་བཛྲ་གྷཎྚེ་ས་བྱ་ཨུཏྟ་ར་ཀ་ར་ཡོ་རྨ་མ་བཛྲ་ས་ཏྭཱཿས་པྲཛྙོ་ད་དཱ་ཏུ་ཧུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨ་ཨཱ་ཨེ་ཨཱི་ཨར་ཨཱར་ཨོ་ཨཽ་ཨལ་ཨཱལ་ཨཾ་ཨཱཾ། སརྦ་བོ་དྷི་ས་ཏྭཿས་བྷཱརྱཿསརྦ་དཱ་སརྦ་ཀཱ་མ་ཨུ་པ་བྷོ་གཾ་བཛྲ་པྲ་ཏཾ་མ་མཱ་
དཱ་དཱནྟུ་སྭཱ་ཧཱ། ཨོཾ་ཧ་ཧཱ་ཡ་ཡཱ་ར་རཱ་ཝ་ཝཱ་ལ་ལཱ་སརྦ་ཀྲོ་དྷ་རཱ་ཛཱཿ། ས་བྷཱརྱཿ། མཻ་ཏྲི་ཀཱ་རུ་ཎཱ་མུ་དི་ཏ་ཨུ་པེཀྵཱ་སརྦ་ས་མ་ཏྭ་སྭ་བྷཱ་ཝཾ་བཛྲ་པཱུརྦཾ་ག་མ་ནཱ་མ་མེ་དཱ་དཱ་ནྟུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨེ་པདྨ་བཛྲ་ཙི་ཧྣཽ་པྲཛྙཽ་བ་ཡཽ་མཎྜལ། ཨ་དྷི་པ་ཏཱི་བཛྲ་སུ་ཁ་ཛྙཱ་ནཱཾ་ག་མ་མ་དད་ཏཱཾ་ཧཾ་ཧཿཧུྃ་ཧཱུྃ་ཕཊ། ཅེས་པས་སོ། །དེ་ནས་དབང་བདུན་
རྣམས་ཀྱིས་དབང་བསྐུར་བར་བདག་ཉིད་བསམ་པར་བྱའོ། །དེ་ནས་དབང་བདུན་ཐོབ་པར་ང་རྒྱལ་བསྐྱེད་པ་ནི། ཨོཾ་སརྦ་ཏཱ་ཐཱ་ག་ཏ་སཔྟ་ཨ་བྷི་ཥེ་ཀ་སཔྟ་བྷཱུ་མི་པྲཱ་པྟོ་ཧཾཿ། ཞེས་པ་སྟེ། དབང་བདུན་རྗེས་སུ་གནང་བའི་ཆོ་གའོ། །དེ་ནས་ཡང་སྔགས་པས་བུམ་པ་ལ་སོགས་པའི་དབང་གོང་མ་དོན་དུ་གཉེར་
བར་བྱ་བའི་ཚིག་འདི་ལྟ་སྟེ། ཨོཾ་ཨོཾ་པྲཛྙཱ་ཨུ་པ་ཡཽ་ཀ་ལ་ཤ་ཨ་བྷི་ཥེ་ཀཾ་མེ་ད་དཱ་ཏཱཾ་ཧུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཨཱཿཔྲཛྙཱ་ཨུ་པཱ་ཡོ་གུ་ཧྱ་ཨ་བྷི་ཥེ་ཀཾ་མེ་ད་དཱ་ཏཱཾ་ཧུྃ་ཧཱུྃ་ཕཊ། ཨོཾ་ཧུྃ་པྲཛྙཱ་ཨུ་པཱ་ཡཽ་བཛྲ་ས་ཏྭ་བི་ཤྭ་མཱ་ཏ་རོ་པྲཛྙཱ་ཛྙཱ་ན་ཨ་བྷི་ཥེ་ཀཾ་མེ་ད་དཱ་ཏཱཾ་ཧུྃ་ཧཱུྃ་ཕཊ། ཅེས་པས་གསོལ་བ་གདབ་བོ། །དེ་ནས་འཆད་པར་འགྱུར་པའི་ཆོ་གས་
བདག་ཉིད་ལ་དབང་བསྐུར་བར་བསམ་མོ། །དེ་ནས་དེ་ལྟར་བསམས་ནས། ཨོཾ་ཨོཾ་བཛྲ་ས་ཏྭེ་ན་བཛྲ་ཀ་ལ་ཤ་ཨ་བྷི་ཥེ་ཀེ་ན་ཨ་བྷི་ཥི་ཀྟོ་བཛྲ་ས་ཏྭ་ཙལཱ་བྷཱུ་མི་ལབ་དྷོ་ཧཾ། ཨོཾ་ཨཱཿབཛྲ་ས་ཏྭེ་ན་གུ་ཧྱ་ཨ་བྷི་ཥེ་ཀེ་ན་ཨ་བྷི་ཥིཀྟོ་བཛྲ་ས་ཏྭ་ཡུ་ཝ་རཱ་ཛྲོ་ན་ཝ་བྷཱུ་མི་ལབ་དྷོ་ཧཾ། ཨོཾ་ཧཱུྃ་བཛྲ་ས་ཏྭེ་ན་པྲ་ཛྙཱ་ཛྙཱ་ནི་ན་ཨ་བྷི་ཥེ་ཀེ་ན་ཨ་བྷི་ཥིཀྟོ་
དྭི་ཏཱི་ཡ་བཛྲ་དྷ་རོ་ཧཾ་དྷརྨྨཱ་མེ་གྷ་བྷཱུ་མི་ལབ་དྷོ་ཧཾ། ཞེས་པས་དབང་གོང་མའི་ང་རྒྱལ་བྱ་བའི་ཆོ་གའོ། །དེ་ནས་དབང་གོང་མ་བས་ཀྱང་གོང་མའི་དབང་བཞི་པའི་མན་ངག་དོན་དུ་གཉེར་བའི་སྔགས་ཀྱི་ཚིག་འདི་ལྟ་སྟེ། ཨོཾ་ཧོ་པྲཛྙཱ་ཨུ་པཱ་ཡ་ཨཱཏྨ་ཀཾ། བཛྲ་ས་ཏྭ་མ་ཧཱ་མུ་དྲཱ་ཛྙཱ་ན་ཨ་བྷི་ཥེ་ཀཾ་མེ་པྲ་ཡཙྪ་ཨོཾ་ཨཱཿཧཱུྃ་ཧོ་ཕཊ།
ཅེས་པས་གསོལ་བ་བཏབ་ནས་བདག་ཉིད་དྲི་མ་དང་བྲལ་བའི་དབང་བསྐུར་བར་བསམས་ལ། དེ་ནས་རྡོ་རྗེ་སེམས་དཔའི་ང་རྒྱལ་བྱ་བ་ནི། ཨོཾ་ཧོ་དྷརྨྨཱ་དྷཱ་ཨཀྵ་ར་ཙ་ཏུ་རྠ་ཨ་བྷི་ཥེ་ཀེ་ན། ཨ་བྷི་ཥིཀྟོ་བཛྲ་ས་ཏྭེ་ན་ཨཱཏྨ་ནཱ་ཧཱཾ། གཉིས་པ་ས་བཅུ་གཉིས་པ་ཐོབ་པ་རྡོ་རྗེ་སེམས་དཔའ་དོན་ཆེན་པོ་མཆོག་ཏུ་མི་འགྱུར་བ་འཇིག་
