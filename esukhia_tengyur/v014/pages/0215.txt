དྷི་དྷྼྀ་དྷུ་དྷླྀ་དྷཾ། ད་དི་དྼྀ་དུ་དླྀ་དཾ། ཐ་ཐི་ཐྼྀ་ཐུ་ཐླྀ་ཐཾ། ཏ་ཏི་ཏྼྀ་ཏུ་ཏླྀ་ཏཾ། རྣམས་དང་རྡོ་རྗེ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དེ་ལྟར་སྟོན་ཟླ་འབྲིང་པོའི་ཚེས་རྣམས་དང་ནི་གཡོན་ན་རྣམ་པར་དག་པའི་རང་བཞིན་ཅན། །འདི་རྣམས་དང་བཅས་
ནུབ་སྒོའི་གཡས་སུ་སྣ་ཐག་ཅན་དང་ཆུ་སྐྱེས་དམར་པོའི་སྟེང་གནས་པ། །ཏཾ་ཡིག་དང་ནི་རྡོ་རྗེ་དག་ལས་བྱུང་བའི་དབང་པོ་སེར་པོ་ཕྱག་ན་རྡོ་རྗེ་དང་། །མདའ་དང་དྲིལ་བུ་གཞུ་རྣམས་འཛིན་པ་ཙ་རྩི་ཀཱ་དང་བཅས་པ་དེ་ལ་ཕྱག་འཚལ་ལོ། །ྈྐ་ྈྐི་ྈྐྲྀ་ྈྐུ་ྈྐླྀ་ྈྐཾ། ཤི་ཤི་ཤྼྀ་ཤུ་ཤླྀ་ཤཾ། ཥ་ཥི་ཥྼྀ་ཥུ་ཥླྀ་ཥཾ། ྉྤ་
ྉྤི་ྉྤྼྀ་ྉྤུ་ྉྤླྀ་ྉྤཾ། ས་སི་སྼྀ་སུ་སླྀ་སཾ། རྣམས་དང་རྩེ་གསུམ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དགུན་ཟླ་ར་བའི་ཚེས་རྣམས་དང་ནི་ལོང་བུའི་ཚིགས་ཀྱི་རང་བཞིན་ཅན་དེ་འདི་རྣམས་དང་། །བཅས་པས་བྱང་སྒོའི་གཡོན་དུ་ཁྱུ་མཆོག་དང་ནི་ཆུ་སྐྱེས་དམར་པོའི་སྟེང་
དུ་ཡང་དག་གནས། །སཾ་ཡིག་དང་ནི་རྩེ་གསུམ་ལས་བྱུང་འཕྲོག་བྱེད་སྔོན་པོ་རྩེ་གསུམ་མདའ་དང་ཁ་ཊྭཱཾ་ག །གཞུ་རྣམས་ཕྱག་གིས་འཛིན་དང་གཤིན་རྗེ་དང་མཚུངས་རི་ཡི་བུ་མོ་པང་གནས་ལ་ཕྱག་འཚལ། །སཱ་སཱི་སྼཱྀ་སཱུ་སླཱྀ་སཱཿ། ྉྤཱ་ྉྤཱི་ྉྤྼཱྀ་ྉྤཱུ་ྉྤླཱྀ་ྉྤཱཿ། ཥཱ་ཥཱི་ཥྼཱྀ་ཥཱུ་ཥླཱྀ་ཥཱཿ། ཤཱ་ཤཱི་ཤྼཱྀ་
ཤཱུ་ཤླཱྀ་ཤཱཿ། ྈྐཱ་ྈྐཱི་ྈྐྼཱྀ་ྈྐཱུ་ྈྐླཱྀ་ྈྐཱཿ། རྣམས་དང་དབྱུག་པ་ལས་བྱུང་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དེ་ལྟར་དགུན་ཟླ་འབྲིང་པོའི་ཚེས་རྣམས་དང་ནི་རྐང་གཡས་ལོང་མོའི་ཚིགས་ཀྱི་རང་བཞིན་ཏེ། །དེ་རྣམས་དང་བཅས་ནུབ་སྒོའི་གཡོན་
དུ་མིང་དང་ཉི་མའི་མདོག་ལྡན་སྟེང་གནས་པ། །ྈྐཱཿཡིག་དཔལ་གྱི་དབྱུག་པ་དག་ལས་བྱུང་བའི་གནོད་སྦྱིན་ནོར་བུ་དབྱུག་པ་སྲེ་མོ་དང་། །པདྨ་རྣམས་ནི་ཕྱག་ན་འཛིན་པའི་ཁྱབ་འཇུག་དང་མཚུངས་ནོར་དབང་པང་གནས་ལ་ཕྱག་འཚལ། །ཏཱ་ཏཱི་ཏྼཱྀ་ཏཱུ་ཏླཱྀ་ཏཱཿ། ཐཱ་ཐཱི་ཐྼཱྀ་ཐཱུ་ཐླཱྀ་
ཐཱཿ། དཱ་དཱི་དྼཱྀ་དཱུ་དླཱྀ་དཱཿ། དྷཱ་དྷཱི་དྷྼཱྀ་བྷཱུ་ཧླཱྀ་དྷཱཿ། ནཱ་ནཱི་ནྼཱྀ་ནཱུ་ནླཱྀ་ནཱཿ། རྣམས་དང་དུང་ལས་བྱུང་བ་པང་གནས་རྣམས་ནི་མངའ་བདག་དང་མཚུངས་རོལ་པས་གནས། །དེ་ལྟར་སྟོན་ཟླ་ཐ་ཆུངས་ཚེས་རྣམས་དང་ནི་པུས་མོའི་ཚིགས་ཀྱི་རང་བཞིན་ཅན་ཉིད་དེ། །ངང་པ་ཆུ་སྐྱེས་དཀར་པོ་
