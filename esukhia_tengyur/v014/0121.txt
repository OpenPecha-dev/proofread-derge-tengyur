སྡུད་དང་བསམ་གཏན་དང་། །སྲོག་རྩོལ་དེ་བཞིན་འཛིན་པ་དང་། །རྗེས་སུ་དྲན་དང་ཏིང་འཛིན་རྣམས། །ཞེས་བྱ་བར་ཞུགས་པའོ། །དེའི་སྟོབས་ཀྱིས་དེ་ནས་མི་འཛག་པའི་ལྷན་ཅིག་སྐྱེས་པ་མི་འགྱུར་ཞེས་བྱ་བ་སྟོང་པ་ཉིད་དང་། སྙིང་རྗེ་དབྱེར་མི་ཕྱེད་པའི་བདེ་བ་འཛག་པ་མི་མངའ་བའི་
ལྷན་ཅིག་སྐྱེས་པ་འབྱུང་ལ། སྟོང་ཉིད་སྙིང་རྗེ་ལ་མི་གནས་པ་གང་ཡིན་པ་དེ་ཁོ་ན་བཞི་པ་ཞེས་ཀྱང་བྱའོ། །གཞན་ཡང་བྱིས་པ་དང་དར་མ་དང་རྒན་པོར་གྱུར་པ་ནི། འགྱུར་བ་དང་བར་ཆད་ཅན་གྱི་བདེ་བ་དཔེར་བྱ་བ་ཡིན་ལ། བཞི་པ་གང་ཡིན་པ་དེ་ནི། འཇིག་རྟེན་གསུམ་གྱི་དཔེར་བྱ་བ་ལས་འདས་པ་ཡིན་ནོ། །
དེ་ཉིད་ཁམས་གསུམ་གྱི་སྤྲོས་པ་ལས་གྲོལ་བ་ཡིན་ནོ། །གང་ལ་སངས་རྒྱས་ཞལ་འདི་ཁ་དང་སྙིང་ག་ལ་གནས་ནས། །ཞེས་བྱ་བ་ནི། སངས་རྒྱས་ཀྱི་གསུང་སྟེ། གསུང་གི་ཡེ་ཤེས་འདི་ཞེས་བྱ་བའོ། །ཐུགས་ལ་གནས་ཞེས་བྱ་བ་ནི་བསྒོམས་ཤིང་ཉམས་སུ་མྱོང་བའོ། །གསུང་ལ་གནས་ཞེས་བྱ་བ་ནི།
སློབ་དཔོན་དེས་སློབ་མ་རྣམས་ལ་ལེགས་པར་སྟོན་པར་བྱེད་པའོ། །དེ་ཡང་དུས་ཐམས་ཅད་ཉམས་སུ་མྱོང་ཞིང་སྟོན་པར་འགྱུར་བ་རྣམས་གང་ཡིན་དེ་ལ། དཔལ་ལྡན་བླ་མ་དེ་ཞེས་བྱ་སྟེ། རྡོ་རྗེ་སེམས་དཔའ་དང་མཚུངས་ཞེས་པའི་དོན་ཏོ། །ད་ནི་རང་གི་སེམས་སྣང་བ་ལ་འཇུག་པ་བརྗོད་པར་བྱ་བའི་ཕྱིར། ནམ་
མཁའ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། དེ་ལ་སྤྱིར་སྦྱོར་བ་གོམས་པ་ནི་རྣམ་པ་གཉིས་ཏེ། ཕ་རོལ་ཏུ་ཕྱིན་པའི་ཐེག་པ་དང་། སྔགས་ཀྱི་ཐེག་པའོ། །དེ་ལ་ཕ་རོལ་ཏུ་ཕྱིན་པའི་ཐེག་པའི་སྦྱོར་བ་གོམས་པ་ནི་བླ་གབ་མེད་པའོ། །སྔགས་ཀྱི་ཐེག་པ་ནི་གང་ནམ་མཁའི་རྣལ་འབྱོར་དུ་བྱེད་པ་སྟེ། །དེ་ཉིད་ནམ་
མཁའ་ཀུན་ཞེས་བྱ་བ་ལ་སོགས་པས་བསྟན་ཅིང་། ནམ་མཁར་དུ་བ་ལ་སོགས་པའི་རྟགས་ལ་བལྟའོ། །<མིས་,མིག་>མི་འཛུམས་པར་རྡོ་རྗེ་ལམ་ལ་རབ་ཏུ་ཞུགས། །ཞེས་བྱ་བ་ནི། མིག་མི་འཛུམས་པའི་ལྷའི་སྤྱན་གྱི་རྟགས་ཡིན་ལ། རྡོ་རྗེ་ལམ་ནི་ཨ་ཝ་དྷཱུ་ཏཱིར་སྲོག་འཇུག་པའོ། །དེ་བས་ནམ་མཁའ་
ཀུན་ཞེས་སེམས་ཀྱི་རྡོ་རྗེ་ལམ་ལ་རབ་ཏུ་ཞུགས། །ཞེས་བྱའོ། །སྟོང་ལ་དུ་བ་སྨིག་རྒྱུ་གསལ་བ་དྲི་མ་མེད་པའི་མེ་ཁྱེར་མར་མེ་ཉིད། །ཅེས་བྱ་བ་ནི། རྣལ་འབྱོར་པས་མཐོང་བར་འགྱུར་བའོ། །དེ་ནས་སྤྲིན་དང་བྲལ་བའི་ནམ་མཁའ་ལྟར་མཐོང་ངོ་། །ནམ་མཁའ་དེ་ལས། ནམ་མཁའ་ལས་བྱུང་རང་བྱུང་བ། །
