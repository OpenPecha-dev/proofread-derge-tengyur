རིམ་པ་བཞིན་དུ། བཅོ་ལྔ་དང་། བཅུ་དང་། ལྔ་དང་། གསུམ་དང་། གཉིས་བྲི་བར་བྱའོ། །དེ་བཞིན་དུ་འཕར་མ་གསུམ་པའི་དང་པོ་ལ་སོགས་པ་ལ་ནི་ལོ་ཟླ་ལ་སོགས་པའི་ཞག་གི་མཚན་མ་སྟོང་བརྒྱད་ཅུ་ལ་སོགས་པའི་རིམ་པ་བཞིན་དུ་བྲིས་ལ་ཚེ་དང་སྲོག་གི་རླུང་རྟོགས་པར་བྱའོ། །ད་ནི་གོང་དུ་བརྗོད་པའི་ལྔ་
ནས་ཉི་ཤུ་ལྔའི་བར་གྱི་བཞི་བཞི་བཅུ་དྲུག་དོར་བ་དེ་རྣམས་བཤད་པར་བྱའོ། །དེ་ཡང་ཇི་སྐད་དུ། ཞག་དྲུག་རྒྱུ་ན་ལོ་གསུམ་ནས་ཉི་ཤུ་རྩ་བཞི་དོར་བས་འཆི། ཞག་བདུན་རྒྱུ་བས་ལོ་གསུམ་ལས། །བཞི་བཅུ་རྩ་བརྒྱད་འདོར་བས་སོ། །ཞག་བརྒྱད་རྒྱུ་ན་ལོ་གསུམ་ལས། །ཟླ་གཉིས་ཞག་བཅུ་གཉིས་ཀྱིས་སོ། །
ཞག་དགུ་རྒྱུ་ན་ལོ་གསུམ་མོ། །ཉིན་མཚན་དགུ་བཅུ་རྩ་དྲུག་འདོར། །གལ་ཏེ་ཞག་ནི་བཅུ་གཅིག་རྒྱུ། །ལོ་གཉིས་ཉིན་ཞག་ཉི་ཤུ་བཞིས། །མ་ཚང་བ་ན་འཆི་བར་བཤད། །ཞག་ནི་བཅུ་གཉིས་ལོ་གཉིས་ལས། །བཞི་བཅུ་རྩ་བརྒྱད་ཕྲི་བས་སོ། །ཞག་ནི་བཅུ་གསུམ་ལོ་གཉིས་ལས། །ཟླ་གཉིས་ཞག་
ནི་བཅུ་གཉིས་དབྲི། །ཞག་ནི་བཅུ་བཞི་ལོ་གཉིས་ལས། །ཟླ་གསུམ་ཞག་དྲུག་དབྲི་བར་བྱ། །ཉིན་ཞག་བཅུ་དྲུག་རྒྱུ་བ་ནི། །ལོ་གཅིག་ཞག་བཅུ་གཉིས་ན་འཆི། །ཉིན་ཞག་བཅུ་བདུན་རྒྱུ་བ་ནི། །ལོ་གཅིག་ཞག་ནི་ཉི་ཤུ་བཞིས། །མ་ཚང་བ་ན་འཆི་བར་བཤད། །ཞག་བཅོ་བརྒྱད་ཀྱིས་ལོ་གཅིག་ལས། །
ཟླ་གཅིག་ཞག་དྲུག་འདོར་བས་སོ། །བཅུ་དགུ་རྒྱུ་ན་ལོ་གཅིག་ལས། །བཞི་བཅུ་རྩ་བརྒྱད་དོར་བས་སོ། །ཉི་ཤུ་གཅིག་ནི་ཟླ་དྲུག་ལས། །ཞག་དྲུག་མ་ཚང་བ་ཡིས་སོ། །ཉི་ཤུ་རྩ་གཉིས་ཟླ་གཉིས་ལས། །ཞག་བཅུ་གཉིས་ནི་འདོར་བས་སོ། །ཉིན་ཞག་ཉི་ཤུ་རྩ་གསུམ་རྒྱུ། །བཅོ་བརྒྱད་རྣམ་པར་འདོར་
བས་སོ། །ཞག་ནི་ཉི་ཤུ་རྩ་བཞི་རྒྱུ། །ཉིན་ཞག་ཉི་ཤུ་རྩ་བཞིས་སོ། །ཉིན་ཞག་སུམ་ཅུ་གསུམ་དེ་བཞིན། །འཆི་བའི་མཚན་མ་འབྱུང་བ་ཡིན། །རླུང་འབྱུང་སྟོབས་ལས་ཤེས་བྱ་བ། །དེ་ཡན་ཆད་ལས་ཡོད་མ་ཡིན། །ཞེས་གསུངས་པ་དང་། གཞན་ཡང་རྡོ་རྗེ་མཁའ་འགྲོ་ལས་མདོར་བསྡུས་ཏེ་བཤད་པ་ནི་
འདི་སྐད་དུ། །དང་པོ་ཉི་མ་ཕྱེད་བྱས་ཉི་མ་རངས་པའམ་ཡང་ན་ཉིན་མཚན་དག་གི་བར་དུ་རྒྱུ། །དེ་ལས་ཞག་གཉིས་དང་ནི་ཞག་གསུམ་འོན་ཏེ་ཞག་བཞི་ཁྱབ་པའི་བར་དུ་ནི། །ཉི་མ་ལ་སྟེ་གཡོན་པ་མིན་པའི་རླུང་དག་དབུགས་ལ་བརྟེན་ཏེ་གང་ཞིག་རྒྱུར་གྱུར་པ། །དེ་ལ་གནས་དང་ཉི་མ་ཕྱོགས་དང་བཀྲ་ཤིས་དྲུག་དང་བཞི་ནི་འདིར་
