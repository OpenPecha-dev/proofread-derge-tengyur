ལ་ཡ་ཀཱི་ལ་ཡ། བཛྲ་མུདྒ་རེ་ཎཱ་ཀོ་ཊ་ཡཱ་ཀོ་ཊ་ཡཱ། བཛྲ་ཙཀྲེ་ཎ་ཙྪེ་ད་ཡ་ཙྪེ་ད་ཡ། བཛྲ་ཀུནྟེ་ན་བྷིནྡ་བྷིནྡ། བཛྲ་དཎྜེ་ཎ་ཏཱ་ཌ་ཡ་ཏཱ་ཌ་ཡ། བཛྲ་པ་ར་ཤུ་ན་ཙྪིནྡ་ཙྪིནྡ། སཱརྡྷ་ཏྲི་ཀོ་ཊི་ཁཎྜ་ཀྲྀ་ཏྭ་ཤྨ་ཤཱན་བྷཱུ་ཏེ་བྷྱོ་བི་ལཾ་ཀུ་རུ་ཀུ་རུ། བཛྲ་ཌ་མ་རུ་ཀེ་ན་བཛྲ་ཌཱ་ཀི་ནཱི་རཱ་བཱ་ཧ་ཡ། བཛྲ་ཌཱ་ཀི་ནཱི་བྷྱོ་
མཱ་ར་ཀཱ་ཡི་ཀཱ་ནཱཾ་རུ་དྷི་རཾ་ནི་བེ་ད་ཡ་བེ་ད་ཡ། པཉྩཱ་མྲྀ་ཏཾ་ནི་རི་ཎཱབྷྱཿ། བཉྩཱ་མྲྀ་ཏཾ་ནི་བེ་ད་ཡ། སརྦ་བཛྲ་ཌཱ་ཀི་ནཱི་ས་ཧི་ཏ། སརྦ་ས་ཏྭཱ་ནཱ་ན་ཤཱནྟི་ཀཾ། །བཽཥྚ་ཀཾ་རཀྵཱ་བ་ར་ཎ་གུཔྟི་ཀུ་རུ་ཀུ་རུ་ཧཱུྃ་ཧཱུྃ་ཕཊ། ཅེས་པ་ནི་ཕྲེང་བའི་སྔགས་སོ། །ཨོཾ་ཨཱཿཧཱུྃ་ཧོཿ[ཧྐྵྨླྺྼྻྃ་]སྭཱ་ཧཱ། སྙིང་པོའོ། །ཨོཾ་ཧྲཱཾ་ཧྲིཾ་ཧྚཱྀ་ཧྲཱཾ་ཧྲཱིཾ་ཧཾཿསྭཱ་ཧཱ། ཉེ་བའི་སྙིང་

ར་མི་ཏཱ་ཡཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་པྲ་ཎི་དྷཱ་ན་པཱ་ར་མི་ཏཱ་ཡཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བ་ལ་པཱ་ར་མི་ཏཱ་ཡཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཛྙཱན་པཱ་ར་མི་ཏཱ་ཡཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཛ་ཡ་གྷ་ཊཻ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བི་ཛ་ཡ་གྷ་ཊེ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་ཙིཧྣོ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་སརྦ་མུ་དྲཱ་བྷྱོ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཙིནྟཱ་མ་ཎ་ཡེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་དྷརྨྨ་གཎྜྱཱ་ཧུྃ་ཕཊ་སྭཱ་
ཧཱ། ཨོཾ་དྷརྨྨཱ་ཤཾ་ཁཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཀལྤ་བྲིཀྵཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨཀྵོ་བྷྱཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨ་མོ་གྷ་སིདྡྷ་ཡེ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་རཏྣ་སཾ་བྷཱ་བཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ཨ་མྲྀ་ཏཱ་བྷཱ་ཡུ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་བཻ་རོ་ཙ་ནཱ་ཡ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་ལོ་ཙ་ནཱ་ཡཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་མཱ་མ་ཀྱཻ་ཧུྃ་ཕཊ་སྭཱ་ཧཱ། ཨོཾ་པཱཎྜ་
