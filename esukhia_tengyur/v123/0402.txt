ཡང་དག་སྒྲུབ་པར་བྱེད། །ཡན་ལག་དྲུག་ཅེས་བྱ་བ་ནི་ངོ་བོ་ཉིད་དོ། །ཡན་ལག་དྲུག་ནི་ཚུལ་ཁྲིམས་དང་ལྡན་པར་གནས་པ་ནས་བསླབ་པའི་གཞི་རྣམས་ལ་ཡང་དག་པར་བླངས་ཏེ། སློབ་པ་ཞེས་བྱ་བའི་བར་ཡིན་ནོ། ཞི་བའི་བསམ་པས་ཟིན་པ་ཞེས་བྱ་བ་ནི་རྒྱུ་
སྟེ། མྱ་ངན་ལས་འདས་པའི་བསམ་པས་ཡང་དག་པར་ལེན་པའི་ཕྱིར་རོ། །བདེ་འགྲོ་གནས་པ་སྦྱིན་བྱེད་དང་། །ཞེས་བྱ་བ་ནི་འབྲས་བུ་སྟེ། ཚུལ་ཁྲིམས་ཀྱིས་བདེ་འགྲོར་འགྲོ་བའི་ཕྱིར་དང་། འགྱོད་པ་མེད་པ་ལ་སོགས་པའི་རིམ་གྱིས་སེམས་གནས་པ་ཐོབ་པའི་ཕྱིར་རོ། །རྟེན་དང་ཞི་དང་
འཇིགས་པ་མེད། །ཅེས་བྱ་བ་ནི་ལས་ཏེ། ཚུལ་ཁྲིམས་ནི་ཡོན་ཏན་ཐམས་ཅད་ཀྱི་རྟེན་ཡིན་ནོ། །ཉོན་མོངས་པའི་ཡོངས་སུ་གདུང་བ་ཞི་བས་ཞི་བའོ། །སྲོག་གཅོད་པ་ལ་སོགས་པའི་རྐྱེན་ཅན་གྱི་རྟེན་འཇིགས་པ་དང་། ཁ་ན་མ་ཐོ་བ་དང་། ཁོན་རྣམས་མི་འབྱུང་བའི་ཕྱིར་འཇིགས་པ་མེད་
པའོ། །བསོད་ནམས་ཚོགས་དང་ལྡན་པ་དང་། །ཞེས་བྱ་བ་ནི་ལྡན་པ་སྟེ། དུས་ཐམས་ཅད་དུ་ལུས་དང་ངག་དང་ཡིད་ཀྱི་ལས་ཡང་དག་པར་ཀུན་ཏུ་སྤྱོད་པའི་ཕྱིར་རོ། །བརྡར་བཏགས་ཆོས་ཉིད་ཐོབ་པ་སྟེ། །སྡོམ་ལ་གནས་པ་རྣམས་ལ་ཡོད། །ཅེས་བྱ་བ་ནི་འཇུག་པ་སྟེ། དེ་ལ་བརྡར་
བཏགས་པས་ཐོབ་པ་ནི་སོ་སོར་ཐར་པའི་སྡོམ་པས་བསྡུས་པའོ། །ཆོས་ཉིད་ཀྱིས་ཐོབ་པ་ནི་བསམ་གཏན་དང་། ཟག་པ་མེད་པའི་སྡོམ་པས་བསྡུས་པའོ། །དེ་ནི་འདིའི་རབ་ཏུ་དབྱེ་བས་འཇུག་པ་སྟེ། རབ་ཏུ་དབྱེ་བ་རྣམ་པ་གསུམ་གྱིས་འཇུག་པའི་ཕྱིར་རོ། །སྡོམ་ལ་གནས་པ་རྣམས་ལ་
ཡོད། །ཅེས་བྱ་བ་ནི་རྟེན་གྱིས་འཇུག་པའོ། །བཟོད་པ་རབ་ཏུ་དབྱེ་བ་ལ་ཚིགས་སུ་བཅད་པ་གཉིས་ཏེ། མི་མཇེད་ཇི་མི་སྙམ་ཤེས་དང་། །སྙིང་རྗེས་ཆོས་ལ་བརྟེན་པའི་ཕྱིར། །ཕན་ཡོན་ལྔར་ནི་ཡང་དག་བཤད། །དེ་ནི་གཉིས་ཀྱི་དོན་བྱེད་དེ། །དཀའ་ཐུབ་གཙོ་དང་ཀུན་ཏུ་ལྡན། །
དེ་ལ་དེ་ནི་རྣམ་གསུམ་འདོད། །བཟོད་པ་དེ་འདྲར་ཡོངས་ཤེས་ནས། །མཁས་པ་ཡང་དག་སྒྲུབ་པར་བྱེད། །མི་མཇེད་ཇི་མི་སྙམ་ཤེས་དང་། །ཞེས་བྱ་བ་ནི་བཟོད་པ་རྣམ་པ་གསུམ་གྱི་ངོ་བོ་ཉིད་ཡིན་ནོ། །གནོད་པ་བྱེད་པ་ལ་མི་མཇེད་པའི་བཟོད་པ་ནི་མི་མཇེད་པའི་བཟོད་པ་སྟེ། མི་
