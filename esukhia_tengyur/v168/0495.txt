བའི་དུས་སུ་ཡིད་ལ་གཅགས་བཞིན་དུ་འཆི་བའི་ཕྱིར་རོ། །ཡིད་ལ་གདུང་བ་སྐྱེ་བ་ལས་ནི་ངན་འགྲོ་སྐྱེ་བར་འགྱུར་ཏེ། འཇིག་རྟེན་ཕ་རོལ་དུ་ངན་སོང་གི་སྡུག་བསྔལ་འབྱུང་བའི་ཕྱིར་བདེ་བ་འགྲུབ་པ་མེད་དོ། །བདེ་བའི་ཐབས་ཞེས་བྱ་བ་ནི།
གང་གི་ཕྱིར་ཁྱིམ་དང་ཁྱིམ་ལ་གནས་པ་བདེ་བའི་རྒྱུའོ་ཞེས་མངོན་པར་ཞེན་པའི་དབང་དུ་བྱས་པ་ནི་ཁྱིམ་ལ་བདེ་བའི་ཐབས་ཞེས་བརྗོད་དོ། །ངན་འགྲོ་བཞིན་དུ་ཞེས་བྱ་བ་ནི་ངན་འགྲོ་བཞིན་དུའོ། །ཁྱིམ་ན་གནས་པ་བདེ་བར་ནི་ག་ལ་བཤད་དེ། དེ་བས་ན་དད་
པས་ཡིད་ཆེས་པར་བྱ་བ་ཡིན་པར་བདག་གིས་མ་མཐོང་ངོ་ཞེས་བཤད་པ་ནི་གཞན་གྱི་ལས་བྱེད་པ་དག་ཀྱང་ཞེས་བྱ་བའི་ཚིགས་བཅད་ཀྱིས་སོ། །བཅིང་བ་དང་གསད་པ་དང་སྡུག་བསྔལ་ནི་བཅིང་ཞིང་གསད་པའི་སྡུག་བསྔལ་ལོ། །དེ་དག་ཉིད་ཀྱི་གཞིར་གྱུར་
པ་ཡིན་ཏེ། ཡོད་པར་གྱུར་ན་བཅིང་བ་དང་། གསད་པའི་སྡུག་བསྔལ་གྱི་གཞིར་གྱུར་པའོ། །འདོད་པ་དེ་དག་གལ་ཏེ་ངོམས་པར་གྱུར་ན་མི་རུང་ན་དེ་ཡང་མེད་དེ། དེ་བསྟན་པའི་ཕྱིར་རྒྱལ་པོ་དག་ཀྱང་ཞེས་བྱ་བའི་ཚིགས་བཅད་ཕྱེད་ཡིན་ནོ། །<ཡངས་,ཡོངས་>
སུ་རྟོགས་པའི་དབང་དུ་བྱས་པའི་ཚིགས་བཅད་ནི། བསྒྲུབ་ཅིང་བསྲུང་བ་ལ་སོགས་པ་ཞེས་བྱ་བའི་ཚིགས་བཅད་དེའི་དོན་རྣམ་པར་འགྲེལ་ཏེ། འདི་ག་ལ་བདེ་ཞེས་བྱ་བ་ནི་བཅིངས་པ་དང་གསད་པའི་སྡུག་བསྔལ་གྱི་གཞིར་གྱུར་པས་ག་ལ་བདེ་སྟེ། དེ་བས་ན་ག་ལ་
བདེའོ། །ཇི་ལྟར་བདེ་ཞེས་བྱ་བ་ནི་རྒྱལ་པོ་དག་ཀྱང་འབྱོར་པས་ཆོག་པ་མེད་དེ། དེས་ན་ཇི་ལྟར་བདེ་བའོ། །ནམ་ཞིག་བདེ་ཞེས་བྱ་བ་ནི་བསྒྲུབ་ཅིང་བསྲུང་ལ་སོགས་པའི་ཚེ་ན་སྡུག་བསྔལ་བའི་ཕྱིར་ནམ་ཞིག་བདེ་བ་ཡིན་ཏེ། དེས་ན་ཡང་རྣམ་པ་གསུམ་གྱིས་བདེ་
བ་མེད་དོ། །ཇི་ལྟར་དེ་ལ་འཇིག་རྟེན་པ་རྣམས་བདེ་བའི་བློ་འབྱུང་ཞེ་ན། སྨྲས་པ། ཡོངས་སུ་རྟོགས་པའི་དབང་གིས་བདེ་བ་འབྱུང་བ་ཡིན་གྱི། ངོ་བོ་ཉིད་ཀྱིས་ནི་མ་ཡིན་ནོ། །དེ་ལྟར་རེ་ཞིག་བདེ་བ་ཡིད་ཀྱིས་ཡོངས་སུ་རྟོགས་པའི་མཐུ་ཡིན་པར་བསྟན་ནས། ཡུལ་
