ཆུང་བ་སྟེ་བུད་མེད་དུ་གྱུར་པའོ། །ད་ནི་སྦྱིན་པའི་དངོས་པོ་ཕུན་སུམ་ཚོགས་པ་དང་། སྦྱིན་པ་པོ་ཕུན་སུམ་ཚོགས་པ་ནི་ཐམས་ཅད་སྒྲོལ་གྱི་སྐྱེས་པའི་རབས་ཡིན་ཏེ། སྦྱིན་པའི་དངོས་པོ་ཕུན་སུམ་ཚོགས་པ་ནི་ཕྱི་རོལ་གྱི་དངོས་པོ་སེམས་
ཅན་དང་སེམས་ཅན་མ་ཡིན་པའི་གྲངས་སུ་འགྲོ་བའི་རྣམ་པ་ཐམས་ཅད་དེ། ཕྱི་རོལ་གྱི་སྦྱིན་པའི་དངོས་པོའོ། །སེམས་ཅན་གྱི་གྲངས་སུ་འགྲོ་བའི་ཕྱི་རོལ་གྱི་སྦྱིན་པ་ནི་གླང་པོ་རིན་པོ་ཆེ་དང་། རྟ་ཆས་དང་བཅས་པ་ལ་སོགས་པ་མཐའ་དག་སྦྱིན་པའོ། །སེམས་
ཅན་མ་ཡིན་པ་ནི་བུ་སྨད་དང་བཅས་པ་ཡིན་ཡང་དེ་ལ་མ་ལྟོས་པར་ཤིང་རྟ་སྦྱིན་པའོ། །བུ་སྦྱིན་པ་དང་། ཆུང་མ་སྦྱིན་པ་ནི་བདག་ཉིད་<གྱི་,ཀྱི་>ཕྱི་རོལ་གྱི་སྦྱིན་པའི་དངོས་པོ་ཁྱད་པར་དུ་གྱུར་པའོ། །སྦྱིན་པ་པོ་ཕུན་སུམ་ཚོགས་པའི་ཁྱད་པར་དུ་གྱུར་པ་དེ་ཉིད་དུ་བསྟན་ཏེ། འཇིག་
རྟེན་དོན་དུ་ང་ཡི་ལུས་འཛིན་ན། །ཕྱི་རོལ་ནོར་ལྟ་སྨོས་ཀྱང་ཅི་ཞིག་དགོས། །ཤི་བི་པ་རྣམས་ཐམས་ཅད་ཀྱིས། །ང་ནི་བསད་དམ་སྤྱུགས་ཀྱང་རུང་། །མི་སྦྱིན་པར་ནི་ང་མི་བྱེད། །དཀའ་ཐུབ་ཚལ་དུ་འགྲོར་ཟད་མོད། །ཅེས་བྱ་བའི་ཚིག་གིས་སོ། །གཞན་ཡང་། སྦྱིན་
ལ་དགའ་བ་གོམས་བྱས་ཤིང་། །སླར་བཟློག་པ་ནི་མ་བསླབས་པས། །རབ་ཏུ་སྡུག་པའི་བུ་གཉིས་ཀྱང་། །ཚམ་ཚོམ་མེད་པར་སྦྱིན་ཞེས་སྨྲས། །ཞེས་བྱ་བ་དང་། གཞན་ཡང་ཕྱི་རོལ་གྱི་དངོས་པོ་སྦྱིན་པ་པོ་ཕུན་སུམ་ཚོགས་པའི་དབང་དུ་བྱས་ནས་བཤད་པ་དེ་ཐམས་
ཅད་ཀྱང་སྦྱར་བར་བྱའོ། །ད་ནི་ཚོང་དཔོན་ཟིལ་གྱིས་མནོན་པའི་སྐྱེས་པའི་རབས་ཀྱིས་སྦྱིན་པ་པོ་ཕུན་སུམ་ཚོགས་པ་དེའི་དབང་དུ་བྱས་ནས་བསྟན་པ་ཡིན་ནོ། །ཕུན་སུམ་ཚོགས་པ་དེ་ཡང་དེ་ཉིད་དུ་བསྟན་ཏེ། འབྱོར་པ་ཟད་པར་གྱུར་ཀྱང་སྦྱིན་པ་ལས་མི་ལྡོག་པ་ཡིན་ཏེ། བརྒྱ་
བྱིན་གྱིས་ནོར་རྣམས་མི་སྣང་བར་བྱས་ཀྱང་ལྟོས་པ་མེད་པས་སོ། །དེ་ལྟར་རེ་ཞིག་གསུམ་གྱིས་ཕུན་སུམ་ཚོགས་པ་ཡིན་ཏེ། སྦྱིན་པ་པོ་དང་། སྦྱིན་པའི་དངོས་པོ་དང་། ལེན་པ་པོའོ། །སྦྱིན་པ་པོ་དང་ལེན་པ་པོ་གཉིས་ཀྱིས་ཕུན་སུམ་ཚོགས་པ་ཡང་ལེན་པ་པོའི་ཡིན་གྱི། སྦྱིན་
