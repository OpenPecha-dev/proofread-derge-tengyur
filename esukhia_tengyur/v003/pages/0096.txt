ཌཎྜི་མ་ཞེས་བྱ་བ་ཆོས་ཐམས་ཅད་མི་དམིགས་པའི་ཏིང་ངེ་འཛིན་ཏོ། །ཎྞ་བཛྫ་ཨ་ཨི་ཞེས་བྱ་བ་ནི་རང་གི་ངང་གིས་འཇུག་པར་བྱའོ་ཞེས་བྱ་བར་དགོངས་པའོ། །ཧེ་རུ་ཀའི་གཟུགས་ཀྱིས་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་ཧེ་རུ་ཀའི་གཟུགས་ནི་ཧེ་རུ་ཀའི་ལུས་སོ། །ཇི་ལྟ་བུ་ཞིག་ཅེ་ན། དྲན་
པ་མི་འཕྲོགས་རྣལ་འབྱོར་ལས། །ཞེས་བྱ་བ་སྟེ། ཏིང་ངེ་འཛིན་བརྟན་པ་ལས་སོ། །དེ་ཉིད་གསུངས་པ། བསྒོམ་པ་ལ་ཆགས་སེམས་ཀྱིས་ནི། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། རྡོ་རྗེ་ཆོས་ཞེས་བྱ་བ་ནི་པདྨ་གར་གྱི་དབང་ཕྱུག་གོ། །སངས་རྒྱས་ཞེས་བྱ་བ་ནི་རྣམ་པར་སྣང་མཛད་ཀྱི་གཟུགས་
སོ། །མ་མོ་ཡིས་ཞེས་བྱ་བ་ནི་བཅོམ་ལྡན་འདས་མ་བདག་མེད་མའོ། །འདི་དག་ཅེས་བྱ་བ་ནི་བསྟན་པའི་མཚན་ཉིད་ཅན་འདི་དག་གོ། །ལེགས་པར་ཞེས་བྱ་བའི་དོན་ཏོ། །ཚོགས་སྲུང་ཞེས་བྱ་བ་ནི་ཚོགས་ཀྱི་འཁོར་ལོ་སྲུང་བའོ། །དབང་བྱེད་ཅེས་བྱ་བ་ནི་དབང་དུ་འགྱུར་བའོ། །
འདིས་ནི་ཞེས་བྱ་བ་ནི་འདིས་སྔགས་བཟླས་པ་ཡང་བྱས་པར་འགྱུར་རོ། །དྲི་ནི་ཞེས་བྱ་བ་ནི་དྲི་ཉིད་དོ། །གླུའི་བྱིན་རླབས་ཀྱིས་རྣལ་འབྱོར་མ་རྣམས་དེར་མཚོན་པར་བྱ་བ་སྟེ་མཚན་མའོ། །སྒྲ་ཞེས་བྱ་བ་ནི་ཀུ་ཅོ་སྟེ་ངང་པ་ལ་སོགས་པའོ། །གླུའི་བྱིན་རླབས་ཀྱིས་མཐའ་ནས་ཐོས་
པར་འགྱུར་རོ། །བ་ལང་འཚེ་ཞེས་བྱ་བ་ནི་ཅེ་སྤྱང་ངོ་། །ཕྱི་རོལ་ཚལ་དུའང་ཞེས་པའང་གི་སྒྲས་ནི་ནམ་མཁའ་ཡང་བསྡུའོ། །འདིས་ནི་གླུ་དང་གར་ཞེས་བྱ་བ་དྲིས་པའི་ལན་བསྟན་པ་ཡིན་ནོ། །དྲིས་པ་གཉིས་པའི་ལན་བསྟན་པའི་ཕྱིར་ཕྱག་རྒྱ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ།
ཕྱག་རྒྱ་དང་རྟགས་དང་མཚན་མ་ཞེས་བྱ་བ་ནི་རྣམ་གྲངས་སོ། །ཕྱག་རྒྱ་འདིས་མཚོན་པར་བྱ་སྟེ་རབ་ཏུ་རྟོགས་པར་བྱའོ། །འདི་དག་གིས་ཅིར་འགྱུར་ཞེ་ན། རིགས་འཆོལ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། རྣལ་འབྱོར་སྒོམ་པར་བྱེད་པ་ན། རིགས་གཞན་གྱིས་རྒྱས་
བཏབ་པར་གྱུར་པ་ན། སྒྲུབ་པ་པོས་དངོས་གྲུབ་འགྲུབ་པར་མི་འགྱུར་གྱི། དམ་ཚིག་ཉམས་པར་གྱུར་པས་དེའི་ཕྱིར་རིགས་ཡོངས་སུ་ཤེས་པར་བྱ་བ་ལ་འབད་པར་བྱའོ་ཞེས་བྱ་བར་དགོངས་པའོ། །བདག་མེད་མ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་ཞེ་སྡང་ནི་མི་བསྐྱོད་པའོ། །གཏི་མུག་ནི་
