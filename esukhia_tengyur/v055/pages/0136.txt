བྱ་བ་སྟེ། དེ་ལྟ་བས་ན་རང་གི་ངོ་བོ་ཉིད་ཀྱིས་རྣམ་པར་དག་པས་ན་རང་བཞིན་གྱིས་དག་པ་སྟེ། གང་གཞན་ལ་རག་ལས་པར་མ་གྱུར་ཅིང་། རང་ཉིད་ཀྱི་ངོ་བོ་ལས་མི་འགྱུར་བའི་རང་བཞིན་འདི་ནི་བདག་སྟེ། བདག་ཉིད་ཀྱི་ཆོས་ཀྱི་ངོ་བོ་ཡིན་ལ། དེའི་རྣམ་པར་དག་པ་ནི་རང་བཞིན་གྱིས་རྣམ་
པར་དག་པའོ། །པས་ཞེས་བྱ་བ་ནི་ཉིད་ཅེས་བྱ་བའི་དོན་ཏོ། །རང་བཞིན་གྱིས་རྣམ་པར་དག་པ་དེ་ཉིད་ཀྱིས་ནའོ། །འདོད་ཆགས་ཡིན་ཡང་དྲི་མ་མེད། །ཅེས་བྱ་བ་ནི། གང་གི་ཕྱིར་འདོད་ཆགས་ནི་འཇིག་རྟེན་ན་ཞེན་པའི་ངོ་བོར་གྲགས་ན། དེ་ཡང་རང་བཞིན་གྱིས་རྣམ་པར་དག་པ་ཡིན་
པའི་ཕྱིར་རྣམ་པར་དག་པ་སྟེ། དོན་དམ་པར་མ་སྐྱེས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཡང་ན་མི་དགེ་བ་བཅུའི་ལས་ཀྱི་ལམ་ཀུན་ནས་མི་སྐྱེད་ཅིང་། བསོད་ནམས་དང་ཡེ་ཤེས་ཀྱི་ཚོགས་ཡོངས་སུ་རྫོགས་པའི་རྒྱུར་གྱུར་པའི་ཕྱིར་སེམས་ཅན་སྨིན་པ་དང་གྲོལ་བ་ལ་ཆགས་པ་ནི་རང་བཞིན་གྱིས་
དག་པ་ཉིད་དོ། །ད་ནི་དེ་བཞིན་གཤེགས་པ་བཞི་པས་ཀྱང་རང་གི་མཚན་མའི་ཕྱག་རྒྱ་བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རྣམ་པར་སྣང་མཛད་ལ་འཁོར་དུ་དབུལ་བར་བསྟན་པའི་ཕྱིར། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པའི་གཞུང་གསུངས་པ་ཡིན་ཏེ། སྒྲིབ་པ་མེད་པའི་ཡེ་ཤེས་ཐུགས་
སུ་ཆུད་པས་ཐམས་ཅད་དུ་གདོན་མི་ཟ་ཞིང་། འབྲས་བུ་ཡོད་པའི་དངོས་གྲུབ་མངའ་བས་ན་གདོན་མི་ཟ་བར་གྲུབ་པའོ། །བཅོམ་ལྡན་འདས་དེ་བཞིན་གཤེགས་པ་རྣམ་པར་སྣང་མཛད་ཀྱི་སྔར་བཤད་པའི་སེམས་དཔའ་རྣམ་པ་བཅུ་དྲུག་གི་མཚན་ཉིད་ཅན་གྱི་དེ་བཞིན་གཤེགས་པ་ཐམས་
ཅད་ཀྱི་ཡེ་ཤེས་ལ་ཕྱག་རྒྱས་གདབ་པའི་ཕྱིར། ལས་ཀྱི་ཕྱག་རྒྱ་སྦྱོར་བའི་དུས་སུ་ཐུགས་ཀའི་ཟླ་བའི་དཀྱིལ་འཁོར་ལ། ལས་ཀྱི་རྡོ་རྗེ་ཉེ་བར་དགོད་པས་རྒྱས་གདབ་ཅིང་རང་གི་རིགས་ཀྱི་མཚན་མ་གཞག་པའི་ཕྱིར་རོ། །བརྩོན་འགྲུས་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཐམས་ཅད་ཀྱི་ངོ་བོ་ཉིད་དེ། ཚོགས་
ཡོངས་སུ་རྫོགས་པར་མཛད་པའི་དུས་སུ་ཐམས་ཅད་དུ་འགྲོ་བ་ཡིན་པས་ན་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཐམས་ཅད་ཅེས་བཤད་དེ། བརྩོན་འགྲུས་འབའ་ཞིག་གྲོགས་གྱུར་པས། །གོ་འཕང་བླ་ན་མེད་པ་བརྙེས། །ཞེས་ཇི་སྐད་བཤད་པ་ལྟ་བུའོ། །ཕ་རོལ་ཏུ་ཕྱིན་པ་ཐམས་ཅད་པ་དེའི་དམ་
