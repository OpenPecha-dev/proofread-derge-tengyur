ཤ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་ཀ་ལ་ཤ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཨོཾ་བཛྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ། ཨོཾ་བུདྡྷ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་རཏྣ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཨོཾ་རཏྣ་མཱ་ལ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་བཛྲ་པཊྚ་ཨ་བ་ལྃ་བ་ན་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་བཛྲ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་བུདྡྷ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་
ཨོཾ། ཨོཾ་རཏྣ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་མུ་དྲ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཨོཾ་བཛྲ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་བུདྡྷ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་རཏྣ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ཨོཾ་པདྨ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་ནཱ་མ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ༀ་བཛྲ་ཙཀྲ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ་
བྷྲཱུཾ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་དྷི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་ཨོཾ་ཨོཾ་ཨོཾ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་དྷི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་དྷི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ་ཏྲཱཾ་ཏྲཱཾ། ཨོཾ་བཛྲ་ཙཀྲ་ཨ་དྷི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿཧྲཱིཿཧྲཱིཿ། ༀ་བཛྲ་ཙཀྲ་ཨ་དྷི་པ་ཏི་ཨ་བྷི་ཥིཉྩ་ཨཿཨཿཨཿ། ༀ་བཛྲ་དྷ་ར་ཎ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ཨོཾ་
བུདྡྷ་དྷ་ར་ཎ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ཨོཾ་རཏྣ་དྷ་ར་ཎ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ༀ་པདྨ་དྷ་ར་ཎ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་དྷ་ར་ཎ་ཨ་བྷི་ཥིཉྩ་ཨཿ། ཨོཾ་བཛྲ་གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཧཱུྃ། ༀ་བུདྡྷ་གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཨོཾ། ༀ་རཏྣ་གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཏྲཱཾ། ༀ་པདྨ་གུ་ཧྱ་ཨ་བྷི་ཥིཉྩ་ཧྲཱིཿ། ཨོཾ་ཀརྨ་གུ་ཧྱ་
ཨ་བྷི་ཥིཉྩ་ཨཿ། ཨོཾ་པྲ་ཛྙཱོ་པ་ཡོ་ས་མ་ཡོ་ག་ཨ་བྷི་ཥིཉྩ་ཧཱུཾ་ཨཱཿཞེས་པས་དབང་བསྐུར་བར་བྱའོ། །དེ་ཡང་མདོ་ལས། རྣམ་བཅུ་དབང་ལྡན་ཆེ་བ་པོ། །ཞེས་གསུངས་སོ། །དེ་ནས་ཚོགས་སོགས་ཀྱི་དཀྱིལ་འཁོར་རྣམས་བདག་ལ་བསྒྲུབ་པར་བསྟན་པའི་ཕྱིར། དེ་ནས་ནམ་
མཁའི་འཁོར་ལོའི་ཚོགས། །ཞེས་པ་གསུངས་ཏེ། དེ་ནས་ནི་དབང་བསྐུར་བའི་དེ་མ་ཐག་པའོ། །ནམ་མཁའི་འཁོར་ལོའི་ཚོགས་ནི་ཚོགས་མདུན་ན་སྤྲུལ་པའི་ལྷ་རྣམས་སོ། །མ་ལུས་བདག་གིས་བསྡུ་བར་བྱ། །ཞེས་པ་ལ་སོགས་པའི་ཚིག་བཞི་གོ་སླའོ། །སླར་གསོ་ལུས་
དང་ལྡན་པར་བྱ། །ཞེས་པ་རྣམ་སྨིན་གྱི་ཕུང་པོ་སྦྱངས་ནས་ཡེ་ཤེས་ཀྱི་ལུས་སྣང་ལ་རང་བཞིན་མེད་པ་གསལ་ལ་རྙོག་པ་མེད་པ་འཇའ་ཚོན་ལྟ་བུ་བསམ་པར་བྱའོ། །དེ་ཡང་རྡོ་རྗེ་གུར་གྱི་རྒྱུད་ལས། ཇི་ལྟར་ཆུ་ནང་ཟླ་བ་དེ། །བདེན་མིན་བརྫུན་པ་མེད་པ་ལྟར། །དེ་བཞིན་རྣལ་འབྱོར་བསྒོམ་པ་
