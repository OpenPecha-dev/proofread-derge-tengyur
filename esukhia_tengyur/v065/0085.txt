ཨོཾ་བྷ་བ་ཀ་ཨགྣ་ཡེ་ག་ག་ན་ཡེ་སྭཱཧཱ། ཞེས་བརྗོད་ལ། ཨོཾ་བྷ་བ་ཀ་ཨགྣི་ཛཿཧཱུཾ་བཾ་ཧོཿ། ཞེས་བྱ་བས་དགུག་པ་དང་གཞུག་པ་དང་། བཅིང་བ་དང་དབང་དུ་བྱ་བ་དང་། དམ་ཚིག་གི་ཕྱག་རྒྱ་བསྟན་ཅིང་། ཨོཾ་བཛྲ་ཨགྣ་ཡེ་མ་ཧཱ་བྷཱུ་ཏ་ཛྭ་ལ་སརྦ་བྷསྨིཾ་ཀུ་རུ་སརྦ་དུཥྚཱ་ན་ཀྲོ་དྷ་ཛཿཧཱུཾ་བཾ་ཧོཿས་མ་
ཡ་སྟྭཾ། ཨ་ཧཱུཾ་ཞེས་བརྗོད་པས་ཕྱག་རྒྱ་བཞིས་རྒྱས་བཏབ་ཅིང་། དེ་ལ་དབང་བསྐུར་སྦྱིན་ནོ། །ཡི་གེ་གསུམ་གྱིས་བྱིན་གྱིས་བརླབས་ལ། ཨོཾ་བྷ་བ་ཀ་བཛྲ་ཨགྣ་ཡེ་བཛྲ་པུཥྤཾ་ཞེས་བྱ་བ་ལ་སོགས་པས་མཆོད་པ་ལྔ་དང་། མཆོད་ཡོན་དབུལ་བར་བྱའོ། །དེ་ནས་ཐོག་མར་མེའི་གཟི་བརྗིད་བསྐྱེད་པའི་ཕྱིར་མར་
གྱི་དགང་བླུགས་བདུན་ནམ་ལྔ་དབུལ་ལོ། །ཨོཾ་བཛྲ་ཨ་ན་ལ་མ་ཧཱ་བྷཱུ་ཏ་ཛྭ་ལ་སརྦ་བྷསྨིཾ་ཀུ་རུ་སརྦ་དུཥྚཱ་ན་ཧཱུཾ་ཕཊ་ཅེས་བརྗོད་ལ་མེའི་མདོག་ལེགས་པར་བརྟགས་ཏེ། དཀར་སེར་འཁྱིལ་ཞིང་འབར་བ་དང་། །མ་ཆད་མ་བྱེ་མི་འཕྲོ་དང་། །སྒྲ་དང་དུ་བ་ཀུན་དུ་སྤངས། །འཇམ་ལ་རྣོ་ཞིང་གསལ་ན་བཟང་། །

ཡཾ་ཤིང་གིའོ། །དེ་བཞིན་དུ་ཀུན་ལ་སྦྱར་བར་བྱའོ། །ཚེ་ལས་འདས་པ་ཆེ་གེ་མོ་སརྦ་པཱ་པཾ་ད་ཧ་ན་ཡ་སྭཱ་ཧཱ་ཏིལ་གྱིའོ། །ཚེ་ལས་འདས་པ་ཆེ་གེ་མོ་སརྦ་པཱ་པཾ་ཛྭ་ལ་རཾ་སྭཱ་ཧཱ། ཡུངས་ཀར་གྱིའོ། །མར་དང་འོ་མ་དང་ཞོ་ལ་སོགས་པ་ཞུ་བའི་རྫས་རྣམས་ནི་མེ་ཉིད་ཀྱི་
སྔགས་ཀྱིས་སོ། །སརྦ་སཾ་བ་དེ་སྭཱ་ཧཱ། འབྲས་ཆན་གྱིའོ། །ཨཱ་ཡུ་ཥེ་སྭཱ་ཧཱ། དཱུར་བའི་མྱུ་གུའི་འོ། །ཨོཾ་ཨ་པྲ་ཏི་ཧ་ཏ་ཡེ་སྭཱ་ཧཱ། ཀུ་ཤའི་འོ། །ཨོཾ་མ་ཧཱ་བེ་ག་ཡེ་སྭཱ་ཧཱ། ནས་ཀྱིའོ། །ཨོཾ་བཛྲ་གྷསྨ་རི་ཡེ་སྭཱ་ཧཱ། གྲོའི་འོ། །ཨོཾ་བཛྲ་བི་ཛ་ཡེ་སྭཱ་ཧཱ། སོ་བའི་འོ། །ཨོཾ་མ་ཧཱ་བ་
ལ་ཡེ་སྭཱ་ཧཱ། སྲན་མའི་འོ། །ཨོཾ་བཛྲ་མུ་ཏ་ཡེ་སྭཱ་ཧཱ། སྲན་མ་མ་ཤའི་འོ། །ཨོཾ་སརྦ་པུཥྚ་ཡེ་སྭཱ་ཧཱ། འབྲས་སོ་བའི་འོ། །ༀ་མ་ཧཱ་ཤོ་པ་ཡེ་སྭཱ་ཧཱ། བིལ་བ་དང་། སྟར་ག་དང་། སྨན་ཐམས་ཅད་ཀྱིའོ། །ཡམ་ཤིང་མེ་འོད་ལའོ། །མར་ཁུ་ཞལ་དུའོ། །འབྲུའི་རྣམ་པ་
