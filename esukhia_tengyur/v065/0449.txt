བཤད་བྱ། །སོར་བཅུ་བཅུ་གཉིས་བཅུ་དྲུག་བཅོ་བརྒྱད་དང་། །མཆོག་གི་ཚད་ནི་སོར་ཡང་ཉི་ཤུའོ། །གསེར་གྱི་རྡོ་རྗེ་རིག་འཛིན་ཉིད་དང་ནི། །ས་ནི་ཐོབ་པར་བྱ་ཕྱིར་རབ་ཏུ་ཤིས། །རྒྱལ་པོའི་དོན་དུ་དངུལ་ནི་མཆོག་ཡིན་ཏེ། །ཟངས་ཀྱི་རིན་ཆེན་འབྱུང་བའི་གླུ་བསྐུལ་ལ། །
ལྷ་མིན་དབང་པོ་འཁྲུལ་འཁོར་གཞིག་པ་དང་། །ཕུབ་ཏུ་འགྲོ་བ་རྡོ་ཡི་རྡོ་རྗེའོ། །དོན་ཀུན་བསྒྲུབ་པ་ལ་ནི་ལྕགས་གསུམ་གྱི། །གནོད་སྦྱིན་སྡེ་འཇོམས་པ་ལ་ལྕགས་ཀྱིའོ། །དོན་གཞན་ལ་ནི་སེང་ལྡེང་དཔལ་དང་ཚེ། །ནད་མེད་ནོར་རྫས་ལ་ནི་བིལ་བའོ། །གནོད་སྦྱིན་མ་བསྒྲུབ་པ་
ལ་མ་དྷུའི་ཤིང་། །ཆོས་ཀྱི་དོན་དུ་ཨ་ཤྭད་ལས་བྱས་ཤིང་། །གསལ་ཞིང་རྩེ་མོ་དགར་རྣམས་ཞིག་པ་ལ། །མི་རྣམས་<པ་སང་,བསད་>པ་ལ་ནི་མི་རུས་ཀྱི། །ཤེལ་ལས་བྱས་པའི་མིག་འཕྲུལ་ཆེན་པོ་ལ། །དབྱེ་ཕྱིར་ནིམ་པ་ལས་བྱས་རབ་ཏུ་ཤིས། །བསད་ཅིང་སྐམ་ལ་བ་རུ་ར་ཡི་ཤིང་། །
ཤ་ཟ་བསྒྲུབ་པ་ལ་ཡང་རྟག་ཏུ་ཤིས། །ལྷ་དང་གནོད་སྦྱིན་དྲི་ཟ་བསྒྲུབ་པ་ལ། །སྒྲོན་ཤིང་ལས་ནི་བྱས་པའི་རྡོ་རྗེའོ། །ཇི་སྟེ་རླུང་ནི་དབང་དུ་བྱེད་འདོད་ན། །དེ་ཚེ་ཀླུ་ཤིང་རྩ་ལས་བྱེད་པའོ། །གཟུགས་བསྒྱུར་པ་ལ་ས་ལས་བྱས་པའོ། །རོ་ལངས་བསྒྲུབ་པ་ལ་ནི་ཀ་དམ་པའི། །བུ་ནན་
ག་དང་ར་ཛ་ཨརྐ་དང་། །ཨ་ཤྭད་ཤིང་གི་རྡོ་རྗེ་ནོར་གྱི་ཕྱིར། །བླ་བ་ཨ་མྲ་ཤྲི་བརྞ་བ། །ཨརྫུ་ལ་སོགས་པ་ལས་བྱས་པ་ནི། །མི་རྣམས་གཡུལ་ལས་རྒྱལ་བར་བྱ་ཕྱིར་རོ། །ཡིད་དུ་འོང་བའི་འདོད་པ་ཐོབ་བྱའི་ཕྱིར། །ཙནྡན་དཀར་པོ་དང་ནི་དམར་པོ་སྟེ། །དབྱིབས་ལེགས་
འཇམ་ཞིང་མ་ཉམས་རྨ་མེད་པའོ། །རྡོ་རྗེ་རྩེ་ལྔ་སྲ་བ་བཅང་བར་བྱ། །དང་པོ་ཟློས་ཚེ་དེ་ནི་ཐོགས་ནས་སུ། །བཟླས་བརྗོད་ཟིན་ན་མྱུར་བ་ཁོ་ནར་ནི། །དད་པའི་སེམས་ཀྱིས་ཕྱག་བྱས་དྲི་སོགས་ཀྱིས། །མཆོད་ནས་རྡོ་རྗེ་རྒྱལ་བའི་ཞབས་དྲུང་གཞག །ང་ཡི་རིགས་ཀྱི་གསང་
སྔགས་སྒྲུབ་པ་པོས། །རྡོ་རྗེ་མེད་ན་འགྲུབ་པར་མི་འགྱུར་རོ། །དེ་ལས་གཅིག་ཀྱང་རྙེད་པར་མ་གྱུར་ན། །ཁུ་ཚུར་བཅངས་ལ་རྩེ་གཅིག་སེམས་ཀྱིས་བཟླས། །ཞེས་འབྱུང་། ལྕགས་གསུམ་ལས་བྱས་དྲིལ་བུ་མཐར། །ཕྱི་ལ་པདྨ་བ་ལང་ཁ། །རྡོ་རྗེ་པདྨ་བདེ་གཤེགས་རིགས། །
