འགྱུར་བཤད། །སྲད་བུ་དག་ནི་ཆོ་ག་བཞིན། །དེ་ཡི་ཉེ་འཁོར་ཤེས་པར་བྱ། །དེ་ལ་སོགས་པའི་རྣམ་པ་ཡིས། །མེ་ཏོག་རྣམས་ནི་དོར་བ་ལས། །བདག་གི་མདུན་དུ་གང་ལྷུང་བ། །དེ་ལ་ཞི་བ་བརྩམ་པར་བྱ། །སྦྱིན་སྲེག་གིས་ནི་ཞི་བྱས་ནས། །དེ་ནི་
སླར་ཡང་དོར་དུ་གཞུག །ལན་གསུམ་སྤོགས་ཏེ་འདོར་ན་ཡང་། །གལ་ཏེ་ལེགས་པར་མ་བབ་ན། །རེ་ཐག་བཅད་དེ་སླར་གཏང་བྱ། །དེ་ནས་མེ་ཏོག་བབས་པ་ལས། །ལྷ་ཡང་དག་པར་ཤེས་བྱས་ནས། །ཕྲེང་བ་བླངས་ཏེ་སྔགས་འདི་བརྗོད། །ཨོཾ་པྲ་ཏི་གྲྀཧྞ་ཏྭ་མི་མཾ་སཏྭ་མ་ཧཱ་
བ་ལ་ཞེས་བརྗོད་ལ་དེའི་མགོ་ལ་བཅིང་ངོ་། །ཡེ་ཤེས་སེམས་དཔའི་མཚན་མ་ཉེ་བར་མཚོན་ནས། །དེ་ལ་མིག་གིས་ཇི་ལྟར་མཐོང་ཞེས་དྲིའོ། །དེས་ཇི་ལྟར་མཐོང་བ་བཞིན་དུ་འགྲུབ་པར་འགྱུར་རོ། །གལ་ཏེ་ཁ་དོག་སྣང་བར་མཐོང་ན། །དེ་ལ་དངོས་གྲུབ་ཐམས་ཅད་ཀྱི་ཡེ་ཤེས་
བསླབ་པར་བྱའོ་ཞེས་རྒྱ་ཆེར་སྦྱར་རོ། །དེ་ནས་འདིས་གདོང་གཡོགས་དགྲོལ་ཏེ། ཇི་ལྟར་མིག་མཁན་རྒྱལ་པོ་ཡིས། །མིག་གི་ལིང་ཏོག་བསལ་བ་ལྟར། །བུ་ཁྱོད་ཀྱིས་ནི་མི་ཤེས་པའི། །རབ་རིབ་རྒྱལ་བ་རྣམས་ཀྱིས་བསལ། །བཛྲ་ཙཀྵུ་ཧེ་བཛྲ་པཱ་ཤ་ཞེས་བརྗོད་དོ། །དེ་ནས་གདོང་
གཡོགས་དཀྲོལ་ནས་ནི། །དཀྱིལ་འཁོར་བཟང་པོ་བསྟན་པར་བྱ། །དེ་ནས་བླ་མ་དགའ་འགྱུར་ནས། །སློབ་མ་དེ་ལ་འདི་སྐད་བསྒོ། །དཀྱིལ་འཁོར་བཟང་པོ་འདི་ལ་ལྟོས། །ད་ནི་དད་པ་རབ་ཏུ་བསྐྱེད། །ཁྱོད་ནི་སངས་རྒྱས་རིགས་སུ་སྐྱེས། །སྔགས་དང་རིགས་པས་བྱིན་གྱིས་
བརླབས། །འགྲོ་བ་ཕུན་སུམ་དངོས་གྲུབ་ཀྱང་། །ཐམས་ཅད་ཁྱོད་ལ་མངོན་དུ་གྱུར། །དངོས་གྲུབ་ཐོབ་ཕྱིར་དམ་ཚིག་སྐྱོངས། །གསང་སྔགས་ལ་ཡང་བརྩོན་པར་གྱིས། །དེ་ནས་དཀྱིལ་འཁོར་གྱི་དེ་ཁོ་ན་བཤད་པར་བྱ་སྟེ། མི་མཉམ་མེད་པས་ཟུར་བཞིར་སྙོམས། །དྲན་དང་
དབང་པོ་སྒོ་བཞི་སྟེ། །བསམ་གཏན་རྟ་བབས་རྣམ་པ་བཞི། །ཏིང་འཛིན་ཁྱམས་སུ་ཤེས་པར་བྱ། །ཁྱམས་ལ་བརྟེན་པ་རྡོ་རྗེའི་གདན། །གཟུངས་རྣམས་ལྷ་མོས་མཆོད་པ་ཡི། །རྒྱན་རྣམས་ཀུན་གྱིས་བཀྲ་བ་ཡང་། །བསམ་པ་ཐམས་ཅད་རྫོགས་བྱེད་པའོ། །གདུལ་བྱ་
