ཅན་དོན་དུ་བསྒྲུབ་ལགས་ཀྱིས། །བྱིན་གྱིས་བརླབ་པར་མཛད་དུ་གསོལ། །དེ་ལྟར་བསོད་ནམས་ཀྱི་ཚོགས་བསགས་པས། །བྱམས་པ་ཆེན་པོ་དང་། སྙིང་རྗེ་དང་དགའ་བ་དང་། བཏང་སྙོམས་ཚད་མེད་པ་བཞི་རྗེས་སུ་དྲན་ནས། སེམས་བསྐྱེད་ཅིང་སྡོམ་པ་བཟུང་ནས། ད་ནི་ཡེ་ཤེས་ཀྱི་ཚོགས་ཡོངས་སུ་
དག་པའི་ཏིང་ངེ་འཛིན་བསྒོམ་པ་སྟོང་པ་ཉིད་ལ་བློ་གཏད་ལ། དེ་སྒྲུབ་པར་བྱེད་པའི་ཚིག་འདི་བརྗོད་དེ་སྒྲའི་དོན་ལ་མོས་པར་བྱའོ། །རྣམ་པར་མི་རྟོག་དབྱིངས་མི་ཟད། །ཡང་དག་བདག་མེད་དེ་བཞིན་ཉིད། །ཡང་དག་མཐའ་ལ་ཡི་གེ་མེད། །ཆོས་དབྱིངས་དམ་པ་ཟད་མི་ཤེས། །ཨོཾ་མ་ཧཱ་ཤུྰནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་
སྭ་བྷཱ་བ་ཨཱཏྨ་ཀོ྅ཧཾ། ཞེས་བརྗོད་པས་ཕྱི་ནང་གིས་བསྡུས་པའི་ཆོས་ཐམས་ཅད་སྟོང་ཞིང་བདག་མེད་པར་གྱུར་ཏེ། ཡེ་ཤེས་ཀྱིས་སྦྱངས་ནས། ཆོས་ཐམས་ཅད་ནམ་མཁའི་དཀྱིལ་ལྟ་བུར་རང་གི་སེམས་ཀྱི་སྣང་བ་ཤར་བ་ལས་ཨཱ་ལི་དང་། ཀཱ་ལིའི་ཕྲེང་བ་བསམས་ལ། ཨཱ་ལི་ལས་ནི་པདྨ་དཀར་པོའི་སྟེང་། ཀཱ་ལི་ལས་
ཆོས་ཀྱི་འབྱུང་གནས་ནམ་མཁའི་མཐའ་དང་མཉམ་པ། དེའི་ཐ་མ་འོད་ཟེར་གྱི་ཕྲེང་བ་ལྔས་བསྐོར་བའི་ནང་དུ་འོད་ཟེར་གྱི་ཕྲེང་བ་ལྔ་འཕྲོས་ཤིང་འདུས་པ་ལས། འོད་ལྗང་གུ་ལས་རླུང་གི་དཀྱིལ་འཁོར་རྡོ་རྗེ་རྒྱ་གྲམ་གྱིས་མཚན་པ། འོད་དམར་པོ་ལས་མེའི་དཀྱིལ་འཁོར་པདྨ་དམར་པོས་མཚན་པ། འོད་དཀར་པོ་
ལས་ཆུའི་དཀྱིལ་འཁོར་རིན་པོ་ཆེས་མཚན་པ། འོད་སེར་པོ་ལས་གསེར་གྱི་ས་གཞི་འཁོར་ལོས་མཚན་པ། འོད་སྔོན་པོ་ལས་རི་རབ་རྡོ་རྗེས་མཚན་པ། དེ་ཡང་ཤར་ཤེལ། ལྷོ་བཻ་ཌཱུརྱ། ནུབ་པདྨ་རཱ་ག །བྱང་གསེར་རྡོ་རྗེས་མཚན་པའོ། །དེའི་སྟེང་དུ་འོད་ཟེར་གྱི་ཕྲེང་བ་ལྔ་ཀ་རིལ་གྱིས་འདུས་པ་ལས། ལྷའི་
གཞལ་ཡས་ཁང་གྲུ་བཞི་པ་ལ་སྒོ་བཞི་པ་སྒོ་ཁྱུད་རིན་པོ་ཆེའི་པ་ཏྲས་མཛེས་པར་བརྒྱན་པ། ཕྱི་རོལ་ན་མུ་ཏིག་གི་ཟ་ར་ཚགས་དང་དྲིལ་བུ་གཡེར་ཁ་གྲྭ་བཞི་ནས་འཕྱང་བ། རིན་པོ་ཆེ་སྣ་ཚོགས་ཀྱི་ཆུན་འཕྱང་ཀྱང་རྟ་བབས་ཀྱི་ཀ་བ་ལ་འཕྱང་བ། རྟ་བབས་ཀྱང་ཏྭ་ར་ཎ་ལ་སོགས་པས་བརྒྱན་ཅིང་། རི་དགས་
དང་ཆོས་ཀྱི་འཁོར་ལོ་དང་། འཕན་དང་རྒྱལ་མཚན་དང་། གདུགས་དང་རིན་པོ་ཆེ་འོད་དུ་འབར་བའི་ཏོག་གིས་མཛེས་པར་བྱས་ཏེ། གཞལ་ཡས་ཁང་གི་ནང་དུ་སེང་གེ་དང་གླང་པོ་ཆེ་དང་། རྟ་མཆོག་དང་། རྨ་བྱ་དང་། ནམ་མཁའ་ལྡིང་གི་སྟེང་དུ་པདྨ་དང་ཟླ་བ་དང་། ཉི་མའི་གདན་མཛེས་པར་བཀྲམ་པ་
