ཨ་མ་ལ་སུ་པ་རི་ཤུདྡྷ་དྷརྨྨ་དྷཱ་ཏུ་ཛྙཱ་ན་གརྦྷ་ཨཱཿ། ཞེས་ནི་བརྗོད་པའི་སྔགས་བཟླས་སོ། །དང་པོའི་སངས་རྒྱས་ང་རྒྱལ་དང་། །ལྡན་པས་དཀྱིལ་འཁོར་བྲི་བའི་ཕྱིར། །ཚིགས་བཅད་འདི་ཡིས་བསྟོད་པར་བྱ། །སངས་རྒྱས་ཀུན་ལ་བསྟོད་བྱ་བ། །རང་ལ་བསྟོད་དེ་ངེས་པའི་མཆོག །འབྲས་
བུ་འབྱིན་པའི་སངས་རྒྱས་ཀྱིས། །ཤི་ལ་ཕན་གདགས་དོན་ཆེད་དུ། །མཆོག་སྦྱིན་རྡོ་རྗེ་མཆོག་ཁྱོད་འདུད། །ཡང་དག་མཐའ་གྱུར་ཁྱོད་ལ་འདུད། །སྟོང་ཉིད་ལས་བྱུང་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་བྱང་ཆུབ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་ཆགས་པ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་
འདོད་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་དགྱེས་པ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་རོལ་ལ་ཕྱག་འཚལ་འདུད། །སངས་རྒྱས་འཛུམ་པ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་བཞད་པ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་གསུང་ཉིད་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་ཐུགས་ལ་ཕྱག་འཚལ་འདུད། །མེད་པ་ལས་
བྱུང་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་འབྱུང་བ་ཁྱོད་ལ་འདུད། །ནམ་མཁའ་ལས་བྱུང་ཁྱོད་ལ་འདུད། །ཡེ་ཤེས་ལས་བྱུང་ཁྱོད་ལ་འདུད། །སྒྱུ་འཕྲུལ་དྲ་བ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་རོལ་སྟོན་ཁྱོད་ལ་འདུད། །ཐམས་ཅད་ཐམས་ཅད་ཁྱོད་ལ་འདུད། །ཡེ་ཤེས་སྐུ་ཉིད་ཁྱོད་ལ་
འདུད། །དེ་ནས་གསོལ་བ་བཏབ་བྱས་ནས། །ཕྱོགས་བཅུའི་སངས་རྒྱས་ནམ་མཁའ་ལ། །མངོན་སུམ་འདྲ་བར་གང་བར་བསྒོམ། །དེ་ནས་གང་ཤི་དེ་ཉིད་ལ། །དམིགས་ལ་ཨ་ཨཱ་ཞེས་བཀོད་དེ། །དེ་ལ་ཕྱོགས་བཅུ་རྒྱལ་བ་དང་། །དང་པོའི་སངས་རྒྱས་ལ་སོགས་པ། །ནམ་
མཁའ་གང་བར་འདིར་གཤེགས་པ། །དེང་འདིར་འདུས་པར་བདེ་གཤེགས་ཏེ། །དེང་བདག་འདུས་པའི་མཆོག་ཡིན་ནོ། །དེ་ནས་རྣམ་རྒྱལ་ཆུས་བཀང་བའི། །བུམ་པས་བཀྲུས་ཏེ་བགེགས་བསྐྲད་ལ། །དེ་ནས་དེ་ལ་བསྲུང་བར་བྱ། །དེ་ནས་ཏིལ་གྱིས་དེ་ལ་ནི། །བསྐོར་ཏེ་སྦྱིན་སྲེག་
དབུལ་བར་བྱ། །ཤི་བ་སྟ་གོན་ལ་གནས་པའི་ཏིང་ངེ་འཛིན་ཏོ། །དེ་ནས་སྔོན་གྱི་སྦྱོར་བ་ཡིས། །ཆོ་ག་རྣམས་ནི་སྤྱད་པར་བྱ། །དང་པོའི་ཆོ་ག་རིམ་པ་ནས། །འཁོར་ལོའི་རིམ་པ་འདི་གསུངས་ཏེ། །ཕྱི་རོལ་བྱུང་སྟེ་གཏོར་མ་སྦྱིན། །དེ་ནས་ཁྲུས་བྱས་མཆོད་པའི་ཚོགས། །མ་
