མོས་སྨྲས་པ། གཞོན་ནུ་མ་རྒྱལ་བུ་འདིས་ཁྱོད་དོན་དུ་གཉེར་གྱིས། སྟ་གོན་གྱིས་ཤིག་ཅེས་ཟེར་ནས། དགེ་སློང་དོན་ཉུང་ངུ་དག་ཅིའི་ཕྱིར་དགེ་སློང་དག །གཅེར་བུ་པ་དང་གཅེར་བུ་མ་ཀུན་དུ་རྒྱུ་དང་། ཀུན་དུ་རྒྱུ་མ་ལ་རང་གི་ལག་ནས་བཅའ་བ་དང་། བཟའ་བ་
སྦྱིན་པར་བྱེད་ཅེས་སྨོད་པར་བྱེད། ཕྱར་ཀ་གཏོང་བར་བྱེད། ཁ་ཟེར་བར་བྱེད་པའི་སྐབས་དེ། དགེ་སློང་རྣམས་ཀྱིས་བཅོམ་ལྡན་འདས་ལ་གསོལ་ཏོ། །དེ་ནས་བཅོམ་ལྡན་འདས་ཀྱིས་སྔ་མ་བཞིན་དུ། ཡང་དགེ་སློང་གང་གཅེར་བུ་པའམ། གཅེར་བུ་མ་ཀུན་དུ་རྒྱུའམ། ཀུན་དུ་རྒྱུ་
མོ་ལ། རང་གི་ལག་ནས་བཅའ་བ་དང་བཟའ་བ་སྦྱིན་པར་བྱེད་ན་ལྟུང་བྱེད་དོ་ཞེས་བཅའ་བ་མཛད་དེ། ཡུལ་ནི་མཉན་ཡོད། གང་ཟག་ནི་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ། །ཉོན་མོངས་པ་ནི་མ་བརྟགས་པར་སྤྱོད་པ། ཉེས་པ་ནི་གཅེར་བུ་མ་ལ་རང་གི་ལག་ནས་ཟན་སྦྱིན་པའོ། །དེ་ལ་ཡང་
དགེ་སློང་གང་ཞེས་བྱ་བ་ནི་ཆོས་འདི་པའོ། །གཅེར་བུ་པའམ་གཅེར་བུ་མ་ཀུན་དུ་རྒྱུའམ་ཀུན་དུ་རྒྱུ་མོ་ལ་ཞེས་བྱ་བ་ནི། རྟགས་འདི་དང་རབ་ཏུ་བྱུང་བ་འདི་མེད་པ་ལའོ། །རང་གི་ལག་ནས་ཞེས་བྱ་བ་ནི། ལག་པས་ལག་པར་རོ། །བཅའ་བ་དང་བཟའ་བ་ཞེས་བྱ་བ་ནི། བཅའ་བ་ལྔ་པོ་དང་
བཟའ་བ་ལྔ་པོ་དག་གོ། །སྦྱིན་ན་ཞེས་བྱ་བ་ནི་བསམ་མི་དགོས་པར་རོ། །ལྟུང་བྱེད་དོ་ཞེས་བྱ་བ་ནི་སྔ་མ་བཞིན་ནོ། །དེ་ལ་འདིར་དངོས་གཞིའི་ལྟུང་བ་རྣམ་པར་གཞག་པ་ནི། དགེ་སློང་གིས་གཞན་མུ་སྟེགས་ཅན་གྱི་རབ་ཏུ་བྱུང་བ་གཅེར་བུ་ཕོ་ཡང་རུང་། མོ་ཡང་རུང་བ། རྟགས་འདི་དང་རབ་ཏུ་
བྱུང་བ་འདི་མེད་པ། ཉེ་དུ་རབ་ཏུ་བྱུང་བ་ལ་ལྟོས་པ་དང་། ནད་པ་དང་། སྡིག་པ་ཅན་གྱི་ལྟ་བའི་རྣམ་པ་དང་། བྲལ་བ་ཉིད་མ་གཏོགས་པ་ལ་རང་གི་ལག་ནས་བཅའ་བ་དང་། བཟའ་བ་ཟས་དེ་ལ་རུང་བ་དག་དེའི་ལུས་སམ། ལུས་དང་འབྲེལ་པར་སྦྱིན་ན། གཅེར་བུ་པ་ལ་སྦྱིན་པའི་ལྟུང་བྱེད་ཀྱི་
དངོས་གཞིར་འགྱུར་རོ། །འདིའི་བྱེད་པ་པོ་ནི། གཞན་མུ་སྟེགས་ཅན་གྱི་རབ་ཏུ་བྱུང་བ་དང་། རང་གིས་སྦྱིན་པ་དང་། ཟས་རུང་བ་དག་གོ། །འདི་ལ་ཡན་ལག་ལྔས་དངོས་གཞིའི་ལྟུང་བར་འགྱུར་ཏེ། གཞན་མུ་སྟེགས་ཅན་གྱི་རབ་ཏུ་བྱུང་བ་གཅེར་བུ་ཕོ་ཡང་རུང་། མོ་ཡང་རུང་བ་རྟགས་འདི་དང་
