སྤྱོད་ཡུལ་གྱི་སྤྱོད་ལམ་མམ། རིགས་ཀྱི་བུད་མེད་དམ། རིགས་ཀྱི་གཞོན་ནུ་མ་དག་ལ་ཀུན་དུ་སྨྲས་བ་འམ། ནོམ་པའམ། ཉུག་པ་བྱས་པར་མ་གྱུར་ཏམ་སྙམ་ནས། རེ་ལྡེ་རི་པའི་བྲམ་ཟེ་དང་། ཁྱིམ་བདག་རྣམས་ཀྱི་དཔོང་སའི་གནས་ན། བྲམ་ཟེ་དང་ཁྱིམ་བདག་ལྔ་བརྒྱ་ཙམ་བྱ་བ་
ཞིག་གི་ཕྱིར་འདུས་ཏེ་མཐུན་པར་གྱུར་པ་ལ་དྲིས་ན། དེ་རྣམས་དབང་པོ་ཕྱིར་བླན་ཏེ་འཁོད་དོ། །དེའི་ཚེ་ན་དགེ་བསྙེན་ཆུ་རྒྱུན་ཅན་འཁོར་དེ་ཉིད་དུ་འདུག་ཅིང་མཐུན་པར་གྱུར་པས། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོའི་ལག་པ་གཉིས་ནས་བཟུང་སྟེ། མཐའ་གཅིག་ཏུ་ཁྲིད་ནས། ཇི་ལྟར་གྱུར་བ་དེ་བརྗོད་
དེ། དོན་འདི་བཅོམ་ལྡན་འདས་ལ་རྒྱས་པར་གསོལ་ཅིག་ཅེས་སྨྲས་ནས། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་རང་གི་སྡུམ་པར་བསོད་སྙོམས་གསོལ་ཏེ། ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོས་ཀྱང་དེ་ལ་ཆོས་བསྟན་ནས། མཉན་ཡོད་དུ་སོང་སྟེ། ཇི་ལྟར་གྱུར་པ་བཅོམ་ལྡན་འདས་ལ་གསོལ་པ་དང་། བཅོམ་ལྡན་འདས་
ཀྱིས་བཀའ་སྩལ་པ། ཀུན་དགའ་བོ་ཁྱེད་དགེ་སློང་གནས་བརྟན་གནས་བརྟན་རྣམས་དང་། ལྷན་ཅིག་རེ་ལྡེ་རི་ལ་སོང་ལ། ལམ་གྱི་བར་དུ་འདུག་སྟེ། དགེ་སློང་འགྲོ་མགྱོགས་དང་། ནབ་སོ་གཉིས་བསྐྲད་པའི་གླེང་བ་པོ་དགེ་སློང་ཆོས་ལྔ་དང་ལྡན་པ་སྤྲོ་བར་བྱས་སྟེ། གསོལ་བ་དང་གཉིས་ཀྱི་ལས་ཀྱིས་
བསྒོ་བར་བྱ་ཞིང་། དེས་གླེངས་ལ། ལྟུང་བ་མ་མཐོང་ངོ་ཞེས་ཟེར་ན་གསོལ་བ་དང་། བཞིའི་ལས་ཀྱིས་དེ་གཉིས་སྐྲོད་ཅིག །ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོས་བཅོམ་ལྡན་འདས་ཀྱི་བཀའ་མཉན་ནས། དེ་ལྟར་བྱ་བར་བརྩམས་པ་དང་། རེ་ལྡེ་རི་ལ་གནས་པའི་དགེ་སློང་དམར་སེར་ཅན་དག་གིས་གཅམ་
ཚོར་ནས། བདག་ཅག་ཀྱང་དགེ་སློང་འགྲོ་མགྱོགས་དང་། ནབ་སོ་གཉིས་ཀུན་དུ་སྤྱོད་པ་མཐུན་པས་བསྐྲད་པར་འགྱུར་རོ་སྙམ་ནས། སྒོ་གཅིག་ཏུ་བྱུང་སྟེ། མཉན་ཡོད་དུ་བྲོས་ནས་དགེ་འདུན་གྱི་ནང་དུ་ལྟུང་བ་བཤགས་ཏེ་ཚུལ་བཞིན་དུ་འཁོད་དོ། །དེ་ནས་ཚེ་དང་ལྡན་པ་ཀུན་དགའ་བོ་དང་། དགེ་སློང་
གནས་བརྟན་གནས་བརྟན་རྣམས་ཀྱིས་དགེ་སློང་འགྲོ་མགྱོགས་དང་། ནབ་སོ་གཉིས་ལ་བསྐྲད་པའི་ལས་བྱས་ནས། མཉན་ཡོད་དུ་དོང་བ་དང་། འགྲོ་མགྱོགས་དང་། <བན་,ནབ་>སོ་གཉིས་ཀྱང་མཐར་གྱིས་མཉན་ཡོད་དུ་ཕྱིན་ཏེ་བལྟས་ན། དགེ་སློང་དམར་སེར་ཅན་དག་མ་བསྐྲད་པར་ཚུལ་བཞིན་དུ་འཁོད་པ་
