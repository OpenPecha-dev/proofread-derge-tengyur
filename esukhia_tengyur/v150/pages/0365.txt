ཁྱོད་དང་ལྷན་ཅིག་སྨྲའམ་འདུག་པ་བདེ་བ་མ་ཡིན་གྱི། འདི་ལྟར་ཁོ་བོ་གཅིག་པུ་སྨྲའམ་འདུག་ན་བདེས་ཁྱོད་སོང་ཤིག་ཅེས་ཟེར་ན་ཞེས་བྱ་བ་ནི་དེ་སྐད་ཟེར་ནའོ། །ལྟུང་བྱེད་དོ་ཞེས་བྱ་བ་ནི་སེམས་ཅན་དམྱལ་བ་དང་། དུད་འགྲོ་དང་། ཡི་དགས་ངན་སོང་དང་། ངན་
འགྲོ་དང་། ལོག་པར་ལྟུང་བ་དག་ཏུ་སྲེག་པར་བྱེད། འཚོད་པར་བྱེད། ལྟུང་བར་བྱེད་པའོ། །གཞན་ཡང་ཇི་སྲིད་དུ་ལྟུང་བ་དེ་མ་བཤགས་ཤིང་སོ་སོར་མ་བཤགས་ན་དགེ་བའི་ཆོས་རྣམས་ལ་སྒྲིབ་པར་བྱེད་པ་སྟེ། དེས་ན་ལྟུང་བྱེད་དོ་ཞེས་གསུངས་སོ། །དེ་ལ་འདིར་དངོས་གཞིའི་
ལྟུང་བ་རྣམ་པར་གཞག་པ་ནི། དགེ་སློང་དག་གིས་དགེ་སློང་ཕ་རོལ་པའི་ནད་ཀྱི་གཉེན་པོའི་དོན་ཉིད་མ་གཏོགས་པར་ཆེད་དུ་བསམས་ཏེ། ཁྱིམ་པ་སྦྱིན་པ་པོའི་ཚིག་གིས་བསྒོ་བས་ཟན་གཅོད་དུ་བཅུག་ནས། ཟན་གཅོད་དུ་འཇུག་པའི་ལྟུང་བྱེད་ཀྱི་དངོས་གཞིར་འགྱུར་རོ། །འདིའི་
བྱེད་པ་པོ་ནི། ཟན་གཅད་པར་བྱ་བ་དགེ་སློང་ཉིད་དག །སྦྱིན་བདག་ཁྱིམ་པ་དང་། ཟས་ཉིད་དོ། །འདི་ལ་ཡན་ལག་ལྔས་དངོས་གཞིའི་ལྟུང་བར་འགྱུར་ཏེ། ཆེད་དུ་བསམས་པ་དང་། ཟན་གཅད་པར་བྱ་བ་དགེ་སློང་ཉིད་དང་། ནད་ཀྱི་གཉེན་པོའི་དོན་ཉིད་མ་གཏོགས་པ་དང་། སྦྱིན་
པ་པོ་ཁྱིམ་པ་དང་། ཚིག་གིས་བསྒོ་བ་དང་ཟན་གཅོད་དུ་འཇུག་པའོ། །དེ་དག་ལས་ཡན་ལག་མ་ཚང་ན་ཅི་རིགས་པར་ན་སྨད་དོ། །ཟན་གཅོད་དུ་འཇུག་པའི་ལྟུང་བྱེད་དེ་ལྔ་བཅུ་གཅིག་པའོ།། །།མེ་ལ་རེག་པའི་ལྟུང་བྱེད་ཀྱི་གླེང་གཞི་ནི། སངས་རྒྱས་བཅོམ་ལྡན་འདས་མ་ག་དྷཱ་དག་ཏུ་
ལྗོངས་རྒྱུ་ཞིང་གཤེགས་པ་ན་རྒྱལ་པོའི་ཁབ་ཏུ་བྱོན་ནས། རྒྱལ་པོའི་ཁབ་ན་འོད་མའི་ཚལ་བྱ་ཀ་ལན་ད་ཀའི་གནས་ན་བཞུགས་སོ། །དེའི་ཚེ་ན་དྲུག་སྡེ་དག །མེས་གཞན་ལ་གཏོང་བས་རྩེ་བར་བྱེད་ཅིང་ཟླ་བ་ཡང་བྱེད། ཉི་མ་ཡང་བྱེད། མགལ་མེའི་འཁོར་ལོ་ཡང་བྱེད་ནས་མུ་སྟེགས་
ཅན་དག་ཤེས་ལན་དག །ཤཱཀྱའི་བུའི་དགེ་སྦྱོང་འདི་དག་མེས་གཞན་ལ་གཏོང་བས་རྩེ་བར་བྱེད། ཟླ་བ་ཡང་བྱེད་ཉི་མ་ཡང་བྱེད། མགལ་མེའི་འཁོར་ལོ་ཡང་བྱེད་དེ། བྱིས་པ་ཁྱེའུ་དག་ཀྱང་དེ་ལྟར་བྱེད་ན། ཤེས་ལྡན་དག་བྱིས་པ་ཁྱེའུ་དག་དང་། ཤཱཀྱའི་བུའི་དགེ་སྦྱོང་འདི་དག་ལ་བྱེ་
