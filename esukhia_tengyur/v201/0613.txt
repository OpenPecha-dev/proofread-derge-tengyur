སྨན་རྣམ་གཉིས་ཟི་ར་དང་། །བྱི་དང་ག་དང་ཟུར་པ་མར། །ཚ་དང་སྐྱུར་རྩི་ཆེན་པོ་ཡི། །ཁུ་བ་དང་ནི་སྒེའུ་གཤེར་དང་། །ལ་ཕུག་སྐམ་པོ་རྒྱ་ཤུག་སྐྱུར། །སེའུ་འབྲུ་ཙུག་ཀྲི་ཀའི་ཁུ་བ། །དར་བ་ཞོ་དང་ཆང་དྭངས་དང་། །རྩ་བ་མོ་དངས་མ་རྩབ་མོ་དང་། །
སྐྱུར་རྩ་བས་དང་ནི་དེ་བསྐོལ་བ། །མེ་དྲོད་སྐྱེད་པའི་དམ་པ་ཡིན། །གཟེར་དང་སྐྲན་དང་དམུ་རྫིང་དང་། །ལུད་པ་དབུགས་མི་བདེ་བ་དང་། །རླུང་དང་བད་ཀན་འཇོམས་པར་བྱེད། །ཡང་ན་སྐྱུར་རྩི་ཆེན་པོ་ཡི། །ཁུ་བར་བསྐོལ་བའི་མར་བླུད་བྱ། །ཞེས་བྱ་བ་ལ། རྩ་བ་ལྔ་ལ་
སོགས་པའི་སྨན་ལྡེ་གུར་བྱས་པ་ཚ་ལ་སོགས་པའི་ཁུ་བའི་རྫས་དང་ལྷན་ཅིག་མཉམ་པོར་བྱས་ཏེ་མར་བསྐོལ་བར་བྱའོ། །དེ་ནས་གལ་ཏེ་བ་མར་བྲེ་གང་བསྐོལ་བ་ཚ་དང་སྐྱུར་རྩི་ཆེན་པོའི་ཁུ་བ་དང་། སྒེའུ་གཤེར་གྱི་ཁུ་བ་དང་། རྒྱ་ཤུག་གི་ཁུ་བ་དང་། ལ་ཕུག་སྐམ་པོ་བསྐོལ་བའི་
ཁུ་བ་དང་། སེའུ་འབྲུའི་ཁུ་བ་དང་། ཙུག་ཀྲི་ཀ་ཞེས་བྱ་བ་ལ་ཨམ་ལི་ཙང་གི་རིའི་ཁུ་བ་དང་དར་བ་དང་། ཞོ་དང་ཆང་གི་དྭངས་མ་མ་ཡིན་པ་དང་། སྐྱུར་རྩབས་ཞེས་བྱ་བ་འབྲས་ཁུའི་རང་སྐྱུར་དག་བྲེ་རེ་རེ་དང་སྦྱར་ལ་མར་དེ་བསྐོལ་བ་ནི་མེ་དྲོད་སྐྱེད་པར་བྱེད་པའི་དམ་པ་ཡིན་ཞིང་
གཟེར་བ་ལ་སོགས་པ་འཇོམས་པར་བྱེད་པ་ཡིན་ནོ། །ཡང་ན་མར་དེ་དང་སྐྱུར་རྩི་ཆེན་པོའི་ཁུ་བ་འབའ་ཞིག་བྱིན་ན་ཡང་ཡོན་ཏན་གོང་མ་དང་འདྲའོ། །འདི་ཡི་ཏིལ་མར་བསྐོལ་བ་ཡིས། །བསྐུ་བྱུག་བྱས་ན་རླུང་སེལ་ཏོ། །ཞེས་བྱ་བ་ལ། ཡང་ན་རྩ་བ་ལྔ་ལ་སོགས་པའི་
མར་དང་བསྐོལ་བའི་སྨན་དང་ཏིལ་མར་བསྐོལ་ལ་བསྐུ་བྱུག་བྱས་ན་ཡང་རླུང་སེལ་བར་བྱེད་དོ། །རླུང་དང་བད་ཀན་མ་འདྲེས་ཞུའམ། །བད་ཀན་རླུང་ཤས་ཆེ་བ་ལ། །སྨན་ནི་འདི་དག་ཉིད་ཀྱི་ཕྱེ།། །ཆུ་དྲོན་སྦྱར་བ་བླུད་པར་བྱ། །ཞེས་བྱ་བ་ལ། རླུང་ཤས་ཆེ་
བའི་བད་ཀན་ལ་སོགས་པའི་ནད་ལ་སྨན་འདི་དག་ཉིད་ཀྱི་ཕྱེ་མ་དང་བསྐོལ་བའི་ཆུ་དྲོན་བླུད་པར་བྱའོ། །མཁྲིས་པས་མེ་དྲོད་ཉམས་གྱུར་པ། །བཀྲུ་སྨན་དག་གམ་སློན་སྨན་གཏང་། །ཁ་དང་ཡང་དང་འཁྲུ་གཅོད་དང་། །ཚ་མི་སྐྱེད་པས་དྲོད་བསྐྱེད་བྱ། །ཕྱེ་མ་
