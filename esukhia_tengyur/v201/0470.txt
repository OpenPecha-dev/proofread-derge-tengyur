ཆུ་སྲང་ལྔ་བརྒྱ་རྩ་བཅུ་གཉིས་ཀྱི་ནང་དུ་བསྐོལ་ཏེ་དེའི་བཞི་ཆ་ལུས་པ་དང་སྨན་ཁུ་དེ་དང་ཆ་མཉམ་པའི་འོ་མ་དང་། མར་སྲང་བརྒྱ་ཉི་ཤུ་རྩ་བརྒྱད་བཅུག་ལ་བསྐོལ་བར་བྱའོ། །དེའི་ནང་དུ་བ་ལ་དང་། ཨ་ཏི་བ་ལ་ལ་སོགས་པ་སྲང་ཕྱེད་ཕྱེད་བཅུག་ལ་དེ་བཞིན་དུ་བསྐོལ་བར་བྱའོ། །དེ་ལྟ་
བུར་བསྐོལ་བའི་ནཱ་ག་པ་ལའི་སྨན་མར་འདིས་ནི་ཁྲག་མཁྲིས་ལ་སོགས་པ་སེལ་བར་བྱེད་པ་དང་། སྟོབས་སྐྱེད་པ་ནས་རྒས་སྲ་བའི་བར་གྱི་ཡོན་ཏན་དང་ལྡན་པ་ཡིན་ནོ། །ཆོ་ག་འདིས་ནི་དྲོད་སྐྱེད་ཅིང་། །མེ་དྲོད་བྲི་བ་སྐྱེད་འཇུ་བྱེད། །གཅོང་ཆེན་བཤད་པའང་གློ་རྡོལ་བསྔགས། །
འཁྲུ་ན་གཅོད་པར་བྱེད་པར་ཕན། །ཞེས་བྱ་བ་ལ། བཤད་མ་ཐག་པའི་སྨན་མར་གྱི་ཆོ་གའི་སྦྱོར་བ་འདིས་ནི་གློ་རྡོལ་དང་ཟད་བྱེད་ཀྱི་ནད་ཀྱིས་བཏབ་པའི་སྟོབས་བསྐྱེད་པས་མེ་དྲོད་སྐྱེད་པར་བྱེད་ཅིང་འབྱུང་བ་དང་། ཕོ་བའི་མེ་དྲོད་འབྲི་བར་བྱེད་པ་དག་དྲོད་སྐྱེད་ཅིང་ཟས་འཇུ་བར་
བྱེད་པ་ཡིན་ནོ། །གཞན་ཡང་གཅོང་ཆེན་པོ་གསོ་བའི་སྐབས་སུ་བཤད་པའི་སྦྱོར་བ་དེ་ཡང་གློ་རྡོལ་བའི་ནད་ལ་བསྔགས་པ་ཡིན་ནོ། །གང་གི་ཕྱིར་གློ་རྡོལ་བའི་མི་དེ་འཁྲུ་བར་གྱུར་ན། གཅོད་པའི་སྦྱོར་བ་གང་ཡིན་པ་དེ་བྱས་ན་བསྔགས་སོ། །རྩ་བ་བཅུ་པ་ཁ་སྲལ་ཕྲེའུ། །
ཤང་ཀ་པུཥྤ་ཟུར་པ་དང་། །བ་ལ་ཧསྟི་པི་པི་ལིང་། །ཨ་པ་མརྒ་ཀྲུ་ཏྲུག་ཏྲེས། །ག་བྲ་པི་པི་ལིང་རྩ་བ། །མ་ནུ་སྲང་ནི་གཉིས་གཉིས་དང་། །ནས་སྲང་དྲུག་ཅུ་རྩ་བཞི་དག །ཨ་རུ་ར་ནི་སྲང་བརྒྱ་དག །ཆུ་སྲང་དྲུག་བརྒྱ་བཞི་བཅུས་བསྐོལ། །ནས་ཚོས་པ་དང་སྨན་ཁུ་དེ། །
བཙགས་ལ་ཨ་རུ་ར་བརྒྱ་དེར། །བུ་རམ་སྲང་བརྒྱ་བཅུག་ལ་བསྐོལ། །མར་དང་ཏིལ་མར་པི་པི་ལིང་། །ཕྱེ་མ་སྲང་ནི་བརྒྱད་པོ་དག །བསྐོལ་ཏེ་བསྒྲངས་ནས་སྦྲང་རྩིར་གཞུག །རྟག་ཏུ་ཨ་རུ་ར་གཉིས་བཅས། །ལྡེ་གུ་བཅུད་ལེན་འདི་ཟོས་ན། །དེས་ནི་སྐྲ་དཀར་གཉེར་
མ་མེད། །ཚེ་རིང་མདོག་སྟོབས་འཕེལ་བར་བྱེད། །ལུད་ལྔ་ཟད་བྱེད་དབུགས་མི་བདེ། །སྐྱིགས་བུར་བཅས་པའི་མི་བཟད་རིམས། །གཅིན་སྙི་བ་ཡི་ནད་དང་ནི། །གཞང་འབྲུམ་སྐྲན་དང་སྙིང་ནད་དང་། །ཆམ་པ་ཡི་ག་འཆུས་པ་སེལ། །བཅུད་ཀྱིས་ལེན་མཆོག་དཔལ་ཡོན་
