འགྱུར་བར་བྱེད་དེ། རླུང་ནད་དེ་ནི་ཕྱིར་དགྱེད་པ་ཞེས་བྱའོ། །ཞེས་བཤད་པ་ལ། ལ་ལ་ན་རེ་ནི་གཞུངས་རེངས་ཞེས་བྱའོ་ཞེས་ཟེར་རོ། །གཞན་དག་ནི་དེ་ལ་ཤུགས་དྲག་པོ་ཞེས་བྱའོ་ཞེས་ཀྱང་བཤད་དོ། །གནད་ལ་གནས་པའི་རྨར་ཞུགས་ཏེ། །རླུང་གིས་བསྐྱོད་པར་གྱུར་པ་དང་། །རྐང་པ་
ནས་ནི་མགོ་བོའི་བར། །ལུས་ལ་ནད་ཀྱིས་ཁྱབ་བྱེད་པ། །སྐོམ་ཞིང་ལུས་ནི་སྐྱ་གྱུར་པ། །རྨ་ཡི་ད་རྒན་དེ་སྤང་བྱ། །ཞེས་བྱ་བ་ལ། གནད་ལ་གནས་ཤིང་བརྟེན་པའི་རྨར་ཞུགས་པའི་རླུང་ལ་སོགས་པའི་ནད་དེ་རླུང་གིས་བསྐྱོད་ཅིང་བསྣན་པར་གྱུར་ན། རྐང་པ་ནས་མགོ་བོའི་བར་གྱི་ལུས་
ཐམས་ཅད་ལ་ནད་ཀྱིས་ཁྱབ་ཅིང་ཟིལ་གྱིས་གནོན་པར་བྱེད་དེ། རྨའི་ད་རྒན་དེ་ལྟ་བུ་དེ་སྐོམ་དད་ཆེ་ཞིང་ལུས་སྐྱ་བ་ལས་བྱུང་བ་དེ་ནི་གསོ་དཀའ་བ་ཡིན་པས་སྤང་ཞིང་དོར་བར་བྱའོ། །ལུས་ལ་འཇུག་པའི་རླུང་ཀུན་ནི། །ཤུགས་དང་བྲལ་ནས་བདེ་བར་འགྱུར། །ཞེས་བྱ་བ་ལ། སྔར་
བཤད་པའི་རླུང་ནད་ལུས་ལ་འཇུག་པ་ལ་སོགས་པ་དེ་རྣམས་ལ་རླུང་གི་ཤུགས་དང་བྲལ་ཞིང་། ཤུགས་ཞི་བར་གྱུར་ན་དེ་ནི་བདེ་ཞིང་ནད་མེད་པར་འགྱུར་བ་ཡིན་ནོ། །ལྕེ་ནི་བྲུད་དྲགས་སྐམ་ཟོས་གནོད། །འགྲམ་པའི་རྩ་བར་གནས་རླུང་འཁྲུགས། །འགྲམ་པ་གཉིས་ནི་བསྐྱོད་བྱས་ནས། །ཁ་
ནི་གདངས་པར་བྱེད་པའམ། །ཡང་ན་འཛུམས་པར་བྱེད་པར་འགྱུར། །འགྲམ་པ་ཉམས་པར་བྱེད་པ་དེས། །ཟ་དང་སྨྲ་བ་དཀའ་བར་འགྱུར། །ཞེས་བྱ་བ་ལ། ལྕེ་བྲུད་དྲགས་ཤིང་སོ་ཤིང་ལ་སོགས་པས་གནོད་པར་བྱས་པ་དང་། ཟས་སྐམ་པོ་ཟོས་ཤིང་བཅས་པ་དང་། ཁུ་ཚུར་དང་རྡོ་
ལ་སོགས་པས་སྣད་པས་འགྲམ་པའི་རྩ་བར་གནས་པའི་རླུང་འཁྲུགས་ཤིང་ལྡང་བར་གྱུར་ནས། འགྲམ་པ་གཉིས་བསྐྱོད་ཅིང་རང་གི་གནས་ནས་འགུལ་བར་བྱས་ནས། ཁ་རྟག་ཏུ་གདངས་པར་བྱེད་པ་དང་། ཡང་ན་ཁ་འཛུམས་ཤིང་འབྱེད་མི་ཤེས་པར་བྱེད་པར་འགྱུར་ཏེ། ནད་དེ་
ནི་འགྲམ་པ་ལ་གནས་ནས་ཉམས་པར་བྱེད་པ་ཞེས་བྱ་སྟེ། འགྲམ་པ་བསྐྱོད་ཅིང་ཉམས་པར་བྱེད་པ་དེས་ཟས་ཟབ་དང་ཚིག་ཏུ་སྨྲ་བ་ཤིན་ཏུ་དཀའ་བར་བྱེད་པར་འགྱུར་རོ། །སྨྲ་བར་བྱེད་པའི་རྩ་གནས་རླུང་། །ལྕེ་ནི་ལྡིབ་པར་བྱེད་འགྱུར་ཞིང་། །ལྕེ་ལྡིབས་དེས་ནི་བཟའ་བཏུང་དང་། །སྨྲ་བ་ལ་སོགས་མི་ནུས་
