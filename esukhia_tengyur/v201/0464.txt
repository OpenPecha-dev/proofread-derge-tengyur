དེ་ནང་ཞུན་མར་བྲེ་གང་བསྐོལ། །འོ་མ་ར་ཤ་སྐྱུ་རུ་ར། །བི་ད་རི་དང་བུ་རམ་ཤིང་། །ཁུ་བ་ར་ཤའི་ཁུ་བར་བཅས། །བསྐོལ་ཏེ་བསྒྲངས་པར་སྦྲང་རྩི་ནི། །བྲེ་ཕྱེད་ཁ་ར་སྲང་ལྔ་བཅུ། །ན་ལེ་ཤམ་དང་སུག་སྨེལ་དང་། །ཤིང་ཚ་གན་དྷ་བྷ་དྲ་དང་། །ནཱ་ག་གེ་སར་སྲང་ཕྱེད་ཕྱེད། །
ཕྱེ་མར་བཏགས་ཏེ་བཅུག་ནས་ནི། །འཇུ་སྟོབས་རྗེས་མཐུན་ཚད་དུ་བླུད། །འདི་ནི་བདུད་རྩིའི་སྨན་མར་ཏེ། །མི་རྣམས་ཀྱི་ནི་བདུད་རྩི་ཡིན། །བདུད་རྩི་འདི་ནི་ཟོས་ནས་སུ། །འོ་མ་ཤ་ཁུ་ལྡན་ཟོས་ན། །ཁུ་ཆུ་ཉམས་དང་གློ་འགྲམས་དང་། །ཉམས་སྟོབས་ཟད་པའི་
གསོ་བ་ཡིན། །བུད་མེད་ལ་གདུ་ལུས་སྐེམ་དང་། །མདོག་སྐད་ཉམས་ལ་གཏང་བར་བྱ། །ལུད་པ་དང་ནི་སྐྱིགས་བུ་རིམས། །དབུགས་མི་བདེ་དང་ལུས་ཚ་དང་། །སྐོམ་དད་ཁྲག་མཁྲིས་སེལ་བྱེད་ཅིང་། །བུ་སྐྱེད་བྱེད་ཅིང་སློན་དང་བརྒྱལ། །མངལ་ནད་གཅིན་ནད་
འཇོམས་བྱེད་ཡིན། །ཞེས་བྱ་བ་ལ། འཚོ་བྱེད་ལ་སོགས་པའི་སྨན་ཞོ་གཉིས་གཉིས་བཏགས་པའི་ལྡེ་གུ་དང་། མར་བྲེ་གང་སྟེ་སྲང་སུམ་ཅུ་རྩ་གཉིས་བསྐོལ་བར་བྱ་སྟེ། འཚོ་བྱེད་ཀྱི་སྡེ་ཚན་དུ་གཏོགས་པ། རྩ་བ་ལྔ་ཆུང་ངུ་སྨན་སྣ་རེ་རེ་ཞིང་ཞོ་གཉིས་གཉིས་གཞུག་
པར་བྱ་ལ། རྒུན་འབྲུ་ལ་སོགས་པའི་སྨན་གཉིས་གཞུག་པར་བྱའོ། །འོ་མ་དང་སྐྱུ་རུ་ར་ལ་སོགས་པའི་ཁུ་བ་ནི་མར་དང་ཆ་མཉམ་དུ་སྦྱར་བར་ཤེས་པར་བྱ་སྟེ། ཁུ་བའི་རྫས་ནི་ལྔ་པོ་སོགས། །རེ་རེའང་མར་དང་མཉམ་པར་གཞུག །ཅེས་བཤད་པ་ཡིན་ནོ། །དེ་ལྟ་བུའི་
སྨན་མར་བསྐོལ་བ་དེར་སྦྲང་རྩི་བྲེ་ཕྱེད་ཅེས་བྱ་བ་སྲང་བཅུ་དྲུག་གཞུག་པར་བྱའོ། །ཤ་ཁར་ནི་སྲང་ལྔ་བཅུ་གཞུག་པར་བྱའོ། །ན་ལེ་ཤམ་ལ་སོགས་པའི་ཕྱེ་མ་སྲང་ཕྱེད་ཕྱེད་ཅེས་པ་ཞོ་བཞི་བཞི་གཞུག་པར་བྱ་སྟེ། དེ་དག་ཕྱེ་མར་ཞིབ་ཏུ་བཏགས་ཏེ་གཞུག་པར་བྱའོ། །
