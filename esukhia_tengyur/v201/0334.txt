བས་ཁྱབ་བྱེད་ལ། །མེས་བསྲེགས་ལྟ་བུར་ཆུ་བུར་འབྱུང་། །མྱུར་དུ་མཆེད་ཅིང་ཁྱབ་པར་བྱེད། །གནད་ཀྱི་རྗེས་འཇུག་མེ་དབལ་ནི། །དེ་ནས་སྟོབས་རབ་སྐྱེས་པའི་རླུང་། །ཡན་ལག་ན་ཞིང་དྲན་པ་སྟོར། །གཉིད་ཡར་དབུགས་ཀྱི་རྔམ་པ་དྲག །སྐྱིགས་བུས་འདེབས་ཤིང་
འདི་ལྟ་བུའི། །གནས་སྐབས་དེས་ནི་ཐོབ་པར་འགྱུར། །ས་དང་མལ་ཁྲི་སྟན་སོགས་ལ། །གང་དུའང་བདེ་བ་མི་འཐོབ་ཅིང་། །འགྲེ་ཞིང་སྡུག་བསྔལ་མྱོང་བ་དང་། །ལུས་སེམས་ངལ་ལས་བྱུང་བ་ཡིས། །གཉིད་ཀྱི་རྨོངས་བག་སད་དཀའ་བ། །དེ་ནི་མེ་འདྲའི་མེ་
དབལ་ཡིན། །ཞེས་བྱ་བ་ལ། རླུང་དང་མཁྲིས་པ་ལས་ནི་མེ་དང་འདྲ་བའི་མེ་དབལ་དུ་འགྱུར་ལ། དེ་ཡང་རིམས་ཀྱིས་འདེབས་པ་ལ་སོགས་པ་དང་ལྡན་པ་དང་། །ལུས་ཀྱི་ཡན་ལག་ཐམས་ཅད་མེ་མདག་འབར་བས་ཁྱབ་ཅིང་གཏོར་བ་ལྟ་བུར་རབ་ཏུ་ཚ་བ་དང་། ཡང་
མེ་དབལ་དེ་ལུས་ཀྱི་ཕྱོགས་གང་དང་གང་དུ་མཆེད་ཅིང་ཁྱབ་པའི་ལུས་ཀྱི་ཡན་ལག་གི་ཕྱོགས་དེ་དང་དེར་མེ་མདག་འབར་བ་ཞི་བ་བཞིན་དུ་གནག་ཅིང་སྔོ་བ་དང་དམར་བས་ཁྱབ་པར་འགྱུར་རོ། །ཡང་མེ་དབལ་དེ་ནི་མེས་བསྲེགས་པ་ལྟ་བུར་ཆུ་བུར་གྱིས་མྱུར་དུ་མཆེད་པར་བྱེད་
ཅིང་ཀུན་ཏུ་ཁྱབ་པར་བྱེད་པར་འགྱུར་ཏེ། དེ་ཡང་རླུང་དང་མཁྲིས་པའི་རྒྱུ་ཉིད་ཀྱིས་མྱུར་བ་ཁོ་ནར་ཁྱབ་པར་བྱེད་པ་ཡིན་ནོ། །ཡང་དེ་བཞིན་དུ་སྙིང་ལ་སོགས་པའི་གནད་ཀྱི་རྗེས་སུ་འཇུག་པར་བྱེད་ཅིང་མེ་དབལ་དེས་དེའི་འོག་ཏུ་སྟོབས་རབ་ཏུ་སྐྱེས་པའི་རླུང་དེས་ཡན་ལག་ན་ཞིང་
གཟེར་བར་བྱེད་པ་དང་། དྲན་པ་སྟོར་ཞིང་བརྒྱལ་བར་བྱེད་པ་དང་། གཉིད་ཡར་ཞིང་མེད་པ་དང་། དབུགས་ཀྱི་རྔམ་པ་དྲག་ཏུ་འབྱུང་བ་དང་། སྐྱིགས་བུའི་ནད་ཀྱིས་འདེབས་པ་དང་། མེ་དབལ་གྱི་ནད་དང་ལྡན་པའི་སྐྱེས་བུ་དེ་འདི་ལྟ་བུའི་གནས་སྐབས་འཐོབ་ཅིང་
མྱོང་བར་འགྱུར་ཏེ། སེམས་མལ་ན་བདེ་བར་མི་གནས་པའི་དབང་དུ་གྱུར་བས་ས་དང་མལ་ཁྲི་དང་སྟན་ལ་སོགས་པ་གང་དུ་གནས་སུ་ཟིན་ཀྱང་ལུས་དང་སེམས་བདེ་བ་མི་འཐོབ་པ་ཡིན་ལ། དེ་ལྟར་མི་བདེ་བ་ཉིད་ཀྱིས་ཕན་ཚུན་འགྲེ་ཞིང་ལྡོག་ལ་སྡུག་བསྔལ་གྱི་ཚོར་བ་
