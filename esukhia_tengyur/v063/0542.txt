སྲིད་པའི་སྦྱོར་བ་དང་། ལྟ་བའི་སྦྱོར་བ་དང་། མ་རིག་པའི་སྦྱོར་བའོ། །སྦྱོར་བ་དེ་རྣམས་ནི་ཤིན་ཏུ་འཇིགས་སུ་རུང་བས་ན་དགོན་པའོ། །དེ་རྣམས་ལས་བྱུང་བ་ནི་ངེས་པར་མཐོང་བ་སྟེ། དེས་ན། སྦྱར་བའི་དགོན་པ་ལས་བྱུང་བ། །ཞེས་བྱ་སྟེ། ངེས་པའི་ཚིག་ནི་ཡང་ཁམས་དང་། འགྲོ་བ་དང་། སྐྱེ་གནས་དང་། ས་རྣམས་སུ་
སྦྱོར་ཞིང་སེམས་ཅན་རྣམས་ཆུ་བོར་ལྷུང་བར་འགྱུར་བས་ན་སྦྱོར་བ་ཞེས་བྱའོ། །ཉོན་མོངས་ཉེ་བའི་ཀུན་ཉོན་མོངས། །བག་ཆགས་བཅས་པར་གཏན་སྤངས་པ། །ཞེས་བྱ་བ་ལ། དེ་ལ་ཉོན་མོངས་པ་ནི་དགུ་བཅུ་རྩ་བརྒྱད་དོ། །ཉེ་བའི་ཉོན་མོངས་པ་ཡང་དེ་ཉིད་དེ། དེ་རྣམས་མདོར་བསྡུས་པ་དང་རྒྱས་པར་ཕྱེ་ནས་གསུངས་
པ་ཡིན་ནོ། །དེ་ལ་ཀུན་དུ་སྦྱོར་བ་དགུ་སྟེ། རྗེས་སུ་ཆགས་པ་དང་། ཁོང་ཁྲོ་བ་དང་། ང་རྒྱལ་དང་། མ་རིག་པ་དང་། ལྟ་བ་དང་མཆོག་ཏུ་འཛིན་པ་དང་། ཐེ་ཚོམ་དང་། སེར་སྣ་དང་། ཕྲག་དོག་གོ། །དེ་ལ་ཁམས་གསུམ་གྱི་འདོད་ཆགས་ཞེན་པའི་མཚན་ཉིད་རྒྱ་སྐྱེགས་ཞུ་བར་སྦྱར་བ་ལྟ་བུའི་ཆོས་ནི་རྗེས་སུ་ཆགས་
པ་སྟེ། རྗེས་སུ་ཆགས་པ་ཉིད་ཀུན་དུ་སྦྱོར་བའོ། །སེམས་ཅན་རྣམས་ལ་གནོད་ཅིང་གསོད་པའི་མཚན་ཉིད་ཅན་ས་བོན་ཁ་བ་ལྟ་བུའི་ཆོས་ནི་ཁོང་ཁྲོ་བ་སྟེ། ཁོང་ཁྲོ་བ་ཉིད་ཀུན་དུ་སྦྱོར་བའོ། །བདག་དང་གཞན་དུ་ཉེ་བར་བཏགས་ནས་སེམས་མཐོ་བའི་མཚན་ཉིད་ཅན་དུ་སྐྱེས་བུ་རེངས་པ་ལྟ་བུ་ནི་ང་རྒྱལ་ལོ། །དེ་ཡང་
རྣམ་པ་བདུན་ཏེ། ང་རྒྱལ་དང་། ཆེ་བའི་ང་རྒྱལ་དང་། ང་རྒྱལ་ལས་ཀྱང་ང་རྒྱལ་དང་། ངའོ་སྙམ་པའི་ང་རྒྱལ་དང་། མངོན་པའི་ང་རྒྱལ་དང་། ཅུང་ཟད་སྙམ་པའི་ང་རྒྱལ་དང་། ལོག་པའི་ང་རྒྱལ་ལོ། །དེ་ལ་ང་རྒྱལ་ནི་ཆུང་ངུ་བས་བདག་རུས་དང་། རིགས་དང་། གཟུགས་དང་། ནོར་དང་། ཐོས་པ་དང་། ཚུལ་
ཁྲིམས་དང་། བཟོ་ལ་སོགས་པ་དག་གིས་བྱེ་བྲག་ཏུ་འཕགས་པའམ། རུས་ལ་སོགས་པ་འདྲ་བ་དང་མཚུངས་སོ་སྙམ་པ་དེའི་རྒྱུ་ལས་བྱུང་བའི་སེམས་ཀྱི་རློམ་སེམས་དེ་ནི་ང་རྒྱལ་ཞེས་བྱའོ། །ཆེ་བའི་ང་རྒྱལ་ནི་མཚུངས་པ་བས་བདག་རུས་ལ་སོགས་པ་དག་གིས་བྱེ་བྲག་ཏུ་འཕགས་པའམ། ཆེ་བ་དང་
མཚུངས་སོ་སྙམ་པ་དེས་དྲངས་པའི་སེམས་ཁེངས་པ་ནི་ཆེ་བའི་ང་རྒྱལ་ལོ། །ང་རྒྱལ་ལས་ཀྱང་ང་རྒྱལ་ནི་ཆེ་བ་བས་ཀྱང་བདག་ཆེས་ཆེའོ་སྙམ་པ་དེས་དྲངས་པའི་སེམས་ཁེངས་པའོ། །ངའོ་སྙམ་པའི་ང་རྒྱལ་ནི་བདག་ཏུ་ལྟ་བའི་རྗེས་ལས་བྱུང་བའི་སེམས་ཁེངས་པའོ། །མངོན་པའི་ང་རྒྱལ་ནི་རྒྱུན་
