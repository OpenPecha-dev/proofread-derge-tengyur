ཅད་ཟད་པའི་ཕྱིར། འབྲས་བུ་ཀུན་ཤེས་པའི་བདག་ཉིད་མངོན་སུམ་དུ་གྲུབ་བོ། །དེའི་ཕྱིར། དུས་གསུམ་སངས་རྒྱས་ལས་བྱེད་པ། །སངས་རྒྱས་ཐོག་མ་ཐ་མ་མེད། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། དང་པོའི་སངས་རྒྱས་རྒྱུ་མེད་པ། །ཡེ་ཤེས་མིག་གཅིག་དྲི་མ་མེད། །དེ་ལྟ་བུའི་རང་བཞིན་དུ་

འབར་བ་དཔལ་གྱི་བེའུ། །ལག་ན་འོད་འབར་སྣང་བ་པོ། །སྨན་པ་ཆེ་མཆོག་གཙོ་བོ་སྟེ། །ཞེས་བྱ་བའི་དོན་ནི་ཡེ་ཤེས་ཀྱི་ལུས་ཀྱི་བདག་ཉིད་ནི་རྣམ་པ་ལྔར་གནས་ལ། སྔགས་ནི་ཡེ་ཤེས་ལྔའི་བརྡ་སྟོན་པའོ། །ཡིད་ནི་དཔལ་བེའུའི་ཚུལ་དུ་ཞུ་བའི་བདག་ཉིད་དུ་གནས་སོ། །
དེ་ཡང་གཞོན་ནུའི་རོལ་པ་དང་སྐྱོ་བས་ཁྱིམ་ནས་ཁྱིམ་མེད་པར་རབ་ཏུ་བྱུང་བ་དང་། ཞི་གནས་ཀྱི་འཁོར་བའི་རྒྱུ་སྤངས་པ། ཤེས་རབ་ཀྱི་སྤྱན་གྱིས་སེམས་ཅན་ལ་གཟིགས་པ་ནི་དུས་གསུམ་གྱི་སངས་རྒྱས་ཀྱི་མཛད་པ་བསྒྲུབ་པའོ། །དཀའ་བའི་གནས་སུ་ཞི་གནས་བསྒོམ་
པའི་ཚུལ་དུ་བཞུགས་ཀྱང་ཡེ་ཤེས་ཆེན་པོ་དང་པོ་སངས་རྒྱས་ཡེ་ཤེས་ལྔའི་ལུས་སུ་ཆོས་ཀྱི་ལོངས་སྤྱོད་པ་དང་། གཟི་བརྗིད་དང་ལྡན་པས་ནི་འོ་ཐུག་གསོལ་བས་མདངས་དང་ལྡན་ནོ། །དེས་ནི་སྤྲུལ་པ་དང་གཞི་རྟེན་དང་པོ་སངས་རྒྱས་སོ། །རྣམ་གྲོལ་ཏོག་དང་ཤེས་རབ་མཆོག །
ཡེ་ཤེས་རྒྱལ་དང་གཟུགས་འཇིགས་པ། །གློ་བུར་འཇིགས་དང་ཐོགས་མེད་གྲགས། །བདེ་བ་ལྡན་དང་ངག་སྙན་རིགས། །གྲགས་དབང་པོ་དང་ཆོས་ཀྱི་བརྩེགས། །འགྲོ་བའི་འོད་དང་ལེགས་པའི་རྒྱལ། །རང་བཞིན་ལུས་དང་ཡེ་ཤེས་གསུང་། །སྣ་ཚོགས་པ་དང་ཉོན་མོངས་སེལ། །
འགྲོ་བའི་མཆོག་དང་སེང་གེ་རིག །ཆོས་ཀྱི་མཚན་དང་སྐལ་བ་བཟང་། །མི་བསྐྱོད་གདོང་དང་འབར་བའི་རྒྱལ། །རྟོག་པའི་གསུང་དང་སྤྱན་བཟང་པོ། །རིག་དོན་བློ་དང་འཇིགས་པ་བྲལ། །དྲི་ཡི་འོད་དང་དགའ་བའི་འོད། །དོན་ཡོད་འགྲོ་དང་ཁྲོ་ཐལ་མདོག །གཟུགས་མཆོག་དང་ནི་བརྟན་
