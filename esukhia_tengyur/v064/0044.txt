ཅད་པའི་སྟོང་པ་བརྒྱད་ཡིན་ལ། དེའི་ཕྱེད་ཀུན་རྫོབ་ཀྱི་རྣམ་པ་བཞི་སྤངས་པའི་སྟོང་པ་ཉིད་བཞི་བསྒོམས་ནས་དེ་དག་ལ་རང་གི་ངང་གིས་ཟུང་དུ་འཇུག་པའི་ཐིག་ལེ་ཅན་ནོ། །དེའི་སྐུ་གསུང་ཐུགས་ཀྱི་ཡན་ལག་མེད་པ་བགྲང་བའི་རྩིས་ལས་འདས་པ་དེ་སྣང་བ་དང་། སྣང་བ་
མཆེད་པ་དང་། སྣང་བ་ཐོབ་པ་དང་། འོད་གསལ་བའི་བསམ་གཏན་བཞིའི་རྩེ་མོ་ཅན་དང་། དེ་ཉིད་ཀྱིས་ན་བསམ་གཏན་ཞི་གནས་དང་། ཡན་ལག་ལྷག་མཐོང་ཀུན་ཤེས་ཤིང་། ཏིང་ངེ་འཛིན་གྱི་རིགས་ཐམས་ཅད་མཁྱེན་ཅིང་ས་བཅུའི་རྒྱུད་ཐམས་ཅད་རིག་པས། ཏིང་
འཛིན་ལུས་ཅན་ལུས་ཀྱི་མཆོག །ཅེས་བྱ་སྟེ་ཆོས་ཀྱི་སྐུ་ཐོབ་པར་གྱུར་པ་དང་། ལོངས་སྤྱོད་རྫོགས་པའི་སྐུ་དང་སྤྲུལ་པའི་སྐུ་ཡང་ཐོབ་པར་འགྱུར་བའོ། །སྐུ་གསུམ་པོ་དེ་ལས་སངས་རྒྱས་སྤྲུལ་པའི་རྒྱུད་ཉིད་ཀྱིས་ཕྱོགས་བཅུར་ཇི་ལྟར་མོས་པའི་འགྲོ་བའི་དོན་བྱེད་པར་འགྱུར་
རོ། །རྟ་མཆོག་གི་དཔའ་བོ་གཅིག་པུའི་དབང་དུ་མཛད་ནས། ལྷའི་བདག་པོ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། ལྷའི་དབང་པོ་བརྒྱ་བྱིན་ནོ། །ལྷའི་ལྷ་ནི་ཚངས་པའོ། །ལྷའི་བདག་པོ་ནི་འདོད་པའི་ལྷའོ། །ལྷ་མིན་བདག་ནི་ཁྱབ་འཇུག་གོ། །འཆི་མེད་ནི་ཉི་མའོ། །ལྷའི་
ལྷ་ནི་པ་བ་སངས་སོ། །འཇོམས་བྱེད་ནི་གཞོན་ནུ་གདོང་དྲུག་གོ། །འཇོམས་བྱེད་དབང་ཕྱུག་ནི་ལྷ་ཆེན་པོའོ། །སྲིད་པའི་དགོན་པ་ལས་བརྒལ་བ་ནི་བདེ་སོགས་མ་ལ་སོགས་པའི་ལྷ་མོ་བརྒྱད་འཕྲོགས་ནས་ལྷ་བརྒྱད་གདན་དུ་བཏིང་བའོ། །དེ་ཉིད་ཀྱིས་ན། སྟོན་པ་གཅིག་པུ་འགྲོ་
བའི་བླ། །ཞེས་བྱའོ། །དེ་བཞིན་གཤེགས་པ་ལྔའི་རྟ་མཆོག་གི་དཀྱིལ་འཁོར་གྱི་དབང་དུ་མཛད་ནས། འཇིག་རྟེན་ཕྱོགས་བཅུ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཕྱོགས་བཅུར་གྲགས་པའི་མི་བསྐྱོད་པ་བདུད་བཞི་ལས་རྒྱལ་བའི་ཆོས་ཀྱི་སྦྱིན་བདག་ཆེ་བ་སྟེ་སྐུ་མདོག་
དཀར་པོ་རྡོ་རྗེ་བསྣམས་པའོ། །མཆོག་སྦྱིན་པའི་ཕྱག་རྒྱས་བྱམས་པའི་གོ་ཆས་བཅིངས་ཤིང་སྙིང་རྗེའི་ཡ་ལད་བགོས་པས་སྐུ་མདོག་སེར་པོ་རིན་པོ་ཆེ་བསྣམས་པའོ། །ཕྱག་བཞིས་ཤེས་རབ་ཀྱི་པོ་ཏི་དང་རལ་གྲི་དང་མདའ་གཞུ་བསྣམས་པ། ཉོན་མོངས་པ་མི་ཤེས་པའི་
