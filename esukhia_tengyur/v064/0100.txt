སྤྲུལ་དང་རབ་གྲུབ་སྦྱིན་པ་དང་། །བར་བཅད་དག་དང་མི་གཡོ་བ། །མངོན་དུ་གཙུག་ཕུད་ཞིང་དང་ནི། །ས་གནོན་རྫུ་འཕྲུལ་ཆེ་ནས་ཆེ། །དེ་ནི་རྩ་བའི་འཁོར་ལོའོ། །ཨ་ཨཱ་ཞེས་བྱ་བ་ནི་རིགས་བྱིན་གྱིས་བརླབས་པའི་དང་པོའི་སངས་རྒྱས་ཀྱི་སྔགས་སོ། །སྠི་ཏོ་ཧྲྀ་དི་ཛྙཱ་ན་མཱུརྟི། དུས་
གསུམ་གྱི་སངས་རྒྱས་ཀྱི་བྱིན་གྱིས་བརླབས་པའི་ཕྱག་གི་སྔགས་སོ། །བཛྲ་ཏཱིཀྵྞ་ཞེས་བྱ་བའི་སྔགས་ནི། ཡེ་ཤེས་སེམས་དཔའི་སྔགས་སོ། །དབྱངས་ཡིག་ལ་བཞི་བསྣན་པ་ནི་ཁྲོ་བོ་བཅུ་དྲུག་གིའོ། །རིགས་ལྔ་ཡུམ་<དད་,དང་>བཅས་པ་ནི་བཛྲ་ཏཱིཀྵྞ་དང་། ཨ་ར་པ་ཙ་ནའི་བར་དུའོ། །
