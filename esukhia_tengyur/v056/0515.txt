བ་སུ་དེ་བ་ཞེས་བྱའོ། །འདིར་ནི་རྡོ་རྗེ་སྒྱུ་མར་བཏགས་སོ། །ཚངས་པ་ནི་ཕས་མཆོད་ཅེས་ཀྱང་བྱ་སྟེ་འདིར་ནི་རྡོ་རྗེ་མི་སྨྲ་བར་བཏགས་སོ། །ས་ནད་གཞོན་ནུ་ནི་ཀརྟི་ཀ་སྟེ་ལྷ་ཆེན་པོའི་སྲས་ཏེ་འདིར་ནི་རྡོ་རྗེ་དྲིལ་བུར་བཏགས་སོ། །དབང་པོ་ནི་བརྒྱ་བྱིན་ཏེ་འདིར་ནི་རྡོ་རྗེ་འཐབ་
ཆར་བཏགས་སོ། །དེ་ནས་རྡོ་རྗེ་རིག་པ་དམ་པས། རང་གི་སྙིང་པོ་གསུངས་པ། ཨོཾ་ནས་ཤཱུ་ལའི་བར་དུའོ། །རྡོ་རྗེ་སྒྱུ་མས་གསུངས་པ་ཨོཾ་ནས་ཕཊ་ཀྱི་བར་དུའོ། །རྡོ་རྗེ་མི་སྨྲ་བས་གསུངས་པ། ཨོཾ་ནས་སྭའི་བར་དུའོ། །རྡོ་རྗེ་དྲིལ་བུས་གསུངས་པ་ཨོཾ་ནས་ཕཊ་ཀྱི་བར་དུའོ། །རྡོ་རྗེ་འཐབ་ཆས་གསུངས་པ་
ཨོཾ་ནས་ཏྲཾ་གི་བར་དུའོ། །དེ་ནས་དེ་དག་གི་དཀྱིལ་འཁོར་འབྱུང་སྟེ་ཞེས་པ་དཀྱིལ་འཁོར་ཇི་ཙམ་ནུས་པ་ཟླུམ་པོར་བྱས་ལ་ཚོན་སྣ་ལྔས་བྲིས་ཏེ། སྒོ་ཤར་ཕྱོགས་སུ་བཏོད་ལ་ཤར་དང་བྱང་གི་གྲྭ་ཨཱི་ཤ་ནའི་གནས་སུ་རྡོ་རྗེ་རིག་པ་དམ་པ་དབང་ཕྱུག་ཆེན་པོ་བྲི་སྟེ། མདོག་དཀར་པོ་ཕྱག་གཡས་པ་ན་གསལ་ཕུར་
ཐོགས་པ་སྐྱིལ་མོ་ཀྲུང་ཕྱེད་བཅས་པ། རློད་སྟན་གྱི་སྟེང་ན་གནས་པ་བྲི། དཀྱིལ་འཁོར་གྱི་གྲྭ་བཞི་སྲད་བུའི་དབུས་ཞེས་བྱ་བ་ནི་དཀྱིལ་འཁོར་ཟླུམ་པོའི་བྱང་ཤར་གྱི་གྲྭའི་ནང་ལོགས་ཞེས་བྱ་བའི་དོན་ཏོ། །ཁ་ཅིག་གིས་དབང་ཕྱུག་ཆེན་པོ་དཀྱིལ་འཁོར་གྱི་དབུས་སུ་བཞག་ལ་ཕྱོགས་བཞིར་ལྷ་གཞན་དག་
དགོད་ཅེས་ཀྱང་འཆད་དོ། །དེར་གཞན་ཡང་རིམ་གྱིས་ཐམས་ཅད་བྲིའམ་བཞག་པའམ་ཞེས་པ་ཤར་ཕྱོགས་སུ་རྡོ་རྗེ་འཐབ་ཆ་སྟེ་ཁ་དོག་དཀར་པོ་སྤྱན་གསུམ་པ་ཕྱག་གཡས་པ་ན་རྡོ་རྗེ་རྩེ་གསུམ་པ་གཡོན་པ་པུས་གོང་དུ་བསྟན་པ་སྐྱིལ་མོ་ཀྲུང་རློད་སྟན་ལ་བཞུགས་པ་བྲིའོ། །ལྷོ་ཕྱོགས་སུ་རྡོ་རྗེ་སྒྱུ་མ་བྲི་སྟེ། ཁ་
དོག་ལྗང་གུ་ཕྱག་བཞི་པ་གཡས་གཅིག་ན་འཁོར་ལོ་རྩིབས་མ་ཅན་བསྣམས་ལ། གཅིག་ཆོས་འཆད་པའི་ཚུལ་མཛད། གཡོན་པ་ཡ་གཅིག་ན་ག་ད་ཞེས་བྱ་བའི་བེ་ཅོན་བསྣམས་པ་ཡ་གཅིག་བླ་གོང་དུ་བསྟན་པའི་སྐྱིལ་མོ་ཀྲུང་བཅས་ཏེ་རློད་སྟན་ལ་བཞུགས་སོ། །ནུབ་ཕྱོགས་སུ་ཚངས་པ་བྲི་སྟེ་ཁ་དོག་དམར་པོ་
དབུ་གསུམ་པ་ཕྱག་བཞི་པ་གཡས་པ་ཡ་གཅིག་ན་སྔགས་སོ། །ཡ་གཅིག་ཆོས་འཆད་པ་གཡོན་པ་ཡ་གཅིག་ན་གླེགས་བམ་ཡ་གཅིག་ན་རིལ་བ་སྤྱི་བླུགས་རློད་སྟན་ལ་སྐྱིལ་མོ་ཀྲུང་དུ་བཞུགས་པར་བྲིའོ། །བྱང་ཕྱོགས་སུ་ས་ནད་གཞོན་ནུ་བྲི་སྟེ་ཁ་དོག་དམར་པོ་ཕྱག་གཡས་ན་དྲིལ་བུ་གཡོན་པ་ཆོས་འཆད་པའི་
