པ་དག་ནི་ཞེས་པ་སྒྲའི་ཚོགས་པ་གཅིག་ལ་ཡན་ལག་གི་དངོས་པོས་འཇུག་བཞིན་པ་དག་ཅེས་པའི་དོན་ཏོ། །ཅི་ལས་གཞི་མཐུན་པ་ཉིད་དུ་འགྱུར་ཞེས་པ་དོན་ཐ་དད་པ་ཉིད་ཀྱི་ཕྱིར་ཏེ། དཔེར་ན་བུམ་པ་དང་སྣམ་བུའི་སྒྲ་དག་ལ་གཞི་མཐུན་པ་ཉིད་མེད་པ་དེ་ལྟར་སྔོན་པོ་དང་ཨུཏྤ་ལའི་སྒྲ་ཚོགས་ཟིན་
པ་དག་ལ་ཡང་ངོ་ཞེས་པ་ཁྱབ་བྱེད་དང་འགལ་བ་གསལ་བར་བྱེད་དོ། །ཁྱད་པར་ནི་སྤྱིའི་ནང་དུ་འདུས་པའི་ཕྱིར་ཞེས་པ། སྔོན་པོ་ཞེས་པ་སྤྱི་སྟེ། དེའི་ཁྱད་པར་རྣམས་ནི་ཨུཏྤ་ལ་ལ་སོགས་པ་རྣམས་སོ། །དེ་རྣམས་དེར་ནང་དུ་འདུས་སོ། །སྔོན་པོ་དང་ཨུཏྤ་ལའི་སྒྲ་དག་ནི་སྤྱི་རྗོད་པར་བྱེད་པ་
དག་སྟེ། ཁྱད་པར་ནང་དུ་འདུས་ནས་འཇུག་པ་དག་གོ། །དེའི་ཕྱིར་ཁྱད་པར་རྣམས་སྤྱིའི་ནང་དུ་འདུས་པའོ་ཞེས་པས། དེ་རྣམས་ཀྱང་སྔོན་པོ་དང་ཨུཏྤ་ལའི་སྒྲ་དག་གི་དོན་རྣམས་སོ། །དེ་ལྟར་སྔོན་པོ་དང་ཨུཏྤ་ལའི་སྒྲ་གཉིས་ཀ་དག་ཀྱང་ཨུཏྤ་ལ་སྔོན་པོ་དོན་དང་ལྡན་པ་དག་གོ། །དེ་ལས་ཀྱང་
རང་གི་དོན་ཡོངས་སུ་མི་སྤོང་ན་ཡང་སྐྱོན་ཅུང་ཟད་ཀྱང་མེད་དོ། །གལ་ཏེ་འོ་ན་སྤྱིའི་སྒྲ་ཡང་ཁྱད་པར་གྱི་དོན་ཅན་ཏེ། དེ་ཁོ་ན་ལས་ཁྱད་པར་རྟོགས་པའི་ཕྱིར་སྒྲ་གཉིས་པ་སྦྱོར་བ་སྤྱིའི་དོན་དུ་ཡིན་ཞེ་ན་སྨྲས་པ། འདྲ་བའི་ངོ་བོ་དག་ཅེས་པ་ལ་སོགས་པ་སྟེ། ངོ་བོ་གང་གིས་སྔོན་པོའི་སྒྲ་སྔོན་པོ་ཉིད་
རྗོད་པ་དེ་ཁོ་ནས་ཨུཏྤལ་སྔོན་པོ་ཡང་ངོ་། །དེའི་ཕྱིར་ངོ་བོ་མཚུངས་པ་ཉིད་ཀྱི་ཕྱིར་ཅི་འདི་སྤྱིའི་དོན་ཁོ་ནའམ་འོན་ཏེ་ཁྱད་པར་གྱི་དོན་ཡང་ཡིན་ཞེས་མི་ཤེས་ཏེ། ཁྱད་པར་གྱི་དོན་དང་ལྡན་པ་ཉིད་དུ་ཤེས་པར་བྱ་བའི་དོན་དུ་སྒྲའི་སྦྱོར་བ་གཉིས་པའོ། །འདིར་རེ་ཞིག་ཅེས་པ་ལ་སོགས་པ་སྟེ། སྔོན་པོའི་སྒྲས་
ནི་སྔོན་པོའི་རིགས་སམ། དེ་དང་སྦྱོར་བའི་ཡོན་ཏན་བརྗོད་པར་བྱའི་ཨུཏྤ་ལ་ལ་སོགས་པའི་རྫས་ཀྱི་རིགས་ནི་མ་ཡིན་པས་ཏེ། དེ་ལས་ཀྱང་ཨུཏྤལ་དམར་པོ་ལ་སོགས་པ་རྣམས་ནི་ཡོན་ཏན་སྔོན་པོའམ་དེ་ལ་འཕྲོད་པ་འདུ་བའི་སྔོན་པོ་ཉིད་ཀྱི་མ་ཡིན་ཏེ། དེས་ན་ཅི་ལྟར་སྔོན་པོའི་སྒྲའི་དོན་གྱི་ནང་དུ་ཨུཏྤལ་
སྔོན་པོ་ལ་སོགས་པ་རྣམས་འདུ་བར་འགྱུར། སྤྱིས་བརྗོད་པ་ཞེས་པས་ཐ་མི་དད་པའི་སྒྲའི་དོན་སྨྲས་སོ། །དེ་འདིས་ནི་གང་རྣམས་སྔོན་པོའི་སྒྲའི་དོན་གྱི་ཁྱད་པར་དུ་འགྱུར་བ་མ་ཡིན་པ་དེ་རྣམས་ནི་དེའི་ནང་དུ་འདུ་བ་མ་ཡིན་ཏེ། དཔེར་ན་བ་ལ་ཀ་ལ་སོགས་པ་རྣམས་བཞིན། སྔོན་པོའི་ཨུཏྤལ་ལ་
