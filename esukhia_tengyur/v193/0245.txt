པར་བྱ་བ་ཐ་དད་པའི་ཆོས་དེ་རྣམས་ནི་དེའི་རྟོགས་པར་བྱེད་པ་རྣམས་ཏེ། རྗེས་སུ་འགྲོ་བ་ལ་སོགས་པ་བཞིན། ཡོད་པ་ཉིད་དང་ཁྱབ་བདག་ཉིད་ལ་སོགས་པ་རྣམས་ཀྱང་དེ་ལྟ་བུའོ་ཞེས་པ་རང་བཞིན་གྱི་ཐལ་བར་འགྱུར་བ་གཉིས་གསུངས་སོ། །འོ་ན་གལ་ཏེ་ཡང་
རྣམ་པ་ཐམས་ཅད་དུ་རྒྱུ་མཚན་དང་རྒྱུ་མཚན་ཅན་གྱི་དངོས་པོ་ཡིན་པ་དེ་ལྟ་ན་ཡང་དེ་ལྟར་མ་ངེས་པའི་ཕྱིར་རྣམ་པ་ཐམས་ཅད་དུ་གོ་བར་བྱ་བ་དང་གོ་བར་བྱེད་པ་ཉིད་དུ་ཐལ་བ་མ་ཡིན་ནོ་ཞེ་ན། ཐུན་མོང་དུ་མངོན་སུམ་པ་ཡང་ངེས་པར་མ་བཟུང་བའི་དོན་ཁྱད་པར་མཐོང་བ་ལས་ཁྱད་པར་དུ་རྟོགས་པར་བྱེད་
དོ་ཞེས་པ་ལ་སོགས་པའི་ཚིག་ལ། རྟགས་ཀྱི་ཤེས་པ་ནི་ངེས་པའི་བདག་ཉིད་ཅན་དུ་འདོད་པ་ཉིད་ཀྱི་ཕྱིར་རོ། །མ་ངེས་པར་འདོད་པ་རྣམས་ཀྱི་ཡང་རྟགས་ཀྱི་ཆོས་རྣམས་གཞན་མ་ཡིན་པ་ཉིད་ཀྱི་ཕྱིར་འདི་ནི་ལན་མ་ཡིན་ནོ། །དེ་ལྟར་འགྱུར་མོད། དེ་ལྟར་གོ་བར་བྱ་བ་དང་གོ་བར་བྱེད་པའི་
དངོས་པོར་འདོད་པ་ཁོ་ནའོ་ཞེ་ན་གསུངས་པ། ཐ་དད་པ་རྣམས་ཀྱི་ཞེས་པ་ལ་སོགས་པ་སྟེ། གང་གི་ཕྱིར་རྗེས་སུ་འགྲོ་བ་ལ་སོགས་པ་རྣམས་ཁོ་ན་བསྟན་བཅོས་སུ་གཏན་ཚིགས་ཉིད་དུ་གཟུང་བར་བྱའི། ཡོད་པ་ཙམ་ལ་སོགས་པ་རྣམས་ནི་མ་ཡིན། གཙོ་བོའི་ཡང་ཡོད་པ་ཉིད་ལ་སོགས་
པ་རྣམས་ཁོ་ན་བསྒྲུབ་པར་བྱ་བ་ཉིད་དུ་ཡིན་གྱི། ཆ་ཤས་མེད་པ་ཉིད་ལ་སོགས་པ་རྣམས་ནི་མ་ཡིན་པ་དེའི་ཕྱིར། རྒྱུ་མཚན་དང་རྒྱུ་མཚན་ཅན་གྱི་དངོས་པོ་ནི་གཙོ་བོ་ཡོད་པ་ཉིད་ལ་སོགས་པ་རྟོགས་པར་བྱེད་པའི་གཏན་ཚིགས་མ་ཡིན་ནོ། །ཡང་ཅི་ལྟར་གཙོ་བོའི་ཆ་ཤས་མེད་པ་ཉིད་ཡིན།
ཇི་ཙམ་གྱིས་སྙིང་སྟོབས་ལ་སོགས་པ་དེའི་ཆ་ཤས་རྣམས་སུ་འགྱུར་བ་ཁོ་ནའོ་ཞེ་ན། དེ་དང་བྲལ་བའི་ཆ་ཤས་ཡོད་པ་མ་ཡིན་པའི་ཕྱིར་ཆ་ཤས་མེད་པ་ཉིད་དོ་ཞེས་པས་སྐྱོན་མེད་དོ། །གཙུག་ཕུད་ཅན་གྱི་དོན་བཅུ་ཞེས་པ་ཡོད་པ་ཉིད་དང་། གཅིག་ཉིད་དང་། དོན་དོན་དང་ལྡན་པ་ཉིད་དང་།
གཞན་གྱི་དོན་ཉིད། གཞན་ཉིད་དང་། བྱེད་པ་པོའི་དངོས་པོ་མ་ཡིན་པ་དང་། སྦྱོར་བ་དང་། འབྲལ་བ་དང་། མང་པོའི་སྐྱེས་བུ་རྣམས་དང་། གནས་པ་སྟེ། ལུས་ཀྱི་ཡང་ལྷག་མར་འཇུག་པས་ཞེས་པ་འདི་རྣམས་སུ་རིག་པར་བྱའོ། །དེའི་ཕྱིར་གང་གང་དུ་རྗེས་སུ་མཐུན་པ་ཞེས་སྦྱར་བར་བྱ་ལ།
