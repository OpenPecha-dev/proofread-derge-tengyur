པ་ཡིན་པའི་སྐྱོན་དུ་མི་བརྩི་བའི་ཕྱིར་རོ། །རྗེའུ་རིགས་ལ་སོགས་པ་ནི་སྐྱོན་དུ་བརྩི་བ་ཡིན་པའི་ཕྱིར་དངོས་གཞིར་འགྱུར་རོ། །ཁབ་རིས་མཁན་ཞེས་བྱ་བ་ནི་སྐུད་རིས་མཁན་ནོ། །བ་ལང་སྲུང་བ་ཞེས་བྱ་བ་ནི་ཕྱུགས་རྫིའོ། །ཟོར་བ་ཞེས་བྱ་བ་ནི་བརྔ་སྤྱད་དོ། །ཕྲག་གདང་ཞེས་བྱ་བ་
ནི་ཕྲག་པའི་ཁུར་རོ། །བེའུ་རས་ཞེས་བྱ་བ་ནི་རས་བལ་གྱི་སྣམ་བུ་སྤུ་ཅན་ནོ། །དར་ལ་ཞེས་བྱ་བ་ནི་རས་སྲབ་མོའོ། །བེའུ་རས་ཡུག་ཆེན་ཞེས་བྱ་བ་ནི་གོས་ཀྱི་རིགས་སྤུ་ཅན་ཡིན་ནོ། །པར་ཏང་ཞེས་བྱ་བ་ནི་གོས་ཀྱི་རིགས་ཁྲ་བོ་སྡུག་པའོ། །ཁྲིའུ་ཚེ་ཞེས་བྱ་བ་ནི་ཐགས་མ་ཁ་དོག་ཅན་ནོ། །གོར་བུ་
ཞེས་བྱ་བ་ནི་ཁྲི་རུ་སྟན་ནོ། །ཞྭ་སྒྲའོ་ཞེས་བྱ་བ་ནི་རྣ་བ་མེད་པའོ། །བྱིས་པའི་གོས་ཞེས་བྱ་བ་ནི་བྱིས་པ་ཆུང་ངུའི་གོས་སོ། །ཤིང་ང་དཔུང་ཆད་ཅེས་བྱ་བ་ནི། ཤིང་ང་དཔུང་ཆད་དུ་བགོ་བ་དྲས་ཏེ་དྲུབས་པའོ། །དོར་མ་ཞེས་བྱ་བ་ནི་རྐང་གོས་སོ། །རྔུས་མ་གར་ཞེས་བྱ་བ་ནི་ཁར་བ་ལ་སོགས་པ་རྔུས་
མགར་རོ། །ཆ་ལང་ཞེས་བྱ་བ་ནི་རོལ་མོའི་སྒྲ་འབྱིན་པའོ། །སྐོམ་ཕོར་ཞེས་བྱ་བ་ནི་སྐོམ་གྱི་སྣོད་དོ། །སྒ་ཞེས་བྱ་བ་ནི་རྟ་ཞོན་པའི་སྒའོ། །དོང་པ་ཞེས་བྱ་བ་ནི་མདའ་སྣོད་དོ། །ཁ་སྤུ་བྲེགས་པ་ཞེས་བྱ་བ་ནི་<སྨར་,སྨ་ར་>བཅོས་པའོ། །བྱིན་པ་བཅོས་པ་ཞེས་བྱ་བ་ནི་རྗེ་ངར་གྱི་སྤུ་བཅོས་པའོ། །དུད་གཟེབ་
པ་ཞེས་བྱ་བ་ནི་གོས་དུ་བས་བདུག་པའི་གཟེབ་བོ། །ཕོ་ཉར་འགྲོ་བ་ཞེས་བྱ་བ་ནི་བདག་ཉིད་མངག་པར་འགྲོ་བའོ། །མངག་པ་ཞེས་བྱ་བ་ནི་གཞན་བསླབས་ཏེ་མངག་པར་བཏང་བའོ། །སོན་པ་ཞེས་བྱ་བ་ནི་སོན་མའོ། །དྲུ་བུ་ཞེས་བྱ་བ་ནི་ཕུད་བུ་ལ་སོགས་པའོ། །བརྒྱང་ཞིང་ཞེས་བྱ་བ་ནི་བཙེམ་
པའི་ཕྱིར་གོས་བརྒྱང་བ་ལ་དགོས་པའོ། །ཚོན་ཞེས་བྱ་བ་ནི་སྔོན་པོ་ལ་སོགས་པའོ། །ཕུར་པ་ཞེས་བྱ་བ་ནི་བཅིང་བའི་ཤིང་ངོ་། །སོ་མངས་ཞེས་བྱ་བ་ནི་ཆ་མཉམ་པར་བྱ་བ་ལ་དགོས་པའི་ཤིང་སོ་མངས་སོ། །སྦུབ་མོ་ཞེས་བྱ་བ་ནི་སོར་མོ་ལ་བསྐོན་པའི་ཀོ་བའོ། །སྦུད་རུ་ཞེས་བྱ་བ་ནི་སྦུད་པའི་
སྣ་ལ་གདགས་པའི་སྦུ་བུའོ། །ཏི་ཚ་ཞེས་བྱ་བ་ནི་འོག་གཞིའི་རྡོའོ། །ཐོ་བ་ཞེས་བྱ་བ་ནི་ལྕགས་ཐོའོ། །རྒྱུས་པ་ཞེས་བྱ་བ་ནི་འཚེམ་པའི་སྐུད་པའོ། །སྨྱུང་བུ་ཞེས་བྱ་བ་ནི་སྨྱུང་ཕྲ་མོའོ། །འཁོར་ལོའི་རྟེན་མ་ཞེས་བྱ་བ་ནི་འཁོར་ལོའི་གཞི་སྟེ་འཁོར་གཏན་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །གཞལ་
