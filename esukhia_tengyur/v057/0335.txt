ཕྱག་རྒྱ་བཅིངས་ལ། ཕྱག་རྒྱ་ཆེན་པོ་སྦྱར་བར་བྱའོ་ཞེས་བཅོམ་ལྡན་འདས་ཀྱིས་གསུངས་པ་ཡིན་ནོ། །དེ་ནས་བསྒྲུབ་པར་བྱའོ་ཞེས་བྱ་བ་ནི་དེ་ལྟར་ཆོ་ག་འདིས་ཕྱག་རྒྱ་བཅིངས་ལ་དེ་བཞིན་གཤེགས་པའི་ཟློས་པ་པོས་ནི། ཛཿཧཱུཾ་བཾ་ཧོཿཞེས་བརྗོད་ལ། ལེགས་པའི་ཆོ་ག་ཡིད་
ཀྱིས་ནི། །ལུས་ལ་སངས་རྒྱས་ཀུན་གཞུག་གོ། །དམ་ཚིག་མཆོག་གི་ཕྱག་རྒྱ་ནི། །སེམས་དཔའི་རྡོ་རྗེ་ལ་སོགས་པས། །དེ་ལྟར་སེམས་དཔའ་ཆེན་པོ་བསྒྲུབ། །ཛཿཧཱུཾ་བཾ་ཧོཿཞེས་བརྗོད་ཅིང་། །ཞེས་བྱ་བའོ། །རྡོ་རྗེ་སེམས་དཔའ་ལ་སོགས་པའི་སྒྲུབ་པ་པོས་ནི། ཕྱག་རྒྱ་
ཆེན་པོ་བཅིངས་ལ། བཛྲ་སཏྭ་ཨཿཞེས་བྱ་བ་ལ་སོགས་པས་མདུན་དུ་རྡོ་རྗེ་དབབ་པ་ཇི་ལྟ་བ་བཞིན་དུ་བྱས་ལ། རང་གི་དམ་ཚིག་གི་ཕྱག་རྒྱ་དང་། བཛྲ་སཏྭ་དྲྀ་ཤྱ་ཞེས་བྱ་བ་ལ་སོགས་པས་བལྟ་བ་བྱས་ལ། དེ་ཉིད་རང་གི་ལུས་ལ་དགུག་པ་དང་། གཞུག་པ་དང་། བཅིང་
བ་དང་། དབང་དུ་བྱས་ལ་བསྒྲུབ་པར་བྱའོ། །འདིས་ནི། རྡོ་རྗེ་དབབ་པ་བྱས་ནས་ནི། །ཕྱག་རྒྱ་ཆེན་པོ་ཆོ་ག་བཞིན། །བཅིངས་ནས་དེ་ཡི་མདུན་ཕྱོགས་སུ། །སེམས་དཔའ་ཆེན་པོ་དེ་བསྒོམ་མོ། །ཡེ་ཤེས་སེམས་དཔའ་དེ་མཐོང་ནས། །རང་གི་ལུས་ལ་བསྒོམ་པར་བྱ། །
བཀུག་ཅིང་བཅུག་ལ་བཅིང་བ་དང་། །དབང་དུ་བྱས་ཏེ་སྒྲུབ་པའོ། །ཞེས་གང་གསུངས་པ་དེ་བཤད་པ་ཡིན་ནོ། །དེ་ནས་དངོས་གྲུབ་ཀྱི་ཆོ་ག་རྒྱས་པ་བཤད་དེ་ཞེས་བྱ་བ་ནི་དངོས་གྲུབ་ཀྱི་དོན་གྱི་ཆོ་ག་རྒྱས་པ་ཡིན་པས་ན། དངོས་གྲུབ་ཀྱི་ཆོ་ག་རྒྱས་པའོ། །ནོར་བསྒྲུབ་པར་
འདོད་པས་རང་གི་ལྷའི་སྦྱོར་བ་རྗེས་སུ་སྤྱད་ནས་ཇི་སྐད་གསུངས་པའི་ཚུལ་གྱིས་རྡོ་རྗེའི་དངོས་གྲུབ་ལ་སོགས་པ་བསམ་པ་ཉིད་དང་། ཤིས་པའི་སྒྲ་བརྗོད་པར་བྱ་བ་ནི། ཨརྠ་སིདྡྷི་ཞེས་བྱ་བ་སྟེ་གང་གི་ཕྱིར་སྙིང་པོ་འདིས་ནོར་གྱི་དངོས་གྲུབ་ཏུ་འགྱུར་བ་ཡིན་ནོ། །དེ་ནས་རྡོ་རྗེ་ཞེས་
