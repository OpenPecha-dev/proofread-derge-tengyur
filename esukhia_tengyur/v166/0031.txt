ཡིད་ཀྱི། །བྱ་བ་རྣམ་པ་གསུམ་པ་ཅན། །དགེ་སློང་ལ་ནི་མི་དགེ་བས། །ལྟུང་བར་འགྱུར་གྱི་སེམས་དག་ན། །མཆོད་རྟེན་སྐུ་ལ་སྐྱེས་པ་ཡི། །རྩྭ་སྔོན་གཅོད་པར་བྱེད་བཞིན་དུ། །ཇི་ལྟར་དགེ་བའི་སེམས་ཀྱིས་ནི། །སྟོན་པ་ལ་ནི་མཆོད་འདོད་པས། །མེ་ཏོག་ཕྲེང་བ་སྤེལ་བའི་ཕྱིར། །

ཡང་། །སྐབས་འོངས་ཞར་ལ་བརྗོད་པ་ཡིན། །དེ་ལྟར་རྣམ་པ་དུ་མ་ཡི། །ལྟུང་བ་སོ་སོའི་སྒོ་ནས་འདིར། །རྩོད་པའི་དབང་དུ་བྱས་པ་ཡི། །ལྟུང་བ་འདི་ནི་མཁས་པས་གསུངས། །གསོལ་བ་ལ་སོགས་ལས་འདི་དག །རུང་བ་མིན་ཞེས་སྨྲ་བ་རྣམས། །རྩོད་པ་དང་ནི་འཐབ་མོ་ཡང་། །
བྱ་བའི་ཡང་ནི་རྒྱུ་རུ་འདོད། །དགེ་འདུན་འབྱེད་པར་གང་བརྩོན་པའི། །བློ་ཆུང་གིས་ནི་ཉེས་བྱས་ཐོབ། །བཞམས་ཏེ་བསྒོས་ཀྱང་མི་གཏོང་ན། །ཉེས་བྱས་ཉིད་དུ་རྣམ་པར་གཞག །དང་པོ་ཉིད་དུ་གསོལ་བའི་ལས། །བརྗོད་ཚེ་མི་གཏོང་སྦོམ་པོ་ཉིད། །བརྗོད་པ་གཉིས་ལའང་དེ་ཡིན་ཏེ། །མཐར་
ནི་དགེ་འདུན་ལྷག་མ་ཉིད། །སྔ་མའི་བརྗོད་པ་གཉིས་ལ་ནི། །སྦོམ་པོའོ་དེ་རྗེས་དེ་ལ་ནི། །གསུམ་པ་དགེ་འདུན་ལྷག་མ་སྟེ། །དེ་ལྟར་འདི་ནི་ལེགས་པར་བརྟག །གང་ཡང་དེ་རྗེས་འབྲང་བ་ཡི། །དགེ་སློང་སྡིག་ཏོ་ཅན་རྣམས་ཀྱང་། །བཞམས་ཏེ་བསྒོ་སོགས་ཆོ་ག་ཡི། །སྔར་བཞིན་འདི་དག་
ཀུན་ལའང་འགྱུར། །ཁྱིམ་སུན་འབྱིན་པ་ཡིས་ནི་གཞན། །སུན་འབྱིན་པ་ནི་རྣམ་གཉིས་གསུངས། །གཅིག་ནི་འགྲོགས་པས་ཡིན་ཏེ་གཞན། །གཉིས་པ་ཡང་ནི་ལོངས་སྤྱོད་ཀྱིས། །འགྲོགས་པ་བུད་མེད་ལྷན་ཅིག་ཏུ། །རྩེ་འཇོ་འཕྱར་གཡེང་ལ་སོགས་བསྟེན། །ལོངས་སྤྱོད་མེ་ཏོག་
