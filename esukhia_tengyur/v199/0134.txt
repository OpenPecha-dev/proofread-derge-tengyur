གྱི་བར་དུ་སྣུམ་ལས་གཞི་བླངས་པའི་བཟའ་བ་དང་བཏུང་བ་སྦྱིན་ནོ། །སྨན་བསྐོལ་བའི་ཆང་ངམ། ཨ་ས་བ་ལེགས་པར་སྦྱར་བའི་ཆང་ཕྱི་དྲོའི་དུས་སུ་སྐོམ་དུ་བླུད་དོ། སྨན་ཤི་རི་ཤ་དང་། ཀ་ཀུ་པ་གཉིས་བསྐོལ་བའི་ཁུ་བའི་ནང་དུ་རས་བལ་སྦངས་པ་མོ་མཚན་གྱི་ནང་དུ་
གཞུག་གོ། །དེ་བཞིན་དུ་ནད་གང་ཡང་རུང་། གློ་བུར་དུ་བྱུང་ན། སོ་སོའི་ནད་དང་མཐུན་པའི་སྨན་སྦྱར་རོ། །རླུང་འཇོམས་པའི་སྨན་རྩ་བ་བཅུ་པའོ། །མར་དང་བསྐོལ་བ་ལས་གཞི་བླངས་པའི་བཟའ་བ་སྦྱིན་ནོ། །དེ་ནས་ཞག་བཅུའི་བར་དུ་ཤ་ཁུ་དང་བཅས་པའི་ཟས་
སྦྱིན་ནོ། །དེ་ནས་ཞག་བཅུའི་བར་དུ་ཟས་ཡང་བ་དང་། ཕན་པ་དང་། ཉུང་བའི་བཟའ་བ་སྦྱིན་ནོ། །དེ་ནས་ཟླ་བ་བཞིའི་བར་དུ་བཟའ་བ་ཉུང་ཟད་སྦྱིན་ཞིང་། འོག་ནས་འབྱུང་བའི་སྨན་ལས་སྦྱར་བའི་ཏིལ་མར་བསྟེན་པར་བྱའོ། །དེ་ཡན་ཆད་ནི་ཅི་བདེ་བར་སྤྱད་དེ་བསྲུང་མི་
དགོས་སོ། །སྨན་བ་ལའི་མར་ཞེས་བྱ་བ་ནི་བ་ལ་ཆམ་ཆོམ་བརྡུངས་ཏེ་ཆུ་དང་ཚུལ་བཞིན་དུ་བསྐོལ་ལ་དྲུག་ཆ་ལུས་པ་དང་མཉམ་པའི་དེ་དང་འོ་མ་དང་བསྲེས་པ་དེའི་ནང་དུ་ནས་དང་། རྒྱ་ཤུག་དང་། རྒྱ་སྲན་དང་། རྩ་བ་བཅུ་པ་འདི་རྣམས་ཆ་གཅིག་དང་། ཏིལ་མར་ཆ་བཅུ་བཞི་དང་།
མེ་ད་རྣམ་གཉིས་དང་། ཐང་ཤིང་དང་། བཙོད་དང་། རམ་ཉེ་བ་རྣམ་གཉིས་དང་། ཐལ་ཏྲེས་དང་། རུ་རྟ་དང་། རྒྱ་སྤོས་དང་། ཛི་བ་ཀ་དང་། ཟངས་རྩི་བ་དང་། ལྕེ་མྱང་ཚ་དང་། གླ་སྒང་དང་། རྡོ་དྲེག་དང་། ཤུ་དག་དང་། ཨ་ཀ་རུ་ནག་པོ་དང་། བ་སྤྲུ་བ་རྣམ་གཉིས་དང་།
ཨ་ཤྭ་གནྡྷ་དང་། ཉེའུ་ཤིང་དང་། པ་ཡས་སྱ་དང་། ཤིང་མངར་དང་། བ་ལ་ར་ས་དང་། ཤུ་ཏི་དང་། མུངྒ་པརྞི་དང་། མ་ཤ་པརྞི་དང་། སུག་སྨེལ་དང་། ཤིང་ཚ་དང་། གནྡྷ་བྷ་དྲ་དང་། འདི་རྣམས་ཞིབ་ཏུ་བཏགས་ཏེ་ལྷན་ཅིག་ཏུ་མེ་མི་ཆེ་བས་བསྐོལ་ལ་ཏིལ་མར་འབའ་ཞིག་
