བར་བྱེད་དོ། །འབྲས་བུ་གསུམ་དང་། ཨ་རུ་ར་དང་། སླེ་ཏྲེས་དང་། པི་པི་ལིང་དང་། འདི་རྣམས་གང་ཡང་རུང་བའི་ཁུ་བ་བསྐོལ་བ་དྲོན་མོའམ་གྲང་མོ་ཡང་རུང་སྟེ། དེ་བཞིན་དུ་མར་དང་། འདག་གུ་དང་། ཕྱེ་མ་དང་།
རི་ལུ་ལ་སོགས་པ་རེ་རེའི་སྦྱོར་བ་བཅུད་ཀྱིས་ལེན་གྱི་ཆོ་ག་བཞིན་དུ་བྱ་བའམ་ཡང་ན་གོ་བྱེའུ་གཉིས་དང་སྦྱར་བ་བྱིན་ན་ནི་རིམས་མི་བཟད་པ་ཞི་བར་བྱེད་དོ། །ཉི་མ་གང་ལ་ཉིན་བཞི་པའི་རིམས་ལ་སོགས་པས་བཏབ་པའི་དུས་སུ་ནད་གཞི་རྣམས་བརྟག་སྟེ། ལུས་
འཕེལ་བར་བྱ་བའི་ཆོ་ག་ཚུལ་བཞིན་དུ་བྱའོ། །སྒོག་སྐྱའི་ཏིལ་མར་དང་བཅས་པ་ཐོ་རངས་སམ་ཁ་ཟས་ཀྱི་གོང་དུ་བཏུང་བའམ་དེ་བཞིན་དུ་མར་རྙིང་པ་དང་། ཞོ་དང་། འོ་མ་དང་། དར་བ་དང་། སྲང་དྲུག་པ་ཞེས་བྱ་བའི་མར་གཅོང་ཆེན་པོའི་སྐབས་སུ་
བཤད་པ་དང་། དགེ་བའི་སྨན་མར་སྨྱོ་བྱེད་ཀྱི་སྐབས་སུ་འབྱུང་བ་དང་། བའི་རྣམ་ལྔའི་སྨན་མར་བརྗེད་བྱེད་ཀྱི་སྐབས་སུ་འབྱུང་བ་དང་། ཏིག་ཏའི་སྨན་མར་མཛེ་ནད་ཀྱི་སྐབས་སུ་བཤད་པ་དང་། བ་ཤ་ཀའི་སྨན་མར་ཁྲག་མཁྲིས་ཀྱི་ནང་ནས་འབྱུང་
བའམ། འདི་རྣམས་གང་ཡང་རུང་བ་ནང་པར་རམ་ཟས་ཀྱི་གོང་དུ་རིམས་མི་བཟད་པ་ལ་སྦྱར་བར་བྱའོ། །འབྲས་བུ་གསུམ་དང་། རྒྱ་ཤུག་དང་། ཨག་ནི་མནྠ་དང་། འདི་རྣམས་བསྐོལ་བའི་ནང་དུ་མར་དང་མར་གྱི་ཉིས་འགྱུར་གྱི་ཞོ་དང་། མར་གྱི་བཞི་
ཆའི་སེང་ཕྲོམ་གྱི་ཤུན་ཕྲགས་གི་ཕྱེ་མ་བཏབ་སྟེ། ཡང་བསྐོལ་བའི་མར་ལས་རན་པར་འཐུངས་ན་རིམས་མི་བཟད་པ་འཇོམས་པའི་མཆོག་ཡིན་ནོ། །འབྲས་བུའི་ཆང་དང་། འབྲུའི་ཆང་ལེགས་པར་སྨིན་པ་དང་། རྨ་བྱ་དང་། སྲེག་པ་དང་། ཁྱིམ་
བྱའི་ཤ་གཙང་བ་དྲོ་བའི་ནུས་པ་དང་ལྡན་པ་ཟས་དང་ལྷན་ཅིག་ཅི་འདོད་པ་ཚིམ་ཚད་དུ་ཟོས་ལ། ཞག་དེ་ཉིད་ལ་གཉིད་ལོག་པའམ། ཡང་ན་སློན་དུ་གཞུག་གོ། །ཡང་ན་སྨན་མར་ཐུན་ཚད་ཆེན་པོ་འཐུངས་ལ་དེའི་འོག་ཏུ་སྐྱུག་སྨན་བྱའོ། །ཐོག་མར་སྣུམ་ཆོས་དང་
