ཅིང་བདེ་བར་འགྱུར་བ་དང་། ལྟེ་བའི་རྩ་བ་སོར་གཅིག་ལ་ཐེ་གུས་བཅིངས་ཏེ། དེ་ནས་སོར་བཞི་ལུས་པ་དང་བཅད་ལ། ཐེ་གུ་གོང་དུ་སྐོ་བ་ལ་བསྟོད་དེ་བཞག་ལ་སྐམ་མོ། །དེའི་འོག་ཏུ་ལྟེ་བ་ལ་རུ་རྟ་ཏིལ་མར་དང་ལྡན་པས་བསྐུའོ། །དེའི་འོག་ཏུ་ཤིང་འོ་མ་ཅན་རྣམས་སམ། ཡང་ན་སྨན་
དྲི་ཞིམ་པོ་རྣམས་ཆུ་དང་བསྐོལ་བ། དྲོ་འཇམ་ཙམ་གྱི་ནང་དུ་གསེར་དང་དངུལ་བསྲེགས་ཏེ་ནང་དུ་སྨྱུགས་ལ་དེས་ལྟེ་བ་བཀྲུའོ། །དེ་ལ་ལག་པ་གཡས་པའི་མཛུབ་མོས་བྱིས་པའི་མཚོག་མའི་སྟེང་དུ་རས་བལ་སྣུམ་ཅན་གྱིས་ལན་གཉིས་ལན་གསུམ་ཅུང་ཞིག་མནན་ཏེ་བཞག་སྟེ། མགོ་བོར་དས་དཀྲིས་
ཏེ་གཞག་གོ། །དེ་ནས་བློ་དང་ཚེ་དང་ཉམས་སྟོབས་སྐྱེད་པའི་ཕྱིར། བི་ཤ་ལ་དང་། བྲཧྨི་དང་། ཤུ་དག་དང་། ཤང་ཀ་པུཥྤ་དང་། དེ་རྣམས་བཏགས་པའི་ཕྱེ་མ་མར་དང་སྦྲང་རྩི་དང་ལྡེ་གུར་བྱས་ཏེ། ཐུན་ཚད་ཧ་རེ་ཎུ་ཀའི་ཚད་ཙམ་བྱའོ། །ཡང་ན་གསེར་དང་། ཤུ་དག་དང་། མུ་མེན་དང་། ཨ་
རུ་ར་དང་། འདི་རྣམས་ཞིབ་ཏུ་བཏགས་པའི་ཕྱེ་མ་མར་དང་སྦྲང་རྩི་དང་སྦྱར་བ་འམ། ཡང་ན་གསེར་དང་སྐྱུ་རུ་རའི་ཕྱེ་མ་སྦྲང་རྩི་དང་སྦྱར་བ་གཏང་ངོ་། །མངལ་གྱི་མི་གཙང་བའི་དྲི་མ་དག་པར་བྱ་བའི་ཕྱིར་ཤུ་དག་དང་རྒྱམ་ཚ་མར་དང་སྦྱར་བས་སློན་བྱའོ། །བྱིས་པ་བཙས་པའི་ལས་དང་ཆོ་ག་ལྷག་
མ་རྣམས་ནི་སྐྱེ་དགུའི་བདག་གིས་བཤད་པ་བཞིན་དུ་བྱས་ན་སྙིང་ལ་གནས་པའི་རྩ་རྣམས་འཕེལ་ཞིང་རྒྱས་པར་འགྱུར་རོ། །དེ་ནས་ཞག་གསུམ་པ་འམ་བཞི་པ་ནས་བུད་མེད་ཀྱི་འོ་མ་འབྱུང་བར་འགྱུར་རོ། །འོ་མ་མི་འབྱུང་བའི་ཞག་དང་པོ་ལ་ནི་མར་དང་སྦྲང་རྩི་དང་སྨན་ཨ་ནན་ཏ་བསྲེས་ཏེ་སྦྱར་བ་ལ་
སྔགས་ཀྱིས་བཏབ་སྟེ་བྱིས་པ་ལ་དུས་གསུམ་དུ་སྦྱིན་ནོ། །ཞག་གཉིས་པ་ལ་ནི་སྨན་ལག་ཤྨ་ན་དང་མར་སྦྱར་བ་སྦྱིན་ནོ། །ཞག་གསུམ་པ་ལ་ནི་ཞུན་མར་ཉི་ཚེ་བླུད་དོ། །ནམ་གྱི་མཚན་སྟོད་ལ་ནི་ཞུན་མར་རམ་མའི་ནུ་ཞོ་ལག་མཐིལ་གང་ཙམ་སྦྱིན་ནོ། །འོ་མ་འམ་མར་སར་ནི་དུས་གཉིས་
སུ་བླུད་དོ། །ནུ་ཞོ་ནི་མ་ཉིད་ཀྱིས་བླུད་ན་ཉམས་སྟོབས་འཕེལ་བར་འགྱུར་རོ། །འོ་མ་མང་བ་དང་བུ་ལ་མཉེས་གཤིན་ཅིང་བརྩེ་བའི་ནུ་མ་བསྣུན་པའི་མ་མ་གཉིས་བྱ་སྟེ། དེ་ཡང་ཡན་ལག་ལྷག་ཆད་མེད་ཅིང་སྨད་པ་མ་ཡིན་པ་དང་། ཚངས་པར་སྤྱོད་པ་དང་རིགས་མཉམ་པ་དང་།
