མིག་སྐྲང་བ་དང་། རིམས་རྣམ་བཞི་དང་འདི་རྣམས་སེལ་ཏོ། །ལ་ལ་ཕུད་དང་། འུ་སུ་དང་། ཟི་ར་དང་། འདི་རྣམས་ཞོ་བཞི་བཞི་དང་། པི་པི་ལིང་གི་རྩ་བ་དང་། ཚ་བ་གསུམ་དང་། འདི་རྣམས་སྲང་རེ་རེ་དང་། སེའུ་འབྲུ་དང་། ཁ་ར་དཀར་པོ་སྲང་བརྒྱད་བརྒྱད་
འདི་རྣམས་ཕྱེ་མར་བྱས་པ་ནི་སེའུ་འབྲུ་ཆ་བརྒྱད་པའི་ཕྱེ་མ་ཞེས་བྱ་སྟེ། དེ་ལས་རན་པར་ཟོས་ན་སྦྱོར་བ་གོང་མའི་ཡོན་ཏན་དང་འདྲའོ། །བད་ཀན་ལས་གྱུར་པའི་འཁྲུ་བའི་ཁ་ཟས་ནི་རླུང་ལས་གྱུར་པའི་འཁྲུ་བའི་ཁ་ཟས་དང་ཐུག་པའི་བྱེ་བྲག་ལ་སོགས་པ་སྦྱར་ཏེ་ཁྱད་པར་དུ་ཙང་གི་རི་དང་། རྒྱ་
ཤུག་སྐྱུར་པོ་རང་གི་ཁུ་བ་དང་། དར་བ་རྣམས་བསྐོལ་བའི་ནང་དུ་བྱི་དང་ག་དང་། ན་ལེ་ཤམ་དང་། ཀ་པིད་ཏ་དང་། དོང་གའི་ཕྱེ་མ་བཏབ་པའི་ཐུག་པ་ནི་ཁ་ལའི་ཐུག་པ་ཞེས་བྱ་སྟེ། དེ་ལས་རན་པར་འཐུངས་ན་བད་ཀན་ལས་གྱུར་པའི་འཁྲུ་བ་ལས་རྒྱལ་ལོ། །བད་
ཀན་ལས་གྱུར་པའི་མ་ཞུ་བའི་འཁྲུ་བ་ཟད་དེ། ཁུ་བར་འཁྲུ་བ་ལ་ནི་རྒྱ་སྐྱེགས་དང་། དོང་ག་དང་། པི་པི་ལིང་ལ་སོགས་པ་སྲང་དྲུག་གོང་དུ་སྨོས་པ་དང་ལྷན་ཅིག་ཏུ་ནས་ཐུག་དང་སྦྱར་བའམ། ཡང་ན་མར་རྙིང་པ་ནས་ཐུག་གམ། ཐུག་པ་རྐྱང་རྐྱང་པོ་དང་འཐུངས་
ན་ཕན་ནོ། །རླུང་དང་བད་ཀན་ལས་གྱུར་པའི་འཁྲུ་བ་འགགས་པ་དང་། བད་ཀན་ལས་གྱུར་པའི་ཤས་ཆེར་འཁྲུ་བ་དང་། ཟུག་གཟེར་དང་བཅས་ཤིང་མཁྲེགས་པ་ལ་ནི་པིད་ཚའི་འཇམ་རྩི་རབ་ཏུ་བསྔགས་ཏེ། དེ་ཡང་ཤུ་དག་དང་། བིལ་བ་དང་། པི་པི་ལིང་དང་།
རུ་རྟ་དང་། ཤུ་ཏི་དང་། རྒྱམ་ཚ་དང་། འདི་རྣམས་བིལ་བའི་མར་དང་ཚུལ་བཞིན་སྦྱར་བའམ། ཡང་ན་ཏིལ་མར་དང་བཅས་ཏེ་གཏང་ངོ་། །རླུང་དང་བད་ཀན་གྱི་འཁྲུ་བ་ཤས་ཆེར་ཉམ་ཐག་པ་ལ་ནི་ཨ་ནུ་བ་ས་ན་དྲོ་འཇམ་ཙམ་གཏང་ངོ་། །བད་ཀན་ཟད་དེ་ཡུན་རིང་པོར་
འཁྲུ་ཞིང་ལུས་ཉམ་ཆུང་བ་དང་། རླུང་གཞང་གི་ནང་དུ་སྟོབས་ཆེར་སྐྱེས་པ་ལ་ནི་དེ་མ་ཐག་རླུང་ཞི་བར་བྱ་བའི་ཆོ་ག་དང་། གང་ཤས་ཆེ་བའི་གསོ་བ་ལ་འབད་པར་བྱའོ། །དེ་ཡང་ཐོག་མར་རླུང་གསོ་བར་བྱ་སྟེ། དེ་ནས་མཁྲིས་པ་དང་། དེ་ནས་བད་ཀན་གསོའོ། །
