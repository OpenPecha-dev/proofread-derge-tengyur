ཡང་ན་རྒྱ་སྲན་ནམ་བལ་བ་ཛའི་ཁུ་བ་འམ། ཨ་ས་བའི་བཏུང་བ་སྦྱར་བ་གང་ཡང་རུང་བ་བླུད་དོ། །ཡང་ན་ཤུ་ཏི་དང་། ཡུངས་ཀར་དང་། ཨ་ཛ་ཛི་ཞེས་བྱ་བ་ཟི་ར་དང་། ན་ལེ་ཤམ་དཀར་པོ་དང་། ཏ་ཏིཀྵྞ་ཀ་དང་། ཀྲུ་ཏྲུག་ཏྲེས་དང་། ཤིང་ཀུན་དང་། རུ་རྟ་དང་། པོ་
སོན་ཆ་དང་། འདི་རྣམས་ཕྱེ་མར་བྱས་པ། གཅིན་དང་། འོ་མ་དང་། ཡུངས་མར་དང་། འདི་རྣམས་ཚུལ་བཞིན་དུ་བསྐོལ་བའི་མར་གྱིས་བཤང་བའི་ལམ་མམ། མོ་མཚན་གྱི་སྒོ་ནས་ཨ་ནུ་བ་ས་ན་བཏང་ན་རབ་ཏུ་ཕན་ནོ། །ཡང་ན་ཤུ་ཏི་དང་། ཤུ་དག་དང་།
ཡུངས་ཀར་དང་། པི་པི་ལིང་དང་། འདི་རྣམས་ཚུལ་བཞིན་དུ་བཏགས་ལ་ལན་ཚྭ་དང་། མར་དང་བཅས་པ་སྦྱར་བ་ནི་རུ་ཧ་མས་གཏོང་ན་ཤ་མ་འབྱིན་པར་བྱེད་དོ། །གལ་ཏེ་ཤ་མ་མ་བྱུང་ན་ནི་རླུང་ནད་ཀྱི་རྒྱུར་འགྱུར་ལ་ཕྱིར་བྱུང་ན་ནི་རླུང་ཡང་ཕྱིར་འབྱིན་པར་བྱེད་དོ། །གོང་
མའི་ཐབས་གང་གིས་ཀྱང་ཤ་མ་མ་ཐོན་ན་བུད་མེད་མཁས་པ་སྔར་གོམས་པ་ཡོད་པས་ལག་པའི་སེན་མོ་བྲེགས་ལ་སྣུམ་གྱིས་བསྐུས་ཏེ། ལག་པ་ནང་དུ་བཅུག་ལ་དལ་བུས་རིམ་གྱིས་དབྱུང་ངོ་། །དེ་ནས་བུ་དང་ཤ་མ་གཉིས་ཀ་བྱུང་ནས་གྲོལ་བ་ལྟ་ན། ཏིལ་མར་གྱིས་མོ་མཚན་
དང་ལུས་ཐམས་ཅད་བསྐུ་མཉེ་བྱའོ། །བུད་མེད་བུ་བྱུང་བའི་འོག་ཏུ་ནད་མག་ཀ་ཞེས་བྱ་བ་འབྱུང་སྟེ། དེའི་མཚན་མ་ནི་འོག་ནས་ཆུ་སོ་དང་། ལྟོ་བ་དང་། བརླ་བྲག་ཏུ་ཟུག་ཅིང་ན་བ་ཡིན་ཏེ། དེ་ལ་ནི་ནས་ཚིག་འཁུས་པའི་ཐལ་བ་ཞིབ་པར་བཏགས་ཏེ། མར་རམ་ཡང་
ན་ཆུ་དྲོན་པོ་བསྐོལ་བ་གང་ཡང་རུང་བ་དང་སྦྱར་བའམ། ཡང་ན་འུ་སུའི་འབྲས་བུ་བསྐོལ་བ་བུ་རམ་དང་། ཚ་བ་གསུམ་དང་དྲི་སྣ་གསུམ་ཞེས་བྱ་བ་ཤིང་ཚ་དང་། སུག་སྨེལ་དང་། གནྡྷ་བྷ་དྲ་རྣམས་ཀྱི་ཕྱེ་མ་དང་སྦྱར་ལ་བླུད་དོ། །ཡང་ན་སྨན་པ་རྒན་པོ་རྣམས་ན་རེ་དྲི་སྣ་གསུམ་
