མ་ཡང་བ་དང་ཕྱི་ས་སླ་བར་འགྱུར་བའི་བག་གང་ཡང་རུང་བ་ཚོད་མར་བྱས་པའི་ནང་དུ་ཤིང་ཀུན་དང་། ཏིལ་དང་། མར་དང་། ཞོའི་སྤྲིས་མ་དང་། འུ་སུ་དང་། ཚ་བ་ལྔ་བཏགས་པའི་ཕྱེ་མ་དང་། འུ་སུའི་གེ་སར་དང་། སྒེའུ་གཤེར་གཤགས་
པ་དང་། འདི་རྣམས་སྤོད་ཐབས་སུ་བཅུག་པ་ལེགས་པར་སྦྱར་ཏེ། ནད་པ་གང་ལ་དགའ་བའི་སྤོས་བསྲེགས་པས་ཚོད་མ་བདུགས་ལ། དེ་ནས་ཟི་ར་དང་། ན་ལེ་ཤམ་དང་། བི་ཊ་ཁ་རུ་ཚ་འདི་དག་བཏགས་པས་ཤས་ཆེར་
བཏབ་སྟེ་ཟོས་ན་རླུང་ཤས་ཆེ་བ་དང་། གཞང་འབྲུམ་དང་། ལུས་རྩུབ་པ་དང་། མེ་དྲོད་ཆུང་བ་དང་། རྟུག་སྐམ་རྣམས་ལ་ཕན་ཏེ་སྦྱར་བར་བྱའོ། །ཟས་ནི་འབྲས་དམར་པོ་ལས་གཡོས་སུ་བྱས་ལ། ཚོད་མ་ནིམ་པ་དང་། སྲོག་ཆགས་གོ་བོ་དང་། ར་དང་།
རྔ་མོ་དང་། ཁྱད་པར་དུ་སྲོག་ཆགས་ཤ་ཟ་བའི་ཤ་ཁུའི་ནང་དུ་ཤིང་ཀུན་ལ་སོགས་པའི་སྤོད་དང་སྦྱར་བ་ཐབས་གོང་མ་བཞིན་དུ་བྱས་ཏེ་ཟོས་ན་ཕན་ཡོན་གོང་མ་དང་འདྲའོ། །བཏུང་བ་ནི་ཆང་དང་ཤ་ཁ་ར་མི་བཏུང་བ་དང་། བུ་རམ་གྱི་ཆང་དང་། དར་བ་དང་། རྩབ་
མོ་དང་། ཨ་རིཥྚའི་ཆང་དང་། ཞོ་ག་ཆུ་དང་འདི་རྣམས་བཏུང་ངོ་། །ཡང་ན་ཆུ་ཉི་ཚེར་འུ་སུ་དང་། བཅའ་སྒ་ཆུ་དང་བསྐོལ་བ་བཏུང་བའམ། ཡང་ན་ཀཎྜ་ཀ་རི་ཆུ་དང་བསྐོལ་བ་ཉུང་ཤས་བཏུང་ངོ་། །བཏུང་བ་དེ་རྣམས་ཀྱང་ཟས་ཀྱི་བར་རམ་ཟས་ཀྱི་འོག་ཏུ་འཐུངས་
ན་རླུང་གི་རྟུག་སྐམ་འབྱིན་ནོ། །རླུང་དང་མཁྲིས་པ་དང་བད་ཀན་ལས་གྱུར་པའི་རྟུག་སྐམ་བདེ་བར་འབྱུང་ཞིང་གཞང་འབྲུམ་གྱི་ནད་ཀྱང་ཞི་བར་འགྱུར་ཏེ། མེའི་དྲོད་ཀྱང་སྐྱེད་པར་འགྱུར་རོ། །རྟུག་སྐམ་གྱི་ཟིལ་གྱིས་མནན་ཏེ་ཁོང་ཤིན་ཏུ་རྩུབ་པ་དང་། རླུང་ལོག་པར་
སོང་བ་དང་། གཟེར་ཞིང་ན་བས་ཉམ་ཐག་པ་ལ་ནི་འཇམ་རྩི་མས་གཏོང་བ་རབ་ཏུ་ཕན་ཏེ། དེ་ཡང་པི་པི་ལིང་དང་། པོ་སོན་ཆ་དང་། བིལ་བ་དང་། ཤུ་ཏི་དང་། ཤིང་མངར་དང་། ཤུ་དག་དང་། རུ་རྟ་དང་། ཟུར་པ་དང་། མ་ནུ་དང་། ཀྲུ་ཏྲུག་ཏྲེས་དང་།
