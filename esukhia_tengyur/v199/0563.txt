ངོ་། །དེ་ལྟར་ལེགས་པར་སྦྱར་བས་ནི་སྨན་རྣོ་བ་ཡང་འཇམ་པོར་འགྱུར་ལ། འཇམ་པོ་ཡང་རྣོ་བར་འགྱུར་རོ། །དེ་ལྟར་གོང་དུ་བཤད་པའི་བཀྲུ་སྨན་དེ་དག་ཀྱང་ཡིད་དང་རྗེས་སུ་མཐུན་པའི་སྨན་ཤིང་ཚ་དང་། ནཱ་ག་གེ་སར་དང་། ཨ་མྲ་ཏ་ཀ་དང་། སེའུ་འབྲུ་དང་། སུག་
སྨེལ་དང་། ཁ་ར་དཀར་པོ་དང་། སྦྲང་རྩི་དང་། ཁ་ལུང་དང་ཆང་ལ་སོགས་པ་དྲི་ཞིམ་པོས་བསྒོས་ཤིང་བཏབ་ལ་ལྡན་པར་བྱས་ལ་བླུད་པར་བྱའོ། །ཆོ་ག་གྲུབ་པའི་གནས་ལས་བཀྲུ་སྨན་བཏང་བའི་ལེའུ་གཉིས་པ་རྫོགས་སོ།། །།དེ་ནས་སྐྱུག་སྨན་
དང་བཀྲུ་སྨན་དང་དེ་གྲུབ་པར་བྱ་བའི་ཆོ་གའི་ལེའུ་བཤད་པར་བྱ་སྟེ། གལ་ཏེ་སྐྱུག་སྨན་དེ་ལྟོ་སྙི་བ་དང་། ཤིན་དུ་ལྟོགས་པ་དང་། བད་ཀན་ཆུང་བ་ལ་གཏང་ངམ། ཡང་ན་སྨན་ཤིན་ཏུ་རྣོ་བ་འམ། གྲང་ཆེས་པའམ། ཐུན་ཆུང་བ་འམ། ཟས་མ་ཞུ་བ་འམ། ཉམ་ཆུང་ཆེས་པ་ལ་བླུད་ན།
སློན་གྱེན་དུ་མི་འབྱུང་བར་ཐུར་དུ་འགྲོ་བར་འགྱུར་ཏེ། ཉེས་པ་ཕྱིར་སོགས་པར་འགྱུར་བས། དེ་བས་ན་དེ་ལ་ནི་ཡང་སྣུམ་ལ་སོགས་པ་གོང་མའི་རིམ་པ་བྱས་ནས་སྨན་ཡང་གཏོང་ངོ་། །བཀྲུ་སྨན་ཡང་ཟས་མ་ཞུ་བ་དང་། བད་ཀན་དང་ལྡན་པ་ལ་གཏང་ངམ། ཡང་ན་བཀྲུ་སྨན་དེ་ཉིད་ཤིན་ཏུ་རྣོ་བ་
དང་། དྲོ་ཆེས་པ་དང་། ལན་ཚྭ་མང་ཞིང་སྨན་ཐུན་ཆེས་པ་དང་། ཡི་གར་མི་འོང་བཞིན་དུ་བྱིན་ན་སླར་སྟོད་དུ་འབྱུང་བར་འགྱུར་ཏེ། དེ་ལ་ནི་གོང་དུ་བཤད་པའི་ཆོ་ག་བཞིན་དུ་བཀྲུ་སྨན་ཡང་བྱའོ། །བཀྲུ་སྨན་དེ་བཏང་ཡང་ལེགས་པར་མ་གྱུར་ཏེ། ཕོ་བ་ལ་བརྟེན་ཞིང་འཁྲུར་མི་བཏུབ་པ་ལ་ནི་
གོམས་ཤིང་ཡི་གར་འོང་བ་སྤངས་ལ། གང་གནོད་པ་མི་སྐྱེད་པའི་བཀྲུ་སྨན་གྱིས་བསྣན་ནོ། །སྐྱེས་བུ་གང་གི་ལུས་སྣུམ་ཆོས་དང་དུགས་ཀྱིས་ཞེན་པར་བྱས་པ་ལ་སྨན་རྙིང་པ་རྩུབ་པའི་བཀྲུ་སྨན་བཏང་བས་ནད་རང་གི་གནས་ནས་བསྐྱོད་ཅིང་འཁྲུ་མ་ནུས་པས་མགོ་འཁོར་བ་དང་། སྐྲང་བ་
དང་། སྐྱིགས་བུ་དང་། མུན་པར་མཐོང་བ་དང་། སྐོམ་དད་ཆེ་བ་དང་། བྱིན་པའི་ཉན་བ་དང་། ལུས་གཡའ་བ་དང་། མགོ་བོ་ཚ་བ་ཆེ་ཞིང་མདོག་འགྱུར་བ་ལ་སོགས་པའི་ནད་དེ་རྣམས་སྐྱེད་པར་བྱེད་དོ། །ནད་པ་དེ་སྣུམ་དང་དུགས་ཀྱིས་ལེགས་པར་བྱས་ཀྱང་མེ་དྲོད་ཆེ་བ་ལ་སྨན་རྙིང་པ་ཐུན་ཆུང་
