བྱེད་པའི་ཚིག་ནི་མེད་པར་བལྟ་བར་བྱའོ། །ད་ནི་ཚིག་ལེའུར་བྱས་པ་དྲུག་གི་ཕན་ཡོན་དང་སྔགས་བཏུ་བ་དང་། བཟླས་པའི་གྲངས་ཀྱང་སྟོན་ཏེ། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། དེ་ནས་ཞེས་བྱ་བའི་སྒྲ་ནི་སྔགས་བཏུ་བ་སྟོན་པའི་དོན་ཏོ། །རྣལ་འབྱོར་གྱི་
དབང་ཕྱུག་ནི་རྣལ་འབྱོར་གྱི་སྤྱི་གཙུག་གི་ནོར་བུའོ། །དེ་རྣམས་དང་འབྲེལ་པ་གསང་བའི་ཡི་གེ་རབ་ཏུ་བཤད་པ་ཞེས་བྱ་བར་སྦྱར་རོ། །མི་བསྟན་པ་ཉིད་ཀྱིས་ན་གསང་བའོ། །ཡང་ན་འདིས་ལྷའི་དེ་ཁོ་ན་ཉིད་གསང་བས་ན་གསང་བའོ། །གསང་བ་ཡང་ཡིན་ལ་ཡི་གེ་
ཡང་ཡིན་ལས་ན་གསང་བའི་ཡི་གེའོ། །དེ་འགྲོ་བ་དམ་པ་ལ་བརྟེན་ནས་རབ་ཏུ་བཤད་པ་སྟེ། བསྟན་པར་བྱ་བའོ་ཞེས་རྡོ་རྗེ་འཆང་གིས་གསུངས་སོ། །ཇི་ལྟ་བུ་ཞེ་ན། ཐབས་ཀྱི་དམ་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། འཇིག་རྟེན་ལས་འདས་པའི་རིགས་པས་
སོ། །རབ་ཏུ་བསྟན་པའི་ཕྱིར་ཞེས་བྱ་བ་ནི་རབ་ཏུ་གསལ་བར་བྱ་བའོ། །གཞན་ཡང་ཇི་ལྟ་བུ་ཞེ་ན། རྣལ་འབྱོར་ཞེས་བྱ་བ་ནི་ལྷ་ཉིད་ཀྱི་ཏིང་ངེ་འཛིན་ཏེ། དེའི་དངོས་གྲུབ་མཐར་ཕྱིན་པའོ། །འབྲས་བུ་ནི་དེ་ཉིད་རྫོགས་པ་སྟེ། ཕྱག་རྒྱ་ཆེན་པོའི་དངོས་གྲུབ་པོ། །རྣམ་
པ་དེ་གཉིས་རབ་ཏུ་བསྟན་པས་སྟེར་ཞེས་བྱ་བ་རྣལ་འབྱོར་གྱི་དངོས་གྲུབ་ཀྱི་འབྲས་བུ་སྟེར་བའོ། །གང་གི་བསམས་པ་ཙམ་གྱིས་ཞེས་བྱ་བ་རྣལ་འབྱོར་གྱི་དངོས་གྲུབ་མ་གཏོགས་པ་བཟླས་པ་ཙམ་གྱིས་ཞེས་བྱ་བའི་དོན་ཏོ། །རྣལ་འབྱོར་པས་མཆོག་སྟེར་བར་ནུས་
པར་འགྱུར་བའོ། །མཆོག་ནི་མངོན་པར་འདོད་པ་སྟེ། རིག་སྔགས་འཆང་ལ་སོགས་པ་ཉིད་དོ། །དང་པོ་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། འདིར་དང་པོའི་ཡི་གེ་ནི་ཨོཾ་མོ། །དེ་སྔགས་དུ་མའི་ཐོག་མར་མཐོང་བ་ཉིད་པས་ན། དེའི་སྦྱོར་བས་རིམ་གྱིས་རྗོད་པར་བྱེད་
པའི་རྣལ་འབྱོར་པ་ཞེས་སྦྱོར་རོ། །དེ་ཅི་ཞིག་བྱེད་ཅེ་ན། དབུགས་རྔུབ་པར་བྱེད་པ་སྟེ། རླུང་གྱེན་དུ་རྒྱུ་བའོ། །རྟག་ཏུ་ཞེས་བྱ་བ་ནི་བསྒོམ་པའི་གནས་སྐབས་ནའོ། །བརྒྱད་པའི་མཐའ་ཞེས་བྱ་བ་ནི་ཡི་གེ་ཧའོ། །ལྡན་པ་ནི་ཡི་གེ་ཨུ་སྟེ་འདྲེས་པར་གྱུར་ཅེས་བྱ་
