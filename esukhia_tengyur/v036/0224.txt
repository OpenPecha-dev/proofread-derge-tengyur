གསུངས་པའི་ཤེས་པས་བསྲུང་བར་བྱ། །ལྷུང་བཟེད་དེ་དག་ལས་ནི་གཅིག་ཐོགས་ནས། །གཉའ་ཤིང་གང་དུ་བལྟ་ཞིང་བསོད་སྙོམས་སྤྱད། །འཛུམ་ཞིང་འགྲོ་ལ་སྨྲ་ཞིང་མིག་ཟུར་ཅན། །ཡན་ལག་ཐམས་ཅད་ཡིད་འཕྲོག་བྱེད་པའི་གཟུགས། །བུད་མེད་གཟུགས་ནི་
མཚོན་ཆ་འོབས་འདྲ་བས། །སྐྱེས་པའི་སེམས་ནི་མངོན་དུ་འཕྲོག་པར་བྱེད། །ལྕགས་ཀྱི་ཐོང་ལྕགས་རབ་ཏུ་འབར་བ་ཡིས། །མིག་ནི་རྣམ་པར་བཀག་པ་ཆེས་སླའོ། །ཉིས་ན་དྲན་པ་ཉམས་བྱེད་བུད་མེད་ཀྱི། །དཀྱུས་རིང་མིག་ཅན་བཞིན་ནི་མ་མཐོང་སླ། །གང་དུ་དགའ་
བར་འཛིན་ཅིང་མཐོང་བྱེད་ལ། །སེམས་ཀྱིས་སེམས་ནི་རྣམ་པར་འདུལ་བཞིན་དུ། །རབ་དང་འབྲིང་ལ་སོགས་པ་སྤངས་ནས་སུ། །མི་སྨྲ་བ་ནི་གཞན་གྱི་ཁྱིམ་དུ་འགྲོ། །གང་ན་བཞོན་བེའུ་ཕྲུ་གུ་ཆུང་བ་དང་། །གང་ན་བུ་མོ་གཞོན་བསྟེན་རྩེ་བ་དང་། །དེ་བཞིན་གང་ན་
གཞོན་ནུ་རྩེ་བ་དང་། །གང་ན་ཁྱེའུ་ཟ་དགའ་སྟོན་བྱེད་པ་དང་། །གང་ན་སྐྱེ་བོ་མང་པོ་འདུས་པ་དང་། །དེ་བཞིན་གང་ན་རོལ་མོ་བྱེད་པ་རྣམས། །དེར་ནི་སྐྱེས་བུ་ནམ་ཡང་འགྲོ་མི་བྱ། །བསོད་སྙོམས་ཅི་རྙེད་ཁྱེར་ཏེ་འོངས་ནས་ཀྱང་། །དང་པོ་ཇི་ལྟར་བདེ་བར་བགོ་བར་
བྱ། །རྐང་བཀྲུས་གློ་བུར་བ་ལ་བྱིན་ནས་སུ། །ལུས་བསྲུང་ཕྱིར་ནི་ཁ་ཟས་བཟའ་བར་བྱ། །དུས་སུ་ཟོས་ནས་དུས་སུ་ཁྲུས་བྱས་ཏེ། །གང་ཚེ་ཟློས་དགའི་སེམས་ནི་ཞི་བ་ལ། །བརྟེན་པ་ཉིད་ཀྱིས་དྲན་པ་རྨོངས་གྱུར་ན། །ལངས་ནས་འཆག་ཅིང་ཕྱོགས་སུ་བལྟ་བ་
དང་། །ཆུ་ཡིས་གདོང་ཡང་ཡོངས་སུ་བཀྲུ་བར་བྱ། །གླིང་བུའི་སྒྲ་དང་རྔ་ཡི་སྒྲ་རྣམས་དང་། །ཤིན་ཏུ་འཇིགས་བྱེད་སྐྱེ་བོའི་སྒྲ་དག་དང་། །གད་རྒྱངས་ཀུ་ཅོ་དག་དང་གླུ་སྒྲ་དང་། །བིང་དང་པི་ཝང་དག་དང་རྫ་རྔའི་སྒྲ། །བྱ་བྲན་སྣ་ཚོགས་སྒྲ་དང་ཆུ་སྒྲ་དང་། །བུད་
མེད་རྒྱན་གྱི་སིལ་ཁྲོལ་སྒྲ་རྣམས་དང་། །ཤ་སྦྲང་ཚ་གྲང་རླུང་གི་གནོད་པ་ཡིས། །མི་ཡི་སེམས་ནི་གཡེང་ཞིང་གཡོ་བར་བྱེད། །སྟངས་དང་མཆོངས་པ་དང་ནི་གར་བྱེད་དང་། །གླུ་ལེན་རྔ་བརྡུང་སྐྱེ་པོས་དབྱས་འགྱུར་དང་། །ལྷག་པའི་ང་རྒྱལ་སྲེད་དང་འདོད་ཆེན་དང་། །
