རབ་སྟེར་བའི། །ཞེས་གནས་སྐབས་དེ་ཉིད་མཐར་ཐུག་ཏུའོ། །དཀྱིལ་འཁོར་འཇིགས་པ་བསམ་པར་བྱ། །ཞེས་པ་བསྒོམ་པ་དེ་ཉིད་དང་རྗེས་སུ་འབྲེལ་པ་བློའི་ཡུལ་དུ་བྲི་བར་བྱ་བ་ཡང་གསལ་གདབ་བའོ། །མཁའ་འགྲོ་དྲ་བའི་བདེ་མཆོག་མཆོད། །ཅེས་པ་ཕྱི་ནང་གི་མཆོད་པ། མེ་ཏོག་
བདུག་སྤོས་མར་མེ་ཞལ་ཟས་བྱུག་པ་སྒྲ་སྙན་ལ་སོགས་པ་དང་། ཡང་གཟུགས་རྡོ་རྗེ་ལ་སོགས་པ་རིག་མ་བཅུ་དྲུག་ལ་སོགས་པ་དང་། ཤ་ལྔ་བདུད་རྩི་དང་། དེ་ཁོ་ན་ཉིད་ཀྱི་མཆོད་པས་མཉེས་པར་བྱ། མཐར་བསྟོད་པ་བྱས་ནས་གསོལ་བ་གདབ་པར་བྱ་བ་ལ་སོགས་པས་སོ། །དེའི་རྗེས་ལ་སློབ་མ་
ལྷག་པར་གནས་པར་བྱ་བ་ལ། ཉེ་རེག་མཐོར་འཐུང་གིས་བསང་བ་བྱ་སྟེ། འབྱུང་པོ་ལ་གཏོར་མ་བྱིན་ནས། གསོལ་བ་གདབ་ཅིང་གདམས་པ་བསྟན་པར་བྱའོ། །བསྲུང་སྐུད་བཅིང་། སྐྱོར་ཆུ་བླུད། ཀུ་ཤའི་སྔས་དང་སྟན་ཁྲོ་བོའི་སྔགས་ཀྱིས་བསངས་པ་བྱིན་ནས། རྨི་ལམ་བརྟག་ཏུ་གཞུག་སྟེ།
མཚན་མ་ངན་ན་བཟློག་པར་བྱའོ། །ད་ནི་རྡོ་རྗེ་སློབ་དཔོན་གྱི་བྱེད་ལས་བསྟན་པ་ལས། དང་པོར་ཐིག་གི་སྦྱོར་བ་བསྟན་པ་ནི། དེ་ནས་སེམས་མེད་སྲད་བུའམ། །ཞེས་སྐྱེས་པ་བུད་མེད་གྲིར་ཤི་བའི་རོའི་རྒྱུས་པ་བསྔགས་ལ། ཡང་ན་རོའི་སྐྲའམ་རས་བ་ལ་རོ་ལ་དཀྲིས་པའམ། ཡང་ན་གཞོན་ནུ་མ་ཉམས་པས་བཀལ་བའི་
རས་བལ་བྱ་བ་ཕྲ་སྦོམ་སྒོའི་ཉི་ཤུ་ཆའོ། །རུ་དྷི་ར་ནི་བསྒོས་པ་ནི་དཔའ་བོ་གཡུལ་དུ་ཤི་བའི་ཁྲག་ཆེན་པོས་བསྒོས་པའི་དོན་ཏོ། །ཀྱང་ནི་བར་མའི་སྒྲ་སྟེ་བདུད་རྩི་ལྔའོ། །མ་ད་ནས་བྱུགས་པ་ཡང་ངོ་། །དཔལ་ལྡན་ཁྲག་འཐུང་གྲོང་ཁྱེར་མཆོག །ཅེས་པ་ཐར་བ་ཆེན་པོའི་ཕོ་བྲང་། འཁོར་ལོ་བདེ་མཆོག་
གི་ཡོན་ཏན་ཕུན་སུམ་ཚོགས་པའི་འབྱུང་གནས་སོ། །དཀྱིལ་འཁོར་ནི་གཞལ་ཡས་ཁང་ཚད་མེད་པ་སྙིང་པོ་ལེན་ཅིང་འཛིན་པ་སྟེ། དུར་ཁྲོད་ཆེན་པོ་བརྒྱད་ཀྱིས་བསྐོར་བ་ནི་འཇིགས་པ་སྟེ་ལོག་འདྲེན་རྣམས་སྐྲག་པར་བྱེད་པའོ། །ཡང་ན་ཐིག་གདབ་པར་བྱ་བའི་སྲད་བུ་ཁ་དོག་ལྔ་པ་ལ། ཧཱུཾ་བྷུཾ་
ཏྲཾ་ཧྲཾ་ཁཾ། འོད་ཟེར་སྤྲོ་བསྡུས་ཡེ་ཤེས་ཀྱི་ཆ་སྤྱན་དྲངས་ཏེ་བསྟིམས་པ་རྗེ་བཙུན་གྱི་སྙིང་པོ་ཉེ་སྙིང་བརྗོད་པས་མཆོད་པས་ལེགས་པར་བསྒྲིལ་བར་བྱའོ། །རིགས་བསྡུས་པའི་དཀྱིལ་འཁོར་ཚད་ངེས་པ་ཡང་། ཁྲུ་བཞི་པའམ་ཁྲུ་བརྒྱད་པ། །ཞེས་པས་ཐ་མ་འབྲིང་ནས་ཁྲུ་བཅུ་དྲུག་པའི་བར་གྱིས་སོ། །
