ཀྱི་སྤྱོད་མཆོག་སྤྱོད་པ་རྣམས། །འདོད་པ་ཐམས་ཅད་བསྟེན་པ་ཡིས། །ཆོས་ཀྱི་ཡི་གེ་མཆོག་བརྙེས་སོ། །<𰵓བད་,བདེ་>དང་སྡུག་བསྔལ་དེ་བདག་མིན། །འདོད་པའི་ཡོན་ཏན་ལྔ་རྣམས་བསྟེན། །ཡེ་ཤེས་དོན་གཉེར་འདོད་ཆགས་ཅན། །སྦྱོར་བས་བསྒྲུབས་ན་མྱུར་དུ་འགྲུབ། །སྐུ་
གསུང་ཐུགས་ནི་ཡང་དག་འགྲུབ། །གང་གཞན་སྨན་པར་བཤད་པ་ནི། །སྔགས་ཀྱི་བཟླས་པས་འགྲུབ་འགྱུར་བ། །སྐུ་གསུང་ཐུགས་ནི་བསྒོམས་པས་སོ། །དེ་ཉིད་ཀྱིས་ནི་རྡོ་རྗེ་གཟུང་། །ཆོས་ཀྱི་དྲིལ་བུ་དཀྲོལ་བར་བྱ། །དམ་ཚིག་གི་ནི་ཕྱག་རྒྱ་ཆེར། །བྱིན་གྱིས་བརླབས་ནས་
སྙིང་པོ་བཟླ། །ཞེས་གང་བཤད་པ་དེ་ཡི་ཕྱིར། །དེ་སྐད་ཅེས་བཤད་པ་ཡིན་ནོ། །སློབ་མ་སྡོམ་པ་གཟུང་བ་དང་། །དེ་བཞིན་ཡོན་ནི་ཕུལ་བ་ཡིས། །དབང་བསྐུར་ཕྱིར་ནི་གསོལ་འདེབས་ཤིང་། །ཚིགས་བཅད་འདིས་ཀྱང་ཕྱག་བྱའོ། །བྱང་ཆུབ་རྡོ་རྗེ་སངས་རྒྱས་ཀྱིས། །མཆོད་
ཆེན་ཇི་ལྟར་གསོལ་བ་བཞིན། །བདག་ཀྱང་རབ་ཏུ་བསྐྱབ་པའི་ཕྱིར། །ནམ་མཁའི་རྡོ་རྗེ་བདག་ལ་སྩོལ། །འཇུག་སྒོར་པདྨ་འདབ་བརྒྱད་པར། །རང་ལྷའི་སྦྱོར་བས་ཁྲི་ལ་གནས། །དུས་གསུམ་ཕྱོགས་ཀུན་ནམ་མཁའ་ཁྱབ། །སངས་རྒྱས་འཁོར་ལོ་རང་སྙིང་བྱུང་། །རོལ་མོ་དྲི་ལ་སོགས་
ལྡན་ཞེས། །སྔ་མ་བཞིན་དུ་རིག་དབང་བསྐུར། །རྡོ་རྗེ་ཆེན་པོའི་དབང་བསྐུར་བ། །ཚིགས་བཅད་འདིས་ནི་དབང་བསྐུར་རོ། །རྡོ་རྗེ་ཆེན་པོའི་དབང་བསྐུར་བ། །ཁམས་གསུམ་གྱིས་ནི་ཕྱག་བྱས་པ། །གསང་བ་གསུམ་གྱི་གནས་ལས་བྱུང་། །སངས་རྒྱས་ཀུན་གྱིས་སྦྱིན་པར་བྱ། །མི་ལྡོག་དབང་བསྐུར་བྱིན་ནས་
ནི། །འཁོར་ལོ་དེ་ཉིད་སྦྱིན་པར་བྱ། །མི་མཉམ་མེད་པ་ཟུར་བཞི་པ། །སངས་རྒྱས་སངས་རྒྱས་མི་མཉམ་ཉིད། །ལུས་ཚོར་སེམས་དང་ཆོས་རྣམས་ཀྱིས། །གཅིག་དང་སྣ་ཚོགས་སྦྱོར་བྲལ་ཉིད། །དེ་དྲན་དེ་ལ་དད་པ་གང་། །ཤར་སྒོར་བྱང་ཆུབ་ཕྱིར་འདོད་དོ། །དེ་ནས་བྱུང་
དང་འབྱུང་འགྱུར་བའི། །ཕྱིན་ཅི་ལོག་སྤང་མ་སྐྱེས་བྱ། །མ་སྐྱེས་སྐྱེས་པའི་ཡང་དག་ཀྱང་། །བསྐྱེད་པ་དང་ནི་གནས་བྱ་བ། །བརྩོན་འགྲུས་རྣམ་བཞི་ལྷོ་ཡི་སྒོ། །མདུན་དང་སྤྲོ་དང་གནས་དང་བློ། །རྫུ་འཕྲུལ་རྐང་དང་དྲན་པ་འདིར། །ནུབ་ཀྱི་སྒོ་ནི་ཡིན་པར་བཤད། །དད་དང་
