རྣམས་ནི་སྤྲོ་བར་བྱ། །ཕྱོགས་བཅུར་གནས་པའི་དེ་བཞིན་གཤེགས། །བྱང་ཆུབ་ལ་སོགས་ཐམས་ཅད་ནི། །དེ་རྣམས་ཀྱིས་ནི་བཀུག་ནས་སུ། །བདག་ཉིད་མདུན་དུ་བཞག་པར་བསམ། །འོད་ཟེར་དེ་རྣམས་མཆོད་པའི་ཕྱིར། །རྡོ་རྗེ་ལ་ནི་ཐིམ་པར་བསམ། །
ཕྱི་ནས་རྡོ་རྗེ་དེ་ཉིད་ནི། །ཟླ་བའི་སྟེང་དུ་ཐིམ་པར་བྱ། །དེ་ནས་ཟླ་བའི་དཀྱིལ་འཁོར་ལ། །ཀ་སོགས་སྡེ་ཚན་དྲུག་པོ་ལ། །ཐུང་དང་རིང་པོའི་དབྱེ་བ་ཡིས། །བཅུ་གཉིས་བརྩེགས་པའི་བདག་ཉིད་བསམ། །ཡི་གེ་དེ་རྣམས་གྱུར་པ་ལས། །གླུ་མཁན་རོལ་མོ་དྲི་མ་དང་། །ཕྲེང་བ་སྤོས་
དང་མར་མེ་དང་། །ལྷ་བཤོས་མ་དང་མི་ཟད་མ། །ལཱ་སྱེ་བཞད་པ་གར་མཁན་མ། །དེ་བཞིན་ལྷ་མོ་འདོད་མ་ནི། །རིམ་པ་ཇི་བཞིན་སྤྲོ་བར་བྱ། །མཆོད་པའི་ལྷ་མོ་རེ་རེ་ཡིས། །འདོད་པའི་ལྷ་སོགས་མཆོད་པར་བྱ། །དེ་ནས་སྔགས་འདི་བརྗོད་ནས་སུ། །སངས་རྒྱས་ལ་སོགས་
མཆོད་པར་བསམ། །དེ་ལ་མཆོད་པའི་སྔགས་ཀྱི་རྐང་པ་རྣམས་ནི། །ཨོཾ་བཛྲ་གནྡྷེ་གནྡྷ་ཨརྩྩ་ནཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་མཱ་ལེ་མཱ་ལ་ཨརྩྩ་ནཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་དྷཱུ་པེ་དྷཱུ་པ་ཨརྩྩ་ནཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་པྲ་དཱི་པེ་པྲ་དཱི་པ་ཨརྩ་ནཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨ་མྲྀ་ཏེ་ནཻ་བཻ་ཏྱེ་ཨརྩ་ནཾ་ཀུ་
རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཨཀྵ་ཏེ་ཨཀྵ་ཏ་ཕ་ལ་ཨརྩྩ་ནཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ལཱ་སྱེ་བསྟྲ་ཨ་བྷ་ར་ཎ་པཱུ་ཛཱཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཧཱ་སྱེ་གཎྚེ་གྷཎྚ་ཨཱ་དརྴ་པཱུ་ཛཱཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཝཱ་དྱེ་ཝཱ་དྱ་པཱུ་ཛཱཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ནྼྀ་ཏྱེ་ནྼྀ་ཏྱ་པཱུ་ཛཱཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་གཱི་ཏི་གཱི་ཏ་པཱུ་ཛཱཾ་
ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། ཨོཾ་བཛྲ་ཀཱ་མེ་སརྦ་བུདྡྷ་བོ་དྷི་སཏྭཱ་ནཱཾ་ཀྲོ་དྷ་ཨཱ་དི་ནཱཾ་བཛྲ་སུ་ར་ཏ་པཱུ་ཛཱཾ་ཀུ་རུ་ཀུ་རུ་སྭཱ་ཧཱ། དེ་ལྟར་མཆོད་པ་བྱས་ནས་སུ། །དེ་ནས་སྡིག་པ་བཤགས་པར་བྱ། །ཐོག་མ་མེད་པའི་འཁོར་བ་ནས། །བདག་གིས་སྡིག་པ་གང་བྱས་པ། །སངས་རྒྱས་ཀུན་གྱི་མདུན་དུ་ནི། །ད་ནི་དེ་ལྟར་
བཤགས་པར་བྱ། །སངས་རྒྱས་དང་ནི་བྱང་ཆུབ་སེམས། །དེ་བཞིན་འཕགས་པའི་དགེ་འདུན་དང་། །སྐྱེས་བུ་ཕལ་བ་རྣམས་ཀྱིས་ནི། །དགེ་བ་བྱས་ལ་ཡི་རང་ངོ་། །ཉིད་དོན་མཛད་པའི་སངས་རྒྱས་རྣམས། །གང་ཞིག་བྱང་ཆུབ་ཤིང་དྲུང་རྩར། །བཞུགས་པ་ཆོས་འཁོར་བསྐོར་བའི་
