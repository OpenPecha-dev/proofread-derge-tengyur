ལ་འདུད། །སངས་རྒྱས་འབྱུང་བ་ཁྱོད་ལ་འདུད། །ནམ་མཁའ་ལས་བྱུང་ཁྱོད་ལ་འདུད། །ཡེ་ཤེས་ལས་བྱུང་ཁྱོད་ལ་འདུད། །སྒྱུ་འཕྲུལ་དྲ་བ་ཁྱོད་ལ་འདུད། །སངས་རྒྱས་རོལ་སྟོན་ཁྱོད་ལ་འདུད། །ཐམས་ཅད་ཐམས་ཅད་ཁྱོད་ལ་འདུད། །ཡེ་ཤེས་སྐུ་ཉིད་ཁྱོད་ལ་འདུད། །
ཅེས་བྱ་བའོ། །དེ་ལྟར་ལན་གསུམ་བརྗོད་ནས་སུ། །འདོད་པའི་ལྷ་ཚོགས་གཤེགས་པར་བསམ། །དེ་ནས་བདུད་རྩི་མྱང་བ་ཡི། །ཆོ་ག་བཞིན་དུ་གཏོར་མ་ནི། །བསྒྲུབ་པ་བྱས་ནས་རྩེ་ལྔ་པའི། །ཕྱག་རྒྱས་སྔགས་འདི་སྦྱིན་པར་བྱ། །དེ་ནས་གཏོར་མའི་སྔགས་ནི། །ཨོཾ་ཨཱཿཧཱུཾ་ཧོ་
ཨུཥྞཱི་ཥ། སུཾ་བྷ་ནི་སུམྦྷ། བིགྷྣཱནྟ་ཀ །ནཱི་ལ་དཎྜ། པྲཛྙཱནྟ་ཏ་ཀ ཊཱ་ཀི་རཱ་ཛ། པཏྨཱནྟ་ཀ །ཨ་ཙ་ལ། ཡ་མཱནྟ་ཀ །མ་ཧཱ་བཱ་ལེ་བྷྱཿས་པ་རི་ཝཱ་རེ་བྷྱཿཨི་དཾ་བ་ལིཾ། གན་དྷཾ། པུཥྤཾ། དྷཱུ་པཾ། དཱི་པཾ། ཨཀྵ་ཏཾ། ད་དཱ་མ་ཧེ། ཏེ་ཛ་ཨཱ་ག་ཏྱེ་ས་པ་རི་ཝཱ་ར། ཤཱི་གྷྲཾ་ཨི་དཾ་བ་ལིཾ། གྲྀཧྞནྟུ་
ཁ་དནྟུ། པི་བནྟུ། ཛཿཧཱུཾ་བཾ་ཧོཿསཾ་ཏྲིཔྟཱཿསརྦ་སཏྭཱ་ནཱཾ་ཤཱནྟིཾ། པུཥྚིཾ། རཀྵ་ཨཱ་ཝ་ར་ཎ། གུཔྟིཾ། ཀུརྦན་ཏུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་བཛྲ་དྷ་ར་ཨཱཛྙཱ་པ་ཡ་ཏི་སྭཱ་ཧཱ། ཞེས་བྱ་བ་ནི་ཁྲོ་བོའི་རྒྱལ་པོ་རྣམས་ཀྱི་གཏོར་མའི་སྔགས་སོ། །སྔགས་འདི་བརྗོད་ནས་གཏོར་མ་ནི། །བྱིན་ནས་ཡིད་དུ་འོང་བ་
ཡི། །དགོས་པ་བསྒྲུབ་པར་བྱ་བའི་ཕྱིར། །དེ་ལ་སྔགས་འདིས་གསོལ་བ་གདབ། །དེ་ནས་ལན་གསུམ་ལག་པའི་རྒྱབ། །བསྟན་ནས་སེ་གོལ་གསུམ་གྱིས་ནི། །གཤེགས་སུ་གསོལ་བར་བྱས་ནས་སུ། །གཏོར་མ་དེ་ཉིད་གསོལ་བར་བསམ། །དེ་ནས་ཀླུ་ལ་སོགས་རྣམས་
ནི། །གལ་ཏེ་གཏོར་མ་འདོད་གྱུར་ན། །དབང་གི་ལེའུ་ལས་གསུངས་པ་ཡི། །ཆོ་ག་ཤེས་ནས་སྦྱིན་པར་བྱའོ། །དེ་བཞིན་རྩ་བའི་སྔགས་སོགས་ནི། །བཞི་པ་ལ་སོགས་ཤེས་ནས་སུ། །དེ་དང་མཐུན་པའི་ཕྲེང་བ་ཡིས། །རྣལ་འབྱོར་པས་ནི་སྔགས་བཟླས་བྱ། །དེ་ལ་དགྲ་དང་གྲོགས་
པོ་ནི། །དེ་བཞིན་དམན་པར་གྱུར་པ་ཡི། །ཡི་གེ་དབྱངས་ཚོགས་ཤེས་ནས་སུ། །སྔགས་ཀྱི་བསྒྲུབ་པ་བསམ་པར་བྱ། །སྒྲུབ་པ་པོ་དང་བསྒྲུབ་བྱ་ཡི། །དང་པོའི་ཡི་གེ་དགྲར་གྱུར་ན། །དེ་ནི་བཟླས་པས་སྒྲུབ་པ་པོ། །འཆི་འགྱུར་དེ་བས་དོར་བར་བྱ། །ཡང་ན་གཉི་གའི་དང་པོ་ཡི། །ཡི་
