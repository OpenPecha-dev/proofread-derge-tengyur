སྨྲ་བའོ། །དེ་ཉིད་ཀྱི་སོར་མོའི་རྩེ་མོ་དག་ནི་རོལ་མོའོ་ཞེས་བྱ་བ་ལ་ནང་ན་གནས་པའི་སྲིན་ལག་གི་རྩེ་མོ་དེ་ཉིད་ཀྱི་སོར་མོའི་རྩེ་མོ་དག་གིས་ཕན་ཚུན་རོལ་མོའི་ཚུལ་དུ་བྱ་སྟེ་དེ་ལྟར་ན་གར་བྱེད་དུ་གཞུག་པའོ། །མཐེའུ་ཆུང་གི་རྡོ་རྗེའོ་ཞེས་བྱ་བ་ལ་རྡོ་རྗེ་བསྡམས་པའི་མཐེའུ་ཆུང་གཉིས་ཀྱི་རྩེ་མོ་ནང་དུ་
གཞུག་པར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་དགོད་པའོ། །དེ་ཉིད་ཀྱི་མཐེ་བོ་གཉིས་ཀྱི་རྩེ་མོ་སེན་མོས་བཙིར་བའོ་ཞེས་བྱ་བ་ལ་འདས་མ་ཐག་ཏུ་བཤད་པའི་མཐེའུ་ཆུང་གཉིས་ཀྱི་རྡོ་རྗེ་ནང་དུ་གནས་པ་དེ་ཉིད་ཀྱི་མཐེ་བོ་གཉིས་ཀྱི་སེན་མོ་བཙིར་བར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་གསོད་པར་འགྱུར་རོ། །
རྡོ་རྗེ་གར་གྱི་མཆོད་པས་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་བ་དན་གསུམ་མཆོག་དང་ཁ་ཊྭཱཾ་ག་སྒེག་པ་དང་བཅས་པར་མཐེའུ་ཆུང་ལ་སོགས་པ་བསྐོར་བའི་ཚུལ་གྱིས་བར་དུ་<མཆོད་,མ་ཆོད་>པ་ཉིད་དུ་དཔལ་རྡོ་རྗེ་སེམས་དཔའི་རྣལ་འབྱོར་བྱས་ལ་ཡི་གེ་ཧཱུཾ་བརྗོད་ཅིང་རྡོ་རྗེའི་ལྟ་བས་བལྟས་ཤིང་གར་བྱ་སྟེ་དེ་ལྟར་ན་
ལྷ་དང་ལྷ་མ་ཡིན་པའི་བུ་མོ་ཐམས་ཅད་དང་། གཞོན་ནུ་མའི་བུ་མོ་ཐམས་ཅད་བསྟན་ནས་ཏེ་རྗེས་སུ་ཆགས་པར་འགྱུར་རོ། །རབ་ཏུ་དགའ་བ་ཞེས་བྱ་བ་ན་དཔལ་རྡོ་རྗེ་སེམས་དཔའི་བདེ་བ་དང་ཡིད་བདེ་བའི་རང་བཞིན་ནོ། །དེ་ཡང་དེ་བཞིན་དུ་གར་གྱི་ཚུལ་གྱིས་རང་གི་མ་མཆུ་ནང་དུ་གཡོ་
བ་མེད་པར་མྱང་བ་དང་བཏུང་བའི་ཚུལ་གྱིས་གྲུབ་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་གར་བྱ་ཞིང་དབུགས་རིང་པོ་བཏང་བར་ཡི་གེ་ཧཱུཾ་བརྗོད་པར་བྱ་སྟེ་དེ་ལྟར་ན་ཆགས་པ་བརྟན་པར་འཆིང་ཞིང་མཐོང་བར་འགྱུར་རོ། །དེ་བཞིན་དུ་གར་བྱ་ཞིང་དཔུང་པ་གཉིས་བསྟོད་ཅིང་བཟླ་བར་བྱ་སྟེ་དེ་ལྟར་ན་དབང་ཕྱུག་
ཐམས་ཅད་ཐོབ་པར་འགྱུར་རོ། །དེ་བཞིན་དུ་གར་བྱ་ཞིང་གར་གྱི་ཚུལ་གྱིས་སྙིང་ག་བསྐྱོད་པར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་ཀྱིས་མཆོད་པར་འགྱུར་རོ། །དེ་བཞིན་དུ་གར་བྱ་ཞིང་མིག་གཉིས་ཀྱིས་བསྒྲུབ་བྱ་དགུག་པའི་ཚུལ་བྱས་ཏེ། དེ་ལྟར་ན་འདོད་པ་ཐམས་ཅད་རྙེད་པར་འགྱུར་རོ། །
དེ་བཞིན་དུ་གར་བྱ་ཞིང་དབུགས་གཏོང་བ་དང་དམན་པའི་གདོང་གིས་བལྟ་བའི་ཚུལ་བྱ་སྟེ་དེ་ལྟར་ན་འདོད་པའི་འདོད་ཆགས་སྐྱེད་པར་བྱེད་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་གར་བྱ་ཞིང་རང་གི་སྙིང་གའི་དཔྲལ་བར་བསྐྱོད་པར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་འབེབས་པར་བྱེད་དོ། །དེ་བཞིན་དུ་གར་
