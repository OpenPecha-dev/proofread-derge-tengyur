བར་བྱ་སྟེ་དེ་ལྟར་ན་སངས་རྒྱས་ཐམས་ཅད་འབྱོན་པའོ། །དེ་ཉིད་བཅིངས་ནས་ཁྲོ་བ་དང་བཅས་པའི་འཇིགས་བྱེད་ཀྱི་ལྟ་བས་གཞན་གྱི་དཔུང་ཐམས་ཅད་འཇོམས་པའོ། །དེ་ཉིད་བཅིངས་ནས་བཙུམ་པ་མེད་པའི་ལྟ་བས་མཚམས་བཅིང་བའོ། །དེ་ཉིད་བཅིངས་ནས་རང་གི་སྙིང་ག་ལས་
རྡོ་རྗེ་འབར་བའི་ཡིད་ཀྱིས་སྤྲོས་ནས་ཕྱོགས་བཅིང་བར་བྱའོ། །དེ་ཉིད་བཅིངས་ནས་བསྐོར་བར་བྱ་སྟེ་དེ་ལྟར་ན་དཀྱིལ་འཁོར་བཅིང་པའོ། །དེ་ཉིད་བཅིངས་ནས་མཛུབ་མོ་བསྒྲེང་བ་བྱས་ནས་ཡོངས་སུ་བསྐོར་ལ་གཙུག་ཏོར་དུ་གཟུང་བར་བྱས་ན་ཤིན་ཏུ་སྲུང་བའོ། །དེ་ཉིད་བཅིངས་ནས་མཛུབ་
མོ་བརྐྱང་བ་ལ་རང་གི་སྙིང་གར་གཟུང་བར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་དེ་ཉིད་བྱེད་པའོ། །ཕྱག་རྒྱ་དེ་ཉིད་རང་གི་སྙིང་གར་དགྲོལ་བར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་དགྲོལ་བའོ། །གཡས་པའི་རྡོ་རྗེ་འཛིན་མ་དེ་ཉིད་ཟུར་དུ་བཞག་ལ་བསྒྲེང་བར་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་བསྒྲེང་པའོ། །དེ་ཉིད་
བཅིངས་ནས་མཛུབ་མོའི་རྩེ་མོས་དབང་དུ་བྱེད་པའི་ཚུལ་བྱ་སྟེ་དེ་ལྟར་ན་ཐམས་ཅད་དབང་དུ་བྱེད་པར་འགྱུར་རོ། །དེ་ཉིད་བཅིངས་ནས་ཁྲོ་བོའི་ལྟ་བས་སྟེང་ནས་འོག་ཏུ་ལྟུང་བའི་ཚུལ་དུ་བྱ་སྟེ་དེ་ལྟར་ན་ལྟུང་བ་ཐམས་ཅད་ལྟུང་བའོ། །འོག་ནས་སྟེང་དུ་དྲངས་པའི་ཚུལ་བྱས་ན་ཀླུ་ཐམས་ཅད་གནོན་པའོ། །
དེ་ཉིད་བཅིངས་ཏེ་ཁྲོས་པས་ཡི་གེ་ཧཱུཾ་བཟླ་བར་བྱ་སྟེ་དེ་ལྟར་ན་གནོད་སྦྱིན་ཐམས་ཅད་ཕྱི་བཞིན་དུ་འབྲང་བར་བྱེད་པའོ། །དེ་ཉིད་བཅིངས་ནས་འཇིགས་བྱེད་ཀྱི་ལྟ་བས་སྲིན་པོ་ཐམས་ཅད་འཇིགས་སུ་བཅུག་པའོ། །དེ་ཉིད་བཅིངས་ནས་ཁྲོ་བོའི་ལྟ་བས་ཁ་ཊྭཱཾ་ག་ཡིད་ཀྱིས་དྲལ་བར་བྱ་སྟེ་དེ་ལྟར་ན་
མཁའ་འགྲོ་མ་ཐམས་ཅད་ལ་སོགས་པའི་གདོན་ཐམས་ཅད་ཞི་བའོ། །དེ་ཉིད་བཅིངས་ནས་རང་གི་སྙིང་གར་འབར་བའི་རྡོ་རྗེ་བསྒོམ་པར་བྱ་སྟེ་དེ་ལྟར་ན་བདུད་ཐམས་ཅད་ཀྱི་སྟོབས་འཇོམས་པའོ། །དེ་ཉིད་བཅིངས་ནས་ཁྲོ་བོའི་ལྟ་བས་ཡིད་ཀྱིས་རྩེ་གསུམ་དྲལ་བར་བྱ་སྟེ་དེ་ལྟར་ན་མ་མོ་ཐམས་
ཅད་ཞི་བར་འགྱུར་རོ། །དེ་ཉིད་བཅིངས་ནས་གདུག་པའི་དཀྱིལ་འཁོར་གཞུག་པར་བྱ་སྟེ་དེ་ལྟར་ན་གདུག་པའི་དཀྱིལ་འཁོར་ཐམས་ཅད་འཐོར་བའོ། །དེ་ཉིད་བཅིངས་ནས་ཡི་གེ་ཧཱུཾ་བཟླ་བར་བྱ་སྟེ་དེ་ལྟར་ན་གདུག་པ་ཐམས་ཅད་ཚར་གཅོད་པའོ། །དེ་ཉིད་བཅིངས་ནས་ཡི་གེ་ཧི་བཟླ་
