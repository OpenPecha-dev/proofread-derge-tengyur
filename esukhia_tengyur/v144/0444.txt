རྒྱུ་ཡང་མི་སྲིད་དོ། །འདུས་མ་བྱས་ནི་སྐྱེ་བ་དང་མི་ལྡན་པའི་ཕྱིར་སྔར་སྐྱེས་པ་མེད་པས་སྐལ་བ་མཉམ་པའི་རྒྱུ་ཡང་མི་སྲིད་དོ། །སེམས་དང་སེམས་ལས་བྱུང་བའི་རང་བཞིན་མ་ཡིན་པའི་ཕྱིར་མཚུངས་པར་ལྡན་པའི་རྒྱུ་ཡང་མེད་དོ། །སྡུག་བསྔལ་ལ་སོགས་པ་མཐོང་བས་སྤང་བར་
བྱ་བ་མ་ཡིན་པའི་ཕྱིར་ཀུན་དུ་འགྲོ་བའི་རྒྱུ་ཡང་མེད་དོ། །འདི་ལ་རྣམ་པར་སྨིན་པའི་རྒྱུ་ཡང་མི་སྲིད་དོ། ལྡན་པ་མ་ཡིན་པའི་ཕྱིར་དང་། རྟག་པའི་ཕྱིར་རོ། །གཞན་ཡང་རྣམ་པར་སྨིན་པ་ནི་ལུང་དུ་མ་བསྟན་པ་ཡིན་ལ། བྲལ་བ་ནི་དགེ་བ་ཡིན་པས་འདི་ལ་རྣམ་པར་སྨིན་པའི་རྒྱུ་
མེད་དོ། །དེ་ལྟར་ན་རྒྱུ་རྣམ་པ་དྲུག་མི་སྲིད་པའི་ཕྱིར་འདུས་མ་བྱས་འབྲས་བུ་མ་ཡིན་ནོ། །འབྲས་བུ་རྣམ་པ་ལྔ་མི་སྲིད་པའི་ཕྱིར་ཡང་འདུས་མ་བྱས་ལ་འབྲས་བུ་མེད་དོ་ཞེས་སྟོན་ཏོ། །འདི་ལྟར་འདུས་མ་བྱས་ལ་བདག་པོའི་འབྲས་བུ་ནི་མི་སྲིད་དོ། བདག་པོའི་འབྲས་བུ་ནི། སྔོན་
བྱུང་མ་ཡིན་འདུས་བྱས་ནི། །འདུས་བྱས་ཁོ་ནའི་བདག་པོའི་འབྲས། །ཞེས་ལྷན་ཅིག་འབྱུང་བ་དང་། ཕྱིས་འབྱུང་བར་འདོད་ན། དེ་ནི་སྐྱེ་བ་དང་ལྡན་པ་ཡང་མ་ཡིན་ནོ། །རྒྱུ་མཐུན་པའི་འབྲས་བུ་ཡང་མི་སྲིད་དོ། རྒྱུ་མཐུན་རྒྱུ་དང་འདྲ་བའོ། །ཞེས་བྱ་བ་ཡིན་ན། འདུས་མ་བྱས་ནི་
སྐྱེ་བ་དང་ལྡན་པ་ཡང་མ་ཡིན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར་སྐྱེས་བུ་བྱེད་པའི་འབྲས་བུ་འང་མི་སྲིད་དེ། འདུས་མ་བྱས་ནི་སྐྱེ་བ་དང་མི་ལྡན་པའི་ཕྱིར་སྐྱེས་བུའི་རྩལ་མེད་པའི་ཕྱིར་རོ། །འདི་ལ་རྣམ་པར་སྨིན་པའི་འབྲས་བུ་ཡང་མི་རུང་སྟེ། རྣམ་པར་སྨིན་པའི་འབྲས་བུ་ནི་མི་དགེ་བ་དང་དགེ་བ་ཟག་
པ་དང་བཅས་པས་བསྐྱེད་པར་བྱ་བ་ཡིན་པའི་ཕྱིར་རོ། །ལམ་གྱི་རང་བཞིན་མ་ཡིན་པའི་ཕྱིར་ན་འདི་ལ་བྲལ་བའི་འབྲས་བུ་ཡང་མི་སྲིད་དེ། དེ་ནི་བར་ཆད་མེད་པའི་ལམ་རྣམས་ཀྱི་འབྲས་བུ་ཡིན་ནོ། །གཞན་ཡང་དུས་ལས་གྲོལ་བ་ལ་འབྲས་བུ་འཛིན་པ་དང་འབྱིན་པའི་མཐུ་མེད་པའི་
ཕྱིར་ཡང་འབྲས་བུ་མེད་དེ། ཇི་སྐད་དུ། ལྔ་ནི་ད་ལྟར་འབྲས་བུ་འཛིན། །ཞེས་རྒྱས་པར་གསུངས་པ་ལྟ་བུའོ། །ཅིའི་ཕྱིར་ལམ་ཞེས་བྱ་བ་རྒྱས་པར་འབྱུང་སྟེ། དེ་ནི་དངོས་པོར་འབྱུང་བ་ལ་བྱེད་པ་དང་བྲལ་བ་ཁོ་ན་ཡིན་པར་འདོད་དོ་སྙམ་པ་ནི་འདྲི་བའི་བསམས་པའོ། །འདི་ལྟར་དེ་
