བཛྲ་ས་མ་ཡ་ཛཿཛཿཛཿཧཱུཾ་བཾ་ཧོཿ། ཞེས་བརྗོད་དེ། སྒོ་དབྱེ་བ་སྔོན་དུ་འགྲོ་བ་བསྡུ་བར་བྱའོ། །དེ་ནས་རྡོ་རྗེ་བསྡུ་བ་དེས་རྡོ་རྗེ་འཛིན་པ་ཆེན་པོ་ལ་སོགས་པ་དང་། འཇིག་རྟེན་གྱི་ཁམས་ཐམས་ཅད་ན་བཞུགས་པའི་སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའི་འཁོར་གྱི་དཀྱིལ་འཁོར་དང་བཅས་པ་བྱོན་
པར་བྱས་པ་དང་། ཨོཾ་སརྦ་ཏ་ཐཱ་ག་ཏ་པཱ་ད་བནྡྷ་ནཾ་ཀ་རོ་མི། ཞེས་བརྗོད་ཅིང་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཞབས་ལ་ཕྱག་བྱས་ནས་ཆེད་དུ་བརྗོད་པ་འདི་ཆེད་དུ་བརྗོད་དོ། །ཨེ་མའོ་བྱང་ཆུབ་སེམས་དཔའ་ཡི། །ཀུན་ཏུ་བཟང་པོ་སྤྱོད་པ་མཆོག །དེ་བཞིན་གཤེགས་པའི་འཁོར་ལོ་
ཡི། །དབུས་ན་དེ་བཞིན་གཤེགས་པ་མཛེས། །དེ་ནས་ཆེད་དུ་བརྗོད་པ་དེ་སྐད་དུ་སྨྲས་ནས། ཕྱོགས་བཅུའི་འཇིག་རྟེན་གྱི་ཁམས་ཐམས་ཅད་ནས་འདུས་པའི་དེ་བཞིན་གཤེགས་པ་བྱང་ཆུབ་སེམས་དཔའི་འཁོར་གྱི་དཀྱིལ་འཁོར་དང་བཅས་པ་དེ་དག་ཐམས་ཅད་དེ་བཞིན་གཤེགས་པ་ཐམས་
ཅད་ཀྱི་བྱིན་གྱི་རླབས་ཀྱིས་བཅོམ་ལྡན་འདས་རྡོ་རྗེ་འཆང་ཆེན་པོའི་ཐུགས་སུ་ཞུགས་སོ། །དེ་ནས་དེ་བཞིན་གཤེགས་པ་དེ་དག་ཐམས་ཅད་ཀྱི་ཐུགས་ཀ་ནས་རྡོ་རྗེ་ཡིད་ལས་བྱུང་བ་ལ་སོགས་པ་ནས། ཆུ་ལྷའི་མཐར་ཐུག་པའི་བར་དུ་ཕྱུང་ནས། རྡོ་རྗེ་ཡིད་ལས་བྱུང་བ་ལ་སོགས་པ་ནས། ཆུ་ལྷའི་
མཐར་ཐུག་པ་དང་བཅས་པ་དང་། གཅིག་ཏུ་གྱུར་ནས་གནས་ཏེ། ཆེད་དུ་བརྗོད་པ་འདི་ཆེད་དུ་བརྗོད་དོ། །ཨེ་མའོ་སངས་རྒྱས་ཐམས་ཅད་ཀྱིས། །རྒྱ་ཆེན་གདོད་ནས་སྐྱེ་བ་མེད། །རྡུར་ཕྲན་ཇི་སྙེད་ཐམས་ཅད་ཀྱི། །སངས་རྒྱས་རྣམས་ནི་གཅིག་ཏུ་གྱུར། །དེ་ནས་བདེ་བ་ཆེན་པོ་ཡིས། །རྡོ་རྗེའི་ཕྱག་རྒྱ་
ཐམས་ཅད་བཅིངས་ནས། ཧཱུྃ་ཟློས་པ་ལ་འབད་པའི་རྣལ་འབྱོར་པ་རྡོ་རྗེ་འཛིན་པ་ཆེན་པོ་ལ་སོགས་པ་དེ་རྣམས་ཀྱིས་བླངས་ནས། དབང་བསྐུར་བ་ཐམས་ཅད་ཀྱིས་དབང་བསྐུར་བར་གྱུར་པར་བདག་ཉིད་བསམས་ལ། དབང་བསྐུར་བ་ཐམས་ཅད་བླང་བར་བྱའོ། །དེ་ལྟར་དབང་བསྐུར་བ་བླངས་
ནས། རྡོ་རྗེ་འཛིན་པ་ཆེན་པོ་ལ་སོགས་པ་ནས་ཆུ་ལྷའི་མཐར་ཐུག་པའི་བར་དུ་རྡོ་རྗེ་གཟུགས་དང་། སྒྲ་དང་། དྲི་དང་། རོའི་ལས་ཀྱི་ཕྱག་རྒྱ་རྣམས་ཀྱིས་བཀུག་ལ་བཅུག་སྟེ། བཅིངས་ལ་དབང་དུ་བྱས་ཏེ། མཆོད་ཡོན་ཕུལ་ཏེ་ཕྱག་རྒྱ་བཞིས་རྒྱས་བཏབ་ལ། རང་གི་དབང་བསྐུར་བས་དབང་བསྐུར་བར་བྱ་སྟེ།
