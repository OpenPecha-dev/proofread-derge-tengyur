ལྔ་པ་གཟུང་ངོ་། །ཐམས་ཅད་སྡུད་པ་ཞེས་བྱ་བ་ནི་ཡི་གེ་ཐམས་ཅད་སྡུད་པ་སྟེ་བཤད་ཟིན་ཏོ། །འཚེ་བ་ཟློག་པར་བྱེད་པ་ཡིན། །ཞེས་བྱ་བ་ནི་ངན་འགྲོ་དང་སྲིད་པའི་འཚེ་བ་ཟློག་པར་བྱེད་པའོ། །བཅུ་དྲུག་པའི་ལས་ནི་གོ་སླ་བས་མ་ཕྱེའོ། །དཀར་པོ་དམར་པོའི་ཏིང་འཛིན་བྱ། །ཞེས་བྱ་
བ་ནི་ཐབས་དང་ཤེས་རབ་ཀྱི་བདག་ཉིད་ཀྱི་ལྷའི་ཏིང་ངེ་འཛིན་བྱའོ། །དེས་ཅིར་འགྱུར་ཞེ་ན། ཨ་ལ་སོགས་དང་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། དེའི་འོག་ཏུ་ཨཱ་ལི་ཀཱ་ལིའི་ཡི་གེ་རྣམས་སྔར་བསྟན་པ་དང་། འོག་ནས་འབྱུང་བ་ལྟར་བདག་དང་གཞན་གྱི་ལུས་ལ་བཀོད་ལ། ཚེགས་མེད་
པར་ལས་རྣམས་འགྲུབ་པོ་ཞེས་བྱ་བའི་དོན་ཏོ། །འདི་ནི་མདོར་བསྡུས་པའི་ཆོ་གའོ། །ད་ནི་ལས་དེ་དག་ཐམས་ཅད་ལ་ཡི་གེ་དེ་དག་ཉིད་ཀྱིས་སྔོན་དུ་རང་གི་ལུས་བསྲུང་བའི་ཆོ་ག་བསྟན་པའི་ཕྱིར། དེ་ལྟར་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། དེ་ལྟར་ཞེས་བྱ་བ་ནི་སྔར་བསྟན་པའི་ལས་
རྣམས་སོ། །ལྷག་མ་ནི་གོ་སླའོ། །བདག་གི་དེ་ཉིད་གསལ་བ་ཡིན། །ཞེས་བྱ་བ་ནི་བདག་གི་ཁམས་བཅུ་དྲུག་བསྲུང་བའི་དེ་ཁོ་ན་ཉིད་དོ། །དེ་ལྟར་དབྱངས་ཀྱི་ལས་རྣམས་བཤད་ནས། གསལ་བྱེད་ཀྱི་ལས་རྣམས་བསྟན་པའི་ཕྱིར། དེ་ནས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། རིག་པ་
དེ་ཉིད་འབྱུང་ཞེས་བྱ་བ་ནི། །ཤེས་རབ་དང་མཉམ་པར་སྦྱོར་བའི་ལྷའི་ཏིང་ངེ་འཛིན་ལས་འབྱུང་བའོ། །ཡི་གེ་དང་པོ་ཞེས་བྱ་བ་ནི་ཡི་གེ་ཀའོ། །ཇི་སྲིད་ལུས་ལ་གནས་ཀྱི་བར། །ཞེས་བྱ་བ་ནི། བསྒྲུབ་པར་བྱ་བ་དེའི་ལུས་ལ་ཡི་གེ་བསམ་པ་དེ་སྲིད་དོ། །རང་གི་ལྷ་ཡི་སྡོམ་པ་
སྟེ། །ཞེས་བྱ་བ་ནི། དཔལ་ཧེ་རུ་ཀའི་ན་བཟའི་ངོ་བོ་ཉིད་དོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཁྱིམ་ཞེས་བྱ་བ་ནི་སྒྲུབ་པར་བྱེད་པའི་ལུས་སོ། །ཐམས་ཅད་དུ་ནི་འབིགས་པར་འགྱུར། །ཞེས་བྱ་བ་ནི། དེའི་འོད་ཟེར་དམར་པོས་ཡན་ལག་གི་ཚིགས་ཐམས་ཅད་དུ་ཁྱབ་པའོ། །དེ་ལ་ཞེས་བྱ་བ་ནི་སྒྲུབ་
པ་པོ་ལའོ། །ད་ནི་ཡི་གེ་ལྔ་པོ་དེའི་ལས་ཀྱི་སྔོན་དུ་འགྲོ་བའི་ཆོ་ག་བསྟན་པའི་ཕྱིར། ཡི་གེ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཆོ་གར་བྱ་ཞེས་བྱ་བ་ནི་ཕྲག་པ་གཉིས་སུ་བཀོད་པའོ། །ཆོ་ག་དེ་ལྟར་བྱས་ནས་ནི། །དཔྱད་མི་དགོས་པར་དངོས་གྲུབ་འགྱུར། །ཞེས་བྱ་བའི་ཆོ་གའི་འབྲས་བུ་
