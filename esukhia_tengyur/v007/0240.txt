ལྷ་མོ་ཆེན་མོ་ནི་རྡོ་རྗེ་སྙེམས་མ་མིང་གཞན་རྡོ་རྗེ་དབྱིངས་ཕྱུག་མའོ། །རྒྱུད་གཞན་དུ་ནི་རྡོ་རྗེ་སེམས་དཔའི་ཤེས་རབ་མཱ་མ་ཀཱི་རུ་ཡང་གསུངས་སོ། །འདིས་ནི་རྡོ་རྗེ་དང་པདྨ་བྱིན་གྱིས་བརླབ་པ་ལ་སོགས་པ་གསལ་བར་བྱས་སོ། །སྲུང་བའི་རྗེས་ཐོགས་སུ་བྱ་བ་
གསུངས་པ། དེ་ལྟར་ཞེས་པ་ལ་སོགས་པའོ། །བདག་ཉིད་རྣམ་སྣང་མཛད་བྱས་ནས། །ཞེས་པ་སྔགས་ཁོ་ན་ཡིས་ཞེས་པའི་རྗེས་ཐོགས་སུ་སྦྲེལ་བར་བྱའོ། །སྟེང་དང་འོག་ཏུ་སྔགས་ཀྱི་ཡི་གེ་ཞེས་པ། ཧཱུྃ་སུཾ་ཧཱུྃ་ཞེས་པའོ། །ཆོ་ག་བཞིན་ཞེས་པ་སྲུང་བའི་
འཁོར་ལོ་རྩིབས་བཅུ་པ་ལ་གནས་པའི་རྡོ་རྗེ་འཛིན་པ་ཡོངས་སུ་གྱུར་པ་གྲུ་གསུམ་པའོ། །དེ་ཉིད་ཀྱི་ཕྱིར་གསུངས་པ། དེ་ངའོ་ཞེས་པ་རྡོ་རྗེ་འཛིན་པ་དང་རྣམ་པར་སྣང་མཛད་དོ། །གནས་ནི་བཞི་སྟེ་ཆོས་འབྱུང་དང་ཁང་པ་བརྩེགས་པའོ། །དེ་ཉིད་ཀྱི་ཕྱིར་ཡང་འདི་
དང་རྗེས་སུ་འབྲེལ་བའི་དོན། རྣམ་པར་སྣང་མཛད་བྱས་ནས། ཞེས་སྔར་གསུངས་སོ། །གྲུ་གསུམ་དེར། སྣ་ཚོགས་པདྨ་ཉི་དཀྱིལ་འཁོར། །ཞེས་པས་སྣ་ཚོགས་རྡོ་རྗེ་ཉེ་བར་མཚོན་པའོ། །རྫོགས་པར་བྱ་ཞེས་པ་དང་སྦྲེལ་ལོ། །ཅི་ལྟ་བུའི་རི་རབ་ཅེ་ན་གསུངས་པ།
རླུང་མེའི་དཀྱིལ་འཁོར་རྣམས་དང་ཞེས་པ་མང་པོའི་ཚིག་ལས་ཆུ་དང་སའི་དཀྱིལ་འཁོར་དག་ཀྱང་ཤེས་པར་བྱའོ། །དེའི་སྟེང་དུ་ཞེས་པ་སྣ་ཚོགས་པདྨ་དང་ཉི་མ་དང་རྡོ་རྗེའི་སྟེང་དུའོ། །སྔགས་ནི་བྷྲཱུཾ་ཡིག་སྟེ། བྷྲཱུཾ་ཡིག་གནས་བསམས་ནས་ཞེས་པའི་གསུང་ལས་སོ། །བྷྲཱུཾ་
ལས་སྐྱེས་པའི་འཁོར་ལོ་ལས་ཡང་དག་པར་འབྱུང་བའི་རྣམ་པར་སྣང་མཛད་ཡོངས་སུ་གྱུར་པའི་ཁང་པ་བརྩེགས་པ་བསམ་པར་བྱའོ་ཞེས་འབྲེལ་ལོ། །ཅི་ལྟ་བུ་ཞེ་ན། དེ་བཞིན་གཤེགས་པ་ཀུན་གྱིས་བསྟེན་ཅིང་སྟེ་ལྷག་པར་བསྟེན་པ་དང་བསྟོད་པ་ཡང་ངོ་། །
