༄༅༅། །རྒྱ་གར་སྐད་དུ། མ་ཧཱ་སཱུ་ཏྲ་ས་མུཙྪ་ཡ་ནཱ་མ། བོད་སྐད་དུ། མདོ་ཀུན་ལས་བཏུས་པ་ཆེན་པོ་ཞེས་བྱ་བ། ཐམས་ཅད་མཁྱེན་པ་ལ་ཕྱག་འཚལ་ལོ། །སངས་རྒྱས་དང་བྱང་
ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་ཕྱག་འཚལ་ལོ། །གཞི་ཆེན་པོའི་མདོ་ལས། ཡང་ཇི་ལྟར་དགེ་སློང་དག་གིས་མདོ་སྡེ་དག་ལས་དོན་ཕྱུང་ནས་གཏམ་དུ་བྱ་བ་སྦྱོར་བར་འཚལ་བར་འགྱུར་ཞིང་ཆོས་བསྟན་
པ་མཐར་གྱིས་ཡང་དག་པར་གནས་པར་འཚལ་བར་འགྱུར། ཇི་ལྟར་ན་འགྱོད་པར་མི་འགྱུར་ཞིང་མདོ་སྡེ་ཉམས་པར་མི་འགྱུར་བ་ཉིད་ལགས་ཞེས་དགེ་སློང་དག་གིས་བཅོམ་ལྡན་འདས་ལ་གསོལ་པ་དང་།
བཅོམ་ལྡན་འདས་ཀྱིས་བཀའ་སྩལ་པ། ངས་གནང་བར་བྱ་ཡིས། མདོ་སྡེ་དག་ལས་དོན་ཕྱུང་ནས་གཏམ་དུ་བྱ་བ་སྦྱར་བར་བྱའོ། །ཆོས་སྟོན་པའི་རིམ་པ་མཐར་གྱིས་ཡང་དག་པར་གནས་པར་བྱ།
མདོ་སྡེ་དག་ཀྱང་དེ་ཁོ་ན་ལྟར་གཞག་པར་བྱའོ། །སྐྱེ་དགུའི་བདག་མོ་ཆེན་མོས་ཁྱོད་ཀྱི་བུ་འདི་ཆོ་འཕྲུལ་ཡ་མ་ཟུང་མཛད་དེ་སྐུའི་སྨད་ནས་མེ་འབར། སྐུའི་སྟོད་ནས་ཆུའི་རྒྱུན་བསིལ་བ་འབབ། སྐུའི་
