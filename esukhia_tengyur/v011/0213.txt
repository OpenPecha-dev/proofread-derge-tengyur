རིང་པོ་དང་། །ཤིན་ཏུ་རིང་པོ་ཡོན་ཏན་མེད་ཅེས་ཡོན་ཏན་དང་བཅས་བུད་མེད་སྐྱེས་པའང་སྐྱེས་པ་བུད་མེད་མིན། །གང་ཞིག་ཀུན་གྱི་རྟེན་གཅིག་བྷ་ག་བཟང་མོ་མཆོག་གི་བྷ་ག་ཁྱོད་ལ་འདུད་དེ་ཁྱོད་ལ་འདུད། ༢༤༨ །གང་ཞིག་མི་ཡི་འཇིག་རྟེན་དག་ན་དཔལ་ལྡན་བླ་མ་རྡོ་རྗེ་འཛིན་པ་སྡིག་
བྲལ་དེ་བདག་སྟེ། །དེ་ལ་གནོད་པ་བྱས་པས་སྲོག་ཆགས་རྣམས་ནི་ངེས་པར་དམྱལ་བར་འགྱུར་ཏེ་འདི་ལ་ངོ་མཚར་མེད། །དེ་མཉེས་པ་ཡིས་བདག་ནི་མགུ་སྟེ་དེ་ལ་ཁྲོ་བའི་མེ་ཡིས་ཁྲོས་པ་ཞེས་པ་ཆེན་པོའོ། །དེ་ཉིད་སེམས་ཅན་རྣམས་ལ་ཐར་པ་བདེ་མཉམ་འབྲས་སྟེར་བུ་རྣམས་དག་གི་ཕྱག་འོས་
མཆོད་འོས་སོ། ༢༤༩ །སྡིག་ཅན་གང་ཡིན་གང་ཞིག་དཔལ་ལྡན་བླ་མའི་ཞབས་པད་བཟང་པོ་དག་ལ་དུས་གསུམ་ཕྱག་མི་བྱེད། །མི་ཤེས་གང་ཡིན་གང་ཞིག་རྣམ་མང་མེ་ཏོག་རྣམས་ཀྱིས་དུས་གསུམ་དུ་ནི་མཎྜལ་མི་བྱེད་པ། །མྱུར་དུ་མནར་མེད་འགྲོ་བ་གང་ཡིན་གང་ཞིག་བདེ་མཉམ་སྟེར་བྱེད་བླ་མ་སྐྱོ་
བ་སྐྱེད་པ་པོ། །ཤེས་རབ་ཡེ་ཤེས་ཐོབ་པ་གང་ཡིན་མ་ཉམས་གང་ཞིག་མཆོག་གི་བླ་མའི་ཞབས་ནི་མི་སྤོང་བ། ༢༥༠ །ཉམས་པ་གང་ཡིན་གང་ཞིག་རྩ་གསུམ་དག་ལ་གནས་པའི་རླུང་ནི་མཐའ་ཡས་དུས་སུ་བསད་པའོ། །དཔའ་བོ་གང་ཡིན་གང་ཞིག་མཉམ་དང་མི་མཉམ་ལམ་ལ་སྲོག་དང་ཐུར་སེལ་
རླུང་ནི་གསོད་བྱེད་པའོ། །སྦྱིན་བདག་གང་ཡིན་གང་ཞིག་དཔལ་ལྡན་བླ་མར་རང་ལུས་བུ་དང་ཆུང་མ་ལ་སོགས་ཐམས་ཅད་སྟེར། །མ་རབས་གང་ཡིན་གང་ཞིག་སླུ་བྱེད་རང་གི་སྙིང་གི་སྡིག་པས་དྲངས་པའི་སེམས་དང་གཡོ་སྒྱུ་ཅན། ༢༥༡ །དཔལ་ལྡན་དཔལ་གྱི་ཆོས་ཀྱི་འཁོར་ལོ་ལྷ་ཡི་མཆོག་
གིས་བཏུད་པའི་གདན་ནི་ཁ་དོག་སྣ་ཚོགས་པ། །དེ་ལ་སངས་རྒྱས་ལྷ་དང་མི་ཡི་བླ་མ་ཉེ་བར་བཞུགས་པས་ཟླ་བའི་ཞུ་བ་རྟོགས་པ་ན། །སེམས་ཅན་དོན་གྱི་སླད་དུ་མི་ཡི་བདག་པོ་རྒྱུད་ཀྱི་རྒྱལ་པོ་དང་པོའི་སངས་རྒྱས་གསུངས་པ་གང་། །དེ་ཡང་ད་ལྟ་བདག་གིས་དེ་རྣམས་ཨཱ་ལི་ཀཱ་ལི་ཉུང་ངུའི་རྒྱུད་
དུ་བཀོད་དོ་ཉི་མ་ཀྱེ། ༢༥༢ །སངས་རྒྱས་ཀྱིས་ནི་གསུངས་པ་དག་ལས་དུས་ཀྱི་འཁོར་ལོ་བདག་གིས་ཕྲེང་འཛིན་སྡེབ་སྦྱོར་བཅིངས་པས་བརྗོད། །པདྨ་དཀར་པོ་རྡོ་རྗེ་འཛིན་པ་ཁྱོད་ཀྱིས་འདི་ནི་སྔགས་པ་རྣམས་ལ་བསྟན་པ་ཉིད་དུ་མཛོད། །ཇི་ལྟར་ཉི་མ་དང་བཅས་ཐུབ་པའི་རིགས་ནི་ཤེས་རབ་ཡེ་ཤེས་
