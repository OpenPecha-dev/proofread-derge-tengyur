མའི་ང་ལྡན་གྱི་བརྗོད་པ་རྣམས་ཡཻ་ཡཱས྄་ཡཱས྄་ཡཱམ྄་ཞེས་པ་འདི་རྣམས་སུ་བསྒྱུར་བར་འགྱུར་རོ། །རྣམ་པར་བརྟག་པས་སུ་ཡིག་ཨཱ་ག་མ་དང་བཅས་པ་དང་སྔ་མ་ཐུང་བའོ། །ཐུང་བ་དང་སུ་ཡིག་ཨཱ་ག་མ་དག་ཡང་ན་འཐོབ་ཅིང་། རྣམ་པར་བརྟག་པ་སྟེ་བསྒྱུར་བ་
རྣམས་ནི་མ་ཡིན་ཏེ་རྣམ་པར་བཤད་པ་ལས་སོ། །དྭོ་ཏཱི་ཡ་སྱཻ། དྭི་ཏཱི་ཡ་སྱཱཿ། དྭི་ཏི་ཡ་སྱཱཿ། དྭི་ཏཱི་ཡ་སྱཱཾ། དྭི་ཏཱི་ཡཱ་ཡཻ། དྭི་ཏཱི་ཡཱ་ཡཱཿ། དྭི་ཏཱ་ཡཱ་ཡཱཿ། དྭི་ཏཱི་ཡཱ་ཡཱམ྄། ཏྲྀ་ཏཱི་ཡ་སྱཻ། ཏྲྀ་ཏཱི་ཡ་སྱཱཿ། ཏྲྀ་ཏཱི་ཡ་སྱཱཿ། ཏྲི་ཏཱི་ཡ་སྱཱམ྄། ཏྲྀ་ཏཱི་ཡཱ་ཡཻ། ཏྲི་ཏཱི་ཡཱ་ཡཱཿ། ཏྲྀ་ཏཱི་ཡཱ་ཡཱཿ། ཏྲྀ་ཏཱི་ཡཱ་ཡཱམ྄། ནཱ་དཱི་
ལས་ཨཻ་ཨཱས྄ ཨཱས྄ ཨཱམ྄འོ། །སུ་དང་བཅས་པ་རྣམས་ཏེ་སྔ་མའི་ཐུང་བ་རྣམས་སོ་ཞེས་པ་ལོག་གོ། །ན་དཱི་ལས་ཕ་རོལ་གྱི་ང་ལྡན་རྣམས་གྲངས་ཇི་ལྟ་བ་བཞིན་དུ་ཨཻ་ཨཱས྄ ཨཱས྄ ཨཱམ྄ ཞེས་པ་འདི་རྣམས་སུ་བསྒྱུར་བར་བྱའོ། །གཽ་ཪྱཱཻ། གཽཪྱཱཿ། གཽ་ཪྱཱཿ། གཽཪྱཱམ྄། བ་དྷཻ། བདྷྭཱཿ་
བ་དྷྭཿ། བ་དྷྭ་མ། ན་དཱི་ལས་ཞེས་པ་ཅི་ཞེ་ན། དྭི་ཏཱི་ཡཱ་ཡེ། སཾ་བུདྡྷི་ལ་ཐུང་ངུའོ། །ན་དཱི་སཾ་བུདྡྷི་ཕྱི་མ་ལ་ཐུང་ངུ་འགྱུར་རོ། །ཧེ་གོ་རི། ཧེ་བ་དྷུ་། སཾ་བུདྡྷི་ལ་ཞེས་པ་ཅི་ཞེ་ན། གཽ་རཱི། བ་དྷཱུཿ། ཨམ་ཤས་དག་གི་དང་པོ་དབྱིའོ། །ན་དཱི་ལས་ཕྱི་མ་ཨམ་ཤས་དག་གི་དང་པོ་དབྱི་བ་འཐོབ་
པོ། །གཽ་རཱིམ྄། གཽ་རཱིཿ། བ་དྷུམ྄། བ་དྷཱུཿ ཨམ་ཤས་དག་གི་ཞེས་པ་ཅི་ཞེ་ན། ནདྱཽ། བ་དྷྭོཿ། ཨཱི་ཡིག་མཐའ་ལས་སིའོ། །ན་དཱིའི་མིང་ཅན་ལས་ཨཱི་ཡིག་མཐའ་ལས་ཕྱི་མའི་སི་དབྱི་བ་འཐོབ་པོ། །གཽ་རཱི། ན་དཱིའི་མིང་ཅན་ལས་ཞེས་པ་ཅི་ཞེ་ན། གྲཱ་མ་ནཱིཿ ཨཱི་ཡིག་མཐའ་
ལས་ཞེས་པ་ཅི་ཞེ་ན། བ་དྷཱུཿ། ལཀྵྨཱི་ལ་སོགས་པ་རྣམས་ལས་སི་དབྱི་བ་མི་འདོད་དེ། ལཀྵྨིཿ། ཏནྟཱིཿ། གསལ་བྱེད་ལས་ཀྱང་ངོ་། །གསལ་བྱེད་མཐའི་རྟགས་ལས་ཕྱི་མའི་སི་དབྱི་བ་འཐོབ་པོ། །ཝཀ྄། བི་དྱུད། རཱ་ཛཱ། འོ་ན་སིའི་རྐྱེན་གྱི་སི་ཡིག་བརྩེགས་པའི་མཐའི་
དབྱིའོ་ཞེས་པས་དབྱི་བ་གྲུབ་སྟེ། དེས་ན་སྦྱོར་བས་ནི་བྱ་ཞེ་ན་འདི་ནི་ཡང་དག་པ་མ་ཡིན་ཏེ། གང་རྟགས་ཀྱི་བརྩེགས་པ་དེའི་མཐའ་དབྱི་བ་ཡིན་ཞིང་། སིའི་རྐྱེན་གྱི་མཐའ་ནི་རྟགས་མ་ཡིན་པའང་ངོ་། །ཨགྣི་ལས་ཨམ་གྱི་ཨ་ཡིག་གོ། །ཨགྣི་ལས་ཕྱི་མའི་ཨམ྄འི་ཨ་
