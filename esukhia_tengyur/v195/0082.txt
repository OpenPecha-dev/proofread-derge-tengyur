དང་ཙའི་ཨ་སྦྱར། ཨ་ཙ་ཀ་མ་ཏའོ། །ཀཱ་རི་ཏའི་མཐའ་ཅན་རྣམས་ལ་ཀྲྀ་ལ་དི་སྦྱར་བར་དུ་ཙན྄་སྦྱར། གོང་དུ་ཨ་དང་ཀྲྀ་གཉིས་པ་སྦྱར་ཏེ། ཨི་ཐོབ་པ་ལ་སོགས་པའོ། །ཨ་ཙི་ཀ་ར་ཏའོ། །ཧྲྀ་ལ་དི་སྦྱར་བར་དུ་ཙཎ་སྦྱར་གོང་བཞིན། ཨ་ཛཱ་ཧ་ར་ཏའོ། །ཨ་ཎཱ་སུ་ཝ་ཙི་ཁྱཱ་ཏི་ལེ་པི་སི་ཙ་ཏྭེ།
ཨ་སུ་ནི་གཏོང་བའོ། །ཝ་ཙ་ནི་སྨྲ་བའོ། །ཁྱཱ་ནི་གྲགས་པའོ། །ལི་པ་ནི་ཉེ་བར་སྟོན། ཥི་ཙ་འཛག་པའོ། །ཧྭེ་ཉ་ནི་འགྲན་པའོ། །འདི་རྣམས་ཀྱི་དྷཱ་ཏུའི་ཕ་རོལ་ན་བྱེད་པ་པོའི་ཚིག་ཅན་གྱི་ཨ་དྱ་ཏ་ནཱི་ཡོད་ན་རྐྱེན་ཨཎ་འབྱུང་བ་སྟེ། ཨ་པ་ལ་ཨ་དྱ་ཏ་ནིའི་ཏ་སྦྱར། བར་དུ་ཨཎ་སྦྱར་གོང་དུ་མ་སྦྱར།
ཨ་དྱ་དེའི་མདོས་ཐ་ཐོབ། ས་དང་ཨ་དང་ཨཎ་གྱི་ཨ་སྦྱར་ཨ་གོང་མ་བསྲིངས་ཏེ། པ་རི་དང་ཨི་ཝཪྞའོ། །ཝཙ་ལ་དི་སྦྱར་བས་ཨོ་ཐོབ་པའོ། །ཨ་བོ་ཙ་ཏའོ། །ཁྱཱ་ལ་དི་སྦྱར། གོང་དུ་ཨཱའོ། ཨ་ཁྱཱ་ཏའོ། །ལི་པ་ལ་དི་སྦྱར། གོང་དུ་ཨའོ། །ཨ་ཁྱཱ་ཏའོ། །ཨཱ་ལི་བཏའོ། །སི་ཙ་ལ་དི་སྦྱར།
གོང་དུ་ཨའོ། །ཨ་སི་ཙཏའོ། པུ་ཥཱ་དི་དྱུ་ཏཱ་དྱླྀ་ཀཱ་རཱ་ནུ་བནྡྷཱརྟིཤརྟི་ཤསྟི་བྷྱཤྩ་པ་ར་སྨཻ། པུ་ཥ་ནི་རྒྱས་པ་སྟེ། དེ་ལ་སོགས་པ་རྣམས་དང་། དྱུ་ཏ་ནི་འབར་བ་སྟེ། དེ་ལ་སོགས་པ་རྣམས་དང་དེ་བཞིན་དུ་ལྀའི་མཐའ་ཅན་གྱི་དྷཱ་ཏུ་རྣམས་དང་། དེ་བཞིན་དུ་ལ་ནི་འགྲོ་བ་དང་རྀ་སྲྀ་ཡང་འགྲོ་བའི་དོན་ཅན་
ནོ། །ཤཱ་སུ་ནི་རྗེས་སུ་སྟོན་པ་སྟེ། དེ་རྣམས་དང་ཕ་རོལ་ནི་བྱེད་པ་པོའི་ཚིག་ལ་ཨདྱ་ཏ་ནཱི་གཞན་གྱི་ཚིག་ཅན་ཡོད་ན་རྐྱེན་ཨན་འབྱུང་བར་འགྱུར་ཏེ། པུ་ཥ་ལ་སོགས་པ་རྣམས་ནི། པུ་ཥ་ལ་དི་སྦྱར་བས་བར་དུ་ཨན྄་འབྱུང་སྟེ། ཥ་དང་སྦྱར་གོང་དུ་ཨའོ། །ཨ་པུ་ཥཏའོ། །ཏུ་ཥ་ལ་དི་ལ་སོགས་
པ་གོང་བཞིན། ཨ་ཏུ་ཥ་ཏའོ། །ཤུ་ཥ་ལ་ཡང་གོང་བཞིན། ཨ་ཤུ་ཥཏའོ། །པུ་ཥ་ནི་རྒྱས་པའོ། །ཏུཥ་ནི་དགའ་བའོ། །ཤུ་ཥ་ནི་སྐེམས་པའོ། །དྱུ་ཏ་ལ་སོགས་པ་རྣམས་ལ། དྱུ་ཏ་ལ་དི་སྦྱར་བས་བར་དུ་ཨན་སྦྱར། གོང་དུ་ཨ་སྦྱར། ཨ་དྱུ་ཏ་ཏ྄འོ། །རུ་ཙ་ལ་དི་སྦྱར་བ་གོང་བཞིན། ཨ་རུ་
ཙན་ཏའོ། །དྱུ་ཏ་ནི་འབར་བའོ། །ལྀ་རྗེས་སུ་འབྲེལ་པ་རྣམས་ལ། གིམླ་ལ་དི་སྦྱར་བས། ལྀ་ཕྱི། བར་དུ་ཨན྄། ན་ཕྱི། གོང་དུ་ཨ་སྦྱར། ཨ་ག་མཏ྄འོ། །ཨ་དི་ལ་དི་སྦྱར་བས། གྷྀསླ་ཐོབ། གོང་དུ་ཨ་སྦྱར། ལྀ་ཕྱི། ཨན་དང་སྦྱར། ཨ་གྷ་སཏ྄འོ། །གམ྄་ལྀ་ནི་འགྲོ་བའོ། །
