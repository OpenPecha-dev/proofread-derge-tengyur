མཿ། བྷུ་ཛ་ལས་ཟས་ཀྱི་དོན་ལ་ཨཱཏྨ་ནེའི་ཚིག་བཤད་པར་བྱ་སྟེ། ཨ་བྷྱ་བ་ཧཱ་རོ་བྷོ་ཛ་ནཾ་བྷུཾ་ཀྵེ་པཱ་ཡ་སཾ། བྷུཛ་སྐྱོང་བ་དང་ཟས་ལའོ། །རུདྷ་སོགས་ལས་དབྱངས་ལས་ཕ་རོལ་དུ་ན་སྒྲའོ། །རུདྷ་སོགས་ཀྱི་བི་ཀ་ར་ཎའི་མཐའ་དབྱིའོ། །ཙ་སྡེའི་ཀིའོ། །སྐྱོང་བར་བྱེད་པ་ལ་ནི་པ་
རསྨཻའི་ཚིག་སྟེ། བྷུ་ནཀྟི་པྲྀ་ཐྭིམ། ཨ་ནུ་དང་པ་རཱ་དག་ལས་ཕ་རོལ་གྱི་ཀྲྀཉ་ལས་པ་རསྨཻ་ཁོ་ན་བཤད་པར་བྱའོ། །ཨ་ནུ་ཀ་རོ་ཏི། པ་རཱ་ཀ་རོ་ཏ་ཏི། ཉ་རྗེས་སུ་འབྲེལ་པ་ལས་གཉི་ག་ཐོབ་པ་ལའོ། །ཨ་དྷི། པྲ་ཏི། ཨ་དྷི། ཨ་བྷི་རྣམས་ལས་ཕ་རོལ་གྱི་ཀྵི་པ་ལས་སོ། །ཨ་དྷི་ཀྵི་པ་ཏི། པྲ་ཏི་ཀྵི་
པ་ཏི། ཡཛ་སོགས་ཉིད་ལས་གཉིས་ཀ་ཐོབ་པའོ། །པྲ་ལས་ཕ་རོལ་གྱི་བ་ཧི་ལས་སོ། །ཕྲ་བ་ཧ་ཏི། སྔར་བཞིན་གཉིས་ཀ་ཐོབ་པ་ལའོ། །པ་རི་ལས་ཕ་རོལ་གྱི་མྲྀཥ་ལས་སོ། །པ་རི་མྲྀ་ཥྱ་ཏི། འདིར་ཡང་གཉི་ག་ཐོབ་པ་ལའོ། །བི། ཨཱང྄། པ་རི། ཨུ་པ་རྣམས་ལས་ཕ་
རོལ་གྱི་རམ྄་ལས་སོ། །བི་ར་མ་ཏི། པ་རི་ར་མ་ཏི། ཨཱ་ར་མ་ཏི། ཨུ་པ་ར་མ་ཏི། རུཙ་སོགས་ཉིད་ལས་ཨཱཏྨ་ནེའི་ཚིག་ཐོབ་པ་ལའོ། །ཇོ་བོ་དྲག་འབྱོར་གྱིས་མཛད་པའི་འགྲེལ་པ་སློབ་མ་ལ་ཕན་པར་ཀུན་བཤད་པའི་རབ་ཏུ་བྱེད་པ་གཉིས་པ་རྫོགས་སོ།། །།
ཨ་བྷྱཱ་ས་མེད་པའི་དང་པོའི་དབྱངས་གཅིག་པའི་གཉིས་བརྗོད་དོ། །ལན་གཉིས་བརྗོད་པ་དག་ནི་གཉིས་དག་སྟེ། ཏ་མ་སོགས་ཉིད་ལས་རྒྱུའི་རྐྱེན་ནོ། །དེ་ནས་གཉིས་བརྗོད་པ་ནི་ལན་གཉིས་བརྗོད་པའོ་ཞེས་དེའི་སྐྱེས་བུའོ། །གང་གི་ཨ་བྷྱཱ་ས་ཡོད་པ་མ་ཡིན་པ་དེ་ནི་ཨ་བྷྱཱ་ས་མེད་
པའོ། །གང་གི་དང་པོ་དབྱངས་གཅིག་ཡིན་པ་དེ་ནི་དབྱངས་གཅིག་པའོ། །ཨ་བྷྱཱ་ས་མེད་པའི་གང་དང་པོའི་དབྱངས་གཅིག་པའི་བྱིངས་དེའི་གཉིས་བརྗོད་པ་མ་བྱས་པ་གཉིས་བརྗོད་པར་འགྱུར་རོ་ཞེས་སྤྱིར་བཏང་བའོ། །ཙཎ་པ་རོ྅ཀྵ་ཞེས་པ་དེར་འདི་རྣམས་ཉེ་བར་གནས་སོ། །ད་ད་རི་
དྲཽ། །ད་རི་དྲ་དབུལ་པོ་ལའོ། །ཨཊའོ། །དང་པོ་དབྱངས་གཅིག་པའི་ཞེས་པས་འདིའི་ད་ལན་གཉིས་བརྗོད་པར་བྱའོ། །ཨཱ་ཡིག་ལས་ཨཊ་ཨཽའོ། །མཚམས་སྦྱོར་གྱི་ཡི་གེ་ཞེས་པས་ཨ་དབྱིའོ། །ཛ་ཛཱ་གཱ་ར། ཛཱ་གྲྀ་གཉིད་སད་པ་ལའོ། །ཨཊའོ། །ཛ་ཞེས་པས་འདི་
