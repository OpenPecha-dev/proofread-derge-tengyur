མའི་མཛོད་འདས་པ་ནི། ནིསྐོ་ཥཱམྦིཿ། གང་གི་བ་ལང་ཁྲ་བོ་ཡིན་པ་དེ། ཙི་ཏྲ་གུཿ། དབྱངས་སྨོས་པ་ནི་མཐའི་ཐུང་ངུའི་དོན་ཏོ། །འདིར་མི་འགྱུར་ཏེ། སུ་ཝཱ་ཀ་ཀུ་ལཾ་ཞེས་པའོ། །གལ་ཏེ་ཡི་གེའི་མཐའི་ཡོངས་སུ་བཤད་པ་ནི་མཐའ་ཉིད་ཀྱི་འགྱུར་རོ་ཞེས་པ་ཅི་ཞེ་ན། དེ་ལྟར་ནི་
རྣམ་པ་གསལ་བའི་དོན་ཏོ། །ཇོ་བོ་དྲག་འབྱོར་གྱིས་མཛད་པའི་འགྲེལ་པ་སློབ་མ་ལ་ཕན་པར་མིང་གི་རབ་ཏུ་བྱེད་པའི་རྐང་པ་བཞི་པ་རྫོགས་སོ།། །།མིང་རྣམས་སྦྱོར་བའི་དོན་བསྡུ་བའོ། །མིང་ནི་མིང་སྟེ་སྔོན་གྱི་སློབ་དཔོན་རྣམས་ཀྱིས་རྟགས་དང་རྣམ་དབྱེའི་མཐའ་ཅན་ལ་བྱས་བའོ། །སྦྱོར་བའང་
འདི་ཡིན་ལ་དོན་ཡང་ཡིན་པ་ནི་སྦྱོར་བའི་དོན་ཏོ། །གང་མིང་རྣམས་སྦྱོར་བའི་དོན་དེ་ནི་བསྡུ་བའི་མིང་ཅན་དུ་འགྱུར་རོ། །ངོ་བོའམ་ཚུལ་གཅིག་ཉིད་དུ་གྱུར་པའི་དོན་དེ་སྦྱོར་བའི་དོན་དེས་བརྗོད་པར་བྱའོ། །ནཱི་ལོཏྤ་ལཾ། རཱ་ཛ་པུ་རུ་ཥཿ། སྦྱོར་བའི་དོན་སྨོས་པ་ཅི་ཞེ་ན། བསྟྲཾ་ནཱི་ལཾ། ཨུདྦ་ལཾ། ཝཪྵ་
སུ་པུ་ཥྱ་ཏི། བྷཱ་རྻཱ་རཱ་ཛྙཿ། པུ་རུ་ཥོ་དེ་ཝ་དཏྟསྱ། བསྡུ་བའི་མིང་གིས་གང་དུ་དགོས་ཤེ་ན། བསྡུ་བའི་མཐར་གྱུར་པ་རྣམས་ཀྱིས་ཞེས་པ་ལ་སོགས་པ་རྣམས་སུའོ། །དེར་གནས་རྣམ་དབྱེ་རྣམས་དབྱི་བར་བྱའོ། །དེ་ལ་གནས་པ་ཞེས་པ་ནི་དེར་གནས་པ་སྟེ་མིང་ལས་ཀྱང་སྠ་ལའང་ངོ་ཞེས་པས་ཀའོ། །
དབྱི་བར་འོས་པ་ཞེས་པས་དབྱི་བར་བྱ་བའོ། །དེར་ལེགས་པ་ལའང་ཡའོ་ཞེས་པའི་འང་གི་སྒྲ་ལས་དབྱི་བར་འོས་པའི་དོན་ལ་ཡའོ། །དེར་སྨོས་པས་ཚིག་སྡུད་ལ་སྙེག་གོ། །ཚིག་སྡུད་ཀྱི་དོན་གྱི་ངག་ཀྱང་ཚིག་སྡུད་ཀྱི་སྒྲས་བརྗོད་པར་བྱ་སྟེ། དབང་པོའི་དོན་གནས་པ་དབང་པོའི་སྒྲས་བཞིན་ནོ། །དེས་ན་
གང་ཚིག་སྡུད་ཀྱི་དོན་གྱི་ངག་ལ་གནས་པའི་རྣམ་དབྱེ་དེ་རྣམས་དབྱི་བར་བྱ་བར་འགྱུར་རོ། །རཱ་ཛྙཱ་པུ་རུ་ཥཿ། རཱ་ཛ་པུ་རུ་ཥཿ། སཾ་ཀུ་ལཱ་ཡ་ཁཎྜཿ། སཾ་ཀུ་ལ་ཁཎྜཿ། དབྱི་བར་བྱའོ་ཞེས་པས་འོས་པའི་རྐྱེན་ངེས་པར་བསྟན་པ་ལས་སྦྱོར་བའི་དབང་གིས་གཞན་གྱིའང་དབྱི་བར་འོས་པ་རྣམས་དབྱི་བར་བྱའོ། །
གང་གི་ཞལ་ཏའི་ཞལ་དང་འདྲ་བ་དེ་ནི། དེ་ཝ་མུ་ཁཿ། མར་གྱིས་གང་བའི་རྫ་མ་ནི། གྷྲྀ་ཏ་གྷཊཿ། གང་ལ་སྡིག་པ་ཡོད་པ་མ་ཡིན་པ་དེ་ནི། ཨ་པཱ་པཾ། ཨི་བ་དང་། པཱུརྞ་དང་། བིདྱ་མཱ་ན་རྣམས་ཀྱིས་དབྱིའོ། །གསུམ་པ་དང་བདུན་པ་དག་ལ་ཡང་ནའོ་ཞེས་པས་འདིའི་ཡང་ན་ཞེས་པ་འཇུག་སྟེ།
