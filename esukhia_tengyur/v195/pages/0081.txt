ཐོབ། བྷུ་གཉིས་པ་བྱས། བྷུ་བོནྟ་ས་འོག་ཏུ་བ་ཐོབ་སྟེ། ཨེ་དང་སྦྱར་རོ། །ཨཱིཀྵ་བ་བྷུ་བའོ། །ཨཱིཀྵ་ནི་ལྟ་བའོ། །ཙ་ཀ་སཱ་མལ་ཨས་སྦྱར་བ་དང་། ཕ་རོལ་ན་ཨེ་ཡོད་པ་ལ་སོགས་པ་གོང་མ་བཞིན། ཙཱ་ཀཱ་སཱ་མཱ་སའོ། །ཙ་ཀཱ་སཱམ྄་ལ་བྷཱུ་སྦྱར་བ་དང་ཨེ་སྦྱར་བ་གོང་
མ་བཞིན། ཙ་ཀཱ་སཱ་བ་བྷུ་བ།། །།སི་ཛཱ་དྱ་ཏ་ནྱཱམ྄། དྷཱ་ཏུའི་ཕ་རོལ་ན་ཨ་དྱ་ཏ་ནཱི་རྣམས་ཡོད་ན་རྐྱེན་སི་ཙ་འབྱུང་བ་སྟེ། ཨི་དང་ཙི་དག་ནི་སི་ཙ་པ་ར་སྨཻའི་མདོ་འདིས་དབྱི་བར་བྱའོ། །ཀྲྀའི་མཐར་སྦྱར་པས་བར་དུ་སི་ཙ་སྦྱར། ཐོག་མར་ཨ་སྦྱར། རྀ་ཨར་དུ་འཕེལ། སས་ཥ་ཐོབ། ཨིད་ཐོབ་སྟེ།
ཨི་དང་སྦྱར། ཧྲི་དང་དི་སྦྱར་བས་བར་དུ་སི་ཙ་དང་ཨི་སྦྱར་བ་སོགས་གོང་བཞིན། ཨ་ཧཱཪྵིཊའོ། །ཌུ་ཀྲིཉ་ནི་བྱེད་པའོ། །ས་ཎ་ནི་ཌསྶི་ཌནྟཱ་ནཱ་མྱུ་བ་དྷཱད་དྲྀ་ཤཿ། དྷཱ་ཏུ་ཥིཊ྄ ཀྱི་མཐའ་ཅན་ནཱ་མིའི་ཉེ་བའི་ཚིག་ཏུ་འགྱུར་པ་ལ་ཨཊ་མ་གཏོགས་པའི་སན་རྐྱེན་དུ་འབྱུང་བ་སྟེ། དྲྀ་ཤིར྄་ཞེས་
བྱ་བ་ནི་ལྟ་བ་སྟེ་དྷཱ་ཏུ་འདི་མ་གཏོགས་པའོ། །བ་ཤའི། ཕ་རོལ་ན་དི་ཡོད་པས་བར་དུས་ན་སྦྱར། ན་ཕྱི། ཤ་དོ་ཀའི་མདོས་ཤས་ཀ་ཐོབ། སས་ཤ་ཐོབ་པ་དང་སྦྱར། ཨ་བིཀྵ་ཊའོ། །ཀྲི་ཤ་ལ་གོང་དུ་ཨ་སྦྱར་བ་ལ་སོགས་པ་གོང་བཞིན་ནོ། །ཨ་ཀྲི་ཀྵ་ཏའོ། །བུ་ཤ་ལ་གོང་དུ་
ཨ་སྦྱར་བ་ལ་སོགས་པའོ། །ཨ་བུཀྵཱ་ཏའོ། །བ་ཤ་ནི་འཇུག་པའོ། །ཀྲྀ་ཤ་ནི་རྣམ་པར་རྐོ་བའོ། །དུ་ཧ་ནི་འཇོ་བའོ། །དྲྀ་ཤ་མ་གཏོགས་པ་ཞེས་ཅིའི་ཕྱིར་སྨོས་ཤེ་ན། དྲྀ་ཤ་ལ་དི་སྦྱར་བ་ཡོད་ཀྱང་བར་དུ་སན་མ་བྱུང་སྟེ། སི་ཙ་བྱུང་ནས་ཨིད་ཀྱང་སྦྱར། ཨ་དྲྀ་ཀྵ་དའོ། །ཤྲཱི་དྲུ་སྲུ་ཀ་མི་
ཀཱ་རི་ཏཱནྟེ་བྱ་ཤྩ་ཀརྟ་རི། དྲུ་ནི་འགྲོ་བའོ། །སྲུ་ཡང་འགྲོ་བའོ། །ཀ་མུ་ནི་འདོད་པའོ། །དེ་རྣམས་ཀྱི་དྷཱ་ཏུ་ཀཱ་རི་ཏའི་མཐའ་ཅན་རྣམས་ལ་ཕ་རོལ་ན་བྱེད་པ་པོའི་ཚིག་ཅན་ཨ་དྱ་ཏ་ནཱི་རྣམས་ཡོད་ན་རྐྱེན་ཙཎ་འབྱུང་བ་སྟེ། ཤྲི་ལ་དི་སྦྱར་བས་གོང་དུ་ཨ་སྦྱར། གོང་དུ་ཤྲི་གཉིས་པ་བྱས་པས་བར་
དུ་ཙཎ་ད་དྷཱ་ཏུ་ཤྲིའི་ཨིས་ཨི་ཡ་ཐོབ། ཡ་དང་ཨའི་ཙ་སྦྱར་རོ། །ཨ་ཤི་ཤྲི་ཙ་ཡ྄ཏའོ། །དྲུ་ལ་དི་སྦྱར་བས་ཙཎ་སྦྱར་བ་སོགས་གོང་བཞིན་ཨ་དུ་དྲུ་བཏྲའོ། །སྲུ་ལ་དི་སྦྱར་པ་སོགས་པའོ། །ཨ་སྲུ་སྲུ་བཏའོ། །ཀ་མུ་ལ་ཨ་དྱ་ཏ་ནཱིའི་ཏ་སྦྱར་བས་ཙན་སྦྱར་ཏེ་ཀ་གཉིས་པ་བྱས་ཨུ་ཕྱི་ནས་བ་
