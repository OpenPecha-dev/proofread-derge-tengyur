ཨ་ཏུས྄་དང་ཨུས྄འོ། །གཉིས་བརྗོད་དོ། །དང་པོའི་ཨའི་ཞེས་པས་རིང་པོའོ། །ར་ཉིད་དོ། །འདིར་ཨཱ་ག་མ་དང་བསྒྱུར་བ་དག་གི་ཨཱ་ག་མའི་སྒྲུབ་པ་སྟོབས་དང་ལྡན་ནོ་ཞེས་པའི་ཚིག་ལས་ཨཏུ་དང་ར་ཉིད་བྱས་ནས་ན་ཡིག་ཨཱ་ག་མ་ཐོབ་པོ་ཞེ་ན། འདི་ནི་སྙིང་པོ་མེད་པ་སྟེ། ཨཱ་ག་མ་ལས་
དབྱངས་བསྒྱུར་བའི་ཆོ་ག་སྟོབས་དང་ལྡན་ནོ་ཞེས་སྔ་མའི་ར་ཉིད་འཇུག་གོ། །དེའི་ཕྱིར་རྀ་ཡིག་མེད་པའི་ཕྱིར་ན་ཨཱ་ག་མ་མེད་དོ། །ཨཤྣོ་ཏིའི་འང་ངོ་། །དེ་ལས་ཏེ་ཨ་ཡིག་རིང་པོར་བྱས་པ་ལས་ཕ་རོལ་གྱི་ཨ་སུ་ཐོབ་པ་ལའོ་ཞེས་པས་འདིའི་དང་པོར་ན་ཡིག་ཨཱ་ག་མ་འགྱུར་རོ། །བྱཱ་ན་ཤེ།

ཨཱ་ཤ་ཏུཿ། བྷ་བ་ཏིའི་ཨའོ། །བྷ་བ་ཏིའི་ཨ་བྷྱཱ་སའི་པ་རོཉཀྵ་ལ་ཨ་ཡིག་ཏུ་བསྒྱུར་བར་འགྱུར་རོ། །བ་བྷུ་བ། བ་བྷུ་བ་ཏུཿ། བ་བྷུ་བུཿ། བྷུ་ཡོད་པ་ལའོ། །ཨ་ཊ་ཨ་ཏུས་ཨུསའོ། །གཉིས་བརྗོད་དོ། །འདིས་ཨུ་ཡིག་གི་ཨ་ཡིག་གོ། །བྷུའི་མཐའ་པ་རོ྅ཀྵ་ཞེས་པས་བ་ཡིག་
ཨཱ་ག་མའོ། །དེ་བཞིན་དུ། ཨ་ནུ་བ་བྷུ་བེ་དེ་བ་དཏྟ་ན། ལས་ལ་ཨཱཏྨ་ནེའི་ཚིག་ཨེའོ། །གཉིས་བརྗོད། གཞན་སྔ་མ་བཞིན་ནོ། །བྷུ་བ་ཞེས་མ་བྱས་པ་ནི་རྣམ་པ་སྣ་ཚོགས་པ་ལས་སོ། །ནི་ཛི་བི་ཛི་བི་ཥ་རྣམས་ཀྱི་བྱིངས་ཐམས་ཅད་པ་ལ་ཡོན་ཏན་ནོ། །ནི་ཛིར་གཙང་བ། བི་ཛིར྄་
དངོས་པོ་ཐ་དད་པ། བི་ཥླི་ཁྱབ་པ་ལའོ། །འདི་རྣམས་ཀྱི་ཨ་བྷྱཱ་སའི་བྱིངས་ཐམས་ཅད་པ་ཕ་རོལ་ལ་ཡོན་ཏན་དུ་འགྱུར་རོ། །ནེ་ནེ་ཀྟི། ནེ་ནི་ཛྱཏ྄། ནེ་ནེཀྟུ། ཨ་ནེ་ནེ་ཀ །བེ་བེཀྟི། བེ་བི་ཛྱཱཏ། བེ་བེཀྟུ། ཨ་བེ་བེཀ྄ི ། བེ་བིཥྚ་ཏ྄། བ་བེཥྜུ། ཨ་བེ་བེཊ། ཏི་དང་ཡཱཏ྄ དང་ཏུ་དང་
དིའོ། །ཨན྄འོ། །ཛུ་ཧོ་ཏི་སོགས་ལས་འཇིག་གོ། །ཛུ་ཧོ་ཏི་སོགས་རྣམས་ཀྱིས་གཉིས་བརྗོད་དོ། །འདིས་ཡོན་ཏན་ནོ། །ཨུ་པ་དྷཱ་ནཱ་མིའི་འང་ཞེས་པས་བྱིངས་ཀྱི་ཡོན་ཏན་ནོ། །ཡཱ་ཏའི་རྐྱེན་ལ་ནི་སྤཏ་མཱི་ལའང་ངོ་ཞེས་པས་ཡོན་ཏན་མེད་དོ། །ཙ་སྡེའི་ཀིའོ། །གསལ་བྱེད་
