ཐོབ་པར་གསུངས་ཏེ། ཨིང་ལ་ཨ་དྱ་ཏ་ནཱི་སྦྱར་བས་ག་ཐོབ། སི་ཙ་དང་། ཨིཊ་སྦྱར་བ་སྲིང་བའི་གོང་དུ་ཨི་དྷི་དང་། ཨ་སྦྱར་རོ། །ཨ་དྷྱ་གཱི་ཥྚའོ། །ཨིང་ལ་ཀྲཱི་ཡ་ཏི་པ་ཏི་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་དྷྱ་གཱི་ཥྱ་ཏའོ། །ཡང་ན་ཨིང་ལ་ཏ་སྦྱར། གོང་དུ་ཨ་དྷི་སྦྱར། སི་ཙ་སྦྱར་ཏེ་ཨི་དང་
ཅ་ཕྱི། སྭར་ཨ་དྷི་ནམ་གྱི་མདོས། ཨིས་ཨཻ་ཐོབ། ཨ་དྷྱཻཥྚའོ། །ཨིད་ལ་སྱ་ཏ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་དྷྱཻ་ཥྱ་ཊའོ།། །།ས་ནཱི་ཤྲི་ངོག་མི། ཨིན་དང་ཨིང་ལ་ག་མ་གཞག་པ་སྟེ། ཕ་རོལ་ན་སན་ཡོད་ནའོ། །ཨིཎ་སན་སྦྱར་བས་ག་མ་ཐོབ་སྟེ་ཨིཊ་དང་སྦྱར། ག་མ་གཉིས་པ་
བྱས། མ་ཕྱི། པ་ནྱ་ཝཎ་སྱའི་མདོས། ཨི་ཐོབ། གས་ཛོའོ། །ཛི་ག་མིཥ་ཏིའོ། །ཨིཎ་ལ་ཡན་སྦྱར་བས་ག་མ་ཐོབ། གོང་དུ་ཨ་དྷི་བྱས། ག་མ་གཉིས་པ་བྱས་པ་ལ་སོགས་སྔ་མ་བཞིན། ཨ་དྷི་ཛི་གཾ་ས་ཏེའོ། །ཁོང་དུ་ཆུད་པའི་དོན་གྱི་ཨིཎ་ལ་ག་མ་མི་འབྱུང་བ་སྟེ། ཨིཎ་གྱི་
གོང་དུ་པྲ་ཏི་བྱས། སན་སྦྱར་བས་སྭ་ར་ཨནྟ་རྣམས་ཀྱི་མདོས་མི་བསྲིངས་ཏེ། ས་མ་ན་དིརྒྷ་སྭ་ར་དྭི་ཡ་སའི་མདོས། ས་གཉིས་པ་བྱས། སནྱ་ཝརྟའི་མདོས་ས་གོང་མས་ཨི་ཐོབ། སས་ཥ་ཐོབ། ཨར་ཐན་པྲ་ཏཱི་ཥི་ཥ་ཏིའོ། །ཨ་ཏེརྦྷུ་ར་སཪྺ་དྷཱ་ཏུ་ཀེ །ཨས་བྷུ་ཞེས་པ་འདི་ལ་
བྷུ་གཞག་པ་སྟེ་ཕ་རོལ་ན་ཨ་སཪྺ་དྷཱ་ཏུ་ཀེ་ཡོད་ནའོ། །ཨ་ས་ལ་ཤྭ་སྟ་ནཱིའི་ཏཱ་སྦྱར་བས་བྷུ་ཐོབ། ཨིཊ་སྦྱར་བས་བྷུ་བྷོར་གྱུར། ཨོ་ཨ་ཝ་དང་སྦྱར་རོ། །བྷ་ཝི་ཏའོ། །ཨས་ལ་ཏི་སྦྱར་བས་བྷུ་ཐོབ་པ་ལ་སོགས་སྔ་མ་བཞིན་ནོ། །བྷ་ཝི་ཥྱ་ཏིའོ། །ཨས་ལེ་སྱ་ཏ་སྦྱར་བས་སྔ་མ་བཞིན།
ཨ་བྷ་ལྦི་ཥྱ་ཏའོ། །ཨས་ལ་ཡཱཏ་སྦྱར་བས་བྷུའོ། །བྷུ་ཡཱ་ཏའོ། །འདི་ནི་ཨ་ས་དི་སྦྱར་བས་བྷུ་ཐོབ། གོང་དུ་ཨ་སྦྱར་བའོ།། །།བྲུ་ཁོ་བ་ཙིཿ། བྲུ་ཉ་ནི་ཚིག་གསལ་བ་ཞེས་བྱ་བ་འདི་ལ་ཝ་ཙི་གཞག་པ་སྟེ། ཕ་རོལ་ན་ཨ་སཪྺ་དྷཱ་ཏུ་ཀེ་ཡོད་ནའོ། །བྲུ་ལ་ཏཱ་སྦྱར་བས། ཝ་ཙ་ཐོབ། ཝརྒ་
སྱའི་མདོས། ཙས་ཀའོ། །ཝསྟཱའོ། །བྲཱུ་ལ་སྱ་ཏི་སྦྱར་བས་སྔ་མ་བཞིན། ཝཀྵྱ་ཏིའོ། །བྲཱུ་ལ་སྱཏ་སྦྱར་བས་སྔ་མ་བཞིན། ཨ་ཝཀྵྱ་ཏའོ། །བྷུ་ལ་ཡཏ་སྦྱར་བས་ཝ་ཙ་ཐོབ། ཝ་ཙ་གཉིས་པ་བྱས། ཙ་ཕྱི། པ་རོ་ཀྵ་ཨ་བྷྱ་སའི་མདོས་ཝས་ཨུ་ཐོབ། ཨ་སྱ་པ་ད་དཱིརྒྷ་ས་
