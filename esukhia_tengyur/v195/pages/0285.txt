དྭཱ་བྷྱམ྄། དྭ་ཡོཿ། ཏས་དང་ཏྲ་ཞེས་པ་འདི་དག་མི་འགྱུར་རོ། །ཐམས་ཅད་པའི་མིང་ཅན་ཞེས་པ་ཅི་ཞེ་ན། བྲྀ་ཀྵ་ཏ། བྲྀ་ཀྵེ། ཐམས་ཅད་པའི་མིང་ཅན་གྱི་བྱ་བ་ཁོ་ན་བྱས་པའི་ཏས་སོགས་རྣམས་ཀྱི་རྣམ་དབྱེའི་མིང་ཅན་དུ་བཤད་པར་བྱ་བའི་བྱ་བ་བཞིན་མ་ཡིན་ནོ། །དེས་
ན། ཏ་དཱ་ཏ། ཀ་དཱ། ཞེས་པ་འདིར་ཀ་ཡིག་རིང་པོའོ་ཞེས་པས་རིང་པོར་མི་འགྱུར་རོ། །དྭི་ཡུཥྨཏ། ཨསྨཏ། བྷ་ཝནྟ། ཀིམ་ཞེས་པ་དྭི་སོགས་སོ། །དེ་ལ་ཨི་དམའི་ཨིའོ། །དེ་ལ་ཨི་དམའི་སྒྲ་ནི་དེ་ལས་ཏེ་རྣམ་དབྱེའི་མིང་ཅན་དེ་རྣམས་ཕ་རོལ་ལ་ཨི་ཡིག་ཏུ་བསྒྱུར་བར་འགྱུར་རོ། །
ཁ་ཅིག་ཏུ་སྔ་མ་དང་ཕྱི་མ་དག་ཅེས་པས་ཐམས་ཅད་ཀྱིའོ། །འདི་ལས། ཨི་ཏཿ། ཨི་ཧ། ཨི་དམ་ལས་ཧའོ་ཞེས་པས་ཧའི་རྐྱེན་ནོ། །ར་ཐ་དག་ལས་ཨེཏ་དང་ཨིཏའོ། །ཨི་དམའི་སྒྲ་ནི་རེ་ཕ་དང་པོ་དང་ཐ་ཡིག་དང་པོའི་རྐྱེན་ཕ་རོལ་ལ་གྲངས་ཇི་ལྟ་བ་བཞིན་དུ། ཨེཏ་དང་ཨིཏ་ཅེས་པ་འདི་
དག་ཏུ་བསྒྱུར་བར་འགྱུར་རོ། །དུས་འདི་ལ། ཨེཏ་རྷི་ཨི་དམ་ལས་རྷི་དང་ཨ་དྷུ་ནཱ་དཱ་ནཱི་མའོ་ཞེས་པས་རྷིའི་རྐྱེན་ནོ། །རྣམ་པ་འདིས། ཨིཏྠཾ། ཨི་དམ་ཀིམ་དག་ལས་ཐ་མུ་ཞེས་པས་ཐ་མུའོ། །དེ་རྣམས་ལས་ཨེ་ཏད་ཨ་ཡིག་ཉིད་དོ། །ཨེ་ཏ་དའི་སྒྲ་ནི་རྣམ་དབྱེའི་མིང་ཅན་དེ་རྣམས་ཕ་
རོལ་ལ་ཨ་ཡིག་ཏུ་འཐོབ་བོ། །དེ་ལས། ཨ་ཏཿ། དེ་ལ། ཨ་ཏྲ། ལྔ་པ་ལས་ཏསའོ། །དྭི་སོགས་མིན་པའི་སརྦིའི་མིང་། །ཀིམ་བ་ཧུ་ལས་ཞེས་པ་རྗེས་སུ་འཇུག་སྟེ་ཕྱི་མར་འང་ངོ་། །ཀིམ་དང་བ་ཧུ་དང་ཐམས་ཅད་པའི་མིང་ཅན་སྟེ་འདི་རྣམས་ལས་ལྔ་པའི་མཐའ་རྣམས་ལས་ཏསའི་རྐྱེན་དུ་
འགྱུར་ཏེ། དྭི་སོགས་དོར་བ་ཉིད་ནའོ། །ཐམས་ཅད་ལས། སརྦ་ཏཿ། གང་ལས། ཡ་ཏཿ། འདི་ལས། ཨ་མུ་ཏཿ། ཅི་ལས། ཀུ་ཏཿ། མང་པོ་རྣམས་ལས། བ་ཧུ་ཏཿ། དྭི་སོགས་མིན་པ་ཅི་ཞེ་ན། དྭཱ་བྷྱཱམ྄ བྷ་ཝ་ཏཿ། ཡུཥྨཏ་ཨསྨཏ་དག་ལ་ནི་ཏས་བཤད་པར་བྱའོ། །ཁྱོད་ལས།
ཏྭཏྟ། བདག་ལས། མཏྟཿ། ཁྱོད་ལས། ཡུཥྨཏྟཿ། བདག་ལས། ཨསྨཏྟཿ། ཐམས་ཅད་པའི་མིང་ཅན་མ་ཡིན་པ་ལས་ཀྱང་། ལྔ་པའི་མཐའ་ལས་ཏས་འདོད་དེ། ཨོཾ་ཧཱག་དང་རུ་ཧི་ཞེས་པ་འདི་དག་གི་སྦྱོར་བ་སྤངས་པ་ཉིད་ནའོ། །གྲོང་ལས། གྲཱ་མ་ཏཿ། དྷརྨ་ཏཿ། ནྱཱ་ཡ་ཏཿ།
