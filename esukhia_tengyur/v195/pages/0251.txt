ཤནྟྲྀངའོ། །ཨན་བི་ཀ་ར་ཎའོ། །ཨ་དང་མཚམས་སྦྱོར་གྱི་ཡི་གེ་ས་དབྱིའོ། །ཨ་དྷཱི་ཡན་པཱ་རཱ་ཡ་ཎཾ། ཨ་དྷི་སྔ་མར་འོ། །ཨིང་དྷཱ་རི་དག་ལས་ཤནྟྲྀང་ཞེས་པས་ཤནྟྲྀངའོ། །ཨན་བི་ཀ་ར་ཎའོ། །ཨད་སོགས་ལས་འཇིག་གོ། །དབྱངས་དང་པོ་ལ་ཨིའི་ཡི་གེ་ཞེས་པས་ཨི་ཡའོ། །ཨཱ་ན་ཤྲའི།

ཨའི་ཨུ་བདྷཱ་ཞེས་པས་རིང་པོའོ། ཨིཥྞུའི། ཀ་ནྱཱ་མ་ལཾ་ཀ་རིཥྞུཿ། ཌུ་ཀྲྀཉ་བྱེད་པ། ཨ་ལཾ་སྔ་མར་འོ། །ཨ་ལཾ་ནི་རབ་ཏུ་ཐོབ་པ་དང་། ཟློག་པ་དང་། རྒྱན་རྣམས་ལའོ། །བྷྲ་ཛི་ཨ་ལཾ་ཀྲྀཉ་ཞེས་པས་ཨིཥྞུའོ། །ཡོན་ཏན་ནོ། །ཀྟྭའི། །ཀ་ཊཾ་ཀྲྀ་ཏྭཱ། བྱེད་པ་པོ་གཅིག་དག་གི་ཞེས་པས་གྷཱའོ། །
ཁལའི། །ཨཱིཥ་དྱ་ཌྷྱ་བྷ་བཾ་བྷ་བ་ཏཱ། བྷཱུ་ཡོད་པ་ལའོ། །ཨཱི་ཥཏ་སྔ་མརའོ། །བྱེད་པ་པོ་དང་ལ་དག་ལའང་བྷཱུ་ཀྲྀཉ་དག་གིའོ་ཞེས་པས་ཁལའོ། །ཡུའི། །ཨཱི་ཥ་ཏསྠ་ནཾ་བྷ་ཝ་ཏཱ། ཥྛ་འགྲོ་བ་ལྡོག་པ་ལའོ། །ཨཱཏ་རྣམས་ལས་ད་རི་དྲ་ཏི་མ་ཡིན་པ་ལས་ཡུའོ་ཞེས་པས་ཡུའོ། །ཡུ་ཝུ་ཛྷ་རྣམས་ཀྱི་
