མཐར་ཡན་སྦྱར་བས། ཨཱི་ཐོབ་ཥས་སའོ། །ཨཱ་བ་སཱི་ཡ་ཏེའོ། །ཧཱ་ལ་ཡང་གོང་བཞིན། ཧཱི་ཡ་ཏེའོ། །ཨཱ་ཤིཥྱེ་ཀཱ་རཿ། མདོ་གོང་མར་སྨོས་པའི་དྷཱ་ཏུ་རྣམས་ཀྱི་ཕ་རོལ་ན་ཨ་ཤཱིཥ་ཨ་གུ་ཎ་ཡོད་ན་ཨེ་འབྱུང་བར་འགྱུར་བ་སྟེ། དཱ་ལ་ཨ་ཤྲཱི་ཥཱ་ཡཏ་སྦྱར་བས་མདོ་འདིས་དཱས་ཨེ་ཐོབ་
པའོ། །དེ་ཡཱཏའོ། །མཱ་ལ་ཡཱཏ་སྦྱར་བས་དཱས་ཨེ་ཐོབ། མེ་ཡཱཏའོ། །གཻ་ལ་ཡཱཏ་སྦྱར་བས། གཻས་ཨེ་ཐོབ་པའོ། །གེ་ཡཱཏའོ། །པཱ་ལ་ཡཱཏ་སྦྱར་བས། པཱས་ཨེ་ཐོབ། པེ་ཡཱཏའོ། །སྠཱ་ལ་ཡང་སྦྱར་བས། སྠཱས་ཨེ་ཐོབ་པའོ། །སྠེ་ཡཱཏའོ། །ཥོ་ལ་གོང་དུ་ཨ་ཝ་སྦྱར། ཥོས་
ཨ་ཐོབ་བོ། །ཨ་ཝ་སེ་ཡཱཏའོ། །ཧ་ལ་ཡཱཏ་སྦྱར་བས་ཨེ་ཐོབ། ཧ་ཡཱཏའོ།། །།ཨ་ན་ཨུསྶི་ཛ་བྷྱསྟ་བི་དཱ་དེ་བྷྱོ་བྷུ་བཿསི་ཙ་དང་ཨ་བྷྱསྟ་དང་། བི་དེ་དང་། ཨཱའི་མཐའ་ཅན་གྱི་ཕ་རོལ་གྱི་ཨན་ལ་ཨུས་གཞག་པར་འགྱུར་བ་སྟེ། བྷུ་ཞེས་པ་འདིའི་ཕ་རོལ་ཏུ་ནི་འབྱུང་བ་མ་ཡིན་ནོ། །སི་
ཙའི་དཔེ་ནི། ཀྲྀ་ལ་ཨན་དང་། བར་དུ་སི་ཙ་སྦྱར་བས། གོང་དུ་ཨ་སྦྱར། ཨན་ཨུ་ཐོབ། རྀ་ཨར་དུ་འཕེལ། སས་ཥ་ཐོབ་པ་དང་སྦྱར་རོ། །ཨ་ཀཱཪྵུཿའོ། །ཨ་བྷྱསྟའི་དཔེ་ནི། ནི་ཛེར་ལ་ཨན་སྦྱར་བས། གོང་དུ་ཨ་དང་ནི་གཉིས་པ་བྱ་སྟེ་གུ་ཎ་ཨན་གྱིས་ཨུས་ཐོབ་བོ། །ཨ་ནེ་
ནི་ཛུཿའོ། །བི་དེའི་དཔེ་ནི། བི་དུ་ལ་ཨན་སྦྱར་བས་ཨན་གྱིས་ཨུ་ཐོབ། ཨ་བི་དུའོ། །ཨཱའི་མཐའ་ཅན་གྱི་དཔེ་ནི། ཡཱ་ལ་ཨུས་སྦྱར་བས་གོང་བཞིན་ནོ། །ཨ་ཡུཿའོ། །ཡཱ་ནི་ཐོབ་པའོ། །བྷུ་མ་ཡིན་པ་ཞེས་ཅིའི་ཕྱིར་སྨོས་ཤེ་ན། བྷཱ་ལ་ཨན་སྦྱར་བས། གོང་དུ་ཨ་སྦྱར།
བྷུ་བྷོ་བོན་ཏས་བར་དུ་ཝ་ཐོབ་སྟེ། ཨ་དང་སྦྱར་རོ། །ཨཱ་བྷུ་ཝནའོ།། །།ཨི་ཙ་ཏ་ལོ་པ། ཨི་ཙའི་ཕ་རོལ་གྱི་རྐྱེན་གྱི་ཏ་དབྱི་བར་བྱ་བ་སྟེ། ཀྲཱི་ལ་ཨཱ་རྱ་ཏ་ནཱི་དི་སྦྱར་བས་གོང་དུ་ཨ་སྦྱར། བར་དུ་ཨི་ཙ་སྦྱར། ཙི་ཕྱི་རྀ་ཨར་དུ་གྱུར་པ་དང་། ཨ་དང་ཀ་སྦྱར། ར་དང་ཨི་སྦྱར། ཨ་ཀཱ་རིའོ། །
ཧྲི་ལ་ཏ་དང་ཨི་ཙ་སྦྱར་བ་གོང་བཞིན། ཨ་ཧཱ་རིའོ། །ཌུ་ཀྲཱིཉ་ནི་བྱེད་པའོ། །ཧྲིཉ་ནི་འཕྲོག་པའོ། །ཧེ་ར་ཀཱ་རཱ་ད་ཧནྟེ། ཨ་ལས་ཕ་རོལ་གྱི་ཨི་ཕྱི་བར་བྱ་སྟེ། ཧན་ཏེའི་ཕ་རོལ་གྱི་ནི་མ་ཡིན་ནོ། །པ་ཙ་ལ་ཧི་སྦྱར་བ་ཕྱི། ཨན་སྦྱར། ཙ་དང་ཨ་སྦྱར་རོ། །པ་ཙའོ། །པ་ཋ་ལ་ཧི་སྦྱར་བ་གོང་
