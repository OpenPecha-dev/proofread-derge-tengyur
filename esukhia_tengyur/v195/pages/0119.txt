པྲ་ཐ་ནི་རབ་ཏུ་གྲགས་པའོ། །ཛ་ནཱི་ནི་རབ་ཏུ་བསྐྱེད་པའོ། །དྲཱི་བྷྱ་ཞེས་པ་འདིའི་དྷ་ཏུ་འཇིགས་པའི་དོན་ལ་ཐུང་བར་འགྱུར་ཏེ། དྲཱི་ལ་ཨིན་སྦྱར་བས། རྀ་ཨཱར་དུ་འཕེལ་ཏེ། ར་དང་ཨི་སྦར། ཨ་དང་སྦྱར་བ་མདོ་འདིས་དཱ་བསྟུངས། ཏི་དང་ཨན་སྦྱར་བ་སྔ་མ་བཞིན། ད་ར་ཡ་ཏིའོ། །ཅིའི་
ཕྱིར་འཇིགས་པ་ཞེས་སྨོས་ཤེ་ན། ཤིང་གཅོད་པ་ཞེས་བྱ་བ་སྟེ། དྲཱི་ལ་ཨིན་སྦྱར་བས་གོང་བཞིན་འཕེལ་ཏེ་ཨ་མ་བསྟུངས་སོ། །ཙ་ལ་ཀམ་པ་ན་ཞེས་པ་འདི་གཡོ་བའི་དོན་ལ་གོང་མ་བཞིན་གསུངས་ཏེ། ཙ་ལ་ཨིན་སྦྱར་བས་ཙ་བསྲིངས་པ་མདོ་འདིས་བསྟུངས། ཏི་དང་ཨན་སྦྱར་བ་སྔ་
མ་བཞིན། ཅིའི་ཕྱིར་གཡོ་བ་ཞེས་སྨོས་ཤེ་ན། ཨུ་ཏྲ་ཙ་ལ་ཨིན་སྦྱར་བས་ཙ་བསྲིངས་ཏེ་མ་བསྟུངས་བའོ། །ཨུ་ཙྩཱ་ལ་ཡ་ཏིའོ། །ཨི་ཙ་བྷཱ། །གང་མ་རྗེས་སུ་འབྲེལ་པ་རྣམས་ཀྱི་ཕ་རོལ་ན་ཀཱ་རི་ཏ་དང་། ཨི་ཙ་ཡོད་ན་དེ་ཐུང་བར་འབྱུང་བའམ། གྷ་ཊ་ཨིཊ་སྦྱར་བས་གྷ་བསྲིངས། ཨ་དྱ་ཏ་ནཱིས་
ཏ་སྦྱར་བས། གོང་དུ་ཨ་སྦྱར། སི་ཙ་སྦྱར་བས་སི་ཙ་ཏ་ལོ་པའི་མདོས། ཏ་ཕྱི། གྷ་བསྟུངས་ཏེ་ཀཱ་རི་ཏ་ཕྱི་ནས། ཊ་དང་ཨི་སྦྱར། ཙ་ཕྱིའོ། །ཨ་གྷ་ཊིའོ། །ཡང་ན་གོང་མ་བཞིན་སྦྱར་ཏེ་གྷཱ་མ་བསྟུངས་སོ། །ཛ་ནི་བ་དྷྱོཤྩ། ཛ་ནའི་དྷཱ་ཏུ་དང་། དེ་བཞིན་དུ་གང་བསྟན་ཏིའི་གནས་སུ་
བནྟ་གཞག་པ་སྟེ། དེ་དག་ཕ་རོལ་ན་ཨི་ཙ་ཡོད་ན་ཉེ་བའི་ཚིག་ཐུང་བར་འགྱུར་རོ། །ཛ་ལ་ཨ་དྱ་ཏཱའི་ཏ་སྦྱར་བས་གོང་དུ་ཨ་སྦྱར། ཨི་ཙ་སྦྱར་བས་ཏ་ཕྱི་སྟེ། ཛ་བསྲིངས་པ། མདོ་འདིས་བསྟུངས་སོ། །ཨ་ཛ་ནའོ། །ཧན་ལ་ཨི་ཙ་སྦྱར་བས། བ་དྷ་ཐོབ། ཨི་སྦྱར་བས་སྔ་མ་བཞིན།
ཨ་བ་དྷིའོ།། །།ཨོ་དཽ་དྦྷྱཾ་ཡིནྣཱ་ཡཱ་སྭ་ར་བཏ། ཨོ་དང་ཨཽ་དག་གི་ཕ་རོལ་ན་ཡིན་དང་། ཨ་ཡིག་དག་ཡོད་ན་དབྱངས་ཀྱི་ངོ་བོ་ལྟ་བུར་འགྱུར་ཏེ། གང་དབྱངས་ཕ་རོལ་ན་ཡོད་པའི་བྱ་བར་བརྗོད་པ་དེ་ཡི་དང་ཨ་ཡི་དག་ལས་འབྱུང་ཞེས་པའི་དོན་ཏོ། །གོ་ལ་ཡིན་སྦྱར་བས་མདོ་འདིས་ཨོས་ཨ་ཝ་
