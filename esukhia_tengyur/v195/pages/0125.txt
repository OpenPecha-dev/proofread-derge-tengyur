ཝ་བསྲིངས། ཨུ་བཱ་ཙའོ། །ཙི་ཀྵིཾ་ཁྱཱཉ། ཙ་ཀྵིང་ནི་ཏིག་གསལ་བ་ཞེས་པ་འདིས་ཁྱཱཉ་ཐོབ་སྟེ། ཨ་སཪྺ་དྷ་ཏུ་ཀེ་ཕ་རོལ་ན་ཡོད་ནའོ། །ཙ་ཀྵིང་ལ་ཏཱ་སྦྱར་བས་ཁྱཱ་ཐོབ། གོང་དུ་ཨཱང་སྦྱར། ཨཱཁྱཱ་ཏཱའོ། །ཙི་ཀྵིང་ལ་སྱ་ཏིང་སྦྱར། སྔ་མ་བཞིན།
ཨཱ་ཁྱཱ་སྱ་ཏིའོ། །ཝཱ་པ་རོ་ཀྵཱ་ཡཾ། པ་རོཀྵཱའི་ཡུལ་གྱི་ཙ་ཀྵིང་ལ་ཁྱཱཉ་བཞག་པའམ། ཙ་ཀྵིང་ལ་ཨེ་སྦྱར། གོང་དུ་ཨཱང་སྦྱར། ཁྱཱ་ཐོབ། ཁྱཱ་གཉིས་པ་བྱས། ཡ་ཕྱི་བས་ཙ་ཐོབ་པའོ། །ཨཱ་ཙ་ཁྱའོ། །ཙ་ཀྵིང་ལ་ཨཱ་ཏེ་སྦྱར་བས་གོང་བཞིན། ཨཱ་ཙ་ཁྱཱ་ཏེའོ། །ཡང་ན་ཙ་ཀྵིང་ལ་ཨེ་སྦྱར།
ཙཀྵིང་གཉིས་པ་བྱས་ཏེ། ཀྵ་ཕྱི་གོང་དུ་ཨ་སྦྱར། ཨཱ་ཙ་ཙ་ཀྵེའོ། །ཨཱ་ཏེ་སྦྱར་བས་གོང་བཞིན་ཏེ། ཨཱ་ཙ་ཙ་ཀྵཱ་ཏེའོ། །ཨི་ཛེ་ཏ་མཱི། ཨ་ཛ་ན་འགྲོ་བ་དང་འདོར་བ་ཞེས་བྱ་བ་འདིའི་ཕ་རོལ་ན་ཨ་སཪྺ་དྷཱ་ཏུ་ཀེ་ཡོད་ན་ཝི་གཞག་པ་སྟེ། ཨ་ཛ་ལ་ཏཱ་སྦྱར་བས་ཕི་ཐོབ་སྟེ། ཨི་ཨེར་གྱུར། བེ་
ཏཱའོ། །ཨ་ཛལ་སྱ་ཏི་སྦྱར་བས་སྔ་མ་བཞིན། བེཥྱ་ཏིའོ། །གྷཉ་དང་། ཨལ་དང་། ཀྱབ་ཕ་རོལ་ན་ཡོད་བཱི་མི་འཐད་དེ། ཨ་ཛ་ལ་གྷཉ་སྦྱར་བས། གྷ་དང་ཉ་ཕྱི། ཨ་དང་ཛ་སྦྱར། གོང་གི་སཾ་དང་ཨ་སྦྱར་བསྲིངས་སོ། །ས་མཱ་ཛཿའོ། །ཨ་ཛའི་གོང་དུ་སམ་སྦྱར། མཐར་ཨལ་
སྦྱར། ལ་ཕྱི། ཨ་དང་ཛ་སྦྱར་རོ། །ས་མཱ་ཛའོ། །ཨ་ཛ་ལ་ཀྱབ་སྦྱར། ཀ་ཕྱི། ཡ་དང་ཛ་སྦྱར། ས་མ་ཛྱའོ། །ཨ་དཱ་དེར་ལུ་གཱི་ཀ་ར་ཎ་སྱེ། ཨ་ད་བྷ་ཀྵ་ཎ་ཞེས་བྱ་བ་ལ་སོགས་པའི་སོགས་པའི་དྷཱ་ཏུའི་ཚོགས་ལས་ཕ་རོལ་གྱི་བི་ཀཱ་ར་ཎ་ཕྱི་བར་བྱ་སྟེ། ཨ་ད་ལ་ཏི་སྦྱར་བས། བར་དུ་ཨན་སྦྱར། མདོ་
འདིས་ཕྱི། རི་ཥེ་ཏ་ཝརྒའི་མདོས། དས་ཏ་ཐོབ་སྟེ་སྦྱར་རོ། །ཨ་དྟིའོ། །ཧ་ན་ལ་ཏི་དང་ཨན་སྦྱར་བ་གོང་བཞིན་ནོ། །ཧིནྟའོ། །དྭེཥ་ལ་ཏི་སྦྱར་བས་སྔ་མ་བཞིན། ཏིས་ཊི་ཐོབ་བོ། །དྭེཥྚིའོ། །ཨ་ད་ནི་བཟའ་བའོ། །ཧ་ན་ནི་འཚེ་བའོ། །དྭེཥ་ནི་སྡང་བའོ།། །།ཛུ་ཧོ་ཏྱཱ་དེཤྩ། ཧུ་ནི་
སྦྱིན་པ་དང་བཟའ་བ་སྟེ། དེ་ལྟ་བུ་ལ་སོགས་པའི་ཇོ་ཏུའི་ཚོགས་ཀྱི་ཕ་རོལ་གྱི་བི་ཀཱ་ར་ཎ་ཕྱི་བར་བྱའོ། །ཧུ་ལ་ཏི་སྦྱར། བར་དུ་ཨན་སྦྱར། མདོ་འདིས། ཧུ་གཉིས་པ་བྱས། ཧས་ཛ་ཐོབ། ཨུ་རྨར་གྱུར། ཛུ་ཧོ་ཏིའོ། །བྷེ་ལ་ཏི་དང་ཨན་སྦྱར་བས་སྔ་མ་བཞིན། བྷས་བ་ཐོབ། བི་
