སྨོས་པ་སྦྲེལ་བར་བྱའོ། །དེ་ལ་དོན་གྱི་དབང་ལས་བདུན་པའི་མཐའ་ཅན་དུ་ཤེས་པར་བྱའོ། །ཧླཱདཱི་བདེ་བ་ལའང་ངོ་ཞེས་པ་འདིའི་ནིཥྛཱ་ཕ་རོལ་ལ་ཐུང་ངུར་འགྱུར་རོ། །པྲཧླནྣཿ། པྲཧླནྣ་ཝཱན། རབ་ཏུ་བདེ་བར་བྱས་སོ་ཞེས་པས་འགྲོ་བའི་དོན་དང་ལས་མེད་པ་ཞེས་པས་ཀྟའོ། །ནིཥྛཱའོ་ཞེས་པས་
ཀྟ་ཝནྟུའོ། །ཌི་ཤྭེ་ཞེས་པས་ཨིཊ་མེད་དོ། །འདིས་ཐུང་ངུའོ། །ད་ལས་སྔ་མའི་དའི་འང་ངོ་ཞེས་པས་ནིཥྛ་ན་ཉིད་དེ། སྔ་མའི་དའི་འང་ནའོ། །ནིཥྛཱ་ལ་ཞེས་པ་ཅི་ཞེ་ན། ཨཱཧླཱ་ད་ཡ་ཏི། ཀུན་ནས་བདེ་བར་བྱེད་པ་ལ་རབ་ཏུ་སྦྱོར་རོ་ཞེས་པས་ཨིནའོ། །ཀྟིའི་རྐྱེན་ལའང་འདོད་དེ། པྲཧླཏྟིཿ། དངོས་པོའི་དོན་ལ་བུད་མེད་
ལ་ཀྟིའོ། །ཚཱ་དིའི་གྷ་ཨིས་མནྟྲ་ན་ཀྭི་པ་རྣམས་ལའང་ངོ་། །ཚཱ་དིའི་གྷལ་སྟེ། ཉེར་བསྒྱུར་གཉིས་མེད་པིའིའོ། །ཚཱད་ཡང་དག་པར་སྒྲིབ་པ་ལའོ་ཞེས་པ་འདིའི་གྷའི་རྐྱེན་ཕ་རོལ་ལ་ཐུང་ངུར་འགྱུར་ཏེ། ཉེར་བསྒྱུར་གཉིས་པའི་ནི་མི་འགྱུར་རོ། །པྲཙྪ་དཿ། དནྟ་ཙྪ་དཿ། ཚཱ་དི་ནི་ཙུར་ལ་སོགས་པ་ཉིད་ཀྱི་ཕྱིར་ཨིནའི་མཐའ་
ཅན་ནོ། །འདིས་སོ་རྣམས་སྒྲིབ་པར་བྱེད་དོ་ཞེས་པས་སྐྱེས་པ་ལ་མིང་ལ་གྷའོ། །གྷ་ཡིག་ནི་ཙ་ཛ་དག་གི་ཀ་ག་དག་གོ་ཞེས་པའི་དོན་ཏོ། །ཐུང་ངུའོ། །ཀཱ་རི་ཏ་དབྱིའོ། །ཉེར་བསྒྱུར་གཉིས་པ་མིན་པའི་ཞེས་པ་ཅི་ཞེ་ན། ས་མུ་པཙྪ་དཿ། སྔ་མ་བཞིན་དུ་གྷའོ། །ཉེར་བསྒྱུར་དུ་མ་ཅན་གྱི་ཚཱ་དིའང་ཐུང་ངུར་འདོད་པ་མ་ཡིན་ཏེ།

པོ་ཉེ་བའི་ཚིག་མི་ཟད་པ་མིན་པའི་ཁ་རྗེས་སུ་འབྲེལ་པ་ལའོ། །ཉེ་བའི་ཚིག་མི་ཟད་པ་མིན་པའི་སྟེ་མི་ཟད་པ་དང་བྲལ་བའི་རིང་པོའི་མཐའ་ཁ་རྗེས་སུ་འབྲེལ་བའི་རྐྱེན་ཡོད་པ་ཐུང་ངུར་འགྱུར་རོ། །ས་ཏིཾ་མནྱ་ཀུ་ལ་ཊཱ། ཏཱ་རུ་ཎཱིཾ་མནྱཱས་བྲིདྡྷཱ། མན་ཤེས་པ་ལའོ། །ས་ཏཱི་དང་ཏ་རུ་ཎཱིའི་སྒྲ་དག་སྔ་མར་རོ། །དམ་པ་མོར་
