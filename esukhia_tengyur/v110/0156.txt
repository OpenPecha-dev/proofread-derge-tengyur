རྒྱས་རྗེས་སུ་དྲན་པ་བླ་ན་མེད་པ་ནི་འདི་ལྟར་བལྟ་བར་བྱ་སྟེ། རྣལ་འབྱོར་པས་ཐོ་རངས་མལ་ནས་ལངས་ཏེ་གཙང་མར་གྱུར་པའམ། མི་གཙང་བ་ཡང་རུང་བས་སྔགས་འདི་ལན་གསུམ་བརྗོད་དོ། །ཨོཾ་སྭ་བྷཱ་བ་ཤུདྡྷ་སྶརྦ་དྷརྨཱ་སྭ་བྷཱ་བ་ཤུདྡྷོ྅ཧཾ། དེ་ནས་འདི་ལྟར་བལྟ་སྟེ། ལུས་འདི་ལྟ་སྟེ། སྐྲ་
དང་། སྨིན་མ་དང་། རྫི་མ་དང་། ཁ་སྤུ་དང་། མཆན་ཁུང་གི་སྤུ་དང་། མདོམས་ཀྱི་སྤུ་དང་། ལུས་གཞན་གྱི་བ་སྤུ་སྟེ། སྤུའི་ཚོགས་དང་། མིག་གཉིས་དང་། རྣ་བ་གཉིས་དང་། སྣ་དང་། སོའི་ཕྲེང་བ་དང་། སེན་མོ་དང་། པགས་པ་དང་། ཤ་དང་། ཚིལ་དང་། ཁྲག་དང་།
