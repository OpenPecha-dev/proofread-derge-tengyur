རབ་རིབ་སྣང་བར་མི་འགྱུར་རོ། །འོ་ན་སངས་རྒྱས་ལ་སྐུ་དང་ཡེ་ཤེས་དང་ཡོན་ཏན་དང་ཕྲིན་ལས་ཀྱང་མི་མངའ་འམ་ཞེ་ན། སྨྲས་པ། རྒྱུ་གསུམ་དང་རྐྱེན་གཅིག་གིས་གདུལ་བྱའི་ངོ་གང་ལ་སྣང་བ་སྟེ། གདུལ་བྱའི་དབྱེ་བས་སྐུ་ཡང་སྣ་ཚོགས་སུ་སྣང་ངོ་། །ཡེ་ཤེས་ཀྱང་རང་འབྱུང་
གི་ཡེ་ཤེས་ཆེན་པོ་ཡིན་ཏེ། བླ་མ་ཨ་ཝ་དྷཱུ་ཏཱི་པ་དང་། བླ་མ་ཏཱ་མྲ་དྭཱི་པའི་ཞལ་ནས། ཡེ་ཤེས་ཆོས་ཀྱི་དབྱིངས་དང་ཐ་མི་དད། །ཆོས་ཀྱི་དབྱིངས་ལ་རང་འབྱུང་ཡེ་ཤེས་བཏགས། །བསམ་དུ་མེད་ཅིང་ཚིག་གི་སྤྲོས་དང་བྲལ། །གདུལ་བྱའི་ངོ་གང་དེ་ཉིད་རྣམ་ལྔར་གསུངས། །ཞེས་སློབ་དཔོན་ཀླུ་སྒྲུབ་ཀྱིས་
གསུངས་ཞེས་གསུངས་སོ། །དེ་བཞིན་དུ་སྟོབས་བཅུ་ལ་སོགས་པ་དང་། ཆོ་འཕྲུལ་གསུམ་དང་། ཕྲིན་ལས་ཉི་ཤུ་རྩ་ལྔ་དང་། ཕྲིན་ལས་སུམ་ཅུ་རྩ་གཉིས་རྣམས་གདུལ་བྱའི་ངོ་གང་ལ་སྣང་ངོ་། །དེ་བས་ན་སློབ་དཔོན་གྱི་ཞལ་སྔ་ནས། མགོན་པོ་སེམས་པ་མི་མངའ་ཞིང་། །རྣམ་རྟོག་གཡོ་བ་མི་མངའ་
ཡང་། །ཁྱོད་ཉིད་(དང་‚ངང་)གིས་སེམས་ཅན་ལ། །སངས་རྒྱས་ཕྲིན་ལས་མཛད་པར་འགྱུར། །ཡང་། རྣམ་རྟོག་རླུང་དམར་ཐམས་ཅད་ཀྱིས། །དཔག་བསམ་ཤིང་ལྟར་མི་བསྐྱོད་ཀྱང་། །དེ་ཡང་སེམས་ཅན་ཐམས་ཅད་ཀྱི། །བསམ་པ་ཡོངས་སུ་རྫོགས་པར་མཛད། །ཡང་།
མགོན་ཁྱོད་སེམས་ཅན་འདུ་ཤེས་ཀྱི། །འཇུག་པ་ཀུན་དུ་མི་མངའ་ཡང་། །སྡུག་བསྔལ་ཅན་གྱི་སེམས་ཅན་ལ། །སྙིང་རྗེ་ཕན་གཏོང་མཛད་པ་ཁྱོད། །ཡང་། གཞན་དོན་ཕུན་སུམ་ཚོགས་པ་ནི། །འབྲས་བུའི་གཙོ་བོ་ཡིན་པར་བཞེད། །སངས་རྒྱས་ཉིད་སོགས་དེ་ལས་གཞན། །
དེ་དག་ཞར་ལ་འབྱུང་བར་བཞེད། །ཅེས་གསུངས་སོ། །ཡང་སློབ་དཔོན་ཨཱརྱ་དེ་བའི་ཞལ་ནས། བསྐལ་པ་གྲངས་མེད་མང་པོ་རུ། །རྟག་ཏུ་གཞན་དོན་འབའ་ཞིག་མཛད། །མཐར་ཐུག་ཆོས་སྐུ་བརྙེས་པ་ན། །ཇི་སྲིད་འགྲོ་བ་གནས་དོན་དུ། །མྱ་ངན་མི་འདའ་གཞན་དོན་མཛད། །
དེ་ཕྱིར་བྱང་ཆུབ་སེམས་དཔའ་ཡིས། །གཞན་དོན་མ་ཡིན་སྤྱད་མི་བྱ། །རང་དོན་ཙམ་ལ་འབད་པ་ནི། །འགྲོ་བ་ལྔ་པོ་ཐམས་ཅད་སྤྱོད། །རྟག་ཏུ་སྙིང་རྗེ་གཞན་གྱི་དོན། །ཉམས་སུ་ལེན་པའི་སྐྱེས་མཆོག་ནི། །འགྲོ་བའི་ཕ་མ་དགའ་དགའ་ན། །ཞེས་གསུངས་སོ། །སྐུ་གསུམ་
