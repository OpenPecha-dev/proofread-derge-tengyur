དང་། ཀླུ་དང་། དྲི་ཟ་དང་། གནོད་སྦྱིན་དང་། མི་འམ་ཅི་དང་། ལྟོ་འཕྱེ་ཆེན་པོ་དང་། ཡི་དགས་དབང་ཆེ་བར་གྲགས་པ་གང་ཡང་རུང་བ་དག་ལ་ཉེས་བྱས་སོ། །དེ་དག་གི་གཟུགས་ནོར་ཅན་དང་། ཚོང་དཔོན་དང་། དེད་དཔོན་དང་། རྒྱལ་པོ་དང་བློན་པོ་ལྟ་བུ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །དེ་
དག་གི་སྒྲ་པི་ཝང་དང་། འོད་མ་དང་། བ་ལ་རི་དང་། མ་ཧ་ཏི་དང་། སུ་གྷོ་ཥ་ཀ་ལྟ་བུ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །དེ་དག་གི་དྲི་ཨ་ཀ་རུ་དང་། ཙན་དན་དང་། གུར་ཀུམ་དང་། ཏ་མ་ལའི་ལོ་མ་ལྟ་བུ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །དེ་དག་གི་རེག་པ་དར་དང་། རས་ཡུག་ཆེན་
དང་། བེའུ་རས་དང་། དར་ལ་དང་། དུ་གུ་ལའི་རས་དང་། ཀོ་ཏམ་པའི་རས་ལྟ་བུ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །གོ་བར་བྱ་བ་དགེ་སློང་མ་ཡིན་པ་ལ་ཉེས་བྱས་སོ། །དེ་དག་འདི་ན་ཡོད་དོ་ཞེའམ་ཕ་གི་ན་ཡོད་དོ་ཞེས་ཟེར་བ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །དེ་དག་གིས་ཁྱོད་
ལ་ཡིད་དུ་འོང་བའི་རྒྱུ་ཉེ་བར་བསྒྲུབ་པར་འགྱུར་རོ་ཞེས་ཟེར་བ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །དེ་དག་བདག་ཉིད་ཀྱིས་གོ་བར་བྱེད་དམ། གཞན་གྱིས་གོ་བར་བྱེད་དུ་བཅུག་པ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །ཡིད་དུ་འོང་བའི་རྒྱུས་དགའ་བ་སྐྱེས་པའམ། མ་སྐྱེས་པ་གང་ཡང་རུང་བ་ལ་
ཉེས་བྱས་སོ། །དེ་དག་གིས་སྐྱོ་བར་བྱེད་པའི་ཆེད་ཉིད་མ་གཏོགས་པ་གང་ཡང་རུང་བ་ལ་ཉེས་བྱས་སོ། །སྦྱོར་བའི་སྦྱོར་བའི་སྟུང་བ་ནི་ཉེས་བྱས་ཕྲ་མོ་སྟེ། སྦྱོར་བ་ཇི་སྙེད་པ་ལ་སྦྱོར་བའི་སྦྱོར་བ་ཡང་དེ་སྙེད་དེ། སྦྱོར་བ་ཉེས་བྱས་ཀྱི་སྦྱོར་བའི་སྦྱོར་བ་ཡང་ཉེས་བྱས་ཏེ་ཉེས་བྱས་ཕྲ་མོའོ། །ལྟུང་བ་
མེད་པ་ནི་སྐྱོ་བར་བྱ་བའི་ཆེད་ཉིད་ལ་ལྟུང་བ་མེད་དོ། །དེའི་ཕྱིར་སྐྱོ་བར་བྱ་བའི་ཆེད་ཉིད་ལ་སེམས་ཅན་དམྱལ་བ་དང་། དུད་འགྲོ་དང་ཡི་དགས་ཀྱི་གཏམ་ལ་སོགས་པ་ལ་ལྟུང་བ་མེད་དོ། །ལས་དང་པོ་པ་དང་། སྨྱོས་པ་དང་། སེམས་འཁྲུགས་པ་དང་། ཚོར་བ་དག་གིས་ཉེན་པ་ལ་ལྟུང་བ་མེད་
དོ། །བསླབ་པར་བྱ་བ་ནི་དགེ་སློང་གིས་ཡུལ་ཐམས་ཅད་ལ་གནས་སྐབས་ཐམས་ཅད་དུ་སྔངས་པར་བྱེད་པའི་ལྟུང་བྱེད་དུ་འགྱུར་བ་རྣམ་པ་ཐམས་ཅད་མི་བསྟེན་པར་བསླབ་པར་བྱའོ། །སྐྱོ་བར་བྱ་བའི་ཆེད་ཉིད་དུ་དེ་དག་བསྒྲག་པར་བྱའོ། །ཕྱིར་བཅོས་པ་ནི་སྔ་མ་བཞིན་ནོ། །སྔངས་པར་
