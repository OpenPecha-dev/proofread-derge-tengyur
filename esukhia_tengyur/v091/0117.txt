པའི་ཕྱིར། གསོལ་པ། བཅོམ་ལྡན་འདས་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོས་ཇི་ལྟར་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ལ་བརྩོན་པར་བགྱིས་ན་བརྩོན་ཞེས་བགྱི་བ་ལགས་ཞེས་བྱ་བ་གསུངས་ཏེ། ཇི་ལྟར་བརྩོན་པར་བགྱིས་ན་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་
པ་ལ་བརྩོན་པར་ཆུད་ཅེས་ཞུས་པའོ། །ཤ་ར་དྭ་ཏིའི་བུ་འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་སེམས་དཔའ་ཆེན་པོ་གཟུགས་སྟོང་པ་ཉིད་ལ་བརྩོན་ན་བརྩོན་ཞེས་བྱའོ་ཞེས་བྱ་བ་ལ་སོགས་པས་བརྩོན་པར་བྱེད་པ་རྣམས་ཀྱི་མཚན་ཉིད་བསྟན་ཏེ། ཕུང་པོ་ལྔ་དང་། སྐྱེ་མཆེད་བཅུ་གཉིས་
དང་། ཁམས་བཅོ་བརྒྱད་དང་། འཕགས་པའི་བདེན་པ་བཞི་དང་། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་བཅུ་གཉིས་དང་། འདུས་བྱས་དང་འདུས་མ་བྱས་པའི་ཆོས་རྣམས་<སྙོང་,སྟོང་>པ་ཉིད་ལ་བརྩོན་པར་བྱེད་ན་བརྩོན་པར་ཆུད་དོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེ་ལ་ཕུང་པོ་དང་། ཁམས་དང་། སྐྱེ་
མཆེད་དང་བདེན་པ་དང་། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་རྣམས་སྨོས་པ་ནི། ཕུང་པོ་ལ་སོགས་པ་སོ་སོའི་མཚན་ཉིད་བྱེ་བྲག་ཏུ་ཕྱེ་སྟེ་སྟོང་པ་ཉིད་དུ་བསྟན་ཏོ། །འདུས་བྱས་དང་། འདུས་མ་བྱས་པའི་ཆོས་གང་ཇི་སྙེད་མིང་དུ་གདགས་སུ་ཡོད་ཅེས་བྱ་བ་དང་། རང་བཞིན་སྟོང་པ་ཉིད་ཅེས་
བྱ་བ་ནི་ཆོས་ཐམས་ཅད་སྤྱིར་བསྡུས་པའི་མཚན་ཉིད་སྟོང་པ་ཉིད་དུ་བསྟན་པའོ། །སྟོང་པ་ཉིད་བདུན་པོ་འདི་དག་ལ་བརྩོན་ན་བརྩོན་ཞེས་བྱའོ་ཞེས་བྱ་བ་ནི། ཕུང་པོ་དང་། ཁམས་ལ་སོགས་པ་སོ་སོར་ཕྱེ་བ་བདུན་པོ་ལ་བྱའོ། །དེ་སྟོང་པ་ཉིད་བདུན་པོ་འདི་དག་གིས་ཤེས་རབ་ཀྱི་ཕ་
རོལ་ཏུ་ཕྱིན་པ་ལ་བརྩོན་པར་བྱེད་ཀྱང་བརྩོན་ཞེ་འམ། མི་བརྩོན་ཞེས་མི་བརྗོད་དེ་ཞེས་བྱ་བ་ནི། དེ་ལྟར་བརྩོན་པ་རྣམས་ཀྱི་མཚན་ཉིད་བསྟན་ན་བརྩོན་པར་བྱེད་པ་དེ་ལྟ་བུ་དོན་དམ་པར་དངོས་པོ་ཡོད་དོ་སྙམ་དུ་རྟོགས་པ་ལས་བཟློག་པ་སྟེ། དེ་ལྟར་བརྩོན་པར་བྱེད་པ་རྣམས་ཀྱི་མཚན་ཉིད་
བསྟན་པ་ཡང་ཀུན་རྫོབ་ཀྱི་ཐ་སྙད་ཙམ་དུ་བསྟན་དུ་ཟད་ཀྱི། དོན་དམ་པར་མཚན་ཉིད་མ་གྲུབ་པའི་ཕྱིར་བརྩོན་ཞེས་བྱ་བའམ་མི་བརྩོན་ཞེས་བྱ་བར་བརྗོད་དུ་མེད་དོ་ཞེས་བསྟན་པའོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། འདི་ལྟར་དེས་གཟུགས་བརྩོན་ཞེས་བྱ་བའམ་མི་བརྩོན་ཞེས་བྱ་བར་ཡང་དག་པར་
