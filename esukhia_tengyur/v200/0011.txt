བསྡུས་མིན་རྒྱས་མིན་པར། །ཞེས་བྱ་བ་ལ། བསྡུས་པ་ནི་གཞུང་ཉུང་ངུ་ཞིག་སྦྱར་བའོ། །རྒྱས་པ་ནི་གཞུང་མང་པོ་སྦྱར་བ་ཡིན་ནོ། །གང་གི་ཕྱིར་ཤིན་ཏུ་བསྡུས་པ་ནི་བློ་དམན་པ་རྣམས་ཀྱི་དོན་དུ་མི་འགྱུར་ལ།
དེ་རྣམས་ཉིད་ལ་ཤིན་ཏུ་རྒྱ་ཆེར་བཤད་ན་ཡང་མཉན་པ་དང་། གོ་བ་དང་ཉམས་སུ་ལེན་པར་མི་ནུས་སོ། །དེ་བས་ན་བློ་དམན་པ་དང་། བར་མ་དང་། མཆོག་དང་ལྡན་པའི་སློབ་མ་རྣམས་ཀྱི་དོན་དུ་རྒྱུད་འདི་བྱས་པ་ཡིན་ནོ། །ཡན་
ལག་བརྒྱད་པའི་སྙིང་པོ་ཞེས་བྱ་བ་ནི། ཡན་ལག་བརྒྱད་པོ་རྣམས་ཀྱི་སྙིང་པོ་སྟེ། དེ་ནི་སྙིང་དང་འདྲ་བས་ན་སྙིང་པོ་སྟེ། སྙིང་པོའི་སྒྲ་ནི་གཅེས་པའི་ཚིག་ཡིན་ཏེ། གཅེས་པར་གྱུར་པ་གཙོ་བོར་བྱས་པས་གཅེས་པའི་དོན་ཏོ། །ཡན་
ལག་བརྒྱད་པོ་དེ་རྣམས་ཀྱང་གང་ཞིག་ཡིན་ཞེ་ན། ལུས་དང་བྱིས་པ་གདོན་ལུས་སྟོད། །མཚོན་དང་མཆེ་བ་རྒས་རོ་ཙ། །གསོ་དཔྱད་གང་ལ་གནས་པ་ཡི། །ཡན་ལག་བརྒྱད་དུ་དེ་བཤད་དོ། །ཞེས་སྨོས་པ་ཡིན་ཏེ། ཚེའི་རིག་བྱེད་
ཅེས་བྱ་བའི་རྒྱུད་དེའི་ཡན་ལག་རྣམ་པ་བརྒྱད་ནི་ལུས་ཀྱི་ཡན་ལག་གསོ་བའི་དཔྱད་གནས་ཤིང་བརྟེན་པའི་ཡན་ལག་བརྒྱད་དུ་དྲང་སྲོང་རྣམས་ཀྱིས་བཤད་ཅིང་བསྟན་ཏོ། །དེ་ལ་གསོ་དཔྱད་ཅེས་བྱ་བ་ནི། ནད་བསལ་ཞིང་ཞི་བར་བྱས་ནས། ཁམས་མཉམ་པར་བྱ་
བའི་དོན་གྱི་བྱ་བའོ། །དེ་ཡང་ལུང་ལས། སྨན་པ་ལ་སོགས་རྣམ་པ་བཞི། །བསྔགས་པས་ཁམས་ནི་འགྱུར་བ་ལ། །ཁམས་སྙོམས་ཕྱིར་ནི་རབ་འཇུག་པ། །གསོ་དཔྱད་ཅེས་ནི་བྱ་བར་བཤད། །ཅེས་འབྱུང་བ་ཡིན་ནོ། །དེ་ལ་ལུས་ཀྱི་གསོ་བ་ནི་ཡན་ལག་
ཐམས་ཅད་ལ་བརྟེན་པའི་ནད་རིམས་དང་། འཁྲུ་བ་དང་། སྐྲང་པ་དང་། ཁྲག་དང་། མཁྲིས་པ་དང་། མཛེ་དང་། གཅིན་རྟུག་འགགས་པ་ལ་སོགས་པ་ཉེ་བར་ཞི་བར་བྱ་བའི་དོན་ཏོ། །བྱིས་པ་གསོ་བ་ནི་གཞོན་ནུ་གསོ་བ་ན་མའི་ནུ་ཞོའི་ཉེས་པ་
