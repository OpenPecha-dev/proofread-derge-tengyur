ནད་ཀྱིས་གཞོམ་མེ་དྲོད་ཆུང་། །རྒྱལ་པོ་ཕྱུག་པོ་ལ་བདེ་བ། །ཡིན་པས་རྟག་ཏུ་བསྟེན་པར་བྱ། །ནད་འཇོམས་ཡང་བ་མེད་པ་དང་། །སྟོབས་སྐྱེད་བཤང་བ་བདེ་བར་འབྱུང་། །ཞེས་བྱ་བ་ལ། སྣུམ་ལས་ཞུན་མར་འབའ་ཞིག་བཏུང་བའི་ཆུང་ངུ་ཐུན་གཉིས་ཀྱིས་འཇུ་བར་འགྱུར་བའི་མཚན་
ཉིད་ཅན་གྱི་ཚད་དང་མཉམ་པར་བྱས་པའི་སྣུམ་གྱི་ཚད། འཇམ་རྩིའི་ཚད་དུ་བཤད་དེ། དེ་ནི་འཇམ་རྩིའི་ཆུང་ཚད་ཅེས་བྱ་བ་ལ་ཚད་དེ་ནི་བྱིས་པ་དང་རྒན་པོ་ལ་སོགས་པས་དུས་རྟག་ཏུ་རྒྱུན་མི་འཆད་པར་བསྟེན་ཅིང་གོམས་པར་བྱའོ། །དེ་བསྟེན་ཅིང་གོམས་པར་བྱེད་པ་ནི་ནད་མེད་
པར་བྱེད་པ་དང་། བདེ་བ་ཡིན་པས་བྱིས་པ་ལ་སོགས་པ་ལ་ཡང་གཏོང་དུ་རུང་བས་གོམས་པར་སྤྱད་པར་བྱའོ། །དེ་བཞིན་དུ་ནད་འཇོམས་པ་སྟེ། རླུང་དང་མཁྲིས་པ་དང་བད་ཀན་ཡང་འཇོམས་ཤིང་སེལ་བར་བྱེད་དོ། །དེ་བཞིན་དུ་ཡང་འདི་ནི་ཡང་བ་མེད་པར་བྱེད་པ་དང་། ཁ་ལག་ཆུ་དྲོན་
གྱིས་བཀྲུ་ཞིང་། །ཞེས་བྱ་བ་ལ་སོགས་པའི་གཞུང་གིས་སྔར་བཤད་པའི་འདམ་ག་མི་དགོས་པ་ཡིན་ལ། དེ་ཉིད་ཀྱིས་ན་བྱིས་པ་ལ་སོགས་པ་ལ་བདེ་བའི་བདག་ཉིད་ཅན་ཡིན་ནོ། །དེ་བཞིན་དུ་སྟོབས་སྐྱེད་པར་བྱེད་པ་ཡིན་ཏེ། ནུས་པ་སྐྱེད་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་བཤང་བ་བདེ་བར་འབྱུང་
བར་བྱེད་པ་ཡིན་ནོ། །གང་ཞིག་བཏང་བས་བཤང་བ་བདེ་བླག་ཏུ་འབྱུང་བར་བྱེད་པ་དེ་ལ་དེ་སྐད་ཅེས་བྱའོ། །མས་གཏོང་བུད་མེད་རོ་སྨད་ཀྱི། །མངལ་དང་མོ་མཚན་ནད་རྣམས་ལ། །བཀྲུས་དྲགས་གཉིས་གསུམ་གྱིས་དག་པར། །མས་གཏོང་གོང་མས་གཏོང་བར་བྱ། །ཞེས་བྱ་བ་ལ།
འདིར་མས་གཏོང་བའི་ལུས་ཀྱི་ཡན་ལག་གི་ཕྱོགས་གཅིག་འབབ་པའི་ཕྱོགས་ནས་གཏང་བ་སྟེ། དེ་ནི་གཅིན་སྲི་བ་ལ་སོགས་པའི་ནད་གང་དག་ཡོད་པའི་སྐྱེས་པ་དང་། བུད་མེད་དང་མ་ནིང་གི་མཚན་ནས་གཏང་བ་གོང་མ་གཏང་བར་བྱའོ། །དེ་དག་ཐམས་ཅད་ཀྱང་ལན་གཉིས་སམ་
ལན་གསུམ་དུ་བཀྲུ་སྨན་དྲག་པོ་མས་གཏོང་བས་དག་ཅིང་སྐྱོན་བསལ་བ་ལ། མས་གཏོང་བ་གོང་མ་གཏང་བར་བྱའོ། །དེ་ཡི་གཅེའུ་ནད་པ་ཡི། །སོར་ཚད་བཅུ་གཉིས་ཚད་དུ་བྱ། །སྣུམ་ཞིང་བ་ལང་མཇུག་མ་འདྲ། །རྩ་བ་དབུས་སུ་གག་པ་གདགས། །རྩེ་མོ་ཡུངས་ཀར་ཤོང་
