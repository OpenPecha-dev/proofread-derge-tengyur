རྫས་སྐམ་པོ་དང་། གཤེར་བ་ཐམས་ཅད་འོ་མ་དང་ལྷན་ཅིག་མི་འཕྲོད་པ་ཡིན་ནོ་ཞེས་བཤད་པ་ཡིན་ནོ། །དེ་བཞིན་རྒྱ་སྲན་ཁྲེ་དག་དང་། །བ་ལ་ཀུ་མུ་ཏ་མི་འཕྲོད། །ཅེས་བྱ་བ་ལ། དེ་བཞིན་དུ་རྒྱ་སྲན་ཀུ་ལ་ད་ཐ་ལ་སོགས་པ་ཡང་འོ་མ་དང་ལྷན་ཅིག་མི་འཕྲོད་པ་
ཡིན་ནོ། །ལ་ཕུག་ལ་སོགས་སྔོ་རྔད་ནི། །ཟོས་ནས་འོ་མ་སྤང་བར་བྱ། །ཞེས་བྱ་བ་ལ། ལ་ཕུག་ལ་སོགས་པའི་སྔོ་རྔད་རྗེན་པ་ཟོས་ནས་འོ་མ་སྤང་བར་བྱའོ། །ལ་སོགས་པའི་སྒྲས་ནི། སྐུ་ཐི་ར་དང་། ཤི་ཀྲུ་དང་། བྱེའུ་རུག་པ་ལ་སོགས་པ་བསྡུའོ། །དེ་ཡང་དྲང་སྲོང་ཙ་
ར་ཀས། མཛེའི་ནད་དུ་འགྱུར་བའི་འཇིགས་པས། ལ་ཕུག་དང་། ཡུངས་ཀར་གྱི་ལོ་མ་དང་། སྒོག་སྐྱ་དང་། གྲི་ན་གནྡྷ་དང་། ཟིན་ཐིག་ལེ་སུ་མུ་ཁ་དང་། སུ་ར་ལ་སོགས་པའི་སྒྲས་ཟོས་ནས། འོ་མ་བཏུང་བར་མི་བྱའོ་ཞེས་བཤད་པ་ཡིན་ནོ། །སྔོ་རྔད་ཅེས་སྨོས་པ་ནི་ལ་ཕུག་
གི་ཁྱད་པར་ལ་བྱ་བར་བསྟན་ཏེ། འདི་ལྟར་སྨོས་པ་ལས་ན་འདི་ནི་རྗེན་པར་ཡང་བཟར་རུང་བར་བསྟན་པ་ཡིན་ལ། སྔར་ཚོད་མའི་སྡེ་ཚན་གྱི་ནང་དུ་འདི་སྨོས་པས་ནི། ཚོད་མ་ཉིད་དུ་བྱར་རུང་བར་སྟོན་པ་ཡིན་ནོ། །དེ་ལྟར་ན་རྗེན་པ་ཟོས་ནས་འོ་མ་མི་བཏུང་བ་ཡིན་གྱི། ཚོད་མར་བྱས་པ་མ་
ཡིན་ཏེ། ལེགས་པར་བཙོས་ནས་ཁུ་བ་བཙིར་བ་དང་། སྣུམ་ལ་སོགས་པས་བརྔོས་ཏེ། ཚོད་མར་གཡོས་སུ་བྱས་པའི་རྗེས་ལ་འོ་མ་མི་བཏུང་བ་ནི་མ་ཡིན་ནོ། །ཕག་རྒོད་བྱི་དུར་ཤ་མི་བཟའ། ཁྱིམ་བྱ་བྲི་ཤ་ཏ་ཞོར་མིན། །ཞེས་བྱ་བ་ལ། ཕག་རྒོད་ཀྱི་ཤ་དང་། བྱི་དུར་ཤ་
ལྷན་ཅིག་མི་བཟའ་བ་དང་། དེ་བཞིན་དུ་ཁྱིམ་བྱ་དང་། བྲི་ཤ་ཏའི་ཤ་ཞོ་དང་ལྷན་ཅིག་མི་བཟའོ། །ཤ་རློན་མཁྲིས་པ་དག་དང་ནི། །ལ་ཕུག་མ་ཤའི་ཚོད་མར་མིན། །ཞེས་བྱ་བ་ལ། ཤ་རློན་པ་མཁྲིས་པ་དག་དང་ལྷན་ཅིག་མི་བཟའ་བ་དང་། དེ་བཞིན་དུ་ལ་ཕུག་མ་ཤའི་ཚོད་
མ་དང་ལྷན་ཅིག་མི་བཟའོ། །ལེ་བརྒན་རྩི་ཚོད་ལུག་ཤར་མིན། །མྱུ་གུའི་ཚོད་མ་པད་རྩར་མིན། །ཞེས་བྱ་བ་ལ། ལེ་བརྒན་རྩིའི་ཚོད་མ་དང་། ལུག་ཤ་ལྷན་ཅིག་མི་བཟའ་བ་དང་། དེ་བཞིན་དུ་འབྲུའི་མྱུ་གུ་སྐྱེས་པའི་ཚོད་མ་དང་། པདྨའི་རྩ་བ་ལྷན་ཅིག་བཟའ་བར་མི་བྱའོ། །
