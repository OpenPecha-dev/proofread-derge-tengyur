རབ་ཐོབ་པར་བྱེད་པའོ། །དེ་བཞིན་དུ་བད་ཀན་རླུང་ནི་སེལ་ཞེས་བྱ་བ་སྟེ། བད་ཀན་དང་རླུང་སེལ་ཞིང་འཇོམས་པར་བྱེད་ལ། དེ་ཡང་མཆོག་ཏུ་ཞེས་ཤིན་ཏུ་སེལ་བར་བྱེད་པའོ། །ཀུ་ཤུ་རྣམ་གཉིས་ལྕི་བ་སྟེ། །མངར་སྐྱུར་བསིལ་དང་ཚ་བ་ཡིན། །ཡིག་འབྱེ་ཞིང་མེ་རབ་འབྲི། །ཞེས་
བྱ་བ་ལ། ཀུ་ཤུ་ནི་ཅོའུ་སྟེ། དེ་ཡང་རྣམ་པ་གཉིས་ཞེས་བྱ་བ་མངར་བ་དང་སྐྱུར་བ་རྣམ་པ་གཉིས་ཡོད་དོ། །དེ་གཉི་ག་ཡང་ལྕི་བ་ཡིན་ལ། དེ་གཉིས་ལས་གོ་རིམས་བཞིན་དུ་མངར་བ་ནི་བསིལ་བ་ཡིན་ལ། སྐྱུར་བ་ནི་ཚ་བའི་ནུས་པ་དང་ལྡན་པ་ཡིན་ནོ། །དེ་བཞིན་དུ་ཡི་ག་འབྱེ་བར་
བྱེད་ཅིང་མེའི་དྲོད་རབ་ཏུ་འབྲི་བར་བྱེད་པ་ཡིན་ནོ། །ཁམ་ཤ་ཡིག་འབྱེད་ཅིང་མངར། །སྨིན་པ་མྱུར་དུ་འཇུ་བར་འགྱུར། །ཤིན་ཏུ་ཚ་མིན་ལྕི་ཉེས་སྐྱེད། །ཅེས་བྱ་བ་ལ། ཁམ་ཤའི་དབྱེ་བ་ནི་རྣམ་པ་བཞི་སྟེ། དེ་ཡང་སྨན་དཔྱད་ཀྱི་ལུང་ལས། ཁམ་ཤ་དང་ནི་ཁམ་རྒོད་དང་། །
ཙི་ལི་སོགས་དང་རྒྱ་ཁམ་སྟེ། །དེ་ནི་རིགས་ཀྱི་ཁྱད་པར་གྱིས། །ཁམ་བུ་རྣམ་པ་བཞིར་ཤེས་བྱ། །ཞེས་བཤད་པ་ཡིན་ནོ། །དེ་བཞི་ཆར་ཡང་ཡི་ག་འབྱེད་ཅིང་ཟས་ལེན་པར་བྱེད་པ་དང་། རོ་མངར་བ་དང་། ཤིན་ཏུ་སྨིན་ཅིང་ཚོས་ནས་ནི་མྱུར་དུ་འཇུ་བར་འགྱུར་བ་ཡིན་ནོ། །
ཤིན་ཏུ་ཚ་མིན་ཞེས་བྱ་བ་ནི། ཅུང་ཟད་ཙམ་ནུས་པ་ཚ་བ་དང་། དེ་བཞིན་དུ་ཤིན་ཏུ་ལྕི་བ་དང་། ཤིན་ཏུ་ཉེས་པ་སྐྱེད་པ་ཡིན་ཏེ། ཅུང་ཟད་ལྕི་བ་དང་། ཅུང་ཟད་ཉེས་པ་སྐྱེད་པའོ། །རྒུན་འབྲུམ་པ་རུ་ཤ་ཀ་དང་། །ཀ་ར་མར་དང་བཙོས་རློན་སྐྱུར། །མཁྲིས་པ་བད་ཀན་སྐྱེད་
ཅིང་འཁྲུ། །ལྕི་ཞིང་ནུས་ཚ་རླུང་འཇོམས་སོ། །ཞེས་བྱ་བ་ལ། རྒུན་འབྲུམ་དང་། པ་རུ་ཤ་ཀ་དང་། ཀ་ར་མར་དང་བཙོས་པ་རློན་པ་སྟེ། སྐམ་པོ་མ་ཡིན་པ་དང་། མ་སྨིན་ཅིང་སྐྱུར་བར་འགྱུར་བ་ནི་མཁྲིས་པ་དང་བད་ཀན་འཕེལ་ཞིང་སྐྱེད་པར་བྱེད་པ་དང་། ཁོང་འཁྲུ་བ་
དང་ལྕི་བ་དང་། རླུང་འཇོམས་ཤིང་སེལ་བ་དང་། ནུས་པའི་སྟོབས་ཚ་བ་ཡིན་ནོ། །རྒྱ་ཤུག་སྐྱུར་ཐུང་ལི་གུ་ཚེ། །ཨ་མྲ་ཀ་ཏ་དེ་བཞིན་ནོ། །ཞེས་བྱ་བ་ལ། སྔར་བཤད་པ་བཞིན་དུ་རྒྱ་ཤུག་ནི་ཁམ་བུའི་འབྲས་བུ་དག་དང་ཆ་འདྲ་བ་ཡིན་ཏེ། རྒྱ་ཤུག་སྐྱུར་པོ་ནི་
