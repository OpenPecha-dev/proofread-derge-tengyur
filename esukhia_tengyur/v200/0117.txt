སྦྱར་བ་བཏུང་བར་མི་བྱ་བ་དང་། དེ་བཞིན་དུ་སྦྲང་རྩི་དང་མར་དང་སྦྱར་བའམ། མར་དང་ཤ་ཁ་ར་དཀར་པོ་དང་། ཤ་ཁ་ར་དང་མི་ལྡན་པར་བཏུང་བ་དང་། སྐྱུ་རུ་ར་མ་ཟོས་པར་མི་བཏུང་ངོ་། །དེ་བཞིན་དུ་རྟག་པར་རྒྱུན་དུ་མི་བཏུང་བ་དང་། མ་ལངས་མིན་ཞེས་བྱ་བ་སྟེ།
ལེགས་པར་མ་ཆགས་ལ། གང་གི་ཚེ་འོ་མའི་རང་བཞིན་མ་སྤངས་ཤིང་ཞོར་ལེགས་པར་མ་ཆགས་པ་དེའི་ཚེ་མ་ལངས་པ་ཞེས་བྱ་སྟེ། དེ་མི་བཏུང་ངོ་། །གཞན་དུ་ཞེས་བྱ་བ་ནི་སྔར་བཤད་པའི་ཆོ་ག་དང་མི་ལྡན་ཞིང་དེ་མ་སྤངས་པར། ལ་ལ་ཞིག་སྤྱོད་པར་བྱེད་ན། དེའི་ཚེ་ཞོ་དེས་རིམས་ལ་
སོགས་པ་སྐྱེད་པར་འགྱུར་རོ། །དར་སར་ཡང་ལ་བསྐ་ཞིང་སྐྱུར། །དྲོད་སྐྱེད་བད་ཀན་རླུང་དག་སེལ། །སྐྲན་ནད་དམུ་རྫིང་གཞང་འབྲུམ་དང་། །ཕོ་ནད་ཆུ་སྲི་ཡི་ག་འཆུས། །མཆེར་པ་སྐྲན་དང་མར་མ་ཞུ། །སྦྱར་དུག་སྐྱ་རྦབ་ནད་རྣམས་སེལ། །ཞེས་བྱ་བ་ལ། དར་སར་ཞེས་བྱ་བ་ལ་ཞོ་
སྲུབ་ཤིང་ལ་སོགས་པས་བསྲུབས་པ་ལས་བྱུང་བ་སྟེ། སྲུབ་ཤིང་སྣུམ་མེད་པ་དང་། མར་གྱི་ཕྱེད་ཕྱུང་བ་དང་། མར་མ་ཕྱུང་བ་སྟེ། གསུམ་ཆར་ཡང་ཡང་བ་ཡིན་ལ། དེ་ལས་རོ་ནི་བསྐ་བ་དང་སྐྱུར་བ་ཡིན་ནོ། །དྲོད་སྐྱེད་པ་ནི་ཕོ་བའི་མེ་དྲོད་འཕེལ་བར་བྱེད་པའོ། །བད་ཀན་རླུང་དག་སེལ་
ཞེས་བྱ་བ་ནི། དེ་ལས་གཉིས་བསྲུབས་པ་ལས་བྱུང་བ། དེ་བཞིན་དུ་སྐྲན་ནད་དང་དམུ་རྫིང་ལ་སོགས་པ་སེལ་བར་འགྱུར་བ་དང་། ཆུ་སྲི་བ་སྟེ། གཅིན་སྲི་བ་དང་མཆེར་བ་ཞེས་པ་ལྟོ་བའི་གཡོན་ལོགས་ན་བརྟེན་པའི་རྣམ་པ་ན་བ་དང་། མར་མ་ཞུ་ཞེས་པ་མར་ལྷག་པར་ཟོས་པའམ།
མ་ལེགས་པར་ཟོས་པ་ལས་བྱུང་བའི་ནད་ལ་ནི་མར་མ་ཞུ་བ་ཞེས་བྱའོ། །སྦྱར་དུག་ནི་བཅོས་མའི་དུག་གོ། །སྐྱ་རྦབ་ཀྱི་ནད་ནི་སྐྱ་རྦབ་ལས་ནད་དུ་གྱུར་པ་སྟེ། ནད་དེ་རྣམས་སེལ་ཞིང་འཇོམས་པ་ཡིན་ནོ། །དེ་འདྲ་ཞོ་ག་ཆུ་འཁྲུ་ཡང་། །རྩ་སྲུབས་སྦྱོང་ལ་རྟུག་སྐམས་སེལ། །ཞེས་བྱ་བ་ལ། དེ་
འདྲ་ཞེས་པ་ནི་དར་བ་སར་པ་སྟེ་དེ་དང་ཞོ་ག་ཆུ་ཡོན་ཏན་འདྲ་ཞིང་མཚུངས་པར་ཤེས་པར་བྱ་ལ། དེ་ཡང་འཁྲུ་བ་སྟེ་ཅུང་ཟད་ཁོང་འཁྲུ་བ་དང་། ཡང་པ་ཞེས་པ་དར་བ་སར་པ་ཡང་ཡང་བ་ཡིན་མོད་ཀྱི། འདི་ནི་ཤིན་ཏུ་ཡང་བས་ན་ཁྱད་པར་སྨོས་པ་ཡིན་ནོ། །རྩ་སྦུབས་སྦྱོང་ཞིང་དག་པར་བྱེད་པ་དང་།
