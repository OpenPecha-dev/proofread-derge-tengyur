བྱས་པ་ལྟ་བུ་ཉིད་དང་། སྐུ་རིམ་པར་འཚམ་པ་ཉིད་དང་། སྐུ་གཙང་བ་ཉིད་དང་། སྐུ་འཇམ་པ་ཉིད་དང་། སྐུ་རྣམ་པར་དག་པ་ཉིད་དང་། མཚན་རྫོགས་པ་ཉིད་དང་། སྐུ་ཁོ་ལག་ཡངས་ཤིང་བཟང་པ་ཉིད་དང་། སྤྱན་མཉམ་པ་ཉིད་དང་། སྤྱན་རྣམ་པར་དག་
པ་ཉིད་དང་། སྐུ་ཤིན་ཏུ་གཞོན་མདོག་ཅན་ཉིད་དང་། སྐུ་ཞུམ་པ་མེད་པ་ཉིད་དང་། སྐུ་རྒྱས་པ་ཉིད་དང་། སྐུ་ཤིན་ཏུ་གྲིམས་པ་ཉིད་དང་། ཡན་ལག་དང་ཉིང་ལག་ཤིན་ཏུ་རྣམ་པར་འབྱེས་པ་ཉིད་དང་། གཟིགས་པ་རབ་རིབ་མེད་ཅིང་རྣམ་པར་དག་པ་ཉིད་དང་། དཀུ་ཟླུམ་པ་ཉིད་
དང་། དཀུ་སྐབས་ཕྱིན་པ་དང་། དཀུ་མ་རྙོངས་པ་ཉིད་དང་། ཕྱལ་ཕྱང་ངེ་བ་ཉིད་དང་། ལྟེ་བ་ཟབ་པ་ཉིད་དང་། ལྟེ་བ་གཡས་ཕྱོགས་སུ་འཁྱིལ་པ་ཉིད་དང་། ཀུན་ནས་མཛེས་པ་ཉིད་དང་། ཀུན་ནས་སྤྱོད་པ་གཙང་བ་ཉིད་དང་། སྐུ་ལ་སྨེ་བ་དང་གནག་བག་མེད་པ་ཉིད་དང་། ཕྱག་
ཤིང་བལ་ལྟར་ཤིན་ཏུ་འཇམ་པ་ཉིད་དང་། ཕྱག་གི་རི་མོ་མདངས་ཡོད་པ་ཉིད་དང་། ཕྱག་གི་རི་མོ་ཟབ་པ་ཉིད་དང་། ཕྱག་གི་སོར་མོ་རིང་བ་ཉིད་དང་། ཞལ་ཧ་ཅང་ཡང་མི་རིང་བ་ཉིད་དང་། ཞལ་གཟུགས་བརྙན་སྣང་བ་ཉིད་དང་། ལྗགས་མཉེན་པ་ཉིད་དང་། ལྗགས་སྲབ་པ་ཉིད་དང་།
ལྗགས་དམར་བ་ཉིད་དང་། གླང་པོ་ཆེའི་ང་རོ་དང་། འབྲུག་སྒྲ་དང་ལྡན་པ་ཉིད་དང་། གསུང་སྙན་པ་ཉིད་དང་། འཇམ་པ་ཉིད་དང་། མཆེ་བ་ཟླུམ་པ་ཉིད་དང་། མཆེ་བ་རྣོ་བ་ཉིད་དང་། མཆེ་བ་དཀར་བ་ཉིད་དང་། མཆེ་བ་མཉམ་པ་ཉིད་དང་། མཆེ་བ་བྱིན་གྱིས་ཕྲ་བ་ཉིད་
དང་། ཤངས་མཐོ་བ་ཉིད་དང་། ཤངས་གཙང་བ་ཉིད་དང་། སྤྱན་རྣམ་པར་དག་པ་ཉིད་དང་། སྤྱན་ཡངས་པ་ཉིད་དང་། རྫི་མ་སྟུག་པ་ཉིད་དང་། སྤྱན་དཀར་ནག་འབྱེས་ཤིང་པདྨའི་འདབ་མ་རངས་པ་ལྟ་བུ་ཉིད་དང་། སྨིན་ཚུགས་གཉིས་རིང་བ་ཉིད་དང་། སྨིན་
མ་ཟུང་འཇམ་པ་ཉིད་དང་། སྨིན་མ་ཟུང་གི་སྤུ་མཉམ་པ་ཉིད་དང་། སྨིན་མ་ཟུང་སྣུམ་པ་ཉིད་དང་། སྙན་ཤལ་སྟུག་ཅིང་རིང་བ་ཉིད་དང་། སྙན་མཉམ་པ་ཉིད་དང་། སྙན་གྱི་དབང་པོ་མ་ཉམས་པ་དང་། འཕྲལ་བ་ལེགས་པར་འབྱེས་པ་ཉིད་དང་། འཕྲལ་བ་དབྱེས་ཆེ་བ་ཉིད་དང་།
