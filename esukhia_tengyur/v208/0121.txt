ཨུ་ད་ཀ་ལ་ཡང་། རྣམ་དབྱེ་དང་པོའི་སུ་ནས་གཉིས་པའི་ཨཽ་ཊའི་བར་ཛྙཱ་དང་འདྲ་སྟེ། ཨུ་ད་ཀཾ། ཨུ་ད་ཀེ། ཨུ་ད་ཀཱ་ནི། ཧེ་ཨུ་ད་ཀ །ཧེ་ཨུ་ད་ཀེ། ཧེ་ཨུ་ད་ཀཱ་ནི། གཉིས་པ་ལ། ཨུ་ད་ཀཾ། ཨུ་ད་ཀེ། ཤས་ལ། ཨུ་དུ་ཀཱ་ནི། ཨུ་དཱ་ནི། ཊཱ་ལ། ཨུ་ད་ཀེ་མ། ལུགས་གཅིག་ལ།
ཨུ་ད་ན་དུ་འགྱུར། ཆུའམ་འོ་ན། ནའི་ཨ་ཏ་ང་བྱི་འོས། ཨུ་པདྷཱའི་ཨ་ཕྱིས། ཨུ་དྣཱ། བྷྱཱཾ་ལ། ཨུ་ད་ཀཱ་བྷྱཱཾ། ལུགས་གཅིག་ལ། མིང་མཐའི་ནད་བྱི་འོས་ན་ཕྱིས། ཨུ་ད་བྷྱཱཾ་བྷི་ས་ལ། ཨུ་ད་ཀཻཿ། ཨུ་ད་བྷིཿ། ངེ་ལ། ཨུ་ད་ནེ། ཨུ་ད་ཀཱ་ཡ། བྷྱཱཾ་བྷྱས་ལ། ཨུ་ད་བྷྱཱཾ། ཨུ་ད་ཀཱ་བྷྱཱ་མ། ཨུ་ད་བྷྱཿ། ཨུ་ད་
ཀེ་བྷྱཿ། ང་སི་ལ། ཨུ་དྣཿ། ཨུ་ད་ཀཱ་ཏ། བྷྱཱཾ་བྷྱཿའདྲ། ང་ས་ལ། ཨུ་དྣཿ། ཨུ་ད་ཀ་སྱ། ཨོ་ས་ལ། ཨུ་ངྣོཿ། ཨུ་ད་ཀ་ཡོཿ། ཨཱ་མ་ལ། ཨུ་དྣཱཾ། ཨུ་ད་ཀཱ་ནཱཾ། ངི་ལ། ཨུ་དྣི། ཨུ་ད་ཀེ། ཨོ་ས་འདྲ། སུ་བ་ལ། ཨུ་ད་སུ། ཨུ་ད་ཀེ་ཥུ། ཨཱ་ས་ནས་ཡང་དང་པོ་སོགས་འདྲ་སྟེ། ཨཱ་ས་ནཾ།
ཨཱ་ས་ནེ། ཨཱ་ས་ནཱ་ནི། ཧེ་ཨཱ་ས་ན། ཧེ་ཨཱ་ས་ནེ། ཧེ་ཨཱ་ས་ནཱ་ནི། ཧེ་ཨཱ་ས་ནཱ་ནི། གཉིས་པ་ལ། ཨཱ་ས་ནི། ཨཱ་ས་ནེ། ཤ་ས་ལ། ཨཱ་སཱ་ནི། ཨཱ་ས་ནཱ་ནི། ཌཱ་ལ། ཨཱ་སྣཱ། ཨཱ་ས་ནེ་ན། བྷྱཱཾ་ལ། ཨཱ་ས་བྷྱཱཾ། ཨཱ་ས་ནཱ་བྷྱཱཾ། ལྷག་མ་རྣམས་ཀྱང་ལུགས་གཉིས་ གདག ཀ་ཨུ་ད་ཀ་དང་འདྲའོ། མགོ །ཤཱིརྵ་ཞེས་

པའི་ཊཱ་ནས་བརྩམས་ཏེ་བདུན་པའི་སུབ་ཀྱི་བར་གང་མཐོང་ཡང་ཤཱིརྵ་ན་ཞེས་པར་ཡང་ན་འགྱུར་བས། ཊཱ་ལ། ཤཱིརྵེ་ཎ། ཤཱིརྵ་ན་དུ་བསྒྱུར་ན། ནའི་ཨ་ཏ་དབྱི་འོས། ཨ་ཕྱིས། ཥ་ར་སོགས་ཀྱིས་ན་ཎ། ཤཱིརྵཱ། བྷྱཱཾ་ལ། ཤཱི་རྵཱ་བྷྱཱཾ། ལུགས་གཅིག་ལ། མིང་མཐའི་ན་དབྱི་འོས། ན་ཕྱིས། ཤཱིརྵ་
བྷྱཱཾ། བྷཾས་ལ། ཤཱིརྵཻཿ། ཤཱིརྵ་བྷིཿ། དེ་ལ། ཤཱིརྵཱ་ཡ། བྷྱཱ་མ་འདྲ། བྷྱས་ལ། ཤཱིརྵེ་བྷྱཿ། ཤཱིརྵ་བྷྱཿ། ང་སི་ལ། ཤཱི་རྵཱ་ཏ། ཤཱིརྵཱ་ཏ། ཤཱིརྵཱཿ། བྷྱཱ་མ་བྷྱས་འདྲ། ངས་ལ། ཤཱིརྵ་སྱ། ཤཱིརྵཱཿ། ཨོ་ས་ལ། ཤཱིརྵ་ཡོཿ། ཤཱི་རྵཱོཿ། ཨཱ་མ་ལ། ཤཱིརྵཱ་ཎཱཾ། ཤཱི་རྵཱཾ། ངི་ལ། ཤཱིརྵེ། ཤཱི་རྵཱི།
