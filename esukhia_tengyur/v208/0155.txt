ཨ་དས་ཀྱི་ཊི། ཨ་དྲི་རུ་འགྱུར། ལ་ལས་སྔ་མ་ཁོ་ནའི་དམར་འགྱུར་བར་འདོད་པས། ཨ་མུ་དྲྱ་ཙ་ཞེས་པར་འགྱུར། རྣམ་དབྱེ་དང་ཕྲད་པའི་ཆོ་ག་སྔ་མ་རྣམས་དང་འདྲ། ཡང་ན་ད་སྔ་མ་ཡང་མར་མ་བསྒྱུར་པ། ཨ་ད་དྲྱ་ཙ། ཞེས་པ་ལ་རྣམ་དབྱེ་རྣམས་སྟེར་ཏེ། དེ་དག་ཨ་དས་ཨ་ཙ་དང་
བཅས་པ་ལ་རྣམ་དབྱེ་འཇུག་ཚུལ་གཟུགས་བཞི་པོ་འདམ་ཀ་ཅན་ནོ། །ཡང་ཨུ་ཏ་སྔ་མའི་ཨ་ཙ། ཨུ་ད་ཙ་ལ། སུ་ཨཽ་ཛས། ཨུ་དང་། ཨུ་དཉྩཽ། ཨུ་དཉྩཿ། ཨ་མ་དང་ཨཽ་ལ། ཨུ་དཉྩཾ། ཨུ་དཉྩཽ། ཨུ་ད་ཙ། ཤས་ལ། ཨུ་དཿ། ཨཱི་ཏ། ཨུ་ཏ་ལས་ཨཱི་ཏའོ། །ཨུ་ཏ་སྔ་མའི་ཨ་ཙའི་ཨ་རྣམ་དབྱེ་ཤས་
ལ་སོགས་པ་ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ཕ་རོལ་ལ་ཨཱི་རུ་འགྱུར་རོ། །ཨུ་དཱི་ཙཿ། ཊཱ་ལ། ཨུ་དཱི་ཙཱ། བྷྱཱ་མ་བྷིས་ལ། ཨུ་ད་གྦྷྱཱཾ། ཨ་ད་གྦྷིཿ། ངེ་ལ། ཨུ་དཱི་ཙེ། བྷྱཱ་མ་བྷྱས་འདྲ། ང་སི་ལ། ཨུ་དཱི་ཙཿ། བྷྱཱ་མ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། ཨུ་དཱི་ཙོཿ། ཨུ་དཱི་ཙཱཾ། བདུན་པ་རྣམས་ལ།
ཨུ་དཱི་ཙི། ཨུ་དཱི་ཙོཿ། ཨུ་ད་ཀྵུ། བོད་ཚིག་རྣམས་རྣམ་དབྱེ་དང་པོ་དང་འདྲའོ། །ས་མཿས་མིཿ། སམ་གྱི་ས་མི་འོ། །ཨ་ཙ་ཞེས་པའི་རྟགས་ཀྱི་སྔ་མའི་ཉེར་བསྐྱུར་སམ་ནི་ས་མི་ཞེས་བར་འགྱུར་ཏེ། ས་མྱ་ཙ་ཞེས་པའང་ཆོ་ག་རྣམས་བྲ་ཙའི་སྐབས་དང་འདྲའོ། །ཏི་རས་ཨ་ཙ། སུ། ཏི་
རཱསྟི་རྻ་ལོ་པི། ཏི་རས་ཀྱི་ཏི་རི་མི་དབྱི་བ་ལའོ། །ཏི་རས་ཞེས་པ་ནི་ཨ་ཙ་ཨ་དབྱི་བ་མ་ཡིན་པ་ཕ་རོལ་ལ་ཏི་རི་ཞེས་པར་འགྱུར་ཏེ། ཏི་རྻ་ཙ། སུ། ཨུ་ཀ་འགྲོ་བ་སོགས་ཀྱིས་ནུམ། ཧལ་དཱི་པ་ཨཱ་བ་སོགས་ཀྱིས་ཡང་དག་པ་སུ་ཕྱིས་ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས་ཅ་ཕྱིས། ཀྭིན་གྱི་ཀུ་འོས་ནང་
ཏི་རྻང་། ཨཽ་དང་ཛས་ལ། ཏིརྻ་ཉྩཽ། ཏི་རྻ་ཉྩཿ། ཨ་མ་དང་ཨཽལ། ཏི་རྻཉྩཾ། ཏི་རྻ་ཉྩཽ། ཤས་ལ། ཨ་ཙའི་འོས། ཨ་ཕྱིས། ཏི་རས་ཀྱི་ཏི་ཇི་མི་དབྱི་བ་ལའོ་ཞེས་པས་ཨ་མ་ཕྱིས་ན་ཏི་རི་ཞེས་པར་འགྱུར་ཡང་འདིར་ཤས་ནས་བཟུང་དབྱངས་ཐོག་མ་ཅན་རྣམས་ལ་ཨ་དབྱི་བས་ཏི་རས་ཉིད་རྐྱེན། དུ་གནས།
དུད་འགྲོ་རྒྱགས་པོ། ས་ཏུའི་ཤ་ཙུ་ལ་ཤ་ཙུ་འོས། ས་ཤ། ཏིར་ཤྩཿ། ཊཱ་ལ། ཏིར་ཤྩཱ། བྷྱཱཾ་བྷིས་ལ། ཏིརྻ་གྦྷྱཾ། ཏིརྻ་གྦྷིཿ། ངེ་ལ། ཏི་ར་ཤྩེ། བྷྱཱམ་བྷྱས་འདྲ། ང་སི་ལ། ཏི་ར་ཤྩཿ། བྷྱཱམ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། ཏི་ར་ཤྩཿ། ཏིར་ཤྩོཿ། ཏི་ར་ཤྩོཿ། ཏིར་ཥྩཱཾ། བདུན་པ་རྣམས་ལ། ཏི་ར་ཤྩི། ཏིར་ཤྩོཿ།
