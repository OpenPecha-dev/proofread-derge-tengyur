རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོས། ཏ་ད། ད་ད་དྦྷཱཾ། ཧ་ལ་དང་པོ་ཅན་རྣམས་འདི་དང་འདྲ། བོད་ཚིག་དང་པོ་དང་འདྲའོ། །ཏ་ཡིག་མཐའ་ཅན་ཛཀྵ་ཏ་ཐོགས་ལ། ཛཀྵ་ཏཱ་ད་ཡཿཥཊ། ཛཀྵ་སོགས་དྲུག་ལའང་ངོ་། །ཛཀྵ་སོགས་དྲུག་དང་ཙཀཱས་རྣམས་ལའང་ཤ་ཏུའི་རྐྱེན་མཐའ་ཅན་ལ་ཡ་བྷ་སྟའི་མིང་ཅན་ཟེར་ཏེ། ནུམ་
འགོག་པའི་དོན་དུའོ། །ཛཀྵ་སོགས་ཤ་ཏུའི་མཐའ་ཅན་ནི། ཛཀྵཏྲ། དཱི་བྷྱ་ཏ། ད་རི་དྲ་ཏ། ཤཱ་ས་ཏ། བྱ་བྱ་ཏ། ཙ་ཀྭས་ད། རྣམས་ཏེ་འདི་རྣམས་ལ་རྣམ་དབྱེ་སྦྱོར་བའི་ཆོ་ག་ཐམས་ཅད་ད་ད་ཏ་དང་འདྲའོ། ༈ །བ་ཡིག་མཐའ་ཅན་ཧ་མཾ་གུ་བ་ལ་སུ། ཧལ་སོགས་ཀྱིས་སུ་ཕྱིས། གནས་མེད་པ་ལ་ཡང་ན་འོས།
དྷརྨ་གུག དྷརྨ་གུག ཨཽ་དང་ཛས་ལ། དྷ་རྨ་ཀུ་བཽ། དྷ་རྨ་གུ་བཿ། གཉིས་པ་རྣམས་ལ། དྷརྨ་གུ་རཾ། དྷ་རྨ་གུ་པཽ། དྷརྨ་གུ་པཿ། །ཊཱ་ལ། དྷ་རྨ་གུ་པཱ། བྷྱཱཾ་བྷིས་ལ། ཛྷ་ཤ་ལ་ཛྷ་ལ་རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོས། པས། དྷ་རྨ་གུ་བྷྱཱཾ། དྷ་རྨ་གུ་བླིཿ། ཆོས་གསང་པ། བཞི་བ་རྣམས་ལ། དྷ་རྨ་གུ་བེ། དྷ་རྨ་གུ་སྦྷྱཱཾ། དྷརྨ་གུ་བྦྷྱཱཿ། ང་སི་
ལ། དྷ་རྨ་གུ་པཿ། བྷྱཱམ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། དྷརྨ་གུ་པཿ། དྷརྨ་གུ་པོཿ། དྷ་རྨ་གུ་པཱཾ། བདུན་པ་རྣམས་ལ། དྷ་རྨ་གུ་པི། དྷ་རྨ་གུ་པོ། དྷ་རྨ་གུ་པྶུ། བོད་ཚིག་རྣམས་ལ། ཧེ་དྷརྨ་གུ་པ། ཧེ་དྷརྨ་གུ་བ། ཧེ་དྷརྨ་གུ་པཽ། ཧེ་དྷརྨ་གུ་པཿ། ༈ །ཤ་ཡིག་མཐའ་ཅན། ཏྱཱ་དྲྀ་ཤ་ལ། སུ། ཧལ་སོགས་ཀྱིས་སུ་
ཕྱིས། ཀྭན་གྱི་ཀུ་འོས། ཤེ་ཀ། གནས་མེད་པ་ལ་ཡང་འོས། ཏྱཱ་དྲྀ་ཀ། ཏྱཱ་དྲི་ག། ཨོ་དང་ཛས་ལ། ཏྱཱ་དྲྀ་ཤཿ། གཉིས་པ་རྣམས་ལ། ཏྱཱ་དྲྀ་ཤཾ། ཏྱཱ་དྲྀ་ཤཾཿ། ཏྱཱ་དྲྀ་ཤཿ། གསུམ་པ་རྣམས་ལ། ཏྱཱ་དྲྀ་ཤཱ། ཏྱཱ་དྲྀ་ལྦྷཱཾ། དེ་ལ། ཏྲཱ་དྲྀ་ཤ། བྷམ་བྷས་འདྲ། ང་སི་ལ། དེ་མཐོང་བ། བྷཱམ་བྷྱས་འདྲ། དྲུག་
བ་རྣམས་ལ། དྱཱདྲྀ་ཤཿ། ཏྱཱ་དྲྀ་ཤོཿ། ཏྱཱ་དྲྀ་ཤཱཾ། བདུན་པ་རྣམས་ལ། ཏྱཱ་དྲྀ་ཤི། ཏྱཱ་དྲྀ་ཤོཿ། ཏྱཱ་དྲྀ་ཀྵཱ། བོད་ཚིག་རྣམ་དབྱེ་དང་པོ་དང་འདྲ། ཏྱ་དཱ་དིཥུ་དྲྀ་ཤོ་ཉནྜ་ལ་ཙ་ནེ་ཀཉྩ། ཏྱ་ད་སོགས་རྣམས་ཀ་དྲྀ་ཤའི་མི་མཐོང་བ་ལ་ཀ་ཉའང་ངོ་། །ཏྱད་སོགས་སྔ་མའི་དྲྀ་ཤ་ལ་མཐོང་བའི་དོན་མ་ཡིན་པ་ལ་ཀྲ་ཏ་སོགས་ཀྱི་
མིང་དུ་བསྒྲུབ་པའི་རྐྱེན་ཀ་ཉ་ཡང་ན་འཇུག་བས། ཀ་ཉ་བྱིན་ན། ཏྱཱ་དྲྀ་ཤ། ཞེས་པ་གྲུབ་སྟེ། དེའི་ཚེ་རྣམ་དབྱེའི་ཆོ་ག་རྣམས་སྤྱིར་བཏང་ཨ་ཐུང་མཐའ་ཅན་རཱ་མའི་སྐབས་དང་འདྲའོ། །འགྲེལ་པར། ས་མཱ་ན་སྔ་མའི་དྲྀ་ཤ་ལ་ཀྲིན་ཁོ་ན་འཇུག་སྟེ། དེའི་ཆེས་མཱ་ན་ཡང་ས་ཞེས་པར་འགྱུར་རོ་ཞེས་བྱུང་པས།
