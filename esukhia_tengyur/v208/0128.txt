དྲུག་པ་རྣམས་ལ། དྲུ་ཧཿ། དྲུ་ཧོཿ། དྲུ་ཧཱི། ངི་ཨོས་ལ། དྲུ་ཧི། དྲུ་ཧོཿ། སུ་པ་ལ། དྷྲུ་ཊྶུ། དྷྲུ་ཊྶུ། སཾ་བུདྡྷི་རྣམ་དབྱེ་དང་པོ་དང་འདྲ། །མུ་ཧ། སྣུ་ཧ། སྣི་ཧ་རྣམས་ཀྱང་འདི་དང་འདྲའོ། །བི་ཤྭ་ཝཱ་ཧ་ལ་སུ། བི་ཤྭ་ཝཱ་ཌ། ཨཽ་དང་ཛས། བི་ཤྭ་ཝཱ་ཧཽ། བི་ཤྭ་ཝཱ་ཧཿ།
ཧེ་བི་ཤྭ་ཝཱ་ཊ། ཧེ་བཾ་ཤྭ་ཝཱ་ཊ། ཧེ་བི་ཤྭ་ཝཱ་ཏོ། ཧེ་བི་ཤྭ་ཝཱ་ཧཿ། ཨ་མ་ལ། བི་ཤྭ་ཝཱ་ཧཾ། ཨཽ་ཊ་ལ། བི་ཤྭ་ཝཱ་ཧཽ། ཤ་ས་ལ། ཝཱ་ཧ་ཨཱུ་ཋ། ཝཱ་ཧའི་ཨཱུ་ཋའོ། ཝཱ་ཧ་ཞེས་པའི་སྒྲ་འདིའི་ སྣ་ཚོགས་ཐོབ། ཝ་ནི་རྣམ་དབྱེ་གཉིས་པའི་ཤས་ནས་བཟུང་ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ལ་ཨཱུ་ཋ་ཞེས་
པར་འགྱུར་རོ། །ཝ་ཨཱུ་ཨེ་ཏི་ཨེ་ངྷ་ཏི་ཨཱུ་ཋ་རྣམས་ལའོ། ཞེས་པས་འཕེལ་བ། བི་ཤྲཽ་ཧཿ། ཊཱ་ལ། བི་ཤྭཽ་ཧཱ། བྷྱཱཾ་བྷི་ས་ལ། བི་ཤྭ་བཱཌྦྷྱཱཾ། བི་ཤྭ་བཱ་དྦྷཱཾཿ། ངེ་ལ། བི་ཤྭཽ་ཧེ། བྷྱཱཾ་བྷྱཿའདྲ། ང་སི་ལ། བི་ཤྭོ་ཧཿ། བྷྱཱཾ་བྷྱ་ས་འདྲ། ང་ས་ལ། བི་ཤྭཽ་ཧཿ། ཨོ་ས་ལ། བུ་ཤྭཽ་ཧོཿ། ཨཱ་མ་ལ།
བུ་ཤྭཽ་ཧཱཾ། ངི་ལ། བུ་ཤྭཽ་ཧི། བི་ཤྭཽ་ཧོཿ། སུ་བ་ལ། བུ་ཤྭ་བཱ་ཊྶུ། བྷཱུ་ཝ་ཧ་ཡང་འདི་དང་འདྲའོ། ཧ་ཡིག་མཐའ་ཅན་ཨ་ན་ཌུ་ཧ་ལ། གླང་ཁྱུག་སུ། ཙ་ཏུ་ར་ན་ཌུ་ཧོ་རཱ་མུ་དཱ་དྷཿ། ཙ་ཏུ་ར་ཨ་ན་ཌུ་ཧ་དག་གི་ཨཱ་མ་ཨུ་དཱ་ཏྟའོ། །ཙ་ཏུ་ར་དང་ཨ་ན་ཌུ་ཧ་དག་ལ། རྣམ་དབྱེ་ཐམས་ཅད་ནས་པའི་མིང་ཅན་
རྣམས་ཕ་རོལ་ལ་ཨཱ་མ་ཨཱ་གམ་སྟེར་ཏེ་དེ་ལ་ཨུ་དཱ་ཏྟའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱེད་དོ། །ཨཱ་མ་ཞེས་པའི་མ་ནི་མ་འགྲོ་བ་ཨ་ཙའི་མཐའ་ལས་ཕ་རོལ་ཞེས་པའི་དོན་ནོ། །ཨ་ན་ཌྭཱ་ཧ། སུ། སཱ་ཝ་ན་ཌུ་ཧཿ། ཨ་ན་ཌུ་ཧའི་སུ་ལའོ། །ཨ་ན་ཌུ་ཧ་ཞེས་པའི་རྟགས་ལ་རྣམ་དབྱེ་དང་པོའི་
གཅིག་ཚིག་སུ་དང་སཾ་བུ་ངྔྷིའི་མིང་ཅན་གྱི་སུ་ཕ་རོལ་ལ་ནུམ་ཨཱ་ག་མ་འགྱུར་ཏེ། ནུ་མ་བྱིན། ཧ་ལ་དཱི་པ་ཨཱ་པ་ཞེས་པས་སུ་ཕྱིས། ཡང་དག་པར་སྦྱོར་བའི་མཐའ་དབྱི་འོས། ཧི་ཕྱིས། ཨ་ན་ཌྭཱ་ན། ཨཽ་དང་ཛ་ས་ལ། ཨ་ན་ཌྭཱ་ཧཽ། ཨ་ན་ཌྭཱ་ཧཿ། ཨ་ན་ཌུ་ཧ་ལ། སཾ་བུ་དྡྷི། ཨ་མ་སཾ་
བུདྡྷཽ། སཾ་བུ་དྡྷི་ལ་ཨ་མའོ། ཨ་ན་ཌུ་ཧ་དང་ཙ་ཏུར་གཉིས་ལ་སཾ་བུདྡྷི་ཕ་རོལ་ལ་ཨ་མ་ཨཱ་ག་མ་སྟེར་རོ། །ཨ་ན་ཌུ་ཧའི་སི་ལ་འོས་ནུ་མ། ཧེ་ཨ་ན་ཌྭ་ན། ཧེ་ཨན་ཌྭཱ་ཧོཾ། ཧེ་ཨ་ན་ཌྭཱ་ཧཿ། ཨ་མ་ལ། ཨ་ན་ཌྭཱ་ཧཾ། ཨཽ་ལ། ཨ་ན་ཌྭཱ་ཧཽ། ཤ་ས་ལ། ཨ་ན་ཌུ་ཧཿ། ཊཱ་ལ། ཨ་ན་
