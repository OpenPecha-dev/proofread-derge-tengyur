ཡཏ་ པྲ་ཏཱི་ཙྱཿ། ཨ་བྱ་ཡཱ་ཏྷྱ་བ། མི་ཟད་པ་ལས་ཏྱ་པའོ། མི་ཟད་པའི་ནང་ཚན་ཨ་མ་ཨི་ཧ། ཀྭ་རྣམས་དང་ཏ་སིལ་དང་ཏྲལ་གྱི་རྐྱེན་མཐའ་རྣམས་ལས་ལྷག་མ་ལ་ཏྱ་པ་འགྱུར་ཏེ། ཨ་མཱ་ཏཿ། ཏྱབ་པ་དབྱངས་དོ་ན། ཏྱབ་ ཨི་ཧ་ཏྱཿ། ཏྱབ་ ཀྭ་ཏྱཿ།[ཐི་ལས་]སྐྱེས་སམ་ཅི་ནས་འོངས་པའམ་ཅི། ཏྱཐ་ཏ་
ནུབ་ཏུ་སྐྱེས་པའམ་ནུབ་ནས་འོངས་པའམ་ནུབ་པ། ཏ་སྷྲཿ། ཏྱབ་ ཏ་ཏྲ་ཏྱཿ། ཏྱཔ་ནེཿདྷྲུ་བེ། ནི་ལས་བརྟན་པ་ལ་ཏྱཔ་འོ། །ནི་ལས་བརྟན་པའི་དོན་ལ་ཏྱབ་འགྱུར་ཏེ། ནི་ཡ་[ཏཾ་དྒྲུ་བཾ]། ཞེས་པའི་ བློན་པོའམ་བློན་པོ་ལས་སྐྱེས་པ། དོན་ལ། ཏྱཔ་ནི་ཏྱཾ། [འདི་མ་]འདི་ལས་སྐྱེས་པའམ་འདི་ནས་འོངས། ཨཻ་ཥ་མོ་ཧྱཿ། ཤྭ་སོ་ཉ་ནྱ་ཏ་ར་
དེ་ལས་སྐྱེས་སམ་དེ་ནས་འོངས་པའམ་དེ་ལ། དེ་ལས་སྐྱེས་སམ་དེ་ནས་འོངས་སམ་དེ་ལ། སྱཱི། ཨཻ་ཥ་མས་ཧྱས་ ཤྭ་ས་ལས་ཡང་ནའོ། །ཨཻ་ཥ་མས་སོགས་རྣམས་ལས་རྣམས་ལས་ལྷག་མ་ལ་ངེས་པའམ་བརྟན་པ། ཏྱཔ་ཡང་ན་འགྱུར་ཏེ། ངེས་པའམ་བརྟན་པའམ་མི་གཡོ་ག ཏྱབ་ ཨཻ་ཥ་མ་སྟྱཾ། འདི་ལས་སྐྱེས་སམ་འདི་ནས་འོངས་
སམ་འདི། ཏྱབ་ ཧྱ་སྟྱཾ། ཏྱབ་ ཤྭ་སྦྱཾ། རླུང་ལས་སྐྱེས་སམ་རླུང་ནས་འོངས་སམ་རླུང་། ཡང་ན། ཨཻ་ཥ་མ་སྟ་ནཾ། སོགས་ཏེ་འོག་ཏུ་འཆད་དོ། །བྲྀ་དྷིཿཡ་སྱཱ་མཱ་དི་སྟ་དྦྲྀ་དྟྷི། གང་གི་ཨཙ་རྣམས་ཀྱི་དང་པོ་འཕལ་བ་དེ་འཕེལ་བའོ། །རྟགས་གང་གིའང་ཨཙ་རྣམས་ཀྱི་དང་པོ་འཕེལ་བའི་ཡི་
གེ་ཡོད་པ་དེ་ལ་ ཁ་སང་སྐྱེས་སམ་ཁ་སང་འོངས་སམ་ཁ་སང་། འཕེལ་པའི་མིང་ཅན་ཞེས་མིང་འདོགས་བྱེད་དེ་དགོས་པ་ནི་འཕེལ་བ་ལས་ཚའི་ཞེས་སོགས་ཀྱི་ཆེད་དེ། བྲྀདྡྷཙྩཿ། འཕེལ་པ་ལས་ཚའོ། །གོང་དུ་འཕེལ་བའི་མིང་ཅན་ཞེས་པ་རྡ་ཆང་བྱས་པ་དེ་ལས་ལྷག་མ་ལ་ཚ་
འགྱུར་ཏེ། གཱ་རྒྱེ་ཛ་ཏཿ། གཱ་རྒྱེ་ཛ་ཏཿ། ཞེས་པའི་དོན་ལ། གཱ་རྒཱི་ཡཿ། གཱ་རྒཱའི་བུ་འམ་ག་རྒའི་བུ་ལས་སྐྱེས་པ། ཨཱ་ག་སྟེ་ཡཿ། ཚ་ཨཱི་ཡ། ཤཱ་ལཱི་ཡཿ། ཏྱ་དཱ་དཱི་ནི་ཙ། ཏྱད་ལས་སོགས་པ་རྣམས་ཀྱང་ངོ་། ཏྱད་ལས་སོགས་པ་རྣམས་ལས་ཀྱང་ལྷག་མ་ལ་ཆ་འགྱུར་ཏེ། ཏྱ་རཱི་ཡཾ། ཚ་ཏ་
[ཤ་རྒའི་]བུ་ལས་སྐྱེས་པ། ག་རྒའི་བུ་ཉིད། གརྒྱ་ལ་ཚ་ཚ་ཨཱི་ཡ་བུ་བརྒྱུད་ཀྱི་ཐ་ན་ལ་ཨཱ་མ་ཡིན་བཞེས་ཡ་[ཕྱིས]། རི་བྱའི་བུའམ་རི་བྱའི་བུ་ལས་སྐྱེས་ག་རབ་དུ་སྐྱེས་པའམ་རབ་ནས་འོངས་པའམ་རག དཱི་ཡཾ། དེ་ལས་སྐྱེས་པའམ་དེ་ནས་འོངས་པའམ་དེ། ཚ་ཡ་དཱི་[ཡཾ་གང་ལས་སྐྱེས་པ་མ་གང་ནས་འོངས་པའམ་གང་།] དེ་ལས་སྐྱེས་པའམ་དེ་ནས་འོངས་པའམ་དེ།
