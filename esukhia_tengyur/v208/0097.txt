ང་སོ༔ཙ། ང་སི་ངས་དག་གི་ཡང་ངོ་། །རྟགས་མཐའ་ཨེང་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོད་ན་ཕ་རོལ་རྣམ་དབྱེ་ལྔ་བའི་ང་སི་དང་། དྲུག་པའི་ངས་གཉིས་ཀྱི་ད་འགྲོ་བའི་མིང་ཅན་ཕྱིས་པའི་ཨ་དེ་ནི་སྔ་མ་ཨེང་གི་ཡི་གེ་གང་མཐོང་བ་དེའི་གཟུགས་སུ་འགྱུར་ཏེ། ཧ་རེ༔། ཧ་རི་བྷྱཱཾ།
ཧ་རི་བྷེ༔། ཧ་རི་ལ། དྲུག་པའི། ངས། ལྔ་བ་དང་འདྲ་བས། ཧ་རེ༔། ཧ་རི། ཨོས། ཨི་ཀ་གི་ཞེས་པས། ཧ་རྻྻོ༔། ཧ་རི། ཨཱ་མ། ཐུང་ངུ་སོགས་ཀྱིས་ནུཊ། ར་ཥ་དག་ལ་ཞེས་པས། ན་ཎ། ནཱ་མ་ལ་འོས་རྟགས་མཐའ་རིང་པོར་འགྱུར། ཧ་རཱི་ཎཱཾ། ཧ་རི་ལ། དི། ཨ་ཙྫུ་གྷེ༔། གྷིའི་ཨ་ཏ་ཡང་
ངོ་། །རྟགས་མཐའ་གྷིའི་མིང་ཅན་གྱི་ཕ་རོལ་དུ་རྣམ་དབྱེ་བདུན་པའི་ངི་བྱུང་ན། རྟགས་མཐའ་གྷིའི་མིང་ཅན་དེ་ཨཏ་དུ་འགྱུར་ལ། ཡང་གི་སྒྲས་ངི་ཡང་ཨཽ་རུ་འགྱུར་ཏེ། ཧ་ར། ཨཽ། ཨེ་ཙ་ལ་འོས། ཧ་རཽ། ཧ་རྻྻྻཽ༔། ཧ་རི་ཥུ། དེ་བཞིན་དུ། ཤྲཱི་པ་ཏི། ཨ་གྣི། ར་བི། ཀ་ཝི་ལ་སོགས་པ་མང་
པོ་དག་ཀྱང་ཧ་རི་དང་བྱ་བ་འདྲའོ། །ས་ཁིའི་སྒྲ་ལ་ཅུང་མི་འདྲ་སྟེ། ས་ཁི་ལ་རྣམ་དབྱེ་དང་པོའི་གཅིག་ཚིག་སུ་བྱིན། ས་ཁི། སུ། ཨ་ནང་སཽ། སུ་ལ་ཨ་ན་ང་ངོ་། །ས་ཁིའི་སྒྲ་ལ་ཀུན་དུ་བོད་པའི་དོན་དུ་སྦྱིན་པའི་སཾ་བུ་དྡྷྲི་མ་ཡིན་པའི། དཔལ་གྱི་བདག་པོ། མེ། ཉི་མ། སྙན་ངག། སུ་ལ་རྟགས་
མཐའ་ཨ་ནང་དུ་འགྱུར་ཏེ། །ས་ཁ་ན། གྲོགས་པོ། སུ། ཨ་ལོ་ནྟྲ་ཏྤཱུ་རྦ་ཨུ་བ་དྷཱ། ཨ་ལ་མཐའ་ལས་སྔ་མ་ཨུ་པ་ངྷཱའོ། །རྟགས་མཐའ་ཨ་ལ་ཏེ་གསལ་བྱེད་དམ་སྐབས་འགར་དབྱངས་ཀྱང་རུང་སྟེ་མཐའི་སྔ་མའི་ཡི་གེ་དེ་ནི་ཨུ་པ་ངྷཱའི་མིང་ཅན་ནོ། །སརྦ་ནཱ་མ་སྠཱ་ནེ་ཙཱ་སཾ་བུ་དྡྷཽ། སཾ་བུ་དྡྷི་མ་ཡིན་པའི་མཐའ་
ཅད་གནས་པའི་མིང་ཅན་ལ་ཡང་ངོ་། །རྟགས་མཐའ་ན་ཡི་གེ་སྲོག་མེད་ཀྱི་ཨུ་པ་དྷཱའི་ཨ་ཅ་དི་ནི་བོད་པའི་དོན་དུ་སྦྱིན་པའི་སུ། སཾ་བུ་དྡྷིའི་མིང་ཅན་མ་ཡིན་པའི་ཐམས་ཅད་གནས་པའི་མིང་ཅན་གྱི་རྣམ་དབྱེ་རྣམས་ལ་རིང་པོར་འགྱུར་རོ། །ས་ཁཱན་སུ། ཨཾ་པྲཾ་ཀྟ་ཨེ་ཀཱ་ལ་པྲ་ཏྱ་ཡ༔། རྐྱེན་ཨལ་
གཅིག་པ་ཨ་པྲྀ་ཀྟའོ། །རྐྱེན་ཨལ་གཅིག་པ་སྟེ་ཡི་གེ་སྲོག་ལྡན་དང་གཉིས་གསུམ་ལྟ་བུ་མ་ཡིན་པའི་གསལ་བྱེད་སྲོག་མེད་གཅིག་པོ་དེ་ལ། །པྲྀ་ཀྟའི་མིང་ཅན་ཞེས་དགོས་པའི་དབང་གིས་མིང་འདོགས་བྱེད་དོ། །ཧ་ལ་དྱཱ་བྷྱོ་དཱི་རྒྷཱ་ཏ་སུ་ཏི་སྱ་པྲྀ་ཀྟཾ་ཧ་ལ། ཧ་ལ་དཱི་པ། ཨཱ་པ། རིང་པོ་
