བྷྱཱཾ་བྷྱ་ས་འདྲ། དྲུག་པ་རྣམས་ལ། ཡ་ཛྭ་ནཿ། ཡ་ཛྭ་ནོཿ། ཡ་ཛྭ་ནཱཾ། བདུན་པ་རྣམས་ལ། ཡ་ཛཱ་ནི། ཡ་ཛྭ་ནོཿ། ཡ་ཛྭ་སུ། བོད་ཚིག་ལ། ཧེ་ཡ་ཛྭ་ན། ཧེ་ཡ་ཛྭཱ་ནཽ། ཧེ་ཡ་ཛྭཱ་ནཿ། ༈ བྲ་ཧྨ་ན་ཡང་འདི་དང་འདྲའོ། །བྲྀ་ཏ་ཧན་ལ་སུ། སཽ་ཙ། སུ་ལའང་ངོ་། །རྟགས་ཨིན་མཐའ་
དང་ཧན་དང་། པུ་ཥ་ན། ཨ་རྻ་མན་རྣམས་ཀྱི་སཾ་བུདྡྷི་མ་ཡིན་པའི་ཡུ་ལ་ཨུ་པ་དྷཱ་རྟག་ཏུ་རིང་པོར་འགྱུར་ཏེ། བྲྀ་ཏྲ་ཧཱ། ཨཽ་ལ། རྗེས་མའི་ཚིག་ཨ་ཙ་གཅིག་པ་ལ་ཎ་འོ་ཞེས་པས། ན་ཎ། བྲྀ་ཏྲ་ཧ་ཎཽ། ཚངས་པ། འདིར་སུ་མ་གཏོགས་རྣམ་ འཕེལ་འཇོམས། དབྱེ་ཐམས་ཅན་གནས་པའི་མིང་ཅན་རྣམས་
ལ་ཨུ་པདྷཱ་རིང་པོར་མི་འགྱུར་ཏེ། ཛ་ས་ལ། བྲྀ་ཏྲ་ཧ་ཎཿ། ཨ་མ་ལ། བྲྀ་ཏྲ་ཧ་ཎཾ། ཨཽ་ལ། བྲྀ་ཏྲ་ཧ་ནཽ། ཤ་ས་ལ། ནའི་ཨ་དབྱི་འོས་ཨ་ཕྱིས། ཧོ་ཧཾ་ཏེ་རྙཱི་ནྣི་ཥ། ཧན་གྱི་ཧའི་ཉ་ཎ་འགྲོ་བ་ན་རྣམས་ལའོ། །བྱིངས་ཏན་འཚེ་བ་ལའོ་ཞེས་པའི་ཧན་གྱི་ཧ་ནི་རྐྱེན་ཉ་ཎ་འགྲོ་བའི་མིང་ཅན་རྣམས་
དང་བ་ཕ་རོལ་ལ་གྷ་རུ་འགྱུར་རོ། །བྲི་ཏྲ་ཧགྷན། ཤས། ཧཾ་ཏེར་ཐྤཱུ་རྦ་སྱ། ཧན་གྱི་ཨ་ཏ་སྔ་མའིའོ། ཧ་ན་ཞེས་པའི་སྒྲའི་སྔོན་ན་ཨ་ཏ་སྟེ་ཨ་ཡིག་ཡོད་ན་ནར་མི་འགྱུར་ཏེ། གོང་དུ་རྗེས་མའི་ཚིག་ཨ་ཙ་གཅིག་པ་ལ་ཎའོ་ཞེས་པའི་སྤྱིར་བཏང་འགོག་པའི་དོན་ནོ། །བྲྀ་ཏྲ་གྷྣཿ། གསུམ་པ་རྣམས་
ལ། བྲྀ་ཏྲ་གྷྣཱ། བྲྀ་ཏྲ་ཧ་བྷྱཱཾ། བྲྀ་ཏྲ་ཧ་བྷིཿ། དེ་ལ། བྲྀ་ཏྲ་གྷྣེ། བྷྱཱཾ་བྷྱ་ས་འདྲ། ང་སི་ལ། བྲྀ་ཏྲ་གྷྣཿ། བྷྱཱཾ་བྷྱ་ས་འདྲ། དྲུག་པ་རྣམས་ལ། བྲྀ་ཏྲ་གྷྣཿ། བྲྀ་ཏྲ་གྷྣཾ། བདུན་པ་རྣམས་ལ། བྲྀ་ཏྲ་གྷྣི། བྲྀ་ཏྲ་གྷྣོཿ། བྲྀ་ཏྲ་ཧ་སུ། བོད་ཚིག་ལ། ཧེ་བྲྀ་ཏྲ་ཧ་ན། ཧེ་བྲྀ་ཏྲ་ཧ་ཎཽ། ཧེ་པྲྀ་ཏྲ་ཧ་ཎཿ།
ཤ་ཎྒི་ན། ཡ་ཤཱ་སྭི་ན། པུ་ཥ་ན། ཨ་རྻཱ་མ་ན་རྣམས་ཀྱང་འདི་དང་འདྲ་ལ། པུ་ཥ་ན་ལ། འགྲེལ་པར་ངི་ལ་ཊི་ཡང་ན་དབྱིའོ་ཞེས་པ་བྱུང་བས། ལུགས་གཅིག་ལ། ཊི་ཕྱིས། པུ་ཥི། པུ་ཥྞི། པུ་ཥ་ཎི་སྟེ་གཟུགས་གསུམ་འབྱུང་ངོ་། ༈ །མ་གྷ་ཝན་ལ། སུ། མ་གྷ་ཝཱ་བ་ཧུ་ལཾ། མགྷ་ཝ་ན་
རི་དྭགས་ཀྲྀ་ཥྞ་སཱ་ར། རྒྱས་ལ། ཉི་མ། གྱི་ཡང་ནའོ། མ་གྷ་ཝ་ན་ཞེས་པའི་སྒྲ་ནི་རྣམ་དབྱེ་གང་མཐོང་ཡང་མ་གྷ་ཝ་ཏྲྀ་ཞེས་པར་ཡང་ན་འགྱུར་རོ། རྀ་འགྲོ་བའི་མིང་ཅན་ཡིན་པས་མི་གནས་ཀྱང་དགོས་པའི་ཆེད་དུའོ། །ཨུ་གྡི་ཙཱ་ས་རྦ་ བརྒྱ་བྱིན། ནཱ་མ་སྠཱ་ནེ་དྷཱ་ཏོཿ། ཨུ་ཀ་འགྲོ་བ་ཨ་ཙ་རྣམས་ཀྱི་
