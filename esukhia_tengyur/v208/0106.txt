སཱུ། སྭ་ཡཾ་བྷཱུ། སོགས་ཀྱང་འདི་དང་འདྲའོ། །བྷཱུ་སུ་དྷཱེ་དག་ལམ་ཡིན་ནོ། །ཡ་ཎར་འགྱུར་བ་མ་ཡིན་ཀྱང་དམིགས་བསལ་བརྵཱ་བྷཱུ་ཞེས་པའི་སྒྲ་ནི་བྷཱུའི་མཐའ་ཡིན་ཀྱང་ཡ་ཎར་འགྱུར་ཏེ། སུ་ལ། བརྵཱ་བྷུཿ། བརྵཱ་བྷཱུཿ། བརྵཱ་བྷཱུ་ལ་ཨཽ། བརྵཱ་བྷཱ་ཤྩ། བརྵཱ་བྷཱུའི་འང་ངོ་། །བརྵཱ་བྷཱུ་ཞེས་པའི་སྒྲ་འདི་སུ་སོགས་
ཀྱི་རྣམ་དབྱེ་ཞེས་པས། །ཨ་ཙ་ཐོག་མ་ཅན་རྣམས་ལ་ཡ་ནར་འགྱུར་ཏེ། །ཨཱུ་ཝར་བསྒྱུར་བས། །བརྵཱ་བྷྭཽ། བརྵཱ་བྷཱཿསོགས། ལོ་དུས་སུ་འབྱུང་བ། ཁ་ལ་པཱུའི་སྐབས་དང་འདྲའོ། །འགྲེལ་པར། དྲྀགྒྷཱུ། ཀཱ་རཱ་བྷཱུ། པུན་རྦྷཱུ་རྣམས་ཀྱང་བརྵཱ་བྷཱུ་དང་འདྲའོ་ཞེས་བྱུང་བས། འདི་རྣམས་ལ་ཡང་རྣམ་དབྱེ་
བྱིན་པའི་བྱ་བ་ཐམས་ཅད་བརྵཱ་བྷཱུ་དང་འདྲ་བས་བཤད་མཁྲག་པའི་ཚོགས་བསྒྲུབ་པའོ། །དེ་ལྟར་ཨཱུ་རིང་མཐའ་ཅན་ལས་སྐྱེས་བའི་རྟགས་ཀྱི་བྱ་བ་རྣམས་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་སྐྱེས་བའི་བྱེད་པ་འབྱུང་བ། སླར་ཡང་འབྱུང་བ། རྟགས་རི་ཡིག་མཐའ་ཅན་གྱི་ཆོ་ག་རྣམས་སྟོན་པར་བྱ་སྟེ། བྷཱ་ཏྲྀ་ལ།
སུ། རྀཏ་ཨུ་ཤ་ནས། བུ་རུ་ད་ཤས་ཨ་ནེ་ཧས་རྣམས་ཀྱི་ཡང་ངོ་། །ཞེས་པས། རྀ་ཨ་ནང་། སཾ་བུ་དྡྷི་མ་ཡིན་པའི་ཞེས་པས་ཨུ་པ་དྷཱ་རིང་པོ། ཧ་ལ་དཱི་ཨཱ་པ་ཅེས་པས་སུ་ཕྱིས། མིང་མཐའི་ན་དབྱེ་འོས། ན་ཕྱིས་པས། བྷཱ་ཏཱ། བྷཱ་ཏྲྀ་ལ་དང་པོའི་ཨཽ་དང་ཇས། གཉིས་པའི་ཨ འཛིན་བྱེད།
དང་ཨཽ་རྣམས་བྱིན། རྀ་ཏ་ཀྱི་དེ་ཐམས་ཅད་གནས་པའི་མིང་ཅན་དག་ལས་སོ། །ཞེས་བས། རྀ་ཨ་ར་དུ་བསྒྱུར། ཨ་བ་ཏྲྀ་ན་ཏྲྀ་ཚ་སོགས་ཀྱི་མདོས། ཨུ་པབྷཱུ་རེད་པོར་དྲང་། བྷཱ་ཏཱ་རཽ། བྷཱ་ཏཱ་རཿ། བྷཱ་ཏཱ་རཾ། བྷཱ་ཏཱ་རཽ། བྷཱ་ཏྲྀ་ལ་ཤས། དང་པོ་དག་ལ་སྔ་མ་འདྲ་བའོ་དང་། དེ་ཡང་ཤས་ཀྱིན་
སྐྱེས་པ་ལ་འོས། བྷཱ་ཏྲཱྀན། གསུམ་པ། བྷཱ་ཏྲཱ། བྷཱ་ཏྲི་བྷྱཱཾ། བྷཱ་ཏྲྀ་བྷིཿ། བཞི་པ་ལ། བྷཱ་ཏྲེ། བྷཱ་ཏྲྀ་བྷྱཱཾ། བྷཱ་ཏྲྀ་བྷྱཿ། བྷཱ་ཏྲྀ། ང་སེ། རྀ་ཏ་ལས་ཨུར་འོ་ཞེས་དང་། ར་ལས་སའིའོ་ཞེས་པས། བྷཱ་ཏུཿ། བྷཱ་ཏྲྀ་བྷྱཱ། བྷཱུ་ཏྲྀ་བྷྱཿ། དྲུག་པའི་གཅིག་ཚིག་ཀྱང་ང་སི་དང་འདྲ་བས། བྷཱ་ཏུཿ། ཨོ་ས་ལ།
བྷཱ་ཏྲོཿ། ཨཱ་མ་ལ། ར་ཨ་ཙ་སོགས་ཀྱིས་ནུཊ། ནཱ་མ་ལ་འོས་རྟགས་མཐའ་རིང་པོ། འགྲེལ་བར་ལུང་དྲངས་པ། རྀབརྞཱནྣ་སྱཎ་ཏྭཾ་བཱཾཙྱ། རྀ་ལས་ནའི་ཎར་བརྗོད་དོ། །ར་ཥ་གཉིས་སུམ་ཟད་དབྱངས་རྀའི་ཕ་རོལ་གྱེན་ཡང་ཎར་འགྱུར་ཏེ། བྷཱ་ཏྲཱྀ་ཎཱཾ། བདུན་པའི་རི་ལ་རྀཏ་ཀྱི་དི་ཐམས་ཅད་གནས་པ་
