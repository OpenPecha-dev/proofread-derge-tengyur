རྟགས་ཨ་ངྒའི་མིང་ཅན་དེ་ཨེ་རུ་འགྱུར་རོ། །ཨེ་ཅའི་ཨ་ཡ་སོགས་དང་། ས་ས་ཛུ་ཥ་སོགས་ཀྱི་མདོས། རཱ་མ་ལ། དྲུག་པའི་ཨཱ་མ། ཧྲ་སྭ་ན་བྱཱ་པོ་ནུཊ། ཐུང་ངུ་ན་དཱི་ཨཱ་པ་རྣམས་ལས་ནུཊ་འོ། །རྟགས་མཐའ་ཐུང་ངུ་རྣམས་དང་ན་དཱི་མིང་ཅན་ཏེ་བུད་མེད་ཀྱི་རྟགས་ཨཱི་ཨཱུ་གཉིས་
དང་ཨཱ་པ་སྟེ་མོ་རྟགས་ཨཱ་རིང་མཐའ་ཅན་རྣམས་ལས་ཕ་རོལ་རྣམ་དབྱེ་དྲུག་པའི་མང་ཚིག་ཨཱ་མ་ལ་ནུཊ་ཨཱ་ག་མ དག་བྱེད་དག་གི། སྟེར་ཏེ། ཊ་ཡིག་ཌྐི་ཏ་སོགས་ཀྱི་དོན་ཨུ་བརྗོད་དོན། རཱ་མ། ནཱ་མ། ནཱམི། ནཱ་མ་ལའོ། ནཱ་མ་ཕ་རོལ་ལ་རྟགས་མཐའི་ཨ་ཅ་གང་ཡིན་ཀྱང་རིང་བོར་འགྱུར་ཏེ།
རྟགས་མཐའི་རིང་པོ་ཨཊ་ཀུ་པུ་སོགས་ཀྱི་མདོས། ནཱ་ཎཱ། རཱ་མཱ་ཎཱཾ། རཱ་མ་ལ། བདུན་པའི་ངེ་། ང་འགྲོ་བའི་མིང་ཅན་ཡིན་པས་ཕྱིས། ཨ་ལ་ཡོན་ཏན་ནོས། རཱ་མེ། རཱ་མ་ལ་བདུན་པའི་ཨོའོ། གང་དྲུག་པའི་སྐབས་ལྟར་བསྒྲུབས་པས། ར་མ་ཡོ༔། རཱ་མ་ལ་བདུན་པའི་མང་ཚིག་སུ་པ་
པ་སོ་སོར་བསྡུ་བའི་དོན་ཡིན་པས་བྱིས། དགའ་བྱེད་རྣམས་ཀྱི། མང་ཚིག་ལ་ཛྷ་ལ་ལ་ཨེ་ཏ་འོས། རཱ་མེ་སུ། ཨཱ་དེ་ཤ་བྲ་ཏྱ་ཡ་ཡོ༔། ཀུན་ནས་བསྒྱུར་བ་དང་། དགའ་བྱེད་ལ། རྐྱེན་དག་ལའོ། །ཆུ་རོལ་ཨི་ཎ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་དང་ཀ་སྡེ། དགའ་བྱེད་དག་ལ། ཡོད་ལ། ཕ་རོལ་ཥ་སར་
བསྒྱུར་ཟིན་པ་ལྟ་བུ་ཀུན་ནས་བསྒྱུར་བ་དང་། རྐྱེན་གྱིས་ཚིག་མཐའ་མིན་པ་རྣམས་ཥ་རུ་འགྱུར་ཏེ། རཱ་མེ་ཥུ། ཚིག་མཐའ་ཡིན་ན་མི་འགྱུར་ཏེ། ཧ་རི་སྟ་ཏྲ། ཞེས་པ་ལྟ་བུ་ལ་ཚུ་རོལ་ཨི་ཎ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཨི་མཐོང་ཡང་ཕ་རོལ་ས་ཥ་ར་མི་འགྱུར་བར་ས་ཉིད་དུ་གནས་པ་ལྟ་བའོ། །དེ་
ལྟར་ཨ་ཙ་མཐའ་ཅན་སྐྱེས་བའི་རྟགས་ལས་སྤྱིར་བཏང་ཨ་ཐུང་མཐའ་ཅན་གྱི་ཆོ་ག་རྣམས་བཤད དགའ་བྱེད་རྣམས། ཟིན་ནས་ཨ་ཐུང་མཐའ་ཅན་ལས་ འཕྲོག་བྱེད་དེ་ལ། དམིགས་བསལ་ས་རྦ་ལ་སོགས་པའི་ཆོ་ག་རྣམས་བཤད་པར་བྱ་སྟེ། སརྦཱ་དཱི་ནི་སརྦ་ནཱ་མཱ་ནི། སརྦ་ལ་སོགས་པ་རྣམས་ལ་སརྦའི་
མིང་ཅན་ནོ། །སརྦ་ལ་སོགས་པ་ནི་འདི་རྣམས་ཏེ། སརྦ། ཐམས་ཅད། བི་ཤྭ། སྣ་ཚོགས། ཨུ་བྷ། གཉིས། ཨུ་བྷ་ཡ། ཌ་ཏ་ར། ཌ་ཏ་མ། ཨི་ཏ་ར། རང་ངམ་ཐ་དད། ཨ་ནྱ། གཞན། ཨ་ནྱ་ཏ་ར། གཞན་དང་གཞན། ཏྭ་ཏ། ཁྱོད། ཏྭ། རང་། ནེ་མ། ཁྲིམས། ས་མ། མཉམ་པ། སི་མ། ས་ཆ་འམ་མངའ་ཞབས།
