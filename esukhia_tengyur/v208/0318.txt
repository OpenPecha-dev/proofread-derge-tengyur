ཨཽ་པ་ག་བ་ཀཾ། [སུ་ཨ]། ཉ་བའི་བ་ལང་གི་ཚོགས། ཞེས་པ་རིགས་དང་། དེ་བཞིན་དུ། ཨཽ་ཀྵ་ཀཾ། སོག་ཅན་ཏེ་གླང་གི་ཚོགས། ཨཽ་ཥ་ཀཾ། བུ་ཉ་ ཨཽ་ར་བྷྲ་ཀཾ བུ་ཉ་ རཱ་ཛ་ཀཾ། བུཉ་ རཱ་ཛ་ཎྱ་ཀཾ། བུ་ཉ་ རཱ་ཛ་བུ་ཏྲ་ཀ། བུཉ་ བཱ་ཏཱ་ཀཾ། བུཉ་ མཱ་རུ་ཥྜྱ་ཀཾ། ཨཱ་ཛ་ཀཾ། བྲྀ་དྡྷྲ་ཙྩ། བྲི་དྡྷ་ལས་ཀྱང་ངོ་། བྲྀ་དྡྷ་ལས་ཀྱང་
[ཨོ་བ་ག་བ་ལ་བུཉ་] ཉ་འཕེལ་བའི་དོན། ཚོགས་པ་བརྗོད་པ་ལ་བུ་ཉ་འགྱུར་ཏེ། [བུ་ཉ་ཉ་འཕེལ་བ་བུ་ཨ་ག་ནའི་དེ་བ་ན་ལའང་ངོས་གང་ཤི་ཨི་ཏ་ལ་རང་ངོས་ཨ་ཕྱིས]། [རྔ་མོ་བཅོས་ས]། [ལྟོ་བྲལ་ཏེ་བཤན་པའི་]ཚོགས། རྒྱལ་པོའི་ཚོགས། རྒྱལ་རིགས་ཚོགས། རྒྱལ་པོའི་དུའི་ཚོགས། [བེའུའི་ཚོགས། མེའི་ཚོགས། བའི་ཚོགས། བཱ་ཎྡྷ་ཀྲཾ། ཀེདཱ་རཱ་དྱ་ཉྩ]། ཀེ་དཱ་ར་ལས་ཡཉ་འང་ངོ་། །
