ལས་ཋ་ཀའོ། །པྲཱ་བྲྀ་ཥའི་སྒྲ་ལས་དེར་སྐྱེས་པའི་དོ་ན་ལ་བདུན་པའི་མཐའ་ལ་ཋ་ཀ་འགྱུར་ཏེ། པྲཱ་བྲྀ་ཥི་ཀཿ། ཏ་ཏྲ་བཧ་བཿ། དེ་ལ་དངོས་པོ་ལའོ། །དེའི་དངོས་པོ་བརྗོད་པ་ལ་བདུན་པའི་མཐའ་ལ་ཨཎ་ལ་སོགས་པ་འགྱུར་ཏེ། མ་ཐུ་རཱ་ཡཱཾ་བྷ་བཿ། ཞེས་པའི་དོན་ལ། མཱ་ཐུ་རཿ། དི་གཱ་
དི་བྷྱོ་ཡ་ཏ། དི་ཤ་ལ་སོགས་པ་རྣམས་ལས་ཡཏའོ། །དི་ཤ་ལ་སོགས་ཋ་ཀ་ཀ་འཕེལ་དོ་ན་ཋཀ་ཨེ་ཀ །དབྱར་སྐྱེས། པ་རྣམས་ལ་དངོས་པོ་ལ་ཡ་ཏ་འགྱུར་ཏེ། ཡ་ཏ་ དི་ཤྱཾ། ཡ་ཏ་ བརྒྱི། ཤ་རི་རཱ་བ་ཡ་བཱ་ཙྩ། ལུས་ཀྱི་ཆ་ཤས་ལས་ཀྱང་ངོ་། །བཅོམ་བརླག་ཉིད། ལུས་ཀྱི་ཆ་ཤས་ བཅོམ་བཪླག་ཉིད་
ལས་ཀྱང་དངོས་པོ་ལ་ཡཏ་འགྱུར་ཏེ། ཡཏ་ ད་ནྷྱི། ཡཏ་ ཀརྞྱཾ། ཡཏ་ ཧ་སྟྱཾ། [ཨཽཥྵྱཾ།] ཡཏ་ པཱ་དྱཾ། འགྲེལ་པར་ལུང་དྲངས་པ། ཨདྷྱཱ་ཏྨཱ་དེཿཀ །ཕྱོགས་ཉིད། སྡེ་ཚན་ཉིད། ཨདྷྱཱ་ཏྨ་ལ་སོགས་པ་རྣམས་ལས་ཋ་ཀའོ། །ཨ་དྷྱཱ་ཏྨ་ལ་སོགས་པ་རྣམས་ལས་དངོས་པོ་ལ་ཋ་ཀ་
འགྱུར་ཏེ། ཨཱ་དྷྱཱ་ཏྨི་ཀཾ། ཨཱ་ཏྷི་དཻ་བ་ སོ་ཉིད། རྣ་བ་ཉིད། ལག་པ་ཉིད། མཆུ་ཉིད། རྐང་པ་ཉིད། ཀྲཾ། ཨཻ་ཧ་ལཽ་ཀི་ཀཾ། ཏ་ཏ་ཨཱ་ག་ཏཿ། དེ་ལས་འོངས་པ་ལའོ། །དེ་ལས་འོངས་པའི་དོན་བརྗོད་པ་ལས་ལྔ་པའི་མཐའ་ལ་ཨིཎ་འགྱུར་ཏེ། མཐུ་རཱ་ཡཱཿཨཱ་ག་ཏཿ། ཞེས་
ཋ་ཀ་ཀ་ལ་འཕེལ་བ་ཎ་ཨེཀ །ནང་པ་ཉིད། ཋཀ་ན་ཨ་ཅུ་ཤར་ཀ་ལ་སོབས་ཀྱིས་སྔ་ཕྱི་གཉིས་འཕེལ་ག པའི་དོན་ལ། མཱ་ཐུ་རཿ། ལྷག་པའི་ལྷ་ཅན། འཇིག་རྟེན་བདེ། ཋཀ་འཕེལ་བ། གནས་ནས་འོངས་པ་ལ་ཋཀའོ། །གནས་ནས་འོངས་པ་བརྗོད་པ་ལ་ལྔ་པའི་མཐའ་ལ་ཋཀ། བཅོམ་བརླག་ནས་འོངས་པ།
འགྱུར་ཏེ། ཋཀ་ཨཱ་པཎི་ཀཾ། སོ྅་སྱ་ནི་བཱ་སཿ། དེ་འདིའི་གནས་ལའོ། །དེ་དེའི་གནས་ བཅོམ་བརླག་ནས་འོངས་པ། བརྗོད་པ་ལ་དྲུག་པའི་མཐའ་ལ་ཨཎ་འགྱུར་ཏེ། མ་ཐུ་རཿཔ་ནི་བཱ་སོ་སྱ། ཞེས་པའི་དོན་ལ། མཱ་ཐུ་རཿ། ཏེ་ན་པྲོ་ཀྟཾ། དེས་རབ་ཏུ་གསུང་བ་ལའོ། དེས་རབ་ཏུ་གསུང་
པའི་དོན་ལ་གསུམ་ རང་གནས་ནས་འོངས། པའི་མཐའ་ལ་ཨཎ་འགྱུར་ཏེ། བྷ་ག་བ་ཏཱ་པྲོ་ཀྟཾ། བཅོམ་ལྡན་འདས་ཀྱིས་རབ་ཏུ་གསུངས་པ། ཞེས་པའི་དོན་ལ། ཨཎ་ བྷྲ་ག་བ་ཏི། བཅོམ་ལྡན་འདས་ཀྱིས་རབ་ཏུ་གསུངས་བ། འདིའི་གནས་བཅོམ་བརླག །བཅོམ་བརླག་པ། མཱ་ན་བཿ། མི་ཡིས་རབ་ཏུ་གསུངས་པ། ཀྲྀ་ཏེ་གྲ་ནྠི། བྱས་
