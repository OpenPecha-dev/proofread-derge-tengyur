སྱཻ། བྷྱཱཾ་བྷྱ་ས་ལ། སརྦཱ་བྷྱཱཾ། སརྦཱ་བྷྱཱཾ། ང་སིལ། སརྦཱ་སྱཱཿ། སརྦཱ་བྷྱཱ། སརྦཱ་བྷྱཿ། དྲུག་པ་ལ། སརྦ་སྱཱཿ། སརྦ་ཡོཿ། སརྦཱ་སཱཾ། བདུན་པ་ལ། །སརྦ་སྱཱཾ། སརྦ་ཡོཿ། སརྦཱ་སུ། དེ་བཞིན་དུ། བི་ཤྭཱ་ལ་སོགས་པ་རྣམས་ཀྱང་རིགས་འགྲེའོ། །ཨུ་དྷ་ར་པཱུ་ནཱ་ཞེས་པ་ལ་རྣམ་དབྱེ་ང་འགྲོ་བ་རྣམས་
མ་གཏོགས་སརྦཱ་དང་འདྲ། ང་འགྲོ་བ་རྣམས་ལ་ངམཱའི་སྐབས་དང་སརྦཱ་སྐབས་གཉིས་ཀར་གྱི་བྱ་བ་འཇུག་པས་འདམ་ཀ་ཅན་ནོ། །ཨུ་དྟ་ར་པཱུ་རྦཱ། ཨུཏྟ་ར་པཱུ་རྦ། ཨུཏྟ་ར་པཱུ་རྦཱཿ། ཧེ་ཨུ་ཏྟ་ར་པཱུ་རྦེ། སྣ་ཚོགས་མ། ཧེ་ཨུ་ཏཱ་ར་པཱུ་ཏྲེ། ཧེ་ཨུ་ཏྟ་ར་པཱུ་རྦཿ། གཉིས་པ་རྣམས་ལ། ཨུ་ཏྟ་ར་པཱུ་རྦཱཾ། ཨུ་ཏྟ་ར་
པཱུ་རྦེ། ཨུ་ཏྟ་ར་པུ་རྦཱཿ། གསུམ་པ་རྣམས་ལ། ཨུཏྟ་ར་པཱུ་རྦ་ཡཱ། ཨུཏྟ་ར་པཱུ་རྦཱ་བྷྱཱཾ། ཨུཏྟ་ར་པཱུ་རྦཱ་ལ། དེ། བིབྷཱ་ཥཱི་དི་ཀ་ས་མཱ་སེ་བ་ཧུ་བྲཱི་ཧཽ། འབྲུ་མང་པོའི་ཚིག་སྡུད་ལ་ཕྱོགས་ལ་འདམ་ཀའོ། །ཕྱོགས་ཀྱི་སྒྲ་འབྲུ་མང་པོའི་ཚིག་སྡུད་བྱས་པ་ཨཱ་པ་མཐའ་ཅན་ནི་རྣམ་དབྱེ་ང་འགྲོ་
བའི་མིང་ཅན་རྣམས་ལ་སརྦཱའི་སྐབས་ལྟར་སྱཱཊ་ཨཱ་ག་མ་བྱིན། རྟགས་མཐའ་ཐུང་དུར་བྱེད་པ་དང་། ར་མཱའི་སྐབས་ལྟར་ལཱཊ་སྟེར་བ་འདམ་ཀ་ཅན་ནོ། །ཨུཏྟ་ར་ཕཱུ་རྦ་སྱཻ། ཨུཏྟ་ར་པཱུ་རྦཱ་བྷྱཱཾ། ཨུཏྟ་ར་པཱུ་རྦཱ་བྷྱཿ། སོགས་སོ། བིབྷཱ་ཥཱ་དྭི་ཏཱི་ཡཱ་ཏྲྀ་ཏཱྀ་ཡཱ་བྷྱཱཾ། དྭི་ཏཱི་ཡཱ་ཏྲྀ་ཏཱི་ཡཱ་དག་ལས་
འདམ་ཀའོ། །དྭི་ཏྲཱི་ཡཱ་དང་ཏྲྀ་ཏཱི་ཡཱ་གཉིས་ལའང་རྣམ་དབྱེ་ང་འགྲོ་བ་རྣམས་ལ་ཡཱ་ཊ་དང་སྱཱ་ཊ་འདམ་ཀ་ཅན་ཏེ། དཔེར་ན། བཞི་པའི་དེ་ལ། དྭི་ཏཱི་ཡཱ་ཡཻ། དྭི་ཏཱི་ཡཱ་ཡཻ། ཏྲྀ་ཏཱི་ཡ་སྱཻ། ཏྲྀ་ཏཱི་ཡ་སྱཻ། ལྟ་བུའོ། །ཨམྦཱ་ཞེས་པ་ཡང་སཾ་བུ་དྦྷི་མ་གཏོགས་རྣམ་དབྱེ་ཐམས་ཅད་ར་མྔའི་སྐབས་དང་འདྲ།
ས་བུདྡྷི་ལ་ཅུང་མི་འདྲ་སྟེ། ཨམྦཱ་ལ། སཾ་བུ་དྡྷི། ཨམྦཱ་རྠ་ན་དྱོཿཧྲ་སྭཿ། ཨ་མྦཱའི་དོན་ན་དཱི་རྣམས་ཀྱི་ཐུང་ངུའོ། །ཨམྦཱའི་སྒྲ་དང་ན་དཱིའི་མིང་ཅན་རྣམས་སཾ་བུདྡྷི་ཕ་རོལ་ལ་རྟགས་མཐའ་ཐུང་ངུར་འགྱུར་ཏེ། ཧེ་ཨམྦ། ཨཀྐཱ་དང་། ཨལླཱ་ཡང་འདི་དང་འདྲའོ། འགྲེལ་པར། ཨམྦཱའི་དོན་ཅན་ཡིན་
གྱང་དེ་ཕན་གྱི་རྐྱེན་ཌ་དང་། ལ་དེང་། ཀ་གནས་པ་དེ་ལ་མོ་རྟགས་སུ་བསྒྲུབ་པའི་རྐྱེན་ཊཱ་པ་བྱིན་པ་ལྟ་བུ་ལ་སཾ་བུདྡྷི་ལ་ཡང་ཐུང་ངུར་མི་འགྱུར་རོ། །ཞེས་བཤད་པས། དཔེར་ན། ཨམྦཱཌཱལ། ཨམྦཱ་ནྦྷི་བྱིན་ཐུང་ངུར ནོར་རྒྱུན་མ། འབིགས་ཁྱེད་ལྷ་མོ། མི་འགྱུར། ར་མཱ་དང་འདྲ་བས། ཧེ་ཨ་མྦཱ་ཌེ། དེ་
