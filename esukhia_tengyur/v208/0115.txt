སྟྲཱི་མ་ཡིན་པ་ཞེས་སྟྲཱིའི་སྒྲ་ནི་ཨི་ཡ་ཨུ་ཝར་འགྱུར་ཡང་ན་དཱིའི་མིང་ཅན་ཡིན་ལ། སྟྲཱི་མ་ཡིན་པ་ཨི་ཡ་ཨུ་ཝ་གནས་པ་སྟེ། ཨི་ཡར་འགྱུར་པ་དང་ཨཱུ་ཨུ་ཝར་འགྱུར་པ་རྣམས་བུད་མེད་དུ་བརྗོད་པའི་སྐབས་འདིར་ཡང་ན་ན་དཱིའི་མིང་ཅན་མ་ཡིན་ཏེ། སྐབས་འདིར་ཤྲཱི་ཡང་། ན་པཱིའི་
མིང་ཅན་མ་ཡིན་བས་སཾ་བུདྡྷི་དབྱི་བ་དང་རྟགས་ཐུང་ངུར་འགྱུར་པ་སོགས་མི་བྱེད། ཧེ་ཤྲཱིཿ། གཉིས་པ་རྣམས་ལ། ཤྲི་ཡཾ། ཤྲི་ཡཽ། ཤྲི་ཡཿ། གསུམ་པ་ལ། ཤྲི་ཡཱ། ཤྲཱ་བྷྱཱཾ། ཤྲཱབྷིཿ། ཤྲཱི་ལ། ངེ་། སྤྱིར་བཏང་ཨི་ཡ་ཨུ་ཝ་ཐོབ་པ་རྣམས་བུད་མེད་དུ་བརྗོད་པའི་
སྐབས་ན་དཱིའི་མིང་ཅན་མ་ཡིན་ནོ་ཞེས་བཀག་ཀྱང་དམིགས་བསལ། ང་འགྲོ་བ་ལ་ཐུང་ངུའི་ཡང་ནའོ་ཞེས་པའི་འགྲེལ་པར། ཨཱི་ཨཱུ་དག་ཀྱང་ཡང་ན་པར་བཤད་པས། རྣམ་དབྱེ་ང་འགྲོ་བའི་མིང་ཅན་རྣམས་ལན་དཱིའི་མིང་ཅན་གྱི་བྱ་བ་ལྟར་ཨཱ་ཊ་སྟེར་བ་དང་ཨི་ཡ་ཨུ་ཝར་འགྱུར་པ་གཉིས་ཀ་ཐོབ་
བས། ཤྲི་ཡེ། ཤྲི་ཡཻ། བྷྱ་མ་བྷྱཿལ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ། ལྔ་པའི་ང་སིལ། ཤྲི་ཡཿ། ཤྲི་ཡཱཿ། བྷྱཱ་མ་བྷྱཿལ། ཤྲཱིབྷྱཱཾ། ཤྲཱི་བྷྱཿ། དྲུག་བའི་ངས་ལ། ཤྲི་ཡཿ། ཤྲི་ཡཱཿ། ཨོ་ས་ལ། ཤྲི་ཡོཿ། ཤྲཱི་ལ། ཨཱ་མ། བཱ་མི། ཨཱམ་ལ་ཡང་ནའོ། །ཨི་ཡ་ཨུ་ཝ་ཐོབ་པ་རྣམས་རྣམ་དབྱེ་
དྲུག་པའི་ཨཱ་མ་ཕ་རོལ་ལ་ན་དཱིའི་མིང་ཅན་ཡང་ནའོ། ཤྲི་ཡཱཾ། ན་དཱིའི་མིང་ཅན་ལྟར་ན། ཐུང་དུ་ནི་དཱིའི་ཞེས་བས། ནུཊ། ཤྲཱི་ཎཱཾ། བདུན་པའི་ངི་ལ། ཤྲི་ཡཱཾ། ཤྲི་ཡི། ཨོ་ས་དང་སུབ་ལ། ཤྲི་ཡོཿ། ཤྲཱི་ཥུ། སུདྡྷི་ལ་སོགས་པ་ཡང་འདི་དང་འདྲའོ། །པྲ་དྷཱི་ཞེས་པའི་སྒྲ་མོ་རྟགས་ལ་
ཡང་ཆོག་ཐམས་ཅད་སྐྱེས་པའི་སྐབས་སུ་བཤད་པ་དང་ཁྱད་པར་མེད་དོ། །དེ་རྣམས་ཀྱིས་བུད་མེད་ཀྱི་རྟགས་ལས་ཨཱི་རིང་མཐའ་ཅན་གྱི་བྱ་བ་རྣམས་བསྟན་ཟིན་ཏོ། ༈ །ད་ནི་མོ་རྟགས་ཨུ་ཐུང་མཐའ་ཅན་རྣམས། བློ་བཟང་མ། བརྗོད་པར་བྱ་སྟེ། དྷེ་ནུ་ལ། སུ། བློ་བར་མ། དྷེ་ནུཿ། ཨཽ་ལ། དང་པོ་
དག་གི་ཞེས་པས། དྷེ་ནཱུ། དྷི་ནུ། ཛས། ཛས་ལ་འང་ངོས། དྷེ་ན་བཿ། གཉིས་པ་རྣམས་ལ། དྷེ་ནུ། ཨ་མ། ཨ་མ་ལ་སྔ་མ་ཞེས་བས། དྷེནུཾ། དྷེནུ། ཨཽ། དྷ་ནཱུ། དྷེ་ནུ། ཤས། དང་པོ་དག་གི་མདོས། ངྷེ་ནཱུཿ། གསུམ་པ་རྣམས་ལ། འཇོ་མ། དྷེ་ནུ་ཊཱ། ཊཱ་ནཱར་མི་འགྱུར། དྷེ་ནཱ། དྷེ་ནུ་བྷྱཱཾ།
