སྒྲ་རྣམས་དང་ཉེར་བསྒྱུར་སུ་སྔོན་ན་ཡོད་པའི་པཱ་དའི་མཐའ་ཡང་འབྲུ་མང་པོའི་བསྡུ་བའི་སྐབས་འདིར་དབྱི་སྟེ། དྭོོ་པཱ་དོོ་ཡ་སྱ། ཞེས་པ་ལ། དྭི་པཱཏ། དྭི་པཱད། དེ་བཞིན་དུ། ས་ཧ་སྲ་པཱ་ངཱཿཡ་སྱ། ཞེས་པ་ལ། ས་ཧ་སྲ་པཱཏ། སུ་བཱ་དོོ་ཡ་སྱ། ཞེས་པ་ལ། སུ་པཱཏ། སུ་
པཱད། བ་ཡ་སི་དནྟ་སྱད་ཏྲཾ། ན་ཚོད་ལ་དནྟའི་ད་ཏྲྀའོ། །འབྲུ་མང་པོའི་བསྡུ་བའི་སྐབས་འདིར་ན་ཚོད་བརྗོད་པའི་གང་གི་རང་བ་གཉིས། དོན་དང་ རྐང་པ་གཉིས། འབྲེལ་བའི་དནྟ་ནི་སུ་དང་གྲངས་སྔོན་ གང་གི་རྐང་བ་སྟོང་། རྐང་བ་སྟོང་། རྐང་བ་སྟོང་། གང་གི་རྐང་པ་མཛེས་བ། དུ་བྱུང་ན་ད་ཏྲྀ་ཞེས་ རྐང་མཛེས།
པར་འགྱུར་ཏེ། དྭོོད་ནྟོོ་ཡ་སྱ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་ཕྱིས། ངནྟ་ད་ཏྲྀ་རུ་འགྱུར། རྀ་ཨུ་ཀ་འགྲོ་བ་ལ་ནུཾ་གྱི་དོན། དྭི་དན། དེ་བཞིན་དུ། ཙ་ཏྭ་རཿདནྟཱཿཡ་སྱ། ཞེས་པ་ལ། ཙ་ཏུ་རྡན། དེ་བཞིན་དུ། ཥཊ་དནྟཱཿཡ་སྱ། ཞེས་པ་ལ། ཥོ་ཌན། སུ་དནྟཱཿཡ་སྱ། ཞེས་
པ་ལ། སུ་དན། གང་གི་སོ་དག་གཉིས། ད་ཏྲྀལ་སུ་ ཨུ་ཀ་འགྲོ་བ་རྣམས་ལ་ནུ་མྀ་ར་དེའི་དོན་ཡིན་པས་ཕྱིས་ ཧལ་སོགས་ཀྱིས་སུ་ཕྱིས་བརྩེགས་པའི་མཐའ་དབྱི་འོས་ར་ཕྱིས། སོ་གཉིས། གང་གི་སོ་བཞི་པ་རྣམས། ན་ཚོད་ཀྱི་དོན་ཅན་ སོ་བཞི་པ། གང་གི་སོ་དྲུག་པ། མིན་ན་ད་ཏྲྀར་མི་ སོ་དྲུག་པ་སྟེ་རྟ་བོང་། གང་གི་སོ་མཛེས། འགྱུར་
ཏེ༈ དྭོོ་ སི་མཛེས་ཏེ་དར་བབ། དནྟོོ་ཡ་སྱ། ཞེས་པ་ལ། དྭི་དནྟ་ཀརཱི། ཨགྲཱ་ཏ་ཤུངྔྷ་ཤུ་བྷྲ་བྲྀ་ཥ་པཱ་རཱ་ཧེ་བྷྱཤྩ། ཨ་གྲ་ཨནྟ་ཤུངྔྷ་ཤུ་བྷྲ་བྲྀ་ཥ་ བཱ་རཱ་ཧ་རྣམས་ལས་ཀྱང་ངོ་། །ཨ་གྲ་ལ་སོགས་པ་འདི་རྣམས་སྔོན་དུ་བྱུང་ན་འབྲུ་མང་པོའི་བསྡུ་བའི་སྐབས་འདིར་དནྟ་ནི་ད་ཏྲྀ་རུ་ཡང་ན་འགྱུར་ཏེ།
ཀུཌྨཱ་ལ་སྱཱ་གྲ་དནྟོ་ཡ་སྱ། ཞེས་པ་ལ། རྣམ་དབྱེ་སོགས་མཆེ་གཉིས་པ་སྟེ་གླང་པོ་ཆེ། ཕྱིས། དནྟ་ད་ཏྲྀ་རུ་ཡང་ན་འགྱུར་བས། ཀུཌྨཱ་ལཱ་གྲ་དནྟཿ། ཀུཌྨ་ལཱ་གྲ་དན། དེ་བཞིན་དུ། ཤཏྔྷཿདནྟོ་ཡ་སྱ། ཞེས་པ་ལ། ཤུཏྔྷ་དནྟཿ། ཤུངྔྷ་དན། ཤུ་བྷྲཿདནྟོ་ཡ་སྱ། ཞེས་པ་ལ། ཤུ་བྷྲ་
དམྟཿ། སོ་དཀར་པོ། ཤུ་ གང་གི་སོ་གུག་བའི་རྩེ་མོ། བྷྲ་དན། བྲྀ་ཥསྱ་དནྟོ་ཡ་སྱ། གང་གི་གླང་གི་སོ། ཞེས་པ་ལ། བྲྀཥ་དནྟཿ། གླང་སོ། བྲྀཥ་དན། བཱ་ སོ་གུག་པའི་རྩེ་མོ། རཱ་ཧ་སྱ་དནྟོ་ཡ་སྱ་ གང་གི་ཕག་པའི་སོ། གང་གིས་གཙང་མ། ཞེས་པ་ལ། སོ་གཙང་མ། གང་གི་སོ་དཀར་པོ། བཱ་རཱ་ཧདནྟཿ། ཕག་སོ། བཱ་རཱ་
