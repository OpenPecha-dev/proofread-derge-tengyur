གྱིས། ཨུ་ཥྞཀ། ཨུ་ཥྜི་ག། ཨཽ་དང་ཛས་ལ། ཨུཥྞི་ཧཽ། ཨུཥྞི་ཧཿ། སོགས་ཏེ། རྣམ་དབྱེ་ཨ་ཙ་ཐོག་མ་ཙ་ན་རྣམས་དེས་རིགས་འགྲེ། བྷཱམ་བྷིས་ལ། ཀྭིན་གྱི་ཀུ། ཛྷ་ཤ་ལ་ཛྷལ་རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོས། ཨུཥྞི་གྦྷི། ཨུཥྞ་གྦྷིཿ། སུ་པ་ལ། ཀྭིན་གྱི་ཀུ། ཁར་ལ་ཙ་ར་འོས། གྷ་ཀ། ཀ་
ཥ་འདུས་པཀྵ། ཨུཥྞི་ཀྵུ། བོད་ཚིག་དང་པོ་དང་འདྲ། ༈ བ་ཡིག་མཐའ་ཅན། དི་བ་ལ། སུ། དི་ཝྲའི་ཝྲ་ཨཽ། དྱཽཿ། ཨཽ་དང་ཛས་ལ། དིབཽ། དི་བཿ། ཨམ་ལ་ རྐྱེན། ཡང་ན་པར་བཤད་པ་ལྟར། དྱཾ། དི་གི ཨཽ་དང་ཤས་ལ། དི་བ། དི་བཿ ཊཱ་ལ། རྐྱེན། དི་ཝཱ། བྷྱཱམ་བྷིས་ལ། དི་བྲའི་ཨུ་ཏ་འོས་
བ་ཨུ། དཱུ་བྷྱཱཾ། དཱུ་བྷི་ན། ང་ལ། དི་བེ། བྷྱཱམ་བྷྱས་འདྲ། ང་སི་ལ། དི་པཿ། བྷྱཱམ་བྷྱས་འདྲ། འོད་གསལ། དྲུག་པ་རྣམས་ལ། དི་བཿ། དི་ངོཿ། དི་བཱཾ། བདུན་པ་རྣམས་ལ། དི་བི། དི་པོཿ། དྱཱཥུ། བོད་ཚིག་ལ། ཧེ་དྱཽཿ། ཧེ་དི་བཽ། ཧེ་དི་བཿ། ༈ དེ་བ་སྔ་མའི་ཏ་ཡིག་མཐའ་ཅན། དྻཱུརྞ་ཏ་འགྲེལ་པར་གཞན་
ནས་ལུང་དྲངས་ཏེ། ཨུ་ཡིག་ཕ་རོལ་ལ་དི་བྲ་ཀྱིབ་ཨུར་འགྱུར་རོ་ཞེས་པ་བྱུང་བས། བ་ཨུ། ཨ་ཀ་ཀི་འདྲ་བ་ལ་རིང་བོ། ཨི་ཀའི་ཨ་ཙ་པ་ཡཎ། དྻཱུརྞ་ཏ་ལ། སུ། ཧ་ལ་སོགས་ཀྱིས་སུ་ཕྱིས། གནས་མེད་པ་ལ་ཡང་ན་འོས། དྻཱུརྞུཏ། དྻཱུརྞུད། ཨཽ་དང་ཛས་ལ། དྻཱུརྞུ་ཏཽ། དྻཱུརྞ་ཏཿ། སོགས་ཏེ་ཨ་ཙ་
