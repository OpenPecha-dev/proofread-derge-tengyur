ཛས་ཨི་ཀ་གི་ཨ་ཙ་ལ་ཡ་ཎ་འོས། བཱ་ཏ་པྲ་མྱཽ། བཱ་ཏ་པྲ་མྱཿ། རྣམདབྱེ་གཉིས་པ་ལ། བཱ་ཏ་པྲ་མཱིཾ། བཱ་ཏ་པྲ་མྱཽ། གསུམ་པ་ལ། བཱ་ཏ་བྲ་མྱཱ། བཱ་ཏ་པྲ་མཱིུ་བྷྲཱིཿ། བཱ་ཏ་པྲ་མཱི་བྷིཿ། བཞི་པ་ལ། བཱ་ཏ་པྲ་ནྱེ། བཱ་ཏ་པྲ་མཱི་བྷཾ། བཱ་ཏ་པྲ་མཱི་བྷྱཿ། ལྔ་པ་ལ། བཱ་ཏ་པྲ་མྱཿ། བཱ་ཏ་པྲ་མཱི་བྷྱཱཾ།
བཱ་ཏ་པྲ་མཱི་བྷྱཿ། དྲུག་པ་ལ། བཱ་ཏ་པྲ་མྱཿ། བཱ་ཏ་པྲ་མྱོཿ། བཱ་ཏ་པྲ་མྱཱཾ། བདུན་པ་ལ། བཱ་ཏ་བྲ་མཱི། བཱ་ཏ་པྲ་མྱོཿ། བཱ་ཏ་པྲ་མཱིཥུ། བོད་ཚིག ཧེ་པཱ་ཏ་པྲ་མཱིཿ། ཧེ་པཱ་ཏ་བྲ་མྱཽ། ཧེ་པཱ་ཏ་ཕྲ་མྱཿ། ཧེ་པཱ་ཏ་པྲ་མྱཿ། རྣམས་གྲུབ་སྟེ་ཆོ་ག་གོ་བདེ་འོ། །དེ་བཞིན་དུ། དེ་ཝ་ཡ་ཛཱི། ཨ་ཏི་ལ་ཀྵཱི་ལ་སོགས་པ་མང་པོ་
དག་ཆོ་ག་འདི་ཁོ་ན་དང་འདྲའོ། །ཨཱི་རིང་མཐའ་ཅན་བྱིངས་གནས་པ་རྣམས་བྱ་བ་ཅུང་མི་འདྲ་སྟེ། སུ་དྡྷི་ལ། སུ། སུ་དྡྷི༔། སུདྡྷཱི། སུ་དྡྷཱིལ། ཨཽ་དང་། ཛས། ཨ་ཙི་ཤཱུ་དྟཱ་ཏུ་བྷཱ་ཝཱཾ་ཡྭོ་རི་ཡ་ངུ་ཝ་ངཽ་། ཨ་ཙ་ལ་ཤྣུ་བྱིངས་བྷྲཱུ་རྣམས་ཀྱི་ཨི་ཨུ་དག་གི་ཨི་ཡ་ཨུ་ཝ་ལྷ་མཆོད་པ། ཤིད་ཏུ་ཕུན་ཚོགས་པ།
དག་གོ། ཕ་རོལ་ཨ་ཙ་སྟེ་དབྱངས་གང་མཐོང་ཡང་རྐྱེན་ཤྣུ་དང་བྱིངས་ཀྱི་ཨི་རིང་ཐུང་ཨུ་རིང་ཐུང་། བློ་བཟང་ངམ་ལེགས་པའི་བློ། བྷྲུའི་སྒྲ་རྣམས་ཀྱི་ཨི་རྣམས་ཨི་ཡ་དང་། ཨུ་རྣམས་ཨུ་ཝ་རུ་འགྱུར་ཞིང་། སྤྱིར་བཏང་ཨི་ཀ་གི་ཨ་ཎ་འོ་ཞེས་པ་ཐོབ་ཀྱང་། ན་བྷཱུ་སུ་དྡྷཱི་དག་ལ་མ་ཡིན་ནོ། །
བྷཱུ་དང་སུ་དྡྷཱི་དག་གི་ཨཱི་དག་ཨ་ཙ་ཕ་རོལ་ལ་ཡཎ་ར་འགྱུར་བ་མ་ཡིན་ཏེ། ཨི་ཡར་བསྒྱུར་བས། སུ་དྡྷི་ཡཽ། སུ་དྡྷི་ཡཿ། རྣམ་དབྱེ་གཉིས་པ་རྣམས་ལ། སུ་དྡྷི་ཡཾ། སུ་ངྡྷི་ཡཽ། སུ་དྡྷི་ཡཿ། གསུམ་པ་ལ། སུ་དྡྷི་ཡཱ། སུ་དྡྷཱི་བྷྱཾཱ། སུདྡྷཱི་བྷིཿ ལྟ་བུ་སྟེ། རྣམ་དབྱེ་ལྷག་མ་རྣམས་ལ་ཡང་ཧ་ལ་ཐོག་

པ་མ་ཡིན་པའི་འོ། །རྟགས་མཐའ་བྱིངས་ཀྱི་ཨཱི་ཡིག་དང་ཨཱུ་ཡིག གཟུགས་མཛེས་སམ་དཔལ་ལམ་ཕུན་ཚོགས་པའམ་བདེ་བར་འབྱོར་བ། ནས་ཚོང་། ཨ་ཙ་གཅིག་པ་མ་ཡིན་པ་སྟེ། བློ་རབ། དབྱངས་གཉིས་སམ་མང་པོ་ཡོད་པ་སྔོན་ན་ཡི་གེ་མང་པོ་བརྩེགས་པ་མེད་པ་དེ་ནི་ཨ་ཙ་ཕ་རོལ་ལ་ཡ་ཎར་འགྱུར་ཏེ།
