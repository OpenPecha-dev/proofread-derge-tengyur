བླ་ཀྵཱ་དི་བྷྱོཉ་ན། བཱ་ཀྵ་ལ་སོགས་པ་རྣམས་ལས་ཨཎ་འོ། །པཱ་ཀྵ་ལ་སོགས་པ་རྣམས་ལས་འབྲས་བུ་བརྗོད་པ་ལ་ཨཎ་འགྱུར་ཞིང་དབྱིབ་འང་མ་ཡིན་ཏེ། བཱ་ཀྵ་སྱ་པ་ལཾ། ཞེས་པའི་དོན་ལ། ཨཎ་ པླཱ་ཀྵཾ། དེ་བཞིན་དུ། ཨཱ་ཤྭཱ་ཏྠ། ཨོ་རཉ། ཨུ་ལས་ཨཉའོ། །རྟགས་ཨུ་ཡིག་རིང་ཐུང་
མཐའི་ཅན་ལས་རྣམ་འགྱུར་ལ་ཨཉ་འགྱུར་ཏེ་འབྲས་བུ་བརྗོད་པ་ལ་འཇིག་གོ། །ཛམྦྲཱ་ཕ་ལཾ། ཞེས་པའི་དོན་ལ། ཛ་མྦའུཿ། ཛ་མྦྦཱཿབཱ། ཛ་མྦུ་ལས་ [བྦ་]ཀྵའི་འབྲས་བུ། ཡང་ནའོ། ཛ་མྦཱུ་ལས། པཱ་ཀྵའི་འབྲས་བུ། རྟག་ནས་ཀྱི་འབྲས་བུ། ཨཎ་སུ་ཨ་མ། འབྲས་བུ་ལ་ཨཎ་འགྱུར་ཏེ། ཛ་མྦཱུའི་འབྲས་བུ། ཛཱ་མྦ་བཾ།
ལུ་བྩ། དབྱི་བའང་ངོ་། །ཛ་མྦཱུ་ལ་འབྲས་བུའི་དོན་དུ་བྱིམ་པའི་ཨཎ་ཡང་ན་དབྱི་སྟེ། ཛམ་བུའི་འབྲས་བུ། ཨན་ཕྱིས་པའོ། །ཛ་མྦུའི་འབྲས་བུ། ཛཱ་མྦ་བཾ། ཛམྦཱུཿ། ཧ་རཱི་ཏ་ཀྱཱ་དི་བྷྱ་ཤྩ། ཧ་རཱི་ཏ་ཀཱི་ལ་སོགས་པ་རྣམས་ཀྱི་འང་ངོ་། །ཧ་རཱི་ཏ་ཀཱི་ལས་སོགས་ཛ་མྦུ་ལས་ཨཎ་འཕེལ་བ་ཨུའི་ཡོན་ཏན་ཏན་ཨཱ་ཙའི་ཨ་ཡ་བ་སུ།
རྣམས་ལས་འབྲས་བུའི་དོན་གྱི་རྐྱེན་རྣམས་དབྱི་ཞིང་བུད་མེད་དུ་བརྗོད་པའི་རྐྱེན་དབྱིབ་མ་ཡིན་ཏེ། ཧ་རཱི་ཏ་ཀྱཱཿཕ་ལཾ། ཞེས་པའི། ཛམྦཱུའི་འབྲས་བུ། ཨཎ་ཕྱིས་བ། དོན་ལ། ཧཱ་རཱི་ཏ་ཀཱི། ཨཎ་ཕྱིས་པའོ། །དྲཱ་ཀྵཱ། ཏ་ཏྲ་ནི་ཡུ་ཀྷཿ། དེ་ལ་ངེས་པར་སྦྱོར་བའོ། །དེ་ལ་ངེས་པར་སྦྱོར་བའི་དོན་ལ་བདུན་
པའི་མཐའ་ལ་ཐག་འགྱུར་ཏེ། དྭ་རེ་ནི་ཡུ་ཀྟཿ། ཞེས་པའི་དོན་ལ། དཽ་བཱ་རི་ཀཿ། སྒོ་སྲུང་། ཀཱ་རུ་ཎི་ ཨ་རུ་རའི་འབྲས་བུ། ཀཿ། ཤཽ་ལྐ་ཤ་ལི་ཀཿ། འགྲེལ་ཡང་བརྗོད་པའི་ ཨ་རུ་རའི་འབྲས་བུ། རྒུན་འབྲུམ། དོན་ལའང་ཐ་ཀ་འགྱུར་རོ་ཞེས་པ་བཤད་པས། སྭ་སྟི་ཏྱ་ཧཿ། ཞེས་པའི་དོན་ལ། སཽ་
བསྟི་ཀཿ། བདེ་ལེགས་སུ་བྱུང་ཞེས་བརྗོད་པ། སྦྲོ་ལ་ངེས་པར་སྦྱོར་བ། དྲཱ་ར་ལ་ཋཀ་དྲཱ་ར་སོགས་ཀྱི་ཨཻ་ཙ་[ཨཱ་ག་མ་ཋ་ཨི་]ཀ། [ཋཀ་འཕེལ་བ་ཋའི་ཨིཀ] །སྙིང་རྗེ་ཅན། ཤོ་གམ་གྱི་བལ་ ནྱ་གྲོ་དྷ་སྱ་ཙ་ཀེ་ལ་སྱ། ནྱ་གྲོ་དྷའི་འང་གཅིག་ཉིད་ལའོ། །ནྱ་གྲོ་དྷའི་འང་ཚིག་གྲོགས་གཞན་མེད་པའི་གཅིག བདེ་ལེགས་སུ་གྱུར་ཅེས་བརྗོད་པ། ཉིད་ལ་དེ་ཕན་གྱི་
[སཱྭ་སྟི་ལ་ཋ་ཀེ་རྭཱ]སོགས་ཀྱི་ཨི་[ཙ་སཱ]་ག་མ་[ཋཨི་ཀཎྡྣ་རྐྱེན་]འཕེལ་བའི་རྒྱུ་མཚན་ཕ་རོལ་ལ་འཕེལ་བ་མི་ཕྱེད་པར་ཨཻ་ཙ་ཨཱ་ག་མ་འགྱུར་ཏེ། ནྱ་གྲོ་དྷ་ཕ་ལཾ། ནྱ་གྲོ་དྷའི་འབྲས་བུ། ཞེས་པའི་དོན་ལ། ནྱ་གྲོ་དྷ་ལ་ཨཎ་ཨཻ་ཙ་ཨཱ་ག་མ། ནེ་ཡ་གྲོ་དྷཾ། ཚིག་གཅིག་ཉིད་མ་ཡིན་ན་ཨཻ་ཙ་མི་སྟེར་ཏེ། ནྱ་གྲོ་དྷ་ལེ་བྷ་པླཿ།་ནྱ་གྲོ་དྷའི་རྩ་བ།
