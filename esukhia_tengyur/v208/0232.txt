བལ་དང་སྦྱོར་བ་ལ། བྡཱུཏཱ་ཡ་བ་ལིཿ། ཞེས་པ་ལ། ཞེས་དབྱེ་བསྡུས། བྟཱུཏབ་ལིཿ། དེ་བཞིན་དུ། དེ་ན་ར་སུ་ཁཿ། དེ་གོ་ཧི་ཏཾ། པཉྩ་མཱི་བྷ་ཡེན། འཇིགས་པས་ལྔ་པའོ། འཇིགས་པའི་རྒྱུའི་དོན་བརྗོད་པའི་སྦྱོར་བ་ལ་ལྔ་པའི་མཐའ་དོན་གཅིག་པར་འགྱུར་
ཏེ། ཙོོོརཱཏ་བྷ་ཡཾ། ཞེས་འབྱུང་པོའི་ཆེད་གཏོར་མ། པ་ལ་རྣམ་དབྱེ་ལྔ་པའི་ འབྱུང་པོའི་གཏོ་རམ། རྗེས་བསྡུས། མི་ཁ་བདེ་བ། མི་ལ་བདེ་བ། འཁང་ལ་བསྲུང་བ། བ་ལང་མ་ཕན་པ། ཅོར་བྷཡཾ། དེ་བཞིན་དུ། ངིས་བྲྀ་ཀ་བྷི་ཏིཿ། བསི་བྲྀ་ཀ་བྷིཿ། ཨ་པེ་ཏཱ་པོ་དྷ་མུཀྟ་པ་ཏི་ཏཱ་པ་ཏྲ་སྟེེར་ལྤ་
ཆོམ་ཆུན་ལས་འཇིགས། ཤཿ། །ཨ་བེ་ཏ་ ཨ་པོ་ཌྷ་ མུཀྟ་ བ་ཏི་ཏ་ ཨ་བོ་ ཏྲསྟ་རྣམས་ཀྱི་ཆུང་ངུ་ལའོ། ཨ་བེ་ཏ་སོགས་ཚེགས་ཆུང་དུ་བརྗོད་ཆོམ་རྐུན་ལས་འཇིགས། བ་ཅན་དང་སྦྱར་བ་ལ་ སྤྱང་ལི་པས་དཱག། ལྔ་པའི་ སྤྱང་ཀི་ལས་སྐྲག། མཐའ་དོན་གཅིག་པར་འགྱུར་ཏེ། །སུ་ཁཱ་ད་
པེ་ཏཿ། ཞེས་པ་ལ། ལྔ་བའི་རྗེས་བསྡུས། སུ་ཁཱ་པེ་ཏཿ། དེ་བཞིན་དུ། ངསི་ཀལྤཱ་པོ་ཌྷཿ། ངསི་ཙ་ཀྲམུ་ཀྟཿ། ངསི་སྲག་པ་ཏི་ཧཿ། ཏ་རཾ་གཱཔ་ཏྲསྟ། སྟོ་ཀཱནྟི་ཀངཱུ་རཱ་རྦཀྲྀ་ཚྲཱཎི་ཀྟེན། སྟོ་ཀ་ཨནྟི་ཀ་རིང་བའི་དོན་ཀྲྀ་ཚྲ་རྣམས་ཀྟའིའོ། ཀྟའི་རྐྱེན་ཅན་སྟོ་ཀ་དང་། ཨནྟི་ག་དང་རིང་
བདེ་བ་ལས་སྦྱོར། བའི་དོན་རྣམས་དང་། ཀྲྀཚྲ བདེ་བསྦྱོར། རྣམས་ཀྟའི་རྐྱེན་ཅན་ སྟོབས་དང་མཉམ། འཁོར་ལོ་ལས་གྲོལ། བདེ་འགྲོ་ལས་ལྷུང་། བརླབས་ལ་སྐྲབ། དང་སྦྱོར་བ་ལ་ལྔ་པའི་མཐའ་མི་དབྱི་ཞིང་རང་སར་གནས་ཀྱང་དེ་དག་ལ་མི་དབྱི་བའི་ཚིག་སྡུད་ཅས་ཟེར་རོ། སྙོ་ཀཱནྨུསྟཿ། ཅུང་ཟད་གྲོལ་
ཨ་ལྕཱནྨུཀྟཿ། ཨནྟིཀཱནླུཀྟཿ། དཱུརཱནྨུཀྟཿ། ཀྲྀཚྲཱནྨུཀྟཿ། ཞེས་པ་རྣམས་ལྔ་པའི་རྗེས་མ་ཕྱིས་པའོ། །ཥཥྵཱི། དྲུག་པའོ། རྣམ་དབྱེ་དྲུག་པའི་མཐའ་ཡང་སུ་པ་མཐའ་དང་སུ་པ་མཐའ་དོན་གཅིག་པར་འགྱུར་ཏེ། རཱཛྙཿཔུརུ་ཥཿ། ཞེས་པ་ལ། རྣམ་དབྱེ་ཕྱིས། རཱཛན་དུ་
ཅུང་ཟད་ལས་གྲོལ། ཉི་ལགས་གྲོལ། རེ་བ་ལས་གྲོལ། །རིང་བ་ལས་གྲོལ། སྡུག་བསྔལ་ལས་གྲོལ། ལོག མིང་མཐའི་ནང་བྱི་འོས། ན་ཕྱིས། རཱཛཔུརུ་ཥཿ། རྒྱལ་པོའི་སྐྱེས་བུ། འགྲེལ་པར། ཀྲྀཏ་སོགས་ཀྱི་རྐྱེན་མཐའ་ཅན་དང་སྦྱོར་བ་ལས། རྒྱལ་པོའི་སྐྱེས་བུ་སྟེ་རྒྱལ་པོའི་མི་སྣ། མཚན་ཉིད་བརྗོད་པ་ལ་
