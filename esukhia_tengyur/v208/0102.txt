པྲ་དྷྱཽ། པྲ་དྷྱཿ། གཉིས་པ་ལ། པྲ་དྷྱཾ། པྲ་དྷཽ། པྲདྷྱཿ། གསུམ་པ། པྲ་དྷཱ། པྲདྷྱི་བྷཱིཿ། པྲ་དྷཱི་བྷི༔། རྣམ་དབྱེ་ལྷག་མ་རྣམས་ཀྱང་རིགས་འགྲོའོ། །འདི་དང་འདྲ་བ་ནི། དི་དྷཱི། སུ་དྷི་ལ་སོགས་པ་རྣམས་ཀྱང་ངོ་། །ཨ་ཙ་མང་པོ་ཞེས་པས་དབྱངས་གཅིག་པ་ལ་བཤད་མ་ཐག་པའི་བྱ་བ་མི་
འཇུག་སྟེ། དཔེར་ན། ནཱི་ཞེས་པ་ཨ་ཙ་གཅིག་པ་ཡིན་པས་པྲ་དྷཱིའི་བྱ་བ་རྣམས་མི་འཇུག། རྣམ་དབྱེ་བདུན་པའི་ངི་མ་གཏོགས་ཆོ་ག་ཐམས་ཅད་སུ་དྡྷཱིའི་སྐབས་དང་འདྲ་སྟེ། འཛིན། བློ་བཟང་། ནཱིཿ། ནཱིཿ། ནི་ཡཽ། ནི་ཡཿ། ལྟ་བུ་རིགས་འགྲེ། བདུན་པའི་ངི་ལ་ཅུང་མི་འདྲ་སྟེ། ནཱི་ལ་ངི་། དེ་རཱི་ན་དྱཱ་མྣཱི་བྷྱ༔།
ངིའི་ཨཱ་མ། ནདཱི་ཨཱ་པ། འདྲེན་པ། ནཱི་རྣམས་ལས་སོ། །རྣམ་དབྱེ་བདུན་པའི་དེ་ནི་བུད་མེད་ཀྱི་རྟགས། ན་ནཱི་མིང་ཅན་རྣམས་དང་། ཨཱ་པ་ཀྱི་མཐའི་རྣམས་དང་། ནཱིའི་སྒྲ་རྣམས་ལ་ཨཱ་མ་དུ་འགྱུར་ཏེ། ནི་ཡཱཾ། གྲཱ་མ་ཎཱི་ཡང་རྣམ་དབྱེ་བདུན་པའི་གཅིག་ཚིག་ངི་མ་གཏོགས་པྲ་བྷཱིའི་སྐབས་དང་
འདྲ། ངི་ལ་བཤད་མ་ཐག་པ་ནཱི་དང་འདྲ་སྟེ། གྲཱ་མ་ཎཱི༔། གྲཱམ་ཎྱཽ། གྲཱ་མཎྱ༔སོགས་དང་། བདུན་པ་ལ། གྲཱ་མ་ཎི་ཡཱཾ། ས་ཁཱི་དང་པ་ཏཱི་རིང་པོའི་མཐའ་ཅན་ཡང་རྣམ་དབྱེ་ལྔ་བའི་ང་སི་དང་། གྲོང་སྡེའི་ཕྲ་མ་མཁན། དྲུག་པའི་ངས་མ་གཏོགས་རྣམ་དབྱེ་གཞན་ཐམས་ཅད་ལ་ཆོ་ག་སྲ་དྷཱི་
དང་འདྲ་ཞིང་། ང་སི། ངས་གཉིས་ལས་ཁཱི་དང་། བ་ཏཱི་ལ། ང་སི་དང་ངས་བྱིན། ཁི་ཏི་དག་ལས་ཕ་རོལ་གྱི་ཨའི་ལོ་ཨ་ཞེས་པས་ང་སེ་དང་ངས་ཀྱི་ཨ་ཨུ་མཚམས་སྦྱར། ས་ཁཱུ༔། བ་ཏྱུ༔། དྲུག་པ་ཡང་གཟུགས་དང་བསྒྲུབ་ཚུལ་གཉིས་ཀ་འདྲའོ། །ལཱུ་ནཱིའི་སྒྲ་ཡང་རྣམ་དབྱེ་བདུན་པའི་
ངི་ལ་ནཱིའི་སྐབས་དང་འདྲ་ཞིང་། ངི་མ་ཡིན་པའི་རྣམ་དབྱེ་གཞན་ཐམས་ཅད་ལ་པྲ་དྷཱིའི་སྐབས་ཀྱི་ཆོ་ག་དང་འདྲའོ། །དེ་ལྟར་སྐྱེས་པའི་རྟགས་ལས་ཨཱི་རིང་མཐའ་ཅན་སྤྱིར་བཏང་དམིགས་པས་དང་བཅས་པར་རགས་པ་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་སྐྱེས། ཞིང་རྔ་བར་བྱེད། པའི་རྟགས་ཨུ་ཐུང་གི་བྱ་བ་
རྣམས་བཤད་པར་བྱ་སྟེ། ཤཾ་བྷུ་ལ། སུ། ས་ས་ཇུ་ཥར་ཞེས་དང་། ཁ་ར་དང་ཞེས་པའི་མདོས། ཤཾ་བྷུ༔། ལྷ་ཆེན་པོ། ཤཾ་བྷུ་ལ། ཨཽ། དང་པོ་དག་གི་ཞེས་པས། ཤཾ་བྷཱུ། ཤཾབྷུ་ལ། ཛས། ཛས་ལ་འང་ངོས། ཨུ་ཨོ་མཚམས་སྦྱར། ཤཾ་བྷ་ཝ༔། ཤཾ་བྷུ་ལ། ཨ་མ། ཨ་མ་ལ་སྔ་མ་འོས།
