ཚིག་གོ། །ཏ་མྱཱ་བྱཱ་ཏྭོ། དེའི་ཁྱེད་རྣམས་ལ་བསྲུང་བར་མཛོད། ཏ་སྱཱ་བྱཱ་ཏཱོ། དེའི་ངེད་རྣམས་ལ་ཡང་བསྲུང་བར་མཛོད། ཅེས་པའི་བོ་དང་ནོ་ནི་ཁྱེད་རྣམས་ལ་ངེད་རྣམས་ལ་ཞེས་གཉིས་པའི་མང་ཚིག་གོ། །སུ་ཁཾ་བོ་ད་དྱཱར། བདེ་བ་ཁྱེད་རྣམས་ལ་སྦྱིན། སུ་ཁཾ་ནོ་ད་བྱཱ་ར། བདེ་བ་
ངེད་རྣམས་ལ་སྦྱིན། ཞེས་པའི། བོ་དང་ནོ་ནི་ཁྱོད་རྣམས་ལ་ངེད་རྣམས་ལ་ཞེས་བཞི་བའི་མང་ཚིག་གོ། །བ་ཏི་བཿཧ་རིཿ། ཁྱོད་རྣམས་ཀྱི་བདག་པོ་འཕྲོག་བྱེད། བ་ཏིནཿབུདྡྷཿ། ངེད་རྣམས་ཀྱི་བདག་པོ་སངས་རྒྱས། ཞེས་པའི། བཿདང་ནཿནི་ཁྱེད་རྣམས་ཀྱི་དང་དེད་རྣམས་ཀྱི་ཞེས་དྲུག་པའི་
མང་ཚིག་གོ། །སླར་ཡང་ཀུན་ནས་བསྒྱུར་བ་དེ་རྣམས་འདོད་ན་ཡིན་གྱི་མི་བྱེད་དུ་མི་རུང་བ་མ་ཡིན་པས་འདམ་ཀ་ཅན་ནོ། །ཚིག་གི་ཐོག་མ་མ་ཡིན་པ་ལ་ཞེས་པས། ཚིག་རེ་རེའི་དང་པོ་སྟེ་ཐོག་མ་ལ། ཏྭཱཾ་མཱཾ་སོགས་རྣམ་འགྱུར་ཐོག་མ་ཁོ་ན་བྱེད་ཀྱི། ཏྭཱ་མཱ་ལ་སོགས་པ་རྣམས་སུ་མི་འགྱུར་ཏེ།
དཔེར་ན། ཏྭཱཾ་པཱ་ཏུ། མཱཾ་པཱ་ཏུ། ཞེས་པ་ལྟ་བུ་དྭཱ་མཱ་ཞེས་པར་མི་འགྱུར་རོ། །དེ་བཞིན་དུ། གཉིས་པའི་མང་ཚིག་གི་དཔེར་བརྗོད། ཡུཥྨཱ་ན་རཀྵ་ཏུ་གོ་བིནྡཿ། ཨསྨཱན་བུདྡྷཿསརྦ་དཱ་པཱ་ཏུ། ཞེས་པ་བས་ནས་སུ་མི་འགྱུར་རོ། །དེ་ལྟར་དྲུག་པའི་གཉིས་ཚིག་གི་དཔེར་བརྗོད། ཡུ་ཝ་ཡོཿཔུ་ཏྲ། ཨཱ་ཝ་ཡོཿ
པུ་ཏྲ། ཞེས ཁྱེད་ལ་བསྲུང་། པ་ཡུ་བ་ཡོཿ། བཱཾ་དང་། ཨཱ་བ་ཡོཿ། ནོ་ཞེས་བར་མི་འགྱུར་རོ། །གཞན་རྣམས་ཀྱང ཁྱོད་ལ་བསྲུང་བའི་མཛོད་ཁྱབ་འཇུག །བདག་ལ་སངས་རྒྱས་ཀྱིས་རྟག་ཏུ་སྐྱོང་བར་མཛོད། དེ་དག་གིས་རིགས་འགྲེའོ། ན་ཙ་བཱ་ཧཱ་ཧེ་བ་ཡུཀྟེ། ཙ་བཱ་ཨ་ཧ་ཧ་ཨེ་ཝ་རྣམས་ ཁྱོད་དག་གི་བུ། ངེད་དག་གི་བུ།
སྦྱོར་བ་ལམ་ཡིན་ནོ། ཙ་ལ་སོགས་པ་ལྔ་པོ་རྣམས་དང་སྦྱོར་བའི་ཚེ་ཡང་ཏེ་མེ་སོགས་སླར་ཡང་ཀུན་ནས་བསྒྱུར་པ་དེ་དག་མི་བྱེད་དོ། །དཔེར་ན། ཏ་ཝ་ཙ་མ་མ་ཙ་མེ་ཏྲཱི་ཙནྡ་ལེ་ཁེ་བ་དྡྷ་ཏེ། ཁྱོད་དང་ང་ཡི་བྱམས་འདྲིས་ནི། ཟླ་བ་ཡར་ངོ་བཞིན་དུ་འཕེལ། །ཞེས་པ་ལྟ་བུ་སྟེ། ཏེ་ཙ་མེ་ཙ་མི་ཟེར་བར། །ཧ་ཝ་
ཙ་མམ་ཙ་ཟེར་བ་ལྟ་བུའོ། །སྡེབ་སྦྱོར་བསྒྲིགས་པ་དང་ཚིག་རྐང་མི་འགྲིག་པའི་སྐྱོན་ཡོད་ན་ཙ་སོགས་དང་སྦྱོར་བའི་སྐབས་ཡིན་ཀྱང་ཏེ་མེ་སོགས་སུ་བསྒྱུར་བ་མི་འགོག་གོ་ཞེས་པ་འགྲེལ་པར་གསུང་ངོ་། །པ་ཤྱཱ་མྦཻ་ཤྱཱ་ནཱ་ལོ་ཙ་ན། ལྟ་བའི་དོན་རྣམས་ཀྱི་ཡང་མིག་གིས་མི་མཐོང་བ་ལའོ། །ལྟ་བའི་དོན་དང་
