རིམ་དྷྱེཥཥྵྱ་བཱ། བཱ་རེ་མ་ངྷྱེ་དུག་པའི་ཡང་ནའོ། །བཱར་མངྷྱེ་དག་དྲུག་པའི་དོན་ལ་མི་ཟད་བའི་བསྔུ་བ་ཚིག་ཕྲད་དུ་ཡང་ན་བྱ་སྟེ། གངྷཱ་ཡཱ་པཱ་རེའམ། ཡང་ན། བཱར་གངྐཾ། དེ་བཞིན་དུ། གངྒཱ་ཡ་མ་ངྷཱེ། ཡང་ན། མངྷྱེགངྒཾ། ཡང་ན་དེའི་སྐྱེས་བུའི་བསླུ་བ་བྱས་པས། གངྒཱ་ཡཱ་པཱརཾ། ཞེས་པ་ལ།
གངྒཱ་པ་རཾ། དེ་བཞིན་དུ། གངྒཱཡཱ་མངྷྱཾ། ཞེས་པ་ལ། གངྒཱམངྷྱཾ། ཞེས་པ་ལྟབུའོ། སཾཁྱཱཾ་བཾ་ཤྱེ་ན། རིགས་ཀྱི་གྲངས་གངྒཱ་ཕ་རོལ་ཏུ། ལའོ། རིགས་གཾ་ག་འཕོལ་ཏུ། ཀྱི་དངོས་པོ། གཾ་གཱའི་དབུས། བརྗོད་པའི། གངྒཱི་དབུས། གྲངས་ཀྱི་སྒྲ་མི་ཟད་པའི་བསྡུ་བར་ཡང་ གངྒཱའི་ཕ་རོལ། འགྱུར་ཏེ། དྭོོ་
གངྒཱའི་ཕ་རོལ། མུ་ནཱི་བཾ་ཤྱོོ། གངྒཱའི་དབུས། ཡང་ན། དྭི་ གངྒཱའི་དབུས། མུ་ནི། དེ་བཞིན་དུ། ཏྲི་མུ་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་བིདྱཱ། ཞེས་པའམ། ཡང་ན། ཏྲི་མུ་ནི་བྱ་ཀ་ར་ཎཾ་པི་དྱཱ། ནདཱིབྷིཤྩ། ཆུ་ཀླུང་རྣམས་ཀྱིའང་ངོ་། ཆུ་ཀླུང་གི་སྐྲ་བརྗོད་པའི་གྲངས་ཀྱི་སྒྲའང་མི་ཟད་པའི་བསྡུ་བར་འགྱུར་ཏེ།
སཔྟཱནཱཾ་གངྒཱ་ ཐུབ་པའི་རིགས་གཉིས། ནཱཾ་ས་མ་ཧ་རཿ། ཞེས་པ་ལ། རྣམ་ ཐུབ་པ་གཉིས། ལུང་སྟོན་པ་ཤེས་པའི་ཐུབ་པ་གསུམ། དབྱེ་དང་རྗེས་འབྲང་བསྡུས། ལུང་སྟོན་པ་ཤེས་པའི་ཐུབ་པ་གསུམ། སབྟགངྨཱ། དེ་བཞིན་དུ། དྭ་ཡོཿཡ་མུན ཡོཿསམ་ཧ་རཾ། ཞེས་པ་ལ། དྭི་ཡ་མུ་ནཾ། ཨནྱཔ་དཱརྠེཙ་
གངྒཱ་བདུན་པ་ཉལ་དུ། སཾ་ཛྙཱ་ཡཾ། ཚིག་དོན་གཞན་ལ་མིང་ལའོ། ཆུ་ཀླུང་རྣམས་ཀྱི་ཚིག་དོན་གཞན་ལ་སུང་ཀྱི་མཐའ་མིང་ལ་མི་ཟད་པའི་བསྡུ་པར་འགྱུར་ཏེ། ཨནྨཏྙ་གངྒཱ་ཡ་སྨིན་དེ། གངྒཱ་འདྲན། ཤེ། ཞེས་པ་ནུའོ་ཡ་མུན་གཉིས་ལན་དུ། ལ། རྣམ་དབྱེ་ཚིག་ ཆུ་བོ་ཡ་མུན་གཉིས། རྗེས་འབྲང་
དང་བཅས་པ་ཕྱིས། ཨུནྨ་ཏྟ་གངླཾ་དེ་ཤེ། མིང་མིན་ན། ཤྲཱི་གྷྲཾ་གངྒཱ་རྔ་ལཾ་བེ་གོ་ཡསྱ། ཞེས་པ་ལ །མི་ཟད་པའི་བསྡུ་བ་མི་འཇུག་གོ། །དེས་ན་འབྲུ་མང་གི་ ཡུལ་གང་ལ་གངྒཱ་སྨྱོན་བ་དང་། བསྡུ་བ་བྱས་པས། ཤཱི་གྷྲ་གངྒཱ་དེ་ཤཿ། ཨ་བྱ་ཡཱི་བྷཱ་བེཤ་རཏ་པྲ་བྲྀ་ཏི་བྷྱཿ། མི་ཟད་པར་གྱུར་
པ་ལང་ཤར་ཏ་སོགས་རྣམས་ཡུལ་གངྒཱའི་སྨྱོན་པ། ལས་སོ། །མི་གང་གི་མྱུར་བ་གངྒཱའི་ཆུ། །ཟད་པར་གྱུར་པའི་བསྡུ་བ་བྱས་པའི་མཐའ་ལས་ཤ་ར་ཏ་སོགས་རྣམས་ལ་ཊ་ཙའི་རྐྱེན་སྦྱིན་པར་བྱ་སྟེ། སྙོན་ལ་ཉེ་བ། ཤ་ར་དཾ་ཨུ་ མཱུང་བར་གངྒཱའི་ཡུལ། པ། ཞེས་པ་ལ། རྣམ་དབྱེ་གཉིས་པའི་རྗེས་ཕྱིས།
