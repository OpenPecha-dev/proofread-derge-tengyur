སྦྱོར་བའི་ཀྲྀཏ་ལས་ལྷག་པར་བྱེད་པ་ལའོ། ཀྲྀཏ་སོགས་ཀྱི་རྐྱེན་དུས་ཀྱི་དོན་བརྗོད་པ་དང་སྦྱོར་བ་ལས་ལྷག་པར་བྱེད་པ་སྟེ་རྣམ་དབྱེ་བདུན་པའི་དོན་ལ་དྲུག་པ་འགྱུར་ཏེ། པཉྩ་ཀྲྀ་ཏྭ་ཉཧྷཿབྷུཾ་ཀྟེ། དྭི་ར་ཧྣོ་པྷུཾཀྟེ། ཀ་རྟྀཀ་རྨ་ཎོཿཀྲྀཏིཿ། བྱེད་པ་པོ་དང་ལས་དག་ལ་ཀྲྀཏ་ལའོ། །བྱེད་པ་པོ་དང་ལས་
དག་བརྗོད་པ་ཡང་ཡིན་ལ་ཀྲྀཏ་སོགས་ཀྱི་རྐྱེན་མཐའ་ཅན་ཀྱང་ཡིན་པ་སྦྱོང་བ་ལས་དྲུག་པ་འགྱུར་ཏེ་བྱེད་པ་པོ་ལ། ཏ་བ་གྲྭ་སཾཀཱ། ཏ་པ་ག་མི་ཀཱ། ལས་ལ། སརྦ་ས་ཏྭཱ་ ཉི་མ་ལྔ་ཟན་ཟ་བར་བྱེད། ཉི་མ་གཉིས་ཟན་ཟ་བར་བྱེད། ནཱཾཔུཥྚིཾཀུརུ། ན་མཿསམནྟབུངྡྷཱ་ནཱཾ། ཨུ་བྷ་ཡ་པྲཱ་པྟོོ་ཀརྨཎི། གཉིས་
ཐོབ་པ་ལ་ལས་ལའོ། །རྣམ་དབྱེ་གཉིས་པ་དང་དྲུག་པ་གཉིས་ཀ་ཐོབ་པ་ལ་ལས་ལ་དྲུག་པ་འགྱུར་ཏེ། ཙི་ཏྲཾ་གབཱཾདོ་ཧོ་གོ་བ་ ཁྱོད་ཟ་བར་བྱེད། ཀེ་ན། ཀྟ་སྱཙབརྟ་མཱནེ། ཀྟའི་འང་ སེམས་ཅན་ཐམས་ཅད་ལ་རྒྱས་བར་མཛོད། སངས་རྒྱས་ཀུན་ལ་ཕྱག་འཚལ་ལོ། ད་ལྟར་བ་ལའོ། ཀྟའི་རྐྱེན་ད་
ལྟར་བའི་དོན་བརྗོད་པ་སྦྱོར་བ་ལ་དྲུག་པ་འགྱུར་ཏེ། སཏཱཾ་ཛྭཱཏཿ། ད་ལྟར་བ་མ་ཡིན་ན་མི་ བ་ལང་སྐྱོང་བ་ཁྲམ་རྣམས་འཇོ་བར་བྱེད། འགྱུར་ཏེ། གྲཱམཾ་གཏཿ། ནསུཾ་སཀེབྷཱབེ། མ་ནིང་ལ་དངོས་པོ་ལའོ། ཀྟའི་རྐྱེན་དངོས་པོའི་དོན་བརྗོད་པ་མ་ནིང་གི་རྟགས་སྦྱོར་བ་ལ་དྲུག་པ་ཡང་ན་འགྱུར་ཏེ། སཏཱཾ་
གཏཿ། སངྦྷིམ་ཏཿ། ཨངྷི་ཀ་ར་ཎབཱཙི་ན་ཤྩ། བདེན་པར་ཤེས་བཞིན་པ། ལྷག་པར་བྱེད་པ་པོ་བརྗོད་པ་ལའང་ངོ་། ཀྟའི་རྐྱེན་མཐའ་ཅན གྲོང་ལ་སོར། དང་སྦྱོར་བ་ལས་ལྷག་པར་བྱེད་པ་སྟེ་བདུན་པའི་དོན་ལ་དྲུག་པ་འགྱུར་ཏེ། ཨིད་མེཥཱ་མཱ་སེ་ཏཾ། ན་ལོ་ཀཱ་བྱ་ཡ་ནི་ཥྵ་ཥ་ལ་རྠ་ཏྲྀ་ཎཾ། འཇིག་རྟེན་
བདེན་པ་རྣམས་སང་། མི་ཟད་པ་བདེན་པ་ཥལ་ཏྲྀཎ་རྣམས་ཀྱི་དོན་ལམ་ཡིན་ནོ། འཇིག་རྟེན་གྱི་སྒྲ་སྦྱོར་བ་དང་མི་ཟད་པའི་སྡེ་རྣམས་དང་སྦྱོར་བ་དང་བདེན་པའི་དོན་ཅན་གྱི་རྐྱེན་དང་ཥལ་གྱི་རྐྱེན་དང་། ཏྲྀཎ་སོགས་རྐྱེན་ལྔ་པོའི་མཐའ་ཅན་རྣམས་དང་ འདི་འདི་རུ་སཱོང་། སྦྱོར་བ་ལ་དྲུག་པ་འགྱུར་བ་མ་ཡིན་ཏེ།
འཇིག་རྟེན་གྱི་སྒྲ་སྦྱོར་བ་ལ། ཙཀྲྀབཱན་སྲྀཥྟིཧཾ་རིཿ། །འཇིག་རྟེན་བྱེད་པོ་འཕྲོག་བྱེད། མི་ཟད་པ་ལ། ཨ་ལཾ་ཀརི་ཥྞུཿ། གང་བར་བྱས། བདེན་པའི་རྐྱེན་མཐའ་ཅན། ཧ་རི་དེེ་བྱཱན་ཧ་ཏབཱན། ལྷ་མིན་འཇོམས་བྱེད་འཕྲོག་བྱེད། བིཥྞུ་ནཱཧ་ཏཱཿདེེ་ཏྱཱཿ། ཁྱབ་འཇུག་གིས་ལྷ་མིན་རྣམས་བསད། ཥལ་གྱི་རྐྱེན་མཐའ། ཨི
