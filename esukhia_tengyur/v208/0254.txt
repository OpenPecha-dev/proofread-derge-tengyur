ངྷིཀཱ་ཤཱི་ཏི། ཞེས་པ་ལ། ཨཥྜཱ་ཨ་ཤཱིཏི། འགྲེལ་པར་ལུང་དྲངས་པ། པྲཱཀ་ཤ་ཏཱ་ཏ་པྲ་ཡོ་གེནཿ། བརྒྱ་ལས་མང་བ་རབ་ཏུ་སྦྱོར་བ་ལ་མ་ཡིན་ནོ། བརྒྱ་ལས་མང་བ་ཞེས་པ་ཉིས་བརྒྱ་བརྒྱད་བརྒྱ་སྟོང་ཕྲག་གཉིས་སྟོང་ཕྲག་བརྒྱད་སོགས་སྦྱོར་བ་ལ་དྭི་དང་ཨཥྚན་གྱི་མཐའ་ཨཱ་ཏ་དུ་འགྱུར་བ་
མ་ཡིན་ཏེ། དྭི་ཤ་ཏཾ། ཨཥྚ་ བརྒྱད། ཤ་ཏཾ་ལྟ་བུའོ། །ཡང་འགྲེལ་པར་ལུང་དྲངས་པ། ཀཔཱ་ལ་དང་གོ་ཕ་རོལ་ལ་ཨཥྚན་གྱིའང་ངོ་། །ཀ་བཱ་ལ་དང་གོ་ཕ་རོལ་དུ་བྱུང་ན་ཨཥྚན་གྱི་མཐའ་ཨཱཏ་དུ་འགྱུར་ཏེ། ཨཥྜཱ་ཀ་པཱ་ལཿ། ཨཥྚཱ་ག་པཿ། ཏྲེ་སྟྲ་ཡཿ། ཏྲིའི་ཏྲ་ཡས་སོ། ཏྲི་ཞེས་པ་ནི་
གྲངས་བརྗོད་ ཉིས་བརྒྱ། བརྒྱད་བརྒྱ། པ་ཕ་རོལ་ལ་ཏྲ་ཡས་ཞེས་པར་འགྱུར་རོ། །ཏྲྱངྷི་ཀཱད་ཤ། ཞེས་པ་ལ། རྣམ་དབྱེ་རྗེས་འབྲེལ་དང་བཅས་པ་ཕྱིས། ཏྲི་ཏྲ་ཡས་སུ་འགྱུར། ཏྲེ་ཡོད་ཤ། དེ་བཞིན་དུ། ཏྱ་ ཀ་དཱ་ལི་བརྒྱད། བ་ལང་བརྒྱད། ངྷི་ཀཱྀ་ཏྲཾ་ཤཏ། ཅེས་པ་ལ། ཏྲཡ་སྟྲཾ་
ཤཏ། ཏྲི་ཙ་ཏྭཱ་རིཾ། བརྒྱ་ལས་མང་བ་སྒྲོང་བ་ཡིན་ན་མི་འགྱུར་ཏེ། ཏྲི་ཤ་ཏཾ། བཅུ་ལས་ལྷག་པོ་གསུམ། ཏྲི་ས་ཧ་སྲཾ། བི་བྷཱ་ཥཱཙ་ཏྭཱརིཾ་ཤཏ་པྲབྷྲྀ་ཏོོསརྦེ་ཥཱ། ཙཏྭཱ་རིཾ་ཤཏ་སོགས་ཐམས་ བཅུ་གསུམ། ཅད་རྣམས་ཀྱི་འདམ་ཀའོ། །བཞི་བཅུ་ནས་བཟུང་བརྒྱ་སྟོང་སོགས་སྒྲོང་བ་མ་ཡིན་ སོ་གསུམ།
བལ་དི་དང་ སོ་བའི། ཨཥྚན་གཉིས་ཀྱི་མཐའ་སཱཏ་དུ་འགྱུར་པ་ སུམ་བརྒྱ། དང་ཏྲི་ཏྲ་ཡས་སུ་འགྱུར་བ་ སུམ་སྟོང་། སོགས་འདམ་ཀ་ཅན་དུ་བྱ་སྟེ། དྭཱ་ཙ་དྭཱ་རི་ཤཏ། ཡང་ན། དྭི་ཅ་ཏྭཱ་རིཾ་ཤཏ། དེ་བཞིན་དུ། ཨཥྚཱ་ཙ་ཏྲཱ་རིཾ་ཤཏ། ཡང་ན། ཨ་ཙ་ཏྭཱ་རིཾ་ཤཏ། ཏྲ་ཡཿཙ་ཏྭཱ་རིཾཤཏ། ཡང་
ན། ཏྲི་ཙ་ཏྭཱ་རིཾ་ཤཏ། ལྟ་བུ་སྙེ་ཐམས་ཅད་ལ་རིགས་འགྲེའོ། །འགྲེལ་བར་ལུང་དྲངས་པ། ཨེ་ཀའི་མཐའ་ཨཱ་ཏའམ་ཡང་ན་ཨུཀའོ། །གྲངས་དང་ ཞེ་གཉིས། སྦྱོར་བའི་ཨེ་ཀ ཞེ་གཉིས། ཞེས་པའི་མཐའ་ཨཱཏ་ ཞེ་བརྒྱད། དུ་འགྱུར་བའམ་ཡང་ ཞེ་བརྒྱད། ཞེ་གསུམ། ན་ཨུ་ཀ་ཨཱ་
ཞེ་གསུམ། གམ་འགྱུར་ཏེ། ཨེ་ཀཱ་དྷིཀཱད་ཤ། བཅུ་ལས་ལྷག་པོ་གཅིག། ཞེས་པ་ལ། ཨེ་ཀཱ་ད་ཤ། བཅུ་གཅིག། ཨེ་ཀོད་ཤ། དེ་བཞིན་དུ། ཨེ་ཀཱན་བིཾ་ཤཏི། ཡང་ན། ཉི་ཤུ་ང་གཅིག་མེད་པ་སྟེ་བཅུ་དགུའོ། ཨེ་ཀོ་ན་བིཾ་ཤ་ཏི་སོགས་སོ། །ཡང་འགྲེལ་པར་ལུང་དྲངས་པ །ཥའི་ཨུ་སྟེད་འང་ཌའོ། །
