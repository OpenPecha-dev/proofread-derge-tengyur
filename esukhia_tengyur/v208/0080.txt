ཞེས་པ་ལ། ལ་ཀྵྷཱི་ཙྪཱ་ཡཱ། ལ་ཀྵྶྭི་ཚཱ་ཡཱ། ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་ས་ནྡྷེ། སཱུ་ཏྲཱ་ཏ་ཧ་ཡ་ནྟ་སྦྷ་ཡ༔ས་མཱ་པྟཱ༔། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མཚམས་སྦྱོར་གྱི་མདོ་ལས་ཧ་ལ་མཐའ་མཚམས་སྦྱོར་རྣམས་རྫོགས་སོ།། །།ལྔ་བ་བྱེ་བྲག་ཏུ་རྣམ་བཅད་ཀྱི་མཚམས་སྦྱོར་བ་ཤད་པ་ནི། བིས་རྫ་ནཱི་ཡ་སྱ་ས།
རྣམ་པར་བཅད དཔལ་གྱི་གྲིབ་མེ། བྱའི་སའོ། །སྔ་མའི་མཐའི་རྣམ་བཅད་ནི་བར་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཕ་རོལ་ལ་སར་འགྱུར་ཏེ། །ཀ༔། ཏ་ནོ་ཏི། ཞེས་པ་ལ། ཀ་སྟ་ནོ་ཏི། བུ་དྡྷ་ཏྲཱ་ཏ། སོགས་སོ། །ཤ་རྨ་རེ་བིས་རྫ་ནཱི་ཡ། ཤར་ཕ་རོལ་ལ་རྣམ་པར་བཅད་བྱའོ། །སྔ་
མའི་མཐའི་རྣམ་བཅད་ནི་ཤར་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཕ་རོལ་ན་ཡོད་པའི་ཁར་དང་པོ་ཅན་ལ་རྣམས་བཅད་ཉིད་དུ་གནས་ཏེ། ཀ༔ཏྶ་རུ་སོགས་སོ། །བཱ་ཤ་རི། །ཅི་འཐག་པར་བྱེད། ཤར་ལ་ཡང་ནའོ། སྔ་ སངས་རྒྱས་སྦྱོང་མའི་མཐའི་རྣམ་བཅད་ནི་ཤར་སོ་སོར་བསྡུ་བ་ཕ་རོལ་ལ་ཡང་ན་ཕ་རོལ་གྱི་གཟུགས་
སུ་བསྒྱུར་ཏེ་ཡང་ན་སའོ། །ཧ་རི༔། ཤེ་ཏེ། ཞེས་པ་ལ། ཧ་རི་ཤྴི་ཏེ། ཀ༔་ཥནྡེ་ཞེས་པ་ལ། །ཀ་ཥྵ་ནྡེ། ཀ༔སཱ་དྷུ༔། ཞེས་པ་ལ། ཅི་ཡུ་བ། ཀ་སྶཱ་དྷུ༔སོགས་སོ། །འགྲེལ་བར་ལུང་དྲངས་པ། །བརྦ་རེ་ཤ་རི་བཱ་བི་ས་རྒ་ལོ་པ། །ཁར་ཕ་རོལ་ལ་ཤར་ལ་རྣམ་པར་བཅད་བྱ་ཡང་ན་དབྱིའོ། །ཁང་ཕ་
རོལ་ན་ཡོད་པའི་ཤར་དང་པོ་ཅན་ཕ་རོལ་ལ་རྣམ་བཅད་ཡང འཕྲོག་བྱེད་ཉལ། ན་དབྱི་སྟེ། ཧ་རིཿ། ཅི་དུམ་བུར་བྱེད། །སྥུ་ར་ཏི་ཞེས་པ་ལ། །ཧ་རི་ཅི་ལེགས། སྥུ་ར་ཏི། །ཧ་རི་སྶྥུར་ཏི། ཧ་རི༔སྥུ་ར་ཏི། འགྲེལ་བར་ལུང་དྲངས་པ། ཏ་དྦྲྀ་ཧ་ཏོ༔ཇ་ར་པ་ཏྱོ་ཥྩོ་ར་དེ་བ་ཡོ༔སུཊ་ཏ་ལོ་པ་ཊྩ། །ཏ་ཏ་བྲྀ་ཧ་ཏ་དག་གི་
ཀ་ར་དང་པ་ཏི་དག་ལ་ཆོམ་རྐུན་དང་ལྷ་དག་ལ་སུཊ་སྟེ་ཏ་དབྱིབ་འང་ངོ་། །ཏ་ཏ་དང་བྲྀ་ཧ་ཏ་དག་གི་ཀ་ར་དང་པ་ཏི་དག་ཕ་རོལ་ལ་རིམ་པ་བཞིན་ འཕྲོག་བྱེད་འཁོར་བར་བྱེད། འཕྲོག་བྱེད་འཁོར་བར་བྱེད། ཆོམ་རྐུན་དང་ལྷ་དག་བརྗོད་པ་ལ་སུཊ་ཨཱ་གམ་འགྱུར་ཞིང་མཐའི་ཏ་འང་དབྱི་སྟེ། ཏ་ཏ་ ཀ་ར༔།
ཞེས་པ་ལ། ཏ་སྐར༔། ཆོམ་རྐུན། བྲྀ་ཧ་ད་ པ་ཏི༔ཞེས་པ་ལ། པྲྀ་ཧ་སྤ་ཏི༔། སུར་བུ། ཀུན་[པྭོཿསྐ་ཁྤཽ་]ཙ། ཀུ་པུ་དག་ལ[་སྐ་ལྤ]་དག་ཀྱང་ངོ་། །སྔ་མའི་མཐའི་རྣམ་བཅད་ནི་ཀུ་སྟེ་ཀ་སྟེའི་དང་པོ་གཉིས། པུ་སྟེ་པ་སྡེའི་དང་པོ་གཉིས་ཕ་རོལ་ལ་རིམ་པ་བཞིན་ཀ་ལ་གཉིས་ལ་ཐྐ་དང་པ་ཕ་བ་གཉིས་
