ཤུ་བྷྲ་སོགས་ཀྱིས་ཌྷ་ཀ་མདོ་འདིས་ཨུ་ཕྱིས་ཌྷ་མ་ཡ་མུ། ཡཿ། ཀ་དྷཱུ་ལ་མི་དབྱི་སྟེ། ཀྵ་དྲ་བེ་ཡཿ། གཟར་བུ་མའི་བྱ། འགྲེལ་པར་ལུང་དྲངས་པ། བཎྜོ་ཤྩ་ན། པཎྜུའི་འང་མ་ཡིན་ནོ། ཌྷའི་རྐྱེན་ཕ་རོལ་ལ་པཎྜུའི་ཨུ་ཡང་མི་དབྱི་སྟེ། པཱ་ཎྜ་བེ་ཡཿ། སྦྱ་པོའམ་སྐྱ་སངས་ཀྱི་བུ། ཨི་ཏ་ཤྩཱ་ནི་ཉཿ།
ཨི་ཏ་ལས་ཀྱང་ཨི་ཉམ་ཡིན་པ་ལས་སོ། ཨ་ཏ་ལས་ཨི་ཉའི་ཞེས་ རིལ་བ་ཅན་གྱི་འུ། པ་ལྟ་བུའི་ ཀདྲཱུ་ལ་ཤུ་བྷྲ་སོགས་ཀྱི་པཌྷ་ཀ་འཕེལ་གཌྷ་ཨེ་ལ་ཨུ་ཏ་ཡོན་ཏན། དེ་ཕན་གྱི་རྐྱེན་ཨིཉ་བྱིན་པའི་ཨིཏ་མཐའ་མ་ཡིན་པའི་ཨི་ཡིག་མཐའ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་ཌྷཀ བཏྜུ་ལ་ཤུ་བྷྲ་སོགས་ཀྱིས་ཌྷ་ག་ཌྷ་ཨི་ཡ་ཨུ་ཡོན་ཏན། འགྱུར་ཏེ།
ཨཱ་ཏྲེ་ཡཿ། རྒྱན་ཤེས་ཀྱི་བུ། ནཻ་དྷེ་ཡཿ། ངེས་འཛིན་གྱི་བུ། ཀ་ལྱཱ་ཎྱཱ་དཱི་ནཱ་མི་ནང་། ཀ་ལྱཱ་ཎཱི་ལ་སོགས་པ་རྣམས་ཀྱི་ཨི་ནང་ངོ་། །ཀ་ལྱཱ་ཎཱི་ལ་སོགས་པ་རྣམས་ཀྱི་བུ་བརྒྱུད་ལ་ཌྷ་ཀ་འགྱུར་ཞིང་ཀ་ལྱཱ་ཎཱི་སོགས་ཀྱི་མཐའ་ཨི་ནང་དུའང་འགྱུར་ཏེ། ཀཱ་ལྱཱ་ཎི་ནེ་ཡཿ། དགེ་བ་མའི་བུ། ཛཽ་བྲྀ་ཎི་ནེ་ཡཿ། རྒན་ལྡན་མའི་བུ།
ཨེ་ཏྲི་དང་ནི་ཌྷིག་མདོ་འདིས་རྟག་འཕེལ་བཌྷ་ཨེ་ཡ་གང་གི་མི་ཏ་ལ་ཡང་ངོས་ཨི་ཕྱིས། ཛྱི་ཥྛ་ནེ་ཡཿ། ཧྲྀ་དྦྷ་ག་སི་ནྡྷྭ་ནྷེ་པཱུ་རྦ་པ་ད་སྱ་ཙ། ཧྲྀ་ད་བྷ་ག་སི་ནྡྷུ་རྣམས་མཐའ་ལ་ཚིག་སྔ་མའི་འང་ངོ་། །དེ་ཕན་གྱི་རྐྱེན་འཕེལ་བའི་རྒྱུ་མཚན་ཅན་ཕ་རོལ་ལ་ཧྲྀ་ཏ་ ཀ་ལྱཱ་ཎཱི་དང་ཛུ་བྲི་ཎཱི་ལ་མདོའི་ཌྷ་ཀ་དང་མཐའ་ཨིཎ་དང་རྗེ[ས་ཕྲེ་ལ་ནའི་ཨ་]བརྗོད་དོན།
སོགས་འདི་རྣམས་ཚིག་ཕྱི་མ་ཡིན་ཞིང་དེའི་སྔོན་དུ་ཚིག རྒན་མོའི་བུ། ཛྱི་ཥྠ་ལ་ཌྷ་ཀ་མ་[བར་མི་ནང་།] གཞན་ཡོད་ན་ཚིག་སྔ་ཕྱི་གཉིས་ཀར་གྱིའང་ཨ་ཙ་དང་པོ་འཕེལ་བར་འགྱུར་ཏེ། སཽ་ཧཱ་རྡཿ། ཡིད་གཅུགས་ཀྱི་བུ། སཽ་བྷཱ་གི་ནེ་ཡཿ། ལེགས་པའི་སྲིང་མོའི་བུ། སཽ་སཻནྡྷ་བཿ། ལེགས་བའི་སེ་ནྡྷའིཿ། འགྲེལ་པར་ལུང་
དྲངས་པ། ཨ་ནུ་ཤ་ཏི་ཀཱ་དཱི་ནཱཾ་ཙ། ཨ་ནུ་ཤ་ཏི་ཀ་ལ་སོགས་པ་རྣམས་ཀྱིའང་ངོ་། སུ་[ཧྲིད་]ལ་དེའི་བུ་བརྒྱུད་ལ་སོགས་ཨཎ་སྔ་ཕྱི་གཉིས་འཕེལ་བ། སུ་བྷ་གི་ནི་ལ་བུད་མེད་རྣམས་ལས་ཌྷ་ཀ་འོས་ཌྷ་ཀ་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་བ། སུ་སཾ་ནྡྷུ་བ་དེའི་བུ་བརྒྱུད་ལ་ངོས་ཨ་ན་མདོ་འདིས་ལྡ་ཕྱི་གཉིས་འཕེལ་བ། ཨ་ནུ་ཤ་
ཏི་ཀ་ལ་སོགས་པ་རྣམས་ཀྱིའང་དེ་ཕན་གྱི་རྐྱེན་འཕེལ་བའི་རྒྱུ་མཚན་ཅན་ཕ་རོལ་ལ་ཚིག་སྔ་ཕྱི་གཉིས་ཀར་གྱི་ཨ་ཙ་དང་པོ་འཕེལ་བར་འགྱུར་ཏེ། ཨ་ནུ་ཤ་རི་ཀ་ལ་དེའི་བུ་བརྒྱུད་ལ་འོས ཨཎ་མདོ་འདིས་སྔ་ཕྱི་གཉིས་འཕེལ་བ། ཨཱ་ཎུ་ཤཱ་ཏི་ཀཾ། [བྴ་པལྱཱི་ལ་ག་ལྡྷྲིཎི་]ལ་སོགས་པ་རྣམས་ཀྱིས་ཨི་ནང་ངོས་ཌྷ་དང་མཐའ་ཨི་ནང་མདོ་འདིས་བཱ་ཕྱི་གཉིས་འཕེལ་བ་
