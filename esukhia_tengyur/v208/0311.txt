ཏྲྀ་ཥྭ་སྲཱི་ཡཿ། མའི་ནུ་མའི་བུའམ་སྲུ་མོའི་བུ། མཱ་ཏྲྀ་ཥྭ་སེ་ཡཿ། མའི་ནུ་མའི་བུའམ་སྲུ་མོའི་བུ། རཱ་[ཛར་]ཤྭ་སུ་རཱ་དྱ་ཏ། རཱ་ཛ་ན་ཤྭ་སུ་ར་ལས་ཡ་ཏའོ། །རཱ་ཛ་ན་དང་ཤྭ་སུར་དག་ལས་བུ་བརྒྱུད་ལ་ཡ་ཏ་འགྱུར་ཏེ། རཱ་ཛ་ནྱཿ། [རྩྱ་ལབོ་ཕུ]། ཤྭ་སུ་རླཿ། མནའ་མའི་ཞ་རཱ་ཛྙོ་ཛཱ་ཏཽ་ཡ་ཏ། རཱ་ཛ་ན་ལས་རིགས་ལ་
མཱ་ཏྲྀ་ཥྭ་སྲི་ལ་ཚཎ་ན་ལ་འཕེལ་བ། ཚ་ཨཱི་ཡ་ཨི་ཀ་གི་ཨ་ཙ་ལ་ཡཎ་སུ། མཱ་ཏྲྀ་ཥཱ་སྲྀ་ལ་ ཌྷ་ཀ་ཀྲིཏ་ལའང་ངོས་འཕེལ་བ། ཌྷ་ཨེ་ཡ་མདོ་འདིས་རི་ཕྱིས། ཡ་ཏའོ། །རཱ་ཛན་ལས་རིགས་བརྗོད་པ་ལའང་ཡཏ་འགྱུར་ཏེ། [དྲ་སྱ་རཱ་ཛྙཿཇཱ་]ཏྀ། ཞེས་པའི་དོན་ལ། རཱ་ཛན་དང་ཤྭ་སུར་གཉིས་ལ་ཡ་ཏ་ཏ་ཏ་ལི་དེའི་དུས་ཀྱི་འོས་ཕྱིས། ཡ་ཏ་སུ་ རཱ་ཛ་ནྲཿ། ཀྵ་ཏྲཱ་དླྷཿ།
ཀྵ་ཏྲ་ལ་གྷའོ། །ཀྵ་ཏྲ་ལ་རིགས་དང་བུ་བརྒྱུད་དག་ལ་གྷ་འགྱུར་ཏེ། ཀྵ་ཏྲི་ཡཿ། འགྲེལ་པར་ལུང་དྲངས་པ། ཀྵ་ཏྲཱ་ཛྡྲཱ་ཧཽ་ཨིཉ། ཀྵ་ཏྲ་ལས་རིགས་བརྗོད་པ་ལ་ཡང་ རྒྱལ་པོའི་རིགས། ཨིཉ་འགྱུར་ཏེ། ཀྵ་ཏྲིཿ། ཀུ་ལཱ་ཏྦཿ། ཀུ་ལ་ལས་ཁའོ། ཀུ་ལའི་སྒྲ་འབའ་ཞིག་ རྒྱལ་རིགས། པ་དང་
ཚིག་གཞན་གྱི་ཕྱི་མར་གྱུར་པ་ལས་ཀྱང་བུ་བརྒྱུད་ལ་ཁ་འགྱུར་ཀྵ་ཏྲ་ལས་བྷ་ཨི་ཡ། རྒྱལ་རོགས་རྒྱལ་པོའི་བུ། ཏེ། ཀུ་ལཱི་ནཿ། ཨཱ་ཌྷྱ་ཀུ་ལཱི་ནཿ། ཨ་པཱུ་རྦ་པ་དཱ་ད་ནྱ་ཏ་ར་སྱཱཾ་ཡ་ཏ་ཌྷ་ཀ་ཉཽ། ཚིག་སྔོན་ན་མེད་པ་ལས། རྒྱལ་རིགས། ཡ་ཏ་ཌྷ་ཀ་ཉ་དག་ཡང་ནའོ། ཀུ་ལ་ལས་སྔོན་ན་ཚིག་གཞན་མེད་པ་ལ་
བུ་བརྒྱུད་ལ་ཡཏ་དང་ཌྷ་ཀཉ་དག་ཡང་ན་འགྱུར་ཏེ། ཀུ་ལྱཿ། ཀཽ་ལེ་ཡ་ཀཿ། ཁར་[སྦྱིན་]ཀུ་རྒྱལ་རིགས་ཀྱི་བུ། ཀུ[ར་ལ་ཁ་ལྔ་ཨཱི་]ན། ཕྱུག་པོའི་རོགས་ཀྱི་བུ། ལཱི་ནཿ། མ་ཧཱ་ཀུ་ལཱ་ད་ཉ་ཁ་ཉཽ། མ་ཧཱ་ཀུ་ལ་ལས་ཨ་ཉ་ཁ་ཉ་དག་གོ། །མ་ཧཱ་ཀུ་ལ་ལས་བུ་བརྒྱུད་ལ་ཨ་ཉ་དང་ཁ་ཉ་དག་ཡང་ན་འགྱུར་ཏེ།
མཱ་ཧཱ་ཀུ་ལཿ། རིགས་ཆེན་པོའི་བུ། མ་ཧཱ་ཀུ་ལཱི་ ཀུ[་ལ་ལ་]ཡ་ཏ། རིགས་ཀྱི་བུ། ཀུ་ལ་ལ་ཌྷ་ཀ་ཉ་ཉ་འབེལ་བའི་[དོན་]དྷ་ཨེ་ཡ། ནཿ། དུཿཀུ་ལཱ་ཏ་ཌྷ་ཀ །དུ་ཥྐུ་ལ་ལས་ཌྷ་ཀའོ། རིགས་ཀྱི་བུ། དུ་ཥྐ་ལ་ལས་བུ་བརྒྱུད་ལ་ཌྷ་ཀ་ཡང་ན་འགྱུར་ཏེ། དཽ་ཥྐུ་ལེ་ཡཿ། དུ་ཥྐུ་ལཱི་ནཿ། སྭ་སུ་ཤྪ། སྭ་སྲྀ་ལས་ཚའོ། སྲ་སྲྀ་ལས་
ལ་ཉ་འཕེལ་བའི་དོན། རིགས་ཆེན་པོའི་བུ། ཉ་འཕེལ་བའི་དོན་བ་ཨཱི་ན། བུ་བརྒྱུད་ལ་ཚ་འགྱུར་ཏེ། སྲིང་མོའི་བུ། སཱ་སྲཱི་ཡཿ། སཱ་ཡྲཾ་ཡ་ཚ་ཚ་ཨཱི་ཡ་མཚམས་སྦྱར། བྷྲཱ་ཏུ་རྦྱ་ཙཱུ། བྷཱ་ཏྲྀ་ལས་བྱ་ཏ་འང་ངོ་། བྷྲཱ་ཏྲྀ་ལས་བུ་བརྒྱུད་ལ་བྱ་ཏ་འགྱུར་ ཌྷ་ཀ་ཀང་སེལ་བའི་དོན་ཌྷ་ཨེ་ཡ། རིགས་ངན་གྱི་བུ། ཁ་ཨཱི་ན། ཞིང་འང་གི་སྒྲས་ཆ་ཡང་འགྱུར་ཏེ།
