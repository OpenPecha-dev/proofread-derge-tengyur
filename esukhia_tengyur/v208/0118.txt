སྭ་སྭཱ། སྭ་སྲྀ་བྷིཿ། ལྷག་མ་རྣམས་ཀྱང་དྷཱ་ཏྲྀའི་སྐབས་དང་འདྲའོ། །གཞན་ཥ་ཊའི་མིང་ཅན་བདུན་པོ་ཐམས་ཅད་ཆོ་ག་འདི་ཁོ་ནས་རིགས་འགྲེའོ། །དེ་རྣམས་ཀྱིས་རྀ་ཏ་མཐའི་བྱ་བ་རྣམས་བསྟན་ཟིན་ཏོ། ༈ །བུད་མེད་ཀྱི་རྟགས་ཨོ་ཡིག་མཐའ་ཅན་སྐྱེས་པའི་རྟགས་ཀྱི་
སྐབས་སུ་བཤད་པའི་གོའི་སྐབས་དང་འདྲ། ཨི་ཡིག་མཐའ་ཅན་རྣམས་རཻ་སྒྲ་དང་འདྲ། ཨཽ་ཡིག་མཐའ་ཅན་རྣམས་གླཽ་ཞེས་པའི་སྐབས་སུ་བཤད་པ་དང་འདྲ་སྟེ། མདོར་ན་ཨོ་མཐའ་དང་ཨཻ་མཐའ་ཅན་རྣམས་སྐྱེས་པའི་རྟགས་ཀྱི་སྐབས་སུ་བཤད་པའི་ཆོ་ག་ཁོ་ན་ལས་སྤྲོས་པ་
མེད་དོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏྲཱ་ད་ཛནྟ་སྟྲཱི་ལིངྒཱ་སྶ་མཱ་པྟཱཿ། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་ཨ་ཙའི་མཐའ་དུ་ད་མེད་ཀྱི་རྟགས་རྣམས་རྫོགས་སོ།། །།དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་གཞུང་འགྲེལ་གྱི་གོ་དོན་ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་
བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བས། ཨ་ཙའ་མཐའ་ཅན་བུད་མེད་ཀྱི་རྟགས་རྣམས་ཀྱི་ཆོ་ག་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པར་རགས་པ་རྣམས་རེ་ཞིག་རྫོགས་སོ།། །།ད་ནི་ཨ་ཙའི་མཐའ་ཅན་མ་ནང་གི་རྟགས་རྣམས་བཤད་པར་བྱ་སྟེ། སྤྱིར་བཏང་ཨ་ཐུང་མཐའ་
ཅན་ཛྙཱན་ཞེས་པའི་སྒྲ་ལ་དཔེར་བརྗོད་བྱ་སྟེ། ཛྙཱ་ན། སུ། ཨ་ཏོ་྅་མྲ། ཨ་ཏ་ལ་ས་ཨ་མའོ། །མ་ནིང་ཨ་ཐུང་གི་མཐའི་ཕ་རོལ་དུ་རྣམ་དབྱེ་དང་པོའི་སུ་དང་གཉིས་པའི་ཨམ་ནི་ཨམ་དུ་འགྱུར་རོ། །སུ། ཨམ། ཨམ་ལ་སྔ་མའོ་ཞེས་པས། ཛྙཱ་ནཾ། ཛྙཱ་ན། ཨཽ། ན་པུཾས་ཀཱཙྩ། མ་ནིང་
ལའང་ངོ་། །མ་ནིང་གི་རྟགས་ལས་རྣམ་དབྱེ་ ཡེ་ཤེས། དང་པོའི་ཨཽ་དང་གཉིས་པའི་ཨཽ་ནི་ཤཱི་ཞེས་པར་འགྱུར་ལ་དེ་ལ་བྷའི་མིང་ཅན་ཞེས་བརྡ་ཆད་དོ། །ཨཽ་ཤཱི། ཡ་སྱེ་ཏི་ཙ། གང་གི་ཨི་ཏ་ལའང་ངོ་། །ཨཱི་ཡིག་བྷའི་མིང་ཅན་དང་དེ་ཕན་གྱི་རྐྱེན་རྣམས་ལ་རྟགས་མཐའི་ཨ་ཡིག་དང་ཨི་
ཡིག་དབྱིའོ་ཞེས་པས་སྤྱིར་བཏང་ཨ་དབྱིར་ཐོབ་ཀྱང་དམིགས་བསལ་འགྲེལ་པར་གཞན་ནས་ལུང་དྲངས་ཏེ། ཨཽ་ཤཱིར་འགྱུར་པ་ལ་ཨ་དབྱི་བ་མ་ཡིན་ནོ། །ཞེས་པ་བྱུང་བས་ཨ་མི་དབྱི་མཚམས་སྦྱར། ཛྙཱ་ནི། ཛྙཱ་ན་ལ། ཛས། ཛཤྴ་སོཿཤིཿ། ཛས་ཤས་དག་གི་ཤིའོ། །མ་ནིང་གི་
