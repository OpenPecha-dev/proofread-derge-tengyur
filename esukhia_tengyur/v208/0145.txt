དོར་ནཱ་དེ་ཤེ། ཡུཥྨད་ཨ་སྨད་དག་གི་ཀུན་ནས་བསྒྱུར་བ་ལའོ། སུ་ཨམ་དུ་བསྒྱུར་བ་ལྟ་བུ་མ་ཡིན་པའི་རྣམ་དབྱེ་ཧ་ལ་ཐོག་མ་ཅན་གཞན་དུ་མ་འགྱུར་བ་རྣམས་ཕ་རོལ་ལ་ཡུཥྨ་ད་ཨ་སྨད་དག་གི་མཐའ་ཨཱ་རུ་འགྱུར་ཏེ། ད་ཨཱུ། ཨ་ཏ་ཀྱི་ཡོན་ཏན་ལའོ་དང་། ཨ་ཀི་གི་འདྲ་བ་ལ་རིང་པོ་འོས། ཡུཝཱ་
བྷྱཱཾ། ཨཱ་ཝཱ་བྷྱཱཾ། བྷིས་ལ། ཡུཥྨ་ད་ཨ་སྨད་དག་གི་ཀུན་ནས་མ་བསྒྱུར་བ་ལའོས། ད་ཨཱ། ཨ་ཀ་གི་འདྲ་བ་ལ་རིང་པོ་འོས། ཡུཥྶཱ་བྷིཿ། ཨསྨཱ་བྷིཿ། ཡུཥྨད་ཨ་སྨད་ལ། ངེ་། དེ་དང་པོ་དག་གི་ཨམ་འོས། དེ་ཨམ། ཏུ་བྷྱ་མ་ཧྻཽད་ཡི། ངེ་ལ་ཏུ་བྷྱ་ཧྻ་དག་གོ། །རྣམ་དབྱེ་བཞི་པའི་དེ་ཕ་རོལ་ལ་ཨད་
བོར་བའི་ཡུཥྨ་ནི་ཏུ་བྷྱ་དང་ཨད་བོར་བའི་ཨསྨྲ་ནི་མཧྻ་ཞེས་པར་འགྱུར་ཏེ། ཏུ་བྷྱ་དང་མ་ཧླ་ར་བསྒྱུར། ལྷག་མ་ལ་དབྱི་འོས་ད་ཕྱིས། ཨཏ་ཀྱི་ཡོན་ཏན་ལའོ་དང་། ཨམ་ལ་སྔ་མ་འོས། ཏུ་བྷྱཾ། མཧྻཾ། བྷྱཱཾ་གསུམ་པའི་སྐབས་དང་འདྲ་སྟེ། ཡུཝཱ་བྷྱཱཾ། ཨཱ་ཝཱ་བྷྱཱཾ། ཡུཥྨད། ཨ་སྨད་ལ་བཞི་པའི་བྷྱས། བྷྱ་སོ་
