བ་རྗོད་པ་ལ་ལས་རབ་ཏུ་སྦྱོར་བ་ཞེས་མིང་འདོགས་བྱེད་དོ། །ཨཱ་ང་མརྻྻ་དཱ་བཙ་ནེ། །ཨཱང་ལས་མཚམས་བརྗོད་པ་ལའོ། །ཨཱང་ལས་མཚམས་ལྟ་བུ་བར་མཚམས་བཟུང་བའི་དོན་བརྗོད་པ་ལས་རབ་ཏུ་སྦྱོར་བ་ཞེས་དགོས་པའི་དབང་གིས་མིང་འདོགས་བྱེད་དོ། །པཉྩ་མྱ་ཨཱང་བ་རི་བྷིཿཨ་

པྲ་ཏི་འགྲན་པ་དང་བརྗེ་བ་དག་ལའོ། །པྲ་ཏི་སྦྱོར་བ་འགྲན་པ་དང་བརྗེ་བ་དག་འཇུག་པའི་ཚེ་ལས་རབ་ཏུ་བརྗོད་པ་ཞེས་མིང་འདོགས་བྱེད་ངོ་། །པྲ་ཏི་ནི་ངྷི་པྲ་ཏི་རཱ་ནེ་ཙ་ཡསྨཱཏ། གང་ འཁོར་བ་ལས་འཕྲོག་བྱེད་བྲལ། འཁོར་བ་ལས་སོ་སོར་འཕྲོག་བྱེད་བྲལ། འཁོར་བ་ལས་གྲོལ་བའི་བར་དུ། ལས་འགྲན་པ་དང་བརྗེ་
པ་ལའང་ངོ་། །པྲཏྭཱི་སྦྱོར་བ་གང་ལས་འགྲན་པ་དང་བརྗེ་བའི་དོན་བརྗོད་པ་གོང་དུ་ལས་རབ་ཏུ་བརྗོད་པ་ཞེས་མི་འདོགས་བྱས་པ་དེ་ལ་རྣམ་དབྱེ་གཉིས་པའི་དོན་ལ་ལྔ་པ་འགྱུར་ཏེ། ཨ་བྷི་མ་ནྱུ་ར་རྫུན་ཏཿཔྲ་ཏི། མཱ་ཥཱ་ན་སྨི་ཏི་ལེ་བྷྱཿཔྲ་ཏི་ཡཙྪ་ཏི། ཨ་ཀ་རྟརྻྲེ་ཎེ་པ་ཉྩ་མཱི། །བྱེད་པ་བོ་མེད་པ་ལ་བུ་ལོན་
ལ་ལྔ་པའོ། །བུ་ལོན་གྱི་རྒྱུ་ལས་བྱེད་པ་པོ་མེད་པ་དེ་ལ་ལྔ་པ་འགྱུར་ཏེ། ཤ་ཏཱ་དྭ་རྔཱཿ། བི་བྷཱ་ཥཱགུཎེ་ཉ་སྟཱི་ཡཱཾ། བུད་མེད་མ་ཡིན་པ་ལ་ཡོན་ཏན་ལ་ཡང་ནའོ། ཆོ་གའི་ཡན། སྲིད་སྦྲུ་བ་ལ་སོ་སོར་ལྡན་པར་བྱེད། སྲན་མ་འདི་དང་ཧི་ལ་རྣམས་སོ་སོར་བརྗེ། ལག་ཏུ་གྱུར་པའི་ཡོན་གྱི་རྒྱུ་ལ་བུད་མེད་ཀྱི་རྟགས་
མ་ཡིན་པ་ལ་ལྔ་བ་ཡང་ན་འགྱུར་ཏེ། ཛཱཌྲྱཱ་དྭདྔྷཿ། །ཛཱ་ཌྱཱེ་ན་པ་ངྔྷཿ། ཡོན་ བརྒྱ་ཐམས་པས་བཅིངས། །ཏན་གྱི་ཡན་ལག་ཡིན་ན་མི་འགྱུར་ཏེ། །དྷ་ནེ་ན་ཀུ་ལཾ། །བུད་མེད་ཀྱི་རྟགས་ཡིན་ན་མི་འགྱུར་ཏེ། །བུ་དྔྷྱྭམུ་ཀྟཿ། པྲྀ་ཐ་གྦི་ནཱ་ནཱ་ནཱ་བྷི་སྟྲྀ་དཱི་ཡཱ་ནྱ་ཏ་ར་སྱཱཾ། །པྲྀཐག་ བི་ནཱ་ ནཱ་ནཱ་རྣམས་ཀྱི་གསུམ་པ་ཡང་ནའོ།
