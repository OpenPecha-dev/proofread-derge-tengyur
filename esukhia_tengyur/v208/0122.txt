ཨོས་ར་དྲ །སུ་པ་ལ། ཤཱིརྵེ་ཥུ། ཤཱིརྵ་སུ། ཧྲ་སྲོན་པུཾས་ཀེ་པྲུ་ཏི་པ་དི་ཀ་སྱ། མ་ནིང་ལས་མིང་གི་ཐུང་ངུའོ། རྟགས་མཐའ་ཨ་ཙ་རིང་པོ་རྣམས་ནི་མ་ནིང་གི་རྟགས་ཀྱི་སྐབས་སུ་ཐུང་ངུར་འགྱུར་ཏེ། དཔེར་ན། ཤཱི་པཱ་ཞེས་པ་ཨཱ་རིང་མཐའ་ཅན་ཡིན་ཀྱང་བཤད་མ་ཐག་པའི་མདོས་ཐུང་ངུར་འགྱུར།
ཤཱེ་པཾ། ཤཱེ་པེ ཤྲཱི་པཱ་ནི་ལྟ་བུ་སྟེ་ཨཱ་རིང་ཐམས་ཅད་ཨ་ཐུང་ངུ་འགྱུར་ཏེ། རྣམ་དབྱེ་དང་ཕྲད་པའི་བྱ་བ་ཐམས་ཅད་ཨ་ཐུང་མཐའ་ཅན་སྤྱིར་བཏང་ཛྙཱ་ནའི་སྐབས་སུ་བཤད་དང་འདྲའོ། དེ་རྣམས་ཀྱིས་ དཔལ་སྤྱོད། མ་ནིང་གི་རྟགས་ལས་ཨཱ་རིང་ཨ་ཐུང་གི་བྱ་བ་རྣམས་བཤད་ཟིན་ཏོ། ༈ །ད་ནི་
མ་ནིང་གི་རྟགས་ཨི་རིང་ཐུང་གི་མཐའ་ཅན་རྣམས་བཤད་པར་བྱ་སྟེ། བཱ་རི། ལ། སུ། སྭ་མོ་རྣ་པུཾས་ཀཱང་། མ་ནིང་ལས་སུ་ཨ་མ་དག་གིའོ། མ་ནིང་གི་རྟགས་ལས་རྣམ་དབྱེ་དང་པོའི་སུ་དང་གཉིས་པའི་ཨ་མ་ནི་འཇིག་གོ། །ཞེས་པས་སུ་འཇིག པཱ་རི། པཱ་རི་ལ། ཨཽ། མ་ནིང་ལ་
འང་ངོས་ཨཽ་ཤཱི། ཨི་ཀོ་ཙི་བི་བྷ་ཀྟཽ། རྣམ་དབྱེ་ཨ་ཙ་ལ་ཨི་ཀ་གི་འོ། ཆུ། མ་ནིང་གི་རྟགས་ཨི་ཀ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཡོང་ལ་རྣམ་དབྱེ་ཨ་ཅ་ཐོག་མ་ཅན་བྱུང་ན་ནུ་མ་ཨཱ་ག་མ་སྟེར་རོ། །བཱ་རི་ཎཱི། ཛས་ལ། ཛས་ཤས་དག་གི་ཤི། རྣམ་དབྱེ་ཨ་ཙ་ལ་ཞེས་པས་ནུ་མ། སམྦུ་དྦྷི་མ་ཡིན་པའི་ཞེས་
པས་ཨུ་པདྷཱ་རིང་པོ། བཱ་རཱི་ཎི། སཾ་བུ་དྦྷི་ལ། ཐུང་ངུའི་ཡོན་ཏན་ནོ་ཞེས་དང་། ཐུང་ངུ་དང་ཨེང་ཞེས་པས་སཾ་བུདྡྷི་ཕྱིས། ཧེ་བཱ་རེ། ཧེ་བཱ་རི་ཎཱི། ཧེ་བཱ་རཱི་ཎི། རྣམ་དབྱེ་གཉིས་པ་རྣམས་དང་པོ་དང་འདྲ། ཊཱ་ལ། བཱརི་ཎཱ། བྷྱཱཾ་ལ། བཱ་རི་བྡྷཱཾ། བྷིཿལ། བཱ་རི་བྷིཿ། ངེ་ལ། བཱ་རི་ཎེ། བྷྱཱཾ་འདྲ།
བྷྱས་ལ། བཱ་རེ་བྷྱཿ། ང་སི་ལ། བཱ་རི་ཎཿ། བྷྱཱཾ་བྷྱཿའདྲ། ང་ས་ལ། བཱ་རི་ཎཿ། ཨོས་ལ། བཱ་རེ་ཎོཿ། བཱ་རཱེ་ཎཱ། ངི་ལ། བཱ་རི་ཎི། ཨོས་འདྲ། སུ་པ་ལ། བཱ་རི་ཥུ། ཨ་ནཱ་དི་ཞེས་པའང་རྣམ་དབྱེ་ཊཱ་ཡན་ཆད་བཱ་རི་དང་འདྲ་སྟེ། དང་པོ་རྣམས་ལ། ཨ་ནཱ་དེ། ཨ་ནཱ་དེ་ནཱི། ཨ་ནཱ་དཱི་ནི། བོང་
ཚིག་ལ། ཧེ་ཨ་ནཱ་དེ། ཧེ་ཨ་ནཱ་དི་ནཱི། ཧེ་ཨ་ནཱ་དཱི་ནི། གཉིས་པ་རྣམས་ཀྱང་དང་པོ་དང་འདྲ། ཨནཱ་དི། ཊཱ། ཏྲྀ་ཏཱི་ཡཱ་དི་ཥུ་བྷཱ་ཥི་ཏ་པུཾ་སྐཾ་པུཾ་བངྒཱལ་ཝ་སྱ། གསུམ་པ་སོགས་རྣམས་ལ་སྐྱེས་པ་བཞིན་ག་ལ་ཝའིའོ། །རྟགས་ཨི་ཀ་སོ་སོར་བསྡུ་བའི་མཐའ་ཅན་ལས་ཕ་རོལ་
