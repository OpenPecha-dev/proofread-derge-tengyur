དྭི་ཛ་སྱཀུརྦ་ཎཿ། སཏཱཾ་ཀུརྦཎཿ། མི་ཟད་པ་ལ། སཏཱཾཀྲྀཏྭཱ། ཏ་བྱ་ལ། །ས་ཏཱཾཀརྟབྱཾ། ཡང་འགྲེལ་པར། ཏ་བྱ་གྱི་རྐྱེན་ཅན་དང་སྦྱོར་བ་ལས་ཀྱང་སྐྲ་བས་འགར་དོན་གཅིག་པར་ཡང་འགྱུར་རོ་ཞེས་བཤད་པས། ངས་དྭི་ཛ་ཀརྟ་བྱཾ། མཉམ་པའི་ལྷག་པའི་བྱེད་པའི་དོན་ལ། རཱ་ཛྙོ་ཉན་ལ་
གཉིས་སྐྱེས་ཀྱིས་བྱས། བདེན་པ་རྣམས་ཀྱིས་བྱས། བདེན་པ་རྣམས་ཀྱིས་བྱིས་ནས། བདེན་པ་རྣམས་ཀྱི་བྱེད། སྱཾ། ཀྟེན་ཙ་པཱུ་ཛཱཡཱཾ། ཀྟའི་ཡང་མཆོད་པ་ལའོ། །ཀྟའི་རྐྱེན་ཅན་ཡང་མཆོད་པའི་དོན་སྦྱོར་བ་ལ་དྲུག་པའི་མཐའ་དོན་ གཉིས་སྐྱེས་ཀྱི་བྱེད། གཅིག་པར་མི་འགྱུར་ཏེ། རཱ་ཛཱི་རྐྱལ་བོའི་ལེ་ལོ་མེད།
མཏོ་བུངྔྷཿཔཱུ་ཛིཏཿ། ཀརྨཎི་ཙ། ལས་ལའང་ངོ་། །ལས་ཀྱི་དོན་གཉིས་ཡོད་པ་དང་སྦྱོར་བ་ལ་དྲུག་པའི་མཐའ་དོན་གཅིག་པར་འགྱུར་བ་མ་ཡིན་ཏེ། ཙཱར་ཡ་ཏིག་བཱཾ་དོཧོ་གོ་པེ་ན། ཏྲྀ་ཛ་ཀཱབྷྱཱཾ་ཀ་རྟ་རཾ། ཏྲྀཙ་ཨ་ཀ་དག་ལས་བྱེད་པ་ལའོ། བྱེད་པའི་དོན་གྱི་རྐྱེན་ཏྲྀ་ཙ་ཨ་ཀ་དག་སྦྱར་པའི་
རྒྱལ་པོ་རྣམས་ཀྱི་ལུགས་སངས་རྒྱས་མཆོད། དྲིཙ། རྟགས་ལས་དྲུག་པའི་མཐའ་དོན་གཅིག་པར་མི་འགྱུར་ཏེ། བྷ་བ་ཏཿཤཱི་མཱན། པ་ཙ་ཀསྕ་ཙ་ཏུརཿ། ཡང་ ཕྱུགས་རྫིས་བའང་རྣམས་འཇོ་ཞིང་འཚོ་བར་བྱེད། འགྲེལ་བར་བདག་གི་སྒྲ་སྦྱོར་བལ་ཏྲྀ་ཙ་མཐའ་ཅན་ཡིན་ཀྱང་ཚིག་སྡུད་ཀྱི་བྱ་བ་འཇུག་གཞེས་
བཤད་པས། ཏྲིཙ་ བྷུ་བཿབྷརྟཿ། ནི་ཏྱཀྲཱི་ཌཱཛཱི་བི་ཀ་ཡོཿ། རྟག་ཏུ་རྩེད་མེ་འཚོ་བ་དག་ ཁྱོད་ཀྱི་དཔལ་ལྡན། འཚོད་པའི་གྲུང་པོ། ལས་སོ། །རྩེད་མོ་དང་འཚོ་བའི་དོན་བརྗོད་པ་ཅན་དང་སྦྱོར་བ་ལ་དྲུག་པའི་མཐའ་རྟག་ཏུ་དོན་གཅིག་པར་འགྱུར་ཏེ། རྩེད་མོལ། ཨུངྔཱ། ལ། །ངས་ཀ་པུཥྤ་
བྷཾཛཾཀཿ། འཚོ་བའི་དོན་སའི་བདག་པོ། ལ། ངས་ཙི་ཏྲ་ལེ་ཁ་ཀཿ། བཱུརྦཱཔརཱ་ངྟ་རོ་ཏྟ་ར་མེཀ་དེ་ཤི་ནེེཀཱངྷི་ཀ་རཎེ། པཱུརྦ་ ཨ་པར་ ཨ་ངྷར་ ཨུ་ཏྟར་རྣམས་ལུས་གཅིག་གིས་གཅིག་ལྷག་པར་བྱེད་པ་ལའོ། །པཱུརྦ་སོགས་དེ་རྣམས་ལུས་གཅིག་གིས་གཅིག་ མེ་ཏོག་གི་དོ་ཤལ་བཅང་།
ལྷག་པར་བྱེད་པ་སྦྱོར་བ་ལ་དྲུག་པའི་མཐའ་དོན་གཅིག་རི་མོ་འདྲི། པར་འགྱུར་ཏེ། པཱུརྦཾཀཱ་ཡསྱ། ལུས་ཀྱི་སྙིང་། ཞེས་པ་ལ་དྲུག་པའི་རྗེས་བསྡུས། པཱུརྦ་ཀཱ་ཡཿ། ལུས་ཀྱི་སྟོད། དེ་བཞིན་དུ། ཨ་པར་ཀཱཡཿ། ལུས་ཀྱི་གཡས། ཨངྷ་ར་ཀཱ་ཡཿ། ལུས་ཀྱི་སྨད། །ཨུཏྟ་ར་ཀཱ་ཡཿ། འགྲེལ་པར།
