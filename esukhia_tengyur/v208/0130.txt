ཨ་ཏུ། ཨས་མཐའི་བྱིངས་མ་ཡིན་པའི་འོ་ཞེས་པའི་འགྲེལ་པར། སཾ་བུདྡྷི་ཡིན་པའི་སུ་ཕ་རོལ་ལ་ཞེས་པ་བྱུང་ཡང་། ཨ་ཝ་ཡཱ་ཛ། ཤྭེ་ཏ་བཱ་ཏ། པུ་རོ་ཌཱས་ཞེས་པ་འདི་རྣམས་ཀྱི་སཾ་བུ་དྡྷི་ལ་ཨུ་པ་དྷཱ་རིང་པོར་འགྱུར་རོ་ཞེས་པ་བྱུང་བས་གཉིས་ཀ་འཇུག་པས་འདམ་ཀ་ཅན་དུ་བཟུང་ངོ་། །
ཧེ་ཤྭེ་ཏ་བཱཿ། ཧེ་ཤྭེ་ཏ་བཿ། ཧེ་ཤྭེ་ཏ་བཱ་ཏཽ། ཧེ་ཤྭེ་ཡ་བཱ་ཧཿ། ཨུ་ཀྠ་ཤཱ་ས། སོགས་ཀྱང་ཤྭེ་ཏ་བཱ་ཧ་དང་འདྲ་སྟེ། སུ་ལ། ཌས། ཊི་ཕྱིས། ཨུ་པ་དྷཱ་རིང་པོ། ཨུ་ཀྠ་ཤཱཿ། ཨཽ་དང་ཛ་ས་ལ། ཨུ་ཀྠ་ཤཱ་སཿ། གཉིས་པ་རྣམས་ལ། ཨུ་ཀྠ་ཤཱ་སཾ། ཨུཀྠ་ཤཱ་སཽ། ཨུ་ཀྠ་
ཤཱ་སཿ། ཊཱ་ལ། ཨུ་ཀྠ་ཤཱ་སཱ། བྷྱཱཾ་ལ། ཌས་ཊི་ཕྱིས། མཚམས་སྦྱར། ཨུ་ཀྠ་ཤོ་བྷྱཱཾ། ཨུ་ཀྠ་ཤོ་བྷིཿ། བཞི་པ་རྣམས་ལ། ཨུ་ཀྠ་ཤཱ་སེ། ཨུ་ཀྠ་ཤོ་བྷྱཱཾ། ཨུ་ཀྠ་ཤོ་བྷྱཱཿ། ང་སི་ལ། ཨུ་ཀྠ་ཤཱ་སཿ། བྷྱཱཾ་བྷྱས་འདྲ། དྲུག་པ་རྣམས་ལ། ཨུ་ཀྠ་ཤཱ་སཿ། ཨུ་ཀྠ་ཤཱ་སོཿ། ཨུ་ཀྠ་ཤཱ་སོཿ། ཨུ་ཀྠ་ཤཱ་སཱཾ། ངི་ལ།
ཨུ་ཀྠ་ཤཱ་སི། ཨོས་འདྲ། སུ་པ་ལ། ཨུ་ཀྠ་ཤཱཿསུ། ཨུ་ཀྠ་ཤཱ་སྶུ། བོད་ཚིག །ཧེ་ཨུ་ཀྠ་ཤཱཿ། ཧེ་ཨུ་ཀྠ་ཤཿ། ཧེ་ཨུ་ཀྠ་ཤཿ། ཧེ་མུ་ཀྠ་ཤཱ་སཽ། ཧེ་ཨུ་ཀྠ་ཤཱ་སཿ། ཨུ་ཝ་ཡཱ་ཛ། བུ་རོ་ཌཱ་ས་དག་ཀྱང་འདི་དང་འདྲའོ། ཏྭ་རཱ་སཱ་ཧ། སུ། ཧའི་ཌྷ་འོས། ཧ་ཌྷ། ས་ཧེ་སྶཱ་ཌྷ་སྶཿ། སཱ་ཧའི་སཱ་ཌྷའི་སའོ། །
སཱ་ཧ་ཞེས་པའི་རྟགས་སཱ་ཌྷ་ཞེས་པར་འགྱུར་བའི་སྔ་མའི་ས་དེ་ཥ་ཡིག་ཏུ་འགྱུར་ཏེ། ཥ་ས་ཧ་ལ་དཱི་པ། ཞེས་པས་སུ་ཕྱིས། གནས་མེད་པ་ལ་འོས་ཌྷ་ཌའམ། ཊ། ཏཱ་རཱ་ཥཱ་ཌ། ཏྭ་རཱ་ཥཱ་ཊ། ཨཽ་དང་ཛ་ས་ལ། བརྒྱ་བྱིན་གྱི་རྟ། ཏྭ་རཱ་སཱ་ཧཽཿ། ཏྭ་རཱ་སཱ་ཧཿ། གཉིས་པ་རྣམས་ལ། ཏྭ་རཱ་སཱ་ཧཾ། ཏྭ་རཱ་
སཱ་ཧཽ། ཏྭ་རཱ་སཱ་ཧཿ། ཊཱ། ཏྭ་རཱ་སཱ་ཧཱ། བྷྱཱཾ་བྷི་ས་ལ། ཏྭ་རཱ་ཥཱ་ཌྦྷྱཱཾ། ཏྭ་རཱ་ཥཱ་ཌྦྷིཿ། དེ་ལ། ཏྭ་རཱ་སཱ་ཧེ། བྷྱཱཾ་བྷྱ་ས། ཏྭ་རཱ་ཥཱ་ཌྦྷྱཱཾ། ཏྭ་རཱ་ཥཱ་ཌྦྷྱཿ། ང་སི་ལ། ཏྭ་རཱ་སཱ་ཧཿ། བྷྱཱཾ་བྷྱ་ས་འདྲ། དྲུག་པ་རྣམས་ལ། ཏྭ་རཱ་སཱ་ཧཿ། ཏྭ་རཱ་སཱ་ཧོཿ། ཏྭ་རཱ་སཱ་ཧཱཾ། ངི་ལ། ཏྭ་རཱ་སཱ་ཧི། ཨོས་འདྲ། སུ་པ་ལ།
