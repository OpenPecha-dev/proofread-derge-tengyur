བྲ་ཏྱ་ཡོ་ཏྟ་ར་པ་ད་ཡོ་ཤྩ། རྐྱེན་དང་ཚིག་ཕྱི་མ་དག་ལས་ཀྱང་ངོ་། །ཡུ་ཥྨ་ད་དང་ཨ་སྨད་གཉིས་ཀྱི་ད་བོར་བའི་ཡུ་ཥྨསྨ་ཨ་སྨ་དག་ནི་རྐྱེན་ཕ་རོལ་དང་ཚིག་གཞན་ཕ་རོལ་ལས་ཀྱང་རིམ་པ་བཞིན་ཏྭ་དང་མ་དག་ཏུ་འགྱུར་ཏེ། ཏྭ་དཱི་ཡཿ། ཡུ་ཥྻ་ད་ཨ་སཱ་ད་ལ། མ་དཱི་ནཡཿ། བདག་
ལས་སྐྱེས་པའམ་བདག །ཚིག་ཕྱི་མ་ལ། ཏྭ་ཏྦུ་ཏྲཿ། བ་དྦུ་ཏྲཿ། བྷ་ཏ་ཥྛ་ཀ་ཚེ་སཽ། །བྷ་བཏ་ལས་ཋ་ཀ་ཞས་དག་གོ། །བྷ་བ་ཏ་ལས་ལྷག་མ་ལ་ཋ་ཀ་དང་ཚས་དག་འགྱུར་ཏེ། [ཚ་ཨཱི་ལ] བྷཱ་བཏྐཿ། བྷ་བཏ་ལ་ཚས་སི་ཏི་ཙ། ས་འགྲོ་བ་བྱོད་[པ་ཐ་སྐྱེས་པའམ་ཀྱོདཱ]། [ཉྱད་ལ་སོགས་བཞེས་པས་ཚ་ཤྲ་སྨ་ཙ་དང་ཨ་སྨ་བུ་འགྱུར]
ལའང་ངོ་། །རྐྱེན་ས་འགོ་བ་དང་ ཁྱོད་ཀྱི་བུ། བདག་གི་བུ། ཡ་ལ་སོགས་པ་ཕ་རོལ་ཏུ་བྱུང་ན་རྐྱེན་སྟེར་ཡུལ་གྱི་མང་དེ་ལ་ཚིག་ཅེས་མིང་འདོགས་བྱེད་དོ། །ཁྱོད་ལས་སྐྱེས་པའམ་ཁྱོད་ལས་འོངས་པའམ་ཁྱོད། བྷ་བ་ཏ་ལ་ཋ་ཀ་ཀ་འཕེལ་བ་ཨིས་ཨུས་ཨུཀ་ཏ་སོགས་ཀྱིས་ཋཀ །དགོས་པ་ནི་མཐའི་ཛྙ་ལ་རྣམས་ཀྱི་ཛ་ཤ་རྣམས་སོ།
ཞེས་པས། ཏ་དར་འགྱུར་བའི་ཆེད་ཡོན་ནོ། བྷ་པ་དཱི་ཡཿ། ཨི་ཉ་ཤྩ། ཨི་ཉ་ལས་ཀྱང་ངོ་། །རིགས་ཀྱི་དོན་དུ་བྱིན་པའི་ཨི་ཉའི་མཐའ་ལས་ལྷག་མ་ལ་ཨཎ་འགྱུར་ཏེ། ངཱ་ཀྵཽ་བྷ་བཿ། དཱ་ཀྵཽ་ཛ་ཏཿ། ཞེས་པའི་དོན་ལ། ཨི་ཉ་ཨཎ་ དཱ་ཀྵཿ། ཀྵ་ཏྲཿ། ཨིཉ་ཨཎ། ག་ཧཱ་དི་[བྦྱཤྯ]། ག་ཧ་ལ་
[བྷ་བཏ་ཝ་ཚས་ས་ཏ་དར་]འགྱུར་བའི་དོན་ཚ་ཨཱི་ཡ། ཁྱོད་ལས་སྐྱེས་སམ་ཁྱོད། སོགས་པ་རྣམས་ལས་ཀྱང་ངོ་། །ག་ཧ་ལ་སོགས་པ་རྣམས་ལས་ཀྱང་ལྷག་མ་ལ་ཚ་འགྱུར་ཏེ། ག་ཧཱི་ཡཿ། མཁས་པའི་རིགས། མཁས་པའི་རིགས་ལས་སྐྱེས་པ། མཁས་པའི་རིགས་ལས་སྐྱེས་པའམ་མཁས་པའི་རིགས། རྒྱལ་པོའི་རིགས་ལས་སྐྱེས་པའམ་རྒྱལ་པོའི་རིགས།
ཛ་ན་པར་ཡོཿཀུ་ཀ་ཙ། ཛན་པར་དག་ལས་ཀུ་ཀའང་ངོ་། །ག་ཧ་སོགས་ཀྱི་ནང་ཚན་ཛན་དང་པར་དག་ལས་རྐྱེན་ཚ་ ཁང་པ་རུ་སྐྱེས་པའམ་ཁང་པ་ནས་འོངས་བའམ་ཁང་པ། ཚ་ཨཱི་ཡ། ཕ་རོལ་ལ་ཀུ་ཀ་ཨཱ་ག་མ་འགྱུར་ཏེ། ཛ་ན་ཀཱི་ཡི། མིའམ་མི་ལས་སྐྱེས་པ། པར་ཀཱི་ཡཾ། འགྲེལ་པར་ལུང་དྲངས་པ།
ད་བ་སྱ་བཱ། དེ་བའི་ཡང་ནའོ། །དེ་བ་ལས་ཚ་ཕ་རོལ་ལ་ཀུ་ཀ་ཨཱ་ག་མ་ཡང་ན་འགྱུར་ཏེ། དེ་བ་ཀཱི་ཡཾ། ལྷའམ་ལྷ་ལས་སྐྱེས་པ། དེ་བཱི་ཡཾ། ཡུ་ཥྨད་དོ་ར་ནྱ་ཏ་ར་སྱཾ་ཁ་ཉྩ། ཡུ་ཥྨད་ ཛ་ན་དང་པར་ལ་ཚ་ཚ་ཨཱི་ཡ་ཀུ་ཀ་ཨཱ་གམ་ཀ་གནས་ཀྱི་དོན་ཨུ་བརྗོད་དོན། ཕ་རོལ་ལམ། ཕ་རོལ་ཏུ་སྐྱེས་པའམ་ཕ་རོལ་ནས་འོངས་པ།
