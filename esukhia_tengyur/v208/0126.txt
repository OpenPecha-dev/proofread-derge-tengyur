ཧེ་པྲ་རི། ཧེ་པྲ་རི་ཎཱི། ཧེ་པྲ་རཱི་ཎི། གཉིས་པ་རྣམས་འདྲ། ཊཱ་སོགས་ལ་མ་ནིང་རང་ལུགས་ལྟར་ན་ནུམ་སྟེར། སྐྱེས་པའི་ལུགས་ལྟར་ན་ཨཻ་ཨཱ་ཡར་འགྱུར་ཏེ། རཻ་ཞེས་པའི་སྐབས་སུ་བཤད་པ་དང་འདྲའོ། །སུ་ནོ་ཞེས་པ། སུ་ནུ་ཞེས་པར་འགྱུར། ཨོ་ཨཱ་ཡ་དང་ཨོ་ཨཱ་ཝའི་ཁྱད་མ་
གཏོགས་པྲ་རཻ་སྐབས་དང་འདྲའོ། །ཨི་ཏི་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་སྱ་སཱུ་ཏཱ་ད་ཇནྷ་ན་པུཾ་ས་ཀ་ལིངྒཱ་ནི་སམཱ་པྟཱཿ། ཞེས་པ་པཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎའི་མདོ་ལས་ཨ་ཙའི་མཐའ་ཅན་མ་ནིང་གི་རྟགས་རྣམས་རྫོགས་སོ། །དེ་རྣམས་ཀྱིས་བརྡ་སྤྲོད་པའི་བསྟན་བཅོས་ཆེན་པོ་བཱ་ཎི་ནི་བྱཱ་ཀ་ར་ཎ་གཞུང་འགྲེལ་གྱི་གོ་དོན་
ཆོ་ག་དཔེར་བརྗོད་སོགས་ཟིན་བྲིས་སུ་བཀོད་པ་ལེགས་བཤད་ནོར་བུ་འདྲེན་པའི་ཤིང་རྟ་ཞེས་བྱ་བ་ལས་ཨ་ཙའི་མཐའ་ཅན་མ་ནིང་གི་རྟགས་རྣམས་ཀྱི་བྱ་བ་སྤྱིར་བཏང་དམིགས་བསལ་དང་བཅས་པ་རགས་པ་རྣམས་རེ་ཞིག་རྫོགས་སོ།། །།ད་ནི་ཧ་ལ་གྱི་མཐའ་ཅན་ཏེ་གསལ་བྱེད་མཐའ་ཅན་
ལས་དང་པོར་ཕོ་རྟགས་རྣམས་བཤད་བར་བྱའོ། །དེ་ཡང་དང་པོ་ཧ་མཐའ་ཅན་ལ། ལ་ཧ་ཞེས་པ་ལ། ཧོ་ཌྷཿ། ཧ་ཡི་ཌྷའོ། ཛྷ་ལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་ཕ་རོལ་ན་ཡོད་པའི་ཆུ་རོལ་གྱི་ཧ་སྲོག་མེད་དང་ཚིག་མཐའི་ཧ་སྲོག་མེད་ནི་ཌྷ་རུ་འགྱུར་རོ། །ལི་ཧ། སུ། ཧ་ཡི་ཌྷ་འོས། ཧ་ཌྷ། ཧ་ལ་
དཱི་པ་ཨྭ་པ་རིང་རྣམས་ལ་ཞེས་བས། སུ་ཕྱིས། གནས་མེད་པ་ལ་ཡང་ན་འོས། མྱོང་། ལི་ཌ། ལི་ཊ། ཨཽ་དང་ཛས་ལ། ཨ་ཙ་དམན་པ་ཕ་རོལ་དུ་རབ་ཏུ་སྦྱོར་རོ། །ཞེས་པས་དྲངས། ལི་ཧཽ། ལི་ཧཿ། གཉིས་པ་རྣམས་ལ། ལི་ཧཾ། ལི་ཧཽ། ལི་ཧཿ། ཊཱ་ལ། ལི་ཧཱ། བྷྱཱ་ལ། ཧ་ཡི་
ཌྷ། ཛྷ་ཤ་ལ་ཛྷ་ལ་རྣམས་ཀྱི་ཛ་རྣམས་སོས། ཌྷ་ཌ་དྲངས། ལི་བྷྱཱཾ། བྷིཿ། ལིཌྦྷིཿ། ངེ་ལ། ལི་ཧེ། བྷྱཱཾབྷྱཿལ། ལིཾཌྦྷྱཱ། ལཌྦྷྱཿ། ངསི་ལ། ལི་ཧཿ། བྷཾཱྱཾབྷྱཿལ། ལིཌྦྷྱཿ། ལཌྦྷྱཿ། ངས་ལ། ལི་ཧཿ། ཨཱ་མལ། ལི་ཧཱཾ། ཨཱ་མ་ལ། ལི་ཧཱཾ། ངི་ལ། ལི་ཧེ། ཨོ་ས་ལ། ལི་ཧོཿ།
སུ་པ་ལ། ཧ་ཡི་ཌྷ་འོས་ཧ་ཌྷ། ཁར་ལ་ཅར་འོས། ཌྷ་ཊ་དྲངས། ལིཊྶུ། བོད་ཚིག་རྣམས་ཀྱང་རྣམ་དབྱེ་དང་པོ་དང་འདྲ་བས། ཧེ་ལི་ཊ། ཧེལིཊ། ཧེ་ལི་ཧཽ། ཧེ་ལི་ཧཿ། འཇོག དུ་ཧ་ཞེས་པ་ཅུང་མི་འདྲ་སྟེ། དུ་ཧ། སུ། བཱ་དེ་རྦྷཱ་ཏོ་རྒྷཿ། ད་སྔ་མའི་བྱིངས་ཀྱི་གྷའོ། །ད་སྔོན་ན་ཡོང་
