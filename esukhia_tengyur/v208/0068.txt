ཏི་དང་། །ཨཱུ་ཋར་བསྒྱུར་བའི་ཨཱུ་ཡིག་རྣམས་བྱུང་ན་འང་འཕེལ་བར་འགྱུར་ཏེ། ཨུ་པ། ཨེ་ཏི་ཞེས་པ་ལ། །ཨུ་པཻ་ཏི། ཨུ་པ། ཨེ་དྷྲ་ཏི་ལ། ཨུ་བཻ་དྷ་ཏི། ཨཱུ་ཋ་ལ། བ་ཧའི་ཨཱུ་ཋ་འོས་པ་ཨཱུ་ཋར་བསྒྱུར་བ་ལྟ་བུ་སྟེ། བྲ་ཤཱ། ཨཱུ་ཧ༔ཞེས་པ་ལ། བྲ་ཤྣཽ་ཧ༔། ཨ་ཀྵ་ལས་ཨཱུ་ཧི་ཎྟི་ལའོ། །
ཨ་ཀྵ་ལས་ཕ་རོལ་ཏུ་ཨཱུ་ཧ་ཎཱི་བྱུང་ན་གཉིས་དག་གིས་གཅིག་འཕེལ་བར་འགྱུར་ཏེ། ཨ་ཀྵ། ཨཱུ་ཧི་ཎཱི་ཞེས་ ཉེ་བར་འགྲོ་བར་བྱེད། ཉེ་བར་འཕེལ་བར་བྱེད། བ་ལ། ཨ་ཀྵཽ་ཧི་ཎཱི། དམག་མིག་གིས་མི་བཟོད་པ། སྭཱ་དཱི་རེ་རི་ཎྱོཿ། སྭལས་ ཨཱི་ར ཨཱི་རི་ཎཱིཿ བྲིས་ཐོག་ དག་ལའོ། སྭ་ལས་ཨཱིར་དང་ཨཱི་རི་ཎཱི་
དག་ཕ་རོལ་ལ་གཉིས་དག་གིས་གཅིག་འཕེལ་བར་འགྱུར་ཏེ། སྭ་ཨཱིར༔ཞེས་པ་ལ། སྭཻར༔། དམག་སྡེ་ཆེན་པོ་གཅིག་གི་མིང་ཡིན་ཏེ། ཇི་སྐད་དུ། སུས་འགྱུར་ཤིང་རྟ་དཔའ་བོ་འབུམ། །རྟ་ནི་འབུམ་ཕྲག་བཅུ་རྐང་ཐང་། །བྱེ་བ་སུམ་ཅུ་རྩ་དྲུག་འཇུག། །ཨ་གྵཽ་ཏི་ཎཱི་ཞེས་བྱར་བརྗོད། །ཅེས་སོ། །སྭ་ཨཱི་རི་ཎཱི་ཞེས་པ་ལ།
སྭཻ་རི་ཎཱི། པྲཱ་དཱུ་ཧོ་ཌྷོ་དྷྱེ་ཥྱེ་ཥྱེ་ཥྱ། བྲ་ལས་ ཨཱུ་ཌྷ་ ཨཱུ་ཌྷི ཨེ་ཥ་ ཨེ་ཥྱ་རྣམས་ལའོ། རང་འདོད་ག པྲ་ལས་ ཨཱུ་ཧ་ ཨཱུ་ཌྷ་ ཨཱུ་ཌྷི་ ཨཱུ་ཌྷ་ ཨེ་ཥ་ ཨེ་ཥྱ་རྣམས་ཕ་རོལ་ལ་འཕེལ་བར་འགྱུར་ཏེ། པྲཽ་ཧ༔། པྲཽ་ཌྷ༔། པྲཽ་ཌྷེ༔། པྲཻ་ཥ༔། པྲཻ་ཥྱ༔། འགྲེལ་པར་སྐབས་ལ་ལར་པྲ་དང་ཨེ་ཥ་ལ་ཡོན་ཏན་
རང་འདོད་མ་སྟ་སྨད་འཚོང་མའོ། དུ་བྱེད་པའང་ཡོད་དོ། །ཞེས་འབྱུང་བས། བྲེ་ཥ༔ རྀ་ཏི་ཙ་ཏཱི་ཡཱས་མཱ་སེ། རྀ་ཏ་ལའང་གསུམ་པའི་ཚིག་སྡུད་ལའོ། །ཨ་ཡིག་མཐའ་ཅན་ནི་རྀ་ཏ་ཕ་རོལ་ལ་ རབ་ཏུ་ཐོག ལུས་རྒྱགས་པ་འམ་འཇིགས་པ་མེད་པ་འམ་བརྟན་པ། བློ་རྐྱས་པའམ་བརྟན་གར་ཏུ་འདོད་པ། རྣམ་དབྱེ་གསུམ་
བའི་ཚིག་སྡད་ལ་གཉིས་དག་གིས་གཅིག་འཕེལ་བར་འགྱུར་ཏེ། རབ་ཏུ་འདོད་པ། སུ་ཁེ་ན་རྀ་ཏ༔ཞེས་པ་ལ། སུ་ཁཱ་རྟ་སོགས་སོ། །པྲ་བ་ཏྶ་བཏྶ་ཏར་ཀ་མྦ་ལ་བ་ན་ད་ཤཱ་རྞཱ་ཎཱ་མྲྀ་ཎེ། པྲ་བ་ཏྶ། བཏྶ་ཏར། ཀ་མྦ་ལ བ་ས་ན ད་ཤ རྀ་ཎ་རྣམས་ཀྱི་རྀ་ཎ་ལའོ། །པྲ་བཏྲ་ པཏྶ་ བཏྶ་ཏ་ར ཀ་མྦ་ལ་
བ་ས་ན ད་ཤ རྀ་ཎ་རྣམས་ཀྱི་ཕ་རོལ་ཏུ་རྀ་ཎ་བྱུང་ན་གཉིས་དག་གིས་གཅིག་འཕེལ་བར་འགྱུར་ཏེ། བདེ་བས་ཟིན། པྲཱ་རྞ༔། རབ་ཏུ་བུ་ལོན། བ་ཏྶ་ཏ་རཱ་རྞ༔། བེའུ་མཆོག་གི་བུ་ལོན། བ་ས་ནཱ་རྞ༔། གོས་ཀྱི་བུ་ལོན། ད་ཤ་རྞ༔། བཅུའི་བུ་ལོན།
