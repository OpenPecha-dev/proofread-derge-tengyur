ཌུ་ཧཱ། ཨ་ན་ཌུ་ཧ་ལ། བྷྱཾ་བྷྱས། བསུ་སྲཾ་སཱུ་དྷྭཾ་སྭ་ན་ཌྭ་ཧཱཾ་དཿ། བ་སུ་སྲཾ་སུདྷྭཾ་སུ་ཨ་ན་ཌུ་ཧ་རྣམས་ཀྱི་དའོ། བ་སུ་དང་སྲཾ་སུ་དང་དྷྭཾ་སུ་དང་ཨ་ན་ཌུ་ཧ་རྣམས་ཀྱིས་མཐའ་རྣམས་དང་ཧ་ཡིག་ནི་ཛྷལ་སོ་སོར་བསྡུ་བའི་ཡི་གེ་གང་རུང་ཕ་རོལ་ན་ཡོད་པ་དང་ཚིག་མཐའ་ལ་ད་རུ་འགྱུ་རོ་
ཨ་ན་ཌུ་དྡྷུཾ། ཨ་ན་ཌུ་དྡྷིཿ། བཞི་པ་རྣམས་ལ། ཨ་ན་ཌུ་ཧེ། ཨ་ན་ཌུ་དྦྷྱཱཾ། ཨ་ན་ཌུ་དྦྷྱཿ། ལྔ་པ་རྣམས་ལ། ཨ་ན་ཌུ་ཧཿ། ཨ་ན་ཌུ་དྦྷྱཱཾ། ཨ་ན་ཌུ་དྦྷྱཿ། དྲུག་པ་རྣམས་ལ། ཨ་ན་ཌུ་ཧཿ། ཨ་ན་ཌུ་ཧོཿ། ཨ་ན་ཌུ་ཧཱཾ། བདུན་པ་རྣམས་ལ། ཨ་ན་ཌུ་ཧི། ཨ་ན་ཌུ་ཧོཿ། ཨ་ན་ཌུ་ཏྶུ། ཤྭེ་ཏ་ཝཱ་ཧ་ཞེས་

ཊིའི་ལོ་ཞེས་པའི་དོན་ཡིན་པས་ཊི་ཕྱིས། ཤྭེ་ཏ་བས་སུ། ཨ་ཏྭ་སཾ་ཏ་སྱ་ཙཱ་དྷཱ་ཏོཿ། ཨ་ཏུ་ཨ་ས་མཐའ་དབྱིངས་མ་ཡིན་པའིའོ། ཨ་ཏུའི་མཐའ་དང་ཨས་ཀྱི་མཐའ་བྱིངས་མ་ཡིན་པ་དག་སཾ་བུདྡྷི་མ་ཡིན་པའི་སུ་ཕ་རོལ་ལ་ཨུ་པདྷཱ་རིང་བོར་འགྱུར་རོ། །ཤྭེ་ཏ་བཱཿ། ཨཽ་དང་ཛ་ས་ལ། ཤྭེ་ཏ་
བཱ་ཧཽ། ཤྭེ་ཏ་བཱ་ཧཿ། ཨ་མ་ལ། ཤྭེ་ཏ་བཱ་ཧཾ། ཨཽ་ལ། ཤྭི་ཏ་བཱ་ཧཽ། ཤྭེ་ཏ་བཱ་ཧ། ཤས། ཤས་ནས་བཟུང་ཨ་ཅ་ཐོག་མ་ཅན་རྣམས་ལ་སྔར་བི་ཤྭ་ཝཱ་ཧའི་སྐབས་སུ་བཤད་པ་ལྟར་ཨཱུ་ཋར་འགྱུར་བ་དང་མི་འགྱུར་བ་ཡང་ན་བ་སྟེ། ཤྭེ་ཏཽ་ཧཿ། ཤྭེ་ཏ་ཝཱ་ཧཿ། ཊཱ་ལ། ཤྭེ་ཏཽ་ཧཱ། ཤྭེ་ཏ་ཝཱ་ཧཱ་
བྷྱཱཾ་ལ། ཌ་སའི་རྐྱེན། ཏི་ཕྱིས། ས་ས་ཇུ་ཥ་དག་གི་རུ། ཨ་ལས་ཡོན་ཏན། ཤྭེ་ཏ་པོ་བྷྱཱཾ། བྷིས་ལ། ཤྭེ་ཏ་བོ་བྷིཿ། ངེ་ལ། ཤྭེ་ཏ་བཱ་ཧེ། ཤྭེ་ཏཽ་ཧེ། བྷྱཱཾ་བྷྱ་ས་ལ། ཤྭེ་ཏ་བོ་བྷྱཱཾ། ཤྭེ་ཏ་བོ་བྷྱཿ། ང་སི་ལ། ཤྭེ་ཏཽ་ཧཿ། ཤྭེ་ཏ་བཱ་ཧཿ། བྷྱཱཾ་བྷྱཿའདྲ། ངས་ལ། ཤྭེ་ཏཽ་ཧཿ། ཤྭེ་ཏ་བཱ་ཧཿ།
