དྷེ་ནུ་བྷིཿ། བཞི་པ་ལ། དྷེ་ནུ། ངེ། ང་འགྲོ་བ་ལ་ཐུང་ངུའི་ཡང་ན་འོས། དྷེ་ནྭཻ། དྷེ་ན་ཝཻ། དྷེ་ནུ་བྷྱཱཾ། དྷེ་ནུ་བྷྱཿ། ལྔ་པའི་ང་སིལ། དྷེ་ནཱཿ། དྷེ་ནོཿ། བྷྱཱ་མ་བྷྱས་འདྲ། དྲུག་པའི་ངས་ལ། དྷེནྭཱཿ། དྷེ་ནོཿ། ཨོས་ལ། དྷེནྭོཿ། ཨཱ་མ་ལེ། དྷེ་ནཱུ་ནཱཾ། བདུན་པའི་ངི་ལ། ངྷེ་ནཱཾ། ངྷེ་ནཱཾ། ངྷེ་ནཽ། ཨོས་
ལཊ། ངྷེ་ནྭོཿ ངྷེ་ནུཥུ། བོད་ཚིག་རྣམས་ལ། ཧེ་དྷེ་ནོ། ཧེ་དྷེ་ནཱ། ཧེ་དྷེ་ན་ཝ་རྣམས་གྲུབ་སྟེ་ཨ་ཙ་མི་གཅིག་པའི་འགྱུར་ཁྱད་ཕྲན་ཙམ་མ་གཏོགས་གོང་དུ་བཤད་པ་མ་ཏེའི་སྐབས་དང་འདྲའོ། །ཏ་ནུ་ལ་སོགས་པ་ཡང་འདི་དང་འདྲའོ། ཨུ་ཐུང་མཐའ་ཅན་ཀྲེཥྚུ་ལ་བཤད་མ་ཐག་པ་
དང་བྱ་བ་མི་འདྲ་སྟེ། སྟྲི་ཡཱཾ་ཙ། བུད་མེད་ལའང་ངོ་། །ཀྲོཥྚུའི་སྒྲ་ནི་བུད་མེད་ཀྱི་རྟགས་ལ་ཏྲྀ་ཙ་བཞིན་རི་མཐར་འགྱུར་པས། ཀྲོཥྚྲྀ་འགྱུར། རྀནྣི་བྷྱོ་དཱི་པ། རྀ་ཏ་ན་རྣམས་ལས་དཱི་པའོ། །སུས་ཕྲ་མ། རྟགས་ཪྀ་ཏ་མཐའ་ཅན་རྣམས་དང་ནམ་མཐའ་ཅན་རྣམས་ལ་བུད་མེད་དུ་བསྒྲུབ་པའི་རྐྱེན་དཱི་པ་
འགྱུར་ཏེ། དཱི་པ་བྱིན། ཀྲོཥྚྲཱ་ཞེས་པར་འགྱུར་རོ། ཅེ་སྦྱང་མོ། ཆོ་ག་ཐམས་ཅད་སྔར་མོ་རྟགས་ཨཱི་རིང་སྐབས་སུ་བཤད་པའི་གཽ་རཱེ་དང་འདྲའོ། །དེ་རྣམས་ཀྱིས་ཨུ་ཐུང་མཐའ་ཅན་མོ་རྟགས་ཀྱི་ཆོ་ག་བསྟན་ཟིན་ཏོ། ༈ །ད་ནི་མོ་རྟགས་ཨཱ་རིང་མཐའ་ཅན་རྣམས་ཀྱི་བྱ་བ་བརྗོད་པར་བྱ་སྟེ། བདྷཱུ། སུ།
བདྷཱུཿ། ཨཽ་དང་། ཛས། བདྷྭཽ། བིདྷྭཿ། བདྷཱུ་ལ། སཾ་བུ་དྡྷི། སུ། ཨམྦཱའི་དོན་ན་དཱི་རྣམས་ཀྱི་ཐུང་དུ་འོས། རྟགས་མཐའ་ཐུང་ངུར་འགྱུར། ཨེང་ཐུང་ངུ་ཞེས་པས་སུ་ཕྱེས། ཧེ་བ་དྷུ། ཧེ་བ་དྷྭོ། ཧེ་བ་དྷཱོཿ། བདྷཱུ་ལ། ཨམ་ལ་སྔོན་མའི་འོས། བདྷཱུཾ། ཨཽ་ལ། བདྷཱཽ། ཤས་ལ། དང་པོ མནའ་མ།
དག་གིས་མདོས། བདྷཱུཿ། གསུམ་པ་རྣམས་ལ། བདྷཱ། བདྷཱུ་བྷྱཱ་མ། བདྷཱུ་བྷིཿ། དེ་ལ། བདྷྭཻ། བདྷཱུ་བྷྱཱཾ། བདྷཱུ་བྷྱཿ། ངས་ལ། བདྷྭཱཿ། བྷྱཱཾ་བྷྱས་འདྲ། ངས་ལ་ཡང་། བདྷཱཿ། ཨོ་ས་ལ། བདྷྭཱོཿ། ཨྭ་མ་ལ། ཐུང་ངུ་ན་དཱི་ཞེས་པས་ནུཊ། བདྷཱུ་ནཱཾ། བིལ། བདྷྭཱཾ། ཨོ་ས་ལ། བདྷཱོཿ། སུ་བ་
