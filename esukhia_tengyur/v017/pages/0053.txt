བཅོས་རྣམས་ལས་འདའ་མི་བྱ། །ཞེས་བྱ་བ་ཡིན་ཏེ། བསྟན་བཅོས་དང་ལྡན་པ་རྣམས་ཀྱིས་རང་དང་གཞན་གྱི་མཚན་མའི་བསྟན་བཅོས་ལས་འདའ་བར་མི་བྱའོ་ཞེས་ཀོངྐ་ནའི་ཞལ་སྔ་ནས་བཞེད་དོ། །སྐལ་ལྡན་གྲགས་པ་ནི་བསྟན་བཅོས་ནི་རིག་བྱེད་ལ་
སོགས་པ་སྟེ། དེ་ལས་མི་འདའ་བ་ནི་སེམས་ཅན་གྱི་བསམ་པའི་དབང་དུ་བྱས་ནས་དེ་ལ་ཡང་གནས་པར་བསྟན་པར་བྱའོ་ཞེས་དགོངས་སོ། །དམ་ཚིག་ཅན་ལ་ཁྲོ་མི་བྱ། །ཞེས་བྱ་བ་ནི་རྟག་ཏུ་ཁོང་ཁྲོ་བར་མི་བྱའོ་ཞེས་ཀོངྐ་ནའི་ཞལ་སྔ་ནས་བཞེད་དོ། །སྐལ་ལྡན་གྲགས་པ་ནི་དམ་ཚིག་
ཅན་ཞེས་བྱ་བ་འཁོར་ལོ་བདེ་མཆོག་ལ་སོགས་པའི་རྣལ་འབྱོར་པ་རྣམས་ལ་སྨད་པར་མི་བྱའོ་ཞེས་འདོད་དོ། །མངོན་པར་བརྩོན་ལས་འདའ་མི་བྱ། །ཞེས་བྱ་བ་དཔལ་འཁོར་ལོ་བདེ་མཆོག་དང་འདུས་པ་ལ་སོགས་པར་གང་ཞིག་དབང་བསྐུར་བ་དེ་ལས་འདའ་བར་མི་བྱའོ་ཞེས་ཀོངྐ་ནའི་
ཞལ་སྔ་ནས་བཞེད་དོ། །སྐལ་ལྡན་གྲགས་པ་ནི་གང་ཞིག་གང་ལ་མངོན་པར་བརྩོན་པ་དེར་དེ་སྨོད་པར་མི་བྱའོ་ཞེས་འདོད་དོ། །ནམ་མཁའ་དང་ནི་རྒྱ་མཚོ་དག །ལན་འགའ་གཞལ་བར་ནུས་སྲིད་ཀྱི། །དེ་ཉིད་ལ་ནི་རྟག་སེམས་བློ། །གཞལ་བར་ནུས་པ་མ་ཡིན་ནོ། །ཞེས་དགོངས་སོ། །དེ་
དང་ལྷན་ཅིག་བདེ་བ་བྱ། །ཞེས་བྱ་བ་ནི་དམ་ཚིག་ལྷན་ཅིག་སྤྱོད་པ་གཅིག་ཏུ་བྱའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཡང་ན་བདེ་བ་ནི་ཐབས་ཏེ་ཕྱག་རྒྱ་ཆེན་པོ་གྲུབ་པའི་ཐབས་བྱ་བ་ཡིན་ནོ་ཞེས་ཀོངྐ་ནའི་ཞལ་སྔ་ནས་བཞེད་དོ། །སྐལ་ལྡན་གྲགས་པ་ནི་གླེགས་བམ་ཁ་ཅིག་ལས། དེ་དང་ལྷན་ཅིག་བདེ་
རྒྱས་བྱ། །ཞེས་ཟེར་བར་བླང་བར་བྱ་བའི་མཆོག་བཀླག་པ་དང་བསམ་གཏན་ལ་སོགས་པ་གདོད་མ་ནས་བྱའོ་ཞེས་འདོད་དོ། །བདག་གི་དོན་དུ་ཐུན་བཞིར་ཞེས་བྱ་བ་ནི་ཉིན་མཚན་ཐུན་བཞིར་བྱ་བའོ། །དེས་འཁོར་ཅི་དགར་སྤྱོད་པར་འགྱུར། །ཞེས་བྱ་བ་ནི་གང་ངེས་པར་འདི་ལྟར་འཁོར་བའི་
རྣལ་འབྱོར་པ་ཅི་འདོད་པ་དང་ལྡན་པར་འགྱུར་བའོ། །མཚན་མོ་རྟག་པར་ནམ་མཁའི་གོས། །ཞེས་བྱ་བ་ནི་གོས་མེད་ཅིང་སྐྲ་གྲོལ་བར་བྱས་ནས་དཔལ་ཧེ་རུ་ཀ་བསྒོམ་པར་བྱའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །གནག་དང་དམར་བར་རྟག་གྱུར་བྱ། །ཞེས་བྱ་བ་ནི་མཚན་མོ་རྒྱུ་བ་ན་འགྱུར་རོ་ཞེས་ཀོངྐ་
