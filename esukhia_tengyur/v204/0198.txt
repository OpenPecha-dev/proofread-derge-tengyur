ཨ་ནི་པྲྀ་ཏཱ་བྱཱ་ཀྲྀ་ཏཱཿ། ཨ་བྷི་ས་མ་ཡཱནྟི་ཀཿ། ཀིཾ་ནཱ་ནཱ་ཀ་ར་ཎཾ། ཨ་སམྤྲ་ཛྙཱ་ནཾ། ནི་རཱ་སྭཱ་དཿ། ཀཱ་མཱ་བ་ཙ་རཿ། རཱུ་བཱ་པ་ཙ་རཿ། ཨོ་མི་ཏྱཱ་ཧ། དཱི་བཿ། ཨུལྐཱ། ཨུལྐཱ་
མ་བསྒྲིབས་པའི་ལུང་དུ་མི་སྟོན་པ། མངོན་པར་རྟོགས་པའི་མཐའ་ལས་བྱུང་བ། ཐ་དད་དུ་བྱ་བ་ཅི་ཡོད། ཤེས་བཞིན་མ་ཡིན་པ། རོ་བྲོ་བ་མེད་པ། འདོད་པ་ན་སྤྱོད་བ། གཟུགས་ན་སྤྱོད་པ། དེ་བཞིན་ནོ་ཞེས་སྨྲས་སོ་ཞེའམ་སྨྲས་པ་དེ་བཞིན་ནོ། །མར་མེ། སྐར་མའམ་སྒྲོན་མའམ་ཏལ་ལ། སྐང་མདའི་
མུ་ཁམ྄། པྲ་དཱི་པཿ། ས་མུ་དི་ཏཿ། ཤླི་ཥོཀྟིཿ། པྲ་བྲྀདྡྷཾ། ཨ་བྷི་སཾ་བྷོཏྶྱ་ཏེ། བོ་དྷི་མ་བྷི་སམྦུདྡྷཿ། ཨ་བྷི་སམྦུདྡྷྱ་ཏི། བུདྡྷ་ཏྭ་མ་བཱ་བྣོ་ཏི། བི་དྱོ་ཏི་
གདོང་ངམ་སྣ། སྒྲོན་མའམ་མར་མེ། ཤར་བའམ་འདུས་པའམ་སྤྱི། ཚིག་ཟུར་གཉིས་སུ་སྦྱར་བ། འཕེལ་བའམ་སྐྱེས་པ། མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བར་འགྱུར། བྱང་ཆུབ་མངོན་བར་རྫོགས་པར་སངས་རྒྱས། མངོན་པར་རྫོགས་པར་འཚང་རྒྱ་བ། སངས་རྒྱས་ཉིད་འཐོབ་བོ། །རྣམ་པར་སྣང་བར་
ཏཿ། པ་རི་ཛྙཱ། སམྦྷ་པ་ཏི། ཨ་སམྦྷ་བཿ། ཨཱ་ཀྲ་མ་ཎམ྄། ཨཱ་ཀྲ་མ་ཡ་ཏི། ཨ་བ་ཀྲ་མ་ཏི། ཨཱ་ཡ་ཏཿ། པྲ་བྷཱ་བ་ནཱ། བྱ་བསྠཱ་ནམ྄། བཱིབྶཱ། ཀྵ་ཎི་ཀཿ། ཨཱ་ལམྦ་ཏེ།
བྱས། ཡོངས་སུ་ཤེས་པ། འབྱུང་ངམ་སྲིད་པ། མི་སྲིད་པའམ་མི་འབྱུང་བ། གནོན་པ། གནོན་པར་བྱེད་པ། འགྲོ་བའམ་འཇུག་པ། ཡུན་རིང་པོའམ་རྒྱ་ཆེ་བའམ་ཡངས་པ། བརྗོད་པ། རབ་ཏུ་ཕྱེ་བའམ་རབ་ཏུ་བཞག་པའམ་རྣམ་པར་བཞག་པ། ཟློས་པ། སྐད་ཅིག་པ། གྲིབ་སོ་བབས།
