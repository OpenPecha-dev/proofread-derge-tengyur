ཡཱི། ཨ་ན་བྷི་སཾ་སྐཱ་ར་པ་རི་ནིརྦཱ་ཡཱི། ཨཱུརྡྷ་ཤྲོ་ཏཱ། ཀཱ་ཡ་སཱ་ཀྵཱི། ཤྲདྡྷཱ་ནུ་སཱ་རཱི། དྷརྨྨཱ་ནུ་སཱ་རཱི། ཤྲདྡྷཱ་དྷི་མུཀྟཿ། དྲྀཥྚི་པྲཱཔྟཿ། ས་མ་ཡ་བི་མུཀྟཿ། ཨ་ས་
ཡོངས་སུ་མྱ་ངན་ལས་འདའ་བ། མངོན་པར་འདུ་བྱེད་མེད་པར་ཡོངས་སུ་མྱ་ངན་ལས་འདའ་བ། གོང་དུ་འཕོ་བ། ལུས་ཀྱིས་མངོན་སུམ་དུ་བྱེད་པ། དད་པས་རྗེས་སུ་འབྲང་བ། ཆོས་ཀྱི་རྗེས་སུ་འབྲང་བ། དད་པས་མོས་པ། མཐོང་བས་ཐོབ་པ། དུས་ཀྱིས་རྣམ་པར་གྲོལ་བ། དུས་དང་མི་
མ་ཡ་བི་མུཀྟཿ། པྲ་ཛྙཱ་བི་མུཀྟཿ། ཨུ་བྷ་ཡ་ཏོ་བྷཱ་ག་བི་མུཀྟཿ། །ཉན་ཐོས་སོ་སོའི་མིང་ལ། ཨཱ་ཛྙཱ་ཏ་ཀཽཎྜི་ནྱཿ། ཀཱ་ཤྱ་པཿ། ཤཱ་རི་པུ་ཏྲཿ། མཽངྒ་ལྱཱ་ནཿ། མ་ཧཱ་ཀཱ་ཏྱཱ་ཡ་ནཿ། སུ་བྷཱུ་ཏིཿ། པཱུརྞྞོ་མཻ་ཏྲེ་ཡ་ཎཱི་པུ་ཏྲཿ། ཨ་ཤྭ་ཛིཏྲ། ཨ་ནི་རུདྡྷཿ། རཱ་ཧུ་ལཿ། ཨཱ་ནནྡཿ། ནནྡཿ། ནནྡ་ཀཿ།
སྦྱོར་བར་རྣམ་པར་གྲོལ་བ། ཤེས་རབ་ཀྱིས་རྣམ་པར་གྲོལ་བ། གཉི་གའི་ཆ་ལས་རྣམ་པར་གྲོལ་བ། ཀུན་ཤེས་ཀཽཎྜི་ནྱ། འོད་སྲུངས། ཤཱ་རིའི་བུ། མཽད་གལ་གྱི་བུ། ཀ་ཏྱའི་བུ་ཆེན་པོ། རབ་འབྱོར། བྱམས་མའི་བུ་གང་པོ། རྟ་ཐུལ། མ་འགགས་པ། སྒྲ་གཅན་འཛིན། ཀུན་དགའ་བོ། དགའ་བོ། དགའ་བྱེད།
ནནྡི་ཀཿ། མ་ཧཱ་ནཱ་མཿ། ཙུནྡཿ། ཏི་ཥྱཿ། ཨུ་པ་ཏིཥྱཿ། ཀོ་ལི་ཏཿ། ཨུ་རུ་བི་ལྭཱ་ཀཱ་ཤྱ་བཿ། ན་དཱི་ཀཱ་ཤྱ་པཿ། ག་ཡཱ་ཀཱ་ཤྱ་པཿ། ག་བཱམྤ་ཏིཿ། བཱ་ཤྱཿ། ཨུ་བ་སེ་ནཿ། ཙཱུ་ཏ་པནྠ་ཀཿ། མ་ཧཱ་པནྠ་ཀཿ། ཤྲོ་ཎ་ཀོ་ཊཱི་བཱིཾ་ཤཿ། ཨུ་དཱ་ཡཱི། སུནད་ར་ནནྡཿ། ཤྲོ་ཎ་ཀོ་
དགའ་ཡོད། མིང་ཆེན། སྐུལ་བྱེད། འོད་ལྡན་པུཥྱ་དང་མ་འདོམ་ན་སྐར་རྒྱལ། ཉེ་རྒྱལ། བང་ནས་སྐྱེས། ལྟེང་རྒྱས་འོད་སྲུངས། ཆུ་ཀླུང་འོད་སྲུངས། ག་ཡ་འོད་སྲུངས། བ་ལང་བདག །རླངས་པ། ཉེ་སྡེ། ལམ་ཕྲན་སྟན། ལམ་ཆེན་སྟན། གྲོ་བཞིན་སྐྱེས་བྱེ་བ་ཉི་ཤུ་བ། འཆར་ཀ། མཛེས་དགའ། གྲོ་བཞིན་སྐྱེས་
