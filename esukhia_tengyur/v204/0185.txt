ཀ་རཿ། དྷརྨྨ་དྷཱ་ཏུ་པ་ར་མཿ། ཨཱ་ཀཱ་ཤདྷཱ་ཏུ་པཪྻ་བ་སཱ་ནཿ། ཨསྠཱ་ན་མེ་ཏཏ྄། ཨུཏྠཱ་པུ་ནམ྄། ས་དྷཱ་མ་དྷྱ་ག་ཏོ་པཱ། རཱ་ཛ་ཀུ་ལ་མ་དྷྱ་ག་ཏོ་བཱ། པཪྵམྨ་དྷྱ་ག་ཏོ་བཱ། ཡུཀྟ་ཀུ་ལམ་དྷྱ་ག་
ཆེས་པ། ཆོས་ཀྱི་དབྱིངས་ཀྱིས་ཀླས་པ། ནམ་མཁའི་ཁམས་ཀྱི་མཐས་གཏུགས་པ། དེ་ནི་གནས་མ་ཡིན། བསླང་བའམ་དྲང་བའམ་བསྐྱེད་པ། འདུས་པའི་ནང་ན་འདུག་ཀྱང་རུང་། རྒྱལ་པོའི་ཕོ་བྲང་འཁོར་གྱི་ནང་ན་འདུག་ཀྱང་རུང་། འཁོར་གྱི་ནང་ན་འདུག་ཀྱང་རུང་། རིགས་ཚོགས་པའི་ནང་ན་འདུག་
ཏོ་བཱ། ཛྙཱ་ཏི་མ་དྷྱ་ག་ཏོ་བཱ། ཨ་ཏཱི་ཏཱཾ་ཤ་ག་ཏཱ། ན་ཀེ་ན་ཙི་དྭི་དྷཱི་ཡ་ཏེ། ཨ་བ་ཀཱ་ཤཿཔྲསྟཱ་བཿཨ་བ་སརཿ། བ་ནཱདྣི་རྦ་ན་མཱ་ག་ཏཿ། ཀཱ་མེ་ཥུནཻཥྐྲ་མྱམ྄། ཤཻ་ལ་མུཀྟི་མི་བ་ཀཱཉྩ་ནཾ། ནི་སེ་བི་ཏཿ། ཛིཧྭེནྡྲི་ཡཾ་ནི་རྣཱ་མ་ཡཱ་མཱ་ས། ཨ་བྷ་ཡ་
ཀྱང་རུང་། ཉེ་དུའི་ནང་ན་འདུག་ཀྱང་རུང་། འདས་པའི་ཆར་གྱུར་པ། གང་གིས་ཀྱང་མི་བྱའོ། །གོ་སྐབས་སམ་པར་རམ་པར་སྐབས། ནགས་ཚལ་ནས་ནགས་ཚལ་མེད་པར་གཤེགས་པར། འདོད་པ་ལས་སྤགས་པ། རི་ལས་ཐལ་བའི་ཉི་མ་བཞིན། བསྙེན་པ། ལྗགས་ཀྱི་དབང་པོ་ཕྱུང་སྟེ། འཇིགས་པའི་དབང་
བ་ཤ་བརྟྟི་ཏཱ། སརྦ་སཏྭཱ་ནཱཾ་ཀིཾ་ཀཱ་ར་ཎཱི་ཡ་ཏཱ། བ་ཧུ་ཛ་ནྱམ྄། པ་རམྤ་ར་ཡཱ། ཏུཥྞཱིམྦྷཱ་བེནཱ་དྷི་བཱ་ས་ཡ་ཏི། སམུལློ་ཀི་ཏ་མུ་ཁཿ། སམྱཀྤྲ་ཏྱཱཏྨཾ་ཛྙཱ་ན་དཪྴ་ནི་པྲ་བརྟྟ་ཏེ། ཨཱ་ཡུཿསཾ་སྐཱ་རཱ་ནུཏྶྲྀ་ཛ་ཏི། ཨུཥྨ་ཧཱ་ནིཿ། མ་ནུ་ཥྱཱ་ཎཱཾ་ས་བྷཱ་ག་ཏཱ་ཡཱ་
དུ་གྱུར་བ་མ་ཡིན་པ། སེམས་ཅན་ཐམས་ཅད་ཀྱི་བྱ་བ་ཅི་ཡོང་སྙམ་པ། སྐྱེ་བོ་མང་པོ་པ། གཅིག་ནས་གཅིག་ཏུ། ཅང་མི་གསུང་བར་གྱུར་པས་གནང་། ཞལ་དུ་བལྟས་བ། བདག་ཉིད་ཀྱིས་ཡང་དག་པར་ཤེས་པ་མཐོང་བ་འཇུག་པ། ཚེའི་འདུ་བྱེད་འདོར་བ། དྲོད་ཡལ་བ། མི་དང་སྐལ་བ་མཉམ་པར་
