ཛྙི་ས་ཏྭཱ། །མི་ཁོམ་པ་བརྒྱད་ཀྱི་མིང་ལ། ཨཥྚཱ་བཀྵཎཱཿ། ན་ར་ཀཱ། ཏིརྻ་ཉྩཿ། པྲེ་ཏཱ། དཱིརྒྷཱ་ཡུ་ཥོ་དེ་བཱ། བྲ་ཏྱནྟ་ཛ་ན་པ་དཾ། ཨིནྡྲི་ཡ་བཻ་ཀ་ལྱཾ། མེ་ཐྱཱ་ད་ཪྴ་ནཾ། ཏ་ཐཱ་ག་ཏཱ་ནཱ་མ་ནུ་ཏྤཱ་དཾ། །ལས་ཀྱི་རྣམ་པར་སྨིན་པ་ལ་སོགས་པའི་མིང་ལ། དྲྀཥྚི་དྷ་རྨ་པེ་ད་ནཱི་ཡཾ། ཨུ་པ་
པའི་སེམས་ཅན། མི་ཁོམ་བརྒྱད། སེམས་ཅན་དམྱལ་བ། དུད་འགྲོ། ཡི་དགས། ལྷ་ཚེ་རིང་པོ། །མཐའ་འཁོབ་ཀྱི་མི། དབང་པོ་མཚང་བ། ལོག་པར་ལྟ་ག་དེ་བཞིན་གཤེགས་པ་མ་བྱུང་བ། མཐོང་བའི་ཆོས་ལ་མྱོང་བར་འགྱུར་བ། སྐྱེས་
པ་དྱ་བེ་ད་ནཱི་ཡཾ། ཨ་པ་ར་པརྻཱ་ཡ་བེ་ད་ནཱི་ཡཾ། ནི་ཡ་ཏ་བེ་ད་ནཱི་ཡཾ། ཨ་ནི་ཡ་ཏ་བེ་ད་ནཱི་ཡཾ། ཀརྨ་སྭ་ཀཿ། ཀརྨ་དཱ་ཡ་དཿ། ཀརྨྨ་ཡོ་ནིཿ། ཀརྨ་པྲ་ཏི་ས་ར་ཎ་མ྄། པྲ་ཡོ་གཿ། མོ་ལཾ། པྲྀཥྚཾ། ཨཱ་སེ་བི་ཏཾ། བྷཱ་
ནས་མྱོང་བར་འགྱུར་བ། ལན་གྲངས་གཞན་ལ་མྱོང་བར་འགྱུར་བ། མྱོང་བར་འགྱུར་བར་ངེས་པ། མྱོང་བར་འགྱུར་བར་མ་ངེས་པ། ལས་བདག་གིར་བྱ་བ། ལས་ཀྱི་བགོ་སྐལ་སྤྱོད་པ། ལས་ཀྱི་སྐྱེ་གནས་པ། ལས་བརྟེན་པར་བྱ་བ། སྦྱོར་བ། དངོས་གཞི། འཇུག །བསྟེན་པའམ་བསྙེན་བ། བསྒོམ་པའམ་གོམས་
བི་ཏཾ། བ་ཧུ་ལཱི་ཀྲྀ་ཏཾ། །མཚམས་མེད་ལྔའི་མིང་ལ། མཱ་ཏྲྀ་གྷཱ་ཏཿ། ཨ་རྷདྦ་དྷཿ། པི་ཏྲྀ་གྷཱ་ཏཿ། སངྒྷ་བྷེ་དཿ། ཏ་ཐཱ་ག་ཏ་སྱཱནྟི་ཀེ་དུཥྚ་ཙིཏྟ་རུ་དྷི་རོ་ཏྦཱ་ད་ནཾ། །མཚམས་མེད་པ་དག་ཉེ་བ་ལྔའི་མིང་ལ། མཱ་ཏུ་རརྷཾནྟྱཱ་དུ་ཥ་ཎཾ། ནི་ཡ་ཏ་བྷུ་མི་སྠི་ཏ་སྱ་བོ་དྷི་སཏྭ་སྱ་ནཱ་ར་ཎཾ།
པར་བྱས་པ། ལན་མང་དུ་བྱས་པ། མ་བསད་པ། དགྲ་བཅོམ་པ་བསད་པ། ཕ་བསད་པ། དགེ་འདུན་གྱི་དབྱེན་བྱས་པ། དེ་བཞིན་གཤེགས་པ་ལ་ངན་སེམས་ཀྱིས་ཁྲག་ཕྱུང་བ། མ་དགྲ་བཅོམ་མ་སུན་ཕྱུང་བ། བྱང་ཆུབ་སེམས་དཔའ་ངེས་པའི་ས་ལ་གནས་པ་བསད་པ།
