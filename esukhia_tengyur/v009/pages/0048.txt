མོངས་སེར་བ་ཞི་བྱེད་དང་། །ཏིང་ངེ་འཛིན་ནི་རྒྱས་བྱེད་ཕྱིར། །རང་གི་ལྷ་ཡི་སྦྱོར་བས་བདག །ཇི་ལྟ་བར་ནི་སེམས་ཀྱིས་བྱ། །དེ་ནས་ཕྱི་ཡི་བཤད་བྱ་སྟེ། །ཐབ་དང་རྫས་ཀྱི་མཚན་ཉིད་དང་། །སྒྲུབ་པ་པོ་དང་ལྷ་དག་དང་། །སྒྲུབ་པའི་ཐབས་དང་རྣམ་ལྔར་འགྱུར། །དེ་ལ་ཐབ་ཁུང་བཤད་པ་ནི། །
སེམས་ཅན་དབྱེ་བས་མཐའ་ཡས་ཏེ། །ལས་ཀྱི་དབྱེ་བས་བཞི་རུ་འགྱུར། །རིགས་ཀྱི་དབྱེ་བས་ལྔ་རུ་འགྱུར། །ཞི་བ་ལ་ནི་ཟླུམ་པོ་སྟེ། །རྒྱས་པ་ལ་ནི་གྲུ་བཞིའོ། །དབང་ལ་གཞུ་ཡི་དབྱིབས་སུའོ། །མངོན་སྤྱོད་ལ་ནི་གྲུ་གསུམ་མོ། །བསྐྲད་དང་དགུག་ལའང་དེ་བཞིན་ནོ། །རྒྱ་རུ་ཁྲུ་གང་ཟབས་
སུ་ནི། །མཐོ་གང་པ་ནི་ཞི་བ་ཡིན། །རྒྱ་རུ་ཁྲུ་དོ་ཟབས་སུ་ནི། །ཁྲུ་གང་པ་ནི་རྒྱས་པར་བཞེད། །རྒྱ་རུ་ཁྲུ་གང་ཟབས་སུ་ནི། །མཐོ་གང་པ་ནི་དབང་ལའོ། །རྒྱ་རུ་སོར་མོ་ཉི་ཤུ་ལ། །ཟབས་སུ་སོར་བཅུ་མངོན་སྤྱོད་ལ། །བསྐྲད་དང་དགུག་ལའང་དེ་བཞིན་ནོ། །ཞི་ལ་ཁ་དོག་དཀར་པོར་བྱ། །རྒྱས་པ་
ལ་ནི་སེར་པོའོ། །དབང་ལ་དམར་པོར་བྱ་བའོ། །མངོན་སྤྱོད་ལ་ནི་ནག་པོའོ། །དགུག་པ་ལ་ནི་དམར་པོར་བྱ། །བསྐྲད་པ་ལ་ནི་ནག་པོར་བྱའོ། །ཞི་ལ་ཁ་ཁྱེར་སོར་བཞིས་བསྐོར། །ནང་དུའང་སོར་བཞི་དོར་བར་བྱ། །རྒྱས་པ་ལ་ནི་སོར་བརྒྱད་དེ། །ནང་དུའང་དེ་བཞིན་དོར་བར་བྱ། །དབང་ལ་ཁ་ཁྱེར་སོར་
བཞིས་བསྐོར། །ནང་དུའང་སོར་བཞི་དོར་བར་བྱ། །མངོན་སྤྱོད་ལ་ནི་སོར་གསུམ་སྟེ། །ནང་དུའང་སོར་གསུམ་དོར་བར་བྱ། །དགུག་དང་བསྐྲད་ལའང་དེ་བཞིན་ནོ། །ཞི་བའི་ཐབ་ཁུང་དབུས་སུ་ནི། །རྡོ་རྗེ་དང་ནི་གྲི་གུག་བྲི། །ཁ་ཁྱེར་ལ་ནི་རྡོ་རྗེས་བསྐོར། །རྒྱས་ལ་དབུས་སུ་རིན་པོ་ཆེའོ། །ཁ་ཁྱེར་
ལའང་རིན་པོ་ཆེའོ། །དབང་ལ་དབུས་སུ་པདྨས་མཚན། །ཁ་ཁྱེར་དེ་དང་འདྲ་བས་བསྐོར། །མངོན་སྤྱོད་ལ་ནི་རལ་གྲིའོ། །ཁ་ཁྱེར་ལ་ཡང་རལ་གྲིས་བསྐོར། །དགུག་ལ་དབུས་སུ་ལྕགས་ཀྱུ་བཞག །ཁ་ཁྱེར་དེ་དང་འདྲ་བས་བསྐོར། །བསྐྲད་ལ་དབུས་སུ་རྡོ་རྗེ་ནི། །རྩེ་གསུམ་པ་ནི་
གཞག་པར་བྱའོ། །ཁ་ཁྱེར་ལ་ཡང་རྩེ་གསུམ་བྲི། །དབུས་སུ་གྲི་གུག་ཀུན་ལ་མཐུན། །ཐབ་ཁུང་ནང་གི་ཕྱོགས་བཞིར་ནི། །ཁ་ཊྭཱཾ་ག་དང་བདེ་གཤེགས་ཀྱི། །ཕྱག་མཚན་ཕྱི་མ་རིམ་བཞིན་གཞག །ཐབ་ཁུང་ཀུན་ལ་ཁ་དོག་དང་། །མཐུན་པའི་དྲི་ཡིས་བྱུག་པར་བྱ། །ཞི་བ་ཕལ་ཆེར་
