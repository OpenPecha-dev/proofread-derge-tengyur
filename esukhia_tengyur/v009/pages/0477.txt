གནས་འདིར་གཤེགས། །གནོད་སྦྱིན་གནོད་སྦྱིན་མོ་སྤྱན་འདྲེན། །ཧཱུྃ་ཧེ་ཧོཿཨཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ་ས་མ་ཡ་ཏི་བེ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། རྡོ་རྗེ་བཅིངས་བསྐོར་སྙིང་གར་གཞག །སྔགས་འདིས་མཉེས་པར་བྱས་ལ་བཏང་། །ཨོཾ་ད་ན་ཏི་བ་ཏི་བ་ཏི་ལ་ཕྲུཾ་སྭཱ་ཧཱ། དེ་ལ་གནོད་སྦྱིན་གྱི་གཏོར་
མའི་ཆོ་གའོ། །རྡོ་རྗེ་སྟབས་ཀྱིས་ལྷོར་གནས་ཏེ། །ཙནྡན་དམར་པོ་ལ་སོགས་པས། །ཕྱི་རོལ་དཀྱིལ་འཁོར་གྲུ་གསུམ་བྱས། །དེར་ནི་མེ་ཏོག་དམར་པོ་དགྲམ། །ཉ་སྐམ་དང་ནི་ཤ་སྐམ་དང་། །སྒོག་པའི་བི་ཁུ་ཁྲག་གི་སྣོད། །ཏིལ་གྱི་ཚྭ་མིག་ཁུ་བ་རྣམས། །ཁྱད་པར་དུ་ནི་བདུད་རྩི་ལྔ། །
རྣམ་པ་སྣ་ཚོགས་བཟའ་བ་དང་། །སྲན་མ་ལྔ་དང་སྲན་ཆན་གྱིས། །གང་བའི་སྣོད་ནི་མདུན་དུ་གཞག །འདོད་རྒྱལ་ཕྱག་རྒྱ་སྔགས་འདིས་དགུག །ཨོཾ་བ་ཏ་ལི་མ་ཧཱ་བ་ཏ་ལི། ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཛཿསྭཱ་ཧཱ། ཨོཾ་ཤིང་གཅིག་དང་ནི་དུར་ཁྲོད་དང་། །རི་དང་བྲག་དང་ཕུག་གིས། །ཡུལ་གྱི་འདབ་དང་ལམ་དང་
ཞིང་། །ཁྱད་པར་དུ་ནི་ཁྱིམ་སྟོང་དང་། །སྣོད་དང་ཐང་དང་ས་སྟེང་དང་། །ཁྱད་པར་གདོལ་པའི་ཁྱིམ་གནས་པ། །ནག་པོ་དྲག་པོ་དྲག་ཆེན་པོ། །ལྷ་སྦྱིན་མཉམ་དུ་བསྟེན་པ་དང་། །ནག་པོ་འཇིགས་བྱེད་འཇིགས་བྱེད་པ། །དགའ་བ་ལས་འདས་བགེགས་རྒྱལ་པོ། །རྒན་བྱེད་མ་དང་དྲག་མོ་
དང་། །འཇིགས་བྱེད་མ་དང་དབུ་མར་བསྟོད། །རྒྱལ་མོ་རྣམ་པར་རྒྱལ་མོ་དང་། །ཐུབ་མེད་གཞན་གྱིས་མི་ཐུབ་མའོ། །ནག་མོ་བཟང་མོ་ནག་ཆེན་མོ། །ནག་སྦོམ་མ་དང་རྣལ་འབྱོར་མ། །དབང་མོ་དང་ནི་ཟླ་བ་མོ། །དྲག་མོ་སྡང་བ་ལམ་བ་ཀི། །འཕྲོག་བྱེད་མ་དང་མར་མེ་མ། །འཇིགས་བྱེད་
ཡུལ་གནས་རྣལ་འབྱོར་མ། །དྲག་མོ་གཟུགས་ཆེན་མ་དང་ནི། །མཆེ་བ་གཙིགས་པ་ཐོད་ཆེན་མའོ། །ཐོད་ཕྲེང་བ་དང་ཁ་ཊྭཱཾ་མ། །སྦྲུལ་ཆེན་རལ་གྲི་དགྲ་སྟ་ཅན། །རྡོ་རྗེ་དེ་བཞིན་གཞུ་ལག་དང་། །ལྷ་ཡི་དབང་ཕྱུག་མ་རྣམས་དང་། །མཁའ་འགྲོ་ལྔ་ཡི་དེ་ཉིད་དང་། །ལས་ཀུན་རྗེས་སྒྲུབ་
བྱེད་པ་ཡི། །སྦྱོར་བའི་དཀྱིལ་འཁོར་རྒྱ་ཆེན་མོ། །དེ་བཞིན་རྡོ་རྗེ་དབང་ཕྱུག་གཙོ། །དེ་བཞིན་གཤེགས་པའི་སྐུ་ཆེན་པོ། །རྡུལ་མེད་སྦྱོར་བ་སྐྱེ་མེད་མ། །རྡོ་རྗེ་དབང་ཕྱུག་མ་བཀའ་ཡིས། །ཐམས་ཅད་ཐམས་ཅད་འོང་བར་བྱོས། །ཨོཾ་ཀཀྐ་ཌྷ་ན་བྷནྡྷ་ན། ཁ་ཁཱ་ད་ན་སརྦ་དུཥྚཾ། ཧ་ན་ཧ་ན་གྷཱ་ཏ་
