བསྲུངས་ན་དངོས་གྲུབ་པས། །རྡོ་རྗེ་བདུད་རྩི་ཆུ་འཐུངས་ཤིག །ཨོཾ་ཨཱཿསྭཱ་ཧཱ། །བདུད་རྩི་ལྔའི་ཆུ་བླུད་པའི་སྔགས་སོ། །དམ་ཚིག་གི་ནི་ཆུ་འཐུངས་ཏེ། །དད་པ་ཉེ་བར་གནས་པ་ལ། །ཅི་སྟེ་འདི་ནི་མི་བརྟན་པར། །དོགས་ཕྱིར་བླ་མས་བརྟན་པར་བྱ། །དེང་ཕྱིན་ཆད་ཁྱོད་ཀྱི་ཕྱག་ན་རྡོ་རྗེ་ང་
ཡིན་གྱིས། །ངས་འདི་བྱོས་ཤིག་ཅེས་བསྒོ་བ་གང་ཡིན་པ་དེ་བྱ་དགོས་སོ། །ཁྱོད་ཀྱིས་ང་ལ་བརྙས་པར་མ་བྱེད་ཅིག །མི་བདེ་བ་མ་སྤངས་པར་ཁྱོད་དུས་བྱས་ནས་དམྱལ་བར་ལྟུང་བར་གྱུར་ཏ་རེ་ཞེས་བརྗོད་པར་བྱའོ། །སེམས་ནི་རབ་ཏུ་དགའ་བྱའི་ཕྱིར། །གལ་ཏེ་འབེབས་པར་
བྱེད་ན་ནི། །སྙིང་གར་རླུང་གི་དཀྱིལ་འཁོར་ལ། །རླུང་གི་ས་བོན་རྣམ་པར་བསྒོམ། །རྐང་པའི་མཐིལ་ནས་རླུང་གིས་གདེག །དེ་ཡི་སྙིང་གར་བསྐུལ་བར་བྱ། །རང་གི་ལྷ་ཡི་སྦྱོར་བ་ཡི། །ང་རྒྱལ་ལྡན་པས་གསང་སྔགས་བཟླས། །སྔགས་ནི་ཨོཾ་ལོ་ལེ་ལ་ལེ་ཧཱུྃ་ཨཱཿསྔགས་པས་
ང་རྒྱལ་བསྐྱེད་ནས་ནི། །རྡོ་རྗེ་དྲིལ་བུ་དཀྲོལ་བྱས་ཏེ། །དེ་ཡི་སྔགས་ཀྱི་རྡོ་རྗེའི་གླུ། །འཇིགས་སུ་རུང་བས་གར་བྱའོ། །ཨོཾ་ཏིཥྛ་བཛྲ་ས་མ་ཡ་པྲ་ཏི་བ་ལིཾ་ཏ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཧོ་ཧོ་ཧོ་སྭཱ་ཧཱ། ཡེ་ཤེས་ཕབ་པ་ལས་བསྐྱོད་པའི་སྔགས་སོ། །དེ་ནས་སེ་གོལ་བརྡབ་བྱ་སྟེ། །མེ་ཏོག་ཕྲེང་བ་བཟང་པོ་
འདིས། །ཨོཾ་པྲ་ཏཱིཙྪ་བཛྲ་ཧོ་ཞེས་བརྗོད། །དཀྱིལ་འཁོར་སྙིང་པོར་རབ་ཏུ་དོར། །གལ་ཏེ་བཟང་པོར་མ་ལྷུང་ན། །ལན་གཉིས་བར་དུ་དོར་བར་བྱ། །དེ་ནས་དེ་ཡི་མགོར་བཅིངས་ཏེ། །ཕྲེང་བའི་སྦྱོར་བ་སྦྱིན་པར་བྱ། །འདི་ཡིས་གདོང་གཡོགས་དབྱེ་བ་སྟེ། །བླ་མས་ཚིགས་བཅད་
འདི་ལྟར་བརྗོད། །ཨོཾ་རྡོ་རྗེ་སེམས་དཔའ་དེང་ཁྱོད་ཀྱིས། །མིག་འབྱིན་པར་ནི་བརྩོན་པར་མཛོད། །མིག་ནི་ཐམས་ཅད་འབྱེད་མཛད་དེ། །རྡོ་རྗེའི་མིག་ནི་བླ་ན་མེད། །ཨོཾ་ཛྙཱ་ན་ཙཀྵུ་ཧཱུྃ་ཨ་སྭཱ་ཧཱ། །འདི་ལྟར་བརྗོད་རྗེས་མིག་ཕྱེ་སྟེ། །བླ་མའི་ཞབས་ལ་ཕྱག་འཚལ་ཏེ། །དེ་ནས་བླ་མས་
སློབ་མ་ལ། །འདི་ལྟར་ཡང་ནི་བརྗོད་པར་བྱ། །དཀྱིལ་འཁོར་འདི་ལ་ད་ལྟ་ཉིད། །དད་པའི་ཤུགས་ཀྱིས་བལྟ་བར་བྱ། །ཁྱོད་ཀྱང་སངས་རྒྱས་རིགས་ལས་སྐྱེས། །སྔགས་དང་ཕྱག་རྒྱས་བྱིན་གྱིས་བརླབས། །སྔོན་མ་བྱུང་བ་མཐོང་བ་ཡི། །སློབ་མ་འདོད་དང་ལྡན་པ་ཡིས། བླ་
