སོ། །འདོད་པའི་དོན་གྲུབ་བྱ་བ་དང་། །བགེགས་ཀྱི་བར་ཆད་ཞི་བྱའི་ཕྱིར། །སྔགས་དང་ཕྱག་རྒྱས་ལྷ་ལ་སོགས། །གཡས་བརྐྱང་བཀུག་ལ་གཏོར་མ་སྦྱིན། །རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་བཅིངས་ཏེ། །རྒྱབ་དང་རྒྱབ་ཏུ་སྦྱར་བར་བྱ། །ས་གདུགས་དང་ནི་པདྨ་སྦྲེལ། །རོ་ཙ་ན་དང་དཀར་
པོ་བརྐྱང་། །དཀར་པོ་བཀུག་ནས་འདྲེན་པ་ནི། །ལྷ་དང་ལྷ་མིན་གྲུབ་པ་དགུག །ཨོཾ་ཨཱཿབཱ་ཧ་ནི་མ་ཧཱ་བཱ་ཧ་ནི་པྲེ་ཏ་བཱ་ཧ་ནི་སཱུརྱ་ཙནྡྲ། སིདྷ་མུ་ཁི་ཛ་ཡ་རཱ་ཛཱི་སོ་མ་རཱ་ཛཱི། ཨུ་མ་དེ་བི་གངྒ་མ། སུ་རཱུ་པཱི་ཏཱ་ར་ཡ། སཾ་གྷ་གྲ་ཧཱ་ན་སརྦེ། ག་ཎ་པ་ཏི། ཀཱ་རི་ཏི་ཀ་ཡ་ཤྲཱི་ས་སི་དེ་བི། ཨ་ག་ལོ་པ་ག་ལ། ཨ་སུ་
རཱ་སུ་རཱ་ཛ་ཡོ། །པུཀྐ་སཱི་མ་ཧཱ་པུཀྐ་སཱི། ཧཱུྃ་ཧོ་ཧོ་ཨཱ་ཀཪྵ་ཡ་ཨཱ་ཀཪྵ་ཡ། ས་མ་ཡཱ་དྷི་པེ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཛ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའི་སྔགས་དང་ཕྱག་རྒྱའོ། །ཨོཾ་བ་ཏ་ལི་བ་ཏ་ལི་མ་ཧཱ་བ་ཏ་ལི་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ། བྷཀྵ་ཡ་བྷཀྵ་ཡ། ས་མ་ཡ་ཏིཥྛ་ཛ་རྫུ་ཕྲུཾ་ཕཊ་སྭཱ་ཧཱ། སྔགས་འདིས་སྤྱན་དྲང་ཕྱག་རྒྱ་ལས། །མཛུབ་
བཅིངས་སེ་གོལ་གསུམ་བརྡབས་གཏང་། །མཉེས་པར་བྱེད་པའི་སྔགས་དང་ཕྱག་རྒྱའོ། །དྲི་ཡི་དཀྱིལ་འཁོར་གྲུ་བཞི་པ། །མེ་ཏོག་དཀར་པོ་དགྲམ་པར་བྱ། །ཤཱ་ལིའི་བཟའ་བ་རྒྱ་ཆེན་པོ། །བུ་རམ་ཞོ་དང་འབྲས་ཆན་མར། །མར་མེ་སོགས་མཆོད་རོལ་སྟབས་ཀྱིས། །ཤར་སྒོ་བ་ཡི་གཏོར་མ་
སྦྱིན། ཨོཾ་ཨེ་ཀ་བཱི་ཀེ་སྨ་ཤཱ་ནེ་པ། པརྦ་ཏེ་ཀནྡ་རེ་གྲྀ་ཧེ་གྲཱ་མ་པཱ་ར་ཤེ་པ་ཀྲེ་ཐེ་ཏྲེ་ཤུ་ནྱ་ག་རེ་བི་ཤི་ཥཏ། བྷྭ་ཛྭ་ནེ་སྛ་ལག་ཏེ། མཱ་ཏ་དྷ་ན་བི་ཤེ་ཥ་ཏཿཀྲྀཥྞ་རུ་དྲ་དེ་བ་དཏྟེ། ས་མཱ་ཤྲཱི་ཏ་ཀྲྀཥྞ་ཨཱ་ཀ་རཱ་ལ་བྷི་བཱ་ཙ་ནནྡ་ཏཱི་ཏ་བི་ནཱ་ཡ་ཀ །ཙ་མུཎྜ་གཽ་རཱི་བི་བྷཙྪ། ཨུ་མ་དེ་བཱི་ཙ་མ་ཏཱ་ར། ཛ་ཡཱན་ཙ། བི་ཛ་ཡཱ་ཎ་ཙེ་བ། ཨ་ཛི་
ཏ། ཨ་པ་རཱ་ཛི་ཏཱ། བྷ་དྲ་ཀཱ་ལེ་མ་ཧཱ་ཀཱ་ལི་སྠཱུ་ལ་ཀཱ་ལི་ཏུ་ཛོ་གི་ནཱིནྡྲི། ཙནྡྲི་གཽ་རཱི། དུཥྚ་ལཾ་བ་ཀི། ཏྲི་ད་ཤི་ཤྭ་རི་ཀཾ་བོ་ཙི་དཱ་པཱི་ཙུཾ་པི་ནྱཱ་གྲཱ་མཱ་ཝསྠི་ཏ་ཡོ་གི་ནི་གྷོ་ར་རཱུ་པི། མ་ཧཱ་རཱུ་པི། དཾ་ཥཱ་རཱུ་པི། ཀ་པཱ་ལི་མཱ་ལཱ་མཱ་ལི་ནྱ་ཁ་ཌངྒི་ན། མ་ཧཱ་དྷི་ཀ །ཁངྒ་པ་ར་ཤུ་ཧ་ཏ་ཡ། བཛྲ་ཧསྟ། དྷ་ནུ་སྟ་ཐཱ། མ་ཧཱ་སུ་ཁ་
མ་ཧཱ་བཛྲ་སརྦྦ་ཀརྨ་མཱ་ནུ་སཱ་དྷ་ཀ། ཡོ་ག་མཎྜལ་མ་ཧཱ་རཱ་ཛཱ། བཛྲཱ་དྷ་པ་ཏི་བྲ་བྷུ་ཐ། ཏ་ཐཱ་ག་ཏ་མ་ཧཱ་ལ་ལ་ནི་རཉྫ་ཡོ་གི་དྷྲྀཥྚི་ཀཱ། ཨི་དཾ་མ་ཧཱ་སུ་ཁ་ཨཛྙེ་ནཱ། ཨ་བ་ཧ་ནཾ་སརྦྦ་སརྦི་ཎི། ཨོཾ་ཀཀྐ་ཌྷ་ན། བབྦནྡྷ་ན། ཁཁྑཱ་ནི། སརྦྦ་དུཥྚཱ་ཎཱཾ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་ཕཊ་ཛཿསྭཱ་ཧཱ། རྡོ་རྗེའི་སྟབས་དང་སྔོན་ཕྱག་རྒྱར། །
