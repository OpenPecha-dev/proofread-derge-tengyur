མར་པདྨ་འདབ་མ་བརྒྱད་པའི་སྟེང་དུ་ཨཱཿདམར་པོ་བསམ། སྙིང་གར་ཉི་མའི་དཀྱིལ་འཁོར་གྱི་སྟེང་དུ་ཧཱུྃ་སྔོན་པོ་བསམས་ལ། ཨོཾ་ཨཱཿཧཱུྃ་རཀྵ་རཀྵ་ཧཱུྃ་ཧཱུྃ་ཧཱུྃ་ཕཊ་སྭཱ་ཧཱ། ཞེས་བྱ་བས་བསྲུངས་ཏེ། བདུད་རྩི་འམ། བའི་རྣམ་ལྔ་ལ་སོགས་པ་ལ་ཨོཾ་ཨཱཿཧཱུྃ་བཟླས་ལ།
ཨོཾ་བཛྲ་རེ་ཁེ་ཧཱུྃ་ཞེས་ཟློས་ཤིང་། མཎྜལ་བྱུག་པར་བྱའོ། །དེ་ནས་ལག་པ་བཤལ་ཏེ་ཐལ་མོ་སྦྱར་ནས་ཕན་ཡོན་བརྗོད་པར་བྱའོ། །ཆུ་བཅས་ལྕི་བ་སྦྱིན་པ་ཡིན། །སྐྱང་ནུལ་བྱས་པ་ཚུལ་ཁྲིམས་ཡིན། །གྲོག་མ་སོགས་བསལ་བཟོད་པ་ཡིན། །བྱ་བ་བརྩམས་པ་བརྩོན་འགྲུས་ཡིན། །
སྐད་ཅིག་རྩེ་གཅིག་བསམ་གཏན་ཡིན། །རི་མོ་སྙོམས་པ་ཤེས་རབ་ཡིན། །ཐུབ་པའི་མཎྜལ་བྱས་པས་ཕ་རོལ་ཕྱིན་དྲུག་དེ་དག་རྙེད་པར་འགྱུར། །ལྷ་མི་ཁྱད་པར་ཅན་སྐྱེ་ཟླ་ལྟར་གཟི་འཕགས་ཤིང་། །ནད་ཀུན་ལས་གྲོལ་གསེར་གྱི་མདོག་དང་འདྲ་བར་འགྱུར། །རྒྱལ་རིགས་འབྲུ་
དང་གསེར་སོགས་འབྱོར་པ་ལྡན་པར་སྐྱེ། །དེས་ན་བདེ་གཤེགས་མཆོག་གི་ཁྱིམ་དུ་ལུས་ལས་བྱ། །ཞེས་བརྗོད་དོ། །དེ་ནས་ལག་པ་གཡོན་པ་ས་ལ་རེག་པར་བྱས་ལ། ཨོཾ་བཛྲ་མེ་དཱི་ནཱི་བཛྲཱི་བྷ་བ་བཛྲ་བནྡྷ་ཧཱུྃ་ཞེས་བརྗོད་པ་ཙམ་གྱིས་ས་གཞི་དེ་སྣ་ཚོགས་རྡོ་རྗེར་གྱུར་པར་བསམ་མོ། །
དེ་ནས་མེ་ཏོག་གི་ཚོམ་བུ་དཀྱིལ་འཁོར་དུ་དགོད་པར་བྱ་སྟེ། དབུས་སུ་ཨོཾ་ཨཱཿཧཱུྃ་ན་མཿ། ཞེས་བརྗོད་དེ་མེ་ཏོག་དོར་རོ། །ཤར་དུ་གྃ་གཽ་རཱི་ན་མཿ། ཞེས་བྱ་བས་མེ་ཏོག་དོར་རོ། །ལྷོར་རྩྃ་རྩ་རཱི་ན་མཿ། ཞེས་བྱ་བས་མེ་ཏོག་དོར་རོ། །ནུབ་ཏུ་པྃ་བེ་ཏཱ་ལཱི་ན་མཿ། ཞེས་བྱ་བས་དོར་རོ། །བྱང་དུ་གྷྃ་གྷསྨ་
དཱི་ན་མཿ། ཞེས་བྱ་བས་དོར་རོ། །དབང་ལྡན་དུ་པྃ་པུཀྐ་སཱི་ན་མཿ། ཞེས་བྱ་བས་དོར་རོ། །མེ་མཚམས་སུ་སཾ་ས་ཝ་རཱི་ན་མཿ། ཞེས་བྱ་བས་དོར་རོ། །བདེན་བྲལ་དུ་ལྃ་ཙཎྜ་ལཱི་ན་མཿ། ཞེས་བྱ་བས་སོ། །རླུང་མཚམས་སུ་ནི་ཌྃ་ཌོམྦི་ནཱི་ན་མཿ། ཞེས་པས་སོ། །དེ་དག་ནི་མེ་ཏོག་གི་ཚོམ་
བུའི་དཀྱིལ་འཁོར་རོ། །དེ་ནས་བདག་ཉིད་ཀྱི་སྙིང་གའི་ཧཱུྃ་ལས་འོད་ཟེར་འཕྲོས་པས་ཕྱོགས་བཅུའི་སངས་རྒྱས་ཐམས་ཅད་ཀྱི་ཤངས་བུག་གཡོན་པ་ནས་ཞུགས་ཏེ། ཐུགས་ཀའི་པདྨ་ལ་བསྣུན་པ་ཙམ་གྱིས། ཐུགས་བསྐུལ་ནས་ཕྱོགས་བཅུའི་སངས་རྒྱས་ཐམས་ཅད་དགྱེས་
