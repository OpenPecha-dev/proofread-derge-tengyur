བསླབ་པ་གཏོང་ཞིང་འདོར་བར་འགྱུར་རོ་ཞེས་གསུངས་ཏེ། དེས་ན་ཚིགས་སུ་བཅད་པ་གཉིས་པོ་འདི་གསུངས་པ་ཡིན་ནོ། །སྡུག་བསྔལ་གྱིས་རྒལ་བར་བྱ་བ་ཡིན་པས་རྒལ་དཀའ་སྟེ་འདའ་བར་དཀའ་བའོ། །དེ་ཐོབ་པའི་ཐབས་ནི་ཐོབ་པར་དཀའ་བའི་ཕྱིར་རོ། །བཟོད་པར་དཀའ་བ་ཞེས་བྱ་
བ་ནི་བཟོད་པར་སླ་བ་མ་ཡིན་པ་སྟེ། གྲོགས་དང་གནས་ཀྱི་ཁྱད་ལ་རག་ལས་པའི་ཕྱིར་རོ། །ཡང་ན་གཏིང་དཔག་དཀའ་བའི་ཕྱིར་ཕ་རོལ་ཏུ་རྒལ་དཀའ་བའོ། །གང་གི་ཕྱིར་དགེ་སྦྱོང་གི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ནི་དཀའ་བས་འཐོབ་པ་ཡིན་ཏེ། རྒྱ་མཚོ་ལྟ་བུའོ། །དེའི་ཕ་རོལ་ནི་ཐར་པའོ། །བཟོད་
དཀའ་དང་ཞེས་བྱ་བ་ནི་འདི་ལ་སྡུག་བསྔལ་གྱིས་ཐེག་ཅིང་བཟོད་པ་ཡིན་པས་བཟོད་དཀའ་བ་སྟེ། བདག་ཉིད་མ་སྦྱངས་པ་དག་དུས་དང་གནས་སྐབས་རྒྱུན་དུ་བསྲུང་བ་མི་བཟོད་པའི་ཕྱིར་རོ། །དེ་ལ་གནས་པ་ན་ཡང་དགའ་བ་ཉི་ཚེར་མ་ཟད་དེ་མཐར་ཐུག་པར་རྟོགས་པར་དཀའ་བའི་ཕྱིར་དང་གི་སྒྲ་
སྨོས་སོ། །འདིར་དགེ་སློང་ནི་རབ་ཏུ་བྱུང་བའི་ཚུལ་ཁྲིམས་སོ། །དེ་ཡང་ཀུན་རྫོབ་དང་དོན་དམ་པའི་དབྱེ་བ་རྣམ་པ་གཉིས་སོ། །གང་གིས་རྒལ་དཀའ་བ་དང་། བཟོད་པར་དཀའ་བ་ཡིན་ཞེ་ན། རྣམ་པར་མ་ཤེས་པས་ཞེས་བྱ་བ་སྟེ། དངོས་པོའི་དེ་བཞིན་ཉིད་ལ་ཀུན་དུ་རྨོངས་པ་སྟེ། དེའི་ངོ་བོ་
དང་རྣམ་པ་ལ་སོགས་པ་མ་རྟོགས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཇི་ལྟ་བུ་ཡིན་ཞེ་ན། སྡུག་བསྔལ་མང་པོ་དེ་དག་འབྱུང་། །ཞེས་བྱ་བ་སྨོས་ཏེ། རབ་ཏུ་བྱུང་བ་ལ་གནོད་པ་དཔག་ཏུ་མེད་པ་འབྱུང་སྟེ། ཉམས་པའི་རྒྱུ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཇི་སྐད་དུ། གཞན་གྱི་ངོར་བལྟས་བསགས་
ཤིང་འཚོ་བ་དང་། །ངན་པ་བཞིན་དུ་སྐྱེ་བོས་མ་བརྩིས་ཤིང་། །གོས་ངན་ཉེ་དུ་གྲོགས་བཤེས་རབ་བྲལ་བས། །ནགས་མཐར་ལུས་བོར་བ་ཡི་དུ་བ་བཞིན། །ཞེས་བྱ་བ་ལྟ་བུའོ། །གང་དག་ཅེས་བྱ་བ་ནི་གང་དག་ལ་སྡུག་བསྔལ་བཟོད་པར་དཀའ་བ་དག་གིས་ཟིལ་གྱིས་མནན་པར་
གྱུར་པ་ནའོ། །ཞན་པ་ནི་རྨོངས་པ་སྟེ། ཡོན་ཏན་དང་སྐྱོན་གཉིས་ལ་རྟོག་པ་མེད་པའི་ཕྱིར་ཏེ། ཇི་སྐད་དུ། རབ་ཏུ་བྱུང་ལ་སྐྱེ་བོ་ཕལ་ཆེར་ནི། །མ་གོམས་པས་ན་རབ་ཏུ་སྐྱེ་བར་གྱུར། །འདོད་པའི་འདོད་ཆགས་ཀྱི་ནི་རྗེས་འབྲང་བས། །གཡང་ས་བཞིན་དུ་རབ་ཏུ་རྟོག་པར་
