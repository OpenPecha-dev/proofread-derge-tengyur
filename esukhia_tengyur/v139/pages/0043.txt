ནང་ན་རྒན་རབས་ལ་རིམ་གྲོ་བྱེད་པ་དང་། དོན་བྱེད་པ་དང་། བྱ་བའི་རིགས་པ་བྱེད་པ་དང་། བསོད་ནམས་བྱེད་པ་དང་། ཚེ་འདི་དང་ཚེ་ཕྱི་མ་ལ་ཁ་ན་མ་ཐོ་བ་རྣམས་ཀྱིས་འཇིགས་པར་ལྟ་བ་དང་། སྦྱིན་པ་དག་སྦྱིན་ཞིང་། བསོད་ནམས་དག་བྱེད་པ་དང་། གསོ་
སྦྱོང་ལ་ཉེ་བར་གནས་པ་དང་། ཚུལ་ཁྲིམས་ཡང་དག་པར་བླངས་ནས་འདུག་པ་དག་མང་ངམ་ཞེས་འཇིག་རྟེན་འདི་དག་ཐམས་ཅད་ཉུལ་ལོ། །ཟླ་བ་ཕྱེད་ཀྱི་ཚེས་བཅྭ་ལྔ་ལ་ནི་རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་དེ་དག་ཉིད་ཅི་མི་དང་མི་མ་ཡིན་པའི་འཇིག་རྟེན་ན་མར་འཛིན་པ་དང་། ཕར་འཛིན་པ་དང་།
དགེ་སྦྱོང་དུ་འཛིན་པ་དང་། བྲམ་ཟེར་འཛིན་པ་དང་། རིགས་ཀྱི་ནང་ན་རྒན་རབས་ལ་རིམ་གྲོ་བྱེད་པ་དང་། དོན་བྱེད་པ་དང་། བྱ་བའི་རིགས་པ་བྱེད་པ་དང་། བསོད་ནམས་བྱེད་པ་དང་། ཚེ་འདི་དང་ཚེ་ཕྱི་མ་ལ་ཁ་ན་མ་ཐོ་བ་རྣམས་ཀྱིས་འཇིགས་པར་ལྟ་བ་དང་། སྦྱིན་པ་དག་སྦྱིན་ཞིང་བསོད་
ནམས་དག་བྱེད་པ་དང་། གསོ་སྦྱོང་ལ་ཉེ་བར་གནས་པ་དང་། ཚུལ་ཁྲིམས་ཡང་དག་པར་བླངས་ནས་འདུག་པ་དག་མང་ངམ་ཞེས་འཇིག་རྟེན་འདི་ཐམས་ཅད་ཉུལ་ལོ། །ཅི་སྟེ་མི་དང་མི་མ་ཡིན་པའི་འཇིག་རྟེན་ན་མར་མི་འཛིན་པ་དང་། ཕར་མི་འཛིན་པ་དང་། དགེ་སྦྱོང་དུ་མི་འཛིན་པ་དང་།
བྲམ་ཟེར་མི་འཛིན་པ་དང་། རིགས་ཀྱི་ནང་ན་རྒན་རབས་ལ་རིམ་གྲོ་མི་བྱེད་པ་དང་། དོན་མི་བྱེད་པ་དང་། བྱ་བའི་རིགས་པ་དག་མི་བྱེད་པ་དང་། བསོད་ནམས་མི་བྱེད་པ་དང་། ཚེ་འདི་དང་ཚེ་ཕྱི་མ་ལ་ཁ་ན་མ་ཐོ་བ་རྣམས་ཀྱིས་འཇིགས་པར་མི་ལྟ་བ་དང་། སྦྱིན་པ་དག་མི་སྦྱིན་ཞིང་བསོད་
ནམས་དག་མི་བྱེད་པ་དང་། གསོ་སྦྱོང་ལ་ཉེ་བར་མི་གནས་པ་དང་། ཚུལ་ཁྲིམས་ཡང་དག་པར་བླངས་ཏེ། མི་འདུག་པ་དག་མང་ན་ནི་རྒྱལ་པོ་ཆེན་པོ་བཞི་པོ་རྣམས་སུམ་ཅུ་རྩ་གསུམ་པའི་ལྷ་རྣམས་ལྷའི་འདུན་ས་ཆོས་བཟངས་སུ་འདུས་ཤིང་འཁོད་པའི་ནང་དུ་སོང་ནས་འདི་སྐད་དུ་
གྲོགས་པོ་དག་འདི་ཤེས་པར་གྱིས་ཤིག །མི་དང་མི་མ་ཡིན་པའི་འཇིག་རྟེན་ན་མར་མི་འཛིན་པ་དང་། ཕར་མི་འཛིན་པ་དང་། དགེ་སྦྱོང་དུ་མི་འཛིན་པ་དང་། བྲམ་ཟེར་མི་འཛིན་པ་དང་། རིགས་ཀྱི་ནང་ན་རྒན་རབས་ལ་རིམ་གྲོ་མི་བྱེད་པ་དང་། དོན་མི་བྱེད་པ་དང་། བྱ་བའི་རིགས་
