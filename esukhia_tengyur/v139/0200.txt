རྒྱལ་བ་ཞེས་བྱ་བ་ནས་རྒྱས་པར་སྦྱར་ཏེ། ཚེ་རིང་ཞིང་ཡུན་རིང་དུ་གནས་པར་འགྱུར་བ་དང་། བཀུར་སྟིར་བྱེད། བླ་མར་བྱེད། རི་མོར་བྱེད། མཆོད་པར་བྱེད་ཅེས་བྱ་བའི་བར་དུ་རྒྱས་པར་སྦྱར་རོ། །གལ་ཏེ་སྐྲ་དང་། ཁ་སྤུ་བྲེགས་ཏེ་ཞེས་བྱ་བ་ནས། རབ་ཏུ་བྱུང་ན་ནི་འཇིག་
རྟེན་དུ་དེ་བཞིན་གཤེགས་པ་དགྲ་བཅོམ་པ་ཡང་དག་པར་རྫོགས་པའི་སངས་རྒྱས་སུ་འགྱུར་ཏེ། སྒྲ་ཆེར་གྲགས་པར་འགྱུར་རོ་ཞེས་བྱ་བའི་བར་དུ་གསུངས་ཏེ། བཅོམ་ལྡན་འདས་ཀྱིས་བཞེད་ན་བསྐལ་པ་འམ། བསྐལ་པ་ལས་ལྷག་པར་ཡང་སྐུ་ཚེ་རིང་ཞིང་ཡུན་རིང་དུ་
གནས་པར་འགྱུར་བ་དང་། བཀུར་སྟིར་བྱེད། བླ་མར་བྱེད་ཅེས་བྱ་བའི་བར་དུ་རྒྱས་པར་སྦྱར་ཏེ། འདི་ནི་དེའི་སྔ་ལྟས་ཡིན་ནོ། །འདི་ལ་ཆོས་ཉིད་ནི་འདི་ཡིན་ཏེ། དེ་ལ་འདི་སྐད་ཅེས་བྱའོ། །གང་ཚེ་རྒྱལ་བ་སོར་མོ་རིང་བར་གྱུར་པ་དང་། །ཞབས་ཀྱི་རྟིང་ཡངས་སྐུ་ཆེ་
དྲང་བར་གྱུར་པ་ནི། །སྲོག་ཆགས་སྲོག་གཅོད་པ་ལས་སླར་ནི་ལོག་གྱུར་དང་། །མ་བྱིན་པ་དང་རྐུ་བ་སྤངས་པར་གྱུར་པ་དང་། །དགེ་སྦྱོང་བྲམ་ཟེ་རྣམས་ལ་གསོང་བོར་སྨྲ་བ་དང་། །དེས་ནི་བླ་མ་རྣམས་ལ་ལྡང་ཞིང་སྟན་བྱིན་པའི། །ལས་དགེ་བདེ་བ་སྐྱེད་པ་རྣམ་
སྨིན་དགེ་བ་དེ། །སྐྱེ་བ་གཞན་དག་ཏུ་ནི་སྟོན་པ་མཛད་པའི་ཕྱིར། །སྟོན་པ་དེ་ལ་ལས་ཀྱི་རྣམ་སྨིན་དེ་འདྲ་བའི། །མཚན་རྣམས་སྐུ་ལས་བྱུང་སྟེ་གསུམ་དང་གསུམ་དུ་སྦྱར། །ཅིའི་ཕྱིར་བྱང་ཆུབ་སེམས་དཔས་སྐྱེས་བུ་ཆེན་པོའི་མཚན་ལག་པ་དང་རྐང་པ་དྲ་བ་དང་ལྡན་
པ་ཐོབ་ཅེ་ན། སྨྲས་པ། བྱང་ཆུབ་སེམས་དཔས་ནི་སྔོན་གྱི་སྲིད་པ་ཞེས་བྱ་བ་ནས་རྒྱས་པར་སྦྱར་ཏེ། བསྡུ་བའི་དངོས་པོ་བཞིས་འཁོར་བསྡུས་ཏེ། འདི་ལྟ་སྟེ། སྦྱིན་པ་དང་། སྙན་པར་སྨྲ་བ་དང་། དོན་སྤྱོད་པ་དང་། དོན་མཐུན་པས་སོ། །དེ་ལས་དེ་དང་ཡང་དག་
པར་བླངས་པ་དེ་ལྟ་བུ་དང་། ཡང་དག་པར་གཟུང་བ་དེ་ལྟ་བུས་ལུས་ཞིག་ནས་དགའ་ལྡན་གྱི་ལྷ་རྣམས་དང་སྐལ་བ་མཉམ་པར་སྐྱེའོ་ཞེས་བྱ་བ་ནས་རྒྱས་པར་སྦྱར་ཏེ། དེ་དེ་ནས་ཤི་འཕོས་ནས་ཁམས་འདིར་མི་རྣམས་དང་སྐལ་བ་མཉམ་པར་སྐྱེའོ་ཞེས་བྱ་བའི་བར་དུ་སྦྱར་རོ། །
