ཁ་དོག་དང་བཅས་པས་ཀྱང་ཁ་དོག་སྣ་ལྔ་པ་སྟེ་རིན་ཆེན་ཆེན་པོའོ། །ཡུངས་ཀར་སྦོམ་པོའི་ཚད་ཙམ་འདི་ལ་ཡོད་པས་ན་དེ་སྐད་ཅེས་བྱ་སྟེ། སྣའི་རྩེ་མོ་ཞེས་བྱ་བ་ནི་བསྒོམ་ཞེས་བྱ་བ་དང་འབྲེལ་ཏོ། །ནན་ཏན་དུ་ཞེས་བྱ་བ་ནི་ཆོས་ཀྱི་དབྱིངས་ཀྱི་སྔགས་ཀྱི་བྱིན་གྱི་རླབས་ཏེ། དེ་ནས་
བསླངས་ལ་སྣའི་རྩེ་མོར་གཞག་པའོ། །འདི་ནི་སྲོག་དང་རྩོལ་བ་ཞེས་བྱ་བ་ཡན་ལག་གསུམ་པའོ། །སྦྱོར་བ་པ་ཡིས་རྟག་ཏུ་བསྒོམ། །ཞེས་བྱ་བ་ལ་སྦྱོར་བ་རྣམས་ནི་རིག་པ་རྣམས་དང་སྦྱོར་བ་སྟེ། སྙིང་གའི་གདན་ལ་རང་གི་སྔགས་བསམས་ཏེ་དེའི་ཐིག་ལེར་རིན་པོ་ཆེ་ཕབ་ལ་དེར་རིག་པ་རྣམས་
དང་སྦྱར་ནས་ངེས་པར་བསྒོམ་པའོ། །རྟག་ཏུ་ཞེས་བྱ་བ་ནི་ཇི་སྲིད་དུ་བརྟན་པར་གྱུར་པ་དེ་སྲིད་ཀྱི་བར་དུ་བསྒོམ་པར་བྱ་བ་སྟེ། འདི་ནི་འཛིན་པའི་ཡན་ལག་བཞི་པའོ། །རྗེས་སུ་དྲན་པ་ནི་སྤྲོ་བའོ། །རེ་ཞིག་དེའི་དུས་གསུངས་པ་རིན་ཆེན་ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། ཕྲ་མོ་བརྟན་པའི་མཚན་
མ་ནི་ཀུན་དུ་སྣང་བའི་བྱེ་བྲག་དག་ཡིན་ན་སྣང་བའི་བྱེ་བྲག་དེ་དག་འདིར་ཅི་འདྲ་བ་ཞིག་སྣང་ན་མཚན་མ་སྤྲོ་བར་བྱ་ཞེ་ན། སངས་རྒྱས་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ་སངས་རྒྱས་ནི་སྟོང་པ་ཉིད་ཀྱི་ནམ་མཁའ་གསལ་བར་བྱེད་པ་ཡིན་ལ། སྲིན་བུ་མེ་ཁྱེར་ཡང་དེ་དང་འདྲ་བར་ནམ་མཁའ་གསལ་བར་
བྱེད་པས་དེ་སྐད་ཅེས་བྱའོ། །འབར་བ་འདྲ་བའི་འོད་ཅེས་བྱ་བ་ལ་མེ་མར་གྱི་རྩེ་མོ་ལྟ་བུ་སྟེ། ཀུན་དུ་འོད་འབར་བའི་ཕྱིར་རོ། །རབ་མཆོག་ཅེས་བྱ་བ་ནི་སྨིག་རྒྱུ་ལྟ་བུ་སྟེ་རབ་མཆོག་གི་ཆོས་ཅན་གྱི་ནོར་བུ་ལས་བྱུང་བའི་ཕྱིར་རོ། །སྤྲིན་ཞེས་བྱ་བ་ནི་སྤྲིན་དང་འདྲ་བ་སྟེ་དུ་བ་ལྟ་བུའོ། །འོད་ཅེས་བྱ་བ་ནི་འོད་
གསལ་བ་སྟེ། སྤྲིན་མེད་པའི་ནམ་མཁའ་ལྟར་སྣང་བ་དང་བཅས་པ་ཞེས་བྱ་བའི་དོན་ཏོ། །འཁོར་ལོ་ཞེས་བྱ་བ་ནི་དེ་ནས་རིན་པོ་ཆེ་རྡོ་རྗེ་ནོར་བུར་ཕབ་སྟེ་བཞག་ལ་དེ་ལས་པདྨའི་ནང་དུ་འོད་ཀྱི་ཐིག་ལེ་ཕྱུང་བས་སྐུ་གཉིས་ཀྱི་ནང་དུ་ཁྱབ་པར་བྱས་ཏེ། བ་སྤུའི་བུ་ག་རྣམས་ནས་འཁོར་ལོ་ལ་སོགས་པར་
སྤྲུལ་ཏེ་སྤྲོ་བར་བྱའོ། །དེ་ཉིད་ཀྱི་ཕྱིར་གསུངས་པ་འཁོར་ལོ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྟེ། འཁོར་ལོ་དང་རྡོ་རྗེ་དང་པདྨ་དང་ཤུགས་ཅན་ཞེས་བྱ་བ་རལ་གྲི་དང་། མཚོན་མཆོག་སྟེ་གཙུག་ཏོར་འཁོར་ལོས་སྒྱུར་བའི་འཁོར་ལོ་རྣམས་ཀྱི་སྤྲིན་དེ་དང་དེ་དག་ནི་སངས་རྒྱས་དང་རྡོ་རྗེ་དང་། ཆོས་དང་རིགས་དང་ཁྲོ་བོ་
