རྣམ་པར་སྤྱོད་པ་ལ། །དེ་བཞིན་ཡང་དག་གྲུབ་པ་ཡིན། །དོན་ལ་དེ་ཉིད་དེ་ཉིད་མིན། །ཀུན་རྫོབ་རྣམས་ལ་དགག་པ་བྱེད། །རྗེས་དཔག་དཔག་པར་བྱ་བའི་དོན། །འདི་ནི་ཐ་སྙད་པ་ལ་གནས། །ཐ་སྙད་རྟོགས་པས་ཡང་དག་གྲུབ། །དམིགས་པར་བྱས་ནས་རྟོགས་པ་ཡིན། །ཅི་ལྟར་དབྱེ་བར་
རབ་གནས་དང་། །ཤེས་པ་རྣམས་ལའང་རྣམ་དབྱེ་བ། །ཆོས་ཅན་ཆོས་ཀྱང་སྣང་བ་ཡིན། །ཐ་སྙད་དེ་ལ་བརྟེན་པའོ། །རང་གིས་རིག་པའི་མངོན་སུམ་གྱིས། །གྲུབ་ཉིད་དེ་ལྟར་ཡིན་ན་ནི། །འབྲས་བུ་རྒྱུ་ཡི་དངོས་པོ་འདི། །ཀུན་དུ་རྣལ་འབྱོར་པ་ཡིས་རྟོགས། །རྒྱུ་དང་འབྲས་བུའི་དངོས་
པོ་ནི། །ཚད་མ་ཡིས་ནི་རབ་གྲུབ་མིན། །དེ་ཕྱིར་གཏན་ཚིགས་འདི་གྲུབ་མིན། །དངོས་པོ་ཐམས་ཅད་སྐུར་འདེབས་པའི། །རྗེས་སུ་འགྲོ་དང་ལྡོག་དག་གིས། །གང་ཞིག་འཇིག་རྟེན་པས་གྲུབ་པ། །དེ་ནི་བརྟགས་པའི་རང་བཞིན་ལས། །གནོད་པར་བྱེད་དང་སྒྲུབ་བྱེད་གྲུབ། །འདི་དག་ཅི་
ལས་གྲུབ་ཅེ་ན། །སྔར་ཉིད་རང་གི་རིག་པ་ལས། །དེ་ལས་གཞན་པ་མི་དམིགས་པས། །དེ་ཉིད་འདི་ཞེས་ངེས་པ་ཡིན། །དེ་ཕྱིར་ལན་ཅིག་མཐོང་གྱུར་པ། །དེ་ཉིད་གཞན་དུའང་དེ་ཡིན་ཏེ། །གང་ཞིག་རྒྱུ་དང་འབྲས་བུའི་དངོས། །འདི་ལྟར་ཉིད་དུ་གྲུབ་པ་ཡིན། །གང་ཞིག་རྟེན་ཅིང་འབྲེལ་
འབྱུང་བ། །དེ་ནི་དེ་ཉིད་མིན་ཞེས་པ། །བདེན་མིན་དཔེ་དང་གནོད་མི་གྲུབ། །འབྲེལ་བ་ཉིད་ཀྱང་གྲུབ་མིན་ཕྱིར། །གཟུགས་བརྙན་ལ་སོགས་དཔེ་ཞེ་ན། །མ་ཡིན་རྣམ་རིག་ཙམ་སྨྲ་ལ། །རང་གི་སེམས་ལས་མ་གཏོགས་པ། །གཟུགས་ལ་སོགས་པ་མ་གྲུབ་ཕྱིར། །དེ་ཉིད་རབ་ཏུ་སྣང་ལ་
སོགས། །ཀུན་རྫོབ་ཉིད་དུ་བསྒྲུབ་པ་ལ། །ཁྱབ་པ་ཉིད་ནི་མ་གྲུབ་ཕྱིར། །གཏན་ཚིགས་མ་ངེས་པ་ཡིན་ནོ། །ཡང་དག་གཏན་ཚིགས་ཀྱང་བརྗོད་བྱ། །ཐམས་ཅད་ཀུན་རྫོབ་གྲུབ་པའི་ཕྱིར། །མ་གྲུབ་ལ་སོགས་སྐྱོན་མི་བསྟན། །ཁྱེད་ལ་ཡང་དག་དེ་ཡོད་མིན། །དངོས་པོ་ཀུན་ལ་
སྐུར་འདེབས་པ། །ཁྱེད་ལ་དངོས་སྟོབས་གཏན་ཚིགས་མེད། །དངོས་པོ་མེད་ལ་གཏན་ཚིགས་ཉིད། །མེད་ཅེས་རིགས་པ་ཤེས་པས་བརྗོད། །དེ་དོན་སྨོས་པས་སྒྲ་དག་དང་། །རྟོགས་པ་སྒྲོ་བཏགས་བདག་ཉིད་ཅན། །རྟགས་མིན་རབ་ཏུ་གྲུབ་དོན་ཏེ། །དོན་ལས་དོན་ནི་རབ་གྲུབ་
