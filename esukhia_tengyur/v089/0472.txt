སྒྲིབ་པ་བྲལ་བ་སྟེ། །བྱེ་བྲག་མཚན་ཉིད་དཀྱུས་ན་གསལ། །གཞི་ཞེས་བསྟན་པར་བྱ་བ་ནི། །རྐང་པ་གཉིས་ཀྱིས་ངོ་བོ་བསྟན། །བཞི་ཡིས་གཞི་ཤེས་དབྱེ་བ་སྟེ། །ལྷག་མ་གཞི་ཤེས་དངོས་ཡིན་ཞེས། །དེ་སྐད་ཐ་སྙད་འདོགས་པ་ཡོད། །གཞི་ཤེས་
གཉིས་ཀ་བསྟན་ཀྱང་ནི། །བསྒོམ་ལམ་སྦྱོར་བ་ཟེར་བའང་ཡོད། །བླང་བྱ་གཙོ་རུ་བསྟན་ཞེས་པ། །སློབ་དཔོན་ཆེན་པོ་སེང་བཟང་གསུང་། །དོར་བྱ་གཙོ་བོར་བསྟན་པ་ནི། །སློབ་དཔོན་བུདྡྷ་ཤྲཱི་ཛྙཱ་གསུང་། །སྦྱོར་བ་རྣམ་བཅུར་བསྟན་ཀྱང་ནི། །གྲངས་
ངེས་{𰶥གོ་རིམས་,གོ་རིམ་}ངེས་པ་མེད། །སྦྱོར་ཡུལ་བཞི་ཡིན་སྦྱོར་བ་དངོས། །གསུམ་ཡིན་གཉིས་འབྲས་གཅིག་དཔེ་ཞེས། །དེ་སྐད་སྡུད་པར་བྱེད་པ་ཡིད། །མཐོང་ལམ་ན་ཡོད་ཁ་ཅིག་ཟེར། །བསྒོམ་ལམ་སྦྱོར་བ་ཟེར་བའང་ཡོད། །ལ་ལ་སྦྱོར་ལམ་ན་ཡོད་དེ། །སྦྱོར་
ལམ་མ་བསྟན་གསུང་པ་ནི། །གཞི་རུ་མ་བསྟན་ལ་དགོངས་གསུང་། །ཟློས་སྐྱོན་སྔར་སྤངས་ངོ་བོ་ནི། །རྗེས་ཀྱི་ལྷག་མཐོང་སྙིང་རྗེ་ཡིན། །རློམ་སེམས་རྣམ་བཞི་བསྟན་པ་ཡང་། །སེང་བཟང་འདོད་པ་དཀྱུས་ན་གསལ། །གཟུགས་ལ་གཟུགས་ཀྱིས་གཟུགས་
སུ་དང་། །བདག་གིར་རློམ་སེམས་མི་བྱེད་པ། །རྣམ་གྲོལ་ཤཱནྟི་རྣམ་གཉིས་བཞེད། །ས་མཚམས་སྦྱོར་བཞིན་ངོ་བོ་ནི། །ཤེས་རབ་ཡིན་ཏེ་བཅུ་པོ་ཡི། །རེ་རེ་ལ་ཡང་བཞི་བཞི་ཡི། །རྣམ་ཡོད་ཅེས་པ་སློབ་དཔོན་ནི། །ཤེས་རབ་འབྱུང་གནས་བློ་གྲོས་ཟེར། །མཐོང་
ལམ་བསྡུས་དོན་སྔར་བཤད་དེ། །འདི་རུ་བཤད་པའི་ཁྱད་པར་ནི། །བཞི་པ་སྐྱེ་བའི་ཚུལ་དང་ནི། །ཐུན་མོང་མ་ཡིན་ཚུལ་གྱིས་སུ། །བསྟན་ཏོ་ཞེས་ནི་བརྗོད་པ་ཡིན། །དོར་བའི་ཚུལ་ནས་བསྟན་ཏོ་ཞེས། །སེང་གེ་བཟང་པོ་དེ་སྐད་གསུང་། །བསྒོམ་ལམ་མི་སྟོན་རྒྱུ་
མཚན་ནི། །སྔར་བསྟན་ཟིན་ཏེ་སྦྱོར་ལམ་ནི། །བྱང་སེམས་ཐམས་ཅད་མཁྱེན་ལ་ལྟོས། །ཉན་ཐོས་ལམ་ཤེས་སུ་བསྟན་ཏོ། །རྣམ་ཀུན་མངོན་རྫོགས་ཡུལ་གྱིས་གཉིས། །ཤེས་དང་ཤེས་བྱེད་རྣམ་གཉིས་སོ། །ཡང་ན་ཤེས་པའི་རྒྱུན་ཐེན་པ། །རྣམ་ཀུན་ཤེས་བྱ་བླང་བའི་
