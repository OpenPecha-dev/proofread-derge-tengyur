བྲལ་བ་སྟེ། དེ་སྐྱེད་པར་བྱེད་པ་ལ་སོགས་པའི་རྡུལ་དང་བྲལ་བའོ། །ཚོར་བ་ཟད་པ་ལ་སོགས་པ་ནི་མྱ་ངན་ལས་འདས་པ་ཉིད་ལ་བརྗོད་པ་ཡིན་ནོ། །དེ་ལྟ་ན་ནི་གཅིག་བརྗོད་པ་མཆོག་སྟེ། གཞན་དགོས་པ་མེད་དོ་ཞེ་ན་བཤད་པ། དེ་ནི་བདེན་མོད་ཀྱི་རྒོལ་བ་དག་ལ་
རིགས་པས་བཤད་པ་ཡིན་ཏེ། དེའི་ཕྱིར་ཁ་ཅིག་ནི་མྱ་ངན་ལས་འདས་པ་ལ་ཚོར་བ་བདེ་བ་ཡིན་ནོ་ཞེས་དེ་ལ་དམིགས་པ་དག་སྣང་སྟེ། བདེ་བ་ནི་ཆགས་པ་ལ་གནས་པ་ཡིན་ནོ། །དེ་བས་ན་ཟག་པ་ཉིད་དོ། །དེ་ལྟར་ཁ་ཅིག་ནི་རྒྱུན་རྟག་པ་ཡིན་ཏེ། སྐད་ཅིག་མ་ནི་མ་
ཡིན་ནོ་ཞེས་པ་སྟེ། དབང་པོ་ལས་སྐྱེས་པའི་མངོན་སུམ་གྱིས་རྟོགས་པའི་ཕྱིར་རོ། །དབང་པོ་ཞན་པས་དམིགས་པའི་ཚུལ་ཡིན་པའི་ཕྱིར་ཏེ། ཇི་སྐད་དུ་ཚངས་པས་བྱས་ཤིང་བསྐྱེད་པའི་ཕྱིར་ཏེ། འབྱུང་དང་རབ་རྒྱས་འཇིག་དང་རབ་ཞུམ་པ། །ཡང་དང་ཡང་དུ་
ཉེར་འགྲོ་སྐྱོ་གྱུར་པས། །ཚངས་པས་ཐར་པའི་གྲོང་དུ་བསྟན་པ་སྟེ། །དེར་འགྲོ་ལམ་ནི་འཇིག་པས་རབ་གྲོལ་ཡིན། །ཞེས་སོ། །ཐར་པ་ལ་ནི་འཁོར་བ་མི་འདོད་པ་ཡིན་ཏེ། འདོད་པའི་བདེ་བས་རྒྱས་པ་ཡིན་པའི་ཕྱིར་དེ་ལས་ཀྱང་ཡོངས་སུ་ཉམས་པ་ཡིན་ཏེ།
ཐམས་ཅད་དེ་དང་ལྷན་ཅིག་ཡོངས་སུ་འཇིག་པ་སྲིད་པ་ཡིན་པའི་ཕྱིར་རོ། །འཆི་བ་དང་བཅས་པའི་མྱ་ངན་ལས་འདས་པའི་དགྲ་བཅོམ་པ་ཞེས་བྱ་བའི་ཚིག་གིས་ཏེ། འགྲོ་བ་མེད་པ་བཞིན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར་འདིར་རྡུལ་ནི་ཡོད་དེ། ཉོན་མོངས་པས་དམིགས་པ་
ཡིན་པའི་ཕྱིར་ཟག་པ་དང་བཅས་པས་ན་གཟུགས་ལྟ་བུ་ཡིན་ཏེ། དེ་བས་ན་རྒོལ་བ་དེ་བསྩལ་བར་བྱ་བའི་ཕྱིར་འདི་དག་བརྗོད་པ་ཡིན་ནོ། །བདག་ཉིད་དུལ་བ་ལས་སྐྱེས་པ་ནི་ཡོན་ཏན་དང་ལྡན་པའི་སྐྱེ་བོ་ཉིད་ཡིན་ནོ་ཞེས་བསྟན་པའི་ཕྱིར། མི་ནི་ལོ་བརྒྱར་ཞེས་བྱ་བ་ལ་སོགས་
པ་སྨོས་སོ། །ལོ་བརྒྱར་ཞེས་བྱ་བ་ནི་དུས་ཚིགས་ཀྱི་ཚད་བརྒྱའོ། །མི་ནི་ཞེས་བྱ་བ་ནི་སྐྱེ་བའི་རྒྱུ་དང་འབྲེལ་པའི་རྒྱུན་ཏེ། འཁོར་བའི་སེམས་ཅན་ཞེས་བྱ་བའི་དོན་ཏོ། །ནགས་ཚལ་ཞེས་བྱ་བ་ནི་དགོན་པ་སྟེ་གཡེང་བ་མེད་པའི་ཕྱིར་རོ། །མེ་ལ་ཡོངས་བརྟེན་ཞེས་བྱ་
