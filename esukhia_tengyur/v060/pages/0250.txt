ནོ། །ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱ་ཉིད་དམ་ཚིག་སྟེ། ཐོག་མ་དང་ཐ་མ་མེད་པའི་ཆོས་ཀྱི་དབྱིངས་སུ་རོ་གཅིག་པའི་ཡེ་ཤེས་ལ་ངེས་པར་སྦྱོར་བའི་ཚུལ་གྱིས་གཟུང་བ་དང་འཛིན་པའི་རྣམ་པར་རྟོག་པ་མེད་པའི་ཕྱིར་དེ་དང་ཡང་དག་པར་ལྡན་པ་ནི་བཅིངས་བ་སྟེ། ཡེ་ཤེས་ཀྱི་ཕྱག་རྒྱ་དང་
ཡང་དག་པའི་སྙིང་པོ་ནི་ཡི་གེ་ཧཱུཾ་དེ་སྟོན་པར་བྱེད་པའམ་དེ་སྐྱེད་པར་བྱེད་པའོ། །དེ་ཉིད་ཀྱིས་ན་དམ་པའོ། །གང་གིས་བསོད་ནམས་མི་ལྡན་པ། །ཞེས་བྱ་བ་ནི་དམ་པའི་ཆོས་སྤོང་བའམ། མཚམས་མེད་པ་ལ་སོགས་པ་མི་དགེ་བ་ཐམས་ཅད་བྱེད་པ་ནི་བསོད་ནམས་མི་ལྡན་པའོ། །དེ་
རྣམས་ཀྱང་གང་གི་དངོས་གྲུབ་ཐམས་ཅད་རྙེད་པར་དཀའ་བ་མ་ཡིན་པར་འགྱུར་རོ། །དེ་བཞིན་དཀྱིལ་འཁོར་མ་ཞུགས་པ། །ཞེས་བྱ་བ་ནི་དཀྱིལ་འཁོར་དུ་ཇི་ལྟ་བ་བཞིན་དུ་མ་ཞུགས་ཏེ་དེ་དག་ཀྱང་ངོ་། །དོན་ཡོད་པ་ནི་འབྲས་བུ་ཡོད་པའོ། །ཐོགས་པ་མེད་པ་ནི་བཟློག་ཏུ་མེད་པའོ། །
མྱུར་བ་ནི་རིངས་པའོ། །མཆོག་ནི་དམ་པར་གྱུར་པའོ། །ཐམས་ཅད་ནི་མ་ལུས་པ་སྟེ་དངོས་གྲུབ་ཐམས་ཅད་རྙེད་པར་དཀའ་བ་མ་ཡིན་པ་སྟེ། སྙིང་པོ་གང་ཞིག་གསུངས་པ་དེས་དོན་ཡོད་པ་དང་། ཐོགས་པ་མེད་པ་དང་། མྱུར་བ་དང་། མཆོག་གི་དངོས་གྲུབ་ཐམས་ཅད་རྣལ་
འབྱོར་པ་རྣམས་ཀྱིས་དངོས་གྲུབ་རྡོ་རྗེ་སེམས་དཔའ་ལ་སོགས་པར་སྐྱེད་པར་གྱུར་པ་དེ་ལ་དེ་སྐད་ཅེས་བྱའོ། །བདེ་བ་ཆེན་པོ་རྡོ་རྗེ་ཆེ། །གསང་བའི་མཆོག་སྟེ་དམ་ཚིག་བརྟན། །ཞེས་བྱ་བ་ནི་བདེ་བ་ཆེན་པོ་རྡོ་རྗེ་གསང་བའི་དཀྱིལ་འཁོར་གྱི་དམ་ཚིག་གི་མཆོག་སྟེ་བསྐྱེད་པའི་
རྒྱུའོ། །བརྟན་པ་ནི་མི་ཤིགས་པའོ། །དངོས་པོ་ཐམས་ཅད་བསྒྲུབ་པ་ནི། །ཞེས་བྱ་བ་ནི་བུམ་པ་བཟང་པོ་ལ་སོགས་པ་དངོས་པོ་ཐམས་ཅད་བསྒྲུབ་པའོ། །རྟོག་པ་ཐམས་ཅད་མྱུར་འགྲུབ་བྱེད། །ཅེས་བྱ་བ་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་དུམ་བུར། ཕྱག་རྒྱ་ཆེན་པོ་
ལ་སོགས་པའི་ཚུལ་གྱིས་རྟོགས་པ་གསུངས་པ་དང་། གཉིས་པར་གང་གསུངས་པ་དེ་རྣམས་བསྒྲུབ་པ་ནི་ཡི་གེ་ཧཱུཾ་སྙིང་པོའི་ཚུལ་གྱིས་བསྒྲུབ་བོ། །ཐོག་མཐའ་མེད་ཅིང་ཞེས་བྱ་བ་ལ་སོགས་པ་ལ་ཐོག་མ་ནི་དང་པོར་སྐྱེ་བའོ། །ཐ་མ་ནི་འཇིག་པའོ། །འདི་ལ་མི་མངའ་བས་ན་
