སྐྱེ་བོ་རྣམས་སྤྲོས་པར་བྱ་བའི་ཕྱིར་བསྟན་ནས། ད་ནི་རིགས་ཐམས་ཅད་ཀྱི་འཁོར་ལོའི་ཁྱད་པར་ཤེས་པས་ཀྱང་དངོས་གྲུབ་འབྱུང་བ་ཡིན་ནོ་ཞེས་རབ་ཏུ་བསྟན་པར་བྱ་བའི་ཕྱིར། དེ་ནས་རིགས་ཐམས་ཅད་འཁོར་ལོའི་རྒྱུད་ཅེས་བྱ་བ་རིགས་ཐམས་ཅད་ཀྱི་འཁོར་ལོའི་ཁྱད་
པར་ཤེས་པའི་རྒྱུད་གསུངས་པ་ཡིན་ཏེ། རིགས་ཐམས་ཅད་ཀྱི་འཁོར་ལོ་ནི་དཀྱིལ་འཁོར་ཏེ། དེ་རྣམས་ཀྱི་ཁྱད་པར་ནི་གསལ་བའི་དབྱེ་བའོ། །རྡོ་རྗེ་འཁོར་ལོ་བསྟན་པ་བཞིན། །ཞེས་བྱ་བ་ལ་རྡོ་རྗེ་སེམས་དཔའ་དང་། དེ་བཞིན་གཤེགས་པ་དག་གི་དཀྱིལ་འཁོར་ནི། རྡོ་
རྗེ་འཁོར་ལོ་སྟེ་སྔར་བྲི་བའི་ཆོ་གར་གཉིས་པོའི་ཁ་དོག་དང་དབྱིབས་ཀྱི་བདག་ཉིད་ཇི་ལྟར་བཤད་པ་དེ་བཞིན་དུ་གཟུང་བར་བྱའོ། །ལྕགས་ཀྱི་འཁོར་ལོ་དམ་པ་སྟེ། །ཞེས་བྱ་བ་ལ་དཀྱིལ་འཁོར་དམ་པ་ནི་ཕུལ་དུ་བྱུང་བ་སྟེ་ནང་གི་དཀྱིལ་འཁོར་ནི་ལྕགས་ཀྱིས་བྲི་བར་བྱའོ། །དེ་
གང་ཡིན་ཞེ་ན། འབར་བའི་འཁོར་ལོ་ཞེས་བྱ་བ་ལ་རབ་ཏུ་དགའ་བའི་རྡོ་རྗེ་ཁྲོ་བོའི་ནང་གི་རིམ་པ་གང་ཡིན་པ་དེ་འབར་བའི་འཁོར་ལོ་སྟེ། ལྕགས་ཀྱི་ཡིན་ནོ་ཞེས་པའི་དོན་ཏོ། །དེ་བཞིན་དུ་པདྨ་འདབ་བརྒྱད་ཅེས་བྱ་བ་ལ་ཇི་ལྟར་སྔར་བཤད་པའི་མཚན་ཉིད་ཅན་གྱིས་རྡོ་རྗེ་
སེམས་དཔའ་དང་། དེ་བཞིན་གཤེགས་པ་དག་གི་དཀྱིལ་འཁོར་དུ་གསུངས་པ་དེ་བཞིན་དུ་མཚན་ཉིད་དེ་ཉིད་ཀྱིས། རྡོ་རྗེ་འཇིག་རྟེན་དབང་ཕྱུག་གི་ནང་གི་དཀྱིལ་འཁོར་པདྨ་འདབ་མ་བརྒྱད་པར་ཤེས་པར་བྱའོ། །ནོར་བུ་ཆེ་ཞེས་བྱ་བ་འཕགས་པ་ནམ་མཁའི་སྙིང་པོའི་
ནང་གི་དཀྱིལ་འཁོར་ཟུར་བརྒྱད་པའི་ནོར་བུ་ཆེན་པོའི་རང་བཞིན་དུ་ཤེས་པར་བྱའོ། །དེ་ལྟར་འཁོར་ལོའི་ཁྱད་པར་ཤེས་པ་བསྟན་ནས། ད་ནི་དཀྱིལ་འཁོར་ཐམས་ཅད་ཀྱི་སྒྲུབ་པ་ཤེས་པས་དངོས་གྲུབ་འབྱུང་བ་ཡིན་ནོ་ཞེས་བསྟན་པར་བྱ་བའི་ཕྱིར། དེ་ནས་དཀྱིལ་
འཁོར་ཐམས་ཅད་སྒྲུབ་པའི་རྒྱུད་ཅེས་བྱ་བ་གསུངས་ཏེ། དཀྱིལ་འཁོར་ཐམས་ཅད་ནི་ཇི་ལྟར་བཤད་པ་བཞིན་ཏེ། དེ་རྣམས་སྒྲུབ་པ་ནི་རྫོགས་པའི་རྒྱུའོ། །དཀྱིལ་འཁོར་བརྟག་པར་བྱས་ནས་ནི། །ཞེས་བྱ་བ་ཇི་ལྟ་བ་བཞིན་དུ་དཀྱིལ་འཁོར་བྲིས་ཏེ་ཡང་དག་པར་
