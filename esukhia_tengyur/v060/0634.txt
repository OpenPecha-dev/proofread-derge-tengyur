རྒྱ་བཞིས་སྔར་བཤད་པའི་བསྒྲུབ་པའི་ཆོ་གས་བསྒྲུབས་པས་ན་དབང་དུ་བྱས་པར་མ་གྱུར་པ་དེ་ནི་མ་གྲུབ་པའོ། །དེས་དངོས་གྲུབ་བསྒྲུབ་པ་ན་འགྲུབ་པ་ཡིན་ལ་མ་བསྒྲུབས་པས་ན་མ་གྲུབ་པའོ། །ལས་ཀྱི་རྒྱུན་ཞེས་བྱ་བ་ནི་ལས་རྣམས་ཀྱི་རྒྱུན་ཏེ། ཐམས་ཅད་བསྲེག་པ་ལ་སོགས་
པའི་ལས་ཀྱི་རྒྱུན་རྣམས་སོ། །ལས་ཀྱི་རྒྱུན་དེའི་དམ་པ་ནི་མཆོག་གོ། །སུ་ཞིག་གི་ཞེ་ན་རྡོ་རྗེ་ཁྲོ་མཆོག་རྒྱལ་པོ་ཞེས་བྱ་བ་གསུངས་ཏེ་རྡོ་རྗེ་ཁྲོ་བོ་ནི་རྡོ་རྗེ་རིགས་ཀྱི་རང་བཞིན་ནོ། །དེ་རྣམས་ཀྱི་མཆོག་ནི་དམ་པའོ། །རྡོ་རྗེ་ཁྲོ་བོ་མཆོག་ཀྱང་ཡིན་རྒྱལ་པོ་ཡང་ཡིན་པས་ན་རྡོ་རྗེ་ཁྲོ་མཆོག་
རྒྱལ་པོ་སྟེ། བཅོམ་ལྡན་འདས་རྡོ་རྗེ་མེ་ལྟར་རབ་ཏུ་འབར་བའོ། །ནི་ཞེས་བྱ་བའི་སྒྲས་ནི་ངེས་པར་གཟུང་བ་སྟེ། རྡོ་རྗེ་ཁྲོ་བོའི་རྒྱལ་པོ་ཉིད་དོ། །གོ་རིམས་བཞིན་དུ་ཞེས་བྱ་བ་ནི་ཇི་སྐད་དུ་བཤད་པའི་བསྒྲུབ་པའི་ཆོ་གས་སོ། །རབ་བཤད་ཅེས་བྱ་བ་ནི་རབ་ཏུ་ཕྱེ་སྟེ་བཤད་པའོ། །འདིས་
ནི་འདི་སྐད་སྟོན་པར་འགྱུར་ཏེ་གང་གི་ཚེ་སྔར་བཤད་པའི་སྔོན་དུ་བསྙེན་པ་བྱས་ནས། ཟླ་བ་ཉ་ལ་མཚན་ཐོག་ཐག་བཟླས་བརྗོད་བྱས་ཏེ་ཆོས་ཀྱི་ཕྱག་རྒྱ་ལ་སོགས་པ་བསྒྲུབ་པ་ན་འགྲུབ་པར་མ་གྱུར་ན་དེའི་ཚེ་འོག་ནས་རྗེས་ཐོགས་ཉིད་ལ་འཆད་པའི་སྒྲུབ་པའི་ལས་བརྒྱ་པོ་བཀླགས་
པ་ཙམ་གྱིས་འགྲུབ་པར་འགྱུར་རོ། །ཐམས་ཅད་བསྲེག་པ་ལ་ནི་ཞེས་བྱ་བ་ནི་རྒྱུ་བ་དང་མི་རྒྱུ་བ་ཐམས་ཅད་བསྲེག་པ་ནི་ཐམས་ཅད་བསྲེག་པའོ། །བསྐྲད་པ་དང་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །འདི་དག་ཐམས་ཅད་ཇི་ལྟར་འགྱུར་ཞེ་ན། རྡོ་རྗེ་ཧཱུཾ་མཛད་ཀྱིས་སོ་ཞེས་བྱ་བ་
གསུངས་ཏེ་རྡོ་རྗེ་ཧཱུཾ་མཛད་ལས་བྱུང་བས་སོ། །དེ་ཡང་སྐེ་ཚེ་དང་བཅས་པའི་ཉུངས་མར་གྱི་སྦྱིན་སྲེག་གཡས་པས་བྱ་བ་དང་དེ་ཉིད་ཀྱིས་ཐམས་ཅད་བཅིང་བ་དང་མི་འགུལ་བ་ཉིད་དང་གཉིད་ལོག་པར་སྟོན་པ་དང་། ཁྲོ་བོའི་རྡོ་རྗེའི་ལྟ་བས་བསྡིགས་པ་དང་། ཕྱག་རྒྱ་ལྷོད་
པར་བྱ་བ་དང་། མཛུབ་མོ་གཉིས་སྦྱར་བར་བྱ་བའི་མཚམས་དབྱེ་བ་དང་། དེ་དག་ཉིད་ཀྱིས་གདུ་བུར་བཅིང་བ་དང་། བཅིངས་ཏེ་བཙིར་བ་དང་། དེ་ནས་དྲལ་བ་དང་། གཡས་ནས་བརྡེག་པ་དང་། མཛུབ་མོ་གཉིས་སྦྱར་བ་དང་། གཡས་པའི་གུང་མོ་འགྱེད་པའི་
