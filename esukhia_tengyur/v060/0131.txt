ཡིད་ཀྱིས་བྲིས་ཏེ། རྡོ་རྗེ་སྐུའི་མཆོག་བཅིངས་ལ་གུང་མོ་གཉིས་ཀྱིས་པདྨ་ལྟར་བྱས་ནས་ངག་འབྱིན་པའི་ཚུལ་དུ་བྱས་ཏེ་དེ་ལྟ་བུས་ནི་གསུང་གི་དངོས་གྲུབ་ཏུ་གྱུར་ལ་འདས་པ་དང་། མ་འོངས་པ་དང་། ད་ལྟར་གྱི་དངོས་པོ་ཐམས་ཅད་བདེན་པར་སྟོན་པའི་
མཚན་ཉིད་ཅན། རྡོ་རྗེ་གསང་བའི་ལྗགས་ཀྱི་མཆོག་གིས་འགྱུར་རོ། །ཐུགས་ཐམས་ཅད་གྲུབ་པ་ལ་ནི་རྡོ་རྗེ་ཐུགས་བསྐྱེད་པས་སོ་ཞེས་བྱ་བ་ལ་དེ་བཞིན་དུ་དཔལ་རྡོ་རྗེ་སེམས་དཔའི་སྦྱོར་བ་བྱས་ལ། རང་གི་སྙིང་གའི་ཟླ་བ་ལ་རྡོ་རྗེ་རྩེ་ལྔ་པ་བསམས་ཏེ། རང་གི་སྙིང་གའི་ཕྱོགས་
སུ་རྡོ་རྗེ་སྐུའི་མཆོག་བཅིངས་ལ། ཨོཾ་བོ་དྷི་ཙིཏྟོ་ཏ་པ་ད་ཡཱ་མི་ཞེས་རྡོ་རྗེའི་སྙིང་པོ་དམ་པ་ཟློས་ཤིང་རྡོ་རྗེ་སེམས་དཔར་བདག་ཉིད་བསྒོམ་པར་བྱ་སྟེ། དེ་ལྟ་བུས་ནི་ཐུགས་ཀྱི་དངོས་གྲུབ་ཐོབ་པར་འགྱུར་རོ། །རྡོ་རྗེ་ཐམས་ཅད་ཀྱི་དངོས་གྲུབ་ནི་རབ་ཏུ་དགའ་བའི་རྡོ་རྗེས་སོ་ཞེས་བྱ་བ་ནི་དེ་
བཞིན་དུ་རྡོ་རྗེ་སེམས་དཔའི་སྦྱོར་བ་བྱས་ལ་རང་གི་སྙིང་གར་ཡི་གེ་ས་ལས་སྣ་ཚོགས་རྡོ་རྗེ་བསམས་ལ་རྡོ་རྗེ་སྐུའི་མཆོག་བཅིངས་ཏེ་གཡས་པའི་མཐེ་བོང་གཡོན་པའི་མཐེ་བོང་གི་བུ་གར་བཅུག་ལ་བསྒོམ་པར་བྱ་སྟེ། དེ་ལྟ་བུས་ནི་རྡོ་རྗེ་ཐམས་ཅད་འགྲུབ་པར་འགྱུར་རོ། །དཀྱིལ་
འཁོར་ཐམས་ཅད་དུ་འཇུག་པ་ལ་ནི་རྡོ་རྗེ་འཁོར་ལོས་སོ་ཞེས་བྱ་བ་ནི་དེ་བཞིན་དུ་དཔལ་རྡོ་རྗེ་སེམས་དཔའི་སྦྱོར་བ་བྱས་ལ་བདག་ཉིད་བྱང་ཆུབ་སེམས་དཔའ་སེམས་བསྐྱེད་མ་ཐག་ཏུ་ཆོས་ཀྱི་འཁོར་ལོ་བསྐོར་བར་བསམས་ལ། རྡོ་རྗེ་ཁུ་ཚུར་གཉིས་བཅིངས་ལ། གློག་འགྱུ་བ་
འདེགས་པའི་ཚུལ་གྱིས་གཡས་ཕྱོགས་ནས་བསྐོར་བ་འདི་ནི་རྡོ་རྗེ་འཁོར་ལོ་སྟེ། འདིས་དཀྱིལ་འཁོར་དུ་འཇུག་པར་འགྱུར་རོ། །ཕྱག་རྒྱ་ཐམས་ཅད་བཅིངས་པའི་དངོས་གྲུབ་ནི་ཞེས་བྱ་བ་ན། ཕྱག་རྒྱ་ཆེན་པོ་ལ་སོགས་པའི་ཕྱག་རྒྱ་ཐམས་ཅད་བཅིངས་པའི་དངོས་གྲུབ་པོ། །དེ་ཇི་
ལྟར་འགྱུར་ཞེ་ན། རྡོ་རྗེ་གསང་བའི་དམ་ཚིག་གི་ཕྱག་རྒྱ་ཆེན་པོས་སོ་ཞེས་བྱ་བ་གསུངས་ཏེ་རྡོ་རྗེ་ནི་དེ་ཁོ་ན་ཉིད་དོ། །དེའི་གསང་བའི་དམ་ཚིག་ནི་དང་པོའི་རྡོ་རྗེའོ། །དེའི་ཕྱག་རྒྱ་ཆེན་པོ་རྡོ་རྗེ་སེམས་དཔའི་ལས་ཀྱི་ཕྱག་རྒྱ་ནི་རྡོ་རྗེ་བསྙེམས་པའོ། །འདི་སྐད་སྟོན་ཏེ། རྡོ་རྗེ་
