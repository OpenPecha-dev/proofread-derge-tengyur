ཐུར་མ་སོར་བཞི་པས་ཚད་མཛེས་པའོ། །དེའི་སྐྱེས་བུ་ཉིད་ཅེས་ཐ་སྙད་དུ་བྱའོ་ཞེས་བྱ་བ་ནི་བཟང་ཞེས་སྦྱར་བར་རོ། །དེ་ནི་ཞེས་བྱ་བ་ནི་ཐུར་མ་སོར་བཞི་པ་ལ་ཇི་ཙམ་པའི་ཐུར་མ་རྣམས་དེ་ཙམ་དུ་སྐྱེས་བུ་གྲིབ་མ་བརྗོད་པར་བྱ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཉིན་མཚན་གྱི་ཆ་སྔ་དྲོ་ལ་
སོགས་པ་དང་ཞེས་བྱ་བ་ནི་རིག་པར་བྱ་ཞེས་བྱ་བ་རྗེས་སུ་འཇུག་གོ། །སྔ་དྲོ་དང་གུང་གི་ཞེས་བྱ་བ་ལ་སོགས་པའི་གཞུང་གིས་གསུངས་སོ། །དུས་ཚོད་ཀྱང་ངོ་ཞེས་བྱ་བ་ནི་རིག་པར་བྱའོ། །དེ་དག་ལྔའོ་ཞེས་བྱ་བ་ནི་དུས་ཚོད་རྣམས་ཏེ། དགུན་གྱི་དུས་དང་། དཔྱིད་ཀྱི་དུས་དང་། དབྱར་
གྱི་དུས་དང་། དབྱར་ཐུང་ངུའི་དུས་དང་། དབྱར་རིང་པོའི་དུས་སོ། །འདི་རྣམས་ཀྱི་ཚད་བརྗོད་པ་ཡིན་ནོ། །དང་པོ་གཉིས་ནི་ཟླ་བ་བཞི་བཞིའོ་ཞེས་བྱ་བ་ནི་དགུན་གྱི་དུས་དང་དཔྱིད་ཀྱི་དུས་སོ། །འོག་མ་ནི་ཟླ་བ་གཅིག་གོ་ཞེས་བྱ་བ་ནི་དབྱར་གྱི་དུས་སོ། །དེའི་འོག་མ་ནི་ཉིན་ཞག་གཅིག་གོ་
ཞེས་བྱ་བ་ནི་དེ་ནི་དབྱར་གྱི་དུས་ལས་འོག་མ་ནི་དབྱར་ཐུང་ངུའི་དུས་ཏེ་ཉིན་ཞག་གཅིག་གོ། །ཐ་མ་ནི་དེས་ཟླ་བ་གསུམ་དུ་མ་ཆོག་པའོ་ཞེས་བྱ་བ་ནི་ཉིན་ཞག་གཅིག་པོ་དེས་མ་ཆོག་པའི་ཟླ་བ་གསུམ་ནི་ཐ་མའི་དུས་ཏེ་དབྱར་རིང་པོའི་དུས་སོ། །དུས་ཚོད་བརྗོད་པའི་དེ་མ་ཐག་ཏུ་བསྙེན་པར་རྫོགས་
པར་བྱས་པ་ཙམ་ལ་ལས་བྱེད་པ་པོས་གནས་རྣམས་བརྗོད་པར་བྱའོ་ཞེས་བྱ་བ་ནི་ཕྱག་དར་ཁྲོད་པ་ལ་སོགས་པ་བཞི་པོ་དེ་རྣམས་སོ། །ལྟུང་བར་འགྱུར་བའི་ཆོས་རྣམས་དང་། དགེ་སྦྱོང་དུ་བྱེད་པའི་ཆོས་རྣམས་ཀྱང་ངོ་ཞེས་བྱ་བ་སྟེ། ཁྲོ་བ་ལ་སླར་མི་ཁྲོ་བ་ལ་སོགས་པ་རྣམས་བརྗོད་པར་བྱའོ་ཞེས་བྱ་
བར་འབྲེལ་ཏོ། །མཆོག་ཏུ་འདོད་པར་བྱ་བ་ཡང་དག་པ་ཉིད་ཀྱིས་རྫོགས་པ་ཉིད་བརྗོད་ནས། ཚུལ་ཁྲིམས་ཀྱིས་མཉམ་པར་གྱུར་པ་ཉིད་བསྒྲུབ་པ་ལ་ངེས་པར་སྦྱར་བར་བྱའོ་ཞེས་བྱ་བ་ནི་མཆོག་ཏུ་འདོད་པར་བྱ་བས་ན་མཆོག་ཏུ་འདོད་པར་བྱ་བ་ཡིན་ཏེ། གང་ཞིག་ཁྱེད་ལ་སྔོན་བྱུང་བ་སྟོན་པར་བྱེད་པའོ། །བདག་གིས་ཅི་
ཞིག་ཐོབ་པར་འགྱུར་རམ་ཞེས་བྱ་བ་ལ་སོགས་པ་དེ་ནི་དེའི་འདོད་པའོ། །དེའི་རྫོགས་པ་ཉིད་ནི་ཡོད་པ་ཉིད་གཟུང་ནས་སུ་འདི་ལ་རབ་ཏུ་བྱུང་བ་དང་བསྙེན་པར་རྫོགས་པ་ཞེས་བྱ་བ་ལ་སོགས་པས་སོ། །སླར་རྫོགས་པ་ཉིད་ལ་མཐུན་པའི་མཁན་པོས་ཞེས་བྱ་བ་ལ་སོགས་པས་ཡང་དག་པ་ཉིད་བརྗོད་པའོ། །རྫོགས་པ་
