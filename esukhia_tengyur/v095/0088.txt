པ་མེད། ཁྲིད་པར་བྱེད་པ་དེ་ཉིད་ཀྱང་མཉམ་པ་ཉིད་ཡིན་ནོ་ཞེས་གསུངས་སོ། །མཉམ་པ་ཉིད་ཆོས་ཀྱི་དབྱིངས་དེ་ལ། ཡང་དག་པར་བཅོམ་པ་མེད་པ་ནི་གང་གིས་ཇི་ལྟར་ཡང་དག་པར་རྒྱུན་བཅད་པ་མེད་པའོ། །འདི་ལྟར་སྣ་ཚོགས་ཉིད་ཀྱིས་ནི་བཅོམ་པ་དང་། འཇོམས་
པར་བྱེད་པ་ལ་ཁྱབ་པར་བྱེད་པ་ཡིན་ལ། སྣ་ཚོགས་ཉིད་དང་མི་མཐུན་པ་ཡང་མཉམ་པ་ཉིད་ཡིན་ནོ། །དེ་ལྟར་སྣ་ཚོགས་ཉིད་ལོག་པར་གྱུར་ན། ཁྱད་པར་བྱ་བ་བཅོམ་པ་རང་ཡང་ལྡོག་པར་འགྱུར་བས། ཡང་དག་པར་བཅོམ་པ་མེད་པ་མཉམ་པ་ཉིད་ཡིན་ནོ། །གཞན་
ཡང་འདས་པ་ནི་འགགས་པའི་ཕྱིར་དང་། མ་འོངས་པ་ནི་མ་ཐོབ་པའི་ཕྱིར་དང་། ད་ལྟར་བྱུང་བ་ནི་སྟོབས་མཚུངས་པའི་ཕྱིར་འཇོམས་པ་མ་ཡིན་ནོ། །དེ་བས་ན་དོན་དམ་པར་གང་གཞོམ་པར་བྱ་བ་དང་། འཇོམས་པར་བྱེད་པ་གང་ཡང་མཉམ་པ་ཉིད་ལ་བསམ་དུ་མེད་དོ། །འཇོམས་
པ་དེ་ལ་ཁྱབ་པར་བྱེད་པའི་སྣ་ཚོགས་ཉིད་དང་། མི་མཐུན་པ་མཉམ་པ་ཉིད་ནི་ཁྲིད་པར་བྱེད་པ་ལ་ཡོད་པར་མཇུག་བསྡུ་ཞིང་མ་གྲུབ་པ་ཡོངས་སུ་སྤང་བའི་ཕྱིར། ཁྲིད་པར་བྱེད་པ་དེ་ཉིད་ཀྱང་མཉམ་པ་ཉིད་ཡིན་ནོ་ཞེས་གསུངས་ཏེ། འདིར་ནི་ཁྲིད་པར་བརྗོད་པ་མཉམ་པ་ཉིད་བསྟེན་པས་ནི་
ཁྲིད་པའི་མི་མཉམ་པའི་ངོ་བོ་སེལ་བར་བྱེད་པ་ཡིན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར་མཉམ་པ་ཉིད་ཡིན་ནོ་ཞེས་བསྟན་ཏོ། །ཀྱང་ཞེས་བྱ་བ་ནི་སྔ་ཕྱི་སྡུད་པར་བྱེད་པའོ། །དེ་ཉིད་ཅེས་བྱ་བ་འདིས་ནི་ཁྲིད་པར་བྱེད་པ་བརྗོད་མ་ཐག་པ་མཚོན་པར་སྟོན་པར་བྱེད་དོ། །དེས་ན་འདི་སྐད་དུ། སྐྱེ་བ་མེད་པར་མཉམ་པ་
ཉིད་མཐོང་བ་ཁོ་ན་ཡང་དག་པར་དགྲ་བཅོམ་པ་ཉིད་ཡིན་ནོ་ཞེས་བསྟན་ཏོ། །སློབ་དཔོན་ཀླུ་སྒྲུབ་ཀྱིས་ཀྱང་། གནས་དང་བཅས་པའི་སེམས་ཡོད་ལ། །ཉོན་མོངས་དུག་ཆེན་ཅིས་མི་འབྱུང་། །ཐ་མལ་འདུག་པའམ་ཉོན་མོངས་པའི། །སྦྲུལ་གདུག་རྣམས་ཀྱིས་ཟིན་པར་འགྱུར། །
སྡུག་ཅེས་པ་ལས་ཆགས་པ་འབྱུང་། །དེ་ལས་བཟློག་པས་འདོད་ཆགས་བྲལ། །སྒྱུ་མའི་སྐྱེས་བུ་ལྟར་དབེན་པར། །མཐོང་ན་མྱ་ངན་འདས་པར་འགྱུར། །ཞེས་བཤད་དོ། །ཕུང་པོ་ལ་སོགས་པ་ཡང་དག་པར་མ་སྐྱེས་པ་རྣམས་ལ། མི་རྟོག་པ་ལ་སོགས་པའི་རྣམ་པར་བསྒོམས་
