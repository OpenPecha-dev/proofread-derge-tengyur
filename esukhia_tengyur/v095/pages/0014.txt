རྒྱ་གར་གྱི་མཁན་པོ་ཀ་མ་ལ་གུཔྟ་དང་། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པ།། ༄༅༅། །རྒྱ་གར་སྐད་དུ། ཨཱརྱ་སཔྟ་ཤ་ཏ་ཀཱ་པྲཛྙཱ་པཱ་ར་མི་ཏ་ཊཱི་ཀ །བོད་སྐད་དུ། འཕགས་པ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་
ཕྱིན་པ་བདུན་བརྒྱ་པའི་རྒྱ་ཆེར་འགྲེལ་པ། བམ་པོ་དང་པོ། ངག་གི་དབང་ཕྱུག་ཟབ་ཅིང་གཏིང་རྟོགས་བློ་མངའ་བ། །ཐུགས་རྗེ་ཆེ་ལྡན་སྐྱབས་ཀྱི་ངང་ལ་ཕྱག་འཚལ་ཏེ། ཀུན་རྨོངས་གནས་ངན་ལེན་གྱི་ནགས་ཆེན་སྲེག་པའི་མེ། །བདུན་བརྒྱ་པ་ནི་མདོར་བསྡུས་ཀུན་ནས་
བཤད་པར་བྱ། །སངས་རྒྱས་སྐྱེད་མ་ཐམས་ཅད་ལ་ཡང་ལྷག་མཐོང་དང་། །ཞི་གནས་དང་ནི་གཉིས་པོ་མཉམ་པའི་དམིགས་པ་འབྱུང་། །བདུན་བརྒྱ་པ་ལའང་བཞི་པོ་གང་དག་བསྟན་མོད་ཀྱི། །འཆད་བྱེད་རྣམས་ཀྱིས་མ་བསྟན་དེ་དག་བརྗོད་པར་བྱ། །གླེང་བསླང་བ་ནི་
མདོར་བསྡུས་དང་། །དེ་ཡི་སྐབས་འབྱེད་པ་དང་བཅས། །འཁོར་དང་གླེང་གཞི་རྒྱུ་དང་དོན། །མཐོང་བ་དང་ནི་བསྒོམ་པའི་ལམ། །རྒྱུ་ནི་མོས་སྤྱོད་རྣམ་པ་གསུམ། །བསྟོད་དང་རབ་བསྟོད་བསྔགས་པ་དང་། །དམིགས་དང་དེ་ཡི་རྣམ་པ་ཡི། །ཚུལ་གྱི་རྗེས་མཐུན་
རྣམ་དྲུག་སྟེ། །དེ་ཉིད་དོན་དམ་དགོངས་པ་དང་། །འཐོབ་དང་མཐའ་གཉིས་སྤངས་པ་དང་། །བསམ་གྱིས་མི་ཁྱབ་པ་དང་འཆད། །དེ་འབྲས་འཐོབ་པའི་མཚན་ཉིད་དང་། །དེ་ཡི་རྗེས་མྱོང་རྣམ་པར་སྒྲོགས། །རང་གི་ངོ་བོའི་ཐོགས་མེད་བློ། །རིགས་དང་བཞུགས་དང་
སྤྱོད་པ་དང་། །ཡོན་ཏན་བཅས་དང་བཟོད་པ་གསུམ། །རྗེས་སུ་གཟུང་དང་སྨོན་པ་དང་། །རྗེས་སུ་བསྟན་པ་དཔེ་དང་བཅས། །མི་རྟོག་པ་ཡི་གཟུགས་བརྙན་གསུངས། །དམ་ཆོས་ཡུན་རིང་གནས་ཆེད་དུ། །ལྷ་རྣམས་ཀྱིས་ནི་རྗེས་སུ་བསྒྲུབས། །སངས་རྒྱས་རྣམས་ཀྱིས་བྱིན་
བརླབས་འདི། །རྣམ་བཤད་མཚན་ཉིད་མདོ་དོན་གྱིས། །བདུན་བརྒྱ་པ་ཡི་དོན་བསྡུས་ཡིན། །དེ་ལ་ཚིག་དང་པོ་གསུམ་ནི་གླེང་བསླང་བའོ། །གླེང་བསླངས་ནས་སྐབས་འབྱེད་པར་རྩོམ་པ་ནི་དུས་གཅིག་ན་ཞེས་བྱ་བ་ལ་སོགས་པ་ནས། ཀུན་དགའ་ར་བ་ན་ཞེས་བྱ་བའི་
