མི་བརྟན་པས་ནི་སྤྲོ་མི་བྱ། །རབ་མཆོག་སྤྲིན་རྣམས་ཀྱིས་སྤྲོ་བྱ། །སངས་རྒྱས་ཡེ་ཤེས་མཉམ་འོད་ཀྱིས། །ཞེས་སོ། །དེ་ནས་ཕུང་པོ་ཁམས་དང་སྐྱེ་མཆེད་ཀྱི་རང་བཞིན་སྤྲོས་པའི་ཆོས་ཐམས་ཅད་ཀྱི་ངོ་བོ་ལྷ་རྣམས་འོད་དུ་བྱས་ནས་འོད་ཟེར་རྣམས་ཀྱང་གསུངས་པའི་སྦྱོར་
བ་ལས་ཕྲ་བའི་རྣལ་འབྱོར་སྟོང་པ་དང་། སྙིང་རྗེ་མི་ཕྱེད་པའི་རང་བཞིན་ལ་དྲིལ་བའི་ཚུལ་དུ་བསྡུས་ནས། སྐད་ཅིག་གིས་རང་རིག་པའི་ཡེ་ཤེས་འབྱུང་བ་ནི་ཏིང་ངེ་འཛིན་ཏོ། །རྒྱུད་གཞན་ནས་བསྟན་པ་དང་བརྒྱུད་པའི་རིམ་པ་ལས་བྱུང་བའི་བཤད་པ་འདི་ཁོ་ནའོ། །
ཡེ་ཤེས་ཞབས་ལུགས་ཀྱི་སྦྱོར་བ་ཡན་ལག་དྲུག་པ་ཞེས་བྱ་བ་རྫོགས་སོ།། {D1902}༄༅༅། །རྒྱ་གར་སྐད་དུ། མཉྫུ་བཛྲ་པཱུ་ཛ་བི་དྷིཿ། བོད་སྐད་དུ། འཇམ་པའི་རྡོ་རྗེ་མཆོད་པའི་ཆོ་ག །
བཅོམ་ལྡན་འདས་འཇམ་པའི་དབྱངས་ལ་ཕྱག་འཚལ་ལོ། །ཨོཾ་བཛྲ་སཏྭ་སརྦ་བིགྷྣཱན་ཨུ་ས་ར་ཡ་ཧཱུྃ་ཕཊ་ཅེས་པས་གུར་གུམ་ལ་སོགས་པ་བདུད་རྩི་ལྔ་དང་བཅས་པས་མཎྜལ་གྲུ་བཞིར་བྱུགས་ལ། སྐད་ཅིག་གིས་འབྱུང་བ་བཞི་དང་། རི་རབ་ཀྱི་སྟེང་དུ་ཡོད་པར་

ཡེ་ཤེས་ཀྱི་དཀྱིལ་འཁོར་སྤྱན་དྲངས་ཏེ་མཎྜལ་གྱི་སྟེང་དུ་བཞུགས་པར་བསམས་ལ་མཆོད་དེ། ཨོཾ་པྲ་ཝ་ར་སཏྐ་ར་ཨརྒྷཾ་པཱ་དྱ་ཨཉྩ་མ་ནཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཨརྒྷཾ་ལ་སོགས་པའི་སྔགས་སོ། །དེའི་དབུས་སུ་ཨོཾ་ཨཱཿབཛྲ་དྷྲྀཀ་དྭེ་ཥ་བཛྲཱ་ཡ་བཛྲ་
པུཥྤེ་ཧཱུྃ་སྭཱ་ཧཱ།ཨོཾ་ཨཱཿམཉྫུ་བཛྲ་ཡེ་པུཥྤེ་ཧཱུྃ་སྭཱ་ཧཱ། ཞེས་བྱ་བས་ཡབ་ཡུམ་ལ་ལན་གསུམ་དུ་མཆོད་ནས། ཤར་དུ་ཨོཾ་ཨ་ཛི་ན་ཛིཀ་མོ་ཧ་བཛྲཱ་ཡ་བཛྲ་པུཥྤེ་ཧཱུཾ་སྭཱ་ཧཱ། ལྷོ་རུ་ཨོཾ་ཨཱཿརཏྣ་དྷྲྀཀ་པི་ཤུ་ན་བཛྲཱ་ཡ་བཛྲ་པུཥྤེ་ཧཱུྃ་སྭཱ་ཧཱ། ནུབ་ཏུ་ཨོཾ་ཨཱཿཨཱ་རོ་ལྀཀ་
