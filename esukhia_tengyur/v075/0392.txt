ཏིཀྵྞ་དུཿ ཁཙྪེད་པྲ་ཛྙཱ་ན་མུརྟ་ཡེ། ཛྙཱ་ན་ཀཱ་ཡ་བཱ་གཱི་ཤྭ་ར་ཨ་ར་པ་ཙ་ན་མཿགྲངས་བརྒྱ་རྩ་བརྒྱད་དུ་མཚན་བརྗོད་པས་ཀྱང་མཚན་ཡང་དག་པར་བརྗོད་པ་དུས་ཇི་ལྟ་བར་བཀླག་པར་བྱ་བ་ཡིན་ནོ། །ཆོས་ཀྱི་དབྱིངས་ཀྱི་ངག་གི་དབང་ཕྱུག་གི་སྒྲུབ་ཐབས་ཨ་བྷ་ཡ་ལས་ཚུལ་ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་
བའོ།། ༄༅༅། །རྒྱ་གར་སྐད་དུ། བཱ་གཱི་ཤྭ་ར་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ངག་གི་དབང་ཕྱུག་གི་སྒྲུབ་ཐབས། སྔར་བཤད་པའི་རིམ་པས་སྟོང་པ་ཉིད་བསྒོམས་པའི་རྗེས་ཐོགས་པདྨ་དང་ཟླ་བ་ལ་གནས་པའི་ཡི་གེ་ཧོཿདམར་པོ་ཡོངས་སུ་གྱུར་པ་ལས་ཞབས་བཞི་ཕྱག་བརྒྱད་པ་སྟེ། རྩ་བའི་
ཞལ་ནི་དམར་སྐྱའོ། །ལྷོ་ནི་གུར་གུམ་དང་སྐྱ་རེངས་ཀྱིའོ། །ནུབ་ནི་པདྨ་རཱ་གའི་འོ། །བྱང་ནི་དམར་སེར་རོ། །གཉིས་ཀྱིས་ནི་མདའ་གཞུ་བསྣམས་པའོ། །གཞན་གཉིས་ཀྱིས་ནི་ཞགས་པ་དང་ལྕགས་ཀྱུ་བསྣམས་པའོ། །གཞན་གཉིས་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་བ་ཏི་དང་རལ་གྲི་བསྣམས་པའོ། །
དེ་བཞིན་དུ་གཞན་གཉིས་ཀྱིས་ནི་རྡོ་རྗེ་དང་དྲིལ་བུ་བསྣམས་པའོ། །ཆགས་པ་ཆེན་པོའི་འཇོ་སྒེག་གི་རོ་མྱོང་བའི་འོད་ཟེར་གྱིས་མཆོག་ཏུ་འབར་བ། རོལ་པས་སྣ་ཚོགས་པདྨ་དང་ཟླ་བའི་གདན་ལ་བཞུགས་པ། ལྷའི་ན་བཟའ་བརྒྱད་དང་ལྡན་པ་རོལ་པ་དང་ཅོད་པན་ལ་སྣང་བ་མཐའ་ཡས་དང་ལྡན་
པའོ། །སྙིང་གའི་ས་བོན་གྱི་འོད་ཟེར་གྱིས་སྤྱན་དྲངས་པའི་སེམས་དཔའ་རང་དབང་གིས་ལེགས་པར་གཞུག་ལ། སྔགས་བཟླས་པར་བྱ་བའོ། །ཨོཾ་མ་ཧཱ་རཱ་ག་ཡ་སརྦ་སཏྭ་ན་ཧོཿཞེས་བྱ་བ་འདི་ནི་ཆོས་ཀྱི་དབྱིངས་ཀྱི་ངག་གི་དབང་ཕྱུག་གི་སྒྲུབ་ཐབས་ལས་དབང་དུ་བྱ་བའི་ཆོ་གའོ། །ཚུལ་
ཁྲིམས་རྒྱལ་མཚན་གྱིས་བསྒྱུར་བའོ།། ༄༅༅། །རྒྱ་གར་སྐད་དུ། བཱ་གཱི་ཤྭ་ར་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། ངག་གི་དབང་ཕྱུག་གི་སྒྲུབ་ཐབས། དེ་ཁོ་ན་ཉིད་བཞིན་དུ་སྟོང་པ་ཉིད་བསྒོམས་པའི་རྗེས་ཐོགས་ལ་ཆུའི་དཀྱིལ་འཁོར་ལ་པདྨ་དཀར་པོ་འདབ་མ་བརྒྱད་པའི་ལྟེ་བར། ཆོས་
ཐམས་ཅད་ནི་འཆི་བ་ལས་ཀུན་ནས་བཟླས་པར་བྱའོ། །ཞེས་ལྷགས་པར་མོས་པར་བྱས་པས་ཟླ་བ་ལ་གནས་པའི་ཡི་གེ་ཨཾ་དཀར་པོ་ཡོངས་སུ་གྱུར་པ་ལས། དཀར་པོ་ཞལ་གཅིག་ཕྱག་གཉིས་པ་རྡོ་རྗེ་སྐྱིལ་མོ་ཀྲུང་གི་སྟེང་ན་ཕྱག་མཉམ་པར་གཞག་པའི་ཕྱག་རྒྱ་མཛད་ཅིང་། གཞོན་ནུའི་རྒྱན་མ་
