ཉེས་པར་སྨྲས་པའི་ཉེས་བྱས་སོ། །གང་དག་གིས་ཤོ་གམ་གཞལ་དགོས་པ་ཤོ་གམ་གྱི་སར། མ་ཕྱིན་པར་དེ་དག་གི་ཤོ་གམ་དེ་ལས་བྱོལ་བར་བྱ་བའི་ཕྱིར་ལམ་གཞན་བསྟན་ན་སྦོམ་པོའོ། །དེ་དག་གི་བདག་གིས་སྐལ་བ་མ་བླངས་པར་བཟླས་ན་སྦོམ་པོའོ། །སར་མ་ཕྱིན་ན་
སྐལ་བའམ། སྐལ་པ་ཕྱེད་དམ་སྐལ་བ་ཚང་བར་བླངས་ཏེ་བཟླས་ན་སྦོམ་པོའོ། །སར་ཕྱིན་ན་ཤོ་གམ་ཕྱེད་དམ། ཤོ་གམ་ཐམས་ཅད་ཀྱང་རུང་སྟེ། གལ་ཏེ་དེ་སྐལ་བ་ཚང་བར་གྱུར་ན་ཕམ་པར་འགྱུར་བའོ། །བདག་གི་སྣོད་ཤོ་གམ་གཞལ་དགོས་པ་ཁྱེར་རམ་ཁྱེར་དུ་བཅུག་ན་དངོས་པོ་ཇི་ལྟ་བ་
བཞིན་ནོ། །བདག་གི་འདི་ནི་མ་ལ་སོགས་པའི་ཡིན་ནོ་སྙམ་དུ་བསྔོས་ལ་མ་སྨྲས་པར་ནམ་མཁའ་འམ། ཁའི་ནང་ངམ། ཚེམ་བུའི་ཕྲག་པའམ། ལམ་གོལ་བ་ནས་བཟླས་ན་སྦོམ་པོའོ། །དེ་བཞིན་དུ་གང་ཆུང་བ་ཉིད་ཀྱིས་རིན་ཐང་གདབ་པར་བྱ་བར་མི་ནུས་པའོ། །གང་ཆེ་བ་ཉིད་ཀྱིས་རིན་ཐང་གདབ་
པར་བྱ་བར་མི་ནུས་ཏེ། གཞལ་དུ་མེད་པ་ནི་དངོས་གཞིའོ། །སྣོད་ཤོ་གམ་གཞལ་དགོས་པ་ཚོང་པ་རྣམས་ཀྱིས་ལྐོག་ཏུ་གཏུར་བུ་ལ་སོགས་པའི་ནང་དུ་བཅུག་པ་ལ་ནི་ལྟུང་བ་མེད་དོ། །རྫས་ཀྱི་སྣོད་རྣམས་ཡལ་བར་གཞག་པར་མི་བྱའི། དགེ་སློང་གཉིས་བཞག་སྟེ་དེ་གཉིས་ལ་བསོད་སྙོམས་སྦྱིན་ནོ། །
གལ་ཏེ་བཟློག་ཀྱང་བཅུག་ན་བསྙེན་པར་མ་རྫོགས་པ་མེད་ན་བདག་ཉིད་ཀྱིས་ཕྱུང་སྟེ་གཞག་པར་བྱའོ། །ཁྱེད་རང་གི་ནོར་ཤེས་པར་གྱིས་ཤིག་ཅེས་བྱའོ། །གཅིག་པུ་འགྲོ་ལ་བསོད་སྙོམས་ལ་སོགས་པའི་ཕྱིར་འགྲོ་དགོས་ན་མཚན་མ་གདབ་པར་བྱའོ། །སླར་འོངས་ནས་ཀྱང་དེ་བརྟག་པར་
བྱའོ། །མ་དང་ཕ་དང་། དཀོན་མཆོག་གསུམ་གྱི་ཕྱིར་ཁྱེར་ན་དེ་དག་གི་བསྔགས་པ་བརྗོད་པར་བྱའོ། །མི་འདོད་ན་བྱིན་ཏེ་འགྲོ་བར་བྱའོ། །གནོད་པ་འབྱུང་དུ་དོགས་ན་དཀོན་མཆོག་གསུམ་གྱི་སྐུ་སྐལ་ལས་གཅིག་ལས་སྦྱིན་པར་བྱ་སྟེ། ཕྱིས་མཉམ་པར་བགོའོ། །ལམ་དུ་ཞུགས་པ་ན་དགེ་
སློང་གཞན་ལ་སྣོད་ཤོ་གམ་གཞལ་དགོས་པ་མེད་གྲང་ཞེས་དྲིས་ཏེ་བླང་བར་བྱའོ། །རྣམ་པ་གཞན་དུ་ན་ཉེས་བྱས་སོ། །སྣོད་ཤོ་གམ་གཞལ་དགོས་པ་ལ་ལམ་དུ་བསྐུར་བར་མི་བྱའོ། །རས་སར་པ་ཞིག་ན་ནི་ཐ་ན་ཁ་ཚར་དང་ཕུན་ཚར་བཅད་དེ་བ་ལང་གི་ལྕི་བ་བཙིར་བས་ཁ་བསྒྱུར་ལ་བསྐུར་བར་བྱའོ། །
