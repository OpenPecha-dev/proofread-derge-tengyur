གང་དག་དགེ་སློང་བྱེ་མ་སྐྱེས་ལ་མ་སྨྱོས་པས་འདུལ་བ་སྩལ་དུ་བཟོད་པ་དེ་དག་ནི་ཅང་མ་གསུང་ཤིག །གང་དག་མི་བཟོད་པ་དེ་དག་ནི་གསུངས་ཤིག །འདི་ནི་ལས་བརྗོད་པ་དང་པོ་སྟེ། དེ་བཞིན་དུ་ལན་གཉིས་ལན་གསུམ་དུ་བཟླས་དགེ་འདུན་གྱིས་བཟོད་ཅིང་གནང་ནས། དགེ་འདུན་གྱིས་དགེ་སློང་བྱེ་མ་
སྐྱེས་ལ་མ་སྨྱོས་པས་འདུལ་བ་སྩལ་ལགས་ཏེ། འདི་ལྟར་ཅང་མི་གསུང་བས་དེ་དེ་བཞིན་དུ་འཛིན་ཏོ་ཞེས་གསུངས་པ་ཡིན་ནོ། །དེ་ལྟར་གདམས་ངག་ལ་སོགས་པ་འཇོག་པས་གཞན་སྤོང་བ་རྒྱུ་མེད་པ་སྨྱོན་པས་བྱས་པས་གླེང་བ་ཉིད་ལས་བྱུང་བའི་འཁྲུག་ལོང་མི་གདམས་པའི་ཕྱིར་རྩོད་པ་མ་སྨྱོས་པས་
འདུལ་བས་དག་པ་སྦྱིན་པས་ཞི་བ་ཡིན་ནོ། །མི་གདམས་པའི་ཕྱིར་རྩོད་པ་ཞི་བར་བྱ་བ་བཤད་ཟིན་ཏོ།། །།ད་ནི་ལྟུང་བའི་ཕྱིར་རྩོད་པ་ཞི་བར་བྱ་བ་བཤད་པར་བྱ་སྟེ། འཁྲུག་ལོང་གི་རྒྱུ་གསུམ་པ་ལྟུང་བ་དང་བཅས་པ་ཉིད་ལས་བྱུང་བའི་འཁྲུག་ལོང་ལྟུང་བའི་ཕྱིར་རྩོད་པ་ནི་ཁས་བླང་བར་བྱ་བ་དང་།
མངོན་སུམ་གྱིས་འདུལ་བ་དང་། རྩྭ་བཀྲམ་པ་ལྟ་བུས་ཕྱིར་བཅོས་པ་དང་། དེའི་ངོ་བོ་ཉིད་ཚོལ་དུ་གཞུག་པས་ངེས་པར་འཛིན་པ་ལ་སྦྱར་བར་བྱ་བ་ཉིད་སྦྱིན་པར་བྱའོ་ཞེས་གསུངས་པ་ལ། འཁྲུག་ལོང་གི་རྒྱུ་གསུམ་པ་ལྟུང་བ་དང་བཅས་པ་ཉིད་ལས་བྱུང་བའི་འཁྲུག་ལོང་ལྟུང་བ་ཕྱིར་རྩོད་པ་ལ་ཁས་
བླང་བར་བྱ་བ་ཉིད་ནི་རྣམ་པ་གཉིས་ཏེ། རྣམ་པར་ཤེས་པ་ནི་རང་གི་ངོ་བོའི་ཉེས་པས་ལྟུང་བ་དང་བཅས་པ་ཉིད་དང་། རྣམ་པར་མི་ཤེས་པ་རང་གི་ངོ་བོའི་ཉེས་པས་ལྟུང་བ་དང་བཅས་པ་ཉིད་དོ། །དེ་ལ་རྣམ་པར་ཤེས་པ་ནི་རང་གི་ངོ་བོའི་ཉེས་པས་ལྟུང་བ་དང་བཅས་པ་ཉིད་ཅེས་བྱ་བ་ནི་གང་ཟག་ལྟུང་
བ་དང་བཅས་པ་གླེང་བ་པོས་གླེངས་ནས་དངོས་པོ་དང་། ཕྱོགས་དང་བསམ་པ་དང་། ཆེ་བཞི་དང་། ཁས་བླངས་པ་ལ་སོགས་པའི་སྒོ་ནས་ལྟུང་བ་རྣམ་པར་ཤེས་པ་ནི་ལྟུང་བ་སྡེ་ལྔ་གང་ཡང་རུང་བ་དག་གི་དངོས་གཞི་འམ། སྦྱོར་བའམ་སྦྱོར་བའི་སྦྱོར་བ་ཞེས་བྱ་བ་ནི་རང་གི་ངོ་བོའི་ཉེས་པས་ལྟུང་བ་
དང་བཅས་པ་ཉིད་དོ། །རྣམ་པར་མི་ཤེས་པ་ན་རང་གི་ངོ་བོའི་ཉེས་པས་ལྟུང་བ་དང་བཅས་པ་ཉིད་ཅེས་བྱ་བ་ནི་གང་ཟག་ལྟུང་བ་དང་བཅས་པ་གླེང་བ་པོས་གླེངས་ནས། ལྟུང་བ་དྲན་པ་དགེ་འདུན་ལ་ཁས་བླངས་ནས། དེའི་འོག་ཏུ་བརྗེད་པས་དངོས་པོ་དང་། ཕྱོགས་དང་། བསམ་པ་དང་། ཆེ་
