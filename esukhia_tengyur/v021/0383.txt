རྗེ་ཉི་མ་རྟ་མཆོག་སྟེ། །གདུག་པ་གོ་ཆ་རྣམས་

ཀྱིས་བསྲུངས། །ཨོཾ་བཾ་རྡོ་རྗེ་ཕག་མོའོ། །ཧཾ་ཡོཾ་ཡ་མི་ནིའོ། །ཧྲིཾ་མོཾ་རྨོངས་བྱེད་མའོ། །ཧྲིཾ་ཧྲཱིཾ་སྐྱོད་བྱེད་མའོ། །ཧཱུཾ་ཧཱུཾ་སྐྲོད་བྱེད་མའོ། །ཕཊ་ཕཊ་ནི་ཙཎྜི་ཀ་སྟེ། ཡན་ལག་ཐམས་ཅད་ལ་མཚོན་ཆ་ཡིན་ཏེ། ལྟེ་བ་དང་སྙིང་ག་དང་དེ་བཞིན་དུ་ཁ་དང་། མགོ་བོ་
དང་། སྤྱི་གཙུག་དག་གི་མཚོན་ཆ་ཉིད་དོ། །མཛུབ་མོ་མདུད་བྱས་གུང་མོ་ངེས་པར་སྦྱར། །མཐེ་བོང་རྡོ་རྗེ་བརྟན་པོར་རབ་ཏུ་སྦྱར། །འདི་ནི་དཔྲལ་བའི་དབྱིངས་སུ་རབ་བཞག་ནས། །བསྐོར་ཞིང་བསྐོར་བས་ཀུན་དུ་བསྐོར་བར་བྱ། །རྐང་པ་མནན་ཅིང་གྱེན་དུ་བལྟ། །གྱེན་དུ་ཕེཾ་གྱི་སྒྲ་སྒྲོགས་
ཤིང་། ཕྱོགས་བཅུ་འཇིག་རྟེན་ཁམས་བཞུགས་པའི། །དཔའ་བོ་རྣལ་འབྱོར་མ་རྣམས་དགུག །ཨོཾ་ཡོ་ག་ཤུདྡྷཿསརྦ་དྷརྨྨཱ་ཡོ་ག་ཤུདྡྷོ྅ཧཾ། དེ་ནས་སྙིང་པོའི་ཕྱག་རྒྱ་སྦྱིན་པར་བྱ་སྟེ། ལག་པ་གཡས་པ་དང་གཡོན་པ་བསྐོར་ཞིང་མཁའ་འགྲོ་མའི་དམ་པ་བདེ་བར་གཞུག་པར་བྱའོ། །
ཨོཾ་ཤྲཱི་བཛྲ་ཧེ་ཧེ་རུ་རུ་ཀཾ་ཧུཾ་ཧཱུཾ་ཕཊ་བཛྲ་ཌཱ་ཀི་ནཱི་ཛཱ་ལ་སམྦ་རཾ་སྭཱ་ཧཱ་སྙིང་པོའི་སྔགས་སོ། །ཨོཾ་ཧྲཱིཿཧ་ཧ་ཧུཾ་ཧཱུཾ་ཕཊ་ཉེ་བའི་སྙིང་པོའི་སྔགས་སོ། །ཨོཾ་བཛྲ་བཻ་རོ་ཙ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་མའི་སྙིང་པོའོ། །ཨོཾ་བཛྲ་སརྦ་བུདྡྷ་ཌཱ་ཀི་ཎཱི་ཡེ་བཛྲ་ཝརྞྞ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། བཅོམ་ལྡན་
འདས་མའི་ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་ཀ་ར་ཀ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་ཙཎྜ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀུ་རུ་ཀུ་རུ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཙཎྜ་ཨཀྵཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བནྡྷ་བནྡྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་བྷཱ་བ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཏྲཱ་ས་ཡ་ཏྲཱ་ས་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་ནཱ་སེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵོ་བྷ་ཡ་ཀྵོ་བྷ་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་

ཙ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཱ་ཡུ་བེ་གེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྷཀྵ་བྷཀྵ་བ་ས་རུ་དྷི་རཱན་ཏྲ་མཱ་ལ་ཨ་བ་ལམྦི་ནེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སུ་ར་བྷཀྵཱི་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་གྲྀཧྞ་གྲྀཧྞ་ས་པ་ཏ་པཱ་ཏཱ་ལ་ག་ཏ་བྷུ་ཛཾ་ག་སརྦཾ་ཝཱ། ཏརྫ་ཡ་ཏརྫ་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤྱཱ་མ་དེ་བཱི་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨ
