བརྗོད་པ་ནི་མ་ཡིན་ཏེ་ཞེས་བྱ་བའོ། །དེའི་རྣམ་པར་རྟོག་པ་གཞན་ཉིད་སྐྱེ་བ་ཞེས་བྱ་བ་འདི་ཉིད་དོན་ཡིན་གྱི། ཤེས་པའམ་དོན་འདས་ཟིན་པ་དེ་ཉིད་སྣང་བ་ནི་མ་ཡིན་ནོ། །དེ་ནི་འདས་པ་ཉིད་ད་ལྟར་སྣང་བའི་མི་འཐད་པའི་ཕྱིར་རོ། །དེ་ཉིད་ཀྱིས་ནི་ཉམས་སུ་མྱོང་བའིའོ། །འདི་ཁོ་ན་ལྟར་དམིགས་པ་
ཞེས་བྱ་བ་ནི་རྨི་ལམ་དུ་མཐོང་བ་དང་སད་པར་མཐོང་བའི་ཤེས་པ་དག་གོ། །གལ་ཏེ་བག་ཆགས་ཡོད་དོ་ཞེས་བྱ་བ་ཙམ་ལ་ཚད་མ་ཡོད་མོད། རྣམ་པར་ཤེས་པ་སྔ་མའི་བྱེ་བྲག་ཡིན་པ་ལ་ནི་ཚད་མ་མེད་དེ་དེ་ནི་འདས་པའི་མངོན་སུམ་མེད་པའི་ཕྱིར་རོ་ཞེ་ན། ཐོག་མ་མེད་པའི་ཞེས་
གསུངས་སོ། །འདི་ཉིད་གསལ་བར་བྱེད་པ་ནི་ཇི་ལྟར་ཞེས་བྱ་བའོ། །ཕྱི་རོལ་ནི་བུམ་པ་ལ་སོགས་པའོ། །ཕྱི་རོལ་ནི་དེའི་རྒྱུ་མཚན་ནི་བག་ཆགས་ཀྱི་རྒྱུ་མཚན་ནོ། །རྒྱུ་མཚན་ཉིད་བརྗོད་པ་ནི་རྒྱུ་མཚན་ཞེས་པའི་བྱེ་བྲག་སྟེ་ཞེས་བྱ་བ་སྟེ། བག་ཆགས་དེ་ཡིན་ནོ། །ཤེས་པའི་བྱེ་བྲག་སྟེ་
ཞེས་བྱ་སྟེ། དེའི་ཞེས་གསུངས་ཏེ། ཤེས་པ་དེའི་དབྱེ་བའང་བག་ཆགས་ཀྱི་བྱེ་བྲག་དེ་ལས་ཡིན་ནོ། །དེའི་ནི་དོན་གྱིའོ། །དེ་ཞེས་བྱ་བ་ནི་དོན་གྱི་དབྱེ་བ་རྒྱུ་མཚན་མེད་པའོ། །འོ་ན་དོན་གྱིས་གཡོ་བའི་རྒྱུ་ཤེས་པའི་དབྱེ་བ་ཡིན་མོད་སྙམ་ན། ཤེས་པའི་དབྱེ་བ་ལས་ཞེས་བྱ་བའོ། །མ་འབྲེལ་བ་ཉིད་དོ་ཞེས་
བྱ་བ་ནི་རང་གི་བདག་ཉིད་ལ་བྱེད་པ་འགལ་བའི་ཕྱིར་རོ། །གལ་ཏེ་དོན་གྱི་དབྱེ་བ་དེ་རང་ཁོ་ནའི་རྒྱུ་མཚན་ཡིན་ན། དེའི་ཚེ་ཉེས་པ་འདིར་འགྱུར་བ་ཞིག་ན། དེ་ལྟར་ནི་འདོད་པ་མ་ཡིན་གྱི་འོན་ཀྱང་དོན་གྱི་དབྱེ་བ་གཞན་ཡིན་ནོ་ཞེ་ན། འོན་ཏེ་ཞེས་བྱ་བའོ། །མི་སྣང་བའི་བག་ཆགས་ཐ་དད་པར་བྱེད་པ་ལ་ཡང་

ལྕགས་སྒྲོག་གིས་བཅིངས་པའི་བདག་ཉིད་མེད་ཀྱང་དེ་ལྟ་བུའི་ཚུལ་དུ་ཞེས་བྱ་བ་འཁོར་བའི་ཚུལ་དུ་འགྲོ་བའི་རྒྱུ་ཡིན་པ་དེ་བཞིན་དུ་བག་ཆགས་ཀྱི་དབྱེ་བས་ཀྱང་ཡིན་ནོ་ཞེས་སྦྱར་རོ། །དེར་ཡང་ནི་འདོད་པ་ལ་སོགས་པའོ། །གལ་ཏེ་དེར་ཕྱི་རོལ་ཉིད་དམིགས་པ་ཡིན་ན། དེའི་འདོད་པ་ཅན་གྱི་
