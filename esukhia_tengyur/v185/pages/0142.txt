བ་ནི་དོན་ཞེས་བྱ་བའོ། །ཤེས་པ་འདི་ནི་རྨི་ལམ་དུ་མཐོང་བ་ལས་མེད་དོ་ཞེས་བརྗོད་པ་ནི་མ་ཡིན་ཏེ་ཞེས་བྱ་བའོ། །དེའི་ཕྱིར་དེ་ནི་ཤེས་པ་ཁོ་ན་ཡིན་ནོ། །འོ་ན་དེ་ནི་སྔོན་པོ་ལ་སོགས་པའི་རང་བཞིན་ཅན་མིང་དང་མཚན་མ་ཡིན་ནོ་སྙམ་ན། འདིས་ནི་ཞེས་གསུངས་སོ། །ཐུན་མོང་མ་ཡིན་པ་ཉིད་དང་།
གཟུང་བ་མ་དམིགས་པ་ཉིད་དུ་བསྟན་པས་སོ། །གཞན་ཡང་ཞེས་བྱ་བ་ནི་འཆད་པར་འགྱུར་བའི་རྣམ་འགྲེལ་གྱི་ཕྱེད་དང་པོའི་དོན་བརྗོད་པའོ། །མིག་ལ་སོགས་པའི་བློ་གང་ཡིན་པ་འདི་ཉིད་དེ་འདོད་པའི་དོན་གང་ཡིན་པ་དེ་དང་ལྡན་པར་རྒྱུ་མཚན་གང་གིས་འདོད། གཟུང་བ་ཉིད་དུ་སྣང་བས་ནི་མ་ཡིན་ཏེ།
རྨི་ལམ་ལ་སོགས་པ་དང་ཐུན་མོང་ཡིན་པའི་ཕྱིར་རོ། །གནོད་བྱེད་མེད་པ་ཡང་སླུ་བ་མེད་པའོ། །མི་སླུ་བ་ཁོ་ན་ཡིན་ལ། དེ་ཡང་ཐོབ་པའི་མཐར་ཐུག་པ་ཡིན་ནོ་ཞེས་བརྗོད་པ་ནི་གལ་ཏེ་དོན་ནུས་ཞེས་བྱ་བའོ། །གཞན་ལ་ནི་རྨི་ལམ་དང་སྤྱི་ལ་སོགས་པའི་ཤེས་པའོ། །དོན་མེད་པ་ཅན་ཐོས་པ་ནི་རྒྱུ་
མཚན་མེད་པའི་ཕྱིར་རོ། །གཞན་ཞེས་བྱ་བ་ལ་སོགས་པས་བཤད་པ་དེ་ཉིད། དེ་ཉིད་རྣམ་འགྲེལ་འཆད་པར་བྱེད་པ་ན་བརྗོད་པ་ནི་གཞན་ཡང་ཞེས་བྱ་བའོ། །འདི་ལ་ཞེས་བྱ་བ་འདི་དོན་གྱིས་བསྐྱེད་པ་ཉིད་དུ་རྣམ་པར་གཞག་པའོ། །འོན་ཏེ་གང་གིས་དེ་ལྟར་རྣམ་པར་འཇོག་པ། མིག་ལ་སོགས་
པའི་ཤེས་པ་ལ་ཁྱད་པར་ཡོད་པ་ཁོ་ནའོ་སྙམ་དུ་དོགས་ན་སྣང་བ་ཉིད་དུ་ཞེས་གསུངས་སོ། །རྗེས་སུ་འགྲོ་བ་དང་ཞེས་བྱ་བ་ནི་ཁྱད་པར་ཡོད་པར་དོགས་པ་བསུ་བའོ། །ཇི་ལྟར་མཐོང་ཞེ་ན། འདི་ལྟར་ཞེས་བྱ་བའོ། །སྔར་ནི་ཐོབ་པ་ལས་སོ། །རང་དང་གཞན་གཉིས་ཀས་ཀྱང་མངོན་སུམ་གྱིས་ནི་མཐོང་
བ་མ་ཡིན་ནོ་ཞེས་བརྗོད་པ་ནི་རང་ཉིད་ཀྱིས་ཞེས་བྱ་བའོ། །རྣམ་པ་དེ་ལྟ་བུའི་སྣང་བ་ལྡོག་པའི་སྟོབས་ལས་བྱུང་བའི་རྗེས་སུ་དཔག་པས་སོ། །མིག་ལ་སོགས་པའི་རྣམ་པར་ཤེས་པ་ནི། བཤད་པའི་ཁྱད་པར་གྱི་དོན་དང་ལྡན་པ་ཡིན་ཡང་རྨི་ལམ་གྱི་ཤེས་པ་ནི་དོན་མེད་པ་ཡིན་ཏེ། གལ་ཏེ་དེ་ལྟ་ན་ནི་
ཞེས་བྱ་བའོ། །དམིགས་པ་དང་བཅས་པ་ལ་སོགས་པའི་རྣམ་པར་དབྱེ་བ་འདི་ནི་ཐ་སྙད་པ་དང་འདའ་བར་བྱ་བ་མ་ཡིན་པ་ཡང་ཡིན་ནོ་ཞེས་བརྗོད་པ་ནི། འདི་ལྟར་ཞེས་བྱ་བའོ། །འདི་ལ་ནི་དོན་མེད་པ་ལ་ཞེས་བྱ་བ་ནི་རྨི་ལམ་གྱི་ཤེས་པ་ལ་སྟེ་དོན་བྱ་བ་མེད་པའི་ཕྱིར་རོ། །འོན་ཏེ་ཡང་གཞན་གྱི་ལྟར་ན་ཤེས་
