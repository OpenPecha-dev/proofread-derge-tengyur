ཡཾ། པྲྀ་ཀ་བྷ་ཡཾ། རཱ་ཛྷཿཔུ་རུ་ཥ། རཱ་ཛ་པུ་རུ་ཥཿ ཨ་ཀྵེཥུ་ཤཽཎྜཿ ཨཀྵ་ཤཽཎྜཿ། ༈ སྐབས་ལ་ལར་ཨམྲ་ལ་སོགས་པའི་མཐའ་ཅན་གྱི་ཚིག་དེ་བསྡུ་བ་བྱས་ནས་སླར་ཕྱི་མར་སྤོ་བ་ཡང་ཡོད་དེ། ཨགྣཽ་ངི ཨཱ་ཧི་ཏཿ ཞེས་པ་ལ་རྣམ་དབྱེ་ཕྱིས་སྔ་ཕྱི་བརྗེས་
པས། ཨཱ་ཧི་ཏཱགྣིཿ སྤྱང་གི་ལས་འཇིགས་སོ། །ངས། རྒྱལ་པོའི་སྐྱེས་བུ་འམ་རྒྱལ་བུའི་མི་སྣ། སུཔྲ། ཤོ་ཀ་ཆོ་ལོ་མཁན། དེ་བཞིན་དུ། པུརྦཾ་ཨམྲ བྷཱུ་ཏཿ བྷཱུ་ཏ་པརྦཿ ༈ ཡང་བསྡུ་བའི་སྐབས་འདིར་ཚིག་གཅིག་ཏུ་བྱེད་པའི་ཚེ་ཚིག་སྔ་མ་ན་ཎར་འགྱུར་མེ་ལ་འབུལ་བའི་སྲེག་རྫས། བའི་རྒྱུ་མཚན་ཅན་ལས་ཕ་རོལ་
མེ་ལ་འབུལ་བའི་སྲེག་རྫས། གྱི་ཚིག་ཕྱི་མ་ན་ཎར་འགྱུར་བ་ཡང་ཡོད་དེ། ཤ་རཱ་ཎཱཾ་བ་ནཾ། ཨཱམྲ། ཞེས་པ་ལ། ཤ་ར་བ་ཎཾ། དེ་བཞིན་དུ། ཨམྲ་ སྔོན་ནས་བྱུང་བའམ་སྔོན་ནས་ཡོད་པ། ཨཱ་མྲཱ་ཎཱཾ་ཨམྲ བ་ནཾ། ཨཱ་མྲཱ་པ་ཎཾ། ༈ པཱ་ན་སྱ་བཱཿ པཱ་ནའི་ཡང་ནའོ། །ཚིག་གཅིག་ཏུ་བསྡུས་པའི་སྔ་མ་ན་ཎར་འགྱུར་བའི་
རྒྱུ་མཚན་ཅན་ལས་ཕ་རོལ་གྱི་པཱ་ནའི་ན་ཡིག་ནི་ཎ་ཡིག་ཏུ་ཡང་ན་འགྱུར་ཨཱམྲ། ཏེ། མདའི་ནགས། སུ་རཱ་ཡཱཿབཱ་ནཾ། ཞེས་པ་ལ། སུ་རཱ་པཱ་ཎི། སུ་རཱ་པཱ་ནཾ། ༈ ན་ཉི། ནཉ། ཨཱམ྄ ཨཱམྲའི་ནགས། ལའོ། །དགག་པའི་སྒྲ་ནཉ་སྔོན་དུ་འཇོག་པ་དེ་ཡང་དེའི་སྐྱེས་བུའི་བསྡུ་བ་ཞེས་བྱའོ། ༈ །ནཱ།
ན་ཨའོ། །ཚིག་སྡུད་ཀྱི་བྱ་བ་ལ་འཇུག་པ་ན་ནཱ་ཀ་སོགས་མ་ཡིན་པའི་དགག་པའི་སྒྲ་ནཉ་ནི་ཨ་ཡིག་ཏུ་འགྱུར་ཏེ། བཏུང་བ་བཅུད་དང་ལྡན་པ་སྟེ་ཆང་། ན་བྲཱཧྨ་ཎཿ ཞེས་པ་ལ། ཨ་བྲཱཧྨ་ཎ། དེ་བཞིན་དུ། དྷརྨཾ་བི་རུ་དྡྷཿ ཞེས་པ་ལ། ཨ་དྷརྨཾ། གྲ་ཧ་ཎ་སྱཱ་བྷ་བཿ ཨ་གྲ་ཧ་ཎཾ།
སོགས་སོ། ༈ །ཨ་ན་སྭ་རེ། དབྱངས་ལ་ཨནའོ། ། ཚིག་སྡུད་ལ་དགག་པའི་ནཉ་སྔོན་དུ་འཇོག་པའི་ཚིག་དེའི་ཐོག་མའི་ཡི་གེ་དེ་དབྱངས་ཡིན་ན་ནཉྲ་ནི། ཟམ་ཟེམ་ཡིན་པ། ཨནྲ་ཞེས་པར་འགྱུར་ཏེ། ཨ་ཤྭཱ་ད་ནྱཿ ཞེས་ ཆོས་འཇོམས པ་ལ་ཨ་ནྱ་ཀྱི་དོན་ནཉྲ་ གཟའ་འཛིན་མི་འབྱུང་། སྔོན་དུ་
བཞག །དེ་ཨནྲ་དུ་བསྒྱུར་བས། ཨ་ན་ཤྭ། རྟ་མ་ཡིན་ནམ་རྟ་ལས་གཞན། ༈ གསུམ་པ་ཟླས་དབྱེའི་བསྡུ་བ་ནི། ཙཱརྠེ་དྭནྡྭ དང་གི་དོན་ལ་ཟླས་དབྱེའོ། །ལས་གཉིས་ལ་བྱ་བ་གཅིག་འཇུག་པ་ལ་ཕན ང་སི། རྟ་ལས་གཞན། ཚུན་མི་འགལ་བ་ལྟ་བུའི་དང་གི་དོན་གྱི་བསྡུ་བ་ནི་ཟླས་དབྱེའི་
