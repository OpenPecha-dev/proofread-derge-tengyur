པྲ་ཙུ་པི་ཏ་མི་ད་མུ་དི་ཏ་མུ་པ་སྠི་ཏ་པཱུརྦཾ། །པཱ་དེ་སྱུརྻ་དི་ནཽ་ས་ནཽ་ན་སཽ་ཙ་ཏྲྀ་ཏཱི་ཡེ། །ཀ་བི་ཡསྟ་ད་བྷི་ག་དནྟི་བརྡྷ་མཱ་ནཾ། །པྲ་ཙུ་པི་ཏ་ས་དྲྀ་ཤ་ཀ་མ་ནྱ་དི་ཧ་བྷ་བ་ཏི་ས་ཀ་ལཾ་པྲ་ཏི་ནཾ་ཡ་ཙ་ར་ཎ་ག་ཏ་ལ་ལཀྵྨ་
དེ་ནས་ན་སྦྱར་ན་ན་ཛ་ཡ་ཉེར་གནས་སྔ་འདི་དལ་འགྲོར་བརྗོད། །གལ་ཏེ་རྐང་པ་གསུམ་པོ་ལ་ནི་ན་དག་ས་ན་དག་དང་། །ན་ས་དེ་ནི་འདིར་གཞན་མཐའ་དག་དལ་འགྲོ་དང་འདྲ་ཞིང་། །རྐང་པའི་གནས་ཀྱི་མཚན་ཉིད་སྒྲུབ་པ་སོ་སོར་ངེས་པ་སྙན་ངག་མཁན་པོ་རྣམས་ཀྱིས། །དེ་ནི་འཕེལ་བ་ཅན་དུ་མངོན་པར་བརྗོད་
བི་དྷཱ་ནཾ། །ཨསྨིནྣེ་བ་ཡ་དཱ་ཏྲྀ་ཏཱི་ཡ་ཀེ་ཏ་ཛ་རེ་ཕཱཿ། །པྲ་ཐ་མེ་ཙ་བི་ར་ཏི་རཱཪྵ་བྷཾ་བ་དནྟི། །ཏཙྪུདྡྷ་བི་རཱཌྦི་ཤེ་ཥ་ཎཾ། །པྲ་ཙུ་པཾ་ཏ་ག་ཎ་ས་མ་མ་ས་རཾ་ཏྲྀ་ཏ་ཡཉྩེཏྲ། །པྲ་ཙུ་པི་ཏ་པྲ་བྷེ་དཱཿ། བི་ཥ་མ་བྲྀཏྟཱ་
བར་གྱུར་པ་དག་ཡིན་ནོ། །གང་ཚེ་གསུམ་པ་ཏ་ཛ་ར་རྣམས་འདི་ཉིད་དང་པོར་ངལ་བསོ། །གལ་ཏེ་གཞན་གསུམ་རྣམས་ནི་དལ་འགྲོ་དག་དང་ཚོགས་དེ་དག །མཚུངས་པར་གྱུར་པ་དག་པ་སྣང་ལྡན་གྱི། །ཁྱད་པར་རབ་མཆོག་ལྡན་དུ་དེ་ནི་རྗོད་པར་བྱེད་པ་ཡིན། །དལ་འགྲོའི་བྱེ་བྲག་རྣམས་ཏེ། མི་མཉམ་པའི་བྲྀཏྟཱ་
ནི།། །།ཨུགྟཱ་བྲྀཏྟ་བི་ཀལྤ་ཨི་ཏི་ཨཱུརྡྷྭཾ་ཛཱ་ཏ་ཡོཉབྷི་དྷཱ་ཏ་བྱཱཿ། །ན་བྱཉྫ་ནེ་ན་མཱ་ཏྲཱ་ལ་གྷུ་ནཻ་ཀ་བྷ་བ་ཏི་གུ་རུ་ཎཱ་དྭེ། །པཉྩ་བྷ་ཛ་སཱ་གུ་རུ་ཡུ་གཾ་ལ་གྷུ་ཀ་ཙ་ཏུཥྐཾ་ཀ་ཎཱཤྩ་ཏུརྨཱ་ཏྲཱཿ། །ཏེ་ཙཱཿཁོ་པི་ཙ་ཏུརླཿཀ་ཨེ་ཀ་མཱ་ཏྲོ་
རྣམས་སོ།། །།བྲྀཏྟཱའི་རྣམ་པར་རྟོག་པ་བཤད་ཟིན་ནས། །འདི་ལས་ཕྱི་ནས་ཛཱ་ཏི་བརྗོད་པར་བྱ། །ཡང་བ་ཕྱི་མོ་གཅིག་སྟེ་ལྕི་བ་ནི། །ཕྱི་མོ་གཉིས་ཏེ་གསལ་བྱེད་དག་གིས་མིན། །བྷ་དང་ཛ་ས་ལྕི་ཟུང་ཡང་བ་བཞི་ཚོགས་ལས། །ཕྱི་མོ་བཞི་པ་དེ་ལྔ་རྣམས་ནི་ཙ་ཞེས་བྱ། །ཡང་བ་བཞི་ནི་ཁ་ཡང་ཡིན། །ཕྱི་མོ་གཅིག་པ་
