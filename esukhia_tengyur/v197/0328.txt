ཙཀྲ་ཀ་ར་ཀཾ། །ཤུ་ཥི་རཱ་བི་དྲུ་མ་ལ་ཏཱ་ཀ་པོ་ཏཱཾ་གྷྲིརྣྣ་ཊཱི་ན་ལཱི། །དྷ་མ་ནྱཾ་ཛ་ན་ཀེ་ཤཱི་ཙ་ཧ་ནུརྷཊྚ་བི་ལཱ་སི་ནཱི། །སུཀྟིཿཤཾ་ཁཿཁུ་རཿཀོ་ལ་ད་ལཾ་ན་ཁ་མ་ཐཱ་ཌྷཱ་ཀཱི། །ཀཀྵཱི་མྲྀཏྶྣཱ་ཏུ་བ་རི་ཀཱ་མྲྀཏྟཱ་ལ་ག་སུ་རཱཥྚྲ་ཛེ། །ཀུ་ཊཱནྣ་ཊཾ་ཧ་ཤ་པུ་
དང་འཁོར་ལོ་བྱེད། །སྦུབས་ཅན་བྱི་རུ་འཁྲི་ཤིང་དང་། །ཕུག་རོན་རྐང་དང་ན་ལཱི་དང་། །ཛ་ན་ཀེ་ཤི་དྷ་མ་ནྱཾ། །ཧ་ནུ་ཧཊྚེ་རྣམ་སྒེག་མ། །ཉ་ཕྱིས་དུང་དང་ཁུ་ར་དང་། །ཀོ་ལ་ད་ལཾ་སེན་མོ་དང་། །དེ་ནས་ཨ་ཌྷཱ་ཀཱི་ཀཀྵཱི། །མྲྀཏྶྣཱ་ཏུ་བ་རི་ཀ་དང་། །མྲྀཏྟཱ་ལ་ཀ་ཡུལ་བཟངས་སྐྱེ། །ལི་ག་དུར་དང་ད་ཤ་
རཾ་བཱ་ནེ་ཡཾ་པ་རི་བེ་ལ་བཾ། །བླ་བ་གོ་པུ་ར་གོ་ནརྡྡ་ཀོ་བརྟྟཱི་མུསྟ་ཀཱ་ནི་ཙ། །གྲནྠི་པརྞྞཾ་ཤུ་ཀཾ་བརྷིཿཔུཥྤཾ་སྠཽ་ཎེ་ཡ་ཀུཀྐུ་རེ། །མ་རུནྨཱ་ལཱ་ཏུ་པི་ཤུ་ནཱ་སྤྲྀཀྐཱ་དེ་བཱི་ལ་ཏཱ་ལ་གྷུཿ། །ས་མུ་དྲཱནྟཱ་བ་དྷཱུ་ཀོ་ཊཱི་བཪྵཱ་ལཾ་ཀོ་པི་ཀེ་ཏྱ་པི། །ཏ་པ་སྭི་ནཱི་ཛ་ཊཱ་ལཱ་ཛ་ཊཱ་
པུར། །ནགས་སྐྱེས་པ་རི་ཕེ་ལ་བཾ། །པླ་བ་གོ་པུར་གོ་ནརྡྷ། །ཀེ་བརྟྟི་མུསྟ་ཀཱ་ནི་ཙ། །ཚིགས་འདབ་ཤུ་ཀཾ་བརྷི་དང་། །མེ་ཏོག་གནས་སྐྱེས་ཀུཀྐུ་རེ། །མ་རུནྨཱ་ལ་བི་ཤུ་ནཱ། །པྲྀཀྐཱ་ལྷ་མོའི་འཁྲི་ཤིང་ཉུང་། །བ་དྷུ་ཀོ་ཌི་རྒྱ་མཚོའི་མཐའ། །བཪྵཱ་ལཾ་ཀོ་པི་ཀེ་ཡང་། །དཀའ་ཐུབ་ཅན་མ་རལ་ཅན་མ། །
མཱཾ་སཱི་ལོ་མ་སཱ་མི་ཤཱི། །ཏྭ་ཀྵནྟྲ་སཱུཏྐ་ཊཾ་བྷྲྀངྒ་ཏྭ་ཙཾ་ཙོ་ཙཾ་བ་རཱངྒ་ཀཾ། །ཀརྩཱུ་ར་ཀོ་ཊྲཱ་བི་ཌ་ཀཿཀཱ་ལྱ་ཀོ་བེ་དྷྲ་མུ་ཁ་ཀཿ། །ཨོ་ཥ་དྷྱོ་ཛཱ་ཏི་མཱ་ཏྲེ་སྱུ་ར་ཛཱ་ཏོ་སརྦྦ་མཽ་ཥ་དྷཾ། །ཤཱ་ཀཱ་ཁྱཾ་པཏྟྲ་པུཥྤཱ་དི་ཏཎྜ་ལཱི་ཡོཉལྤ་མཱ་རི་ཥཿ། །བི་ཤལྤཱགྣི་
སྤང་སྤོས་སྤུ་ཅན་མཱཾ་སིའོ། །པགས་ལོ་སུཏྐཊ་བྷྲྀངྒ་དང་། །པགས་པ་ཙོ་ཙམ་ཡན་ལག་མཆོག །ཀརྩཱུ་ར་ཀོ་བི་ཌ་ཀ །ཀཱལྤ་ཀ་དང་རིག་བྱེད་གདོང་། །ཨོ་ཥ་དྷི་ནི་རིགས་ཙམ་ཡིན། །ར་ཛཱ་ཏོ་ནི་ཀུན་ལ་སྨན། །ཤཱ་ཀཱ་ཁྱཾ་འདབ་མེ་ཏོག་སོགས། །ཏཎྜུ་ལཱི་ཡོ་ཆུང་དང་འཕགས། །བི་ཤལ་བ་དང་མེ་
