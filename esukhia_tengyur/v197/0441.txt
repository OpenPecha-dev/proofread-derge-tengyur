༄༅། །གྷཾ་སྡིག་སྡུག་ཉོན་མོངས་ལ། །འདོད་དང་ཉུང་ལ་ལ་གྷུའོ། །གྷ་མཐའ་རྣམས་སོ། །ཀཱ་ཙ་དྲ་བའི་ཐག་པ་དང་། །མཆིང་བུ་དང་ནི་ལིང་ཐོག་གོ། །ཕྱིར་ལོག་རྒྱ་ཆེན་པྲ་པཉྩ། །ཤུ་ཅི་མེ་དང་ཟླ་བ་དང་། །ཤིན་ཏུ་གནོན་པ་དག་དང་ནི། །སྐྱེས་བུ་རྣམ་དག་དཀར་པོ་གསུམ། །མངོན་འཁྱུད་དང་ནི་
ཧཱ་ཡཱཉྩ་ག་བྷསྟཽ་ཙ་རུ་ཙིཿསྟྲི་ཡཱམ། །ཀེ་ཀི་ཏཱརྐྵྱཱ་བ་ཧི་བྷུ་ཛཽ་དནྟ་བི་པྲཱཎྜ་ཛཱ་དྭི་ཛཱཿ། །ཨ་ཛཱ་བིཥྞུ་ཧ་རཙྪ་གཱ་གོཥྛ་དྷྭ་ནི་བ་ཧཱ་བྲ་ཛཱཿ། །དྷརྨྨ་ར་ཛཽ་ཛི་ན་ཡ་མཽ་ཀུཉྫོ་དནྟེ་པི་ནསྟྲི་ཡཱཾ། །བ་ལ་ཛེ་ཀྵེ་ཏྲ་པཱུརྡྭཱ་རེ་བ་ལ་ཛཱ་བལྒུ་དཪྴ་ནཱ། །
