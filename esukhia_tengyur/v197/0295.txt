ཏོ་མུ་ཁཾ། །ཨམྦྷོ྅རྞྞ་སྟོ་ཡ་པཱ་ནཱི་ཡ་ནཱི་རཀྵཱི་རཱམྦུ་སམྦ་རཾ། །མེ་གྷ་པུཥྤཾ་གྷ་ན་ར་སསྟྲི་ཥུ་དྭེ་ཨཱ་སྱ་མམྨ་ཡཾ། །བྷངྒསྟ་རཾ་ག་ཨཱུརྨི་བཱ་སྟྲི་ཡཱཾ་བཱི་ཙི་ར་ཐོརྨྨི་ཥུ། །མ་ཧཏྶྡཱུལློ་ལ་ཀལློ་
ཀུན་སྒོ་བསྙེན་བྱེད་དང་། །བཏང་འགྲོ་སྐྱོང་བྱེད་བཏུང་བྱ་དང་། །ཟ་བྱེད་བཏུང་བྱེད་བདེ་བ་མཆོག །སྤྲིན་གྱི་མེ་ཏོག་སྤྲིན་གྱི་བཅུད། །གསུམ་གཉིས་བྱེད་པ་རྣམ་འགྱུར་རོ། །དབའ་རླབས་གཉེར་མ་ཁྲེམ་གཉེར་རིམ། །དེ་ནས་ཀློང་དང་དཔའ་ཆེན་འཁྲུག །སྟེང་འཕར་དང་ནི་རབ་འཁྲུག་
ལླཽ་སྱཱ་དཱ་བརྟྟོ྅མྦྷ་སཱཾ་བྷྲ་མཿ། །པྲྀ་ཥནྟི་བིནྡུ་པྲི་ཥཏཱཿསུ་མཱཾ་སོ་བི་སྲུ་ཥཿསྟྲི་ཡཿ། །བ་ཀྲཱ་ཎི་པུ་ཀྲ་བྷེ་དཱཿསྱཱུ་བྷྲ་མཱཤྩ་ཛ་ལ་ནིརྒྒ་མཱཿ། །ཀཱུ་ལཾ་རོ་དྷཤྩ་ཏཱི་རཉྩ་པྲ་ཏཱི་རཉྩ་ཏ་ཊནྟྲི་ཥུ། །པཱ་རཱ་བཱ་རེ་པ་རཱརྦྦཱ་ཙཱི་ཏཱི་རེ་པཱ་ཏྲནྡ་
ཡིན། །འཁྱིལ་ཅན་འཁོར་ཆུ་ཆུ་འཁོར་དང་། །ཆུ་ཡི་ཐིགས་པ་ཆུ་འཐོར་ཕོ། །ཆུ་ཡི་ཟེར་མ་མོ་ཡིན་ནོ། །རྒྱུན་འཁྱོག་དང་ནི་གྱ་གྱུ་འགྲོས། །ཡུར་བ་ངང་ནི་རྣང་མའོ། །འགྲམ་དང་འགོག་དང་ཆུ་ངོགས་དང་། །འཇུག་ངོགས་དང་ནི་ཆུ་འགྲམ་གསུམ། །ཕ་རོལ་ཚུ་རོལ་ཕན་ཚུན་ངོགས། །དེ་ཡི་བར་ན་
དནྟ་རཾ། །དྭི་པོ྅སྟྲི་ཡཱ་མནྟ་རཱི་པཾ་ཡ་དནྟརྦཱ་རི་ཎསྟ་ཊ། །ཏོ་ཡོཏྠི་ཏནྟཏྤུ་ལི་ནཾ་སཻ་ཀ་ཏཾ་སི་ཀ་ཏཱཾ་མ་ཡཾ། །ནི་ཥདྦ་རཱསྟུ་ཛམྦཱ་ལཿཔཾ་ཀྐོ྅སྟྲི་ཤཱ་ད་ཀརྡྲ་མཽ། །ཛ་ལོཙྪྭཱ་སཱཿཔ་རཱི་བཱ་ཧཱཿཀཱུ་པ་ཀསྟུ་བི་དཱ་ར་ཀཱཿ། །ནཱ་བྱནྟྲི་ལིངྒཾ་
རྒྱུན་ལམ་མོ། །ཆུ་བོ་གཉིས་ལྡན་མོ་མིན་ཞིང་། །བར་གླིང་གང་ལ་ཆུ་འགྲམ་དབུས། །ཆུ་བོས་ཉེར་བཞག་དེ་ནི་སྒང་། །བྱེ་མའི་ངོགས་དང་བྱེ་མའི་ཐང་། །འབྱིང་བྱེད་རྫབ་དང་འདམ་དང་ནི། །མོ་མིན་འདམ་དང་འཇིམ་པའོ། །ཆུ་བོ་སྒྱུར་བ་ཆུ་འབབ་པ། །བུད་དང་ཤྭ་ཞེས་བྱ་བའོ། །གྲུ་རབ་རྟགས་གསུམ་
