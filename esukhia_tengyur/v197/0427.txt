ཏཿ། །ཨ་ཎཱ་ཡཱ་ས་ཀྲྀ་ཏཾ་ཕཱཎྞཾ་སྭ་ནི་ཏཾ་དྷྭ་ནི་ཏཾ་ས་མི། །བདྡྷེ་སནྡཱ་ནི་ཏཾ་མཱུརྞྞ་མུདྡྭི་ཏཾ་སནྡི་ཏཾ་སི་ཏཾ། །ནིཥྤ་ཀྭཾ་ཀྭ་ཐི་ཏཾ་པཱ་ཀེ་ཀྵཱི་རཱ་ཛྱཱ་པ་ཡ་སཱཾ་ཤྲྀ་ཏཾ། །ནི་བཱ་ནོ་མུ་ནི་བ་ཧྣྱཱ་དཽ་ནིརྦཱ་ཏ་སྟུ་ག་ཏེཉནི་ལེ། །པ་ཀྭཾ་པ་རི་ཎ་ཏེ་

གཱུ་ཎཾ་ཧནྣེ་མཱི་ཌྷཾ་ཏུ་མཱུ་ཏྲི་ཏེ། །པུཥྚེ་ཏུ་པུ་ཥི་ཏ་སོ་ཌྷེ་ཀྵཱནྟ་མུ་དྭཱ་ན་མུངྒ་ཏེ། །དཱནྟསྟུ་ད་མི་ཏེ་ཤཱནྟཿཤ་མི་ཏེ་པྲཱརྠི་ཏེཉརྡི་ཏཿ། །ཛྙཔྟསྟུ་ཛྙ་པི་ཏེ་ཙྪནྣཾ་ཤྪཱ་དི་ཏེ་པཱུ་ཛི་ཏེཉཉྩི་ཏཾ། །པཱུརྞྞསྟུ་པཱུ་རི་ཏེ་ཀླིཥྚ་ཀླི་ཤི་ཏེཉབ་སི་ཏེ་སི་ཏཿ། །
བའོ། །བཤང་བ་གཅི་བ་བཅགས་བྱས་པ། །རྒྱས་པ་དང་ནི་གང་བའོ། །དུས་པ་དང་ནི་བཟོད་པའོ། །རྒྱས་བཏབ་པ་དང་ཨབ་བཏབ་པ། །ཉོན་མོངས་བཟོད་པ་དུལ་བྱ་སྟེ། །ཞི་དུལ་སློང་དང་ཞུ་བའོ། །བསྒོ་བ་དང་ནི་སློབ་པའོ། །གཡོགས་པ་དང་ནི་ཕུབ་པའོ། །མཆོད་དང་ཕུལ་དང་གང་དང་རྫོགས། །
