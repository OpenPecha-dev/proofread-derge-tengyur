བ་པཱ་རི་པླ་བ་པ་རི་པླ་བེ། །ཨ་ཏི་རིཀྟཿས་མཱ་དྷི་ཀོ་དྲྀ་ཌྷ་སནྡྷིསྟུ་སཾ་ཧ་ཏཿ། །ཀཀྑ་ཊཾ་ཀ་ཋི་ནཾ་ཀྵཱུ་རཾ་ཀ་ཋོ་རཾ་ནིཥྛུ་རཾ་དྲྀ་ཌྷཾ། །ཛ་ཋ་རཾ་མཱུརྟྟི་མནྨཱུརྟྟཾ་པྲ་བྲྀདྡྷཾ་པྲཽ་ཌྷ་མེ་དྷི་ཏཾ། །པུ་རཱ་ཎ་པྲ་ཏ་ན་པྲཏྣ་པུ་རཱ་ཏ་ན་ཙི་རནྟ་ནཱཾཿ། །
སིག་དང་། །ཡོངས་སུ་འགུལ་བ་ཞེས་པའོ། །སྟོང་པ་དང་ནི་ཏིང་ངེ་འཛིན། །བརྟན་པོ་དང་ནི་མི་འགྱུར་བ། །སྲ་མཁྲེགས་མི་ཤིགས་ཀ་ཋོ་རཾ། །མི་འཇིགས་མི་ཕེད་སྲན་དང་ནི། །གོང་བུ་དང་ནི་གོར་མའོ། །འཕེལ་བ་རྒན་པོ་ཆེན་པོ་དང་། །རྙིང་པ་སྔོན་སྐྱེས་དང་པོ་པ། །རྙིང་མ་དང་ནི་ཡུན་རིང་མ། །

གསར་པ་སོ་མ་ཐོག་མ་དང་། །ཡག་མ་གཞོན་ནུ་གསར་པའོ། །རབ་གཞོན་ཤིན་ཏུ་གཞོན་པ་དང་། །འཇམ་པ་དང་ནི་མཉེན་དང་འབོལ། །རྗེས་འགྲོ་ཕྱི་འབྲང་རྗེས་འཇུག་དང་། །རྟིང་འབྲང་མ་ནིང་མི་ཟད་པ། །མངོན་སུམ་དང་ནི་དབང་པོའི་ཡུལ། །ལྐོག་ཏུ་གྱུར་དང་མངོན་སུམ་མིན། །གཅིག་སྔ་གཅིག་སྔོན་གཅིག་
