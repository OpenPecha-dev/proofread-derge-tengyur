ལྟར་བ་ལ་ཏཀྲ་ཀྱང་ངོ་། །ད་ལས་ཏའི་ན་སྟེ་དའི་ཡང་ངོ་། །ཨཏྲ་ཀྱི་ཀ་འགྲོ་བ་ལ་ཛགྷྲའོ། །ར་ལས་སོ། །ལཱུ་སོགས་དང་ཨོཏྲ་འགྲོ་བ་ལས་སོ། །ཏ་ལ་དཱའི་ད་དྲའོ། །དབྱངས་ལས་ཏ་ཡང་ནའོ། །བས྄་ཀྵུ་དྷྲ་དག་ལ་ཨིཊ྄འོ། །གྷས྄་སོགས་ཀྱི་ཥའོ། །ཏྲཎ་སོགས་ནི་ནི་པཱ་ཏ་ཡང་
ནའོ། །སམ྄་ པ་རི་ ཨུ་པ་རྣམས་ལས་ཀྲྀ་ལ་རྒྱན་གྱི་དོན་ལ་སུཊ྄འོ། །དངོས་པོ་ལ་གྷཉྲའོ། །ཛནྲའི་ཛཱའོ། །པཙ྄་ལས་བའོ། །ཤུ་ཥ་ལ་ཀའོ། །ཀྵཱ་སོགས་ལ་མའོ། །སྤཱ་ཡའི་སྥཱིའོ། །ཀྭ་སུ་ཀཱ་ན་དག་ཎབྲ་ཨེ་བཞིན་ནོ། །བྱ་བ་ལ་ཤྀཏྲ་ཤཱ་ན་དག་སྟེ་ཏིཔྲ་ཏེ་བཞིན་ནོ། །ཨཱ་ན་ལས་

ཨ་ཐུའོ། །ཌུ་འགྲོ་བ་ལས་དེ་འདས་པ་ལ་ཏྲི་མྐྲའོ། །ནཊ྄་ཀི་དག་གོ །པྲཙྪ་བིཙྪ་དག་གི་ན་ལ་ཤའོ། །ཡུཊ྄འོ། །ཨུཏྲ་ལས་སྠ་སོགས་ཀྱིས་དབྱིའོ། །སྒྲུབ་ཐབས་དང་རྟེན་དག་ལ་འང་ངོ་། །ཨཱིཥ྄། དུཿ་སུ་རྣམས་ལ་ཁལ་ཡཱུཊ྄འོ། །ཏ་བྱ་ཨཱ་ནཱི་ཡ་དག་གོ །དབྱངས་ལས་ཡའོ། །པུ་ཤ་
