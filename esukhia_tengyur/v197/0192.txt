སོགས་དང་། ཨཙྪཻཏྶཱིཏྲ། ཨཙྪཻཏྟཱཾ། ཨཙྪེཏྶུཿསོགས་སོ། ༈ དི་བུ་རྩེད་མོ་ལའོ་ཞེས་པ་ལ། ཨིཊྲ་འཇུག་པས་ཨིཊྲ་བྱིན། བྱིངས་ལ་ཡོན་ཏན་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་དེ་བཱིཏྲ། ཨ་དེ་བཱིཥྚཱཾ། ཨ་དེ་བི་ཥུཿ། ཨ་དེ་བཱིཿསོགས་དང་དེ་བཞིན་དུ། ༈ སི་བུ་བསྙེན་པར་བྱེད་པ་ལ། ཨ་ས་
བཱིཏྲ། ཨ་སེ་བིཥྚཱཾ། ཨ་སེ་བི་ཥུཿསོགས་སོ། ༈ དྲྀ་ཤིརྲ་ལྟ་བ་ལ། ཨིརྲ་ལ་ཡང་ནའོས་པར་རྐྱེན་ང་དང་སི་ཡང་ན་བར་བྱེད་པས། ངའི་ཕྱོགས་ལ། ཨ་དརྴཏྲ། ཨ་དརྴ་ཏཱཾ། ཨ་དརྴནྲ་སོགས་དང་། སི་ཉིད་ཀྱི་ཕྱོགས་ལ། ༈ རཱ་རོ་ཛྷ་སཾ་དྲྀ་ཤཱཾ། ཛྷ་ས་ལ་དྲྀ་ཤ་རྣམས་ཀྱི་ཨར་ནི་རའོ། །བྱུང་
ཟིན་གྱི་རྣམ་དབྱེའི་སྐབས་འདིར་རྐྱེན་དང་རྣམ་དབྱེ་ཛྷ་སའི་ཡི་གེ་དང་པོ་ཕ་རོལ་ལ་བྱིངས་དྲྀ་ཤ་ལ་སོགས་པ་རྣམས་ལ་ཡོན་ཏན་བྱས་པའི་ཨར་ནི་ར་ཞེས་པར་འགྱུར་ཏེ། བྱིངས་ཀྱི་རྀ་ཨརྲ་དེ་ར་རུ་བསྒྱུར། དེ་རིང་པོར་བསྒྱུར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་དྲཱཀྵཱིཏྲ། ཨ་དྲཱཥྚཱཾ། ཨ་དྲཱཀྵུཿ ཨ་དྲཱཀྵིཿ ཨ་དྲཱཥྚཾ།
ཨདྲཱཥྚ། ཨ་དྲཱཀྵཾ། ཨ་དྲཱཀྵྭ། ཨ་དྲཱཀྵྨ། དེ་བཞིན་དུ་སྤྲྀ་ཤ་རེག་པ་ལ། ཨ་སྤྲཱཀྵཱིཏྲ། ཨ་སྤྲཱཥྚཱཾ། ཨསྤྲཱཀྵུཿ། ཨ་སྤྲཱཥྚཾ། ཨསྤྲཱཥྚཿ། ཨསྤྲཱཀྵཾ་སོགས་སོ། མྲྀ་ཛུཥྲ་དག་པ་ལ། ཉེར་བསྒྱུར་བར་སྔོན་དུ་བཞག་ཆོ་ག་སྔོན་མ་དང་འདྲ་བས། པ་རཱ་མྲཱ་ཀྵཱིཏྲ། པ་རཱ་མྲཱཥྚཱཾ། པ་རཱ་མྲཱཀྵུཿསོགས་
སོ། ༈ བྷི་དིརྲ་འཇོམས་པ་ལ་ཨིཊ་མི་འཇུག་པ་རསྨཻ་པ་ད་རྣམས་ལ། ཨིཊྲ་མེད་པའི་ནཱ་མི་ལ་འཕེལ་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་བྷཻཏྶཱཏྲ། ཨ་བྷཻ་ཏྟཱཾ། ཨ་བྷཻཏྶུཿསོགས་དང་། ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ༈ སི་སྱོཿ སི་སཱི་དག་ལའོ། །ཨིཊྲ་མེད་པའི་རྐྱེན་སི་དང་སཱིཥྚ་སོགས་
ཕ་རོལ་ལ་བྱིངས་ཀྱི་ཨུ་པ་དྷཱ་ལ་ཡོན་ཏན་འབྱུང་བ་མ་ཡིན་ཏེ། བྱིངས་ལ་ཡོན་ཏན་མི་བྱེད་རྣམ་དབྱེ་ཛྷ་ས་དང་པོ་ཅན་རྣམས་ལ་སི་དབྱི་བ་སོགས་ཀྱིས། ཨབྷིཏྟ། ཨ་བྷིཏྶཱ་ཏཱཾ། ཨ་བྷིཏྶ་ཏ། ཨ་བྷིཏྠཱཿ ཨ་བྷིཏྶཱ་ཐཱཾ། ཨ་བྷི་དྷྭཱཾ། ཨ་བྷི་ཏྶི། ཨ་བྷིཏྶྭ་ཧི། ཨ་བྷིཏྶྨ་ཧི་སོགས་སོ། ༈ བྲྀ་ཏུ་འཇུག་
པ་ལ་ཨཱཏྨ་ནེ་པ་ད་རྣམས་སྦྱོར་བ་ལས། ཨིཊྲ་སྟེར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཨ་བརྟིཥྚཾ། ཨ་བརྟི་ཥཱ་ཏཱཾ། ཨ་བརྟི་ཥ་ཏ། ཨ་བིརྟིཥྛཱཿ ཨ་བརྟིཥཱ་ཐཱཾ། ཨ་བརྟི་དྷྭཱཾ། ཨ་བརྟི་ཥི། ཨ་བརྟི་ཥྭ་ཧི། ཨ་བརྟིཥྨ་ཧི། ༈ ག་མླྀ་དང་སྲྀ་པླྀ་འགྲོ་བ་ལའང་པ་རསྨཻ་པ་ད་རྣམས་སྦྱོར་བ་
