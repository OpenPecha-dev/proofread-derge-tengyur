ཚིག་དང་མང་ཚིག་ལ། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ ལྔ་པ་གསུམ་ལ། ཤྲི་ཡཱཿ ང་སི། ཤྲི་ཡཿ ང་སི། ཤྲཱི་བྷྱཱཾ། ཤྲཱི་བྷྱཿ དྲུག་པའི་གཅིག་ཚིག་དང་གཉིས་ཚིག་ལ། ཤྲི་ཡཱཿ ཤྲི་ཡཿ ངསྲ། ཤྲི་ཡོཿ ཨོསྲ། མང་ཚིག་ལ་ཨཱམྲ་ལ་ནུཊྲ་ཡང་ན་བར་བཤད་པས། ཤྲི་ཡཱཾ། ཨཱམྲ། ཤྲཱིཎཱཾ། ཨཱམྲ། བདུན་པ་གསུམ་ལ། ཤྲི་ཡཱཾ། ཤྲི་ཡཱི། ངི།
ཤྲི་ཡོཿ ཨོསྲ། ཤྲཱི་ཥུ། སུཔྲ། ༈ དེ་བཞིན་དུ། དྷཱི། ཧྲཱི་སོགས་ཏེ་ཨཱིཔྲ་མཐའ་མ་ཡིན་པ་ཐམས་ཅད་དོ། །བྷཱུའི་སྒྲ་དང་བྷྲཱའི་སྒྲ་སོགས་ཀྱང་འདི་དང་འདྲ་ལ། བ་དྷཱུའི་སྒྲ་དང་ཛམྦཱུའི་སྒྲ་སོགས་མང་པོ་ཞིག་ནི་ན་དཱིའི་སྒྲ་དང་འདྲ་བ་སྟེ། ༈ བ་དྷཱུཿ བ་དྷྲཽ། ཨཽ། བདྷཱཿ ཛས྄ སོགས་སོ། །རྀ་ཡིག་མཐའ་ཅན་བུད་མེད་ཀྱི་
རྟགས་ལ་མཱ་ཏྲྀ་ཞེས་པ་ལ་མཚོན་ན། བློ། ངོ་ཚ། དེ་ཡང་ཐམས་ཅད་སྐྱེས་པའི་རྟགས་པི་ཏྲིའི་སྐབས་སུ་བཤད་པ་དང་འདྲ་སྟེ། ༈ མཱ་ཏཱ། སི། མཱ་ཏ་རཽ། །ཨཽ། མཱ་ཏ་རཿ ཛསྲ། ཤིང་ཛམྦཱུ་ཤིང་། ཧེ་མཱ་ཏཿ སི། མཱ་ཏ་རཾ། ཨམྲ། མཱ་ཏ་རཽ། ཨཽ། མཱ་ཏྲཱིཿ བུ་མོ། སོགས་ཏེ་ཤས྄་ལ་རིང་པོ་ཉིད་ལས་ས྄་ནྲར་མི་འགྱུར་བ་གཞན་རྣམས་འགྲོའོ། །
