བིདྨ། ཆོག་འབྱུང་ལིང་ལ། བི་དྱཱཏྲ། བིདྱཱཏཱཾ། བི་དྱུཿསོགས་དང་། ཤིས་བྱེད་ལོཊྲ་ལ། བེཏྟུ། བིཏྟུ་ཏ། བཏྟཏྲ། བིཏྟཱཾ བི་དནྟུ་སོགས་དང་། དེ་རིང་མ་ཡིན་པ་ལང་ལ། ཨ་བེ་ཏྲ། ཨ་བིཏྟཱཾ། ཨ་བི་དུཿ་སོགས་སོ། ༈ །ཡང་བིད་ལཨཱ་ཏྨ་ནེ་པ་ད་རྣམས་ཀྱང་འཇུག་
ཅིང་དེའི་ཚེ་ཨ་པྲ་བི་ཀཱ་ར་ཎ་ཡང་མི་འཇིག་ལ་བྱིངས་མཐར་རུཊ་ཨཱ་ག་མ་སྟེར་བ་དང་མི་སྟེར་བ་འདམ་ཀ་ཅན་དུ་བཤད་པས། བི་དྲ་ཏེ། བི་དྲཱ་ཏེ། བི་དྲཱ་ཏེ། བི་དྲ་ནྟེ་སོགས་སམ། བི་ད་ཏེ། བི་དཱ་ཏེ། བི་ད་ནྟེ་སོགས་སོ། ༈ །ཨ་སྲ་ཡོད་པ་ལ། ན་མྲ་ལའོས། ང་འགྲོ་བ་ཕ་
རོལ་ལ་ཨ་སྲ་ཀྱི་ཨ་དབྱི་བ་སོགས་ཀྱི་བསྒྲུབས་པས། བརྟ་ནའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སྟི་སྟཿ ས་ནྟི་སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ༈ སི་སཿ སི་སའོ། །རྣམ་དབྱེ་སི་པྲ་ཕ་རོལ་ལ་ཨ་སྲ་ཀྱི་སྲ་དབྱི་བར་བྱ་སྟེ། ས་ཕྱིས་པས། ཨ་སི་ལྷག་མ་
རྣམས་ལ། སྠཿ སྠ། ཨ་སྨི། སྭཿ སྨཿ ཆོག་འབྱུང་ལིང་ལ། སྱཱ་ཏཱཾ། སྱུཿསོགས་སོ། །ཤིས་བྱེད་ལོཊྲའི་སྐྱེས་བུ་དང་པོ་གསུམ་ལ། ཨ་སྟུ། སྟཱ་ཏྲ། ས་ནྟུ། སྐྱེས་བུ་དབུས་མའི་གཅིག་ཚིག་ལ། ༈ ཛ་ཧྱེ་དྷི་ཤཱ་དྷི། ཛ་ཧི་ཨི་དྷི་ཤཱ་དྷིའོ། །བྱིངས་
ཧནྲ། ཨ་སྲ། ཤསྲ་རྣམས་ནི་རྣམ་དབྱེ་ཧི་དང་ལྷན་ཅིག་པ་རིམ་པ་བཞིན་ཛ་ཧི་ཨེ་དྷི་ཤཱ་དྷི་རྣམས་སུ་ཚིག་ཕྲད་ཀྱིས་བསྒྱུར་སྟེ། ཨ་སྲ་ལ་ཧི། ཨཔྲ་འཇིག །ཨེ་དྷི། ལྷག་མ་རྣམས་ལ། སྟཾ། སྟ། ཨ་སཱ་ནི། ཨ་སཱབ། ཨ་སཱ་མ། དེ་རིང་མ་ཡིན་པ་ལང་གི་སྐྱེས་བུ་
དང་པོ་གཅིག་ཚིག་ལ། ༈ ཨ་སྟེ་རཱིཊྲ། ཨ་སྟི་ལས་ཨཱི་ཊྲའོ། །བྱིངས་ཨསྲ་ལས་ཕ་རོལ་གྱི་རྣམ་དབྱེ་དི་པྲ་དང་སི་པྲ་ལ་ཨཱིཊ་ཨཱ་ག་མ་སྟེར་བར་བྱ་སྟེ། ཨཱི་ཊྲ་བྱིན་པས། ཨཱ་སཱིཏྲ། ལྷག་རྣམས་ལ་སི་པྲ་ལ་ཨཱིཏྲ། ཨ་མི་པྲ་ལ་ཨ་མི་དབྱི། ང་འགྲོ་བ་རྣམས་ལ་
ཨ་དབྱི་བར་བྱེད་ཀྱང་དི་པྲ་སོགས་ཀྱི་ཨཊྲ་དང་། དབྱངས་དང་པོའི་ཨཊྲ་མཚམས་སྦྱར་བས་གཟུགས་འདྲ་སྟེ། ཨཱ་སྟཾ། ཨཱ་སནྲ། ཨཱ་སྟཾ། ཨཱསྟ། ཨཱསཾ། ཨཱསྨ། ༈ ལི་ཧྲ་རོ་ཀུན་ཏུ་མྱང་བ་ལ། བརྟ་མ་ནའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ།
