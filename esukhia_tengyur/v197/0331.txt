བ་ལ་བྷ་དྲི་ཀཱ །བིཥྭཀྶེ་ནཱ་པྲི་ཡཱ་གྲྀཥྚི་བཱ་རཱ་ཧཱི་བ་ད་རེ་ཏྱ་པི། །མཱརྐ་བོ་བྷྲྀདྒ་རཱ་ཛཿསྱཱཏྐཱ་ཀ་མཱ་ཙི་ཏུ་བ་ཡ་སཱི། །ཤ་ཏ་པུཥྤཱ་ཤི་ཏ་ཙྪཏྟྲཱཉབྷི་ཙྪཏྟྲཱ་མ་དྷུ་རཱ་མི་སི། །ཨ་བཀྤུཥྤཱི་ཀཱ་ར་བཱི་ཙ་ས་ར་ཎཱི་ཏུ་པྲ་ཤཱ་ར་ཎཱི། །ཏ་སྱཱ་
༄༅། །བྱེད། །སྟོབས་བཟང་སྣ་ཚོགས་སྡེ་དགའ་བྱེད། །གྲྀཥྚི་ཕག་མོ་རྒྱ་ཤུག་ཀྱང་། །མཱརྐ་བ་དང་བུང་བའི་རྒྱལ། །ཀཱ་ཀ་མཱ་ཙི་པོ་གན་ནོ། །མེ་ཏོག་བརྒྱ་བ་གདུགས་དཀར་དང་། །གདུགས་བཟང་མངར་མོ་མི་སིའོ། །ཟི་ར་ནག་པོ་ཀཱ་ར་བཱི། །ས་རི་ཎཱ་དང་རབ་ཏུ་བརྐྱང་། །དེ་ཡི་ཀ་ཊུཾ་
ཀ་ཊཾ་བྷ་རཱ་རཱ་ཛ་བ་ལཱ་བྷ་དྲ་བ་ལེ་ཏི་ཙ། །ཛ་ནཱི་ཛཱ་ཏུ་ཀཱ་ར་ཛ་ནཱི་ཛ་ཏུ་ཀྲྀཙྩཀྲ་བརྟྟི་ནཱི། །སཾ་སྤཪྴཱ་ཐས་ཊཱི་གནྡྷ་མཱུ་ལི་ཥདྒྲནྠི་ཀེ་ཏྱ་པི། །ཀ་ཙཱུ་རོ་པི་པ་ལཱ་ཤོ་པི་ཀཱ་ར་བེལླཿཀ་ཊིལླ་ཀཿ། །སུ་ཁ་བཱི་ཙཱ་ཐ་ཀཱུ་ལ་ཀཾ་པ་ཊོ་ལ་སྟིཀྟ་ཀཿཔ་ཊུཿ། །ཀུཥྨཱཎྜ་ཀསྟུ་ཀརྐྐཱ་
བྷ་ར་དང་། །རྒྱལ་པོའི་སྟོབས་དང་སྟོབས་བཟང་དང་། །ཛ་ནཱི་ཛཱ་ཏུ་ར་ཛ་ནཱི། །ཛ་ཏུ་བྱེད་དང་འཁོར་ལོས་སྒྱུར། །རེག་བཟང་དེ་ནས་ས་ཊཱི་དང་། །དྲི་ཡི་རྩ་བ་ཚིགས་དྲུག་པ། །ཀ་ཙུ་ར་པི་པ་ལ་ཤ། །ཀཱ་ར་བེལླ་ཀ་ཊིལླ་ཀ །བདེ་ལྡན་དེ་ནས་རིགས་ལྡན་དང་། །པ་ཊོ་ལ་དང་ཏིགྟ་གསལ། །ཀུཥྨཱཎྜ་ཀ་ཕ་ཏ་སི། །
