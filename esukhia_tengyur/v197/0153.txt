ད་དྷཱ་ཏཱཾ། ཨ་ད་དྷ་ཏ། ཨ་དྷདྠཱཿ ཨ་ད་དྷཱ་ཐཱཾ། ཨ་དྷདྡྷྭཾ། ཨ་ད་དྷི། ཨ་ད་དྷྭ་ཧི། ཨ་དདྷྨ་ཧི། སོགས་ཏེ། དེ་རྣམས་ཀྱི་རིགས་འགྲེ་དེད་དེ་ཤེས་པར་བྱའོ། ༈ །ད་ནི་བྱིངས་དིབྲ་ལ་སོགས་པ་ཨཔྲ་ཀྱི་ཡུལ་ལ་ཡ་བི་ཀཱ་ར་ཎ་འཇུག་པ་རྣམས་ཀྱི་བྱ་བ་རྣམས་
བརྗོད་པར་བྱ་སྟེ། དེ་ཡང་དེ་བུ་སྟེ་བ་དང་། མངོན་པར་འདོད་པ་དང་། ཐ་སྙད་དང་། སྡུག་གུའམ་འདུ་འགོད་དང་། གསལ་བ་དང་། བསྟོད་པ་དང་། རྨི་ལམ་དང་། མཛེས་པ་དང་། དགའ་བ་དང་། མྱོས་པ་དང་། འགྲོ་བ་རྣམས་ལའོ། །ཞེས་པའི་བྱིངས་ལ་བརྟ་མ་ན་སོགས་ཀྱི་
པ་རསྨཻ་པ་ད་རྣམས་སྦྱོར་ཏེ། བརྟ་མ་ནའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ལ། དིཔྲ། ཏིཔྲ་ཅེས་གནས་པ་ལ། ༈ དི་བཱ་དེརྻཿ དིཔྲ་ལ་སོགས་པ་ལ་ཡའོ། །བྱིངས་དིཔྲ་ལ་སོགས་པའི་སྡེ་ཚན་རྣམས་ལ་བྱེད་པ་པོའི་དོན་གྱི་ཨཔྲ་བི་ཀཱ་ར་ཎ་འི་ཡུལ་ལ་ཡ་བི་ཀཱ་ར་ཎ་འཇུག་སྟེ་ཡ་བི་ཀཱ་ར་ཎ་སྦྱར།
ཨི་ཨུ་སོགས་ཀྱི་མདོས་རིང་པོར་བསྒྱུར་བས། དཱི་བྱ་ཏཿ དཱི་བྱནྟི། དེ་བཞིན་དུ་བསྒྲུབས་པས་གཉིས་ཚིག་དང་མང་ཚིག་ལ། དཱི་བྱ་ཏཿ དཱི་བྱནྟི། སྐྱེས་བུ་དབུས་མ་དང་ཕྱི་མ་ལ། དཱི་བྱ་སི། དཱི་བྱ་ཐཿ དཱི་བྱ་ཐ། དཱི་བྱཱ་མི། དཱི་བྱཱ་བཿ དཱི་བྱཱ་མཿ ༈ དེ་བཞིན་དུ་བསྒྲུབས་པས་ཆོག་
འབྱུང་ལིངྲ་ལ། དཱི་བྱེཏྲ། དཱི་བྱེ་ཏཱཾ། དཱི་བྱེ་ཡུཿ དཱི་བྱེཿ དཱི་བྱེ་ཏཱཾ། དཱི་བྱེ་ཏ། དཱི་བྱེ་ཡཾ། དཱི་བྱེ་བ། དཱི་བྱེ་མ། ༈ ཤིས་བྱེད་ལོཊྲ་ལ། དཱི་བྱ་ཏུ། དཱི་བྱ་ཏཱཏྲ། དཱི་བྱ་ཏཱཾ། དཱི་བྱནྟུ། དི་བྱ། དི་བྱ་ཏཱཏྲ། དཱི་བྱ་ཏཱཾ། དཱི་བྱ་ཏ། དཱི་བྱཱ་ནི། དཱི་བྱཱ་བ། དཱི་བྱཱ་མ། ༈ དེ་རིང་མ་ཡིན་པ་ལང་ལ།
ཨ་དཱི་བྱཏྲ། ཨ་དཱི་བྱ་ཏཱཾ། ཨ་དཱི་བྱནྲ། ཨ་དཱི་བྱཿ ཨ་དཱི་བྱ་ཏཾ། ཨ་དཱི་བྱ་ཏ། ཨ་དཱི་བྱཱཾ། ཨ་དཱི་བྱཱ་བ། ཨ་དཱི་བྱཱ་མ་སོགས་སོ། ༈ དེ་ནས་ཥུཉྲ་ལ་སོགས་པ་ནུ་བི་ཀཱ་ར་ཎ་འཇུག་པ་རྣམས་ཀྱི་བྱ་བའི་དཔེ་ཙམ་བརྗོད་པར་བྱ་སྟེ། ཥུཉྲ་གཅོད་ཅིང་བརྡུང་བ་ལའོ་ཞེས་པའི་བྱིངས་ལ་བརྟ་མ་ན་
སོགས་ཀྱི་བདག་དོན་དང་གཞན་དོན་གཉིས་ཀ་སྦྱོར་ཏེ། དང་པོ་པ་རསྨཻ་པ་ད་རྣམས་ལས་བརྟ་མ་ན་ལཊྲའི་སྐྱེས་བུ་དང་པོའི་གཅིག་ཚིག་ཏིཔྲ་སྦྱར་བྱིངས་མཐའི་ཉྲ་གཉིས་ཀའི་ཚིག་ཅན་གྱི་དོན། ཥུ། ཏིབྲ་ཅེས་གནས་པ་ལ། ༈ ཨཱདྱཿཥྞཿསྣཿ དང་པོའི་ཥ་ཎ་ནི་ས་ནའོ། །བྱིངས་ཀྱི་
