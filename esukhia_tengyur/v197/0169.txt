ཡཱཾ་བ་བྷཱུ་བ་སོགས་དང་ཨཱཏྨ་ནེ་པ་ད་རྣམས་ལ། ཀ་ལ་ཡཱཾ་ཙ་ཀྲེ། ཀ་ལ་ཡཱ་མཱ་སེ། ཀ་ལ་ཡཱཾ་བ་བྷཱུ་བེེ་སོགས་སོ། ༈ བིད་ཤེས་པ་ལའོ་ཞེས་པ་ལ། ཨཱ་མི་བི་དེརྣ་གུ་ཎཿ ཨཱམྲ་ལ་བིད་ཀྱི་ཡོན་ཏན་མ་ཡིན་ནོ། །ཨཱམྲ་ཕ་རོལ་ལ་བྱིངས་བིད་ལ་ཡོན་ཏན་འབྱུང་བ་མ་ཡིན་ཏེ།
ཡོན་ཏན་མི་བྱེད། བི་དཱཾ་ཙ་ཀཱ་ར། བི་དཱ་མཱ་སེ། བི་དཱཾ་བ་བྷཱུ་བ་སོགས་སོ། ༈ ཨཱིཀྵ་ལྟ་བ་དང་མཚན་མ་ལའོ་ཞེས་པ་ལ། ཨཱིཀྵཱཾ་ཙ་ཀྲེ། ཨཱིཀྵུ་མཱ་སེ། ཨཱིཀྵཱཾ་བ་བྷཱུ་བ། ཨཱུཧྲ་རྣམ་པར་བརྟགས་པ་ལ། ཨཱུ་ཧཱཾ་ཙ་ཀྲེ། ཨཱུ་ཧཱ་མཱ་སེ། ཨཱུ་ཧཱཾ་བ་བྷཱུ་བེ་སོགས་སོ། ༈ ཨིཧ་རྩེད་མོ་ལའོ་ཞེས་པ་
ལ། ཨི་ཧཱཾ་ཙ་ཀྲེ། ཨི་ཧཱ་མཱ་སེ། ཨི་ཧཱཾ་བ་བྷཱུ་བེེ་སོགས་སོ། །དེ་ནས་རྐྱེན་ས་དང་བཅས་པའི་ཨཱམྲ་གྱི་བྱ་བ་རྣམས་བཤད་པར་བྱ་སྟེ། ༈ ཨིཙྪ་ཡཱ་མཱཏྨ་ནཿསཿ བདག་གི་འདོད་པ་ལོ་སའོ། བྱིངས་ལ་བདག་ཉིད་ཀྱི་འདོད་པའི་དོན་ལ་ས་རྐྱེན་དུ་འགྱུར་ཏེ། ཌུ་ཀྲྀཉ་བྱེད་པ་ལ་སྦྱར་བ་ལ།
ཀྲྀ་ ས། ༈ རྀ་ཏྲ་ཨིརྲ། རྀཏྲ་ཨིརྲའོ། །བྱིངས་མཐའི་རྀ་ཡིག་ནི་ཀྲ་ཡིག་དང་ངྲ་ཡིག་འགྲོ་བའི་མིང་ཅན་ཕ་རོལ་ལ་ཨིརྲ་དུ་འགྱུར་ཏེ། རྀ་ཨིརྲ་དུ་བསྒྱུར་བས། ཀིརྲ་ས། ཞེས་པ་ལ། ༈ སཱཏྲ་དྭིཤྩ། ས་ལས་ཀྱང་གཉིས་སོ། །བདག་གི་འདོད་པའི་དོན་གྱི་ས་ཕ་རོལ་ལས་
ཀྱང་བྱིངས་གཉིས་སུ་འགྱུར་ཏེ། གཉིས་བརྗོད། ཀིརྲ་ཀིརྲ། ས་ཞེས་གནས་པ་ལ། ༈ པུརྦ་ཧ་སཱ་དིཿཤེ་ཥཿ སྔོན་མའི་ཧ་ས་དང་པོ་ལྷག་པོའོ། །བྱིངས་གཉིས་བརྗོད་པའི་སྔོན་མའི་ཧ་ས་དང་པོ་དེ་ལྷག་པོ་སྟེ་གནས་ལ་ཧ་ས་གཞན་རྣམས་དབྱི་བར་བྱ་སྟེ། རྲ་གོང་མ་ཕྱིས། ཀུ་
དང་ཧའི་ཙུ་འོས་ཀ་ཙ། ཨིརྲ་སོགས་ཀྱིས་རིང་པོ། །ཀུ་ཨི་ལ་ཞེས་པས་ས་ཥར་བསྒྱུར་བས། ཙི་ཀཱིཪྵ་ཞེས་གནས་པ་ལ། ཀཱསྲ་ཞེས་པས་ཨཱམྲ་སྦྱོར་བ་སོགས་ཀྱིས་བསྒྲུབས་པས། ཙི་ཀཱིཪྵཱཾ་ཙ་ཀཱ་ར། ཙི་ཀཱིརྵཱ་མཱ་ས། ཙི་ཀཱི་རྵཱཾ་བ་བྷཱུ་བ་སོགས་སོ། །ཆོ་ག་དེ་བཞིན་དུ་བྱས་པས།
བྷུཛྲ་སྐྱོང་བ་ལ་སྟེ་ཟས་གཡོ་བ་ལའང་ངོ་ཞེས་པ་ལ། བུ་བྷུཀྵཱཾ་ཙ་ཀཱ་ར། བུ་བྷུཀྵཱ་མཱ་ས། བུ་བྷུཀྵཱཾ་བ་བྷཱུ་བ་སོགས་སོ། ༈ ནཱམྣོ་ཡ་ཨཱི་ཙཱ་སྱ། མིང་ལ་ཡ་སྟེ་ཨའི་ཡང་ཨཱིའོ། །བདག་གི་འདོད་པས་བྱེད་དོ་སྔོན་བྱས་སོ། །མིང་ལས་བདག་གི་འདོད་པའི་དོན་ལ་ཡ་རྐྱེན་དུ་འགྱུར་ཞིང་ཡ་ཕ་རོལ་ལ་མིང་
