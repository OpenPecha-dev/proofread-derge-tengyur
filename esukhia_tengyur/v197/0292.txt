སྭ་བྷཱ་བཤྩ་ནི་སརྒཤྩཱ་ཐ་པེ་པ་ཐུ། །ཀམྦོ྅ཐ་ཀྵ་ཎ་ཨུདྡྷཱཪྵོ་མ་ཧཱ་ཨུངྦ་ཧ་ཨུཏྶ་བཿ། །སྭརྒ་བརྒཿ།། ༈ །།ཨ་དྷོ་བྷུ་བ་ན་པཱ་ཏཱ་ལ་བ་ལི་སདྨ་ར་སཱ་ཏ་ལཾ། །ནཱ་ག་ལོ་ཀོ྅ཐ་ཀུ་ཧ་རཾ་བི་བ་རཾ་ཤུ་ཥི་རཾ་བི་ལཾ། །ཙྪི་དྲཾ་ནིརྦྱ་
ཉིད་དང་དངོས། །དེ་ནས་གཡོ་དང་འདར་བའོ། །དེ་ནས་སྐད་ཅིག་ཉེར་དགའ་དང་། །མཆོད་ཆེན་དགའ་སྟོན་ལྟད་མོའོ། །མཐོ་རིས་ཀྱི་སྡེའོ།། ༈ །།འོག་ས་རྐང་འོག་སྟོབས་ལྡན་གནས། །བཅུད་འོག་ཀླུ་ཡི་འཇིག་རྟེན་ནོ། །དེ་ནས་བུ་ག་གློ་བུག་དང་། །སྐར་ཁུང་དང་ནི་ཁུང་བུ་དང་། །བུ་གུ་ཕུ་གུ་པད་
ཐ་ནཾ་རོ་ཀཾ་རནྡྷྲཾ་ཤྭ་བྷྲམྦ་པཱ་ཤུ་ཥིཿ། །གརྟྟོ་བ་ཌྷཽ་བྷུ་བི་ཤྭ་བྷྲེ་ས་རནྡྷྲེ་ཤུ་ཥི་རཾ་ཏྲྀ་ཥུ། །ཨནྡྷྲ་ཀཱ་རོཉསྟྲི་ཡཱཾ་དྷྭཱནྟནྟ་མི་ཤྲནྟི་མི་རནྟ་མཿ། །དྷྭཱནྟེ་གཱ་ཌྷེཉནྡྷ་ཏ་མ་སཾ་ཀྵཱི་ཎེཉབ་ཏ་མ་སནྟ་མཿ། །བི་ཤྭཱ་ཀྶནྟ་མ་སནྣཱ་གཱཿཀཱ་དྲ་
མིག་དང་། །བུ་ག་དང་ནི་གསལ་བའོ། །མཐོང་བ་རླུང་འབྱུང་མཁའ་ངོས་དང་། །སའི་མཐོང་བུག་སྒོ་སའི་སྒོ་གསུམ། །མདོངས་བྱེད་མོ་མིན་མུན་པ་དང་། །སྣང་མེད་རབ་རིབ་སྨག་མེད་དང་། །མུན་ནག་མུན་འཐིབས་ཤིན་ཏུ་སྡུག །མུན་མདངས་ཤེ་མུན་ཟ་ཟི་དང་། །ཀུན་ཁྱབ་དང་ནི་མུན་ཁྱབ་པོ། །རི་གནས་
བེ་ཡཱསྟ་དཱི་ཤྭ་རཿ། །ཤེ་ཤོ྅ནནྟོ་བཱ་སུ་ཀིསྟུ་སརྦ་རཱ་ཛོ྅ཐ་གོ་ན་སེ། །ཏི་ལིཏྶཿསྱཱ་ད་ཛ་ག་རེ་ཤ་ཡུརྦཱ་ཧ་ས་ཨི་ཏྱུ་བྷཽ། །ཨ་ལ་གརྡྷོ་ཛ་ལ་བྱཱ་ལཿས་མཽ་རཱ་ཛི་ལ་ཌུཎྜུ་བྷཽ། །མ་ལུ་དྷཱ་ནོ་མཱ་ཏུ་ལཱ་ཧིརྣྣི་མུཀྟོ་རྨུཀྟ་ཀཉྩུ་ཀཿ། །
ཀ་དྲུའི་བུ་དང་ནི། །དེ་ཡི་དབང་ཕྱུག་ལྷག་མ་དང་། །མཐའ་ཡས་ནོར་རྒྱལ་སྤྲུལ་རྒྱལ་དང་། །དེ་ནས་གླང་སྣ་ཐིག་ལེའི་སྣ། །དོང་གནས་ཉལ་བྱེད་རླུང་ཟས་གཉིས། །མྱུར་འགྲོ་དང་ནི་ཆུ་སྦྲུལ་ལོ། །མཚུངས་པའི་གྲལ་རིས་དུག་མེད་སྤྲུལ། །ཁྲི་ཅན་དང་ནི་རྐང་བཞི་དང་། །ཤུན་པ་བརྗེས་དང་གོ་ཆ་གྲོལ། །
