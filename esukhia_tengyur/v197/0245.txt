ཧ་ར་ཎཾ། ཧ་བ་ནཾ། བྷཱུ་ཥ་ཎཾ། བ་ཧ་ནཾ། ཨུ་ཏྲ་སྔོན་མའི་སྠཱ་ལ་ཡུཊ། ༈ ཨུ་དསྠཱ་དེས་ལོ་པཿ ཨུ་ཏྲ་ལས་སྠཱ་སོགས་ཀྱིས་དབྱིའོ། །ཉེར་བསྒྱུར་ཨུཏྲ་སྔོན་མ་ལས་བྱིངས་སྠཱ་ལ་སོགས་པའི་ས་ཡིག་ནི་རྐྱེན་ཡུཊྲ་ཕ་རོལ་ལ་དབྱི་བར་བྱ་སྟེ། ཨུཏྠཱ་ནི། དེ་
བཞིན་དུ། ཨུ་ཏྲ། འཕྲོག་པ། སྦྱིན་སྲེག །རྒྱན། ཐོབ་པ། སྔོན་མའི་སྟཾ་བྷ་ལ། ཨུཏྟ་མྦྷ་ནཾ་སོགས་སོ། ༈ །སཱ་ངྷ་ནཱཾ་ངྷཱ་ར་ཡོཤྩ། སྒྲུབ་ཐབས་དང་རྟེན་དག་ལའང་ངོ་། །བྱིངས་ལས་སྒྲུབ་ཐབས་དང་རྟེན་གྱི་དོན་ལའང་རྐྱེན་ཡུཊྲ་འཇུག་སྟེ། པ་ཙ་ལ་ཡུཊྲ། མེས་འཚེད། པ་ཙ་ནོ་ཉགྣིཿ སྟེང་གི་གནས་

གཞོང་བར་བཅོས། སུ་ཀ་རཿ ཡཱུཊྲ ཨཱི་ཥ་དྱོ་ངྷ་ནཿ ཡཱུཊ དུརྻ་ངྷོནཿ སུ་ཡོ་ངྷ་ནཿ ཡཱུ་ སོགས་སོ། ༈ །ཏ་བྱཱ་ནཱི་ཡཽ། ཏ་བྱ་ཨཱ་ནཱི་ཡ་དག་གོ། །བྱིངས་བྷཱུ་ལ་སོགས་པ་རྣམས་ལ་མ་འོངས་པའི་དུས་བརྗོད་པའི་དོན་ལ་རྐྱེན་ཏ་ ཅུང་ཟད་ཡོད། སྡུག་བསྔལ་ཡོད། བདེ་བ་ཡོད། ཅུང་ཟད་བྱེད། སྡུག་བསྔལ་བྱེད། བྱ་དང་
ཨཱ་ནཱི་ཡ་དག་འགྱུར་ཏེ། བདེ་བ་བྱེད། ཅུང་ཟད་འཐབས། འཐབ་དཀའ་བ། བདེ་བར་འཐབ། བྷ་བི་ཏ་བྱཾ། བྷ་བཱ་ནཱི་ཡཿ ཨཱ་སི་ཏ་བྱཾ། ཨཱ་སཱ་ནཱི་ཡཾ། ཀརྟ་བྱཾ། ཀ་རཱ་ཎཱི་ཡ། གྲ་ཧི་ཏ་བྱཾ། གྲ་ཧཱ་ནཱི་ཡཾ་སོགས་སོ། ༈ །སྭ་རཱ་དྱཿ དབྱངས་ལས་ཡའོ། །བྱིངས་དབྱངས་མཐའ་ཅན་ལ་རྐྱེན་ཡ་འགྱུར་ཏེ། ཀྵེ་ཡཾ། ཛེ་ཡཾ།

ཐོབ་བྱ། མཐའ་ཅན་དང་ཤ་ཀ་སོགས་ལའང་རྐྱེན་ཡ་འགྱུར་ཏེ། ལ་བྷྱཾ། ཐོབ་པ། ཡ་བྷྱཾ། འཁྲིག་པ། ཛ་བྱཾ། བཟླས་པ། ཤ་ཀྱ། ནུས་པ། ས་ཧྱཾ། དགའ་བ། དང་པོའང་འདོད་པ་དང་འཛས་པ། རུ་ཅྱ་སོགས་སོ། ༈ །ཨཱི་ཙཱ་ཏཿ ཨཱ་ཏྲ་ལས་ཨཱིའང་ངོ་། །བྱིངས་ཨཱ་ཏྲ་མཐའ་ཅན་ལའང་རྐྱེན་ཡ་འགྱུར་ཞིང་ཡ་ཕ་རོལ་ལ་ཨཱ་ཏ་དེ་ཨི་
