མཱ་ཎི་ཀྱ་བྷཱ་ཥྱ་སིནྡཱུ་ར་ཙཱི་ར་ཙཱི་བ་ར་པཉྫ་རཾ། །ལོ་ཀཱ་ཡ་ཏཾ་ཧ་རི་ཏཱ་ལཾ་བི་ད་ལ་སྠལ་ཐིལ་མིང་བཱཧླ་བཾ།། །།པུནྣ་པུནྶ་ཀ་ཡོཿཤེ་ཥོ་྅་ཨརྡྷཱ། ྅ རྡྭ་རྩྩ། རྡྷཱ་རིག་བྱེ་དྲ་གྲངས། རྩ་པི་ཎྱ་ཀ་ཀཎྜ་ཀཱཿ། །མོད་ལ་དུ་ཀོ་དཎྜ་ཀ་ཚོང་པའི་སྟོང་པ་རི་ཀ། སྟདྐ་སྡེབ་སྦྱོར་རམ་རས་རིང་པོ། ཤཱ་ཊ་ཀཿཀརྦ་ཊོ྅་རྡོ་འཇོག། རྦུ་
སྙན་ངག་མཁན་བྱས་ནོར་བུ་དང་ནི་སྐད། །ལི་ཁྲི་གོས་དང་ཆོས་གོས་དང་ནི་གུར། །འཇིག་རྟེན་རྒྱང་པན་དང་ནི་བ་བླ་དང་། །བདུག་རྫས་སྟེར་མ་དང་ནི་གུར་གུམ་མོ།། །།སྐྱེས་བུ་མ་ནིང་དག་ལའོ་ལྷག་མ་ཡི། །ཕྱད་མཆོད་འབའ་ཆ་དང་ནི་ཚེར་མ་དང་། །ཟས་ཀྱི་བྱེ་བྲག་རྒྱུན་ཆགས་མཐུན་བྲལ་དང་། །གོས་དང་བརྒྱ་གྲོང་དེ་བཞིན་ཤར་
དཿ། །པ་ཏ་ཀོ་དྻོ་ག་ཙ་ར་ཀ་ཏ་མཱ་ལཱ་མཱ་ལ་ན་ཌཿ། མཛེ་ནང་ལའང་། །ཀུཥྛ་མུཎྜ་ཤཱི་དྷུ་བསྟ་འབྲུ་སོགས་ཕྱེ་མར་བྱས། ཀྵྭེ་ཌི་ཏཾ་ཀྵེ་མ་ཀུཊྚི་འོག་གི་ཞལ་བ། མཾ། །སཾ་ག་མ་ཤ་ཏ་མཱ་ཡོན་ཏན་ནཱརྨྨ་སམྦ་ལཱ་བྱ་ཡ་ཏཱཎྜ་བཾ། །ཀ་བི་ཡཾ་ཀརྨྨ་ཀརྤྱཱ་ས་པཱ་ར་བཱ་རཾ་ཡུ་གནྡྷ་རཾ། །
དང་ནི། །ལྟུང་བ་དང་ནི་ཉེར་སྦྱོར་སྨན་དཔྱད་དང་། །ཏ་མཱ་ལཱ་དང་གྲོང་མཐའ་ལྡུམ་ར་དང་། །རུ་རྟ་མགོ་བོའི་ཆང་དང་བས་ཏ་དང་། །སེང་སྒྲ་དང་ནི་བདེ་དང་བུམ་པ་དང་། །ཡང་དག་འགྲོགས་དང་བརྒྱ་ཚད་མིག་ནད་དང་། །རྒྱགས་དང་མི་ཟད་པ་དང་གར་སྟབས་སོ། །ཁ་ཡི་སྒྱེ་དང་ལས་དང་རས་བལ་དང་། །ཕ་རོལ་ཚུ་རོལ་དང་ནི་གཉའ་ཤིང་
ཡཱུ་པཾ། ཡཱུ་ཡཾ་རང་ཟེར། པྲ་གྲཱི་བ་པཱ་ཏྲི་བེ་ཡཱུ་སྲན་ཕྱེ། ཥཉྩ་སྲན་སོགས་ཁུ་བ། མ་ས་ཙི་ཀྐ་སྔོ་གྲོ་ནས་བྱེ་སཽ། །ཙིཀྐ་སཽ། །ཨརྡྷ་རྩྩ་རིག་བྱེད་གྲངས། དཽ་གྷྲྀ་ཏཱ་དཱི་ནཱཾ་པུ་སྟྭཱ་དྱཾ་བཻ་དི་ཀཾ་དྷྲུ་བཾ། །ཏནྣོ་ཀྟ་མི་ཧ་ལོ་ཀེ་པི་ཏཙྩ་དསྟྱ་སྟུ་ཤེ་ཥ་བཏ།། །།སྟྲཱི་པུཾ་ས་ཡོ་ར་པ་ཏྱནྟ་དྭི་ཙ་ཏུཿ་ཥཊྤཱ་
འཛིན། །མཆོད་སྡོང་པྲ་གྲཱི་རབ་མགྲིན་བ་དང་ལྷུང་བཟེད་དང་། །གའུ་ཟླ་བ་དང་ནི་ཆང་གི་སྣོད། །ཕྱེད་མཆོད་ལ་སོགས་དང་ནི་མར་ལ་སོགས། །སྐྱེས་པ་ཉིད་ཕྱིར་གང་ཞིག་ངེས་པར་རིག །དེ་འདིར་བཤད་པ་མིན་པ་འཇིག་རྟེན་ན། །གལ་ཏེ་ཡོད་ནའང་ལྷག་མ་བཞིན་དུ་གྱིས།། །།བུད་མེད་སྐྱེས་པ་དག་ལའོ་བུ་རྒྱུད་མཐའ། །གཉིས་
