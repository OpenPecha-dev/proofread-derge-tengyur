༄༅། །གླང་པོ་སྨྱོན་པ་གླང་པོ་རྒོད། །གླང་ཆུ་ཚོགས་པ་གླང་པོ་འདུས། །གླང་མོ་བ་དང་བ་གླང་མོ། །འགྲམ་པ་ཀ་ཊ་ཆང་སྦྱིན་དང་། །གླང་པོའི་མགོ་དང་ལག་ལྡན་མཆོག །བུམ་པ་དང་ནི་གླང་འབུར་གཉིས། །དེ་དག་དབུས་ནི་བི་དུ་ཕོ། །ཨ་བ་གྲ་ཧ་དཔྲལ་བ་ཡིན། །དཱི་ཥིཀཱ་ནི་མིག་འབྲས་དག །

གླང་བུའོ། །མཱ་ལཱ་ནཾ་ནི་འདོགས་པའི་ཀ། །ལྕགས་ཐག་ཨཾ་དུ་ཀ་དང་སྒྲོག །མོ་མིན་ལྕགས་ཀྱུ་སྲྀ་ཎི་མོ། །འཛེག་སྟེགས་དང་ནི་ཡ་གདོང་ངོ་། །རྟོགས་ལྡན་སྐྱེ་བོ་དམ་པ་མཚུངས། །གླང་གཡོགས་ལ་བ་ཁེབས་དང་ནི། །སྟན་དང་ཀུ་ཐ་གཉིས་དག་གོ །བཱི་ཏཾ་གླང་པོ་ཕལ་བའོ། །བཱ་རཱི་གླང་པོའི་
