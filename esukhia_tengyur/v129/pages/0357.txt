དྲན་པ་དང་། བློ་གྲོས་དང་རྟོག་པ་ལ་སོགས་པ་ཡོན་ཏན་འཕེལ་བ་དང་། གཞན་ཡིད་ལ་མི་བྱེད་པ་དང་། རྣམ་པ་ཐམས་ཅད་དུ་སེམས་ཅན་ཡོངས་སུ་སྨིན་པར་བྱ་བ་ལ་བརྩོན་པ་དང་། འཇིག་རྟེན་པའི་བཟོ་སྒྲུབ་པ་དང་། དགེ་བའི་རྩ་བ་རྣམ་པར་དག་པ་དང་། སྐྱེ་བ་དང་
མཐུ་ལས་མདོར་བསྟན་པར་རིག་པར་བྱའོ། །རྒྱས་པར་བསྟན་པ་ནི་འདི་ལྟ་སྟེ། ས་བཅུའི་ས་ཤིན་ཏུ་སྦྱང་དཀའ་བ་ལས་འབྱུང་བ་བཞིན་ནོ། །བདེན་པ་རྣམས་ལ་རྣམ་པར་ངེས་པའི་ཤེས་པ་ནི་ཤིན་ཏུ་སྦྱང་དཀའ་བ་ཞེས་བྱ་སྟེ། འདིར་ཡང་དེ་བསྟན་པས་ན་དེའི་ཕྱིར་ས་དེ་ལ་ཤིན་ཏུ་སྦྱང་དཀའ་བ་ཞེས་
བྱ་སྟེ། དོན་དེ་ཉིད་ཀྱི་བདེན་པ་དང་ལྡན་པའི་ལྷག་པའི་ཤེས་རབ་ལ་གནས་པ་ཡིན་པར་བལྟ་བར་བྱའོ། །དེ་ལ་བྱང་ཆུབ་སེམས་དཔའ་རྣམས་ཀྱི་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་དང་ལྡན་པའི་ལྷག་པའི་ཤེས་རབ་ལ་གནས་པ་གང་ཞེ་ན། འདི་ལ་བྱང་ཆུབ་སེམས་དཔའ་བདེན་པ་དང་ལྡན་པའི་ལྷག་པའི་
ཤེས་རབ་ལ་གནས་པ་ལ་ཐོག་མ་ཉིད་དུ་ཆོས་མཉམ་པ་ཉིད་བཅུ་ཐོབ་སྟེ། གཞུང་ནི་མདོ་ལས་ཇི་སྐད་འབྱུང་བ་བཞིན་དུ་རིག་པར་བྱའོ། །དེ་དག་ཆེ་བའི་ཕྱིར་དང་ཡོངས་སུ་རྫོགས་པའི་ཕྱིར་གནས་པ་འདི་ལ་འཇུག་སྟེ། ཆོས་ཐམས་ཅད་ལ་དོན་དམ་པར་ཡོད་པའི་ངོ་བོ་ཉིད་ཀྱི་མཚན་མ་མེད་པའི་མཉམ་
པ་ཉིད་དང་། བརྗོད་པ་མངོན་པར་འདུ་བྱེད་པ་སྣང་བ་མཚན་ཉིད་མེད་པའི་མཉམ་པ་ཉིད་དང་། དེ་ཉིད་མཚན་ཉིད་མེད་པས་བདག་ཉིད་མ་སྐྱེས་པའི་མཉམ་པ་ཉིད་དང་། རྒྱུ་ལས་ཀྱང་མ་སྐྱེས་པའི་མཉམ་པ་ཉིད་དང་། རང་དང་རྒྱུ་ལས་ཀྱང་མ་སྐྱེས་པས་གཟོད་མ་ནས་ཤིན་ཏུ་ཞི་བའི་མཉམ་པ་ཉིད་དང་། ཡོད་
པའི་དངོས་པོ་འཛིན་པའི་ཤེས་པ་སྤྲོས་པ་མེད་པའི་མཉམ་པ་ཉིད་དང་། བླང་བ་དང་དོར་བའི་མངོན་པར་འདུ་བྱེད་པ་དང་བྲལ་བའི་མཉམ་པ་ཉིད་དང་། དེ་ཉིད་ཉོན་མོངས་པ་དང་སྡུག་བསྔལ་གྱི་ཀུན་ནས་ཉོན་མོངས་པ་དང་བྲལ་བའི་ཕྱིར་དབེན་པའི་མཉམ་པ་ཉིད་དང་། རྣམ་པར་བརྟགས་པའི་ཤེས་བྱའི་ངོ་བོ་
ཉིད་སྒྱུ་མ་དང་སྤྲུལ་པ་ལྟ་བུའི་མཉམ་པ་ཉིད་དང་། རྣམ་པར་མི་རྟོག་པའི་ཡེ་ཤེས་ཀྱི་སྤྱོད་ཡུལ་གྱི་ངོ་བོ་ཉིད་ཀྱི་དངོས་པོ་དང་དངོས་པོ་མེད་པ་གཉིས་སུ་མེད་པའི་མཉམ་པ་ཉིད་དེ་ནི་ཆོས་མཉམ་པ་ཉིད་བཅུ་པོ་དེ་དག་གི་དོན་རྣམ་པར་དབྱེ་བ་ཡིན་པར་རིག་པར་བྱའོ།། །།བམ་པོ་བཅུ་དགུ་པ། གནས་པ་
