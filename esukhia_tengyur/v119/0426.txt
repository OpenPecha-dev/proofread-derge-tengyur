ལས་རྨོངས་པའི་སྒྲིབ་པ་ཞེས་བསྟན་པ་ནི་འདི་ལྟར་འདོད་ཆགས་ཉིད་རྨོངས་པའི་སྒྲིབ་པ་ཡིན་པའི་ཕྱིར་རྨོངས་པའི་སྒྲིབ་པ་ཞེས་བྱ་སྟེ། འདི་ནི་ལས་ཡོངས་སུ་འཛིན་པའི་རྣམ་པར་བཤད་པའོ། །ཡང་ན་རྨོངས་པ་ནི་ཉོན་མོངས་པའི་མིང་ཡིན་ཏེ། འདི་ལྟར་ཤེས་བྱའི་སྒྲིབ་པ་ལ་རྨོངས་པ་ཞེས་བསྟན་པ་ནི་ངོ་བོ་ཉིད་
གཅིག་པའི་ཕྱིར་རོ། །བཤད་པ་འདི་དང་སྦྱར་ན་ནི་བདག་པོའི་དབང་དུ་བྱས་པའི་རྣམ་པར་བཤད་པ་ཡིན་ཏེ། རྨོངས་པའི་སྒྲིབ་པ་ཡིན་པའི་ཕྱིར་རོ། །དེ་ལ་འདོད་པའི་འདོད་ཆགས་ཀྱི་སྒྲིབ་པ་ཞེས་བྱ་བ་ནི་བདག་པོའི་དབང་དུ་བྱས་པའི་རྣམ་པར་བཤད་པའོ། །ཡང་ན་ཆོས་ལ་སྲེད་པ་ནི་འདོད་ཆགས་ཡིན་པའི་ཕྱིར་ལས་
ཡོངས་སུ་འཛིན་པའི་རྣམ་པར་བཤད་པའོ། །ས་བཞི་པ་ལ་ནི་སྙོམས་པར་འཇུག་པ་ལ་སྲེད་པ་དང་ཆོས་ལ་སྲིད་པའི་སྒྲིབ་པའི་གཉེན་པོར་གྱུར་པ་ཡིན་ནོ་ཞེས་བྱ་བ་ནི་ས་བཞི་པའི་སྒྲིབ་པ་བསྟན་པའོ། །བསྟན་བཅོས་རྣམ་པར་རིག་པ་ཙམ་དུ་གྲུབ་པ་ལས་ནི་ཡིད་ཀྱི་རྣམ་པར་ཤེས་པ་དང་ལྷན་ཅིག་སྐྱེས་པའི་
བདག་ཏུ་ལྟ་བ་དང་། སྙོམས་པར་འཇུག་པ་ལ་སྲེད་པ་དང་ཆོས་ལ་སྲེད་པའི་ངོ་བོ་ཉིད་དོ་ཞེས་བཤད་དོ། །དེའི་ཕྱིར་བསྟན་བཅོས་དེ་ཉིད་ལས་བཞི་པ་ནི་ཉོན་མོངས་པ་ཕྲ་མོ་ཀུན་དུ་འབྱུང་བའི་སྒྲིབ་པ་ནི་འདི་ལྟ་སྟེ། ཤེས་བྱའི་སྒྲིབ་པ་ལས་ལྷན་ཅིག་སྐྱེས་པའི་ཕྱོགས་གཅིག་དང་ཡིད་ཀྱི་རྣམ་པར་ཤེས་པ་དང་
ལྷན་ཅིག་པའི་འཇིག་ཚོགས་ལ་ལྟ་བ་ལ་སོགས་པས་བསྡུས་པ་གང་ཡིན་པའོ། །དེ་ལས་བཞི་པའི་བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་ལ་སྲེད་པར་བྱེད་དེ། ས་བཞི་པ་ལ་ཞུགས་པ་ལ་ཡོངས་སུ་སྤོང་བར་བྱེད་དོ་ཞེས་འབྱུང་སྟེ། ཞིབ་ཏུ་དེ་ཉིད་ལས་བསྟན་པ་བཞིན་ནོ། །དེ་ལ་འཇིག་ཚོགས་ལ་ལྟ་བ་ལ་སོགས་པ་
ཞེས་བྱ་བའི་ཚིག་གིས་ནི་ཐོག་མ་མེད་པའི་ཤེས་བྱའི་སྒྲིབ་པ་དང་། སྙོམས་པར་འཇུག་པའི་སྲེད་པ་དང་། ཆོས་ལ་སྲེད་པ་ཡང་བསྡུ་བ་སྟེ། སྙོམས་པར་འཇུག་པ་ལ་སྲེད་པ་དང་། ཆོས་ལ་སྲེད་པ་དེ་དག་ནི་ས་གསུམ་པ་དག་གིས་ཀྱང་འཕེལ་བར་བྱེད་པས་ས་བཞི་པ་ལ་ཞུགས་པའི་ཚེ། གདོད་ཡོངས་
སུ་སྤོང་བར་བྱེད་དེ། བྱང་ཆུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆོས་ནི་དེ་དག་དང་ཤིན་ཏུ་མི་མཐུན་པའི་ཕྱིར་རོ། །གཞན་ཡང་མདོ་འདི་ཉིད་ལས་འོག་ནས་ཀུན་དུ་རྨོངས་པ་རྣམ་པ་གཉིས་སོ་ཞེས་སྟོན་པར་མཛད་དེ། སྙོམས་པར་འཇུག་པ་ལ་སྲེད་པའི་ཀུན་དུ་རྨོངས་པ་དང་། ཆོས་ལ་སྲེད་པའི་ཀུན་དུ་རྨོངས་པ་དང་།
