རང་བཞིན་གྱི་ཚོགས་དེ་གང་དུ་ཐལ་ལོ་ཞེས་ཤེས་པ་མ་ཡིན་ཏེ་མཐར་ཐུག་གི་དོན་ཏོ། །འདིར་ནི་ཀུན་དུ་སྒྲས་མ་བཤད་པ་ཡང་ཀུན་ལས་བསྡུ་བར་བྱེད་པ་སྟེ། །རྗེས་སུ་གཞིག་པའི་དཔེ་ཡང་མཚོན་པར་བྱའོ། །དེ་ལ་ཇི་ལྟར་དྲི་མ་མེད་པའི་སྟོན་ཀའི་མཚོའམ།
ཆུ་བོའི་ཆུ་ལ་གཟུགས་བརྙན་དུ་གྱུར་པའི་ཟླ་བའི་དཀྱིལ་འཁོར་དེ་ཉིད་དང་གཞན་ཉིད་དག་གིས་བརྗོད་དུ་མེད་པ་དེ་བཞིན་དུ། བདག་དང་གཞན་གྱི་རྒྱུ་ལ་གནས་པའི་འོད་གསལ་བ་བརྗོད་དུ་མེད་པ་ཤེས་པར་བྱ་སྟེ་མཐར་ཐུག་ཁོ་ནའོ། །སླར་ཡང་སྤྱོད་ལམ་གྱི་རྣམ་པར་དག་པ་
བཤད་པར་བྱ་བའི་ཕྱིར། འགྲོ་དང་འདུག་དང་ཉལ་དང་ཟ། །མིག་བཙུམས་འབྱེད་དང་དགོད་པ་ན། །བསམ་གཏན་སྦྱོར་བ་འདི་ཡིས་ནི། །དེ་ཉིད་རིག་པ་རྟག་ཏུ་གནས། །ཞེས་བྱ་བ་གསུངས་ཏེ། །དེ་ཉིད་རིག་པ་ནི་ཐམས་ཅད་སྟོང་པའི་གོ་འཕང་ལ་གནས་པའི་རྣལ་འབྱོར་
པས་འགྲོ་བ་ལ་སོགས་པའི་སྤྱོད་ལམ་ལ་འོད་གསལ་བའི་ཏིང་ངེ་འཛིན་གྱིས་གནས་པར་བྱ་སྟེ་སྒྲ་ཇི་བཞིན་པའོ། །འོད་གསལ་བ་ལ་ཞུགས་པས་སེམས་ཅན་གྱི་དོན་ཇི་ལྟར་བྱེད་པ་ཡིན་ནོ་ཞེས་བཤད་པར་བྱ་བའི་ཕྱིར། གཟུགས་བཅས་འོད་ཟེར་དེ་ཉིད་དང་། །རྣམ་ཤེས་རླུང་
དང་ལྡན་པ་ཡིས། །བདག་བྱིན་བརླབ་པའི་རིམ་གྱིས་ཀྱང་། །སེམས་ཅན་དོན་དུའང་ལན་འགའ་འགྱུར། །ཞེས་བྱ་བ་གསུངས་ཏེ། འོད་གསལ་བ་ལ་ཞུགས་པའི་རྣལ་འབྱོར་པས་གལ་ཏེ་སྔོན་གྱི་སྨོན་ལམ་དབང་གིས་སེམས་ཅན་གྱི་དོན་བྱེད་ན། དེའི་ཚེ་བདག་ལ་བྱིན་
གྱིས་བརླབ་པའི་རིམ་པས་བྱ་བ་ཡིན་ནོ། །འོ་ན་སྒྱུ་མ་ལྟ་བུའི་ཏིང་ངེ་འཛིན་ནི་གསུམ་པ་ཡིན་པའི་ཕྱིར། དེ་ཁོ་ན་ཉིད་བཞི་པ་དང་། འདི་ལ་ཇི་ལྟར་མངོན་དུ་བྱ་བ་ཡིན་ཞེ་ན། སྨྲས་པ། བདག་ལ་བྱིན་གྱིས་བརླབ་པའི་རིམ་པ་རྣམ་པ་གཉིས་ཏེ། །གཅིག་ནི་རིམ་པ་གསུམ་པ་
རྗོད་པར་བྱེད་པའོ། །གཉིས་པ་ནི་ཟུང་འཇུག་གི་རིམ་པ་རྗོད་པར་བྱེད་པ་སྟེ། །དེ་ཡང་འོག་ནས་བཤད་པར་འགྱུར་བ་འདིར་མངོན་དུ་བྱས་ཏེ་འགལ་བ་མེད་དོ། །ཐ་མ་ནི་བདག་ལ་བྱིན་གྱིས་བརླབ་པ་ཞེས་ཇི་ལྟར་ཡིན་ཞེ་ན། སྨྲས་པ། བདག་དང་བདག་ནི་བདག་སྟེ་ཀུན་
