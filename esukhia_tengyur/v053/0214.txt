མི་ཕེབས་ཏེ། །བླ་མའི་བརྡ་ལས་ངེས་མཐོང་བྱ། །ཉམས་སུ་མྱོང་བ་ངོས་གཟུང་བྲལ། །འདི་ནི་རྣལ་འབྱོར་མ་ཡི་ཡུལ། །ཞེས་གསུང་ངོ་། །རྩ་བའི་དོན་དབང་བཞི་ལས་རྙེད་པ་ཨ་བ་དྷཱུ་ཏཱི་པའི་ཞལ་སྔ་ནས། མི་མཐུན་ལམ་དུ་བསླང་བ་ཡིས། །འགལ་བའི་རྐྱེན་རྣམས་
གྲོགས་སུ་འགྱུར། །འདིས་ནི་སྣ་ཚོགས་ལམ་དུ་སློངས། །<བགགས་,བགེགས་>སོལ་སྣང་བ་རང་སར་གྲོལ། །ཞེས་གསུང་ངོ་། །ཟླ་བའི་འོད་ཟེར་མ་གྷ་མ་ལི་ཏིའི་ཞལ་སྔ་ནས། ནམ་མཁའ་མདུང་ཚུགས་བསམ་གཏན་མཁར། །རིག་པ་སོར་བཞག་གནད་ཀྱི་མཚང་། །བདེ་གསལ་
རྒྱུན་མི་འཆད་པར་བྱེད། །ཡུལ་ལ་ཡེངས་པར་མི་བྱའོ། །ཞེས་གསུང་ངོ་། །རྟེན་འབྲེལ་གྱི་འགྲོས་ལ་བྱང་བ་ཀྲྀཥྞ་པའི་ཞལ་སྔ་ནས། ཀྱེ་མ་ངོ་བོ་ཉིད་ཀྱིས་ཟབ། །གཟུགས་ཀྱི་རང་བཞིན་དེ་ལྟོས་དང་། །སྟོང་པའི་མཚན་ཉིད་གསལ་འགྱུར་གྱིས། །གཟུགས་ཀྱི་མཚང་མཐོང་དེ་ཉིད་

རྗེས་མི་འབྲང་། །སྣང་བ་ཧད་པོས་དེར་འཛིན་གྲོལ། །སོར་བཞག་ཚོགས་དྲུག་ལྷུག་པའི་ངང་། །བསྒོམ་བློ་འདུན་པ་དྲི་མ་ཡིན། །ཞེས་གསུང་ངོ་། །མོས་གུས་ཀྱིས་རྒྱུད་སྨིན་པ་ཨཱརྱ་སིངྒའི་ཞལ་སྔ་ནས། ཆོས་རྣམས་མ་ལུས་རྣམ་ཀུན་ནི། །རྗོད་བྱེད་ཚིག་དང་བརྗོད་བྱའི་དོན། །
རྣམ་པ་གཉིས་ཀྱིས་སྟོང་འགྱུར་བས། །གཞི་རྩ་མེད་པའི་བདག་ཉིད་ཅན། །ཞེས་གསུང་ངོ་། །བཟོད་པའི་སྟོབས་ལྡན་མ་ན་ཏ་རའི་ཞལ་སྔ་ནས། ཚོགས་དྲུག་ཡུལ་གྱི་སྣང་བ་ནི། །སྣ་ཚོགས་རྐྱེན་གྱི་སྣང་བ་སྟེ། །རྐྱེན་སྣང་སྒྱུ་མ་ལས་མ་འདས། །རང་བཞིན་མེད་དེ་ངོ་མཚར་
