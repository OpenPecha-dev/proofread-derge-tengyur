ཐམས་ཅད་ཀྱིས་སོ། །གཅིག་ཉིད་དགའ་བ་ཆེན་པོ་ནི་ལྷན་ཅིག་སྐྱེས་པའི་དགའ་བའོ། །ལྔ་རུ་འགྱུར་བའི་དབྱེ་བ་ནི། །ཞེས་པ་ནི་དེ་དབྱེ་བའི་གྲངས་ཀྱིས་སོ། །དེ་བཞིན་གཤེགས་པའི་དབྱེ་བས་རྡོ་རྗེ་སེམས་དཔའི་དབྱེ་བ་སྟེ་རིགས་ལྔ་རྣམས་སུའོ། །ཡང་དེ་བཞིན་གཤེགས་པ་རྣམས་ཉིད་ནང་
གི་དབྱེ་བའི་བསམ་པས་སོ། །དེའི་ཕྱིར་གཅིག་གི་ངོ་བོ་ནི་ཞེས་པ་འདི་ཉིད་ཀྱི་ཕྱིར་ཐམས་ཅད་སྐྱེ་བ་མེད་པ་དེ་ཉིད་ཀྱི་རང་བཞིན་ཉིད་དེ། ཇི་ལྟར་ས་ལས་བྱུང་བ་བུམ་པ་དང་ཁམ་ཕོར་ལ་སོགས་པ་སའི་གཟུགས་མ་སྤངས་པ་ལྟ་བུའོ། །དེ་ཉིད་ཀྱིས་གསུངས་པ། མཆོག་ཏུ་དགའ་བ་ལས་བྱུང་བ། །
ཞེས་པ་སྟེ། དེ་བས་ན་དེ་ཉིད་ལས་བྱུང་བ་ཐམས་ཅད་དུ་མཐའ་ཡས་པའི་རིགས་གཞན་རྣམས་ཀྱང་ཞེས་བྱ་བའི་དོན་ཏོ། །བརྟག་པ་གཉིས་པ་ལས་དངོས་གྲུབ་གཏན་ལ་དབབ་པ་སྟེ་ལེའུ་གཉིས་པའི་བཤད་པའོ།། །།དེ་ནས་རྡོ་རྗེ་ཅན་ཞེས་པ་རྡོ་རྗེ་སེམས་དཔའོ། །རྒྱུད་ཐམས་
ཅད་ཀྱི་གླེང་གཞི་ཞེས་པ་ནི་རྒྱུད་ཀུན་གྱི་གླེང་གཞི་སྟེ་གང་གི་དོན་གྱིས་རྒྱུད་མ་ལུས་པ་ལ་འཇུག་པ་དེའི་ཐབས་ཞེས་བྱའོ། །རྣལ་འབྱོར་མ་རྣམས་ཀྱི་ཞེས་བྱ་བ་ནི་རྣལ་འབྱོར་མ་དག་གི་ཕྱིར་རམ། རྣལ་འབྱོར་མ་རྣམས་ཏེ་ཐབས་གང་གིས་འགྲུབ་བའོ། །དེའི་ཐབས་བཀའ་སྩལ་པ། དེ་ཅི་
སྙམ་པ་ལ། སྡོམ་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་བཞིའོ། །སྡོམ་པ་ནི་བདེ་བའི་མཆོག་གམ་ཆོས་ཐམས་ཅད་ཀྱི་བདག་ཉིད་དེ་བཞིན་གཤེགས་པས་བསྡུས་པའོ། །དབང་བསྐུར་བ་ལས་ཤེས་པར་བྱ། །ཞེས་པ་ནི་གང་དུ་ཨེ་དང་ཝཾ་ལས་དབང་བསྐུར་བ་ཤེས་པར་བྱེད་པའོ། །ཡང་དག་
ཅེས་པ་ནི་མཆོག་གི་དབང་བསྐུར་བས་དྲི་མ་མ་ལུས་པར་འཁྲུ་བར་བྱེད་པའོ། །དེ་ཅི་སྙམ་པ་ལ་གསུངས་པ། བདེ་བ་ཆེན་པོ་སྟེ་བདེ་བ་ཆེན་པོའི་ཡེ་ཤེས་སོ། །དེ་ཉིད་ཀྱི་དོན་གསལ་བར་བྱ་བའི་ཕྱིར་རྣལ་འབྱོར་མ་རྣམས་ཀྱིས་ཞུས་པ་དེ་ནས་ཤེས་པའོ། །ཝཾ་ནི་ཐབས་སུ་གྱུར་པའོ། །བརྒྱན་
པ་ནི་རྒྱན་བྱས་པའོ། །དེ་གསུངས་པ། ཨེ་ནི་ཤེས་རབ་ཅེས་བྱར་གསུངས། །ཝཾ་ནི་ཐབས་ཞེས་བྱ་བ་གང་། །ཝཾ་ཞེས་བྱ་བས་མཛེས་པ་ཡང་། །ཨེ་ནི་ངེས་པར་མཛེས་པ་ཡིན། །སྐད་ཅིག་ཡེ་ཤེས་ཞེས་པ་སྐད་ཅིག་གི་དབྱེ་བས་དགའ་བ་ལ་སོགས་པ་ཡོངས་སུ་ཤེས་ན་བདེ་བའི་ཡེ་
