བས། །ཇི་ལྟར་ཤེས་པའི་བདེ་ཆེན་པོ། །དེ་ཡི་འདི་ནི་བྱིན་རླབས་ཡིན། །གཞན་ལས་དཀྱིལ་འཁོར་འབྱུང་བ་མེད། །ཅེས་པ་སྟེ་ཇི་ལྟར་སྒྱུ་མ་རྨི་ལམ་ལ་སོགས་པར་ཤེས་པ་ཁོ་ན་ཕྱག་རྒྱ་ཆེན་པོ་དང་། འཇིག་རྟེན་པའི་སྤྱོད་པ་ལས་འགྲོལ་བའི་དབང་
དང་། ཡུམ་སྒྱུ་འཕྲུལ་ཆེན་མོ་དང་། ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པ་ཞེས་བྱ་བ་ཡིན་ནོ། །དེའི་ཕྱིར། འདི་ཉིད་ཤེས་པར་བྱ་བའི་ཕྱིར། །བདེ་བ་གནག་ཅིང་བདེ་བ་སེར། །བདེ་བ་དམར་པོ་བདེ་བ་དཀར། །བདེ་བ་ལྗང་གུ་བདེ་བ་སྔོ། །བདེ་བ་རྒྱུ་དང་མི་རྒྱུ་ཀུན། །བདེ་བ་ཤེས་རབ་
བདེ་བ་ཐབས། །ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། དྲང་དང་ངེས་པའི་བྱེ་བྲག་ལས། །བསྐྱེད་དེ་སྤྲོས་ལ་ཞེན་སྤང་ཕྱིར། །དཀྱིལ་འཁོར་ཁ་དོག་ཡང་ཡང་བསྒྱུར། །ལས་ལ་སོགས་པའང་འགྲུབ་འགྱུར་ལ། །རྡོ་རྗེ་འཆང་ཆེན་དྲུག་ཀྱང་ཡིན། །གཞན་དུ། ཤེས་རབ་ཡེ་ཤེས་མེ་བོ་
ཆེ། །ཞེས་བྱ་བ་ལ་སོགས་པས་དུ་བ་ལ་སོགས་པའི་ལམ་གྱི་མཚན་མར་གསུངས་ཏེ། སོ་སོར་སྡུད་པའི་བདེ་བ་ཉིད་དེ་ལྟར་སྣང་བའི་ཕྱིར་རོ། །བདེ་བཞིན་དུ་ས་བོན་གྱི་བདེ་བ་དེ་ཉིད་ཀྱི་དུས་སུ་གང་ཅུང་ཟད་དམིགས་པའི་གཟུགས་དང་ཁ་དོག་དང་རྒྱུ་བ་དང་མི་རྒྱུ་བ་རྣམས་བདེ་བ་ལས་གཞན་པར་མ་
མཐོང་བའི་ཕྱིར་རོ། །དེ་བཞིན་ཀུན་དུ་རུ་སྐྱེས་བདེ། །བདེ་བ་དངོས་དང་དངོས་མེད་བདེ། །བདེ་བ་རྡོ་རྗེ་སེམས་དཔར་བརྗོད། །ཅེས་གསུངས་ཏེ། དེ་བས་འདོད་ཆགས་ཀྱི་མེས་ཚིག་པ་རྣམས་ལ་ཀུན་དུ་རུའི་བདེ་བ་ཉིད་མཆོག་ཏུ་བཤད་དེ། དངོས་དང་དངོས་མིན་གྱི་རང་བཞིན་གྱིས་རྡོ་རྗེ་
སེམས་དཔའི་བདེ་བ་འཐོབ་པར་བྱེད་པའི་ཕྱིར་ལ་མི་འགྱུར་བ་རྣམས་ལ་ཇི་བཞིན་པའི་རྡོ་རྗེ་སེམས་དཔའི་བདེ་བའོ། །ད་ནི། སྤྲོས་པ་རྨི་ལམ་ལྟ་བུ་དང་། །ཞེས་པས། རྫོགས་པའི་རིམ་པ་ཤེས་པར་བྱ། །ཞེས་པའི་གསུང་གི་དགོངས་པ་དགྲོལ་བའི་ཕྱིར། རྡོ་རྗེ་སྙིང་པོས་གསོལ་པ། རྫོགས་
པའི་རིམ་པའི་རྣལ་འབྱོར་འདི། དེ་ཡི་བདེ་བ་བདེ་ཆེན་བརྗོད། །རྫོགས་པ་བསྒོམ་པ་མེད་པ་སྟེ། །བསྐྱེད་པ་ཡིས་ནི་ཅི་ཞིག་འཚལ། །ཞེས་བྱ་བ་ལ། ཀྱེ་བཅོམ་ལྡན་འདས་དབང་པོ་གཉིས་ཀྱི་བདེ་བ་ཉིད་ཀྱིས་སངས་རྒྱ་བ་ལགས་ན། རྫོགས་པའི་བསྒོམ་པ་གཞན་མ་མཆིས་ཏེ། དེ་ལས་
