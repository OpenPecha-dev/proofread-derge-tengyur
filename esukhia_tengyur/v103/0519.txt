དེ་ནི་ཐེ་ཚོམ་གྱི་ཀུན་དུ་སྦྱོར་བ་ཞེས་བྱའོ། །ཕྲག་དོག་ནི་གཞན་གྱི་ཡོན་ཏན་གྱིས་གདུང་བ་སྟེ་ཆོས་གང་དང་ལྡན་ན་གང་དང་གང་དུ་ཕ་རོལ་དུ་ཐོས་པ་དང་། ཚུལ་ཁྲིམས་དང་། གཏོང་བ་དང་། གཟུགས་ལ་སོགས་པའི་ཡོན་ཏན་ཚོགས་པ་ཉེ་བར་དམིགས་པ་དེ་
དང་དེར་གང་ཟག་དེ་ལ་ནང་དུ་ཤིན་ཏུ་གདུང་བ་འབྱུང་སྟེ། ཆོས་དེ་ནི་ཕྲག་དོག་ཅེས་བྱ་བར་བརྗོད་དེ། ཆོས་དེ་ནི་ཕྲག་དོག་གི་ཀུན་དུ་སྦྱོར་བའོ། །ཆོས་དང་མཁས་པའི་བདག་ཉིད་ལ་སོགས་པ་སྦྱིན་པ་དང་རབ་ཏུ་འགལ་བ་སེམས་ཀྱིས་ཀུན་ཏུ་འཛིན་པ་ནི་སེར་སྣ་སྟེ། དེ་ནི་སེར་སྣའི་ཀུན་
དུ་སྦྱོར་བ་ཞེས་བྱའོ། །ཀུན་དུ་སྦྱོར་བ་དགུ་བཤད་ཟིན་ཏོ། །ངེས་པར་འཆིང་བ་ཞེས་བྱ་བ་ལ་ངེས་པར་འཆིང་བ་ནི་རྣམ་པ་གསུམ་སྟེ། འདོད་ཆགས་ཀྱི་ངེས་པར་འཆིང་བ་དང་། ཞེ་སྡང་གི་ངེས་པར་འཆིང་བ་དང་། གཏི་མུག་གི་ངེས་པར་འཆིང་བའོ། །འཆིང་བ་འདི་དག་གིས་
ཁམས་གསུམ་པོ་རྣམས་རབ་ཏུ་བཅིངས་པས་ན་འདི་དག་ནི་ངེས་པར་འཆིང་བས་ཕྲ་རྒྱས་ཞེས་བྱ་བ་ལ་འདིར་དྲུག་ནི་ཕྲ་རྒྱས་ཀྱི་རྩ་བ་སྟེ། འདོད་ཆགས་ཀྱི་ཕྲ་རྒྱས་དང་། ཁོང་ཁྲོའི་ཕྲ་རྒྱས་དང་། ང་རྒྱལ་གྱི་ཕྲ་རྒྱས་དང་། མ་རིག་པའི་ཕྲ་རྒྱས་དང་། ལྟ་བའི་ཕྲ་རྒྱས་དང་། ཐེ་ཚོམ་
གྱི་ཕྲ་རྒྱས་ཞེས་བྱ་བ་སྟེ། ཕྲ་རྒྱས་དྲུག་པོ་འདི་དག་ལས་འདོད་པའི་འདོད་ཆགས་དང་། སྲིད་པའི་འདོད་ཆགས་ཞེས་བྱ་བས་འདོད་ཆགས་རྣམ་པ་གཉིས་སུ་དབྱེ་བས་ཡང་རྣམ་པ་བདུན་དུ་འགྱུར་རོ། །ཕྲ་རྒྱས་དྲུག་པོ་འདི་དག་ལས་ཡང་ལྟ་བའི་ཕྲ་རྒྱས་ལ་འཇིག་ཚོགས་
ལ་ལྟ་བ་ལ་སོགས་པ་རྣམ་པ་ལྔའི་དབྱེ་བས་ཕྲ་རྒྱས་བཅུར་འགྱུར་ཏེ། ཕྲ་རྒྱས་བཅུ་པོ་འདི་དག་རང་གི་ངོ་བོ་ཡང་ཁམས་ཀྱི་དབྱེ་བ་དང་། བྱེ་བྲག་གི་དབྱེ་བས་དགུ་བཅུ་རྩ་བརྒྱད་དུ་འགྱུར་རོ། །དེ་ལ་ཁམས་ནི་གསུམ་སྟེ། འདོད་པའི་ཁམས་དང་། གཟུགས་ཀྱི་ཁམས་དང་།
གཟུགས་མེད་པའི་ཁམས་སོ། །དེ་ལ་འདོད་པའི་ཁམས་ནི་གནས་ཉི་ཤུ་འདི་ལྟ་སྟེ། དམྱལ་བ་ཆེན་པོ་བརྒྱད་ལ་ཡང་སོས་དང་། ཐིག་ནག་དང་། བསྡུས་འཇོམས་དང་། ངུ་འབོད་དང་། ངུ་འབོད་ཆེན་པོ་དང་། ཚ་བ་དང་། རབ་ཏུ་ཚ་བ་དང་། མནར་མེད་པ་ཞེས་བྱ་བའོ། །
