བསྟན་པ་ཡིན་ནོ། །འདིར་འདི་བརྗོད་པར་བྱ་སྟེ། ཇི་སྟེ་རྟ་ཁོ་ན་ཡིན་ནམ། འོན་ཏེ་སྲིན་པོ་ཡིན་ཞེ་ན། དེ་སྲིན་པོ་ཡིན་ཏེ། སྦྱོར་བ་འགའ་ཞིག་གིས་དེ་ལ་རག་ལུས་པ་ཡིན་ནོ་ཞེས་གནས་པའི་སྦྱོར་བ་འདི་ཁོ་ནས་དེ་ཐར་བར་གྱུར་པའི་ཡིན་ནོ་ཞེས་བླ་མ་དག་སྟོན་པར་མཛད་དོ། །བ་
ལང་སྐྱོང་གི་མ་ཞི་བ་ཡང་ཇི་ལྟར་སྨྲ་བར་འགྱུར་ཞེ་ན། འདི་ལ་ཡང་ཞི་བ་ལ་ལྷས་རྨི་ལམ་དུ་བསྟན་པའི་དུས་དེ་ཤེས་པ་དེས་སྨྲས་པ་ཡིན་ནོ་ཞེས་བསྟན་པར་འགྱུར་རོ། །དུས་ཤེས་པ་ཞེས་བྱ་བ་ནི་ཁྲོ་བོའི་དུས་ལ་མཁས་པའོ། །མཚན་ཉིད་དང་ལྡན་པའི་གླང་པོ་ཆེའི་མཚན་ཁུང་གི་
ཕྱོགས་སུ་རྐང་པ་གཞག་པར་བྱ་བའི་ཕྱིར་པགས་པའི་རྒྱན་གྱིས་བསྟད་པའི་ཚེའོ། །སྐད་ཕྱུང་བ་ནི་གླང་པོ་ཆེ་སྒྲ་སྒྲོགས་པའོ། །དཔག་ཚད་བརྒྱ་སོང་བའི་འོག་ཏུ་བྱང་ཕྱོགས་ཀྱི་རྒྱ་མཚོར་སོང་ནས་ཆུ་བཏུང་བར་བྱའོ་ཞེའོ། །སྨྲས་པ་ནི་གནད་དུ་བསྣུན་པས་སྣུན་པར་བྱེད་པའོ། །བྱའི་
རིགས་ཏི་ཏི་ར་དེ་རབ་ཏུ་སྐད་འབྱིན་པའི་ཕྱིར་སྤྱོད་ཡུལ་ནགས་ཐིབས་པོ་ཡིན་པ་དེ་བས་ན་སྤྱོད་ཡུལ་ཞེས་བསྟན་ཏོ། །འདོད་པའི་ཕྱིར་ཞེས་བྱ་བ་ནི་གསད་པར་འདོད་པ་བདག་གིར་བྱེད་པའོ། །གཡོ་བར་བྱེད་ཅེས་བྱ་བ་ནི་ཕྱོགས་གང་གནས་པ་དེ་ཉིད་ལས་ཅུང་ཞིག་གཡོ་བར་
བྱེད་པའོ། །རབ་ཏུ་གཡོ་བར་བྱེད་ཅེས་བྱ་བ་ནི་ཕྱོགས་གཞན་དུ་འཕོས་པས་ཀུན་དུ་གཡོ་བར་བྱེད་པའོ། །སོང་བ་ནི་ཆུ་ངེས་པར་འབྱུང་བའི་ཐབས་ཀྱིས་སོ། །གོ་བར་བྱེད་པ་ཞེས་བྱ་བ་ནི་རིགས་པའི་དོན་ཁོང་དུ་ཆུད་པར་བྱས་པས་སོ། །མངོན་པར་གཙེས་པ་ནི་
རྣམ་པར་འཚེ་བར་བྱེད་པའོ། །ནང་ན་གནས་པ་ཞེས་བྱ་བ་ལ་གནས་ནི་ཕྱི་རོལ་གྱི་སྐྱེ་བོ་ཆེན་པོ་འདུག་པའི་གནས་ཡིན། ནང་ནི་ནང་དུ་གྱུར་པ་དག་ཡིན་ནོ། །རིགས་པ་སྨྲ་བ་ཞེས་བྱ་བ་ནི་སྔགས་ཀྱི་ཚུལ་ལ་མཁས་པའོ། །མིའི་ལག་པ་རིག་པས་ཨུཏྤལ་སྔོན་པོའི་ལག་
པར་བྱས་ནས་བྱིན་ཏེ། མི་རྟོགས་པ་ནི་མིར་ཡོངས་སུ་སྒྲོགས་པའོ། །གྲོང་གི་སྲང་ནི་ཁང་བཟངས་ཆེན་པོའོ། །མ་མོའི་དཀྱིལ་འཁོར་ཞེས་བྱ་བ་ལ། སྲིན་མོ་བརྒྱད་ལ་མ་མོར་བརྗོད་ལ། དེའི་དམ་ཚིག་ནི་དཀྱིལ་འཁོར་ཡིན་ཏེ། དེ་ལ་བརྟེན་པའི་ཆོ་གས་དཀྱིལ་འཁོར་
