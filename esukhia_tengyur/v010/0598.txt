ཁྱད་པར་གྱི་ནི་ཚིག་ཀྱང་བརྗོད། །མགོན་པོ་ཁྱོད་ལ་འདི་འཚལ་ལོ། །དེ་ནས་སློབ་མ་ཕྱིར་གྱེད་བྱ། །ནང་མིན་ཧ་ཅང་རིང་བའང་མིན། །རིམ་པས་བུ་མོ་དབང་བསྐུར་བྱ། །དེ་ལ་གསང་ཆོས་བཤད་པར་བྱ། །ཕྱག་རྒྱའི་བརྡ་དང་དབང་གི་དོན། །སྐད་ཅིག་རྣམས་ཀྱང་དེ་ལྟར་རོ། །
དེ་ནས་གཅིག་ཏུ་སྦྱོར་བ་ཡིས། །ལེགས་དགྱེས་དགྱེས་པ་རབ་བགྱེད་དོ། །སྣང་བ་རེག་བྱ་འདུས་པ་ནི། །ཡེ་ཤེས་གསུམ་ནི་འབྱུང་བར་འགྱུར། །ཡང་དག་ཡེ་ཤེས་གསུམ་སྐྱེས་ནི། །བྱང་ཆུབ་སེམས་ནི་སྤང་མི་རུང་། །སྤངས་ན་འགྲོ་བའི་གནས་སུ་ནི། །བདེ་ཆེན་སེམས་ནི་
འགྲོ་བར་འགྱུར། །འགྲོ་ན་དེ་ནི་ཆད་པ་ཡིན། །ཤེས་རབ་ཡེ་ཤེས་གསལ་ཡིན་ནོ། །དཔལ་གསང་འདུས་པའི་ནང་ནས་ཀྱང་། །རྡོ་རྗེ་ཉི་ནུབ་ཞེས་ནི་འབྱུང་། །མཚམས་ནི་སྣང་བ་ཐོབ་པ་ཡིན། །རིམ་པ་ལྔ་པ་ལས་ཀྱང་གསུངས། །དེ་ནི་གསང་བར་བཤད་པ་ཡིན། །གསང་
བར་སྣང་བ་མཆེད་པ་བཤད། །འདུད་ནས་དེ་ནི་སྣང་བ་མཆེད། །དེ་ནི་གཅིག་པུ་ཕོ་བྲང་གསང་། །དེར་ནི་རང་གིས་ཤེས་པར་བྱ། །སེམས་ནི་འདམ་སྐྱེས་ནམ་མཁར་མིན། །དེ་ནས་སློབ་མ་དགུག་པར་བྱ། །དེ་ནས་རས་ཀྱིས་གདོང་ཡང་དགབ། །པདྨ་རབ་ཀྱི་ཁ་དབུས་ནས། །རྡོ་རྗེ་
རྒྱལ་པོ་དྲང་བར་བྱ། །བུ་ཡི་ཁ་པད་ནང་དུ་ནི། །སྦོམ་པོ་ཐ་ལྟག་གིས་མནན་ནོ། །རྩ་གཉིས་མཆོག་གི་ནང་ནས་ནི། །ལོངས་སྤྱོད་འཁོར་ལོའི་ནང་དུ་བཏང་། །སྐྱེས་བུ་བདུད་རྩི་ཇི་ལྟ་བུ། །ཁྱོད་ཀྱིས་སྨྲོས་ཤིག་དགའ་བ་ཡིན། །ཨེ་མའོ་མཆོག་གི་བདུད་རྩི་ཡིན། །ཨེ་མའོ་ཡེ་ཤེས་
བདུད་རྩི་རབ། །ཨེ་མའོ་རོ་ནི་བརྗོད་མི་ལང་། །ཨེ་མའོ་རྡོ་རྗེ་བདུད་རྩི་རབ། །{𰶱དི་རིང་,དེ་རིང་}བདག་སྐྱེས་འབྲས་བུ་མཆོག །{𰶲དི་རིང་,དེ་རིང་}བདག་ནི་རབ་ཏུ་རྒྱལ། །{𰶳དི་རིང་,དེ་རིང་}བདག་ནི་འཁོར་ལས་རྒྱལ། །{𰶴དི་རིང་,དེ་རིང་}བདག་ལ་ཐེ་ཚོམ་མེད། །ཁྱོད་ཀྱི་བཀའ་དྲིན་ཡང་དག་མཆོག །བརྗོད་མེད་གནས་ནི་བདག་
གིས་རྙེད། །དེ་ནས་ཡང་ནི་ཞུ་བ་བྱ། །བསྟོད་པ་སྣ་ཚོགས་རབ་བགྱིས་ཏེ། །ཡེ་ཤེས་བདེ་བའི་གནས་བཞུགས་ཏེ། །རྒྱལ་འགྱུར་རྒྱལ་འགྱུར་མགོན་པོ་ཁྱོད། །ཨཱ་ལི་ཀཱ་ལི་གཉིས་བདག་ཅན། །ལེགས་དགྱེས་བདེ་མཚོ་རྒྱལ་གྱུར་ཅིག །མཆོག་དང་ཆད་པའི་དབུས་སུ་ནི། །རོལ་
