ལེ་ཡི། །རྒྱུད་ལས་རྡོ་རྗེ་སེམས་དཔས་གསུངས། །སློབ་མའི་མཚན་ཉིད་བཅུ་གཉིས་ཏེ། །ཁ་སྦུབ་དུག་དང་རྡོལ་བ་མེད། །དབང་གི་སྣོད་རབ་ཡོད་པ་དང་། །ཐོས་ཆེ་བ་དང་དད་ཆེ་བ། །འདོད་ཆགས་ཆེ་དང་ཞི་དྲང་དང་། །ཐམས་ཅད་ཡོད་དང་གསང་ཐུབ་དང་། །གཏོང་ཕོད་ཕྲག་དོག་རྐྱང་བའོ། །
གནས་བརྟན་བ་ལ་རྣམ་གཉིས་ཏེ། །རུང་མི་རུང་བ་བརྟག་བའོ། །མི་རུང་བ་ལ་རྣམ་པ་གསུམ། །རྒྱལ་པོ་ཆོམ་རྐུན་མུ་སྟེགས་ཀྱིས། །འཇིགས་པའི་ས་ནི་ངན་པ་ཡིན། །རུང་བ་ཡི་ནི་སྐྱོན་མེད་པའོ། །དཀྱིལ་འཁོར་ལ་ཡང་རྣམ་གསུམ་སྟེ། །རྡུལ་ཚོན་མེ་ཏོག་ཏིང་ངེ་འཛིན། །དབང་པོ་རབ་འབྲིང་དབང་དུ་བྱས། །
གསུམ་ནི་རིགས་ཆེན་རྒྱ་མཚོ་ཡིས། །རྒྱུད་ལས་དེ་སྐད་འབྱུང་བའོ། །དབང་བསྐུར་བ་ལ་རྣམ་བཞི་སྟེ། །བུམ་གསང་ཤེས་རབ་ཡེ་ཤེས་དང་། །དེ་ལྟར་དེ་བཞིན་བཞི་པའི་དབང་། །བུམ་པའི་དབང་ལ་བཅུ་གཅིག་སྟེ། །ཆུ་དང་དབུ་རྒྱན་རྡོ་རྗེ་དྲིལ། །མིང་དང་དམ་ཚིག་བརྟུལ་ཞུགས་དང་། །
བཀའ་སྩལ་ལུང་བསྟན་སློབ་དཔོན་དབང་། །དབང་གསུམ་གཞན་དུ་ཤེས་པར་བྱ། །རྒྱུད་ཀུན་ལ་ནི་རྣམ་གཉིས་ཏེ། །ཐབས་དང་ཤེས་རབ་དག་རིག་བྱ། །ཐབས་ནི་དེ་ལ་རྣམ་བཞི་སྟེ། །ཀྲྀ་ཡ་ཙརྱ་ཡོ་ག་དང་། །ཡོ་གོཏྟ་ར་དང་རྣམས་སོ། །ཤེས་རབ་ཡོ་ག་ནེ་རོཏྟའོ། །དབང་དེ་རྣམས་ལ་
