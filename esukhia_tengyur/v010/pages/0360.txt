ནས་བརྩམས་ནས་ཁྱོད་ཀྱི་དཔལ་ཧེ་རུ་ཀ་ང་ཡིན་གྱི་གང་ངས་འདི་བྱོས་ཤིག་ཅེས་བསྒོ་བ་དེ་ཁྱོད་ཀྱིས་བྱ་དགོས་ཏེ། ད་ཁྱོད་ངལ་བརྙས་པའི་ལས་མ་བྱེད་ཅིག །བྱས་ན་ཁྱོད་ཀྱི་མི་བདེ་བ་མ་སྤངས་པར། ཚེའི་དུས་བྱས་ནས་དམྱལ་བར་ལྟུང་བར་གྱུར་ཏ་རེ། དེ་ནས་འདི་སྐད་བརྗོད་དོ། །
དེང་ཁྱོད་དེ་བཞིན་གཤེགས་པ་ཐམས་ཅད་ཀྱི་ཡེ་ཤེས་ཀྱང་འགྲུབ་པར་དཀའ་བ་མ་ལགས་ན། དངོས་གྲུབ་གཞན་ལྟ་སྨོས་ཀྱང་ཅི་དགོས་ཞེས་པས་ཡོལ་བའི་ནང་དུ་དྲངས་ཏེ། ཡེ་ཤེས་དབབ་པ་བླ་མའི་མན་ངག་བཞིན་བྱའོ། །དེ་ནས་མཐོང་བ་དྲི་བ་དང་མེ་ཏོག་གི་ཕྲེང་བ་ཐོགས་
ཏེ། པྲ་ཏཱིཙྪ་བཛྲ་ཧོ། །ཞེས་བརྗོད་ཅིང་ཕྲེང་བ་དོར་བ་དང་། མིག་དབྱེ་བར་བྱ་སྟེ། ཨོཾ་རྡོ་རྗེ་སེམས་དཔའ་དེང་ཁྱོད་ཀྱི། །མིག་འབྱེད་པ་ནི་བརྩོན་པ་སྟེ། །མིག་ནི་ཐམས་ཅད་འབྱེད་མཛད་པས། །རྡོ་རྗེའི་མིག་ནི་བླ་ན་མེད། །ཧེ་བཛྲ་པ་ཤྱ། དེ་ཉིད་སྔགས་སུ་བརྗོད་པ་ནི། །ཨོཾ་བཛྲ་
སཏྭ་སྭ་ཡཾ་དྱ་ཡ། ཙཀྵུ་ངྒྷཊྚ་ཡ་ཏཏྤ་ར། ཨུདྒྷ་ཊ་ལ་ཏི་སརྦཱཀྵོ། བཛྲ་ཙཀྵུ་རཱ་ནུཏྟ་ར། ཧེ་བཛྲ་པྲེ་བེ་ཤ །དེ་ནས་གདོང་གཡོགས་ཕུད་ཅིང་འདི་སྐད་བརྗོད་དོ། །ད་ནི་དད་པ་རབ་ཏུ་བསྐྱེད། །དཀྱིལ་འཁོར་བཟང་པོ་འདི་ལ་ལྟོས། །ཁྱོད་ནི་རྣལ་འབྱོར་མ་རིག་སྐྱེས། །སྔགས་དང་ཕྱག་རྒྱས་བྱིན་
གྱིས་བརླབས། །འགྲོ་བའི་ཕུན་སུམ་ཚོགས་ཀུན་དང་། །དངོས་གྲུབ་ཐམས་ཅད་མངོན་དུ་གྱུར། །དམ་ཚིག་ཐམས་ཅད་བསྐྱེད་པ་དང་། །གསང་སྔགས་ལ་ཡང་བརྩོན་པར་གྱིས། །ཞེས་བརྗོད་པ་དང་། མེ་ཏོག་གང་ལ་ལྟུང་བའི་ལྷ་ངོ་བསྟན་ཅིང་ལྷག་པར་དབོག །སྔགས་བསླབ་ཅིང་
དེ་ཁོ་ན་ཉིད་བཤད་པར་བྱའོ། །དེ་ལ་གཞལ་ཡས་ཁང་གྲུ་བཞིར་བྱུང་བ་ནི་ཚད་མེད་པའི་རང་བཞིན་གྱིས་རྣམ་པར་དག་པའོ། །སྒོ་བཞི་ནི་དྲན་པ་ཉེ་བར་གཞག་པ་བཞི་རྣམ་པར་དག་པའི་རང་བཞིན་ནོ། །རྟ་བབས་བཞི་ནི་ཡང་དག་པར་སྤོང་བ་བཞི་རྣམ་པར་དག་པའི་རང་བཞིན་ནོ། །
དྲ་བ་ནི་ཉོན་མོངས་པ་ཆེན་པོ་རྣམ་པར་དག་པའོ། །དྲ་བ་ཕྱེད་པ་ནི་ཉོན་མོངས་པ་ཆུང་ངུ་རྣམ་པར་དག་པའོ། །ཀ་བ་བརྒྱད་ནི། འཕགས་པའི་ལམ་ཡན་ལག་བརྒྱད་རྣམ་པར་དག་པའོ། །དར་གྱི་པ་ཏི་ལ་སོགས་པའི་རྒྱན་རྣམས་ནི་ཚུལ་ཁྲིམས་རྣམ་པར་དག་པའི་མཚན་ཉིད་དོ། །
