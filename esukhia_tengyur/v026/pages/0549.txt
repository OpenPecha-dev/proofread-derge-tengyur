ཅད་ཀྱིས། །རབ་ཏུ་མྱོང་བར་དབང་བསྐུར་འགྱུར། །རྣལ་འབྱོར་པ་ཡི་བདེ་བ་ནི། །ཐབས་དང་ཤེས་རབ་གཉིས་སུ་མེད་པའི་དོན་རྟགས་གཉིས་ཀྱི་བདེ་བ་དང་ལྡན་པ་སྟེ། དེས་ནི་དྲི་མའི་ཚོགས་ཐམས་ཅད་འཁྲུད་པར་བྱེད་ཅིང་། བྷ་ག་དང་ལིངྒ་མཚན་མ་ལས་ངེས་
པར་བྱེད་པས་རིན་པོ་ཆེའི་ཚོགས་དང་ལྡན་ཏེ། སྤྱི་བོ་དང་ལྕེ་སྟེང་དང་སྙིང་ག་ལ་སོགས་པའི་དབང་པོ་དང་ཉམས་སུ་མྱོང་བའི་སྒོ་ནས་བདེ་བའི་ཡེ་ཤེས་ཀྱིས་རྒྱས་ཤིང་མྱོང་བའི་ཚུལ་དུ་དབང་བསྐུར་བར་བྱའོ། །དེ་ཡང་རྩ་བའི་རྒྱུད་ལས། ཟག་པ་མེད་པའི་དགའ་བདེ་ཡི། །རོ་མྱང་
བདེ་ཆེན་གནས་ལ་བཞུགས། །དེ་རྟགས་དབང་བསྐུར་ལས་བྱུང་བ། །ཞུ་བདེ་གཉིས་ཀྱི་ཆུ་བོའི་རྒྱུན། །རིན་ཆེན་ཚོགས་ཀུན་གྱིས་བརྒྱན་པས། །རང་གི་ཡེ་ཤེས་གནས་ལ་སྦྱོར། །དབང་བསྐུར་གསང་བ་བླ་ན་མེད། །གནས་ཀུན་དགའ་བྱེད་འབྲས་ཚོགས་བསྐྱེད། །ཅེས་
ཏེ། འདིས་ནི་གཟུགས་བརྙན་གྱི་དཀྱིལ་འཁོར་དང་། ཚོགས་ཀྱི་འཁོར་ལོས་སློབ་མ་དབང་བསྐུར་བའི་བྱེ་བྲག་ཏུ་ཡང་ཤེས་པར་བྱའོ། །སྦྱོར་བ་བཞི་དང་ལྡན་པའི་རྣལ་འབྱོར་པས་ལྷ་མོ་ལ་སོགས་པའི་ལོངས་སྤྱོད་རྣམས་ལ་སྤྱད་པའི་ཕྱིར། ལྷ་མོ་ཀུན་གྱི་ལོངས་སྤྱོད་པ། །ཅི་
བདེ་བར་ནི་བསྟེན་བྱ་བ། །གང་ཟག་བཞིའི་སྒོ་ནས་འོད་ཀྱི་རིག་མ་ལ་སོགས་པའི་ཕྱག་རྒྱ་མོ་དང་ལོངས་སྤྱོད་ཀྱི་ཚོགས་ལ་དུས་དང་རྟོགས་པའི་བྱེ་བྲག་དང་ཁམས་དང་གྲོགས་ལ་སོགས་པའི་སྒོ་ནས་ཅི་བདེ་བར་བསྟེན་པར་བྱའོ། །དེ་ཡང་རྩ་བའི་རྒྱུད་ལས། རྣལ་འབྱོར་ལོངས་
སྤྱོད་ཐམས་ཅད་ལ། །དུས་ལ་སོགས་པའི་བྱེ་བྲག་གིས། །བདག་གཞན་དོན་དུ་འགྱུར་བའི་ལས། །ཅི་བདེ་བར་ནི་བསྟེན་པར་བྱ། །ཞེས་སོ། །དེ་ཡང་སྦྱོར་བ་བཞིའི་རྣམ་པ་གསལ་བས་ཉམས་སུ་བླངས་ཏེ། དང་པོའི་སྦྱོར་བ་ནི། བདག་ཉིད་ཆེ་བའི་ལྷར་སྦྱོར་བས། །
བདག་ལ་བདག་གིས་རབ་ཏུ་མཆོད། །མངོན་པར་བྱང་ཆུབ་པ་རྣམ་པ་ལྔས་རྒྱུའི་རྡོ་རྗེ་འཆང་ཆེན་པོའི་སྐུར་གྱུར་པ་ནི་ཆེ་བའི་ལྷ་སྟེ་སངས་རྒྱས་ཀུན་གྱི་བདག་ཉིད་དོ། །དེ་ཉིད་སྐུ་གསུང་ཐུགས་ཀྱི་ཕྱག་རྒྱར་སྦྱོར་ཞིང་། རྒྱན་ཆ་དང་ཆ་ལུགས་དང་དབང་བསྐུར་ཞིང་བྱིན་གྱིས་
