༄༅༅། །རྒྱ་གར་སྐད་དུ། ཤིཀྵ་ས་མུཙྪ་ཡ་ཀཱ་རི་ཀཱ། བོད་སྐད་དུ། བསླབ་པ་ཀུན་ལས་བཏུས་པའི་ཚིག་ལེའུར་བྱས་པ། སངས་རྒྱས་དང་བྱང་ཆུབ་སེམས་དཔའ་ཐམས་ཅད་ལ་
ཕྱག་འཚལ་ལོ། །འདི་ལྟར་བདག་དང་གཞན་དག་ཀྱང་། །འཇིགས་དང་སྡུག་བསྔལ་མི་འདོད་ན། །བདག་དང་བྱེ་བྲག་ཅི་ཡོད་ན། །གང་དེ་བསྲུང་ལ་གཞན་མི་བསྲུང་། །སྡུག་བསྔལ་མཐར་འབྱིན་
འདོད་པ་དང་། །བདེ་བའི་མཐར་ཡང་འགྲོ་འདོད་པས། །དད་པའི་རྩ་བ་བརྟན་བྱ་སྟེ། །བྱང་ཆུབ་ལ་ཡང་བློ་བརྟན་བྱ། །བྱང་ཆུབ་སེམས་དཔའི་སྡོམ་པ་ནི། །རྒྱས་པར་ཐེག་པ་ཆེ་ལས་འབྱུང་། །གང་
གིས་ལྟུང་བར་མི་འགྱུར་བའི། །གནད་ཀྱི་གནས་རྣམས་འདིར་རིག་བྱ། །བདག་གི་ལུས་དང་ལོངས་སྤྱོད་དང་། །དགེ་བ་དུས་གསུམ་སྐྱེས་པ་རྣམས། །སེམས་ཅན་ཀུན་ལ་བཏང་བ་དང་། །དེ་བསྲུང་
དག་པ་སྤེལ་བའོ། །སེམས་ཅན་རྣམས་ཀྱིས་ལོངས་སྤྱད་ཕྱིར། །ལུས་ལ་སོགས་པས་བསྲུང་བར་བྱ། །མ་བསྲུངས་ལོངས་སྤྱོད་གང་ལས་འོང་། །གང་མི་སྤྱོད་པ་ཅི་ཞིག་སྦྱིན། །དེ་བས་སེམས་
