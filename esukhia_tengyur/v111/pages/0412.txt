བ་སྐྱེ་བོ་གཞན་རྣམས་ཀྱིས། །བདེ་གཤེགས་ཉིད་ཐོབ་བྱ་ཕྱིར་ཡོངས་སུ་སྔོ་བྱེད་ན། །དེ་ཡི་བྱམས་པ་འདི་ནི་ཤིན་ཏུ་མཆོག་འགྱུར་ཞིང་། །རང་གི་བྱང་ཆུབ་སེམས་ནི་རྡོ་རྗེ་ལྟར་བརྟན་བྱེད། །གཞན་ཡང་བདེ་བར་གཤེགས་པ་རྣམས་ཀྱི་དགེ་བ་དང་། །དེ་བཞིན་རྒྱལ་སྲས་དང་ནི་
སེམས་ཅན་གཞན་གྱི་ཡང་། །དེ་དག་མ་ལུས་པ་ཡང་འགྲོ་བ་སྲིད་པའི་བཙོན་ར་ཡི། །བཅིངས་པ་གྲོལ་བར་བྱ་ཕྱིར་ཡོངས་སུ་སྔོ་བྱེད་པ། །སྣ་ཚོགས་སྙིང་རྗེའི་སྙིང་པོ་ཅན། །ཡོངས་སུ་བསྔོ་བ་གང་དག་ཡིན། །དེ་ཀུན་བྱམས་པ་ཉིད་ཀྱི་ནི། །རྗེས་སུ་འཇུག་པར་ཐེ་ཚོམ་མེད། །དེ་
དག་ནི་ཞེ་སྡང་ལ་ཀུན་དུ་སྤྱོད་པའི་གཉེན་པོའོ།། །།གཞན་ཡོད་པ་ལ་བརྟེན་ནས་ནི། །མེ་ལོང་གི་ནི་གཟུགས་བརྙན་བཞིན། །གང་གི་དངོས་པོ་སྣང་གྱུར་པ། །དེ་ལ་རང་དངོས་ག་ལ་ཡོད། །དེ་ཕྱིར་རྐྱེན་ལས་བྱུང་བ་ཡི། །དངོས་པོ་རྣམས་ནི་དངོས་མེད་ཉིད། །རང་གི་
