རཱེ་བ་ལིཿ།། ༣༩ །།ཨོཾ་བྷ་ག་ཝ་ཏི་ཨུཥྞཱི་ཥ་པི་ཛ་ཡེ་མ་མ་མྲྀ་ཏྱུཾ་ནཱ་ཤ་ཡ་ནཱ་ཤ་ཡ། བི་ནཱ་ཤ་ཡ་བི་ནཱ་ཤ་ཡ། བི་དྷྲནྶ་ཡ་བི་དྷྭནྶ་ཡ། ཨི་དཾ་བ་ལིཾ་གྲྀཧྞ་གྲྀཧྞ་དཱིརྒྷཱ་ཡུཥྐ་མེ་ཀུ་རུ་སྭསྟི་སྭཱ་ཧཱ། ཨུཥྞཱྀ་ཥ་བི་ཛ་ཡཱ་བ་ལིཿ།། ༤༠ །།ཨོཾ་སིདྡྷཻ་ཀ་བཱི་རཱ་ཡ་བཱི་རེ་ཤཱ་ཡ། ཀ་རུ་ཎ་པེ་ཤ་
གྱི་གཏོར་མའོ།། །།བཅོམ་ལྡན་འདས་མ་གཙུག་ཏོར་རྣམ་པར་རྒྱལ་མ་བདག་གི་འཆི་བདག་ཉམས་པར་གྱིས་ཉམས་པར་གྱིས། རྣམ་པར་ཉམས་པར་གྱིས་རྣམ་པར་ཉམས་པར་གྱིས། རྣམ་པར་ཆོམས་རྣམ་པར་ཆོམས། བདག་གི་ཚེ་སྲིངས་ཤིག །བདེ་ལེགས་གཞི་ཚུགས། གཙུག་ཏོར་རྣམ་པར་རྒྱལ་མའི་གཏོར་མའོ།། །།དཔའ་བོ་གཅིག་ཏུ་གྲུབ་པ་ལ་དཔའ་བོའི་དབང་པོ་ལ་
ལ་ཧྲི་ད་ཡཱ་ཡ། ནཱི་ལོཏྤ་ལ་ཧསྟཱ་ཡ། ཨི་དཾ་བ་ལིཾ་གྲྀ་ཧཱི་ཏྭོ་པ་བྷུ་ཛྱ་ཤཱནྟིཾ་ཀུ་རུ་པུཥྚིཾ་ཀུ་རུ་སརྦ་སིདྡྷིམྐུ་རུ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། སིདྡྷཻ་ཀ་བཱི་ར་བ་ལིཿ།། ༤༡ །།ཨོཾ་ཡ་མཱནྟ་ཀཱ་ཡ་ཡ་མ་བྷཱི་ཥ་ཎཱ་ཡ་ཏྲཱིཿཥྚྲཱྀཿབི་ཀྲྀ་ཏཱ་ན་ནཱ་ཡ་ནཱ་ག་བྷ་ར་ཎཱ་ཡ། ཏྲཻ་ལོ་ཀྱ་བི་མརྡ་ན་དཎྜ་ཧསྟཱ་ཡ་སརྦ་མཱ་ར་བ་ལ་བི་ནཱ་ཤ་ནཱ་
སྙིང་རྗེ་དང་དེས་པའི་ཐུགས་ཅན་མ་ཨུཏྤལ་སྔོན་པོ་ཕྱག་ན་བསྣམས་པ་ལ་གཟུང་སྟེ་ལོངས་སྤྱད་དེ། དཔའ་བོ་གཅིག་གྲུབ་ཀྱི་གཏོར་མའོ།། །།གཤིན་རྗེ་མཐར་བྱེད་གཤིན་རྗེ་འཇིགས་པར་བྱེད་པ་རྣམ་པར་འགྱུར་བའི་ཞལ་ཅན་ཀླུའི་རྒྱན་ཅན་འཇིག་རྟེན་གསུམ་རྣམ་པར་འཇོམས་པའི་དབྱུག་པ་ཕྱག་ན་
ཡ་ཨི་མཾ་བ་ལིཾ་གྲྀཧྞ་གྲྀཧྞ་མ་མ་བིགྷྣཱན྄། ཧ་ར་ཧ་ར། ཙ་ཏུརྨཱ་རཱན། ནི་བཱ་ར་ཡ་ནི་བཱ་ར་ཡ་ཧཱུཾ་ཧཱུཾ་ཕཊ྄་ཕཊ྄་སྭཱ་ཧཱ། ཡ་མཱནྟ་ཀ་བ་ལིཿ།། ༤༢ །།ཨོཾ་བིགྷྣཱནྟ་ཀ་མ་ཧཱ་ཀྲོ་དྷ་མ་ཧཱ་བ་ལ་པ་རཱ་ཀྲ་མ་ཨི་མཾ་པ་ལིཾ་གྲྀཧྞ་གྲྀཧྞ་ཁ་ཁ་ཁཱ་ཧི་ཁཱ་ཧི། མ་མ་སརྦ་བིགྷྣཱནྲ། ཙྪེ་ད་ཙྪེ་ད། བྷེ་ད་བྷེ་ད། དྲུ་ཏཾ་ད་ཧ་ད་ཧ། པ་ཙ་པ་ཙ།
བསྣམས་པ་བདུད་ཀྱི་སྟོབས་ཐམས་ཅད་རྣམ་པར་འཇོམས་པ། གཤིན་རྗེ་གཤེད་ཀྱི་གཏོར་མའོ།། །།བགེགས་མཐར་བྱེད་ཁྲོ་བོ་ཆེན་པོ་སྟོབས་པོ་ཆེ་ཕ་རོལ་གནོན་པ་བདག་གི་བགེགས་ཐམས་ཅད་
