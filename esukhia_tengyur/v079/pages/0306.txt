འོང་སྟེར་བར་འགྱུར། །འདོད་ཆགས་བྲལ་བ་སྟོང་ཕྲག་ལ། །ཉིན་རེ་བཟའ་བ་མཆོག་ཕུལ་ལས། །བྱུང་བའི་བསོད་ནམས་གང་ཡིན་དེ། །ཉ་རྣམས་ཚིམ་པར་བྱས་ལས་འབྱུང་། །རང་རྒྱལ་གྱི་ནི་དགེ་འདུན་ལ། །བསོད་སྙོམས་ཕུལ་ལས་དགེ་གང་འབྱུང་། །ཉ་རྣམས་ཚིམ་པར་བྱས་པ་ལས། །བསོད་ནམས་དེ་ནི་ཐོབ་པར་འགྱུར། །དེ་ནི་ས་འོག་ན་གནས་
པའི་གཟུགས་ཅན་གྱི་ཉའི་གཏོར་མའི་ཆོ་གའོ། །གང་གི་སྤྱོད་ལམ་རྣམ་བཞི་རུ། །རྣལ་འབྱོར་པ་ཡིས་འགྲུབ་འགྱུར་བ། །ཕྱོགས་སྤྱོད་གཏོར་མ་བསྒྲུབ་པ་ནི། །དེ་ནས་གཞན་ཡང་རབ་བཤད་བྱ། །རབ་ཏུ་གྲགས་པའི་བཞི་མདོ་རུ། །རྒྱལ་པོའི་སྐྱེད་མོས་ཚལ་བརྟེན་ནས། །རྡུལ་ཚོན་ཁ་དོག་ལྔ་པ་ཡིས། །དེ་རུ་དཀྱིལ་འཁོར་བྲི་
བར་བྱ། །པདྨ་འདབ་བརྒྱད་ལྟེ་བར་བཅས། །པདྨ་སྣ་ཚོགས་དང་ལྡན་པ། །རྣལ་འབྱོར་པ་ཡིས་དེར་བྲི་བྱ། །དབུས་སུ་ཐོད་པ་བྲི་བར་བྱ། །སྡིགས་མཛུབ་རྡོ་རྗེ་འགྱིང་བཅས་པས། །ཤིན་ཏུ་མི་བཟད་འཇིགས་མཛད་པ། །སྤྲིན་ནི་སྔོན་པོའི་མདོག་དང་ལྡན། །འབར་བ་སྔོན་པོ་རབ་ཏུ་ཁྱབ། །གཽ་རཱི་གཽ་རཱིའི་བདག་པོ་ཡིས། །སྙིང་གར་གཡས་
བརྐྱང་ཞབས་ཀྱིས་མནན། །ལྟེ་བའི་དབུས་སུ་བཅོམ་ལྡན་འདས། །རྡོ་རྗེ་ཧཱུཾ་མཛད་བསྒོམ་པར་བྱ། །ཤར་འདབ་ཚངས་པའི་ལྷ་མོ་བསམ། །ཁྱབ་འཇུག་གི་ནི་ལྷོ་ན་གནས། །གཞོན་ནུ་ཡི་ནི་ནུབ་ཏུ་བསྒོམ། །གཤིན་རྗེ་བྱང་གི་ཕྱོགས་ན་གནས། །སྲིན་པོའི་ཕྱོགས་སུ་ལུས་ངན་པོ། །མེ་ཡི་ཕྱོགས་སུ་ཙ་མུཎྜཱི། །དབང་ལྡན་ཕྱོགས་སུ་དབང་ལྡན་
མ། །རླུང་གི་ལྷ་མོ་རང་ཕྱོགས་གནས། །ཞལ་གཅིག་སྤྱན་ནི་གསུམ་པ་སྟེ། །ཕྱག་གཉིས་གར་ལ་རབ་བརྩོན་པ། །གཡས་པ་ཡི་ནི་ཕྱག་རྣམས་སུ། །ཕྲེང་བ་དེ་བཞིན་འཁོར་ལོ་དང་། །རྩེ་གསུམ་དབྱུག་པ་དབྱུག་ཏོ་དང་། །རལ་གྲི་མདུང་དང་གོས་ཉིད་དེ། །ཕྱག་མཚན་འདི་རྣམས་བསམ་པར་བྱ། །སྤྱི་བླུགས་དང་ནི་པདྨ་དང་། །ཕྲེང་བ་དང་
ནི་དེ་བཞིན་མགོ། །སེང་གེའི་རྒྱལ་མཚན་ཐོད་པ་དང་། །ཁ་ཊྭཱཾ་ག་དང་ངོ་མཚར་གཞན། །གཡོན་པ་རྣམས་སུ་བསམ་པར་བྱ། །དམར་པོ་ལྗང་གུ་སེར་དང་མཚུངས། །སྔོན་པོ་དམར་པོ་དམར་དཀར་འོད། །ཨིནྡྲ་ནཱི་ལའི་འོད་ལྡན་པ། །མི་མོ་མཆོག་ནི་འདི་རྣམས་བསམ། །སྔགས་དང་དམ་ཚིག་ཕྱག་རྒྱ་ཡིས། །དེ་རུ་སྤྱན་ནི་དྲང་བར་བྱ། །དེ་
ནས་གཏོར་མ་རྒྱ་ཆེན་པོ། །སྔགས་པས་སྦྱིན་བྱ་ལུང་རྗེ་བཞིན། །ཨོཾ་ན་མོ་བྷ་ག་ཝ་ཏེ་བཛྲ་ཧཱུཾ་ཀ་ར། ཧྲཱིཿཥྚྲྀ་སརྦ་བྷཱུ་ཏ་ནི་སྟཾ་བྷ་ཡ་སྭཱ་ཧཱ། ཨོཾ་བྲཧྨ་ཎི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཻཥྞ་བཱི་ཡེ་ཧྲཱིཿཕཊ། ཨོཾ་ཀཽ་མཱ་རཱི་ཡེ་ཧྲཱིཿཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མཱ་ནཱི་ཡེ་ཧྲཱིཿཧཱུཾ་ཕཊ། ཨོཾ་ཀུ་བཻ་རཱ་ཡེ་ཧྲཱིཿཧཱུཾ་ཕཊ། ཨོཾ་ཙཱ་མུཎྜཱ་ཡེ་ཧྲཱིཿཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཥ་བྷྱོ་ཧྲཱིཿཧཱུཾ་ཕཊ། ཨོཾ་བ་ཡ་བྱཻ་ཧྲཱིཿཧཱུཾ་ཕཊ།
