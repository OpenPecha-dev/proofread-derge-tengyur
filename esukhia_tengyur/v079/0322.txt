བ་གང་བསགས་པ། །དེ་ཡིས་དེང་སང་སྐྱེ་བོ་མང་པོ་བདེ་གཤེགས་ཉིད་ནི་རབ་ཐོབ་ཤོག །བདག་ཀྱང་སྲིད་པ་འདི་རུ་ནི། །རྡོ་རྗེ་སེམས་དཔར་འགྱུར་བར་ཤོག །མ་ལུས་བདེ་གཤེགས་བསམ་བྱ་བ། །ལྷ་ཆེན་སོགས་ཀྱིས་བསམ་མི་ཁྱབ། །རྟོག་པའི་རླབས་ནི་འཇོམས་བྱེད་བློ་མཆོག་སྐལ་ལྡན་སྐྱེ་བོ་ཡི། །སྙིང་ལ་རྟག་ཏུ་
ཤིན་ཏུ་བརྟན་པར་གནས་པར་བྱེད་པར་ཤོག །རྒྱུད་སྣ་ཚོགས་ལས་བཏུས་པའི་གཏོར་མའི་ཆོ་ག་སློབ་དཔོན་དཔལ་ཀླུ་སྒྲུབ་ཀྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།། །།རྒྱ་གར་གྱི་མཁན་པོ་རྡོ་རྗེ་སློབ་དཔོན་ཆེན་པོ་ཀརྨ་བཛྲའི་ཞལ་སྔ་ནས་དང་། ལོ་ཙཱ་བ་དགེ་སློང་གཞོན་ནུ་ཚུལ་ཁྲིམས་ཀྱིས་བསྒྱུར་ཅིང་ཞུས་པའོ།། །།

དབེན་ཞིང་གཙང་བའི་ས་ཕྱོགས་སུ། །དབང་ཆེན་དཀྱིལ་འཁོར་རྣམ་ལྔས་བྱུག །དེ་སྟེང་པད་སྣོད་མཛེས་པའི་ནང་། །སྔར་བསགས་རྫས་ཀྱིས་ཡོངས་སུ་དགང་། །བྱང་ཆུབ་མཆོག་ཏུ་སེམས་བསྐྱེད་ལ། །འོག་ཏུ་གཡོ་བའི་དཀྱིལ་འཁོར་བསྒོམ། །དེ་སྟེངས་འབར་བ་གྲུ་གསུམ་བསམ། །འབྲུ་གསུམ་སྟེང་དུ་པད་སྣོད་
བསམ། །རླུང་བསྐྱོད་མེ་འབར་རྫས་རྣམས་ཞུ། །ཐོག་མར་ཉི་མ་ཤར་ལྟ་བུ། །དེ་ནས་སྦྱང་དང་སྦར་རྟོགས་ནི། །ཟླ་བ་རྡོ་རྗེ་ཨོཾ་གྱིས་བསྒྲུབ། །ཡེ་ཤེས་བདུད་རྩིས་སངས་རྒྱས་སོགས། །འབད་པས་དེ་ཉིད་ཐིམ་པར་བསམ། །སྐུ་གསུང་ཐུགས་ཀྱིས་བྱིན་བརླབ་བྱ། །རང་སྙིང་ས་བོན་འོད་ཟེར་གྱིས། །ཕྱོགས་
བཅུའི་སངས་རྒྱས་ཞིང་རྣམས་ན། །འཇིག་རྟེན་འཇིག་རྟེན་འདས་པ་ཡི། །ཆོས་སྐྱོང་སྲུང་མ་ཐམས་ཅད་རྣམས། །མདུན་གྱི་ནམ་མཁར་སྤྱན་དྲངས་ལ། །ཐོག་མར་ཞལ་བསིལ་བསང་གཏོར་བྱ། །དེ་ནས་མེ་ཏོག་སོགས་ལྔ་མཆོད། །རྗེས་སུ་གཏོར་མ་དབུལ་བ་ནི། །རྡོ་རྗེ་ཐལ་མོའི་སྟེང་བཞག་
