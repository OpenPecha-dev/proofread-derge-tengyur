རྒྱས་ཀྱི་སྒྲ་ནི་བདེ་བར་གཤེགས་པ་ངེས་ཚིག་དང་བཅས་པ་ལ་བརྗོད་དེ། འདི་ལྟར། འདི་སྐད་བདག་གིས་ཐོས་པ་དུས་གཅིག་ན། བཅོམ་ལྡན་འདས་མཉན་ཡོད་ན་རྒྱལ་བྱེད་ཀྱི་ཚལ་མགོན་མེད་ཟས་སྦྱིན་གྱི་ཀུན་དགའ་
ར་བ་ན་བཞུགས་ཏེ། དེ་ནས་བྲམ་ཟེ་གཞན་ཞིག་བཅོམ་ལྡན་འདས་ག་ལ་བ་དེར་ཉེ་བར་སོང་སྟེ། ཉེ་བར་སོང་ནས་བཅོམ་ལྡན་འདས་དང་ཐབས་ཅིག་ཡང་དག་པར་དགའ་བ་དང་། ཡང་དག་པར་མགུ་བ་དང་། རབ་ཏུ་དགའ་བར་འགྱུར་
བའི་གཏམ་རྣམ་པ་སྣ་ཚོགས་བྱས་ཏེ་ཕྱོགས་གཅིག་ཏུ་འདུག་གོ། །ཕྱོགས་གཅིག་ཏུ་འདུག་ནས་བྲམ་ཟེ་དེས་བཅོམ་ལྡན་འདས་ལ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །གཽ་ཏ་མ་སངས་རྒྱས་སངས་རྒྱས་ཞེས་བརྗོད་པ་ཅི་ཁྱོད་ཀྱི་མིང་ཕ་དང་མས་བྱས་
སམ། ངེས་ཚིག་གང་གིས་ན་སངས་རྒྱས་ཞེས་བརྗོད། །དེ་ནས་བྲམ་ཟེ་དེས་དེའི་ཚེ་ཚིགས་སུ་བཅད་པ་འདི་སྐད་ཅེས་གསོལ་ཏོ། །འཇིག་རྟེན་རྣམ་པར་འདྲེན་པ་ལས། །མིང་མཆོག་སངས་རྒྱས་ཞེས་འབྱུང་བ། །ཕ་དང་མ་ཡིས་བྱས་སམ་
ཅི། །ངེས་ཚིག་གང་གིས་སངས་རྒྱས་བརྗོད། །བྲམ་ཟེ་ལ་ནི་རྗེས་བརྩེ་བས། །དེ་ལ་སྟོན་པས་ལུང་བསྟན་པ། །བྲམ་ཟེ་ཇི་ལྟར་སངས་རྒྱས་ཀྱི། །མཚན་ཉིད་བརྗོད་པ་རྙེད་པ་ནི། །སངས་རྒྱས་འདུ་བྱེད་འཇིག་པ་ཡི། །ཆོས་ཅན་འདས་པའི་
