སྱ་དརྤྤ་པྲཱི་ཏི་ར་ཛཱ་ཡ་ཏ། །ཨ་ཐ་བུདྡྷཱ་བྷི་དྷོ་བྷྱེ་ཏྱ་ཤྲ་མ་ཎཱརྷ་ཏྭཱ་མ་ག་ཏཿ། །རྀདྡྷྱཱ་ཙཀྲེ་པ་རཱ་ཛི་ཏྱཾ་ཏཾ་པྲ་པཱ་ད་མ་ཡོ་ད་ཡཾ། །ཛ་གཏྐཱཪྻ་ཏ་ཏཿཀྲྀ་ཏྭཱ་ཀཱ་ལེ་ན་བྷ་ག་བཱཉྫི་ནཿ། །ཀལྤ་ཡ་ན་པ་རི་ནིརྦྦཱ་ཎཾ་པཤྩི་མེ་ཤ་
དྲེགས་དང་དགའ་བ་རབ་ཏུ་སྐྱེས། །དེ་ནས་དགེ་སློང་དགྲ་བཅོམ་ཐོབ། །སངས་རྒྱས་ཞེས་བྱ་མངོན་ཕྱོགས་ཏེ། །དེ་ནི་རྫུ་འཕྲུལ་གྱིས་བཏུལ་ནས། །རབ་ཞི་རང་བཞིན་རྒྱས་པར་བྱས། །དེ་ནས་དུས་ཀྱིས་བཅོམ་ལྡན་འདས། །རྒྱལ་བས་འགྲོ་བའི་དོན་མཛད་ནས། །ཡོངས་སུ་མྱང་འདས་ལ་རྟོག་ཅིང་། །ཐ་མའི་གཟིམས་
ཡ་ནེསྠི་ཏ། །པྲ་ད་དྷྱཽ་བི་ནེ་ཡེ་ཏཱ་པཱདྦི་ན་ཡཱ་བི་བེ་ཤི་ཏཱཿ། །ཨ་བ་ཤིཥྚཿསུ་བྷ་དྲོ་མེ་གནྡྷརྦྷ་ས་ཙ་སུ་པྲི་ཡཿ། །ཨ་ཀླེ་ཤེ་ན་སུ་བྷ་དྲསྟུ་བི་ན་ཡཾ་པྲ་ཏི་པ་དྱ་ཏེ། །ཀྲྀཙྪཱ་ནུ་ཤཱ་སྱོ་གནྡྷརྦྷསྟཱ་རུ་ཎྱ་བི་བྷ་བོནྨ་དཿ། །ཨི་
མལ་དག་ལ་གནས། །རེ་ཞིག་གདུལ་བྱ་འདུལ་བ་ལ། །རྣམ་པར་བཀོད་བྱེད་རབ་ཏུ་བསམས། །རབ་བཟང་དང་ནི་དྲི་ཟ་དག །རབ་དགའ་དེ་ཡང་བདག་གི་ལྷག །རབ་བཟང་ཚེགས་ནི་ཆུང་ངུ་ཡིས། །འདུལ་བ་རབ་ཏུ་ཐོབ་པར་འགྱུར། །དྲི་ཟ་ལང་ཚོ་འབྱོར་པ་ཡིས། །མྱོས་པ་ཚེགས་ཀྱིས་འདུལ་བར་འགྱུར། །ཞེས་པ་
ཏི་སཾ་ཙིཾ་ཏྱ་གནྡྷརྦྷ་པུ་རཾ་ག་ཏྭཱ་ཏ་ཐཱ་ག་ཏཿ། །གནྡྷརྦྷ་རཱ་ཛ་མ་པ་རཾ་ནིརྨྨི་མེ་ཏྱ་དྷི་ཀ་དྱུ་ཏཾ། །ས་ཧ་སྲ་ཏཾནྟྲི་བཻ་ཌཱུཪྱ་དཎྜཱཾ་བཱི་ཎཱཾ་ས་བཱ་ད་ཡན། །སུ་པྲི་ཡ་སྱ་མ་དཙྪེ་ད་སྤརྡྷ་ཡ་ཀརྟྟུ་མུདྻ་ཡཽ། །ཀྲ་མེ་ཎ་ཏནྟྲཱིཙྪེ་
བསམས་ནས་དེ་བཞིན་གཤེགས། །དྲི་ཟའི་གྲོང་ཁྱེར་དག་ཏུ་བྱོན། །དྲི་ཟའི་རྒྱལ་པོ་གཞན་པ་ནི། །མཛེས་པ་ལྷག་པར་མངོན་པར་སྤྲུལ། །བཻཌཱུརྱ་ཡི་དབྱུག་པ་ཅན། །རྒྱུད་མང་རྒྱུད་སྟོང་སྦྲེང་བྱེད་ཅིང་། །འགྲན་པས་རབ་དགའི་རྒྱགས་པ་ནི། །ཉམས་པར་བྱེད་ལ་བརྩོན་པར་གྱུར། །རིམ་གྱིས་རྒྱུད་ནི་བཅད་
