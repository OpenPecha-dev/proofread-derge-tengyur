དྟ་ནརྫཱ་ཏིརྡྡྷ་ན་བི་དྱཱ་དྷ་ནནྡྷརྨྨོ་དྷ་ནཾ་ཡ་ཤཿ། །ཀིནྡྷ་ནེ་ན་བི་ཧཱི་ནཱ་ནཱཾ་ཡཱཙྙཱ་བྷིརྫཱི་ཏཻརྒུ་ཎཻཿ། །པ་རོ་པཱ་ཀཱ་རཱ་ཎཱི་བྷཱུ་ཏཻཿགིནྡ་རི་དྲ་སྱ་སངྒུ་ཎཻཿ། །ཀླེ་ཤཱརྠཻརྦྷྲ་རི་ཀ་སྱེ་བ་མ་ཎི་ཀཱཉྩ་ན་བྷཱུ་ཥ་ཎཻཿ། །ད་རི་དྲོ་
ནོར་རིགས་ནོར་ནི་རིག་པ་སྟེ། །ནོར་ཆོས་ནོར་ནི་གྲགས་པ་ཡིན། །ནོར་གྱིས་དམན་ཞིང་བསླངས་པ་ཡིས། །འཚོ་བའི་ཡོན་ཏན་རྣམས་ཀྱིས་ཅི། །གཞན་གྱི་ཉེ་མཁོར་གྱུར་པ་ཡིས། །དབུལ་པོའི་ཡོན་ཏན་དམ་པས་ཅི། །ཁུར་དུབབ་ལ་ནོར་བུ་ཡིས། །གསེར་རྒྱན་ངལ་དུབ་དོན་གྱིས་བཞིན། །དབུལ་པོ་སྦྱིན་
དཱ་རི་དྱཱཾ་པྲ་བི་ཤ་ཏི་པུ་ནརྡཱ་ན་བི་ར་ཧཱད་ད་རི་དྲཿཔཱ་པཱཏྨ་བྷ་བ་ཏི་དྷ་ན་ལིཔྶཱ་པ་ར་བ་ཤཿ། །ད་ར་དྲཱ་སཙྪྭཱ་སེ་མྲྀ་ཏ་ཨི་ཏི་ན་ཙཱ་དྲསྟི་བི་མ་ཏིར་ད་རི་དྲ་སྱེ་བཻ་ཏཱ་ནི་ཛ་ཛ་ན་བི་ཧཱི་ནོ་ད་ཤ་དི་ཤཿ། །ཨ་བྷྱཙྩ་
པ་དག་དང་བྲལ་བས་སླར་ཡང་དབུལ་པོར་རབ་ཏུ་འཛུད། །དབུལ་པོ་ནོར་ལ་ཆགས་པས་གཞན་དབང་སྡིག་པའི་བདག་ཉིད་དག་ཏུ་འགྱུར། །དབུལ་པོ་དབུགས་དང་བཅས་ཀྱང་ཤི་ཞེས་འདི་ལ་ཁྱད་པར་བློ་གྲོས་མེད། །དབུལ་པོ་ཕྱོགས་བཅུ་འདི་དག་རང་གི་སྐྱེ་བོ་དག་གིས་དམན་པ་བཞིན། །དེ་སླད་དམན་
ཡཱ་བཿསུ་ག་ཏཾ་ཏ་སྨཱཏྐྲྀ་པ་ཎ་བཏྤ་ལཾ། །ཨ་ན་རཱ་དྷི་ཏ་བུདྡྷཱ་ནཱཾ་མུགྡྷཱ་ནཱཾ་ཀུ་ཤ་ལཾ་ཀུ་ཏཿ། །ཨེ་ཥ་པདྨ་ད་ལཱ་ཤཱཀྵཿཀྵི་པ་ཏྱཱ་པནྣཱ་བཱནྦྷ་བཿ། །ཡ་ཏྲ་ཡ་ཏྲ་དྲི་ཤཾ་ཛཱ་ནེ་ཏ་ཏྲ་ཏ་ཏྲཱ་ཤྲ་ཡ་ཤྲི་ཡཿ། །ཨི་ཏི་པ་
ལ་མཉེས་གཤིན་པའི། །བདེ་བར་གཤེགས་པ་མཆོད་པར་བྱ། །སངས་རྒྱས་མཉེས་པར་མ་བྱས་པའི། །རྨོངས་ལ་དགེ་བ་ག་ལ་ཡོད། །པདྨའི་འདབ་མ་ལྟ་བུའི་སྤྱན། །ཕོངས་པའི་གཉེན་འདིས་གང་གང་དུ། །སྤྱན་ནི་འཕངས་པ་དེ་དང་དེར། །དཔལ་ནི་གནས་པ་དག་གིས་ཤེས། །དེ་སྐད་བདག་པོའི་
