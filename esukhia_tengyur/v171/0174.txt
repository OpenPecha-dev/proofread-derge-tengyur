ད་ཡཾ་ཙིཏྟ་བི་སླ་བཿ། །ལཛྫ་བ་ཧཱ་བི་ཀཱ་ར་སྱ་ཀེ་ཡ་མ་ཏྱཏྟ་ཏཱ་ཏ་བ། །ཏྱ་ཛནྟི་དྷྲི་ཏི་མཪྻཱ་དཱཾ་བྷཱུ་དྷ་ར་ན་ཏུ་སཱ་དྷ་བཿ། །ཨི་ཏི་ཏ་སྱ་བ་ཙཿཤྲུ་ཏྭཱ་ཏཱ་མུ་བཱ་ཙ་བ་ས་སྨི་ཏཿ། །བྷཱ་དྲེ་ནཱསྟྱ་བ་མ་ན་སཿསྭཔྣེ་པི་མ་མ་བི་སླ་བཿ། །
གང་འདི་རྣམ་པར་བསླད་པ་སྐྱེས། །ཁྱོད་ཀྱི་རྣམ་འགྱུར་ངོ་ཚའི་རྒྱུན། །ཤིན་ཏུ་ཆེ་ལྡན་འདི་ནི་ཅི། །ས་འཛིན་དག་ཀྱང་བརྟན་པའི་ལུགས། །གཏོང་གི་དམ་པ་ཡོངས་མ་ཡིན། །ཞེས་པ་དེ་ཡི་ཚིག་ཐོས་ནས། །འཛུམ་ལྡན་དེ་ཡིས་དེ་ལ་སྨྲས། །རྨི་ལམ་ན་ཡང་བདག་གི་ཡིད། །བཟང་མོ་ངེས་པར་བསླད་པ་མེད། །

འོན་ཀྱང་འདིར་ནི་སྦྲུལ་ནག་འདི། །འཇིགས་སུ་རུང་བ་ཡོངས་སུ་རྒྱུ། །ཁྱོད་ཀྱི་ལག་པ་མལ་སྟན་ལས། །འཕྱང་བ་འཇིགས་ནས་བདག་གིས་བསྲུངས། །ཁྱོ་ཡི་ཚིག་འདི་ཐོས་གྱུར་ནས། །དོགས་བཏང་དགའ་བ་དེས་སྨྲས་པ། །བདེན་པའི་ངང་ཚུལ་ཁྱོད་ཀྱི་བློ། །འདོད་པས་དྲི་མར་བྱས་པ་མེད། །གདུག་པའི་

སྦྲུལ་ནི་ཁེགས་པ་སྟེ། །ཆགས་པའི་ལྟོ་འཕྱེས་མཆོག་ཏུ་འཇིགས། །སྦྲུལ་གྱིས་ལུས་གཅིག་འཇོམས་པ་སྟེ། །འདོད་པས་ལུས་ནི་བརྒྱ་མཐར་བྱེད། །དེ་སླད་རྣམ་འགྱུར་འདི་དག་བསྲུངས། །ཞེས་སྨྲས་དེ་ནི་ཆགས་དང་བྲལ། །དེ་ཡི་ཤིན་ཏུ་དུལ་བ་ཡང་། །པིསྤ་ལ་སྐྱེས་དག་གིས་བསྔགས། །དུས་ཀྱི་ནྱ་གྲོ་དྷ་
