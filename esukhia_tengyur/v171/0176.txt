སཱ་རཱཿསུ་ཁ་སམྤ་དཿ། །ཨཱ་སྭཱ་དྱ་མཱ་ནཱཿཀུརྦྦནྟི་ན་ཌ་ཤཱ་ཁཱ་ཨི་བ་བྱརྠཱི། །ཀླེ་ཤ་ཤཻ་བཱ་ལ་ཛཱ་ལེ་ཥུ་བཱ་པ་སཾ་ཀེ་ཥུ་བེཤྨིནཿ། །ཨ་བ་སནྣ་གྲྀ་ཧེ་ཥྭེ་བ་ནིམྣེ་ཥྭ་བ་ཛ་རདྒ་བཿ། །ཏསྨཱདྒྲྀ་ཧ་ས་མཱ་རཾ་བྷ་སྟྱ་ག་ཡཽ་གྱཱ་ཡརྨ་བ་ཡོཿ། །ཨི་ཏྱུཀྟཱ་
མཇུག་ཏུ་རོ་བྲལ་སྙིང་པོ་མེད། །མྱངས་ན་འདམ་བུའི་འདམ་མ་བཞིན། །གདུང་བ་དག་ནི་སྐྱེད་པར་བྱེད། །ཁྱིམ་པ་རྣམས་ནི་སྡིག་པའི་འདམ། །ཉོན་མོངས་ཉ་ལྕིབས་དྲ་བ་ཅན། །ཁྱིམ་ཉིད་དུ་ནི་ཉམས་འགྱུར་ཏེ། །རླུབས་སུ་གླང་ནི་རྒན་པོ་བཞིན། །དེ་སླད་ཁྱིམ་གྱི་རྩོལ་བ་འདི། །ཡུ་ཅག་གིས་ནི་གཏོང་བར་འོས། །ཞེས་སྨྲས་ཆུང་
ནུ་མ་ཏཿཔཏྣྱ་ཤཱནྟཽནིཤྩ་ཡ་མཱ་ད་དྷཻ། །ས་ཏྭཱ་དྷ་ན་མརྠི་བྷྱེ་གྲྀ་ཧཾ་ཙ་ས་པ་རིཙྪ་དཾ། །ནིཤྩ་ཀྲཱ་མ་ས་མུནྨུཀྟ་སརྦྦཱ་ཤཱ་པཱ་ཤ་ནནྡ་ནཿ། །སོ་ཐཱ་ཀཱ་ཤྱ་པ་གོ་ཏཱ་ཏྭཱ་ཏ་མ་ཧཱ་ཀཱ་ཤྱ་པ་ཏཱཾ་ག་ཏཿ། །ཏཏྐཱ་ལ་སམྱཀྶཾ་བུདྡྷ་ཀཱ་ཤྱ་པ་
མས་ཁས་བླངས་པས། །ཞི་བ་ལ་ནི་ངེས་པར་བྱས། །ཁྱིམ་དང་ཡོ་བྱད་བཅས་པ་ཡི། །ནོར་རྣམས་སློང་ལ་དེས་བྱིན་ནས། །རེ་བའི་ཞགས་པས་འཆིང་བ་ཀུན། །རྣམ་པར་གྲོལ་ཏེ་ངེས་པར་བྱུང་། །དེ་ནས་འོད་སྲུངས་རིགས་ཉིད་ཕྱིར། །འོད་སྲུངས་ཆེན་པོ་ཉིད་གྱུར་ཏེ། །དེ་དུས་ཡང་དག་རྫོགས་སངས་རྒྱས། །འོད་

སྲུངས་དྲུང་དུ་དེ་སོང་གྱུར། །མཆོད་རྟེན་བུ་མང་ཞེས་པ་ཡི། །རྩ་བར་གནས་པ་དེ་ཅན་ཕྱིན། །འདུལ་བའི་ཆོས་ནི་དེས་བྱིན་པས། །དག་པའི་བྱང་ཆུབ་ཐོབ་པར་གྱུར། །བཟང་མོ་ཡིས་ཀྱང་ཆགས་བྲལ་གྱི། །བསྟན་པའི་ལམ་གྱི་འདུལ་བ་ཐོབ། །སྔོན་གྱི་བསོད་ནམས་ཡོངས་སྨིན་པས། །དགེ་བ་དྲི་མ་མེད་པ་བསྟེན། །
