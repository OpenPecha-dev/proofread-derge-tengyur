ལ་ཏཱ་བྲྀ་ཏཿ། །ས་ད་དཪྴཱམྦུ་ཛ་རཛཿཔུཉྫཻཿསུ་ར་བྷི་པིཉྫ་རཾ། །ཧེ་མ་ཀུཾ་བྷཻརྫ་ལནྟ་ཏྲ་ན་ཡནྟཱིཿཀིནྣ་རཱཾ་ག་ནཱཾ། །ཀུམ་བྷོཏྐྵེ་པེ་ཤྲཱ་མཱརྟྟཱི་ཡ་སྟ་ཏྲེ་ཀ་སྱཱཿས་མ་བྷྱེ་ཏྱ་སཿ། །ཧསྟཱ་ལམྦེ་ན་བ་ཧཱ་
འཁྲི་ཤིང་གིས་བསྒྲིབས་ནས། །དེར་དེས་ཆུ་སྐྱེས་རྡུལ་ཚོགས་ཀྱིས། །དྲི་བཟང་སེར་པོར་གྱུར་པའི་ཆུ། །མིའམ་ཅི་མོ་རྣམས་དག་གིས། །གསེར་གྱི་བུམ་པར་ལེན་པ་མཐོང་། །དེར་ནི་བུམ་འདེགས་ངལ་བ་ཡིས། །གཟིར་བ་ཅིག་ཅན་དེས་མངོན་ཕྱོགས། །ལག་པས་རྟེན་ནས་གྲོགས་
ཡཾ་ཀྲྀ་ཏྭཱ་ས་པྲཙྪ་ཏཱཾ་ཤ་ནཻཿ། །མཱ་ཏཿཀ་སྱ་ཀྲྀ་ཏེ་ཏོ་ཡ་མི་དཾ་ཡཏྣེ་ན་ནཱི་ཡ་ཏེ། །ཡངྦྷཀྟྱཱ་ག་ཎྱ་ཏེ་ནཱ་ཡཾ་བྷ་བ་ཏཱི་བྷིཿཔ་རི་ཤྲ་མཿ། །ཨི་ཏི་ཏེ་ན་པྲི་ཡ་གི་རཱ་པཱ་པྲིཥྚཱ་ཏ་མ་བྷཱཥ་ཏ། །མཱ་དྷུཪྻ་དྷུཪྻ་སཽནྡཪྻ་བཀྵ་པཱ་ཏ་
བྱས་ཏེ། །དལ་བུ་ཡིས་ནི་དེ་ལ་དྲིས། །མ་མ་སུ་ཡི་ཆེད་དུ་ནི། །ཆུ་འདི་འབད་ནས་ལེན་པར་བྱེད། །གང་ལ་གུས་པས་ཁྱེད་རྣམས་ནི། །ཡོངས་སུ་ངལ་འདི་རྩི་མི་བྱེད། །དེ་ལྟར་ཚིག་སྙན་གྱིས་དེས་དྲིས། །མཛེས་སྡུག་སྙན་སྒྲ་ལྡན་པ་ཡིས། །ཕྱོགས་སུ་སྐད་ཅིག་གིས་ལྷུང་མ། །དེ་
པ་ཏཱི་ཀྵ་ཎཱཏ། །མརྟྱ་མོ་དཱ་ས་ནོ་དཱ་ཡ་ས་དཱ་སུ་ར་བྷི་བཱ་རི་བྷིཿ། །པི་ཏུཿཀིནྣ་རརཱ་ཛསྱ་གི་རཱ་སྣཱ་ཏི་མ་ནོ་ཧ་རཱ། །ཏ་ཡེ་ཏི་ཀ་ཐི་ཏཾ་ཤྲུ་ཏྭཱ་སུ་དྷཱ་སིཀྟ་ཨི་བཀྵ་ཎཱ་ཏ། །ས་ཧེ་མ་ཀུཾ་བྷེ་བིཀྵེ་བཏ་ད་བྷི་ཛྙཱཾ་གུ་ལཱི་
ཡིས་དེ་ལ་རབ་སྨྲས་པ། །མི་ཡི་དིང་དག་བྱའི་སླད། །རྟག་ཏུ་དྲི་བཟང་ཆུ་རྣམས་ཀྱིས། །མི་འམ་ཅི་རྒྱལ་ཡབ་ཀྱི་བཀས། །ཡིད་འཕྲོག་མ་ནི་ཁྲུས་བྱེད་དོ། །ཞེས་པ་དེ་ཡིས་བརྗོད་ཐོས་ནས། །སྐད་ཅིག་བདུད་རྩིས་བྲན་པ་བཞིན། །དེ་ཡི་རྟགས་མཚན་སོར་གདུབ་ནི། །གསེར་གྱི་བུམ་པར་དེ་
