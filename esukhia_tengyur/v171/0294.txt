ཏཱི་བྷིརྨྨཱ་ཀྲི་ཡ་ཏཱ་མི་ཏྱུ་བཱ་ཙ་ཏཱ། །ཏ་ཏསྟཱ་བྷིརྣྣི་རུདྡྷཱ་ཀྵཱི་དྲྀ་ཌྷ་བནྡྷེ་ན་བཱ་པ་སཱ། །སཱ་སཱུ་ཏ་བཱ་ལ་ཡུ་ག་ལཾ་པྲ་ཏཔྟ་ཀ་ན་ཀ་པྲ་བྷཾ། །དྷྲྀ་ཏྭཱ་ཏཽ་བསྟྲ་བི་ཧི་ཏེ་མཉྫུ་ཥེ་གཱུ་ཌྷ་མུ་དྲྀ་ཏེ། །གི་གཱ་ཛ་ལེ་ནིཿཀ་རུ་ཎཿཀུ་མཱ་རཽ་
པ་ནི། །མཛོད་ཅེས་དེ་དག་རྣམས་ལ་སྨྲས། །དེ་ནས་དེ་དག་རྣམས་ཀྱིས་མིག །རས་ཀྱིས་དམ་དུ་བཅིངས་པས་བཀབ། །བཙོ་མ་གསེར་གྱི་འོད་ལྡན་པའི། །བྱིས་པ་ཟུང་ནི་དེ་ལས་བཙས། །གཞོན་ནུ་དེ་དག་སྒྲོམ་བུར་བཅུག །གོས་ཀྱིས་དཀྲིས་ཤིང་རྒྱ་བཏབ་ནས། །བུད་མེད་སྙིང་རྗེ་མེད་རྣམས་ཀྱིས། །གང་གཱའི་ཆུ་ལ་
བིཀྵིཔྟ་སྟྲི་ཡཿ། །ཀྲྀ་ཏྭཱ་རཀྟཱཀྟ་བ་ད་ནཱ་མཱུ་ཙུཿཔདྨཱ་བ་ཏཱིཾ་ཙ་ཏཱཿ། །ཤ་ཀ་ལ་དྭ་ཡ་མ་སྱནྡཾ་ཛཱ་ཏཾ་ཏེ་ཀྵིཔྟ་མཾ་བྷ་སེ། །བི་པུ་ལོཏྤ་བ་པནྣདྡྷཿཀིཾ་ཛཱ་ཏ་མི་ཏི་བྷཱུ་པ་ཏིཿ། །ཨ་པྲིཙྪ་དཱནྟ་པུ་རི་ཀཱནྤུ་ཏྲ་སནྡཪྴ་ནོཏྤུ་ཀཱཿ། །ཏེ་ཏཾ་བ་
གསང་སྟེ་འཕངས། །པདྨ་ཅན་གདོང་ཁྲག་གིས་ནི། །བསྒོས་ཏེ་དེ་དག་གིས་སྨྲས་པ། །ཁྱོད་ལ་དུམ་བུ་གཡོ་མེད་གཉིས། །བཙས་པ་དག་ནི་ཆུ་ལ་འཕངས། །དགའ་སྟོན་རྒྱ་ཆེ་ཆས་བཅུག་ཅིང་། །བུ་ནི་བལྟ་བར་འདོད་པ་ཡིས། །ས་ཡི་བདག་པོས་ཅི་སྐྱེས་ཞེས། །ཕོ་བྲང་འཁོར་པ་རྣམས་ལ་དྲིས། །དེ་ལ་དེ་དག་
བྷཱ་ཥི་རི་དེ་བ་ས་དྲྀ་ཤཾ་སུ་ཏཱ་ཏྭདྡྭ་ཡ། །སཾ་ཛཱ་ཏཾ་བྷཀྵི་ཏཾ་དེ་བྱཱ་པི་ཤཱ་ཙྱེ་བ་ཀི་མུཙྱ་ཏེ། །ཏྲསྟཿཤྲུ་ཏྭེ་ཏི་ནྲྀ་པ་ཏིརྒ་ཏྭཱནྟཿཔུ་ར་མནྡི་རཾ། །རཀྟཱཀྟ་བ་ད་ནཱནྡེ་བཱིཾ་དྲྀཥྚྭཱ་ས་ཏྱ་མ་མ་ནྱ་ཏ། །བ་དྷཱ་ཡ་བ་ཧི་རུཏྶྲྀཥྚཱ་ས་
གིས་སྨྲས་ལྷ། །སྲས་ནི་ཁྱོད་དང་མཚུངས་པ་གཉིས། །ཡང་དག་འཁྲུངས་པ་ལྷ་མོས་ཟོས། །ཤ་ཟ་མོ་ཉིད་ཅི་ཞིག་བརྗོད། །འདི་ཐོས་མི་བདག་སྐྲག་པ་ཡིས། །ཕོ་བྲང་འཁོར་གྱི་ཁང་པར་སོང་། །ལྷ་མོ་བཞིན་རས་ཁྲག་གིས་ནི། །སྤགས་པ་མཐོང་ནས་བདེན་པར་བསམས། །ས་བདག་རབ་ཏུ་ཁྲོས་པ་ཡིས། །དེ་
