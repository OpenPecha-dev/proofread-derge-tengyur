ཏ་ཏཿཏ་སྱ་པྲ་བ་ཧ་ཎཾ་ཀཱུ་ལེ་པྲཱ་ཛྱ་མ་བྷ་ཛྱ་ཏ། །ཏ་སྱོཏྶིཀྟ་མ་ཏེརྨཱ་ཏུ་རཱ་ཤཱ་བྷཾ་ག་གུ་རོཏྦྷ་བཏ། །བྷགྣེ་པྲ་བ་ཧ་ཎེ་ཏསྨིནྨོ་ཧེཿཤཱ་པ་ཨི་བཱ་བྷ་བཏ། །ཏ་ཏཿཕ་ལ་ཧ་ཀཱ་བཱཔྟྱཱ་ཤ་ནཻཿསཾ་ཛྙཱ་མ་བཱ་སྱ་སཿ། །ཏཱི་ར་མཱ་
སོག་ལེ་དྲག་པོས་དྲལ་བར་བྱས། །འཇུག་ངོགས་དག་ཏུ་ཞིག་པར་གྱུར། །བློ་གྲོས་རབ་འཕྱར་དེ་ཡིས་ནི། །མ་ཡི་བསམ་བཅོམ་ཁུར་ལྕི་བས། །གཟིངས་ནི་ཞིག་པར་གྱུར་པ་དེས། །དམོད་པ་བཞིན་དུ་རྨོངས་པ་ཐོབ། །དེ་ནས་ཤིང་གི་ཚལ་པ་ཐོབ། །དལ་གྱིས་འདུ་ཤེས་ཐོབ་པ་དེས། །ཆུ་གཏེར་གྱི་
སཱ་དྱ་ཛ་ལ་དྷི་རྫ་གཱ་ཧེ་ག་ཧ་ཎཾ་མ་ཧཏ། །ཨེ་ཀཱ་ཀཱི་ཧཱ་རི་ཏ་དྷྲྀ་ཏིཿས་བྲ་ཛནྦི་ཛ་ནེ་ཙི་རཱཏ། །ར་མྱཾ་ར་མ་ཎ་ཀཾ་ནཱ་མ་ན་ག་རཾ་དི་བྱ་མཱ་བྷ་བཏ། །ཨཔྤ་རོ་བྷིཤྩ་ཏ་སྲེ་བྷིརྦ་ད་ནེནྡུ་ཏྭི་ཥཱཾ་ཙ་ཡཻཿ། །ཛྱོཏྶྣ་པུ་རཾ་སྲྀ་ཛནྟཱི་
ནི་ངོགས་ཕྱིན་ནས། །ནགས་ཚལ་ཆེན་པོའི་ནང་དུ་ཞུགས། །གཅིག་པུ་བརྟན་ཉམས་འགྲོ་བ་དེས། །སྐྱེ་བོ་མེད་པར་ཡུན་རིང་ན། །གྲོང་ཁྱེར་རྩེ་དགའ་ཅན་ཞེས་པ། །དགའ་བ་མཆོག་ནི་ཐོབ་པར་གྱུར། །བཞིན་རས་ཟླ་ཟེར་ཚོགས་རྣམས་ཀྱིས། །ཟླ་བའི་འོད་ནི་སྤྲོ་བྱེད་པ། །ལྷ་ཡི་བུ་མོ་བཞི་དག་
བྷིཿཀྲྀ་ཏཱ་ཏི་ཐྱཿས་སཱ་ད་རཾ། །ཧེ་མ་པུཥྐ་རི་ཎཱི་ཏཱི་རེ་ར་མྱ་རཏྣཾ་ལ་ཏཱ་བ་ནེ། །མ་ཎི་མནྡི་ར་ཤཡྻཱ་སུ་ཏ་ཏྲ་སཾ་བྷོ་ག་བཱན་བྷཱུཏ། །བི་པ་དཿསུ་ཁ་པཪྻནྟཱཿཀླེ་ཤ་པྲཱི་ཏཱཤྩ་སམྤ་དཿ། །ཨཱ་བརྟྟི་ནྱོ་བྷ་བཾ་ཏྱེ་བ་པཪྻཱ་ཡེ་
གིས། །གུས་བཅས་མགྲོན་ནི་བྱས་པ་དེ། །གསེར་གྱི་རྫིང་བུ་དག་གི་འགྲམ། །རིན་ཆེན་འཁྲི་ཤིང་དགའ་བའི་ཚལ། །ནོར་བུའི་ཁང་པ་ཁྲི་སྟན་ཅན། །དེར་ནི་ལོངས་སྤྱོད་ལྡན་པར་གྱུར། །རྒུད་པ་དག་ནི་བདེ་བའི་མཐའ། །ཕུན་ཚོགས་ཉོན་མོངས་དག་ལ་མཛའ། །ལུས་ཅན་རྣམས་ལ་རེ་མོས་ཀྱིས། །
