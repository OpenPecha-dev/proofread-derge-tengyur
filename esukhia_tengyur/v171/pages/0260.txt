ཏཾ་དྲྀཥྚྭཱ་པ་ཏི་ཏཾ་མནྟྲཱི་བི་ཥ་མེ་ཤྭ་བྷྲ་ཀོ་ཊ་རེ། །ཙཱུ་ཌཱ་མ་ཎི་ས་མཱ་དཱ་ཡ་ག་ཏྭཱ་རཱཛྙེ་ནྱ་བེ་ད་ཡཏ། །ཀུ་མཱ་རོ་སྱ་ཛ་ནཱ་ཁྱ་ན་ལཀྵེ་ཎ་ཤྭ་བྷྲ་བཱ་སི་ནཱ། །རཀྵི་ཏཿཀྵི་ཏི་མཱ་སཱ་དྱ་པཀྵི་བནྣ་བྱ་པ་དྱ་ཏ། །སུ་དྷརྨྨཱ་པི་པ་ཏི་ཏཱཾ་ཤྲུ་
མི་བཟད་གཡང་ས་གཅོད་དག་ཏུ། །ལྷུང་བ་དེ་མཐོང་བློན་པོ་ཡིས། །གཙུག་གི་ནོར་བུ་ཡང་དག་བླངས། །སོང་ནས་རྒྱལ་པོ་དག་ལ་བཤད། །གནོད་སྦྱིན་ཨ་ཛ་ལ་ཞེས་པ། །གཡང་སར་གནས་པས་གཞོན་ནུ་ཡང་། །བསྲུངས་ཏེ་འདབ་ཆགས་བཞིན་དུ་ནི། །ས་ལ་ཕྱིན་ཅིང་ཤི་མ་གྱུར། །བུ་ནི་ལྷུང་ཐོས་ཆོས་བཟང་དག །
ཏྭཱ་པུ་ཏྲཾ་སྭ་ནི་དྷ་ནེ་ཥི་ཎཱི། །སུ་ཏཱ་ཏེ་ཛཱི་བ་ཏཱི་ཏྱུ་ཀྟྭཱ་རཀྵི་ཏཱ་དི་བྱ་ཡོ་ཥི་ཏཱ། །ཀུ་མཱ་རོ་པི་ཁ་ལ་བྱཱ་གྷྲ་ཀྵུ་ར་ཀྵུརྞྞ་ཤི་ལཱ་ཏ་ལཾ། །ག་ཛཱ་སྲྀཏྨཏྟ་ཤཱརྡཱུ་ལ་དཱ་རུ་ཎཾ་བ་ན་མཱ་བི་ཤཏ། །ཏ་ཏྲ་སིཾ་ག་ལ་ཀ་ཀྵི་ན་ལུབྡྷ་ཀེ་ན་ནི་བེ་དི་ཏཾ། །
རང་ཉིད་འཆི་བ་འཚོལ་བ་ནི། །ལྷ་མོ་རྣམས་ཀྱིས་ཁྱོད་ཀྱི་བུ། །འཚོ་ཞེས་བརྗོད་ནས་རབ་ཏུ་བསྲུངས། །གཞོན་ནུའང་སྟག་རྨིག་རྣོན་པོ་ཡིས། །རྡོ་བའི་གཞི་ནི་ཉམས་བྱས་ཤིང་། །གླང་པོའི་ཁྲག་གིས་མྱོས་པ་ཡིས། །མི་བཟད་སྟག་ལྡན་ནགས་སུ་ཞུགས། །དེ་ནི་རྔོན་པ་མིག་སེར་ཅན། །ཞེས་པ་ཡིས་ནི་རབ་བཤད་དེ། །

ལམ་ནི་ཐོབ་པའི་སྐྱེས་བུ་དག །ལུས་ཀུན་བཅད་པ་མཐོང་བར་གྱུར། །དེ་མཐོང་སྙིང་རྗེ་སྐྱེས་པ་ཡི། །མི་བདག་བུ་ཡིས་རབ་ཏུ་དྲིས། །ཁྱོད་ཀྱི་སྡུག་བསྔལ་གནས་སྐབས་འདི། །མི་མེད་ནགས་སུ་ཅི་ཡིས་ཐོབ། །དེས་སྨྲས་རིང་པོ་མིན་འདི་ན། །གཏུམ་པོ་གཏུམ་པོའི་སྤྱོད་ཚུལ་ཅན། །བཟོད་དཀའ་སུ་དཱ་ས་ཞེས་པ། །སྐྱེས་
