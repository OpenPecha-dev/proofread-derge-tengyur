ཡ་དཱ་ཡ་དཱ་པལླ་བ་སེ་ས་ལཱཾ་གཱི་དེ་ཧཾ་ས་མུཏྶྲྀཥྚ་བ་ཏཱི་པ་ཏི་སཱ། །ཏ་དཱ་ཏ་དཱ་མནྠ་ཐ་མོ་ཧི་ཏཱཾ་ཏཱཾ་ད་ཡཱརྡྲ་ཙཀྵུརྦྷ་ག་བཱནྡ་རཀྵ། །ཏ་ཏཿཀ་ད་ཙིདྦྷ་ག་བཱནྤནཱནྟེ་ཏཏྐཽ་ཏུ་ཀཱདྦྷི་ཀྵུ་ག་ཎེ་ན་པྲྀཥྚཿ།
དག་པ་ཡལ་འདབ་ལྟར་མཉེན་ལུས་དང་ལྡན་པ་དེས། །གང་གང་ཚེ་ན་ལུས་ནི་རབ་ཏུ་འཕངས་གྱུར་པ། །དེ་དེ་ཚེ་ན་ཡིད་སྲུབས་ཀྱིས་ནི་རྨོངས་པ་དེ། །བརྩེ་བས་བརླན་པའི་སྤྱན་ལྡན་བཅོམ་ལྡན་འདས་ཀྱིས་བསྲུངས། །དེ་ནས་ནམ་ཞིག་ནགས་ཀྱི་མཐར་ནི་བཅོམ་ལྡན་ལ། །དེ་ཡི་དགེ་མཚན་དག་ལས་དགེ་སློང་ཚོགས་ཀྱིས་
ཛ་གཱ་ད་དནྟ་དྱུ་ཏི་ཙནྡྲི་ཀ་བྷིརྣི་པ་ར་ཡནྲཱ་ག་མི་བཱ་དྷ་ར་སྱ། །ཡ་ཤོ་དྷ་རཱ་མདྦི་ར་ཧེ་ཎ་ནི་ཏྱཾ་ཀ་རོ་ཏི་ཡཏྶཱ་ཧ་ས་མ་ཧ་ཏཱརྟྟི། །ཧ་རནྟི་དྷཻཪྻཾ་བེ་ཏ་རནྟི་མོ་ཧ་མེ་ཥ་སྭ་བྷཱ་བཿསྨ་ར་བི་བྷྲ་
དྲིས། །ཚེམས་ཀྱི་འོད་ཟེར་ཟླ་བའི་འོད་རྣམས་དག་གིས་ནི། །མཆུ་ཡི་དམར་བ་འགེབས་པར་མཛད་པ་བཞིན་དུ་གསུངས། །བདག་དང་བྲལ་བས་རྟག་ཏུ་གཟིར་བར་གྱུར་པ་ཡི། །གྲགས་འཛིན་མ་ནི་གང་ཞིག་གཟུ་ལུམས་འདི་དག་བྱེད། །བརྟན་པ་འཕྲོག་ཅིང་རྨོངས་པ་རྣམ་པར་སྟེར་བ་འདི། །ཆགས་པའི་རྣམ་འཕྲུལ་ལྡན་པ་རྣམས་ཀྱི་རང་
མཱ་ཎཱཾ། །མ་མཱ་པི་ཏ་སྱཱ་བི་ར་ཧེ་པཱུརྦྦཱཾ་ཛནྨཱཾ་ཏ་རེ་མཱ་ར་བི་མོ་ཧི་ཏེ་ན། །སཾ་སཀྟ་སཾ་ཏཱ་པ་ནི་མིཏྟ་བྷུ་ཏཿཁེ་དཿཔྲ་བྷུ་ཏ་བྱ་པ་ནོ་ནུ་བྷཱུ་ཏཿ། །པུ་རཱ་ཛི་ཏཱ་མ་ར་པུ་རེ་མཏྤུ་རེ་ཧསྟི་ན་སུ་རེ། །རཱ་ཛ་དྷ་ནཱ་བི་དྷཱ་ནོ་བྷཱུནྣེ་དྷཱ་
བཞིན་ཉིད། །སྔོན་གྱི་སྐྱེ་བ་གཞན་ལ་དེ་དང་བྲལ་བ་ཡིས། །འདོད་པས་རྣམ་པར་རྨོངས་པར་བྱས་ཏེ་བདག་གིས་ཀྱང་། །རྒྱུན་དུ་ཡོངས་སུ་གདུང་བ་དག་གི་རྒྱུར་གྱུར་པ། །རབ་མང་གདུང་བའི་ཉོན་མོངས་དག་ནི་ཉམས་སུ་མྱོང་། །གྲོང་མཆོག་ཧསྟི་ན་ཡི་གྲོང་། །འཆི་མེད་ཡོན་ཏན་ཐུལ་བར་སྔོན། །ཡོན་ཏན་ཕུན་སུམ་ཚོགས་པའི་གཏེར། །
