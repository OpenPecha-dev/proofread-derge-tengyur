བི་ན་ཡཱཏྤཱུ་ཛི་ཏསྟ་དྲ་བྲཧྨ་དཏྟེ་ན་བྷཱུ་བྷུ་ཛཱ། །ཏཏྶཾ་ད་ཀྵི་ཎཾ་ཙཀྲེ་དྷ་ནྱཱ་ནཱ་མ་ནུ་ཤཱ་ས་ནཾ། །ཏ་ཏྲ་པཽ་རཱཿས་མ་བྷྱེ་ཏྱ་བྷ་ག་བནྟཾ་བྱ་ཛཱིརྞྞ་ཥུཿ། །སརྦྦ་སདྦི་ནི་པ་ཏེ་ཥུ་ཏྲཱ་ཏཱ་རཾ་སརྦ་དེ་ཧི་
དེར་ནི་ས་བདག་ཚངས་བྱིན་གྱིས། །དུལ་བ་ཡི་ནི་མཆོད་པ་བྱས། །དེ་ཡིས་ཡང་དག་རྗེས་མཐུན་པར། །སྐལ་ལྡན་རྣམས་ལ་རྗེས་བསྟན་མཛད། །དེར་ནི་གྲོང་པ་རྣམས་འོངས་ཏེ། །རྒུད་པ་ཀུན་ཏུ་ལུས་ཅན་རྣམས། །རྣམ་པར་ལྷུང་ལ་སྐྱོབ་མཛད་པའི། །བཅོམ་ལྡན་འདས་ལ་རབ་གསོལ་
ནཱཾ། །བྷ་ག་བནྣསྟི་ན་ག་ར་སྱཱནྟེ་སཱ་ཥཱ་ཎསརྦྦ་ཏཿ། །ཏ་ཏྲ་གོ་པཱ་ལ་ཀོ་ནཱ་གཿཀྲཱུ་རོ་བ་ས་ཏི་དུཿས་ཧཿ། །ས་པ་ཤཱུ་ནཱཾ་ཛི་ནཱ་ནཱཉྩ་ཤ་སྱཱ་ནཱཉྩ་མ་ཧཱ་ཤ་ནིཿ། །སམྤནྣཱ་ནཱ་མ་བྷཱ་བཱ་ཡ་ན་བིདྨཿཀེ་ན་ནིརྨྨི་ཏཿ། །ཨ་དཱནྟ་ད་མ་
པ། །གྲོང་ཁྱེར་མཐའ་ན་བཅོམ་ལྡན་འདས། །རྡོ་བ་ཡི་ནི་རི་ཞིག་ཡོད། །དེ་ན་ཀླུ་ནི་བ་ལང་སྐྱོང་། །དྲག་པོ་བཟོད་པར་དཀའ་བ་གནས། །དེ་ནི་ཕྱུགས་དང་སྐྱེ་བོ་དང་། །འབྲུ་རྣམས་ཀུན་གྱི་རྡོ་རྗེ་ཆེ། །ཕུན་སུམ་ཚོགས་རྣམས་དངོས་མེད་སླད། །ཅ་ཡིས་རྣམ་སྤྲུལ་རིག་མ་གྱུར། །མ་དུལ་འདུལ་
ན་སྱསྨིནྣ་ཤཱནྟ་ཤ་མ་ན་སྱ་ཏེ། །ཨུ་པ་སརྒེ་ཀྲྀ་པཱསྨཱ་ཀཾ་ཤར་ཎ་ཤ་ར་ཎཻ་ཥི་ཎཱ་མ། །ཨི་ཏྱུཀྟྭཱ་ཏེ་ཥུ་དྷཱ་ཏེ་ཥུ་བྷ་ག་བཱན་ཀ་རུ་ཎཱ་ནི་དྷིཿ། །ཨནྟརྷི་ཏ་པྲ་བྷཱ་མ་དྷྱཱ་ཏ་ཡ་ཡཽ་པཱ་ཥཱ་ཎ་བྷཱུ་དྷ་རཿ། །ཏ་ཊེ་བི་སཾ་
མཛད་མ་ཞི་བ། །ཞི་མཛད་ཁྱོད་ཀྱི་བརྩེ་བ་ནི། །ཉེ་བར་འཚེ་བ་འདི་དག་ལ། །བདག་ཅག་སྐྱབས་འཚོལ་རྣམས་ཀྱི་སྐྱབས། །ཞེས་བརྗོད་དེ་དག་སོང་བའི་ཚེ། །བཅོམ་ལྡན་སྙིང་རྗེའི་གཏེར་དག་ནི། །འདུས་པའི་དབུས་ལས་མི་སྣང་བར། །རྡོ་བའི་ས་འཛིན་དག་ཏུ་གཤེགས། །དེ་ངོས་ཤིན་ཏུ་
