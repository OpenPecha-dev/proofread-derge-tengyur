སུ་ག་ཏ་པྲ་ཏཾ་སྱ་ས་ཏྭ་ཤྲི་ཡཿཤུ་ཙི་ཏ་ཡཱ་ཏི་ལ་ཀཱ་ཡ་མཱ་ནཾ། །སརྦྦཱཏྨ་ནཱ་སུ་ཙ་རི་ཏཱ་བྷ་ར་ཎ་སྱ་པུཾ་སཿཤཱི་ལོ་པ་མཾ་ན་ཧི་པི་བྷཱུ་ཥ་ཎ་མསྟི་ཀིཉྩི་ཏ། །ཨི་ཏྱུཀྟྭཱ་སཾ་ཡ་ཏཿཤཱུ་ནྱ་པཱ་ཏྲ་ཨེ་བ་ཛ་གཱ་མ་སཿ། །ཨ་པཱ་ཡ་ཤ་ལྱ་བི་
སྤྱད་རྒྱན་ལྡན་སྐྱེས་བུ་ལ། །བདེ་གཤེགས་བཏུལ་ཞུགས་ཅོད་པན་ཉིད་དུ་རབ་གྱུར་ཅིང་། །སྙིང་སྟོབས་དཔལ་ནི་གཙང་བས་ཐིག་ལེ་ཉིད་གྱུར་པ། །ཚུལ་ཁྲིམས་མཚུངས་པའི་རྒྱན་ནི་ཅུང་ཟད་ཡོད་མ་ཡིན། །ཞེས་པ་བརྗོད་ནས་དུལ་ལྡན་དེ། །གནོད་པའི་ཟུག་རྔུ་མི་བཟད་ཅན། །མཆོད་པ་རབ་བཏང་མྱུར་
ཥ་མཱཾ་པཱུ་ཛཱ་མུདྶྲྀ་ཛྱ་པ་ཏྭ་ར། །ཨ་ཐ་མནྟྲཱ་བ་ལཱནྣཱ་མ་ཙཎྜཱ་ལཱི་བི་ཤ་ཀཱ་ལི་ནཱི། །བ་ཤྱཱ་ཀཪྵཱ་དི་ནི་ཡནྟྲ་མནྟྲ་ཏནྟྲཱ་དི་ཏཏྤ་རཱཾ། །ཨཱ་ཧཱུ་ཡ་བི་ནི་བེ་དྱཱསྨི་ནི་ཛཾ་སྨ་ར་པ་རཱ་བྷ་བཾ། །ཡ་ཡཱ་ཏེ་དྷ་ན་སཾ་ཏུཥྚིཾ་རཱ་
པ་དང་། །བཅས་པ་སྣོད་སྟོང་ཉིད་དུ་སོང་། །དེ་ནས་སྔགས་སྟོབས་ཞེས་པ་ཡི། །སྨད་འཚོང་གཏུམ་མོ་འདོད་ལྡན་མ། །དབང་དགུག་སོགས་ལ་འཁྲུལ་འཁོར་དང་། །སྔགས་དང་རྒྱུད་སོགས་དེ་སྦྱོར་མ། །བོས་ནས་དེ་ལ་རང་ཉིད་ནི། །ཆགས་པའི་ཟིལ་གྱིས་མནན་པ་བཤད། །ནོར་གྱིས་ཚིམ་བྱས་ཆགས་པ་ཡིས། །གཟིར་མས་
གཱརྟཱ་པྲི་ཡཱཾ་སཾ་ག་མཾ། །ཨ་ཐ་མནྟྲ་བ་ལཱདྐྲྀཥྚསྟ་ཡཱ་ས་ཡ་དི་པདྨ་ཀཿ། །བི་བ་ཤསྟཱཾ་ས་མ་ཐྱེ་ཏྱ་ཀིཾ་ཀ་རོ་མཱི་ཏྱ་བྷཱ་ཥཊ། །སཱ་ཏ་མཱུ་ཙེ་ཤ་ཤི་མུ་ཁཱིཾ་བྷ་ཛ་པྲ་ཎ་ཡི་ནཱི་མི་མཱཾ། །པུ་ཎྱཻ་རུ་པ་ན་ཏཱཾ་ཀཱནྟཱཾ་དཱིཔྟ་བཱ་བཧྣི་མཱ་བི་ཤཏ། །པདྨ་ཀོ་བྱ་
དགའ་དང་འགྲོགས་པ་ཞུས། །དེ་ནས་དེ་ཡིས་འཕྲལ་ལ་ནི། །སྔགས་སྟོབས་ཀྱིས་དྲངས་པདྨ་ཅན། །དབང་མེད་དེ་ལ་མངོན་ཕྱོགས་ཏེ། །བདག་གིས་ཅི་ཞིག་བྱ་ཞེས་སྨྲས། །དེ་ལ་དེས་སྨྲས་བསོད་ནམས་ཀྱིས། །ཉེར་གནས་མཛེས་མ་ཟླ་བཞིན་མ། །མཛའ་མོ་འདི་ནི་རབ་ཏུ་བསྟེན། །ཡང་ན་འབར་བའི་མེ་ལ་ཞུགས། །པདྨོ་ཅན་
