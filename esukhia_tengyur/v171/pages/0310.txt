གཱ་དྷིཥྛི་ཏཾ་ཛྙཱ་ཏྭཱ་བྷཱི་ར་ས་ཏྭཿས་མཱ་དྷི་ནཱ། །ས་སཾ་ཀྵོ་བྷཾ་བི་ད་དྷེ་ན་ནཱ་གཱ་ནཾ་ཀཾ་བི་ཏཱ་བ་ནིཿ། །ཤསྟྲཱགྣི་བྲྀཥྚཱ་བུཏྤྲིཥྚཱ་ཏཻ་ར་མཪྴ་བི་ཀཱ་རི་བྷིཿ། །པྲ་བྷཱ་བེ་ན་བྷ་བཏྟ་སྱ་མཱུརྡྷི་པདྨོཏྤ་ལཱ་བ་ལཱི། །ཏ་ཏསྟཾ་ཛ་ག་དུརྣཱ་གཱསྟིངྦཪྻོ་ཏྐཪྵ་
བྱིན་བརླབས་དེ་ཤེས་ནས། །ས་གཞི་འདར་བྱེད་དེ་ཡིས་ནི། །སྙིང་སྟོབས་བརྟན་ལྡན་ཏིང་འཛིན་གྱིས། །ཀླུ་རྣམས་ཀུན་ཏུ་བསྐྱོད་པ་བསྒྲུབས། །རྣམ་འགྱུར་ཁྲོ་ལྡན་དེ་དག་གིས། །མཚོན་དང་མེ་ཡི་ཆར་སྤྲོས་པ། །མཐུ་ཡིས་དེ་ཡི་སྤྱི་བོར་ནི། །པདྨ་ཨུཏྤལ་འཕྲེང་བར་གྱུར། །དེ་ནས་དེ་ཡི་བརྩོན་འགྲུས་ཀྱིས། །རབ་ཏུ་ཡ་
བི་སྨི་ཏཱཿ། །ཡཱ་བནྦཪྻཾ་ཀ་བནྡྷོ་ཏྲ་དེ་ཤསྟཱ་བནྦ་ཤེ་ཏ་བ། །ཨུཀྟྭེ་ཏི་ཏསྨཻ་པཪྻཾ་ཀ་བནྡྷ་ཏུ་ལྱཾ་པྲ་མཱ་ཎ་ཏཿ། །ན་བ་དྲོ་ཎ་མུ་ཁཱ་ཡཱ་ལཾ་པྲ་ད་དུཿཤཱུ་མཎྜ་ལཾ། །ས་ཏ་ཏྲ་ན་ག་ར་གྲཱ་མ་པནྣི་བེ་ཤ་བི་ཤེ་ཥ་ཀྲྀཏ། །སྠི་ཏཾ་བ་བནྡྷ་
མཚན་ཀླུས་དེ་ལ། །སྨྲས་པ་ཇི་སྲིད་སྐྱིལ་ཀྲུང་བཅས། །དེ་སྲིད་ཡུལ་འདིར་ཁྱོད་ཀྱི་གནས། །ཞེས་པ་བརྗོད་ནས་དེ་ལ་ནི། །སྐྱིལ་ཀྲུང་ཡངས་པར་བཅིངས་པ་ཡི། །ཚད་དང་མཚུངས་པ་སྟོང་པ་ཡི། །དཀྱིལ་འཁོར་དག་ནི་རབ་ཏུ་བྱིན། །དེར་ནི་དེ་ཡི་གྲོང་ཁྱེར་དང་། །གྲོང་གི་ཁྱད་པར་འགོད་བྱེད་ཅིང་། །དགྲ་བཅོམ་ལྔ་བརྒྱ་
ས་ཧི་ཏཿཤ་ཏཻཿཔཉྩ་བྷི་རརྷ་ཏཱཾ། །ཏ་ཏྲཱ་ཀྵ་ཡཾ་བྷ་ག་བ་ཏཿཔནྣི་བེ་ཤྱ་དྷརྨྨཾ་ཀྲྀ་ཏྭ་བི་ཧཱ་ར་རུ་ཙི་རཱ་བྷ་ར་ཎཱཾ་དྷ་རི་ཏྲཱིཾ། །ཏཾ་གནྡྷ་མཱ་ད་ན་ཏ་ཊོདྡྷྲྀ་ཏ་ལབྡྷཱ་སཱ་རཻརྦྱཱཔྟཱ་ཙ་ཀཱ་ར་ན་བ་ཀུཾ་ཀུ་མ་ཀནྡ་བྲྀནྡཻཿ། །ཨི་ཏི་ཀྵེ་
དག་དང་ནི། །ལྷན་ཅིག་རབ་ཏུ་གནས་པར་བྱས། །དེར་ནི་མི་ཟད་བཅོམ་ལྡན་ཆོས་དག་རྣམ་པར་བཀོད་བྱས་ནས། །མཛེས་པའི་གཙུག་ལག་ཁང་གི་རྒྱན་དང་ལྡན་པའི་འཛིན་བྱེད་དེ། །སྤོས་ངད་ལྡང་གི་ངོས་ལས་བླངས་ཤིང་སྙིང་པོ་ཐོབ་གྱུར་པ། །གུར་གུམ་གསར་པའི་རྩ་བའི་ཚོགས་ཀྱིས་རབ་ཏུ་ཁྱབ་པར་བྱས། །ཞེས་པ་དགེ་
