རམྨི་ཐཿ། །སྭ་ཡཾ་ཤུ་ཤྲཱ་བ་སུ་ག་ཏཿཔྲཱི་ཏི་ཡུཀྟཿསུ་བྷཱ་ཥི་ཏཾ། །སུ་བྷཱ་ཥི་ཏ་བྱ་ས་ནི་ཏཱཾ་ཏཱཾ་དྲྀཥྚྭཱ་ཏ་སྱ་བྷིཀྵ་བཿ། །ཏ་མཱུ་ཙུསྟ་བ་སརྦ་ཛྙ་ཀ་ཐ་མསྨཱདྦ་ཙཿཔྲི་ཡཾ། །ཨི་ཏྱུཀྟེ་བྷིཀྵཱུ་སཾ་གྷེ་ན་བྷ་ག་བཱ་ན་བྷྱ་དྷཛྫི་ནཿ། །ཛནྨཱནྟ་རེ་
དག་གིས། །ལེགས་བཤད་བདེ་བར་གཤེགས་པ་ནི། །དགྱེས་པ་དང་ལྡན་རང་གིས་གསན། །དེ་ཡི་ལེགས་བཤད་ལ་གཅགས་པ། །དེ་མཐོང་དགེ་སློང་རྣམས་ཀྱིས་ནི། །དེ་ལ་གསོལ་བ་ཀུན་མཁྱེན་ཉིད། །ངེད་ཀྱི་ཚིག་ལ་ཇི་ལྟར་དགྱེས། །དགེ་སློང་ཚོགས་ཀྱིས་དེ་བརྗོད་ཚེ། །བཅོམ་ལྡན་རྒྱལ་བས་བཀའ་སྩལ་པ། །བདག་ནི་སྐྱེ་བ་
ཥྭ་པི་པ་ར་བལླ་བྷ་མེ་སུ་བྷཱ་ཥི་ཏཾ། །ཤི་བ་བ་ཏྱཱཾ་པུ་རཱ་པུཪྻཱཾ་ཤི་བིརྣྣཱ་མ་ན་རེ་ཤྭ་རཿ། །བ་བྷཱུ་བ་སརྦྦ་བྷཱུ་ཏཱ་ནཱཾ་ད་ཡཱ་ད་ཡི་ཏ་བཱནྡྷ་བཿ། །ཏཏྤུ་ཎྱ་ཤཱ་ས་ནཱ་སཱི་ནཻཿསྭརྒྒཱ་མཱ་པཱུ་རི་ཏཾ་ཛ་ནཻཿ། །དྲྀཥྚྭཱ་ཤ་ཏ་ཀྲ་ཏུ་ཛྙཱ་ཏུཾ་སྭ་ཏྭཾ་ཏ་སྱ་ས་མཱ་ཡ་
གཞན་དུ་ཡང་། །ལེགས་བཤད་དག་ལ་མཆོག་ཏུ་དགའ། །གྲོང་ཁྱེར་ཞི་ལྡན་དག་ཏུ་སྔོན། །མི་དབང་ཤི་བི་ཞེས་བྱ་བ། །འབྱུང་པོ་ཀུན་ལ་བརྩེ་བ་ཡི། །ཡིད་འོང་གཉེན་དུ་གྱུར་པ་བྱུང་། །དེ་ཡི་བསོད་ནམས་བཀར་གནས་པའི། །མི་ཡི་མཐོ་རིས་ཀུན་གང་བ། །མཐོང་ནས་དེ་ཡི་སྙིང་སྟོབས་ནི། །རྟག་ཏུ་མཆོད་སྦྱིན་བརྒྱ་པ་
ཡཽ། །རཏྣཱ་པྲཱ་སཱ་ད་ཤི་ཁ་ར་ཀྲཱནྟཾ་ནྲྀ་པ་ཏི་མེ་བྷྱ་སཿ། །རཀྵོ་རཱུ་པཱཾ་བི་དྷཱ་ཡོ་གྲཾ་ཏ་ད་གྲཱ་བྷི་མུ་ཁོ་བྲ་བཱིཏྲ། །ཨ་ནི་ཏྱཱཿསསྐཱཾ་རཱསྟ་ར་ལ་ཏ་ར་བི་དྱུདྡྻུ་ཏི་ནི་བྷཿས་མུཏྤནྣོཏྤནྣ་པྲ་ལ་ཡ་པ་རི་ཎཱ་མ་པྲ་ཎ་ཡི་ནཿ། །
འོངས། །དེ་ནི་རིན་ཆེན་ཁང་བཟང་རྩེར། །འཛེགས་པའི་མི་བདག་ཅན་སོང་ནས། །འཇིགས་རུང་སྲིན་པོའི་གཟུགས་བསྒྲུབས་ཏེ། །དེ་ཡི་མདུན་དུ་མངོན་ཕྱོགས་སྨྲས། །འདུ་བྱེད་མི་རྟག་ཤིན་ཏུ་གཡོ་བའི་གློག་རབ་མཛེས་པ་དག་དང་མཚུངས། །རབ་ཏུ་སྐྱེས་ཤིང་སྐྱེས་ཤིང་འཇིག་པའི་ཡོངས་སུ་འགྱུར་བ་དག་ལ་ཆགས། །
