ནཱཿ། །པཱ་ད་བྷྱཱ་མ་བ་ག་མྱ་ཏེ་ཡ་དི་པུ་ན་སྟེ་ནཻ་བཱ་སེ་བཱ་པུ་ནཱ་ཏཏྐོ་ནཱ་མ་གུ་ཎཿཀྵ་ཎཾ་ཀྵི་ཏི་བྷུ་ཛཱ་དྲྀཥྚེ་ན་པྲྀཥྚེ་ན་བཱ། །ཨི་ཏི་དྱཱ་ཏྭཱ་དྷ་རཱ་དྷི་ཤཿཀྲྀ་བ་བ་སུ་དྷ་ཡཱ་དྷི་ཡཱ། །ཏདྦཱཾཙྪ་སིདྡྷི་སནྣདྡྷ་པྲ་ཏི་བོ་དྷྱ་ཏ་མ་
གིས། །གལ་ཏེ་རྐང་པ་ཉིད་ཀྱིས་འགྲོ་ཞིང་སླར་ཡང་དེ་ནི་བསྟེན་བྱ་དེ་ལྟ་ན། །དེ་ནི་ངེས་པར་ས་བདག་སྐད་ཅིག་མཐོང་དང་མགུ་བ་ལ་ཡང་ཡོན་ཏན་ཙི། །བློ་གྲོས་བརྩེ་བའི་ནོར་འཛིན་གྱིས། །དེ་ལྟར་བསམས་ནས་ས་འཛིན་དབང་། །དེ་འདོད་བསྒྲུབ་པར་ཆས་གྱུར་པས། །རབ་སད་བྱས་ཏེ་
བྷྱ་དྷཏ། །ཨུཏྟིཥྛ་བི་པྲ་སཾ་པྲསྟུ་སྟྭཏ་ས་མཱི་ཧི་ཏ་སིདྡྷ་ཡེ། །ཨ་བི་ལམྦ་ཕ་ལཱ་བཱསྟི་ཏུ་པཱ་ཡ་པ་ར་མོ་མ་ཡཱ། །ཙྪི་ཏྭཱ་མ་མ་ཤི་རསྟཱ་བཏྲ་བྲ་ཧྨ་དཏྟཱ་ཡ་བྷཱུ་བྷུ་ཛེ། །ག་ཏྭཱ་པྲ་ཡཙྪ་ཏཏྤྲྀ་ཏྱཱ་པ་ཏེ་བིཏྟཾ་པྲ་དཱ་
དེ་ལ་སྨྲས། །ལོངས་ཤིག་བྲམ་ཟེ་ཁྱོད་ཀྱི་ནི། །འདོད་པ་བསྒྲུབ་སླད་མི་ཐོགས་པར། །འབྲས་བུ་ཐོབ་པའི་ཐབས་ཀྱི་མཆོག །བདག་གིས་ཡང་དག་རྙེད་པར་གྱུར། །རེ་ཞིག་བདག་གི་མགོ་བཅད་ནས། །སོང་སྟེ་ས་བདག་ཚངས་བྱིན་ལ། །བྱིན་ན་དེ་ནི་དགའ་བ་ཡིས། །ཁྱོད་ལ་ནོར་དག་སྟེར་བར་
སྱ་ཏི། །ཏ་སྱཱརྠེ་ཙནྡ་ན་ཏ་བྷོཿཤྲུ་ཏྭེ་ཏདྦ་ཙ་ན་དྭི་ཛཿ། །ཨུ་པ་ཙ་ཏཔྟ་པཱུ་ཙྱེ་བ་བིདྡྷཿཀརྞྞ་པྲ་བིཥྚ་ཡཱ། །ཏ་བ་ཏྲཻ་ལོ་ཀྱ་སཱ་ར་སྱ་ཛ་གཏྤུ་ཎྱཱཔྟ་ཛནྨ་ནཿ། །ཤསྟྲཾ་ཀཎྛེ་ཤ་ཋེ་ནཱསྨནྡ་པཱ་པེ་ན་ཡ་ཏྱ་ཧ་ན། །
འགྱུར། །སློང་བའི་ཙནྡན་ལྗོན་པ་དེའི། །ཚིག་དེ་ཐོས་ནས་གཉིས་སྐྱེས་ནི། །རྣ་བར་རབ་ཞུགས་གདུང་བ་ཡིས། །ཁབ་ཀྱིས་ཕུག་བཞིན་རབ་སྨྲས་པ། །སྲིད་གསུམ་སྙིང་པོ་སྐྱེ་བ་ནི། །འགྲོ་བའི་བསོད་ནམས་ཀྱིས་ཐོབ་པ། །ཁྱོད་ཀྱི་མགྲིན་པ་འདི་ལ་མཚོན། །བརླང་པོ་སྡིག་ཅན་སུ་ཞིག་བསྣུན། །
