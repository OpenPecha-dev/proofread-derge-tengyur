ཀྲྀཙྪཱ་ག་ཏསྟ་པནྟེ། །ནི་དྷཱི་ཡ་ཏི་མུརྡྷྲཱེ་མ་མོ་གྲ་ཤཱ་པཿས་ཤི་ཏ་ལོཥྨཱ་པྲྀ་ཐུ་པཱ་པ་ཏཱ་པཱཏ། །ཨེ་ཏྱུཀྟ་མཱ་ཏྲེ་བ་སུ་དྷཱ་དྷི་པེ་ན་ཏཽ་བཛྲ་བྷགྣཱ་བི་བ་བྷགྣ་དྷཻཪྻཽ། །བྷཱི་ཏྱེ་བ་ཤོ་ཀ་སྱ་སུ་དུཿས་ཧ་སྱ་མ་ཧཱི་ཡཱ་སིཾ་མོ་ཧ་
སྡུག་བསྔལ་གྱུར་ནས་ཆུ་འགྲམ་ཉལ། །སྡིག་པའི་གདུང་བ་ཆེ་ལས་ཚ་གྲང་དང་། །བཅས་པའི་དམོད་དྲག་བདག་གི་སྤྱི་བོར་དོར། །ནོར་འཛིན་བདག་པོས་དེ་སྐད་བརྗོད་ཙམ་ལ། །རྡོ་རྗེས་བཅོམ་བཞིན་བརྟན་པ་ཉམས་དེ་དག །ཤིན་ཏུ་བཟོད་དཀའི་མྱ་ངན་ལ་འཇིགས་བཞིན། །རབ་ཏུ་ཆེ་བ་
གུ་ཧཱ་པྲ་བིཥྚཽ། །སིཀྟཽ་ཏ་ཏཿཤཱི་ཏ་ཛ་ལེ་ན་རཱ་ཛྙཱ་ཏཽ་ལབྡྷ་སཾ་ཛྙཱ་བ་ཡ་པཽ་ཀ་ཐཾ་ཙིཏ། །པུ་ཏྲཱནྟི་ཀཾ་བྷཱུ་པ་ན་ཡ་ཏྭ་མཱ་བཱ་མི་ཏྱུཙ་ཏུཥྚཽ་ཀ་རུ་ཎ་པྲ་ལཱ་པཽ། །ནཱི་ཏཽ་ཏ་ཏསྟེ་ན་པ་རིཏྤ་མཱི་པ་མ་དྭཱི་པ་དཱི་པེ་བྱ་བ་ནེ་ནི་མགྣཽ། །ཧསྟེ་ན་སཾ་སྤྲྀ་ཤྱ་སུ་
རྨོངས་པའི་ཕུག་ནང་ཞུགས། །དེ་ནས་རྒྱལ་པོས་ཆུ་གྲང་གཏོར་དེ་དག །ཇི་ཞིག་ལྟར་སྟེས་འདུ་ཤེས་ཆ་ཤས་ཐོབ། །ས་བདག་ཁྱོད་ཀྱིས་བདག་ཅག་བུ་དྲུང་དུ། །ཁྲིད་ཅེས་སྙིང་རྗེ་རབ་སྨྲ་དེ་དག་སྨྲ། །དེ་ནས་དེ་ཡིས་ཆུ་ཀླུང་ཉེ་བར་ཁྲིད། །གླིང་དང་སྒྲོན་མེ་ཉེ་བར་བྱིང་གྱུར་པ། །ལག་པས་མདའ་བསྣུན་བུ་ལ་
ཏཾ་ཤ་རཱཥྚཾ་ཏདྦཱ་ཎ་བིདྡྷ་དི་པ་བི་ཏ་ཏུསྟཽ། །ཀསྨཱ་ད་ཀསྨཱདྒུ་རུ་བཏྶ་ལོ་བི་བྲྀདྡྷཱ་བ་ནཱ་ཐཽ་བི་ཏ་རཽ་ཏྭ་མནྡྷཽ། །ཏྱཀྟྭ་དི་པཾ་གཙྪ་སི་པུ་ཏྲ་ཀེ་ཏི་གཱ་ཌྷཾ་ས་རཾ་ཥྭ་ཛྱ་ཏ་མཱུ་ཙ་ཏུསྟཽ། །ཨཱ་རཱ་དྷི་ཏཽ་བྷཀྟི་ས་པཪྻ་ཡཻ་བ་ནིརྦྱཱ་ཛ་མཱ་བཾ་གུ་རུ་
རིག་བྱས་ནས། །དེ་ཡི་མདའ་ཡིས་ཕུག་བཞིན་དེ་དག་འགྱེལ། །བུ་གཅིག་བླ་མ་དག་ལ་མཉེས་གཤིན་ཡང་། །ཕ་མ་རྒན་ལོང་མགོན་མེད་བཏང་ནས་ནི། །ཅི་སླད་གློ་བུར་མཐོ་རིས་ཁྱོད་འགྲོ་ཞེས། །དེ་དག་གིས་ནི་དེ་ལ་དམ་འཁྱུད་སྨྲས། །བུ་ཁྱོད་མེ་ཡི་སྦྱིན་སྲེག་ལ་དགའ་ཞིང་། །བླ་མའི་ལྷ་དང་
