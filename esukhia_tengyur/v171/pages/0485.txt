བི་སྨ་ཡཱཿ། །དྲྀཥྚཱ་དྲྀཥྚ་སུ་ཁེ་སུ་བྷཱ་ཥི་ཏ་རུ་ཙིརྨྨཱརྒཱཿཔྲ་བྷཱ་དཱི་པ་ཡཏ་སཾ་བཱ་དེ་ན་སུ་བྷཱ་ཥི་ཏཱ་མྲྀ་ཏ་ར་སཿཔྲཱི་ཏིཾ་པ་རཾ་བཪྵ་ཏི། །སརྦྦ་ཛྙོ་པི་སུ་བྷཱ་ཥི་ཏཱརྠ་བི་ཥ་ཡེ་དྱརྠ་ནཱི་ཡརྠི་ནཱཾ་
གྱིས་རབ་ཏུ་བསྔགས། །མཐོང་དང་མ་མཐོང་བདེ་ལ་ལམ་ནི་ལེགས་བཤད་ཀྱིས་གསལ་སྒྲོན་མ་ལྟ་བུར་སྤྱོད། །ལེགས་བཤད་བདུད་རྩིའི་རོ་ནི་ཉམས་མྱོང་ལྡན་པས་མཆོག་ཏུ་དགའ་བའི་ཆར་དག་འབེབས། །ལེགས་བཤད་དོན་གཉེར་ཡུལ་ལ་གཉེར་བྱ་ཀུན་མཁྱེན་དག་ཀྱང་
ཏསྨཱནྣཱ་སྟི་སུ་བྷཱ་ཥི་ཏེ་ན་ས་དྲྀ་ཤཾ་ཀིཾ་ཙིཏྶ་ཏཱཾ་བལླ་བྷམ། །ཨི་ཏི་ཀྵེ་མེནྡྲ་བི་ར་ཙི་ཏཱ་ཡཱཾ་བོ་དྷི་ས་ཏྭཱ་བ་དཱ་ན་ཀལྤ་ལ་ཏཱ་ཡཱཾ་ཤི་བི་སུ་བྷཱ་ཥི་ཏཱ་བ་དཱ་ན་མེ་ཀ་ན་བ་ཏི་ཏ་མཿ
དོན་གཉེར་ཉིད་དུ་འགྱུར། །དེ་སླད་དམ་པ་རྣམས་ལ་ལེགས་བཤད་དང་མཚུངས་དགའ་བ་ཅི་ཡང་ཡོད་མ་ཡིན། །ཞེས་པ་དགེ་བའི་དབང་པོས་བྱས་པའི་བྱང་ཆུབ་སེམས་དཔའི་རྟོགས་པ་བརྗོད་པ་དཔག་བསམ་གྱི་འཁྲི་ཤིང་ལས་ཤི་པིའི་ལེགས་བཤད་ཀྱི་རྟོགས་པ་བརྗོད་པའི་ཡལ་
པལླ་བཿ།། །།གུ་རཱུ་ཎ་མ་སྱ་གྲེ་གུ་རུ་ཏ་ར་པ་དེ་སཻ་བ་མ་ཧ་ཏཱི་ས་དཱ་སདྦྷིཿཔཱུ་ཛྱཱ་ཏྲི་ད་ཤ་ཏ་ཊི་ནཱི་པཱ་བ་ན་ཏ་ནུཿ། །དྷ་རི་ཏྲཱི་བྷཱུཾ་ཏཱ་ནཱཾ་ས་ཀ་ལ་ར་ས་སཱ་ར་པྲ་ས་ཙི་ནཱ་ཤ་རི་རཾ་པུཥྞཱ་
འདབ་སྟེ་དགུ་བཅུ་རྩ་གཅིག་པའོ།། །།བླ་མ་རྣམས་ཀྱི་མདུན་དག་ཏུ་ཡང་ཤིན་ཏུ་བླ་མའི་གནས་སུ་ཆེ། །འབྱུང་པོ་རྣམས་ནི་འཛིན་བྱེད་མཐའ་དག་རོ་ཡི་སྙིང་པོ་རབ་ཏུ་བསྐྲུན། །ལུས་ནི་མཛའ་གཅུགས་བཞིན་དུ་རྒྱས་བྱེད་སྐབས་གསུམ་ཆུ་བོ་གཙང་བའི་ལུས། །མ་དེ་ཉིད་ནི་རྟག་ཏུ་
