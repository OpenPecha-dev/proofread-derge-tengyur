བྱའི་ཕྱིར། །ཤ་རུས་ལ་སོགས་བཟའ་བར་བྱ། །ཛཿཧཱུཾ་བཾ་ཧོཿ། ཤྲཱི་ཧེ་རུ་ཀ་མ་ཧཱ་བཛྲ་ས་མ་ཡ། མ་ཧཱ་མཾ་ས། མ་ཧཱ་རཀྟ། མ་ཧཱ་ཙིཏྟ། མ་ཧཱ་བ་སུ་ཏ། མ་ཧཱ་པཉྩ་ཨ་མྲྀ་ཏ། མ་ཧཱ་ཨརྒྷཾ་པྲ་ཏཱིཙྪ་ཧོ། ཁ་ཁ་ཁ། ལེ་ལེ་ལེ། ཧཱུཾ་བྷྱོ་ཧཱུཾ་བྷྱོ་ཧཱུཾ་བྷྱོ། ཀོ་ཏ་ཤ། ཀོ་ཏ་པ། ཀོ་ཏ་ཀོ་ཏ།
བ་ས་པ། ཀི་ར་རཱ་ག །ཀོ་ལ་ཤ། ཀོ་ལ་བ། ཀོ་ལ་ཀོ་ལ་བ། ཀ་ལ་བ། ཤྚཐ། རྣམ་དག་སྒྲིབ་སྦྱང་སྲིད་གསུམ་སྐྱོན་བྲལ་ཞིང་། །སྟོན་ཀའི་ཟླ་གསལ་དྲི་མེད་ཞལ་མངའ་བ། །བཅོམ་ལྡན་བདེ་གཤེགས་དྲི་མེད་རྡོ་རྗེའི་དཀྱིལ་འཁོར་དབྱིངས། །དུས་གསུམ་རྒྱལ་བ་ཐམས་ཅད་བསྐྱེད་པའི་
ཡབ། །ཤེས་རབ་ལྷ་མོ་རིགས་ཀྱི་ཡུམ་རྣམས་དང་། །ཕྱག་རྒྱ་གཏུམ་ཆེན་རྔམ་པའི་ཚོགས་ལ་སོགས། །བསོད་ནམས་ཡེ་ཤེས་ཚོགས་ཀྱི་ཞིང་རྣམས་རྫོགས། །དགྱེས་ཤིང་རོལ་པའི་དུས་མཆོད་སྦྱར་བ་འདི། །བཞེས་ཏེ་ཨ་ལ་ལ་སྟེ་དགྱེས་པར་མཛོད། །ཨོཾ་བཛྲ་ཨི་ས་
ཧཱུཾ། བེ་ས་ཧཱུཾ། ས་མ་ཡ་བག་<ཤཧཱུཾ,ཤ་ཧཱུཾ>། རུ་ལུ་རུ་ལུ་ཧཱུཾ་བྷྱོ་ཧཱུཾ། ཛཿཧཱུཾ་བཾ་ཧོཿ། པྲ་ཏི་ཛཿཧཱུཾ་བཾ་ཧོཿ། ཡེ་ཤེས་མཉམ་ཉིད་ཆོས་སྐུ་དག་པ་ལས། །ལོག་རྟོག་དབང་གིས་སྒྲོ་དང་སྐུར་བཏབ་སྟེ། །བདེ་དང་སྡུག་བསྔལ་སྣ་ཚོགས་ངོ་བོར་སྨིན། །དེ་བཟློག་དོན་དུ་རྒྱལ་བའི་དམ་མནོས་པས། །རྒྱལ་

བྱེ་བ་སྟོང་ཕྲག་གི །འོད་ཀྱིས་ཕོག་པས་འཇིག་རྟེན་ཁམས་སུ་སྣང་བ་དང་། །ལུས་ངག་སེམས་དང་ཚེ་ནི་རྡོ་རྗེ་རབ་གནས་ཏེ། །རེ་སྨོན་ཐམས་ཅད་དུས་འདིར་གྲུབ་པར་མཛད་དུ་གསོལ། །ཡེ་ཤེས་སེམས་དཔའི་སྔགས་འཁོར་ལོ། །ལྷ་རྣམས་ཀུན་གྱི་གནས་ཞུགས་པ། །ཡན་
