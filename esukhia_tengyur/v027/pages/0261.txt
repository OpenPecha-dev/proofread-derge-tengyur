རྒྱ་མཆོག་རྣམས་དང་། །ཁ་ར་སྦྲང་རྩི་བྱུག་པ་ཡིས། །ཨ་མྲ་ལ་སོགས་ཚི་གུས་བྱུག །ཧ་ས་ཀ་རའི་མར་ལ་སོགས། །མཉམ་པའི་དྲི་ཡིས་བསྐུ་བར་བྱ། །ཡུངས་དང་རྒྱ་སྐྱེགས་ལ་སོགས་པའི། །ཚོན་གྱིས་མཐུན་པར་ཁ་བསྒྱུར་ནས། །ཤུན་པ་དྲོ་བ་ལ་
སོགས་བྱས། །མཆོད་པའི་ཚེ་ན་དབུལ་བར་བྱ། །ཐོད་པའི་ནང་དུ་སྦྱར་བྱས་པའི། །བཟའ་བཏུང་བཅའ་བ་ལ་སོགས་རྣམས། །ཧ་ཧོཿཧྲཱིཿཡིས་སྲེག་སྦྱང་བརླབ། །མཁའ་འགྲོའི་འཁོར་ལ་བསྔོ་བར་བྱ། །གཏོར་མ་རྫས་ཆེན་མ་ལུས་པ། །དང་པོ་དྲི་ཡི་ཆུས་སྦྱངས་ཤིང་། །
སྣ་ཚོགས་སྒྲ་སྙན་བ་དན་སོགས། །བཀོད་པའི་མཆོད་གཏོར་བརླབས་ལ་སྤེལ། །སངས་རྒྱས་རྡོ་རྗེ་སེམས་དཔའ་ཉིད། །རྟ་མཆོག་གཟུགས་སུ་སྤྲུལ་བྱས་ནས། །མཛེས་པའི་ཕྱག་རྒྱ་སྔགས་ལྡན་པ། །མཁའ་འགྲོའི་ཚོགས་རྣམས་བསྡུ་བར་བྱ། །ཨོཾ་བཛྲ་ཊཱ་ཀི་ནཱི་བྷྱོ། །ཨེ་ཨ་ར་
ལི་ཛ་ཛ། འཁོར་བཅས་བསྡུ་བར་འོངས་པ་དང་། །དམ་ཚིག་ཕྱག་རྒྱ་རྣམ་གསུམ་བསྟན། །ཨོཾ་བཛྲ་གུ་ཧྱེ་དྲྀ་ཤྱ་ཧོ། །ས་མ་ཡ་སྟྭཾ། ཨོཾ་བཛྲ་གུ་ཧྱེ་དྲྀ་ཤྱ་ཧོ། །ས་མ་ཡ་སྟྭཾ། ཨོཾ་བཛྲ་གུ་ཧྱེ་དྲྀ་ཤྱ་ཧོ། །ས་མ་ཡ་ཕཊ། ཨོཾ་ཨརྒྷཾ་པུཥྤེ་དྷཱུ་པེ་གནྡྷེ་པཱུ་ཛཱ་སྥ་ར་ཎ་ཨི་དཾ་བ་ལིཾ་ཁ་ཁ་ཁཱ་ཧི། ཡཾ་
ལས་སྐྱེས་པའི་གཞུ་འདྲའི་དབྱིབས། །བ་དན་གཡོ་བས་མཚན་པ་སྟེ། །རཾ་བྱུང་མེ་ཡི་དཀྱིལ་འཁོར་འབར། །དམར་པོ་གྲུ་གསུམ་རཾ་གྱིས་མཚན། །ཀཾ་བྱུང་ཐོད་པ་མཚན་ལྡན་ནང་། །འོད་གསལ་ཆགས་མེད་གནས་པ་ཡི། །མི་ཤ་ལ་སོགས་ཕྱོགས་དང་མཚམས། །ཨ་ལས་
བྱུང་བའི་ཟླ་བ་དང་། །ཧཱུཾ་བྱུང་རྡོ་རྗེས་བཀབ་བྱས་ནས། །ཧཱུཾ་ཨོཾ་ཧྲཱིཿཏྲཱཾ་བཞི་རྣམས་སོ། །ཨོཾ་ཨཱཿཧཱུཾ། སརྦ་པཉྩ་ཨ་མྲྀ་ཏ་ཧཱུཾ་ཧྲཱིཿཐཱ། རིམ་བཞིན་སྔགས་དང་ཕྱག་མཚན་དང་། །སྐུར་འགྱུར་འཕྲོ་དང་འདུ་བ་ཡིས། །ཡེ་ཤེས་བདུད་རྩི་དགུག་བྱ་སྟེ། །རྡོ་རྗེ་བཞུགས་ནས་སྣོད་བཟང་
བལྟམས། །ཟླ་འདྲ་འོད་ཟེར་ལྔ་བཅས་པས། །དཀྱིལ་འཁོར་འཁོར་བཅས་ཚོགས་མཉེས་བྱ། །ཨོཾ་འཇིག་རྟེན་ཀུན་གཙོ་མངའ་མཛད་པའི། །འཇིགས་བྱེད་དྲེགས་བྱེད་སྟོབས་ལྡན་པ། །བདེ་གཤེགས་སྲས་སོགས་བྲན་མཛད་པའི། །བྱང་ཆུབ་སེམས་ལྡན་སྙིང་རྗེ་ཅན། །
