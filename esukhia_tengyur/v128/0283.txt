ཀྱི་སོ་སོ་རང་གི་རིག་པ་དང་། རྗེས་སུ་དཔག་པའི་བྱ་བ་ཡང་དེ་ཡིན་ལ། གྲགས་པའི་ཆོས་ཉིད་ཀྱང་དེ་ཡིན་ཏེ། དེ་ལྟར་བྱ་བ་བྱེད་པའི་རིགས་པ་དང་། འཐད་པ་སྒྲུབ་པའི་རིགས་པ་དང་། ཆོས་ཉིད་ཀྱི་རིགས་པས་ཡོངས་སུ་ཚོལ་བར་བྱེད་དོ། །དེ་ལ་ཁམས་རབ་ཏུ་དབྱེ་བ་ལ་
དམིགས་པ་ལྷག་མཐོང་ལ་རབ་ཏུ་བརྩོན་པ། ཇི་ལྟར་དོན་ཡོངས་སུ་ཚོལ་བར་བྱེད་ཅེ་ན། ཁམས་ཀྱི་དོན་ནི་རིགས་ཀྱི་དོན་དང་། ས་བོན་གྱི་དོན་དང་། རྒྱུའི་དོན་དང་། རང་བཞིན་གྱི་དོན་ཏོ་ཞེས་དོན་སོ་སོར་རྟོག་པར་བྱེད་དེ། དེ་ལྟར་དོན་ཡོངས་སུ་ཚོལ་བར་བྱེད་དོ། །ས་ལ་སོགས་པ་ཁམས་དྲུག་
པོ་དག །ནང་གི་དང་ཕྱི་རོལ་གྱི་ཡིན་པར་མོས་པར་བྱེད་པ་ནི་དངོས་པོ་ཡོངས་སུ་ཚོལ་བར་བྱེད་པ་ཡིན་ནོ། །སའི་མཚན་ཉིད་ནི་སྲ་བའོ་ཞེས་བྱ་བ་ནས། རླུང་གི་མཚན་ཉིད་ནི་ཡང་ཞིང་གཡོ་བའོ། །རྣམ་པར་ཤེས་པའི་མཚན་ཉིད་ནི་རྣམ་པར་ཤེས་པར་བྱེད་པའོ། །ནམ་མཁའི་ཁམས་ནི་
སྦུབས་ཅན་གྱི་མཚན་ཉིད་དང་། གཟུགས་ཀྱི་རྣམ་པ་གསལ་བའི་མཚན་ཉིད་དོ་ཞེས་བྱ་བའི་བར་དུ་དེ་ལྟར་རང་གི་མཚན་ཉིད་ཡོངས་སུ་ཚོལ་བར་བྱེད་དོ། །ཁམས་དེ་དག་ཐམས་ཅད་ནི་མི་རྟག་པ་ཉིད་དང་། བདག་མེད་པ་ཉིད་ཀྱི་བར་གྱིས་མཉམ་ཞིང་མཚུངས་པ་ཡིན་ནོ་ཞེས་དེ་ལྟར་
སྤྱིའི་མཚན་ཉིད་ཡོངས་སུ་ཚོལ་བར་བྱེད་དོ། །ཁམས་སྣ་ཚོགས་མི་ཤེས་ཤིང་རིལ་པོར་འདུ་ཤེས་པ། ཁམས་སྣ་ཚོགས་ཅན་གྱི་ལུས་འདིས་ཁེངས་ཤིང་རློམ་སེམས་སུ་བྱེད་པ་གང་ཡིན་པའི་ཆོས་དེ་ནི། ཉོན་མོངས་པ་དང་བཅས་པ་སྟེ། ནག་པོའི་ཕྱོགས་ཡིན་ནོ། །དེ་ལས་བཟློག་པ་
ནི་ཉོན་མོངས་པ་མེད་པ་སྟེ། དཀར་པོའི་ཕྱོགས་ཡིན་ནོ་ཞེས་དེ་ལྟར་ཕྱོགས་ཡོངས་སུ་ཚོལ་བར་བྱེད་དོ། །འདས་པ་དང་། མ་འོངས་པ་དང་། ད་ལྟར་བྱུང་བའི་དུས་རྣམས་སུ་ཁམས་དྲུག་པོ་དག་ལ་བརྟེན་ཏེ། མའི་ལྟོ་བར་མངལ་དུ་འཇུག་པར་འགྱུར་རོ་ཞེས་དེ་ལྟར་དུས་ཡོངས་སུ་ཚོལ་བར་
བྱེད་དོ། །འདི་ལྟ་སྟེ། དཔེར་ན། རྩྭ་དང་། ཤིང་གི་རྐྱེན་ལ་བརྟེན་ཏེ། ནམ་མཁའ་ཀུན་ནས་དཀྲིས་ན། ཁང་པ་ཁང་པ་ཞེས་བྱ་བའི་གྲངས་སུ་འགྲོ་བ་དེ་བཞིན་དུ། ཁམས་དྲུག་པོ་དག་ལ་ཡང་བརྟེན། རུས་པ་དང་། ཆུ་རྒྱུས་དང་། པགས་པ་དང་། ཤ་དང་། ཁྲག་གི་རྐྱེན་ལ་ཡང་
