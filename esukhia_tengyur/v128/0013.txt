གློ་གཡས་པ་ཕབ་ནས་རྐང་པ་རྐང་པའི་སྟེང་དུ་བཞག་སྟེ། སྣང་བའི་འདུ་ཤེས་དང་། དྲན་པ་དང་ཤེས་བཞིན་དང་། ལྡང་བ་ཁོ་ནའི་འདུ་ཤེས་ཡིད་ལ་བྱེད་བཞིན་དུ་ཉལ་བར་བྱེད་པ་དང་། དེ་མཚན་མོའི་ཐུན་ཐ་མ་ལ་ཡང་མྱུར་བ་མྱུར་བ་ཁོ་ནར་སད་པར་བྱས་ནས། འཆག་པ་དང་། འདུག་
པ་དག་གིས་སྒྲིབ་པར་གྱུར་པའི་ཆོས་རྣམས་ལས་སེམས་ཡོངས་སུ་སྦྱོང་བར་བྱེད་པ་ཡིན་ཏེ། དེ་ནི་ནམ་གྱི་ཆ་སྟོད་དང་ནམ་གྱི་ཆ་སྨད་ལ་མི་ཉལ་བར་སྦྱོར་བའི་རྗེས་སུ་བརྩོན་པ་ཉིད་ཅེས་བྱའོ། །ཤེས་བཞིན་དུ་སྤྱོད་པ་ཉིད་གང་ཞེ་ན། དེ་ལྟར་མི་ཉལ་བར་སྦྱོར་བའི་རྗེས་སུ་བརྩོན་པར་གྱུར་པ་
དེ་འགྲོ་བ་དང་ལྡོག་པ་ལ་ཤེས་བཞིན་དུ་སྤྱོད་པར་བྱེད་པ་དང་། ཅུང་ཟད་བལྟས་པ་དང་། རྣམ་པར་བལྟས་པ་དང་། བསྐུམ་པ་དང་། བརྐྱང་བ་དང་། སྣམ་སྦྱར་དང་། ཆོས་གོས་དང་། ལྷུང་བཟེད་བཅང་བ་དང་། ཟོས་པ་དང་། འཐུངས་པ་དང་། འཆོས་པ་དང་། མྱངས་པ་དང་།
སོང་བ་དང་། སྡོད་པ་དང་། འདུག་པ་དང་། ཉལ་བ་དང་། མི་ཉལ་བ་དང་། སྨྲ་བ་དང་། མི་སྨྲ་བ་དང་། གཉིད་ཀྱིས་སྙོམ་པ་བསང་བ་ལ་ཡང་ཤེས་བཞིན་དུ་སྤྱོད་པ་ཡིན་ཏེ། དེ་ནི་ཤེས་བཞིན་དུ་སྤྱོད་པ་ཉིད་ཅེས་བྱའོ། །རབ་ཏུ་དབེན་པར་གནས་པ་གང་ཞེ་ན། དེ་ཆོས་
དེ་དག་གིས་ཡོངས་སུ་སྦྱོང་བར་བྱ་བའི་ས་སྦྱངས་ནས་གནས་མལ་དབེན་པ་དགོན་པ་དག་གམ། ཤིང་དྲུང་དག་གམ། ཁང་སྟོང་དག་གམ། རི་ཁྲོད་དམ། རི་ཕུག་གམ། རྩྭ་ཁང་ངམ། བླ་གབ་མེད་པའམ། དུར་ཁྲོད་དམ། ནགས་འདབ་བམ། བས་མཐའི་གནས་མལ་
དག་ཏུ་གནས་པར་བྱེད་པ་ཡིན་ཏེ། དེ་ནི་རབ་ཏུ་དབེན་པར་གནས་པ་ཞེས་བྱའོ། །སྒྲིབ་པ་རྣམ་པར་དག་པ་གང་ཞེ་ན། དེ་དགོན་པ་ན་འདུག་ཀྱང་རུང་། ཤིང་དྲུང་ན་འདུག་ཀྱང་རུང་། ཁང་སྟོང་ན་འདུག་ཀྱང་རུང་སྟེ། སྒྲིབ་པ་ལྔ་པོ་འདོད་པ་ལ་འདུན་པ་དང་། གནོད་སེམས་དང་། རྨུགས་
