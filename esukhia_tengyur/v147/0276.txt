བས་བཅུ་དྲུག་ཏུ་འགྱུར་རོ། །རེ་ཞིག་དེ་དག་ནི་ཚ་བའི་དམྱལ་བ་བརྒྱད་ཡིན་ནོ། །ཆུ་བུར་ཅན་སོགས་གྲང་བརྒྱད་གཞན། །འདི་ལྟ་སྟེ། ཆུ་བུར་ཅན་དང་། ཆུ་བུར་རྡོལ་བ་དང་། སོ་ཐམ་ཐམ་པ་དང་། ཨ་ཆུ་དང་། ཀྱི་ཧུད་དང་། ཨུཏྤལ་ལྟར་གས་པ་དང་། པདྨ་ལྟར་གས་
པ་དང་། པདྨ་ལྟར་ཆེར་གས་པའོ། །མིང་འདི་རྣམས་ནི་སེམས་ཅན་གྲང་བ་མི་བཟད་པས་ཉེན་པ་དེ་དག་གི་ལུས་དང་སྒྲ་འགྱུར་བ་དང་མཐུན་པ་དག་ཡིན་ནོ། །ཉི་མ་དང་ཟླ་བ་འདི་ནས་ཐག་ཇི་སྲིད་ཅེ་ན། ཉི་ཟླ་ལྷུན་པོའི་ཕྱེད་ནའོ། །རི་གཉའ་ཤིང་འཛིན་གྱི་རྩེ་མོ་དང་མཉམ་
པར་རྒྱུའོ། །ཚད་ཀྱང་། དཔག་ཚད་ལྔ་བཅུ་གཅིག་དང་བཅས། །རིམ་པ་བཞིན་དུའོ། །གླིང་བཞི་པོ་དག་ཏུ་ཉི་མས་ཅིག་ཅར་བྱ་བ་བྱེད་དམ་ཞེ་ན། མ་ཡིན་ཏེ། འོན་ཀྱང་། ནམ་ཕྱེད་ཉི་མ་ནུབ་པ་དང་། །ཉི་མ་ཕྱེད་དང་འཆར་དུས་གཅིག །གང་གི་ཚེ་བྱང་སྒྲ་མི་སྙན་དུ་ནམ་ཕྱེད་པ་
དེའི་ཚེ་ཤར་ལུས་འཕགས་སུ་ཉི་མ་ནུབ་བོ། །འཛམ་བུའི་གླིང་དུ་ཉི་མ་ཕྱེད་དོ། །བ་ལང་སྤྱོད་དུ་ཉི་མ་ཤར་ཏེ། དེ་བཞིན་དུ་གཞན་དག་ལ་ཡང་སྦྱར་བར་བྱའོ། །ཉི་མ་འགྲོ་བའི་བྱེ་བྲག་ལས་ཉིན་མོ་དང་མཚན་མོ་རྣམས་རིང་ཐུང་དུ་འགྱུར་ཏེ། དེ་ལ་ཡང་། དབྱར་གྱི་ཟླ་བ་གཉིས་
པ་ཡི། །ཐ་མའི་ཚེས་དགུ་ནས་མཚན་རིང་། །དབྱར་ཟླ་ཐ་{𰶉ཆུངས་,ཆུང་}ཡར་གྱི་ངོའི་ཚེས་དགུ་ནས་མཚན་མོ་རིང་ངོ་། །དགུན་རྣམས་ཀྱི་ནི་བཞི་པ་ལ། །ཐུང་ངུ་འགྱུར་རོ། །དགུན་ཟླ་ཐ་{𰶊ཆུངས་ཀྱི་,ཆུང་གི་}ཕྱོགས་གཉིས་པའི་ཚེས་དགུ་ནས་སོ། །ཉིན་མོ་བཟློག །གང་གི་ཚེ་ཉིན་མོ་རིང་བ་
དེའི་ཚེ་མཚན་མོ་ཐུང་བར་འགྱུར་རོ། །ཇི་ཙམ་ཞིག་གིས་རིང་བར་འགྱུར་ཞེ་ན། ཉིན་མཚན་རིང་བ་ཐང་གཅིག་གོ། །དེ་དག་ཀྱང་{𰶋གོ་རིམས་,གོ་རིམ་}བཞིན་དུ། ཉི་མ་ལྷོ་བྱང་འགྲོ་བའི་ཚེའོ། །དཀར་པོའི་ཕྱོགས་ཀྱི་དང་པོ་ཟླ་བའི་དཀྱིལ་འཁོར་མ་རངས་པར་སྣང་བ་དེ་ལ་རྒྱུ་ཅི་ཞེ་ན། ཉི་
མ་དང་ནི་ཉེ་བ་ལས། །རང་གི་གྲིབ་མས་བསྒྲིབས་པར་སྣང་། །གང་གི་ཚེ་ཟླ་བའི་གཞལ་མེད་ཁང་ཉི་མའི་གཞལ་མེད་ཁང་དང་ཉེ་བར་རྒྱུ་བ་དེའི་ཚེ་ཉི་མའི་འོད་དེ་གཞལ་མེད་ཁང་ལ་འབབ་སྟེ། དེས་ན་དེའི་ཕ་རོལ་གྱི་ངོས་སུ་གྲིབ་མ་འབབ་པས་ཟླ་བའི་དཀྱིལ་འཁོར་མ་རངས་
