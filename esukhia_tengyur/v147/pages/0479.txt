བཅོས་ལ་སྒྲོན་མ་ཞེས་བྱའོ། །ཡང་ན་ཆོས་མངོན་པའི་སྒྲོན་མ་མཛད་པ་ལ་སྒྲོན་མ་མཛད་པ་ཞེས་བྱའོ། །སྒྲོན་མ་མཛད་པས་སྒྲོན་མ་མཛད་དེ། དེའི་ཕྱིར་སྒྲོན་མ་མཛད་པ་དེ་ལ་ཕྱག་འཚལ་ལོ། །དེས་ལོག་པར་བཤད་པའི་མུན་པ་བསལ་བས་སྒྲོན་མ་དང་འདྲ་བའི་ཕྱིར་སྒྲོན་མ་ཞེས་བྱ་སྟེ། ཆོས་ཀྱི་ངོ་
བོ་ཉིད་མ་ནོར་བར་བསྟན་པ་བསྟན་བཅོས་མཛད་པ་དེ་ལ་ཤིན་ཏུ་གུས་པར་གྱུར་ནས་བསྟོད་དོ། །དེ་ལྟར་ཚིགས་སུ་བཅད་པ་གསུམ་གྱིས་ཇི་ལྟར་འདོད་པ་རྣམས་ལ་ཕྱག་འཚལ་ནས། ལ་ལ་དག་ནི་ཞེས་བྱ་བ་ལ་སོགས་པའི་རབ་ཏུ་བྱེད་པ་རྩོམ་མོ། །བློ་ཆུང་བ་དང་འབྲིང་དང་ཆེན་པོ་ཐ་དད་པའི་ཕྱིར་
མིང་མ་སྨོས་སོ། །ནི་ཞེས་བྱ་བ་ནི་གང་གི་ཕྱིར་ཞེས་བྱ་བའི་དོན་ཏོ། །ཐུབ་པའི་དབང་པོ་ནི་ཐུབ་པ་རྣམས་ཀྱི་དབང་པོ་ཡིན་པས་ཐུབ་པའི་དབང་པོའོ། །ཟག་པ་མེད་པའི་ལུས་དང་ངག་དང་ཡིད་ཐུབ་པ་ཡོད་པའི་ཕྱིར་ཉན་ཐོས་རྣམས་དང་། རང་སངས་རྒྱས་རྣམས་ནི་ཐུབ་པ་ཞེས་བྱ་བར་རྣམ་པར་གཞག་གོ། །
བཅོམ་ལྡན་འདས་ནི་ཆོས་ཐམས་ཅད་ལ་མངའ་བརྙེས་པའི་ཕྱིར་དེ་དག་གི་རྗེ་དང་རྒྱལ་པོ་ཡིན་ནོ། །ཇི་སྟེ་ཐུབ་པ་དག་གི་རྣམ་པར་གཞག་པ་གང་ཡིན་པ་བསྟན་ཀྱང་གང་གིས་ཐུབ་པ་དེ་དག་རྣམ་པར་གཞག་པ་དེ་མི་ཤེས་སོ་ཞེ་ན། ཆོས་ཉིད་ཀྱིས་སོ། །འདིར་ནི་ཆོས་ཉིད་ཅེས་བྱ་བ་འཕགས་པའི་
ལམ་ལ་བྱའོ། །འདི་ནི་ཐུབ་པའི་དབང་པོའི་ཡིན་པས་ཐུབ་པའི་དབང་པོའོ། །དེ་གང་ཞེ་ན། གསུང་རབ་སྟེ་གསུང་རབ་ཏུ་གཏོགས་པ་ཡན་ལག་བཅུ་གཉིས་སོ། །མཐའ་དག་ནི་མ་ལུས་པའོ། །དོན་དང་ཚིག་ཅེས་བྱ་བ་ལ་གོང་མས་ནི་བཤད་པ་བཟུང་བ་སྟོན་ཏོ། །འོག་མས་ནི་ཡི་
གེའོ། །ཡང་ན་དོན་གྱིས་ནི་ངེས་པར་ལེགས་པ་བསྒྲུབ་པའོ། །ཚིག་གིས་ནི་རང་དང་སྤྱིའི་མཚན་ཉིད་ཁོང་དུ་ཆུད་པའོ། །སྐལ་པ་ཡོད་པའི་ངང་ཚུལ་ཅན། །ཞེས་བྱ་བ་ནི་ནུས་པར་སྟོན་ཏོ། །སྐལ་བ་ཉིད་སྐལ་བ་ཡོད་པའི་ངང་ཚུལ་ཅན་ཏེ། ལེན་པ་དང་འཛིན་པ་དང་ཁོང་དུ་ཆུད་པར་བྱེད་ནུས་
ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དགེ་བ་དག་ཅེས་བྱ་བ་ནི་སྐྱེས་བུ་དམ་པ་སྟེ་ཚུལ་བཟང་པོ་ལ་མངོན་པར་དགའ་བ་སོ་སོའི་སྐྱེ་བོ་དགེ་བ་དག་གོ། །ཡིན་ཞེས་བྱ་བ་ནི་དེ་ལྟར་གྱུར་པའོ། །ཟིན་ཀྱང་ཞེས་བྱ་བ་ནི་སྲིད་པ་སྟེ་ཕྱི་རོལ་གྱི་རྐྱེན་གྱི་སྤྱོད་པ་སྟོན་ཏོ། །ཇི་སྐད་དུ་འཇིག་རྟེན་གྱི་མགོན་པོས་ཡང་དག་པའི་ལྟ་བ་སྐྱེ་
