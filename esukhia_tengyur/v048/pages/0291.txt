པ་དྲུག་ཏུ་བྱ་བ་སྟེ། །རྫས་དང་སྔགས་དང་ཏིང་ངེ་འཛིན། །རླུང་དང་མཉམ་མེད་དག་དང་ནི། །སྲུང་བ་མེད་པའི་བསྲུང་བའོ། །དེ་ལ་རེ་ཞིག་རྣལ་འབྱོར་པས། །སེང་ལྡེང་དབྱུག་པ་གླ་རྩི་དང་། །གི་ཝཾ་ཤུ་དག་གུ་གུལ་དང་། །གནྡྷི་ཏིལ་དང་ཡུངས་ཀར་དང་། །མུ་མེན་བྱི་རུ་
སི་ལ་དང་། །ཙནྡན་དམར་སོགས་དེ་བཞིན་དུ། །ལི་ཁྲི་མཚལ་དཀར་འདུས་བྱས་ལ། །སྲུང་བ་མེ་ལས་བཅུ་གསུམ་ལྟར། །བསྐྱེད་ལ་བཅུ་གསུམ་གནས་སུ་གཞག །ཧཱུཾ་ཀྵེ་མ་མེ་ད་ཛ་ས་དོ་རུ་ཡཙྪ་ནི་ར། སྙིང་ག་མིག་དང་རྣ་བ་སྣ། །ཁ་དང་ལྟེ་བ་འོག་སྒོ་གཉིས། །བརླ་གཉིས་ཕྲག་
པ་གཉིས་དང་ནི། །དེ་བཞིན་སྤྱི་བོའི་གཙུག་ལ་རེག །ཇི་ལྟར་ཞེ་སྡང་གཤེད་སོགས་བསྒོམ། །རྫས་རྣམས་བརྡུངས་ཏེ་ལུས་ལ་བཅད། །ཚངས་དང་ཁྱབ་འཇུག་དྲང་སྲོང་སོགས། །ལོག་པར་བྱེད་པའི་རིག་སྔགས་ཀྱིས། །ནམ་ཡང་ཚུགས་པར་མི་འགྱུར་རོ། །ཕྱི་ཡོ་བྱད་རྫས་ཀྱིས་
ཕྱིར་བཟློག་པའོ། །ཤིན་ཏུ་ངན་པར་བྱེད་པ་ཡི། །རིག་འཛིན་ལ་སོགས་མེད་གྱུར་ན། །དེ་བཟློག་པ་ཡི་ཆོ་ག་ནི། །ཚད་མེད་སེམས་སོགས་སྔོན་འགྲོ་བས། །གསང་སྔགས་གསུམ་གྱིས་དག་པར་བྱ། །ཨོཾ་ཤཱུ་ནྱ་ཏཱ་ཛྙཱ་ན་བཛྲ་སྭ་བྷཱ་བ་ཨཱཏྨ་ཀོ྅ཧཾ། ནམ་མཁའི་དཀྱིལ་དུ་བསམ་བྱ་
བ། །པཾ་རཾ་ལས་བྱུང་པད་ཉིའི་སྙིང་། །སྣ་ཚོགས་རྡོ་རྗེ་ཡོངས་གྱུར་པའི། །སྲུང་བའི་འཁོར་ལོ་བསྒོམ་བྱ་སྟེ། །རྩིབས་བཅུ་མུ་ཁྱུད་ལྟེ་བར་བཅས། །བྷྲཱུཾ་ལས་བྱུང་བའི་གནས་བཅུ་ལ། །ཧཱུཾ་བྱུང་རྒྱུ་དང་བསྲེ་བ་ལས། །ཇི་ལྟར་འབྱུང་བའི་ཁྲོ་བོ་བསྐྱེད། །དབུས་སུ་རང་ཉིད་གཤིན་རྗེ་གཤེད། །
མདོག་གནག་ཁྲོས་པ་མ་ཧེའི་སྟེང་། །དེ་ལྟར་བསྐྱེད་ནས་ཆོ་ག་བཞིན། །རྡོ་རྗེ་གསུམ་དུ་བྱ་བ་དང་། །གཞུག་དང་བཅིང་སོགས་ལེགས་བྱས་ནས། །ཌཀྐི་རཱ་ཛ་གྲངས་མེད་པས། །གནོད་བྱེད་གདུང་ཞིང་དགུག་ལ་གཞུག །རྡོ་རྗེའི་གོ་གོན་སྙིང་ག་ཡི། །སེམས་དཔའི་གནས་སུ་
བསྲུང་བྱ་ནི། །དམིགས་ལས་བཟླས་ཏེ་འཁོར་བས་སོ། །དུས་དུས་གཏོར་མ་སྦྱིན་པར་བྱ། །གཟིར་བའི་སྦྱིན་སྲེག་མཚན་ཕྱེད་ན། །ཁྲོ་བོ་བཅུ་ལ་སྦྱིན་པར་བྱ། །ཕྱིའི་བྱད་སྔགས་ཀྱིས་ཕྱིར་བཟློག་པའོ། །གལ་ཏེ་རྣལ་འབྱོར་ལམ་པ་ལ། །རྟོག་དང་མཚན་མས་གཡེངས་པའི་བྱད། །
