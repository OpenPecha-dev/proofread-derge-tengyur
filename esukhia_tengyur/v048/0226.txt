སྦལ་རྒྱབ་འདྲའི་སེན་མོ་དམར། །ནུ་མ་མཁྲེགས་ཤིང་འཇམ་པ་དང་། །རྐང་མཐིལ་མཉམ་ནས་བྲང་ཡང་མཛེས། །བ་སྤུ་འཁྱིལ་ཞིང་ངང་པའི་སྐད། །རི་མོ་གསུམ་ཅན་ཆུ་སྐྱེས་མཆོག །པདྨའི་དྲི་ལྡན་གླང་ཆེན་འགྲོས། །པདྨའི་འདབ་ལྟར་རེག་ན་འཇམ། །པདྨའི་བྱེད་པས་བསྟེན་ཕྱིར་
སྟེ། །པདྨ་ཅན་གྱི་དངོས་གྲུབ་འགྱུར། །གླང་པོ་ཅན་ཞེས་བྱ་བ་ནི། །སྣ་གཡོན་བ་སྤུ་འཁྱིལ་བ་དང་། །འདོད་པས་མྱོས་ཤིང་འདོད་ལ་སྲེད། །ཆང་གི་དྲི་ལྡན་བྱིན་པ་སྦོམ། །ལུས་ནི་སྦོམ་ལ་མི་སྲུན་པ། །བཞིན་གྱི་མདངས་ཆེ་ང་རོ་ཆེ། །གླུ་དང་རོལ་མོ་ལ་དགའ་བ། །བྱེད་པ་བཅིངས་པའི་
<སྤྱོར་,སྦྱོར་>བ་ཡིས། །གླང་པོ་ཆེ་ཡི་མཆོག་སྒྲུབ་བྱེད། །དུང་ཅན་མ་ཞེས་གྲགས་པ་ནི། །ཧ་ཅང་སྐེམ་ཞིང་སྦོམ་པ་མིན། །སྐྲ་རིང་སྣ་ཡང་ཅུང་ཟད་རིང་། །ནུ་མ་ན་ར་ཀ་འབྲས་ལྟར། །ཞོ་དང་འོ་མའི་ཚོགས་ལ་དགའ། །བྱ་རོག་སྐད་ཅན་བུང་བའི་སྒྲ། །བ་ལང་ལྕེ་ལྟར་རྩུབ་པ་སྟེ། །གི་ཝང་ར་ཏའི་
བཅིངས་པས་བསྟེན། །རི་མོ་ཅན་ཡང་བཤད་པར་བྱ། །ཤིན་ཏུ་ཁྲོ་ཞིང་ངོ་ཚ་སྤངས། །རྟག་ཏུ་རྩོད་པ་ལ་དགའ་ཞིང་། །ལུས་ཐུང་དེ་ཡི་བྲང་རྣམས་མཛེས། །པ་ལའི་འབྲས་ལྟར་ནུ་མ་ཟབ། །བྱིན་པ་ཕྲ་ཞིང་ཕྱིས་གན་ཉལ། །གསུས་པ་སྦོམ་ཞིང་མཆུ་ཡང་འཕྱང་། །ཤ་བའི་དྲི་ལྡན་ཕུག་
རོན་སྐད། །དེ་ནི་རི་མོ་ཅན་ཞེས་བྱ། །པི་ལ་ས་ཡི་བཅིངས་པས་བསྟེན། །རི་མོ་ཅན་གྱི་དངོས་གྲུབ་འགྱུར། །བཅུ་དྲུག་ལོན་པའི་སྐད་བཟང་མ། །གཟུགས་དང་ལང་ཚོ་ཕུན་སུམ་ཚོགས། །དང་པོར་ལེགས་པར་སྦྱང་བ་དང་། །དབང་བསྐུར་ནས་ནི་སྒྲུབ་པ་བརྩམ། །སྐུ་གསུང་ཐུགས་དང་རྡོ་རྗེ་
ནི། །དཔྲལ་བ་ལ་སོགས་རྣམས་སུ་དགོད། །སྭཱ་ཧཱ་གསང་བ་གཙུག་ཏོར་དུའོ། །དེ་ནས་པདྨ་རྡོ་རྗེ་སྦྱང་། །ཨ་ཡིས་པདྨའི་འདབ་བརྒྱད་ལ། །ཨ་ཡིས་ཟེ་འབྲུ་རབ་ཏུ་བརྟག །ཧཱུཾ་གིས་རྡོ་རྗེ་རྩེ་ལྔ་པ། །ཕཊ་ཀྱིས་སྒོ་ལ་རབ་ཏུ་བརྟག །དེ་ལྟར་རྡོ་རྗེ་པདྨ་བཅས། །བྷ་གར་ལིངྒ་བཞག་ནས་ནི། །བྱང་ཆུབ་
སེམས་ནི་སྤྲོ་མི་བྱ། །རྣལ་འབྱོར་པས་ནི་ཧཱུཾ་ཕཊ་བསམ། །བྷ་གར་རྡོ་རྗེ་འཛིན་པར་བྱེད། །ཁམས་གསུམ་པོ་ནི་མ་ལུས་པ། །སངས་རྒྱས་ངོ་བོར་བསྒོམ་པར་བྱ། །རྒྱུ་མཐུན་ལ་སོགས་རབ་དབྱེ་བས། །ཤིན་ཏུ་རྣལ་འབྱོར་བསྒོམ་པར་བྱ། །རྡོ་རྗེ་དབྱིངས་ཀྱི་དབང་ཕྱུག་མའི། །རྩ་ནི་ཉེ་བར་བཙལ་བྱ་སྟེ། །
