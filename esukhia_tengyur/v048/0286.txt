ཕྲེང་རིལ་བ་དབྱུག་པ་ཅན། །དྲང་སྲོང་ནག་པོས་བསྐོར་ཞིང་གནག །དེ་ནས་དགུག་ཅིང་སྤྱན་དྲངས་ཏེ། །ཨོཾ་ཨཱཿཧཱུྃ། ཨོཾ་ཨེ་ཧྱེ་ཧི་མ་ཧཱ་བྷཱུ་ཏ་དེ་བ་རྀ་ཥི་དྭི་ཛ་ཛཿཧཱུྃ་བཾ་ཧོཿ ཨོཾ་ཨགྣེ་ཀྲོ་དྷ་དེ་ཛ་ཕཊ། ཨོཾ་བཛྲ་ནི་རི་ཏི་ཁཾ་ཕཊ། ཨོཾ་བཛྲ་ཨརྒྷཾ་ཕཊ། ཨོཾ་བཛྲ་པུཥྤེ་ཕཊ། ཨོཾ་བཛྲ་དྷཱུ་པེ་ཕཊ། ཨོཾ་བཛྲ་
ཨཱ་ལོ་ཀེ་ཕཊ། ཨོཾ་བཛྲ་ནཻ་བི་དྱཻ་ཕཊ། ཨོཾ་བཛྲ་ཤཔྟེ་ཕཊ། མཆོད་པའོ། །ཛི་ཨ་ཀུ་རུ་ཀུ་རུ། པུཥྤ། དྷཱུ་པ། དཱི་པ། གནྡྷ། ནཻ་བི་དྱཻ། ཤཔྡ། ཨོཾ་དང་སྭཱ་ཧཱས་ཐོག་མཐར་བརྒྱན། །སྦྱང་བའོ། །མཆུ་གཉིས་ཡ་ར་དེ་བཞིན་དུ། །མེ་ལྷའི་ཁ་ལ་རབ་བཀོད་ལ། །པུས་མོའི་བར་བཅུག་ལག་
པ་ཡི། །གསུམ་མམ་བདུན་དུ་གཡས་ནས་དབུལ། །ཨགྣེ། སརྦ་པཱ་པཾ་ད་ཧ་ན། བཛྲ་ཡ། བཛྲ་པུ་ཤ་ཡ། སརྦ་པཱ་པཾ་བྷ་ཝ་ཏེ། ཨ་པྲ་ཏི་ཧ་ཏ་ཡ། ཧ་བི་ཀ་ཡ། མ་ཧཱ་བ་ལ། བཛྲ་གྷ་སྨ་རི། བོ་དྷི་བྲྀ་ཀྵ་ཡ། ཨོཾ་དང་སྭཱ་ཧཱས་ཐོག་མཐར་བརྒྱན། །ཨོཾ་ཨགྣེ་དེ་ཝ་རྀ་ཥི་མ་ཧཱ་ཏེ་ཛ་གནོད་
བྱེད་མཱ་ར་ཡ་ཕཊ། དེ་ནས་སྙིང་གར་ལྷ་ཡི་གཟུགས། །ཇི་བཞིན་བསྐྱེད་ལ་དེ་བཞིན་མཆོད། །ཨོཾ་ཡ་མཱནྟ་ཀ་ཨ་མུ་ཀ་ཏ་མཱ་ར་ཡ་ཕཊ། ཨོཾ་ཧྲཱིཿཥྚྀཿཝི་ཀྲྀ་ཏཱ་ན་ན་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ། ཨ་མུ་ཀ་མཱ་ར་ཡ་ཕཊ། ཨོཾ་བཛྲ་བེ་ཏཱ་ལཱི། ཨ་མུ་ཀ་མཱ་ར་ཡ་ཕཊ། ཛི་ན་ཛིཀ །རཏྣ་དྷྲྀཀ །ཨཱ་རོ་ལྀཀ །

ཇི་ལྟ་བ་ཡི་སྦྱོར་བས་འགྲུབ། །དེ་ལྟར་གཤིན་རྗེ་གཤེད་དཔལ་མཆོད་པ་ཡིས། །སྲིད་གསུམ་ཉོན་མོངས་བུད་ཤིང་ཀུན་བསྲེགས་ནས། །འགྲོ་བའི་མུན་སེལ་ཡེ་ཤེས་ལྔ་འོད་འབར། །རྡོ་རྗེ་མེ་ཡི་གོ་འཕང་ལ་གནས་ཤོག །དཔལ་གཤིན་རྗེ་གཤེད་དམར་པོའི་སྲེག་བྱེད་མེའི་ཆོ་ག་
