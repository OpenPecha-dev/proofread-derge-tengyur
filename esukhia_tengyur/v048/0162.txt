གསུམ་རབ་ཏུ་སྟོན་པའོ། །བདུད་བཞི་ཁྲག་གི་རོ་དང་བཅུད་ལེན་དཔག་ཏུ་མེད་པ་མྱོང་བས་ན་ལྷ་མོའི་གཡོན་གྱི་ཕྱག་ན་ཐོད་པའོ། །གཡས་ཀྱི་གྲི་གུག་གི་དངོས་པོ་དང་། དངོས་པོ་མེད་པའི་རྣམ་པར་རྟོག་པ་རབ་ཏུ་གཅོད་པའི་ཕྱིར་རོ། །གདུག་པ་ཅན་འདུལ་བ་ནི་ཁྲོ་བོའི་གཟུགས་ཅན་ནོ། །གསུས་
པ་འཕྱང་བ་ནི་མྱ་ངན་ལས་མི་འདའ་བའོ། །དབུ་སྐྲ་སེར་པོ་གྱེན་དུ་བརྫེས་པ་ནི། སྡུག་བསྔལ་བ་དང་ཀུན་འབྱུང་བ་དང་། འགོག་པ་དང་། ལམ་གྱི་བདེན་པ་རབ་ཏུ་སྟོན་པའོ། །སྒོ་བཞི་ནི་བདུད་བཞི་སྐྲག་པར་མཛད་པའོ། །ཟུར་བཞི་ནི་ཚངས་པའི་གནས་བཞི་རབ་ཏུ་གང་བའོ། །ཀ་བ་བརྒྱད་
ནི་རྨི་ལམ་ལྟ་བུ་དང་དྲི་ཟའི་གྲོང་ཁྱེར་ལྟ་བུ་དང་། བྲག་ཅ་དང་སྒྱུ་མའི་བདག་ཉིད་དང་། སྨིག་རྒྱུ་དང་མེ་ལོང་གི་གཟུགས་བརྙན་དང་། ཆུའི་ཟླ་བ་ལྟ་བུ་དང་། ནམ་མཁའི་འཇའ་ཚོན་ཏེ་དངོས་པོ་དང་དངོས་པོ་མེད་པའི་ཉོན་མོངས་པ་འཇོམས་པའི་ཕྱིར་རོ། །ནང་གི་རྟོག་པ་འཇོམས་པ་ནི་དྲ་བ་ཕྱེད་
པའོ། །རྟ་བབས་བཞི་ནི་སྦྱིན་པ་དང་སྙན་པར་སྨྲ་བ་དང་། དོན་མཐུན་པ་དང་དོན་དཔྱོད་པ་དང་ལྡན་ཞིང་<བབ་,རབ་>ཏུ་སྤང་བ་བཞི་དང་ལྡན་པའོ། །གཤིན་རྗེ་གཤེད་དམར་སྒྲུབ་པའི་ཐབས། །བྱས་པའི་བསོད་ནམས་གང་བསགས་པ། །དེ་ཡིས་འགྲོ་བ་མ་ལུས་ཀུན། །བདུད་བཞི་ལས་ནི་ཐར་བར་

ལ་ཕྱག་འཚལ་ལོ། །གང་ཞིག་ཤིན་ཏུ་གོམས་པ་ཡིས། །སྤྲོས་པ་ཡིད་ཀྱིས་ཡིད་གཡེངས་ན། །བླ་མའི་མན་ངག་ལ་བརྟེན་ནས། །སྤྲོས་པ་དེ་ཉིད་སྤྲོས་མེད་བྱ། །རི་བོའི་རྩེ་སོགས་རབ་དབེན་པར། །མཎྜལ་ལ་སོགས་ཚོགས་བྱས་ལ། །བདག་བསྐྱེད་མཆོད་དང་བུམ་པ་བཟླས། །ཇི་སྐད་
བཤད་བཞིན་འགྲུབ་པ་ན། །ཤིན་ཏུ་དད་དང་ལྡན་པ་ཡི། །སློབ་མ་རྗེས་སུ་གཟུང་བར་བྱ། །ཁྲུས་སོགས་སྔོན་འགྲོ་བགེགས་བསྐྲད་ལ། །ཐུན་མོང་པ་དང་ཁྱད་པར་གྱི། །ཡོངས་སུ་རྫོགས་པའི་སྡོམ་པ་སྦྱིན། །རང་སྙིང་ལ་གནས་དབྱུག་པ་དཀར། །རྡོ་རྗེས་མཚན་པ་བསྒོམ་པར་
