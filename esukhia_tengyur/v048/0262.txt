ལ་རྩིབས་བཅུ་དྲུག་པ། རྩིབས་བརྒྱད་པའི་ཕྱོགས་སུ་ཡ་མ་ལ་སོགས་པའི་ཡི་གེ་དགོད། མཚམས་ཀྱི་འདབ་མ་ལ་བསྒྲུབ་བྱའི་མིང་གི་ཡི་གེ་དགོད། རྩིབས་བཅུ་དྲུག་པའི་རྩིབས་ཀྱི་བར་གྱི་རྩིབས་བརྒྱད་ལ། དེ་ལྟར་བསྒྲུབ་བྱ་ཟུར་རྣམས་སུའོ། །འཁོར་ལོ་དང་པོའི་དབུས་ཀྱི་ལྟེ་བར།
ཡ་ནི་རབ་ཏུ་འབྱུང་བར་འགྱུར། ཤར་དུ་མ། རྩིབས་བཅུ་གཉིས་པའི་ཤར་དུ་རཱ། གཉིས་པར་ཛ། གསུམ་པར་ས། བཞི་པར་དོ། །རྩིབས་བརྒྱད་པའི་ལྷོའི་རྩིབས་ལ་མེ། སླར་ཡང་རྩིབས་བཅུ་གཉིས་པའི་འདབ་མ་ལྔ་པ་ལ་རུ། དྲུག་པ་ལ་ཎ། བདུན་པ་ལ་ཡོ། སླར་ཡང་
རྩིབས་བརྒྱད་ཀྱི་ནུབ་ཀྱི་འདབ་མ་ལ་ད། ཡང་རྩིབས་བཅུ་གཉིས་པའི་རྩིབས་བརྒྱད་པ་ལ་ནི། དགུ་པ་ལ་རཱ། བཅུ་པ་ལ་ཡ། སླར་ཡང་རྩིབས་བརྒྱད་པའི་བྱང་གི་རྩིབས་ལ་ཀྵེ། ཡང་རྩིབས་བཅུ་གཉིས་པའི་བཅུ་གཅིག་པ་ལ་ཙྪ། བཅུ་གཉིས་པ་ལ་ནི། དེ་བཞིན་དུ་གསུངས་པ། ཡ་མ་
རཱ་ཛ་ས་དོ་མེ་ཡ། ཡ་མེ་ད་རུ་ཎ་ཡོ་ད་ཡ། ཡ་ད་ཡོ་ནི་ར་ཡ་ཀྵེ་ཡ། ཡཀྵེ་ཡཙྪ་ནི་རཱ་མ་ཡ། ཞེས་པ་རྩ་བའི་སྔགས་སོ། །ཡི་གེ་བཅུ་དྲུག་ནི་རྩིབས་བཅུ་དྲུག་ལའོ། །གཅིག་ནི་གཙོ་བོའི་སྟེ་འདི་ལྟར་དབུས་སུ་རབ་ཏུ་འབྱུང་བར་འགྱུར་རོ། །ཡ་མ་རཱ་ཛ་ས་དོ་མེ་རུ་ཎ་ཡོ། དེ་ནས་དབུས་ཀྱི་སྙིང་པོའི་

ཆད་པ་བཏབ་པ་ལ་ཎ། སེམས་མནན་པ་ལ་རུ། རྒྱལ་བ་ལ་ཛ། རབ་ཏུ་གདུང་བ་ལ་ཡོ། དགྲ་བརླག་པར་བྱ་བ་ལ་ནི། འདིར་ལས་ཀྱི་རང་བཞིན་གྱིས་ཡི་གེ་དབུས་སུ་གཙོ་བོར་གྱུར་པ་དེའི་གནས་སུ་ཡ་བྲི་བར་བྱ། གཤིན་རྗེ་གཤེད་ཀྱི་འཁོར་ལོ་ལ་ངེས་པ་འདི་དང་པོའི་སངས་རྒྱས་
