སིདྡྷིམྨེ་པྲ་ཡཙྪ། སརྦ་ཀརྨཱ་སུ་ཙ་མེ་ཙིཏྟཾ་ཤྲེ་ཡཿ ཀུ་རུ་ཧཱུཾ་ཧ་ཧ་ཧ་ཧ་ཧོཿ བྷ་ག་བཱན་སརྦ་ཏ་ཐཱ་ག་ཏ་བཛྲ་མཱ་མེ་མུཉྩ་བཛྲཱི་བྷ་བ་མ་ཧཱ་ས་མ་ཡ་སཏྭ་ཨཿ ཞེས་པས་ཡི་གེ་བརྒྱ་པའི་སྔགས་སོ། །འདི་བརྗོད་ནས་འདོད་པའི་ལྷའི་ང་རྒྱལ་གྱིས་ཇི་ལྟ་བར་གནས་པ་བྱའོ། །གཤིན་རྗེ་གཤེད་དམར་པོ་མཎྜལ་
དུ་མཆོད་པའི་ཆོ་ག་ཞེས་བྱ་བ། དཔལ་འཛིན་གྱིས་མཛད་པ་རྫོགས་སོ།། །།ལྷའི་སྐད་ལ་ལེགས་པར་ཞུགས་པའི་ཤཱཀྱའི་དགེ་སློང་ཉི་མས་སྐད་གཉིས་སྨྲ་བ་རྣམས་ཀྱི་གདན་ས་དཔལ་ཐར་པ་གླིང་གི་གཙུག་ལག་ཁང་དུ་བསྒྱུར་བའོ།།  ༄༅༅། །རྒྱ་གར་སྐད་དུ། ཀྲྀཥྞ་
ཡ་མཱ་རི་རཀྟ་ཡ་མཱ་རི་ཤྩ་པཱུ་ཛ་བི་དྷིཿ། བོད་སྐད་དུ། གཤིན་རྗེ་གཤེད་ནག་པོ་དང་དམར་པོ་མཆོད་པའི་ཆོ་ག །དཔལ་གཤིན་རྗེ་གཤེད་ལ་ཕྱག་འཚལ་ལོ། །དེ་ནས་བཟླས་པ་རྗེས་སུ་དྲན་པར་བྱ་སྟེ། ཨོཾ་ཧྲཱིཿཥྚྲྀཿཝི་ཀྲྀ་ཏཱ་ན་ན་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ་སྭཱ་ཧཱ། སྙིང་པོའི་སྔགས་སོ། །ཨོཾ་ཧྲཱིཿཀཱ་ལ་རཱུ་པ་ཧཱུཾ་
ཁཾ། ཉེ་བའི་སྙིང་པོའོ། །ཨོཾ་ཧྲཱིཿབྷྲཱུཾ་བྷྲཱུཾ་ཤྲུཾ་ཧཱུཾ་ཀཿ སྙིང་པོ་ཕྱི་མའོ། །ཨོཾ་ཧྲཱིཿཧཿབྷོ་མ་ཧཱ་ཀྲོ་དྷ་ཨཱ་གཙྪ་ཨཱ་གཙྪ། ཨསྨཱ་ཏ་པཱུ་ཛ་པྲ་ཏི་གྲྀཧྞ་པྲ་སཱ་དྷཾ་མེ་དྷི་མན་ཀུ་རུ་སྭཱ་ཧཱ། སྤྱན་དྲང་བའོ། །ཨོཾ་ཧྲཱིཿཥྚྲྀཿཨཱཿམཆོད་ཡོན་ནོ། །ཨོཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཕཊ། ཞབས་བསིལ་བའོ། །ཨོཾ་ཝི་ཀྲྀ་ཏཱ་ན་ན་དུཥྚ་སཏྭ་ད་མ་ཀ་
གཿགཿ དྲིའི་སྔགས་སོ། །ཨོཾ་ཀུ་མཱ་ར་རཱུ་པི་ཎི་ཛཿཛཿཧཱུཾ་ཕཊ། མེ་ཏོག་དབུལ་བའི་སྔགས་སོ། །ཨོཾ་ཧྲཱིཿཧཿཧེ་ཕཊ། བདུག་སྤོས་སོ། །ཨོཾ་ཀཱ་ལ་རཱུ་པ་སརྦ་དུཥྚཱཾ་སཏྭ་ད་མ་ཀ་མ་ཧཱ་བྷཱུ་ཏ་པ་ཏི། ཨི་མཾ་བ་ལིཾ་གྲྀཧྞཱ་པ་ཡ་སྭཱ་ཧཱ། གཏོར་མའི་སྔགས་སོ། །ཨོཾ་དཱིཔྟ་ལོ་ཙ་ནི་ཝི་ཀྲྀ་ཏཱ་ན་ན་མ་ཧཱ་ཊྚཱཊྚ།
ཧཱ་ས་ནཱ་དི་ནི་དཱིཔྟ་དཱིཔྟཱ་ཡ་སྭཱ་ཧཱ། མར་མེའི་སྔགས་སོ། །ཨོཾ་ཛ་ཡ་ཛ་ཡ། སུ་ཛ་ཡ། བི་ཤྭ་རཱུ་པ་དྷ་རི་གཙྪ་གཙྪ། སྭ་བྷཱ་བ་ནཱཾ་ཀྲྀ་ཏེ་ན། ཏ་ཐཱ་ག་ཏཾ་ཤྩ་སརྦ་བུངྡྷཾཤྩ་བི་ས་ར་ཛ་ཡ་མ་མ་ཀཱཪྻཾ་ཀུ་རུ་སྭཱ་ཧཱ། གཤེགས་སུ་གསོལ་བའི་སྔགས་སོ། །གཤིན་རྗེ་གཤེད་ནག་པོ་དང་གཤིན་རྗེ་གཤེད་
དམར་པོའི་མཆོད་པའི་ཆོ་ག །སློབ་དཔོན་ཆེན་པོ་པཎྜི་ཏ། དཔལ་འཛིན་གྱི་ཞལ་སྔ་ནས་མཛད་པ་རྫོགས་སོ།། །།པཎྜི་ཏ་ཉི་མའི་དབང་པོའི་འོད་ཟེར་དང་། ཆག་ཆོས་རྗེ་དཔལ་གྱིས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།། ༄༅༅། །རྒྱ་གར་སྐད་དུ། རཀྟ་ཡ་མཱ་རི་ས་མ་དྷི་
