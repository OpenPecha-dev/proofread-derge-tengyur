སུ་སྨད་གྱུར་ཅིང་། །འདི་དག་སློང་བས་ཡིད་བྱུང་ལ། །བསླངས་པར་གྱུར་ཀྱང་བཀྲེས་པ་ན། །ཀྱེ་མ་འདི་དག་ཉེས་བྱས་ཉིད། །དེ་ལྟར་ཡུན་རིང་བསམས་བྱས་ནས། །ཀུན་གྱི་ཉོན་མོངས་གཞོམ་བརྩོན་པས། །འགྲོ་བ་དབུལ་མིན་བྱ་བའི་སླད། །རིན་ཆེན་དོན་གཉེར་རྒྱ་མཚོར་སོང་། །ཕ་ཡིས་ཇི་ཞིག་ལྟར་སྟེས་

ཉིད། །རྗེས་གནང་ངེས་པ་བརྟན་པ་དེ། །<གཟངས་,གཟིངས་>ལ་ཡང་དག་ཞོན་བྱས་ནས། །རིན་ཆེན་གླིང་དུ་ཕྱིན་པར་གྱུར། །དེར་ནི་ཚོང་པ་གཟིངས་དག་ལ། །ཞོན་ནས་ལྷན་ཅིག་འོངས་རྣམས་ལ། །དེ་ཡིས་སྨྲས་པ་ཁྱེད་ཅག་རྣམས། །རིན་ཆེན་འདོད་པ་བསྡུ་བར་མཛོད། །ཐུན་མོང་རིན་ཆེན་འདི་དག་གིས། །བདག་ལ་

དགོས་པ་ཡོད་མ་ཡིན། །བདག་ཅག་མཛོད་ན་རིན་ཆེན་མཆོག །འོད་དང་ལྡན་པ་མང་པོ་ཡོད། །འོན་ཀྱང་ཡིད་བཞིན་ནོར་ཐོབ་སླད། །བདག་གི་རྩོམ་པ་ཆེན་པོ་འདི། །དེ་ཡིས་ས་ནི་དབུལ་བ་ཉིད། །ཉམས་པ་དག་ཏུ་བྱ་བར་འདོད། །རྒྱ་མཚོ་ཞེས་བྱ་ཀླུ་ཡི་རྒྱལ། །ཆུ་གཏེར་ཆེ་ལ་རབ་ཏུ་གནས། །བསམ་
