ཛྱ་ཏཱཾ་པུ་ཏྲ་སྠི་རཱ་བུདྡྷིརྦི་དྷཱ་ཡ་ཏཱཾ། །བ་ད་ར་དྭི་པ་ཡཱ་ཏྲཱ་ཡཱ་མུདྶཱ་ཧཿཔ་རི་གྲྀ་ཧྱ་ཏཱཾ། །ཤྲུ་ཡ་ཏཱ་མེ་ཥཱ་ཏཏ་པྲཱ་པྟྱཻ་དིདྨཱ་ཏྲཱ་ནུ་ཀྲ་མཿཀྲ་མཿ། །སྥཱི་ཏཿས་ཏྭ་པྲ་བྷཱ་བསྟེ་སཾ་སཱ་རོཏྟ་ར་ཎ་ཀྵ་མ། །ཨ་སྟི་པཤྩི་མ་
གདུང་བ་ཐོང་། །བརྟན་པའི་བློ་གྲོས་རྣམ་པར་བསྒྲུབས། །བ་ད་ར་ཡི་གླིང་འགྲོ་ལ། །རབ་ཏུ་སྤྲོ་བ་ཡོངས་སུ་ཟུངས། །ཁྱོད་ནི་སྙིང་སྟོབས་མཐུ་རྒྱས་ཤིང་། །འཁོར་བ་རབ་ཏུ་སྒྲོལ་བར་བཟོད། །དེ་ཐོབ་བྱ་ཕྱིར་ཕྱོགས་དང་ནི། །ལམ་གྱི་རིམ་པ་འདི་དག་ཉོན། །ནུབ་ཀྱི་ཕྱོགས་ཀྱི་ཆ་ན་
དིགྦྷཱ་གེ་ས་མུལླཾ་གྷྱ་མ་ཧཱ་ཡ་སཱཾ། །ཤ་ཏཱ་ནི་སཔྟ་དྭཱི་པཱ་ནཱཾ་ཏ་ཐཱ་སཔྟ་མ་ཧཱ་ཙ་ལཱན། །སཔྟཱ་ས་གཱཤྩཱ་ནུ་ལོ་མ་པྲ་ཏི་ལོ་མ་བྷི་དྷོམྦུ་དྷིཿ། །ཨ་ནུ་ཀཱུ་ལཱ་ནི་ལཻཪྻསྨིན་པཱ་ར་མཱསྣོ་ཏི་པུ་ཎྱ་བཱན། །ཏ་ཏསྟཏྟུ་ལྱ་ནཱ་མཱ་དྲིརྦཱ་ཏཻསྟི་
ཡོད། །ཤིན་ཏུ་ཆེ་བའི་གླིང་དག་ནི། །བདུན་བརྒྱ་རབ་ཏུ་འདས་གྱུར་ཅིང་། །དེ་བཞིན་དུ་ནི་རི་ཆེན་བདུན། །ཆུ་ཀླུང་བདུན་དང་རྗེས་འགྲོ་དང་། །བཟློག་འགྲོ་ཞེས་པའི་ཆུ་ཡི་གཏེར། །རྗེས་སུ་མཐུན་པའི་རླུང་གིས་དེར། །ཕ་རོལ་འཐོབ་བོ་བསོད་ནམས་ཅན། །དེ་ནས་དེ་མཚུངས་མིང་ཅན་རི། །རླུང་
མི་ར་མོ་ཧ་ཀྲྀ་ཏ། །ཡ་ཏཱ་ཀྵྞོརྡི་ཤ་ཏི་སྭཱསྠྱ་མ་མོ་གྷཱ་ཁྱཱ་མ་ཧཽ་ཥ་དྷིཿ། །ཨ་ཐཱ་བརྟྱ་བྷི་དྷོཾ་བྷོ་དྷི་བཻ་རམྦྷེཪྻ་ཏྲ་བ་ཡུ་བྷི། །མཛྫ་ནོནྨ་ཛྫ་ནེ་རྫནྟུཿསཔྟ་བརྟེ་ཥུ་ཏཱཪྻ་ཏེ། །ཨ་བརྟཱ་ཁྱསྟ་ཏཿཤཻ་
གིས་རབ་རིབ་རྨོངས་པར་བྱེད། །སྨན་ཆེན་དོན་ཡོད་ཅེས་པ་ཡིས། །མིག་གཉིས་གང་དུ་རྣལ་གནས་སྟེར། །དེ་ནས་ཀུན་འཁོར་ཞེས་ཆུ་གཏེར། །གང་དུ་རླུང་ནི་བེ་རཱམ་བྷས། །འབྱིང་ཞིང་འབྱུང་བས་སྲོག་ཆགས་རྣམས། །ཀུན་འཁོར་བདུན་ལས་སྒྲོལ་བར་བྱེད། །དེ་ནས་ཀུན་འཁོར་ཞེས་པའི་རི། །
