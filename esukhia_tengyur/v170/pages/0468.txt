སུ་ཏསྟ་སྱ་མཱནྡྷ་ཏཱ་བྷཱུ་ད་ཡོ་ནི་ཛཿ། །ཛ་ག་ཏྱེ་ཀ་ཏ་པ་ཏྲ་སྱ་ཡ་སྱ་བཾ་ཤོ་མ་ཧ་ན་བྷཱུ་ཏ། །བཾ་ཤེ་ས་ཧ་སྲ་བཾ་ཤ་སྱ་ཀྲྀ་ཀཱིསྟ་སྱཱ་བྷ་བནྣྲྀ་པཿ། །ཙིཏྟ་པྲ་སཱ་ད་མ་ཀ་རོད་བྷ་ག་བཱན་ཡ་སྱ་ཀཱ་ཤྱ་པཿ། །ཨི་
འཁོར་ལོས་སྒྱུར། །སྐྱེ་གནས་སྐྱེས་མིན་ང་ལས་ནུ། །གང་གི་རིགས་བརྒྱུད་འགྲོ་བ་ན། །གདུགས་གཅིག་ཉིད་ལྡན་ཆེན་པོ་བྱུང་། །དེ་ཡི་རིགས་ནི་སྟོང་ཕྲག་གི། །རིགས་ལས་མི་བདག་ཀྲྀ་ཀཱི་བྱུང་། །གང་གི་ཐུགས་ནི་བཅོམ་ལྡན་འདས། །འོད་སྲུངས་དག་གིས་དང་བར་མཛད། །དེ་བརྒྱུད་
ཀྵྭཱ་ཀུ་ར་ནྭ་ཡེ་ཏ་སྱ་ཏ་སྱ་ཙཱ་བྷཱུད་བི་རཱུ་ཌྷ་ཀཿ། །པྲཱྀ་ཏྱཱ་ཀཱ་ནི་ཡ་ས་སཱུ་ནོ་ཛྱོཥྛསྟེ་ན་བི་བཱ་སི་ཏཱཿ། །ཨེ་ཀཱི་བྷཱུ་ཡ་ཏ་ཏཿསརྦེ་སྭ་དེ་ཤ་བི་ག་ཏེ་སྤྲྀ་ཧཱཿ། །ཀུ་མཱ་རཿཀ་པི་ལཱ་ཁྱ་སྱ་མ་ཧཪྵི་རཱ་ཤྲ་མཾ་ཡ་ཡུཿ། །དྷྱཱ་ན་ཀཱ་ལཱནྟ་རཱ་
བུ་རམ་ཤིང་པ་བྱུང་། །དེ་ལས་ཡང་ནི་འཕགས་སྐྱེས་པོ། །དེ་ཡི་བུ་ནི་ནུ་བོ་ལ། །དགའ་བས་ཕུ་བོ་གནས་ནས་བསྐྲད། །དེ་ནས་གཞོན་ནུ་ཐམས་ཅད་ནི། །གཅིག་ཏུ་འདུས་ནས་རང་ཡུལ་ལ། །ཆགས་དང་བྲལ་བས་ཐུབ་པ་ཆེ། །སེར་སྐྱ་ཞེས་པའི་གནས་སུ་སོང་། །བྱིས་པ་སྒྲ་ཆེན་སྒྲོགས་
ཡཱ་ཎཱཾ་བཱ་མྱཱ་དུཙྩཻཿབྲ་ལཱ་པི་ནཾ། །སོ་ནྱ་ཏྲ་ནིརྨྨ་མེ་ཏེ་ཥཱཾ་པུ་རཾ་ཀ་པི་ལ་བསྟྭེ་ཏི། །ཀཱ་ལ་ན་པུ་ཏྲ་བཱཏྶ་ལྱཱ་ད་ནུ་ཏཱ་པ་ན་བྷཱུ་པ་ཏིཿ། །ཨཱ་ནཱི་ཡནྟཱཾ་ཀུ་མ་རཱ་སྟེ་ས་ཙི་བཱ་ནི་ཏྱ་བྷཱ་ཥ་ཏ། ཏ་མཱུ་ཙུརྨནྟྲི་ཎཿསརྦེ་
པ་ཡིས། །བསམ་གཏན་དུས་ཀྱི་གེགས་བྱེད་པ། །དེ་དག་གྲོང་ཁྱེར་གཞན་དུ་ནི། །སེར་སྐྱའི་གཞི་ཞེས་དེ་ཡིས་སྤྲུལ། །དུས་ཀྱིས་བདག་བུ་ལ་ནི། །མཉེས་གཤིན་རྗེས་སུ་གདུང་བ་ཡིས། །བློན་པོ་རྣམས་ལ་གཞོན་ནུ་ནི། །དེ་དག་ཁྱེར་ལ་ཤོག་ཅེས་སྨྲས། །བློན་པོས་དེར་སྨྲས་རྒྱལ་པོ་
