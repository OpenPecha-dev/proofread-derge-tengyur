ཨེ་ཏ་དཱ་ཀ་རྞྱ་ནྲྀ་པ་ཏི་ནིརྒ་ཏྱ་མཱ་ཏྱ་སམྨ་ཏཿ། །ད་དཪྴ་དྭི་ར་ད་དྭཱ་རི་ཀཻ་ལཱ་ཤ་མི་བ་ཛཾ་ག་མཾ། །ཨུདྡཱ་མ་སཽ་ར་བྷཱ་ཧཱུ་ཏཻ་རྦྷྲ་མ་རེ་རྒཎྜ་དིཎྜི་མཻཿ། །ཤྲྀཾ་གཱ་ར་བྷཱ་ར་ཎོ་དཱ་རཾ་བ་སནྟ་མི་བ་སེ་པི་ཏཾ། །དནྟ་བརྻ་ནྟབི་
དེ་དག་ཐོས་ནས་མི་བདག་ནི། །བློན་ལ་ཡང་དག་བགྲོས་ནས་སོང་། །སྒོར་ནི་སོ་གཉིས་ཀཻ་ལ་ཤ །གཡོ་བ་ལྟ་བུ་མཐོང་གྱུར་ཏེ། །རབ་ལྷག་དྲི་བཟང་གིས་བོས་པ། །བུང་བ་འགྲམ་པའི་རྔ་རྣམས་ཀྱིས། །སྒེག་པའི་རྒྱན་ནི་ཡིད་འོང་བ། །དཔྱིད་ཀ་བཞིན་དུ་རབ་ཏུ་བསྟེན། །འབིགས་བྱེད་ཀྱི་
ཤྲཱནྟ་ཀ་རཾ་མི་ལི་ཏ་ལོ་ཙ་ནཾ། །སྨ་རནྟ་བིན་དྷྱ་ཀ་ད་ལཱི་ཤལླ་ཀཱི་ཀཱ་ན་ན་ཤྲི་ཡཿ། །ཨ་གསྟྱཱ་ཤཱ་ས་ནཱདྨཱ་ཏཾ་བྷུ་བི་ཀུཉྫ་ར་རཱ་ཛ་ཏཱཾ། །སྥུརྫྫཏྶཔྟ་ཙྪ་དཱ་མོ་དཾ་བིནྡྷྱ་ཙ་ལ་མི་བོནྣ་ཏཾ། །ཏཾ་བི་ལོ་ཀྱ་ཀྵི་
ནི་ཆུ་ཤིང་དང་། །ཤ་ལ་ཀི་ཡི་ནགས་ཀྱི་དཔལ། །དྲན་ལས་མིག་ནི་ཟུམ་གྱུར་ཅིང་། །སོ་ཡི་བར་དུ་ལག་པ་བརྐྱངས། །འདབ་མ་བདུན་པའི་དྲིས་རྒྱས་ཤིང་། །ཨ་གསྟྱཱ་ཡིས་བསྟན་པ་ལས། །འབིགས་བྱེད་མི་གཡོ་ས་ལ་ནི། །གླང་པོའི་རྒྱལ་པོར་གྱུར་ལྟར་མཐོ། །སོ་ཡི་ཀ་བས་རབ་
ཏི་པ་ཏི་རྡནྟ་སྟ་མྦྷ་བི་བྷཱུ་ཥི་ཏཾ། །ལཀྵྨཱི་བི་ལཱ་ས་བྷ་བ་ནཾ་བིསྨ་ཡཱ་དི་ཏྱ་ཙཱི་ཏ་ཡ་ཏ། །ཨ་ཧོ་ན་བ་ན་བོཏྐཪྵཱ་ནིརྨཱ་ཎཱ་ཤྩ་ཪྱ་ཤཱ་ལི་ནཱཾ། །ཀརྨ་ཎཱ་མ་ན་བཙྪིནྣཱ་སཾ་སཱ་ར་སརྒ་སནྟ་ཏིཿ། །ཨ་མནྠེ་
བརྒྱན་ཅིང་། །དཔལ་འབྱོར་རྣམ་པར་རྩེ་བའི་གནས། །དེ་ནི་རྣམ་གཟིགས་ས་ཡི་བདག །ཡ་མཚན་གྱུར་ནས་འདི་ལྟར་བསམས། །ཨེ་མ་གསར་པ་གསར་པར་ནི། །རབ་གྱུར་ངོ་མཚར་སྤྲུལ་བྱེད་པ། །བསྔགས་འོས་ལས་རྣམས་འཁོར་བར་ནི། །སྤྲོས་པའི་རྒྱུན་དག་ཆད་པ་མེད། །བདུད་
