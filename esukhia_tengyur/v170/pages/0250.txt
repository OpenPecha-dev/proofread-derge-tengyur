རཾ་ཡ་སྱེ་དྲྀ་ཤཾ་ཕ་ལཾ། །བྷ་ག་བཱ་ནི་ཏེ་ཏཻཿཔྲྀཥྚཿསརྦ་ཛྙསྟཱ་ན་བྷཱ་ཥཱ་ཏྭཾ། །ནི་ཛ་ཀརྨ་བི་ས་ཀེན་ཤཱཀྱཱ་ནཱཾ་མེ་ཥ་སཾ་ཀྵ་ཡཿ། །ཀྲྀཥྚཽ་པུ་ར་མ་ཧཱམཏྶཽ་དྷི་བ་རཻཿས་རི་ཏོནྟ་རཱཏ། །ས་དཱ་ནི་ཀྲྀ་ཏཽ་མཏྶ་སྱ་
བསྒྲུབས། །འཇིགས་རུང་གང་གི་འབྲས་འདི་འདྲ། །དེ་སྐད་དེས་དྲིས་བཅོམ་ལྡན་འདས། །ཐམས་ཅད་མཁྱེན་པས་དེར་བཀའ་སྩལ། །རང་གི་ལས་ཀྱི་རྣམ་སྨིན་གྱིས། །ཤཱཀྱ་རྣམས་ནི་བརླག་པ་འདི། །ཉ་པ་རྣམས་ཀྱིས་ཆུ་ནང་ནས། །སྔོན་ཚེ་ཉ་ཆེན་གཉིས་ཤིག་དྲངས། །ཉ་ནི་རྟག་ཏུ་རབ་བཅད་པའི། །
བྷཱུ་ཡ་སྟདྦྱ་ཐ་ཡ་ན་ཏཽ། །ཀཱ་ལེ་ན་ཙཽ་ར་ཏཱཾ་ཡཱ་ཏཻསྟཻ་རེ་བཱ་པ་ར་ཛནྨ་ནི། །ཧ་ཏཽ་གྲི་ཧ་པ་ཏཱི་དགྡྷཱ་ཏཱ་བེ་བ་དྷ་ན་ཧཱ་རཱི་བྷཱིཿ། །ཏཽ་མཏྶཽ་ཏཽ་གྲྀ་ཧ་སྠཽ་བི་རཱུ་ཌྷ་ཀ་པུ་རཽ་ཧི་ཏོ། །དཱ་སཱ་ནཱཾ་ཏསྐ་རཱ་ཎཱཾ་ཙ་ཤཱཀྱཱ་ནཱཾ་མྲྀ་ཏྱུ་
གདུང་བ་ཆེན་པོ་དག་གིས་བསད། །དུས་ཀྱིས་སྐྱེ་བ་གཞན་ལ་ནི། །དེ་རྣམས་ཆོམ་རྐུན་ཉིད་གྱུར་པ། །ནོར་ནི་འཕྲོག་པས་དེ་དག་ཉིད། །ཆོམ་རྐུན་དུ་གྱུར་བསྲེགས་ནས་བསད། །ཉ་དེ་དག་དང་ཁྱིམ་བདག་དེ། །འཕགས་སྐྱེས་པོ་དང་མདུན་ན་འདོན། །ཉ་པ་དང་ནི་ཆོམ་རྐུན་པའི། །ཤཱཀྱ་རྣམས་ཀྱིས་

བསད་པར་གྱུར། །བཅོམ་ལྡན་འདས་ཀྱིས་འདི་ཐོས་ནས། །དགེ་སློང་རྣམས་ཀྱིས་ལས་རྣམས་ཀྱི། །འབྲས་བུའི་རྒྱུ་ནི་བསླུ་བ་དག །མེད་པ་ཉིད་དུ་ངེས་པར་གྱུར། །དེ་ནས་རྒྱལ་བས་ངན་རྒྱགས་པའི། །འཕགས་སྐྱེས་པོ་ནི་རང་གྲོང་སོང་། །བུ་ནི་རྒྱལ་བྱེད་ཅེས་བྱ་བའི། །བྱིས་པ་རྩེ་བས་མཛའ་ལ་སྨྲས། །
