པལླ་བཿ།། །།ཤོ་བྷནྟེ་བྷཱུ་བ་ནེ་ཥུ་བྷ་བྱ་མ་ན་སཾ་ཡནྣཱ་ཀ་ཀཱནྟཱ་ཀ་ར་པྲོ་ཌྷོ་དཉྩི་ཏ་ཙཱ་རུ་ཙཱ་མ་ར་སི་ཏ་ཙྪ་ཏྲ་སྨི་ཏཱཿསཾ་པ་དཿ། །ཡཙྪོཏྶརྤ་ཏི་ཏརྤི་ཏ་ཤྲུ་ཏི་ཡ་ཤཿཀརྤཱུ་ར་པཱུ་
སྟེ་གསུམ་པའོ།། །།གང་ཞིག་མཐོ་རིས་མཛེས་མའི་ལག་པས་མཆོག་ཏུ་རབ་བསྐྱོད་མཛེས་པའི་རྔ་ཡབ་དག་དང་ནི། །གདུགས་དཀར་འཛུམ་པའི་ཕུན་ཚོགས་དག་གིས་འཇིག་རྟེན་ན་ནི་ལེགས་ལྡན་ཐུགས་ནི་རྣམ་མཛེས་དང་། །གང་ཡང་གྲགས་སྙན་ག་པུར་རྒྱུན་གྱིས་དྲི་མེད་ཐོས་པ་སིམ་བྱེད་
རོཛྫྭ་ལཾ་སྭལྤ་དཱ་ནཱ་ཀ་ཎ་སྱ་ཏཏྥ་ལ་མ་ཧོ་དཱ་ནཾ་ནི་དཱ་ན་ཤྲི་ཡཿ། །ཨ་བྷཱུ་དུ་པོ་ཥ་དྷོ་ནཱ་མ་བྷཱུ་བྷྲྀཏྻ་སྱ་བི་བྷཱ་བ་ཏེ། །བི་བུ་དྷཱ་བྷི་མ་ཏཱ་ཀཱིརྟིཿསུ་དྷཱ་དུགྡྷོ་ད་དྷེ་རི་བ། །བ་སུ་དྷཱ་མ་བ་ཏོ་ཡ་སྱ་བ་སུ་དྷཱ་
རབ་ཏུ་འཕྲོ་བ་དག །སྦྱིན་པའི་ཆ་ཤས་དེ་འབྲས་ཤིན་ཏུ་ཆུང་ངུའོ་ཨེ་མ་སྦྱིན་པ་དཔལ་གྱི་དང་པོའི་རྒྱུ། །འོ་མའི་མཚོ་ལས་བདུད་རྩི་བཞིན། །མཁས་རྣམས་མངོན་འདོད་གྲགས་པ་ཡིས། །གང་ཞིག་རྣམ་གསལ་ས་འཛིན་ནི། །གསོ་སྦྱོང་འཕགས་ཞེས་བྱ་བ་བྱུང་། །ནོར་འཛིན་བསྲུང་བ་ནོར་དང་ནི། །གཟི་བརྗིད་ལྡན་
མ་བ་ཏཿཔུ་རཿ། །ན་ནཱ་མ་པྲ་ཎ་ཏོ་ཀ་སྱ་ན་ནཱ་མ་ནྲྀ་པ་ཏེཿཤི་རཿ། །ཤུདྡྷཱ་དྷཱི་རི་བ་དྷརྨེ་ཎ་དཱ་ནེ་ནེ་བ་ད་ཡཱ་ལུ་ཏཱ། །བི་བྷཱུ་ཏེརྦི་ན་ཡེནེ་བ་བྷཱུ་ཥི་ཏཱ་ཡེ་ན་བྷཱུ་ར་བྷཱུཏ། །གུ་ཎི་ནཾཿཔྲི་ཤུ་བཾ་ཤ་སྱ་བ་བྷཱུ་བེནྡུ་དྻུ་ཏེཿསྠི་ཏཿ། །ཡ་སྱ་
པ་གང་མདུན་དུ། །མི་ཡི་བདག་པོའི་སྤྱི་བོ་དག །ངེས་པར་ཕྱག་ལ་སུས་མ་བཏུད། །ཆོས་ཀྱིས་དག་པའི་བློ་གྲོས་བཞིན། །སྦྱིན་པས་བརྩེ་བ་ལྡན་ཉིད་བཞིན། །དུལ་བས་རྣམ་པར་འབྱོར་པ་བཞིན། །གང་གིས་ས་ནི་བརྒྱན་པར་གྱུར། །ཡོན་ཏན་ཀུན་ལྡན་རིགས་སུ་མཐོ། །ཀུན་གྱི་གདུང་སྐྱོབ་ཟླ་བའི་འོད། །གང་གི་
