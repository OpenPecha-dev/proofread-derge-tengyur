ད་བཿ། །ཙི་ཏྲེ་སྭཔྣེ་ཐ་སཾ་ཀལྤེ་པྲྀ་ཐུ་པུ་ཎྱཻཿས་དྲྀ་ཤྱ་ཏེ། །ཨི་ཏི་མནྡྲཾ་བ་ཙཿཤྲུ་ཏྭཱ་ཏ་ཐེ་ཏྱུཀྟྭཱ་མ་ཧཱི་པ་ཏིཿ། །ག་ཏྭཱ་བྷ་ག་བ་ཏེ་ན་མྲཿཏ་མེ་བཱརྠཾ་བྱ་བེ་ད་ཡཏ། །ཨ་ནུ་ཛྙཱ་ཏ་སྟ་ཏསྟེ་ན་ནྲྀ་པཤྩི་ཏྲ་ཀ་རཱན་
བསམ་ཤིང་། །རི་མོ་རྨི་ལམ་ཀུན་རྟོག་ལའང་། །དེ་ནི་བསོད་ནམས་ཆེ་བས་མཐོང་། །ཞེས་པ་བློན་པོའི་ཚིག་ཐོས་ནས། །དེ་བཞིན་ཞེས་བརྗོད་ས་ཡི་བདག །སོང་ནས་བཅོམ་ལྡན་ལ་བཏུད་དེ། །དོན་དེ་ཉིད་ནི་རབ་ཏུ་གསོལ། །དེ་ནས་དེ་ཡིས་རྗེས་གནང་སྟེ། །རི་མོ་མཁན་པོ་མཆོག་
བ་རཱན། །ཨཱ་དི་དེ་ཤཱ་ཤུ་བྷ་ག་བ་ཏེཿཔྲ་ཏི་མོལླེ་ཁ་ཀརྨྨ་ཎི། །ཛི་ན་སྱཱ་ལོ་ཀ་ཡནྟསྟེ་མཱུརྟྟིཾ་རཱུ་པ་བ་ཤཱི་ཀྲྀ་ཏཱཿ། །ཡ་ཡུཿཔྲ་མཱ་ཎ་གྲ་ཧ་ཎེ་པྲ་གལྦྷཱ་ཨ་སྱ་ཤཀྟ་ཏཱཾ། །སཾ་ཀྲནྟཾ་ནིརྨྨ་ལ་པ་ཊེ་ཙྪཱ་ཡཱཾ་བྷ་ག་བ་
རྣམས་ལ། །བཅོམ་ལྡན་སྐུ་གཟུགས་བྲི་བའི་ལས། །མི་ཡི་བདག་པོས་མྱུར་བར་བསྐོས། །རྒྱལ་བའི་སྐུ་ནི་ལྟ་བྱེད་པ། །དེ་དག་གཟུགས་ཀྱི་དབང་བྱས་ནས། །སྤོབས་དང་ལྡན་ཡང་ཚད་དག་ནི། །རབ་ཏུ་གཟུང་བར་ནུས་མ་གྱུར། །དེ་ནས་བཅོམ་ལྡན་སྐུ་གཟུགས་ནི། །དྲི་མེད་རས་ལ་

འཕོས་པ་དག །གཟུགས་བཟང་བསྒོམ་པ་མངོན་ཤེས་དེས། །དལ་བུས་ཡང་དག་རྫོགས་པར་བྱས། །འགྲོ་བའི་མིག་གི་བསོད་ནམས་རྣམས། །གཟུགས་ཅན་ཡང་དག་བསགས་པ་བཞིན། །སངས་རྒྱས་སྐུ་གཟུགས་ཐང་ཀ་དེ། །དེ་ནས་ས་གཞི་སྐྱོང་བས་སྤྲིངས། །གཟུགས་ཅན་སྙིང་
