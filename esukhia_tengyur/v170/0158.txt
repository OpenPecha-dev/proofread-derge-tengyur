བྷཱ་ཥེ་བྷ་ག་བཱན་ཛི་ནཿ། །ཡཱ་སཽ་པུ་རཱ་རོ་ཧི་ཏཱ་ཁྱཱ་དཱ་སི་ཤཱཀྱཱ་གྲྀ་ཧ་བྷ་བཏ། །སཻ་བ་མུཀྟཱ་ལ་ཏཱ་ཛཱ་ཏཱ་སཏྐརྨྨ་པྲ་ཎི་དྷཱ་ན་ཏཿ། །ཨ་བྷཱུད་མ་ཧཱ་དྷ་ནོ་ནཱ་མ་བཱ་རཱ་ཎ་སྱཱཾ་བ་ཎི་ཀ་པུ་རཱ། །ཏ་སྱ་རཏྣ་པ་ཏཱི་ནཱ་མ་པཏྣཱི་པུ་ཎྱོ་
བཅོམ་ལྡན་རྒྱལ་བས་གསུངས། །གང་འདི་ཤཱཀྱའི་ཁྱིམ་དུ་སྔོན། །རོ་ཧི་ཏ་ཞེས་འབངས་མོར་གྱུར། །དེ་ཉིད་མུ་ཏིག་འཁྲི་ཤིང་དུ། །སྨོན་ལམ་དམ་པའི་ལས་ཀྱིས་སྐྱེས། །ནོར་ཆེན་ཞེས་པའི་ཚོང་པ་ནི། །བཱ་རཱ་ཎ་སཱིར་སྔོན་བྱུང་གྱུར། །དེ་ཡི་ཁྱིམ་ཐབ་བསོད་ནམས་ཅན། །རིན་ཆེན་ལྡན་
ཅི་ཏཱ་བྷ་བཏ། །སཱ་པ་ཏྱཽ་པཉྩ་ཏཱ་ཡཱ་ཏེ་ནིཿཔུ་ཏྲཱ་སྟཱུ་པ་ཤེ་ཁ་རེ། །པཱུ་ཛཱཾ་ཀྲྀ་ཏྭཱ་མ་ཧཱ་ཧཱ་རཾ་བྷཀྟི་ཡུཀྟཱ་ནྱ་བེ་ད་ཡཏ། །ཏེ་ན་པུ་ཎྱ་བི་པཱ་ཀེ་ན་སིཾ་ཧ་ལཱ་དྷི་པ་ཏེཿསུ་ཏཱ། །ཛཱ་ཏཱ་མུཀྟཱ་ལ་ཏཱ་སཻ་བ་པྲཱཔྟཱ་ཙ་པ་རི་ནིརྦྲྀ་
ཞེས་བྱ་བར་གྱུར། །བདག་པོ་ལྔ་པ་ཉིད་གྱུར་ཚེ། །བུད་མེད་དེ་ཡིས་མཆོད་རྟེན་རྩེར། །དོ་ཤལ་ཆེན་པོ་གུས་ལྡན་པས། །མཆོད་པ་བྱས་ནས་ཕུལ་བར་གྱུར། །བསོད་ནམས་རྣམ་སྨིན་དེས་དེ་ཉིད། །སིང་ག་ལ་ཡི་བདག་པོ་ཡི། །བུ་མོ་མུ་ཏིག་འཁྲི་ཤིང་གྱུར། །ཡོངས་སུ་མྱ་ངན་འདས་པའང་
ཏིཾ། །སཻ་བ་ཛནྨ་ནི་ཙཱ་ནྱསྨིནྣཻཤྭདྱ་མ་ད་མོ་ཧི་ཏཱ། །པཱུ་ཛྱཱ་དྷི་ཀྵེ་པ་ད་ཀྵ་བྷཱུཏ་དཱ་སཱི་ཏེ་ནཱ་ཏི་བཙྩ་རཾ། །ཛནྨ་ཏེ་ཛ་ནེ་ནོཔྟཾ་ཡདྻད་ཀརྨ་ཤུ་བྷཱ་ཤུ་བྷཾ། །ཏ་སྱ་ཏ་སྱ་ས་ཏད་རཱུ་པཾ་བྷུཾ་ཀྟེ་པ་རི་ཎ་ཏི་ཕ་ལཾ། །ནི་
ཐོབ། །དེ་ཉིད་སྐྱེ་བ་གཞན་ལ་ནི། །ཕྱུག་པས་རྒྱགས་ཤིང་རྨོངས་གྱུར་ནས། །མཆོད་ལ་སྨོད་ཅིང་སྒྲིན་གྱུར་པ། །དེ་ཡིས་ཡུན་རིང་འབངས་མོར་གྱུར། །དགེ་དང་མི་དགེའི་ལས་གང་གང་། །སྐྱེ་བའི་ཞིང་ལ་སྐྱེ་བོས་བཏབ། །དེ་དང་དེ་ཡི་རང་བཞིན་དེ། །ཡོངས་སུ་སྨིན་པའི་འབྲས་བུ་སྤྱོད། །དགེ་བ་
