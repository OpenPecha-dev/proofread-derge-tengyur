རརྒ་ལེ། །མུ་ན་ཡོ་བྷཱ་རྒ་བ་མུ་ཁཱ་ནྲྀ་པཱ་དུཥྤྲ་ས་ཧཱ་ད་ཡཿ། །བརྟ་མཱ་ནེ་མ་ཁེ་ཏ་སྨིན་ནིཿསཾ་ཁྱ་བ་སུ་བཪྵི་ཎི། །རཀྵ་རཱུ་པ་ས་མུཏྟ་སྠོ་བ་ཧྣི་མ་དྷྱཏ་སུ་རེ་ཤྭ་རཿ། །ཨུ་པ་སྲྀ་ཏྱ་ས་བྷཱུ་པཱ་ལཾ་ཀྲྀ་ཤོ་བི་ཀྲྀ་
སོགས། །འདོད་པ་ཀུན་གྱིས་མ་བསྡམས་པའི། །མཆོད་སྦྱིན་དེར་ནི་ཡང་དག་འོངས། །གྲངས་མེད་ནོར་གྱི་ཆར་འབེབས་པའི། །སྦྱིན་སྲེག་མངོན་དུ་གྱུར་པ་དེའི། །མེ་ཡི་དབུས་ནས་ལྷ་ཡི་དབང་། །སྲིན་པོའི་གཟུགས་སུ་ཡང་དག་ལངས། །རིད་ཅིང་རྣམ་འགྱུར་ལུས་ཅན་དེ། །ས་སྐྱོང་ལ་ནི་ཉེར་བཏུད་
ཏ་བི་གྲ་ཧཿ། །ཀྵུཏྤི་པཱ་ས་རྡི་ཏོ་སྨཱི་ཏི་ཡ་ཡཱ་ཙེ་པཱ་ན་བྷོ་ཛ་ནཾ། །ཤཱ་ས་ནཱ་ད་ཐ་བྷཱུ་བྷརྟུསྟསྨཻ་བི་བི་དྷ་བྷོ་ཛ་ནཾ། །ཨུ་པ་ནི་ནྻུཿཔ་རི་ཙི་ཏཱཿཔཱ་ནཾ་ཙ་པ་རི་ཙཱ་ར་ཀཱཿ། །ཏ་ཏཿཀིཉྩིད་བི་ཧ་སྱཻ་བཀྵི་ཏི་བཾ་ས་ཧ་རཱཀྵ་སཿ། །ནེ་
དེ། །བདག་ནི་བཀྲེས་སོ་སྐོམ་མོ་ཞེས། །བཟའ་དང་བཏུང་བ་སློང་བར་གྱུར། །དེ་ནས་ས་ཡི་བདག་པོའི་བཀས། །རིམ་གྲོ་བ་ཡི་འཁོར་རྣམས་ཀྱིས། །བཟའ་བ་རྣམ་པ་སྣ་ཚོགས་དང་། །བཏུང་བ་དག་ཀྱང་ཉེ་བར་བསྟབས། །དེ་ནས་སྲིན་པོས་ས་བདག་ལ། །ཅུང་ཟད་དགོད་བཞིན་རབ་ཏུ་སྨྲས། །རྒྱལ་
ད་མསྨཏ་པྲི་ཡཾ་རཱ་ཛན་བ་ཡཾ་ཧི་པི་ཤི་ཏཱ་ཤ་ནཱཿ། །ས་དྱོ་ཧ་ཏ་སྱ་མཱཾ་སེ་ན་རུ་དྷི་རེ་ཎ་ཙ་བྷཱུ་ཡ་པཱ། །ཏྲྀཔྟི་རུཏྤ་དྱ་ཏེ་སྨཱ་ཀཾ་དཱི་ཡ་ཏཱཾ་ཡ་ད་བྷཱི་སྶི་ཏཾ། །སརྦ་ཀཱ་མ་པྲ་དོ་སཱི་ཏི་ཏྭཱ་མ་ཧཾ་ས་མུ་པཱ་ག་ཏཿ། །ད་དཱ་མཱི་ཏི་
པོ་འདི་ལ་བདག་མི་དགའ། །གང་ཕྱིར་བདག་ནི་ཤ་ཡིས་འཚོ། །འཕྲལ་ལ་བསད་པའི་ཤ་དང་ནི། །ཤིན་ཏུ་མང་བའི་ཁྲག་དག་གིས། །བདག་ཅག་ཚིམ་པ་རབ་ཏུ་སྐྱེ། །གང་ཞིག་འདོད་པ་གནང་བར་མཛོད། །འདོད་ཀུན་རབ་ཏུ་སྟེར་རོ་ཞེས། །ཁྱོད་ཅན་བདག་ནི་ཡང་དག་འོངས། །སྦྱིན་ནོ་ཞེས་ནི་རབ་
