ཏ་སྱཻ་བཱ་ནུ་ཙ་ར་བྷཱུ་ཏྭཱ་བི་ཙ་ཙཱ་ར་མ་ཏ་བྲ་ཏཿ། །ས་དྲྀཥྚཱ་ཧཪྵ་ཀཻ་བཱ་རི་ཀྵཏྟཱཏྐྵེ་ཏྟྲཾ་བརྟྟི་ཏཱཾ། །ནིརྡི་ཥྚེ་ཡ་ཐཱ་ཡནྟཾ་བི་སྨ་ཡཱ་དི་ཏྱ་ཙིཾ་ཏ་ཡཏ། །ཨ་ཧོ་པི་ཧི་ཏ་མརྒེ་ཎ་གཙྩ་ཏཱ་མ་སྱ་ཙེ་ཏ་
དེ་ཡི་རྗེས་འབྲང་ཉིད་གྱུར་ནས། །བརྟུལ་ཞུགས་ཀྱིས་བསྡམས་རྣམ་པར་རྒྱུ། །ཞིང་པ་རྣམས་ཀྱིས་བསྟན་པ་ཡིས། །ཞིང་ནས་ཞིང་ལ་འཇུག་པའི་ཆུ། །ལམ་ནས་བགྲོད་པ་དེ་མཐོང་ནས། །ཡ་མཚན་ལས་ནི་འདི་ལྟར་བསམས། །ཀྱེ་མ་སེམས་མེད་ཆུ་རྣམས་ཀྱང་། །བསྒྲུབས་པའི་ལམ་ནས་བགྲོད་
སཾ། །ཛ་ལཱ་ནཱཾཀརྨྨ་སཾ་སིདྡྷིརྟྲི་ཤྱེ་ཏེ་ན་ཏུ་དེ་ཧི་ནཱཾ། །སཾ་བི་ཏྱེ་ཧི་པྲ་ཛནྣ་གྲེ་དྲྀཥྚཱ་སྱཥྚཱི་ཀྲྀ་རཾ། །ས་ཏ་པྟ་མི་ཥུ་ཀ་རེ་ཎ་པྲ་ད་གྷཽ་དྷི་མ་ཏཱམྦ་རཿ། །ཏཱ་སཏྤྲ་གུ་ཎ་ཏཱ་མེ་ཏེ་ཡཱནྟི་ནིཤྩེ་
པ་ཡིས། །ལས་ནི་ཡང་དག་སྒྲུབ་མཐོང་སྟེ། །ལུས་ཅན་རྣམས་ལ་མ་ཡིན་ནོ། །ཞེས་བསམས་བགྲོད་པས་མདུན་དུ་ནི། །མདའ་མཁན་དག་གིས་མདའ་བསྲོས་ཤིང་། །དྲང་པོར་བྱེད་པ་མཐོང་གྱུར་ནས། །བློ་ལྡན་མཆོག་དེས་རབ་ཏུ་བསམས། །མདའ་ནི་སེམས་མེད་འདི་དག་རྣམས། །གདུང་བས་
ཏ་ནཱཿཤ་རཿ། །ན་ཏུ་སཾ་སཱ་ར་སཾ་ཏཔྟ་མ་པི་བཀྟྲཱཿཤ་རི་རིཎཿ། །ཨི་ཏི་གྷཱ་ཡ་ན་བི་ལོ་ཀྱཱ་པྲེ་ཏ་ཛྙཱ་ཤ་ཀ་ཊ་ཙཀྲ་ནཱཾ། །ནཱི་ཏཱ་ནི་དྲྀ་ཌྷ་རཱུ་པཱ་ཎི་པུ་ནིཤྩིཏྟཱཾ་ས་མཱ་ཡ་ཡཽ། །ཨ་ཧོ་ནུ་དྱ་ཌྷ་ནཱ་ཡོ་གཱཏ་
དྲང་པོ་ཉིད་འགྱུར་ཏེ། །ལུས་ཅན་འཁྱོག་པོ་རྣམས་དག་ནི། །འཁོར་བའི་གདུང་བས་ཀྱང་མ་ཡིན། །ཞེས་སེམས་མདུན་དུ་ཤིང་མཁན་གྱིས། །ཤིང་རྟའི་འཁོར་ལོ་ཉིད་བྱས་པའི། །བརྟན་པའི་གཟུགས་རྣམས་མཐོང་གྱུར་ནས། །སླར་ཡང་ཡང་དག་སེམས་པར་གྱུར། །ཀྱེ་མ་སེམས་མེད་ཤིང་རྣམས་ནི། །
