དྷརྨྨ་ཛི་ཛྙཱ་ས་ཡཱ་རཱ་ཛྙཱ་པི་ཤུ་ན་པྲེ་རི་ཏེ་ན་སཿ། །ནཻ་པཱ་ཀ་རོ་ད་སདྐཱཪྻ་ནི་ཡུཀྟོ་པི་པུ་ནཿཔུ་ནཿ། །ཨཱ་གྷཱ་ཏཾ་པྲེ་ཀྵི་ཏཿསོ྅ཐ་མི་ཐྱཱ་ཀོ་པ་ན་བྷཱུ་བྷུ་ཛཱ། །ཏ་ཐཱ་པྱ་དྷརྨྨ་སཾ་ཡུཀྟཾ་ན་ཤཱ་ས་ན་མ་ནྱ་ཏཱ། །ཨེ་ཀ་
ཕྲ་མ་ཅན་གྱིས་བསྐུལ་བ་ཡི། །རྒྱལ་པོས་ཆོས་ནི་བརྟག་འདོད་ནས། །དམ་པའི་བྱ་བ་མིན་པའི་ལས། །ཡང་ཡང་བསྐོས་ཀྱང་དེས་མ་བྱས། །དེ་ནས་མི་བདག་ཁྲོ་རྫུན་གྱིས། །དེ་ནི་གསོད་དུ་བསྐུལ་བར་གྱུར། །དེ་ལྟ་ན་ཡང་ཆོས་མིན་དང་། །ཡང་དག་ལྡན་པའི་བཀའ་མ་མནོས། །སྐྱེ་བ་གཅིག་
ཛནྨ་སུ་ཁཱ་ཡཻ་བ་བ་ཧུ་ཛནྨ་ཤ་ཏཱརྡི་ཏུཾ། །ན་སཱ་དྷུ་ནིནྡ་ཏཾ་ཀརྨྨ་ཀ་རོ་མཱི་ཏི་ཛཱ་ག་ད་སཿ། །བྷ་ཡ་དྷརྨོ་པ་དྷཱ་ཤུདྡྷཿཔྲ་ཏི་མུཀྟཿས་བྷཱུ་བྷུ་ཛཱ། །དཱ་ན་ཤ་ཏྲི་མ་བིཙྪིནྣ་མ་ཀ་རོ་ད་ཁི་ལརྠི་ནཱཾ། །སརྦ་ཏྲ་བི་ཤྲུ་ཏེ་ཏ་སྱ་
ཉིད་བདེ་བའི་སླད། །སྐྱ་བ་བརྒྱ་མང་ཉམས་འགྱུར་བ། །སྨད་པའི་ལས་ནི་དགེ་བ་བདག །མི་བགྱིད་དོ་ཞེས་དེ་ཡིས་སྨྲས། །ཆོས་ཀྱིས་འཇིགས་ཤིང་ཆོ་འབྲང་གཙང་། །དེ་ནི་ས་བདག་གིས་རབ་བཏང་། །སློང་བ་རྣམས་ནི་མ་ལུས་ལ། །རྒྱུན་ཆད་མེད་པའི་སྦྱིན་ཁང་བྱས། །གྲགས་སྙན་ལྡན་པ་དེ་ཡི་
དཱ་ན་ཤ་ཏྲི་ཡ་ཤ་སྭི་ནཿ། །ཀལྤ་བྲྀཀྵཱ་ད་རཿཔུཾ་སཱཾ་པ་རཾ་པྲ་ཏ་ནུ་ཏཱཾ་ཡ་ཡཽ། །ཨ་ཏྲཱནྟ་རེ་མུ་ནི་ག་ཎསྟཱིརྠཱརྠཱ་དཀྵི་ཎཱ་པ་ཐཱཏ། །ཨཱ་ག་ཏ་ཀྲྀཙྪྭ་ཀཱནྟཱ་ར་མ་པི་ཤན་ནི་རྫ་ནཾ་བ་ནཾ། །ཏ་ཏྲ་ཏྲྀཥྞཱརྟུ་
ནི། །སྦྱིན་ཁང་ཀུན་ཏུ་གྲགས་པའི་ཚེ། །དཔག་བསམ་ཤིང་ལ་སྐྱེས་བུ་རྣམས། །གུས་པ་མཆོག་ཏུ་ཆུང་བར་གྱུར། །གནས་སྐབས་དེར་ནི་ཐུབ་པའི་ཚོགས། །འཇུག་ངོགས་དོན་གཉེར་ལྷོ་ཕྱོགས་ནས། །འོངས་པ་འབྲོག་དགོན་ཉམ་ང་བ། །སྐྱེ་བོ་མེད་པའི་གནས་སུ་ཞུགས། །དེར་ནི་ཐམས་ཅད་
