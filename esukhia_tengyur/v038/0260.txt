ཡང་། དམ་པ་ཡིས་ནི་བྱ་བ་སངས་རྒྱས་ཐམས་ཅད་ལ་བྱ་སྟེ། དེ་དག་གིས་རབ་ཏུ་བསྔགས་ཤིང་སྟོན་པར་མཛད་པ་ཡིན་ནོ། །དེ་ཡང་གང་ཞེ་ན། བསྙེན་པའི་ཆོ་ག་དང་པོ་སྟེ། །གཉིས་པ་ཉེ་བར་སྒྲུབ་པ་དང་། །གསུམ་པ་བསྒྲུབ་པར་བྱ་བ་དང་། །བཞི་པ་སྒྲུབ་པ་ཆེན་པོའོ། །ཞེས་
གསུངས་ཏེ། རྩ་བའི་རྒྱུད་ལེའུ་བཅུ་གཉིས་པ་ལས་ནི། བསྙེན་པའི་དམ་ཚིག་སྦྱོར་བ་དང་། །ཉེ་བར་སྒྲུབ་པ་འབྱུང་བ་དང་། །སྒྲུབ་པའི་དོན་གྱི་དམ་ཚིག་དང་། །སྒྲུབ་པ་ཆེན་པོ་དག་དང་བཞི། །ཞེས་གསུངས་ཏེ། དེ་ལ་བསྙེན་པ་ཞེས་བྱ་བ་ནི་ཐེག་པ་གཞན་ལས་ཁྱད་པར་དུ་གྱུར་པ་
རྡོ་རྗེ་ཐེག་པའི་མྱ་ངན་ལས་འདས་པ་ལ་བསྙེན་ཅིང་འཁོར་བ་ལས་ཐག་བསྲིངས་པར་གྱུར་པ་ལས་མི་འདའ་བའོ། །ཉེ་བར་སྒྲུབ་པ་ཞེས་བྱ་བ་ནི་ལྷའི་ཏིང་ངེ་འཛིན་རྫོགས་པས་དེའི་སྐུ་གསུང་ཐུགས་སུ་གྲུབ་པ་ལ་ཉེ་བའི་རྒྱུར་གྱུར་པའོ། །སྒྲུབ་པ་ཞེས་བྱ་བ་ནི་བསྐུལ་བ་སྟེ། ལྷ་དང་ཡུམ་
བསྐུལ་བས་རང་གི་ལུས་ངག་ཡིད་གསུམ་དེའི་རང་བཞིན་དུ་གྲུབ་པར་བྱེད་པའོ། །སྒྲུབ་པ་ཆེན་པོ་ཞེས་བྱ་བ་ནི་མཉེས་པ་བསྐྱེད་ནས་ཆོས་ཀྱི་རྒྱལ་པོ་ཆེན་པོའི་དབང་ཐོབ་པར་བྱེད་པའི་རྒྱུའོ། །རྩ་བའི་རྒྱུད་དེ་ཉིད་ལས་རྡོ་རྗེའི་བྱེ་བྲག་ཅེས་བྱ་བ་ལ། དེ་ནས་ལས་རྣམས་བསྒྲུབ་པར་བྱ། །ཞེས་
གསུངས་པ་དང་ལེའུ་འདི་ལས། ཐུན་མོང་མཆོག་གི་བྱེ་བྲག་གིས། །ཞེས་གསུངས་པས་དེ་ལྟར་གྲངས་བཞི་བསྟན་པ་དེ་ལ་ཡང་རྣམ་པ་བཞི་སྟེ། སྤྱིའི་བསྙེན་སྒྲུབ་བཞི་དང་། དངོས་གྲུབ་ཐུན་མོང་གི་བསྙེན་སྒྲུབ་བཞི་དང་། མཆོག་གི་བསྙེན་སྒྲུབ་བཞི་དང་། གསང་བའི་བསྙེན་སྒྲུབ་བཞི་
སྟེ། སོ་སོར་ཕྱེ་ན་བཅུ་དྲུག་ཏུ་འགྱུར་བས་དེ་ཡང་ཆུང་ངུ་འབྲིང་དང་ཆེན་པོར་དབྱེ་བའི་དོན་ཡང་རིག་པར་བྱས་ལ། དངོས་གྲུབ་ཐུན་མོང་དང་ཐུན་མོང་མ་ཡིན་པ་གང་དང་གང་སྒྲུབ་ན་ཡང་དེས་བསྒྲུབ་པར་བྱ་སྟེ། ཅིའི་ཕྱིར་ཞེ་ན། བསྙེན་སྒྲུབ་རྣམ་པ་བཞི་ནི་རྒྱུད་འདིར་གསུངས་པའི་ལས་
དང་སྒྲུབ་པ་ཐམས་ཅད་ཀྱི་རྒྱུར་གྱུར་པའི་ཕྱིར་རོ། །སྒྲུབ་པ་ཕྲ་མོ་ཁ་ཅིག་ལ་བསྙེན་སྒྲུབ་ཡན་ལག་བཞི་ཚང་བར་སྒོམ་མི་དགོས་པ་ཡང་ཡོད་དེ། དེ་ཡང་རྩ་བའི་རྒྱུད་བཀྲལ་བའི་སྐབས་ཀྱི་ཊཱི་ཀཱ་ཉིད་ལས་གསལ་བར་ཤེས་པར་བྱའོ། །དེ་ལ་དང་པོར་བསྟན་པའི་བཞི་པོ་འདི་ནི་རྣམ་
