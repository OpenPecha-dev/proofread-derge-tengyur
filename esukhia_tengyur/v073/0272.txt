ཤུདྡྷཿསརྦ་དྷརྨཱཿསྭ་བྷཱ་བ་བི་ཤུདྡྷོ྅་ཧཾ། གང་ཞིག་ཀུན་རྟོག་མཚན་ཉིད་མེད། །དེ་ནི་བདག་སེམས་འོད་གསལ་ཉིད། །འགྲོ་འོང་གནས་དང་རྟེན་བྲལ་བ། །ད་ནི་སྒྱུ་མའི་སྐྱེས་བུ་བཞིན། །ཨོཾ་སྭ་ཙིཏྟ་ཤཱུ་ནྱ་ཏཱ་བཛྲ་སྭ་བྷཱ་བ་ཨཱཏྨ་ཀོ྅ཧཾ། སྟོང་པ་ཉིད་སེམས་སྒྱུ་མའི་ཚུལ། །སྤྲུལ་པའི་མཛད་པ་ཀུན་སྒྲུབ་
ཕྱིར། །སྐྱེ་བ་མེད་ལས་སྐྱེ་བའི་དངོས། །པདྨོ་ཟླ་དབུས་འཕགས་མཆོག་བསྐྱེད། །ཐུགས་རྗེའི་སྙིང་པོ་ལས་བྱུང་བ། །སྤྲོ་དང་བསྡུ་དང་སྡིག་སེལ་བའི། །མཚན་མ་རྒྱལ་བ་སྡུག་བསྔལ་གཅོད། །བཅུ་གཅིག་ཞལ་མངའ་རྣམ་པར་བསྒོམ། །ཕྱག་གཉིས་ཐུགས་ཀར་ཐལ་སྦྱར་ཞིང་། །བགྲང་
ཕྲེང་པདྨོ་གཡས་གཡོན་གཉིས། །རིལ་པ་མཆོག་སྦྱིན་འཁོར་ལོ་དང་། །རྡོ་རྗེ་མདའ་གཞུ་འཛིན་པར་བརྟག །ཏི་སེའི་གངས་དང་སྟོན་ཟླ་སོས་ཀའི་ཉི་འོད་ཅན། །མུ་ཏིག་གསེར་སོགས་ཇི་སྙེད་རྒྱན་མང་ཀུན་མངའ་བ། །ཞལ་ནི་བཅུ་གཅིག་ནོར་བུ་མེ་མདོག་སྐེ་རགས་ཅན། །རི་དགས་ལྤགས་འཛིན་
འཛུམ་ཞལ་གཟི་བརྗིད་འབར་བའོ། །དེ་ནས་སྙིང་པོའི་ས་བོན་གྱིས་སྤྱན་དྲངས་ནས་སྐུ་གསུང་ཐུགས་ལེགས་པར་གནས་པས་དབང་བསྐུར་བའི་མཐར་ཐུག་པའི་སྒྲུབ་པའོ། །སྐྱོ་བར་གྱུར་ན་སྔགས་བཟླས་པས་འཁོར་ལོ་དགྱེ་བའི་སྦྱོར་བ་བརྩམ་པར་བྱ་སྟེ། ཨོཾ་དྷ་ར་དྷ་ར། དྷི་རི་དྷི་རི། དྷུ་རུ་
དྷུ་རུ། ཨིཊྚེ་བིཊྚེ། ཙ་ལེ་ཙ་ལེ། པྲ་ཙ་ལེ་པྲ་ཙ་ལེ། ཀུ་སུ་མེ་ཀུ་སུ་མ་བ་རེ། ཨི་ལི་མི་ལི། ཙི་ཏེ་ཛྭ་ལ་མ་པ་ན་ཡེ་སྭཱ་ཧཱ། ཞེས་བྱ་བས་སོ། །གལ་ཏེ་བློ་ཞན་རྟུལ་མི་ཕོད། །ཞལ་ལྔ་ཕྱག་བརྒྱད་ཞལ་གསུམ་དང་། །ཕྱག་དྲུག་དེ་བཞིན་ཞལ་གཅིག་དང་། །ཕྱག་གཉིས་པ་ཡང་བསྒོམ་པར་

འཛུམ་བཅས་དབུས་བཞུགས་ཟླ་བའི་ཞལ་མངའ་དཔལ་ཆེན་དོན་ཡོད་ཞགས་པ་འཆར་ཀའི་མདོག །བདུད་འཇོམས་རྟ་མཆོག་མེ་ལྕེའི་ཁ་དོག་ཁྲོས་(མདུང་‚མ་དུང་)མདོག་སྐར་མའི་འོད་ལྡན་བྷྲྀ་ཀུ་ཊཱི། །ལྕགས་ཀྱུ་ཞགས་པ་ལྕགས་སྒྲོག་དྲིལ་འཛིན་ཉི་ཤུ་དང་ནི་བཅུ་དྲུག་མཆོད་པའི་ལྷ་མོས་བརྒྱན། །
