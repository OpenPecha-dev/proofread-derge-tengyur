མཐུན་ན་ཡང་ཁྱད་པར་བརྗོད་པ་རྟོག་པ་དང་མཚུངས་པ་ཡིན་ནོ། །དཔེར་ན་སྔ་མ་བཞིན་དུ་བུམ་པ་དང་ཆོས་མཐུན་པས་མི་རྟག་པར་བྱས་ན། ཆོས་མཐུན་པ་དེ་ཡོད་ཀྱང་ཆོས་པར་བྱ་བ་ཉིད་དང་། མིག་གིས་གཟུང་བ་ཉིད་ལ་སོགས་པས་བུམ་པ་ཉིད་མི་རྟག་པ་ཡིན་གྱི། གཞན་ནི་མ་ཡིན་ནོ་
ཞེས་བརྗོད་པ་ལྟ་བུའོ། །གལ་ཏེ་འདིར་ཡང་དབྱིབས་ཉིད་ལ་སོགས་པའི་འབྲས་བུ་སྤྱིས་སྒྲུབ་པར་བྱེད་པ་མ་ཡིན་ནམ། རི་དང་བུམ་པ་ལ་སོགས་པའི་དབྱིབས་ཡོངས་སུ་བརྟགས་པས། འབྲེལ་པ་ཅན་གྱི་སྒོ་ནས་ཁྱད་པར་ཡོངས་སུ་རྟོག་ན་ནི་འབྲས་མཚུངས་བརྗོད་པ་ཉིད་དོ་ཞེ་ན། དེ་ནི་མི་
བདེན་ཏེ། དབྱིབས་ལ་སོགས་པ་དག་གི་སྤྱི། །བློ་སྔོན་བྱེད་པ་ལས་བྱུང་མིན། །དེ་ནི་ཤིང་སོགས་གཞན་ལ་ཡང་། །མཐོང་བའི་ཕྱིར་ཞེས་བརྗོད་ཟིན་ཏེ། །འབྲེལ་པ་ཡི་ནི་ཁྱད་པར་དག །སྤངས་ནས་འགར་ནི་སྒྲུབ་བྱེད་གང་། །དེ་ནི་ཤིན་ཏུ་འབྲེལ་མེད་དེ། །མི་རྟག་བྱ་རོག་ནག་པོས་
བཞིན། །འདིར་གཞན་དག་སྨྲ་བ་ནི། གལ་ཏེ་སློ་སྔོན་དུ་བྱེད་པ་རྗེས་སུ་འགྲོ་བར་མ་མཐོང་བས་དབྱིབས་ལ་སོགས་པ་དེ་ཉིད་སྒྲུབ་བྱེད་དུ་མི་འགྱུར་མོད། དབྱིབས་དབྱིབས་ཞེས་བྱ་བ་ལ་སོགས་པའི་སྒྲས་བརྗོད་པར་བྱ་བ་ནི་རྗེས་སུ་འགྲོ་བ་ཡིན་ཏེ། དེས་ན་དེས་མཚན་ཉིད་གཅིག་པའི་ཕྱིར་གཅིག་
གི་འབྲས་བུར་གྲུབ་པོ་ཞེ་ན། དེ་ནི་མ་ཡིན་ཏེ། འདི་ལྟར། རྟོག་པ་ཙམ་གྱིས་བཀོད་པའི་ཆོས། །དོན་ཉེ་བ་དང་བྲལ་བ་ལས། །དོན་དེ་ཡོད་པར་གྱུར་ན་ནི། །གང་དུ་འདོད་པ་དེར་འགྱུར་རོ། །རིགས་གཞན་ལས་ནི་རབ་གྲུབ་པ། །སྒྲ་ཡི་སྤྱི་ནི་མཐོང་བ་ལས། །སྒྲུབ་བྱེད་མི་རིགས་ངག་
ལ་སོགས། །གཽ་ཡིན་ཕྱིར་རྭ་ཅན་བཞིན། །ངག་ལ་སོགས་པ་དང་མཚན་ཉིད་མི་མཐུན་པ་རིགས་གཞན་ལ་རབ་ཏུ་གྲུབ་པའི་རྭ་དང་འབྲེལ་པ་དེ། ཅི་གོ་ཞེས་ཚིག་ཐ་མི་དད་པ་ཙམ་གྱིས་རྗེས་སུ་དཔག་པར་བྱ་བའམ། དོན་གཅིག་ན་ཡང་སྐྱ་བོ་ལས། །རྗེས་སུ་དཔོག་པ་མིན་ཞེས་བརྗོད། །
གང་ལ་དོན་ཀྱང་ཡོད་མིན་ན། །སྒྲ་ཙམ་མཚུངས་པར་སྨོས་ཅི་དགོས། །རེ་ཞིག་དོན་ནི་མངོན་པར་འདོད་པའི་བསྒྲུབ་བྱ་དང་འབྲེལ་པ་སྲིད་པ་ཉིད་དོ། །དེ་ལ་ཡང་རྟོགས་པ་ཞིབ་མོ་དང་ལྡན་པ་དག་གིས་སྒྲུབ་པར་བྱེད་པ་མ་ཡིན་པར་བརྗོད་ན་གང་ལ་བསྒྲུབ་བྱ་དང་འབྲེལ་པ་འདི་ཙམ་ཡང་རིང་དུ་
