གཞན་མི་སྲིད་པའི་ཕྱིར་རོ། །གལ་ཏེ་འདི་ཉིད་ལས་འབྱུང་བ་ཞེས་བྱ་བ་ཡིན་ན་ནི། ཤུགས་ཀྱིས་གཞན་ལས་མི་འབྱུང་བ་ཡིན་ནོ་ཞེས་བྱ་བའི་དོན་འདིར་འགྱུར་བ་ཡིན་ནོ། །དེ་ལ་ཡང་རྣམ་པར་རྟོག་པ་གཉིས་ཏེ། དེའི་ཚེའམ་འོན་ཏེ་གཞན་གྱི་ཚེ་ཡིན། གལ་ཏེ་
དེའི་ཚེ་འདི་ཉིད་ལས་ཡིན་ནོ་ཞེས་འདོད་ན། འོ་ན་ནི་གཞན་གྱི་ཚེ་ནི་གཞན་ལས་ཡིན་ནོ་ཞེས་བྱ་བར་འགྱུར་རོ། །ད་ལྟར་ཡང་ཡུལ་གཞན་དུ་གཞན་ལས་མི་འབྱུང་ངོ་ཞེས་གང་ལས་ཡིན། དེ་ལྟ་བུ་ནི་རྒྱུ་དང་འབྲས་བུའི་དངོས་པོ་མ་ཡིན་ཏེ། ངེས་པ་མེད་པའི་ཕྱིར་རོ། །འདི་ལས་འབྱུང་བ་
ཉིད་དོ་ཞེས་ངེས་པར་འབྱུང་བ་ཡང་མ་ཡིན་ཏེ། དུས་གཞན་དུ་རེས་འགའ་མི་འབྱུང་བའི་ཕྱིར་རོ། །དེས་ན་མངོན་སུམ་གྱིས་ནི་རྒྱུ་དང་འབྲས་བུ་རྟོགས་པ་མ་ཡིན་ནོ། །འདི་ལས་ཀྱང་འབྱུང་བ་གཞན་ལས་ཀྱང་འབྱུང་བ་ཡིན་ནོ་ཞེས་བྱ་བའི་ཕྱོགས་འདི་ལ་ནི་རང་ཉིད་ཁོ་ནས་འཁྲུལ་པས་ཁས་བླངས་པའི་ཕྱིར་
རྒྱུ་དང་འབྲས་བུའི་དངོས་པོར་མི་འགྱུར་རོ་ཞེས་བསྟན་པར་འགྱུར་ཏེ། དེས་ན་ཕྱོགས་འདི་ཡང་མ་ཡིན་ནོ། །ཅི་སྟེ་དེ་མེད་ན་མེད་པའི་ཕྱིར་རོ་ཞེས་མི་དམིགས་པ་ལ་རྟོགས་སོ་ཞེ་ན། དེ་ལྟར་གྱུར་ན་ཡང་གཅིག་ལ་གཅིག་བརྟེན་པའི་ཉེས་པ་ཡོད་དོ། །འདི་ལྟར། གཅིག་མི་དམིགས་པ་མངོན་སུམ་ལས། །གཞན་ནམ་དེ་
བཞིན་གཞན་མ་ཡིན། །མངོན་སུམ་ལས་ནི་གཞན་མིན་ན། །མིང་ཙམ་སྒྲུབ་བྱེད་ཡིན་པར་འགྱུར། །རྗེས་སུ་དཔག་པ་ཉིད་དུ་ནི། །རྟོགས་པ་འཇུག་པ་མེད་འགྱུར་ཏེ། །གང་ཕྱིར་རྗེས་འགྲོ་ལྡོག་པ་ཡིས། །འབྲེལ་ལ་རྗེས་སུ་དཔོག་ཕྱིར་རོ། །གལ་ཏེ་མི་དམིགས་པ་མངོན་སུམ་ཉིད་ཡིན་ན་ནི་མངོན་སུམ་ལ་
བརྗོད་པའི་སྐྱོན་དེ་ཉིད་དོ། །དེ་སྟེ་རྗེས་སུ་དཔག་པ་ཡིན་ན། དེའི་འབྲེལ་པ་ནི་རྗེས་སུ་འགྲོ་བ་དང་ལྡོག་པ་དག་ལས་འཇུག་པ་ཡིན་ཏེ། དེའི་ཕྱིར་རྗེས་སུ་དཔག་པ་མེད་ན་རྗེས་སུ་འགྲོ་བ་དང་ལྡོག་པ་འཛིན་པ་མེད་ལ། དེ་མེད་ན་ཡང་རྗེས་སུ་དཔག་པ་འཇུག་པ་མ་ཡིན་ནོ། །རྗེས་སུ་དཔག་པ་ཐུག་པ་
མེད་པར་ཡང་འགྱུར་རོ། །འོན་ཏེ་དངོས་པོ་མེད་པའི་ཚད་མས་དེ་མེད་ན་མེད་པར་འཛིན་པ་ཡིན་ནོ་ཞེ་ན། དེ་ཡང་མི་འཐད་དེ། དེ་ལས་ཀྱང་། ཡུལ་དང་དུས་ཉེ་བ་ཅན་དག་གིས་དངོས་པོ་མེད་པར་རྟོགས་པར་འགྱུར་བའི་ཕྱིར་རོ། །གཞན་དག་ནི་རྒྱུ་དང་འབྲས་བུའི་ངོ་བོ་ནི་དཔྱོད་པའི་ཚད་མས་རྟོགས་
