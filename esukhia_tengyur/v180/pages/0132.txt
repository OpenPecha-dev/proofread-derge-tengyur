མི་འཛིན་པ་ཉིད་སྤྱི་ཡིན་ནོ་ཞེས་ཉེ་བར་བསྡུས་པ་ཡིན་ནོ། །དེའི་ཕྱིར་ཐུན་མོང་བ་ཞེས་བྱ་བའོ། །མཆོད་སྦྱིན་ལ་སོགས་པའི་བྱེ་བྲག་རྣམས་ཀྱིས་བྱེད་པའི་དོན་ནི་ངེས་པར་འཛིན་པ་མེད་པ་ཁོ་ན་ཡིན་པར་བསྟན་པ་ནི། དེ་བས་ན་བྱེད་པ་ཞེས་བྱ་བའོ། །ངེས་པར་སྦྱོར་བའི་ཡུལ་གྱི་
སྐད་ཀྱི་དབྱིངས་ཀྱི་དོན་དང་། བྱེད་པའི་དོན་དག་ནི་རྟོགས་པས་ཕྱེ་བ་ཡིན་ནོ། །དེ་ལྟ་མོད་ཀྱི་ཡུལ་མེད་པ་ནི་མ་ཡིན་ནོ་སྙམ་དུ་སེམས་པ་ལ། རང་གི་བྱ་བ་དང་ཞེས་བྱ་བ་སྨོས་ཏེ། སྔ་མའི་བག་ཆགས་དང་དོན་ཡང་ཤེས་པ་དེའི་རྒྱུ་མཚན་ཡིན་གྱི། རེ་རེའི་ནི་མ་
ཡིན་ནོ། །གཏན་མེད་པའི་དོན་ལ་སྒྲ་ནི་ཤེས་པའི་བྱེད་པ་ཡིན་ནོ་སྙམ་དུ་སེམས་པས་སྨྲ་བ་ནི། མངོན་སུམ་ནི་ཞེས་བྱ་བ་སྟེ། གཏན་མེད་བ་ཞེས་བྱ་བ་ནི་ད་ལྟར་བ་མ་ཡིན་པ་འབྱུང་བར་འགྱུར་བ་དང་བྱུང་ཟིན་པའི་ངོ་བོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ལན་དུ་མཐོང་བ་མེད་པའི་དོན་དེ་ནི་ཞེས་
བྱ་བ་སྟེ། ད་ལྟར་བ་མ་ཡིན་པའི་ཡུལ་ལ་ཡང་སྒྲའི་ཤེས་པ་འབྱུང་བ་ནི་དོན་ཡོད་པའི་ངོ་བོར་གྲུབ་པ་ཡིན་ན་ནི་བདེན་ན། རྟོགས་པ་འདི་ཉིད་ལ་དོན་ཡོད་པ་ནི་མ་ཡིན་ཏེ། རང་གི་ངོ་བོར་རིག་པའི་ཕྱིན་མཐོང་བ་མེད་པ་ཞེས་བྱ་བའི་དོན་ཡིན་ནོ། །འབྲས་བུ་ལ་བྱ་བ་བྱེད་པ་པོའི་ངོ་བོ་མི་ཤེས་
པ་ཡིན་ན། དེའི་བྱེད་པ་མེད་ལ་སྦྱིན་སྲེག་བྱས་པས་མཐོ་རིས་ཐོབ་པོ་ཞེས་བྱ་བའི་ཤེས་པ་རྟོགས་པར་བྱེད་པ་མ་ཡིན་པར་བསྟན་པ་ནི། མེའི་སྦྱིན་སྲེག་ལ་སོགས་པ་ལ་ནི་འབྲས་བུའི་རྒྱུ་ཉིད་ཅེས་བྱ་བའོ། །མི་འཇུག་པ་འཇུག་པར་བྱེད་པ་ངེས་སྦྱོར་མ་ཡིན་ནམ། སྐྱེས་བུ་ནི་དེ་ལ་སྦྱར་
བྱ་ཡིན་པ་ཉིད་ཀྱིས་དེ་དང་འབྲེལ་པ་ཅན་ཡིན་གྱི་བྱེད་པ་པོ་ཉིད་ཀྱིས་ནི་མ་ཡིན་ནོ། །སྐད་ཀྱི་དབྱིངས་ཀྱི་དོན་ནི་ཡུལ་ཉིད་ཀྱི་འབྲེལ་པ་ཅན་ཡིན་གྱི་བྱེད་པ་ཉིད་ཀྱིས་མ་ཡིན་ལ། དེ་ལིང་ལ་སོགས་པའི་བྱ་བ་ནི་མ་ཡིན་ཏེ། འཛུད་པར་བྱེད་པའི་བྱ་བ་ཉིད་དང་། ཚད་མ་གཞན་གྱི་སྤྱོད་
ཡུལ་ཡིན་པའི་ཕྱིར་དང་། མ་བཅད་པ་གཅོད་པ་པོ་ཉིད་མེད་པར་ཐལ་བར་འགྱུར་རོ། །འདི་ནི་རྫོགས་བྱ་ཉིད་ཀྱི་འབྲས་བུ་ཡང་ཡིན་ལ། འཇུག་པ་ལ་ཕྱོགས་པའི་གེགས་བྱེད་པས་དགག་པ་ཡང་ཡིན་ཏེ། སྐྱེས་བུའི་འཛུད་བྱེད་དང་། སྒྲིབ་བྱེད་དག་གི་བསྒྲུབ་པ་དང་དགག་པའི་
