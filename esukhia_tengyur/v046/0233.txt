སོགས་པའོ། །བསྒྲུབ་བྱ་དགུག་པའི་དུས་སུ་ཇི་ལྟར་བསྒོམ་གསུངས་པ་ནི་བུད་མེད་ཀྱི་ནི་པདྨ་རུ་ཞེས་བྱ་བ་སྟེ། མངོན་པར་འདོད་པའི་བསྒྲུབ་བྱ་པདྨའི་གེ་སར་ལ་གནས་པར་རོ། །རྡོ་རྗེ་ལུས་མེད་ཀྱི་ནི་སྔགས་ཏེ་སྔགས་དང་ལྷ་དབྱེར་མེད་པར་རོ། །
དེ་ཡང་སཱི་ཏ་ཞེས་བྱ་བའོ། །དོན་ནི་འདི་ཡིན་ཏེ། བསྒྲུབ་བྱ་དགུག་པའི་དུས་སུ་ཇི་སྐད་བཤད་པའི་གནས་སུ་སྔགས་བསྒོམ་པར་བྱའོ། །རྣ་བའི་དྲུང་དུ་བཀང་བའི་མདའ་འཕང་བར་བྱ་སྟེ། མྱོས་པར་བྱེད་པ་དང་རྨོངས་པར་བྱེད་པ་དང་འདར་བར་བྱེད་པ་དང་། ལྟུང་བར་
བྱེད་པ་དང་བརྒྱལ་བར་བྱེད་པའི་མདའ་ལྔས། མགོ་བོ་དང་། མགྲིན་པ་དང་། སྙིང་ག་དང་། ལྟེ་བ་དང་། གསང་བ་རྣམས་ཕུག་ནས་སྙིང་པོའི་སྔགས་ཀྱི་འོད་ཀྱིས་བཀུག་པར་གྱུར་པར་བསམ་པར་བྱའོ། །ཡི་གེ་སཱི་ཏ་ལས་ཡང་དག་བྱུང་ཞེས་བྱ་བ་ནི་སཱི་ཏ་ལས་སྐྱེས་པའོ། །བསླུས་
ཤིང་ཞེས་བྱ་བ་ནི་འདོད་པར་བྱ་བ་ལ་མོས་པར་བྱའོ། །གཡེངས་པ་ནི་མྱོས་པའོ། །མྱོས་ནས་ཞེས་བྱ་བ་ནི་གཡོ་བའོ། །བདེ་བ་འདོད་པ་ནི་འདོད་པ་ལ་ཞེན་པའོ། །བཀུག་ནས་རྐང་པའི་དྲུང་དུ་བཏུད་པ་གོས་དམར་པོས་ཡོངས་སུ་དཀྲིས་པར་བསམ་པར་བྱའོ། །སྔགས་བཏུ་བ་
གསུངས་པ། དེར་ནི་ཞེས་བྱ་བ་ལ་བསྒོམ་པ་དེ་ལའོ། །ཡི་གེ་ཨོཾ་ནི་རིང་པོར་དྲང་། །ཞེས་བྱ་བ་ནི་དང་པོར་ཡི་གེ་ཨོཾ་ཉེ་བར་མཚོན་པར་བྱའོ། །ཡི་གེ་སྭཱ་ཧཱ་ཐ་མར་སྦྱིན། །ཞེས་བྱ་བ་ནི་མཐར་དེ་སྦྱིན་པའོ། །འདི་དག་གི་བར་དུ་ཡི་གེ་སཱི་ཏའི་སྔགས་བརྗོད་པ་ནི་བཟླས་པའོ། །
སྔགས་བསྡུས་པ་འདི་ཡིན་ཏེ། ཨོཾ་ཧྲི་སཱི་ཏ་སྭཱ་ཧཱ་ཞེས་བྱ་བའོ། །ཧྲི་ཞེས་བྱ་བ་ནི་མན་ངག་ལས་ཤེས་པར་བྱའོ། །ཆེ་གེ་མོ་ཞིག་བདག་དབང་གྱུར། །ཅེས་བྱ་བ་ནི་སྒྲུབ་པའི་དུས་སུ་བསྐུལ་ཚིག་དང་ལྡན་པའོ། །ཐམས་ཅད་དུ་ནི་བསྒོམ་བྱ་མིན། །ཞེས་བྱ་བ་ནི་རྟག་ཏུའོ། །
ཞག་བདུན་བར་དུ་ཞེས་བྱ་བའི་ཚིག་ཡོད་པ་དེར་ནི་སྒྲུབ་པའི་དུས་སུ་ཉི་མ་བདུན་བསྒོམ་པའོ། །འདིར་ཅིར་འགྱུར་ཞེ་ན་གསུངས་པ། འདོད་པ་དེ་འཐོབ་འགྱུར་ཞེས་པའོ། །གང་རྡོ་རྗེ་ལུས་མེད་ཀྱི་སྒྲུབ་ཐབས་གསུངས་པ་དེ་ལ་ཡང་དེ་སྐད་ཅེས་བྱའོ། །དཔལ་གཤིན་
