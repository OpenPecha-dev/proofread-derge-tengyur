མི་ཁྱབ་པ་ནི། རྡོ་རྗེ་སེམས་དཔའི་སྦྱོར་བ་ཡིས། །ཞེས་བྱ་སྟེ་བསྐྱེད་པ་འདི་ལ་རབ་ཏུ་གནས་པའོ། །དེ་དག་བསམ་གྱིས་མི་ཁྱབ་པ་ཡིན་ཡང་དངོས་པོ་ཉིད་དུ་ནི་ཡོད་དོ་ཞེས་ཐེ་ཚོམ་ཟ་བ་བསལ་བའི་ཕྱིར། དངོས་<པ་,པོ་>ཐམས་ཅད་དངོས་མེད་བཤད། །ཅེས་བྱ་བ་གསུངས་ཏེ། ཇི་ལྟར་དངོས་
པོ་འདི་དག་ཐམས་ཅད་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བས་སོ་སོར་སྣང་བ་དེ་བཞིན་དུ་དངོས་པོར་བརྗོད་པར་མི་བྱ་ཞིང་དངོས་པོར་བརྗོད་པར་བྱ་བ་མ་ཡིན་ཏེ། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བའི་ཕྱིར་རོ་ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་དང་དེ་བརྟེན་གང་འབྱུང་བ། །རང་གི་ངོ་བོར་དེ་མ་སྐྱེས། །རང་གི་
ངོ་བོར་གང་མ་སྐྱེས། །དེ་ལ་སྐྱེས་ཞེས་ཇི་ལྟར་བྱ། །ཞེས་གསུངས་པ་ཡིན་ནོ། །དེ་ལྟ་ན་ནི་དངོས་པོ་འདི་དག་ཉིད་དངོས་པོ་མེད་པར་འགྱུར་རོ་ཞེས་ཐེ་ཚོམ་ཟ་བ་བསལ་བའི་ཕྱིར། དངོས་པོ་མེད་ཅེས་མ་གྲགས་པ། །ཞེས་གསུངས་ཏེ། རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བའི་དངོས་པོ་
དེ་དངོས་པོ་མེད་པར་གྲགས་ཏེ། དངོས་པོ་མེད་པར་ངོ་བོར་མི་རྟག་ཅེས་བྱ་བའི་དོན་ཏོ། །གལ་ཏེ་དེ་ལྟར་ན་རྣལ་འབྱོར་པ་ཤེས་རབ་དང་ཐབས་ལ་སྙོམས་པར་ཞུགས་པའི་རྣལ་འབྱོར་ཉིད་དུ་ཇི་ལྟར་འབྲེལ་ཞེ་ན། དེའི་ཕྱིར་གསུངས་པ། ཆད་པ་མ་ཡིན་ཞེས་བྱ་བ་ལ་ཆད་པ་ནི་མེད་
པའོ། །རྟག་པ་མེད་པ་ཞེས་བྱ་བ་ལ་རྟག་པ་ནི་ཐེར་ཟུག་གོ། །འང་གི་སྒྲས་ནི་རྟག་པ་དང་མི་རྟག་པ་དང་། རྟག་པ་ཡང་མ་ཡིན་མི་རྟག་པ་ཡང་མ་ཡིན་པ་དེ་ལ་ཅུང་ཟད་ཀྱང་བསམ་དུ་མེད་པའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །ཕྱག་དང་ཞལ་གྱི་སྒོ་ནས་གསུངས་པ། ཡན་ལག་མེད་ཅིང་ཁ་
གདོང་མེད། །ཅེས་བྱ་བ་ལ་སོགས་པ་ལ། ཕྱག་དང་ཞལ་ལ་སོགས་པའི་རྟོག་པ་ཡང་མི་རྟག་པའོ། །བསམ་གྱིས་མི་ཁྱབ་པའི་ལྷན་ཅིག་སྐྱེས་པའི་རིག་པའི་ངོ་བོ་ཡིན་པའི་ཕྱིར་ཞེས་དགོངས་པའོ། །དེ་ཉིད་གསུངས་པ། མཆོག་ཏུ་དགའ་བའི་མཚན་ཉིད་དོ། །ཞེས་བྱ་བ་ལ་མཆོག་ནི་བདག་དང་
གཞན་གྱི་དོན་ཕུན་སུམ་ཚོགས་པའོ། །དགའ་བ་གང་ལ་དེ་ཡོད་པ་དེ་ནི་མཆོག་ཏུ་དགའ་བ་སྟེ། ལྷན་ཅིག་སྐྱེས་པ་ཡང་དག་པར་རིག་པའི་བདེ་བ་ཆེན་པོའི་ངོ་བོའོ། །དེ་ཉིད་ཀྱི་མཚན་ཉིད་གང་ཡིན་པའི་ཕྱག་དང་ཞལ་ལ་སོགས་པ་ནི་མཆོག་ཏུ་དགའ་བའི་མཚན་ཉིད་དོ། །འདིས་མཚོན་
