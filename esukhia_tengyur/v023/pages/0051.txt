དྲེཾ་དྷཾ་དྲེཾ་དྷཾ། གྲཾ་གྲཾ། ཧེ་ཧེ་ཧ་ཧ། བྷཱི་མེ། ས་ཧ་ས་ཧ། བཱི་རེ་ཧཱ་ཧཱ་ཧཱ་ཧཱ་ཧོ་ཧུཾ་ཧཱུཾ། ཏྲཻ་ལོཀྱ་བི་ནཱ་ཤ་ནི། ཤ་ཏ་ས་ཧ་སྲ་ཀོ་ཊི་ཏ་ཐཱ་ག་ཏ་པ་རི་ཝཱ་རི་ཏེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ། སིཾ་ཧ་རཱུ་པེ་ཁཿག་ཛ་རཱུ་པེ་གཿཏྲཻ་ལོཀྱ་ད་རེ། མ་ཧཱ་ས་མུ་དྲ་མེ་ཁ་ལེ། གྲ་ས་གྲ་ས་ཧཱུཾ་ཕཊ། པཱི་ར་
དྭཻ་ཏེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧཱ་ཧཱ་མ་ཧཱ་པ་ཤུ་མོ་ཧ་ནི་མ་ཧཱ་ཡོ་གཱི་ཤྭ་རི་ཏྭཾ། ཌཱ་ཀི་ནཱི་ལོ་ཀཱ་ནཱཾ་བནྡྷ་ན། སདྱཿཔྲ་ཏྱ་ཡ་ཀ་རི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། བྷཱུ་ཏ་ཏྲཱ་ས་ནི་མ་ཧཱ་བཱི་རེ་པ་ར་མ་སིདྷྱེ་བིདྱེ་ཤྭ་རི་ཕཊ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ། བཅོམ་ལྡན་འདས་མའི་རྩ་བའི་སྔགས་སོ།། །།ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པི་
ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙུམྦི་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པ་རཱ་བྲྀ་ཏཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ས་བཱ་ལི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ནུ་ཝརྟྟཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་མེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡོ་གཱི་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷ་དྲཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་པཱ་ལི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
ཀཾ་ཀཱ་ལ་ཀཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱ་ཛ་བརྟྟཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁཎྜ་རོ་ཧཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤྨ་ཤཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་དྲ་པཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུ་རུ་ཀུལླི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རུ་ད་ན་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ན་ཊཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།
ཨོཾ་བཛྲ་བྷཻ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤི་ཁཎྜཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ཊཱ་ལཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རུ་དཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་མེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་ཤྭ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁཎྜ་
རོ་ཧེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པདྨ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཱུ་པི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་རཏྣ་ཌཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་ཙཎྜཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁཎྜ་ཀ་པཱ་ལི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙཎྜཱ་ཀྵཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧཱ་ཀཾ་ཀ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་བྷཱ་ཝ་ཏི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཾ་ཀ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ།
ཨོཾ་བཛྲ་མ་ཧཱ་ནཱ་སེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་ཀ་ཊ་དཾཥྚྭ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཱི་ར་མ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སུ་རཱ་བཱི་རི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཁརྦ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་མི་ཏཱ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཾ་ཀེ་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དྲཱུ་མཙྪཱ་ཡེ་ཧུཾ་ཧཱུཾ་
