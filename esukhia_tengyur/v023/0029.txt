སྟོབས་དང་། དྲག་ཅན་གྱི་བྱིན་པ་དང་། ཆོས་ཀྱི་དབང་ཕྱུག་དང་། གཟི་བརྗིད་འོད་དང་། འོད་ཟེར་ཆེན་པོ་དང་། རིན་ཆེན་གྲགས་པ་དང་། ཚོགས་ཅན་གསལ་བ་དང་། གྲགས་པ་མཐའ་ཡས་པ་དང་། འོད་ཟེར་དོན་ཡོད་པ་དང་། ལྷའི་དྲང་སྲོང་དང་། སྐྱེ་
བའི་དབང་པོ་དང་། དགེ་འདུན་བརྟན་པ་དང་། ཕྱོགས་{𰵌བཟངས་,བཟང་}དང་། རྟོག་དང་། ཡུལ་འཁོར་མེ་ཏོག་དང་། ཆོས་ཀྱི་བློ་གྲོས་དང་། རླུང་གི་ཤུགས་ལྟར་གཤེགས་པ་དང་། ལེགས་པའི་གྲགས་པ་དང་། སྣང་ལྡན་དང་། ལྷའི་ཚོགས་དང་། ཡོན་ཏན་སྲུང་དང་། དོན་གྱི་
བློ་གྲོས་དང་། བསྙེངས་པ་མེད་པ་དང་། གྲགས་པ་བརྟན་པ་དང་། འོད་གནས་མཚུངས་པ་དང་། ནོར་བུའི་ཞབས་དང་། ཐར་པའི་གཟི་བརྗིད་དང་། ངོས་{𰵍བཟངས་,བཟང་}དང་། བློ་{𰵎བཟངས་,བཟང་}དང་། ཀུན་དུ་གཟི་བརྗིད་དང་། ཡེ་ཤེས་མཆོག་དང་། ཚངས་པར་གནས་དང་།
བདེན་པར་གསུང་དང་། བློ་{𰵏བཟངས་,བཟང་}དང་། སྟོབས་བྱིན་དང་། སེང་གེའི་སྟབས་དང་། མེ་ཏོག་གི་ཏོག་དང་། ཡེ་ཤེས་འབྱུང་གནས་དང་། མེ་ཏོག་བྱིན་དང་། ཡོན་ཏན་སྙིང་པོ་དང་། རིན་ཆེན་གྲགས་པ་དང་། གྲགས་པ་རྨད་དུ་བྱུང་བ་དང་། མི་ཚུགས་པ་དང་།
བསྙེངས་པ་མེད་པ་དང་། ཉི་མའི་འོད་དང་། ཚངས་པར་གཤེགས་པ་དང་། ལྷའི་མཐུ་རྩལ་དང་། ཡེ་ཤེས་དགྱེས་དང་། ལྷའི་བདེན་པ་དང་། ནོར་བུའི་སྙིང་པོ་དང་། ཡོན་ཏན་གྲགས་པ་དང་། ཡེ་ཤེས་དཔལ་དང་། བཅིང་བ་མེད་པ་དང་། བརྟུལ་ཞུགས་བརྟན་པ་དང་། ལྷའི་གཟི་
བརྗིད་དང་། ཚངས་ཐུབ་དང་། བརྟུལ་ཞུགས་དཀའ་ཐུབ་དང་། འོད་ཟེར་ཕུང་པོ་དང་། གཟི་བརྗིད་ཆེ་བ་དང་། ཙམ་པ་ཀ་དང་། དགའ་མཛད་དང་། ཚོགས་ཅན་བཟང་པོ་དང་། དབང་པོའི་རྒྱལ་མཚན་དང་། དགྱེས་པ་ཆེན་པོ་དང་། སྣ་མའི་མེ་ཏོག་འོད་དང་། ཚོགས་ཅན་
འོད་དང་། སྐྲུན་མཛད་དང་། མདངས་འགྲོ་དང་། དོན་ལེགས་ངེས་པ་དང་། ཁྱུ་མཆོག་དང་། ལག་{𰵐བཟངས་,བཟང་}དང་། འོད་ཟེར་ཆེན་པོ་དང་། བསམ་པས་བྱིན་པ་དང་། བསོད་ནམས་འོད་དང་། རིན་ཆེན་དབྱངས་དང་། རྡོ་རྗེའི་སྡེ་དང་། འབྱོར་ལྡན་དང་། སེང་གེའི་སྟོབས་དང་།
