ཡས་སྐྱེ་མཆེད་མ། །ཅིའང་མེད་སྐྱེ་མཆེད་མ་དེ་ནས། །འདུ་ཤེས་འདུ་ཤེས་མེད་མིན་མ། །དམྱལ་མོ་{𰵗ཡི་དགས་,ཡི་དྭགས་}མ་དེ་བཞིན། །དུད་འགྲོ་མི་མོ་ལྷ་མིན་མོ། །གཞལ་མེད་མཁའ་སྤྱོད་མ་དེ་བཞིན། །ཟླ་མོ་ཉི་མོ་གཤིན་རྗེ་མོ། །དབང་མོ་འཁོར་སྒྱུར་མར་ཤེས་བྱ། །རང་
གི་འཁོར་ལོའི་མདོག་བཞིན་བྱ། །ཕྱག་དང་མཚོན་ཆ་སྔར་བཞིན་ནོ། །ཤེས་རབ་ཐབས་ཀྱི་རང་བཞིན་ཏེ། །ཉེ་བའི་གྲོང་མཐར་ཡང་དག་གནས། །ས་ནི་མོས་པས་སྤྱོད་པའོ། །དེ་ནས་དེ་ཡི་ཕྱི་ཁྱམས་ལ། །འཕགས་པ་དགྱེས་པ་དང་། མཐུ་སྟོབས་དང་། སྐར་མ་ལ་དགྱེས་པ་དང་།
སྤྲིན་གྱི་རྔ་སྒྲ་དང་། སྤྱན་སྡུག་ཞལ་དང་། ཡེ་ཤེས་བཟང་ལྡན་དང་། འབྱོར་ལྡན་དང་། ཡོན་ཏན་ཕུང་པོ་དང་། གསལ་བ་དང་། ཆོས་ཀྱི་རྒྱལ་མཚན་དང་། ཡེ་ཤེས་ང་རོ་དང་། ནམ་མཁའ་དང་། མཆོད་སྦྱིན་ང་རོ་དང་། ཤེས་རབ་རྣམ་དགོད་ང་རོ་དང་། ཡོན་ཏན་འོད་ཟེར་གཟི་
བརྗིད་དང་། དྲང་སྲོང་དབང་པོ་དང་། བློ་ལྡན་དང་། སྟོབས་པའི་ཚོགས་དང་། མཆོད་སྦྱིན་བཟང་པོ་དང་། ཟླ་བའི་ཞལ་དང་། ཤིན་ཏུ་གཟིགས་དང་། རྡུལ་བྲལ་དང་། ཡོན་ཏན་ཚོགས་དང་། ཏོག་ལྡན་དང་། བསོད་ནམས་རྒྱལ་མཚན་དང་། ཡུལ་འཁོར་སྤོབས་པ་དང་།
རིན་ཆེན་རབ་སྦྱིན་དང་། ཟླ་དགྱེས་དང་། སེང་གེའི་སྟོབས་དང་། དབང་སྒྱུར་རྒྱལ་པོ་དང་། བདུད་རྩི་གསལ་བ་དང་། བསམ་གཏན་སྙོམས་དང་། འཁྲུལ་པ་མེད་པ་དང་། དྲི་མ་རབ་ཏུ་ཞི་བ་དང་། ཕྱོགས་མ་བསླད་དང་། མཛེས་པ་དང་། ཞལ་{𰵘བཟངས་,བཟང་}དང་། ཤུགས་གནས་ཡེ་
ཤེས་ལྡན་དང་། གསུང་གི་དབང་པོ་དང་། གཟི་བརྗིད་ཆེན་པོ་དང་། བློ་གྲོས་ཟབ་མོ་དང་། བདུད་རྩི་དང་། ཆོས་ཀྱི་སྟོབས་དང་། མཆོད་པར་ལྡན་པ་དང་། མེ་ཏོག་འོད་དང་། ཁམས་གསུམ་དག་གི་མཆོད་པའི་གནས་དང་། སྒྲ་གཅན་ཉི་མའི་སྙིང་པོ་དང་། ལྷའི་མཆོད་པ་དང་།
ཐར་པའི་རྒྱལ་མཚན་དང་། དགེ་བའི་གཙུག་དང་། བདུད་རྩིའི་འོད་དང་། རྡོ་རྗེ་དང་། བརྟན་པ་དང་། རིན་ཆེན་ཕུང་པོ་དང་། མཛེས་པར་གཤེགས་དང་། ཉི་མ་ལྟ་བུར་གཤེགས་པ་དང་། དག་པའི་འོད་དང་། ཡོན་ཏན་གཙུག་དང་། དཔལ་རྫོགས་པ་དང་། སེང་གེའི་
