ལྷན་ཅིག་སྐྱབས་ཡོད་པ་ན་འགྲེང་ན་ལྟུང་བའོ། །ལྟུང་བའི་ཁྱད་པར་ནི་གོང་མ་དང་འདྲའོ། །ལྟུང་བ་བརྒྱད་ཅུ་གསུམ་པ་རྫོགས་སོ།། །།སྟོན་པ་མཉན་དུ་ཡོད་པ་ན་སྟེ། དགེ་སློང་མ་སྦོམ་དགའ་མོས་བསམས་པ་དབེན་པ་སྐྱབས་ཡོད་པར་ནི་འགྲེང་ཞིང་གཏམ་
ཟེར་དུ་བཅོམ་ལྡན་འདས་ཀྱིས་མ་གནང་བས། བླ་གབ་མེད་པར་འགྲེང་ཞིང་གཏམ་བྱས་ཀྱང་ཉེས་པ་མེད་དོ་སྙམ་ནས། འགྲེང་ཞིང་གཏམ་རྣམ་པ་སྣ་ཚོགས་ཟེར་བ་ལས་བཅས་སོ། །ཡང་དགེ་སློང་མ་གང་སྐྱེས་པ་དང་། ལྷན་ཅིག་བླ་གབ་མེད་པ་ན་འགྲེང་ན་ལྟུང་
བའོ། །ལྟུང་བའི་ཁྱད་པར་ནི་གོང་མ་དང་འདྲའོ། །ལྟུང་བ་བརྒྱད་ཅུ་བཞི་པ་རྫོགས་སོ།། །།སྟོན་པ་མཉན་དུ་ཡོད་པ་ན་སྟེ། སྦེད་མ་དང་འཆར་བ་གཉིས་བླ་གབ་མེད་པར་འགྲེང་ཞིང་གཏམ་རྣམ་པ་སྣ་ཚོགས་ཟེར་བ་ལས་བཅས་ཏེ། ཡང་དགེ་སློང་མ་གང་
གཅིག་པུ་དགེ་སློང་དང་ལྷན་ཅིག་བླ་གབ་མེད་པ་ན་འགྲེང་ན་ལྟུང་བའོ། །འདིའི་ཁྱད་པར་ཡང་གོང་མ་དང་འདྲའོ། །ལྟུང་བ་བརྒྱད་ཅུ་ལྔ་པ་རྫོགས་སོ།། །།སྟོན་པ་མཉན་དུ་ཡོད་པ་ན་སྟེ། དགེ་སློང་མ་སྦོམ་དགའ་མོ་གཅིག་པུ་གནས་ཁང་གཅིག་ན་གནས་པ་ལས་ཕྱེ་མ་
རེད་ཁར་སྤོས་འཚོང་གི་ཁྱེའུ་ཁྲིའི་འོག་ཏུ་སྦས་ཏེ་བཞག་ནས། དེའི་འོག་ཏུ་བརྗེད་དེ་ཁྱེའུ་དེ་བྱུང་ནས་སྦོམ་དགའ་མོ་ཡང་སྐྲག་པར་གྱུར་ནས་རྐུན་པོ་བྱུང་ངོ་ཞེས་ཁུས་བཏབ་པ་དང་། དགེ་སློང་མ་དག་གིས་སྨྲས་ཏེ་སྤོས་འཚོང་ཡིན་པ་དང་འཕྱ་བས། བཅོམ་ལྡན་འདས་ཀྱིས་
ཁྱིམ་ཟླ་མེད་པར་དགེ་སློང་མ་གཅིག་པུ་མ་གནས་ཤིག །གཅིག་པུ་གནས་ན་ལྟུང་བར་འགྱུར་རོ་ཞེས་བཅས། ཡང་དགེ་སློང་མ་གང་གཅིག་པུ་གནས་ཁང་ན་གནས་ན་ལྟུང་བའོ། །དགེ་སློང་མས་ཤེས་བཞིན་དུ་གཅིག་པུ་གནས་ཁང་གཅིག་གི་ནང་དུ་གནས་པར་བྱེད་ན་ལྟུང་
བའོ། །བསམ་གཏན་གྱི་མཐུ་ཁུགས་སུ་ཉེ་བའམ་ཁུགས་ཟིན་ན་ཉེས་པ་མེད་དོ། །ལྟུང་བ་བརྒྱད་ཅུ་དྲུག་པ་རྫོགས་སོ།། །།སྟོན་པ་མཉན་དུ་ཡོད་པ་ན་སྟེ། སྦོམ་དགའ་མོས་སྐྱེས་པའི་རྣ་བར་བཤུབས་པ་ལས་བཅས། ཡང་དགེ་སློང་མ་གང་སྐྱེས་པའི་རྣ་བར་བཤུབས་
