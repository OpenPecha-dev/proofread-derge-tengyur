ཨོཾ་མཧཱ་བཛྲ་ཀྲཱུ་ར་ཀྲོ་དྷ་པ་ཏ་ཡེ་སརྦ་དུཥྚཱཾ་མཎྜལཾ་བི་ནཱ་ཤ་ཡ། སརྦ་དུཥྚ་ས་མ་ཡཾ་བ་ཀི་ར་བི་དྷྭནྶ་ཡ་སྥོ་ཊ་ཡ། བྷཉྫ་སརྦ་དུཥྚཱཾ་ས་མ་ཡ་མུ་དྲ་བནྡྷ། གྲྀཧྞ་ཧ་ན་ད་ཧ་པ་ཙ་སརྦ་དུཥྚཱཾ་ས་མ་ཡ་ས་ཏྭཱཾ་བཛྲ་ས་མ་ཡ་ཧཱུཾ་ཕཊ་ཅེས་བྱ་བ་དང་ཁྲོ་བོ་ཆེན་པོ་གདུག་པ་ཕྱག་དང་ཞལ་སྟོང་མངའ་བས་མ་
རུངས་པ་ཐམས་ཅད་བཅོམ་ནས་རྒྱ་མཚོ་ཆེན་པོར་ལྟུང་བར་བྱས་པར་བསམ་མོ། །དེ་ནས་འདིའི་ཕྱག་རྒྱ་བཤད་དེ། གཡོན་པའི་རྡོ་རྗེ་མཛུབ་མོ་ལ། །གཡས་པའི་མཛུབ་མོ་བཀུག་པས་གཟུང་། །རབ་ཏུ་ཁྲོས་ནས་དྲག་གཏོགས་ནས། །རི་རབ་ཀྱང་ནི་ལྷུང་བར་བྱེད། །ཨོཾ་ཧ་ན་ཧ་ན་བཛྲ་
ཧཱུཾ་ཕཊ། ཅེས་བྱ་བའི་དེ་བཞིན་གཤེགས་པའི་བཀའ་དང་། ཨོཾ་བཛྲ་ཨཱ་ཀཪྵ་ཡ་ཤཱི་གྷྲཾ་སརྦ་དུཥྚཱཾ་ད་ཧ་ན་བཛྲ་དྷ་ར་སཏྱེ་ན་ཧཱུཾ་ཛཿཞེས་བྱ་བའི་ཕྱག་རྒྱ་དང་བཅས་པས་ཕྲ་མེན་ལ་སོགས་པ་གདོན་མ་རུངས་པ་ཐམས་ཅད་དགུག་པར་བྱའོ། །ཕྱག་རྒྱ་ནི་འདི་ཡིན་ཏེ། གཡོན་པའི་རྡོ་རྗེ་མཛུབ་མོ་ལ། །
གཡས་པའི་མཛུབ་མོ་ལྕགས་ཀྱུས་གཟུང་། །རབ་ཏུ་ཁྲོས་ཤིང་དགུག་པ་ཡིས། །གདོན་རྣམས་ཐམས་ཅད་དབང་དུ་བྱེད། །ཨོཾ་བཛྲ་པྲ་ཏི་གྲྀཧྣ་ཧྲྀ་ད་ཡ་ཨཱ་ཀཪྵ་ཨཱདྱ་ཡན་སཏྭ་ཨརྡྷེ་ན་མྲི་ཡ་ཏེ་ཏ་སྱ་ཧྲྀ་ད་ཡཾ་ནིཥྛ་མ་ཏུ་ས་མ་ཡ་ཧཱུྃ་ཛ་ཞེས་བྱ་བ་དང་། ཕྱག་རྒྱ་འདིས་འཆི་བར་གྱུར་པ་དེ་རྣམས་
ཀྱི་མཚན་མོ་སྙིང་ཆུང་བ་ལ་ཕྲ་མན་མ་གདོན་རྣམས་ལ་སྦྱིན་ནོ། །དེ་རྣམས་དགའ་སྟེ་ཐམས་ཅད་མཁྱེན་པའི་ཡེ་ཤེས་གྲུབ་པར་བྱེད་དོ། །ཕྱག་རྒྱ་དེ་ནི་འདི་ཡིན་ཏེ། རྡོ་རྗེ་བཅིང་བ་བྱས་ནས་ནི། །ལག་ངར་སྙིང་གར་ཤིན་ཏུ་བསྡམས། །རྡོ་རྗེ་མཛུབ་མོ་རང་མཆན་ཁུང་། །བྲུད་པ་ཡི་ནི་
སྙིང་སྦྱིན་ནོ། །ཨོཾ་ཧཱུཾ་ཕཊ་ཅེས་བྱ་བ་དེ་བཞིན་གཤེགས་པའི་བཀའ་དང་། ཨོཾ་བཛྲ་ས་མ་ཡ་ཨ་ན་ཡ་སརྦ་དུཥྚ་ཛྭ་རཱ་དི་ན་ཨཱ་ཀཪྵ་ཡ་ཧཱུཾ་ཕཊ། ཅེས་ཟློས་ཤིང་ཕྱག་རྒྱ་དང་བཅས་པས་རིམས་ལ་སོགས་པ་དགུག་པར་བྱའོ། །ཕྱག་རྒྱ་ནི་འདི་ཡིན་ཏེ། མཛུབ་མོ་གཉིས་ཀྱི་རྩེ་མོ་ནི། །མཐེ་བོང་
གཉི་གའི་དྲུང་དུ་ནི། །ཡིད་ཀྱིས་ཁྲོས་ནས་བཙིར་བར་བྱ། །རིམས་ལ་སོགས་པར་དགུག་པར་བྱ། །ཨོཾ་བཛྲ་ཀརྨ་བི་ཤོ་དྷ་ཡ་སརྦ་ཨཱ་བ་ར་ཎ་ནི་བུདྡྷ་སཏྱེ་ན་ས་མ་ཡ་ཧཱུཾ་ཞེས་བྱ་བ་འདིས་ཕྱག་རྒྱ་དང་བཅས་པས་རང་ངམ་གཞན་གྱི་རིམས་ལ་སོགས་པའི་ནད་བསལ་བར་བྱའོ། །དེ་ལ་ཕྱག་
