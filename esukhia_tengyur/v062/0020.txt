ཕྱག་རྒྱའི་གཟུགས་ཀྱིས་བདག་ཉིད་དུ་སྙིང་པོ་འདིས་དབབ་པར་བྱའོ། །བཛྲ་རཱ་ཛ་ཨ། བཛྲ་རཱ་ག་ཨ། བཛྲ་སཱ་དྷུ་ཨ། བཛྲ་རཏྣ་ཨ། བཛྲ་ཏེ་ཛ་ཨ། བཛྲ་ཀེ་ཏུ་ཨ། བཛྲ་ཧཱ་སྱ་ཨ། བཛྲ་དྷརྨ་ཨ། བཛྲ་ཏཱིཀྵྞ་ཨ། བཛྲ་ཧེ་ཏུ་ཨ། བཛྲ་བྷཱ་ཥ་ཨ། བཛྲ་ཀརྨ་ཨ། བཛྲ་རཀྵ་ཨ། བཛྲ་ཡཀྵ་
ཨ། བཛྲ་སནྡྷི་ཨ། བཛྲ་ལཱ་སྱ་ཨ། བཛྲ་མཱ་ལེ་ཨ། བཛྲ་གཱི་ཏི་ཨ། བཛྲ་ནཱིརྟི་ཨ། བཛྲ་དྷཱུ་པེ་ཨ། བཛྲ་པུཥྤེ་ཨ། བཛྲ་ཨཱ་ལོ་ཀེ་ཨ། བཛྲ་གནྡྷེ་ཨ། བཛྲ་ཨཾ་ཀུ་ཤ་ཨ། བཛྲ་པཱ་ཤ་ཨ། བཛྲ་སྥོ་ཊ་ཨ། བཛྲ་ཨཱ་བེ་ཤ་ཨ། དེ་ནས་དེ་རྣམས་ཀྱི་ཕྱག་རྒྱའི་ང་རྒྱལ་བསྒོམ་པར་བྱ་སྟེ། ཨཾ་ཀུ་ཤོ་ཧཾ།
ཤ་རོ་ཧཾ། ཏུཥྚི་ར་ཧཾ། རཏྣ་མ་ཧཾ། བཛྲ་སཱུརྱོ་ཧཾ། ཀེ་ཏུ་ར་ཧཾ། སྨི་ཏ་མ་ཧཾ། པདྨོ་ཧཾ། ཁཌ་གོ་ཧཾ། ཙཀྲ་མ་ཧཾ། ཛི་ཧྭཱ་ཧཾ། ཀརྨ་བཛྲོ་ཧཾ། བ་ར་མོ་ཧཾ། དཾཥྚོ་ཧཾ། མུཥྚི་ར་ཧཾ། བཛྲ་དྭ་ཡ་མ་ཧཾ། རཏྣ་མཱ་ལ་ཧཾ། བཱི་ན་ཧཾ། བཛྲ་ནྲི་ཏྱ་ཀ་ར་པལླ་བོ་ཧཾ། དྷཱུ་པོ་ཧཾ། པུཥྤོ་ཧཾ། དཱི་པོ་ཧཾ། གྷནྡྷོ་ཧཾ། ཨཾ་ཀུ་ཤོ་
ཧཾ། པཱ་ཤོ་ཧཾ། སྥོ་ཊོ་ཧཾ། གྷཎྚོ་ཧཾ། ཕྱག་རྒྱའི་གཟུངས་དེ་ཡང་རང་རང་གི་ལྷའི་སྐུར་བསྒོམ་པར་བྱ་སྟེ། བཛྲ་རཱ་ཛོ་ཧཾ། བཛྲ་རཱ་གོ་ཧཾ། བཛྲ་སཱ་དྷུ་ར་ཧཾ། བཛྲ་གརྦྷོ་ཧཾ། བཛྲ་པྲ་བྷོ་ཧཾ། བཛྲ་ཡཥྚི་ར་ཧཾ། བཛྲ་པྲཱི་ཏི་ར་ཧཾ། བཛྲ་ཎེ་ཏྲོ་ཧཾ། བཛྲ་བུདྡྷི་ར་ཧཾ། བཛྲ་མཎྜོ་ཧཾ། བཛྲ་བཱ་ཙོ་ཧཾ། བཛྲ་
བི་ཤྭ་ཧཾ། བཛྲ་བཱི་རྱོ་ཧཾ། བཛྲ་ཙཎྜོ་ཧཾ། བཛྲ་མུཥྚི་ར་ཧཾ། བཛྲ་ལཱ་སྱེ་ཧཾ། བཛྲ་མཱ་ལེ་ཧཾ། བཛྲ་གཱི་ཏི་ཧཾ། བཛྲ་ནཱི་རྟི་ཧཾ། བཛྲ་དྷཱུ་པོ་ཧཾ། བཛྲ་པུཥྤོ་ཧཾ། བཛྲ་དཱི་པོ་ཧཾ། བཛྲ་གནྡྷོ་ཧཾ། བཛྲ་ཨཾ་ཀུ་ཤོ་ཧཾ། བཛྲ་པཱ་ཤཱོ་ཧཾ། བཛྲ་སྥོ་ཊོ་ཧཾ། བཛྲ་གྷཎྜོ་ཧཾ། དེ་ནས་རྡོ་རྗེ་ལྕགས་ཀྱུ་ལ་སོགས་པའི་ལས་
ཀྱི་ཕྱག་རྒྱ་དང་། སྙིང་པོ་འདི་རྣམས་ཀྱིས་འདི་ཉིད་དུ་དགུག་པ་དང་། གཞུག་པ་དང་། བཅིང་བ་དང་དབང་དུ་བྱ་བའོ། །བཛྲ་ཨཾ་ཀུ་ཤ་ཛཿ བཛྲ་པཱ་ཤ་ཧཱུཾ། བཛྲ་སྥོ་ཊ་བཾ། བཛྲ་གཎྜེ་ཨ། དེ་ནས་རང་རང་གི་དམ་ཚིག་གི་ཕྱག་རྒྱ་བཅིངས་ལ། སྙིང་པོ་འདི་རྣམས་ཀྱིས་བལྟ་བར་བྱ་
སྟེ། བཛྲ་རཱ་ཛེ་དྲྀ་ཤྱ། བཛྲ་རཱ་ག་དྲྀ་ཤྱ། བཛྲ་སཏྭ་དྲྀ་ཤྱ། བཛྲ་རཏྣ་དྲྀ་ཤྱ། བཛྲ་ཏེ་ཛ་དྲྀ་ཤྱ། བཛྲ་ཀེ་ཏུ་དྲྀ་ཤྱ། བཛྲ་ཧཱ་ས་དྲྀ་ཤྱ། བཛྲ་དྷརྨ་དྲྀ་ཤྱ། བཛྲ་ཏཱིཀྵྞ་དྲྀ་ཤྱ། བཛྲ་ཧེ་ཏུ་དྲྀ་ཤྱ། བཛྲ་བྷཱ་ཥ་དྲྀ་ཤྱ། བཛྲ་ཀརྨ་དྲྀ་ཤྱ། བཛྲ་རཀྵ་དྲྀ་ཤྱ། བཛྲ་ཡཀྵ་དྲྀ་ཤྱ། བཛྲ་སནྡྷི་དྲྀ་ཤྱ།
