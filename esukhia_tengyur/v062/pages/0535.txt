རྡོ་རྗེ་ལྕགས་ཀྱུ་ལ་སོགས་པ་རྣམས་དགུག་པ་དང་གཞུག་པར་བྱའོ། །བཅིང་བ་དང་དབང་དུ་བྱ་བ་རྣམས་བྱས་ཏེ། མཆོད་ཡོན་ལ་སོགས་པ་དབུལ་བར་བྱའོ། །ཨོཾ་བཛྲཱཾ་ཀུ་ཤ་ཛཿ། བཛྲ་པཱ་ཤ་ཧཱུཾ། བཛྲ་སྥོ་ཊ་བཾ། ཨོཾ་བཛྲཱ་བེ་ཤ་ཧོཿ། དེ་ནས་མཆོད་ཡོན་གྱི་ཕྱག་རྒྱས་
མཆོད་ཡོན་དབུལ་བར་བྱའོ། །ཨོཾ་བཛྲ་དྷཱ་ཏུ་པྲ་བཱ་ར་སཏྐཱ་རཱརྒྷཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ལན་གསུམ་དུའོ། །ཨོཾ་བཛྲ་དྷཱ་ཏུ་པྲ་བཱ་ར་སཏྐཱ་ར་པཱ་དྱཾ་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཞབས་བསིལ་བར་བྱའོ། །ཨོཾ་བཛྲ་དྷཱ་ཏུ་པྲ་བཱ་ར་སཏྐཱ་ར་ཨཉྩ་མན་པྲ་ཏཱིཙྪ་སྭཱ་ཧཱ། ཞལ་བསིལ་ལོ། །ཨོཾ་བཛྲ་རཀྵ་ཧཱུཾ། བསང་གཏོར་བྱའོ། །དེ་ནས་དཀྱིལ་
འཁོར་ལ་མེ་ཏོག་དགོད་པར་བྱ་སྟེ། ཨོཾ་བཛྲ་དྷཱ་ཏུ་ཧཱུཾ་དབུས་སོ། །ཨོཾ་དྷཱ་ཏུ་ཧཱུཾ་ཤར་དུའོ། །ཨོཾ་བཛྲ་རཏྣ་ཏྲཱཾ་ལྷོ་རུའོ། །ཨོཾ་བཛྲ་དྷརྨ་བཛྲ་ཧྲཱིཿནུབ་ཏུའོ། །ཨོཾ་ཀརྨྨ་བཛྲ་ཨཿབྱང་དུའོ། །ཨོཾ་བཛྲ་སཏྭ་ཧཱུཾ་མི་བསྐྱོད་པའི་མདུན་དུའོ། །ཨོཾ་བཛྲ་རཱ་ཛ་ཛཿམི་བསྐྱོད་པའི་གཡས་སོ། །ཨོཾ་བཛྲ་རཱ་ག་ཧོ་མི་
བསྐྱོད་པའི་གཡོན་ནོ། །ཨོཾ་བཛྲ་སཱ་དྷུ་སཿམི་བསྐྱོད་པའི་རྒྱབ། ཨོཾ་བཛྲ་རཏྣ་ཏྲཾ། ཨོཾ་བཛྲ་ཏེ་ཛ་ཨཾ། ཨོཾ་བཛྲ་ཀེ་ཏུ་ཏྲཾ། ཨོཾ་བཛྲ་ཧཱ་ས་ཧཱུཾ། མི་བསྐྱོད་པ་བཞིན་དུ་རིན་ཆེན་འབྱུང་ལྡན་གྱི་ཡང་ངོ་། །ཨོཾ་བཛྲ་དྷརྨ་ཧྲཱིཿཨོཾ་བཛྲ་ཏཱིཀྵ་དྷཱཾ། ཨོཾ་བཛྲ་ཧེ་ཏུ་མཱཾ། ཨོཾ་བཛྲ་བྷཱ་ཥ་རཾ། སྔོན་མ་བཞིན་དུ་འོད་དཔག་མེད་ཀྱི་
ཡང་ངོ་། །ཨོཾ་བཛྲ་ཀརྨྨ་ཀཾ། ཨོཾ་བཛྲ་རཀྵ་ཧཱུཾ། ཨོཾ་བཛྲ་ཡཀྵ་ཧཱུཾ། ཨོཾ་བཛྲ་སནྡྷི་བཾ། སྔོན་མ་བཞིན་དུ་དོན་ཡོད་གྲུབ་པའིའོ་ཞེས་པ་འདིའི་སྙིང་པོ་རེ་མིག་གོ། །ཨོཾ་བཛྲ་ལཱ་སྱེ་ཧཱུཾ། ཨོཾ་བཛྲ་མཱ་ལེ་ཏྲཱཾ། ཨོཾ་བཛྲ་གཱི་ཏི་ཧྲཱིཿཨོཾ་བཛྲ་ཎྲྀ་ཏྱེ་ཨཱཿ། ཕྱིའི་མེ་ཏོག་ལ་སོགས་པའི་དཀྱིལ་འཁོར་གྱི་མཚམས་
བཞི་ལ་ཕྱིའི་དཀྱིལ་འཁོར་གྱི་ཟུར་བཞི་རྣམས་སུ་བདུག་སྤོས་མ་ལ་སོགས་པ་རྣམས་སོ། །ཨོཾ་བཛྲ་དྷཱུ་པེ་ཧཱུཾ། ཨོཾ་བཛྲ་པུཥྤེ་ཏྲཱཾ། ཨོཾ་བཛྲ་ཨཱ་ལོ་ཀེ་ཧྲཱིཿཨོཾ་བཛྲ་གནྡྷེ་ཨཿདེ་ནས་ཤར་ལ་སོགས་པའི་སྒོའི་ལོགས་དག་ཏུ། ཨོཾ་མཻ་ཏྲི་ཡ་ཧ་ར་ཎཱ་ཡ་སྭཱ་ཧཱ། ཨོཾ་ཨ་མོ་གྷཱ་དརྴི་ནི་ཧཱུཾ། ཨོཾ་སརྦ་པཱ་པཾ་ཛ་ཧ་
སརྦ་པཱ་པཾ་བི་ཤོ་དྷ་ནི་ཧཱུཾ། ཨོཾ་བཛྲ་ཤོ་ཀ་ཏ་མོ་ན་རྒྷ་ཏ་ན་མ་ཏི་ཧཱུཾ། ཤར་དུ། ཨོཾ་བཛྲ་གནྡྷ་ཧ་སྟི་ནེ་ཧཱུཾ། ཨོཾ་སཱུ་རཾ་ག་མེ་ཧཱུཾ། ཨོཾ་ག་ག་ནེ་ག་ན་ལོ་ཙ་ནེ་ཧཱུཾ། ཨོཾ་ཛྙཱ་ན་ཀེ་ཏུ་ཛྙཱ་ན་བ་ཏི་ཧཱུཾ། ལྷོ་རུའོ། །ཨོཾ་ཨ་མྲི་ཏ་བྷཻ་ཨ་མྲྀ་ཏ་བ་ཏི་ཧཱུཾ། ཨོཾ་ཙནྡྲེ་ཙནྡྲ་སྠེ་ཙནྡྲ་བྱ་བ་ལོ་ཀ་ནི་སྭཱ་ཧཱ། ཨོཾ་བྷ་དྲ་བ་ཏི་བྷ་དྲ་པཱ་ལེ་སྭཱ་ཧཱ། ཨོཾ་
