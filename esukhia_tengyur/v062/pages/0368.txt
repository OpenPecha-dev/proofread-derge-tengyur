མངོན་པར་བྱང་ཆུབ་ལྔས་ལྷར་བསྐྱེད་ལ། བྱིན་གྱིས་བརླབ་པ་དང་། ཏིང་ངེ་འཛིན་གྱི་བསྒོམ་པ་དང་། སྙིང་གར་དབང་བསྐུར་བ་དང་། ཕྲིན་ལས་ཀྱི་རྡོ་རྗེའི་བརྟུལ་ཞུགས་བླང་བ་དང་། རྣལ་འབྱོར་ཆེན་པོ་བསྐྱེད་པ་དང་། རྗེས་སུ་རྣལ་འབྱོར་དེ་བཞིན་གཤེགས་པའི་ཁམས་གཞུག་པ་
དང་། བྱིན་གྱིས་བརླབ་པ་དང་། དབང་བསྐུར་བ་དང་། མཆོད་པ་དང་བསྟོད་པ་བྱས་ལ། བདག་ཉིད་ཐབས་ཀྱི་འཁོར་ལོ་རྫོགས་པར་བྱས་ལ། ཡེ་ཤེས་ཀྱིས་སྦྱོང་བ་སྟོང་པ་ཉིད་བསྒོམས་པ་དང་། རྡོ་རྗེའི་ས་སྦྱང་བ་འབྱུང་བ་རིམ་བརྩེགས་གཞལ་ཡས་ཁང་གདན་དང་བཅས་པ་བསྐྱེད་
པ་དང་། འཆི་བའི་སྲིད་པ་རི་རབ་ཀྱི་རྩེ་མོར་ལུས་གཞག་པ་དང་། སྲིད་པ་བར་མ་དོ་བསམ་གཏན་གཟུགས་མེད་པ་བསྒོམ་པ་དང་། སྐྱེ་བའི་རྣལ་འབྱོར་བྱང་ཆུབ་སེམས་དཔའ་དོན་གྲུབ་ཏུ་བསྐྱེད་པ་དང་། མི་གཡོ་བའི་ཏིང་ངེ་འཛིན་ལས་བསླངས་ནས་དང་པོའི་སྦྱོར་བ་དང་།
དཀྱིལ་འཁོར་རྒྱལ་པོ་མཆོག་དང་། ལས་རྒྱལ་མཆོག་བསྒོམས་ལ། ཡེ་ཤེས་སེམས་དཔའ་ཞུགས་པའི་ལས་བརྒྱད་ཀྱི་ཡེ་ཤེས་སེམས་དཔའ་བཅུག་ལ། རྒྱས་བཏབ་ནས་དབང་བསྐུར་བ་དང་། མཆོད་པ་ཐམས་ཅད་ཀྱིས་མཆོད་ལ། ཇི་ལྟ་བ་བཞིན་དུ་བསྒོམ་པར་བྱའོ། །དེ་ལྟར་
མཚན་མ་མ་ཐོབ་ཀྱི་བར་དུ་བསྒོམ་པའམ། གཙོ་བོ་ལ་འབུམ་དང་། འཁོར་རྣམས་ལ་ཁྲི་ཁྲི་བཟླས་པ་དབུལ་བ་དང་། རྒྱུད་ཕྱི་མའི་ཕྱི་མ་བཞིན་དུ། ཟླ་བ་བཞི་འམ་ཟླ་བ་དྲུག་ཏུ་བསྙེན་པ་བྱའོ། །དེ་ལྟར་བསྙེན་པ་བྱས་ནས་ས་སྦྱང་བའམ། མ་སྦྱངས་པའི་གནས་སའི་ཆོ་ག་ཇི་ལྟ་བ་
བཞིན་དུ་བྱས་ཏེ། ལྷག་པར་གནས་པར་བྱའོ། །དེ་ལ་མཆོད་པ་བཤམ་པ་དང་། རྫས་བསགས་ལ་རྒྱན་ཐམས་ཅད་ཀྱིས་བརྒྱན་ལ་གདོང་དྲི་ཞིམ་པར་བྱས་ལ། མེ་ཏོག་གི་ཕྲེང་བ་བཏགས་ཏེ། རྒྱུ་སྐར་རྒྱལ་གྱི་ཞག་དང་གཟའ་ཕུར་བུ་ལ་སྲུང་བའི་འཁོར་ལོ་རྟེན་དང་བཅས་པ་དང་།
ཚོགས་སོགས་པའི་དམིགས་པའི་ཡུལ་དང་བཅས་པ་དང་། ཚོགས་ཀྱི་ལམ་དངོས་སེམས་བསྐྱེད་པ་དང་བཅས་པ་དང་། རྡོ་རྗེ་དབབ་པ་ཆོ་ག་བཞིན་དུ་བྱ་བ་དང་། ཕྱག་རྒྱ་ཆེན་པོ་ཆོ་ག་བཞིན་དུ་བཅིང་བ་དང་། མདུན་དུ་སེམས་དཔའ་ཆོ་ག་བཞིན་དུ་བསྐྱེད་དེ་བདག་ཉིད་རྫོགས་
