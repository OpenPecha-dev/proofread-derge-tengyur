སྦྱིན་པར་བྱའོ། །སློབ་དཔོན་དགོངས་སུ་གསོལ། ཇི་ལྟར་འཕགས་པ་དགྲ་བཅོམ་པ་དེ་རྣམས་ཀྱིས་ཇི་སྲིད་འཚོའི་བར་དུ་སྲོག་གཅོད་པ་སྤངས་ཤིང་སྲོག་གཅོད་པ་ལས་སླར་ལོག་པ་དེ་བཞིན་དུ། བདག་མིང་འདི་ཞེས་བགྱི་བ་ཇི་སྲིད་འཚོའི་བར་དུ་སྲོག་གཅོད་པ་སྤོང་སྟེ་སྲོག་
གཅོད་པ་ལས་སླར་ལྡོག་པར་བགྱིད་དོ། །དགེ་ཚུལ་གྱི་བསླབ་པ་ཡང་དག་པར་བླང་བར་བགྱིའོ། །དེ་བཞིན་དུ་དེ་ལྟར་བདག་ཡན་ལག་དང་པོ་འདིས་འཕགས་པ་དགྲ་བཅོམ་པ་དེ་རྣམས་ཀྱི་བསླབ་པ་ལ་རྗེས་སུ་སློབ་བོ། །རྗེས་སུ་བགྱིད་དོ། །རྗེས་སུ་བསྒྲུབ་བོ། །ཇི་ལྟར་འཕགས་
པ་དགྲ་བཅོམ་པ་དེ་རྣམས་ཀྱིས་ཇི་སྲིད་འཚོའི་བར་དུ་མ་བྱིན་པར་ལེན་པ་དང་མི་ཚངས་པར་སྤྱོད་པ་དང་། བརྫུན་དུ་སྨྲ་བ་དང་འབྲུའི་ཆང་དང་བཅོས་པའི་ཆང་མྱོས་པར་གྱུར་པ་བག་མེད་པའི་གནས་དང་། གླུ་གར་དང་རོལ་མོའི་སྒྲ་དང་། ཕྲེང་བ་དང་། དྲི་དང་བྱུག་པ་དང་ཁ་དོག་འཆང་བ་
དང་། ཁྲི་སྟན་མཐོན་པོ་དང་། ཁྲི་སྟན་ཆེན་པོ་དང་། དུས་མ་ཡིན་པའི་ཁ་ཟས་དང་། གསེར་དངུལ་ལེན་པ་རྣམས་སྤངས་ཏེ། དེ་བཞིན་དུ་དེ་ལྟར་བདག་མིང་འདི་ཞེས་བགྱི་བས་ཇི་སྲིད་འཚོའི་བར་དུ་མ་བྱིན་པར་ལེན་པ་ལས་ལྡོག་ཅིང་དགེ་ཚུལ་གྱི་བསླབ་པ་བདག་གིས་ཡང་དག་
པར་བླང་བར་བགྱིའོ། །བདག་མིང་འདི་ཞེས་བགྱི་བས་ཇི་སྲིད་འཚོའི་བར་དུ་མི་ཚངས་པར་སྤྱོད་པ་ལས་སླར་ལྡོག་ཅིང་བསླབ་པའི་གནས་བདག་གིས་ཡང་དག་པར་བླང་བར་བགྱིའོ། །བདག་མིང་འདི་ཞེས་བྱ་བས་ཇི་སྲིད་འཚོའི་བར་དུ། བརྫུན་དུ་སྨྲ་བ་ལས་སླར་ལྡོག་ཅིང་བསླབ་པའི་
གནས་བདག་གིས་ཡང་དག་པར་བླང་བར་བགྱིའོ། །བདག་མིང་འདི་ཞེས་བགྱི་བས་ཇི་སྲིད་འཚོའི་བར་དུ་འབྲུའི་ཆང་དང་བཅོས་པའི་ཆང་མྱོས་པར་འགྱུར་བ་བག་མེད་པའི་གནས་ལས་སླར་ལྡོག་ཅིང་བསླབ་པའི་གནས་བདག་གིས་ཡང་དག་པར་བླང་བར་བགྱིའོ། །བདག་མིང་འདི་
ཞེས་བགྱི་བས་ཇི་སྲིད་འཚོའི་བར་དུ་གླུ་གར་དང་རོལ་མོའི་སྒྲ་ལས་སླར་ལྡོག་ཅིང་བསླབ་པའི་གནས་བདག་གིས་ཡང་དག་པར་བླང་བར་བགྱིའོ། །བདག་མིང་འདི་ཞེས་བགྱི་བས་ཇི་སྲིད་འཚོའི་བར་དུ་ཕྲེང་བ་དང་དྲི་དང་བྱུག་པ་ལས་སླར་ལྡོག་ཅིང་བསླབ་པའི་གནས་བདག་གིས་ཡང་
