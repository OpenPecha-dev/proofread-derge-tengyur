འབྲས་བུ་མཆིས་པར་མཛོད། །དེ་ལྟར་གདོན་ལ་མཆོད་པ་བྱས་ནས་ནི། །དེ་ནས་ཕྱོགས་སྐྱོང་རྣམས་ལ་གཏོར་མ་སྦྱིན། །ཤར་ཕྱོགས་རྡོ་རྗེ་ཅན་དེ་ལྷ་ཚོགས་བཅས། །གཏོར་མ་ཕུལ་བ་འདི་དག་བཞེས་སུ་གསོལ། །མེ་དང་གཤིན་རྗེ་བདེན་བྲལ་སྲིན་པོའི་
བདག །ཆུ་བདག་རླུང་བདག་ནོར་གྱི་བདག་པོ་སྟེ། །དབང་ལྡན་ལྷ་དེ་འབྱུང་པོའི་བདག་པོ་དང་། །སྟེང་གི་ཉི་ཟླ་ཚངས་པ་ཆེན་པོ་དང་། །ལྷ་རྣམས་དང་ནི་ས་ཡི་ཀླུ་རྣམས་དང་། །རི་ཡི་ལྷ་དང་གནོད་སྦྱིན་ཚོགས་རྣམས་ལ། །ས་ས་སོ་སོར་གཏོར་མ་ཕུལ་བ་རྣམས། །རབ་ཏུ་དགྱེས་
པ་ཡིས་ནི་བཞེས་ནས་ཀྱང་། །རང་རང་ཕྱོགས་སུ་ལེགས་པར་བཞུགས་ནས་སུ། །རང་གི་བུ་སྨད་མཛའ་བཤེས་གཉེན་འདུན་དང་། །ནང་མི་དམག་དང་དཔུང་བཅས་ཐབས་ཅིག་ཏུ། །མེ་ཏོག་དྲི་དང་བྱུག་པ་གཏོར་མར་བཅས། །འདི་རྣམས་བཏུང་ཞིང་ལེགས་པར་གསོལ་ནས་
ཀྱང་། །བདག་གི་ལས་འདི་འབྲས་བུ་མཆིས་པར་མཛོད། །དེ་ནས་ཉེ་རེག་བྱས་ལ། དཔལ་རྡོ་རྗེ་སེམས་དཔས་ཞི་བའི་སྦྱིན་སྲེག་གི་ཆོ་གས། སྦྱིན་སྲེག་བརྒྱ་རྩ་བརྒྱད་དབུལ་ལོ། །དཔལ་རྣམ་པར་སྣང་མཛད་ལ་སོགས་པའི་སྙིང་པོས་ནི། སྲེག་བླུགས་བདུན་བདུན་དབུལ་ལོ། །དེ་ནས་ཀྱང་དེ་
བཞིན་གཤེགས་པ་ཐམས་ཅད་ལ་ཕྱག་བྱས་ལ། བདག་མིང་འདི་ཞེས་བགྱི་བ་ཡིས། །རྡོ་རྗེ་སློབ་དཔོན་དཀའ་ཐུབ་ཆེས། །སེམས་ཅན་ཀུན་ལ་ཕན་པའི་ཕྱིར། །སློབ་མ་རྣམས་ནི་གཞུག་པར་བགྱིད། །བཅོམ་ལྡན་འདས་དཀྱིལ་འཁོར་ཆེན་པོ་འདིར་ཞུགས་པ་ལ། སྣོད་དུ་
གྱུར་པ་དང་། མ་གྱུར་པ་བརྟག་མི་འཚལ་ལོ། །དེ་ཅིའི་ཕྱིར་ཞེ་ན། སེམས་ཅན་ཁ་ཅིག་ནི། སྡིག་པ་ཆེན་པོ་བགྱིས་པ་ཡང་མཆིས་ཏེ། དེ་དག་གིས་རྡོ་རྗེ་དབྱིངས་ཀྱི་དཀྱིལ་འཁོར་ཆེན་པོ་འདིར་ཞུགས་ཏེ་མཐོང་ན་ངན་སོང་ཐམས་ཅད་དང་བྲལ་བར་འགྱུར་རོ། །བཅོམ་
ལྡན་འདས་ནོར་དང་བཟའ་བ་དང་། བཏུང་བ་དང་། འདོད་པའི་ཡོན་ཏན་ཐམས་ཅད་ལ་སོགས་པ་དང་དམ་ཚིག་ལ་མི་དགའ་བ་དང་། སྔར་བཤད་པ་ལ་སོགས་པ་མི་ནུས་པའི་སེམས་ཅན་ཡང་མཆིས་ཏེ། དེ་དག་ཀྱང་འདིར་ཇི་ལྟར་འཚལ་བ་བཞིན་དུ་བགྱིད་ཅིང་ཞུགས་
