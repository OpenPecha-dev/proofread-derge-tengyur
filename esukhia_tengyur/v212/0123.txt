ཙོ་ཀྲུ་ཤ་ཏི། མུ་ད་དགའ་བ་ལའོ། །མོ་མུ་དཱི་ཏི། མོ་མོཏྟོཿ མོ་མུཏྟཿ མོ་མུ་ད་ཏི། སཔྟ་མཱི། མོ་མུ་དྱཱ་ཏི། པཉྩ་མཱི། མོ་མུ་དི་ཏུ། མོ་མོཏྟུ། མོ་མུཏྟཱ་ཏ། མོ་མུདྡཱ་མ། མོ་མུད་ཏུ། ཧུ་དྷུཊ་དག་ལས་ཧིའི་དྷིའོ་ཞེས་པས་དྷིའི་དངོས་པོའོ། །མོ་
མུདྟི། མོ་མུཏྟཱ་ཏ། མོ་མུཏྟཱམ། མོ་མུཏྟ། མོ་མུ་དི་ནི། མོ་མུ་དཱ་པ། མོ་མུ་དཱ་མི། ཧྱ་སྟ་ནཱི། ཨ་མོ་མུ་དཱི་ཏ། ཨ་མོ་མོ་ཏ། ཨ་མོ་མུཏྟཱམ། ཨ་མོ་མུ་དུཿ ཨ་དྱ་ཏ་ནཱི། ཨིཊ་ནི་ལས་ཀྱང་ཨཱིཊ་ལའོ་ཞེས་པས་སི་ཙའི་དབྱིའོ། །ཨ་མོ་མོ་དཱི་ཏ། ཨ་མོ་མོ་དཱི་ཏ། ཨ་མོ་མོ་
དཥྚཱི་མ། ཨ་མོ་མོ་དི་ཥུཿ ཨ་མོ་མོ་དཱིཿ ཁཱ་དྲྀ། ཨ་ཙ་ཁཱ་དཱིཥྚཱམ། ཨ་ཙཱ་ཥཱ་དཱི་ཥུཿ ཨཱ་ཙཱ་ཥཱ་དཱིཿ གྲྀ་དྷུ། མངོན་པར་འདོད་པ་ལའོ། །རྀ་ལྡན་པའི་རཱིའོ་ཞེས་པས་རཱི་ཨཱ་ག་མའོ། །ཛ་རཱི་གྲྀ་དྷཱི་ཏི། རི་དང་ར་དག་ཀྱང་ཙེ་ཀྲྀ་ཡི་ཏ་ཞིག་པ་ལའོ་ཞེས་པས་རི་དང་ར་དག་ཏུ་
ཡང་འགྱུར་རོ། །ཛཱ་རི་གྲྀ་དྷཱི་ཏི། ཛ་རྒྲྀ་དྷཱི་ཏི། ཛ་རི་གརྡྷི། ཛ་རི་གརྡྡྷི། ཛརྒྒིརྡྡྷི། ཧྱསྟ་ནཱི། ཨ་ཛ་རཱི་གྲྀ་དྷཱི་ད། ཨ་ཛ་རཱི་གྲྀ་དྷཱི་ཏ། ཨ་ཛ་རཱྀ་གྲྀ་དྷི་ཏ། ཨ་ཛརྒྲཾ་དྷཱི་ཏ། ཨ་ཛ་རཱི་གྷརྟྟ། ཨ་ཛ་རཱི་གྷརྟྟ། ཨ་ཛརྒྒརྟྟ། ཨ་ཛ་རཱྀ་གྲྀདྡྷཱ་མ། ཨ་ཛ་རི་གྲྀ་དྡྷཱམ། ཨ་ཛ་རྒྲྀདྡྷཱ་མ། ཨ་ཛ་

པ་ལའོ། །ལ་ལར་དེ་ལ་གཞན་རབ་ཏུ་སྦྱོར་བ་ལ་ཡཱ་ཏི་གང་ཡིན་པ་དེ། དེའི་རྒྱུར་འགྱུར་ཞེས་པས་བྱིངས་ལས་ཀྱང་རྒྱུ་ལའོ་ཞེས་པས་ཨིན་རྐྱེན་ནོ། །ན་ཡིག་ནི་ཨའི་ཨུ་པ་དྷཱའི་ཞེས་སོགས་ཀྱི་ཁྱད་པར་གྱི་དོན་ཏོ། །ཨརྟྟི་ཧྲཱི་བླཱྀ་རི་ཞེས་སོགས་ཀྱིས་པ་ཡིག་ཨཱ་ག་མའོ། །
