ཏ་ཏིའོ་དེ་ནས་པྲ་པཱ་ཏ། །ཆུ་སྒྲ་གཡང་ས་ཨ་བ་ཊ། །པྲ་བྱེ་ཏ་ནི་ཐེ་རི་བ་དང་། །པཱ་བྱེའ་དེ་ནས་པྲ་མྦི་ཏ། །རབ་ཏུ་སོང་དང་ཤི་བ་ལ། །པྲ་བྲྀཏྟི་ནི། །འགྲེལ་པ་དང་། །སྤྱད་མཐའ་རབ་འབག་རབ་འཇུག་མ། །པྲ་སཱུ་ཏི་ནི་རབ་སྐྱེས་དང་། །སྐྱེ་དང་འདབ་ན་བུ་མོ་ལ། །
པྲ་སཱུ་ཏི་ནི་མེ་ཏོག་དང་། །ལྕུག་མ་དང་ནི་སྐྱེ་བ་ལ། །པྲ་སཱུ་ཏཱ་ནི་སྐྱེ་དགུ་དང་། །ཐ་མ་དང་ནི་རབ་སྐྱེ་ལ། །པྲ་སྲྀ་ཧ་ནི། །སྟེང་ཐལ་སྦྱོར། །དེ་རབ་འགྲོ་དང་། །འཇུག་རྣམ་དེས། །པྲ་ཧ་ཏི་ནི། །ཀྵུཎྞ་མཐོ། །པྲརྠི་ཏྭཾ་ནི་སློང་བ་དང་། །དགྲ་བོ་འགོག་དང་རྣམ་པར་བཅོམ།
པྲོ་ཀྵི་ཏི་ནི་རྣམ་པར་བཅོམ། །བརྡྷི་ཏི་ནི་བཀངས་པ་དང་། །བཀབ་དང་། །སྤེལ་དང་། འབྲས་རྒན་པོ། །པྲྀ་ཧཱི་ཧྟཱི། །མ་ཧཱ་ཏཱི། །ཀཎྜ་ཀཱ་རཱི་བུམ་པ་ཅན། །ཤུ་དག །ལོ་རྒྱུས་ཕྲ་མོ་དང་། །སྡེབ་སྦྱོར་དབྱེ་བ་གོང་རྒྱན་ལ། །བྷ་ར་ཏ་ནི་དགའ་སྟོན་སྐྱེས། །གར་དང་གར་གྱི་བསྟན་

རྣམས་ལའོ། །བྷཱ་སནྟ་ནི་སྣང་བ་དང་། །མཁའ་འགྲོ་དང་ནི་མཛེས་པ་དང་། །འོད་དང་། གསལ་དང་། ཉི་མ་དང་། །ས་འཛིན། པྲྀ་བ་གལའོ། །མཐི་ཏཾ་ནི། བསྲུབས་དང་། དཀྲུགས། །ཆུ་མེད་དང་ནི། །བ་སྲི་ཏའོ། །མ་རུ་ཏི་ཞེས་བྱ་བ་ནི། །ལྷ་དང་། རླུང་དང་། རྒྱལ་སྲིད་
ཆེ། །མ་ན་ཏཱི་ནི་མི་སྦྱིན་གྱི། །པི་ཝང་དང་ནི་ས་གཞི་ལའོ། །མཱ་ལ་ཏཱི་ནི། ཛཱ་ཏི་དང་། །ན་ཆུང་ཡོཏྶྣཱ། ནིཀྵུ་དང་། །ཆུ་ཀླུང་གི་ནི་དབྱེ་བ་ལ། །མུ་ཥི་ཏཾ་ནི་མེའི་རྩེ་མོ། །ཀཱ་ཀ་མཱ་ཙཱི་དུམ་བུ་སྙིང་། །མཱུརྩྪི་ཏཾ་ནི་རྨོངས་པར་འགྱུར། །དེ་བསྒྲེང་པ་དང་བརྟེན་པ་ལ། །ར་ཛ་ཏཾ་ནི་
དངུལ་དོ་ཤལ། །བཙུན་པ་དང་ནི་རྒྱས་པ་ལ། །ར་མ་ཏི་ནི་མཐོ་རིས་འདྲེན། །ར་སི་ཏི་ནི། སྤྲིན་དང་སྒྲ། །གསེར་ལ་སོགས་པས་སྦྲས་པ་ལའོ། །རེ་བ་ཏཱི་ནི་མ་མོ་དང་། །སྐར་མའི་བྱེ་བྲག་འབྲས་ལྡན་མའོ། །རཻ་བ་ཏ་ནི་རིའི་བྱེ་བྲག །གསེར་གྱི་ཕྲེང་དང་མདུང་ཅན་ལའོ། །
