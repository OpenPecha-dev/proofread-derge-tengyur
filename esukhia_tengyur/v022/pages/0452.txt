ལོ་ཀཱ་ནི་བནྡ་ནི་ས་བྱཿ། པྲཏྱ་ཡ་ཀ་རི་ཎི་ཧཱུཾ་ཧཱུཾ་ཕཊ། བྷུ་ཏ་ཏྲཱ་ས་ནི་མ་ཧཱ་བཱི་རེ་པ་ར་མ་སིདྡྷེ་ཡོ་གཱི་ཤྭ་རི་ཕཊ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕཊ་སྭཱ་ཧཱ་ཞེས་སོ། །ལྷ་སོ་སོའི་སྔགས་ནི། ཨོཾ་ཀ་ར་ཀ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་ཙཎྜེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀུ་རུ་ཀུ་རུ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཙཎྜཱ་ཀྵཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བནྡྷ་བནྡྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པྲ་བྷ་བ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲཽཾ་ཧྲཽཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁརྦ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲཿཧྲཿཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ལཾ་ཀེ་ཤྭ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཕེཾ་ཕེཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དྲུ་མཙྪཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཕཊ་ཕཊ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཻ་ར་བ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ད་ཧ་ད་ཧ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་
བྷཻ་ར་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་པ་ཙ་པ་ཙ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཱ་ཡུ་བེ་གེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བྷཀྵ་བྷཀྵ་བ་ཤ་རུ་དྷི་རཱནྟྲ་མ་ལཱ་བ་ལམྦི་ནེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སུ་རཱ་བྷཀྵི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་གྲྀཧྞ་གྲྀཧྞ་སཔྟ་པཱ་ཏཱ་ལ་ག་ཏ་བྷུ་ཛཾ་ག་སརྦམྦཱ་ཏརྫ་ཡ་ཏརྫ་ཡ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤྱཱ་མ་དེ་བཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨཱ་ཀཌྜྷྱ་ཨཱ་ཀཌྜྷྱ་ཧུཾ་
ཧཱུཾ་ཕཊ། ཨོཾ་སུ་བྷ་དྲེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲིཾ་ཧྲིཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧ་ཡ་ཀརྞེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཛྙཽཾ་ཛྙཽཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁ་གཱ་ན་ནེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀྵྨཱ་ཀྵྨཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཙཀྲ་བེ་གེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧཱཾ་ཧཱཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁཎྜ་རོ་ཧེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧྲཱིཾ་ཧྲཱིཾ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤཽཎྜི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧུཾ་ཧུཾ་ཧུཾ་
ཧཱུཾ་ཕཊ། ཨོཾ་ཙཀྲ་བརྨི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀི་ལི་ཀི་ལི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སུ་བཱི་རེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སི་ལི་སི་ལི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་བ་ལེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཧི་ལི་ཧི་ལི་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཙཀྲ་བརྟི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་དྷི་ལི་དྷི་ལི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་མ་ཧཱ་བཱིརྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཀཱ་ཀཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
ཨུ་ལཱུ་ཀཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཤྭཱ་ནཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་སཱུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་དཱ་དྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཱུ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་དཾཥྚྲཱི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཡ་མ་མ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཌཱ་ཀི་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ལཱ་མཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཁཎྜ་རོ་ཧཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་རཱུ་པི་
ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཅེས་སོ། །དེ་ནས་ལྡང་བར་འདོད་ན་བདག་ཉིད་ཀྱི་དཔྲལ་བ། སྤྱི་བོ། རྣ་བ་གཡས། ལྟག་པ། རྣ་བ་གཡོན། སྨིན་མཚམས། མིག་གཉིས། ཕྲག་པ་གཉིས། མཆན་ཁུང་གཉིས། ནུ་མ་གཉིས། ལྟེ་བ། སྣ་རྩེ། ཁ། མགྲིན་པ། སྙིང་ག །
