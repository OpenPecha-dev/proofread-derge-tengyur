བྱའི་སྒྲིབ་པ་སྤངས་པའོ། །སྔགས་སུ་རླུང་གི་འཁོར་ལོ། དབྱིབས་སུ་འོག་ཏུ་བསྲུབ་བའི་ཉི་མ་དང་ཟླ་བའི་རྩའོ། །མི་མགོ་རློན་པའི་ཕྲེང་བ་ནི་ཆོས་སུ་འདུ་བྱེད་ལྔ་བཅུ་རྩ་གཅིག་གོ། །སྔགས་སུ་གཡས་གཡོན་གྱི་རླུང་འབྱུང་འཇུག་འཁོར་བ། དབྱིབས་སུ་ས་ནཱ་དང་ལ་ནཱའི་རྩའོ། །ཕྱག་རྒྱ་དྲུག་ནི་
ཆོས་སུ་ཕ་རོལ་ཏུ་ཕྱིན་པ་དྲུག །སྔགས་སུ་དབང་པོ་དྲུག་གི་རླུང་། དབྱིབས་སུ་དབང་པོ་དྲུག་གི་རྩའོ། །དེའི་སྤྱན་སྔར་བཅོམ་ལྡན་འདས་མ་ཡུམ་སྐུ་མདོག་དམར་པོ་ཞེས་བྱ་བ་ལ། དམར་བ་ནི་ཐུགས་རྗེ་ཆེན་པོས་སེམས་ཅན་གྱི་དོན་ལ་ཆགས་པའི་རྟགས་སོ། །གཡོན་པའི་ཐོད་པ་ནི་
འཆི་བདག་གི་བདུད་བཅོམ་པའོ། །གྲི་གུག་ནི་ཡེ་ཤེས་ཏེ་ཉོན་མོངས་པའི་བདུད་བཅོམ་པའོ། །རུས་པའི་རྒྱན་ནི་ཕུང་པོའི་བདུད་བཅོམ་པའོ། །འཇིགས་བྱེད་དང་དུས་མཚན་ཞབས་ཀྱིས་མནན་པ་ནི་ལྷའི་བུའི་བདུད་བཅོམ་པའོ། །ཕྱག་རྒྱ་ལྔ་དང་ལྡན་པ་ནི་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་
པས་ས་བཅུ་གཅིག་ཐོབ་པའོ། །ཞལ་ནས་ཁྲག་གི་རྒྱུན་འཛག་པ་ནི་གསུང་རབ་ཡན་ལག་བཅུ་གཉིས་འཆད་པའོ། །མཁའ་འགྲོ་དེ་བཞིན་མཛེས་མ་དང་། །ཞེས་བྱ་བ་ལ་སོགས་པ་ལ། མཁའ་འགྲོ་མ་ལ་སོགས་པ་ཡུམ་གྱི་ཕྱག་རྒྱ་བཞི་ནི་ཚད་མེད་པ་སྟེ། ཁ་ཊྭཱཾ་ག་དང་ཌ་མ་རུ་ནི་ཞི་
གནས་དང་ལྷག་མཐོང་ངོ་། །གྲི་གུག་དང་ཀ་པཱ་ལ་ནི་སྙིང་རྗེ་དང་ཤེས་རབ་བོ། །སྔགས་དང་དབྱིབས་ནི་སྔ་མ་བཞིན་ནོ། །མཚམས་རྣམས་སུ་ནི་བྱང་ཆུབ་ཀྱི་སེམས་ལ་སོགས་པའི་སྣོད་བཞི་དགོད་པ་ཞེས་བྱ་བ་ལ་བུམ་པ་བཞི་ནི་རིགས་བཞིའི་རང་བཞིན་ནོ། །ཐོད་པ་བཞི་ནི་བདུད་
རྩིའི་སྣོད་དེ་བདུད་བཞི་བཅོམ་པའོ། །དང་པོ་ཐུགས་ཀྱི་འཁོར་ལོ་ནི་རྩིབས་བརྒྱད་པ་སྟེ་མདོག་སྔོ་ནག །རྡོ་རྗེ་ཕྲེང་བས་རབ་བརྒྱན་པ། ཞེས་བྱ་བ་ལ་ནག་པོ་ནི་མི་བསྐྱོད་པའི་རིགས་སོ། །རྩིབས་བརྒྱད་ནི་རྣམ་པར་ཐར་པ་བརྒྱད་དོ། །རྡོ་རྗེ་ཕྲེང་བ་ནི་ཐུགས་ཀྱི་རྟགས་
སོ། །པུལླཱི་ར་མཱ་ལ་ཡ་སོགས་འཁོར་ལོའི་རྩིབས་རྣམས་ཀྱི་མིང་ནི་སྔོན་གྲུབ་པ་འཛམ་བུའི་གླིང་གི་གནས་སོ། །དེ་ཡང་གནས་བཞི་ནི་ལྷའོ། །ཉེ་བའི་གནས་ནི་དྲི་ཟའོ། །ཐོད་པའི་དུམ་བུ་དང་། རབ་ཏུ་གཏུམ་མོ་ལ་སོགས་པའི་དཔའ་མོ་དང་། དཔའ་བོའི་མིང་ནི་དྲག་
