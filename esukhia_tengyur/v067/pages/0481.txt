ཆུང་བརྟག་པར་བྱ། བའི་ཆུ་དང་ལྕི་བས་བྱུག །ས་གཟུང་བ་དང་སྦྱང་བའི་ཆོ་གའོ། །དེ་ནས་གོས་དང་རྒྱན་སྣ་ཚོགས་ཀྱིས་བརྒྱན། ཁྲུས་བྱས་ལུས་དྲི་ཆུས་བྱུགས། རྡོ་རྗེ་གནོད་སྦྱིན་དུ་བསྐྱེད། མིག་གཉིས་སུ་མ་དང་ཊ་ལས་ཟླ་བ་དང་ཉི་མ་དགོད། རྐང་པ་ལ་ཨ་ལས་
སྣ་ཚོགས་རྡོ་རྗེ་བསྐྱེད། ལག་པ་གཉིས་ཀྱིས་རྡོ་རྗེ་དྲིལ་བུ་བཟུང་སྟེ་ཡི་གེ་ཧཱུཾ་གི་སྒྲ་སྒྲོགས་ཤིང་། གཡས་ནས་བསྐོར་ལ་དེངས་ཞེས་བརྗོད། མཚམས་དང་ར་བ་དང་དྲ་བ་དག་ཀྱང་བྱ། བདུད་དང་སྡང་བ་བཟློག །དེ་ནས་ཕུར་བུ་བྱིན་གྱིས་བརླབས་ལ་ཕུར་བུ་གདབ་པར་བྱའོ། །འདིས་ནི།
ཇི་བཞིན་པ་ཡི་ཆོ་ག་ཡིས། །ཞེས་བྱ་བའི་དོན་ཏོ། །དེ་ནས་སྦྱོར་བ་དང་པོ་དང་། དཀྱིལ་འཁོར་རྒྱལ་མཆོག་དང་། ལས་རྒྱལ་མཆོག་བསྒོམས་ལ་དཀྱིལ་འཁོར་ནམ་མཁའ་ལ་བཏེག། །ལྷའི་གནས་སུ་ཙནྡན་ལ་སོགས་པའི་ཟླུམ་པོ་དང་། གྲུ་བཞིའི་ཐིག་ལེ་བྱ། དཀྱིལ་འཁོར་
བརྟགས་པ་བསྡུས། ལྕགས་ཀྱུ་ལ་སོགས་པས་དགུག་གཞུག་བྱ། བགེགས་བསྐྲད་པ་དང་། ར་བ་དང་། དྲ་བ་བྱ། སྒོ་བཅད་སྲུང་བ་བརྟན་པར་བྱ། མཆོད་ཡོན་དབུལ་བ་སྔོན་དུ་འགྲོ་བས་ཕྱག་རྒྱ་བཞིས་རྒྱས་གདབ། མཆོད་པ་རྣམས་ཀྱིས་མཆོད་རང་རང་གི་སྔགས་ཅི་མང་དུ་བཟླས་ཏེ།
བུམ་པ་ལྷག་པར་གནས་པ་དང་སློབ་མ་ལྷག་པར་གནས་པ་དང་། ཕྱི་རོལ་དུ་གཏོར་མ་སྦྱིན་པ་དང་། ཉེ་རེག་དང་། སྦྱིན་སྲེག་དང་། མཆོད་ཅིང་གསོལ་བ་གདབ་པ་ལ་སོགས་བྱའོ། །འདི་ཙམ་གྱིས་ནི། གནས་ནི་བྱིན་གྱིས་བརླབ་པ་བརྩམ། །ཞེས་བྱ་བའི་དོན་ཡིན་ནོ། །བྱིན་གྱིས་བརླབས་
པའི་གནས་དེར་ནི། །ཞེས་བྱ་བ་ནི་དེ་ཉིད་དུའོ། །ཕྱི་ཡི་དཀྱིལ་འཁོར་བྲི་བར་བྱ། །ཞེས་པ་ནི་དཀྱིལ་འཁོར་གྱི་སྒོ་གཉིས་པའོ། །ཕྱོགས་ཀུན་མཉམ་པ་གྲྭ་དང་བཅས་པས་གྲུ་བཞི་པ་ལ་ཕྱོགས་ཀྱི་ཆ་རེ་རེ་ཞིང་སྒོ་བཞི་ཡོད་པས་སྒོ་བཞི་པ་ཞེས་བྱའོ། །རྟ་བབས་བཞི་དང་ལྡན་པ་དང་། །ཐེམ་
སྐས་བཞི་ཡི་ཕྲེང་བ་དང་། །སེང་གེ་བ་གླང་རྣམས་ཀྱིས་བརྒྱན། །ཞེས་བྱ་བ་ནི། ཕྱོགས་རེ་རེ་ན་རྟ་བབས་རེ་དང་། ཐེམ་སྐས་ཕྲེང་བ་དང་། སེང་གེ་བ་གླང་རྣམས་ཀྱིས་མཛེས་པར་བྱས་པས་དེ་སྐད་ཅེས་བྱའོ། །དར་དང་ལྡ་ལྡི་དོ་ཤལ་ཕྲེང་། །དྲིལ་བུ་དང་ནི་རྔ་མས་བརྒྱན། །ཞེས་བྱ་བ་
