འཕགས་པ་རང་སངས་རྒྱས་སེར་སྐྱ་དང་། འོད་མཛེས་དང་། ཀུན་འགྲོགས་དང་། ལག་བརྐྱང་དང་། ཉི་མའི་ཏོག་དང་། ཤཱཀྱ་འོད་དང་། པདྨ་ཅན་དང་ལེགས་བཟངས་རྣམས་བདེ་བར་གཤེགས་པའི་ཆ་ལུགས་ཀྱིས་པདྨའི་སྟེང་དུ་བཞུགས་པ་བྲིའོ། །ནུབ་
ཕྱོགས་ཀྱི་རྩིབས་བརྒྱད་ལ་ནི་ཁྲོ་བོ་ཆེན་པོ་སྟོབས་པོ་ཆེ་དང་། མི་གཡོ་མགོན་དང་། གཤིན་རྗེ་གཤེད་དང་། རྟ་མགྲིན་དང་། བདུད་རྩི་འཁྱིལ་པ་དང་། དབྱུག་པ་སྔོན་པོ་དང་། གཞན་ལས་རྒྱལ་བ་དང་། ཁམས་གསུམ་རྣམ་པར་རྒྱལ་བ་རྣམས་པདྨའི་སྟེང་
དུ་མཆེ་བ་རྣམ་པར་གཙིགས་ཤིང་སྐུ་ལ་སྦྲུལ་གྱི་རྒྱན་ཆ་ལྔས་བརྒྱན་པ། སྟག་གི་པགས་པས་སྨད་དཀྲིས་པ། རང་རང་གི་ཕྱག་རྒྱས་མཚན་པ་རྣམས་བྲི་པར་བྱའོ། །བྱང་གི་རྩིབས་བརྒྱད་ལ་ནི་དེ་དག་གི་ཕོ་ཉའི་ཚོགས་བརྒྱད་དེ། སེང་གེའི་མགོ་དང་། སྟག་
གི་མགོ་དང་། འུག་པའི་མགོ་དང་། བྱ་རོག་གི་མགོ་དང་། ཁྱུ་མཆོག་གི་མགོ་ཅན་དང་། སྦྲུལ་མགོ་ཅན་དང་། ཤ་བའི་མགོ་ཅན་དང་། ཕག་གི་མགོ་ཅན་རྣམས་བྲི་བར་བྱའོ། །དེ་དག་གི་ཕྱི་རོལ་གྱི་འཁོར་ལོ་རྩིབས་དྲུག་ཅུ་རྩ་བཞི་ལ་ཤར་གྱི་རྩིབས་དང་
པོ་དང་། དེ་བཞིན་དུ་ལྷོ་དང་ནུབ་དང་། བྱང་གི་རྩིབས་དང་པོ་རྣམས་སུ་ནི་ཡུལ་འཁོར་སྲུང་དང་། འཕགས་སྐྱེས་པོ་དང་། མིག་མི་བཟང་དང་། རྣམ་ཐོས་སྲས་རྣམས་རང་རང་གི་ཕྱག་མཚན་དང་ལྡན་པ་བྲི་བར་བྱའོ། །དེ་ནས་དེ་དག་གི་གཡོན་ཕྱོགས་རྣམས་སུ་ནི་
ས་བདག་དང་། གཤིན་རྗེ་དང་། ཆུ་ལྷ་དང་། གནོད་སྦྱིན་དང་། མེ་ལྷ་དང་། སྲིན་པོ་དང་། རླུང་ལྷ་དང་། བདེ་བྱེད་རྣམས་རང་རང་གི་བཞོན་པ་དང་། ཁ་དོག་དང་མཚན་མས་མཚན་པ་བྲི་བར་བྱའོ། །དེ་ནས་ཡང་དེ་དག་གི་གཡོན་ཕྱོགས་རྣམས་སུ་ཉི་མ་དང་། ཟླ་བ་དང་།
གཟའ་ལྷག་དང་། མིག་དམར་དང་། ཕུར་བུ་དང་། པ་སངས་དང་། སྤེན་པ་དང་། སྒྲ་གཅན་རྣམས་རང་རང་གི་གདན་དང་མཚན་མས་མཚན་པ་བྲི་བར་བྱའོ། །དེ་ནས་ཡང་ཤར་ཕྱོགས་ལ་སོགས་པའི་རིམ་པ་ཇི་ལྟ་བ་བཞིན་དུ་སྨིན་དྲུག་དང་། སྣར་མ་དང་། མགོ་དང་།
