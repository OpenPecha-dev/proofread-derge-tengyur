རྒྱལ་པོ་དེའི་བུ་དང་། བྲམ་ཟེ་དང་། རྒྱལ་རིགས་དང་། མི་དང་། བྲན་དང་། ཕྱུགས་དང་། གཞན་ཡང་གང་ཡང་རུང་བ་དག་ལ་ནད་དང་ཡི་དགས་དང་། འདྲེས་མི་འཇིགས་པར་བགྱི། སྲིན་པོ་ལ་སོགས་པས་མི་འཇིགས་པར་བགྱི། དུས་མ་ལགས་པས་འཆི་བས་འཇིགས་
པར་མི་བགྱི་སྟེ། དུས་ཐམས་ཅད་དུ་བསྲུང་བ་དང་། བསྐྱབ་པ་དང་སྦ་བར་བགྱིའོ། །དེ་ནས་ཀླུའི་རྒྱལ་པོ་ཆུ་ལྷས་ཀྱང་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་ཅག་གིས་ཀྱང་དུས་ཐམས་ཅད་དུ་རྒྱལ་པོ་དེ་དང་དེའི་ཡུལ་འཁོར་ཐམས་ཅད་སོ་སོར་བསྐྱབ་པར་
བགྱིའོ། །ཀུན་དུ་བསྲུང་བར་བགྱི། འབྲུ་རྣམས་བསྐྱེད་པར་བགྱི། ཀླུའི་ཉེས་པ་རྣམས་མི་འབྱུང་བར་བགྱི། དུག་དང་སེར་བ་རྣམས་མི་འབབ་པར་བགྱི། དུས་མ་ཡིན་པར་འཆི་བ་རྣམས་སོ་སོར་དགག་པར་བགྱིའོ། །དེ་ནས་རླུང་གི་བདག་པོས་ཀྱང་འདི་སྐད་ཅེས་གསོལ་
ཏོ། །བཅོམ་ལྡན་འདས་བདག་གིས་ཀྱང་སེམས་ཅན་ཆེན་པོ་དེ་ལ་རླུང་གི་འཇིགས་པ་མི་འབྱུང་བར་བགྱི། འབྲུ་དང་། མེ་ཏོག་དང་། འབྲས་བུ་ཐམས་ཅད་བསྐྱེད་པར་བགྱི། འཇིགས་པ་ཐམས་ཅད་སོ་སོར་དགག་པར་བགྱིའོ། །དེ་ནས་གནོད་སྦྱིན་གྱི་རྒྱལ་པོ་ལུས་
ངན་པོས་ཀྱང་བཅོམ་ལྡན་འདས་ལ་ཕྱག་བཙལ་ནས་འདི་སྐད་ཅེས་གསོལ་ཏོ། །བཅོམ་ལྡན་འདས་བདག་གིས་ཀྱང་གནོད་སྦྱིན་གྱི་སྡེ་དཔོན་ཆེན་པོ་ཉི་ཤུ་རྩ་བརྒྱད་དང་ལྷན་ཅིག་ཏུ་མཆིས་ཏེ། སྟོབས་དང་ལྡན་པས་སེམས་ཅན་ཆེན་པོ་དེ་ལ་དུས་ཐམས་ཅད་དུ་འཇིགས་པ་
ཐམས་ཅད་སོ་སོར་དགག་པར་བགྱིའོ། །ནོར་དང་འབྲུ་ཐམས་ཅད་བསྐྱེད་ཅིང་ལེགས་པར་སྤེལ་བར་བགྱིའོ། །རང་གི་ངོ་བོ་དང་། །ཡོངས་ཀྱི་སྐྱེ་བོ་དང་། ཡུལ་དང་། གནས་དང་། བྲན་དང་། སྣག་གི་གཉེན་མཚམས་དང་། གྲོགས་དང་། བུ་དང་། བུ་མོ་དང་། ཆུང་མ་ལ་སོགས་
པ་བསྲུང་བར་བགྱིའོ། །དེ་བཞིན་དུ་གླང་དང་། བ་དང་། བོང་བུ་དང་། རྔ་མོ་དང་། ལུག་དང་། མ་ཧེ་དང་། བ་ལང་དང་། རྟ་དང་། ར་ལ་སོགས་པ་ལའང་བསྲུང་བར་བགྱིའོ། །དེ་ནས་འབྱུང་པོ་ཐམས་ཅད་ཀྱི་བདག་པོ་དབང་བྱེད་ཀྱིས་ཀྱང་བཅོམ་ལྡན་འདས་ལ་ཕྱག་འཚལ་ནས་འདི་སྐད་ཅེས་
