ཁོ་ན་ཡིན་ཏེ། ལྟ་བའི་བྱ་བ་དང་ལྡན་པའི་ཕྱིར་རོ་ཞེས་བྱ་བའི་ཚིག་གི་དོན་ཏོ་ཞེས་བྱ་བ་ནི་མིག་དེ་ལྟ་བའི་ངོ་བོ་ཉིད་ཡིན་ན། སྔ་ནས་ལྟ་བའི་བྱ་བ་དང་ལྡན་པའི་ཕྱིར་དེ་ལ་ལྟ་བའི་བྱ་བ་གཞན་དོན་མེད་པ་ཉིད་དེ་ལྟ་བའི་ངོ་བོ་ཉིད་ཡིན་པ་ལ་སྒྲའི་བསྟན་བཅོས་ལས་བྱེད་པ་པོ་ལ་བྱ་བའི་རྐྱེན་བརྗོད་ནས་བྱེད་པ་པོ་ཉིད་དུ་གྲུབ་པའི་
ཕྱིར་ལྟ་བ་ཞེས་བྱ་བ་དེ་མི་རུང་ངོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ཅི་སྟེ་དེ་ལྟ་བ་མ་ཡིན་པ་ལ་བྱ་ན་ཡང་ཞེས་བྱ་བ་ནི་ཅི་སྟེ་བརྟག་པ་ཕྱི་མ་ལྟར་འདོད་ན་ཡང་ངོ་། །དེས་ཅིར་འགྱུར་ཞེ་ན། གང་ཚེ་ཅུང་ཟད་མི་ལྟ་བ། །ལྟ་བར་བྱེད་པ་མ་ཡིན་ནོ། །དེའི་ཚེ་དེ་ལྟ་བའི་ངོ་བོ་ཉིད་མ་ཡིན་ན་བོང་བ་ལ་སོགས་པ་བཞིན་དུ་ལྟ་བའི་བྱ་བ་དང་
བྲལ་བ་ཡིན་པས་ལྟ་བ་མ་ཡིན་པ་ཡང་ལྟ་བར་བྱེད་པ་མ་ཡིན་པ་ཉིད་དོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། ལྟ་བའི་ངོ་བོ་ཉིད་མ་ཡིན་པ་ལ་སྒྲའི་བསྟན་བཅོས་ལས་བྱེད་པ་པོ་ལ་བྱ་བའི་རྐྱེན་བརྗོད་ནས་བྱེད་པ་པོ་ཉིད་དུ་གྲུབ་པའི་ཕྱིར་ལྟ་བར་བྱེད་པས་ལྟ་བ་ཞེས་བྱ་བ་དེ་མི་རུང་ངོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །དེའི་ཕྱིར་དེ་ལྟར་གཉི་ག་ཡང་
ལྟ་བའི་བྱ་བ་དང་མི་ལྡན་པའི་ཕྱིར། ལྟ་བ་ལྟ་ཉིད་མ་ཡིན་ཏེ། །ལྟ་བ་མིན་པ་མི་ལྟ་ཉིད། །ཅེས་བྱ་བ་ནི་བརྟག་པ་སྔ་མ་གཉི་གའི་མཇུག་བསྡུ་བ་དང་ཚད་མའི་འབྲས་བུ་ཡིན་ནོ། །ཅི་སྟེ་དགག་པ་གཉིས་ཡོད་ན་སྐབས་ཀྱི་དོན་གོ་བས་དེའི་ཕྱིར་མིག་ལྟ་བའི་བྱ་བ་སྐྱེས་པས་མཐོང་ངོ་ཞེ་ན་ཞེས་བྱ་བ་ནི་ཅི་སྟེ་ཕ་
རོལ་པོ་དག་ན་རེ་ཁྱེད་དབུ་མ་པས། ལྟ་བ་མིན་པ་མི་ལྟ་ཉིད། །ཅེས་བཤད་པའི་ཚིག་ལེའུར་བྱས་པ་དེ་ལ་དགག་པ་གཉིས་ཡོད་ན་སྐབས་ཀྱི་དོན་གོ་བས་འདིར་ཁོ་བོ་ཅག་འདོད་པའི་སྐབས་ཀྱི་དོན་ནི་ལྟ་བས་ལྟ་བ་ཉིད་ཡིན་ན་དེའི་ཕྱིར་ཁྱེད་ཀྱིས། ལྟ་བ་མིན་པ་མི་ལྟ་ཉིད། །ཅེས་བཤད་པ་དེས་ན་མིག་ལྟ་བའི་
བྱ་བ་སྐྱེས་པས་ལྟ་བས་ཁྱེད་རང་གིས་ཁས་བླངས་པ་ལ་གནོད་པར་འགྱུར་རོ་ཞེས་ཟེར་ནའོ། །དེ་སྐད་ཅེས་ཟེར་བ་དེའི་ལན་དུ་དེ་ཉིད་བཟང་པོ་མ་ཡིན་ཏེ་ཡོད་པ་དང་མེད་པའི་རྐྱེན་དགག་པ་བཞིན་དུ་ཡོད་པ་དང་མེད་པའི་བྱེད་རྒྱུ་ཉིད་བཀག་པའི་ཕྱིར་རོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། ཀླན་ཀ་དེ་ནི་བཟང་པོ་མ་ཡིན་
ཏེ། སྔར་རྐྱེན་བརྟག་པའི་རབ་ཏུ་བྱེད་པའི་སྐབས་སུ། མེད་དམ་ཡོད་པའི་དོན་ལ་ཡང་། །རྐྱེན་ནི་རུང་བ་མ་ཡིན་ཏེ། །མེད་ན་གང་གི་རྐྱེན་དུ་འགྱུར། །ཡོད་ན་རྐྱེན་གྱིས་ཅི་ཞིག་བྱ། །ཞེས་བཤད་པས་ཡོད་པ་དང་མེད་པའི་རྐྱེན་དགག་པ་བཞིན་དུ་འདིར་ཡང་ལྟ་བའི་བྱ་བ་ཡོད་པ་དང་མེད་
