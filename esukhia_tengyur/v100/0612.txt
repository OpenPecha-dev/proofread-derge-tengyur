བསྟན་པའི་ཕྱིར་ཁྱེད་ཕ་རོལ་པོ་དག་ལ་སྒྲུབ་པ་དེ་ལྟ་བུ་མེད་པས་ཁྱེད་ཀྱིས་ལྟ་བ་ལ་སོགས་པ་དངོས་པོ་གང་ལ་ཡང་གཞན་ཉིད་ཅེས་བྱ་བའི་ཚིག་གི་དོན་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲའི་སྤྱོད་ཡུལ་ཉིད་ཀྱི་ཆོས་མེད་དོ་ཞེས་སྟོན་པར་མི་ནུས་པའི་ཕྱིར་ཁོ་བོ་ཅག་གི་གཏན་ཚིགས་དེ་གྲུབ་པ་ལ་སྒྲུབ་པ་ཡང་མ་ཡིན་ནོ། །
དེ་ཉིད་ཀྱི་རྗེས་སུ་དཔག་པ་བསྟན་པའི་ཕྱིར་དོན་དམ་པར་གཞན་ཉིད་ལ་གཞན་ཉིད་ཀྱི་ངོ་བོ་ཉིད་མེད་དེ། སྤྱི་དང་ཁྱད་པར་དང་ལྡན་པའི་ཕྱིར་དཔེར་ན་གཟུགས་ཉིད་བཞིན་ནོ་ཞེས་བྱ་བ་སྨྲས་ཏེ། དེ་དག་ནི་ཆོས་ཅན་དང་། བསྒྲུབ་པར་བྱ་བ་ཆོས་དང་། སྒྲུབ་པའི་ཆོས་དང་། དཔེ་ཡིན་པར་གོ་རིམས་
བཞིན་དུ་སྦྱར་ཏེ། དཔེར་ན་གཟུགས་ཉིད་ནི་གཟུགས་ཞེས་བྱ་བ་ཉི་ཚེའི་སྤྱི་དང་སྤྱི་ཆེན་པོ་དང་བུམ་པའི་གཟུགས་དང་སྣམ་བུའི་གཟུགས་ཞེས་བྱ་བའི་ཁྱད་པར་དང་ལྡན་པའི་ཕྱིར། དེ་ལ་གཞན་ཉིད་ཅེས་བྱ་བའི་ཚིག་གི་དོན་གྱི་ངོ་བོ་ཉིད་མེད་པ་དེ་བཞིན་དུ། བྱེ་བྲག་པས་བརྟགས་པའི་གཞན་ཉིད་
ཅེས་བྱ་བའི་ཚིག་གི་དོན་དེ་ཡང་གཞན་མ་ཡིན་པ་ལས་ལོག་ཅིང་གཞན་ཞེས་བྱ་བའི་སྤྱི་དང་བུམ་པ་དང་ལྡན་པའི་གཞན་དང་། སྣམ་བུ་དང་ལྡན་པའི་གཞན་ཞེས་བྱ་བའི་ཁྱད་པར་དང་ལྡན་པའི་ཕྱིར་དེ་ལ་ཡང་གཞན་ཉིད་ཀྱི་ངོ་བོ་ཉིད་མེད་དེ། དེས་ནི་གཞན་ཉིད་ཅེས་བྱ་བའི་ཚིག་གི་དོན་གྱི་
ངོ་བོ་ཉིད་ཀྱི་དགག་པ་བསྟན་ཏོ། །ཡང་ན་དོན་དམ་པར་གཞན་ཉིད་ནི་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲ་སྐྱེ་བའི་རྒྱུ་ཉིད་མ་ཡིན་ཏེ། བློ་དང་སྒྲ་ཁྱད་པར་ཅན་གྱི་རྒྱུ་ཡིན་པའི་ཕྱིར་དཔེར་ན་གཟུགས་ཉིད་བཞིན་ནོ་ཞེས་བྱ་བ་དག་ཀྱང་ཆོས་ཅན་དང་། བསྒྲུབ་པར་བྱ་བ་ཆོས་དང་། སྒྲུབ་པའི་ཆོས་དང་། དཔེ་ཡིན་
པར་གོ་རིམས་བཞིན་དུ་སྦྱར་ཏེ། དཔེར་ན་གཟུགས་ཉིད་ནི་གཟུགས་མཐོང་ན་གཟུགས་ཀྱི་བློ་དང་སྒྲ་ཁྱད་པར་ཅན་སྐྱེ་བའི་རྒྱུ་ཡིན་པའི་ཕྱིར་བྱེ་བྲག་པ་དག་གིས་ཡོངས་སུ་བརྟགས་པའི་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲ་སྐྱེ་བའི་རྒྱུ་ཉིད་མ་ཡིན་པ་དེ་བཞིན་དུ་བྱེ་བྲག་པ་དག་གིས་ཡོངས་སུ་བརྟགས་པ་གཞན་ཉིད་
ཅེས་བྱ་བའི་ཚིག་གི་དོན་ཀྱང་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲ་ཁྱད་པར་ཅན་སྐྱེ་བའི་རྒྱུ་ཡིན་པའི་ཕྱིར་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲ་སྐྱེ་བའི་རྒྱུ་ཉིད་མ་ཡིན་ཏེ། དེས་ནི་བྱེ་བྲག་པ་དག་གིས་ཡོངས་སུ་བརྟགས་པ་གཞན་ཉིད་ཅེས་བྱ་བའི་ཚིག་གི་དོན་དེ་གཞན་ཉིད་ཀྱི་བློ་དང་སྒྲ་སྐྱེ་བའི་རྒྱུ་ཉིད་མ་ཡིན་པར་བསྟན་ཏོ། །ཡང་ན་
