པ་སྟེ། །དྲི་མ་གསུམ་གྱིས་ལྟུང་བར་འགྱུར། །འཇིགས་པས་གསོད་པར་རབ་ཏུ་བཤད། །ས་གདུགས་ལས་ཀུན་བྱེད་པ་ཡིན། །ཨཱ་ལི་ཞེས་བྱའི་ཕྱག་རྒྱ་དང་། །ས་བོན་བརྒྱད་ཀྱིས་བསྐུལ་བྱ་སྟེ། །རྡོ་རྗེ་མཆོག་ཏུ་འབར་བའི་འོད། །མེ་ཡི་ས་བོན་སྟེང་སྒོར་དགོད། །ས་ཡི་
མིག་ལ་བསམ་པར་བྱའོ། །ནམ་མཁའ་རྣ་བ་སྣ་དག་ལ། །བདུད་རྩི་ཆུ་ཡི་ས་བོན་ནི། །དཀར་པོ་ལྕེ་ལ་ཀུན་དུ་བསམ། །སྡེ་ཚན་བརྒྱད་པའི་གསུམ་པ་ཡིས། །རྩ་བ་ཡིས་ནི་སྟེང་དུ་བཙིར། །ར་ནི་གསུམ་པ་ཉིད་ཡིན་ཏེ། །འོག་ཏུ་རླུང་སྦྱར་བཞི་པའོ། །སྒྲ་དང་གཞོན་ནུ་ཐིག་ལེ་ཅན། །ཡི་གེ་བཞི་པའི་
མིང་ཅན་ཏེ། །ཆུ་སྟེ་གཏི་མུག་མགྲིན་པར་རོ། །ཉོན་མོངས་པ་ནི་ནམ་འཚོད་དགོད། །སྙིང་གར་རླུང་གིས་བསྐྱོད་པར་བྱ། །ལྟ་བར་མེ་སྟེ་དེ་བཞིན་ཕྱུགས། །ཡི་གེའི་དགོད་པས་བཀོད་ནས་ནི། །ས་བོན་ལས་ནི་ལྷ་ཡི་ཚུལ། །ཟླ་བའི་དཀྱིལ་འཁོར་དབུས་གནས་པའི། །རྡོ་རྗེ་སེམས་དཔའ་བསམ་པར་
བྱ། །པདྨའི་གདན་ལ་བཞུགས་པ་སྟེ། །གངས་དང་ཀུན་ད་ལྟ་བུའི་འོད། །ཕྱག་གཉིས་སེམས་དཔའི་སྐྱིལ་མོ་ཀྲུང་། །སངས་རྒྱས་ལྔ་ཡི་ཅོད་པན་ཅན། །རྡོ་རྗེ་སྙིང་གར་བཟུང་ནས་ནི། །གཡོན་གྱི་དྲིལ་བུ་དཀུ་རུ་བརྟེན། །སྐུ་ལ་རྒྱན་རྣམས་ཐམས་ཅད་དང་། །ཤིན་ཏུ་མཛེས་པའི་ན་བཟའ་ཀླུབས། །འོད་
ཟེར་སྣ་ཚོགས་འབར་བ་ཅན། །རྡོ་རྗེ་སེམས་དཔའ་བདག་ཅེས་བརྗོད། །རང་གི་ཡེ་ཤེས་ས་བོན་ནི། །སྒྲ་དང་ཐིག་ལེར་ཡང་དག་བཅས། །ཕྱག་འཚལ་བསྲེག་པའི་མཐའ་ཅན་ཏེ། །འབུམ་ཕྲག་གསུམ་བཟླས་སྔར་བསྙེན་བྱ། །རྣལ་འབྱོར་མ་ནི་བརྒྱད་པོ་དག །ཁྲི་ཕྲག་བཞིར་ནི་བསྙེན་པ་བྱ། །
སྦོམ་མོ་ཁྲོ་གཉེར་ལ་སོགས་པ། །ཉི་ཤུ་སྟོང་ཕྲག་ལྔ་རུ་བཟླས། །ཕྱི་རོལ་གནས་ཀྱི་འཇིག་རྟེན་པ། །རྒྱལ་དབང་དངོས་གྲུབ་ཤིན་ཏུ་དྲག །སྟོང་ཕྲག་གསུམ་དུ་བཟླས་ནས་ནི། །ཀུན་ནས་ཀུན་དུ་འགྲུབ་པར་འགྱུར། །སྔོན་དུ་བསྙེན་པས་འགྲུབ་འགྱུར་ཏེ། །མ་བསྙེན་པ་ནི་དཀྱིལ་འཁོར་མིན། །
སྔགས་དང་རྣལ་འབྱོར་དེ་དག་གིས། །རིམ་གྱིས་བྱིན་གྱིས་རློབ་པར་འགྱུར། །སྔོན་དུ་བསྙེན་པས་ལྷ་མཉེས་པར་བྱ་བའི་ཆོ་ག་སྟེ་གཉིས་པའོ།། །།དེ་ནས་བརྟན་དང་གཡོ་ལ་སོགས། །སྔགས་བསྙེན་པ་ལ་འབད་པ་ཡིས། །དེ་མ་ཐག་ཏུ་འབེབས་གྱུར་པ། །སྔགས་ནི་ཡང་དག་
