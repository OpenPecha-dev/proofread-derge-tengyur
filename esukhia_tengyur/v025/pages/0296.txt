ཐམས་ཅད་བདག་སྐྱེས་ངོ་བོ་གཅིག་ལ་བསྟོད་ཕྱག་འཚལ། །འཁོར་བའི་རྒྱ་མཚོ་རྨོངས་པ་གཅོད་ལ་བསྟོད་ཕྱག་འཚལ། །དེ་ཉིད་གཅིག་ལ་རྟག་ཏུ་བདག་འདུད་བསྟོད་ཕྱག་འཚལ། །རང་དགར་སེམས་ཅན་དགྲོལ་བ་ཡི། །རྒྱུ་མེད་དངོས་ལ་འདུད་པར་བགྱི། །ཏེ་ན་ཧཱུཾ། ཏེ་ན་ཏེ་ཏེ་
ཧཱུཾ། སྟོན་ཀའི་ཟླ་ལྟར་རབ་མཛེས་ཤིང་། །དུང་ལྟར་དཀར་བའི་སྐུ་འཛིན་པ། །ཏེ་ན་ཧཱུཾ། ཏེ་ན་ཏེ་ཏེ་ཧཱུཾ། རྡོ་རྗེ་དྲིལ་བུའི་ཕྱག་རྒྱ་ཅན། །གཉིས་འཁྱུད་སྦྱོར་བ་འཛིན་པའོ། །ཏེ་ན་ཧཱུཾ། ཏེ་ན་ཏེ་ཏེ་ཧཱུཾ། སྒྱུ་མའི་ལུས་སུ་ཐིམ་པ་ཡིས། །འགྲོ་བ་འཛིན་པ་རྒྱུ་མེད་དངོས། ཏེ་ན་ཧཱུཾ། ཏེ་ན་
ཏེ་ཏེ་ཧཱུཾ། སྲིད་གསུམ་འབར་བའི་སྐུ་ནི་རྗེས་ཆགས་འདུལ་བར་ལྡན། །འོད་ཟེར་རྣོན་པོ་སྟོང་ལྡན་ཉི་མ་སྟོང་འདྲར་ལྡན། །རྣམ་མང་སྐུ་དང་ཆ་བྱད་རྗེས་ཆགས་སེམས་དང་ལྡན། །རྡོ་རྗེ་སེམས་དཔའ་དབང་ཕྱུག་མཆོག་རོལ་གར་མཛད་པ། །ཞེས་བྱ་བས་ཁྱད་པར་དུ་བསྟོད་ཅིང་གླུ་
དང་བསྟོད་པས་མཆོད་པས་མཉེས་པར་བྱ། །བར་བར་དུ་ཧཱུཾ་གི་སྒྲ་དང་དྲིལ་བུ་བསྒྲགས་ལ་སྨོན་ལམ་རྣམ་པ་གསུམ་གདབ་པར་བྱའོ། །རྡོ་རྗེ་ཐལ་མོ་དམ་བཅིངས་ནས། །སྙིང་གའི་ཕྱོགས་སུ་བཙིར་བར་བྱ། །ཨོཾ་སྭ་བྷཱ་བ་ཤུདྡྷཿསརྦ་དྷརྨཱ་སྭ་བྷཱ་བ་ཤུདྡྷོ྅ཧཾ། ཨོཾ་བཛྲ་ཤུདྡྷཿསརྦ་
དྷརྨཱ་བཛྲ་ཤུདྡྷོ྅ཧཾ། ཨོཾ་ཡོ་ག་ཤུདྡྷཿསརྦ་དྷརྨཱ་ཡོ་ག་ཤུདྡྷོ྅ཧཾ། ཞེས་བརྗོད་ལ་ལྷའི་རྣལ་འབྱོར་ལ་གནས་པར་བྱའོ། །ཏིང་ངེ་འཛིན་གྱིས་སྐྱོན་སྔགས་བཟླ་སྟེ། ཕྲེང་བ་དེ་ཉིད་དང་ལྡན་པས་སྔགས་འདིས་སྦྱང་ངོ་། །ཨོཾ་པ་ཏེ་པ་ཏེ་མ་ཧཱ་ཛྙཱ་ན་སརྦ་བུདྡྷ་མ་ཧཱ་བ་བེ་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཧོ་ཧོ་ཧོ་
ཨ་སྭཱ་ཧཱ། ཨོཾ་ཛྲུཾ་སྭཱ་ཧཱ། རྩ་བའི་སྔགས་སོ། །ཨོཾ་ཧཱུཾ་སྭཱ་ཧཱ། ཡེ་ཤེས་མཁའ་འགྲོ་མའོ། །ཨོཾ་སུཾ་སྭཱ་ཧཱ། ཨོཾ་ཀྵུཾ་སྭཱ་ཧཱ། ཨོཾ་ཡུཾ་སྭཱ་ཧཱ། ཨོཾ་ཧཱུཾ་སྭཱ་ཧཱ། ཨོཾ་སྨྼྻུཾ་སྭཱ་ཧཱ། ཨོཾ་ཧྨྼྻུཾ་སྭཱ་ཧཱ། ཨོཾ་ཡྨྼྻུཾ་སྭཱ་ཧཱ། ཨོཾ་ཀྵྨྼྻུཾ་སྭཱ་ཧཱ། ཨོཾ་ཕུཾ་སྭཱ་ཧཱ། ཨོཾ་ཕེཾ་སྭཱ་ཧཱ། ཨོཾ་ཕྲོཾ་སྭཱ་ཧཱ། ཨོཾ་ཕྲིཾ་སྭཱ་ཧཱ།
ཞེས་འབུམ་ཕྲག་གསུམ་བཟླས་པས་སྔོན་དུ་བསྙེན་པ་བྱས་པར་འགྱུར་རོ། །ཕྱི་ནས་ལས་ལ་སྦྱར་རོ། །དེ་ནས་ལངས་པར་འདོད་ན་སྨོན་ལམ་གདབ་སྟེ། མ་ལུས་སེམས་ཅན་ཀུན་གྱི་ཕྱིར། སྨོན་ལམ་དག་ནི་གདབ་པར་བྱ། །སེམས་ཅན་ཐམས་ཅད་ནད་མེད་ཅིང་། །
