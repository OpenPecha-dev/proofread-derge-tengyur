བཅས་པ་ནི། །སངས་རྒྱས་གཙོ་བོ་ཞེས་བྱར་གྲགས། །ཟླ་བའི་ཟེར་ལྟར་དྲི་མེད་པའི། །རིགས་ཀྱིས་ཁྱབ་པའི་འོད་ཟེར་འཕྲོ། །གལ་ཏེ་སྣ་ཚོགས་སྒྲུབ་འདོད་ན། །ཤེས་རབ་ཅན་གྱིས་འདི་བསམ་བྱ། །དེ་ནས་དྲོད་ཀྱི་གསུམ་པ་དང་། །བརྒྱད་ཀྱི་མཐའ་དང་བདུན་པ་ཡི། །དང་པོ་དེ་བཞིན་
དྲོད་པའི་མཐའ། །སྒྲ་དང་བཅས་པས་ཡང་དག་བརྒྱན། །ཤར་གྱི་ཕྱོགས་སུ་མཁའ་འགྲོ་མ། །དེ་བཞིན་བྱང་དུ་དྲག་མོ་བསྒོམ། །ནུབ་ཀྱི་ཕྱོགས་སུ་རོ་ལངས་མ། །ལྷོ་ཕྱོགས་མཁའ་འགྲོ་གཏུམ་མོ་སྟེ། །སྔོན་དུ་བསྟན་པའི་མདོག་བཞིན་དུ། །ཕྱོགས་ཀྱི་འདབ་མར་བསྒོམ་པར་བྱ། །མཚམས་
སུ་ཁ་དོག་འདྲེན་མ་རུ། །བརྩེགས་པའི་ས་བོན་བཞི་པོ་བསམ། །ཡན་ལག་བརྒྱད་ཀྱི་གོ་ཆ་གང་། །སྔགས་དེ་ཉིད་ཀྱིས་རྣལ་འབྱོར་ལྡན། །འོད་ཟེར་འབར་བའི་ཕྲེང་འཁྲུག་པའི། །སྔགས་དེ་དག་ནི་བལྟ་བར་བྱ། །ལས་བཞི་ཡི་ནི་གནས་སྐབས་ལ། །རྗེས་སུ་མཐུན་པའི་མི་ཤིགས་
བསྒོམས། །དེ་བཞིན་ལས་བཞི་སྒྲུབ་བྱེད་པས། །ལུས་ཀྱང་ཁ་དོག་བཞི་རུ་དགོད། །ཞལ་གཅིག་ཕྱག་གཉིས་ལ་སོགས་པ། །ཞལ་ཕྱག་ཇི་སྙེད་འདོད་པར་བྱ། །དྲག་ལ་ཁྲོ་བོར་བསྒོམ་པར་བྱ། །ཞི་བ་ལ་ནི་ཞི་བའི་ལུས། །རླུང་དང་མེ་ཡི་སྟེང་དུ་ནི། །ཉི་མ་ལ་གནས་དྲག་པོར་བྱ། །ས་ཆུ་ཟླ་
བ་ལ་གནས་ཏེ། །ཞི་བའི་ལས་ནི་བསྒོམ་པར་བྱ། །ཨོཾ་སུཾ་ཀྵུཾ་ཡུཾ་ཧཱུཾ་ཧཱུཾ། སྨྼྻུཾ་ཧྨྼྻུཾ་ཡྨྼྻུཾ་ཀྵྨྼྻུཾ། ཕུཾ་ཕེཾ་ཕྲོཾ་ཕྲིཾ། དཀྱིལ་འཁོར་གྱི་ས་བོན་ནོ། །ཨོཾ་ཧྲཱི་ཊ་ཊ་དྷ་ར་ཧཱུཾ་ཧཱུཾ་ཧཱུཾ་ཕུ་ཕུ་ཕུ། །ཨཱ་ཨཱ་ཨཱ་ཕཊ་ཕཊ་ཕཊ། ཀླུ་བརྒྱད་ཀྱི་དུག་དབྲོག་པའོ། །ཨོཾ་ཧྲཱི་ཀྲིཾ་ཀྵུཾ་སྭཱ་ཧཱ། །གྲོང་ཁྱེར་དཀྲུགས་པའིའོ། །ཨོཾ་བཏྟཱ་ལི་
རཀྵ་རཀྵ་ཧཱུཾ་ཧཱུཾ་ཕུ་ཕུཾ་ཕཊ། དམ་ཚིག་སྲུང་བའི་འཁོར་ལོ་འདི་བྲིས་ནས། དཔུང་པར་བཏགས་ཏེ་བསྲུང་བར་མོས། །ནད་པ་ནད་ལས་གྲོལ་གྱུར་ཞིང་། །དུས་ལ་བབ་པས་མཐོ་རིས་འགྲོ། །རྡོ་རྗེ་འཛིན་པར་སྙོམས་ཞུགས་ནས། །གཞན་གྱི་གདན་ནི་བསྟན་པ་སྟེ། །དངོས་གྲུབ་སྣ་
ཚོགས་བྱེད་པའི་གནས། །དངོས་གྲུབ་ཐམས་ཅད་ཀྱིས་ནི་གནས། །ཚེས་གཅིག་སོགས་ལ་རྐང་གཡོན་གྱི། །མཐེ་བོང་སོགས་འཕེལ་མགོ་བོར་གནས། །དེ་ནས་གཡས་ཀྱི་ཆ་ལ་ནི། །ཟླ་བ་གཉིས་ཀྱིས་འཇུ་བར་འགྱུར། །རླུང་དང་མེ་དང་ཉི་མ་ལ། །གནས་པའི་རང་ལ་ས་བོན་
