སྒྱུ་མ་དྲན་བྱ་ཞིང་། །དམ་པའི་ཆོས་ལ་བླ་མར་བྱོས། །ཞེས་གསུངས་པ་ཡང་། དམ་ཆོས་ཆོས་རྒྱལ་གསལ་བར་ལྡན། །འཇིག་རྟེན་སྣང་བར་བྱེད་པའི་མཆོག །ཅེས་སོ། །འདི་སྐད་བདག་གིས་ཞེས་པ་ལ་སོགས་པའི་རྡོ་རྗེའི་ཚིག་རྣམས་ཀྱི་དོན་ཡང་བཤད་པའི་རྒྱུད་ལ་འཇུག་པར་
བྱ་སྟེ། ཨེ་ཝཾ་རྡོ་རྗེ་ཆུ་སྐྱེས་དང་། །ནོར་བུའི་ནང་གནས་ཐིག་ལེ་འཛིན། །དེ་ཉིད་ཐ་དད་མེད་བདག་ཉིད། །མ་མཱ་ཞེས་པས་རབ་གསལ་བྱས། །འདི་སྐད་བདག་གིས་ཐོས་ཞེས་བརྗོད། །འཁོར་ལོས་སྒྱུར་བའི་གཟུགས་ཅན་གྱིས། །རྡོ་རྗེ་སེམས་དཔའ་བདེ་བ་ཆེ། །དེ་ནི་སྟོན་བྱེད་བདེ་བ་
ང་། །ཐོས་པ་ཞེས་པ་བདག་གིས་ནི། །བདག་ཉིད་ཁོ་ནས་བདག་ཉིད་ཀྱིས། །ཞེས་སོ། །གསུངས་པའང་། ང་ལས་འགྲོ་བ་ཐམས་ཅད་བྱུང་། །ང་ལས་ས་གསུམ་པོ་ཡང་བྱུང་། །ང་ཡིས་འདི་ཀུན་ཁྱབ་པ་སྟེ། །འགྲོ་བ་རང་བཞིན་གཞན་མ་མཐོང་། །ཞེས་སོ། །སླར་ཡང་འདི་
ཉིད་དུ་གསུངས་པ། །འཆང་བ་པོ་ང་ཆོས་ཀྱང་ང་། །རང་གི་ཚོགས་ལྡན་ཉན་པ་ང་། །འགྲོ་བའི་སྟོན་པ་བསྒྲུབ་བྱ་ང་། །འཇིག་རྟེན་པ་ང་འཇིག་རྟེན་དང་། །ལྷན་ཅིག་སྐྱེས་དགའི་རང་བཞིན་ང་། །མཆོག་མཐའ་དགའ་བྲལ་དང་པོ་ཅན། །དངོས་དང་དངོས་པོ་མེད་པ་ང་། །དངོས་པོ་རྟོག་
ཕྱིར་སངས་རྒྱས་ང་། །གང་ཞིག་ལེ་ལོས་བསྣུན་པ་དང་། །རྨོངས་པ་གང་གིས་ང་མི་ཤེས། །རྡོ་རྗེ་བཙུན་མོའི་བྷ་ག་ནི། །བདེ་བ་ཅན་དུ་ང་གནས་སོ། །ཞེས་སོ། །དཔལ་ཐམས་ཅད་མཁྱེན་པ་རྡོ་རྗེ་འཇིགས་བྱེད་ཀྱི་རྒྱུད་དུ་ཡང་གསུངས་པ། སྤྲོ་དང་སྡུད་པའི་གནས་ལ་ནི། །གསང་
ཆེན་དམ་པ་རྨད་བྱུང་བར། །ཤེས་རབ་བདུད་རྩིའི་མཚོ་གནས་སུ། །བཅོམ་ལྡན་འདས་ནི་སྐྱེ་མེད་བཞུགས། །སྟོང་པ་མ་ཡིན་མི་སྟོང་མིན། །རྣམ་པར་མི་རྟོག་གཞི་མེད་པ། །སྤྲོས་པ་མེད་ཅིང་སྣང་བ་མེད། །རྡོ་རྗེ་སེམས་དཔའ་བདེ་བ་ཆེ། །ཞེས་སོ། །དཔལ་དང་པོའི་སངས་རྒྱས་
སུ་ཡང་གསུངས་པ། །ཕུང་པོ་ལྔའི་རང་བཞིན་རྣམས་ཀྱིས་རྡོ་རྗེ་བཙུན་མོའི་བྷ་ག་རྣམས་ལ་ཐུགས་རྗེས་བཞུགས། །ངེས་པ་རྣམ་པར་མ་བསྒྲུབས་བསོད་ནམས་མིན་པ་བསགས་པའི་སེམས་ཅན་རྣམས་སྨིན་དོན་དུ་སྟེ། །གཙང་མའི་གནས་ལ་སོགས་པར་བཅོམ་ལྡན་གང་ཞིག་རྣམ་པར་
