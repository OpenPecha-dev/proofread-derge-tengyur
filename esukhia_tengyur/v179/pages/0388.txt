འགྱུར་བ་ནི། །དེ་མེད་དེས་ན་མེད་འདོད་ན། །དེས་བསྐྱེད་པ་ཡི་ནུས་པ་ཉིད། །ཡོད་པ་ཉིད་ཕྱིར་རྟོགས་པར་འགྱུར། །དེ་ལྟ་མིན་ན་ད་རྟོགས་པས། །དེ་ཡོད་པས་ནི་རྟོགས་མི་འགྱུར། །འོན་ཏེ་གང་ལས་གང་སྐྱེས་པ། །དེ་ལ་དེ་ཡོད་པར་རྟོགས་ན། །སྐྱེས་པའི་སྔོན་རོལ་ཤེས་མེད་པ། །དེ་ལས་
དེ་མིན་ཅི་ལྟར་རྟོགས། །དེ་ལྟ་སྐྱེ་བ་ཡོད་པ་དང་། །འབྲེལ་པ་རབ་ཏུ་འགྲུབ་མ་ཡིན། །ཅི་ལྟར་རྒྱུ་ལས་འབྲས་ཡིན་པ། །དེ་ཡོད་ལས་ཀྱང་དེ་འབྱུང་འགྱུར། །འབྲས་གང་གང་ཞིག་སྔོན་འགྲོ་ཅན། །དེ་ནི་དེའི་ཞེས་གནས་ཕྱིར་རོ། །དེ་ཡོད་སྔོན་འགྲོ་ཅན་འབྲས་བུ། །དེ་དེ་ཉིད་ཀྱིས་ཅི་ལྟར་འགྱུར། །
གཞན་བཞིན་འབྲས་བུ་ཐམས་ཅད་ནི། །ནུས་ལས་འབྱུང་བའི་ཕྱིར་ཞེ་ན། །ནུས་པ་ཉིད་ལས་ནུས་འབྱུང་འགྱུར། །ཐ་དད་མེད་ན་ཅི་ལྟར་འགྱུར། །གལ་ཏེ་ནུས་ལས་གསལ་སྐྱེ་ན། །ནུས་ལ་གསལ་བ་གཞན་ནས་ཅི། །གཞན་ནམ་མེད་པས་བསྐྱེད་བྱ་མིན། །གལ་ཏེ་ཡོད་ན་ཅི་ལྟར་བསྐྱེད། །
གསལ་བ་བློ་ཡིན་དེ་ཡང་ཡོད། །དེས་ན་དེ་ནི་བསྐྱེད་བྱ་མིན། །འོན་ཏེ་གསལ་བ་མྱོང་ཡིན་ན། །བསྐྱེད་བྱ་མིན་དེ་ཅི་ལྟར་སྐྱེ། །དེས་ན་འབྲས་བུ་རང་གིས་ནི། །བསྐྱེད་པ་པོས་གསལ་ཚད་ལྡན་མིན། །བུམ་པ་མུན་ནང་གནས་པ་ལ། །རིག་པའི་ཚད་མ་རྗེས་དཔག་ཡོད། །བུམ་སོགས་འཇིམ་
གོང་ལ་སོགས་ལ། །ཡོད་ལ་རིག་པའི་ཚད་མ་མེད། །དེས་ན་ཐམས་ཅད་གསལ་བ་བཞིན། །བྱས་པའི་ཕྱིར་ན་མི་རྟག་ཉིད། །དེས་ན་རྟགས་ཉིད་ཕྱོགས་སུ་ནི། །བྱས་ཏེ་རྗེས་དཔག་ཅེས་བྱས་གནོད། །འོ་ན་མངོན་སུམ་གྱི་དོན་དང་རྗེས་སུ་དཔག་པ་དང་། ཡིད་ཆེས་པ་དང་གྲགས་པས་རང་གི་ཆོས་
ཅན་ལ་མ་བསལ་བ་ཞེས་རང་གི་ཆོས་ཅན་སྨོས་པ་ལ་དོན་ཅི་ཡོད། འདི་ལྟར། ཚད་མས་གནོད་པ་ཕྱོགས་མིན་ཞེས། །འདོད་པ་འདི་ནི་རིགས་པ་ཡིན། །ཆོས་ཅན་དེ་འམ་གཞན་དག་ལ། །བྱ་བ་འདི་ནི་གང་ལ་དགོས། །ཆོས་ཅན་གཞན་ལ་གནོད་པ་ནི། །གནོད་པ་མ་ཡིན་པར་མི་འགྱུར་རོ། །རྣམ་
པ་ཐམས་ཅད་དུ་གནོད་པ་ནི་བསྒྲུབ་པར་ནུས་པ་མ་ཡིན་པའི་ཕྱིར་གཏན་ཚིགས་ཀྱི་ཡུལ་མེད་པས་ཕྱོགས་མ་ཡིན་ནོ་ཞེས་རིགས་སོ་ཞེ་ན། འདིར་བཤད་པ། ཀུན་ཏུ་རྒོལ་བ་བདག་ཉིད་ཀྱི། །བསྒྲུབ་བྱར་འདོད་པའི་ཆོས་གང་ཡིན། །ཆོས་དེ་ལྡན་ལ་གནོད་འགྱུར་གྱི། །ཆོས་གཞན་གྱི་ནི་ཆོས་གཞན་མིན། །
