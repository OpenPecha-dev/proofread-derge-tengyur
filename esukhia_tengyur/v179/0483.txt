རོ་ཞེས་ད་ལྟར་ཁོ་ན་བསྟན་པ་ཉིད་དོ། །དེས་ན་དཔེ་གཅིག་ཉིད་ཀྱིས་གཉི་ག་བསྟན་པར་འགྱུར་བའི་ཕྱིར་ནི་བྱས་པའི་ཕྱིར་རོ་ཞེས་བསྟན་པ་ནི་དོན་མེད་པ་ཁོ་ནའོ་ཞེ་ན། དེ་ནི་བདེན་མོད་ཀྱི་འོན་ཀྱང་། འདི་ཉིད་ཀྱིས་ནི་རང་བཞིན་ཉིད། །གྲུབ་ཀྱང་ཐ་དད་བྱས་པ་ནི། །
འབྲས་དང་ལྷན་ཅིག་བསྟན་པ་ན། །ཐམས་ཅད་དེ་འདྲར་ཤེས་མི་འགྱུར། །འབྲས་བུ་དང་ལྷན་ཅིག་བསྟན་པ་ན་དེ་དང་ལྷན་ཅིག་སྤྱོད་པ་ཡིན་པའི་ཕྱིར་རང་བཞིན་གྱི་གཏན་ཚིགས་མཐའ་དག་ནི་འདི་ལྟ་བུ་ཉིད་ཡིན་ཏེ། རང་བཞིན་གྱི་གཏན་ཚིགས་ཀྱང་མཐུན་པའི་ཕྱོགས་གཅིག་ལ་འཇུག་
པ་ཉིད་ཡིན་ནོ་ཞེས་འདོགས་པར་འགྱུར་བ་དེ་རྣམ་པར་གཅད་པའི་གང་ཞིག་མཐུན་པའི་ཕྱོགས་ལ་ཁྱབ་པའི་རང་བཞིན་གྱི་གཏན་ཚིགས་ཀྱི་རྣམ་པ་གཞན་ཀྱང་ཡོད་དོ་ཞེས་བྱ་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་འདི་ལོགས་སུ་བསྟན་ཏོ། །གལ་ཏེ། དེ་ལ་རིགས་མཐུན་པ་ཡོད་དང་། །རྣམ་གཉིས་
མེད་ལ་མེད་གང་ཡིན། །དེ་ནི་གཏན་ཚིགས་དེ་ལས་བཟློག །འགལ་བ་གཞན་ནི་མ་ངེས་ཉིད། །བཤད་པ་མ་ཡིན་ནམ་ཅིའི་ཕྱིར་ལོགས་སུ་བརྗོད་ཅེ་ན། འདི་ལྟར་བཤད་པ། རྗེས་སུ་དཔག་པར་བཤད་ཟིན་དོན། །གཏན་ཚིགས་བརྗོད་བྱས་རྟོགས་དོན་དུ། །རབ་ཏུ་དབྱེ་བ་ཙམ་བཤད་
དེ། །མཚན་ཉིད་ལ་ནི་ཐ་དད་མེད། །ཐ་དད་པར་བཤད་པས་ནི་དེ་ཉིད་གསལ་བར་འགྱུར་བ་ཡིན་ནོ། །དེ་ལ་རིགས་མཐུན་པ་ཡོད་དང་། །རྣམ་གཉིས་ཞེས་བྱ་བ་ནི་དཔེར་ན་སྒྲ་ནི་མི་རྟག་སྟེ་བྱས་པའི་ཕྱིར་ཞེས་བྱ་བ་དང་བརྩལ་མ་ཐག་ཏུ་བྱུང་བའི་ཕྱིར་རོ་ཞེས་བྱ་བ་ལྟ་བུའོ་ཞེས་བྱ་བ་ཡིན་ནོ། །
དེ་ལྟ་མ་ཡིན་ཏེ་གཉི་ག་ཡང་མ་བསྟན་ན་རྣམ་པ་གཉིས་ཞེས་བྱ་བའི་མཚན་ཉིད་འདི་ཉིད་སྲིད་པ་ཡིན་གྱི། རིགས་མཐུན་པ་དང་པོ་ཡོད་པ་ཉིད་དོ་ཞེས་བྱ་བ་འདི་ནི་དཔེ་མེད་པའི་ཕྱིར་མཚན་ཉིད་མི་སྲིད་པ་ཡིན་ནོ་ཞེས་བྱ་བར་འགྱུར་རོ། །གལ་ཏེ་མི་མཐུན་པའི་ཕྱོགས་ལ་ལྡོག་པ་ཡིན་ན་དེ་ཙམ་
གྱིས་ཐམས་ཅད་བསྡུས་པའི་ཕྱིར་དོགས་པར་མི་རིགས་སོ་ཞེ་ན། དེ་ནི་བདེན་ཏེ་དེ་ལྟ་ན་ཡང་དཔེར་བརྗོད་པའི་སྤྲོས་པ་འདི་ནི་རྗེས་སུ་དཔག་པའི་མཚན་ཉིད་དེ་ཉིད་རྟོགས་པར་བྱ་བའི་ཕྱིར་ཡིན་ནོ། །མཚན་ཉིད་ནི་རང་གི་དོན་རྗེས་སུ་དཔག་པར་བསྟན་པ་དེ་ཉིད་ཡིན་ནོ། །འོ་ན་ནི་གཏན་ཚིགས་
