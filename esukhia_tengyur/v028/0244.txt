ལ་ནི། །དྲི་ཟའི་གྲོང་ཁྱེར་མཐོང་བ་དང་། །{𰵶ཡི་དགས་,ཡི་དྭགས་}ཤ་ཟ་འཇིགས་རུང་བའི། །གཞན་དག་ཀྱང་ནི་མཐོང་འགྱུར་དང་། །ཤིན་ཏུ་འདར་བར་འགྱུར་བ་དང་། །ཡང་དང་ཡང་དུ་བརྒྱལ་བ་དང་། །སྐྱུགས་པ་བཤང་དང་གཅི་བ་ན། །གསེར་རམ་དངུལ་གྱི་མདོག་ཏུ་མཐོང་། །དེ་
ཡིས་རེ་རེ་མཐོང་འགྱུར་ནའང་། །ཟླ་བ་གཅིག་གིས་འཆི་བར་འགྱུར། །ལྐོག་མ་མཆུ་དང་རྐན་དང་ལྕེ། །ཆོད་དང་བུ་ག་དེ་བཞིན་ལུས། །མཐོང་ན་དེ་བཞིན་ཟླ་དྲུག་གི །ནང་དུ་ངེས་པར་འཆི་བར་འགྱུར། །གྲེ་བ་མཆུ་དང་རྐན་དང་ལྕེ། །རྒྱུན་དུ་སོ་སོར་སྐམས་ན་ནི། །
དེ་ནི་ཟླ་དྲུག་ནང་དུ་ནི། །ངེས་པར་འཆི་བར་སྟོན་པ་ཡིན། །སྐར་མའི་ཚོགས་དང་དྲང་སྲོང་བདུན། །ཕྱོགས་སུ་མཚན་མོ་མ་མཐོང་ངམ། །ལུས་ནི་གློ་བུར་འབར་འགྱུར་ན། །ཞག་བདུན་ན་ནི་འཆི་བར་འགྱུར། །སྒྲིབ་མེད་དང་ནི་སྣར་མ་དང་། །བསྟན་དང་དེ་བཞིན་གྲིབ་པའི་
ལམ། །མཚན་མོ་གློ་བུར་དུ་བ་མཐོང་། །ཟླ་བ་དྲུག་ལ་འཆི་བར་འགྱུར། །ཟླ་བ་ར་རི་མེད་པ་དང་། །ཉི་མ་འོད་དང་བྲལ་བ་དང་། །མེ་ནི་སྣ་ཚོགས་འོད་བཅས་དང་། །ཟླ་བ་ཡང་ནི་འོད་ཟེར་བཅས། །མཚན་མོ་ཉི་མ་ཉིན་པར་མཚན། །དེ་བཞིན་རང་མིག་འབར་བ་དང་། །སྐར་
མ་རི་རབ་དག་གི་ཚད། །རྒྱ་མཚོ་ཆུ་བོ་ལྟ་བུ་དང་། །ལྟེ་བའི་སྐྱིབས་སུ་བཤང་སྒོ་ལ། །སྦྲིད་པ་བཞིན་མདོག་པདྨ་དང་། །ལྐོག་མར་འབྲུམ་པ་དམར་པོ་དང་། །ལུས་ལ་ཁ་དོག་སྣ་ཚོགས་དང་། །སྙིང་ག་ནས་སྨྲེ་མཆི་མར་བཅས། །ཡང་ན་རྐན་ནི་འགུལ་པ་དང་། །ཟླ་
ཟེར་མེ་ལྟར་འཚེར་བ་དང་། །ཉི་ཟེར་ཁ་བ་ལྟ་བུར་བསིལ། །ཟླ་བ་ཉི་མའི་བུ་ག་དང་། །ས་དང་གནམ་གྱི་བུ་ག་དང་། །གང་ཞིག་བདག་ཉིད་ཁོ་ནས་མཐོང་། །དེ་ནི་ཟླ་ཕྱེད་ནས་འཆི་མཐོང་། །ཁུ་བ་དང་ནི་བཤང་གཅི་རྣམས། །དུས་ནི་མཚུངས་པར་འབབ་འགྱུར་ན། །ལོ་
གཅིག་ནས་འཆི་དེ་ལ་ནི། །སྨན་པའི་ཐབས་ཀྱིས་གསོ་བར་བྱ། །གང་གི་ལུས་ནི་ཕྱེད་དག་ནི། །གྲང་ཞིང་དེ་བཞིན་ཕྱེད་དྲོ་ན། །ཞག་བདུན་ནས་ནི་འཆི་འགྱུར་གྱིས། །འཇིག་རྟེན་ཕ་རོལ་རྣམ་བསམ་བྱ། །ཧ་ཞེས་བྱ་བ་གྲང་བར་ཚོར། །ཕུ་ནི་མེ་དང་འདྲ་འགྱུར་ན། །གང་
