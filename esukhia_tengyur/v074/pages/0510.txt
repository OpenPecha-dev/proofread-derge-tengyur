ཆུབ་ཏུ་སངས་རྒྱས་ཀྱི་བར་དུ་སུ་ཡང་སེམས་ཅན་དམྱལ་བ་དང་ངན་འགྲོར་འགྲོ་བར་མ་གྱུར་ཅིག །ལྷ་ཡུལ་བའི་བདེ་བ་དང་། མིའི་བདེ་བ་མྱོང་བར་གྱུར་ཅིག །བདག་ཅག་ལ་སོགས་པ་སེམས་ཅན་གང་དག་དབུལ་པོར་གྱུར་ཏེ། ཟས་དང་། སྐོམ་དང་། ནོར་དང་། འབྲུ་དང་།
མལ་ཆ་དང་། སྤོས་དང་། ཕྲེང་བ་དང་། ཡོ་བྱད་མི་བདོག་པ་དེ་དག་ཐམས་ཅད་བྱང་ཆུབ་ཀྱི་མཐར་ཐུག་གི་བར་དུ་དབྱིག་དང་། གོས་དང་། ནོར་དང་། འབྲུ་དང་། མལ་ཆ་དང་། སྤོས་དང་། ཕྲེང་བ་དང་། ཡོ་བྱད་འཕེལ་བར་གྱུར་ཅིག །བདག་ཅག་ལ་སོགས་པའི་སེམས་ཅན་གང་
དག་བཅིང་བ་དང་། བརྡེག་པ་དང་། གོ་རར་བཞུག་པ་དང་། མཚོན་ཆའི་ཚོགས་རྣམ་པ་སྣ་ཚོགས་ཀྱིས་ལུས་ཡོངས་སུ་གཟིར་བ་དེ་དག་ཐམས་ཅད་བྱང་ཆུབ་ཀྱི་མཐར་ཐུག་གི་བར་དུ་བཅིང་བ་དང་། བརྡོག་པ་ཐམས་ཅད་ལས་ཡོངས་སུ་ཐར་བར་གྱུར་ཅིག །ལུས་ཀྱི་གདུང་བ་
ཐམས་ཅད་རྒྱུན་ཆད་པར་གྱུར་ཅིག །བདག་ཅག་ལ་སོགས་པ་སེམས་ཅན་གང་དག་མགོན་མེད་པར་གྱུར་པ་ལུས་ལ་གདུང་བའི་མེ་འབར་བ་དང་། སེང་གེ་དང་། སྟག་དང་། དོམ་དང་། དྲེད་དང་། སྦྲུལ་གདུག་པའི་ནང་དུ་ཆུད་དེ། སྲོག་ལ་འབབ་ཅིང་ཉམ་ཐག་པ་དེ་དག་ཐམས་
ཅད་བྱང་ཆུབ་ཀྱི་མཐར་ཐུག་གི་བར་དུ་ལུས་ལ་གདུང་བའི་མེ་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག །སེང་གེ་དང་། སྟག་དང་། དོམ་དང་། དྲེད་དང་། སྦྲུལ་གདུག་པ་དེ་དག་ཀྱང་བྱམས་པ་ལ་གནས་པར་གྱུར་ཅིག །བདག་ཅག་ལ་སོགས་པ་སེམས་ཅན་གང་དག་ཕན་ཚུན་འཐབ་
རྩོད་ཅིང་འགྱེད་ལ། འཁྲུགས་ཤིང་འཇིགས་སྐྲག་ནས་སྔངས་པ་དེ་དག་ཐམས་ཅད་བྱང་ཆུབ་ཀྱི་མཐར་ཐུག་གི་བར་དུ་འཐབ་རྩོད་དང་འགྱེད་པ་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་གྱུར་ཅིག །ཕན་ཚུན་བྱམས་པ་ལ་གནས་པར་གྱུར་ཅིག །བདག་ཅག་ལ་སོགས་པ་སེམས་ཅན་གང་དག་
ཆུའི་ནང་དུ་སོན་ཏེ་གྲུར་ཞུགས་པ། གླིང་མེད་པར་ཕྱིན་པ། སྲོག་འདོད་པ། ཆུ་ཀླུང་དང་། རྒྱ་མཚོའི་ནང་དུ་སོན་ཏེ། སྨྲེ་སྔགས་སྣ་ཚོགས་འདོན་པ། དེ་དག་ཐམས་ཅད་བྱང་ཆུབ་ཀྱི་མཐར་ཐུག་གི་བར་དུ་ཆུ་ཀླུང་ཆེན་པོ་དེ་དག་ལས་དབུགས་འབྱིན་པ་སྦྱིན་པར་གྱུར་ཅིག །འཇིགས་
