མ་ངན་པ་ཐམས་ཅད་རབ་ཏུ་ཞི་བར་མཛད་པ། སྔོན་སངས་རྒྱས་འདུས་པ་ཆེན་པོའི་ཚེ། ཐེག་པ་ཆེན་པོའི་མདོ་སྡེ་དང་། འདུལ་བ་དང་། ཆོས་མངོན་པ་དང་། གཟུངས་སྔགས་དང་། རིག་སྔགས་དང་། སྙིང་པོ་དང་། ཕྱག་རྒྱ་ལ་སོགས་པ་ཟབ་ཅིང་རྒྱ་ཆེན་པོ་བཤད་པའི་ཚེ་ན། སངས་
རྒྱས་བཅོམ་ལྡན་འདས་ཀྱིས་ང་ཡོངས་སུ་མྱ་ངན་ལས་འདས་པའི་འོག་ཏུ་ཕྱི་མའི་དུས་ཀྱི་ཚེ་ངའི་བསྟན་པ་དམ་པའི་ཆོས་དང་། དགེ་སློང་དང་། དགེ་སློང་མ་དང་། དགེ་བསྙེན་དང་། དགེ་བསྙེན་མའི་འཁོར་རྣམ་བཞི་དང་། རྒྱལ་པོ་དང་། བློན་པོ་དང་། རིགས་བཞི་ཆོས་དང་ལྡན་ཞིང་
ཆོས་སྤྱོད་པ་ཐམས་ཅད་དང་། ལྷ་ཡུལ་དང་། ཡུལ་འཁོར་དང་། གྲོང་དང་། གྲོང་ཁྱེར་ལ་སོགས་པ་འབྱུང་པོའི་ཚོགས་དགེ་བ་རྣམས་བསྲུངས་ཤིག་ཅེས་ཞལ་གྱིས་བཀའ་སྩལ་ཞིང་། དེ་དག་གིས་ཀྱང་དམ་ཚིག་མནོས་ཏེ་ལྷ་དང་མིའི་མགོན་མཛད་པ། ཕྱག་བྱ་བའི་གནས། བཀུར་སྟི་
བྱ་བའི་གནས། བླ་མར་བྱ་བའི་གནས། བཙུན་པར་བྱ་བའི་གནས། འདི་ལྟ་སྟེ། འོག་མིན་ན་ལྷའི་དབང་ཕྱུག་ཆེན་པོ་གནས་གཙང་མ་གཞན་གྱི་རིས་ཀྱི་ལྷ་དང་། བསམ་གཏན་བཞི་པ་དང་། གསུམ་པ་དང་། གཉིས་པ་དང་། བསམ་གཏན་དང་པོའི་རིས་ཀྱི་ལྷ་དང་། གཞན་འཕྲུལ་དབང་བྱེད་དང་།
འཕྲུལ་དགའ་དང་། དགའ་ལྡན་དང་། འཐབ་བྲལ་དང་། སུམ་ཅུ་རྩ་གསུམ་དང་། རྒྱལ་ཆེན་བཞིའི་རིས་ཀྱི་ལྷ་དང་། ཀླུ་དང་། གནོད་སྦྱིན་དང་། དྲི་ཟ་དང་། ལྷ་མ་ཡིན་དང་། ནམ་མཁའ་ལྡིང་དང་། མིའམ་ཅི་དང་། ལྟོ་འཕྱེ་ཆེན་པོ་དང་། མི་དང་མི་མ་ཡིན་པ་ལ་སོགས་པ་བདག་ཉིད་
ཆེན་པོ་ཁྱད་པར་ཅན་གྱི་ཚོགས་ཐམས་ཅད་དང་། དམ་ཚིག་ཅན་དང་། སངས་རྒྱས་དང་། ཆོས་དང་། དགེ་འདུན་ལ་སྐྱབས་སུ་སོང་བ། སངས་རྒྱས་ཀྱི་བསྟན་པ་བསྲུང་བ། དམ་པའི་ཆོས་སྤེལ་བར་དམ་བཅས་པ། ཚངས་པ་དང་། བརྒྱ་བྱིན་དང་། རྒྱལ་པོ་ཆེན་པོ་བཞི་དང་། ཕྱོགས་སྐྱོང་བཅུ་
