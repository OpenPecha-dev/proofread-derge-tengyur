བི་དཱ་ར་ཎ་དྷཱ་ར་ཎཱི་སཱ་དྷ་ནཱ་མ། བོད་སྐད་དུ། རྡོ་རྗེ་རྣམ་པར་འཇོམས་པའི་གཟུངས་ཀྱི་སྒྲུབ་ཐབས་ཞེས་བྱ་བ། བཅོམ་ལྡན་འདས་དཔལ་ཀུན་དུ་བཟང་པོ་ལ་ཕྱག་འཚལ་ལོ། །མི་ཤིགས་ཀུན་འཇོམས་རྡོ་རྗེ་འཛིན། །ཁྲོས་ཤིང་གཏུམ་པའི་སྐུ་མངའ་བ། །རྡོ་རྗེ་ཡོན་ཏན་རྫོགས་བཏུད་དེ། །
དེ་ཉིད་བསྒྲུབ་པའི་རིམ་པ་བཤད། །དད་བརྟན་གཡོ་དང་ང་རྒྱལ་མེད། །སེར་སྣ་ཆུང་ཞིང་བླ་མར་གུས། །འདི་ལ་ལྷག་པར་ཆེ་དད་པས། །བསྒྲུབ་པ་འདི་འབད་བདེ་ལེགས་ཤོག །དང་པོར་ཡིད་འོང་ཕྱོགས་སུ་ཡང་། །འགྲོ་བ་བསྒྲལ་བའི་བསམ་པ་ཡིས། །རྟག་ཏུ་འདི་ལ་བརྩོན་པར་
བྱ། །མཎྜལ་མཆོད་དང་གཏོར་མ་དང་། །ལས་དང་པོ་ཡི་རྣལ་འབྱོར་པས། །དམིགས་པའི་རྟེན་ཀྱང་བྲི་བར་བྱ། །རང་ཉིད་ཕྱི་ནང་ཁྲུས་བྱས་ཏེ། །ཁྱད་པར་དུ་ནི་ལུས་ཀྱི་ཁྲུས། །དྲི་ཡི་ཆུ་ཡིས་རྟག་ཏུ་བྱ། །བདེ་བའི་སྟན་ལ་འདུག་ནས་ནི། །རང་ཉིད་ཟླ་བ་ས་བོན་བསམ། །དེ་ཡི་འོད་
ཀྱིས་བླ་མ་སོགས། །ནམ་མཁའི་ཕྱོགས་སུ་དམིགས་ནས་ནི། །བྱིན་གྱིས་བརླབ་པར་གསོལ་བ་གདབ། །དེ་ནས་སེམས་ཅན་ཐམས་ཅད་ལ། །ལྷག་པར་བྱམས་པ་བསྐྱེད་པར་བྱ། །དེ་བཞིན་སྙིང་རྗེ་དགའ་བ་དང་། །བཏང་སྙོམས་ཚུལ་བཞིན་བསྒོམ་པར་བྱ། །དེ་རྗེས་རྣམ་པར་མི་
རྟོག་པའི། །རྡོ་རྗེའི་རང་བཞིན་བསྒོམ་པར་བྱ། །དེ་ལ་ཅུང་ཟད་དམིགས་ནས་ནི། །རྣམ་པར་རྟོག་པར་ལྡང་བར་བྱ། །གཟུགས་ལ་སོགས་པ་མི་དམིགས་ཀྱང་། །སྔགས་ཀྱི་སྒྲ་རུ་བསྒོམ་པར་བྱ། །དེ་རྗེས་རང་གི་ཤེས་པ་ནི། །སྔགས་ཀྱི་ཕྲེང་བར་བསྒོམ་པར་བྱ། །གསུང་གི་རྣམ་
པར་རྫོགས་པ་ཡི། །ཕྲ་ཞིང་གསལ་པའི་རྣམ་པར་རོ། །དེ་ནས་ཟླ་བའི་དཀྱིལ་འཁོར་དེ། །གསལ་ཞིང་ཆ་མ་འདྲེས་པར་བསམ། །དེ་ཡི་འོད་ཀྱིས་འགྲོ་བ་རྣམས། །རྣམ་པར་འཇོམས་པ་བྱས་ནས་ནི། །ཚུར་ལ་བསྡུ་བའི་ཚུལ་གྱིས་ནི། །རང་སྙིང་རྣམ་པར་འཇོམས་པར་བསྒོམ། །ཞབས་
