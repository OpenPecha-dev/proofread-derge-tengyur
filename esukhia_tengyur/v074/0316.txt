དང་། གཟའ་ཆེན་པོ་བརྒྱད་དང་། རྒྱུ་སྐར་ཉི་ཤུ་རྩ་བརྒྱད་དང་། ཕྱོགས་ཀྱི་གླང་པོ་དང་། སྡེ་དཔོན་དང་། ལྷ་མོ་དང་། ཕྱོགས་ཀྱི་ལྷ་དང་། ཕྱོགས་མཚམས་ཀྱི་ལྷ་དང་། བཙུན་མོ་དང་། སྲས་དང་། འཁོར་དང་། ཕོ་ཉ་དང་། མངག་པ་དང་། ཁྱབ་འཇུག་དང་། ཚོགས་བདག་དང་།
དགའ་བྱེད་དབང་ཕྱུག་དང་། སྨིན་དྲུག་གི་བུ་དང་། ནག་པོ་ཆེན་པོ་དང་། སྟོབས་ཅན་དང་། སྟོབས་པོ་ཆེ་དང་། གནོད་འཛིན་དང་། པདྨ་དང་། པདྨ་ཆེན་པོ་དང་། དུང་དང་། དུང་ཆེན་དང་། གང་པོ་དང་། ཤིན་ཏུ་གང་བ་དང་། མགྲིན་པ་དང་། མགྲིན་བཟངས་དང་། བསོད་ནམས་ལྷུན་པོ་དང་། གང་བ་བཟང་པོ་
དང་། ནོར་བུ་བཟང་པོ་དང་། བུམ་རྣ་དང་། དྲིལ་རྣ་དང་། རྣ་གསུམ་དང་། རྣ་བོ་ཆེ་དང་། ཡང་དག་ཤེས་དང་། ཡང་དག་རྒྱལ་བ་དང་། རྒྱལ་བ་ཁྱུ་མཆོག་དང་། འབྲོག་གནས་དང་། གང་གནས་དང་། བདེ་རི་དང་། ལྔ་རྩེན་དང་། ལྔ་ལེན་དང་། ཚིགས་ལྔ་སེར་དང་། མགྲིན་བཅུ་དང་།
འཇིགས་རུང་དང་། འབྲུག་སྒྲ་དང་། ནམ་མཁའི་<དབྱངས་,དབྱིངས་>དང་། སྟོང་གསུམ་དང་། རྒྱ་མཚོ་དང་། དགའ་བོ་དང་། ཉེ་དགའ་དང་། མ་དྲོས་དང་། གཟི་ཅན་དང་། ནོར་རྒྱས་དང་། གདོན་ལ་འཇེབས་དང་། དྲང་སྲོང་དང་། རིག་སྔགས་འཆང་དང་། ཆར་ལྷ་དང་། སྤྲིན་ལྷ་དང་། ལོ་ཏོག་
ལྷ་དང་། ལམ་ལྷ་དང་། སའི་ལྷ་དང་། ཆུའི་ལྷ་དང་། མེའི་ལྷ་དང་། རླུང་གི་ལྷ་དང་། ནམ་མཁའི་ལྷ་དང་། རིའི་ལྷ་དང་། ཤིང་གི་ལྷ་དང་། རྫིང་གི་ལྷ་དང་། དུས་ཚིགས་ཀྱི་ལྷ་དང་། ཉིན་མཚན་གྱི་ལྷ་དང་། དུས་མཚམས་ཀྱི་ལྷ་དང་། ཚེས་གྲངས་ཀྱི་ལྷ་དང་། སྐད་ཅིག་དང་། ཐང་
ཅིག་དང་། ཡུད་ཙམ་གྱི་ལྷ་དང་། ཧ་ནུ་མ་ཐ་ལ་སོགས་པ་མཐུ་བོ་ཆེ་དང་། ཚིག་ལ་དབང་བ་དང་། དགའ་དབྱངས་དང་། མདངས་བུ་དང་། དཔལ་གྱི་ལྷ་མོ་དང་། འོད་ཟེར་ཅན་དང་། དཀར་ཤམ་དང་། བརྗོད་སྐྱོབ་དང་། གངྒཱ་དང་། ཡ་མུ་ན་དང་། བུ་མངས་དང་། དུང་ཅན་དང་། རི་
མཁར་དང་། ནག་མོ་དང་། མ་བདུན་དང་། སྲིང་མོ་བདུན་དང་། སྲིང་མོ་བཞི་དང་། ནག་མོ་ཆེན་མོ་དང་། རལ་པ་གཅིག་མ་ལ་སོགས་པ་དང་། ལྷ་མོ་ཆེན་མོ་དང་། གནོད་སྦྱིན་ཆེན་མོ་དང་། སྲིན་མོ་དང་། ཤ་ཟ་ཆེན་པོ་དང་། གཞོན་ནུ་དང་། རྫུ་འཕྲུལ་ཅན་དང་། དབང་བདག་དང་།
