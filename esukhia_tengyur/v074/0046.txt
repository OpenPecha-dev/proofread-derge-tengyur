ན་ལ་བྷིནྡ་ན་མཎྜ་ལ་བུདྡྷ་སརྦ་བྷསྨི་ཀུ་རུ་སརྦ་དུཥྚཱཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཧྱེ་ཧི་བཛྲཱ་ཀཱ་ལ་དཎྜ་ཧ་ན་ཧ་ན་ཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཧྱེ་ཧི་བཛྲ་ཀི་རི་ཊི་སྟཾ་རཀྵ་ས་དེ་བ་ཏ་ཡཻ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧྱ་ཧི་ནཱ་ག་བཛྲ་ཨཱ་ན་ཡ་སརྦ་དྷ་ན་དྷ་ནྱ། སརྦ་ན་མ་ནི་མུཀྟཱ་ལཾ་ཀ་ར་དྷ་ར་སརྦ་ནཱ་ག་ན་ར་ནི་ཙ་བཛྲ་དྷ་ར་ཡ། ས་མ་ཡཱ་མ་ནུ་སྨཱ་ར་ཨ་ཀཌྜྷྱ་ཨ་ཀ་ཌྜྷྱ་བནྡྷ་
བནྡྷ་ཧ་ར་ཧ་ར་པྲ་ཎ་ན་མ་ཧཱ་ཙྪེ་ད་ཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཧྱེ་ཧི་བཛྲ་ཨ་ནི་ལ་མ་ཧཱ་བཻ་ག །ཨཱ་ན་ཡ་སརྦ་དུཥྚཱཾ་ཧཱུཾ་ཕཊ། ཨོཾ་ཨེ་ཧྱེ་ཧི་ཀུ་བེ་ར་ཡཀྵཱ་དྷི་བ་ཏ་ཡེ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧྱེ་ཧི་དྷི་དྷུ་བྷི་ན་བཛྲ་ཤཱུ་ལ་སྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧྱེ་ཧི་ན་མཿས་མནྟ་བུདྡྷཱ་ནཱཾ། ཨོཾ་ཕྲི་ཉི་བྷྱཿསྭཱ་ཧཱ། ཨོཾ་ཨེ་ཧྱེ་ཧི་ན་མཿས་མནྟ་བུདྡྷ་ན་མཿ། ཧེ་ཤོརྱ་པྲ་ཏོ་ཛྱོ་ཏི་ར་མཱ་ཡ་
སྭཱ་ཧཱ། ལྷ་རྣམས་སོ་སོའི་ཕྱག་རྒྱ་ཡིས། །མཆོད་ཡོན་ལ་སོགས་གཏོར་མ་ལ། །སྔགས་ནི་བདུན་ཙམ་ཅིག་དག་གིས། །ངེས་པར་འབུལ་དང་བསྟབས་པར་བྱའོ། །དེ་ནས་མགོན་པོ་མཉེས་པ་ཡི། །ཚིག་འདི་རྣམས་ཀྱང་བརྗོད་པར་བྱའོ། །ཧཱུཾ་ལྷ་དང་ལྷ་མིན་གནོད་སྦྱིན་ལྟོ་འཕྱེ་གྲུལ། །
ནམ་མཁའ་ལྡིང་དང་འདབ་{𰵍བཟངས་,བཟང་}ལུས་སྲུལ་པོ། །དྲི་ཟ་སྲིན་པོ་གདོན་གྱི་རིགས་རྣམས་དང་། །གང་དག་ས་ལ་གནས་པའི་རྫུ་འཕྲུལ་ཅན། །བུ་སྨད་བཅས་ཤིང་གཡོག་འཁོར་བཅས་པ་ཡང་། །བདག་གི་པུས་མོ་ས་ལ་བཙུགས་ནས་སུ། །ཐལ་མོ་སྦྱར་ནས་ཁྱེད་ལ་གསོལ་བ་འདེབས། །
གསན་ནས་ཕན་གདགས་ཕྱིར་ནི་འདིར་སྤྱོན་ཅིག །དབང་པོ་རྡོ་རྗེ་འཆང་བའི་ལྷ། །ལྷ་ཡི་རྒྱལ་པོ་གདོན་གྱི་བདག །ཤར་ཕྱོགས་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་པར་བགྱིད་ཅིང་ཕྱག་ཀྱང་འཚལ། །མེ་སྟེ་བསྲེག་བྱ་ཟ་བའི་ལྷ། །འབྱུང་པོའི་རྒྱལ་པོ་གདོན་གྱི་བདག །ཤར་ལྷོ་གདོན་
གྱི་ཚོགས་བཅས་ལ། །མཆོད་པར་བགྱིད་ཅིང་ཕྱག་ཀྱང་འཚལ། །གཤིན་རྗེ་བེ་ཅོན་འཆང་བའི་ལྷ། །མ་མོའི་རྒྱལ་པོ་གདོན་གྱི་བདག །ལྷོ་ཕྱོགས་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་པར་བགྱིད་ཅིང་ཕྱག་ཀྱང་འཚལ། །བདེན་དང་བྲལ་བ་ལེགས་ལྡན་ལྷ། །སྲིན་པོའི་རྒྱལ་པོ་གདོན་
གྱི་བདག །ལྷོ་ནུབ་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་པར་བགྱིད་ཅིང་ཕྱག་ཀྱང་འཚལ། །ཞགས་པ་ཐོགས་པ་ཆུ་ཡི་ལྷ། །ཀླུ་ཡི་རྒྱལ་པོ་གདོན་གྱི་བདག །ནུབ་ཕྱོགས་གདོན་གྱི་ཚོགས་བཅས་ལ། །མཆོད་པར་བགྱིད་ཅིང་ཕྱག་ཀྱང་འཚལ། །རླུང་སྟེ་སྲོག་ཀྱང་འཛིན་པའི་ལྷ། །
