འཁོར་ཀུན་དུ་བསྐོར། །རྡོ་རྗེ་ཕྱེད་པས་སྒོ་ཟུར་གྱི། །ཀ་བའི་རྩེ་མོ་དབུས་ནས་མཚན། །ཕྱི་རུ་དུར་ཁྲོད་འཇིགས་པ་ཡི། །ཁ་དོག་སྣ་ཚོགས་མཆོག་གི་མཆོག་སྣ་ཚོགས་ཤིང་རྣམས་ཉམས་དགས་མཚན། །ཁྱི་དང་ཅེ་སྤྱང་དག་གིས་འཇིགས། །ཁྭ་དང་འུག་པ་བྱ་རྒོད་ཁྲ། །ཝ་
དང་ཙི་ལི་ཀ་དེ་བཞིན། །ཀེང་རུས་མདུང་གིས་ཕུག་པ་དང། །འཕྱང་བ་དང་ནི་ཕྱེད་བསྲེགས་དང། །སྤྱི་བོ་ཀོ་སྐོ་པུས་མོ་དང། །ལག་པ་མགོ་བོ་རྣམས་ཀྱིས་བརྒྱན། །སྒྲ་སྒྲོགས་དེ་བཞིན་གླང་མགོ་དང། །རྣམ་བ་སྣ་ཚོགས་སྤྲུལ་གྱིས་དགའ། །སེང་གེ་སྟག་དང་ཕག་རྒོད་
དང༑ །དེ་བཞིན་དོམ་དང་ར་རྣམས་དང། །དཔའ་བོ་དང་ནི་རྣལ་འབྱོར་མ། །རྣམ་བ་སྣ་ཚོགས་འདུས་པ་མཆོག་ཐོད་པ་ཁ་ཊཱཾ་ག་ནི་བསྣམས། །གྲི་གུག་ཅད་ཏེའུ་དེ་བཞིན་ནོ། །གནོད་སྦྱིན་རོ་ལངས་སྲིན་པོ་རྣམས། །དེ་ལྟར་སྣ་ཚོགས་དུར་ཁྲོད་དོ། །དེ་རུ་ཉི་མའི་དཀྱིལ་འཁོར་བལྟ། །
ཨཱ་ལི་ཀཱ་ལི་རང་ས་བོན། །དེ་བཞིན་ཉིད་དང་ཡང་དག་མཐའ། །སྟོང་པ་ཉིད་དང་ཆོས་ཀྱི་དབྱིངས། །མཉམ་པ་ཉིད་དང་སོ་སོར་རྟོག་བྱ་བ་གྲུབ་པ་བསྒོམ་པ་དང། །ཤིན་ཏུ་རྣམ་པར་དག་པ་དམིགས། །རྣལ་འབྱོར་འདི་ནི་མཆོག་གི་མཆོག་རྡོ་རྗེ་སེམས་དཔའ་རང་རིག་བཅས་རིམ་
པ་འདི་ཡིས་རྣམ་པར་བསམ། །དཀར་པོ་སེར་པོ་དེ་བཞིན་དམར། །ལྗང་གུ་དང་ནི་ཞལ་རྣམ་བཞི། །རྡོ་རྗེ་དྲིལ་བུས་ཡང་དག་འཁྱུད། །མདུང་དང་དགྲ་སྟ་ཅང་ཏེའུ་དང། །གྲི་གུག་ཐོད་པ་ཁ་ཊཱཾ་ག་ཚངས་པའི་པགས་པ་ན་བཟར་མན་པས། །ཞགས་པ་ཚངས་པའི་མགོ་དེ་བཞིན། །
ཕྱག་བཅུ་གཉིས་ཀྱིས་བརྒྱན་བྱས་པ། །སེམས་དཔའི་སྐྱིལ་ཀྲུང་གིས་བཞུགས་པ། །ཐོད་པའི་ཅོད་པན་གྱིས་མི་བཟད། །རྒྱན་རྣམས་ཀུན་གྱིས་ཡང་དག་རྫོགས། །མཚན་དང་དཔེ་བྱད་རྣམས་ཀྱིས་བརྒྱན། །སྙིང་ག་ཡེ་ཤེས་དམ་ཚིག་སྟེ། །རྡོ་རྗེ་འཆང་བའི་ཅོད་པན་ཅན། །ཐབས་
