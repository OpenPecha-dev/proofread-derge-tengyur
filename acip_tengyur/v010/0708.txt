བ་དངོས་ཅི་འདྲ། །ཁོང་དུ་ཆུད་དབང་གྱུར་ལས་ནི། །ནང་གི་རེག་བྱ་རྙེད་པ་ཡིན། །ཚད་མེད་པ་ཡི་ཡོན་ཏན་གཏེར། །བརྗོད་དུ་མེད་པ་དེ་ཉིད་དོ། །རང་གི་བདེ་བ་ཀུན་དུ་གསོལ། །ཉམས་སུ་མྱོང་བ་ཞུམ་པ་ཡི། །དེ་ནི་བརླག་པར་རབ་ཏུ་བརྗོད། །ཡེ་ཤེས་སྐུ་ཡི་རང་ངོ་བོ། །དངོས་པོ་མིན་པ་
དེ་ལྟར་མིན། །བསྒོམ་དང་བྲལ་བའི་དངོས་པོ་ཡིན། །དབང་པོ་བདེ་བའི་བྱེ་བྲག་ནི། །གསོལ་བ་ལོངས་སྤྱོད་དུ་ནི་གྲགས། །སྲིད་དང་མྱ་ངན་འདས་ཡོངས་གསོལ། །ལོངས་སྤྱོད་བདེ་བ་ཡིས་ནི་དབང། །དགའ་གསུམ་ཐ་དད་རབ་ཏུ་གཞི། །གང་གི་ས་ལ་དེ་ནི་བརླག་དཔེ་མེད་
རང་བཞིན་དུ་ནི་གྲགས། །རྣལ་འབྱོར་ཡེ་ཤེས་རྙེད་པས་སོ། །དེ་ནི་རིན་ཆེན་རྡོ་རྗེ་རྩེར། །གྱེན་གྱི་ལམ་དུ་འགྲོ་བ་རུ། །སྙིང་ག་ཡི་ནི་ཕྱོགས་སུ་ཉེ། །བྱང་ཆུབ་ཀྱི་ནི་སེམས་ལ་དབང། །རང་གི་འདུག་པ་ནང་དུ་ཡང། །ལོངས་སྤྱོད་བདེ་བ་གྱ་ནོམ་པ། །འཛིན་པའི་གདུ་བུ་ཡིས་ནི་ས། །
སླེབས་པ་ཙམ་གྱིས་ནང་དུ་ནི། །བདེ་བའི་ངོ་བོ་ཉམས་སུ་མྱོང། །དེ་ནི་ནང་གི་རེག་བྱ་ཡིན། །བདེ་ཆེན་གྲགས་པ་རང་བཞིན་བརྗོད་དུ་མེད། །སྐྱེ་དང་གནས་དང་འགག་པ་རྣམ་པར་སྤངས། །རྣམ་པར་རྟོག་པ་ཙམ་གྱིས་བློ་རྙེད་མིན། །རྣམ་རྟོག་ཕྱི་དོན་བདེ་བ་མིན་ངོ་བོ། །དེ་བས་ཅི་མིང་གང་
ལ་རག་ལས་བསྒོམ། །བསྒོམ་པའི་དོན་གཞན་གྲུབ་པ་དེས་ལས་གཞན། །སྐད་ཅིག་རྟོག་པ་ཐམས་ཅད་དང་བྲལ་བས། །རྣམ་པར་རྟོག་མེད་ངོ་བོ་ཉིད་རང་རིག་གལ་ཏེ་ཚོར་བ་ཡིན་པ་དེ་ནི་རེག་རེག་ན་སྒྲ་ཡི་རྣམ་རྟོག་གཟུགས་སུ་འགྱུར། །སྒྲ་མེད་གཉིས་སུ་མེད་པས་ཀླགས་སྐབས་
མེད། །དངོས་བྲལ་དེ་ནི་དེ་ཉིད་དོན་དམ་པར། །འདི་ལྟ་དེ་ཉིད་ཡིན་གྱུར་ཀྱང། །དེ་བཞིན་དེ་ལ་བྱ་བ་ཡིན་པ་གཞུང། །དངོས་པོའི་བྱ་བས་ལྟ་བ་ལོག་མི་འགྱུར། །ལྟ་བའི་རིམ་པ་དེ་ཉིད་རིན་ཆེན་རྙེད། །ཇི་ལྟར་བུར་ཤིང་ས་ལ་ལུད་བཏབ་ན། །དེ་ལས་ཁུ་མང་ཤིང་རབ་ཆེ་བ་སྐྱེས། །དོན་གྱི་ཕྱིར་
