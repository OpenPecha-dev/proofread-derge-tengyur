རྣལ་འབྱོར་དང། ཚད་མ་དང། དག་པ་དང། མ་དག་པ་དང། ནང་གི་བདག་དང། སྐྱེས་བུ་དང། ཁྱབ་བདག་དང། སྲོག་དང། གང་ཟག་དང། རྣམ་པར་ཤེས་པ་དང། ཀུན་གཞི་དང། ཤེས་པ་པོ་དང། མཐོང་བ་པོ་དང། གཟུང་བ་དང། འཛིན་པ་དང། ཤེས་པ་དང། ཤེས་བྱ་དང་
ཤེས་པར་བྱ་བ་དང། ཤེད་ལས་སྐྱེས་དང། ཤེད་བུ་དང། བརྗོད་པར་བྱ་བ་དང། རྗོད་པར་བྱེད་པ་དང། བརྗོད་པར་བྱ་བ་མ་ཡིན་པར་རྟོག་སྟེ། གསང་བའི་བདག་པོ་བདག་གི་ཁྱད་པར་དེ་ལྟ་བུ་ལ་སོགས་པ་རྣལ་འབྱོར་བ་དག་གིས་ཐར་པ་འདོད་པས་སྔོན་བརྟགས་པ་ཡིན་ནོ་ཞེས་གསུངས་པ་ལྟ་
བུའོ༑ །དེ་ནི་འཇིག་རྟེན་དང་འཇིག་རྟེན་ལས་འདས་པའི་ཆོས་དེ་དག་གཉི་ག་ཡང་བྱེ་བྲག་མེད་པར་མཚན་ཉིད་མེད་པར་རྟོགས་པར་བྱེད་པའི་ཕྱིར། གསུམ་པ་མཚན་ཉིད་མེད་པར་ཞེས་བྱ་བ་སྨོས་ཏེ་མཚན་ཉིད་ནི་ཆོས་རྣམས་ཀྱི་རང་གི་ཐུན་མོང་མ་ཡིན་པའི་རང་བཞིན་སྲབ་དང། གྲང་བ་དང། རིག་
པ་དང། བེམས་པོ་ལ་སོགས་པ་སོ་སོ་རང་གི་མཚན་ཉིད་མ་ལུས་པའོ། །དེ་མེད་པར་དོན་དམ་པར་རིགས་པའི་མཐུ་མི་ཐེག་པས་ན་མེད་ཅེས་བརྗོད་ཀྱི། །རི་བོང་གི་རྭ་ལྟར་བུ་ནི་མ་ཡིན་ཏེ། མ་བརྟགས་ན་ཉམས་དགའ་བའི་ངོ་བོ་སྒྱུ་མ་ཙམ་དུ་དེ་དག་རང་རང་གི་མཚན་ཉིད་དུ་ཡོད་པ་ནི་མངོན་སུམ་དུ་གྱུར་
པའི་ཕྱིར་རོ། །དེ་ལྟར་མཚན་ཉིད་མེད་པ་དེ་དག་ཀྱང་མཚན་གཞི་ཐ་དད་པ་སྲབ་དང། ཚ་བ་ལ་སོགས་པ་ཐ་དད་པ་ལྟར། རང་རང་གི་ཡོད་པའི་མཚན་ཉིད་ལས་ལོག་པ་ཙམ་དུ་གནས་སོ་སྙམ་དུ་རྟོགས་ལ། དོན་དམ་པ་ནི་ཆོས་རྣམས་ཀྱི་སྤྱིའི་མཚན་ཉིད་ཡིན་པའི་ཕྱིར། དེ་ལྟར་ནི་མི་བསམ་
བར་སྟོན་པ་བཞི་པ་གཉིས་སུ་མེད་པའི་མཚན་ཉིད་དུ་ཞེས་བྱ་བ་སྨོས་ཏེ། གཟུགས་ཀྱི་སྟོང་པ་ཉིད་དང་ཚོར་བ་ལ་སོགས་པའི་སྟོང་བ་ཉིད་གང་དག་ཡིན་པ་དེ་དག་ནི་གཉིས་སུ་མེད་ཅིང་གཉིས་སུ་བྱར་མེད་དེ་སོ་སོར་མ་ཡིན་ཐ་མི་དག་པར་རྟོགས་པར་བྱེད་ཅེས་བྱ་བའི་དོན་ཏོ། །རྟོགས་པ་ཞེས་བཞི་
ཚར་གྱི་མཐར་ཆོས་ཐམས་ཅད་ཅེས་བྱ་བ་བསྟན་པ་ནི་རྟོགས་པ་དེ་དག་རེ་རེ་ཡང་ཀུན་ནས་ཉོན་མོངས་པ་དང། རྣམ་པར་བྱང་བའི་དངོས་པོ་མ་ལུས་པར་བསྡུས་པའི་ཕྱིར་རོ། །དེ་བཞིན་དུ་སྐད་ཅིག་མ་གཅིག་པའི་མཚན་ཉིད་ཅེས་བྱ་བ་སྨོས་པ་ཡང་རྟོགས་པ་དེ་དག་བཞི་ཆར་ཡང་ཁྱད་པར་
