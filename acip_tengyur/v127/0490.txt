ཕྱི་མ་ལ་མ་བསྒོ་བར་འགྲོ་བ་མེད་པ་དང། ཆོས་གོས་ཅི་དགོས་པ་ཚོལ་དུ་རུང་བ་དང། སྟོན་ཟླ་ཐ་ཆུངས་ནས། དཔྱིད་ཟླ་ར་བའི་བར་དུ་དབྱར་གནས་པར་ཁས་བླངས་པའི་དགེ་སློང་གི་དགེ་འདུན་རྣམས་ཀྱིས་རྙེད་པ་མ་བགོས་ཤིང། གཞག་པ་དེ་ཡང་སྲ་བརྐྱང་
བཏིང་བ་རྣམས་ཀྱིས་མ་དད་ན་གཞན་ལ་མི་སྦྱིན་ནོ། །དེ་ནི་སྲ་བརྐྱང་གི་བྱ་བ་ཞེས་བྱའོ། །ཆུ་ཚགས་ཀྱི་བྱ་བ་གང་ཞེ་ན། དགེ་སློང་ཆུ་ཚགས་མེད་པར་ལམ་དུ་འགྲོ་བར་མི་བྱའོ། །གལ་ཏེ་ཆུ་རྒྱུན་དུ་འབབ་པ་དེ་ཡང། བཏུང་བར་འགྱུར་ཏེ། ཆུ་གཞན་དང་མི་འདྲེས་པ་ཞིག་གམ་
ཁྲོན་པ་རུང་བ་ཞིག་གམ། རྒྱང་གྲགས་ཕྱེད་ཕྱེད་དུ་གཙུག་ལག་ཁང་རྒྱུན་མ་ཆད་པར་ཡོད་ན་ནི་མ་གཏོགས་སོ། །དེ་ནི་ཆུ་ཚགས་ཀྱི་བྱ་བ་ཞེས་བྱའོ། །འཆད་པའི་བྱ་བ་གང་ཞེ་ན། བཅོམ་ལྡན་འདས་ཀྱིས་འཆགས་གཉིས་རྗེས་སུ་གནང་སྟེ། བསམ་གཏན་ལ་
ཞུགས་པ་དང། ཁ་ཏོན་ལ་ཞུགས་པའོ། །བསམ་གཏན་ལ་ཞུགས་པའི་འཆག་ས་ནི་ཁྲུ་བཅོ་བརྒྱད་དོ། །ཁ་ཏོན་ལ་ཞུགས་པའི་འཆགས་ས་ནི་ཁྲུ་བཅུ་གཉིས་ཏེ། དེ་ནི་འཆག་སའི་བྱ་བ་ཞེས་བྱའོ་འཆག་སའི་བྱ་བ་གང་ཞེ་ན། དགེ་སློང་འཆག་སར་འདོད་ན་
དེས་དྲག་ཏུ་ཡང་མ་ཡིན། དལ་ནས་ཀྱང་མ་ཡིན་པར་བཅག་པར་བྱ། ཤེས་བཞིན་དུ་བཅག་པར་བྱ། དྲན་པ་ཉེ་བར་གཞག་པ་དང། དབང་པོ་ནང་དུ་གཞུག་པ་དང། ཡིད་ཕྱིར་མི་འབྱུང་བར་བྱ། རྡེག་མི་འཆོས་པར་བྱ། ཕན་ཚུན་མི་འཁོར་བར་བྱ། དྲང་
པོར་བཅག་པར་བྱ། གལ་ཏེ་དྲང་པོར་བཅག་མི་ནུས་ན། སྲད་བུ་བྲེ་བར་བྱ་ཞིང་སྦུབས་སུ་བཅུག་ནས། དེ་ལ་བཟུང་ཞིང་བཅག་པར་བྱ་སྟེ། འདི་ནི་བཅག་པའི་བྱ་བ་ཞེས་བྱའོ། །འཕྱན་གྱི་བྱ་བ་གང་ཞེ་ན། སྟོན་པའི་མདུན་དུ་མ་ཡིན། མཆོད་རྟེན་གྱི་དྲུང་དུ་མ་
ཡིན༑ སློབ་དཔོན་དང་མཁན་པོའི་དྲུང་དུ་མ་ཡིན། བླ་མའི་དྲུང་དུ་མ་ཡིན། བླ་མའི་གནས་ལྟ་བུའི་དྲུང་དུ་མ་ཡིན། གྲོང་གི་ནང་དུ་སྐྱེ་བོ་མང་པོ་འདུ་བའི་ནང་དུ་མ་ཡིན། ཕྱི་ཕྱིར་ཞིང་གཏང་བར་མི་བྱ། གལ་ཏེ་ཕྱིར་ཞིང་གཏོང་བ་ཞིག་ཏུ་གྱུར་ན། གནས་གུང་དུ་སོང་
