༄༅། །སྟེ༑ རང་གི་དགེ་བའི་རྩ་བའི་སྟོབས་བསྐྱེད་པས་སམ། གཞན་གྱིས་བསྒྲུབས་པས་ཀྱང་རུང་སྟེ། སེམས་ཀྱི་རྒྱུ་བ་གསལ་བ་ནི་དགེ་བའི་སེམས་ཡོད་པའི་འཆི་བར་བལྟའོ། །རང་གི་མི་དགེ་བའི་སེམས་བསྐྱེད་པས་སམ། གཞན་གྱིས་བསྒྲུབས་པས་ཀྱང་རུང་སྟེ་
སེམས་རྒྱུ་བ་གསལ་བ་ནི་མི་དགེ་བའི་སེམས་ཡོད་པའི་འཆི་བར་བལྟའོ། །སེམས་ཀྱི་རྒྱུ་བ་གསལ་ཡང་རུང་མི་གསལ་ཡང་རུང་རྐྱེན་དེ་གཉི་ག་མ་ཚང་བའི་ཕྱིར་མངོན་པར་འདུ་བྱེད་མི་ནུས་པ་ཡང་རུང་སྟེ། ལུང་དུ་མ་བསྟན་པའི་སེམས་ཡོད་པའི་འཆི་བར་བལྟའོ། །དགེ་བ་བྱེད་
པའི་ལུས་ནི་ཐོབ་མ་ཉིད་སྨད་གྲང་བར་འགྱུར། མི་དགེ་བ་བྱེད་པའི་ལུས་ནི་སྟོད་གྲང་བར་འགྱུར་རོ། །དེ་ལ་མི་དགེ་བ་བྱེད་པའི་བར་མ་དོའི་སྲིད་པ་མངོན་པར་འགྲུབ་པ་ནི། དཔེར་ན་ཕྱར་བ་ནག་པོའམ། མཚན་མོ་མུན་པ་མུན་ནག་ཏུ་སྣང་བ་ལྟ་བུའོ། །དགེ་བ་བྱེད་པ་ནི་དཔེར་ན་
སྣམ་བུ་དཀར་པོའམ། ཟླ་བའི་འོད་དང་བཅས་པའི་མཚན་མོར་སྣང་བ་ལྟ་བུའོ། །བར་མ་དོའི་སྲིད་པ་ནི་འདོད་པའི་ཁམས་དང་གཟུགས་ཀྱི་ཁམས་སུ་སྐྱེ་བ་ནའོ། །གཟུགས་མེད་པ་ནས་ནི་འཆི་འཕོ་བ་ནའོ། །དེ་ནི་ཡིད་ལས་བྱུང་བ་དང་དྲི་ཟ་ཞེས་ཀྱང་བྱའོ། །དེ་ལྟར་ཐོགས་ན་
ཞིག་བདུན་གནས་ཏེ། དེ་ཚུན་ཆད་ཀྱིས་འཆི་འཕོའོ། །ཇི་སྟེ་ན་ལྡོག་གོ། །དེ་ན་གནས་པ་ཡང་ལས་སོག་གོ། །སྐལ་པ་མཉམ་པའི་སེམས་ཅན་རྣམས་ཀྱང་མཐོང་ངོ། །གང་དུ་སྐྱེ་བར་འགྱུར་བ་དེའི་བྱད་གཟུགས་སོ། །འགྲོ་བ་ལ་ཐགས་ཐོགས་མེད་དོ། །རྫུ་འཕྲུལ་
དང་ལྡན་པ་བཞིན་དུ་མྱུར་དུ་འགྲོ་ཞིང་སྐྱེ་བའི་གནས་སུ་ཐོགས་སོ། །སྲང་མདའི་མཐོན་དམན་གྱི་ཚུལ་དུ་འཆི་འཕོ་ཞིང་ཉིད་མཚམས་སྦྱོར་རོ། །བར་མ་དོའི་སྲིད་པ་ན་གནས་པ་ནི་སྐྱེ་བའི་གནས་ལ་འདོད་ཆགས་བསྐྱེད་དེ། །དེ་ལས་གཞན་པའི་ཉོན་མོངས་པ་ནི་རྐྱེན་དུ་འགྱུར་རོ། །འདོད་
ཆགས་དང་བཅས་པར་བར་མ་དོའི་སྲིད་པ་འགག་ཅིང་ནུར་ནུར་པོ་རྣམ་པར་ཤེས་པ་དང་བཅས་པ་འབྱུང་སྟེ། དེ་ནི་རྣམ་པར་སྨིན་པའོ། །དེའི་འོག་ཏུ་རྟེན་ཅིང་འབྲེལ་པར་འབྱུང་བ་ལས་ཇི་སྐད་འབྱུང་བ་བཞིན། སྐྱེ་གནས་བཞི་པོ་སྒོ་ང་ལས་སྐྱེ་བ་དང། མངལ་ནས་སྐྱེ་བ་དང། དྲོད་གཤེར་ལས་
