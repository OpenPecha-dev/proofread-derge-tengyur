ལུས་སྙིང་པོར་ནི། །གྱུར་པས་མཐུ་ཞེས་བཤད་པ་དག །དོན་དང་མཐུན་པར་བཏགས་པ་ཡིན། །གཅིས་པ་མང་དུ་ཡོད་ན་ཡང་། །ཐ་སྙད་གཙོ་བོ་ཉིད་ཀྱིས་ཡིན། །དེ་བས་གོ་ནི་བཟློག་པའི་ཕྱིར། །དེ་ནི་རོ་ལ་{𰶉སྩོགས་,སོགས་}པ་ལ། །སྲིད་དུ་ཟིན་ཀྱང་མ་བཤད་དེ། །དེ་བས་ལྕི་བ་
ལ་སོགས་ཡིན། །གཞན་དག་ཚ་དང་བསིལ་ཉིད་གཉིས། །སྟོབས་ཡིན་ཞེས་ཀྱང་སྟོན་པར་བྱེད། །སྨན་ནི་སྣ་ཚོགས་བདག་ཉིད་ཅན། །ཉི་མ་ཟླ་བའི་སྟོབས་ཅན་ཡིན། །འགྲོ་འདི་གསལ་དང་མི་གསལ་ལས། །ནམ་ཡང་འདའ་བར་མི་འགྱུར་རོ། །དེ་ལ་ཚ་བ་མགོ་
འཁོར་སྐོམ། །རྨྱ་རྡུལ་ཚ་ཞིང་མྱུར་འཇུ་བྱེད། །རླུང་དང་བད་ཀན་ཞི་བར་བྱེད། །བསིལ་བ་དག་ནི་ཚིམ་བྱེད་ཅིང་། །འཚོ་དང་རེངས་པར་བྱེད་པ་དང་། །ཁྲག་དང་མཁྲིས་པ་རབ་དང་བྱེད། །ཕོ་བའི་མེ་དང་ཕྲད་པ་ལས། །གང་ཞིག་རོ་གཞན་བྱུང་གྱུར་པའི། །རོ་རྣམས་
ཀྱི་ནི་མཐར་ཞུ་བ། །དེ་ནི་ཞུ་བ་ཞེས་བཤད་དོ། །མངར་དང་ལན་ཚྭ་ཞུ་རྗེས་མངར། །སྐྱུར་བའི་ཞུ་རྗེས་སྐྱུར་བ་ཡིན། །ཁ་དང་བསྐ་དང་ཚབ་རྣམས། །ཞུ་རྗེས་ཕལ་ཆེར་ཁ་བ་ཡིན། །རོ་དང་ཞུ་རྗེས་འབྲས་བུ་མཉམ། །དེ་ལ་སྨན་ནི་བཟང་དང་ངན། །ལ་ལ་རོ་ཡིས་ལས་བྱེད་
ལ། །གཞན་དག་འཇུ་བར་བྱེད་པ་ཡིན། །ཡོན་ཏན་གཞན་དང་ནུས་པ་དང་། །མཐུ་ཡིས་ལ་ལ་བྱེད་པ་ཡིན། །སྨན་གྱི་རོ་{𰶊སྩོགས་,སོགས་}གང་དང་གང་། །སྟོབས་ལྡན་ཉིད་ཀྱིས་གནས་བྱེད་དེ། ཤིན་ཏུ་ཟིལ་མནན་དེ་དང་དེ། །སྒྲུབ་བྱེད་རྒྱུ་ནི་ཉིད་དུ་གནས། །ཡོན་
ཏན་མི་འཕྲོད་སྦྱར་བ་ནི། །མང་པོས་ཉུང་ངུ་ཟིལ་གྱིས་གནོན། །རོ་དང་འཇུ་བའི་མཐུ་དག་ནི། །དེ་ཡི་སྟོབས་ཀྱི་ཟིལ་གྱིས་གནོན། །རོ་{𰶋སྩོགས་,སོགས་}དེ་ལྟར་སྟོབས་ནི་མཉམ། །མ་བཅོས་པ་ཡི་སྟོབས་དང་ལྡན། །རོ་{𰶌སྩོགས་,སོགས་}མཉམ་པའི་ལས་གང་ནི། །ཁྱད་འཕགས་
དེ་ནི་མཐུ་ལས་སྐྱེས། །དན་ཏི་དང་ནི་ཙི་ཏྲ་ཀ། །རོ་{𰶍སྩོགས་,སོགས་}མཚུངས་ཀྱང་དན་ཏི་འཁྲུ། །ཤིང་མངར་རྒུན་འབྲུམ་འདྲ་ཡང་འཁྲུ། །འོ་མ་མར་འདྲ་མར་དྲོད་སྐྱེད། །དེ་ལྟར་སྨན་ལ་སོགས་པ་ཡི། །སྤྱི་ཡི་ལས་ཏེ་དེ་ཡང་ནི། །སྣ་ཚོགས་རྐྱེན་གྱི་དབྱེ་བ་ཡིས། །
