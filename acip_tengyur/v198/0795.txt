མཁྲིས་པ་སྒྲུབ་བྱེད་ཅེས་བྱ་བའི་གནས་ནི་སྙིང་ལ་བརྟེན་ཅིང་གནས་སོ། །དེའི་ལས་ནི་སེམས་དཔའི་བློ་དང་། ཡིད་གཞུངས་པ་དང་། སྙེམས་པའི་ང་རྒྱལ་དུ་སེམས་པ་ལ་སོགས་པས་སྙིང་ལ་བསམས་པའི་དོན་མཁྲིས་པ་དེ་ཡོད་དམ་སྒྲུབ་པའི་གྲོགས་དོན་སྒྲུབ་ཅེས་བྱའོ། །མཁྲིས་
པ་མཐོང་བྱེད་ཅེས་བྱ་བའི་གནས་ནི་མིག་ལ་བརྟེན་ནོ། །དེའི་ལས་ནི་གཟུགས་རྣམས་མཐོང་བར་བྱེད་པ་ཡིན་ནོ། །མཁྲིས་པ་མདོག་གསལ་བྱེད་ཅེས་བྱ་བའི་གནས་ནི་མིའི་པགས་པ་ལ་བརྟེན་ཏེ་གནས་སོ། །ལས་ནི་པགས་པའི་མདོག་འཇམ་ཞིང་བཟང་ལ་མདོག་གསལ་བར་བྱེད་
