ལྷ་མཚམས་ཀྱིས་ནི་མཆོག་རབ་སྟེར། །ཡང་དག་སྐར་མ་དང་མཐུན་པས། །ཇི་སྐད་བཤད་པ་ཐམས་ཅད་བྱང། །གསུམ་པ་བདུན་པ་དྲུག་པ་དང། །བཅུད་དང་བཅུ་གཅིག་ཟླ་བ་རྣམས། །དང་པོ་ལྔ་པ་གསུམ་པ་དང། །ཐ་མ་ཡང་ནི་རྣམ་པར་སྤང། །དགུ་དང་བརྒྱད་དང་བཅུ་གཉིས་གཉིས། །བཞི་
དང་ལྔ་དང་བརྵཾ་ཉིད། །རྣམ་པ་གཞན་དུ་བྱེད་པ་ལ། །བཟློག་པའི་འབྲས་བུ་འབྱུང་པར་འགྱུར། །སེམས་ཅན་དམན་པའི་དབང་མཛད་ནས། །ཟླ་བ་ལ་སོགས་བཟང་པ་འདིའོ། །མཆོག་གི་དབང་དུ་མཛད་ནས་ནི། །ངེས་པ་ཉིད་ནི་རིགས་མ་ཡིན། །དེ་ལྟར་ཡང༑ །དབྱར་ནི་དངོས་
