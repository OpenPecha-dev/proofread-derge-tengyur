།རྒྱ་གར་སྐད་དུ། ཤྲཱི་བུདྡྷ་ཀ་པཱ་ལི་ཏནྟྲ་སྱ་པཉྩི་ཀཱ་ཛྙཱ་ན་བ་ཏི་ནཱ་མ། བོད་སྐད་དུ། དཔལ་སངས་རྒྱས་ཐོད་པའི་རྒྱུད་ཀྱི་དཀའ་འགྲེལ་ཡེ་ཤེས་ལྡན་པ་ཞེས་བྱ་བ། དཔལ་ཧེ་ཏུ་ཀ་ལ་ཕྱག་འཚལ་ལོ། །བསྒྲུབ་བྱ་ཕྲ་གཟུགས་མཆོག་ཏུ་བདེ་འབྱུང་དགའ་
བ་བཅུ་དྲུག་སྟེར་བྱེད་དམ། །མདུན་ནང་ཐིག་ལེའི་སྟེང་སྒྲ་ཡོངས་ཤེས་སེམས་དངོས་རང་བཞིན་ཏེ། རང་གི་དགའ་བའི་རྟེན་གཟུགས་ཡོན་ཏན་སྡུད་པ་བྱང་ཆུབ་ཚོགས་སྙིང་པོ། །དངོས་དང་དངོས་མེད་རང་བཞིན་སྲིད་པའི་འཇིགས་སེལ་སྣ་ཚོགས་སྡེ་ཕྱག་འཚལ། །སྣ་ཚོགས་སྡེ་
མས་བཤད་པ་ཡི། །སངས་རྒྱས་ཐོད་པ་ཡི་ནི་རྒྱུད། །ཇི་ལྟར་བླ་མས་བཤད་པ་དང། །ཇི་ལྟར་རྒྱུད་གཞན་དག་མཐོང་བ། །དེ་ལྟར་ཅུང་ཟད་བདག་གིས་བཤད། །དེ་ལ་རེ་ཞིག་རྗོད་པ་དང། བརྗོད་པར་བྱ་བ་དང། དགོས་པ་དང། དགོས་པའི་ཡང་དགོས་པ་སྟོན་ཏེ། རྗོད་
པར་བྱེད་པ་ནི་རྒྱུད་དོ། །བརྗོད་པར་བྱ་བ་ནི་རྒྱུད་ཀྱི་དོན་ཏོ། །དགོས་པ་ནི་ལྷའི་རྣལ་འབྱོར་མངོན་དུ་བྱེད་པའོ། །དགོས་པའི་ཡང་དགོས་པ་ནི་སེམས་ཅན་གྱི་དོན་གྱིས་བྱ་བ་བྱེད་པའོ། །གལ་ཏེ་ཡང་བཅོམ་ལྡན་འདས་ཉིད་སྟོན་པ་པོ་ལ་སོགས་པའི་གཟུགས་སུ་བཞུགས་མོད་ཀྱི་
འོན་ཀྱང་བདག་ཉིད་བཅོམ་ལྡན་འདས་མའི་གཟུགས་སུ་ལོགས་སུ་བསྟན་ཏོ། དེ་ནས་བཅོམ་ལྡན་འདས་མ་ལས་ཕྱག་ན་རྡོ་རྗེས་ཐོས་སོ། །རྒྱུད་སྡུད་པར་བཞེད་ནས་དེར་འདི་སྐད་ཅེས་བྱ་བ་གསུངས་པ་ནི། སྡུད་པར་བྱེད་པའི་ཚིག་གོ། །ཐོས་པ་ཞེས་བྱ་བ་ནི་རྣམ་པ་འདི་ལྟར་བཅོམ་
ལྡན་འདས་མ་ལས་བདག་གིས་ཐོས་ཞེས་སྦྱར་རོ། །འདི་སྐད་བདག་གིས་ཞེས་པ་ལ། ཨེ་ནི་སར་གསུངས་ཏེ། དེ་ནི་སྤྱན་མའི་རང་བཞིན་ནོ། །ཝཾ་ནི་ཆུར་གསུངས་ཏེ། དེ་ནི་མཱ་མ་ཀཱིའི་རང་བཞིན་ནོ། །མ་ནི་མེར་གསུངས་ཏེ། དེ་ནི་གོས་དཀར་མོའི་རང་བཞིན་ནོ། །ཡཱ་ནི་རླུང་
དུ་གསུངས་ཏེ། དེ་ནི་སྒྲོལ་མའི་རང་བཞིན་ནོ། །ཡི་གེ་བཞི་ནི་ཕྱག་རྒྱ་བཞིར་ཡང་དག་པར་བཤད་དོ། །སྟོང་པ་ཉིད་ཀྱི་རང་བཞིན་གྱིས་འགྲོ་བ་རྒྱས་འདེབས་པས་ན་ཕྱག་རྒྱའོ། །བདག་གིས་ཐོས་ཞེས་བྱ་བ་ལ། རྣ་བའི་རྣམ་པར་ཤེས་པས་ཐོས་པ་ནི་མ་ཡིན་ནོ། །འོ་ན་ཅི་ཞེ་
