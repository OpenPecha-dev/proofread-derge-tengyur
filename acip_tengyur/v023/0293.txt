༄༅། །གཉིས་པ་ཉིད་དུ་ངེས་པར་བསྟན་པ། །གནས་ནི་ཛཱ་ལནྡྷ་རར་བཤད། །དེ་བཞིན་ཨོ་ཌི་ཡཱ་ན་ཉིད། །གནས་ནི་པཱུརྞྞ་གི་རི་ཉིད། །དེ་བཞིན་ཀཱ་མ་རཱུ་པ་ཡང། །ཉེ་གནས་མཱ་ལ་ཝ་རུ་བརྗོད། །སིནྡྷུ་ན་ག་ར་ཉིད་དང། །ཞིང་ནི་མུ་མུ་ནི་རུ་བཤད། །ཞིང་ནི་བྱེད་པའི་བྲང་ཉིད་དོ། །
དེ་ཝཱི་ཀོ་ཊ་དེ་བཞིན་ཞིང། །ཞིང་ནི་ལྕགས་པའི་བྲང་ཉིད་དོ། །ཉེ་ཞིང་ཀུ་ལུ་ཏཱ་རུ་བརྗོད། །དེ་བཞིན་ཨརྦུ་ད་ཡང་ངོ། །གོ་དཱ་ཝ་རཱི་གངས་ཀྱི་རི། །ཉེ་བའི་ཞིང་ནི་མདོར་བསྡུས་པའོ། །འདུན་རྟོག་ཧ་རི་ཀེ་ལ་དང། །ལན་ཚྭའི་རྒྱ་མཚོའི་ནང་སོན་པའོ། །ལམྦཱ་ཀ་དང་ལཱཉྩི་ཀ། །དེ་བཞིན་ཡུལ་འཁོར་
བཟང་པོའོ། །ཉེ་བའི་འདུན་རྟོག་ཀ་ལིངྒ། །འདི་རྣམས་ལྡན་པ་གླིང་ཉིད་དོ། །ཉེ་བའི་འདུན་རྟོག་ཀངྒ་ཎ། །མདོར་བསྡུས་པས་ནི་མངོན་པར་བརྗོད། །འཐུང་གཅོད་གྲོང་གི་མཐའ་དང་ནི། །ན་ག་ར་ཡང་འཐུང་གཅོད་དོ། །ཙ་རི་ཏྲ་དང་ཀོ་ས་ལ། །འབིགས་བྱེད་གཞོན་ནུ་མ་ཡི་གྲོང། །ཉེ་བའི་
འཐུང་གཅོད་དེ་དང་ཉེ། །རྡོ་རྗེ་སྙིང་པོ་སྙིང་རྗེ་ཆེ། །དུར་ཁྲོད་ཡི་དགས་འདུས་པ་དང། །རྒྱ་མཚོའི་འགྲམ་ཡང་དུར་ཁྲོད་དོ། །སྐྱེད་ཚལ་རྫིང་བུའི་འགྲམ་ཉིད་ནི། །ཉེ་བའི་དུར་ཁྲོད་ངེས་པར་བརྗོད། །ཅེས་གསུངས་སོ། །དེ་ལ་པོལླ་གི་རི་ནི་པཱུརྞྞ་གི་རིའོ། །དེ་ཡང་ཀོལླ་གི་རི་དང་
པུལླཱི་ར་མ་ལ་ཡ་དང་མཱ་ལ་ཝ་དང། རྡོ་རྗེའི་གནས་ཞེས་པ་རྣམས་རྣམ་གྲངས་སོ། །ཁ་སྦྱོར་ཐིག་ལེ་ལས། གནས་ནི་ཀོལླ་གི་རི་ཉིད། །དེ་བཞིན་ཨརྦུ་ད་ཡང་ངོ། །ཞེས་བཤད་པ་མ་ཡིན་ནམ། ཨརྦུ་ད་གནས་སུ་ཡང་བསྟན་པ་འདིར་ཉེ་བའི་ཞིང་དུ་བརྗོད། །དེ་བཞིན་དུ་ཀཱ་མ་རཱུ་བ་འདིར་
གནས་ཉིད་དུ་བསྟན། ཕྱག་རྒྱ་ཆེན་པོ་ཐིག་ལེའི་རྒྱུད་དུ་ཡང། གནས་ཛཱ་ལནྡྷ་ར་དང་ཨོ་ཌི་ཡ་ན་དང་པཱུརྞྞ་གི་རི་དང་ཀཱ་མ་རཱུ་པ་ཡང་ངོ་ཞེས་སོ། །སམྤུ་ཊར་ནི། དེ་བཞིན་ཀཱ་མ་རཱུ་པར་བརྗོད། །ཨོ་ཏྲེ་ཞིང་ཉིད་དུ་ཡང་བརྗོད། །གནས་ཉིད་ལ་སོགས་པར་རྣམ་པར་གཞག་པ་ཡང་ཐིག་ལེ་དང། རྡོ་རྗེ་
མཁའ་འགྲོ་ཕལ་ཆེར་མཚུངས་པ་ཉིད་དོ། །དེ་ཅི་ལྟར་མི་འགལ་ཞེ་ན། གནས་དང་ཉེ་བའི་ཞིང་དག་ཚ་གྲང་དང་དངོས་པོ་དང་དངོས་མེད་བཞིན་དུ་ཕན་ཚུན་འགལ་བ་ནི་མ་ཡིན་ཏེ། གཅིག་ལ་ཡང་མིང་སྣ་ཚོགས་ཇི་ལྟར་མི་འགལ་ཞེས་སོ། །འདིར་ན་ག་ར་དང། པཱ་ཊ་ལི་པུ་ཊ་དང། མཱ་ལ་ཝ་
