༄༅། །སེམས་ཞེས་གང་ལ་བགྱི། །ཞེས་འདིར་སྨྲས་པའི་ལན་བསྟན་པའི་ཕྱིར། ཐོག་མ་ཐ་མ་མེད་ཞི་བ། །ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། དངོས་དང་དངོས་མེད་མི་ཟད་ཅེས་བྱ་བ་ནི་དངོས་པོ་དང་དངོས་པོ་མེད་པའི་རྫས་དང་བྲལ་བའོ། །སེམས་གཙོ་ཞེས་བྱ་བ་ནི་མིག་ཡོར་གྱི་
ཚུལ་གྱིས་ཁྱབ་པའོ། །སྟོང་ཉིད་སྙིང་རྗེ་མི་ཕྱེད་པ། །ཞེས་བྱ་བ་ནི་ཆོས་ཀྱི་དེ་ཁོ་ན་ཉིད་ཀྱི་བདག་ཉིད་ཅན་ནོ། །དེ་བཞིན་རིག་པའི་སྐྱེས་བུ་ཅི། །ཞས་བྱ་བའི་ལན་བསྟན་པར་བྱ་བའི་ཕྱིར་སྐུ་གསུང་ཐུགས་ཀྱི་རྡོ་རྗེ་ཡེ་ཤེས་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་ཏེ། མི་ཕྱེད་པ་ནི་ཞེས་བྱ་བ་ལ་ཐབས་
དང་ཤེས་རབ་ནི་སྐུ་གསུང་ཐུགས་རྡོ་རྗེ་ཉིད་ཀྱི་གཉིས་སུ་མེད་པའི་ཚུལ་དུ་ངོ་བོ་ཉིད་གཅིག་པས་མི་ཕྱེད་པས་སོ། །ཕྱེ་བའི་དངོས་ཞེས་བྱ་བ་ནི་དེ་དག་ཉིད་རྣམ་པར་དབྱེ་བའི་དབྱེ་བས་ཕྱེ་བས་སོ། །བཛྲྀ་དྷྲཀ་ཅེས་གང་ལ་བགྱི། །ཞེས་བྱ་བ་ལ་སོགས་པ་དྲིས་པ་ལྔ་ལ་ལན་ལྔའི་ཚིགས་སུ་བཅད་པ་
བསྟན་པ་ནི་རྒྱུ་ལྔ་དང་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་གསུངས་པ་ཡིན་ཏེ། རྒྱུ་ལྔ་ནི་རྒྱུར་གྱུར་པའོ། །འབྲས་བུ་ལྔ་ནི་འབྲས་བུར་གྱུར་པ་སྟེ། རྡོ་རྗེའི་རྩེ་ལྔ་ནི་སངས་རྒྱས་ལྔའི་བདག་ཉིད་ཅན་ཡིན་ལ། དེ་ཉིད་སྟེང་དང་འོག་གི་ཆས་རྒྱུ་དང་འབྲས་བུ་རྣམ་པར་དག་པའི་གཟུགས་ཅན་ནི་རྡོ་རྗེ་ཞེས་བྱ་
བ་ལ་དེ་འཆང་བར་མཛད་པས་ནི་རྡོ་རྗེ་འཛིན་པའོ། །ཆོས་ཀྱི་དབྱིངས་རྣམ་པར་དག་པ་ནི་རྣམ་པར་ཤེས་པ་སྟེ། དེ་ཉིད་ནི་ཡིད་ཅེས་ཀྱང་བྱའོ། །ཡོད་དང་མེད་དང་དབུས་ལས་འདས། །ཞེས་བྱ་བ་ནི་ཡོད་པ་དང་མེད་པའོ་ཞེས་བྱ་བའི་རྣམ་པར་རྟོག་པ་དང་བྲལ་ཞིང་འབྱུང་བ་དང་འབྱུང་བ་ལས་གྱུར་པའི་
མེ་ལོང་རྣམ་པར་དག་པའི་གཟུགས་ཅན་ཡིན་ལ། དེ་ཉིད་ནི་སེམས་ཅན་ཀུན་གྱི་ལུས་ཡིན་ཏེ། ཀུན་གཞིའི་རྣམ་པར་ཤེས་པའོ། །དེ་ནི་འདི་ལྟར་ཛིན་ཛིཀ་ཅེས་བྱའོ། །ཛིན་ཞེས་བྱ་བའི་སྒྲ་ཉིད་ཀྱི་དོན་བསྟན་པར་བྱ་བའི་ཕྱིར། སྐྱེ་ལས་རྒྱལ་བ་ཞེས་བྱ་བར་གསུངས་ཏེ། ཛི་ནའི་སྒྲ་ནི་སྐྱེ་བ་ལ་
བྱ་ལ་དེ་ལས་རྒྱལ་བར་མཛད་པས་ན་ཛི་ན་ཛིཀ་ཅེས་བྱ་སྟེ། བརྟགས་པའི་བདག་ཉིད་དུ་མི་སྐྱེའོ་ཞེས་བྱ་བའི་དོན་ཏོ། །དོན་རྣམས་ཐམས་ཅད་ཅེས་བྱ་བ་ནི་དོན་ཐམས་ཅད་ལ་གང་མཉམ་པ་ཉིད་དེ་ལས་ཀུན་དུ་འབྱུང་བ་ནི་དེ་རྣམ་པར་དག་པའི་སེམས་ཉིད་ནི་རིན་པོ་ཆེ་དེ་ཡིན་ལ། དེ་ཡང་ཚོར་བ་པོས་
