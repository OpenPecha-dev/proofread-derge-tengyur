༄༅། །རིགས་པར་ཡང། ཀུ་ཤའི་ཆུན་པོས་སྣོད་དང་པོར། །ལས་ཀུན་བྱེད་ལས་ཆུ་ཡིས་ནི། །བསང་གཏོར་ལན་གསུམ་དབུལ་བར་བྱ། །དེ་ཉིད་དུ་ནི་ཞལ་བསིལ་ཏེ། །མཛུབ་མོས་མེ་ཏོག་བླངས་ནས་ནི། །ཁུ་ཚུར་འོག་ཏུ་ཁ་བསྟན་ནས། །མཐེའུ་ཆུང་སྲིན་ལག་གུང་མོ་དང། །མཛུབ་མོ་རིམ་
གྱིས་དགྲོལ་ཞིང་དབུལ། །གཉིས་པའི་དུང་གི་ཆུ་ཡིས་ནི། །མཆོད་ཡོན་བཟེད་ཞལ་གཉིས་པ་རུ། །རྡོ་རྗེའི་སྙིམ་པ་ཁ་ཕྱེ་བས། །དབུལ་ཞིང་ཞབས་བསིལ་གསུམ་པར་ནི། །ཁུ་ཚུར་བཅས་ནས་མཐེ་བ་མཛུབ་ཀྱི། །སྐམ་པས་མེ་ཏོག་བླངས་ནས་སུ། །སངས་རྒྱས་རྣམས་ཀྱི་སྤྱན་སྔར་ནི། །ལན་གསུམ་བསྐོར་
བའི་སོར་མོ་ནི། །མཛུབ་མོ་གུང་མོ་མིང་མེད་དང། །མཐེའུ་ཆུང་རིམ་གྱིས་དཀྲོལ་བས་སོ། །ཞེས་བྱ་བའི་གདམས་པ་ཁྱད་པར་ཁོ་བོའི་བླ་མ་རྣམས་ཀྱིས་གསུངས་སོ། །ཕྱོགས་གཞན་ནི་དཔལ་སམྤུ་ཊའི་རྒྱུད་ལས། སྔོན་དུ་མཆོད་ཡོན་གྱི་སྣོད་ཅེས་གསུངས་པའི་དགོངས་པའོ། །མདོར་བསྡུས་པའི་
རིམ་པའོ། །རྒྱས་པར་ནི་ཨོཾ་ཨཱ་ཛཿ་ཧཱུཾ་ཞེས་པ་འདིས་བཀུག་ནས་ཐབ་ཁུང་གི་མེའི་ཕྱོགས་ཀྱི་ལོགས་ན་ཀུ་ཤའི་ཆུན་པོ་ལ་འདུག་པ་ལ། མཆོད་ཡོན་ལ་སོགས་པ་ཕུལ་ཏེ། དེ་ཉིད་གསུམ་གྱིས་སྤེལ་བའི་ཛཧཱུཾ་བཾ་ཧོཞེས་བྱ་བའི་རིམ་གྱིས་རྡོ་རྗེ་དང་བཅས་པའི་རྡོ་རྗེ་དྲིལ་བུས་འཁྱུད་པའི་ཚུལ་དང་ལྡན་པའི་ཕྱག་
གཉིས་ཀྱིས་སེ་གོལ་གཏོགས་པ་སྔོན་དུ་འགྲོ་བས་དམ་ཚིག་སེམས་དཔའ་ལ་རབ་ཏུ་བཅུག་ལ་བཅིང་བ་དང་དབང་དུ་བྱས་ནས་ཡང་མཆོད་ཡོན་ལ་སོགས་པ་ཐམས་ཅད་བྱའོ། །དེ་ནས་ཨོཾ་བཛྲ་ཨ་ན་ལ་མ་ཧཱ་བྷཱུ་ཏ་ཛྭ་ལ་ཛྭ་ལ་ཡ་སརྦ་བྷ་སྨའི་ཀུ་རུ་སརྦ་དུཥཊཱ་ན་ཧཱུཾ་ཕཊ་དྲྀ་ཤྱ་ཛཧཱུཾ་བཾ་ཧོས་མ་ཡ་སྟྭཾ་ས་མ་ཡ་ཧཾ་ཞེས་ལན་
གསུམ་དུ་དམ་ཚིག་བསྒྲག་པར་བྱའོ། །དེའི་རྗེས་ལ་ལག་པ་གཡོན་པ་གན་རྐྱལ་དུ་བྱས་ལ་བྱང་ཆུབ་མཆོག་གི་ཁུ་ཚུར་གྱིས་མར་གྱིས་གང་བའི་དགང་གཟར་བཟུང་ནས་ལག་པ་གཡས་པ་ནི་ཁ་སྦུབ་པས་བླུགས་གཟར་གན་རྐྱལ་དུ་བྱས་པ་མར་གྱིས་གང་བ་དང། དེ་བཞིན་དུ་དྲག་པོའི་ལས་ལ་ནི་དུག་དང་བཅས་པའི་
ཚ་བའི་མར་ནག་གིས་གང་བའི་དགང་གཟར་གཡས་ཀྱིས་བཟུང་ཞིང་གཡོན་པས་བླུགས་གཟར་ཡང་སྟེ་དགང་གཟར་བླུགས་གཟར་གྱི་ཁ་ལ་རང་གི་ལྷའི་ས་བོན། ཡི་གེ་གསུམ་གྱིས་སྤེལ་བའི་འོད་ཟེར་དང་བཅས་པ་དང། མེ་ལྷའི་ཞལ་དུ་རཾ་རྣམ་པར་བཀོད་ལ་དགང་གཟར་གྱི་ཁར་བླུགས་གཟར་གྱི་ཁ་བཞག་སྟེ་ལག་
