།རྒྱ་གར་སྐད་དུ། བཱ་གའི་རཱ་ཊ་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། གསུང་གི་མངའ་བདག་གི་
སྒྲུབ་ཐབས། དངོས་གྲུབ་ཐམས་ཅད་རབ་བྱིན་པའི། །ཡེ་ཤེས་རང་བཞིན་མགོན་ལ་འདུད། །ཀྲི་ཡའི་རྒྱུད་ཀྱི་རྗེས་འབྲངས་ལ། །དེ་ཡི་སྒྲུབ་ཐབས་བདག་གིས་བཤད། །དེ་ལ་རེ་ཞིག་དང་པོར་སྔགས་པས་ཡིད་དང་མཐུན་པའི་གནས་སུ་ཇི་ལྟར་བདེ་བར་ཉེ་བར་གནས་ལ། སྡུག་བསྔལ་
གྱི་ཆུ་བོར་བྱིང་བར་གྱུར་པའི་སེམས་ཅན་གྱི་ཁམས་རྣམས་ལ་བལྟས་པས་སྨོན་ལམ་གདབ་པར་བྱ་སྟེ། སྡུག་བསྔལ་བར་གྱུར་པའི་སེམས་ཅན་འདི་དག་རྣམས་འཁོར་པ་ན་འཁོར་བར་གྱུར་པ་དེ་དག་འཇམ་པའི་དབྱངས་སུ་གྱུར་ནས། བླ་ན་མེད་པའི་གོ་འཕང་ལ་ལེགས་པར་དགོད་པར་བྱའོ་
