འོག་ཏུ་ཐད་དྲང་རྣམས་ལ་དགོད། །ལུག་གི་སྨིན་དྲུག་རྒྱལ་ཡིན་ཏེ། །དེ་བཞིན་སྦྲུལ་ནི་སྣར་མ་མགོ །བྱི་བ་དབོ་དང་མོན་གྲེར་བརྗོད། །ཁྱུ་མཆོག་མཆུ་དང་ཆུ་སྟོད་ཡིན། །སྤྲེའུ་གྲོ་བཞིན་ཆུ་སྨད་དྲན། །རྟ་ནི་ཐ་སྐར་མོན་གྲུའོ། །གླང་པོ་བྲ་ཉིན་མ་གྲུར་བརྗོད། །དེ་ནས་སྤྱང་
ཁུ་ཡག་དང་སྣྲུབས། །བྱི་བཞིན་ཁྲུམས་སྟོད་ནེའུ་ལེ། །བྱི་ལ་དེ་བཞིན་ནབ་སོ་སྐག །སྟག་ནི་ནག་པས་གར་བརྗོད། །ཁྲི་སྙན་སྣོན་དང་ལྷ་མཚམས་བཤད། །མ་ཧེ་མེ་བཞི་ས་རིར་བཤད། །སེང་གེ་ཁྲུམས་དང་གྲེ་ཡིན་ཏེ། །འོག་དང་གོང་གི་རེ་མིག་ལ། །ལུག་སོགས་
སྦྱང་སོགས་རིམ་བཞིན་དགོད། །དེ་ལྟར་རྟ་དང་མ་ཧེ་དགྲ། །གླང་དང་སྟག་དག་མི་མཐུན་ཏེ། །སྦྲུལ་དང་ནེ་ལེ་མཛའ་བ་མེད། །ལྷ་མོ་གླང་ཆེན་སེང་གེ་དགྲ། །ལུག་དང་སྤྱང་ཁུ་མཐུན་པ་མེད། །མཛེས་མ་བྱི་བ་བྱི་ལ་ཡང་། །དེ་བཞིན་སྤྲེའུ་ཁྲི་སྙན་དྲན། །གཡུལ་དུ་ཤེས་བྱ་
ལྷས་མཆོད་མ། །ཇི་ལྟར་མིང་གི་སྐར་མ་ནི། །མཆུ་དང་ཆུ་སྟོད་ཁྱུ་མཆོག་མིང་། །སེང་གེའི་མིང་ལ་ཁྲུམས་སྨད་གྲེ། །དེ་ལྟར་ཐམས་ཅད་ཤེས་པར་བྱ། །གང་གི་རིགས་ནི་ཤེས་པ་ལས། །སྟོབས་ཀྱི་ཆེ་ཆུང་ཤེས་པར་འགྱུར། །མིང་གི་སྐར་མའི་མིང་ཤེས་པར། །མཛེས་
དགའ་གཡུལ་སོགས་ཇི་ལྟར་བྱེད། །རྒྱལ་དང་ཐ་སྐར་ནག་སོ་མགོ །ནམ་གྲུ་ལྷ་མཚམས་ས་རི་དང་། །གྲོ་བཞིན་མེ་བཞིར་བཅས་བ་རྣམས། །ལྷ་ཡི་རིགས་སུ་ཤེས་པར་བྱ། །ཆུ་སྟོད་ཁྲུམས་སྟོད་ཁྲུམས་སྨད་གྲོ། །དབོ་དང་ཆུ་སྟོད་བྲ་ཉེ་ལག །སྣར་མའི་མིང་གི་བྱེ་བྲག་
རྣམས། །མོ་ཡི་རིགས་ཡིན་བཞིན་{𰵶བཟངས་,བཟང་}མ། །མཆུ་དང་ནག་པ་སྨིན་དྲུག་དང་། །མོན་གྲེ་མོན་གྲུས་ག་དང་། །སྣུབས་སྐག་སྣོན་དང་བཅས་པ་རྣམས། །སྲིན་པོའི་རིགས་ལས་གྲུབ་པ་ཡིན། །རང་གི་རིགས་རྣམས་མཉམ་པར་མཐུན། །ལྷ་དང་མིན་འབྲིང་པོ་སྟེ། །ལྷ་དང་
སྲིན་པོ་དེ་བཞིན་ནོ། །མི་དང་སྲིན་པོ་ལྷན་ཅིག་འཆི། །གལ་ཏེ་ཆུང་མ་སྲིན་པོར་འགྱུར། །ཁྱོ་ནི་མི་ཡི་རིགས་ལས་སྐྱེས། །དེ་ཚེ་ཁྱོ་ནི་འཆི་བར་འགྱུར། །ཤིན་ཏུ་རྒྱལ་རིང་འདུག་པར་བྱ། །མི་མཛའ་ལྷན་ཅིག་འཆི་བ་སྟེ། །རང་གི་རིགས་མཐུན་པ་དེ་བར་བྱེད། །གལ་ཏེ་ལྷ་ཡི་
