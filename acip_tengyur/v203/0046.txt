ཀྱང་ཆར་པ་ཆེན་པོ་མི་འབེབས་པ་དང་། ཕྱོགས་ཐམས་ཅད་ཀྱི་སའི་མདངས་ཀྱང་སྔོག་ནག་ཏུ་འགྱུར་ཞིང་མིག་སལ་བ་དང་། རྩུབ་པར་སྣང་བ་དང་། བ་ལང་དང་། ར་ལུག་དང་། བྱ་དང་། {𰵎རི་དགས་,རི་དྭགས་}ཀྱང་འཇིགས་པ་མེད་པར་ཡང་སྐྲག་ཅིང་འདྲོག་པ་དང་། ཕྱོགས་བསླད་ཅིང་ཕན་ཚུན་ཐོབ་པར་འགྱུར་
རོ། །ཡང་རྒྱུ་སྐར་ས་ག་དང་། སྨིན་དྲུག་དང་། རྒྱལ་དང་། ཁྲུམས་སྟོད་དང་། བྷ་གའི་ལྷ་གང་ཡིན་པའི་འདི་དང་། མཆུ་དང་། གཤིན་རྗེའི་ལྷ་གང་ཡིན་པ་འདི་དག་དབང་བའི་དུས་སུ་ས་གཡོས་པའམ། གནམ་གྲུམ་པ་འབྱུང་བར་གྱུར་ན་མིའི་ཕྱོགས་ཀྱི་འཁོར་ལ་བབ་པའི་ཤུགས་ཀྱིས་གནོད་པ་འགྱུར་
བར་རིག་པར་བྱ་སྟེ། མཁྲིས་པའི་རིགས་ལས་གྱུར་བའི་གནོད་པས་སྐྱེ་དགུ་ཤིན་ཏུ་གཟིར་བ་དང་། བུད་མེད་རྣམས་ལ་ཁྲག་ནད་འབྱུང་བ་དང་། བ་དང་མ་ཧེ་ཞོ་སྐམ་པ་དང་། བསོད་ནམས་བྱ་བར་བརྩམས་ཤིང་བསམ་པ་ཐམས་ཅད་ཀྱང་འགྲུབ་པར་མི་འགྱུར་ལ། སེམས་ཅན་རྣམས་ལ་ཡམས་ནད་འབྱུང་
བ་དང་། དེ་བཞིན་དུ་ནད་དང་། རྒྱལ་པོ་ལས་ཀྱང་གྱུར་པ་འབྱུང་བ་དང་། སྐྱེ་དགུ་རྣམས་ཀྱང་ཕན་ཚུན་དགའ་བ་དག་བྱེད་ནས་མི་མཐུན་པར་ཡང་འགྱུར་ཞིང་ཉིན་མཚན་དུ་ཕྱོགས་རྣམས་སུ་གནས་མེ་ཞི་བར་ཡང་མི་འགྱུར། སྤྲིན་ཡང་ཆར་བ་འབེབས་པར་མི་འགྱུར། དེ་བཞིན་དུ་འདོད་འཇོ་དག་ཀྱང་འོ་མ་མེད་པར་
འགྱུར་བ་དང་། ཤིང་ལྗོན་པའི་རིགས་དང་། ལྕུག་མ་དག་ཀྱང་མེ་ཏོག་དང་། འབྲས་བུ་ཉུང་བ་དང་། ཆུའི་གནས་ཁྲོན་པ་དང་། རྫིང་དང་། ལྟེང་ཀ་ལ་སོགས་པ་ཡང་སྐམ་པར་འགྱུར་ཞིང་། མུ་གེ་དང་ནད་ཀྱིས་ཉེན་བ་འབྱུང་བ་དང་། ཡུལ་འཁྲུག་པར་ཡང་འགྱུར། ཆོམ་རྐུན་གྱི་འཇིག་རྟེན་རྣམས་ཀྱི་མཐུ་
དང་། {𰵏རི་དགས་,རི་དྭགས་}རྣམས་ཀྱང་འདྲེན་པར་འགྱུར། ཤ་སྦྲང་དང་། སྦྲང་བུ་མཆུ་{𰵐རིངས་,རིང་}རྣམས་དང་། དེ་བཞིན་དུ་རིགས་ངན་གྱི་རིགས་དག་ཀྱང་འཕེལ་བར་འགྱུར་ཏེ། དེ་ལྟར་མེའི་ཕྱོགས་ཀྱི་དཀྱིལ་འཁོར་དུ་ཆོས་ལུགས་དེ་རྣམས་འབྱུང་བར་ཤེས་པར་བྱའོ། །ཁྲུམས་སྨད་དང་། ཆུ་ལྷ་དང་། དེ་བཞིན་དུ་ནམ་གྲུ་དང་།
བདེན་བྲལ་གྱི་ལྷ་དང་། སྐག་དང་། དེ་བཞིན་ལག་ནི་ཆུའི་དཀྱིལ་འཁོར་ཡིན་ཏེ། དེ་ནི་བསྟན་བཅོས་རིག་པ་རྣམས་ཀྱིས་བརྗོད་པ་ཡིན་ནོ། །གལ་ཏེ་འདི་རས་གཡོ་བ་དང་། གནམ་གྲུམ་པ་བྱུང་བ་དང་། ཉི་མ་དང་། ཟླ་བ་ཡང་ཉེ་བའི་ཆགས་པ་དང་ལྡན་པ་དང་། མཇུག་{𰵑རིངས་,རིང་}དང་། དེ་བཞིན་དུ་ལྟས་སུ།
