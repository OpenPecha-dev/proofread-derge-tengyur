ལྷ་དང་ལྷ་མིན་རྣམས་དང་ནི། །སྟོང་ཕྲག་དག་ནི་བརྒྱ་སྙེད་ཀྱིས། །དད་པའི་དབང་པོ་སོགས་ཐོབ་ནས། །སྲིད་པ་ལས་ནི་གྲོལ་བར་གྱུར། །ལྷས་སྦྱིན་ཁྱིམ་ན་གནས་ན་ནི། །འཁོར་ལོས་སྒྱུར་བའི་སྟོབས་ལྡན་ཞིང། །འཁོར་དང་སྟོབས་ནི་ཐོབ་གྱུར་ནས། །བསྟན་པ་འདི་ནི་ཉམས་པར་
