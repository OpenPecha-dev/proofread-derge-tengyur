།རྒྱ་གར་སྐད་དུ། བཛྲ་ཙཎྜ་མཎྜལ་བིདྷི་བཱ་ས་ཛ་ཀ་བྷ་དྲ་ནཱ་མ། བོད་སྐད་དུ། རྡོ་རྗེ་གཏུམ་པོའི་དཀྱིལ་འཁོར་གྱི་ཆོ་ག་བཟང་པོ་མདོར་བསྡུས་པ་ཞེས་བྱ་བ། གསང་བའི་བདག་པོ་རྡོ་རྗེ་གཏུམ་པོ་ལ་ཕྱག་འཚལ་ལོ། །གསང་བ་
བསྟན་པའི་རྒྱུད་དང་ནི། །བླ་མའི་གདམས་པའི་ལམ་ཞུགས་ནས། །ཆོ་ག་བཟང་པོ་མདོར་བསྡུས་འདི། །རང་དགའ་ཉེས་པ་དོར་ཏེ་བྲི། །སློབ་དཔོན་མཚན་ཉིད་ཀུན་ལྡན་ཞིང། །རྒྱུད་ཀྱི་དབང་ཐོབ་དམ་ཚིག་བརྟན། །གཞན་ཕན་བསྒྲུབ་ལ་དགའ་བ་ལ། །གཞན་དག་གུས་
