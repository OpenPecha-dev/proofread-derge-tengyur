རྒྱལ་པོ་དེས་སྲས་ཀྱི་དོན་དུ་སྣ་ཚོགས་པའི་ལྷ་ཉེ་བར་མཆོད་པ་ལ་རྩེ་གཅིག་པའི་དུས་གཅིག་གི་ཚེ་ན་བཙུན་མོའི་ལྟོ་བར་བྱང་ཆུབ་སེམས་དཔའ་བཙས་པར་གྱུར་ཏེ། ཇི་ལྟ་ཇི་ལྟར་མངལ་གྱིས་ཀུན་ལ་ལེ་ལོ་ཅན། །
དེ་ནི་སྐྱེ་བ་ཡིད་འཕྲོག་བཞིན་ལྡན་འབྱུང་འགྱུར་བ། །དེ་ལྟ་དེ་ལྟར་དེ་ལ་ལྟ་ཞིང་མི་སྐྱོང་ནི། །ཕྲད་པར་འདོད་པས་བ་དེ་བླག་ཏུ་ནི་ཉེར་འགྲོ་འགྱུར། །སྲས་ནི་རིང་པོར་བྱིས་པ་ཉིད་དུ་བདག་ལ་རོལ་འགྱུར་བ། །གཞོན་ནུ་ཉིད་ནི་རིགས་ཀྱི་ཁུར་བུ་ཆེན་པོ་འཛིན་གྱུར་ཏེ། །བདག་ནི་འཇིག་
རྟེན་གཞན་དུ་འགྲོ་བ་ཤིད་ནི་སྟེར་བྱེད་པ། །དེ་ལྟར་སྐྱེ་བོ་དེ་ཕྱིར་ངན་སོང་སྡུག་བསྔལ་མཐོང་མི་འགྱུར། །དེ་ནས་དུས་ཀྱི་རིམ་པ་ལས་བཙུན་མོ་མཚན་ཐམས་ཅད་ཀྱིས་བརྒྱན་པའི་ལུས་ལྷའི་གཞོན་ནུ་དང་འདྲ་བའི་གཟུགས་དང་ལྡན་པ་གནས་གྱི་ཁྱད་པར་འདོད་པའི་རྒྱལ་པོའི་
ཡོན་ཏན་དུ་མ་རྣམས་སོ་སོར་སྐྱོང་བཞིན་པ་དང་འདྲ་བའི་དཔལ་ལྡན་སྲས་སྐྱེས་པར་གྱུར་ཏེ། དེ་ནས་དེར་ནི་སྲས་སྐྱེས་པ་ལ་སྲས་ནི་འདོད་པ་ཡི། །བསོད་ནམས་བདག་ཉིད་ནོར་མང་བ་ཡི་ནོར་ཅན་རྒྱལ་པོ་ཡི། །གྲོང་ཁྱེར་དེ་ལ་འཕྲོ་བའི་རིན་ཆེན་འོད་ཟེར་སྣང་བ་ཅན། །ས་ཡི་གཏེར་རྣམས་ཀུན་
ལས་རབ་ཏུ་འབྱུང་བར་གྱུར་པ་ནི། འཕྲོས་པ་དྲི་མ་མེད་པའི་འོད་ཅན་རིན་ཆེན་རྣམས་ཁ་ཅིག་དྲི་མ་མེད་པའི་འོད་ལྡན་མུ་ཏིག་ཚོགས་ནི་གཞན་དག་ཀྱང། །ས་སྐྱོང་རྣམས་ཀྱིས་རྒྱ་མཚོའི་འགྲམ་ལ་ལྷག་གནས་བྱས་པ་ནི། །སྣ་ཚོགས་པ་ཡི་ནོར་ནི་ནོར་ཅན་རྣམས་ལ་རབ་ཏུ་སྐུར། །
དེ་སྐྱེས་པ་ལ་གང་ཕྱིར་ནོར། །ནོར་ཅན་གྱི་ནི་དུ་མར་འགྱུར། །དེ་ཕྱིར་ནོར་བཟངས་ཞེས་བྱ་ཉིད། །སྐྱེ་བོ་དེ་ཡི་མིང་བཏགས་གྱུར། །དེ་ནས་དུས་གཅིག་གི་ཚེ་ན་རིག་པ་འཛིན་པར་ནུས་པ་ནོར་བཟངས་ཤེས་ནས་གཞུའི་རིག་བྱེད་ཤེས་པ་རྒྱལ་པོའི་བསྟན་བཅོས་འཆད་པ་ལ་
མཁས་པ་རྣམས་དང། གླང་པོ་ཆེའི་བསླབ་པའི་རང་གི་མཚན་ཉིད་ཤེས་པའི་མཁན་པོ་རྣམས་ལ་ནོར་ཅན་གྱིས་སློབ་མ་ཉིད་དུ་སོང་བར་གྱུར་ཏེ། སྤོབས་པའི་གྲུ་ནི་མ་རྡོལ་བ། །རིང་པོར་འགྲོ་བ་ལ་ཞུགས་ནས། །རིག་པའི་ཆུ་བོ་ཟབ་རྣམས་ཀྱི། །ཕ་རོལ་དུ་ནི་ཀུན་སོང་
