༄༅། །རྒྱལ་བ་དམ་པས་གསུངས། །ཞེས་བྱ་བ་ཡིན་ནོ། །དེ་ལྟར་ཀུན་མཁྱེན་གྱིས་གསུངས་བཀའ། །ཀུན་ཕན་མ་ལུས་བརྟགས་ནས་ནི། །མདོར་བསྡུས་ནས་ནི་བསྟན་པ་འདི། །དེ་ཉིད་ཤེས་རྣམས་མགུ་ཕྱིར་བྱས། །དེ་ལྟར་ལམ་ལ་རྣམ་ཞུགས་སེམས་ནི་ངེས་པར་མི་
གཡོ་གང། །གཡོ་བར་གྱུར་ནའང་དེ་ཚེ་ཡང་དག་བཀའ་ནི་རྣམས་བསྒོམས་ལ། །རྒྱལ་བའི་གསུང་གི་ཚོགས་ནི་ཐོག་མ་བར་དང་མཐའ་དག་ལ། །བརྟེན་ཅིང་མི་ཕྱེད་ཚུལ་རྣམས་ཀུན་གྱིས་རྣམ་པར་བརྟགས་པས་བརྟེན་པར་འགྱུར། །དེ་ལྟར་མ་ལུས་བདེ་གཤེགས་བཀའ། །བདག་གིས་
ཡང་དག་བཤད་པ་ཡི། །བསོད་ནམས་གང་ཡིན་དེ་འགྲོ་ཀུན། །དེ་ཡི་གཞུང་ལ་ཅི་དགར་ཤོག་དེ་ཁོ་ན་ཉིད་ལ་འཇུག་པ་ཞེས་བྱ་བ་བདེ་བར་གཤེགས་པའི་བཀའ་མ་ལུས་པ་མདོར་བསྡུས་ཏེ་བཤད་པའི་རབ་ཏུ་བྱེད་པ། སློབ་དཔོན་དཔལ་ཡེ་ཤེས་གྲགས་པས་མཛད་པ་རྫོགས་སོ།། །།
རྒྱ་གར་གྱི་མཁན་པོ་པདྨཱ་ཀ་ར་ཝརྨ་དང། ཞུ་ཆེན་གྱི་ལོ་ཙཱ་བ་དགེ་སློང་རིན་ཆེན་བཟང་པོས་བསྒྱུར་ཅིང་ཞུས་ཏེ་གཏན་ལ་ཕབ་པའོ།།
