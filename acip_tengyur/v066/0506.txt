།རྒྱ་གར་སྐད་དུ། ཨཱརྱཱ་བ་ལོ་ཀི་ཏེ་ཤྭ་ར་ཙིནྟ་མ་ཎི་ཙཀྲ་ཝརྟྟི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། འཕགས་པ་འཇིག་རྟེན་དབང་ཕྱུག་གི་ཡིད་བཞིན་གྱི་ནོར་བུ་འཁོར་ལོས་སྒྱུར་པའི་སྒྲུབ་ཐབས། ཐུགས་རྗེ་ཆེན་པོ་ལ་ཕྱག་འཚལ་ལོ། །འགྲོ་བ་
སྐྱོབ་མཛད་ཐུགས་རྗེ་ཆེན་པོའི་གཙོ། །ཟླ་བའི་ནོར་ལྟའི་སྐུ་མངའི་ཐུགས་རྗེ་དང། །མཆོག་སྦྱིན་དེ་ལ་གསུམ་གྱིས་ཕྱག་འཚལ་ནས། །ཉི་མའི་ཟེར་གྱིས་མུན་བསལ་ཆུ་སྐྱེས་ཁ་འབྱེད་ལྟར། །ཐུགས་རྗེའི་འོད་ཀྱིས་མ་རིག་མུན་སེལ་སྙིང་གི་པདྨ་འབྱེད་མཛད་པའི། །སྒྲུབ་ཐབས་དམ་པ་བཀའ་
དྲིན་དྲན་ཕྱིར་བྲི་བར་བྱ། །མེ་ཏོག་ར་བ་ལ་སོགས་གཙང་མ་ཡིད་དགའི་གནས་སུ་ནི། །རྣམ་ལྔ་ཧྲཱི་ཡིས་བསྔགས་པའི་མཎྜལ་ལེགས་པར་བྱ། །ཁ་དོག་དཀར་པོའི་མེ་ཏོག་ལ་སོགས་མཆོད་པ་ལྡན་བྱས་ལ། །ཡི་གེ་གསུམ་ལྡན་ཁྲུས་ཀྱིས་དྲི་མ་གཉིས་བཀྲུས་ཏེ། །ཉོན་མོངས་མཐའ་
དག་གཞིག་ཕྱིར་བདེ་བའི་སྟན་ལ་འདུག་ཡི་གེ་གསུམ་ལྡན་ཧྲཱི་ཡིས་གནས་བདག་རྣལ་འབྱོར་བསྲུང། །སྙིང་གར་ཟླ་བའི་དཀྱིལ་འཁོར་སྟེང་དུ་ཧྲཱི་ཡིག་དཀར་པོ་བསམ། །འོད་ཀྱིས་འགྲོ་བའི་དོན་བྱས་འཇིག་རྟེན་དབང་ཕྱུག་སྤྱན་དྲངས་ཏེ། །ཡི་གེ་གསུམ་ལྡན་ཧྲཱི་ཡིས་མེ་ཏོག་
