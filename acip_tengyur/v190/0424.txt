མོང་བར་ཡང་བརྟགས་པ་ནི། དུ་མ་དེའི་བདག་ཉིད་དུ་ཡོད་པ་བཞིན་ནོ། །དེ་ལྟར་ཡང་སྔོན་པོ་ནི་དབྱིབས་མཐོང་ན་མཐོང་བ་དང་དེའི་རང་བཞིན་དུ་དམིགས་པ་མ་ཡིན་ཏེ། དབྱིབས་ལོག་ཀྱང་སྔོན་པོ་དམིགས་པའི་ཕྱིར་དང་། དབྱིབས་ཉེ་ཡང་རྡུལ་ཕྲ་རབ་ཙམ་གྱི་ཆའི་སྔོན་པོ་དབྱིབས་ཀྱི་
བདག་ཉིད་མ་ཡིན་པར་དམིགས་པའི་ཕྱིར་རོ། །དབྱིབས་ཉིད་སྔོན་པོ་དེ་མཐོང་ན་སྣང་ཞིང་དེ་མེད་ན་མི་སྣང་ལ། སྣང་ན་ཡང་དེའི་རང་བཞིན་དུ་སྣང་བ་ཡིན་ཏེ། སྔོན་པོར་སྣང་བ་བཏང་ན་དབྱིབས་མི་རྟོགས་པའི་ཕྱིར་རོ། །དེའི་ཕྱིར་སྔོན་པོ་ཉིད་དེ་ལ་དུ་མ་ཡིན་ཞིང་དུ་མའི་ཐུན་མོང་མ་ཡིན་པ་
དང་། གཞི་ཡང་ཡིན་ནོ། །དབྱིབས་ནི་དེ་ལྟ་བུ་དུ་མ་ཡིན་གྱི། འོན་ཀྱང་དུ་མའི་ཐུན་མོང་དང་། དེ་ལ་བརྟེན་པ་ཡིན་ནོ། །གང་གི་ཕྱིར་དེ་མེད་པར་མི་སྣང་ཞིང་སྔོན་པོ་ལ་སོགས་པ་ཡོད་ན་ནི་<𰵧ད་,དེ་>བདག་ཉིད་དུ་རྟོགས་པར་འགྱུར་བ་དེའི་ཕྱིར་སྔོན་པོ་ནི་སྐད་ཅིག་མ་ཉིད་ལ་བརྟེན་པ་མ་ཡིན་ཏེ། དེ་མ་
རྟོགས་ཀྱང་སྔོན་པོ་ནི་དེ་མ་ཡིན་པའི་རང་བཞིན་དུ་སྣང་བའི་ཕྱིར་རོ། །སྐད་ཅིག་མ་ཉིད་ནི་དེ་ལ་བརྟེན་པ་ཡིན་ནོ། །སྔོན་པོ་ལ་སོགས་པའི་དངོས་པོ་མཐོང་ན་དེ་རྟོགས་པའི་ཕྱིར་དང་མཐོང་བ་དེའི་བདག་ཉིད་དུ་རྣམ་པར་འཇོག་པའི་ཕྱིར་རོ། །ཅི་སྟེ་ཐུན་མོང་དུ་རྟོགས་པ་འདི་དུ་མར་ཡོད་པ་ཉིད་དུ་
ཁས་བླངས་ནས་དེའི་རང་བཞིན་དུ་ཡོད་པར་རྣམ་པར་འཇོག་པ་གང་ཡིན་པ་དེ་ལ་ཡིད་བརྟན་དུ་ཅི་ཡོད་ཅེ་ན། བཤད་པ་ཐ་དད་པ་ཉིད་དོན་བྱེད་པ་ལ་ཉེ་བར་སྦྱོར་བར་སྣང་བའི་ཕྱིར་རོ། །མཐོང་བའི་གཟུང་བ་ཁྱད་པར་ཁོ་ན་དོན་བྱེད་པ་ལ་ཉེ་བར་སྦྱོར་བ་མཐོང་གི་སྤྱི་ནི་མ་ཡིན་ནོ། །ཡོད་པ་ནི་དོན་
བྱེད་པའི་རྒྱུ་མཚན་ཅན་ཡིན་ནོ། །དེ་ཉིད་ཀྱི་ཕྱིར་སྐད་ཅིག་མ་ཉིད་ནི་སྣང་བ་སྒྲ་ལ་སོགས་པའི་རང་བཞིན་དུ་དོན་བྱེད་པའི་མཚན་ཉིད་དུ་ཡོད་པ་མཐོང་བ་ལས་སྒྲུབ་པར་བྱེད་དོ། །དེའི་སྒྲུབ་པར་བྱེད་པ་ཡང་ཁྱད་པར་ཅན་གྱི་ཆོས་ཅན་ལ་བརྟེན་པ་ཡིན་ནོ། །དེ་ལྟ་ན་འོ་ན་དབྱིབས་ཀྱང་དོན་བྱེད་ནུས་
པར་མཐོང་བའི་ཕྱིར་ཇི་ལྟར་དངོས་པོ་མ་ཡིན། འདི་ལྟར་དེའི་རང་བཞིན་ཡོད་པ་ཉིད་ནི་དབྱིབས་ཀྱི་ཐབས་ཀྱི་དུང་གིས་བསྒྲུབ་པར་བྱ་བའི་འབྲས་བུ་ཐོབ་པར་བྱེད་དོ། །དེའི་ཕྱིར་དབྱིབས་རང་བཞིན་གཞན་ཉིད་དེ་ཐོབ་པར་བྱེད་པ་ཡིན་པའི་ཕྱིར་དོན་བྱེད་པ་ལ་དེ་ཉིད་ཀྱི་ནུས་པ་མཐོང་བ་ཡིན་ནོ། །
