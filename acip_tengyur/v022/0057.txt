༄༅། །བཛྲ་ཙ་ཀྲ་བཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲྀ་ཀྵ་རཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པྲྀ་ཀྵ་ཎ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཀྐ་བའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཀྐ་བི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ལ་ཀཱ་ཀཱི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛ་ལ་ཀཱ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བི་ལཱ་ཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
བི་ལ་ཌ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་གྲི་བཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་གྲི་བ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཱ་རི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཱ་རི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སེ་ནཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སེ་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུ་མ་ལོ་ལཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུཾ་ཀུ་མ་ལོ་ལ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
བཱ་ཊི་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཱ་ཊི་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀ་ཛི་གྷ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀ་ཛཾ་ཧ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་ས་མའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་ས་མ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་ལེ་ཧ་སྲི་ཥྚ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་ལེ་ཧ་སྲིཥྚ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་དད་ད་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་དད་ད་རི་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་གཱ་རི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་མྲྀ་གཱ་རི་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་སིཾ་གྷཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་སིཾ་གྷ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་བྱཱ་གྷྲི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་བྱཱ་གྷྲ་ཧཱུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་བཧི་བཧི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་བྷིམྦྷ་ཧུཾ་ཧཱུཾ་ཕཊ།ཨོཾ་བཛྲ་ཤ་ཤཱི་
ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤ་ཤི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ག་ཛ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མྲྀ་གཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོམ་བཛྲ་མྲྀ་ག་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མརྫ་རི་ཀཱ་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མརྫ་རི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གཱ་བ་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གཱ་བ་ཧུཾ་ཧཱུཾ་
ཕཊ། ཨོཾ་བཛྲ་མ་ཧི་ཥཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧི་ཥ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མ་ཧི་ཥ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏུ་ར་གའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏུ་ར་ག་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཛཾ་བུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བུ་ཀ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གི་ཎཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོམ་བཛྲ་གི་ཎཌ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙ་མ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ་
ཨོཾ་བཛྲ་ཙ་མ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མུ་ཥཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མུ་ཥ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གརྡ་བྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གརྡ་བྷ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཧི་ཊཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཧི་ཊ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཛ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཨ་ཛ་ཀ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
