།རྒྱ་གར་སྐད་དུ། ཤྲཱི་བཛྲ་ཡོ་གི་ནཱི་སཱ་དྷ་ནཾ། བོད་སྐད་དུ། དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྒྲུབ་ཐབས། །དཔལ་རྡོ་རྗེ་རྣལ་འབྱོར་མ་ལ་
ཕྱག་འཚལ་ལོ། །སྟོང་ཉིད་སྙིང་རྗེའི་བདག་ཉིད་ཅན། །ཁམས་གསུམ་པ་ཡི་ངོ་བོ་ཉིད། །བསྐལ་བའི་མེ་ལྟར་འབར་བའི་འོད། །རྡོ་རྗེ་རྣལ་འབྱོར་མ་དཔལ་འདུད། །བླ་མ་བརྒྱུད་པའི་རིམ་པ་ནི། །རྟོགས་པ་གཅིག་གི་སྤྱོད་ཡུལ་ཏེ། །རྡོ་རྗེ་རྣལ་འབྱོར་མའི་སྒྲུབ་ཐབས། །མདོར་བསྡུས་བདག་གིས་
བཤད་པར་བྱ། །སྔགས་པས་ཐོག་མར་རེ་ཞིག་བླ་མ་དང་སངས་རྒྱས་རྣམས་ལ་མངོན་པར་གུས་པའི་ཡིད་ཀྱིས། །བྱང་ཆུབ་ཀྱི་སེམས་བརྟན་པོར་བཟུང་སྟེ། དབང་བསྐུར་བ་ཡང་དག་པར་ཐོབ་ནས། ཆོས་ཐམས་ཅད་ངོ་བོ་ཉིད་ཀྱིས་འོད་གསལ་བའི་རང་བཞིན་ཡིན་ཡང། རང་དང་གཞན་གྱི་དོན་ཕུན་སུམ་
