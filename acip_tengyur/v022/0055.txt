༄༅། །བཛྲ་ཀྲྀ་ཥི་ཀཱ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙརྨ་ཀཱ་རི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཙརྨ་ཀཱ་ར་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བྷདྲ་ཀལྤ་བུདྡྷེ་བྱོ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱ་ཀཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤྭ་ནཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་སཱུ་ཀ་རཱ་སྱེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཡ་མ་དཱ་དྷཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་དའུ་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་དཾ་ཥྚི་ཎཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཡ་མ་ཐ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀིནྣ་རཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀིནྣ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་ཧ་རི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གནྡྷ་ཧ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧུ་ཊུ་
ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧུ་ཊུ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཊ་བའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་པཱ་ཊ་བ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བའི་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བའི་ཎ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཾ་ཤཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་བཾ་ཤི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མུ་ཀུ་ཎཌཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀུཎྜ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
མུ་རྫཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མུ་རྫ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གགཧ་རཱི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱཾ་པཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀཱཾ་པ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤཻ་ལེནྡྲི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཤཻལེནྡྲི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གའི་ཏཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་གའི་ཏ་
ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་ར་ཊཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཀ་ར་ཊ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏྨ་ཌཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་མ་ཌ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནྲྀ་ཏྱཱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནྲི་ཏྱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་སྱ་ཡཻ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ལཱ་སྱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཊཀྐཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་
བཛྲ་ཊཀྐ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ལི་ཤ་ར་ཎཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏ་ལི་ཤ་ར་ཎ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུཾ་དུ་བཧི་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་དུཾ་དུ་བཧི་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཽ་འརི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཽ་དྲི་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཏཱ་ན་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་
མཱ་ཏུ་ནཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་མཱ་ཏུ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་བའི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ནཱ་ལ་བཱ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌྷཾ་བ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌྷཾ་བ་ཀ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་མ་རཱི་ཡེ་ཧཱུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཌ་མ་ར་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཊུ་ཎ་ཊུ་ཀཱི་ཡེ་ཧུཾ་ཧཱུཾ་ཕཊ། ཨོཾ་བཛྲ་ཊུ་ཎ་ཀ་
