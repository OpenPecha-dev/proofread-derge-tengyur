།རྒྱ་གར་སྐད་དུ། དེ་ཤ་ན་སྟ་བ་བྲྀ་ཏྟི། བོད་སྐད་དུ་
བཤགས་པའི་བསྟོད་པའི་འགྲེལ་བ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །འདི་ན་བསྟོད་པ་བྱེད་པའི་དུས་སུ་བསྟོད་པ་རྩོམ་པ་རྣམས་སངས་རྒྱས་ཀྱི་རྒྱུ་དང་འབྲས་བུ་དང། དགོས་པ་མཛད་པའི་མཚན་ཉིད་ཀྱི་གནས་སྐབས་རྣམ་པ་གསུམ་ལས་བརྩམས་ཏེ་
སྨྲ་བར་བྱེད་ལ། བསྟོད་པ་བྱེད་པ་དེ་ཡང་དད་པ་དང། །མང་དུ་ཐོས་པ་དང། ཉམ་ང་བའི་གནས་སྐབས་ཐོབ་པ་སྟེ། རྣམ་པ་གསུམ་ཡིན་ནོ། །དེ་ལ་འདི་ནི་གཙོ་ཆེར་མང་དུ་ཐོས་པའི་རྒྱུས་དགོས་པ་མཛད་པའི་གནས་སྐབས་ལ་བརྟེན་ཏེ་བཅོམ་ལྡན་འདས་ཀྱི་ཡོན་ཏན་བརྗོད་
