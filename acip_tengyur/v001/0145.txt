།རྒྱ་གར་སྐད་དུ། ནི་རུ་པ་མ་སྟ་བཾ། བོད་སྐད་དུ། དཔེ་མེད་པར་བསྟོད་པ། འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །
གང་ཞིག་ལྟ་བས་ཕོངས་པ་ཡི། །འཇིག་རྟེན་འདི་ལ་ཕན་བརྩོན་ཁྱོད། །དངོས་པོ་མེད་པའི་དོན་རིག་པས། །དཔེ་མེད་ཁྱོད་ལ་ཕྱག་འཚལ་བསྟོད། །གང་ཞིག་ཁྱོད་ཀྱིས་ཅུང་ཞིག་ཀྱང། །སངས་རྒྱས་སྤྱན་གྱིས་མ་གཟིགས་པ། །ཁྱོད་ཀྱིས་གཟིགས་པ་བླ་མེད་དེ། །དེ་ཉིད་དོན་ནི་རིག་པ་
ལགས༑ །དོན་དམ་པ་ཡི་ཡོད་པ་ཉིད། །རྟོགས་དང་རྟོགས་བྱ་མི་མངའ་ཞིང། །ཨེ་མའོ་མཆོག་ཏུ་རྟོགས་དཀའ་བའི། །ཆོས་ཉིད་སངས་རྒྱས་རྣམས་ཀྱིས་རྟོགས། །ཁྱོད་ཀྱིས་ཅུང་ཞིག་མ་བསྐྱེད་ཅིང། །ཆོས་རྣམས་བཀག་པའང་མ་ལགས་ལ། །མཉམ་པ་ཉིད་ཀྱི་ལྟ་བ་ཡིས། །བླ་
ན་མེད་པའི་གོ་འཕང་བརྙེས། །འཁོར་བ་སྤངས་པར་གྱུར་པ་ཡིས། །མྱ་ངན་འདས་ཁྱོད་མི་བཞེད་ཀྱི། །འཁོར་བ་མ་དམིགས་པ་ཉིད་ཀྱི། །ཞི་དེ་མགོན་པོ་ཁྱོད་ཀྱིས་རྟོགས། །ཁྱོད་ཀྱིས་ཀུན་ནས་ཉོན་མོངས་དང། །རྣམ་བྱང་རོ་གཅིག་གྱུར་རིག་པས། །ཆོས་དབྱིངས་མངོན་པར་དབྱེར་མེད་
པ། །ཀུན་དུ་རྣམ་པར་དག་པར་གྱུར། །གཙོ་བོ་ཁྱོད་ཀྱིས་འགའ་ཞིག་ཏུ། །ཡི་གེ་གཅིག་ཀྱང་མ་གསུངས་ལ། །གདུལ་བྱའི་འགྲོ་བ་མ་ལུས་པ། །ཆོས་ཀྱི་ཆར་གྱིས་ཚིམ་པར་མཛད། །མཁའ་དང་མཉམ་པའི་ཐུགས་མངའ་ཁྱོད། །ཕུང་པོ་ཁམས་དང་སྐྱེ་མཆེད་ལ། །ཆགས་པར་གྱུར་
