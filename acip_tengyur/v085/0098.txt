།རྒྱ་གར་སྐད་དུ། པྲཛྙཱ་པྲ་དའི་པ་མའུ་ལ་མ་དྷྱཱ་མི་ཀ་པྲྀཏྟི། བོད་སྐད་དུ། དབུ་མའི་
རྩ་བའི་འགྲེལ་པ་ཤེས་རབ་སྒྲོན་མ། བམ་པོ་དང་པོ། །དཀོན་མཆོག་གསུམ་ལ་ཕྱག་འཚལ་ལོ། །འཇམ་དཔལ་གཞོན་ནུར་གྱུར་པ་ལ་ཕྱག་འཚལ་ལོ། །སློབ་དཔོན་འཕགས་པ་ཀླུ་སྒྲུབ་ལ་ཕྱག་འཚལ་ལོ། །སློབ་དཔོན་བཙུན་པ་ལེགས་ལྡན་འབྱེད་ལ་ཕྱག་འཚལ་ལོ། །གང་གིས་ཆོས་ཀྱི་
དེ་ཉིད་རྣམ་རྟོག་ལེགས་བྲལ་བ། །སྤྲོས་པ་ཀུན་ཞི་བློ་ངན་རབ་རིབ་ཀུན་འཇོམས་ཅན། །ཚིག་གི་ཡུལ་མ་ཡིན་ཡང་དེ་ཉིད་སྒྲོ་བཏགས་ནས། །ས་ཀུན་འབྱོར་ཕྱིར་བསྟན་པ་དེ་ལ་ཕྱག་འཚལ་ལོ། །སློབ་དཔོན་གྱི་ཞལ་སྔ་ནས་ཀྱི་ཚིག་ལེའུ་བྱས་པ་དག་ཁོ་ནས་རྗེས་སུ་དཔག་པ་དང། སུན་དབྱུང་བའི་
གསལ་པ་དང། དེ་ཁོ་ན་དག་བསྟན་པ་དང། ལྟ་བ་ངན་པའི་དྲ་བ་ཞི་བར་བྱེད་པ་དང་ལྡན་པ་ཤེས་རབ་ཀྱི་ཕ་རོལ་ཏུ་ཕྱིན་པའི་ཚུལ་བཀའ་སྩལ་མོད་ཀྱི། ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་རྣམས་ལས་ཁ་ཅིག་གིས་མི་རྟོགས་པས། དེའི་ཕྱིར་དེ་དག་གིས་རྟོགས་པར་བྱ་བར་འདོད་ནས་ལུང་ཇི་ལྟ་བ་བཞིན་དུ་
