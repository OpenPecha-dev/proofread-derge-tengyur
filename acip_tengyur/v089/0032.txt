རྒྱུ་དང་རྐྱེན་གྱི་གཞན་དབང་དུ་གྱུར་བ་དེ་ཉིད་སྡུག་བསྔལ་བའོ། །དེའི་རྒྱུར་གྱུར་པའི་འདུ་བྱེད་རྣམས་ཀྱང་འདུ་བྱེད་ཀྱི་སྡུག་བསྔལ་གྱིས་སྡུག་བསྔལ་བའོ། །གཅོག་པའི་ཕྱིར་ཞེས་པ་ནི་ཇི་ལྟར་ཟོ་ཆུན་གྱི་འཁྲུལ་འཁོར་ཉི་མ་རེ་
རེ་ཞིང་ཟོ་བ་ཆག་པར་གྱུར་པ་དེ་བཞིན་དུ་འགྲོ་བ་རྣམས་ཉི་མ་རེ་རེ་ཞིང་མི་རྟག་པ་ཉིད་ཀྱིས་གཞན་དུ་གྱུར་བའོ། །ཟོ་ཆུན་བརྒྱུད་མོའི་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་འགྲོ་བ་ཟོ་ཆུན་གྱི་འཁྲུལ་འཁོར་གྱི་གནས་སྐབས་དང་རྗེས་སུ་མཐུན་པ་ཉིད་དུ་བསྟན་ཏོ། །དེའི་སྡུག་བསྔལ་གྱི་སྡུག་བསྔལ་ཞེས་པ་
ནི་འགྲོ་བ་རྣམས་ཀྱི་སྡུག་བསྔལ་གྱིས་སྡུག་བསྔལ་བ་གང་ཡིན་པའོ། །དེ་ལ་ཤིན་ཏུ་སྙིང་རྗེས་དམིགས་ནས་ཞེས་པ་ནི་སྙིང་རྗེ་དེས་སྡུག་བསྔལ་བའི་སེམས་ཅན་རྣམས་ལ་དམིགས་ནས་སོ། །བསྐྱབ་པར་སྤྲོ་བ་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་གོ་སླའོ། །དེ་ལ་མཇུག་བསྡུ་བ་བསྟན་པར་བྱ་བའི་
ཕྱིར༑ འདི་ནི་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ་གོ་སླའོ། །དེ་ལྟར་སེམས་ཅན་ལ་དམིགས་པའི་སྙིང་རྗེ་ལ་ཕྱག་འཚལ་ནས། ཆོས་ལ་དམིགས་པ་དང། དམིགས་པ་མེད་པའི་སྙིང་རྗེ་ལ་ཕྱག་འཚལ་བར་བྱ་བའི་ཕྱིར་འབྲེལ་པ་སྟོན་པ་ནི་ཆོས་ལ་དམིགས་པ་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ་གོ་
སླའོ༑ །འགྲོ་བ་གཡོ་བའི་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི་ཅི་ལྟར་རླུང་གིས་གཡོ་བར་གྱུར་པའི་ཆུའི་ནང་དུ་ཟླ་བའི་གཟུགས་བརྙན་གཡོ་བའི་རང་བཞིན་དུ་མཐོང་བ་དང། གདུང་བ་སེལ་བ་ལ་སོགས་པའི་དོན་བྱེད་མི་ནུས་པས་རང་བཞིན་མེད་པར་མཐོང་བ་དེ་བཞིན་དུ་བྱང་ཆུབ་སེམས་དཔས་ཀྱང་རླུང་དང་ཉི་
མ་ལ་སོགས་པའི་རྐྱེན་གཞན་དང་ལྡན་པའི་སྐད་ཅིག་མ་སོ་སོ་ལ་ནུས་པ་ཐ་དད་པས་འགྲོ་བ་སྐད་ཅིག་མར་མཐོང་བ་དང། རང་ལས་མ་ཡིན་གཞན་ལས་མ་ཡིན་ཞེས་བྱ་བ་ལ་སོགས་པས་དོན་དམ་པར་སྐྱེ་བ་མེད་པའི་ཕྱིར་འགྲོ་བ་རང་བཞིན་གྱིས་སྟོང་པར་མཐོང་ནས་སྡུག་བསྔལ་ལས་འདྲེན་པར་འདོད་པའི་
རང་བཞིན་ཅན་གང་ཡིན་པ་དེ་ལ་འདུད་དོ། །སྙིང་རྗེར་གྱུར་གང་ཞེས་བྱ་བ་ལ་སོགས་པ་ནི། འགྲོ་ལ་སྙིང་རྗེར་གྱུར་གང་དེ་ལ་འདུད། །ཅེས་སྔར་བསྟན་པའི་ཚིག་རྐང་དེ་འདིར་སྦྱར་བའོ། །དེ་འགྲེལ་པས་བསྟན་པར་བྱ་བའི་ཕྱིར། འདི་ལྟར་ཞེས་བྱ་བ་ལ་སོགས་པ་སྨོས་ཏེ། ཆུ་རྙོག་མ་ཅན་ལ་
