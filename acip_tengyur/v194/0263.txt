རྒྱུ་མཚན་ཅན་གྱི་ཡང་མེད་དོ་ཞེས་པས་ཋའི་ཐའོ། །ཨ་ཛ། བྲ་ཛ་འགྲོ་བ་ལའོ། །ཤྲྀ་འཚེ་བ་ལའོ། །སྑ་ད་འཚེ་བ་ལའོ། །ཥྛཱ་འགྲོ་བ་ལས་ལྡོག་པ་ལའོ། །ཨིཥུ་འདོད་པ་ལའོ། །ཙྪི་དིར་རྣམ་པ་གཉིས་སུ་བྱེད་པ་ལའོ། །བྷི་དིར་རྣམ་པར་འཇོམས་པ་ལའོ། །ཨི་
ཡིག་དང་རེ་ཕ་ནི། ཨིར་འགྲོ་བ་ལ་ཡང་ནའོ་ཞེས་པའི་རྟགས་དོན། མུ་ཧ་རྣམ་པར་རྨོངས་པ་ལའོ། །མུ་ཙྪི་ཐར་པ་ལའོ། །རུ་ཙ་འབར་བ་ལའོ། །ཤྲ་ཐ་ལྷོད་པ་ལའོ། །སྠི་ཡཱི་ཨེ་སྱཱ་ཡཱི་འཕེལ་བ་ལའོ། །སྒྲ་འདི་རྣམས་ལས་ཀི་ར་ཙ་ཕ་རོལ་དང་ཚིག་ཕྲད་མཐའོ། །དེ་ལ་ཏིཥྛ་ཏིའི་ཨ་ཙ་ལ་ཀིཏ་
ལའང་ཨཱཏ་ཀྱི་དབྱིའོ་ཞེས་པས་ཨཱ་དབྱིའོ། །སྠི་རཿ། ཡུ་གྷཉ་ཨས་ཀྱ་རྣམས་མིན་པ་ལ་ཨ་ཛིའི་བཱིའི་འོ་ཞེས་པས་བཱིའི་དངོས་པོར་མི་འགྱུར་ཏེ་ཚིག་ཕྲད་ལས་སོ། །ཨ་ཛི་ར་མ། ཤྲྀ་ཎི་གཉིས་ཉིད་དང་། ཨཏ་ཀྱི་ཨི་ཉིད་དང་། ར་ཡིག་དབྱིའོ། །ཤི་ཤི་རཾ། སྑ་དིའི་ས་ཡིག་དབྱིའོ། །
ཁ་དི་རཿ། སྠཱ་ཞེས་པའི་ཐུང་ངུ་ཉིད་དང་། བརྟན་པའམ་སྨྲ་བ། བུཀ་ཨཱ་ག་མ། ཀ་ཡིག་ཀིཏ་བྱ་བའི་དོན་ཏོ། །སྠ་བི་རཿ། ཨི་ཥི་རཿ། ཙྪི་དི་རཿ། བྷི་དི་རཿ། མུ་ཧི་རཿ། ཁྱིམ། མུ་ཙི་རཿ། དཱ་ཏཱ། རུ་ཙི་རཿ། ཤྲ་ཐི་རཿ། ཤྲ་ཐི་ར། སྥི་ཡི་རཾ། དེ་བཞིན་དུ་གཞན་དགུན། ཡང་བལྟ་བར་བྱའོ། །ཏཉྩི།
སེང་ལྡེང་། བཉྩི། ཤ་ཀི། །ཀྵི་པི། ཀྵུ་དི། རུ་དི། མ་དི། མནྡི། ཙནྡི་རྣམས་ལས་རཀའོ། །ཏཉྩུ་ཡང་དག་གནས་བརྟན། མདའ། རྣམ་པ་གཉིས་སུ་བྱེད་པ། རྡོ་རྗེ། རྨོངས་པ། པར་བསྐུལ་བ་ལའོ། །བཉྩུ་འགྲོ་མཛེས་པ། ཤྲ་ཐི་ར། རྒྱས་པ། ཕ་ལའོ། །ཤ་ཀླྀ་ནུས་པ་ལའོ། །ཀྵི་པ་དེད་པ་ལའོ། །ཀྵུ་
ད་ཡང་དག་པར་བསྔགས་པ་ལའོ། །རུ་དིར་མཚི་མ་འཛག་པ་ལའོ། །མ་དཱི་དགའ་བ་ལའོ། །མ་དི་བླུན་པ་ལའོ། །ཙ་དི་ཀུན་དུ་ཚིམ་པ་ལའོ། །འདི་རྣམས་ལས་རཀ་ཕ་རོལ་ཏུ་འགྱུར་རོ། །ཀ་ཡིག་ཀིཏུ་ལའོ་ཞེས་པའི་རྟགས་དོན། ཀཏ྄་ལ་ཧལ་གྱི་ཨི་དིཏ་མ་ཡིན་པའི་མཐའ་ཉེའིའོ་
ཞེས་པས་ན་དབྱིའོ། །ནྱཾ་ཀུ་ལ་སོགས་པ་རྣམས་སོ་ཞེས་པས་ཀུ་ཉིད་དོ། །ཏ་ཀྲཾ། ད་རའམ་ཞོ་དཀྲོགས་པ། བ་ཀྲཾ། འཁྱོག་པོ། ཤ་ཀྲཿ། ལྷའི་རྒྱལ་པོ། ཀྵི་པྲཾ། མྱུར་བ། ཀྵུ་དྲཿ། ཕྲ་མོའམ་ཞན་པ། རུ་དྲཿ། དྲག་པོའམ་ལྷ་ཆེན་པོ། ཀིཏ྄་ལ་ཧལ་གྱི་ཨི་དིཏ་མ་ཡིན་པའི་མཐའ་ཉེའི་འོ་ཞེས་པས་ན་དབྱི་བར་མི་འགྱུར་
