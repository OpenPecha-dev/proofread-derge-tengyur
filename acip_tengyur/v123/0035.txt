རྣམས་སྨིག་རྒྱུ་འདྲར་ཤེས་ནས། །བདུད་ཀྱི་མེ་ཏོག་མཆོག་ནི་འདིར་གཅོད་པ། །འཆི་བདག་རྒྱལ་པོས་བལྟར་ཡང་མེད་པར་འགྲོ། །འཇིག་རྟེན་འདི་དག་བུམ་པ་འདྲར་རྟོགས་ཤིང། །དེ་བཞིན་ཆོས་རྣམས་སྨིག་རྒྱུ་འདྲར་ཤེས་ནས། །བདུད་ཀྱི་མེ་ཏོག་མཆོག་ནི་འདིར་གཅོད་པ།།
