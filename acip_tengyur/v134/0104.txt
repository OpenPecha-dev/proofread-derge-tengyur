ཡིན། །དེ་ཉིད་ཁྲུམས་སྟོད་ཟླ་བ་ཡི། །ཚེས་གཅིག་ནས་ནི་ཕྱི་མར་གྲགས། །ཟླ་གསུམ་དབྱར་ནི་འགྲོ་བ་ཡི། །སྤྱོད་པ་དགེ་སློང་རྣམས་ཀྱིས་སྤང། །བྱ་རྣམས་ཀྱིས་ཀྱང་རང་གི་གནས། །དབྱར་གྱི་དུས་སུ་མི་སྤོང་ངོ། །ཆུ་སྟོད་ཅན་གྱི་ཟླ་ཕྱེད་ནས། །ཀུན་ནས་ཉེ་ཕྱིར་འབད་པ་

ཚངས་པ་མཚུངས་པར་སྤྱོད་པ་དང། །ཡོན་ཏན་ཅན་དང་ཐོས་པ་འཛིན། །སྙིང་རྗེ་ཅན་ཡིད་དེར་མངོན་ནས། །ཐོག་མར་ལྟུང་བ་མི་འབྱིན་ཅིང། །བྱུང་བ་སླར་ཡང་འཆོས་བྱེད་པ། །དེ་ལ་སོགས་པའི་རྣམ་པ་ཡིས། །གནས་ནི་ཡོངས་སུ་བརྟག་པར་བྱ། །སྨན་དང་བསོད་སྙོམས་དག་
