༄༅། །ཅད་དུ་རྒྱུ་ནི་འདུལ་བ་ཡིན་པར་གསུངས་ཏེ། དེས་ཚུལ་ཁྲིམས་ལ་གནས་ཚུལ་ཁྲིམས་ལ་བརྟེན་ཏེ་ལམ་བསྒོམ་པའི་ཕྱིར་རོ། །ད་ནི་བསྟན་པའི་གནས་རྒྱུ་ཉིད་ཀྱིས་སྟོད་པར་བྱེད་དེ། ཇི་སྲིད་འགྲོ་འདིར་རྒྱལ་བ་ཐམས་ཅད་ཀྱི། །མཛོད་མཆོག་འདུལ་བ་རྫོགས་པ་
འདི་གནས་པ། །དེ་སྲིད་ཆོས་སྒྲོན་འདི་ནི་གནས་འགྱུར་ཞིང། །བཟློག་པར་གྱུར་ན་ཞི་བ་ཐོབ་པར་འགྱུར། །ཞེས་བྱ་བ་ལ་རྫོགས་པའི་ཞེས་བྱ་བ་ནི་ཚང་བ་སྟེ། གསོལ་བ་དང་ལས་ལ་སོགས་པ་ལ་འཇུག་ནུས་པའོ། །དེ་སྲིད་ཆོས་སྒྲོན་འདི་ནི་གནས་འགྱུར་ཞིང་ཞེས་བྱ་བའི་སྒྲ་ནི་ངེས་པར་
གཟུང་བའི་དོན་ཏོ། །དེས་འདིར་དོན་དུ་ན་དུས་ཇི་སྲིད་དུ་འགྲོ་བ་འཇིག་རྟེན་འཛམ་བུའི་གླིང་འདིར་རྒྱལ་བ་ཐམས་ཅད་ཀྱི་དམ་པའི་ཆོས་ཀྱི་མཛོད་ཀྱི་མཆོག་འདུལ་བ་རྫོགས་པ་འདི་གནས་པར་འགྱུར་བའི་དུས་དེ་སྲིད་ཁོ་ནར་དེ་བཞིན་གཤེགས་པའི་དམ་པའི་ཆོས་ཀྱི་སྒྲོན་མ་ལྷག་མ་འདི་ཡང་གནས་
པར་འགྱུར་ཞིང། དེ་ལས་བཟློག་པར་གྱུར་པ་འདུལ་བ་རྫོགས་པར་ནུབ་པར་གྱུར་ན་དམ་པའི་ཆོས་ཀྱི་སྒྲོན་མ་ལྷག་མ་ཡང་ཞི་བ་ཐོབ་པར་འགྱུར་ཏེ་ནུབ་པ་ཁོ་ནར་འགྱུར་རོ་ཞེས་བྱ་བའི་ཐ་ཚིག་གོ། །ད་ནི་དཀོན་པ་ཉིད་ཀྱིས་སྟོད་པར་བྱེད་དེ། སངས་རྒྱས་མི་བཞུགས་འཇིག་རྟེན་ཚིགས་བཅད་དང་
མདོ་སྡེའི་ཡན་ལག་དེ་དང་དེ་དག་སྤྱོད། །བསླབ་འཇུག་མེད་ན་འདུལ་བའང་མེད་འགྱུར་བས། །དེའི་ཕྱིར་རྒྱལ་བ་དག་པས་ཆེས་འདི་དཀོན། །ཞེས་བྱ་བ་ལ། མི་བཞུགས་ཞེས་བྱ་བས་ནི་མེད་པའི་དོན་བསྟན་ཏོ། །དེ་དང་དེ་དག་ཅེས་བྱ་བ་ནི་ཟློག་པའི་ཚིག་སྟེ་མང་བ་ཉིད་དུ་བསྟན་ཏོ། །དེས་ནི་
འདི་ལྟར་སངས་རྒྱས་མི་བཞུགས་སུ་ཟིན་ཀྱང་འཇིག་རྟེན་འཛམ་བུའི་གླིང་འདི་ན། འདོད་པ་ཁྱོད་ཀྱི་རྩ་བ་ཤེས། །ཁྱོད་ནི་ཀུན་དུ་རྟོག་ལས་བྱུང། །ཁྱོད་ལ་ཀུན་དུ་བརྟག་མི་བྱ། །དེས་ན་ང་ལ་ཁྱོད་མི་འབྱུང། །ཞེས་བྱ་བ་ལ་སོགས་པ་ཚིགས་བཅད་དེ་དང་དེ་ལྟ་བུ་དག་དང། ལྷའི་མདོ་ལ་
སོགས་པ་མདོ་སྡེའི་ཡན་ལག་དམ་དུམ་མང་པོ་དེ་དང་དེ་དག་སྤྱོད་པ་ཡོད་ཀྱི། །བཅོམ་ལྡན་འདས་བྱུང་དུ་ཟིན་ཀྱང་བསླབ་པ་ལ་འཇུག་པ་མེད་ཅིང་བསླབ་པ་བཅའ་བ་མེད་ན་འདུལ་བའང་མེད་ཅིང་སྤྱོད་པར་མི་འགྱུར་བས་དེའི་ཕྱིར་རྒྱལ་བ་སངས་རྒྱས་བཅོམ་ལྡན་འདས་དག་པས་ཀྱང་ཆོས་འདུལ་བ་
